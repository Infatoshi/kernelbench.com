"""Single-launch W4A16 decode megakernel for the Kimi-Linear motif.

The module hierarchy deliberately mirrors reference.py so its state_dict loads
strictly.  The timed path consists of allocations (which do not enqueue GPU
work) followed by exactly one raw CUDA kernel launch.
"""
from __future__ import annotations

import os
import sys

import torch
import torch.nn as nn
import torch.utils.cpp_extension as _ce
from torch.utils.cpp_extension import load_inline


GROUP = 128
MAX_CONTEXT = 20000


class QuantLinear(nn.Module):
    def __init__(self, in_f: int, out_f: int, group: int = GROUP):
        super().__init__()
        self.in_f, self.out_f, self.group = in_f, out_f, group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(in_f // group, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(in_f // group, out_f, dtype=torch.bfloat16))


class QuantExperts(nn.Module):
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, in_f // group, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, in_f // group, out_f, dtype=torch.bfloat16))


class KDA(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        d, c, h = cfg.hidden, cfg.kda_heads * cfg.kda_head_dim, cfg.kda_heads
        self.q_proj = QuantLinear(d, c, cfg.group)
        self.k_proj = QuantLinear(d, c, cfg.group)
        self.v_proj = QuantLinear(d, c, cfg.group)
        self.g_proj = QuantLinear(d, c, cfg.group)
        self.beta_proj = nn.Linear(d, h, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, c, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(c, d, cfg.group)


class MLA(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        h, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, h * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, h * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(h * cfg.v_head, d, cfg.group)


class MoE(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        d, m, e = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, e, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(e, d, m, cfg.group)
        self.up = QuantExperts(e, d, m, cfg.group)
        self.down = QuantExperts(e, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)


class Block(nn.Module):
    def __init__(self, cfg, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)


_CUDA = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <mma.h>
#include <vector>
#include <cmath>

using bf16 = __nv_bfloat16;
constexpr int NB = 188;
constexpr int NT = 128;
constexpr int D = 2304;
constexpr int H = 32;
constexpr int HD = 128;
constexpr int C = 4096;
constexpr int M = 1024;
constexpr int ACTIVE = 8;
constexpr int MAXL = 20000;

struct QW { const unsigned char* q; const bf16* s; const bf16* z; };
struct EW { const unsigned char* q; const bf16* s; const bf16* z; };
struct Params {
  const bf16 *hin, *cold, *krold;
  bf16 *hout, *cnew, *krnew;
  unsigned char* scratch;
  float* S[3]; bf16* cq[3]; bf16* ck[3]; bf16* cv[3];
  const bf16* norm[4][2];
  QW katt[3][5]; const bf16* beta[3]; const bf16* conv[3];
  QW matt[4];
  const bf16* router[4];
  EW ex[4][6];
  int L;
};

__device__ __forceinline__ float b2f(bf16 x) { return __bfloat162float(x); }
__device__ __forceinline__ bf16 f2b(float x) { return __float2bfloat16_rn(x); }

// Scratch layout.  The first cache line is initialized on the CPU and holds
// the reusable barrier.  All vector temporaries are shared by successive stages.
__device__ __forceinline__ bf16* sx(Params& p) { return (bf16*)(p.scratch + 256); }
__device__ __forceinline__ bf16* sn(Params& p) { return (bf16*)(p.scratch + 4864); }
__device__ __forceinline__ bf16* sp0(Params& p) { return (bf16*)(p.scratch + 9472); }
__device__ __forceinline__ bf16* sp1(Params& p) { return (bf16*)(p.scratch + 17664); }
__device__ __forceinline__ bf16* sp2(Params& p) { return (bf16*)(p.scratch + 25856); }
__device__ __forceinline__ bf16* sp3(Params& p) { return (bf16*)(p.scratch + 34048); }
__device__ __forceinline__ float* pred(Params& p) { return (float*)(p.scratch + 42240); }
__device__ __forceinline__ bf16* sov(Params& p) { return (bf16*)(p.scratch + 58624); }
__device__ __forceinline__ bf16* satt(Params& p) { return (bf16*)(p.scratch + 66816); }
__device__ __forceinline__ int* routes(Params& p) { return (int*)(p.scratch + 71424); }
__device__ __forceinline__ float* rweight(Params& p) { return (float*)(p.scratch + 71456); }
__device__ __forceinline__ float* stats(Params& p) { return (float*)(p.scratch + 71552); }
__device__ __forceinline__ float* ff(Params& p) { return (float*)(p.scratch + 72064); }
__device__ __forceinline__ float* aq(Params& p) { return (float*)(p.scratch + 108928); }
__device__ __forceinline__ float* zl(Params& p) { return (float*)(p.scratch + 174464); }
__device__ __forceinline__ float* scores(Params& p) { return (float*)(p.scratch + 240000); }

__device__ __forceinline__ void gsync(Params& p, int& sense) {
  __syncthreads();
  int* count = (int*)p.scratch;
  int* epoch = count + 1;
  if (threadIdx.x == 0) {
    __threadfence();
    int ticket = atomicAdd(count, 1);
    if (ticket == NB - 1) {
      atomicExch(count, 0);
      __threadfence();
      atomicAdd(epoch, 1);
    } else {
      while (atomicAdd(epoch, 0) == sense) __nanosleep(64);
    }
  }
  __syncthreads();
  ++sense;
}

__device__ __forceinline__ float wval(QW w, int k, int n, int N) {
  unsigned char v = w.q[(k >> 1) * N + n];
  int q = (k & 1) ? (v >> 4) : (v & 15);
  float s = b2f(w.s[(k >> 7) * N + n]);
  float z = b2f(w.z[(k >> 7) * N + n]);
  return ((float)q - z) * s;
}

__device__ __forceinline__ float ewval(EW w, int e, int k, int n, int K, int N) {
  size_t qo = (size_t)e * (K >> 1) * N;
  size_t so = (size_t)e * (K >> 7) * N;
  unsigned char v = w.q[qo + (size_t)(k >> 1) * N + n];
  int q = (k & 1) ? (v >> 4) : (v & 15);
  float s = b2f(w.s[so + (size_t)(k >> 7) * N + n]);
  float z = b2f(w.z[so + (size_t)(k >> 7) * N + n]);
  return ((float)q - z) * s;
}

__device__ __forceinline__ float qdot(const bf16* x, QW w, int K, int N, int n) {
  float a0=0.0f,a1=0.0f,a2=0.0f,a3=0.0f;
  for (int g = 0; g < K / 128; ++g) {
    float sc = b2f(w.s[g * N + n]);
    float ze = b2f(w.z[g * N + n]);
    int kb = g * 128;
    #pragma unroll 2
    for (int j = 0; j < 128; j += 8) {
      int k=kb+j; unsigned char v0=w.q[(size_t)(k>>1)*N+n];
      unsigned char v1=w.q[(size_t)(k>>1)*N+N+n];
      unsigned char v2=w.q[(size_t)(k>>1)*N+2*N+n];
      unsigned char v3=w.q[(size_t)(k>>1)*N+3*N+n];
      a0=fmaf(b2f(x[k]),((float)(v0&15)-ze)*sc,a0);a0=fmaf(b2f(x[k+1]),((float)(v0>>4)-ze)*sc,a0);
      a1=fmaf(b2f(x[k+2]),((float)(v1&15)-ze)*sc,a1);a1=fmaf(b2f(x[k+3]),((float)(v1>>4)-ze)*sc,a1);
      a2=fmaf(b2f(x[k+4]),((float)(v2&15)-ze)*sc,a2);a2=fmaf(b2f(x[k+5]),((float)(v2>>4)-ze)*sc,a2);
      a3=fmaf(b2f(x[k+6]),((float)(v3&15)-ze)*sc,a3);a3=fmaf(b2f(x[k+7]),((float)(v3>>4)-ze)*sc,a3);
    }
  }
  return (a0+a1)+(a2+a3);
}

// The newly appended latent can dominate the norm of a prefilled random cache
// after the preceding MoEs.  Keep oracle-style bf16 dequant rounding and a
// deterministic accumulation order for this one small projection.
__device__ __forceinline__ float qdot_round(const bf16* x,QW w,int K,int N,int n){
  float a=0;
  for(int g=0;g<K/128;++g){float sc=b2f(w.s[g*N+n]),ze=b2f(w.z[g*N+n]);int kb=g*128;
    for(int j=0;j<128;j+=2){int k=kb+j;unsigned char v=w.q[(size_t)(k>>1)*N+n];
      a=fmaf(b2f(x[k]),b2f(f2b(((float)(v&15)-ze)*sc)),a);
      a=fmaf(b2f(x[k+1]),b2f(f2b(((float)(v>>4)-ze)*sc)),a);}}
  return a;
}

__device__ __forceinline__ void gemv(const bf16* x, bf16* y, QW w, int K, int N) {
  int tid = blockIdx.x * blockDim.x + threadIdx.x;
  int step = NB * NT;
  for (int n = tid; n < N; n += step) y[n] = f2b(qdot(x, w, K, N, n));
}

__device__ __forceinline__ void gemv4(const bf16* x, bf16* y0, bf16* y1, bf16* y2, bf16* y3,
                                      QW w0,QW w1,QW w2,QW w3,int K,int N) {
  int tid=blockIdx.x*blockDim.x+threadIdx.x,step=NB*NT;
  for(int t=tid;t<4*N;t+=step){int which=t/N,n=t-which*N;QW w=which==0?w0:(which==1?w1:(which==2?w2:w3));
    bf16*y=which==0?y0:(which==1?y1:(which==2?y2:y3));y[n]=f2b(qdot(x,w,K,N,n));}
}

__device__ __forceinline__ void gemv_mla_pair(const bf16*x,bf16*q,bf16*kv,QW wq,QW wk){
  int tid=blockIdx.x*NT+threadIdx.x,step=NB*NT;
  for(int t=tid;t<6720;t+=step){if(t<6144)q[t]=f2b(qdot(x,wq,D,6144,t));else kv[t-6144]=f2b(qdot_round(x,wk,D,576,t-6144));}
}

// Warp-level bf16 tensor-core GEMV.  Sixteen identical activation rows turn a
// 1xK GEMV tile into an m16n16k16 MMA; only row zero is retained.  The int4 B
// tile exists solely in per-warp shared memory and is overwritten by C.
__device__ __forceinline__ void tc_gemv(const bf16*x,bf16*y,QW w,int K,int N){
  using namespace nvcuda::wmma;
  __shared__ __align__(16) unsigned char raw[4][1024];
  int warp=threadIdx.x>>5,lane=threadIdx.x&31;
  int gw=blockIdx.x*4+warp,nw=NB*4,tiles=N/16;
  for(int tile=gw;tile<tiles;tile+=nw){int nb=tile*16;
    fragment<accumulator,16,16,16,float> cf;fill_fragment(cf,0.0f);
    for(int kb=0;kb<K;kb+=16){bf16*ab=(bf16*)raw[warp];bf16*bb=ab+256;
      for(int u=lane;u<256;u+=32){int r=u>>4,c=u&15;ab[u]=x[kb+c];bb[u]=f2b(wval(w,kb+r,nb+c,N));}
      __syncwarp();fragment<matrix_a,16,16,16,bf16,row_major> af;fragment<matrix_b,16,16,16,bf16,row_major> bf;
      load_matrix_sync(af,ab,16);load_matrix_sync(bf,bb,16);mma_sync(cf,af,bf,cf);__syncwarp();}
    float*dst=(float*)raw[warp];store_matrix_sync(dst,cf,16,mem_row_major);__syncwarp();if(lane<16)y[nb+lane]=f2b(dst[lane]);__syncwarp();}
}

__device__ __forceinline__ void tc_gemv4(const bf16*x,bf16*y0,bf16*y1,bf16*y2,bf16*y3,
                                         QW w0,QW w1,QW w2,QW w3,int K,int N){
  using namespace nvcuda::wmma;
  __shared__ __align__(16) unsigned char raw[4][1024];int warp=threadIdx.x>>5,lane=threadIdx.x&31;
  int gw=blockIdx.x*4+warp,nw=NB*4,one=N/16;
  for(int tile=gw;tile<4*one;tile+=nw){int which=tile/one,nb=(tile%one)*16;QW w=which==0?w0:(which==1?w1:(which==2?w2:w3));
    fragment<accumulator,16,16,16,float> cf;fill_fragment(cf,0.0f);
    for(int kb=0;kb<K;kb+=16){bf16*ab=(bf16*)raw[warp];bf16*bb=ab+256;
      for(int u=lane;u<256;u+=32){int r=u>>4,c=u&15;ab[u]=x[kb+c];bb[u]=f2b(wval(w,kb+r,nb+c,N));}
      __syncwarp();fragment<matrix_a,16,16,16,bf16,row_major> af;fragment<matrix_b,16,16,16,bf16,row_major> bf;
      load_matrix_sync(af,ab,16);load_matrix_sync(bf,bb,16);mma_sync(cf,af,bf,cf);__syncwarp();}
    float*dst=(float*)raw[warp];store_matrix_sync(dst,cf,16,mem_row_major);__syncwarp();if(lane<16){bf16*y=which==0?y0:(which==1?y1:(which==2?y2:y3));y[nb+lane]=f2b(dst[lane]);}__syncwarp();}
}

__device__ __forceinline__ void tc_mla_pair(const bf16*x,bf16*q,bf16*kv,QW wq,QW wk){
  tc_gemv(x,q,wq,D,6144);
  int tid=blockIdx.x*NT+threadIdx.x;
  for(int n=tid;n<576;n+=NB*NT)kv[n]=f2b(qdot_round(x,wk,D,576,n));
}

__device__ __forceinline__ void rms(const bf16* x, const bf16* w, bf16* y) {
  if (blockIdx.x != 0) return;
  __shared__ float sh[NT];
  float v = 0.0f;
  for (int i = threadIdx.x; i < D; i += NT) { float a=b2f(x[i]); v=fmaf(a,a,v); }
  sh[threadIdx.x] = v; __syncthreads();
  for (int s=NT/2; s; s>>=1) { if(threadIdx.x<s) sh[threadIdx.x]+=sh[threadIdx.x+s]; __syncthreads(); }
  float rr = rsqrtf(sh[0] / (float)D + 1.0e-6f);
  for (int i=threadIdx.x; i<D; i+=NT) y[i]=f2b(b2f(x[i])*rr*b2f(w[i]));
}

__device__ __forceinline__ void kda(Params& p, int b, int ki, int& sense) {
  bf16 *x=sx(p), *xn=sn(p), *q=sp0(p), *k=sp1(p), *v=sp2(p), *gr=sp3(p);
  rms(x,p.norm[b][0],xn); gsync(p,sense);
  gemv4(xn,q,k,v,gr,p.katt[ki][0],p.katt[ki][1],p.katt[ki][2],p.katt[ki][3],D,C);
  int tid=blockIdx.x*NT+threadIdx.x;
  if (tid < H) {
    float z=0; const bf16* bw=p.beta[ki]+(size_t)tid*D;
    for(int j=0;j<D;++j) z=fmaf(b2f(xn[j]),b2f(bw[j]),z);
    stats(p)[tid]=1.0f/(1.0f+expf(-b2f(f2b(z))));
  }
  gsync(p,sense);

  for(int n=tid;n<C;n+=NB*NT) {
    bf16* cs[3]={p.cq[ki],p.ck[ki],p.cv[ki]};
    bf16* ps[3]={q,k,v};
    for(int a=0;a<3;++a) {
      bf16* cp=cs[a];
      float z=fmaf(b2f(cp[n]),b2f(p.conv[ki][(a*C+n)*4]),0.0f);
      z=fmaf(b2f(cp[C+n]),b2f(p.conv[ki][(a*C+n)*4+1]),z);
      z=fmaf(b2f(cp[2*C+n]),b2f(p.conv[ki][(a*C+n)*4+2]),z);
      z=fmaf(b2f(ps[a][n]),b2f(p.conv[ki][(a*C+n)*4+3]),z);
      bf16 cur=ps[a][n]; cp[n]=cp[C+n]; cp[C+n]=cp[2*C+n]; cp[2*C+n]=cur;
      ps[a][n]=f2b(z/(1.0f+expf(-z)));
    }
  }
  gsync(p,sense);

  for(int t=tid;t<C;t+=NB*NT) {
    int h=t/HD, dv=t%HD; float z=0;
    for(int dk=0;dk<HD;++dk) {
      int kd=h*HD+dk;
      float decay=1.0f/(1.0f+expf(b2f(gr[kd])));
      z=fmaf(p.S[ki][(size_t)kd*HD+dv]*decay,b2f(k[kd]),z);
    }
    pred(p)[t]=z;
  }
  gsync(p,sense);
  for(int t=tid;t<H*HD*HD;t+=NB*NT) {
    int dv=t%HD, kd=t/HD, h=kd/HD;
    float decay=1.0f/(1.0f+expf(b2f(gr[kd])));
    float old=p.S[ki][t]*decay;
    p.S[ki][t]=fmaf(stats(p)[h]*b2f(k[kd]),b2f(v[h*HD+dv])-pred(p)[h*HD+dv],old);
  }
  gsync(p,sense);
  for(int t=tid;t<C;t+=NB*NT) {
    int h=t/HD,dv=t%HD; float z=0;
    for(int dk=0;dk<HD;++dk) z=fmaf(p.S[ki][(size_t)(h*HD+dk)*HD+dv],b2f(q[h*HD+dk])*0.08838834764831845f,z);
    sov(p)[t]=f2b(z);
  }
  gsync(p,sense);
  gemv(sov(p),satt(p),p.katt[ki][4],C,D); gsync(p,sense);
  for(int n=tid;n<D;n+=NB*NT) x[n]=f2b(b2f(x[n])+b2f(satt(p)[n]));
  gsync(p,sense);
}

__device__ __forceinline__ void mla(Params& p, int b, int& sense) {
  bf16 *x=sx(p),*xn=sn(p),*q=sp0(p),*kv=sp1(p);
  int tid=blockIdx.x*NT+threadIdx.x;
  rms(x,p.norm[b][0],xn); gsync(p,sense);
  gemv_mla_pair(xn,q,kv,p.matt[0],p.matt[1]); gsync(p,sense);
  int LL=p.L+1;
  for(int t=tid;t<H*32;t+=NB*NT) {
    int h=t/32,i=t%32; float inv=powf(10000.0f,-(2.0f*i)/64.0f), a=p.L*inv;
    float co=cosf(a),si=sinf(a); int j=h*192+128+2*i;
    float e=b2f(q[j]),o=b2f(q[j+1]); q[j]=f2b(e*co-o*si); q[j+1]=f2b(o*co+e*si);
  }
  for(int i=tid;i<32;i+=NB*NT) {
    float inv=powf(10000.0f,-(2.0f*i)/64.0f),a=p.L*inv,co=cosf(a),si=sinf(a);
    float e=b2f(kv[512+2*i]),o=b2f(kv[513+2*i]);
    p.krnew[(size_t)p.L*64+2*i]=f2b(e*co-o*si); p.krnew[(size_t)p.L*64+2*i+1]=f2b(o*co+e*si);
  }
  for(int i=tid;i<p.L*512;i+=NB*NT) p.cnew[i]=p.cold[i];
  for(int i=tid;i<p.L*64;i+=NB*NT) p.krnew[i]=p.krold[i];
  for(int i=tid;i<512;i+=NB*NT) p.cnew[(size_t)p.L*512+i]=kv[i];
  gsync(p,sense);

  for(int t=tid;t<H*512;t+=NB*NT) {
    int h=t/512,c=t%512; float z=0;
    for(int d=0;d<128;++d) z=fmaf(b2f(q[h*192+d]),wval(p.matt[2],c,h*256+d,8192),z);
    aq(p)[t]=z;
  }
  gsync(p,sense);
  for(int t=tid;t<H*LL;t+=NB*NT) {
    int h=t/LL,l=t%LL; float z=0;
    const bf16* cp=p.cnew+(size_t)l*512;
    for(int c=0;c<512;++c) z=fmaf(b2f(cp[c]),aq(p)[h*512+c],z);
    const bf16* rp=p.krnew+(size_t)l*64;
    for(int d=0;d<64;++d) z=fmaf(b2f(q[h*192+128+d]),b2f(rp[d]),z);
    scores(p)[h*MAXL+l]=z*0.07216878364870322f;
  }
  gsync(p,sense);
  if(blockIdx.x<H) {
    __shared__ float sh[NT]; int h=blockIdx.x; float mx=-INFINITY;
    for(int l=threadIdx.x;l<LL;l+=NT) mx=fmaxf(mx,scores(p)[h*MAXL+l]);
    sh[threadIdx.x]=mx; __syncthreads();
    for(int s=NT/2;s;s>>=1){if(threadIdx.x<s)sh[threadIdx.x]=fmaxf(sh[threadIdx.x],sh[threadIdx.x+s]);__syncthreads();}
    mx=sh[0]; float sm=0;
    for(int l=threadIdx.x;l<LL;l+=NT) sm+=expf(scores(p)[h*MAXL+l]-mx);
    sh[threadIdx.x]=sm;__syncthreads();
    for(int s=NT/2;s;s>>=1){if(threadIdx.x<s)sh[threadIdx.x]+=sh[threadIdx.x+s];__syncthreads();}

class Model(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)
        # Non-persistent so strict reference state_dict loading sees no extra key.
        # 3 MiB covers scores for 20k tokens plus all other intermediates.
        self.register_buffer("_scratch", torch.zeros(3_000_000, dtype=torch.uint8), persistent=False)

    @staticmethod
    def _q(args, q):
        args.extend((q.w_q, q.scales, q.zeros))

    @staticmethod
    def _e(args, e):
        args.extend((e.w_q, e.scales, e.zeros))

    def step(self, hidden, state):
        global _kernel
        if _kernel is None:
            _kernel = _load_kernel()
        old_c = state[3]["c_kv"]
        old_r = state[3]["k_rope"]
        length = old_c.shape[0]
        if length >= MAX_CONTEXT:
            raise RuntimeError(f"context length {length} exceeds fused-kernel limit {MAX_CONTEXT}")
        out = torch.empty_like(hidden)
        new_c = torch.empty((length + 1, 512), device=hidden.device, dtype=torch.bfloat16)
        new_r = torch.empty((length + 1, 64), device=hidden.device, dtype=torch.bfloat16)
        args = [hidden, out, self._scratch, old_c, old_r, new_c, new_r]
        for i in range(3):
            st = state[i]
            args.extend((st["S"], st["cq"], st["ck"], st["cv"]))
        for block in self.blocks:
            args.extend((block.attn_norm, block.moe_norm))
            if block.kind == "K":
                a = block.attn
                for q in (a.q_proj, a.k_proj, a.v_proj, a.g_proj, a.o_proj):
                    self._q(args, q)
                args.extend((a.beta_proj.weight, a.conv_w))
            else:

                    self._q(args, q)

            args.append(m.router.weight)
            for e in (m.gate, m.up, m.down, m.s_gate, m.s_up, m.s_down):
                self._e(args, e)

        state[3]["k_rope"] = new_r
