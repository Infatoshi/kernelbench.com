"""Kimi-Linear W4A16 hybrid decode: single-launch megakernel.

The entire per-token forward (3x KDA + 1x MLA block, each followed by a
64-expert MoE, every int4 dequant-GEMV fused, causal conv, KDA delta-rule
state update, absorbed MLA latent attention, MoE router + experts, RMSNorms,
residuals, latent-cache append) is one persistent Triton kernel launched
exactly once per step().

Grid: 188 programs (one per SM on the RTX PRO 6000), cooperative launch so
all blocks are co-resident; cross-phase sequencing is a single-slot
generation barrier (every block arrives exactly once per phase; the last
arriver bumps the generation word).

W4A16 (matches reference.py): w_q (K/2, N) uint8 packs two int4 along `in`
(lo nibble = even k, hi nibble = odd k); scales/zeros (K/128, N) bf16;
w[k,n] = (unpack(w_q)[k,n] - zeros[k//128,n]) * scales[k//128,n].
Dequant is fused into every GEMV; no bf16 weight is ever materialized.

MLA uses the absorb identities
    q_nope . (c_kv @ W_uk) == (W_uk^T q_nope) . c_kv
    sum_l p_l (c_kv @ W_uv) == (sum_l p_l c_kv) @ W_uv ,
so attention runs in the 512-d latent space and per-token K/V expansion is
never materialized.
"""
from __future__ import annotations

import math
import os
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "kimi_linear_w4a16_decode"
HARDWARE_REQUIRED = ["RTX_PRO_6000"]
EPS = 1.0e-6
GROUP_SIZE = 128

HDIM = tl.constexpr(2304)
CQV = tl.constexpr(4096)
KV = tl.constexpr(512)
NQ = tl.constexpr(6144)
ROUTE_SCALE = tl.constexpr(2.446)

NSLOT = tl.constexpr(128)
LSJ = tl.constexpr(40)
EPSJ = tl.constexpr(1.0e-6)
GRID = 188
LTH32 = tl.constexpr(math.log2(10000.0) / 32.0)

# python-side layout values
LS = 40
GB = 4 * LS
NPAR = GB + 31

# jit-side layout values as constexpr globals
def _mk():
    import triton.language as _tl
    names = ["GP_ACC", "GP_H", "GP_QKV", "GP_BETA", "GP_OACC", "GP_ISEL", "GP_WSEL",
             "GP_INTERM", "GP_QBUF", "GP_QLAT", "GP_QR", "GP_WS", "GP_BAR", "GP_CACHE",
             "GP_KRC", "GP_S0", "GP_S1", "GP_S2", "GP_CQ0", "GP_CK0", "GP_CV0",
             "GP_CQ1", "GP_CK1", "GP_CV1", "GP_CQ2", "GP_CK2", "GP_CV2", "GP_OLAT", "GP_OBUF"]
    g = {}
    for i, n in enumerate(names):
        g[n] = _tl.constexpr(GB - 0 + i if False else 160 + i)
    return g

_g = _mk()
GP_ACC = _g["GP_ACC"]; GP_H = _g["GP_H"]; GP_QKV = _g["GP_QKV"]; GP_BETA = _g["GP_BETA"]
GP_OACC = _g["GP_OACC"]; GP_ISEL = _g["GP_ISEL"]; GP_WSEL = _g["GP_WSEL"]
GP_INTERM = _g["GP_INTERM"]; GP_QBUF = _g["GP_QBUF"]; GP_QLAT = _g["GP_QLAT"]
GP_QR = _g["GP_QR"]; GP_WS = _g["GP_WS"]; GP_BAR = _g["GP_BAR"]; GP_CACHE = _g["GP_CACHE"]
GP_KRC = _g["GP_KRC"]; GP_S0 = _g["GP_S0"]; GP_S1 = _g["GP_S1"]; GP_S2 = _g["GP_S2"]
GP_CQ0 = _g["GP_CQ0"]; GP_CK0 = _g["GP_CK0"]; GP_CV0 = _g["GP_CV0"]
GP_CQ1 = _g["GP_CQ1"]; GP_CK1 = _g["GP_CK1"]; GP_CV1 = _g["GP_CV1"]
GP_CQ2 = _g["GP_CQ2"]; GP_CK2 = _g["GP_CK2"]; GP_CV2 = _g["GP_CV2"]
GP_OLAT = _g["GP_OLAT"]; GP_OBUF = _g["GP_OBUF"]; GP_DBG = tl.constexpr(189); GP_XNS = tl.constexpr(190)


@triton.jit
def _p_u8(P, i):
    return tl.cast(tl.load(P + i), tl.pointer_type(tl.uint8))


@triton.jit
def _p_bf16(P, i):
    return tl.cast(tl.load(P + i), tl.pointer_type(tl.bfloat16))


@triton.jit
def _p_f32(P, i):
    return tl.cast(tl.load(P + i), tl.pointer_type(tl.float32))


@triton.jit
def _p_i32(P, i):
    return tl.cast(tl.load(P + i), tl.pointer_type(tl.int32))


@triton.jit
def _gbar(bar_ptr, slot: tl.constexpr, G: tl.constexpr):
    """Slot barrier: each phase transition uses its own counter so a late
    straggler's extra increment can never pollute the next barrier."""
    tl.atomic_add(bar_ptr + slot, 1)
    g = tl.atomic_add(bar_ptr + slot, 0)
    while g < G:
        g = tl.atomic_add(bar_ptr + slot, 0)


@triton.jit
def _prof_tick(P, i):
    pp = _p_i32(P, GP_DBG)
    z = tl.zeros((1,), tl.int64)
    tns = tl.inline_asm_elementwise("mov.u64 $0, %globaltimer;", "=l,l", [z], dtype=tl.int64, is_pure=False, pack=1)
    tl.store(pp + 100 + i * 2 + tl.arange(0, 1), tns)


@triton.jit
def _rrms(x_ptr, IS_BF16: tl.constexpr):
    offs = tl.arange(0, 4096)
    m = offs < HDIM
    if IS_BF16:
        x = tl.load(x_ptr + offs, mask=m, other=0.0, cache_modifier=".cg").to(tl.float32)
    else:
        x = tl.load(x_ptr + offs, mask=m, other=0.0, cache_modifier=".cg")
    return tl.rsqrt(tl.sum(x * x) / HDIM + EPSJ)


@triton.jit
def _xn(x_ptr, an_ptr, rrms, idx, IS_BF16: tl.constexpr):
    if IS_BF16:
        x = tl.load(x_ptr + idx, cache_modifier=".cg").to(tl.float32)
    else:
        x = tl.load(x_ptr + idx, cache_modifier=".cg")
    w = tl.load(an_ptr + idx, cache_modifier=".cg").to(tl.float32)
    return (x * rrms * w).to(tl.bfloat16).to(tl.float32)


@triton.jit
def _xfm(x_ptr, idx, msk, IS_BF16: tl.constexpr):
    if IS_BF16:
        return tl.load(x_ptr + idx, mask=msk, other=0.0, cache_modifier=".cg").to(tl.float32)
    else:
        return tl.load(x_ptr + idx, mask=msk, other=0.0, cache_modifier=".cg")


@triton.jit
def _xnm(x_ptr, an_ptr, rrms, idx, msk, IS_BF16: tl.constexpr):
    if IS_BF16:
        x = tl.load(x_ptr + idx, mask=msk, other=0.0, cache_modifier=".cg").to(tl.float32)
    else:
        x = tl.load(x_ptr + idx, mask=msk, other=0.0, cache_modifier=".cg")
    w = tl.load(an_ptr + idx, mask=msk, other=0.0, cache_modifier=".cg").to(tl.float32)
    return (x * rrms * w).to(tl.bfloat16).to(tl.float32)


@triton.jit
def _xf(x_ptr, idx, IS_BF16: tl.constexpr):
    if IS_BF16:
        return tl.load(x_ptr + idx, cache_modifier=".cg").to(tl.float32)
    else:
        return tl.load(x_ptr + idx, cache_modifier=".cg")


@triton.jit
def _prep_xn(P, pid, x_ptr, an_ptr, IS_BF16: tl.constexpr, NORMED: tl.constexpr, K2: tl.constexpr):
    # Fill pid-private scratch: xlo[r] (K2 bf16) at +0, xhi[r] at +2048.
    # Applies rms-norm with reference bf16 rounding when NORMED.
    xns = _p_bf16(P, GP_XNS) + pid * 4096
    rr = _rrms(x_ptr, IS_BF16) if NORMED else 0.0
    for base in range(0, K2, 256):
        idx = base + tl.arange(0, 256)
        msk = idx < K2
        ilo = 2 * idx
        if NORMED:
            lo = _xnm(x_ptr, an_ptr, rr, ilo, msk, IS_BF16)
            hi = _xnm(x_ptr, an_ptr, rr, ilo + 1, msk, IS_BF16)
        else:
            lo = _xfm(x_ptr, ilo, msk, IS_BF16)
            hi = _xfm(x_ptr, ilo + 1, msk, IS_BF16)
        tl.store(xns + idx, lo.to(tl.bfloat16), mask=msk)
        tl.store(xns + 2048 + idx, hi.to(tl.bfloat16), mask=msk)


@triton.jit
def _gemv_core(P, pid, wb, sb, zb, NSZ: tl.constexpr, n0, r0, KCH: tl.constexpr, BN: tl.constexpr):
    # Dequant-GEMV over packed rows [r0, r0+KCH) -> acc (BN,) fp32.
    xns = _p_bf16(P, GP_XNS) + pid * 4096
    acc2 = tl.zeros((64, BN), tl.float32)
    wq0 = tl.load(wb + (r0 + tl.arange(0, 64))[:, None] * NSZ + n0 + tl.arange(0, BN)[None, :], cache_modifier=".cg")
    for r in range(0, KCH, 64):
        rn = tl.minimum(r + 64, KCH - 64)
        wq = tl.load(wb + (r0 + rn + tl.arange(0, 64))[:, None] * NSZ + n0 + tl.arange(0, BN)[None, :], cache_modifier=".cg")
        g = (2 * (r0 + r)) // 128
        s0 = tl.load(sb + g * NSZ + n0 + tl.arange(0, BN), cache_modifier=".cg").to(tl.float32)
        z0 = tl.load(zb + g * NSZ + n0 + tl.arange(0, BN), cache_modifier=".cg").to(tl.float32)
        xlo = tl.load(xns + r + tl.arange(0, 64)).to(tl.float32)
        xhi = tl.load(xns + 2048 + r + tl.arange(0, 64)).to(tl.float32)
        wlo = ((wq0 & 0xF).to(tl.float32) - z0[None, :]) * s0[None, :]
        whi = ((wq0 >> 4).to(tl.float32) - z0[None, :]) * s0[None, :]
        acc2 += xlo[:, None] * wlo + xhi[:, None] * whi
        wq0 = wq
    return tl.sum(acc2, 0)


@triton.jit
def _dq64(wq, sb, zb, gN, n0, NC: tl.constexpr):
    s0 = tl.load(sb + n0 + tl.arange(0, NC), cache_modifier=".cg").to(tl.float32)
    z0 = tl.load(zb + n0 + tl.arange(0, NC), cache_modifier=".cg").to(tl.float32)
    wlo = ((wq & 0xF).to(tl.float32) - z0[None, :]) * s0[None, :]
    whi = ((wq >> 4).to(tl.float32) - z0[None, :]) * s0[None, :]
    return wlo, whi


@triton.jit
def _gemv_kda_qkvg(pid, P, LB, x_ptr, IS_BF16: tl.constexpr, G: tl.constexpr):
    qkv = _p_bf16(P, GP_QKV)
    an = _p_bf16(P, LB + 17)
    for it in range(pid, 138, G):
        if it < 128:
            m = it // 32
            n0 = (it % 32) * 128
            wb = _p_u8(P, LB + m)
            sb = _p_bf16(P, LB + 4 + m)
            zb = _p_bf16(P, LB + 8 + m)
            _prep_xn(P, pid, x_ptr, an, IS_BF16, True, HDIM // 2)
            acc = _gemv_core(P, pid, wb, sb, zb, CQV, n0, 0, HDIM // 2, 128)
            tl.store(qkv + m * CQV + n0 + tl.arange(0, 128), acc.to(tl.bfloat16), cache_modifier=".cg")
        elif it == 128:
            bw = _p_bf16(P, LB + 12)
            beta = _p_bf16(P, GP_BETA)
            _prep_xn(P, pid, x_ptr, an, IS_BF16, True, HDIM // 2)
            xns = _p_bf16(P, GP_XNS) + pid * 4096
            accb = tl.zeros((32,), tl.float32)
            for r in range(0, HDIM // 2, 128):
                ilo = 2 * (r + tl.arange(0, 128))
                lo = tl.load(xns + r + tl.arange(0, 128)).to(tl.float32)
                hi = tl.load(xns + 2048 + r + tl.arange(0, 128)).to(tl.float32)
                wlo = tl.load(bw + tl.arange(0, 32)[:, None] * HDIM + ilo[None, :], cache_modifier=".cg").to(tl.float32)
                whi = tl.load(bw + tl.arange(0, 32)[:, None] * HDIM + ilo[None, :] + 1, cache_modifier=".cg").to(tl.float32)
                accb += tl.sum(wlo * lo[None, :], 1) + tl.sum(whi * hi[None, :], 1)
            tl.store(beta + tl.arange(0, 32), accb.to(tl.bfloat16), cache_modifier=".cg")
        else:
            oaccp = _p_f32(P, GP_OACC)
            off = (it - 129) * 256 + tl.arange(0, 256)
            tl.store(oaccp + off, tl.zeros((256,), tl.float32))


@triton.jit
def _conv4(bufp, win_in, win_out, cw, cidx):
    p0 = tl.load(win_in + cidx, cache_modifier=".cg").to(tl.float32)
    p1 = tl.load(win_in + 4096 + cidx, cache_modifier=".cg").to(tl.float32)
    p2 = tl.load(win_in + 8192 + cidx, cache_modifier=".cg").to(tl.float32)
    val = tl.load(bufp + cidx, cache_modifier=".cg").to(tl.float32)
    w0 = tl.load(cw + cidx * 4 + 0).to(tl.float32)
    w1 = tl.load(cw + cidx * 4 + 1).to(tl.float32)
    w2 = tl.load(cw + cidx * 4 + 2).to(tl.float32)
    w3 = tl.load(cw + cidx * 4 + 3).to(tl.float32)
    o = p0 * w0 + p1 * w1 + p2 * w2 + val * w3
    o = o * tl.sigmoid(o)
    tl.store(win_out + cidx, p1.to(tl.bfloat16))
    tl.store(win_out + 4096 + cidx, p2.to(tl.bfloat16))
    tl.store(win_out + 8192 + cidx, val.to(tl.bfloat16))
    return o


@triton.jit
def _gemv_kda_delta(pid, P, LB, sin_p, cq_in, ck_in, cv_in, LI: tl.constexpr, G: tl.constexpr):
    qkv = _p_bf16(P, GP_QKV)
    beta_b = _p_bf16(P, GP_BETA)
    convw = _p_bf16(P, LB + 13)
    obuf = _p_bf16(P, GP_OBUF)
    s_out = _p_f32(P, GP_S0 + LI)
    cq_out = _p_bf16(P, GP_CQ0 + 3 * LI)
    ck_out = _p_bf16(P, GP_CK0 + 3 * LI)
    cv_out = _p_bf16(P, GP_CV0 + 3 * LI)
    for it in range(pid, 64, G):
        h = it // 2
        dvs = it % 2
        dv0 = dvs * 64
        c128 = h * 128 + tl.arange(0, 128)
        c64 = h * 128 + dv0 + tl.arange(0, 64)
        qf = _conv4(qkv, cq_in, cq_out, convw, c128) * 0.08838834764831845
        kf = _conv4(qkv + CQV, ck_in, ck_out, convw + 16384, c128)
        vf = _conv4(qkv + 2 * CQV, cv_in, cv_out, convw + 32768, c64)
        gf = tl.load(qkv + 3 * CQV + c128, cache_modifier=".cg").to(tl.float32)
        eg = tl.exp(-tl.where(gf > 20.0, gf, tl.log(1.0 + tl.exp(gf))))
        bt = tl.sigmoid(tl.load(beta_b + h, cache_modifier=".cg").to(tl.float32))
        soff = h * 16384 + tl.arange(0, 128)[:, None] * 128 + dv0 + tl.arange(0, 64)[None, :]
        S = tl.load(sin_p + soff, cache_modifier=".cg")
        Sd = S * eg[:, None]
        pred = tl.sum(Sd * kf[:, None], 0)
        S2 = Sd + bt * kf[:, None] * (vf - pred)[None, :]
        tl.store(s_out + soff, S2, cache_modifier=".cg")
        o = tl.sum(S2 * qf[:, None], 0)
        tl.store(obuf + c64, o.to(tl.bfloat16), cache_modifier=".cg")


@triton.jit
def _gemv_oproj(pid, P, LBw, G: tl.constexpr):
    oaccp = _p_f32(P, GP_OACC)
    obuf = _p_bf16(P, GP_OBUF)
    wb = _p_u8(P, LBw)
    sb = _p_bf16(P, LBw + 1)
    zb = _p_bf16(P, LBw + 2)
    for it in range(pid, 72, G):
        nc = it % 18
        ks = it // 18
        n0 = nc * 128
        r0 = ks * 512
        _prep_xn(P, pid, obuf, obuf, True, False, CQV // 2)
        acc = _gemv_core(P, pid, wb, sb, zb, HDIM, n0, r0, 512, 128)
        tl.atomic_add(oaccp + n0 + tl.arange(0, 128), acc)


@triton.jit
def _phase_router(pid, P, LBn, x_ptr, IS_BF16: tl.constexpr, G: tl.constexpr):
    if pid == 0:
        hb = _p_bf16(P, GP_H)
        accp = _p_f32(P, GP_ACC)
        oaccp = _p_f32(P, GP_OACC)
        mn = _p_bf16(P, LBn)
        rt = _p_bf16(P, LBn + 1)
        offs = tl.arange(0, 256)
        for r in range(0, HDIM, 256):
            o = tl.load(oaccp + r + offs, cache_modifier=".cg")
            if IS_BF16:
                xv = tl.load(x_ptr + r + offs, cache_modifier=".cg").to(tl.float32)
            else:
                xv = tl.load(x_ptr + r + offs, cache_modifier=".cg")
            h = (xv + o).to(tl.bfloat16)
            tl.store(hb + r + offs, h, cache_modifier=".cg")
            tl.store(accp + r + offs, h.to(tl.float32), cache_modifier=".cg")
        big = tl.arange(0, 4096)
        msk = big < HDIM
        hf = tl.load(hb + big, mask=msk, other=0.0, cache_modifier=".cg").to(tl.float32)
        rr = tl.rsqrt(tl.sum(hf * hf) / HDIM + EPSJ)
        accl = tl.zeros((64,), tl.float32)
        for r in range(0, HDIM, 256):
            idx = r + offs
            xv = tl.load(hb + idx, cache_modifier=".cg").to(tl.float32)
            wn = tl.load(mn + idx, cache_modifier=".cg").to(tl.float32)
            xn = (xv * rr * wn).to(tl.bfloat16).to(tl.float32)
            wt = tl.load(rt + tl.arange(0, 64)[:, None] * HDIM + idx[None, :], cache_modifier=".cg").to(tl.float32)
            accl += tl.sum(wt * xn[None, :], 1)
        lg = accl.to(tl.bfloat16).to(tl.float32)
        mx = tl.max(lg, 0)
        ex = tl.exp(lg - mx)
        probs = ex / tl.sum(ex, 0)
        isel = _p_i32(P, GP_ISEL)
        wsel = _p_f32(P, GP_WSEL)
        wtot = 0.0
        ar64 = tl.arange(0, 64)
        p = probs
        for j in tl.static_range(8):
            w = tl.max(p, 0)
            m = tl.argmax(p, 0)
            tl.store(isel + j, m.to(tl.int32))
            tl.store(wsel + j, w)
            wtot += w
            p = tl.where(ar64 == m, -float("inf"), p)
        tl.store(wsel + 8, ROUTE_SCALE / (wtot + 1e-9))


@triton.jit
def _phase_gateup(pid, P, EB, LBn, G: tl.constexpr):
    hb = _p_bf16(P, GP_H)
    mn = _p_bf16(P, LBn)
    inter = _p_bf16(P, GP_INTERM)
    isel = _p_i32(P, GP_ISEL)
    gw = _p_u8(P, EB)
    gs = _p_bf16(P, EB + 1)
    gz = _p_bf16(P, EB + 2)
    uw = _p_u8(P, EB + 3)
    us = _p_bf16(P, EB + 4)
    uz = _p_bf16(P, EB + 5)
    sgw = _p_u8(P, EB + 9)
    sgs = _p_bf16(P, EB + 10)
    sgz = _p_bf16(P, EB + 11)
    suw = _p_u8(P, EB + 12)
    sus = _p_bf16(P, EB + 13)
    suz = _p_bf16(P, EB + 14)
    for it in range(pid, 144, G):
        slot = it // 16
        n0 = (it % 16) * 64
        if slot < 8:
            e = tl.load(isel + slot, cache_modifier=".cg").to(tl.int64)
            gwb = gw + e * 1152 * 1024
            gsb = gs + e * 18432
            gzb = gz + e * 18432
            uwb = uw + e * 1152 * 1024
            usb = us + e * 18432
            uzb = uz + e * 18432
        else:
            gwb = sgw
            gsb = sgs
            gzb = sgz
            uwb = suw
            usb = sus
            uzb = suz
        _prep_xn(P, pid, hb, mn, True, True, HDIM // 2)
        gac = _gemv_core(P, pid, gwb, gsb, gzb, 1024, n0, 0, HDIM // 2, 64)
        uac = _gemv_core(P, pid, uwb, usb, uzb, 1024, n0, 0, HDIM // 2, 64)
        hv = (gac * tl.sigmoid(gac)) * uac
        tl.store(inter + slot * 1024 + n0 + tl.arange(0, 64), hv.to(tl.bfloat16), cache_modifier=".cg")


@triton.jit
def _phase_down(pid, P, EB, G: tl.constexpr):
    inter = _p_bf16(P, GP_INTERM)
    accp = _p_f32(P, GP_ACC)
    isel = _p_i32(P, GP_ISEL)
    wsel = _p_f32(P, GP_WSEL)
    dw = _p_u8(P, EB + 6)
    ds = _p_bf16(P, EB + 7)
    dz = _p_bf16(P, EB + 8)
    sdw = _p_u8(P, EB + 15)
    sds = _p_bf16(P, EB + 16)
    sdz = _p_bf16(P, EB + 17)
    for it in range(pid, 162, G):
        slot = it // 18
        n0 = (it % 18) * 128
        if slot < 8:
            e = tl.load(isel + slot, cache_modifier=".cg").to(tl.int64)
            wgt = tl.load(wsel + slot, cache_modifier=".cg") * tl.load(wsel + 8, cache_modifier=".cg")
            dwb = dw + e * 512 * HDIM
            dsb = ds + e * 18432
            dzb = dz + e * 18432
        else:
            wgt = 1.0
            dwb = sdw
            dsb = sds
            dzb = sdz
        _prep_xn(P, pid, inter + slot * 1024, inter, True, False, 512)
        acc = _gemv_core(P, pid, dwb, dsb, dzb, HDIM, n0, 0, 512, 128)
        tl.atomic_add(accp + n0 + tl.arange(0, 128), acc * wgt)


@triton.jit
def _phase_mla_qkv(pid, P, LB, L, MODE, ckvi, kroi, G: tl.constexpr):
    qb = _p_bf16(P, GP_QBUF)
    qrb = _p_bf16(P, GP_QR)
    cache = _p_bf16(P, GP_CACHE)
    krc = _p_bf16(P, GP_KRC)
    accp = _p_f32(P, GP_ACC)
    an = _p_bf16(P, LB + 12)
    qw = _p_u8(P, LB + 0)
    qs = _p_bf16(P, LB + 1)
    qz = _p_bf16(P, LB + 2)
    vw = _p_u8(P, LB + 3)
    vs = _p_bf16(P, LB + 4)
    vz = _p_bf16(P, LB + 5)
    oaccp = _p_f32(P, GP_OACC)
    pos = L.to(tl.float32)
    inv = tl.exp2(tl.arange(0, 32).to(tl.float32) * (-LTH32))
    cosf = tl.math.cos(pos * inv)
    sinf = tl.math.sin(pos * inv)
    ncopy = (L + 7) // 8
    for it in range(pid, 114 + ncopy, G):
        if it < 96:
            n0 = it * 64
            _prep_xn(P, pid, accp, an, False, True, HDIM // 2)
            acc = _gemv_core(P, pid, qw, qs, qz, NQ, n0, 0, HDIM // 2, 64)
            if it % 3 == 2:
                e, o = tl.split(tl.reshape(acc, (32, 2)))
                re_ = e * cosf - o * sinf
                ro_ = o * cosf + e * sinf
                outv = tl.reshape(tl.join(re_, ro_), (64,))
                tl.store(qrb + (it // 3) * 64 + tl.arange(0, 64), outv.to(tl.bfloat16), cache_modifier=".cg")
            else:
                tl.store(qb + n0 + tl.arange(0, 64), acc.to(tl.bfloat16), cache_modifier=".cg")
        elif it < 105:
            c = it - 96
            n0 = c * 64
            _prep_xn(P, pid, accp, an, False, True, HDIM // 2)
            acc = _gemv_core(P, pid, vw, vs, vz, 576, n0, 0, HDIM // 2, 64)
            if c == 8:
                e, o = tl.split(tl.reshape(acc, (32, 2)))
                re_ = e * cosf - o * sinf
                ro_ = o * cosf + e * sinf
                outv = tl.reshape(tl.join(re_, ro_), (64,))
                tl.store(krc + L * 64 + tl.arange(0, 64), outv.to(tl.bfloat16), cache_modifier=".cg")
            else:
                tl.store(cache + L * 512 + n0 + tl.arange(0, 64), acc.to(tl.bfloat16), cache_modifier=".cg")
        elif it < 114:
            off = (it - 105) * 256 + tl.arange(0, 256)
            tl.store(oaccp + off, tl.zeros((256,), tl.float32))
        else:
            if MODE == 1:
                cc = it - 114
                rows = cc * 8 + tl.arange(0, 8)
                m8r = rows < L
                ridx = tl.where(m8r, rows, 0)
                v = tl.load(ckvi + ridx[:, None] * 512 + tl.arange(0, 512)[None, :], mask=m8r[:, None], cache_modifier=".cg")
                tl.store(cache + ridx[:, None] * 512 + tl.arange(0, 512)[None, :], v, mask=m8r[:, None])
                v2 = tl.load(kroi + ridx[:, None] * 64 + tl.arange(0, 64)[None, :], mask=m8r[:, None], cache_modifier=".cg")
                tl.store(krc + ridx[:, None] * 64 + tl.arange(0, 64)[None, :], v2, mask=m8r[:, None])


@triton.jit
def _phase_mla_qlat(pid, P, LB, G: tl.constexpr):
    qb = _p_bf16(P, GP_QBUF)
    qlat = _p_bf16(P, GP_QLAT)
    wb = _p_u8(P, LB + 6)
    sb = _p_bf16(P, LB + 7)
    zb = _p_bf16(P, LB + 8)
    for it in range(pid, 64, G):
        h = it // 2
        kh = it % 2
        c0 = kh * 256
        x = tl.load(qb + h * 192 + tl.arange(0, 128), cache_modifier=".cg").to(tl.float32)
        for r in range(0, 128, 64):
            rows = kh * 128 + r + tl.arange(0, 64)
            wq = tl.load(wb + rows[:, None] * 8192 + h * 256 + tl.arange(0, 128)[None, :], cache_modifier=".cg")
            g = (c0 + 2 * r) // 128
            wlo, whi = _dq64(wq, sb + g * 8192, zb + g * 8192, 0, h * 256, 128)
            ye = tl.sum(x[None, :] * wlo, 1)
            yo = tl.sum(x[None, :] * whi, 1)
            y = tl.reshape(tl.join(ye, yo), (128,))
            tl.store(qlat + h * 512 + c0 + 2 * r + tl.arange(0, 128), y.to(tl.bfloat16), cache_modifier=".cg")


@triton.jit
def _phase_mla_attn(pid, P, L, CHL, G: tl.constexpr):
    cache = _p_bf16(P, GP_CACHE)
    krc = _p_bf16(P, GP_KRC)
    qlat = _p_bf16(P, GP_QLAT)
    qrb = _p_bf16(P, GP_QR)
    ws = _p_f32(P, GP_WS)
    Lp = L + 1
    scale = 0.07216878364870323
    for it in range(pid, NSLOT, G):
        hg = it % 2
        ch = it // 2
        h0 = hg * 16
        l0 = ch * CHL
        l1 = tl.minimum(l0 + CHL, Lp)
        jr = tl.arange(0, 16)
        mrun = tl.full((16,), -float("inf"), tl.float32)
        lrun = tl.zeros((16,), tl.float32)
        oac0 = tl.zeros((16, 128), tl.float32)
        oac1 = tl.zeros((16, 128), tl.float32)
        oac2 = tl.zeros((16, 128), tl.float32)
        oac3 = tl.zeros((16, 128), tl.float32)
        for lt in range(l0, l1, 64):
            lr = lt + tl.arange(0, 64)
            val = lr < l1
            acc = tl.zeros((64, 16), tl.float32)
            for kk in range(0, KV, 64):
                ct = tl.load(cache + lr[:, None] * 512 + kk + tl.arange(0, 64)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
                bq = tl.load(qlat + (kk + tl.arange(0, 64))[:, None] + (h0 + jr)[None, :] * 512, cache_modifier=".cg")
                acc += tl.dot(ct, bq)
            krt = tl.load(krc + lr[:, None] * 64 + tl.arange(0, 64)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
            bqr = tl.load(qrb + tl.arange(0, 64)[:, None] + (h0 + jr)[None, :] * 64, cache_modifier=".cg")
            acc += tl.dot(krt, bqr)
            acc = acc * scale
            acc = tl.where(val[:, None], acc, -float("inf"))
            mnew = tl.maximum(mrun, tl.max(acc, 0))
            alpha = tl.where(mnew == -float("inf"), 1.0, tl.exp(mrun - mnew))
            p = tl.exp(acc - mnew[None, :])
            lrun = lrun * alpha + tl.sum(p, 0)
            pb = tl.trans(p.to(tl.bfloat16))
            c0t = tl.load(cache + lr[:, None] * 512 + tl.arange(0, 128)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
            oac0 = oac0 * alpha[:, None] + tl.dot(pb, c0t)
            c1t = tl.load(cache + lr[:, None] * 512 + 128 + tl.arange(0, 128)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
            oac1 = oac1 * alpha[:, None] + tl.dot(pb, c1t)
            c2t = tl.load(cache + lr[:, None] * 512 + 256 + tl.arange(0, 128)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
            oac2 = oac2 * alpha[:, None] + tl.dot(pb, c2t)
            c3t = tl.load(cache + lr[:, None] * 512 + 384 + tl.arange(0, 128)[None, :], mask=val[:, None], other=0.0, cache_modifier=".cg")
            oac3 = oac3 * alpha[:, None] + tl.dot(pb, c3t)
            mrun = mnew
        base = ws + it * (16 * 514)
        tl.store(base + jr[:, None] * 514 + tl.arange(0, 128)[None, :], oac0)
        tl.store(base + jr[:, None] * 514 + 128 + tl.arange(0, 128)[None, :], oac1)
        tl.store(base + jr[:, None] * 514 + 256 + tl.arange(0, 128)[None, :], oac2)
        tl.store(base + jr[:, None] * 514 + 384 + tl.arange(0, 128)[None, :], oac3)
        tl.store(base + jr * 514 + 512, mrun)
        tl.store(base + jr * 514 + 513, lrun)


@triton.jit
def _phase_mla_ohead(pid, P, LB, G: tl.constexpr):
    ws = _p_f32(P, GP_WS)
    olat = _p_bf16(P, GP_OLAT)
    obuf = _p_bf16(P, GP_OBUF)
    wb = _p_u8(P, LB + 6)
    sb = _p_bf16(P, LB + 7)
    zb = _p_bf16(P, LB + 8)
    for it in range(pid, 64, G):
        h = it // 2
        sub = it % 2
        mv = tl.load(ws + tl.arange(0, NSLOT) * (16 * 514) + h * 514 + 512, cache_modifier=".cg")
        lv = tl.load(ws + tl.arange(0, NSLOT) * (16 * 514) + h * 514 + 513, cache_modifier=".cg")
        mm = tl.max(mv, 0)
        lsum = tl.sum(tl.exp(mv - mm) * lv, 0)
        acc = tl.zeros((512,), tl.float32)
        for s0 in range(0, NSLOT, 8):
            sidx = s0 + tl.arange(0, 8)
            m8 = tl.load(ws + sidx * (16 * 514) + h * 514 + 512, cache_modifier=".cg")
            e8 = tl.exp(m8 - mm)
            tile = tl.load(ws + sidx[:, None] * (16 * 514) + h * 514 + tl.arange(0, 512)[None, :], cache_modifier=".cg")
            acc += tl.sum(tile * e8[:, None], 0)
        ol = (acc / lsum).to(tl.bfloat16)
        tl.store(olat + h * 512 + tl.arange(0, 512), ol, cache_modifier=".cg")
        n0 = h * 256 + 128 + sub * 64
        _prep_xn(P, pid, olat + h * 512, olat, True, False, 256)
        o = _gemv_core(P, pid, wb, sb, zb, 8192, n0, 0, 256, 64)
        tl.store(obuf + h * 128 + sub * 64 + tl.arange(0, 64), o.to(tl.bfloat16), cache_modifier=".cg")


@triton.jit(do_not_specialize=['L', 'CHL', 'MODE', 'DBG'])
def _mega(x0_ptr, out_ptr, P, L, CHL, MODE, DBG,
          s0i, s1i, s2i, cq0i, ck0i, cv0i, cq1i, ck1i, cv1i, cq2i, ck2i, cv2i,
          ckvi, kroi,
          G: tl.constexpr):
    pid = tl.program_id(0)
    bar = _p_i32(P, GP_BAR)
    t = 0
    for li in tl.static_range(0, 3):
        LB = li * LSJ
        if li == 0:
            _gemv_kda_qkvg(pid, P, LB, x0_ptr, True, G)
        else:
            _gemv_kda_qkvg(pid, P, LB, _p_f32(P, GP_ACC), False, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
        if li == 0:
            _gemv_kda_delta(pid, P, LB, s0i, cq0i, ck0i, cv0i, 0, G)
        elif li == 1:
            _gemv_kda_delta(pid, P, LB, s1i, cq1i, ck1i, cv1i, 1, G)
        else:
            _gemv_kda_delta(pid, P, LB, s2i, cq2i, ck2i, cv2i, 2, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
        _gemv_oproj(pid, P, LB + 14, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
        if li == 0:
            _phase_router(pid, P, LB + 18, x0_ptr, True, G)
        else:
            _phase_router(pid, P, LB + 18, _p_f32(P, GP_ACC), False, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
        _phase_gateup(pid, P, LB + 20, LB + 18, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
        _phase_down(pid, P, LB + 20, G)
        t += 1
        _gbar(bar, t, G)
        if DBG == -2 and pid == 0:
            _prof_tick(P, t)
        if DBG == t:
            return
    LBM = 3 * LSJ
    _phase_mla_qkv(pid, P, LBM, L, MODE, ckvi, kroi, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_mla_qlat(pid, P, LBM, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_mla_attn(pid, P, L, CHL, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_mla_ohead(pid, P, LBM, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _gemv_oproj(pid, P, LBM + 9, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_router(pid, P, LBM + 13, _p_f32(P, GP_ACC), False, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_gateup(pid, P, LBM + 15, LBM + 13, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    _phase_down(pid, P, LBM + 15, G)
    t += 1
    _gbar(bar, t, G)
    if DBG == -2 and pid == 0:
        _prof_tick(P, t)
    if DBG == t:
        return
    accp = _p_f32(P, GP_ACC)
    for off in range(pid * 256, HDIM, G * 256):
        v = tl.load(accp + off + tl.arange(0, 256), cache_modifier=".cg")
        tl.store(out_ptr + off + tl.arange(0, 256), v.to(tl.bfloat16))
    if pid == 0:
        tl.store(bar + tl.arange(0, 32), tl.zeros((32,), tl.int32))

from dataclasses import dataclass, field


@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: torch.dtype = field(default=torch.bfloat16)


def build_config(shape: dict) -> Config:
    return Config(n_experts=int(shape.get("n_experts", 64)))


class QuantLinear(nn.Module):
    def __init__(self, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))


class QuantExperts(nn.Module):
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))


class KDA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        self.q_proj = QuantLinear(d, H * Dk, cfg.group)
        self.k_proj = QuantLinear(d, H * Dk, cfg.group)
        self.v_proj = QuantLinear(d, H * Dk, cfg.group)
        self.g_proj = QuantLinear(d, H * Dk, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, H * Dk, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(H * Dk, d, cfg.group)
        self.scale = Dk ** -0.5


class MLA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5


class MoE(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(E, d, m, cfg.group)
        self.up = QuantExperts(E, d, m, cfg.group)
        self.down = QuantExperts(E, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)


class Block(nn.Module):
    def __init__(self, cfg: Config, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)


class Model(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)
        self._ready = False

    def _init_ws(self, device):
        dev = device
        ptrs = []
        for i, blk in enumerate(self.blocks):
            seg = []
            if blk.kind == "K":
                a = blk.attn
                seg += [a.q_proj.w_q, a.k_proj.w_q, a.v_proj.w_q, a.g_proj.w_q,
                        a.q_proj.scales, a.k_proj.scales, a.v_proj.scales, a.g_proj.scales,
                        a.q_proj.zeros, a.k_proj.zeros, a.v_proj.zeros, a.g_proj.zeros,
                        a.beta_proj.weight, a.conv_w,
                        a.o_proj.w_q, a.o_proj.scales, a.o_proj.zeros,
                        blk.attn_norm, blk.moe_norm, blk.moe.router.weight]
            else:
                a = blk.attn
                seg += [a.q_proj.w_q, a.q_proj.scales, a.q_proj.zeros,
                        a.kv_a.w_q, a.kv_a.scales, a.kv_a.zeros,
                        a.kv_b.w_q, a.kv_b.scales, a.kv_b.zeros,
                        a.o_proj.w_q, a.o_proj.scales, a.o_proj.zeros,
                        blk.attn_norm, blk.moe_norm, blk.moe.router.weight]
            m = blk.moe
            seg += [m.gate.w_q, m.gate.scales, m.gate.zeros,
                    m.up.w_q, m.up.scales, m.up.zeros,
                    m.down.w_q, m.down.scales, m.down.zeros,
                    m.s_gate.w_q, m.s_gate.scales, m.s_gate.zeros,
                    m.s_up.w_q, m.s_up.scales, m.s_up.zeros,
                    m.s_down.w_q, m.s_down.scales, m.s_down.zeros]
            assert len(seg) <= LS, (i, len(seg))
            seg += [seg[0]] * (LS - len(seg))
            ptrs += seg
        self._acc = torch.zeros(2304, dtype=torch.float32, device=dev)
        self._h = torch.zeros(2304, dtype=torch.bfloat16, device=dev)
        self._qkv = torch.zeros(4 * 4096, dtype=torch.bfloat16, device=dev)
        self._beta = torch.zeros(32, dtype=torch.bfloat16, device=dev)
        self._oacc = torch.zeros(2304, dtype=torch.float32, device=dev)
        self._isel = torch.zeros(8, dtype=torch.int32, device=dev)
        self._wsel = torch.zeros(9, dtype=torch.float32, device=dev)
        self._interm = torch.zeros(9 * 1024, dtype=torch.bfloat16, device=dev)
        self._qbuf = torch.zeros(6144, dtype=torch.bfloat16, device=dev)
        self._qlat = torch.zeros(32 * 512, dtype=torch.bfloat16, device=dev)
        self._qr = torch.zeros(32 * 64, dtype=torch.bfloat16, device=dev)
        self._ws = torch.zeros(128 * 16 * 514, dtype=torch.float32, device=dev)
        self._bar = torch.zeros(40, dtype=torch.int32, device=dev)
        cap = 16384 + 512
        self._cap = cap
        self._cache = torch.zeros(cap * 512, dtype=torch.bfloat16, device=dev)
        self._krc = torch.zeros(cap * 64, dtype=torch.bfloat16, device=dev)
        self._S = [torch.zeros(32 * 128 * 128, dtype=torch.float32, device=dev) for _ in range(3)]
        self._cw = [[torch.zeros(3 * 4096, dtype=torch.bfloat16, device=dev) for _ in range(3)] for _ in range(3)]
        self._olat = torch.zeros(32 * 512, dtype=torch.bfloat16, device=dev)
        self._obuf = torch.zeros(4096, dtype=torch.bfloat16, device=dev)
        self._dbg = torch.zeros(4096, dtype=torch.float32, device=dev)
        self._xns = torch.zeros(GRID * 4096, dtype=torch.bfloat16, device=dev)
        self._xns = torch.zeros(GRID * 4096, dtype=torch.bfloat16, device=dev)
        ws = [self._acc, self._h, self._qkv, self._beta, self._oacc, self._isel,
              self._wsel, self._interm, self._qbuf, self._qlat, self._qr, self._ws,
              self._bar, self._cache, self._krc,
              self._S[0], self._S[1], self._S[2],
              self._cw[0][0], self._cw[0][1], self._cw[0][2],
              self._cw[1][0], self._cw[1][1], self._cw[1][2],
              self._cw[2][0], self._cw[2][1], self._cw[2][2],
              self._olat, self._obuf, self._dbg, self._xns]
        assert len(ptrs) == GB and len(ws) == 31
        allt = ptrs + ws
        table = torch.tensor([t.data_ptr() for t in allt] + [0] * (NPAR - len(allt)), dtype=torch.int64)
        self._table_hold = allt
        self._params = table.to(dev)
        self._pool = [torch.zeros(2304, dtype=torch.bfloat16, device=dev),
                      torch.zeros(2304, dtype=torch.bfloat16, device=dev)]
        self._parity = 0
        self._state = [
            {"S": self._S[0].view(32, 128, 128),
             "cq": self._cw[0][0].view(3, 4096), "ck": self._cw[0][1].view(3, 4096), "cv": self._cw[0][2].view(3, 4096)},
            {"S": self._S[1].view(32, 128, 128),
             "cq": self._cw[1][0].view(3, 4096), "ck": self._cw[1][1].view(3, 4096), "cv": self._cw[1][2].view(3, 4096)},
            {"S": self._S[2].view(32, 128, 128),
             "cq": self._cw[2][0].view(3, 4096), "ck": self._cw[2][1].view(3, 4096), "cv": self._cw[2][2].view(3, 4096)},
            {"c_kv": None, "k_rope": None},
        ]
        # compile the kernel once with the real argument signature
        out0 = self._pool[0]
        k = _mega[(GRID,)](
            out0, out0, self._params, 8, 1, 0, -1,
            self._S[0], self._S[1], self._S[2],
            self._cw[0][0], self._cw[0][1], self._cw[0][2],
            self._cw[1][0], self._cw[1][1], self._cw[1][2],
            self._cw[2][0], self._cw[2][1], self._cw[2][2],
            self._cache, self._krc,
            GRID, num_warps=8, launch_cooperative_grid=True)
        torch.cuda.synchronize()
        self._kernel = k
        self._ready = True

    def step(self, hidden, state):
        if not self._ready:
            self._init_ws(hidden.device)
        ckv = state[3]["c_kv"]
        L = ckv.shape[0]
        handoff = (ckv.data_ptr() != self._cache.data_ptr())
        if handoff and L + 256 > self._cap:
            newcap = L + 512
            self._cache = torch.zeros(newcap * 512, dtype=torch.bfloat16, device=hidden.device)
            self._krc = torch.zeros(newcap * 64, dtype=torch.bfloat16, device=hidden.device)
            self._cap = newcap
            hold = self._table_hold
            hold[GB + 13] = self._cache
            hold[GB + 14] = self._krc
            table = torch.tensor([t.data_ptr() for t in hold] + [0] * (NPAR - len(hold)), dtype=torch.int64)
            self._params = table.to(hidden.device)
        out = self._pool[self._parity]
        self._parity ^= 1
        if handoff:
            sargs = (state[0]["S"], state[1]["S"], state[2]["S"],
                     state[0]["cq"], state[0]["ck"], state[0]["cv"],
                     state[1]["cq"], state[1]["ck"], state[1]["cv"],
                     state[2]["cq"], state[2]["ck"], state[2]["cv"],
                     ckv, state[3]["k_rope"])
            mode = 1
        else:
            sargs = (self._S[0], self._S[1], self._S[2],
                     self._cw[0][0], self._cw[0][1], self._cw[0][2],
                     self._cw[1][0], self._cw[1][1], self._cw[1][2],
                     self._cw[2][0], self._cw[2][1], self._cw[2][2],
                     self._cache, self._krc)
            mode = 0
        chl = (L + 1 + 63) // 64
        dbg = int(os.environ.get('MEGA_DBG', '-1'))
        if dbg != -1:
            self._bar.zero_()
        k = self._kernel
        lau = k.run
        lau.launch(GRID, 1, 1, torch.cuda.current_stream().cuda_stream, k.function,
                   lau.launch_cooperative_grid, lau.launch_pdl, None, None,
                   k.packed_metadata, None, None, None,
                   hidden, out, self._params, L, chl, mode, dbg,
                   sargs[0], sargs[1], sargs[2], sargs[3], sargs[4], sargs[5], sargs[6],
                   sargs[7], sargs[8], sargs[9], sargs[10], sargs[11], sargs[12], sargs[13],
                   GRID)
        self._state[3]["c_kv"] = self._cache[: (L + 1) * 512].view(L + 1, 512)
        self._state[3]["k_rope"] = self._krc[: (L + 1) * 64].view(L + 1, 64)
        return out, self._state
