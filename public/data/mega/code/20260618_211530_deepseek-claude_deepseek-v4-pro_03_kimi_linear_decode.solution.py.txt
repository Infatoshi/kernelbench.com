"""Fused int4-dequant GEMV solution for Kimi-Linear W4A16 hybrid decode.

Key optimisations:
1. Fused int4 dequant-GEMV Triton kernel — never materialises full bf16 weights
2. MLA KV absorption — cache projected k_nope/v, skip O(ctx) kv_b recomputation
"""
from __future__ import annotations

from dataclasses import dataclass, field

import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl

# ---------------------------------------------------------------------------
# Inlined utilities (Config, state, helpers) — built-in, not imported
# ---------------------------------------------------------------------------

OP_TYPE = "kimi_linear_w4a16_decode"
HARDWARE_REQUIRED = ["RTX_PRO_6000"]
EPS = 1.0e-6
GROUP_SIZE = 128


@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: torch.dtype = field(default=torch.bfloat16)


def build_config(shape: dict) -> Config:
    return Config(n_experts=int(shape.get("n_experts", 64)))


def _pack_int4(w_q: torch.Tensor) -> torch.Tensor:
    lo = w_q[0::2] & 0xF
    hi = w_q[1::2] & 0xF
    return (lo | (hi << 4)).contiguous()


def _unpack_int4(w_packed: torch.Tensor, K: int) -> torch.Tensor:
    out = torch.empty((K, w_packed.shape[1]), dtype=torch.uint8, device=w_packed.device)
    out[0::2] = w_packed & 0xF
    out[1::2] = (w_packed >> 4) & 0xF
    return out


def _quantize(w_io: torch.Tensor, group: int = GROUP_SIZE):
    K, N = w_io.shape
    ng = K // group
    wg = w_io.view(ng, group, N).float()
    wmin = wg.min(dim=1, keepdim=True).values
    wmax = wg.max(dim=1, keepdim=True).values
    scales = (wmax - wmin).clamp_min(1e-8) / 15.0
    zeros = (-wmin / scales).round().clamp(0, 15)
    w_q = ((wg / scales) + zeros).round().clamp(0, 15).to(torch.uint8).view(K, N)
    return _pack_int4(w_q), scales.squeeze(1).to(torch.bfloat16), zeros.squeeze(1).to(torch.bfloat16)


def _rmsnorm(x: torch.Tensor, w: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    xf = xf * torch.rsqrt(xf.pow(2).mean(-1, keepdim=True) + EPS)
    return (xf * w.float()).to(x.dtype)


def _rope_cossin(pos: int, dim: int, theta: float, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos * inv
    return torch.cos(ang), torch.sin(ang)


def _apply_rope(x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    even, odd = xf[..., 0::2], xf[..., 1::2]
    out = torch.empty_like(xf)
    out[..., 0::2] = even * cos - odd * sin
    out[..., 1::2] = odd * cos + even * sin
    return out.to(x.dtype)


def init_state(cfg: Config, context_len: int, seed: int) -> list:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed)
    H, Dk = cfg.kda_heads, cfg.kda_head_dim
    C = H * Dk
    state = []
    for kind in cfg.pattern:
        if kind == "K":
            state.append({
                "S": torch.randn(H, Dk, Dk, device=dev, generator=g) * 0.05,
                "cq": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "ck": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "cv": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
            })
        else:
            state.append({
                "c_kv": torch.randn(context_len, cfg.kv_lora, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "k_rope": torch.randn(context_len, cfg.qk_rope, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
            })
    return state


def init_token(cfg: Config, seed: int) -> torch.Tensor:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed + 1)
    return torch.randn(cfg.hidden, device=dev, generator=g, dtype=cfg.dtype) * 0.25


# ---------------------------------------------------------------------------
# Base modules (weights/buffers same as reference, needed for state_dict compat)
# ---------------------------------------------------------------------------

class _QuantLinearBase(nn.Module):
    """Quantised linear layer.  Buffers match reference QuantLinear."""
    def __init__(self, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        assert in_f % group == 0 and in_f % 2 == 0
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))

    def init_random(self, gen: torch.Generator, std: float = 0.02) -> None:
        w = torch.randn(self.in_f, self.out_f, generator=gen) * std
        wq, s, z = _quantize(w, self.group)
        self.w_q.copy_(wq)
        self.scales.copy_(s)
        self.zeros.copy_(z)


class _QuantExpertsBase(nn.Module):
    """Expert weight bank.  Buffers match reference QuantExperts."""
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))

    def init_random(self, gen: torch.Generator, std: float = 0.02) -> None:
        for e in range(self.n):
            w = torch.randn(self.in_f, self.out_f, generator=gen) * std
            wq, s, z = _quantize(w, self.group)
            self.w_q[e].copy_(wq)
            self.scales[e].copy_(s)
            self.zeros[e].copy_(z)


# ---------------------------------------------------------------------------
# Triton fused int4 dequant-GEMV kernel
# ---------------------------------------------------------------------------

@triton.jit
def _fused_int4_gemv_kernel(
    w_q_ptr, scales_ptr, zeros_ptr, x_ptr, y_ptr,
    K,
    stride_wq_k, stride_sc_k,
    BLOCK_N: tl.constexpr, GROUP_SIZE: tl.constexpr, N: tl.constexpr,
):
    HALF_GROUP: tl.constexpr = GROUP_SIZE // 2

    pid = tl.program_id(0)
    n_offs = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offs < N

    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    num_groups = K // GROUP_SIZE

    for g in range(num_groups):
        k_start = g * GROUP_SIZE

        sc_offs = g * stride_sc_k + n_offs
        scale = tl.load(scales_ptr + sc_offs, mask=n_mask, other=0.0).to(tl.float32)
        zero = tl.load(zeros_ptr + sc_offs, mask=n_mask, other=0.0).to(tl.float32)

        x_even_offs = k_start + 2 * tl.arange(0, HALF_GROUP)
        x_odd_offs = x_even_offs + 1
        x_even = tl.load(x_ptr + x_even_offs, mask=x_even_offs < K, other=0.0).to(tl.float32)
        x_odd = tl.load(x_ptr + x_odd_offs, mask=x_odd_offs < K, other=0.0).to(tl.float32)

        wq_k_start = k_start // 2
        wq_k_offs = wq_k_start + tl.arange(0, HALF_GROUP)
        wq_row_mask = wq_k_offs < (K // 2)
        wq_2d_offs = wq_k_offs[:, None] * stride_wq_k + n_offs[None, :]
        wq_2d_mask = wq_row_mask[:, None] & n_mask[None, :]
        wq_bytes = tl.load(w_q_ptr + wq_2d_offs, mask=wq_2d_mask, other=0)

        w_even = (wq_bytes & 0xF).to(tl.float32)
        w_odd = ((wq_bytes >> 4) & 0xF).to(tl.float32)

        acc += tl.sum((w_even - zero[None, :]) * scale[None, :] * x_even[:, None], axis=0)
        acc += tl.sum((w_odd - zero[None, :]) * scale[None, :] * x_odd[:, None], axis=0)

    tl.store(y_ptr + n_offs, acc.to(tl.bfloat16), mask=n_mask)


def _pick_blk_n(N: int) -> int:
    """Pick BLOCK_N for best occupancy. Prefer 128 or 256; avoid tiny grids."""
    candidates = [64, 128, 256]
    for c in [128, 256, 64, 512, 32]:
        if c <= N and triton.cdiv(N, c) >= 4:
            return c
    # Fallback: smallest power-of-2 that gives enough blocks
    for c in [512, 256, 128, 64, 32]:
        if c <= N:
            return c
    return 32


def _fused_int4_gemv(
    x: torch.Tensor,        # (K,) bf16
    w_q: torch.Tensor,      # (K//2, N) uint8
    scales: torch.Tensor,   # (K//group, N) bf16
    zeros: torch.Tensor,    # (K//group, N) bf16
    group_size: int = GROUP_SIZE,
) -> torch.Tensor:
    """Fused int4 dequant-GEMV: y = x @ dequant(w_q, scales, zeros)."""
    K = w_q.shape[0] * 2
    N = w_q.shape[1]
    y = torch.empty(N, dtype=torch.bfloat16, device=x.device)

    blk_n = _pick_blk_n(N)
    grid = (triton.cdiv(N, blk_n),)

    _fused_int4_gemv_kernel[grid](
        w_q.contiguous(), scales.contiguous(), zeros.contiguous(),
        x.contiguous(), y,
        K,
        w_q.stride(0), scales.stride(0),
        BLOCK_N=blk_n, GROUP_SIZE=group_size, N=N,
    )
    return y


@triton.jit
def _fused_int4_gemv_multi_kernel(
    w_q_ptr, scales_ptr, zeros_ptr, x_ptr, y_ptr,
    K,
    stride_wq_e, stride_sc_e,
    stride_wq_k, stride_sc_k,
    SHARED_X: tl.constexpr,
    BLOCK_N: tl.constexpr, GROUP_SIZE: tl.constexpr, N: tl.constexpr,
):
    """2-D grid: pid0=expert, pid1=N-tile.

    If SHARED_X, all experts read from the same x_ptr.
    """
    HALF_GROUP: tl.constexpr = GROUP_SIZE // 2

    pid_e = tl.program_id(0)
    pid_n = tl.program_id(1)

    wq_e = w_q_ptr + pid_e * stride_wq_e
    sc_e = scales_ptr + pid_e * stride_sc_e
    z_e = zeros_ptr + pid_e * stride_sc_e
    if SHARED_X:
        x_e = x_ptr
    else:
        x_e = x_ptr + pid_e * K
    y_e = y_ptr + pid_e * N

    n_offs = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offs < N

    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    num_groups = K // GROUP_SIZE

    for g in range(num_groups):
        k_start = g * GROUP_SIZE

        sc_offs = g * stride_sc_k + n_offs
        scale = tl.load(sc_e + sc_offs, mask=n_mask, other=0.0).to(tl.float32)
        zero = tl.load(z_e + sc_offs, mask=n_mask, other=0.0).to(tl.float32)

        x_even_offs = k_start + 2 * tl.arange(0, HALF_GROUP)
        x_odd_offs = x_even_offs + 1
        x_even = tl.load(x_e + x_even_offs, mask=x_even_offs < K, other=0.0).to(tl.float32)
        x_odd = tl.load(x_e + x_odd_offs, mask=x_odd_offs < K, other=0.0).to(tl.float32)

        wq_k_start = k_start // 2
        wq_k_offs = wq_k_start + tl.arange(0, HALF_GROUP)
        wq_row_mask = wq_k_offs < (K // 2)
        wq_2d_offs = wq_k_offs[:, None] * stride_wq_k + n_offs[None, :]
        wq_2d_mask = wq_row_mask[:, None] & n_mask[None, :]
        wq_bytes = tl.load(wq_e + wq_2d_offs, mask=wq_2d_mask, other=0)

        w_even = (wq_bytes & 0xF).to(tl.float32)
        w_odd = ((wq_bytes >> 4) & 0xF).to(tl.float32)

        acc += tl.sum((w_even - zero[None, :]) * scale[None, :] * x_even[:, None], axis=0)
        acc += tl.sum((w_odd - zero[None, :]) * scale[None, :] * x_odd[:, None], axis=0)

    tl.store(y_e + n_offs, acc.to(tl.bfloat16), mask=n_mask)


def _fused_int4_gemv_multi(
    x: torch.Tensor,            # (num_experts, K) bf16 — per-expert inputs
    w_q: torch.Tensor,          # (num_experts, K//2, N) uint8 (pre-selected experts)
    scales: torch.Tensor,       # (num_experts, K//group, N) bf16
    zeros: torch.Tensor,        # (num_experts, K//group, N) bf16
    group_size: int = GROUP_SIZE,
) -> torch.Tensor:
    E = w_q.shape[0]
    K = w_q.shape[1] * 2
    N = w_q.shape[2]
    y = torch.empty(E, N, dtype=torch.bfloat16, device=x.device)

    blk_n = _pick_blk_n(N)
    grid = (E, triton.cdiv(N, blk_n))

    _fused_int4_gemv_multi_kernel[grid](
        w_q.contiguous(), scales.contiguous(), zeros.contiguous(),
        x.contiguous(), y,
        K,
        w_q.stride(0), scales.stride(0),
        w_q.stride(1), scales.stride(1),
        False,
        BLOCK_N=blk_n, GROUP_SIZE=group_size, N=N,
    )
    return y


def _fused_int4_gemv_multi_shared(
    x: torch.Tensor,            # (K,) bf16 — shared input
    w_q: torch.Tensor,          # (num_experts, K//2, N) uint8 (pre-selected experts)
    scales: torch.Tensor,       # (num_experts, K//group, N) bf16
    zeros: torch.Tensor,        # (num_experts, K//group, N) bf16
    group_size: int = GROUP_SIZE,
) -> torch.Tensor:
    E = w_q.shape[0]
    K = w_q.shape[1] * 2
    N = w_q.shape[2]
    y = torch.empty(E, N, dtype=torch.bfloat16, device=x.device)

    blk_n = _pick_blk_n(N)
    grid = (E, triton.cdiv(N, blk_n))

    _fused_int4_gemv_multi_kernel[grid](
        w_q.contiguous(), scales.contiguous(), zeros.contiguous(),
        x.contiguous(), y,
        K,
        w_q.stride(0), scales.stride(0),
        w_q.stride(1), scales.stride(1),
        True,
        BLOCK_N=blk_n, GROUP_SIZE=group_size, N=N,
    )
    return y


# ---------------------------------------------------------------------------
# Fused QuantLinear
# ---------------------------------------------------------------------------

class QuantLinear(_QuantLinearBase):
    """W4A16 linear: fused int4 dequant-GEMV."""
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return _fused_int4_gemv(x, self.w_q, self.scales, self.zeros, self.group)


# ---------------------------------------------------------------------------
# MoE FFN helpers
# ---------------------------------------------------------------------------

def _ffn_routed(x, gate, up, down, idx, w):
    """Routed MoE FFN with fused multi-expert GEMV (single launch per projection).

    Returns (hidden,) fp32.
    """
    # Select expert weights
    wq_g = gate.w_q[idx].contiguous()
    sc_g = gate.scales[idx].contiguous()
    z_g = gate.zeros[idx].contiguous()

    wq_u = up.w_q[idx].contiguous()
    sc_u = up.scales[idx].contiguous()
    z_u = up.zeros[idx].contiguous()

    wq_d = down.w_q[idx].contiguous()
    sc_d = down.scales[idx].contiguous()
    z_d = down.zeros[idx].contiguous()

    # Fused multi-expert GEMV (shared input for gate/up, batched for down)
    gate_out = _fused_int4_gemv_multi_shared(x, wq_g, sc_g, z_g)
    up_out = _fused_int4_gemv_multi_shared(x, wq_u, sc_u, z_u)

    h = F.silu(gate_out.float()) * up_out.float()  # (k, m)

    down_out = _fused_int4_gemv_multi(h.to(torch.bfloat16), wq_d, sc_d, z_d)
    return (w[:, None].float() * down_out.float()).sum(0)


def _ffn_shared(x, gate, up, down, n_shared):
    """Shared MoE FFN.  Returns (hidden,) fp32."""
    idx = torch.arange(n_shared, device=x.device)

    wq_g = gate.w_q[idx].contiguous()
    sc_g = gate.scales[idx].contiguous()
    z_g = gate.zeros[idx].contiguous()
    wq_u = up.w_q[idx].contiguous()
    sc_u = up.scales[idx].contiguous()
    z_u = up.zeros[idx].contiguous()
    wq_d = down.w_q[idx].contiguous()
    sc_d = down.scales[idx].contiguous()
    z_d = down.zeros[idx].contiguous()

    gate_out = _fused_int4_gemv_multi_shared(x, wq_g, sc_g, z_g)
    up_out = _fused_int4_gemv_multi_shared(x, wq_u, sc_u, z_u)
    h = F.silu(gate_out.float()) * up_out.float()
    down_out = _fused_int4_gemv_multi(h.to(torch.bfloat16), wq_d, sc_d, z_d)
    return down_out.float().sum(0)


# ---------------------------------------------------------------------------
# MoE layer
# ---------------------------------------------------------------------------

class MoE(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.cfg = cfg
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = _QuantExpertsBase(E, d, m, cfg.group)
        self.up = _QuantExpertsBase(E, d, m, cfg.group)
        self.down = _QuantExpertsBase(E, m, d, cfg.group)
        self.s_gate = _QuantExpertsBase(cfg.n_shared, d, m, cfg.group)
        self.s_up = _QuantExpertsBase(cfg.n_shared, d, m, cfg.group)
        self.s_down = _QuantExpertsBase(cfg.n_shared, m, d, cfg.group)

    def step(self, x):
        cfg = self.cfg
        probs = torch.softmax(self.router(x).float(), dim=-1)
        w, idx = torch.topk(probs, cfg.n_active)
        w = (w / (w.sum() + 1e-9) * cfg.routed_scaling).to(x.dtype)
        out = _ffn_routed(x, self.gate, self.up, self.down, idx, w)
        out = out + _ffn_shared(x, self.s_gate, self.s_up, self.s_down, cfg.n_shared)
        return out.to(torch.bfloat16)


# ---------------------------------------------------------------------------
# KDA layer
# ---------------------------------------------------------------------------

class KDA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        self.q_proj = QuantLinear(d, H * Dk, cfg.group)
        self.k_proj = QuantLinear(d, H * Dk, cfg.group)
        self.v_proj = QuantLinear(d, H * Dk, cfg.group)
        self.g_proj = QuantLinear(d, H * Dk, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, H * Dk, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(H * Dk, d, cfg.group)
        self.scale = Dk ** -0.5

    def _short_conv(self, val, prev, idx):
        win = torch.cat([prev, val[None]], dim=0)
        w = self.conv_w[idx].float().transpose(0, 1)
        out = (win.float() * w).sum(0)
        return F.silu(out).to(val.dtype), win[1:]

    def step(self, x, st):
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)
        q, st["cq"] = self._short_conv(q, st["cq"], 0)
        k, st["ck"] = self._short_conv(k, st["ck"], 1)
        v, st["cv"] = self._short_conv(v, st["cv"], 2)
        q = q.view(H, Dk).float() * self.scale
        k = k.view(H, Dk).float()
        v = v.view(H, Dk).float()
        g = (-F.softplus(self.g_proj(x).float())).view(H, Dk)
        beta = torch.sigmoid(self.beta_proj(x).float())
        S = st["S"] * g.exp()[:, :, None]
        pred = (S * k[:, :, None]).sum(1)
        S = S + beta[:, None, None] * k[:, :, None] * (v - pred)[:, None, :]
        o = (S * q[:, :, None]).sum(1)
        st["S"] = S
        return self.o_proj(o.reshape(H * Dk).to(torch.bfloat16))


# ---------------------------------------------------------------------------
# MLA layer with KV absorption
# ---------------------------------------------------------------------------

class MLA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5

    @staticmethod
    def _dequant_weight(ql: QuantLinear) -> torch.Tensor:
        """Return the dequantised bf16 weight matrix (K, N).  Used once per
        MLA layer for the initial full-cache kv_b projection."""
        K = ql.in_f
        wu = _unpack_int4(ql.w_q, K).to(torch.bfloat16)
        s = ql.scales.repeat_interleave(ql.group, dim=0)
        z = ql.zeros.repeat_interleave(ql.group, dim=0)
        return (wu - z) * s

    def step(self, x, st):
        cfg = self.cfg
        H = cfg.mla_heads
        pos = st["c_kv"].shape[0]

        q = self.q_proj(x).view(H, cfg.qk_nope + cfg.qk_rope)
        q_nope = q[:, :cfg.qk_nope].float()
        q_rope = q[:, cfg.qk_nope:]

        kv = self.kv_a(x)
        c_kv_new = kv[:cfg.kv_lora]
        k_rope_new = kv[cfg.kv_lora:]

        cos, sin = _rope_cossin(pos, cfg.qk_rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin).float()
        k_rope_new = _apply_rope(k_rope_new, cos, sin)

        st["c_kv"] = torch.cat([st["c_kv"], c_kv_new[None]], 0)
        st["k_rope"] = torch.cat([st["k_rope"], k_rope_new[None]], 0)

        # KV absorption with pre-allocated caches to avoid torch.cat copies
        if "k_nope_cache" not in st:
            # First call: dequant kv_b and matmul for the full cache.
            w_kvb = self._dequant_weight(self.kv_b)
            kvb_full = (st["c_kv"] @ w_kvb).view(-1, H, cfg.qk_nope + cfg.v_head).float()

            curr_len = st["c_kv"].shape[0]
            # Pre-allocate with room for future tokens (one extra per expected step)
            capacity = curr_len + 128
            st["k_nope_cache"] = torch.empty(capacity, H, cfg.qk_nope, dtype=torch.float32,
                                             device=x.device)
            st["v_cache"] = torch.empty(capacity, H, cfg.v_head, dtype=torch.float32,
                                        device=x.device)
            st["k_nope_cache"][:curr_len] = kvb_full[..., :cfg.qk_nope]
            st["v_cache"][:curr_len] = kvb_full[..., cfg.qk_nope:]
            st["_cache_len"] = curr_len
        else:
            # Write new token into pre-allocated slot (no copy)
            idx = st["_cache_len"]
            new_kvb = self.kv_b(c_kv_new).view(1, H, cfg.qk_nope + cfg.v_head).float()
            st["k_nope_cache"][idx] = new_kvb[0, :, :cfg.qk_nope]
            st["v_cache"][idx] = new_kvb[0, :, cfg.qk_nope:]
            st["_cache_len"] = idx + 1

        k_nope = st["k_nope_cache"][:st["_cache_len"]]
        v = st["v_cache"][:st["_cache_len"]]

        scores = (
            torch.einsum("hd,lhd->lh", q_nope, k_nope)
            + torch.einsum("hd,ld->lh", q_rope, st["k_rope"].float())
        ) * self.scale
        p = torch.softmax(scores, dim=0)
        o = torch.einsum("lh,lhd->hd", p, v)
        return self.o_proj(o.reshape(H * cfg.v_head).to(torch.bfloat16))


# ---------------------------------------------------------------------------
# Block and Model
# ---------------------------------------------------------------------------

class Block(nn.Module):
    def __init__(self, cfg: Config, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)

    def step(self, x, st):
        h = x + self.attn.step(_rmsnorm(x, self.attn_norm), st)
        return h + self.moe.step(_rmsnorm(h, self.moe_norm))


class Model(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)
        self.reset_parameters()

    def reset_parameters(self):
        g = torch.Generator(device="cpu").manual_seed(1234)
        for mod in self.modules():
            if isinstance(mod, (_QuantLinearBase, _QuantExpertsBase)):
                mod.init_random(g)
            elif isinstance(mod, nn.Linear):
                nn.init.normal_(mod.weight, 0.0, 0.02, generator=g)
            elif isinstance(mod, KDA):
                nn.init.normal_(mod.conv_w, 0.0, 0.1, generator=g)

    def step(self, hidden, state):
        for i, blk in enumerate(self.blocks):
            hidden = blk.step(hidden, state[i])
        return hidden, state
