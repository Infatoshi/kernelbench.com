"""Fast W4A16 decode for one Kimi-Linear hybrid motif (batch=1).

Beats baseline.py three ways:
  1. Fused int4 unpack + per-group dequant directly into the GEMV (grouped over
     experts, gather-free so no expert weight is ever materialized). The int4
     weights stream once -- a quarter of the bf16 traffic.
  2. MLA *absorb*: absorb kv_b into q/v so every cache op stays O(L*kv_lora)
     instead of materializing an [L, H, 256] tensor (~268MB at L=16384).
  3. CUDA-graph capture of the whole motif step (only the new hidden + position
     are copied in), killing the ~200 kernel-launch overhead per step. The
     growing MLA cache is handled by a fixed padded buffer + an in-graph
     scatter-append + length-masked softmax, so a single static graph serves an
     entire decode run.

Module tree / buffer names are identical to the reference so its state_dict
loads with strict=True. All int4 dequant-GEMV kernels are hand-written Triton.
"""
from __future__ import annotations

import os
import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl
from dataclasses import dataclass, field

GROUP_SIZE = 128
EPS = 1.0e-6
_USE_GRAPH = os.environ.get("KBH_NO_GRAPH", "0") != "1"


@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: "torch.dtype" = field(default=torch.bfloat16)


def build_config(shape: dict) -> Config:
    return Config(n_experts=int(shape.get("n_experts", 64)))


# --------------------------------------------------------------------------- #
# fused int4 dequant-GEMV, grouped over experts, gather-free.
#   y[e, n] = sum_k x_e[k] * (unpack(wq[e])[k,n] - z[e,k//g,n]) * s[e,k//g,n]
# pack: wq[e, k//2, n] uint8, even-K = low nibble, odd-K = high nibble.
# When IDX is given the expert for program slot pid_e is IDX[pid_e] (routed MoE),
# avoiding a gather/copy of the selected expert weights.
# --------------------------------------------------------------------------- #
@triton.jit
def _w4_grp_kernel(
    X, WQ, S, Z, IDX, Y,
    K, N,
    stride_xe, stride_xk,
    stride_we, stride_wqk, stride_wqn,
    stride_se, stride_sg, stride_sn,
    stride_ye,
    GROUP: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    SPLIT_K: tl.constexpr, IS_PARTIAL: tl.constexpr, HAS_IDX: tl.constexpr,
):
    pid_e = tl.program_id(0)
    pid_n = tl.program_id(1)
    pid_k = tl.program_id(2)
    e_real = tl.load(IDX + pid_e) if HAS_IDX else pid_e
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = offs_n < N

    x_base = X + pid_e * stride_xe
    wq_base = WQ + e_real * stride_we
    s_base = S + e_real * stride_se
    z_base = Z + e_real * stride_se

    ng = K // GROUP
    tiles_per_split = (ng + SPLIT_K - 1) // SPLIT_K
    g_start = pid_k * tiles_per_split
    g_end = tl.minimum((pid_k + 1) * tiles_per_split, ng)

    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    pk = tl.arange(0, BLOCK_K // 2)
    for g in range(g_start, g_end):
        s = tl.load(s_base + g * stride_sg + offs_n * stride_sn, mask=n_mask, other=0.0).to(tl.float32)
        z = tl.load(z_base + g * stride_sg + offs_n * stride_sn, mask=n_mask, other=0.0).to(tl.float32)
        acc_g = tl.zeros([BLOCK_N], dtype=tl.float32)
        gbase = g * GROUP
        for ck in tl.static_range(0, GROUP, BLOCK_K):
            kbase = gbase + ck
            x_even = tl.load(x_base + (kbase + 2 * pk) * stride_xk).to(tl.float32)
            x_odd = tl.load(x_base + (kbase + 2 * pk + 1) * stride_xk).to(tl.float32)
            w_off = wq_base + (kbase // 2 + pk)[:, None] * stride_wqk + offs_n[None, :] * stride_wqn
            wq = tl.load(w_off, mask=n_mask[None, :], other=0)
            we = (wq & 0xF).to(tl.float32)
            wo = (wq >> 4).to(tl.float32)
            contrib = (we - z[None, :]) * x_even[:, None] + (wo - z[None, :]) * x_odd[:, None]
            acc_g += tl.sum(contrib, axis=0)
        acc += acc_g * s

    y_base = Y + pid_e * stride_ye
    if IS_PARTIAL:
        tl.store(y_base + pid_k * N + offs_n, acc, mask=n_mask)
    else:
        tl.store(y_base + offs_n, acc.to(tl.bfloat16), mask=n_mask)


def _launch(X, wq, scales, zeros, K, N, E, group, stride_xe, idx=None,
            num_warps=4, num_stages=2):
    ng = K // group
    BLOCK_N = 64
    BLOCK_K = 128
    n_blocks = (N + BLOCK_N - 1) // BLOCK_N
    target = 256
    denom = max(E * n_blocks, 1)
    split_k = (target + denom - 1) // denom
    split_k = min(max(split_k, 1), ng, 8)
    has_idx = idx is not None
    idx_arg = idx if has_idx else X
    wq = wq.contiguous()
    scales = scales.contiguous()
    zeros = zeros.contiguous()
    X = X.contiguous()
    B0 = wq.shape[0]
    stride_we, stride_wqk, stride_wqn = (wq.stride(0), wq.stride(1), wq.stride(2)) if wq.ndim == 3 \
        else (0, wq.stride(0), wq.stride(1))
    sse, ssg, ssn = (scales.stride(0), scales.stride(1), scales.stride(2)) if scales.ndim == 3 \
        else (0, scales.stride(0), scales.stride(1))
    if split_k == 1:
        Y = torch.empty(E, N, dtype=torch.bfloat16, device=X.device)
        grid = (E, n_blocks, 1)
        _w4_grp_kernel[grid](
            X, wq, scales, zeros, idx_arg, Y, K, N,
            stride_xe, X.stride(-1) if X.ndim >= 1 else 1,
            stride_we, stride_wqk, stride_wqn, sse, ssg, ssn, N,
            GROUP=group, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
            SPLIT_K=1, IS_PARTIAL=False, HAS_IDX=has_idx,
            num_warps=num_warps, num_stages=num_stages)
        return Y
    part = torch.empty(E, split_k, N, dtype=torch.float32, device=X.device)
    grid = (E, n_blocks, split_k)
    _w4_grp_kernel[grid](
        X, wq, scales, zeros, idx_arg, part, K, N,
        stride_xe, X.stride(-1) if X.ndim >= 1 else 1,
        stride_we, stride_wqk, stride_wqn, sse, ssg, ssn, split_k * N,
        GROUP=group, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
        SPLIT_K=split_k, IS_PARTIAL=True, HAS_IDX=has_idx,
        num_warps=num_warps, num_stages=num_stages)
    return part.sum(1).to(torch.bfloat16)


def qlin_fwd(x, ql):
    """QuantLinear: x[K] -> [N]."""
    return _launch(x, ql.w_q.unsqueeze(0), ql.scales.unsqueeze(0), ql.zeros.unsqueeze(0),
                   ql.in_f, ql.out_f, 1, ql.group, stride_xe=0)[0]


def qexp_shared(x, qe, idx=None):
    """Shared-x grouped GEMV over selected (or all) experts -> [E', N]."""
    E = qe.n if idx is None else idx.numel()
    return _launch(x, qe.w_q, qe.scales, qe.zeros, qe.in_f, qe.out_f, E, qe.group,
                   stride_xe=0, idx=idx)


def qexp_perexpert(X, qe, idx=None):
    """Per-expert-input grouped GEMV. X:[E', K] -> [E', N]."""
    E = X.shape[0]
    return _launch(X, qe.w_q, qe.scales, qe.zeros, qe.in_f, qe.out_f, E, qe.group,
                   stride_xe=qe.in_f, idx=idx)


def _dequant_full(w_q, scales, zeros, K, group=GROUP_SIZE):
    N = w_q.shape[-1]
    wu = torch.empty(K, N, dtype=torch.uint8, device=w_q.device)
    wu[0::2] = w_q & 0xF
    wu[1::2] = (w_q >> 4) & 0xF
    s = scales.repeat_interleave(group, dim=0)
    z = zeros.repeat_interleave(group, dim=0)
    return (wu.to(torch.bfloat16) - z) * s


# --------------------------------------------------------------------------- #
# quantized linear / experts (storage matches the reference exactly)
# --------------------------------------------------------------------------- #
class QuantLinear(nn.Module):
    def __init__(self, in_f, out_f, group=GROUP_SIZE):
        super().__init__()
        assert in_f % group == 0 and in_f % 2 == 0
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))

    def fwd(self, x):
        return qlin_fwd(x, self)


class QuantExperts(nn.Module):
    def __init__(self, n, in_f, out_f, group=GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))


# --------------------------------------------------------------------------- #
# helpers
# --------------------------------------------------------------------------- #
def _rmsnorm(x, w):
    xf = x.float()
    xf = xf * torch.rsqrt(xf.pow(2).mean(-1, keepdim=True) + EPS)
    return (xf * w.float()).to(x.dtype)


def _rope_cossin(pos, dim, theta, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos * inv
    return torch.cos(ang), torch.sin(ang)


def _rope_cossin_t(pos_t, dim, theta, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos_t.to(torch.float32) * inv
    return torch.cos(ang), torch.sin(ang)


def _apply_rope(x, cos, sin):
    xf = x.float()
    even, odd = xf[..., 0::2], xf[..., 1::2]
    out = torch.empty_like(xf)
    out[..., 0::2] = even * cos - odd * sin
    out[..., 1::2] = odd * cos + even * sin
    return out.to(x.dtype)

class KDA(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        C = H * Dk
        self.C = C
        self.q_proj = QuantLinear(d, C, cfg.group)
        self.k_proj = QuantLinear(d, C, cfg.group)
        self.v_proj = QuantLinear(d, C, cfg.group)
        self.g_proj = QuantLinear(d, C, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, C, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(C, d, cfg.group)
        self.scale = Dk ** -0.5

    def _conv(self, val, prev, idx):
        win = torch.cat([prev, val[None]], dim=0)
        w = self.conv_w[idx].float().transpose(0, 1)
        out = (win.float() * w).sum(0)
        return F.silu(out).to(val.dtype), win[1:]

    def core(self, x, b):
        """b holds persistent S, cq, ck, cv buffers (mutated in place)."""
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        q = self.q_proj.fwd(x)
        k = self.k_proj.fwd(x)
        v = self.v_proj.fwd(x)
        q, nq = self._conv(q, b["cq"], 0); b["cq"].copy_(nq)
        k, nk = self._conv(k, b["ck"], 1); b["ck"].copy_(nk)
        v, nv = self._conv(v, b["cv"], 2); b["cv"].copy_(nv)
        q = q.view(H, Dk).float() * self.scale
        k = k.view(H, Dk).float()
        v = v.view(H, Dk).float()
        g = (-F.softplus(self.g_proj.fwd(x).float())).view(H, Dk)
        beta = torch.sigmoid(self.beta_proj(x).float())
        S = b["S"]
        Snew = S * g.exp()[:, :, None]
        pred = (Snew * k[:, :, None]).sum(1)
        Snew = Snew + beta[:, None, None] * k[:, :, None] * (v - pred)[:, None, :]
        o = (Snew * q[:, :, None]).sum(1)
        S.copy_(Snew)
        return self.o_proj.fwd(o.reshape(H * Dk).to(torch.bfloat16))

    def step(self, x, st):
        # eager fallback (state dict carries the windows directly)
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        q = self.q_proj.fwd(x)
        k = self.k_proj.fwd(x)
        v = self.v_proj.fwd(x)
        q, st["cq"] = self._conv(q, st["cq"], 0)
        k, st["ck"] = self._conv(k, st["ck"], 1)
        v, st["cv"] = self._conv(v, st["cv"], 2)
        q = q.view(H, Dk).float() * self.scale
        k = k.view(H, Dk).float()
        v = v.view(H, Dk).float()
        g = (-F.softplus(self.g_proj.fwd(x).float())).view(H, Dk)
        beta = torch.sigmoid(self.beta_proj(x).float())
        S = st["S"]
        S = S * g.exp()[:, :, None]
        pred = (S * k[:, :, None]).sum(1)
        S = S + beta[:, None, None] * k[:, :, None] * (v - pred)[:, None, :]
        o = (S * q[:, :, None]).sum(1)
        st["S"] = S
        return self.o_proj.fwd(o.reshape(H * Dk).to(torch.bfloat16))


class MLA(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5
        self._kvb_bf = None

    def _get_kvb(self):
        if self._kvb_bf is None or self._kvb_bf.device != self.kv_b.w_q.device:
            self._kvb_bf = _dequant_full(self.kv_b.w_q, self.kv_b.scales,
                                         self.kv_b.zeros, self.kv_b.in_f, self.kv_b.group)
        return self._kvb_bf

    def core(self, x, b):
        """b holds persistent c_kv_buf, k_rope_buf, pos buffers. Pad-masked absorb."""
        cfg = self.cfg
        H = cfg.mla_heads
        LPAD = b["c_kv"].shape[0]
        c_kv_buf = b["c_kv"]
        k_rope_buf = b["k_rope"]
        pos_in = b["pos"].reshape(())

        q = self.q_proj.fwd(x).view(H, cfg.qk_nope + cfg.qk_rope)
        q_nope = q[:, : cfg.qk_nope].float()
        q_rope = q[:, cfg.qk_nope:]
        kv = self.kv_a.fwd(x)
        c_new = kv[: cfg.kv_lora]
        kr_new = kv[cfg.kv_lora:]
        cos, sin = _rope_cossin_t(pos_in, cfg.qk_rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin).float()
        kr_new = _apply_rope(kr_new, cos, sin)
        c_kv_buf.index_copy_(0, pos_in.reshape(1), c_new.reshape(1, cfg.kv_lora))
        k_rope_buf.index_copy_(0, pos_in.reshape(1), kr_new.reshape(1, cfg.qk_rope))
        new_len = pos_in + 1

        kvbf = self._get_kvb().view(cfg.kv_lora, H, cfg.qk_nope + cfg.v_head).float()
        kn = kvbf[:, :, : cfg.qk_nope]
        vv = kvbf[:, :, cfg.qk_nope:]
        qa = torch.einsum("hd,rhd->rh", q_nope, kn)
        scores_nope = c_kv_buf.float() @ qa                      # [LPAD, H]
        scores_rope = k_rope_buf.float() @ q_rope.t()           # [LPAD, H]
        scores = (scores_nope + scores_rope) * self.scale
        idx = torch.arange(LPAD, device=x.device)
        mask = (idx >= new_len).unsqueeze(1)
        scores = scores.masked_fill(mask, float("-inf"))
        p = torch.softmax(scores, dim=0)
        lv = torch.einsum("lh,lr->hr", p, c_kv_buf.float())
        o = torch.einsum("hr,rhd->hd", lv, vv)
        b["pos"].copy_(new_len.reshape(b["pos"].shape))
        return self.o_proj.fwd(o.reshape(H * cfg.v_head).to(torch.bfloat16))

    def step(self, x, st):
        # eager fallback: materialize-free absorb over the real (cat-grown) cache
        cfg = self.cfg
        H = cfg.mla_heads
        pos = st["c_kv"].shape[0]
        q = self.q_proj.fwd(x).view(H, cfg.qk_nope + cfg.qk_rope)
        q_nope = q[:, : cfg.qk_nope].float()
        q_rope = q[:, cfg.qk_nope:]
        kv = self.kv_a.fwd(x)
        c_new = kv[: cfg.kv_lora]
        kr_new = kv[cfg.kv_lora:]
        cos, sin = _rope_cossin(pos, cfg.qk_rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin).float()
        kr_new = _apply_rope(kr_new, cos, sin)
        st["c_kv"] = torch.cat([st["c_kv"], c_new[None]], 0)
        st["k_rope"] = torch.cat([st["k_rope"], kr_new[None]], 0)
        kvbf = self._get_kvb().view(cfg.kv_lora, H, cfg.qk_nope + cfg.v_head).float()
        kn = kvbf[:, :, : cfg.qk_nope]
        vv = kvbf[:, :, cfg.qk_nope:]
        c_kv = st["c_kv"].float()
        k_rope = st["k_rope"].float()
        qa = torch.einsum("hd,rhd->rh", q_nope, kn)
        scores_nope = c_kv @ qa
        scores_rope = k_rope @ q_rope.t()
        scores = (scores_nope + scores_rope) * self.scale
        p = torch.softmax(scores, dim=0)
        lv = torch.einsum("lh,lr->hr", p, c_kv)
        o = torch.einsum("hr,rhd->hd", lv, vv)
        return self.o_proj.fwd(o.reshape(H * cfg.v_head).to(torch.bfloat16))


class MoE(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.cfg = cfg
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(E, d, m, cfg.group)
        self.up = QuantExperts(E, d, m, cfg.group)
        self.down = QuantExperts(E, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)

    def _moe(self, x):
        cfg = self.cfg
        probs = torch.softmax(self.router(x).float(), dim=-1)
        w, idx = torch.topk(probs, cfg.n_active)
        w = w / (w.sum() + 1e-9) * cfg.routed_scaling
        g = qexp_shared(x, self.gate, idx).float()
        u = qexp_shared(x, self.up, idx).float()
        gi = F.silu(g) * u                                  # [na, m]
        d_out = qexp_perexpert(gi.to(torch.bfloat16), self.down, idx).float()  # [na, d]
        out = (w[:, None] * d_out).sum(0)
        sg = qexp_shared(x, self.s_gate).float()
        su = qexp_shared(x, self.s_up).float()
        sgi = F.silu(sg) * su                               # [1, m]
        sd = qexp_perexpert(sgi.to(torch.bfloat16), self.s_down).float()       # [1, d]
        return (out + sd[0]).to(torch.bfloat16)

    def core(self, x, b):
        return self._moe(x)

    def step(self, x):
        return self._moe(x)


class Block(nn.Module):
    def __init__(self, cfg, kind):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)

    def core(self, x, b):
        h = x + self.attn.core(_rmsnorm(x, self.attn_norm), b)
        return h + self.moe.core(_rmsnorm(h, self.moe_norm), b)

    def step(self, x, st):
        h = x + self.attn.step(_rmsnorm(x, self.attn_norm), st)
        return h + self.moe.step(_rmsnorm(h, self.moe_norm))


class Model(nn.Module):
    def __init__(self, cfg):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)
        self._graphs = {}
        self._pos_py = None

    # -- graph infrastructure ------------------------------------------------ #
    def _pad(self, ctx):
        return ((ctx + 64 + 127) // 128) * 128

    def _alloc_bufs(self, lpad, device):
        cfg = self.cfg
        H, Dk = cfg.kda_heads, cfg.kda_head_dim
        C = H * Dk
        bufs = []
        for i, kind in enumerate(cfg.pattern):
            if kind == "K":
                bufs.append({
                    "S": torch.empty(H, Dk, Dk, dtype=torch.float32, device=device),
                    "cq": torch.empty(cfg.short_conv - 1, C, dtype=cfg.dtype, device=device),
                    "ck": torch.empty(cfg.short_conv - 1, C, dtype=cfg.dtype, device=device),
                    "cv": torch.empty(cfg.short_conv - 1, C, dtype=cfg.dtype, device=device),
                })
            else:
                bufs.append({
                    "c_kv": torch.empty(lpad, cfg.kv_lora, dtype=cfg.dtype, device=device),
                    "k_rope": torch.empty(lpad, cfg.qk_rope, dtype=cfg.dtype, device=device),
                    "pos": torch.zeros(1, dtype=torch.int64, device=device),
                })
        return bufs

    def _copy_state_in(self, state, bufs):
        cfg = self.cfg
        for i, kind in enumerate(cfg.pattern):
            st = state[i]
            b = bufs[i]
            if kind == "K":
                b["S"].copy_(st["S"])
                b["cq"].copy_(st["cq"])
                b["ck"].copy_(st["ck"])
                b["cv"].copy_(st["cv"])
            else:
                L = st["c_kv"].shape[0]
                b["c_kv"][:L].copy_(st["c_kv"])
                b["c_kv"][L:].zero_()
                b["k_rope"][:L].copy_(st["k_rope"])
                b["k_rope"][L:].zero_()
                b["pos"].fill_(L)

    def _repoint_state(self, state, bufs):
        """Make the state dict reference the persistent buffers (views, not copies)."""
        for i, kind in enumerate(self.cfg.pattern):
            if kind == "K":
                state[i]["S"] = bufs[i]["S"]
                state[i]["cq"] = bufs[i]["cq"]
                state[i]["ck"] = bufs[i]["ck"]
                state[i]["cv"] = bufs[i]["cv"]

    def _core_step(self, h, bufs):
        for i, blk in enumerate(self.blocks):
            h = blk.core(h, bufs[i])
        return h

    def _capture(self, state, hidden, lpad):
        device = hidden.device
        bufs = self._alloc_bufs(lpad, device)
        h_in = torch.empty_like(hidden)
        self._copy_state_in(state, bufs)
        h_in.copy_(hidden)
        # warmup on a side stream to materialize lazy CUDA state
        s = torch.cuda.Stream()
        s.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(s):
            for _ in range(3):
                _ = self._core_step(h_in, bufs)
        torch.cuda.current_stream().wait_stream(s)
        # reset (warmup mutated buffers), then capture
        self._copy_state_in(state, bufs)
        h_in.copy_(hidden)
        g = torch.cuda.CUDAGraph()
        with torch.cuda.graph(g):
            h_out = self._core_step(h_in, bufs)
        self._copy_state_in(state, bufs)
        h_in.copy_(hidden)
        kda0 = self.cfg.pattern.index("K")
        return {"graph": g, "bufs": bufs, "h_in": h_in, "h_out": h_out,
                "last_ptr": state[kda0]["S"].data_ptr()}

    def _step_graph(self, hidden, state):
        cfg = self.cfg
        kda0 = cfg.pattern.index("K")
        mla0 = cfg.pattern.index("M")
        ctx = state[mla0]["c_kv"].shape[0]
        lpad = self._pad(ctx)
        ent = self._graphs.get(lpad)
        if ent is None:
            ent = self._capture(state, hidden, lpad)
            self._graphs[lpad] = ent
            self._repoint_state(state, ent["bufs"])
            self._pos_py = ctx
        else:
            if state[kda0]["S"].data_ptr() != ent["last_ptr"]:
                self._copy_state_in(state, ent["bufs"])
                self._repoint_state(state, ent["bufs"])
                self._pos_py = ctx
                ent["last_ptr"] = state[kda0]["S"].data_ptr()
        ent["h_in"].copy_(hidden)
        ent["graph"].replay()
        self._pos_py += 1
        L = self._pos_py
        state[mla0]["c_kv"] = ent["bufs"][mla0]["c_kv"][:L]
        state[mla0]["k_rope"] = ent["bufs"][mla0]["k_rope"][:L]
        return ent["h_out"], state

    def _step_eager(self, hidden, state):
        for i, blk in enumerate(self.blocks):
            hidden = blk.step(hidden, state[i])
        return hidden, state

    def step(self, hidden, state):
        if _USE_GRAPH and getattr(self, "_graph_ok", True):
            try:
                return self._step_graph(hidden, state)
            except Exception:
                self._graph_ok = False
                self._graphs.clear()
                return self._step_eager(hidden, state)
        return self._step_eager(hidden, state)
