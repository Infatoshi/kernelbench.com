"""Hybrid W4A16 Kimi-Linear decode.

Strategy: dequant non-MoE weights to bf16 once (cached), then use cuBLAS GEMV.
MoE experts still use fused Triton kernels. MLA uses the absorb trick.
"""

from __future__ import annotations

from dataclasses import dataclass, field

import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl

# --------------------------------------------------------------------------- #
# Contract types and helpers
# --------------------------------------------------------------------------- #

OP_TYPE = "kimi_linear_w4a16_decode"
HARDWARE_REQUIRED = ["RTX_PRO_6000"]
EPS = 1.0e-6
GROUP_SIZE = 128


@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: torch.dtype = field(default=torch.bfloat16)


def build_config(shape: dict) -> Config:
    return Config(n_experts=int(shape.get("n_experts", 64)))


def _rmsnorm(x: torch.Tensor, w: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    xf = xf * torch.rsqrt(xf.pow(2).mean(-1, keepdim=True) + EPS)
    return (xf * w.float()).to(x.dtype)


def _rope_cossin(pos: int, dim: int, theta: float, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos * inv
    return torch.cos(ang), torch.sin(ang)


def _apply_rope(x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    even, odd = xf[..., 0::2], xf[..., 1::2]
    out = torch.empty_like(xf)
    out[..., 0::2] = even * cos - odd * sin
    out[..., 1::2] = odd * cos + even * sin
    return out.to(x.dtype)


def init_state(cfg: Config, context_len: int, seed: int) -> list:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed)
    H, Dk = cfg.kda_heads, cfg.kda_head_dim
    C = H * Dk
    state = []
    for kind in cfg.pattern:
        if kind == "K":
            state.append({
                "S": torch.randn(H, Dk, Dk, device=dev, generator=g) * 0.05,
                "cq": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g,
                                  dtype=cfg.dtype) * 0.1,
                "ck": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g,
                                  dtype=cfg.dtype) * 0.1,
                "cv": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g,
                                  dtype=cfg.dtype) * 0.1,
            })
        else:
            state.append({
                "c_kv": torch.randn(context_len, cfg.kv_lora, device=dev,
                                    generator=g, dtype=cfg.dtype) * 0.1,
                "k_rope": torch.randn(context_len, cfg.qk_rope, device=dev,
                                      generator=g, dtype=cfg.dtype) * 0.1,
            })
    return state


def init_token(cfg: Config, seed: int) -> torch.Tensor:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed + 1)
    return torch.randn(cfg.hidden, device=dev, generator=g, dtype=cfg.dtype) * 0.25

# QuantLinear helpers

def _wbf_int4(w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor,
              K: int, group: int = GROUP_SIZE) -> torch.Tensor:
    """Dequant int4 weights to bf16 [K, N]."""
    wu = torch.empty((K, w_q.shape[1]), dtype=torch.uint8, device=w_q.device)
    wu[0::2] = w_q & 0xF
    wu[1::2] = (w_q >> 4) & 0xF
    s = scales.repeat_interleave(group, dim=0)
    z = zeros.repeat_interleave(group, dim=0)
    return (wu.to(torch.bfloat16) - z) * s

# Triton kernels (MoE only)
# --------------------------------------------------------------------------- #

@triton.jit
def _gemv_kernel(
    x_ptr, wq_ptr, scales_ptr, zeros_ptr, y_ptr,
    K, N, N_stride,
    G: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    pid = tl.program_id(0)
    n_offs = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offs < N
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    ng = K // G
    for g in range(ng):
        s = tl.load(scales_ptr + g * N_stride + n_offs, mask=n_mask, other=0.0)
        z = tl.load(zeros_ptr + g * N_stride + n_offs, mask=n_mask, other=0.0)
        sf = s.to(tl.float32)
        zf = z.to(tl.float32)
        kb = g * G
        for j in range(0, G, 2):
            k0 = kb + j
            x0 = tl.load(x_ptr + k0).to(tl.float32)
            x1 = tl.load(x_ptr + k0 + 1).to(tl.float32)
            prow = k0 // 2
            b = tl.load(wq_ptr + prow * N_stride + n_offs, mask=n_mask, other=0)
            w0 = (b & 0xF).to(tl.float32)
            w1 = ((b >> 4) & 0xF).to(tl.float32)
            acc += x0 * (w0 - zf) * sf
            acc += x1 * (w1 - zf) * sf
    tl.store(y_ptr + n_offs, acc.to(tl.bfloat16), mask=n_mask)


@triton.jit
def _gemv_batched_same_x_kernel(
    x_ptr, wq_ptr, scales_ptr, zeros_ptr, y_ptr,
    idx_ptr,
    K, N, N_stride,
    E_stride_wq, E_stride_scale, E_stride_zero,
    G: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    b = tl.program_id(0)
    pid = tl.program_id(1)
    n_offs = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offs < N
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    e = tl.load(idx_ptr + b)
    wq_e = wq_ptr + e * E_stride_wq
    sc_e = scales_ptr + e * E_stride_scale
    zr_e = zeros_ptr + e * E_stride_zero
    ng = K // G
    for g in range(ng):
        s = tl.load(sc_e + g * N_stride + n_offs, mask=n_mask, other=0.0)
        z = tl.load(zr_e + g * N_stride + n_offs, mask=n_mask, other=0.0)
        sf = s.to(tl.float32)
        zf = z.to(tl.float32)
        kb = g * G
        for j in range(0, G, 2):
            k0 = kb + j
            x0 = tl.load(x_ptr + k0).to(tl.float32)
            x1 = tl.load(x_ptr + k0 + 1).to(tl.float32)
            prow = k0 // 2
            wb = tl.load(wq_e + prow * N_stride + n_offs, mask=n_mask, other=0)
            w0 = (wb & 0xF).to(tl.float32)
            w1 = ((wb >> 4) & 0xF).to(tl.float32)
            acc += x0 * (w0 - zf) * sf
            acc += x1 * (w1 - zf) * sf
    tl.store(y_ptr + b * N + n_offs, acc.to(tl.bfloat16), mask=n_mask)


@triton.jit
def _gemv_batched_diff_x_kernel(
    x_ptr, wq_ptr, scales_ptr, zeros_ptr, y_ptr,
    idx_ptr,
    K, N, N_stride,
    E_stride_wq, E_stride_scale, E_stride_zero,
    G: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    b = tl.program_id(0)
    pid = tl.program_id(1)
    n_offs = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offs < N
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    e = tl.load(idx_ptr + b)
    wq_e = wq_ptr + e * E_stride_wq
    sc_e = scales_ptr + e * E_stride_scale
    zr_e = zeros_ptr + e * E_stride_zero
    x_e = x_ptr + b * K
    ng = K // G
    for g in range(ng):
        s = tl.load(sc_e + g * N_stride + n_offs, mask=n_mask, other=0.0)
        z = tl.load(zr_e + g * N_stride + n_offs, mask=n_mask, other=0.0)
        sf = s.to(tl.float32)
        zf = z.to(tl.float32)
        kb = g * G
        for j in range(0, G, 2):
            k0 = kb + j
            x0 = tl.load(x_e + k0).to(tl.float32)
            x1 = tl.load(x_e + k0 + 1).to(tl.float32)
            prow = k0 // 2
            wb = tl.load(wq_e + prow * N_stride + n_offs, mask=n_mask, other=0)
            w0 = (wb & 0xF).to(tl.float32)
            w1 = ((wb >> 4) & 0xF).to(tl.float32)
            acc += x0 * (w0 - zf) * sf
            acc += x1 * (w1 - zf) * sf
    tl.store(y_ptr + b * N + n_offs, acc.to(tl.bfloat16), mask=n_mask)


@triton.jit
def _mla_q_absorb_kernel(
    q_nope_ptr, wq_ptr, scales_ptr, zeros_ptr, y_ptr,
    qk_nope: tl.constexpr,
    kv_lora: tl.constexpr,
    n_heads: tl.constexpr,
    total_N: tl.constexpr,
    G: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    h = tl.program_id(0)
    n_start = tl.program_id(1) * BLOCK_N
    BLOCK_PAIRS: tl.constexpr = BLOCK_N // 2
    col_off = h * qk_nope
    pair_start = n_start // 2
    pair_ids = pair_start + tl.arange(0, BLOCK_PAIRS)
    n_even = pair_ids * 2
    n_odd = n_even + 1
    mask_even = n_even < kv_lora
    mask_odd = n_odd < kv_lora
    acc_even = tl.zeros((BLOCK_PAIRS,), dtype=tl.float32)
    acc_odd = tl.zeros((BLOCK_PAIRS,), dtype=tl.float32)
    for i in range(qk_nope):
        qv = tl.load(q_nope_ptr + h * qk_nope + i).to(tl.float32)
        wq = tl.load(wq_ptr + pair_ids * total_N + col_off + i, mask=mask_even, other=0)
        wl = (wq & 0xF).to(tl.float32)
        wh = ((wq >> 4) & 0xF).to(tl.float32)
        g_even = n_even // G
        se = tl.load(scales_ptr + g_even * total_N + col_off + i,
                     mask=mask_even, other=0.0).to(tl.float32)
        ze = tl.load(zeros_ptr + g_even * total_N + col_off + i,
                     mask=mask_even, other=0.0).to(tl.float32)
        acc_even += qv * (wl - ze) * se
        g_odd = n_odd // G
        so = tl.load(scales_ptr + g_odd * total_N + col_off + i,
                     mask=mask_odd, other=0.0).to(tl.float32)
        zo = tl.load(zeros_ptr + g_odd * total_N + col_off + i,
                     mask=mask_odd, other=0.0).to(tl.float32)
        acc_odd += qv * (wh - zo) * so
    tl.store(y_ptr + h * kv_lora + n_even, acc_even.to(tl.bfloat16), mask=mask_even)
    tl.store(y_ptr + h * kv_lora + n_odd, acc_odd.to(tl.bfloat16), mask=mask_odd)


@triton.jit
def _mla_o_absorb_kernel(
    c_kv_w_ptr, wq_ptr, scales_ptr, zeros_ptr, y_ptr,
    kv_lora: tl.constexpr,
    v_head: tl.constexpr,
    qk_nope: tl.constexpr,
    n_heads: tl.constexpr,
    total_N: tl.constexpr,
    G: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    h = tl.program_id(0)
    n_start = tl.program_id(1) * BLOCK_N
    n_offs = n_start + tl.arange(0, BLOCK_N)
    n_mask = n_offs < v_head
    col_off = n_heads * qk_nope + h * v_head
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    ng = kv_lora // G
    for g in range(ng):
        s = tl.load(scales_ptr + g * total_N + col_off + n_offs, mask=n_mask, other=0.0)
        z = tl.load(zeros_ptr + g * total_N + col_off + n_offs, mask=n_mask, other=0.0)
        sf = s.to(tl.float32)
        zf = z.to(tl.float32)
        kb = g * G
        for j in range(0, G, 2):
            k0 = kb + j
            x0 = tl.load(c_kv_w_ptr + h * kv_lora + k0).to(tl.float32)
            x1 = tl.load(c_kv_w_ptr + h * kv_lora + k0 + 1).to(tl.float32)
            prow = k0 // 2
            b = tl.load(wq_ptr + prow * total_N + col_off + n_offs, mask=n_mask, other=0)
            w0 = (b & 0xF).to(tl.float32)
            w1 = ((b >> 4) & 0xF).to(tl.float32)
            acc += x0 * (w0 - zf) * sf
            acc += x1 * (w1 - zf) * sf
    tl.store(y_ptr + h * v_head + n_offs, acc.to(tl.bfloat16), mask=n_mask)

def _single_gemv(x, w_q, scales, zeros, K, N, G=GROUP_SIZE) -> torch.Tensor:
    y = torch.empty(N, dtype=torch.bfloat16, device=x.device)
    BN = 128
    grid = (triton.cdiv(N, BN),)
    _gemv_kernel[grid](x, w_q, scales, zeros, y, K, N, N, G=G, BLOCK_N=BN)
    return y

def _batched_gemv(x, w_q, scales, zeros, idx, K, N, G=GROUP_SIZE,
                  same_x: bool = True) -> torch.Tensor:
    E = idx.numel()
    y = torch.empty(E, N, dtype=torch.bfloat16, device=x.device)
    BN = 128
    grid = (E, triton.cdiv(N, BN))
    E_stride_wq = w_q.stride(0)
    E_stride_scale = scales.stride(0)
    E_stride_zero = zeros.stride(0)
    if same_x:
        _gemv_batched_same_x_kernel[grid](
            x, w_q, scales, zeros, y, idx,
            K, N, N, E_stride_wq, E_stride_scale, E_stride_zero,
            G=G, BLOCK_N=BN,
        )
    else:
        _gemv_batched_diff_x_kernel[grid](
            x, w_q, scales, zeros, y, idx,
            K, N, N, E_stride_wq, E_stride_scale, E_stride_zero,
            G=G, BLOCK_N=BN,
        )
    return y


# --------------------------------------------------------------------------- #
# QuantLinear  (keeps int4 buffers for state-dict load, uses cached bf16)
# --------------------------------------------------------------------------- #

class QuantLinear(nn.Module):
    def __init__(self, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        assert in_f % group == 0 and in_f % 2 == 0
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self._w_bf16 = None

    def _cache_bf16(self):
        if self._w_bf16 is None:
            self._w_bf16 = _wbf_int4(self.w_q, self.scales, self.zeros,
                                      self.in_f, self.group)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        self._cache_bf16()
        return x @ self._w_bf16


class QuantExperts(nn.Module):
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))

# KDA
# --------------------------------------------------------------------------- #

class KDA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        C = H * Dk
        self.q_proj = QuantLinear(d, C, cfg.group)
        self.k_proj = QuantLinear(d, C, cfg.group)
        self.v_proj = QuantLinear(d, C, cfg.group)
        self.g_proj = QuantLinear(d, C, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, C, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(C, d, cfg.group)
        self.scale = Dk ** -0.5

    def _short_conv(self, val, prev, idx):
        win = torch.cat([prev, val[None]], dim=0)
        w = self.conv_w[idx].float().transpose(0, 1)
        out = (win.float() * w).sum(0)
        return F.silu(out).to(val.dtype), win[1:]

    def step(self, x, st):
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)
        q, st["cq"] = self._short_conv(q, st["cq"], 0)
        k, st["ck"] = self._short_conv(k, st["ck"], 1)
        v, st["cv"] = self._short_conv(v, st["cv"], 2)
        q = q.view(H, Dk).float() * self.scale
        k = k.view(H, Dk).float()
        v = v.view(H, Dk).float()
        g = (-F.softplus(self.g_proj(x).float())).view(H, Dk)
        beta = torch.sigmoid(self.beta_proj(x).float())
        S = st["S"] * g.exp()[:, :, None]
        pred = (S * k[:, :, None]).sum(1)
        S = S + beta[:, None, None] * k[:, :, None] * (v - pred)[:, None, :]
        o = (S * q[:, :, None]).sum(1)
        st["S"] = S
        return self.o_proj(o.reshape(H * Dk).to(torch.bfloat16))

# MLA
# --------------------------------------------------------------------------- #

class MLA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5

    def step(self, x, st):
        cfg = self.cfg
        H = cfg.mla_heads
        pos = st["c_kv"].shape[0]
        q = self.q_proj(x).view(H, cfg.qk_nope + cfg.qk_rope)
        q_nope = q[:, :cfg.qk_nope]
        q_rope = q[:, cfg.qk_nope:]
        kv = self.kv_a(x)
        c_kv_new = kv[:cfg.kv_lora]
        k_rope_new = kv[cfg.kv_lora:]
        cos, sin = _rope_cossin(pos, cfg.qk_rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin).float()
        k_rope_new = _apply_rope(k_rope_new, cos, sin)
        st["c_kv"] = torch.cat([st["c_kv"], c_kv_new[None]], 0)
        st["k_rope"] = torch.cat([st["k_rope"], k_rope_new[None]], 0)
        q_absorb = torch.empty(H, cfg.kv_lora, dtype=torch.bfloat16, device=x.device)
        BN_Q = 128
        grid_q = (H, triton.cdiv(cfg.kv_lora, BN_Q))
        _mla_q_absorb_kernel[grid_q](
            q_nope, self.kv_b.w_q, self.kv_b.scales, self.kv_b.zeros, q_absorb,
            qk_nope=cfg.qk_nope, kv_lora=cfg.kv_lora, n_heads=H,
            total_N=H * (cfg.qk_nope + cfg.v_head), G=cfg.group, BLOCK_N=BN_Q,
        )
        scores_nope = torch.einsum("hd,td->ht", q_absorb.float(), st["c_kv"].float())
        scores_rope = torch.einsum("hd,td->ht", q_rope, st["k_rope"].float())
        scores = (scores_nope + scores_rope) * self.scale
        attn_w = torch.softmax(scores, dim=1)
        c_kv_weighted = torch.einsum("ht,td->hd",
                                     attn_w.to(torch.bfloat16), st["c_kv"])
        o_absorb = torch.empty(H, cfg.v_head, dtype=torch.bfloat16, device=x.device)
        BN_O = 128
        grid_o = (H, triton.cdiv(cfg.v_head, BN_O))
        _mla_o_absorb_kernel[grid_o](
            c_kv_weighted, self.kv_b.w_q, self.kv_b.scales, self.kv_b.zeros,
            o_absorb,
            kv_lora=cfg.kv_lora, v_head=cfg.v_head, qk_nope=cfg.qk_nope,
            n_heads=H, total_N=H * (cfg.qk_nope + cfg.v_head), G=cfg.group,
            BLOCK_N=BN_O,
        )
        return self.o_proj(o_absorb.reshape(H * cfg.v_head))

# MoE
# --------------------------------------------------------------------------- #

class MoE(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(E, d, m, cfg.group)
        self.up = QuantExperts(E, d, m, cfg.group)
        self.down = QuantExperts(E, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)
        self._gu_wq = None

    def _get_fused_gate_up(self):
        if self._gu_wq is None:
            self._gu_wq = torch.cat([self.gate.w_q, self.up.w_q], dim=2).contiguous()
            self._gu_sc = torch.cat([self.gate.scales, self.up.scales], dim=2).contiguous()
            self._gu_zr = torch.cat([self.gate.zeros, self.up.zeros], dim=2).contiguous()
        return self._gu_wq, self._gu_sc, self._gu_zr

    def step(self, x: torch.Tensor) -> torch.Tensor:
        cfg = self.cfg
        m = cfg.moe_inter
        probs = torch.softmax(self.router(x).float(), dim=-1)
        w, idx = torch.topk(probs, cfg.n_active)
        w_norm = (w / (w.sum() + 1e-9) * cfg.routed_scaling).to(torch.bfloat16)

        gu_wq, gu_sc, gu_zr = self._get_fused_gate_up()
        gate_up = _batched_gemv(x, gu_wq, gu_sc, gu_zr,
                                idx, cfg.hidden, 2 * m, G=cfg.group, same_x=True)
        h_gate, h_up = gate_up[:, :m], gate_up[:, m:]
        hh = F.silu(h_gate.float()) * h_up.float()

        h_down = _batched_gemv(hh, self.down.w_q, self.down.scales,
                               self.down.zeros, idx,
                               m, cfg.hidden, G=cfg.group, same_x=False)
        routed = (w_norm[:, None] * h_down).sum(0)

        # Shared expert: dequant once, use bf16 matmul (small, fixed)
        sh_gate_w = _wbf_int4(self.s_gate.w_q[0], self.s_gate.scales[0],
                              self.s_gate.zeros[0], cfg.hidden)
        sh_up_w = _wbf_int4(self.s_up.w_q[0], self.s_up.scales[0],
                            self.s_up.zeros[0], cfg.hidden)
        sh_down_w = _wbf_int4(self.s_down.w_q[0], self.s_down.scales[0],
                              self.s_down.zeros[0], m)
        sh = F.silu((x @ sh_gate_w).float()) * (x @ sh_up_w).float()
        sh_down = (sh.to(torch.bfloat16) @ sh_down_w).to(torch.bfloat16)
        return (routed + sh_down).to(torch.bfloat16)


# --------------------------------------------------------------------------- #
# Block
# --------------------------------------------------------------------------- #

class Block(nn.Module):
    def __init__(self, cfg: Config, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)

    def step(self, x, st):
        h = x + self.attn.step(_rmsnorm(x, self.attn_norm), st)
        return h + self.moe.step(_rmsnorm(h, self.moe_norm))


# --------------------------------------------------------------------------- #
# Model
# --------------------------------------------------------------------------- #

class Model(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)

    def step(self, hidden, state):
        for i, blk in enumerate(self.blocks):
            hidden = blk.step(hidden, state[i])
        return hidden, state
