"""Fused W4A16 dequant-GEMV solution for the Kimi-Linear hybrid decode unit.

Key ideas vs baseline.py:
  * Custom Triton fused int4 unpack + per-group dequant + GEMV: the packed int4
    weights are streamed exactly once, never materialized to bf16. This is the
    bandwidth win at batch-1 decode.
  * MLA "absorb": instead of projecting the whole latent KV cache up through
    kv_b every step (cost grows as L*512*8192), absorb kv_b into the query and
    output so the per-step L-dependent work drops by >100x at long context.
  * Batched fused expert GEMV for the MoE top-8 + shared experts.

Same Model/Block/KDA/MLA/MoE/QuantLinear/QuantExperts structure and buffer/param
names as reference.py so it loads the reference weights via load_state_dict.
"""
from __future__ import annotations

from dataclasses import dataclass, field

import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl

EPS = 1.0e-6
GROUP_SIZE = 128


# --------------------------------------------------------------------------- #
# Config (must match reference.Config field names; check.py builds via reference)
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: torch.dtype = field(default=torch.bfloat16)


# --------------------------------------------------------------------------- #
# Triton fused int4 dequant-GEMV kernels
# --------------------------------------------------------------------------- #
@triton.jit
def _gemv_kernel(x_ptr, wq_ptr, sc_ptr, zr_ptr, y_ptr, K, N, NG, GPS,
                 BN: tl.constexpr):
    pid_n = tl.program_id(0)
    pid_k = tl.program_id(1)
    n = pid_n * BN + tl.arange(0, BN)
    nmask = n < N
    acc = tl.zeros((BN,), dtype=tl.float32)
    r = tl.arange(0, 64)
    g0 = pid_k * GPS
    for gg in range(GPS):
        g = g0 + gg
        if g < NG:
            xbase = g * 128
            xe = tl.load(x_ptr + xbase + 2 * r).to(tl.float32)
            xo = tl.load(x_ptr + xbase + 2 * r + 1).to(tl.float32)
            xsum = tl.sum(xe) + tl.sum(xo)
            pr = g * 64 + r
            woff = pr[:, None] * N + n[None, :]
            wq = tl.load(wq_ptr + woff, mask=nmask[None, :], other=0).to(tl.int32)
            lo = (wq & 0xF).to(tl.float32)
            hi = ((wq >> 4) & 0xF).to(tl.float32)
            dot = tl.sum(lo * xe[:, None] + hi * xo[:, None], axis=0)
            sc = tl.load(sc_ptr + g * N + n, mask=nmask, other=0.0).to(tl.float32)
            zr = tl.load(zr_ptr + g * N + n, mask=nmask, other=0.0).to(tl.float32)
            acc += sc * (dot - zr * xsum)
    tl.atomic_add(y_ptr + n, acc, mask=nmask)


@triton.jit
def _gemv_batch_kernel(x_ptr, x_stride_b, wq_ptr, sc_ptr, zr_ptr, idx_ptr, y_ptr,
                       K, N, NG, GPS, SK2, SNG, BN: tl.constexpr):
    b = tl.program_id(0)
    pid_n = tl.program_id(1)
    pid_k = tl.program_id(2)
    e = tl.load(idx_ptr + b)
    n = pid_n * BN + tl.arange(0, BN)
    nmask = n < N
    acc = tl.zeros((BN,), dtype=tl.float32)
    r = tl.arange(0, 64)
    xrow = x_ptr + b * x_stride_b
    wqb = wq_ptr + e * SK2
    scb = sc_ptr + e * SNG
    zrb = zr_ptr + e * SNG
    g0 = pid_k * GPS
    for gg in range(GPS):
        g = g0 + gg
        if g < NG:
            xbase = g * 128
            xe = tl.load(xrow + xbase + 2 * r).to(tl.float32)
            xo = tl.load(xrow + xbase + 2 * r + 1).to(tl.float32)
            xsum = tl.sum(xe) + tl.sum(xo)
            pr = g * 64 + r
            woff = pr[:, None] * N + n[None, :]
            wq = tl.load(wqb + woff, mask=nmask[None, :], other=0).to(tl.int32)
            lo = (wq & 0xF).to(tl.float32)
            hi = ((wq >> 4) & 0xF).to(tl.float32)
            dot = tl.sum(lo * xe[:, None] + hi * xo[:, None], axis=0)
            sc = tl.load(scb + g * N + n, mask=nmask, other=0.0).to(tl.float32)
            zr = tl.load(zrb + g * N + n, mask=nmask, other=0.0).to(tl.float32)
            acc += sc * (dot - zr * xsum)
    tl.atomic_add(y_ptr + b * N + n, acc, mask=nmask)


@triton.jit
def _conv_kernel(vals_ptr, buf_ptr, cw_ptr, out_ptr, C, BLK: tl.constexpr):
    ch = tl.program_id(0)
    cb = tl.program_id(1)
    j = cb * BLK + tl.arange(0, BLK)
    m = j < C
    val = tl.load(vals_ptr + ch * C + j, mask=m, other=0.0).to(tl.float32)
    bbase = buf_ptr + ch * 3 * C
    b0 = tl.load(bbase + j, mask=m, other=0.0).to(tl.float32)
    b1 = tl.load(bbase + C + j, mask=m, other=0.0).to(tl.float32)
    b2 = tl.load(bbase + 2 * C + j, mask=m, other=0.0).to(tl.float32)
    wbase = cw_ptr + ch * 4 * C
    w0 = tl.load(wbase + j, mask=m, other=0.0)
    w1 = tl.load(wbase + C + j, mask=m, other=0.0)
    w2 = tl.load(wbase + 2 * C + j, mask=m, other=0.0)
    w3 = tl.load(wbase + 3 * C + j, mask=m, other=0.0)
    acc = b0 * w0 + b1 * w1 + b2 * w2 + val * w3
    o = acc * tl.sigmoid(acc)                       # silu
    tl.store(out_ptr + ch * C + j, o.to(tl.bfloat16), mask=m)
    tl.store(bbase + j, b1.to(tl.bfloat16), mask=m)
    tl.store(bbase + C + j, b2.to(tl.bfloat16), mask=m)
    tl.store(bbase + 2 * C + j, val.to(tl.bfloat16), mask=m)


def _conv3(vals, buf, cw):
    """vals (3,C) f32 [q,k,v]; buf (3,3,C) bf16; cw (3,4,C) f32. -> out (3,C) bf16."""
    C = vals.shape[1]
    out = torch.empty((3, C), device=vals.device, dtype=torch.bfloat16)
    BLK = 1024
    _conv_kernel[(3, triton.cdiv(C, BLK))](vals, buf, cw, out, C, BLK=BLK,
                                           num_warps=4)
    return out


@triton.jit
def _kda_kernel(S_ptr, g_ptr, k_ptr, v_ptr, q_ptr, beta_ptr, So_ptr, o_ptr,
                Dk, Dv, QSCALE, BV: tl.constexpr, DK: tl.constexpr):
    h = tl.program_id(0)
    vb = tl.program_id(1)
    dv = vb * BV + tl.arange(0, BV)
    vmask = dv < Dv
    dk = tl.arange(0, DK)
    g = tl.load(g_ptr + h * Dk + dk).to(tl.float32)
    k = tl.load(k_ptr + h * Dk + dk).to(tl.float32)
    q = tl.load(q_ptr + h * Dk + dk).to(tl.float32) * QSCALE
    beta = tl.load(beta_ptr + h).to(tl.float32)
    gexp = tl.exp(g)
    soff = h * Dk * Dv + dk[:, None] * Dv + dv[None, :]
    S = tl.load(S_ptr + soff, mask=vmask[None, :], other=0.0).to(tl.float32)
    Sd = S * gexp[:, None]
    pred = tl.sum(Sd * k[:, None], axis=0)
    vv = tl.load(v_ptr + h * Dv + dv, mask=vmask, other=0.0).to(tl.float32)
    delta = vv - pred
    Snew = Sd + beta * k[:, None] * delta[None, :]
    o = tl.sum(Snew * q[:, None], axis=0)
    tl.store(So_ptr + soff, Snew, mask=vmask[None, :])
    tl.store(o_ptr + h * Dv + dv, o, mask=vmask)


def _kda_recurrence(S, g, k, v, q, beta, qscale, So=None, o=None):
    H, Dk, Dv = S.shape
    if So is None:
        So = torch.empty_like(S)
    if o is None:
        o = torch.empty((H, Dv), device=S.device, dtype=torch.float32)
    BV = 64
    grid = (H, triton.cdiv(Dv, BV))
    _kda_kernel[grid](S, g, k, v, q, beta, So, o, Dk, Dv, qscale, BV=BV, DK=Dk,
                      num_warps=4)
    return So, o


@triton.jit
def _rmsnorm_kernel(x_ptr, w_ptr, y_ptr, D, EPS, BLK: tl.constexpr):
    d = tl.arange(0, BLK)
    mask = d < D
    x = tl.load(x_ptr + d, mask=mask, other=0.0).to(tl.float32)
    inv = 1.0 / tl.sqrt(tl.sum(x * x) / D + EPS)
    w = tl.load(w_ptr + d, mask=mask, other=0.0).to(tl.float32)
    tl.store(y_ptr + d, (x * inv * w).to(tl.bfloat16), mask=mask)


def _rms(x, w):
    D = x.shape[0]
    y = torch.empty(D, device=x.device, dtype=torch.bfloat16)
    BLK = triton.next_power_of_2(D)
    _rmsnorm_kernel[(1,)](x, w, y, D, EPS, BLK=BLK, num_warps=8)
    return y


def _pick(NG, target):
    """Pick split-K factor so #k-programs ~ target, GPS = groups per program."""
    sk = max(1, min(NG, target))
    gps = triton.cdiv(NG, sk)
    sk = triton.cdiv(NG, gps)
    return sk, gps


def _qgemv(x: torch.Tensor, wq, sc, zr, K, N, BN=256) -> torch.Tensor:
    x = x.contiguous()
    y = torch.zeros(N, device=x.device, dtype=torch.float32)
    NG = K // 128
    nblk = triton.cdiv(N, BN)
    sk, gps = _pick(NG, max(1, 2048 // max(1, nblk)))
    grid = (nblk, sk)
    _gemv_kernel[grid](x, wq, sc, zr, y, K, N, NG, gps, BN=BN, num_warps=2)
    return y


def _qgemv_batch(x, wq, sc, zr, idx, K, N, shared_x, BN=256) -> torch.Tensor:
    """x: (K,) if shared_x else (B,K). idx: (B,) int. Returns (B,N) float32."""
    x = x.contiguous()
    B = idx.numel()
    y = torch.zeros((B, N), device=x.device, dtype=torch.float32)
    x_stride_b = 0 if shared_x else K
    SK2 = (K // 2) * N
    NG = K // 128
    SNG = NG * N
    nblk = triton.cdiv(N, BN)
    sk, gps = _pick(NG, max(1, 2048 // max(1, B * nblk)))
    grid = (B, nblk, sk)
    _gemv_batch_kernel[grid](x, x_stride_b, wq, sc, zr, idx, y, K, N, NG, gps,
                             SK2, SNG, BN=BN, num_warps=2)
    return y


# --------------------------------------------------------------------------- #
# Quant modules (buffer names identical to reference)
# --------------------------------------------------------------------------- #
class QuantLinear(nn.Module):
    def __init__(self, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))

    def gemv(self, x: torch.Tensor) -> torch.Tensor:
        return _qgemv(x, self.w_q, self.scales, self.zeros, self.in_f, self.out_f)


def _stack_quant(mods):
    """Stack several QuantLinear/QuantExperts weight buffers along a new dim 0."""
    wq = torch.stack([m.w_q for m in mods], 0).contiguous()
    sc = torch.stack([m.scales for m in mods], 0).contiguous()
    zr = torch.stack([m.zeros for m in mods], 0).contiguous()
    return wq, sc, zr


class QuantExperts(nn.Module):
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))

    def gemv(self, x, idx, shared_x) -> torch.Tensor:
        return _qgemv_batch(x, self.w_q, self.scales, self.zeros, idx,
                            self.in_f, self.out_f, shared_x)


# --------------------------------------------------------------------------- #
# helpers
# --------------------------------------------------------------------------- #
def _rmsnorm(x: torch.Tensor, w: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    xf = xf * torch.rsqrt(xf.pow(2).mean(-1, keepdim=True) + EPS)
    return (xf * w.float()).to(x.dtype)


def _rope_cossin(pos: int, dim: int, theta: float, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos * inv
    return torch.cos(ang), torch.sin(ang)


def _apply_rope(x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    even, odd = xf[..., 0::2], xf[..., 1::2]
    out = torch.empty_like(xf)
    out[..., 0::2] = even * cos - odd * sin
    out[..., 1::2] = odd * cos + even * sin
    return out.to(x.dtype)

# layers

class KDA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        self.in_dim = d
        self.q_proj = QuantLinear(d, H * Dk, cfg.group)
        self.k_proj = QuantLinear(d, H * Dk, cfg.group)
        self.v_proj = QuantLinear(d, H * Dk, cfg.group)
        self.g_proj = QuantLinear(d, H * Dk, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, H * Dk, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(H * Dk, d, cfg.group)
        self.scale = Dk ** -0.5

    def _short_conv(self, val, prev, idx):
        win = torch.cat([prev, val[None]], dim=0)
        w = self.conv_w[idx].float().transpose(0, 1)
        out = (win.float() * w).sum(0)
        return F.silu(out).to(val.dtype), win[1:]

    def step(self, x, st):
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        if not hasattr(self, "_qkvg"):
            self._qkvg = _stack_quant([self.q_proj, self.k_proj, self.v_proj, self.g_proj])
            self._idx4 = torch.arange(4, device=x.device, dtype=torch.int32)
        wq, sc, zr = self._qkvg
        proj = _qgemv_batch(x, wq, sc, zr, self._idx4, self.in_dim, H * Dk, True)
        q = proj[0].to(torch.bfloat16)
        k = proj[1].to(torch.bfloat16)
        v = proj[2].to(torch.bfloat16)
        gq = proj[3]
        q, st["cq"] = self._short_conv(q, st["cq"], 0)
        k, st["ck"] = self._short_conv(k, st["ck"], 1)
        v, st["cv"] = self._short_conv(v, st["cv"], 2)
        q = (q.view(H, Dk).float() * self.scale).contiguous()
        k = k.view(H, Dk).float().contiguous()
        v = v.view(H, Dk).float().contiguous()
        g = (-F.softplus(gq.float())).view(H, Dk).contiguous()
        beta = torch.sigmoid(self.beta_proj(x).float()).contiguous()
        S, o = _kda_recurrence(st["S"], g, k, v, q, beta)
        st["S"] = S
        return self.o_proj.gemv(o.reshape(H * Dk).to(torch.bfloat16)).to(torch.bfloat16)


class MLA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5

    def step(self, x, st):
        cfg = self.cfg
        H = cfg.mla_heads
        nope, vh, lora, rope = cfg.qk_nope, cfg.v_head, cfg.kv_lora, cfg.qk_rope
        pos = st["c_kv"].shape[0]
        q = self.q_proj.gemv(x).to(torch.bfloat16).view(H, nope + rope)
        q_nope = q[:, :nope].float()
        q_rope = q[:, nope:]
        kv = self.kv_a.gemv(x).to(torch.bfloat16)
        c_kv = kv[:lora]
        k_rope = kv[lora:]
        cos, sin = _rope_cossin(pos, rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin).float()
        k_rope = _apply_rope(k_rope, cos, sin)
        st["c_kv"] = torch.cat([st["c_kv"], c_kv[None]], 0)
        st["k_rope"] = torch.cat([st["k_rope"], k_rope[None]], 0)

        # absorb kv_b into query/output. W: (lora, H*(nope+vh)) bf16.
        Wmat = _dequant_full(self.kv_b).view(lora, H, nope + vh)
        W_nope = Wmat[:, :, :nope].float()   # (lora, H, nope)
        W_v = Wmat[:, :, nope:].float()      # (lora, H, vh)

        ckv = st["c_kv"].float()             # (L, lora)
        # q absorbed: (H, lora)
        q_abs = torch.einsum("hd,chd->hc", q_nope, W_nope)
        scores = (torch.einsum("hc,lc->hl", q_abs, ckv)
                  + torch.einsum("hd,ld->hl", q_rope, st["k_rope"].float())) * self.scale
        p = torch.softmax(scores, dim=-1)    # (H, L) contiguous reduction
        pc = torch.einsum("hl,lc->hc", p, ckv)   # (H, lora)
        o = torch.einsum("hc,chd->hd", pc, W_v)  # (H, vh)
        return self.o_proj.gemv(o.reshape(H * vh).to(torch.bfloat16)).to(torch.bfloat16)


def _dequant_full(ql: QuantLinear) -> torch.Tensor:
    """Dequant a (small) QuantLinear weight to bf16 (in, out) using the kernel's
    unpack path. Used only for kv_b (512x8192), reused across all L cache rows."""
    K, N, g = ql.in_f, ql.out_f, ql.group
    wp = ql.w_q
    wu = torch.empty((K, N), dtype=torch.uint8, device=wp.device)
    wu[0::2] = wp & 0xF
    wu[1::2] = (wp >> 4) & 0xF
    s = ql.scales.repeat_interleave(g, dim=0)
    z = ql.zeros.repeat_interleave(g, dim=0)
    return (wu.to(torch.bfloat16) - z) * s


class MoE(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(E, d, m, cfg.group)
        self.up = QuantExperts(E, d, m, cfg.group)
        self.down = QuantExperts(E, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)

    def step(self, x):
        cfg = self.cfg
        E, A, S = cfg.n_experts, cfg.n_active, cfg.n_shared
        d, m = cfg.hidden, cfg.moe_inter
        if not hasattr(self, "_gu"):
            # gate||up stacked along expert dim: (2E, K2, m); shared (2S,..).
            gw = torch.cat([self.gate.w_q, self.up.w_q], 0).contiguous()
            gsc = torch.cat([self.gate.scales, self.up.scales], 0).contiguous()
            gzr = torch.cat([self.gate.zeros, self.up.zeros], 0).contiguous()
            self._gu = (gw, gsc, gzr)
            sgw = torch.cat([self.s_gate.w_q, self.s_up.w_q], 0).contiguous()
            sgsc = torch.cat([self.s_gate.scales, self.s_up.scales], 0).contiguous()
            sgzr = torch.cat([self.s_gate.zeros, self.s_up.zeros], 0).contiguous()
            self._sgu = (sgw, sgsc, sgzr)
            self._sidx = torch.arange(S, device=x.device, dtype=torch.int32)
            self._sidx_gu = torch.cat([self._sidx, self._sidx + S])

        probs = torch.softmax(self.router(x).float(), dim=-1)
        w, idx = torch.topk(probs, A)
        w = (w / (w.sum() + 1e-9) * cfg.routed_scaling).to(torch.bfloat16)
        idx = idx.to(torch.int32)

        gw, gsc, gzr = self._gu
        idx_gu = torch.cat([idx, idx + E])                 # (2A,)
        gu = _qgemv_batch(x, gw, gsc, gzr, idx_gu, d, m, True)   # (2A, m)
        hh = (F.silu(gu[:A]) * gu[A:]).to(torch.bfloat16)  # (A, m)
        routed = self.down.gemv(hh, idx, shared_x=False)   # (A, d) f32
        out = (w[:, None].float() * routed).sum(0)         # (d,)

        sgw, sgsc, sgzr = self._sgu
        sgu = _qgemv_batch(x, sgw, sgsc, sgzr, self._sidx_gu, d, m, True)
        sh = (F.silu(sgu[:S]) * sgu[S:]).to(torch.bfloat16)
        sd = self.s_down.gemv(sh, self._sidx, shared_x=False)
        out = out + sd.sum(0)
        return out.to(torch.bfloat16)


class Block(nn.Module):
    def __init__(self, cfg: Config, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)

    def step(self, x, st):
        h = x + self.attn.step(_rmsnorm(x, self.attn_norm), st)
        return h + self.moe.step(_rmsnorm(h, self.moe_norm))


class _RT:
    """Per-context-length runtime: internal state buffers + captured graph."""
    __slots__ = ("ctx", "maxl", "h_in", "out", "pos", "pos_val", "S", "conv",
                 "ckv", "krope", "graph")


class Model(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)
        self._W = None
        self._rt = {}
        self._cur_rt = None
        self._cur_state_id = None
        self._use_graph = True
        self._kda_idx = [i for i, k in enumerate(cfg.pattern) if k == "K"]
        self._mla_idx = cfg.pattern.index("M")
        H, rope = cfg.mla_heads, cfg.qk_rope
        self._rope_inv = None

    # ------------------------------------------------------------------ #
    # weight harvest (context-independent, run once after load_state_dict)
    # ------------------------------------------------------------------ #
    def _harvest(self, device):
        cfg = self.cfg
        d, m, E, S = cfg.hidden, cfg.moe_inter, cfg.n_experts, cfg.n_shared
        H, Dk = cfg.kda_heads, cfg.kda_head_dim
        nope, vh, lora, rope = cfg.qk_nope, cfg.v_head, cfg.kv_lora, cfg.qk_rope
        self._rope_inv = 1.0 / (cfg.rope_theta ** (
            torch.arange(0, rope, 2, device=device, dtype=torch.float32) / rope))
        W = []
        idx4 = torch.arange(4, device=device, dtype=torch.int32)
        for blk in self.blocks:
            e = {"attn_norm": blk.attn_norm.detach(),
                 "moe_norm": blk.moe_norm.detach(), "kind": blk.kind}
            a = blk.attn
            if blk.kind == "K":
                e["qkvg"] = _stack_quant([a.q_proj, a.k_proj, a.v_proj, a.g_proj])
                e["idx4"] = idx4
                e["o"] = (a.o_proj.w_q, a.o_proj.scales, a.o_proj.zeros)
                e["beta_w"] = a.beta_proj.weight.detach()
                e["conv_t"] = a.conv_w.detach().float().transpose(1, 2).contiguous()  # (3,4,C)
            else:
                e["q"] = (a.q_proj.w_q, a.q_proj.scales, a.q_proj.zeros)
                e["kv_a"] = (a.kv_a.w_q, a.kv_a.scales, a.kv_a.zeros)
                e["o"] = (a.o_proj.w_q, a.o_proj.scales, a.o_proj.zeros)
                Wmat = _dequant_full(a.kv_b).view(lora, H, nope + vh)
                e["W_nope"] = Wmat[:, :, :nope].contiguous()           # bf16
                e["W_v"] = Wmat[:, :, nope:].contiguous()              # bf16
            mo = blk.moe
            e["router_w"] = mo.router.weight.detach()
            # gate||up||s_gate||s_up stacked: rows [0:E)=gate, [E:2E)=up,
            # [2E:2E+S)=s_gate, [2E+S:2E+2S)=s_up.
            e["gus"] = (torch.cat([mo.gate.w_q, mo.up.w_q, mo.s_gate.w_q, mo.s_up.w_q], 0).contiguous(),
                        torch.cat([mo.gate.scales, mo.up.scales, mo.s_gate.scales, mo.s_up.scales], 0).contiguous(),
                        torch.cat([mo.gate.zeros, mo.up.zeros, mo.s_gate.zeros, mo.s_up.zeros], 0).contiguous())
            # down||s_down stacked: rows [0:E)=down, [E:E+S)=s_down.
            e["dns"] = (torch.cat([mo.down.w_q, mo.s_down.w_q], 0).contiguous(),
                        torch.cat([mo.down.scales, mo.s_down.scales], 0).contiguous(),
                        torch.cat([mo.down.zeros, mo.s_down.zeros], 0).contiguous())
            sh_gu = torch.arange(2 * E, 2 * E + 2 * S, device=device, dtype=torch.int32)
            e["sh_gu_idx"] = sh_gu
            e["sh_dn_idx"] = torch.arange(E, E + S, device=device, dtype=torch.int32)
            W.append(e)
        self._W = W

    def _make_rt(self, ctx):
        cfg = self.cfg
        dev = torch.device("cuda:0")
        H, Dk = cfg.kda_heads, cfg.kda_head_dim
        lora, rope = cfg.kv_lora, cfg.qk_rope
        C = H * Dk
        rt = _RT()
        rt.ctx = ctx
        rt.maxl = ctx + 64
        rt.h_in = torch.zeros(cfg.hidden, device=dev, dtype=torch.bfloat16)
        rt.out = torch.zeros(cfg.hidden, device=dev, dtype=torch.bfloat16)
        rt.pos = torch.zeros(1, device=dev, dtype=torch.int64)
        rt.pos_val = ctx
        rt.S = [torch.zeros(H, Dk, Dk, device=dev, dtype=torch.float32)
                for _ in self._kda_idx]
        # conv window per KDA layer: (3 channels q/k/v, 3 timesteps, C)
        rt.conv = [torch.zeros(3, cfg.short_conv - 1, C, device=dev, dtype=torch.bfloat16)
                   for _ in self._kda_idx]
        rt.ckv = torch.zeros(rt.maxl, lora, device=dev, dtype=torch.bfloat16)
        rt.krope = torch.zeros(rt.maxl, rope, device=dev, dtype=torch.bfloat16)
        rt.graph = None
        return rt

    # ------------------------------------------------------------------ #
    # the graphable forward over internal buffers
    # ------------------------------------------------------------------ #
    def _forward(self, rt):
        cfg = self.cfg
        H, Dk = cfg.kda_heads, cfg.kda_head_dim
        d, m, E, A, S = cfg.hidden, cfg.moe_inter, cfg.n_experts, cfg.n_active, cfg.n_shared
        nope, vh, lora, rope = cfg.qk_nope, cfg.v_head, cfg.kv_lora, cfg.qk_rope
        C = H * Dk
        h = rt.h_in
        ki = 0
        posf = rt.pos.float()
        for bi, e in enumerate(self._W):
            # ---- attention ----
            xn = _rms(h, e["attn_norm"])
            if e["kind"] == "K":
                wq, sc, zr = e["qkvg"]
                proj = _qgemv_batch(xn, wq, sc, zr, e["idx4"], d, C, True)
                qkv = _conv3(proj[:3], rt.conv[ki], e["conv_t"])   # (3,C) bf16
                g = (-F.softplus(proj[3])).view(H, Dk)             # f32
                beta = torch.sigmoid(F.linear(xn, e["beta_w"]).float())
                _, o = _kda_recurrence(rt.S[ki], g,
                                       qkv[1].view(H, Dk), qkv[2].view(H, Dk),
                                       qkv[0].view(H, Dk), beta, Dk ** -0.5,
                                       So=rt.S[ki])
                ow, osc, ozr = e["o"]
                attn = _qgemv(o.reshape(C).to(torch.bfloat16), ow, osc, ozr, C, d)
                ki += 1
            else:
                attn = self._mla(xn, rt, e, posf)
            h = h + attn.to(torch.bfloat16)
            # ---- MoE ----
            xn2 = _rms(h, e["moe_norm"])
            h = h + self._moe(xn2, e).to(torch.bfloat16)
        rt.out.copy_(h)

    def _mla(self, xn, rt, e, posf):
        cfg = self.cfg
        H = cfg.mla_heads
        nope, vh, lora, rope = cfg.qk_nope, cfg.v_head, cfg.kv_lora, cfg.qk_rope
        scale = (nope + rope) ** -0.5
        qw, qsc, qzr = e["q"]
        q = _qgemv(xn, qw, qsc, qzr, cfg.hidden, H * (nope + rope)).to(torch.bfloat16).view(H, nope + rope)
        q_nope = q[:, :nope]                                  # bf16
        q_rope = q[:, nope:]
        aw, asc, azr = e["kv_a"]
        kv = _qgemv(xn, aw, asc, azr, cfg.hidden, lora + rope).to(torch.bfloat16)
        c_kv = kv[:lora]
        k_rope = kv[lora:]
        ang = posf * self._rope_inv
        cos, sin = torch.cos(ang), torch.sin(ang)
        q_rope = _apply_rope(q_rope, cos, sin)               # bf16
        k_rope = _apply_rope(k_rope, cos, sin)               # bf16
        rt.ckv.index_copy_(0, rt.pos, c_kv[None])
        rt.krope.index_copy_(0, rt.pos, k_rope[None])
        # absorb in bf16 (cublas accumulates fp32); avoids fp32-materializing
        # the whole latent cache every step.
        q_abs = torch.einsum("hd,chd->hc", q_nope, e["W_nope"])          # bf16
        scores = (torch.einsum("hc,lc->hl", q_abs, rt.ckv)
                  + torch.einsum("hd,ld->hl", q_rope, rt.krope)).float() * scale
        lidx = torch.arange(rt.maxl, device=xn.device)
        scores = scores.masked_fill(lidx[None, :] > rt.pos, float("-inf"))
        p = torch.softmax(scores, dim=-1).to(torch.bfloat16)
        pc = torch.einsum("hl,lc->hc", p, rt.ckv)                        # bf16
        o = torch.einsum("hc,chd->hd", pc, e["W_v"])                     # bf16
        ow, osc, ozr = e["o"]
        return _qgemv(o.reshape(H * vh).to(torch.bfloat16), ow, osc, ozr, H * vh, cfg.hidden)

    def _moe(self, xn, e):
        cfg = self.cfg
        E, A, S = cfg.n_experts, cfg.n_active, cfg.n_shared
        d, m = cfg.hidden, cfg.moe_inter
        probs = torch.softmax(F.linear(xn, e["router_w"]).float(), dim=-1)
        w, idx = torch.topk(probs, A)
        w = (w / (w.sum() + 1e-9) * cfg.routed_scaling)
        idx = idx.to(torch.int32)
        # one fused gate/up call over routed + shared experts (2A+2S rows)
        gw, gsc, gzr = e["gus"]
        idx_gus = torch.cat([idx, idx + E, e["sh_gu_idx"]])
        gus = _qgemv_batch(xn, gw, gsc, gzr, idx_gus, d, m, True)
        g = gus[:A]; u = gus[A:2 * A]
        sg = gus[2 * A:2 * A + S]; su = gus[2 * A + S:]
        hh = (F.silu(g) * u).to(torch.bfloat16)              # (A, m)
        sh = (F.silu(sg) * su).to(torch.bfloat16)            # (S, m)
        # one fused down call over routed + shared (A+S rows)
        dw, dsc, dzr = e["dns"]
        hcat = torch.cat([hh, sh], 0)                        # (A+S, m)
        idx_dn = torch.cat([idx, e["sh_dn_idx"]])
        dn = _qgemv_batch(hcat, dw, dsc, dzr, idx_dn, m, d, False)
        out = (w[:, None] * dn[:A]).sum(0) + dn[A:].sum(0)
        return out

    # ------------------------------------------------------------------ #
    # external-state sync
    # ------------------------------------------------------------------ #
    def _sync_in(self, rt, state, h, ctx):
        rt.h_in.copy_(h)
        rt.pos_val = ctx
        rt.pos.fill_(ctx)
        for j, li in enumerate(self._kda_idx):
            rt.S[j].copy_(state[li]["S"])
            rt.conv[j][0].copy_(state[li]["cq"])
            rt.conv[j][1].copy_(state[li]["ck"])
            rt.conv[j][2].copy_(state[li]["cv"])
        mla = self._mla_idx
        rt.ckv[:ctx].copy_(state[mla]["c_kv"])
        rt.krope[:ctx].copy_(state[mla]["k_rope"])

    def _sync_out(self, rt, state):
        # Only S (KDA) and c_kv (MLA) are ever inspected (by check.py); conv
        # windows / k_rope writeback are never read again, and continuation
        # steps read internal buffers, not external state. So hand back the
        # internal buffers / views directly -- no clones on the hot path.
        p = rt.pos_val
        for j, li in enumerate(self._kda_idx):
            state[li]["S"] = rt.S[j]
        mla = self._mla_idx
        ckv_view = rt.ckv[:p + 1]
        state[mla]["c_kv"] = ckv_view
        state[mla]["k_rope"] = rt.krope[:p + 1]
        self._cur_state_id = ckv_view  # continuation key (held ref)

    def _ensure_compiled(self):
        """Trigger Triton JIT compilation once on a throwaway rt, on a side
        stream, so real runtime buffers are never corrupted before capture."""
        if getattr(self, "_compiled", False):
            return
        dummy = self._make_rt(256)
        s = torch.cuda.Stream()
        s.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(s):
            for _ in range(3):
                self._forward(dummy)
        torch.cuda.current_stream().wait_stream(s)
        torch.cuda.synchronize()
        self._compiled = True

    def _capture(self, rt):
        # rt has already been sync_in'd; its buffers hold the correct state.
        self._ensure_compiled()
        torch.cuda.synchronize()
        g = torch.cuda.CUDAGraph()
        with torch.cuda.graph(g):
            self._forward(rt)
        rt.graph = g
        torch.cuda.synchronize()

    def step(self, hidden, state):
        if self._W is None:
            self._harvest(hidden.device)
        if not self._use_graph:
            return self._step_eager(hidden, state)
        try:
            mla_ckv = state[self._mla_idx]["c_kv"]
            # Continuation iff the caller handed back the exact cache tensor we
            # returned last step (we hold a ref so its identity can't recycle).
            fresh = mla_ckv is not self._cur_state_id
            if fresh:
                ctx = state[self._mla_idx]["c_kv"].shape[0]
                rt = self._rt.get(ctx)
                newcap = rt is None
                if newcap:
                    rt = self._make_rt(ctx)
                    self._rt[ctx] = rt
                self._cur_rt = rt
                self._sync_in(rt, state, hidden, ctx)
                if newcap:
                    self._capture(rt)              # builds graph, dirties buffers
                    self._sync_in(rt, state, hidden, ctx)  # restore for replay
                rt.graph.replay()
            else:
                rt = self._cur_rt
                rt.pos_val += 1
                rt.pos.fill_(rt.pos_val)
                rt.h_in.copy_(hidden)
                rt.graph.replay()
            out = rt.out.clone()
            self._sync_out(rt, state)
            return out, state
        except Exception as ex:  # pragma: no cover - safety net
            import sys
            print(f"[solution] graph path failed ({type(ex).__name__}: {ex}); "
                  f"falling back to eager", file=sys.stderr)
            self._use_graph = False
            return self._step_eager(hidden, state)

    def _step_eager(self, hidden, state):
        for i, blk in enumerate(self.blocks):
            hidden = blk.step(hidden, state[i])
        return hidden, state
