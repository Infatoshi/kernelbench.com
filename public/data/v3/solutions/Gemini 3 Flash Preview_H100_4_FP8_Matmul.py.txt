import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, M: int, K: int, N: int, use_e4m3: bool = True):
        super().__init__()
        self.M = M
        self.K = K
        self.N = N
        self.use_e4m3 = use_e4m3

        if use_e4m3:
            self.fp8_dtype = torch.float8_e4m3fn
            self.fp8_max = 448.0
        else:
            self.fp8_dtype = torch.float8_e5m2
            self.fp8_max = 57344.0

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.weight = nn.Parameter(torch.randn(K, N) * 0.02)
        torch.random.set_rng_state(rng_state)
        
        self.register_buffer("w_fp8_t", None)
        self.register_buffer("w_inv_scale", None)

    def _prepare_weights(self, device):
        if self.w_fp8_t is None or self.w_fp8_t.device != device:
            w_t = self.weight.t().contiguous()
            w_amax = w_t.abs().max()
            w_scale = self.fp8_max / w_amax.clamp(min=1e-12)
            self.w_fp8_t = (w_t * w_scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)
            self.w_inv_scale = (1.0 / w_scale).to(torch.float32).view(1)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        self._prepare_weights(x.device)
        
        B, L, K = x.shape
        x_2d = x.view(-1, K)
        
        # Using tensors for scales to avoid synchronization
        x_amax = x_2d.abs().max()
        x_scale = self.fp8_max / x_amax.clamp(min=1e-12)
        x_inv_scale = (1.0 / x_scale).to(torch.float32).view(1)
        
        x_fp8 = (x_2d * x_scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)
        
        out = torch._scaled_mm(
            x_fp8,
            self.w_fp8_t.t(),
            scale_a=x_inv_scale,
            scale_b=self.w_inv_scale,
            out_dtype=x.dtype,
        )
        
        return out.view(B, L, self.N)

def get_inputs():
    return [torch.randn(8, 2048, 4096, device='cuda', dtype=torch.float16)]

def get_init_inputs():
    return [8 * 2048, 4096, 4096, True]
