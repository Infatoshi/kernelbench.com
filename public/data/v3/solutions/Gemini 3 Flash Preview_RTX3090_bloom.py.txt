import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def _bloom_kernel(
    image_ptr, output_ptr,
    batch, channels, height, width,
    stride_b, stride_c, stride_h, stride_w,
    threshold, intensity, kernel_size,
    half_k,
    BLOCK_SIZE_H: tl.constexpr, BLOCK_SIZE_W: tl.constexpr
):
    pid_h = tl.program_id(0)
    pid_w = tl.program_id(1)
    pid_bc = tl.program_id(2)
    
    b_idx = pid_bc // channels
    c_idx = pid_bc % channels

    base_ptr = image_ptr + b_idx * stride_b + c_idx * stride_c
    out_base_ptr = output_ptr + b_idx * stride_b + c_idx * stride_c

    offs_h = pid_h * BLOCK_SIZE_H + tl.arange(0, BLOCK_SIZE_H)
    offs_w = pid_w * BLOCK_SIZE_W + tl.arange(0, BLOCK_SIZE_W)
    
    sum_bright = tl.zeros((BLOCK_SIZE_H, BLOCK_SIZE_W), dtype=tl.float32)

    # Slighly better reuse pattern for the convolution-like operation
    for r in range(-half_k, half_k + 1):
        curr_h = offs_h + r
        h_mask = (curr_h >= 0) & (curr_h < height)
        for c in range(-half_k, half_k + 1):
            curr_w = offs_w + c
            w_mask = (curr_w >= 0) & (curr_w < width)
            
            mask = h_mask[:, None] & w_mask[None, :]
            val = tl.load(base_ptr + curr_h[:, None] * stride_h + curr_w[None, :] * stride_w, mask=mask, other=0.0)
            bright = tl.maximum(val - threshold, 0.0)
            sum_bright += bright
            
    blurred = sum_bright / (kernel_size * kernel_size)
    
    mask_hw = (offs_h[:, None] < height) & (offs_w[None, :] < width)
    img_val = tl.load(base_ptr + offs_h[:, None] * stride_h + offs_w[None, :] * stride_w, mask=mask_hw)
    
    res = img_val + intensity * blurred
    res = tl.where(res < 0.0, 0.0, res)
    res = tl.where(res > 10.0, 10.0, res)
    
    tl.store(out_base_ptr + offs_h[:, None] * stride_h + offs_w[None, :] * stride_w, res, mask=mask_hw)

class Model(nn.Module):
    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel
        self.half_k = blur_kernel // 2

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        if not image.is_cuda:
            image = image.cuda()
        
        batch, channels, height, width = image.shape
        output = torch.empty_like(image)
        
        # Grid settings
        BLOCK_SIZE_H = 16
        BLOCK_SIZE_W = 16
        
        grid = (
            triton.cdiv(height, BLOCK_SIZE_H),
            triton.cdiv(width, BLOCK_SIZE_W),
            batch * channels
        )
        
        _bloom_kernel[grid](
            image, output,
            batch, channels, height, width,
            image.stride(0), image.stride(1), image.stride(2), image.stride(3),
            float(self.threshold), float(self.intensity), int(self.blur_kernel),
            int(self.half_k),
            BLOCK_SIZE_H=BLOCK_SIZE_H, BLOCK_SIZE_W=BLOCK_SIZE_W,
            num_warps=4, num_stages=2
        )
        return output

def get_inputs():
    return [torch.randn(2, 3, 1024, 1024, device='cuda') * 1.5 + 0.2]

def get_init_inputs():
    return [1.0, 0.6, 5]
