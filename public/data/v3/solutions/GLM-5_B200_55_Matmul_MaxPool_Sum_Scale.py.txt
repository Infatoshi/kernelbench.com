import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_matmul_maxpool_sum_scale_kernel(
    x_ptr, w_ptr, bias_ptr, output_ptr,
    batch_size, in_features, out_features, kernel_size, scale_factor,
    stride_xb, stride_xi,
    stride_wo, stride_wi,
    BLOCK_K: tl.constexpr,
):
    """
    Fused kernel that computes:
    1. MatMul: output[b, o] = x[b, :] @ w[o, :] + bias[o]
    2. MaxPool1d: pool[b, p] = max(output[b, 2p], output[b, 2p+1])
    3. Sum: result[b] = sum(pool[b, :])
    4. Scale: result[b] = result[b] * scale_factor
    """
    pid = tl.program_id(0)
    
    if pid >= batch_size:
        return
    
    # Number of output pairs after maxpool
    out_pairs = out_features // kernel_size
    
    # Accumulator for final sum (after maxpool)
    sum_acc = 0.0
    
    # Process output pairs (for maxpool with kernel_size=2)
    for pair_idx in range(out_pairs):
        out_idx_0 = pair_idx * kernel_size      # e.g., 0, 2, 4, ...
        out_idx_1 = pair_idx * kernel_size + 1  # e.g., 1, 3, 5, ...
        
        # Accumulators for two consecutive output features
        acc_0 = 0.0
        acc_1 = 0.0
        
        # Compute dot product for both output features
        for k_start in range(0, in_features, BLOCK_K):
            k_offsets = k_start + tl.arange(0, BLOCK_K)
            k_mask = k_offsets < in_features
            
            # Load input x[pid, k_start:k_start+BLOCK_K]
            x = tl.load(x_ptr + pid * stride_xb + k_offsets * stride_xi, mask=k_mask, other=0.0)
            
            # Load weights for output feature 0: w[out_idx_0, k_start:k_start+BLOCK_K]
            w_0 = tl.load(w_ptr + out_idx_0 * stride_wo + k_offsets * stride_wi, mask=k_mask, other=0.0)
            acc_0 += tl.sum(x * w_0).to(tl.float32)
            
            # Load weights for output feature 1: w[out_idx_1, k_start:k_start+BLOCK_K]
            w_1 = tl.load(w_ptr + out_idx_1 * stride_wo + k_offsets * stride_wi, mask=k_mask, other=0.0)
            acc_1 += tl.sum(x * w_1).to(tl.float32)
        
        # Add bias
        if bias_ptr is not None:
            acc_0 += tl.load(bias_ptr + out_idx_0).to(tl.float32)
            acc_1 += tl.load(bias_ptr + out_idx_1).to(tl.float32)
        
        # Maxpool: take max of the two consecutive features
        sum_acc += tl.maximum(acc_0, acc_1)
    
    # Handle remaining output feature if out_features is not divisible by kernel_size
    remaining_idx = out_pairs * kernel_size
    if remaining_idx < out_features:
        acc = 0.0
        for k_start in range(0, in_features, BLOCK_K):
            k_offsets = k_start + tl.arange(0, BLOCK_K)
            k_mask = k_offsets < in_features
            x = tl.load(x_ptr + pid * stride_xb + k_offsets * stride_xi, mask=k_mask, other=0.0)
            w = tl.load(w_ptr + remaining_idx * stride_wo + k_offsets * stride_wi, mask=k_mask, other=0.0)
            acc += tl.sum(x * w).to(tl.float32)
        if bias_ptr is not None:
            acc += tl.load(bias_ptr + remaining_idx).to(tl.float32)
        sum_acc += acc
    
    # Scale and store result
    tl.store(output_ptr + pid, sum_acc * scale_factor)


def fused_matmul_maxpool_sum_scale(x, weight, bias, kernel_size, scale_factor):
    """
    Fused operation: matmul -> maxpool -> sum -> scale
    """
    batch_size, in_features = x.shape
    out_features = weight.shape[0]
    
    output = torch.empty(batch_size, dtype=torch.float32, device=x.device)
    
    grid = (batch_size,)
    
    # Choose block size based on input features
    BLOCK_K = 128 if in_features >= 128 else in_features
    
    fused_matmul_maxpool_sum_scale_kernel[grid](
        x, weight, bias, output,
        batch_size, in_features, out_features, kernel_size, scale_factor,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        BLOCK_K=BLOCK_K,
    )
    
    return output


class Model(nn.Module):
    """
    Model that performs matrix multiplication, max pooling, sum, and scaling.
    """
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        
        # Store weight and bias like nn.Linear
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize like nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size,).
        """
        return fused_matmul_maxpool_sum_scale(
            x, self.weight, self.bias, self.kernel_size, self.scale_factor
        )


batch_size = 128
in_features = 4096
out_features = 4096
kernel_size = 2
scale_factor = 0.5


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, kernel_size, scale_factor]
