import torch
import torch.nn as nn
import triton
import triton.language as tl

# Fused kernel: GroupNorm + Scale + MaxPool2d(2x2) + Clamp
# Input: (B, C, H, W) - conv output
# Output: (B, C, H//2, W//2) - after maxpool
#
# We run one block per (batch, group), reading C_per_group * H * W elements.
# Each block:
#   1. Computes mean and variance across the group
#   2. Normalizes each element, applies scale*weight+bias, multiplies by scale param
#   3. Writes maxpool-reduced result

@triton.jit
def groupnorm_scale_maxpool_clamp_kernel(
    # Input tensor (B, C, H, W) - row-major
    inp_ptr,
    # GroupNorm params
    gn_weight_ptr,  # (C,)
    gn_bias_ptr,    # (C,)
    # Scale param (C,)
    scale_ptr,
    # Output (B, C, H//pool, W//pool)
    out_ptr,
    # Dimensions
    B: tl.constexpr,
    C: tl.constexpr,
    H: tl.constexpr,
    W: tl.constexpr,
    G: tl.constexpr,       # num_groups
    pool: tl.constexpr,    # maxpool kernel size (stride=pool)
    clamp_min: tl.constexpr,
    clamp_max: tl.constexpr,
    eps: tl.constexpr,
    # derived
    C_per_G: tl.constexpr,
    H_out: tl.constexpr,
    W_out: tl.constexpr,
    # block sizes
    BLOCK_ELEMS: tl.constexpr,   # elements per block for reduction
):
    # Grid: (B * G,) blocks
    # Each block handles one (batch, group) pair
    pid = tl.program_id(0)
    b = pid // G
    g = pid % G

    # Channel range for this group
    c_start = g * C_per_G

    # Total elements in this group: C_per_G * H * W
    total = C_per_G * H * W

    # Base pointer for this batch
    batch_offset = b * C * H * W

    # Pass 1: compute mean and variance
    sum_val = tl.zeros([1], dtype=tl.float32)
    sum_sq  = tl.zeros([1], dtype=tl.float32)

    # We iterate over blocks of BLOCK_ELEMS elements
    for block_start in range(0, total, BLOCK_ELEMS):
        offs = block_start + tl.arange(0, BLOCK_ELEMS)
        mask = offs < total

        # Map linear index to (c_local, h, w)
        c_local = offs // (H * W)
        hw = offs % (H * W)
        h_idx = hw // W
        w_idx = hw % W
        c_abs = c_start + c_local

        linear_idx = batch_offset + c_abs * H * W + h_idx * W + w_idx
        x = tl.load(inp_ptr + linear_idx, mask=mask, other=0.0).to(tl.float32)

        sum_val += tl.sum(tl.where(mask, x, 0.0))
        sum_sq  += tl.sum(tl.where(mask, x * x, 0.0))

    mean = sum_val[0] / total
    var  = sum_sq[0] / total - mean * mean
    inv_std = tl.rsqrt(var + eps)

    # Pass 2: normalize, scale, maxpool, clamp, write output
    # Output dims per group: C_per_G * H_out * W_out
    out_batch_offset = b * C * H_out * W_out

    for oh in range(0, H_out):
        for ow in range(0, W_out):
            # MaxPool 2x2: gather pool*pool values for each channel
            for c_local in range(0, C_per_G):
                c_abs = c_start + c_local

                # Load gn_weight, gn_bias, scale for this channel
                gn_w = tl.load(gn_weight_ptr + c_abs).to(tl.float32)
                gn_b = tl.load(gn_bias_ptr + c_abs).to(tl.float32)
                sc   = tl.load(scale_ptr + c_abs).to(tl.float32)

                max_val = -1e38

                for ph in tl.static_range(0, pool):
                    for pw in tl.static_range(0, pool):
                        ih = oh * pool + ph
                        iw = ow * pool + pw
                        idx = batch_offset + c_abs * H * W + ih * W + iw
                        raw = tl.load(inp_ptr + idx).to(tl.float32)
                        normed = (raw - mean) * inv_std * gn_w + gn_b
                        scaled = normed * sc
                        max_val = tl.maximum(max_val, scaled)

                # Clamp
                clamped = tl.minimum(tl.maximum(max_val, clamp_min), clamp_max)

                out_idx = out_batch_offset + c_abs * H_out * W_out + oh * W_out + ow
                tl.store(out_ptr + out_idx, clamped)


# A more efficient version using vectorized loads with tl.arange for the maxpool dimension
@triton.jit
def groupnorm_scale_maxpool_clamp_v2(
    inp_ptr,
    gn_weight_ptr,
    gn_bias_ptr,
    scale_ptr,
    out_ptr,
    B: tl.constexpr,
    C: tl.constexpr,
    H: tl.constexpr,
    W: tl.constexpr,
    G: tl.constexpr,
    pool: tl.constexpr,
    clamp_min: tl.constexpr,
    clamp_max: tl.constexpr,
    eps: tl.constexpr,
    C_per_G: tl.constexpr,
    H_out: tl.constexpr,
    W_out: tl.constexpr,
    BLOCK_HW: tl.constexpr,  # number of output HW pixels per block iteration
):
    # Grid: (B * G * C_per_G, cdiv(H_out * W_out, BLOCK_HW))
    pid0 = tl.program_id(0)
    pid1 = tl.program_id(1)

    bgc = pid0
    b = bgc // (G * C_per_G)
    gc = bgc % (G * C_per_G)
    g = gc // C_per_G
    c_local = gc % C_per_G
    c_abs = g * C_per_G + c_local

    batch_offset = b * C * H * W
    c_offset = c_abs * H * W

    # Total elements for this (b, g): C_per_G * H * W
    total_group = C_per_G * H * W

    # First compute mean/var for this group
    # Each channel-block participates in the group reduction
    # BUT: all C_per_G channels for same (b,g) need the SAME mean/var
    # We need to share mean/var across c_local within same (b,g)
    # This is tricky in Triton without shared memory primitives
    # Alternative: pre-compute mean/var in a separate pass

    # For simplicity, each (b, g, c_local) block recomputes the full group stats
    # This is redundant but avoids complex synchronization
    c_grp_start = g * C_per_G

    sum_val = tl.zeros([1], dtype=tl.float32)
    sum_sq  = tl.zeros([1], dtype=tl.float32)

    # Iterate over all elements in the group
    BLOCK_RED: tl.constexpr = 256
    group_total = C_per_G * H * W
    for blk in range(0, tl.cdiv(group_total, BLOCK_RED)):
        offs = blk * BLOCK_RED + tl.arange(0, BLOCK_RED)
        mask = offs < group_total
        cg_local = offs // (H * W)
        hw = offs % (H * W)
        h_idx = hw // W
        w_idx = hw % W
        c_g = c_grp_start + cg_local
        idx = batch_offset + c_g * H * W + h_idx * W + w_idx
        x = tl.load(inp_ptr + idx, mask=mask, other=0.0).to(tl.float32)
        sum_val += tl.sum(tl.where(mask, x, 0.0))
        sum_sq  += tl.sum(tl.where(mask, x * x, 0.0))

    mean = sum_val[0] / group_total
    var  = sum_sq[0] / group_total - mean * mean
    inv_std = tl.rsqrt(var + eps)

    gn_w = tl.load(gn_weight_ptr + c_abs).to(tl.float32)
    gn_b = tl.load(gn_bias_ptr + c_abs).to(tl.float32)
    sc   = tl.load(scale_ptr + c_abs).to(tl.float32)

    # Now handle output pixels for this channel
    out_batch_offset = b * C * H_out * W_out
    out_c_offset = c_abs * H_out * W_out
    hw_out_total = H_out * W_out

    # Process BLOCK_HW output pixels
    hw_start = pid1 * BLOCK_HW
    hw_offs = hw_start + tl.arange(0, BLOCK_HW)
    mask_hw = hw_offs < hw_out_total

    oh = hw_offs // W_out
    ow = hw_offs % W_out

    # MaxPool 2x2: find max of 4 input pixels
    max_val = tl.full([BLOCK_HW], -1e38, dtype=tl.float32)
    for ph in tl.static_range(0, pool):
        for pw in tl.static_range(0, pool):
            ih = oh * pool + ph
            iw = ow * pool + pw
            idx = batch_offset + c_abs * H * W + ih * W + iw
            raw = tl.load(inp_ptr + idx, mask=mask_hw, other=0.0).to(tl.float32)
            normed = (raw - mean) * inv_std * gn_w + gn_b
            scaled = normed * sc
            max_val = tl.maximum(max_val, scaled)

    clamped = tl.minimum(tl.maximum(max_val, clamp_min), clamp_max)
    out_idx = out_batch_offset + out_c_offset + hw_offs
    tl.store(out_ptr + out_idx, clamped, mask=mask_hw)


# Most efficient version: separate mean/var kernel + fused normalize/maxpool/clamp
@triton.jit
def compute_group_stats_kernel(
    inp_ptr,
    mean_ptr,
    rstd_ptr,
    B: tl.constexpr,
    C: tl.constexpr,
    H: tl.constexpr,
    W: tl.constexpr,
    G: tl.constexpr,
    eps: tl.constexpr,
    C_per_G: tl.constexpr,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // G
    g = pid % G
    c_start = g * C_per_G
    batch_offset = b * C * H * W
    group_total = C_per_G * H * W

    acc = tl.zeros([1], dtype=tl.float32)
    acc2 = tl.zeros([1], dtype=tl.float32)

    for blk in range(0, tl.cdiv(group_total, BLOCK)):
        offs = blk * BLOCK + tl.arange(0, BLOCK)
        mask = offs < group_total
        c_local = offs // (H * W)
        hw = offs % (H * W)
        c_abs = c_start + c_local
        idx = batch_offset + c_abs * H * W + hw
        x = tl.load(inp_ptr + idx, mask=mask, other=0.0).to(tl.float32)
        acc  += tl.sum(tl.where(mask, x, 0.0))
        acc2 += tl.sum(tl.where(mask, x * x, 0.0))

    mean = tl.sum(acc) / group_total
    var  = tl.sum(acc2) / group_total - mean * mean
    rstd = tl.rsqrt(var + eps)

    tl.store(mean_ptr + pid, mean)
    tl.store(rstd_ptr + pid, rstd)


@triton.jit
def normalize_maxpool_clamp_kernel(
    inp_ptr,
    mean_ptr,
    rstd_ptr,
    gn_weight_ptr,
    gn_bias_ptr,
    scale_ptr,
    out_ptr,
    B: tl.constexpr,
    C: tl.constexpr,
    H: tl.constexpr,
    W: tl.constexpr,
    G: tl.constexpr,
    pool: tl.constexpr,
    clamp_min: tl.constexpr,
    clamp_max: tl.constexpr,
    C_per_G: tl.constexpr,
    H_out: tl.constexpr,
    W_out: tl.constexpr,
    BLOCK_HW: tl.constexpr,
):
    # Grid: (B * C, cdiv(H_out*W_out, BLOCK_HW))
    pid0 = tl.program_id(0)
    pid1 = tl.program_id(1)

    b = pid0 // C
    c = pid0 % C
    g = c // C_per_G

    # Load mean/rstd for this (b, g)
    stats_idx = b * G + g
    mean = tl.load(mean_ptr + stats_idx)
    rstd = tl.load(rstd_ptr + stats_idx)

    gn_w = tl.load(gn_weight_ptr + c).to(tl.float32)
    gn_b = tl.load(gn_bias_ptr + c).to(tl.float32)
    sc   = tl.load(scale_ptr + c).to(tl.float32)

    # Combined affine: (x - mean) * rstd * gn_w + gn_b) * sc
    # = x * (rstd * gn_w * sc) + (gn_b * sc - mean * rstd * gn_w * sc)
    alpha = rstd * gn_w * sc
    beta  = (gn_b - mean * rstd * gn_w) * sc

    hw_out_total = H_out * W_out
    hw_start = pid1 * BLOCK_HW
    hw_offs = hw_start + tl.arange(0, BLOCK_HW)
    mask_hw = hw_offs < hw_out_total

    oh = hw_offs // W_out
    ow = hw_offs % W_out

    batch_offset = b * C * H * W
    max_val = tl.full([BLOCK_HW], -1e38, dtype=tl.float32)

    for ph in tl.static_range(0, pool):
        for pw in tl.static_range(0, pool):
            ih = oh * pool + ph
            iw = ow * pool + pw
            idx = batch_offset + c * H * W + ih * W + iw
            raw = tl.load(inp_ptr + idx, mask=mask_hw, other=0.0).to(tl.float32)
            val = raw * alpha + beta
            max_val = tl.maximum(max_val, val)

    clamped = tl.minimum(tl.maximum(max_val, clamp_min), clamp_max)
    out_batch_offset = b * C * H_out * W_out
    out_idx = out_batch_offset + c * H_out * W_out + hw_offs
    tl.store(out_ptr + out_idx, clamped, mask=mask_hw)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool = nn.MaxPool2d(kernel_size=maxpool_kernel_size)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

        self.out_channels = out_channels
        self.num_groups = num_groups
        self.pool = maxpool_kernel_size

    def forward(self, x):
        # Step 1: Conv2d (use aten op directly - cuDNN backend)
        x = torch.ops.aten.conv2d(
            x,
            self.conv.weight,
            self.conv.bias,
            [1, 1],   # stride
            [0, 0],   # padding
            [1, 1],   # dilation
            1         # groups
        )

        B, C, H, W = x.shape
        G = self.num_groups
        C_per_G = C // G
        pool = self.pool
        H_out = H // pool
        W_out = W // pool

        # Ensure contiguous float32
        x = x.contiguous()

        # Allocate stats and output buffers
        mean_buf = torch.empty(B * G, device=x.device, dtype=torch.float32)
        rstd_buf = torch.empty(B * G, device=x.device, dtype=torch.float32)
        out = torch.empty(B, C, H_out, W_out, device=x.device, dtype=x.dtype)

        # Get GroupNorm weight/bias as float32
        gn_w = self.group_norm.weight.to(torch.float32).contiguous()
        gn_b = self.group_norm.bias.to(torch.float32).contiguous()
        sc   = self.scale.view(C).to(torch.float32).contiguous()

        # Kernel 1: compute group stats
        BLOCK_RED = 512
        grid_stats = (B * G,)
        compute_group_stats_kernel[grid_stats](
            x, mean_buf, rstd_buf,
            B, C, H, W, G, 1e-5, C_per_G,
            BLOCK=BLOCK_RED,
        )

        # Kernel 2: normalize + maxpool + clamp
        BLOCK_HW = 128
        hw_out_total = H_out * W_out
        grid_norm = (B * C, triton.cdiv(hw_out_total, BLOCK_HW))
        normalize_maxpool_clamp_kernel[grid_norm](
            x, mean_buf, rstd_buf, gn_w, gn_b, sc, out,
            B, C, H, W, G, pool,
            self.clamp_min, self.clamp_max,
            C_per_G, H_out, W_out,
            BLOCK_HW=BLOCK_HW,
        )

        return out


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
