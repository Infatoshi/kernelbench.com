import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cuda_fp16.h>
#include <float.h>

template <typename T>
__device__ __forceinline__ float to_float(T val) { return (float)val; }

template <typename T>
__global__ void cross_entropy_kernel(const T* __restrict__ predictions, const int64_t* __restrict__ targets, float* __restrict__ output, int batch_size, int num_classes) {
    int row = blockIdx.x;
    if (row >= batch_size) return;

    int target = targets[row];
    const T* row_ptr = predictions + row * num_classes;

    // 1. Find max for numerical stability
    float row_max = -FLT_MAX;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        row_max = fmaxf(row_max, to_float(row_ptr[i]));
    }

    // Block reduction for max
    for (int offset = 16; offset > 0; offset /= 2) {
        row_max = fmaxf(row_max, __shfl_down_sync(0xFFFFFFFF, row_max, offset));
    }
    extern __shared__ float shared_mem[];
    if (threadIdx.x % 32 == 0) shared_mem[threadIdx.x / 32] = row_max;
    __syncthreads();
    if (threadIdx.x < 32) {
        row_max = (threadIdx.x < (blockDim.x / 32)) ? shared_mem[threadIdx.x] : -FLT_MAX;
        for (int offset = 16; offset > 0; offset /= 2) {
            row_max = fmaxf(row_max, __shfl_down_sync(0xFFFFFFFF, row_max, offset));
        }
        shared_mem[0] = row_max;
    }
    __syncthreads();
    row_max = shared_mem[0];

    // 2. Compute sum of exps and target val together
    float sum_exp = 0.0f;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        sum_exp += expf(to_float(row_ptr[i]) - row_max);
    }
    float target_val = to_float(row_ptr[target]);

    // Block reduction for sum
    for (int offset = 16; offset > 0; offset /= 2) {
        sum_exp += __shfl_down_sync(0xFFFFFFFF, sum_exp, offset);
    }
    if (threadIdx.x % 32 == 0) shared_mem[threadIdx.x / 32] = sum_exp;
    __syncthreads();
    if (threadIdx.x < 32) {
        sum_exp = (threadIdx.x < (blockDim.x / 32)) ? shared_mem[threadIdx.x] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2) {
            sum_exp += __shfl_down_sync(0xFFFFFFFF, sum_exp, offset);
        }
        shared_mem[0] = sum_exp;
    }
    __syncthreads();
    sum_exp = shared_mem[0];

    if (threadIdx.x == 0) {
        float log_sum_exp = logf(sum_exp);
        output[row] = log_sum_exp - (target_val - row_max);
    }
}

torch::Tensor cross_entropy_forward(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    auto losses = torch::empty({batch_size}, torch::TensorOptions().dtype(torch::kFloat32).device(predictions.device()));
    
    int threads = 256;
    int shared_mem_size = (threads / 32) * sizeof(float);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, predictions.scalar_type(), "cross_entropy_kernel", ([&] {
        cross_entropy_kernel<scalar_t><<<batch_size, threads, shared_mem_size>>>(
            predictions.data_ptr<scalar_t>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }));

    return losses.mean();
}
'''

ext = load_inline(
    name='ce_ext',
    cpp_sources='torch::Tensor cross_entropy_forward(torch::Tensor predictions, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [torch.randn(batch_size, num_classes, device='cuda'), torch.randint(0, num_classes, (batch_size,), device='cuda')]

def get_init_inputs():
    return []
