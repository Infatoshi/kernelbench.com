import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
    ],
    key=['Batch', 'In_Feat', 'Out_Feat'],
)
@triton.jit
def fused_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    Batch, In_Feat, Out_Feat,
    Scale,
    stride_xm, stride_xk,
    stride_wk, stride_wn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    Kernel_Size: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """
    Computes (X @ W + B), MaxPool1d, Sum, and Scale.
    Linear: (M, K) @ (K, N) + (N,) -> (M, N)
    MaxPool1d: (M, 1, N) -> (M, 1, N // Kernel_Size)
    Sum: (M, N // Kernel_Size) -> (M,)
    Scale: (M,) * Scale -> (M,)
    """
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(Batch, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(Out_Feat, BLOCK_SIZE_N)
    pid_m = pid % num_pid_m
    
    # We will compute the entire row sum for each BLOCK_SIZE_M
    # To avoid complex synchronization, let's process full rows (N) in one program or use atomic adds.
    # Actually, it's better to have each program handle a subset of M and the whole N reduction.
    
    rm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    m_mask = rm < Batch
    
    # Accumulator for the final sum of the scaled max-pooled values
    # Shape: (BLOCK_SIZE_M,)
    row_sum = tl.zeros((BLOCK_SIZE_M,), dtype=tl.float32)
    
    # MaxPool kernel size reduction
    # Each MaxPool window is Kernel_Size elements from the N dimension.
    # We iterate through the N dimension in steps of BLOCK_SIZE_N.
    # Assuming Out_Feat is divisible by Kernel_Size as per MaxPool1d behavior.
    
    # Note: MaxPool1d on (M, 1, N) with kernel_size K returns (M, 1, N // K).
    # Each output element is max(X[m, k*K : (k+1)*K]).
    # Since we then sum over the pooled dimension, we are summing N // K max values.
    
    # We must ensure BLOCK_SIZE_N is divisible by Kernel_Size.
    # To use tl.reshape, the dimension sizes must be constexpr.
    
    for n_start in range(0, Out_Feat, BLOCK_SIZE_N):
        # Compute Matmul tile (BLOCK_SIZE_M, BLOCK_SIZE_N)
        acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
        rn = n_start + tl.arange(0, BLOCK_SIZE_N)
        n_mask = rn < Out_Feat
        
        for k_start in range(0, In_Feat, BLOCK_SIZE_K):
            rk = k_start + tl.arange(0, BLOCK_SIZE_K)
            k_mask = rk < In_Feat
            
            x = tl.load(X_ptr + (rm[:, None] * stride_xm + rk[None, :] * stride_xk), mask=m_mask[:, None] & k_mask[None, :], other=0.0)
            w = tl.load(W_ptr + (rk[:, None] * stride_wk + rn[None, :] * stride_wn), mask=k_mask[:, None] & n_mask[None, :], other=0.0)
            acc += tl.dot(x, w)
            
        bias = tl.load(B_ptr + rn, mask=n_mask, other=0.0)
        acc += bias[None, :]
        
        # acc is (BLOCK_SIZE_M, BLOCK_SIZE_N).
        # To perform max pooling over chunks of size Kernel_Size:
        # We can't use tl.reshape with a variable Kernel_Size if it's not constexpr.
        # But Kernel_Size is passed as a parameter. Let's make it a constexpr.
        
        reshaped_acc = tl.reshape(acc, (BLOCK_SIZE_M, BLOCK_SIZE_N // Kernel_Size, Kernel_Size))
        pooled = tl.max(reshaped_acc, axis=2) # (BLOCK_SIZE_M, BLOCK_SIZE_N // Kernel_Size)
        
        row_sum += tl.sum(pooled, axis=1)
        
    out = row_sum * Scale
    tl.store(Out_ptr + rm, out, mask=m_mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        nn.init.kaiming_uniform_(self.weight, a=torch.sqrt(torch.tensor(5.0)).item())
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / torch.sqrt(torch.tensor(fan_in)).item()
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        batch_size = x.shape[0]
        out = torch.empty((batch_size,), device=x.device, dtype=torch.float32)
        
        # Grid and block sizes
        grid = lambda META: (triton.cdiv(batch_size, META['BLOCK_SIZE_M']),)
        
        fused_kernel[grid](
            x, self.weight.t(), self.bias, out,
            batch_size, self.in_features, self.out_features,
            self.scale_factor,
            x.stride(0), x.stride(1),
            self.weight.stride(1), self.weight.stride(0),
            Kernel_Size=self.kernel_size,
        )
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 2, 0.5]
