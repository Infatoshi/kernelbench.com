import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ w,
    const scalar_t* __restrict__ b,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    int K, int stride, int pad,
    int H_out, int W_out) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c = (idx / (W_out * H_out)) % C;
    int n = idx / (W_out * H_out * C);

    scalar_t sum = 0;

    for (int ky = 0; ky < K; ++ky) {
        for (int kx = 0; kx < K; ++kx) {
            int h_in = h_out * stride + ky - pad;
            int w_in = w_out * stride + kx - pad;

            if (h_in >= 0 && h_in < H && w_in >= 0 && w_in < W) {
                int x_idx = ((n * C + c) * H + h_in) * W + w_in;
                int w_idx = (c * K + ky) * K + kx;
                sum += x[x_idx] * w[w_idx];
            }
        }
    }

    if (b != nullptr) sum += b[c];

    int y_idx = ((n * C + c) * H_out + h_out) * W_out + w_out;
    y[y_idx] = sum;
}

Tensor depthwise_conv2d_forward(
    Tensor x, Tensor w, c10::optional<Tensor> b_opt,
    int stride, int pad) {

    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int K = w.size(2);

    int H_out = (H + 2 * pad - K) / stride + 1;
    int W_out = (W + 2 * pad - K) / stride + 1;

    auto y = at::empty({N, C, H_out, W_out}, x.options());

    int total = N * C * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    const bool has_bias = b_opt.has_value();

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "dwconv", ([&] {
        const scalar_t* b_ptr = has_bias ? b_opt.value().data_ptr<scalar_t>() : nullptr;

        depthwise_conv2d_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            w.data_ptr<scalar_t>(),
            b_ptr,
            y.data_ptr<scalar_t>(),
            N, C, H, W, K, stride, pad, H_out, W_out);
    }));

    return y;
}
'''

ext = load_inline(
    name='dwconv_ext',
    cpp_sources='torch::Tensor depthwise_conv2d_forward(torch::Tensor x, torch::Tensor w, c10::optional<torch::Tensor> b, int stride, int pad);',
    cuda_sources=cuda_src,
    functions=['depthwise_conv2d_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.depthwise_conv2d_forward(x, self.conv2d.weight, self.conv2d.bias, self.conv2d.stride[0], self.conv2d.padding[0])


def get_inputs():
    batch_size = 16
    in_channels = 3
    width = 256
    height = 256
    x = torch.randn(batch_size, in_channels, height, width, device='cuda')
    return [x]


def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
