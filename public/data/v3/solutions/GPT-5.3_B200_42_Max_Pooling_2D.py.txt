import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>


template <typename scalar_t>
__global__ void maxpool2d_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    int outH, int outW,
    int k, int stride, int pad, int dilation
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * outH * outW;
    if (idx >= total) return;

    int ow = idx % outW;
    int oh = (idx / outW) % outH;
    int c = (idx / (outW * outH)) % C;
    int n = idx / (outW * outH * C);

    int h_start = oh * stride - pad;
    int w_start = ow * stride - pad;

    scalar_t max_val = -1e20;

    for (int kh = 0; kh < k; ++kh) {
        for (int kw = 0; kw < k; ++kw) {
            int ih = h_start + kh * dilation;
            int iw = w_start + kw * dilation;

            if (ih >= 0 && ih < H && iw >= 0 && iw < W) {
                int in_idx = ((n * C + c) * H + ih) * W + iw;
                scalar_t v = x[in_idx];
                if (v > max_val) max_val = v;
            }
        }
    }

    y[idx] = max_val;
}


torch::Tensor maxpool2d(torch::Tensor x, int k, int stride, int pad, int dilation) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int outH = (H + 2*pad - dilation*(k-1) - 1)/stride + 1;
    int outW = (W + 2*pad - dilation*(k-1) - 1)/stride + 1;

    auto y = torch::empty({N, C, outH, outW}, x.options());

    int total = N * C * outH * outW;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "maxpool2d", ([&] {
        maxpool2d_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            N, C, H, W, outH, outW,
            k, stride, pad, dilation
        );
    }));

    return y;
}
'''

ext = load_inline(
    name='maxpool2d_ext',
    cpp_sources='torch::Tensor maxpool2d(torch::Tensor x, int k, int stride, int pad, int dilation);',
    cuda_sources=cuda_src,
    functions=['maxpool2d'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.k = kernel_size
        self.stride = stride
        self.pad = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d(x, self.k, self.stride, self.pad, self.dilation)


def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width, device='cuda')
    return [x]


def get_init_inputs():
    return [2, 2, 1, 3]
