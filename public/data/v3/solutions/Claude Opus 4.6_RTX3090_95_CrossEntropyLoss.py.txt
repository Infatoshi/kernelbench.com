import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def cross_entropy_kernel(
    logits_ptr,      # [batch_size, num_classes]
    targets_ptr,     # [batch_size]
    losses_ptr,      # [batch_size]
    num_classes: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one row (one sample in the batch)
    row_idx = tl.program_id(0)
    
    # Pointer to the start of this row
    row_start = logits_ptr + row_idx * num_classes
    
    # Load target for this row
    target = tl.load(targets_ptr + row_idx)
    
    # Since num_classes=1024 and BLOCK_SIZE=1024, single pass
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < num_classes
    x = tl.load(row_start + cols, mask=mask, other=float('-inf'))
    
    # Step 1: Find max for numerical stability
    m = tl.max(x, axis=0)
    
    # Step 2: Compute sum of exp(x - max)
    s = tl.sum(tl.exp(x - m), axis=0)
    
    # Step 3: Load the logit at the target position
    logit_target = tl.load(row_start + target)
    
    # Step 4: Compute cross-entropy loss for this sample
    loss = tl.log(s) + m - logit_target
    
    # Store per-sample loss
    tl.store(losses_ptr + row_idx, loss)


@triton.jit  
def reduce_mean_kernel(
    losses_ptr,
    output_ptr,
    N: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Single program to compute mean
    s = 0.0
    for off in tl.static_range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        x = tl.load(losses_ptr + cols, mask=mask, other=0.0)
        s += tl.sum(x, axis=0)
    tl.store(output_ptr, s / N)


def cross_entropy_triton(predictions, targets):
    batch_size, num_classes = predictions.shape
    
    # Allocate per-sample losses
    losses = torch.empty(batch_size, device=predictions.device, dtype=torch.float32)
    
    # Use predictions in float32 for accuracy
    if predictions.dtype != torch.float32:
        predictions = predictions.float()
    
    # Choose BLOCK_SIZE - power of 2 >= num_classes for efficiency
    BLOCK_SIZE = triton.next_power_of_2(num_classes)
    
    # Launch one program per batch element
    cross_entropy_kernel[(batch_size,)](
        predictions, targets, losses,
        num_classes=num_classes,
        BLOCK_SIZE=BLOCK_SIZE,
        num_warps=8,
    )
    
    # Compute mean
    output = torch.empty(1, device=predictions.device, dtype=torch.float32)
    
    reduce_mean_kernel[(1,)](
        losses, output,
        N=batch_size,
        BLOCK_SIZE=min(triton.next_power_of_2(batch_size), 4096),
        num_warps=8,
    )
    
    return output.squeeze()


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return cross_entropy_triton(predictions, targets)
