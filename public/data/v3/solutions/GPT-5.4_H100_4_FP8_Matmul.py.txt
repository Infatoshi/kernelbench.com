import torch
import torch.nn as nn

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ['fp4', 'fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['H100', 'B200']


class Model(nn.Module):
    def __init__(self, M: int, K: int, N: int, use_e4m3: bool = True):
        super().__init__()
        self.M = M
        self.K = K
        self.N = N
        self.use_e4m3 = use_e4m3

        if use_e4m3:
            self.fp8_dtype = torch.float8_e4m3fn
            self.fp8_max = 448.0
        else:
            self.fp8_dtype = torch.float8_e5m2
            self.fp8_max = 57344.0

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.weight = nn.Parameter(torch.randn(K, N) * 0.02)
        torch.random.set_rng_state(rng_state)

        self.register_buffer('w_scale_inv', torch.empty([], dtype=torch.float32), persistent=False)
        self.register_buffer('w_fp8_t', torch.empty(0, dtype=self.fp8_dtype), persistent=False)
        self._weight_version = -1

    def _refresh_weight_cache(self):
        version = self.weight._version
        if version == self._weight_version and self.w_fp8_t.numel() != 0:
            return
        with torch.no_grad():
            w = self.weight
            amax = w.abs().max()
            scale = self.fp8_max / amax.clamp(min=1e-12)
            w_t = w.t().contiguous()
            w_fp8 = (w_t * scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)
            self.w_fp8_t = w_fp8
            self.w_scale_inv = (1.0 / scale).to(torch.float32)
            self._weight_version = version

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        self._refresh_weight_cache()

        input_dtype = x.dtype
        batch_size, seq_len, _ = x.shape
        x_2d = x.reshape(-1, self.K)

        x_amax = x_2d.abs().max()
        x_scale = self.fp8_max / x_amax.clamp(min=1e-12)
        x_fp8 = (x_2d * x_scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)
        x_scale_inv = (1.0 / x_scale).to(torch.float32)

        out = torch._scaled_mm(
            x_fp8,
            self.w_fp8_t.t(),
            scale_a=x_scale_inv,
            scale_b=self.w_scale_inv,
            out_dtype=input_dtype,
        )
        return out.reshape(batch_size, seq_len, self.N)


batch_size = 8
seq_len = 2048
M = batch_size * seq_len
K = 4096
N = 4096
use_e4m3 = True


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [M, K, N, use_e4m3]
