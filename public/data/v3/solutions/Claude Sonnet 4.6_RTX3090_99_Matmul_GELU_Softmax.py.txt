import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CUTLASS_INCLUDE = "/home/infatoshi/cuda/KernelBench-v3/docs/refs/cutlass/include"

# -----------------------------------------------------------------------
# Fused Matmul + GELU + Softmax
# Strategy:
#   1. GEMM via CUTLASS (FP32 SIMT, bias via separate kernel)
#   2. Fused GELU + Softmax via custom CUDA kernel (one block per row)
# -----------------------------------------------------------------------

fused_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>
#include <math.h>

// -----------------------------------------------------------------------
// Fast GELU (erf-based, matching PyTorch default)
// -----------------------------------------------------------------------
__device__ __forceinline__ float gelu_f32(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.7071067811865476f));
}

// -----------------------------------------------------------------------
// Warp reductions
// -----------------------------------------------------------------------
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, offset));
    return val;
}

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, offset);
    return val;
}

// -----------------------------------------------------------------------
// Fused GELU + Softmax
// One block per row.  BLOCK_SIZE should be >= 32 and a multiple of 32.
// -----------------------------------------------------------------------
template <int BLOCK_SIZE>
__global__ void gelu_softmax_kernel(
    const float* __restrict__ input,
    float*       __restrict__ output,
    int rows, int cols)
{
    int row = blockIdx.x;
    if (row >= rows) return;

    const float* row_in  = input  + (long long)row * cols;
    float*       row_out = output + (long long)row * cols;

    int tid     = threadIdx.x;
    int lane_id = tid & 31;
    int warp_id = tid >> 5;
    int nwarps  = BLOCK_SIZE >> 5;

    __shared__ float smem[32];  // max 32 warps

    // ---- Pass 1: apply GELU and find row-wise max ----
    float thread_max = -FLT_MAX;
    for (int col = tid; col < cols; col += BLOCK_SIZE) {
        float v = gelu_f32(row_in[col]);
        row_out[col] = v;
        thread_max = fmaxf(thread_max, v);
    }

    thread_max = warp_reduce_max(thread_max);
    if (lane_id == 0) smem[warp_id] = thread_max;
    __syncthreads();

    if (warp_id == 0) {
        float v = (lane_id < nwarps) ? smem[lane_id] : -FLT_MAX;
        v = warp_reduce_max(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    float row_max = smem[0];

    // ---- Pass 2: exp(gelu - max) and sum ----
    float thread_sum = 0.0f;
    for (int col = tid; col < cols; col += BLOCK_SIZE) {
        float v = __expf(row_out[col] - row_max);
        row_out[col] = v;
        thread_sum += v;
    }

    thread_sum = warp_reduce_sum(thread_sum);
    if (lane_id == 0) smem[warp_id] = thread_sum;
    __syncthreads();

    if (warp_id == 0) {
        float v = (lane_id < nwarps) ? smem[lane_id] : 0.0f;
        v = warp_reduce_sum(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    float inv_sum = 1.0f / smem[0];

    // ---- Pass 3: normalize ----
    for (int col = tid; col < cols; col += BLOCK_SIZE) {
        row_out[col] *= inv_sum;
    }
}

torch::Tensor gelu_softmax_cuda(torch::Tensor input) {
    TORCH_CHECK(input.is_cuda() && input.is_contiguous());
    int rows = input.size(0);
    int cols = input.size(1);
    auto output = torch::empty_like(input);

    // Choose block size: must cover cols efficiently
    // cols=4096, use 512 threads (each handles 8 cols)
    const int BLOCK_SIZE = 512;
    int nwarps = BLOCK_SIZE / 32;
    // smem is declared as fixed 32 floats in kernel

    gelu_softmax_kernel<BLOCK_SIZE><<<rows, BLOCK_SIZE>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        rows, cols
    );
    return output;
}

// -----------------------------------------------------------------------
// GEMM with bias using CUTLASS TF32 tensor cores
// C = A * B^T + bias
// A: [M, K], B (weight): [N, K], bias: [N], C: [M, N]
// -----------------------------------------------------------------------
#include <cutlass/cutlass.h>
#include <cutlass/gemm/device/gemm.h>

// Use TF32 Tensor Core GEMM for FP32 inputs on SM86
using ElementInputA_TC   = float;
using ElementInputB_TC   = float;
using ElementOutput_TC   = float;
using ElementAccum_TC    = float;
using ElementEpilogue_TC = float;

// TF32 GEMM with tensor cores
using GemmTF32 = cutlass::gemm::device::Gemm<
    float,                                         // A type
    cutlass::layout::RowMajor,                     // A layout
    float,                                         // B type
    cutlass::layout::ColumnMajor,                  // B layout (weight.T)
    float,                                         // C/D type
    cutlass::layout::RowMajor,                     // C layout
    float,                                         // accumulator
    cutlass::arch::OpClassTensorOp,                // Tensor Core
    cutlass::arch::Sm80,                           // SM80 for TF32
    cutlass::gemm::GemmShape<128, 128, 32>,        // threadblock tile
    cutlass::gemm::GemmShape<64, 64, 32>,          // warp tile
    cutlass::gemm::GemmShape<16, 8, 8>,            // instruction shape (TF32)
    cutlass::epilogue::thread::LinearCombination<
        float, 4, float, float
    >,
    cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<>,
    3                                              // pipeline stages
>;

torch::Tensor linear_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias) {
    TORCH_CHECK(input.is_cuda() && weight.is_cuda() && bias.is_cuda());
    TORCH_CHECK(input.is_contiguous() && weight.is_contiguous() && bias.is_contiguous());

    int M = input.size(0);
    int K = input.size(1);
    int N = weight.size(0);

    auto output = torch::empty({M, N}, input.options());

    // We do GEMM: C = A * B^T  (no bias in epilogue to keep it simple)
    // Then add bias separately
    cutlass::gemm::GemmCoord problem_size(M, N, K);

    typename GemmTF32::Arguments args(
        problem_size,
        {input.data_ptr<float>(), K},
        {weight.data_ptr<float>(), K},
        {output.data_ptr<float>(), N},  // C (source, used as bias source if alpha/beta set)
        {output.data_ptr<float>(), N},  // D (output)
        {1.0f, 0.0f}                    // alpha=1, beta=0
    );

    GemmTF32 gemm_op;
    cutlass::Status status = gemm_op.initialize(args);
    TORCH_CHECK(status == cutlass::Status::kSuccess,
        "CUTLASS GEMM init failed: ", cutlass::cutlassGetStatusString(status));
    status = gemm_op();
    TORCH_CHECK(status == cutlass::Status::kSuccess,
        "CUTLASS GEMM exec failed: ", cutlass::cutlassGetStatusString(status));

    // Add bias broadcast
    // output: [M, N], bias: [N]
    output.add_(bias.unsqueeze(0));

    return output;
}
'''

cpp_fused_source = r'''
torch::Tensor gelu_softmax_cuda(torch::Tensor input);
torch::Tensor linear_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias);
'''

ext = load_inline(
    name='fused_matmul_gelu_softmax_v2',
    cpp_sources=cpp_fused_source,
    cuda_sources=fused_source,
    functions=['gelu_softmax_cuda', 'linear_forward'],
    verbose=True,
    extra_cuda_cflags=[
        '-O3',
        '--use_fast_math',
        '-arch=sm_86',
        f'-I{CUTLASS_INCLUDE}',
        '-std=c++17'
    ],
    extra_cflags=['-O3', '-std=c++17'],
)


class Model(nn.Module):
    """
    Fused: Linear -> GELU -> Softmax(dim=1)
    """
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)

    def forward(self, x):
        # CUTLASS GEMM (TF32 tensor cores) + bias
        x = ext.linear_forward(x, self.linear.weight, self.linear.bias)
        # Fused GELU + Softmax
        x = ext.gelu_softmax_cuda(x)
        return x


batch_size = 128
in_features = 4096
out_features = 4096


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features]
