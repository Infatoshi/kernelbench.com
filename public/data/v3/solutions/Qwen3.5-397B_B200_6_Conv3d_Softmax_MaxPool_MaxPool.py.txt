import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <algorithm>
#include <cmath>

// Fused kernel: Conv3d + Softmax + 2x MaxPool3d
template<typename scalar_t>
__global__ void fused_conv3d_softmax_maxpool_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size,
    int in_channels,
    int in_depth,
    int in_height,
    int in_width,
    int out_channels,
    int kernel_size,
    int pool_kernel_size
) {
    // Output dimensions after conv: (D-2, H-2, W-2) for kernel_size=3
    int conv_d = in_depth - kernel_size + 1;
    int conv_h = in_height - kernel_size + 1;
    int conv_w = in_width - kernel_size + 1;
    
    // After two poolings with kernel_size=2
    int out_d = conv_d / (pool_kernel_size * pool_kernel_size);
    int out_h = conv_h / (pool_kernel_size * pool_kernel_size);
    int out_w = conv_w / (pool_kernel_size * pool_kernel_size);
    
    int out_elements = out_d * out_h * out_w;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_output = batch_size * out_channels * out_elements;
    
    if (idx >= total_output) return;
    
    int tmp = idx;
    int ow = tmp % out_w;
    tmp /= out_w;
    int oh = tmp % out_h;
    tmp /= out_h;
    int od = tmp % out_d;
    tmp /= out_d;
    int c = tmp % out_channels;
    int n = tmp / out_channels;
    
    // Map output position to conv position (before pooling)
    int cd = od * pool_kernel_size * pool_kernel_size;
    int ch = oh * pool_kernel_size * pool_kernel_size;
    int cw = ow * pool_kernel_size * pool_kernel_size;
    
    // Compute 4x4 pooling region max after softmax
    float max_val = -1e20f;
    
    // First pass: find max for softmax stability and pooling
    for (int pd = 0; pd < 4 && cd + pd < conv_d; pd++) {
        for (int ph = 0; ph < 4 && ch + ph < conv_h; ph++) {
            for (int pw = 0; pw < 4 && cw + pw < conv_w; pw++) {
                float conv_out = 0.0f;
                int id_start = cd + pd;
                int ih_start = ch + ph;
                int iw_start = cw + pw;
                
                // Convolution
                for (int ic = 0; ic < in_channels; ic++) {
                    for (int kd = 0; kd < kernel_size; kd++) {
                        for (int kh = 0; kh < kernel_size; kh++) {
                            for (int kw = 0; kw < kernel_size; kw++) {
                                int id = id_start + kd;
                                int ih = ih_start + kh;
                                int iw = iw_start + kw;
                                if (id >= 0 && id < in_depth && 
                                    ih >= 0 && ih < in_height && 
                                    iw >= 0 && iw < in_width) {
                                    int in_idx = ((n * in_channels + ic) * in_depth + id) * in_height * in_width + ih * in_width + iw;
                                    int w_idx = ((c * in_channels + ic) * kernel_size + kd) * kernel_size * kernel_size + kh * kernel_size + kw;
                                    conv_out += (float)input[in_idx] * (float)weight[w_idx];
                                }
                            }
                        }
                    }
                }
                conv_out += (float)bias[c];
                
                if (conv_out > max_val) max_val = conv_out;
            }
        }
    }
    
    // Second pass: compute softmax and find pooling max
    float exp_sum = 0.0f;
    float pool_max = -1e20f;
    
    for (int pd = 0; pd < 4 && cd + pd < conv_d; pd++) {
        for (int ph = 0; ph < 4 && ch + ph < conv_h; ph++) {
            for (int pw = 0; pw < 4 && cw + pw < conv_w; pw++) {
                float conv_out = 0.0f;
                int id_start = cd + pd;
                int ih_start = ch + ph;
                int iw_start = cw + pw;
                
                for (int ic = 0; ic < in_channels; ic++) {
                    for (int kd = 0; kd < kernel_size; kd++) {
                        for (int kh = 0; kh < kernel_size; kh++) {
                            for (int kw = 0; kw < kernel_size; kw++) {
                                int id = id_start + kd;
                                int ih = ih_start + kh;
                                int iw = iw_start + kw;
                                if (id >= 0 && id < in_depth && 
                                    ih >= 0 && ih < in_height && 
                                    iw >= 0 && iw < in_width) {
                                    int in_idx = ((n * in_channels + ic) * in_depth + id) * in_height * in_width + ih * in_width + iw;
                                    int w_idx = ((c * in_channels + ic) * kernel_size + kd) * kernel_size * kernel_size + kh * kernel_size + kw;
                                    conv_out += (float)input[in_idx] * (float)weight[w_idx];
                                }
                            }
                        }
                    }
                }
                conv_out += (float)bias[c];
                
                float softmax_val = expf(conv_out - max_val);
                exp_sum += softmax_val;
                
                if (softmax_val > pool_max) pool_max = softmax_val;
            }
        }
    }
    
    // Third pass: normalize and write output
    float inv_sum = 1.0f / (exp_sum + 1e-9f);
    
    for (int pd = 0; pd < 4 && cd + pd < conv_d; pd++) {
        for (int ph = 0; ph < 4 && ch + ph < conv_h; ph++) {
            for (int pw = 0; pw < 4 && cw + pw < conv_w; pw++) {
                float conv_out = 0.0f;
                int id_start = cd + pd;
                int ih_start = ch + ph;
                int iw_start = cw + pw;
                
                for (int ic = 0; ic < in_channels; ic++) {
                    for (int kd = 0; kd < kernel_size; kd++) {
                        for (int kh = 0; kh < kernel_size; kh++) {
                            for (int kw = 0; kw < kernel_size; kw++) {
                                int id = id_start + kd;
                                int ih = ih_start + kh;
                                int iw = iw_start + kw;
                                if (id >= 0 && id < in_depth && 
                                    ih >= 0 && ih < in_height && 
                                    iw >= 0 && iw < in_width) {
                                    int in_idx = ((n * in_channels + ic) * in_depth + id) * in_height * in_width + ih * in_width + iw;
                                    int w_idx = ((c * in_channels + ic) * kernel_size + kd) * kernel_size * kernel_size + kh * kernel_size + kw;
                                    conv_out += (float)input[in_idx] * (float)weight[w_idx];
                                }
                            }
                        }
                    }
                }
                conv_out += (float)bias[c];
                
                float softmax_val = expf(conv_out - max_val) * inv_sum;
                
                // Check if this position contributes to output
                int out_idx = ((n * out_channels + c) * out_d + od) * out_h * out_w + oh * out_w + ow;
                
                // We need to track which position has the max in the 4x4x4 region
                // For simplicity, write the softmax value to a temp location
                // and do pooling separately - but that defeats fusion
                // Instead, just accumulate the max
            }
        }
    }
    
    // Write the pooled max of softmax values
    int out_idx = ((n * out_channels + c) * out_d + od) * out_h * out_w + oh * out_w + ow;
    output[out_idx] = (scalar_t)(pool_max * inv_sum);
}

// Simpler approach: separate kernels for better optimization
template<typename scalar_t>
__global__ void conv3d_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size,
    int in_channels,
    int in_depth,
    int in_height,
    int in_width,
    int out_channels,
    int kernel_size
) {
    int conv_d = in_depth - kernel_size + 1;
    int conv_h = in_height - kernel_size + 1;
    int conv_w = in_width - kernel_size + 1;
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * out_channels * conv_d * conv_h * conv_w;
    
    if (idx >= total) return;
    
    int tmp = idx;
    int w = tmp % conv_w;
    tmp /= conv_w;
    int h = tmp % conv_h;
    tmp /= conv_h;
    int d = tmp % conv_d;
    tmp /= conv_d;
    int c = tmp % out_channels;
    int n = tmp / out_channels;
    
    float acc = 0.0f;
    for (int ic = 0; ic < in_channels; ic++) {
        for (int kd = 0; kd < kernel_size; kd++) {
            for (int kh = 0; kh < kernel_size; kh++) {
                for (int kw = 0; kw < kernel_size; kw++) {
                    int id = d + kd;
                    int ih = h + kh;
                    int iw = w + kw;
                    int in_idx = ((n * in_channels + ic) * in_depth + id) * in_height * in_width + ih * in_width + iw;
                    int w_idx = ((c * in_channels + ic) * kernel_size + kd) * kernel_size * kernel_size + kh * kernel_size + kw;
                    acc += (float)input[in_idx] * (float)weight[w_idx];
                }
            }
        }
    }
    acc += (float)bias[c];
    
    int out_idx = ((n * out_channels + c) * conv_d + d) * conv_h * conv_w + h * conv_w + w;
    output[out_idx] = (scalar_t)acc;
}

template<typename scalar_t>
__global__ void softmax_kernel(
    scalar_t* __restrict__ x,
    int batch_size,
    int channels,
    int depth,
    int height,
    int width
) {
    int spatial = depth * height * width;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * spatial;
    
    if (idx >= total) return;
    
    int n_spatial = idx;
    int w = n_spatial % width;
    n_spatial /= width;
    int h = n_spatial % height;
    n_spatial /= height;
    int d = n_spatial % depth;
    int n = n_spatial / depth;
    
    // Find max and sum for this spatial position across channels
    float max_val = -1e20f;
    float sum_val = 0.0f;
    
    for (int c = 0; c < channels; c++) {
        int x_idx = ((n * channels + c) * depth + d) * height * width + h * width + w;
        float val = (float)x[x_idx];
        if (val > max_val) max_val = val;
    }
    
    for (int c = 0; c < channels; c++) {
        int x_idx = ((n * channels + c) * depth + d) * height * width + h * width + w;
        float exp_val = expf((float)x[x_idx] - max_val);
        sum_val += exp_val;
    }
    
    float inv_sum = 1.0f / (sum_val + 1e-9f);
    for (int c = 0; c < channels; c++) {
        int x_idx = ((n * channels + c) * depth + d) * height * width + h * width + w;
        x[x_idx] = (scalar_t)(expf((float)x[x_idx] - max_val) * inv_sum);
    }
}

template<typename scalar_t>
__global__ void maxpool3d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_depth,
    int in_height,
    int in_width,
    int kernel_size
) {
    int out_depth = in_depth / kernel_size;
    int out_height = in_height / kernel_size;
    int out_width = in_width / kernel_size;
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * channels * out_depth * out_height * out_width;
    
    if (idx >= total) return;
    
    int tmp = idx;
    int ow = tmp % out_width;
    tmp /= out_width;
    int oh = tmp % out_height;
    tmp /= out_height;
    int od = tmp % out_depth;
    tmp /= out_depth;
    int c = tmp % channels;
    int n = tmp / channels;
    
    float max_val = -1e20f;
    for (int kd = 0; kd < kernel_size; kd++) {
        for (int kh = 0; kh < kernel_size; kh++) {
            for (int kw = 0; kw < kernel_size; kw++) {
                int id = od * kernel_size + kd;
                int ih = oh * kernel_size + kh;
                int iw = ow * kernel_size + kw;
                if (id < in_depth && ih < in_height && iw < in_width) {
                    int in_idx = ((n * channels + c) * in_depth + id) * in_height * in_width + ih * in_width + iw;
                    float val = (float)input[in_idx];
                    if (val > max_val) max_val = val;
                }
            }
        }
    }
    
    int out_idx = ((n * channels + c) * out_depth + od) * out_height * out_width + oh * out_width + ow;
    output[out_idx] = (scalar_t)max_val;
}

torch::Tensor fused_conv3d_softmax_maxpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_depth = input.size(2);
    int in_height = input.size(3);
    int in_width = input.size(4);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int conv_d = in_depth - kernel_size + 1;
    int conv_h = in_height - kernel_size + 1;
    int conv_w = in_width - kernel_size + 1;
    
    // After two poolings
    int out_d = conv_d / (pool_kernel_size * pool_kernel_size);
    int out_h = conv_h / (pool_kernel_size * pool_kernel_size);
    int out_w = conv_w / (pool_kernel_size * pool_kernel_size);
    
    auto conv_out = torch::empty({batch_size, out_channels, conv_d, conv_h, conv_w}, 
                                  input.options());
    auto pool1_out = torch::empty({batch_size, out_channels, 
                                    conv_d / pool_kernel_size,
                                    conv_h / pool_kernel_size, 
                                    conv_w / pool_kernel_size},
                                  input.options());
    auto pool2_out = torch::empty({batch_size, out_channels, out_d, out_h, out_w},
                                  input.options());
    
    int threads = 256;
    int conv_total = batch_size * out_channels * conv_d * conv_h * conv_w;
    int softmax_total = batch_size * conv_d * conv_h * conv_w;
    int pool1_total = batch_size * out_channels * (conv_d / pool_kernel_size) * 
                      (conv_h / pool_kernel_size) * (conv_w / pool_kernel_size);
    int pool2_total = batch_size * out_channels * out_d * out_h * out_w;
    
    if (input.dtype() == torch::kFloat32) {
        conv3d_kernel<float><<<(conv_total + threads - 1) / threads, threads>>>(
            input.data_ptr<float>(), weight.data_ptr<float>(), bias.data_ptr<float>(),
            conv_out.data_ptr<float>(), batch_size, in_channels, in_depth, in_height, in_width,
            out_channels, kernel_size);
        
        softmax_kernel<float><<<(softmax_total + threads - 1) / threads, threads>>>(
            conv_out.data_ptr<float>(), batch_size, out_channels, conv_d, conv_h, conv_w);
        
        maxpool3d_kernel<float><<<(pool1_total + threads - 1) / threads, threads>>>(
            conv_out.data_ptr<float>(), pool1_out.data_ptr<float>(),
            batch_size, out_channels, conv_d, conv_h, conv_w, pool_kernel_size);
        
        maxpool3d_kernel<float><<<(pool2_total + threads - 1) / threads, threads>>>(
            pool1_out.data_ptr<float>(), pool2_out.data_ptr<float>(),
            batch_size, out_channels, conv_d / pool_kernel_size, conv_h / pool_kernel_size,
            conv_w / pool_kernel_size, pool_kernel_size);
    } else {
        // FP16/BF16
        conv3d_kernel<at::Half><<<(conv_total + threads - 1) / threads, threads>>>(
            input.data_ptr<at::Half>(), weight.data_ptr<at::Half>(), bias.data_ptr<at::Half>(),
            conv_out.data_ptr<at::Half>(), batch_size, in_channels, in_depth, in_height, in_width,
            out_channels, kernel_size);
        
        softmax_kernel<at::Half><<<(softmax_total + threads - 1) / threads, threads>>>(
            conv_out.data_ptr<at::Half>(), batch_size, out_channels, conv_d, conv_h, conv_w);
        
        maxpool3d_kernel<at::Half><<<(pool1_total + threads - 1) / threads, threads>>>(
            conv_out.data_ptr<at::Half>(), pool1_out.data_ptr<at::Half>(),
            batch_size, out_channels, conv_d, conv_h, conv_w, pool_kernel_size);
        
        maxpool3d_kernel<at::Half><<<(pool2_total + threads - 1) / threads, threads>>>(
            pool1_out.data_ptr<at::Half>(), pool2_out.data_ptr<at::Half>(),
            batch_size, out_channels, conv_d / pool_kernel_size, conv_h / pool_kernel_size,
            conv_w / pool_kernel_size, pool_kernel_size);
    }
    
    return pool2_out;
}

ext = load_inline(

    cpp_sources='''

torch::Tensor fused_conv3d_softmax_maxpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size
);
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("forward", &fused_conv3d_softmax_maxpool, "Fused Conv3d + Softmax + 2x MaxPool3d");

    cuda_sources=cuda_source,

class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies Softmax, and performs two max pooling operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        weight = self.conv.weight
        bias = self.conv.bias
        return ext.forward(x, weight, bias, self.pool_kernel_size)

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
