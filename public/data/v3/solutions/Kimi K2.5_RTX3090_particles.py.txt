import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

__global__ void particle_kernel_float2(
    const float2* __restrict__ positions,
    const float2* __restrict__ velocities,
    float4* __restrict__ output,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;
    
    float2 pos = positions[idx];
    float2 vel = velocities[idx];
    
    // Update velocity: v = v + g * dt
    vel.x += gravity_x * dt;
    vel.y += gravity_y * dt;
    
    // Update position: p = p + v * dt
    pos.x += vel.x * dt;
    pos.y += vel.y * dt;
    
    // Boundary collision and bounce
    if (pos.x > 1.0f) {
        pos.x = 1.0f;
        vel.x = -vel.x * damping;
    } else if (pos.x < -1.0f) {
        pos.x = -1.0f;
        vel.x = -vel.x * damping;
    }
    
    if (pos.y > 1.0f) {
        pos.y = 1.0f;
        vel.y = -vel.y * damping;
    } else if (pos.y < -1.0f) {
        pos.y = -1.0f;
        vel.y = -vel.y * damping;
    }
    
    output[idx] = make_float4(pos.x, pos.y, vel.x, vel.y);
}

__global__ void particle_kernel_half2(
    const __half2* __restrict__ positions,
    const __half2* __restrict__ velocities,
    __half2* __restrict__ output,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;
    
    // Load as half2 and convert to float2 for computation
    float2 pos = __half22float2(positions[idx]);
    float2 vel = __half22float2(velocities[idx]);
    
    // Update in float32 for numerical stability
    vel.x += gravity_x * dt;
    vel.y += gravity_y * dt;
    pos.x += vel.x * dt;
    pos.y += vel.y * dt;
    
    // Boundary collision
    if (pos.x > 1.0f) {
        pos.x = 1.0f;
        vel.x = -vel.x * damping;
    } else if (pos.x < -1.0f) {
        pos.x = -1.0f;
        vel.x = -vel.x * damping;
    }
    
    if (pos.y > 1.0f) {
        pos.y = 1.0f;
        vel.y = -vel.y * damping;
    } else if (pos.y < -1.0f) {
        pos.y = -1.0f;
        vel.y = -vel.y * damping;
    }
    
    // Store back as half2
    output[idx * 2] = __float22half2_rn(pos);
    output[idx * 2 + 1] = __float22half2_rn(vel);
}

__global__ void particle_kernel_bfloat2(
    const __nv_bfloat162* __restrict__ positions,
    const __nv_bfloat162* __restrict__ velocities,
    __nv_bfloat162* __restrict__ output,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;
    
    // Load as bfloat2 and convert to float2
    float2 pos = __bfloat1622float2(positions[idx]);
    float2 vel = __bfloat1622float2(velocities[idx]);
    
    // Update in float32
    vel.x += gravity_x * dt;
    vel.y += gravity_y * dt;
    pos.x += vel.x * dt;
    pos.y += vel.y * dt;
    
    // Boundary collision
    if (pos.x > 1.0f) {
        pos.x = 1.0f;
        vel.x = -vel.x * damping;
    } else if (pos.x < -1.0f) {
        pos.x = -1.0f;
        vel.x = -vel.x * damping;
    }
    
    if (pos.y > 1.0f) {
        pos.y = 1.0f;
        vel.y = -vel.y * damping;
    } else if (pos.y < -1.0f) {
        pos.y = -1.0f;
        vel.y = -vel.y * damping;
    }
    
    // Store back as bfloat2
    output[idx * 2] = __float22bfloat162_rn(pos);
    output[idx * 2 + 1] = __float22bfloat162_rn(vel);
}

torch::Tensor particle_forward(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    torch::Tensor gravity,
    float damping
) {
    int n = positions.size(0);
    auto output = torch::empty({n, 4}, positions.options());
    
    float gravity_x = gravity[0].item<float>();
    float gravity_y = gravity[1].item<float>();
    
    const int threads = 256;
    const int blocks = (n + threads - 1) / threads;
    
    // Ensure contiguous memory layout for vectorized access
    auto pos_contig = positions.contiguous();
    auto vel_contig = velocities.contiguous();
    
    if (positions.scalar_type() == torch::kFloat16) {
        particle_kernel_half2<<<blocks, threads>>>(
            (__half2*)pos_contig.data_ptr<at::Half>(),
            (__half2*)vel_contig.data_ptr<at::Half>(),
            (__half2*)output.data_ptr<at::Half>(),
            dt, gravity_x, gravity_y, damping, n
        );
    } else if (positions.scalar_type() == torch::kBFloat16) {
        particle_kernel_bfloat2<<<blocks, threads>>>(
            (__nv_bfloat162*)pos_contig.data_ptr<at::BFloat16>(),
            (__nv_bfloat162*)vel_contig.data_ptr<at::BFloat16>(),
            (__nv_bfloat162*)output.data_ptr<at::BFloat16>(),
            dt, gravity_x, gravity_y, damping, n
        );
    } else {
        // Default to float32
        particle_kernel_float2<<<blocks, threads>>>(
            (float2*)pos_contig.data_ptr<float>(),
            (float2*)vel_contig.data_ptr<float>(),
            (float4*)output.data_ptr<float>(),
            dt, gravity_x, gravity_y, damping, n
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor particle_forward(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    torch::Tensor gravity,
    float damping
);
'''

ext = load_inline(
    name='particle_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        return ext.particle_forward(positions, velocities, dt_value, gravity, self.damping)

def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]

def get_init_inputs():
    return [0.9]
