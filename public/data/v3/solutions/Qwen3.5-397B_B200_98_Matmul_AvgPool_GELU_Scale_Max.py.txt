import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

#define BLOCK_SIZE 256
#define POOL_KERNEL 4

__device__ float gelu(float x) {
    return 0.5f * x * (1.0f + tanhf(sqrtf(2.0f / M_PI) * (x + 0.044715f * x * x * x)));
}

// Fused kernel: Matmul + AvgPool + GELU + Scale + Max
__global__ void fused_matmul_avgpool_gelu_scale_max(
    const float* input,      // [batch, in_features]
    const float* weight,     // [out_features, in_features]
    const float* bias,       // [out_features]
    float* output,           // [batch]
    int batch_size,
    int in_features,
    int out_features,
    float scale_factor
) {
    int batch_idx = blockIdx.y;
    int out_row = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    int pooled_out_features = out_features / POOL_KERNEL;
    
    if (out_row >= pooled_out_features) return;
    
    // Compute avgpool output for this position
    float sum = 0.0f;
    for (int k = 0; k < POOL_KERNEL; k++) {
        int out_idx = out_row * POOL_KERNEL + k;
        if (out_idx < out_features) {
            float acc = 0.0f;
            for (int j = 0; j < in_features; j++) {
                acc += input[batch_idx * in_features + j] * weight[out_idx * in_features + j];
            }
            if (bias) acc += bias[out_idx];
            sum += acc;
        }
    }
    float avg = sum / POOL_KERNEL;
    
    // Apply GELU
    float gelu_out = gelu(avg);
    
    // Apply scale
    float scaled = gelu_out * scale_factor;
    
    // Store intermediate result for max reduction
    output[batch_idx * pooled_out_features + out_row] = scaled;
}

// Max reduction kernel
__global__ void max_reduction(
    const float* input,
    float* output,
    int batch_size,
    int features
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    float max_val = -FLT_MAX;
    for (int i = 0; i < features; i++) {
        float val = input[batch_idx * features + i];
        if (val > max_val) max_val = val;
    }
    output[batch_idx] = max_val;
}

torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float scale_factor) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    int pooled_out_features = out_features / POOL_KERNEL;
    
    auto intermediate = torch::empty({batch_size, pooled_out_features}, input.options());
    auto output = torch::empty({batch_size}, input.options());
    
    dim3 block(BLOCK_SIZE);
    dim3 grid((pooled_out_features + BLOCK_SIZE - 1) / BLOCK_SIZE, batch_size);
    
    fused_matmul_avgpool_gelu_scale_max<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        intermediate.data_ptr<float>(),
        batch_size,
        in_features,
        out_features,
        scale_factor
    );
    
    dim3 grid2(batch_size);
    max_reduction<<<grid2, 256>>>(
        intermediate.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        pooled_out_features
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_op(torch::Tensor, torch::Tensor, torch::Tensor, float);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        weight = self.matmul.weight
        bias = self.matmul.bias
        return ext.fused_op(x, weight, bias, self.scale_factor)

def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
