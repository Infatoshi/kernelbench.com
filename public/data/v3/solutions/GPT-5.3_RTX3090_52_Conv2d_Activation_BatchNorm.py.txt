import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ inline float softplus(float x){return log1pf(expf(x));}

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    const float* __restrict__ mean,
    const float* __restrict__ var,
    float eps,
    float* __restrict__ out,
    int N,int Cin,int H,int W,
    int Cout,int kH,int kW,
    int Ho,int Wo)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*Cout*Ho*Wo;
    if(idx>=total) return;

    int wo = idx % Wo; idx/=Wo;
    int ho = idx % Ho; idx/=Ho;
    int co = idx % Cout; idx/=Cout;
    int n = idx;

    float v = 0.0f;
    for(int ci=0; ci<Cin; ++ci){
        for(int kh=0; kh<kH; ++kh){
            for(int kw=0; kw<kW; ++kw){
                int hi = ho + kh;
                int wi = wo + kw;
                int x_idx = ((n*Cin + ci)*H + hi)*W + wi;
                int w_idx = ((co*Cin + ci)*kH + kh)*kW + kw;
                v += x[x_idx] * w[w_idx];
            }
        }
    }
    if(bias) v += bias[co];

    float a = tanhf(softplus(v)) * v;

    float norm = (a - mean[co]) * rsqrtf(var[co] + eps);
    float y = norm * gamma[co] + beta[co];

    int out_idx = ((n*Cout + co)*Ho + ho)*Wo + wo;
    out[out_idx] = y;
}

torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor bias,
    torch::Tensor gamma,
    torch::Tensor beta,
    torch::Tensor mean,
    torch::Tensor var,
    double eps)
{
    x = x.contiguous(); w = w.contiguous();
    gamma = gamma.contiguous(); beta = beta.contiguous();
    mean = mean.contiguous(); var = var.contiguous();

    int N=x.size(0), Cin=x.size(1), H=x.size(2), W=x.size(3);
    int Cout=w.size(0), kH=w.size(2), kW=w.size(3);
    int Ho = H - kH + 1;
    int Wo = W - kW + 1;

    auto out = torch::empty({N,Cout,Ho,Wo}, x.options().dtype(torch::kFloat32));

    int total = N*Cout*Ho*Wo;
    int threads = 256;
    int blocks = (total + threads -1)/threads;

    fused_kernel<<<blocks,threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.numel()?bias.data_ptr<float>():nullptr,
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        mean.data_ptr<float>(),
        var.data_ptr<float>(),
        (float)eps,
        out.data_ptr<float>(),
        N,Cin,H,W,Cout,kH,kW,Ho,Wo
    );

    return out;
}
'''

ext = load_inline(
    name='fused_conv_act_bn',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor bias, torch::Tensor gamma, torch::Tensor beta, torch::Tensor mean, torch::Tensor var, double eps);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        x = x.float().contiguous()
        return ext.fused_forward(
            x,
            self.conv.weight,
            self.conv.bias,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.eps
        )

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width, device='cuda')]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
