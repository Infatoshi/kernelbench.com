import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cfloat>

// Cross-entropy loss kernel for FP32
__global__ void cross_entropy_loss_fp32_kernel(
    const float* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    int target = targets[batch_idx];
    
    const float* pred_row = predictions + batch_idx * num_classes;
    
    // Find max for numerical stability
    float max_val = -FLT_MAX;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        max_val = fmaxf(max_val, pred_row[i]);
    }
    
    // Shared memory for reduction
    extern __shared__ float shared_mem[];
    float* max_vals = shared_mem;
    float* sum_vals = shared_mem + blockDim.x;
    
    max_vals[threadIdx.x] = max_val;
    __syncthreads();
    
    // Reduce to find global max
    if (threadIdx.x == 0) {
        float global_max = max_vals[0];
        for (int i = 1; i < blockDim.x; i++) {
            global_max = fmaxf(global_max, max_vals[i]);
        }
        max_vals[0] = global_max;
    }
    __syncthreads();
    
    float global_max = max_vals[0];
    
    // Compute exp(x - max) and sum
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        local_sum += expf(pred_row[i] - global_max);
    }
    
    sum_vals[threadIdx.x] = local_sum;
    __syncthreads();
    
    // Reduce sum
    if (threadIdx.x == 0) {
        float total_sum = 0.0f;
        for (int i = 0; i < blockDim.x; i++) {
            total_sum += sum_vals[i];
        }
        sum_vals[0] = total_sum;
    }
    __syncthreads();
    
    float log_sum_exp = global_max + logf(sum_vals[0]);
    
    // Compute loss for this sample
    if (threadIdx.x == 0) {
        losses[batch_idx] = log_sum_exp - pred_row[target];
    }
}

// Cross-entropy loss kernel for BF16
__global__ void cross_entropy_loss_bf16_kernel(
    const at::BFloat16* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    int target = targets[batch_idx];
    
    const at::BFloat16* pred_row = predictions + batch_idx * num_classes;

    float max_val = -1e10f;

    }

    }
    __syncthreads();
    
    float log_sum_exp = global_max + logf(sum_vals[0]);
    
    if (threadIdx.x == 0) {
        float target_val = static_cast<float>(pred_row[target]);
        losses[batch_idx] = log_sum_exp - target_val;
    }
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    auto losses = torch::empty({batch_size}, predictions.options().dtype(torch::kFloat32));
    
    int threads = 256;
    int blocks = batch_size;
    int shared_mem = threads * sizeof(float) * 2;
    
    if (predictions.dtype() == torch::kFloat32) {
        cross_entropy_loss_fp32_kernel<<<blocks, threads, shared_mem>>>(
            predictions.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kHalf) {
        cross_entropy_loss_fp16_kernel<<<blocks, threads, shared_mem>>>(
            predictions.data_ptr<at::Half>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kBFloat16) {
        cross_entropy_loss_bf16_kernel<<<blocks, threads, shared_mem>>>(
            predictions.data_ptr<at::BFloat16>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }
    
    // Sum all losses and divide by batch size (mean reduction)
    auto total_loss = losses.sum();
    return total_loss / batch_size;
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_100', '-O3'],
)

class Model(nn.Module):
    """
    A model that computes Cross Entropy Loss for multi-class classification tasks.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_loss(predictions, targets)

batch_size = 4096
num_classes = 1024
input_shape = (num_classes,)
dim = 1

def get_inputs():
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

def get_init_inputs():
    return []
