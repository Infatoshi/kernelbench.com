import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <c10/util/BFloat16.h>

template <typename T>
__device__ __forceinline__ float to_float(T val) {
    return static_cast<float>(val);
}

template <>
__device__ __forceinline__ float to_float<c10::Half>(c10::Half val) {
    return __half2float(val);
}

template <>
__device__ __forceinline__ float to_float<c10::BFloat16>(c10::BFloat16 val) {
    return float(val);
}

template <typename T>
__device__ __forceinline__ T from_float(float val) {
    return static_cast<T>(val);
}

template <>
__device__ __forceinline__ c10::Half from_float<c10::Half>(float val) {
    return __float2half(val);
}

template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, T* __restrict__ output, int rows, int cols) {
    int row = blockIdx.x;
    if (row >= rows) return;
    
    const T* row_input = input + row * cols;
    T* row_output = output + row * cols;
    
    float local_max = -1e20f;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        float val = to_float(row_input[i]);
        local_max = fmaxf(local_max, val);
    }
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, offset));
    }
    
    __shared__ float shared_max[32];
    int lane = threadIdx.x & 31;
    int warp_id = threadIdx.x >> 5;
    
    if (lane == 0) shared_max[warp_id] = local_max;
    __syncthreads();
    
    if (threadIdx.x < 32) {
        local_max = (threadIdx.x < (blockDim.x + 31) >> 5) ? shared_max[lane] : -1e20f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, offset));
        }
        if (lane == 0) shared_max[0] = local_max;
    }
    __syncthreads();
    
    float row_max = shared_max[0];
    
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        float val = expf(to_float(row_input[i]) - row_max);
        local_sum += val;
        row_output[i] = from_float<T>(val);
    }
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_xor_sync(0xffffffff, local_sum, offset);
    }
    
    if (lane == 0) shared_max[warp_id] = local_sum;
    __syncthreads();
    
    if (threadIdx.x < 32) {
        local_sum = (threadIdx.x < (blockDim.x + 31) >> 5) ? shared_max[lane] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            local_sum += __shfl_xor_sync(0xffffffff, local_sum, offset);
        }
        if (lane == 0) shared_max[0] = local_sum;
    }
    __syncthreads();
    
    float row_sum = shared_max[0];
    
    for (int i = threadIdx.x; i < cols; i += blockDim.x) {
        float val = to_float(row_output[i]);
        row_output[i] = from_float<T>(val / row_sum);
    }
}

torch::Tensor softmax_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int rows = input.size(0);
    int cols = input.size(1);
    
    const int threads = 1024;
    const int blocks = rows;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "softmax_cuda", [&] {
        softmax_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            rows, 
            cols
        );
    });
    
    return output;
}
'''

cpp_source = '''
torch::Tensor softmax_cuda(torch::Tensor input);
'''

ext = load_inline(
    name='softmax_b200',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs a Softmax activation.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features).

        Returns:
            torch.Tensor: Output tensor with Softmax applied, same shape as input.
        """
        return ext.softmax_cuda(x)

batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]
