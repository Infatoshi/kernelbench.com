import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define THREADS 256

template <typename scalar_t>
__global__ void ce_row_kernel(const scalar_t* __restrict__ logits, const int64_t* __restrict__ targets, float* __restrict__ losses, int B, int C){
    int row = blockIdx.x;
    if(row >= B) return;

    const scalar_t* row_ptr = logits + row * C;

    __shared__ float smax;
    __shared__ float ssum;

    float local_max = -1e30f;
    for(int i = threadIdx.x; i < C; i += blockDim.x){
        float v = (float)row_ptr[i];
        local_max = fmaxf(local_max, v);
    }

    __shared__ float redmax[THREADS];
    redmax[threadIdx.x] = local_max;
    __syncthreads();

    for(int s = blockDim.x/2; s>0; s>>=1){
        if(threadIdx.x < s){
            redmax[threadIdx.x] = fmaxf(redmax[threadIdx.x], redmax[threadIdx.x+s]);
        }
        __syncthreads();
    }

    if(threadIdx.x==0) smax = redmax[0];
    __syncthreads();

    float local_sum = 0.f;
    for(int i = threadIdx.x; i < C; i += blockDim.x){
        float v = (float)row_ptr[i];
        local_sum += expf(v - smax);
    }

    __shared__ float redsum[THREADS];
    redsum[threadIdx.x] = local_sum;
    __syncthreads();

    for(int s = blockDim.x/2; s>0; s>>=1){
        if(threadIdx.x < s){
            redsum[threadIdx.x] += redsum[threadIdx.x+s];
        }
        __syncthreads();
    }

    if(threadIdx.x==0) ssum = redsum[0];
    __syncthreads();

    if(threadIdx.x==0){
        int t = targets[row];
        float logit_t = (float)row_ptr[t];
        float logsumexp = logf(ssum) + smax;
        losses[row] = -(logit_t - logsumexp);
    }
}

__global__ void reduce_mean_kernel(const float* __restrict__ losses, float* out, int B){
    __shared__ float buf[THREADS];
    int tid = threadIdx.x;
    float v = 0.f;
    for(int i = tid; i < B; i += blockDim.x) v += losses[i];
    buf[tid] = v;
    __syncthreads();

    for(int s = blockDim.x/2; s>0; s>>=1){
        if(tid < s) buf[tid] += buf[tid+s];
        __syncthreads();
    }

    if(tid==0) *out = buf[0] / B;
}

torch::Tensor cross_entropy_forward(torch::Tensor logits, torch::Tensor targets){
    int B = logits.size(0);
    int C = logits.size(1);

    auto losses = torch::empty({B}, logits.options().dtype(torch::kFloat));

    dim3 grid(B);
    dim3 block(THREADS);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, logits.scalar_type(), "ce_row_kernel", ([&]{
        ce_row_kernel<scalar_t><<<grid, block>>>(
            logits.data_ptr<scalar_t>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            B, C);
    }));

    auto out = torch::empty({}, logits.options().dtype(torch::kFloat));
    reduce_mean_kernel<<<1, THREADS>>>(losses.data_ptr<float>(), out.data_ptr<float>(), B);

    return out;
}
'''

ext = load_inline(
    name='cross_entropy_cuda',
    cpp_sources='torch::Tensor cross_entropy_forward(torch::Tensor logits, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets)

batch_size = 4096
num_classes = 1024
input_shape = (num_classes,)

def get_inputs():
    return [torch.randn(batch_size, *input_shape, device='cuda'), torch.randint(0, num_classes, (batch_size,), device='cuda')]

def get_init_inputs():
    return []
