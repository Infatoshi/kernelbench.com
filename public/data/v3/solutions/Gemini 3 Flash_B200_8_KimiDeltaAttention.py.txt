
import torch
import torch.nn as nn
import torch.nn.functional as F
from fla.ops import chunk_kda

class Model(nn.Module):
    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        # Combine projections for efficiency
        # q_proj, k_proj, v_proj, a_proj, b_proj, g_proj
        # Total channels: 3 * num_heads * head_dim_qk (q,k) + 4 * num_heads * head_dim_v (v, a, g) + num_heads (beta)
        # Note: actually q and k are both qk_dim. q: H*D_qk, k: H*D_qk, v: H*D_v, a: H*D_v, b: H, g: H*D_v
        # Let's just group them carefully.
        self.W_qkvabg = nn.Linear(hidden_size, 
                                  num_heads * (2 * head_dim_qk + 3 * head_dim_v + 1), 
                                  bias=True)
        # However, the reference uses no bias for q, k, v, g.
        # It's better to stick to original structure or be very careful.
        
        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.a_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=True)
        self.b_proj = nn.Linear(hidden_size, num_heads, bias=True)
        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        if use_short_conv:
            self.q_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.k_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.v_conv = nn.Conv1d(
                num_heads * head_dim_v, num_heads * head_dim_v,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_v,
                padding=conv_kernel_size - 1
            )

        self.o_norm = nn.LayerNorm(head_dim_v)
        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use autocast to speed up computations on B200
        with torch.cuda.amp.autocast(dtype=torch.bfloat16):
            batch_size, seq_len, _ = x.shape
            
            # Parallel projections
            q = self.q_proj(x)
            k = self.k_proj(x)
            v = self.v_proj(x)
            a = self.a_proj(x)
            b = self.b_proj(x)
            g_gate = self.g_proj(x)

            if self.use_short_conv:
                q = F.silu(self.q_conv(q.transpose(1, 2))[:, :, :seq_len].transpose(1, 2))
                k = F.silu(self.k_conv(k.transpose(1, 2))[:, :, :seq_len].transpose(1, 2))
                v = F.silu(self.v_conv(v.transpose(1, 2))[:, :, :seq_len].transpose(1, 2))

            q = q.view(batch_size, seq_len, self.num_heads, self.head_dim_qk)
            k = k.view(batch_size, seq_len, self.num_heads, self.head_dim_qk)
            v = v.view(batch_size, seq_len, self.num_heads, self.head_dim_v)

            # Gate computations
            a = torch.sigmoid(a).view(batch_size, seq_len, self.num_heads, self.head_dim_v)
            beta = torch.sigmoid(b).view(batch_size, seq_len, self.num_heads)
            g_log = a.clamp(min=1e-6).log()

            # Core Attention - fla chunk_kda is already optimized for bf16
            o, _ = chunk_kda(q, k, v, g_log, beta, scale=self.scale)

            # Post-attention
            o = self.o_norm(o)
            g = torch.sigmoid(g_gate).view(batch_size, seq_len, self.num_heads, self.head_dim_v)
            o = (o * g).reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
            out = self.o_proj(o)
            
        return out.to(x.dtype)

def get_inputs():
    return [torch.randn(4, 2048, 2048, device='cuda')]

def get_init_inputs():
    return [2048, 16, 128, 128]
