import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void gemv_int4_kernel(
    const int8_t* __restrict__ x,
    const int8_t* __restrict__ w,
    float scale,
    half* __restrict__ y,
    int B, int K, int O)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * O;
    if (idx >= total) return;

    int b = idx / O;
    int o = idx % O;

    const int8_t* x_row = x + b * K;
    const int8_t* w_row = w + o * K;

    int acc = 0;

    #pragma unroll 4
    for (int i = 0; i < K; ++i) {
        acc += (int)x_row[i] * (int)w_row[i];
    }

    float val = acc * scale;
    y[idx] = __float2half(val);
}


torch::Tensor gemv(torch::Tensor x_q, torch::Tensor w_q, float scale) {
    int B = x_q.size(0);
    int K = x_q.size(1);
    int O = w_q.size(0);

    auto y = torch::empty({B, O}, torch::dtype(torch::kFloat16).device(x_q.device()));

    int total = B * O;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    gemv_int4_kernel<<<blocks, threads>>>(
        x_q.data_ptr<int8_t>(),
        w_q.data_ptr<int8_t>(),
        scale,
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        B, K, O
    );

    return y;
}
'''

ext = load_inline(
    name='gemv_fp4_ext',
    cpp_sources='torch::Tensor gemv(torch::Tensor x_q, torch::Tensor w_q, float scale);',
    cuda_sources=cuda_src,
    functions=['gemv'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        scale = float(x_scale.item() * self.weight_scale.item())
        return ext.gemv(x_q.contiguous(), self.weight_q.contiguous(), scale)


def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8, device='cuda')
    x_scale = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    return [x_q, x_scale]


def get_init_inputs():
    return []
