import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def swish_scale_kernel(
    C_ptr, D_ptr,
    stride_cm, stride_cn,
    stride_dm, stride_dn,
    M, N,
    scaling_factor,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
):
    """Element-wise kernel for Swish activation and scaling."""
    pid = tl.program_id(axis=0)
    num_blocks_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_blocks_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    block_m = pid // num_blocks_n
    block_n = pid % num_blocks_n
    
    rows = block_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    cols = block_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    
    mask_m = rows < M
    mask_n = cols < N
    mask = mask_m[:, None] & mask_n[None, :]
    
    c_ptr = C_ptr + rows[:, None] * stride_cm + cols[None, :] * stride_cn
    c = tl.load(c_ptr, mask=mask, other=0.0)
    
    # Swish: x * sigmoid(x)
    sigmoid_x = tl.sigmoid(c)
    swish_x = c * sigmoid_x
    
    # Scale
    d = swish_x * scaling_factor
    
    d_ptr = D_ptr + rows[:, None] * stride_dm + cols[None, :] * stride_dn
    tl.store(d_ptr, d, mask=mask)


@triton.jit
def matmul_swish_scale_kernel(
    A_ptr, B_ptr, bias_ptr,
    C_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    scaling_factor,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
    BIAS: tl.constexpr,
):
    """Fused GEMM + Swish + Scaling kernel."""
    pid = tl.program_id(axis=0)
    num_blocks_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_blocks_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_blocks_in_group = GROUP_SIZE_M * num_blocks_n
    
    group_id = pid // num_blocks_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_blocks_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_blocks_in_group) // group_size_m
    
    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptr = A_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    b_ptr = B_ptr + offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn
    
    acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        k_offs = k * BLOCK_SIZE_K + offs_k
        mask_a = (offs_am[:, None] < M) & (k_offs[None, :] < K)
        mask_b = (k_offs[:, None] < K) & (offs_bn[None, :] < N)
        
        a = tl.load(a_ptr, mask=mask_a, other=0.0)
        b = tl.load(b_ptr, mask=mask_b, other=0.0)
        
        acc = tl.dot(a, b, acc)
        
        a_ptr += BLOCK_SIZE_K * stride_ak
        b_ptr += BLOCK_SIZE_K * stride_bk
    
    # Add bias if present
    if BIAS:
        bias = tl.load(bias_ptr + offs_bn, mask=offs_bn < N, other=0.0)
        acc = acc + bias[None, :]
    
    # Apply Swish and scaling
    sigmoid_acc = tl.sigmoid(acc)
    swish_acc = acc * sigmoid_acc
    out = swish_acc * scaling_factor
    
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptr = C_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    
    mask_c = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptr, out, mask=mask_c)


class MatmulSwishScaleFn(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, scaling_factor):
        M, K = x.shape
        N = weight.shape[0]
        
        assert x.shape[1] == weight.shape[1], "Dimension mismatch"
        
        output = torch.empty((M, N), dtype=x.dtype, device=x.device)
        
        BLOCK_SIZE_M = 128
        BLOCK_SIZE_N = 128
        BLOCK_SIZE_K = 32
        GROUP_SIZE_M = 8
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        
        matmul_swish_scale_kernel[grid](
            x, weight, bias,
            output,
            M, N, K,
            x.stride(0), x.stride(1),
            weight.stride(1), weight.stride(0),
            output.stride(0), output.stride(1),
            scaling_factor,
            BLOCK_SIZE_M=BLOCK_SIZE_M,
            BLOCK_SIZE_N=BLOCK_SIZE_N,
            BLOCK_SIZE_K=BLOCK_SIZE_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
            BIAS=bias is not None,
        )
        
        return output


class Model(nn.Module):
    """
    Optimized model that performs a matrix multiplication, applies Swish activation, and scales the result.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
        # Remove the default bias from Linear since we handle it ourselves
        # But keep it for state_dict compatibility
        self.matmul = nn.Linear(in_features, out_features, bias=True)

    def forward(self, x):
        weight = self.matmul.weight
        bias = self.matmul.bias if self.matmul.bias is not None else None
        
        return MatmulSwishScaleFn.apply(x, weight, bias, self.scaling_factor)


batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, scaling_factor]
