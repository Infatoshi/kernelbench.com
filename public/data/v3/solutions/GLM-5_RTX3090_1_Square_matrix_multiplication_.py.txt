import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>
#include <cuda_fp16.h>

using namespace nvcuda::wmma;

// Tile dimensions
constexpr int BLOCK_M = 128;
constexpr int BLOCK_N = 128;
constexpr int BLOCK_K = 32;

// Threadblock and warp configuration
constexpr int WARP_SIZE = 32;
constexpr int WARPS_PER_BLOCK_M = 4;
constexpr int WARPS_PER_BLOCK_N = 2;
constexpr int THREADS_PER_BLOCK = 256;

// WMMA fragment dimensions
constexpr int WMMA_M = 16;
constexpr int WMMA_N = 16;
constexpr int WMMA_K = 16;

__global__ void __launch_bounds__(256) gemm_kernel(
    const half* __restrict__ A, 
    const half* __restrict__ B, 
    half* __restrict__ C, 
    int N) {
    
    // Warp indices
    int warpIdx = threadIdx.x / WARP_SIZE;
    int laneIdx = threadIdx.x % WARP_SIZE;
    
    // This warp's output tile (16x16 per warp, 4x2 warps per block)
    int warpM = warpIdx % 4;  // 0-3
    int warpN = warpIdx / 4;  // 0-1
    
    // Global row/col for this warp's output
    int gRow = blockIdx.y * BLOCK_M + warpM * WMMA_M;
    int gCol = blockIdx.x * BLOCK_N + warpN * WMMA_N;
    
    // Shared memory for double buffering
    __shared__ half smemA[2][BLOCK_M][BLOCK_K];
    __shared__ half smemB[2][BLOCK_K][BLOCK_N];
    
    // Accumulator in FP32 for precision
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc;
    fill_fragment(acc, 0.0f);
    
    // Loop over K tiles
    int numKTiles = (N + BLOCK_K - 1) / BLOCK_K;
    
    for (int kTile = 0; kTile < numKTiles; kTile++) {
        // Cooperative load of A and B tiles into shared memory
        // Each thread loads multiple elements
        for (int i = threadIdx.x; i < BLOCK_M * BLOCK_K; i += THREADS_PER_BLOCK) {
            int row = i / BLOCK_K;
            int col = i % BLOCK_K;
            int gArow = blockIdx.y * BLOCK_M + row;
            int gAcol = kTile * BLOCK_K + col;
            half val = __float2half(0.0f);
            if (gArow < N && gAcol < N) {
                val = A[gArow * N + gAcol];
            }
            smemA[0][row][col] = val;
        }
        
        for (int i = threadIdx.x; i < BLOCK_K * BLOCK_N; i += THREADS_PER_BLOCK) {
            int row = i / BLOCK_N;
            int col = i % BLOCK_N;
            int gBrow = kTile * BLOCK_K + row;
            int gBcol = blockIdx.x * BLOCK_N + col;
            half val = __float2half(0.0f);
            if (gBrow < N && gBcol < N) {
                val = B[gBrow * N + gBcol];
            }
            smemB[0][row][col] = val;
        }
        
        __syncthreads();
        
        // Perform WMMA for each K-step within the tile
        for (int kStep = 0; kStep < BLOCK_K; kStep += WMMA_K) {
            // Load A fragment (row-major)
            fragment<matrix_a, WMMA_M, WMMA_N, WMMA_K, half, row_major> a_frag;
            load_matrix_sync(a_frag, &smemA[0][warpM * WMMA_M][kStep], BLOCK_K);
            
            // Load B fragment (row-major, but we treat it as column-major for C = A*B)
            fragment<matrix_b, WMMA_M, WMMA_N, WMMA_K, half, row_major> b_frag;
            load_matrix_sync(b_frag, &smemB[0][kStep][warpN * WMMA_N], BLOCK_N);
            
            // Matrix multiply-accumulate
            mma_sync(acc, a_frag, b_frag, acc);
        }
        
        __syncthreads();
    }
    
    // Store accumulator to shared memory, then to global memory
    __shared__ half smemC[BLOCK_M][BLOCK_N];
    
    // Convert FP32 accumulator to half and store
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, half> acc_half;
    #pragma unroll
    for (int i = 0; i < acc.num_elements; i++) {
        acc_half.x[i] = __float2half(acc.x[i]);
    }
    store_matrix_sync(&smemC[warpM * WMMA_M][warpN * WMMA_N], acc_half, BLOCK_N, mem_row_major);
    
    __syncthreads();
    
    // Write from shared memory to global memory
    for (int i = threadIdx.x; i < BLOCK_M * BLOCK_N; i += THREADS_PER_BLOCK) {
        int row = i / BLOCK_N;
        int col = i % BLOCK_N;
        int gCrow = blockIdx.y * BLOCK_M + row;
        int gCcol = blockIdx.x * BLOCK_N + col;
        if (gCrow < N && gCcol < N) {
            C[gCrow * N + gCcol] = smemC[row][col];
        }
    }
}

torch::Tensor gemm(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    
    // Convert to half precision for tensor cores
    auto A_half = A.to(torch::kFloat16);
    auto B_half = B.to(torch::kFloat16);
    auto C_half = torch::empty({N, N}, torch::kFloat16).cuda();
    
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (N + BLOCK_M - 1) / BLOCK_M);
    dim3 block(THREADS_PER_BLOCK);
    
    gemm_kernel<<<grid, block>>>(
        reinterpret_cast<const half*>(A_half.data_ptr<at::Half>()),
        reinterpret_cast<const half*>(B_half.data_ptr<at::Half>()),
        reinterpret_cast<half*>(C_half.data_ptr<at::Half>()),
        N
    );
    
    // Convert back to original precision
    return C_half.to(A.scalar_type());
}
'''

cpp_source = r'''
torch::Tensor gemm(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math', '-std=c++17'],
    verbose=True,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemm(A.cuda(), B.cuda())

N = 2048
OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp16", "bf16", "fp32"]

def get_inputs():
    A = torch.randn(N, N)
    B = torch.randn(N, N)
    return [A, B]

def get_init_inputs():
    return []
