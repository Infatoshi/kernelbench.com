import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def _attention_fwd_kernel(
    Q_ptr, K_ptr, V_ptr, Out_ptr,
    stride_qb, stride_qh, stride_qs, stride_qd,
    stride_kb, stride_kh, stride_ks, stride_kd,
    stride_vb, stride_vh, stride_vs, stride_vd,
    stride_ob, stride_oh, stride_os, stride_od,
    num_heads, seq_len, head_dim,
    scale,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_D: tl.constexpr,
):
    """Flash attention kernel for H100."""
    start_m = tl.program_id(0)
    head_id = tl.program_id(1)
    batch_id = tl.program_id(2)
    
    # Offsets for this batch and head
    q_batch_head = batch_id * stride_qb + head_id * stride_qh
    k_batch_head = batch_id * stride_kb + head_id * stride_kh
    v_batch_head = batch_id * stride_vb + head_id * stride_vh
    o_batch_head = batch_id * stride_ob + head_id * stride_oh
    
    # Initialize accumulators for this block of Q
    m_i = tl.zeros([BLOCK_M], dtype=tl.float32) - float('inf')
    l_i = tl.zeros([BLOCK_M], dtype=tl.float32)
    acc = tl.zeros([BLOCK_M, BLOCK_D], dtype=tl.float32)
    
    # Row indices for Q block
    m_range = start_m * BLOCK_M + tl.arange(0, BLOCK_M)
    q_mask = m_range < seq_len
    
    # Load Q block
    q_ptrs = Q_ptr + q_batch_head + m_range[:, None] * stride_qs + tl.arange(0, BLOCK_D)[None, :] * stride_qd
    q = tl.load(q_ptrs, mask=q_mask[:, None], other=0.0).to(tl.float32)
    
    # Loop over K, V blocks
    for start_n in range(0, seq_len, BLOCK_N):
        n_range = start_n + tl.arange(0, BLOCK_N)
        k_mask = n_range < seq_len
        
        # Load K block
        k_ptrs = K_ptr + k_batch_head + n_range[:, None] * stride_ks + tl.arange(0, BLOCK_D)[None, :] * stride_kd
        k = tl.load(k_ptrs, mask=k_mask[:, None], other=0.0).to(tl.float32)
        
        # Load V block  
        v_ptrs = V_ptr + v_batch_head + n_range[:, None] * stride_vs + tl.arange(0, BLOCK_D)[None, :] * stride_vd
        v = tl.load(v_ptrs, mask=k_mask[:, None], other=0.0).to(tl.float32)
        
        # Compute Q @ K^T * scale
        qk = tl.dot(q, tl.trans(k)) * scale
        
        # Apply causal mask if needed (not needed for self-attention without causal)
        # But we need to mask out-of-bounds
        qk = tl.where(k_mask[None, :], qk, float('-inf'))
        
        # Softmax with running max
        m_new = tl.maximum(m_i, tl.max(qk, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(qk - m_new[:, None])
        l_new = alpha * l_i + tl.sum(p, axis=1)
        
        # Update accumulator
        acc = acc * alpha[:, None] + tl.dot(p, v)
        
        # Update running stats
        m_i = m_new
        l_i = l_new
    
    # Normalize
    acc = acc / l_i[:, None]
    
    # Store output
    out_ptrs = Out_ptr + o_batch_head + m_range[:, None] * stride_os + tl.arange(0, BLOCK_D)[None, :] * stride_od
    tl.store(out_ptrs, acc, mask=q_mask[:, None])


@triton.jit
def _layer_norm_fwd_kernel(
    X_ptr, Out_ptr, Weight_ptr, Bias_ptr,
    stride_xb, stride_xs, stride_xd,
    stride_ob, stride_os, stride_od,
    seq_len, embed_dim,
    eps,
    BLOCK_D: tl.constexpr,
):
    """LayerNorm kernel."""
    batch_id = tl.program_id(0)
    seq_id = tl.program_id(1)
    
    # Compute mean
    d_range = tl.arange(0, BLOCK_D)
    x_offset = batch_id * stride_xb + seq_id * stride_xs
    x = tl.load(X_ptr + x_offset + d_range * stride_xd, mask=d_range < embed_dim, other=0.0).to(tl.float32)
    mean = tl.sum(x, axis=0) / embed_dim
    
    # Compute variance
    x_centered = x - mean
    var = tl.sum(x_centered * x_centered, axis=0) / embed_dim
    rstd = 1.0 / tl.sqrt(var + eps)
    
    # Normalize and apply weight/bias
    weight = tl.load(Weight_ptr + d_range, mask=d_range < embed_dim, other=0.0).to(tl.float32)
    bias = tl.load(Bias_ptr + d_range, mask=d_range < embed_dim, other=0.0).to(tl.float32)
    out = x_centered * rstd * weight + bias
    
    # Store output
    out_offset = batch_id * stride_ob + seq_id * stride_os
    tl.store(Out_ptr + out_offset + d_range * stride_od, out, mask=d_range < embed_dim)


class FusedAttention(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, in_proj_weight, in_proj_bias, out_proj_weight, out_proj_bias,
                ln_weight, ln_bias, num_heads, eps=1e-5):
        B, C, H, W = x.shape
        seq_len = H * W
        embed_dim = C
        head_dim = embed_dim // num_heads
        
        # Reshape x to (seq_len, B, C) - same as reference
        x_reshaped = x.view(B, C, seq_len).permute(2, 0, 1).contiguous()
        
        # Compute Q, K, V projections manually
        # in_proj_weight shape: (3*embed_dim, embed_dim)
        # in_proj_bias shape: (3*embed_dim,)
        qkv = torch.nn.functional.linear(x_reshaped, in_proj_weight, in_proj_bias)
        # qkv shape: (seq_len, B, 3*embed_dim)
        
        # Split into Q, K, V and reshape for attention
        # Reference uses: qkv viewed as (seq_len, B, num_heads, 3, head_dim) then permuted
        qkv_reshaped = qkv.view(seq_len, B, num_heads, 3, head_dim)
        # Permute to (B, num_heads, seq_len, head_dim, 3) then split
        qkv_perm = qkv_reshaped.permute(1, 2, 0, 4, 3).contiguous()
        # Shape: (B, num_heads, seq_len, head_dim, 3)
        
        Q = qkv_perm[:, :, :, :, 0].contiguous()  # (B, num_heads, seq_len, head_dim)
        K = qkv_perm[:, :, :, :, 1].contiguous()
        V = qkv_perm[:, :, :, :, 2].contiguous()
        
        scale = 1.0 / (head_dim ** 0.5)
        
        # Output buffer for attention
        attn_out = torch.zeros_like(Q)
        
        # Grid for attention kernel
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_D = triton.next_power_of_2(head_dim)
        
        grid = (triton.cdiv(seq_len, BLOCK_M), num_heads, B)
        
        _attention_fwd_kernel[grid](
            Q, K, V, attn_out,
            Q.stride(0), Q.stride(1), Q.stride(2), Q.stride(3),
            K.stride(0), K.stride(1), K.stride(2), K.stride(3),
            V.stride(0), V.stride(1), V.stride(2), V.stride(3),
            attn_out.stride(0), attn_out.stride(1), attn_out.stride(2), attn_out.stride(3),
            num_heads, seq_len, head_dim,
            scale,
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_D=BLOCK_D,
        )
        
        # Reshape attention output back to (seq_len, B, embed_dim)
        # attn_out is (B, num_heads, seq_len, head_dim)
        attn_out = attn_out.permute(2, 0, 1, 3).reshape(seq_len, B, embed_dim).contiguous()
        
        # Apply output projection
        proj_out = torch.nn.functional.linear(attn_out, out_proj_weight, out_proj_bias)
        
        # Residual connection
        residual = x_reshaped + proj_out
        
        # LayerNorm
        ln_out = torch.zeros_like(residual)
        ln_grid = (B, seq_len)
        _layer_norm_fwd_kernel[ln_grid](
            residual, ln_out, ln_weight, ln_bias,
            residual.stride(0), residual.stride(1), residual.stride(2),
            ln_out.stride(0), ln_out.stride(1), ln_out.stride(2),
            seq_len, embed_dim,
            eps,
            BLOCK_D=triton.next_power_of_2(embed_dim),
        )
        
        # Reshape back to (B, C, H, W)
        ln_out = ln_out.permute(1, 2, 0).view(B, C, H, W)
        
        return ln_out


class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        
    def forward(self, x):
        B, C, H, W = x.shape
        
        # Get parameters from the original modules
        in_proj_weight = self.attn.in_proj_weight
        in_proj_bias = self.attn.in_proj_bias
        out_proj_weight = self.attn.out_proj.weight
        out_proj_bias = self.attn.out_proj.bias
        ln_weight = self.norm.weight
        ln_bias = self.norm.bias
        
        return FusedAttention.apply(
            x, in_proj_weight, in_proj_bias, out_proj_weight, out_proj_bias,
            ln_weight, ln_bias, self.num_heads
        )
