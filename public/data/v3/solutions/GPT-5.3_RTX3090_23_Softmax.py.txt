import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

template <typename scalar_t>
__global__ void softmax_row_kernel(const scalar_t* __restrict__ x, scalar_t* __restrict__ y, int rows, int cols){
    int row = blockIdx.x;
    if(row >= rows) return;

    const scalar_t* row_x = x + row * cols;
    scalar_t* row_y = y + row * cols;

    int tid = threadIdx.x;

    float local_max = -1e30f;
    for(int i = tid; i < cols; i += blockDim.x){
        float v = (float)row_x[i];
        local_max = fmaxf(local_max, v);
    }

    __shared__ float smax[256];
    smax[tid] = local_max;
    __syncthreads();

    for(int s = blockDim.x/2; s>0; s>>=1){
        if(tid < s) smax[tid] = fmaxf(smax[tid], smax[tid+s]);
        __syncthreads();
    }

    float maxv = smax[0];

    float local_sum = 0.f;
    for(int i = tid; i < cols; i += blockDim.x){
        float v = expf((float)row_x[i] - maxv);
        local_sum += v;
    }

    __shared__ float ssum[256];
    ssum[tid] = local_sum;
    __syncthreads();

    for(int s = blockDim.x/2; s>0; s>>=1){
        if(tid < s) ssum[tid] += ssum[tid+s];
        __syncthreads();
    }

    float sumv = ssum[0];

    for(int i = tid; i < cols; i += blockDim.x){
        float v = expf((float)row_x[i] - maxv) / sumv;
        row_y[i] = (scalar_t)v;
    }
}

torch::Tensor softmax_forward(torch::Tensor x){
    int rows = x.size(0);
    int cols = x.size(1);

    auto y = torch::empty_like(x);

    const int threads = 256;

    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "softmax_row_kernel", ([&]{
        softmax_row_kernel<scalar_t><<<rows, threads>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            rows,
            cols
        );
    }));

    return y;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_forward(torch::Tensor x);',
    cuda_sources=cuda_src,
    functions=['softmax_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_cuda:
            x = x.cuda()
        x = x.contiguous()
        return ext.softmax_forward(x)

batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
