import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

template <unsigned int blockSize>
__global__ void gemv_kernel_optimized_v2(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C, int M, int K) {
    extern __shared__ float sdata[];
    int row = blockIdx.x;
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    const float* row_ptr = A + row * K;
    
    int k = tid * 4;
    for (; k + 3 < K; k += blockSize * 4) {
        float4 a_val = reinterpret_cast<const float4*>(&row_ptr[k])[0];
        float4 b_val = reinterpret_cast<const float4*>(&B[k])[0];
        sum += a_val.x * b_val.x;
        sum += a_val.y * b_val.y;
        sum += a_val.z * b_val.z;
        sum += a_val.w * b_val.w;
    }
    // Remaining elements if K is not a multiple of 4*blockSize
    // The previous loop leaves out elements for whole blockSize * 4 range
    // Let's just finish up with a simple loop
    for (int i = blockSize * 4 * (K / (blockSize * 4)) + tid; i < K; i += blockSize) {
        sum += row_ptr[i] * B[i];
    }
    
    sdata[tid] = sum;
    __syncthreads();
    
    if (blockSize >= 1024) { if (tid < 512) { sdata[tid] += sdata[tid + 512]; } __syncthreads(); }
    if (blockSize >= 512) { if (tid < 256) { sdata[tid] += sdata[tid + 256]; } __syncthreads(); }
    if (blockSize >= 256) { if (tid < 128) { sdata[tid] += sdata[tid + 128]; } __syncthreads(); }
    if (blockSize >= 128) { if (tid <  64) { sdata[tid] += sdata[tid +  64]; } __syncthreads(); }
    
    if (tid < 32) {
        volatile float* vshm = sdata;
        if (blockSize >= 64) vshm[tid] += vshm[tid + 32];
        if (blockSize >= 32) vshm[tid] += vshm[tid + 16];
        if (blockSize >= 16) vshm[tid] += vshm[tid + 8];
        if (blockSize >= 8)  vshm[tid] += vshm[tid + 4];
        if (blockSize >= 4)  vshm[tid] += vshm[tid + 2];
        if (blockSize >= 2)  vshm[tid] += vshm[tid + 1];
    }
    
    if (tid == 0) {
        C[row] = sdata[0];
    }
}

torch::Tensor gemv(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    auto C = torch::empty({M, 1}, A.options());
    
    const int threads = 256;
    gemv_kernel_optimized_v2<threads><<<M, threads, threads * sizeof(float)>>>(
        A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, K);
    
    return C;
}
'''

cpp_source = 'torch::Tensor gemv(torch::Tensor A, torch::Tensor B);'

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        # Check if we need to move to CUDA
        if not A.is_cuda:
            A = A.cuda()
        if not B.is_cuda:
            B = B.cuda()
        return ext.gemv(A, B)

def get_inputs():
    M = 256
    K = 131072
    A = torch.randn(M, K, device='cuda')
    B = torch.randn(K, 1, device='cuda')
    return [A, B]

def get_init_inputs():
    return []
