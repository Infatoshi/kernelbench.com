import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def fused_post_pool_kernel(
    x_ptr,
    y_ptr,
    H,
    W,
    OH,
    OW,
    sub1,
    sub2,
    total,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    offs = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offs < total

    ow = offs % OW
    tmp = offs // OW
    oh = tmp % OH
    c = (tmp // OH) % 16
    n = tmp // (OH * 16)

    h0 = oh * 2
    w0 = ow * 2

    base = ((n * 16 + c) * H)
    idx00 = (base + h0) * W + w0
    idx01 = idx00 + 1
    idx10 = idx00 + W
    idx11 = idx10 + 1

    v00 = tl.load(x_ptr + idx00, mask=mask, other=0.0)
    v01 = tl.load(x_ptr + idx01, mask=mask, other=0.0)
    v10 = tl.load(x_ptr + idx10, mask=mask, other=0.0)
    v11 = tl.load(x_ptr + idx11, mask=mask, other=0.0)

    z00 = v00 - sub1
    z01 = v01 - sub1
    z10 = v10 - sub1
    z11 = v11 - sub1
    e00 = tl.exp(2.0 * z00)
    e01 = tl.exp(2.0 * z01)
    e10 = tl.exp(2.0 * z10)
    e11 = tl.exp(2.0 * z11)
    t00 = (e00 - 1.0) / (e00 + 1.0)
    t01 = (e01 - 1.0) / (e01 + 1.0)
    t10 = (e10 - 1.0) / (e10 + 1.0)
    t11 = (e11 - 1.0) / (e11 + 1.0)
    out = (t00 + t01 + t10 + t11) * 0.25 - sub2
    tl.store(y_ptr + offs, out, mask=mask)


class Model(nn.Module):
    """
    Faster model for this fixed benchmark shape.
    Uses conv2d for the convolution, then fuses subtract/tanh/subtract/avgpool in
    a single Triton kernel over the convolution output, avoiding multiple PyTorch ops.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        x = self.conv(x)
        if x.is_cuda and self.kernel_size_pool == 2 and x.ndim == 4 and x.shape[1] == 16:
            N, C, H, W = x.shape
            OH = H // 2
            OW = W // 2
            y = torch.empty((N, C, OH, OW), device=x.device, dtype=x.dtype)
            total = y.numel()
            grid = lambda meta: (triton.cdiv(total, meta['BLOCK']),)
            fused_post_pool_kernel[grid](
                x, y,
                H, W, OH, OW,
                self.subtract1_value,
                self.subtract2_value,
                total,
                BLOCK=1024,
            )
            return y
        x = x - self.subtract1_value
        x = torch.tanh(x)
        x = x - self.subtract2_value
        x = torch.nn.functional.avg_pool2d(x, self.kernel_size_pool)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
