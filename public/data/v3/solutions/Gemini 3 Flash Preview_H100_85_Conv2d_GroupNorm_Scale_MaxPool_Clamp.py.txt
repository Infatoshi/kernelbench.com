cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>

__global__ void conv_gn_scale_maxpool_clamp_kernel(
    const float* __restrict__ input,      // [N, InC, H, W]
    const float* __restrict__ weight,     // [OutC, InC, KH, KW]
    const float* __restrict__ bias,       // [OutC]
    const float* __restrict__ gn_gamma,   // [OutC]
    const float* __restrict__ gn_beta,    // [OutC]
    const float* __restrict__ scale,      // [OutC, 1, 1]
    const float* __restrict__ gn_mean,    // [N, NumGroups]
    const float* __restrict__ gn_var_inv, // [N, NumGroups]
    float* __restrict__ output,            // [N, OutC, H_out, W_out]
    int N, int InC, int OutC, int H, int W,
    int KH, int KW, int NumGroups,
    int PoolK, float clamp_min, float clamp_max,
    int H_conv, int W_conv, int H_out, int W_out
) {
    int n = blockIdx.z;
    int oc = blockIdx.y;
    int oh = blockIdx.x / W_out;
    int ow = blockIdx.x % W_out;

    if (n >= N || oc >= OutC || oh >= H_out || ow >= W_out) return;

    int g_size = OutC / NumGroups;
    int g = oc / g_size;
    float m = gn_mean[n * NumGroups + g];
    float v_inv = gn_var_inv[n * NumGroups + g];
    float s = scale[oc];
    float gamma = gn_gamma[oc];
    float beta = gn_beta[oc];

    float max_val = -FLT_MAX;

    for (int ph = 0; ph < PoolK; ++ph) {
        int ch = oh * PoolK + ph;
        for (int pw = 0; pw < PoolK; ++pw) {
            int cw = ow * PoolK + pw;
            
            if (ch < H_conv && cw < W_conv) {
                // Compute Conv2d at (n, oc, ch, cw)
                float conv_val = bias[oc];
                for (int ic = 0; ic < InC; ++ic) {
                    for (int kh = 0; kh < KH; ++kh) {
                        for (int kw = 0; kw < KW; ++kw) {
                            int ih = ch + kh;
                            int iw = cw + kw;
                            if (ih < H && iw < W) {
                                float val = input[((n * InC + ic) * H + ih) * W + iw];
                                float w = weight[((oc * InC + ic) * KH + kh) * KW + kw];
                                conv_val += val * w;
                            }
                        }
                    }
                }
                
                // GroupNorm: (x - mean) * var_inv * gamma + beta
                float gn_val = (conv_val - m) * v_inv * gamma + beta;
                // Scale
                float scaled_val = gn_val * s;
                
                if (scaled_val > max_val) max_val = scaled_val;
            }
        }
    }

    // Clamp
    if (max_val < clamp_min) max_val = clamp_min;
    if (max_val > clamp_max) max_val = clamp_max;

    output[((n * OutC + oc) * H_out + oh) * W_out + ow] = max_val;
}

    int W_conv = W - KW + 1;

    int W_out = W_conv / pool_k;

    auto output = torch::empty({N, OutC, H_out, W_out}, x.options());

    // Before main kernel, we need mean and variance for GN.
    // Since we're not allowed to use F.conv2d, we'll do it naively or within the kernel.

# Let's rethink. If we cannot use F.conv2d, we must implement it.
# But for GN, we need the mean of the conv output.
# So we must compute the conv output first.

# Let's use Triton for efficiency and to avoid forbidden PyTorch ops.

import triton
import triton.language as tl

@triton.jit
def conv_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    N, InC, OutC, H_in, W_in, KH, KW, H_out, W_out,
    stride_in_n, stride_in_c, stride_in_h, stride_in_w,
    stride_out_n, stride_out_c, stride_out_h, stride_out_w,
    BLOCK_SIZE_H: tl.constexpr, BLOCK_SIZE_W: tl.constexpr,
    BLOCK_SIZE_C: tl.constexpr
):
    # This is a basic 2D convolution kernel
    n = tl.program_id(0)
    oc = tl.program_id(1)
    # block_h and block_w
    pid_hw = tl.program_id(2)
    num_blocks_w = tl.cdiv(W_out, BLOCK_SIZE_W)
    block_h = pid_hw // num_blocks_w
    block_w = pid_hw % num_blocks_w

    offs_h = block_h * BLOCK_SIZE_H + tl.arange(0, BLOCK_SIZE_H)
    offs_w = block_w * BLOCK_SIZE_W + tl.arange(0, BLOCK_SIZE_W)
    
    # Init accumulator with bias
    acc = tl.zeros((BLOCK_SIZE_H, BLOCK_SIZE_W), dtype=tl.float32)
    bias_val = tl.load(bias_ptr + oc)
    acc += bias_val
    
    for ic in range(InC):
        for kh in range(KH):
            for kw in range(KW):
                # Load input patch
                h_in = offs_h[:, None] + kh
                w_in = offs_w[None, :] + kw
                
                # Check bounds for input
                mask = (h_in < H_in) & (w_in < W_in)
                input_ptrs = input_ptr + n * stride_in_n + ic * stride_in_c + h_in * stride_in_h + w_in * stride_in_w
                in_val = tl.load(input_ptrs, mask=mask, other=0.0)
                
                # Load weight
                w_val = tl.load(weight_ptr + oc * (InC * KH * KW) + ic * (KH * KW) + kh * KW + kw)
                
                acc += in_val * w_val
    
    # Store result
    out_mask = (offs_h[:, None] < H_out) & (offs_w[None, :] < W_out)
    output_ptrs = output_ptr + n * stride_out_n + oc * stride_out_c + offs_h[:, None] * stride_out_h + offs_w[None, :] * stride_out_w
    tl.store(output_ptrs, acc, mask=out_mask)

@triton.jit
def gn_scale_pool_clamp_kernel(
    x_ptr, gamma_ptr, beta_ptr, scale_ptr, y_ptr,
    N, C, H, W, NumGroups, PoolK, clamp_min, clamp_max,
    H_out, W_out,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_yn, stride_yc, stride_yh, stride_yw
):
    n = tl.program_id(0)
    g = tl.program_id(1)
    
    group_size = C // NumGroups
    
    # Compute mean and var for this group
    # We'll use a simple but possibly slow way for mean/var
    sum_val = 0.0
    sum_sq_val = 0.0
    count = group_size * H * W
    
    for c_idx in range(group_size):
        c = g * group_size + c_idx
        for h in range(H):
            for w in range(W):
                val = tl.load(x_ptr + n * stride_xn + c * stride_xc + h * stride_xh + w * stride_xw)
                sum_val += val
                sum_sq_val += val * val
    
    mean = sum_val / count
    var = (sum_sq_val / count) - (mean * mean)
    inv_std = 1.0 / tl.sqrt(var + 1e-5)
    
    # Process each pixel in this group
    for oh in range(H_out):
        for ow in range(W_out):
            # For each channel in the group
            for c_idx in range(group_size):
                c = g * group_size + c_idx
                
                # Max pooling over PoolK x PoolK
                max_val = -1e38 # Very small
                for ph in range(PoolK):
                    for pw in range(PoolK):
                        ih = oh * PoolK + ph
                        iw = ow * PoolK + pw
                        if ih < H and iw < W:
                            # GN and scale
                            val = tl.load(x_ptr + n * stride_xn + c * stride_xc + ih * stride_xh + iw * stride_xw)
                            gamma = tl.load(gamma_ptr + c)
                            beta = tl.load(beta_ptr + c)
                            scale = tl.load(scale_ptr + c)
                            
                            norm_val = (val - mean) * inv_std * gamma + beta
                            scaled_val = norm_val * scale
                            
                            if scaled_val > max_val:
                                max_val = scaled_val
                
                # Clamp
                if max_val < clamp_min: max_val = clamp_min
                if max_val > clamp_max: max_val = clamp_max
                
                # Store
                tl.store(y_ptr + n * stride_yn + c * stride_yc + oh * stride_yh + ow * stride_yw, max_val)

# Wait, the kernels above are quite inefficient. Let's try to leverage PyTorch for Mean/Var and then use kernels.
# But "no PyTorch operator fallbacks". Does that include torch.mean?
# "No PyTorch operator fallbacks (torch.matmul, torch.mm, torch.conv2d, F.linear, F.conv2d)".
# These are the ones explicitly forbidden. Other ops like torch.mean or torch.sum might be okay,
# but it's safer to avoid them.

# Actually, the most efficient way for H100 would be to use CUTLASS for Conv.
# But for simplicity and to meet the requirement, I'll use a better Triton implementation.

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool = nn.MaxPool2d(kernel_size=maxpool_kernel_size)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.num_groups = num_groups
        self.pool_k = maxpool_kernel_size

    def forward(self, x):
        # Instead of writing custom Triton for conv, which can be hard to match performance 
        # of cuDNN, let's see if we can use a trick or if just 'torch.conv2d' is forbidden.
        # The prompt says: "No PyTorch operator fallbacks (torch.matmul, torch.mm, torch.conv2d, F.linear, F.conv2d)".
        # Okay, so I MUST avoid torch.conv2d.
        
        # Let's write a Triton-based Conv2d and a Fused GN/Scale/Pool/Clamp.
        N, InC, H_in, W_in = x.shape
        OutC, _, KH, KW = self.conv.weight.shape
        H_conv = H_in - KH + 1
        W_conv = W_in - KW + 1
        
        conv_out = torch.empty((N, OutC, H_conv, W_conv), device=x.device, dtype=x.dtype)
        
        grid_conv = (N, OutC, triton.cdiv(H_conv, 16) * triton.cdiv(W_conv, 16))
        conv_kernel[grid_conv](
            x, self.conv.weight, self.conv.bias, conv_out,
            N, InC, OutC, H_in, W_in, KH, KW, H_conv, W_conv,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            conv_out.stride(0), conv_out.stride(1), conv_out.stride(2), conv_out.stride(3),
            BLOCK_SIZE_H=16, BLOCK_SIZE_W=16, BLOCK_SIZE_C=1
        )
        
        H_out = H_conv // self.pool_k
        W_out = W_conv // self.pool_k
        y = torch.empty((N, OutC, H_out, W_out), device=x.device, dtype=x.dtype)
        
        grid_gn = (N, self.num_groups)
        gn_scale_pool_clamp_kernel[grid_gn](
            conv_out, self.group_norm.weight, self.group_norm.bias, self.scale.view(-1), y,
            N, OutC, H_conv, W_conv, self.num_groups, self.pool_k, self.clamp_min, self.clamp_max,
            H_out, W_out,
            conv_out.stride(0), conv_out.stride(1), conv_out.stride(2), conv_out.stride(3),
            y.stride(0), y.stride(1), y.stride(2), y.stride(3)
        )
        return y
