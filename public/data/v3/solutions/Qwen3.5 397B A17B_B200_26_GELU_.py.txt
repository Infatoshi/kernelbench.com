import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// GELU approximation using fast math
// gelu(x) = 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))

__device__ __forceinline__ float fast_gelu(float x) {
    const float sqrt_2_over_pi = 0.7978845608028654f;
    const float c = 0.044715f;
    float x3 = x * x * x;
    float inner = sqrt_2_over_pi * (x + c * x3);
    float tanh_inner = tanhf(inner);
    return 0.5f * x * (1.0f + tanh_inner);
}

__device__ __forceinline__ __half fast_gelu_half(__half x) {
    float xf = __half2float(x);
    float result = fast_gelu(xf);
    return __float2half(result);
}

__device__ __forceinline__ __nv_bfloat16 fast_gelu_bf16(__nv_bfloat16 x) {
    float xf = __bfloat162float(x);
    float result = fast_gelu(xf);
    return __float2bfloat16(result);
}

// FP32 kernel with float4 vectorized access
__global__ void gelu_kernel_fp32_vec(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = blockDim.x * gridDim.x;
    
    // Process 4 floats at a time
    int n_vec = n / 4;
    int vec_idx = idx;
    
    while (vec_idx < n_vec) {
        float4 v = reinterpret_cast<const float4*>(x)[vec_idx];
        v.x = fast_gelu(v.x);
        v.y = fast_gelu(v.y);
        v.z = fast_gelu(v.z);
        v.w = fast_gelu(v.w);
        reinterpret_cast<float4*>(y)[vec_idx] = v;
        vec_idx += total_threads;
    }
    
    // Handle remainder
    int rem_start = n_vec * 4;
    for (int i = rem_start + idx; i < n; i += total_threads) {
        y[i] = fast_gelu(x[i]);
    }
}

// FP32 kernel for smaller tensors
__global__ void gelu_kernel_fp32(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    
    for (int i = idx; i < n; i += stride) {
        y[i] = fast_gelu(x[i]);
    }
}

// FP16 kernel with vectorized access (using half2)
__global__ void gelu_kernel_fp16_vec(const __half* __restrict__ x, __half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = blockDim.x * gridDim.x;
    
    // Process 2 halfs at a time using half2
    int n_vec = n / 2;
    int vec_idx = idx;
    
    while (vec_idx < n_vec) {
        half2 v = reinterpret_cast<const half2*>(x)[vec_idx];
        float x0 = __half2float(v.x);
        float x1 = __half2float(v.y);
        float r0 = fast_gelu(x0);
        float r1 = fast_gelu(x1);
        v.x = __float2half(r0);
        v.y = __float2half(r1);
        reinterpret_cast<half2*>(y)[vec_idx] = v;
        vec_idx += total_threads;
    }
    
    // Handle remainder
    int rem_start = n_vec * 2;
    for (int i = rem_start + idx; i < n; i += total_threads) {
        y[i] = fast_gelu_half(x[i]);
    }
}

// BF16 kernel with vectorized access (using __nv_bfloat162)
__global__ void gelu_kernel_bf16_vec(const __nv_bfloat16* __restrict__ x, __nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = blockDim.x * gridDim.x;
    
    // Process 2 bfloat16 at a time using __nv_bfloat162
    int n_vec = n / 2;
    int vec_idx = idx;
    
    while (vec_idx < n_vec) {
        __nv_bfloat162 v = reinterpret_cast<const __nv_bfloat162*>(x)[vec_idx];
        float x0 = __bfloat162float(v.x);
        float x1 = __bfloat162float(v.y);
        float r0 = fast_gelu(x0);
        float r1 = fast_gelu(x1);
        v.x = __float2bfloat16(r0);
        v.y = __float2bfloat16(r1);
        reinterpret_cast<__nv_bfloat162*>(y)[vec_idx] = v;
        vec_idx += total_threads;
    }
    
    // Handle remainder
    int rem_start = n_vec * 2;
    for (int i = rem_start + idx; i < n; i += total_threads) {
        y[i] = fast_gelu_bf16(x[i]);
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    if (n == 0) return y;
    
    int threads = 256;
    int blocks = 512;  // Fixed number of blocks for good occupancy
    
    if (x.dtype() == torch::kFloat32) {
        if (n >= 512) {
            gelu_kernel_fp32_vec<<<blocks, threads>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), n);
        } else {
            gelu_kernel_fp32<<<1, threads>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), n);
        }
    } else if (x.dtype() == torch::kFloat16) {
        if (n >= 512) {
            gelu_kernel_fp16_vec<<<blocks, threads>>>(
                reinterpret_cast<__half*>(x.data_ptr<at::Half>()), 
                reinterpret_cast<__half*>(y.data_ptr<at::Half>()), n);
        } else {
            gelu_kernel_fp16_vec<<<1, threads>>>(
                reinterpret_cast<__half*>(x.data_ptr<at::Half>()), 
                reinterpret_cast<__half*>(y.data_ptr<at::Half>()), n);
        }
    } else if (x.dtype() == torch::kBFloat16) {
        if (n >= 512) {
            gelu_kernel_bf16_vec<<<blocks, threads>>>(
                reinterpret_cast<__nv_bfloat16*>(x.data_ptr<at::BFloat16>()), 
                reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
        } else {
            gelu_kernel_bf16_vec<<<1, threads>>>(
                reinterpret_cast<__nv_bfloat16*>(x.data_ptr<at::BFloat16>()), 
                reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
        }
    } else {
        // Fallback for other types - convert to fp32
        auto x_fp32 = x.to(torch::kFloat32);
        auto y_fp32 = torch::empty_like(x_fp32);
        gelu_kernel_fp32<<<blocks, threads>>>(
            x_fp32.data_ptr<float>(), y_fp32.data_ptr<float>(), n);
        y = y_fp32.to(x.dtype());
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(cudaGetErrorString(err));
    }
    
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    """
    Optimized GELU activation using custom CUDA kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_op(x)

batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []  # No special initialization inputs needed
