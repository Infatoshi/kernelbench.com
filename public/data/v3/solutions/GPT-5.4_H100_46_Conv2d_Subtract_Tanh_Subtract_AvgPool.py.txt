import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

_cpp_src = r'''
#include <torch/extension.h>
torch::Tensor fused_post(torch::Tensor x, double sub1, double sub2);
'''

_cuda_src = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

#if !defined(__CUDA_ARCH__) || __CUDA_ARCH__ >= 600
#else
#error "Requires CUDA arch >= 600"
#endif

template <typename scalar_t, typename acc_t>
__global__ void fused_post_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    acc_t sub1, acc_t sub2) {

    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int nc = blockIdx.z;

    int HWo = W >> 1;
    int HHo = H >> 1;
    if (ow >= HWo || oh >= HHo) return;

    int c = nc % C;
    int n = nc / C;

    int h0 = oh << 1;
    int w0 = ow << 1;

    int base = ((n * C + c) * H + h0) * W + w0;
    int idx00 = base;
    int idx01 = base + 1;
    int idx10 = base + W;
    int idx11 = base + W + 1;

    acc_t v00 = tanh((acc_t)x[idx00] - sub1) - sub2;
    acc_t v01 = tanh((acc_t)x[idx01] - sub1) - sub2;
    acc_t v10 = tanh((acc_t)x[idx10] - sub1) - sub2;
    acc_t v11 = tanh((acc_t)x[idx11] - sub1) - sub2;

    acc_t out = (v00 + v01 + v10 + v11) * acc_t(0.25);
    int out_idx = ((n * C + c) * HHo + oh) * HWo + ow;
    y[out_idx] = (scalar_t)out;
}

torch::Tensor fused_post(torch::Tensor x, double sub1, double sub2) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA tensor");
    TORCH_CHECK(x.dim() == 4, "x must be NCHW 4D");
    TORCH_CHECK(x.size(2) % 2 == 0 && x.size(3) % 2 == 0, "H and W must be even");
    auto xc = x.contiguous();
    auto N = (int)xc.size(0);
    auto C = (int)xc.size(1);
    auto H = (int)xc.size(2);
    auto W = (int)xc.size(3);
    auto y = torch::empty({N, C, H / 2, W / 2}, xc.options());

    dim3 block(16, 16);
    dim3 grid((W / 2 + block.x - 1) / block.x,
              (H / 2 + block.y - 1) / block.y,
              N * C);

    cudaStream_t stream = at::cuda::getDefaultCUDAStream();

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, xc.scalar_type(), "fused_post", [&] {
        using acc_t = float;
        fused_post_kernel<scalar_t, acc_t><<<grid, block, 0, stream>>>(
            xc.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(), N, C, H, W,
            (acc_t)sub1, (acc_t)sub2);
    });
    return y;
}
'''

_ext = load_inline(
    name='fused_conv_post_h100_v1',
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=['fused_post'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        x = self.conv(x)
        if x.is_cuda and self.kernel_size_pool == 2 and x.shape[-1] % 2 == 0 and x.shape[-2] % 2 == 0:
            return _ext.fused_post(x, float(self.subtract1_value), float(self.subtract2_value))
        x = x - self.subtract1_value
        x = torch.tanh(x)
        x = x - self.subtract2_value
        x = F.avg_pool2d(x, self.kernel_size_pool)
        return x

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
