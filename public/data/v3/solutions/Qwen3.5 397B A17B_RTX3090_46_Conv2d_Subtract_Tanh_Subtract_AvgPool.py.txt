import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <math.h>

#define BLOCK_SIZE 16
#define POOL_SIZE 2

// Optimized kernel with better memory access pattern
__global__ void fused_conv_tanh_pool_kernel_v2(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int batch_size,
    int in_channels,
    int in_height,
    int in_width,
    int out_channels,
    int kernel_size,
    int out_height,
    int out_width,
    float subtract1_value,
    float subtract2_value,
    int pool_kernel
) {
    int pooled_h = (out_height + pool_kernel - 1) / pool_kernel;
    int pooled_w = (out_width + pool_kernel - 1) / pool_kernel;
    
    int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    int out_c = blockIdx.z % out_channels;
    int out_b = blockIdx.z / out_channels;
    
    if (out_b >= batch_size || out_c >= out_channels || out_x >= pooled_w || out_y >= pooled_h) {
        return;
    }
    
    float sum = 0.0f;
    int count = 0;
    
    // Iterate over pooling region
    for (int py = 0; py < pool_kernel; py++) {
        for (int px = 0; px < pool_kernel; px++) {
            int conv_out_x = out_x * pool_kernel + px;
            int conv_out_y = out_y * pool_kernel + py;
            
            if (conv_out_x >= out_width || conv_out_y >= out_height) {
                continue;
            }
            
            // Convolution
            float conv_sum = bias[out_c];
            for (int ic = 0; ic < in_channels; ic++) {
                for (int ky = 0; ky < kernel_size; ky++) {
                    for (int kx = 0; kx < kernel_size; kx++) {
                        int in_x = conv_out_x + kx;
                        int in_y = conv_out_y + ky;
                        
                        if (in_x < in_width && in_y < in_height) {
                            int in_idx = ((out_b * in_channels + ic) * in_height + in_y) * in_width + in_x;
                            int w_idx = ((out_c * in_channels + ic) * kernel_size + ky) * kernel_size + kx;
                            conv_sum += input[in_idx] * weight[w_idx];
                        }
                    }
                }
            }
            
            // Apply activation and subtraction
            float val = conv_sum - subtract1_value;
            val = tanhf(val);
            val = val - subtract2_value;
            
            sum += val;
            count++;
        }
    }
    
    int out_idx = ((out_b * out_channels + out_c) * pooled_h + out_y) * pooled_w + out_x;
    output[out_idx] = sum / count;
}

torch::Tensor fused_conv_tanh_pool_v2(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float subtract1_value,
    float subtract2_value,
    int pool_kernel
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_height = in_height - kernel_size + 1;
    int out_width = in_width - kernel_size + 1;
    int pooled_h = (out_height + pool_kernel - 1) / pool_kernel;
    int pooled_w = (out_width + pool_kernel - 1) / pool_kernel;
    
    auto output = torch::empty({batch_size, out_channels, pooled_h, pooled_w}, input.options());
    
    dim3 block(BLOCK_SIZE, BLOCK_SIZE);
    dim3 grid((pooled_w + BLOCK_SIZE - 1) / BLOCK_SIZE, 
              (pooled_h + BLOCK_SIZE - 1) / BLOCK_SIZE,
              batch_size * out_channels);
    
    fused_conv_tanh_pool_kernel_v2<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_channels, in_height, in_width,
        out_channels, kernel_size, out_height, out_width,
        subtract1_value, subtract2_value, pool_kernel
    );
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor fused_conv_tanh_pool_v2(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float subtract1_value,
    float subtract2_value,
    int pool_kernel
);
'''

# Compile the CUDA extension
ext = load_inline(
    name='fused_conv_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_tanh_pool_v2'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    Optimized fused model: Conv2d -> Subtract -> Tanh -> Subtract -> AvgPool2d
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        weight = self.conv.weight
        bias = self.conv.bias
        return ext.fused_conv_tanh_pool_v2(x, weight, bias, 
                                           self.subtract1_value, 
                                           self.subtract2_value,
                                           self.kernel_size_pool)


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    subtract1_value = 0.5
    subtract2_value = 0.2
    kernel_size_pool = 2
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
