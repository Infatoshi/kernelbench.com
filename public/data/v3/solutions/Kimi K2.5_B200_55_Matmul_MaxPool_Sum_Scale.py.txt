import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# CUDA kernel for fused Linear + MaxPool1d(kernel=2) + Sum + Scale
# Optimized for B200 (sm_100) but using sm_90 for compatibility
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename T>
__device__ __forceinline__ float to_float(T x) {
    return static_cast<float>(x);
}

template <> 
__device__ __forceinline__ float to_float(at::Half x) {
    return __half2float(x);
}

template <> 
__device__ __forceinline__ float to_float(at::BFloat16 x) {
    return __bfloat162float(x);
}

template <typename T>
__global__ void fused_kernel(
    const T* __restrict__ x,
    const T* __restrict__ W,
    const T* __restrict__ b,
    float* __restrict__ out,
    int in_features,
    int out_features,
    float scale
) {
    int batch_idx = blockIdx.x;
    int tid = threadIdx.x;
    int num_pairs = out_features / 2;
    
    float thread_sum = 0.0f;
    const T* x_row = x + batch_idx * in_features;
    
    // Grid-stride loop over output pairs
    for (int pair_idx = tid; pair_idx < num_pairs; pair_idx += blockDim.x) {
        int idx0 = pair_idx * 2;
        int idx1 = idx0 + 1;
        
        float sum0 = (b != nullptr) ? to_float(b[idx0]) : 0.0f;
        float sum1 = (b != nullptr) ? to_float(b[idx1]) : 0.0f;
        
        // Compute dot products
        #pragma unroll 4
        for (int i = 0; i < in_features; ++i) {
            float xv = to_float(x_row[i]);
            sum0 += xv * to_float(W[idx0 * in_features + i]);
            sum1 += xv * to_float(W[idx1 * in_features + i]);
        }
        
        // MaxPool (take max of pair)
        thread_sum += (sum0 > sum1) ? sum0 : sum1;
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        thread_sum += __shfl_down_sync(0xFFFFFFFF, thread_sum, offset);
    }
    
    // Shared memory reduction across warps
    __shared__ float sdata[256];
    sdata[tid] = thread_sum;
    __syncthreads();
    
    if (tid < 128) sdata[tid] += sdata[tid + 128];
    __syncthreads();
    if (tid < 64) sdata[tid] += sdata[tid + 64];
    __syncthreads();
    
    if (tid < 32) {
        float val = sdata[tid];
        val += __shfl_down_sync(0xFFFFFFFF, val, 16);
        val += __shfl_down_sync(0xFFFFFFFF, val, 8);
        val += __shfl_down_sync(0xFFFFFFFF, val, 4);
        val += __shfl_down_sync(0xFFFFFFFF, val, 2);
        val += __shfl_down_sync(0xFFFFFFFF, val, 1);
        
        if (tid == 0) {
            out[batch_idx] = val * scale;
        }
    }
}

torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor W,
    torch::Tensor b,
    float scale
) {
    int batch_size = x.size(0);
    int in_features = x.size(1);
    int out_features = W.size(0);
    
    auto out = torch::zeros({batch_size}, torch::dtype(torch::kFloat32).device(x.device()));
    
    const int threads = 256;
    dim3 grid(batch_size);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "fused_kernel", [&] {
        using T = scalar_t;
        fused_kernel<T><<<grid, threads>>>(
            x.data_ptr<T>(),
            W.data_ptr<T>(),
            b.defined() ? b.data_ptr<T>() : nullptr,
            out.data_ptr<float>(),
            in_features,
            out_features,
            scale
        );
    });
    
    return out;
}
'''

cpp_source = r'''
torch::Tensor fused_forward(torch::Tensor x, torch::Tensor W, torch::Tensor b, float scale);
'''

# Compile the extension
ext = load_inline(
    name='fused_linear_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-gencode=arch=compute_90,code=sm_90'],
    verbose=False,
)

# Model configuration
batch_size = 128
in_features = 4096
out_features = 4096
kernel_size = 2
scale_factor = 0.5


class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        x_c = x.contiguous()
        W_c = self.weight.contiguous()
        b_c = self.bias.contiguous()
        return ext.fused_forward(x_c, W_c, b_c, float(self.scale_factor))


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, kernel_size, scale_factor]
