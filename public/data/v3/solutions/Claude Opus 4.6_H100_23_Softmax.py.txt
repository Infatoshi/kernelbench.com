import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>
#include <cmath>

// Warp-level reduction for max
__device__ __forceinline__ float warpReduceMax(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xFFFFFFFF, val, offset));
    return val;
}

// Warp-level reduction for sum
__device__ __forceinline__ float warpReduceSum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_xor_sync(0xFFFFFFFF, val, offset);
    return val;
}

// Block-level reduction for max using shared memory
__device__ __forceinline__ float blockReduceMax(float val, float* shared) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    val = warpReduceMax(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    // Only first warp does the final reduction
    int num_warps = blockDim.x >> 5;
    val = (threadIdx.x < num_warps) ? shared[lane] : -FLT_MAX;
    if (wid == 0) val = warpReduceMax(val);
    
    // Broadcast result
    if (threadIdx.x == 0) shared[0] = val;
    __syncthreads();
    return shared[0];
}

// Block-level reduction for sum using shared memory
__device__ __forceinline__ float blockReduceSum(float val, float* shared) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    val = warpReduceSum(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    int num_warps = blockDim.x >> 5;
    val = (threadIdx.x < num_warps) ? shared[lane] : 0.0f;
    if (wid == 0) val = warpReduceSum(val);
    
    if (threadIdx.x == 0) shared[0] = val;
    __syncthreads();
    return shared[0];
}

// Online softmax kernel: one block per row
// Each thread processes multiple elements using a stride
__global__ void softmax_kernel(const float* __restrict__ input, 
                                float* __restrict__ output,
                                int n_cols) {
    __shared__ float smem[32]; // enough for 32 warps
    
    int row = blockIdx.x;
    const float* row_in = input + row * n_cols;
    float* row_out = output + row * n_cols;
    
    // Pass 1: Find max
    float local_max = -FLT_MAX;
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        local_max = fmaxf(local_max, row_in[i]);
    }
    float row_max = blockReduceMax(local_max, smem);
    
    // Pass 2: Compute exp and sum
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        local_sum += __expf(row_in[i] - row_max);
    }
    float row_sum = blockReduceSum(local_sum, smem);
    
    // Pass 3: Normalize and write
    float inv_sum = 1.0f / row_sum;
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        row_out[i] = __expf(row_in[i] - row_max) * inv_sum;
    }
}

// Vectorized version using float4 for coalesced memory access
__global__ void softmax_kernel_vec4(const float* __restrict__ input, 
                                     float* __restrict__ output,
                                     int n_cols) {
    __shared__ float smem[32];
    
    int row = blockIdx.x;
    const float4* row_in = reinterpret_cast<const float4*>(input + row * n_cols);
    float4* row_out = reinterpret_cast<float4*>(output + row * n_cols);
    int n_cols4 = n_cols >> 2;  // n_cols / 4
    
    // Pass 1: Find max
    float local_max = -FLT_MAX;
    for (int i = threadIdx.x; i < n_cols4; i += blockDim.x) {
        float4 v = row_in[i];
        local_max = fmaxf(local_max, fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w)));
    }
    float row_max = blockReduceMax(local_max, smem);
    
    // Pass 2: Compute exp and sum
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < n_cols4; i += blockDim.x) {
        float4 v = row_in[i];
        local_sum += __expf(v.x - row_max) + __expf(v.y - row_max) 
                   + __expf(v.z - row_max) + __expf(v.w - row_max);
    }
    float row_sum = blockReduceSum(local_sum, smem);
    
    // Pass 3: Normalize and write
    float inv_sum = 1.0f / row_sum;
    for (int i = threadIdx.x; i < n_cols4; i += blockDim.x) {
        float4 v = row_in[i];
        float4 out;
        out.x = __expf(v.x - row_max) * inv_sum;
        out.y = __expf(v.y - row_max) * inv_sum;
        out.z = __expf(v.z - row_max) * inv_sum;
        out.w = __expf(v.w - row_max) * inv_sum;
        row_out[i] = out;
    }
}

// Two-pass online softmax using shared memory to cache values
// This reduces global memory reads from 3 passes to 2
__global__ void softmax_kernel_2pass(const float* __restrict__ input, 
                                      float* __restrict__ output,
                                      int n_cols) {
    extern __shared__ float shared_data[];
    __shared__ float smem_reduce[32];
    
    int row = blockIdx.x;
    const float4* row_in = reinterpret_cast<const float4*>(input + row * n_cols);
    float4* row_out = reinterpret_cast<float4*>(output + row * n_cols);
    int n_cols4 = n_cols >> 2;
    
    // Online softmax: single pass for max and sum
    float local_max = -FLT_MAX;
    float local_sum = 0.0f;
    
    for (int i = threadIdx.x; i < n_cols4; i += blockDim.x) {
        float4 v = row_in[i];
        float m = fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w));
        
        if (m > local_max) {
            local_sum *= __expf(local_max - m);
            local_max = m;
        }
        local_sum += __expf(v.x - local_max) + __expf(v.y - local_max) 
                   + __expf(v.z - local_max) + __expf(v.w - local_max);
    }
    
    // Block-level online reduction for max+sum
    // Warp-level first
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other_max = __shfl_xor_sync(0xFFFFFFFF, local_max, offset);
        float other_sum = __shfl_xor_sync(0xFFFFFFFF, local_sum, offset);
        if (other_max > local_max) {
            local_sum = local_sum * __expf(local_max - other_max) + other_sum;
            local_max = other_max;
        } else {
            local_sum = local_sum + other_sum * __expf(other_max - local_max);
        }
    }
    
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    if (lane == 0) {
        smem_reduce[wid] = local_max;
        smem_reduce[wid + 16] = local_sum;  // store sum in upper half
    }
    __syncthreads();
    
    int num_warps = blockDim.x >> 5;
    if (wid == 0) {
        local_max = (lane < num_warps) ? smem_reduce[lane] : -FLT_MAX;
        local_sum = (lane < num_warps) ? smem_reduce[lane + 16] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            float other_max = __shfl_xor_sync(0xFFFFFFFF, local_max, offset);
            float other_sum = __shfl_xor_sync(0xFFFFFFFF, local_sum, offset);
            if (other_max > local_max) {
                local_sum = local_sum * __expf(local_max - other_max) + other_sum;
                local_max = other_max;
            } else {
                local_sum = local_sum + other_sum * __expf(other_max - local_max);
            }
        }
        
        if (lane == 0) {
            smem_reduce[0] = local_max;
            smem_reduce[1] = local_sum;
        }
    }
    __syncthreads();
    
    float row_max = smem_reduce[0];
    float row_sum = smem_reduce[1];
    float inv_sum = 1.0f / row_sum;
    
    // Pass 2: Normalize and write
    for (int i = threadIdx.x; i < n_cols4; i += blockDim.x) {
        float4 v = row_in[i];
        float4 out;
        out.x = __expf(v.x - row_max) * inv_sum;
        out.y = __expf(v.y - row_max) * inv_sum;
        out.z = __expf(v.z - row_max) * inv_sum;
        out.w = __expf(v.w - row_max) * inv_sum;
        row_out[i] = out;
    }
}


torch::Tensor softmax_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n_rows = input.size(0);
    int n_cols = input.size(1);
    
    // Use vectorized 2-pass kernel
    int threads = 256;
    softmax_kernel_2pass<<<n_rows, threads, 0>>>(
        input.data_ptr<float>(), output.data_ptr<float>(), n_cols
    );
    
    return output;
}
