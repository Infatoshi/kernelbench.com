import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized GEMV for H100: y = x @ W^T
// x: (M, K), W: (N, K), y: (M, N)
// y[m, n] = sum_k x[m, k] * W[n, k]

__global__ void gemv_fp16_kernel(
    const at::Half* __restrict__ x,
    const at::Half* __restrict__ weight,
    at::Half* __restrict__ y,
    int M, int N, int K) {
    
    int m = blockIdx.x;
    int tid = threadIdx.x;
    
    if (m >= M) return;
    
    const at::Half* x_row = x + m * K;
    
    // Each thread computes one output element
    for (int n = tid; n < N; n += blockDim.x) {
        const at::Half* w_row = weight + n * K;
        float acc = 0.0f;
        
        // Simple loop for correctness
        for (int k = 0; k < K; k++) {
            acc += __half2float(x_row[k]) * __half2float(w_row[k]);
        }
        
        y[m * N + n] = __float2half(acc);
    }
}

torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor weight) {
    int M = x.size(0);
    int K = x.size(1);
    int N = weight.size(0);
    
    auto y = torch::empty({M, N}, x.options());
    
    // Use multiple threads per row
    int threads = 256;
    gemv_fp16_kernel<<<M, threads>>>(
        x.data_ptr<at::Half>(),
        weight.data_ptr<at::Half>(),
        y.data_ptr<at::Half>(),
        M, N, K);
    
    return y;
}
'''

ext = load_inline(
    name='gemv_fp16_ext',
    cpp_sources='torch::Tensor gemv_fp16(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gemv_fp16'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        return ext.gemv_fp16(x_fp16, self.weight)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
