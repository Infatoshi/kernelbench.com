import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda;
using namespace nvcuda::wmma;

#define CUDA_CHECK(call) \
    do { \
        cudaError_t err = call; \
        if (err != cudaSuccess) { \
            fprintf(stderr, "CUDA error at %s:%d: %s\\n", __FILE__, __LINE__, \
                    cudaGetErrorString(err)); \
            exit(1); \
        } \
    } while(0)

// Warp-level GEMM with WMMA for INT4 dequantized weights
template <int WMMA_M, int WMMA_N, int WMMA_K>
__launch_bounds__(256)
__global__ void int4_gemm_wmma_kernel(
    const __half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const __half* __restrict__ scales,
    __half* __restrict__ output,
    int batch_seq,
    int K,
    int N,
    int num_groups,
    int group_size
) {
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    const int bz = blockIdx.z;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    
    // Block tiling: 8 warps per block (256 threads)
    // 4 warps in M dimension, 2 warps in N dimension
    const int WARPS_M = 4;
    const int WARPS_N = 2;
    const int BLOCK_M = WMMA_M * WARPS_M;  // 64
    const int BLOCK_N = WMMA_N * WARPS_N;  // 32
    
    const int row0 = by * BLOCK_M;
    const int col0 = bx * BLOCK_N;
    
    if (row0 >= N || col0 >= batch_seq) return;
    
    const int warp_m = warp_id % WARPS_M;
    const int warp_n = warp_id / WARPS_M;
    
    const int tile_row = row0 + warp_m * WMMA_M;
    const int tile_col = col0 + warp_n * WMMA_N;
    
    // WMMA fragments
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, __half, wmma::col_major> b_frag;
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, __half> acc_frag;
    
    wmma::fill_fragment(acc_frag, __float2half_rn(0.0f));
    
    // Loop over K in chunks of WMMA_K (16)
    for (int k = 0; k < K; k += WMMA_K) {
        // Load activation tile (16x16)
        __half a_tile[WMMA_M][WMMA_N];
        const int k_limit = min(k + WMMA_K, K);
        
        for (int i = 0; i < WMMA_M; ++i) {
            for (int j = 0; j < WMMA_N; ++j) {
                const int k_idx = k + i;
                const int col_idx = tile_col + j;
                if (k_idx < k_limit && col_idx < batch_seq) {
                    a_tile[i][j] = x[(int64_t)k_idx * batch_seq + col_idx];
                } else {
                    a_tile[i][j] = __float2half_rn(0.0f);
                }
            }
        }
        
        // Load A fragment
        for (int i = 0; i < WMMA_M; ++i) {
            for (int j = 0; j < WMMA_N; ++j) {
                a_frag.x[i * WMMA_N + j] = a_tile[i][j];
            }
        }
        
        // Load weight tile with INT4 unpack and dequant
        __half b_tile[WMMA_M][WMMA_N];
        const int group_idx = k / group_size;
        
        for (int i = 0; i < WMMA_M; ++i) {
            for (int j = 0; j < WMMA_N; ++j) {
                const int n_idx = tile_row + i;
                const int k_idx = k + j;
                
                if (n_idx < N && k_idx < k_limit) {
                    const int packed_idx = k_idx / 2;
                    const uint8_t packed = weight_packed[(int64_t)n_idx * (K/2) + packed_idx];
                    
                    int w_int;
                    if (k_idx % 2 == 0) {
                        w_int = packed & 0xF;
                    } else {
                        w_int = (packed >> 4) & 0xF;
                    }
                    
                    // Dequantize: scale * (w_int - 8)
                    const __half w_half = __int2half_rn(w_int);
                    const __half eight_half = __float2half_rn(8.0f);
                    const __half w_offset = __hsub(w_half, eight_half);
                    const float scale_f = __half2float(scales[(int64_t)n_idx * num_groups + group_idx]);
                    const float w_deq = __half2float(w_offset) * scale_f;
                    b_tile[i][j] = __float2half_rn(w_deq);
                } else {
                    b_tile[i][j] = __float2half_rn(0.0f);
                }
            }
        }
        
        // Load B fragment
        for (int i = 0; i < WMMA_M; ++i) {
            for (int j = 0; j < WMMA_N; ++j) {
                b_frag.x[i * WMMA_N + j] = b_tile[i][j];
            }
        }
        
        wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
    }
    
    // Store result
    wmma::store_matrix_sync(
        output + bz * batch_seq * N + row0 * batch_seq + col0,
        acc_frag,
        batch_seq,
        wmma::mem_row_major
    );
}

// Kernel launcher
torch::Tensor int4_gemm_forward(
    torch::Tensor x,           // (batch, seq, K)
    torch::Tensor weight_packed,  // (N, K/2)
    torch::Tensor scales,      // (N, num_groups)
    int group_size
) {
    auto batch_size = x.size(0);
    auto seq_len = x.size(1);
    auto K = x.size(2);
    auto N = weight_packed.size(0);
    auto num_groups = K / group_size;
    
    auto batch_seq = batch_size * seq_len;
    
    auto output = torch::empty({batch_size, seq_len, N}, x.options());
    
    // Flatten x for 2D GEMM: (batch_seq, K)
    auto x_flat = x.view({batch_seq, K});
    
    // Use the WMMA kernel for best performance
    const int BLOCK_M = 64;
    const int BLOCK_N = 32;
    
    dim3 block(256);
    dim3 grid(
        (batch_seq + BLOCK_N - 1) / BLOCK_N,
        (N + BLOCK_M - 1) / BLOCK_M,
        batch_size
    );
    
    // Cast pointers explicitly
    const __half* x_ptr = static_cast<const __half*>(x_flat.data_ptr());
    const uint8_t* w_ptr = static_cast<const uint8_t*>(weight_packed.data_ptr());
    const __half* s_ptr = static_cast<const __half*>(scales.data_ptr());
    __half* out_ptr = static_cast<__half*>(output.view({batch_seq, N}).data_ptr());
    
    int4_gemm_wmma_kernel<16, 16, 16><<<grid, block>>>(
        x_ptr,
        w_ptr,
        s_ptr,
        out_ptr,
        batch_seq,
        K,
        N,
        num_groups,
        group_size
    );
    
    CUDA_CHECK(cudaGetLastError());
    
    return output;
}
'''

cpp_source = '''
torch::Tensor int4_gemm_forward(
    torch::Tensor x,
    torch::Tensor weight_packed,
    torch::Tensor scales,
    int group_size
);
'''

ext = load_inline(
    name='int4_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['int4_gemm_forward'],
    verbose=False,
    extra_cuda_cflags=['-gencode=arch=compute_86,code=sm_86', '-std=c++17'],
)

class Model(nn.Module):
    def __init__(self, K, N, group_size=128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size
        
        assert K % group_size == 0
        assert K % 2 == 0
        
        # Use same initialization as reference
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)
    
    def forward(self, x):
        return ext.int4_gemm_forward(
            x,
            self.weight_packed,
            self.scales,
            self.group_size
        )

def get_inputs():
    batch_size = 4
    seq_len = 2048
    K = 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]

def get_init_inputs():
    K = 4096
    N = 11008
    group_size = 128
    return [K, N, group_size]
