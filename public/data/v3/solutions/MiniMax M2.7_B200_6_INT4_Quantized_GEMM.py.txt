import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// INT4 Weight-Only Quantized GEMM for B200 (SM100)
// Key optimization: Fused unpack + dequant + GEMM, never materialize dequantized weights

// Optimized kernel: Each thread computes one output element
// Grid: (N+31)/32 x (M+31)/32, Block: 256 threads
__global__ void int4_woq_gemm_kernel(
    const __half* __restrict__ A,      // (M, K) FP16 activation
    const uint8_t* __restrict__ B,      // (N, K/2) packed INT4 weights  
    const __half* __restrict__ scales,  // (N, num_groups) FP16 scales
    __half* __restrict__ C,             // (M, N) output
    int M, int N, int K,
    int group_size,
    int num_groups
) {
    // Thread computes one output element C[m_idx, n_idx]
    int n_idx = blockIdx.x * 32 + threadIdx.x % 32;
    int m_idx = blockIdx.y * 32 + threadIdx.x / 32;
    
    if (m_idx >= M || n_idx >= N) return;
    
    float result = 0.0f;
    
    // Sequential K accumulation with INT4 unpacking and dequantization
    for (int k = 0; k < K; k++) {
        // Load activation A[m_idx, k]
        float a_val = __half2float(A[m_idx * K + k]);
        
        // Unpack INT4 weight from B[n_idx, k/2]
        int k_byte = k >> 1;  // k / 2
        int k_bit = k & 1;    // k % 2
        
        uint8_t packed_val = B[n_idx * (K >> 1) + k_byte];
        int w_int = (k_bit == 0) ? (packed_val & 0x0F) : ((packed_val >> 4) & 0x0F);
        
        // Get scale for this group: scales[n_idx, k/group_size]
        int g = k / group_size;
        float scale_val = __half2float(scales[n_idx * num_groups + g]);
        
        // Dequantize: W = scale * (w_int - 8)
        float w_dequant = scale_val * (float)(w_int - 8);
        
        // Accumulate: result += a * W
        result += a_val * w_dequant;
    }
    
    C[m_idx * N + n_idx] = __float2half_rn(result);
}

// Kernel for small K values (unroll loops for efficiency)
__global__ void int4_woq_gemm_unroll_kernel(
    const __half* __restrict__ A,
    const uint8_t* __restrict__ B,
    const __half* __restrict__ scales,
    __half* __restrict__ C,
    int M, int N, int K,
    int group_size,
    int num_groups
) {
    int n_idx = blockIdx.x * 32 + threadIdx.x % 32;
    int m_idx = blockIdx.y * 32 + threadIdx.x / 32;
    
    if (m_idx >= M || n_idx >= N) return;
    
    float result = 0.0f;
    
    // Special case K = 4096 with unrolling for better ILP
    #pragma unroll 16
    for (int k = 0; k < 4096; k++) {
        float a_val = __half2float(A[m_idx * 4096 + k]);
        uint8_t packed_val = B[n_idx * 2048 + (k >> 1)];
        int w_int = ((k & 1) == 0) ? (packed_val & 0x0F) : ((packed_val >> 4) & 0x0F);
        float scale_val = __half2float(scales[n_idx * 32 + (k >> 7)]);
        result += a_val * scale_val * (float)(w_int - 8);
    }
    
    C[m_idx * N + n_idx] = __float2half_rn(result);
}

// Batch kernel: Process all batch elements in a flattened manner
__global__ void int4_woq_gemm_batch_kernel(
    const __half* __restrict__ A,
    const uint8_t* __restrict__ B,
    const __half* __restrict__ scales,
    __half* __restrict__ C,
    int M, int N, int K,
    int group_size,
    int num_groups
) {
    int n_idx = blockIdx.x * 32 + threadIdx.x % 32;
    int m_idx = blockIdx.y * 32 + threadIdx.x / 32;
    
    if (m_idx >= M || n_idx >= N) return;
    
    float result = 0.0f;
    
    for (int k = 0; k < K; k++) {
        float a_val = __half2float(A[m_idx * K + k]);
        int k_byte = k >> 1;
        int k_bit = k & 1;
        
        uint8_t packed_val = B[n_idx * (K >> 1) + k_byte];
        int w_int = (k_bit == 0) ? (packed_val & 0x0F) : ((packed_val >> 4) & 0x0F);
        
        int g = k / group_size;
        float scale_val = __half2float(scales[n_idx * num_groups + g]);
        float w_dequant = scale_val * (float)(w_int - 8);
        
        result += a_val * w_dequant;
    }
    
    C[m_idx * N + n_idx] = __float2half_rn(result);
}

// Wrapper for 2D matmul
torch::Tensor int4_woq_gemm_2d(
    torch::Tensor A,      // (M, K) FP16
    torch::Tensor B,      // (N, K/2) packed INT4 uint8
    torch::Tensor scales, // (N, num_groups) FP16
    int group_size
) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(0);
    int num_groups = K / group_size;
    
    auto C = torch::empty({M, N}, A.options());
    
    const __half* A_ptr = reinterpret_cast<const __half*>(A.data_ptr<torch::Half>());
    const uint8_t* B_ptr = reinterpret_cast<const uint8_t*>(B.data_ptr());
    const __half* scales_ptr = reinterpret_cast<const __half*>(scales.data_ptr<torch::Half>());
    __half* C_ptr = reinterpret_cast<__half*>(C.data_ptr<torch::Half>());
    
    // Grid configuration
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    dim3 block(256);
    
    int4_woq_gemm_kernel<<<grid, block>>>(
        A_ptr, B_ptr, scales_ptr, C_ptr,
        M, N, K, group_size, num_groups
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\\n", cudaGetErrorString(err));
    }
    
    return C;
}

// Wrapper for batch matmul
torch::Tensor int4_woq_gemm_batched(
    torch::Tensor A,      // (batch, seq, K) or (M, K) FP16
    torch::Tensor B,      // (N, K/2) packed INT4 uint8
    torch::Tensor scales, // (N, num_groups) FP16
    int group_size
) {
    // Handle both 2D and 3D inputs
    bool is_3d = (A.dim() == 3);
    int batch = 1, seq = 1;
    
    if (is_3d) {
        batch = A.size(0);
        seq = A.size(1);
        A = A.view({batch * seq, A.size(2)});
    }
    
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(0);
    int num_groups = K / group_size;
    
    auto C = torch::empty({M, N}, A.options());
    
    const __half* A_ptr = reinterpret_cast<const __half*>(A.data_ptr<torch::Half>());
    const uint8_t* B_ptr = reinterpret_cast<const uint8_t*>(B.data_ptr());
    const __half* scales_ptr = reinterpret_cast<const __half*>(scales.data_ptr<torch::Half>());
    __half* C_ptr = reinterpret_cast<__half*>(C.data_ptr<torch::Half>());
    
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    dim3 block(256);
    
    int4_woq_gemm_batch_kernel<<<grid, block>>>(
        A_ptr, B_ptr, scales_ptr, C_ptr,
        M, N, K, group_size, num_groups
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\\n", cudaGetErrorString(err));
    }
    
    if (is_3d) {
        C = C.view({batch, seq, N});
    }
    
    return C;
}
'''

ext = load_inline(
    name='int4_woq_ext',
    cpp_sources='''
torch::Tensor int4_woq_gemm_2d(
    torch::Tensor A,
    torch::Tensor B, 
    torch::Tensor scales,
    int group_size
);

torch::Tensor int4_woq_gemm_batched(
    torch::Tensor A,
    torch::Tensor B,
    torch::Tensor scales,
    int group_size
);
''',
    cuda_sources=cuda_source,
    functions=['int4_woq_gemm_2d', 'int4_woq_gemm_batched'],
    extra_include_paths=['/usr/local/cuda/include'],
    verbose=True
)


class Model(nn.Module):
    """
    Optimized INT4 Weight-Only Quantized GEMM.
    
    Key optimizations over reference:
    1. Fused kernel: unpacks INT4 + dequantizes + matmuls in one pass
    2. Never materializes dequantized FP16 weight matrix
    3. Minimal memory footprint - only reads packed weights
    """

    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even for INT4 packing"

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Fused INT4 weight-only quantized matmul.
        
        Input x: (batch, seq, K) or (M, K) in FP16
        Output: (batch, seq, N) or (M, N) in FP16
        """
        return ext.int4_woq_gemm_batched(
            x,
            self.weight_packed,
            self.scales,
            self.group_size
        )


def get_inputs():
    batch_size = 4
    seq_len = 2048
    K = 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16, device='cuda')]


def get_init_inputs():
    K = 4096
    N = 11008
    group_size = 128
    return [K, N, group_size]
