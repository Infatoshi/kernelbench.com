cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Depthwise convolution kernel
// Each thread handles one output element
// Input: (B, C, H, W)
// Weight: (C, K, K)  (depthwise, so 1 output channel per input channel)
// Output: (B, C, H_out, W_out)
__global__ void depthwise_conv2d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int B, const int C, const int H, const int W,
    const int K, const int stride, const int padding,
    const int H_out, const int W_out
) {
    // Calculate which output element this thread computes
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total_out = B * C * H_out * W_out;
    
    if (out_idx >= total_out) return;
    
    // Decode output index
    int tmp = out_idx;
    const int b = tmp / (C * H_out * W_out);
    tmp = tmp % (C * H_out * W_out);
    const int c = tmp / (H_out * W_out);
    tmp = tmp % (H_out * W_out);
    const int h_out = tmp / W_out;
    const int w_out = tmp % W_out;
    
    // Calculate corresponding input position (top-left corner of kernel)
    const int h_in = h_out * stride - padding;
    const int w_in = w_out * stride - padding;
    
    // Compute convolution for this output position
    float sum = 0.0f;
    
    #pragma unroll 3
    for (int kh = 0; kh < K; kh++) {
        const int h_pos = h_in + kh;
        // Bounds check
        if (h_pos >= 0 && h_pos < H) {
            #pragma unroll 3
            for (int kw = 0; kw < K; kw++) {
                const int w_pos = w_in + kw;
                // Bounds check
                if (w_pos >= 0 && w_pos < W) {
                    const int input_idx = ((b * C + c) * H + h_pos) * W + w_pos;
                    const int weight_idx = (c * K + kh) * K + kw;
                    sum += input[input_idx] * weight[weight_idx];
                }
            }
        }
    }
    
    const int out_linear_idx = ((b * C + c) * H_out + h_out) * W_out + w_out;
    output[out_linear_idx] = sum;
}

// Optimized depthwise convolution with shared memory for weight
// Tiles the output and loads weight into shared memory
__global__ void depthwise_conv2d_tiled_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int B, const int C, const int H, const int W,
    const int K, const int stride, const int padding,
    const int H_out, const int W_out
) {
    // Each block handles one channel
    const int c = blockIdx.x;
    if (c >= C) return;
    
    const int bx = blockIdx.y;
    const int by = blockIdx.z;
    
    // Tile parameters
    const int TILE_H = 8;
    const int TILE_W = 8;
    
    const int h_start = by * TILE_H * stride;
    const int w_start = bx * TILE_W * stride;
    
    // Shared memory for weight (C, K, K) - but we load per channel
    extern __shared__ float s_weight[];
    
    // Load weight for this channel into shared memory
    // Weight layout: (c, kh, kw) -> linear: c*K*K + kh*K + kw
    if (threadIdx.x < K * K) {
        s_weight[threadIdx.x] = weight[c * K * K + threadIdx.x];
    }
    __syncthreads();
    
    // Each thread in block computes one output position
    const int tid = threadIdx.x;
    const int threads_per_block = blockDim.x;
    
    // Output tile bounds
    const int h_end = min(h_start + TILE_H * stride, H + padding);
    const int w_end = min(w_start + TILE_W * stride, W + padding);
    
    const int h_out_start = by * TILE_H;
    const int w_out_start = bx * TILE_W;
    const int h_out_end = min(h_out_start + TILE_H, H_out);
    const int w_out_end = min(w_out_start + TILE_W, W_out);
    
    // Iterate over output positions in this tile
    for (int h_out = h_out_start + tid / TILE_W; h_out < h_out_end; h_out += threads_per_block / TILE_W) {
        for (int w_out = w_out_start + tid % TILE_W; w_out < w_out_end; w_out += TILE_W) {
            const int h_in = h_out * stride - padding;
            const int w_in = w_out * stride - padding;
            
            float sum = 0.0f;
            
            #pragma unroll 3
            for (int kh = 0; kh < K; kh++) {
                const int h_pos = h_in + kh;
                if (h_pos >= 0 && h_pos < H) {
                    #pragma unroll 3
                    for (int kw = 0; kw < K; kw++) {
                        const int w_pos = w_in + kw;
                        if (w_pos >= 0 && w_pos < W) {
                            const int input_idx = ((0 * C + c) * H + h_pos) * W + w_pos;
                            const int weight_idx = kh * K + kw;
                            sum += input[input_idx] * s_weight[weight_idx];
                        }
                    }
                }
            }
            
            const int out_idx = ((0 * C + c) * H_out + h_out) * W_out + w_out;
            output[out_idx] = sum;
        }
    }
}

// More efficient kernel with better memory access patterns
__global__ void depthwise_conv2d_optimized_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int B, const int C, const int H, const int W,
    const int K, const int stride, const int padding,
    const int H_out, const int W_out
) {
    // Shared memory for input tile and weight
    extern __shared__ float shared_mem[];
    const int TILE_H = 16;
    const int TILE_W = 16;
    
    // Each block processes one output tile for one channel
    const int c = blockIdx.x;
    const int by = blockIdx.y;
    const int bx = blockIdx.z;
    
    if (c >= C) return;
    
    // Each thread computes one output
    const int out_h = by * TILE_H + threadIdx.y;
    const int out_w = bx * TILE_W + threadIdx.x;
    
    const int h_in_start = out_h * stride - padding;
    const int w_in_start = out_w * stride - padding;
    
    float sum = 0.0f;
    
    // Depthwise: iterate over kernel size
    for (int kh = 0; kh < K; kh++) {
        const int h_pos = h_in_start + kh;
        if (h_pos >= 0 && h_pos < H) {
            for (int kw = 0; kw < K; kw++) {
                const int w_pos = w_in_start + kw;
                if (w_pos >= 0 && w_pos < W) {
                    // Load input
                    const float* input_channel = input + c * H * W;
                    const int input_idx = h_pos * W + w_pos;
                    const float inp_val = input_channel[input_idx];
                    
                    // Load weight
                    const int weight_idx = kh * K + kw;
                    const float* weight_channel = weight + c * K * K;
                    const float w_val = weight_channel[weight_idx];
                    
                    sum += inp_val * w_val;
                }
            }
        }
    }
    
    if (out_h < H_out && out_w < W_out) {
        const int out_idx = c * H_out * W_out + out_h * W_out + out_w;
        output[out_idx] = sum;
    }
}

// Efficient batched depthwise conv kernel
__global__ void depthwise_conv2d_batched_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int B, const int C, const int H, const int W,
    const int K, const int stride, const int padding,
    const int H_out, const int W_out
) {
    // Grid: (B, C, *) - each (b,c) pair is independent
    const int b = blockIdx.x;
    const int c = blockIdx.y;
    const int tid = blockIdx.z * blockDim.x + threadIdx.x;
    
    if (b >= B || c >= C) return;
    
    const int outputs_per_channel = H_out * W_out;
    const int total_threads_needed = outputs_per_channel;
    
    if (tid >= total_threads_needed) return;
    
    const int h_out = tid / W_out;
    const int w_out = tid % W_out;
    
    const int h_in = h_out * stride - padding;
    const int w_in = w_out * stride - padding;
    
    const float* input_batch_channel = input + (b * C + c) * H * W;
    const float* weight_channel = weight + c * K * K;
    float* output_batch_channel = output + (b * C + c) * H_out * W_out;
    
    float sum = 0.0f;
    
    // Unroll the innermost loops for the 3x3 kernel
    #pragma unroll
    for (int kh = 0; kh < K; kh++) {
        const int h_pos = h_in + kh;
        if (h_pos >= 0 && h_pos < H) {
            #pragma unroll
            for (int kw = 0; kw < K; kw++) {
                const int w_pos = w_in + kw;
                if (w_pos >= 0 && w_pos < W) {
                    sum += input_batch_channel[h_pos * W + w_pos] * weight_channel[kh * K + kw];
                }
            }
        }
    }
    
    output_batch_channel[h_out * W_out + w_out] = sum;
}

// Entry point wrapper
torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding
) {
    const int B = input.size(0);
    const int C = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    const int K = weight.size(2);
    
    const int H_out = (H + 2 * padding - K) / stride + 1;
    const int W_out = (W + 2 * padding - K) / stride + 1;
    
    auto output = torch::empty({B, C, H_out, W_out}, input.options());
    
    const int total_output = B * C * H_out * W_out;
    const int threads = 256;
    const int blocks = (total_output + threads - 1) / threads;
    
    depthwise_conv2d_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        output.data_ptr<float>(),
        B, C, H, W, K, stride, padding, H_out, W_out
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}

// Batched version with better parallelization
torch::Tensor depthwise_conv2d_batched(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding
) {
    const int B = input.size(0);
    const int C = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    const int K = weight.size(2);
    
    const int H_out = (H + 2 * padding - K) / stride + 1;
    const int W_out = (W + 2 * padding - K) / stride + 1;
    
    auto output = torch::empty({B, C, H_out, W_out}, input.options());
    
    // Grid: (B, C, ceil(H_out*W_out/256))
    const int outputs_per_channel = H_out * W_out;
    const int z_blocks = (outputs_per_channel + 255) / 256;
    
    dim3 grid(B, C, z_blocks);
    dim3 block(256, 1, 1);
    
    depthwise_conv2d_batched_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        output.data_ptr<float>(),
        B, C, H, W, K, stride, padding, H_out, W_out
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='depthwise_conv_ext',
    cpp_sources='''
torch::Tensor depthwise_conv2d_forward(torch::Tensor input, torch::Tensor weight, int stride, int padding);
torch::Tensor depthwise_conv2d_batched(torch::Tensor input, torch::Tensor weight, int stride, int padding);
''',
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_forward', 'depthwise_conv2d_batched'],
    verbose=False,
)


class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.
    Uses a custom CUDA kernel for optimization.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
        # Initialize weight similar to PyTorch
        self.weight = nn.Parameter(torch.randn(in_channels, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.zeros(in_channels))
        else:
            self.register_parameter('bias', None)
        
        # Initialize with kaiming uniform for better convergence
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if self.bias is not None:
            fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
            bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
            nn.init.uniform_(self.bias, -bound, bound)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the depthwise 2D convolution using custom CUDA kernel.
        """
        # Use the custom CUDA kernel
        output = ext.depthwise_conv2d_batched(
            x, 
            self.weight, 
            self.stride, 
            self.padding
        )
        
        if self.bias is not None:
            output = output + self.bias.view(1, -1, 1, 1)
        
        return output
