import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused CUDA kernel for Tanh + Scaling + BiasAdd + MaxPool
# This fuses 3 element-wise operations with max pooling into a single kernel

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Optimized kernel with proper parallelization - processes multiple elements in parallel
__global__ void fused_kernel_coalesced(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ bias,
    float scaling_factor,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int pool_kernel_size,
    int pool_stride
) {
    // Sequential thread ID
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_out = batch_size * channels * out_height * out_width;
    
    if (tid >= total_out) return;
    
    // Decode output index
    int b = tid / (channels * out_height * out_width);
    int remainder = tid % (channels * out_height * out_width);
    int c = remainder / (out_height * out_width);
    int out_h = (remainder % (out_height * out_width)) / out_width;
    int out_w = remainder % out_width;
    
    // Compute input region
    int in_h_start = out_h * pool_stride;
    int in_w_start = out_w * pool_stride;
    
    // Max pooling with fused ops
    float max_val = -INFINITY;
    
    for (int kh = 0; kh < pool_kernel_size; kh++) {
        for (int kw = 0; kw < pool_kernel_size; kw++) {
            int in_h = in_h_start + kh;
            int in_w = in_w_start + kw;
            
            if (in_h < in_height && in_w < in_width) {
                int in_idx = ((b * channels + c) * in_height + in_h) * in_width + in_w;
                float val = input[in_idx];
                val = tanhf(val);
                val = val * scaling_factor;
                val = val + bias[c];
                
                if (val > max_val) max_val = val;
            }
        }
    }
    
    output[tid] = max_val;
}

torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_kernel_size,
    int pool_stride
) {
    // input is already the output of conv2d
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    int out_height = (in_height - pool_kernel_size) / pool_stride + 1;
    int out_width = (in_width - pool_kernel_size) / pool_stride + 1;

    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                               torch::TensorOptions().device(input.device()).dtype(torch::kFloat));

    int total_out = batch_size * channels * out_height * out_width;
    int block_size = 256;
    int grid_size = (total_out + block_size - 1) / block_size;
    
    fused_kernel_coalesced<<<grid_size, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        bias.data_ptr<float>(),
        scaling_factor,
        batch_size, channels, in_height, in_width,
        out_height, out_width,
        pool_kernel_size, pool_stride
    );

    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }

    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_kernel_size,
    int pool_stride
);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that fuses Tanh + Scaling + BiasAdd + MaxPool operations.
    Conv2d uses PyTorch's cuDNN-accelerated implementation, while the 
    element-wise ops and pooling are fused into a single CUDA kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        # Same as reference - no padding
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size
        self.pool_stride = pool_kernel_size

    def forward(self, x):
        # Conv2d (handled by cuDNN - well optimized)
        conv_out = self.conv(x)
        
        # Fused kernel: Tanh + Scale + Bias + MaxPool
        return ext.fused_forward(
            conv_out,
            self.bias,
            self.scaling_factor,
            self.pool_kernel_size,
            self.pool_stride
        )

# Input generation
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
