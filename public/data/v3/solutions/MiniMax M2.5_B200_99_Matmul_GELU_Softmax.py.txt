import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Shared memory tile sizes for GEMM
#define TILE_M 16
#define TILE_N 16  
#define TILE_K 16

// Tiled GEMM with GELU activation
template <typename T, int TM, int TN, int TK>
__global__ void gemm_gelu_tiled(
    const T* __restrict__ A,
    const T* __restrict__ B,
    const T* __restrict__ bias,
    T* __restrict__ C,
    const int M,
    const int N,
    const int K) {
    
    // Shared memory for tiles
    __shared__ T As[TM][TK];
    __shared__ T Bs[TK][TN];
    
    const int BLOCK_M = blockIdx.z;
    const int BLOCK_N = blockIdx.y;
    
    const int GM = BLOCK_M * TM + threadIdx.y;
    const int GN = BLOCK_N * TN + threadIdx.x;
    
    // Thread ID within block
    const int tid = threadIdx.y * TN + threadIdx.x;
    const int num_threads = TM * TN;
    
    // Accumulator
    T sum = 0;
    
    // Number of K tiles
    const int num_k_tiles = (K + TK - 1) / TK;
    
    // Loop over K tiles
    for (int tile_k = 0; tile_k < num_k_tiles; tile_k++) {
        // Cooperative load of A tile
        const int a_col = tile_k * TK;
        const int a_row = GM;
        
        // Calculate which elements this thread loads
        for (int i = tid; i < TM * TK; i += num_threads) {
            int row = i / TK;
            int col = i % TK;
            int global_row = a_row + row;
            int global_col = a_col + col;
            if (global_row < M && global_col < K) {
                As[row][col] = A[global_row * K + global_col];
            } else {
                As[row][col] = T(0);
            }
        }
        
        // Cooperative load of B tile
        const int b_row = tile_k * TK;
        const int b_col = GN;
        
        for (int i = tid; i < TK * TN; i += num_threads) {
            int row = i / TN;
            int col = i % TN;
            int global_row = b_row + row;
            int global_col = b_col + col;
            if (global_row < K && global_col < N) {
                Bs[row][col] = B[global_col * K + global_row];
            } else {
                Bs[row][col] = T(0);
            }
        }
        
        __syncthreads();
        
        // Compute partial dot product - each thread computes inner product
        #pragma unroll
        for (int k = 0; k < TK; k++) {
            sum += As[threadIdx.y][k] * Bs[k][threadIdx.x];
        }
        
        __syncthreads();
    }
    
    // Write result with bias and GELU
    if (GM < M && GN < N) {
        T result = sum;
        if (bias != nullptr) {
            result += bias[GN];
        }
        
        // GELU activation using fast approximation
        // x * 0.5 * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
        T x = result;
        T c = T(0.044715f);
        T sqrt_2_over_pi = T(0.7978845608028654f);
        T gelu = x * T(0.5f) * (T(1.0f) + tanhf(sqrt_2_over_pi * (x + c * x * x * x)));
        
        C[GM * N + GN] = gelu;
    }
}

// Softmax kernel - one block per row
template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, 
                               T* __restrict__ output,
                               const int rows,
                               const int cols) {
    const int row = blockIdx.x;
    if (row >= rows) return;

    const T* row_ptr = input + row * cols;
    T* out_ptr = output + row * cols;

    // Shared memory for reduction
    extern __shared__ __align__(16) char smem[];
    T* shared_max = (T*)smem;
    T* shared_sum = (T*)smem + blockDim.x;

    // Thread position
    const int tid = threadIdx.x;
    const int stride = blockDim.x;

    // First pass: find maximum
    T thread_max = -INFINITY;
    for (int col = tid; col < cols; col += stride) {
        thread_max = max(thread_max, row_ptr[col]);
    }
    shared_max[tid] = thread_max;
    __syncthreads();

    // Tree reduction for max
    for (int s = stride / 2; s > 32; s >>= 1) {
        if (tid < s) {
            shared_max[tid] = max(shared_max[tid], shared_max[tid + s]);
        }
        __syncthreads();
    }
    
    // Warp reduction for final steps
    if (tid < 32) {
        #pragma unroll
        for (int s = 32; s > 0; s >>= 1) {
            if (s > blockDim.x/2) continue;
            shared_max[tid] = max(shared_max[tid], shared_max[tid + s]);
        }
    }
    T row_max = shared_max[0];

    // Second pass: compute exp(x - max) and sum
    T thread_sum = 0;
    for (int col = tid; col < cols; col += stride) {
        T val = exp(row_ptr[col] - row_max);
        out_ptr[col] = val;
        thread_sum += val;
    }
    shared_sum[tid] = thread_sum;
    __syncthreads();

    // Tree reduction for sum
    for (int s = stride / 2; s > 32; s >>= 1) {
        if (tid < s) {
            shared_sum[tid] += shared_sum[tid + s];
        }
        __syncthreads();
    }
    
    // Warp reduction for final steps
    if (tid < 32) {
        #pragma unroll
        for (int s = 32; s > 0; s >>= 1) {
            if (s > blockDim.x/2) continue;
            shared_sum[tid] += shared_sum[tid + s];
        }
    }
    T row_sum = shared_sum[0];

    // Third pass: normalize
    for (int col = tid; col < cols; col += stride) {
        out_ptr[col] = out_ptr[col] / row_sum;
    }
}

// Host wrapper for GEMM + GELU
torch::Tensor gemm_gelu_op(torch::Tensor input, torch::Tensor weight, torch::Tensor bias) {
    auto M = input.size(0);
    auto K = input.size(1);
    auto N = weight.size(0);
    
    // Allocate output
    auto output = torch::empty({M, N}, input.options());
    
    // Launch config: each block does TM x TN elements
    const int BLOCK_M = 16;
    const int BLOCK_N = 16;
    const int BLOCK_K = 16;
    
    // Grid dimensions
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M, 1);
    dim3 block(BLOCK_N, BLOCK_M);
    
    // Choose kernel based on dtype
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "gemm_gelu", ([&] {
        gemm_gelu_tiled<scalar_t, 16, 16, 16><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.defined() ? bias.data_ptr<scalar_t>() : nullptr,
            output.data_ptr<scalar_t>(),
            M, N, K
        );
    }));
    
    cudaGetLastError();
    return output;
}

// Host wrapper for Softmax
torch::Tensor softmax_op(torch::Tensor input) {
    auto rows = input.size(0);
    auto cols = input.size(1);
    
    auto output = torch::empty_like(input);
    
    const int BLOCK_SIZE = 256;
    dim3 grid(rows);
    dim3 block(BLOCK_SIZE);
    size_t shared_mem = 2 * BLOCK_SIZE * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "softmax", ([&] {
        softmax_kernel<scalar_t><<<grid, block, shared_mem>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            rows, cols
        );
    }));
    
    cudaGetLastError();
    return output;
}
'''

ext = load_inline(
    name='matmul_gelu_softmax_ext',
    cpp_sources=r'''
torch::Tensor gemm_gelu_op(torch::Tensor input, torch::Tensor weight, torch::Tensor bias);
torch::Tensor softmax_op(torch::Tensor input);
''',
    cuda_sources=cuda_source,
    functions=['gemm_gelu_op', 'softmax_op'],
    verbose=False,
    extra_cuda_cflags=['-std=c++17', '-O3', '--use_fast_math', '-arch=sm_90']
)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))
        
        # Initialize using Kaiming
        nn.init.kaiming_normal_(self.weight, mode='fan_in', nonlinearity='relu')
        nn.init.zeros_(self.bias)
    
    def forward(self, x):
        # Custom CUDA kernel for GEMM + GELU
        x = ext.gemm_gelu_op(x, self.weight, self.bias)
        # Custom CUDA kernel for Softmax
        x = ext.softmax_op(x)
        return x

batch_size = 128
in_features = 4096
out_features = 4096

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features]
