import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused kernel: takes conv output and applies subtract1, tanh, subtract2, avgpool
# Conv is done via im2col + GEMM using cuBLAS (called via torch.mm on reshaped tensors... wait that's forbidden)
# Let me use a custom approach

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <math.h>

// Fused kernel: conv2d (direct, 3x3) + subtract1 + tanh + subtract2 + avgpool(2x2)
// Input: [N, C_in, H, W]
// Weight: [C_out, C_in, KH, KW]
// Bias: [C_out]
// Output: [N, C_out, H_out_pool, W_out_pool]
// where H_conv = H - KH + 1, W_conv = W - KW + 1
// H_out_pool = H_conv / pool_size, W_out_pool = W_conv / pool_size

// This kernel computes one output element of the final (pooled) output
// Each output element corresponds to a pool_size x pool_size region in conv output space
// For each element in that region, we compute the conv, subtract, tanh, subtract, then average

__global__ void fused_conv_sub_tanh_sub_avgpool_kernel(
    const float* __restrict__ input,   // [N, C_in, H, W]
    const float* __restrict__ weight,  // [C_out, C_in, KH, KW]
    const float* __restrict__ bias,    // [C_out]
    float* __restrict__ output,        // [N, C_out, H_pool, W_pool]
    int N, int C_in, int H, int W,
    int C_out, int KH, int KW,
    float subtract1, float subtract2,
    int pool_size,
    int H_conv, int W_conv,
    int H_pool, int W_pool
) {
    // Each thread computes one output element
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_pool * W_pool;
    if (idx >= total) return;
    
    int w_pool = idx % W_pool;
    int h_pool = (idx / W_pool) % H_pool;
    int co = (idx / (W_pool * H_pool)) % C_out;
    int n = idx / (W_pool * H_pool * C_out);
    
    float pool_sum = 0.0f;
    int pool_count = 0;
    
    int h_start = h_pool * pool_size;
    int w_start = w_pool * pool_size;
    
    for (int ph = 0; ph < pool_size; ph++) {
        for (int pw = 0; pw < pool_size; pw++) {
            int h_conv_idx = h_start + ph;
            int w_conv_idx = w_start + pw;
            
            if (h_conv_idx < H_conv && w_conv_idx < W_conv) {
                // Compute conv at (n, co, h_conv_idx, w_conv_idx)
                float val = bias[co];
                for (int ci = 0; ci < C_in; ci++) {
                    for (int kh = 0; kh < KH; kh++) {
                        for (int kw = 0; kw < KW; kw++) {
                            int h_in = h_conv_idx + kh;
                            int w_in = w_conv_idx + kw;
                            float inp = input[((n * C_in + ci) * H + h_in) * W + w_in];
                            float w_val = weight[((co * C_in + ci) * KH + kh) * KW + kw];
                            val += inp * w_val;
                        }
                    }
                }
                
                // subtract1 + tanh + subtract2
                val = tanhf(val - subtract1) - subtract2;
                pool_sum += val;
                pool_count++;
            }
        }
    }
    
    output[idx] = pool_sum / (float)pool_count;
}

torch::Tensor fused_conv_sub_tanh_sub_avgpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float subtract1,
    float subtract2,
    int pool_size
) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int C_out = weight.size(0);
    int KH = weight.size(2);
    int KW = weight.size(3);
    
    int H_conv = H - KH + 1;
    int W_conv = W - KW + 1;
    int H_pool = H_conv / pool_size;
    int W_pool = W_conv / pool_size;
    
    auto output = torch::empty({N, C_out, H_pool, W_pool}, input.options());
    
    int total = N * C_out * H_pool * W_pool;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    
    fused_conv_sub_tanh_sub_avgpool_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C_in, H, W,
        C_out, KH, KW,
        subtract1, subtract2,
        pool_size,
        H_conv, W_conv,
        H_pool, W_pool
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_conv_sub_tanh_sub_avgpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float subtract1,
    float subtract2,
    int pool_size
);
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_sub_tanh_sub_avgpool'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        return ext.fused_conv_sub_tanh_sub_avgpool(
            x,
            self.conv.weight,
            self.conv.bias,
            self.subtract1_value,
            self.subtract2_value,
            self.kernel_size_pool
        )

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
