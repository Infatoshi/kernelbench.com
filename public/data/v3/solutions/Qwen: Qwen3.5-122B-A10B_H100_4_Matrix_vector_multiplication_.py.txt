import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Highly optimized GEMV using shared memory tiling
template<typename scalar_t>
__global__ void gemv_tiled_kernel(const scalar_t* __restrict__ A,
                                   const scalar_t* __restrict__ B,
                                   scalar_t* __restrict__ C,
                                   int M, int K) {
    extern __shared__ unsigned char smem_raw[];
    scalar_t* smem = reinterpret_cast<scalar_t*>(smem_raw);
    
    int m = blockIdx.x;
    int tid = threadIdx.x;
    
    scalar_t sum = scalar_t(0);
    
    // Tile size - process 128 elements at a time
    const int TILE_SIZE = 128;
    const int num_tiles = (K + TILE_SIZE - 1) / TILE_SIZE;
    
    for (int tile = 0; tile < num_tiles; ++tile) {
        int k_start = tile * TILE_SIZE;
        int k_end = min(k_start + TILE_SIZE, K);
        
        // Load B elements into shared memory
        for (int i = tid; i < k_end - k_start; i += blockDim.x) {
            smem[i] = B[k_start + i];
        }
        __syncthreads();
        
        // Compute partial dot product
        const scalar_t* A_row = A + m * K;
        for (int k = k_start + tid; k < k_end; k += blockDim.x) {
            sum += A_row[k] * smem[k - k_start];
        }
        __syncthreads();
    }
    
    C[m] = sum;
}

// FP16 version using native half operations
__global__ void gemv_tiled_kernel_fp16(const half* __restrict__ A,
                                        const half* __restrict__ B,
                                        half* __restrict__ C,
                                        int M, int K) {
    extern __shared__ unsigned char smem_raw[];
    half* smem = reinterpret_cast<half*>(smem_raw);
    
    int m = blockIdx.x;
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    
    const int TILE_SIZE = 128;
    const int num_tiles = (K + TILE_SIZE - 1) / TILE_SIZE;
    
    for (int tile = 0; tile < num_tiles; ++tile) {
        int k_start = tile * TILE_SIZE;
        int k_end = min(k_start + TILE_SIZE, K);
        
        // Load B elements into shared memory
        for (int i = tid; i < k_end - k_start; i += blockDim.x) {
            smem[i] = B[k_start + i];
        }
        __syncthreads();
        
        // Compute partial dot product
        const half* A_row = A + m * K;
        for (int k = k_start + tid; k < k_end; k += blockDim.x) {
            sum += __half2float(A_row[k]) * __half2float(smem[k - k_start]);
        }
        __syncthreads();
    }
    
    C[m] = __float2half(sum);
}

// BF16 version using nv_bfloat16
__global__ void gemv_tiled_kernel_bf16(const nv_bfloat16* __restrict__ A,
                                        const nv_bfloat16* __restrict__ B,
                                        nv_bfloat16* __restrict__ C,
                                        int M, int K) {
    extern __shared__ unsigned char smem_raw[];
    nv_bfloat16* smem = reinterpret_cast<nv_bfloat16*>(smem_raw);
    
    int m = blockIdx.x;
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    
    const int TILE_SIZE = 128;
    const int num_tiles = (K + TILE_SIZE - 1) / TILE_SIZE;
    
    for (int tile = 0; tile < num_tiles; ++tile) {
        int k_start = tile * TILE_SIZE;
        int k_end = min(k_start + TILE_SIZE, K);
        
        // Load B elements into shared memory
        for (int i = tid; i < k_end - k_start; i += blockDim.x) {
            smem[i] = B[k_start + i];
        }
        __syncthreads();
        
        // Compute partial dot product
        const nv_bfloat16* A_row = A + m * K;
        for (int k = k_start + tid; k < k_end; k += blockDim.x) {
            sum += __bfloat162float(A_row[k]) * __bfloat162float(smem[k - k_start]);
        }
        __syncthreads();
    }
    
    C[m] = __float2bfloat16(sum);
}

torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    TORCH_CHECK(A.size(1) == B.size(0), "Dimension mismatch");
    TORCH_CHECK(B.size(1) == 1, "B must be a column vector");
    
    auto C = torch::empty({M, 1}, A.options());
    
    int threads = 256;
    int blocks = M;
    size_t smem_size = threads * A.element_size();
    
    if (A.scalar_type() == torch::kFloat32) {
        gemv_tiled_kernel<float><<<blocks, threads, smem_size>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    } else if (A.scalar_type() == torch::kFloat16) {
        gemv_tiled_kernel_fp16<<<blocks, threads, smem_size>>>(
            reinterpret_cast<const half*>(A.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(B.data_ptr<at::Half>()),
            reinterpret_cast<half*>(C.data_ptr<at::Half>()),
            M, K
        );
    } else if (A.scalar_type() == torch::kBFloat16) {
        gemv_tiled_kernel_bf16<<<blocks, threads, smem_size>>>(
            reinterpret_cast<const nv_bfloat16*>(A.data_ptr<at::BFloat16>()),
            reinterpret_cast<const nv_bfloat16*>(B.data_ptr<at::BFloat16>()),
            reinterpret_cast<nv_bfloat16*>(C.data_ptr<at::BFloat16>()),
            M, K
        );
    } else {
        // Fall back to float32 for other types
        A = A.to(torch::kFloat32);
        B = B.to(torch::kFloat32);
        C = torch::empty({M, 1}, A.options());
        gemv_tiled_kernel<float><<<blocks, threads, smem_size>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    }
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources='torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemv_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A, B):
        return ext.gemv_op(A, B)

def get_inputs():
    return [torch.randn(256, 131072), torch.randn(131072, 1)]

def get_init_inputs():
    return []
