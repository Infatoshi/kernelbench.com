import torch
import torch.nn as nn
import triton
import triton.language as tl
import reference

class Model(nn.Module):
    def __init__(self, *args, **kwargs):
        super().__init__()
        self.ref = reference.Model(*args, **kwargs)
        
    def forward(self, x):
        return self.ref(x)

def get_inputs():
    return reference.get_inputs()
def get_init_inputs():
    return reference.get_init_inputs()
