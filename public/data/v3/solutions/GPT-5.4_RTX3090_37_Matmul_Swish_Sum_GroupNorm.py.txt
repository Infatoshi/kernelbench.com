import os
import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 32, 'GROUP_M': 8}, num_warps=4, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 32, 'GROUP_M': 8}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 64, 'BLOCK_K': 32, 'GROUP_M': 8}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 32, 'GROUP_M': 8}, num_warps=8, num_stages=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def fused_matmul_swish_bias_kernel(
    a_ptr, w_ptr, lin_bias_ptr, add_bias_ptr, out_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = tl.minimum(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k0 in range(0, tl.cdiv(K, BLOCK_K)):
        k_offs = k0 * BLOCK_K + offs_k
        a = tl.load(a_ptr + offs_m[:, None] * stride_am + k_offs[None, :] * stride_ak,
                    mask=(offs_m[:, None] < M) & (k_offs[None, :] < K), other=0.0)
        w = tl.load(w_ptr + offs_n[None, :] * stride_wn + k_offs[:, None] * stride_wk,
                    mask=(offs_n[None, :] < N) & (k_offs[:, None] < K), other=0.0)
        acc += tl.dot(a, w)

    lin_b = tl.load(lin_bias_ptr + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
    add_b = tl.load(add_bias_ptr + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
    acc = acc + lin_b[None, :]
    sig = tl.sigmoid(acc)
    acc = sig * acc + add_b[None, :]

    tl.store(out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on,
             acc, mask=(offs_m[:, None] < M) & (offs_n[None, :] < N))


@triton.autotune(
    configs=[
        triton.Config({'BLOCK': 128}, num_warps=4, num_stages=2),
        triton.Config({'BLOCK': 256}, num_warps=4, num_stages=2),
        triton.Config({'BLOCK': 512}, num_warps=8, num_stages=2),
    ],
    key=['N'],
)
@triton.jit
def groupnorm_stats_kernel(
    x_ptr, mean_ptr, rstd_ptr,
    M, N,
    stride_xm, stride_xn,
    groups: tl.constexpr,
    group_size: tl.constexpr,
    eps: tl.constexpr,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    row = pid // groups
    g = pid % groups
    if row >= M:
        return
    base = g * group_size
    acc = tl.zeros((BLOCK,), dtype=tl.float32)
    acc2 = tl.zeros((BLOCK,), dtype=tl.float32)
    for off in range(0, group_size, BLOCK):
        cols = off + tl.arange(0, BLOCK)
        mask = cols < group_size
        vals = tl.load(x_ptr + row * stride_xm + (base + cols) * stride_xn, mask=mask, other=0.0).to(tl.float32)
        acc += vals
        acc2 += vals * vals
    s = tl.sum(acc, axis=0)
    s2 = tl.sum(acc2, axis=0)
    mean = s / group_size
    var = s2 / group_size - mean * mean
    tl.store(mean_ptr + pid, mean)
    tl.store(rstd_ptr + pid, tl.rsqrt(var + eps))


@triton.autotune(
    configs=[
        triton.Config({'BLOCK': 128}, num_warps=4, num_stages=2),
        triton.Config({'BLOCK': 256}, num_warps=4, num_stages=2),
        triton.Config({'BLOCK': 512}, num_warps=8, num_stages=2),
    ],
    key=['N'],
)
@triton.jit
def groupnorm_apply_kernel(
    x_ptr, mean_ptr, rstd_ptr, gamma_ptr, beta_ptr, y_ptr,
    M, N,
    stride_xm, stride_xn,
    stride_ym, stride_yn,
    groups: tl.constexpr,
    group_size: tl.constexpr,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    row = pid
    if row >= M:
        return
    cols = tl.arange(0, BLOCK)
    for off in range(0, N, BLOCK):
        idx = off + cols
        mask = idx < N
        g = idx // group_size
        mean = tl.load(mean_ptr + row * groups + g, mask=mask, other=0.0)
        rstd = tl.load(rstd_ptr + row * groups + g, mask=mask, other=0.0)
        x = tl.load(x_ptr + row * stride_xm + idx * stride_xn, mask=mask, other=0.0).to(tl.float32)
        gamma = tl.load(gamma_ptr + idx, mask=mask, other=1.0).to(tl.float32)
        beta = tl.load(beta_ptr + idx, mask=mask, other=0.0).to(tl.float32)
        y = (x - mean) * rstd
        y = y * gamma + beta
        tl.store(y_ptr + row * stride_ym + idx * stride_yn, y, mask=mask)


class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.num_groups = num_groups
        self.out_features = out_features
        self.eps = self.group_norm.eps

    def forward(self, x):
        if not x.is_cuda:
            x = x.cuda()
        x = x.contiguous()
        M, K = x.shape
        N = self.out_features
        tmp = torch.empty((M, N), device=x.device, dtype=torch.float32)
        fused_matmul_swish_bias_kernel[(triton.cdiv(M, 64) * triton.cdiv(N, 64),)](
            x, self.matmul.weight, self.matmul.bias, self.bias, tmp,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            tmp.stride(0), tmp.stride(1),
        )
        means = torch.empty((M, self.num_groups), device=x.device, dtype=torch.float32)
        rstds = torch.empty((M, self.num_groups), device=x.device, dtype=torch.float32)
        group_size = N // self.num_groups
        groupnorm_stats_kernel[(M * self.num_groups,)](
            tmp, means, rstds,
            M, N,
            tmp.stride(0), tmp.stride(1),
            groups=self.num_groups,
            group_size=group_size,
            eps=self.eps,
        )
        out = torch.empty_like(tmp)
        groupnorm_apply_kernel[(M,)](
            tmp, means, rstds, self.group_norm.weight, self.group_norm.bias, out,
            M, N,
            tmp.stride(0), tmp.stride(1),
            out.stride(0), out.stride(1),
            groups=self.num_groups,
            group_size=group_size,
        )
        return out


batch_size = 128
in_features = 4096
out_features = 4096
num_groups = 32
bias_shape = (out_features,)


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, num_groups, bias_shape]
