import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def tanh(x):
    """Compute tanh using exponential."""
    # tanh(x) = (exp(x) - exp(-x)) / (exp(x) + exp(-x))
    # = (exp(2x) - 1) / (exp(2x) + 1)
    # = 1 - 2 / (exp(2x) + 1)
    exp_2x = tl.exp(2.0 * x)
    return 1.0 - 2.0 / (exp_2x + 1.0)


@triton.jit
def fused_conv_act_bn_kernel(
    # Input tensor
    x_ptr,
    # Conv weights and bias
    w_ptr, b_ptr,
    # BN parameters
    running_mean_ptr, running_var_ptr, weight_ptr, bias_ptr,
    # Output tensor
    out_ptr,
    # Dimensions
    N, C_IN, C_OUT, H_IN, W_IN, KH, KW,
    # BN epsilon
    eps,
    # Strides for input
    stride_x_n, stride_x_c, stride_x_h, stride_x_w,
    # Strides for weights
    stride_w_out, stride_w_in, stride_w_h, stride_w_w,
    # Strides for output
    stride_out_n, stride_out_c, stride_out_h, stride_out_w,
    # Output dimensions
    H_OUT, W_OUT,
    # Block size for output tile
    BLOCK_H: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    """Fused Conv2d + Activation + BatchNorm kernel."""
    
    # Program ID - flatten the 3D grid
    pid = tl.program_id(0)
    pid_n = pid // (C_OUT * ((H_OUT + BLOCK_H - 1) // BLOCK_H) * ((W_OUT + BLOCK_W - 1) // BLOCK_W))
    remainder = pid % (C_OUT * ((H_OUT + BLOCK_H - 1) // BLOCK_H) * ((W_OUT + BLOCK_W - 1) // BLOCK_W))
    pid_c = remainder // (((H_OUT + BLOCK_H - 1) // BLOCK_H) * ((W_OUT + BLOCK_W - 1) // BLOCK_W))
    remainder2 = remainder % (((H_OUT + BLOCK_H - 1) // BLOCK_H) * ((W_OUT + BLOCK_W - 1) // BLOCK_W))
    pid_h = remainder2 // ((W_OUT + BLOCK_W - 1) // BLOCK_W)
    pid_w = remainder2 % ((W_OUT + BLOCK_W - 1) // BLOCK_W)
    
    # Bounds check
    if pid_n >= N or pid_c >= C_OUT:
        return
    
    # Output starting position
    h_start = pid_h * BLOCK_H
    w_start = pid_w * BLOCK_W
    
    # Output offsets
    h_offsets = h_start + tl.arange(0, BLOCK_H)
    w_offsets = w_start + tl.arange(0, BLOCK_W)
    
    # Mask for output bounds
    h_mask = h_offsets < H_OUT
    w_mask = w_offsets < W_OUT
    out_mask = h_mask[:, None] & w_mask[None, :]
    
    # Initialize accumulator for convolution output
    acc = tl.zeros((BLOCK_H, BLOCK_W), dtype=tl.float32)
    
    # Convolution: iterate over input channels and kernel
    for ic in range(C_IN):
        for kh in range(KH):
            for kw in range(KW):
                # Weight value
                w_val = tl.load(w_ptr + pid_c * stride_w_out + ic * stride_w_in + 
                               kh * stride_w_h + kw * stride_w_w)
                
                # Input position
                h_in = h_offsets + kh
                w_in = w_offsets + kw
                
                # Input mask
                in_mask = (h_in[:, None] >= 0) & (h_in[:, None] < H_IN) & \
                          (w_in[None, :] >= 0) & (w_in[None, :] < W_IN) & out_mask
                
                # Load input values
                x_ptr_offset = pid_n * stride_x_n + ic * stride_x_c
                x_val = tl.load(
                    x_ptr + x_ptr_offset + h_in[:, None] * stride_x_h + w_in[None, :] * stride_x_w,
                    mask=in_mask,
                    other=0.0
                )
                
                # Accumulate
                acc += x_val * w_val
    
    # Add conv bias
    conv_bias = tl.load(b_ptr + pid_c)
    acc += conv_bias
    
    # Activation: x * tanh(softplus(x))
    # softplus(x) = log(1 + exp(x))
    # Use numerically stable softplus
    abs_acc = tl.abs(acc)
    softplus = tl.where(acc >= 0, 
                        abs_acc + tl.log(1.0 + tl.exp(-abs_acc)),
                        tl.log(1.0 + tl.exp(acc)))
    tanh_softplus = tanh(softplus)
    act_out = acc * tanh_softplus
    
    # Batch Normalization (inference mode)
    mean = tl.load(running_mean_ptr + pid_c)
    var = tl.load(running_var_ptr + pid_c)
    gamma = tl.load(weight_ptr + pid_c)
    beta = tl.load(bias_ptr + pid_c)
    
    # BN: (x - mean) / sqrt(var + eps) * gamma + beta
    inv_std = tl.rsqrt(var + eps)
    bn_out = (act_out - mean) * inv_std * gamma + beta
    
    # Store output
    out_ptr_offset = pid_n * stride_out_n + pid_c * stride_out_c
    tl.store(
        out_ptr + out_ptr_offset + h_offsets[:, None] * stride_out_h + w_offsets[None, :] * stride_out_w,
        bn_out,
        mask=out_mask
    )


class Model(nn.Module):
    """
    Fused Conv2d + Activation + BatchNorm using Triton.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.eps = eps
        self.momentum = momentum
        
        # Create reference modules for state_dict compatibility
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
    
    def forward(self, x):
        N, C_IN, H_IN, W_IN = x.shape
        C_OUT = self.out_channels
        KH = KW = self.kernel_size
        H_OUT = H_IN - KH + 1
        W_OUT = W_IN - KW + 1
        
        # Allocate output
        out = torch.empty(N, C_OUT, H_OUT, W_OUT, device=x.device, dtype=x.dtype)
        
        # Get strides
        stride_x = x.stride()
        stride_w = self.conv.weight.stride()
        stride_out = out.stride()
        
        # Grid and block sizes
        BLOCK_H = 16
        BLOCK_W = 16
        
        # Calculate total number of blocks
        n_h_blocks = (H_OUT + BLOCK_H - 1) // BLOCK_H
        n_w_blocks = (W_OUT + BLOCK_W - 1) // BLOCK_W
        total_blocks = N * C_OUT * n_h_blocks * n_w_blocks
        
        # Launch kernel with 1D grid
        grid = (total_blocks,)
        
        fused_conv_act_bn_kernel[grid](
            x,
            self.conv.weight, self.conv.bias,
            self.bn.running_mean, self.bn.running_var, 
            self.bn.weight, self.bn.bias,
            out,
            N, C_IN, C_OUT, H_IN, W_IN, KH, KW,
            self.eps,
            stride_x[0], stride_x[1], stride_x[2], stride_x[3],
            stride_w[0], stride_w[1], stride_w[2], stride_w[3],
            stride_out[0], stride_out[1], stride_out[2], stride_out[3],
            H_OUT, W_OUT,
            BLOCK_H=BLOCK_H,
            BLOCK_W=BLOCK_W,
        )
        
        return out


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
