import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized GELU kernel for H100
# Uses vectorized loads/stores and fast GELU approximation
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fast GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
// This is the formula used in Google BERT and other popular implementations
__device__ __forceinline__ float fast_gelu(float x) {
    const float sqrt_2_over_pi = 0.7071067811865476f;
    const float coeff = 0.044715f;
    return 0.5f * x * (1.0f + tanhf(sqrt_2_over_pi * (x + coeff * x * x * x)));
}

__device__ __forceinline__ half fast_gelu_half(half x) {
    const float sqrt_2_over_pi = 0.7071067811865476f;
    const float coeff = 0.044715f;
    float x_f = __half2float(x);
    float result = 0.5f * x_f * (1.0f + tanhf(sqrt_2_over_pi * (x_f + coeff * x_f * x_f * x_f)));
    return __float2half(result);
}

// Vectorized kernel using float4 for better memory bandwidth
__global__ void gelu_kernel_float4(const float4* __restrict__ x, 
                                     float4* __restrict__ y, 
                                     const int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float4 val = x[idx];
        val.x = fast_gelu(val.x);
        val.y = fast_gelu(val.y);
        val.z = fast_gelu(val.z);
        val.w = fast_gelu(val.w);
        y[idx] = val;
    }
}

// Fallback kernel for non-vectorized case
__global__ void gelu_kernel_float(const float* __restrict__ x, 
                                   float* __restrict__ y, 
                                   const int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = fast_gelu(x[idx]);
    }
}

// Half precision kernel using vectorized loads
__global__ void gelu_kernel_half(half* __restrict__ y, 
                                   const half* __restrict__ x, 
                                   const int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = fast_gelu_half(x[idx]);
    }
}

__global__ void gelu_kernel_bf16(const float* __restrict__ temp_x,
                                   float* __restrict__ temp_y,
                                   const int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x_val = temp_x[idx];
        temp_y[idx] = fast_gelu(x_val);
    }
}

torch::Tensor gelu_op_float(torch::Tensor x) {
    auto y = torch::empty_like(x);
    const int n = x.numel();
    const int vec_n = n / 4;
    
    const int block_size = 256;
    const int grid_size = (vec_n + block_size - 1) / block_size;
    
    gelu_kernel_float4<<<grid_size, block_size, 0, 0>>>(
        (const float4*)x.data_ptr<float>(),
        (float4*)y.data_ptr<float>(),
        vec_n
    );
    
    // Handle remainder elements
    const int remainder = n % 4;
    if (remainder > 0) {
        const float* x_ptr = x.data_ptr<float>() + n - remainder;
        float* y_ptr = y.data_ptr<float>() + n - remainder;
        
        const int rem_grid = 1;
        gelu_kernel_float<<<rem_grid, remainder, 0, 0>>>(
            x_ptr, y_ptr, remainder
        );
    }
    
    return y;
}

torch::Tensor gelu_op_half(torch::Tensor x) {
    auto y = torch::empty_like(x);
    const int n = x.numel();
    
    const int block_size = 256;
    const int grid_size = (n + block_size - 1) / block_size;
    
    gelu_kernel_half<<<grid_size, block_size, 0, 0>>>(
        (half*)y.data_ptr<torch::Half>(),
        (const half*)x.data_ptr<torch::Half>(),
        n
    );
    
    return y;
}

torch::Tensor gelu_op_bf16(torch::Tensor x) {
    // Convert to float, apply gelu, convert back
    auto x_float = x.to(torch::kFloat);
    auto y_float = torch::empty_like(x_float);
    const int n = x_float.numel();
    
    const int block_size = 256;
    const int grid_size = (n + block_size - 1) / block_size;
    
    gelu_kernel_float<<<grid_size, block_size, 0, cudaStreamByIndex(0)>>>(
        x_float.data_ptr<float>(),
        y_float.data_ptr<float>(),
        n
    );
    
    return y_float.to(x.dtype());
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources=r'''
torch::Tensor gelu_op_float(torch::Tensor x);
torch::Tensor gelu_op_half(torch::Tensor x);
torch::Tensor gelu_op_bf16(torch::Tensor x);
''',
    cuda_sources=cuda_source,
    functions=['gelu_op_float', 'gelu_op_half', 'gelu_op_bf16'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dtype = x.dtype
        if dtype == torch.float32:
            return ext.gelu_op_float(x)
        elif dtype == torch.float16:
            return ext.gelu_op_half(x)
        elif dtype == torch.bfloat16:
            return ext.gelu_op_bf16(x)
        else:
            # Fallback to PyTorch for unsupported types
            return torch.nn.functional.gelu(x)

batch_size = 256
dim = 16384

def get_inputs():
    return [torch.randn(batch_size, dim)]

    return []
