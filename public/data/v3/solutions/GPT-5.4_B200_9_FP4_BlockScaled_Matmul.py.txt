import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ['fp4', 'fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['B200']
BLOCK_SIZE = 32


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 64, 'GROUP_M': 8}, num_warps=4, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 64, 'GROUP_M': 8}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 64, 'BLOCK_K': 64, 'GROUP_M': 8}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 64, 'GROUP_M': 8}, num_warps=8, num_stages=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def fp4_blockscaled_gemm_kernel(
    x_ptr, w_ptr, s_ptr, out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn_bytes,
    stride_sn_blocks,
    stride_om, stride_on,
    QUANT_BLOCK: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr, GROUP_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = tl.minimum(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_m = offs_m < M
    mask_n = offs_n < N

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    k_iter = 0
    while k_iter < K:
        offs_k = k_iter + tl.arange(0, BLOCK_K)
        mask_k = offs_k < K

        x = tl.load(
            x_ptr + offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk,
            mask=mask_m[:, None] & mask_k[None, :],
            other=0.0,
        ).to(tl.float32)

        pair_idx = offs_k // 2
        packed = tl.load(
            w_ptr + offs_n[:, None] * stride_wn_bytes + pair_idx[None, :],
            mask=mask_n[:, None] & mask_k[None, :],
            other=0,
        )
        packed_i = packed.to(tl.int32)
        low = packed_i & 0xF
        high = (packed_i >> 4) & 0xF
        nib = tl.where((offs_k[None, :] & 1) == 0, low, high)
        q = nib.to(tl.int32) - 8

        block_idx = offs_k // QUANT_BLOCK
        scales = tl.load(
            s_ptr + offs_n[:, None] * stride_sn_blocks + block_idx[None, :],
            mask=mask_n[:, None] & mask_k[None, :],
            other=0.0,
        ).to(tl.float32)
        w = q.to(tl.float32) * scales

        acc += tl.dot(x, tl.trans(w))
        k_iter += BLOCK_K

    out = acc.to(tl.float16)
    tl.store(out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on, out, mask=mask_m[:, None] & mask_n[None, :])


class Model(nn.Module):
    def __init__(self, K: int, N: int):
        super().__init__()
        self.K = K
        self.N = N
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        w_fp16 = torch.randn(N, K) * 0.02
        torch.random.set_rng_state(rng_state)
        self.weight_packed, self.scales = self._quantize_fp4(w_fp16)

    def _quantize_fp4(self, w: torch.Tensor):
        N, K = w.shape
        n_blocks = (K + BLOCK_SIZE - 1) // BLOCK_SIZE
        w_padded = torch.zeros(N, n_blocks * BLOCK_SIZE, dtype=w.dtype)
        w_padded[:, :K] = w
        w_blocked = w_padded.view(N, n_blocks, BLOCK_SIZE)
        amax = w_blocked.abs().amax(dim=-1, keepdim=True).clamp(min=1e-12)
        scales = amax / 7.0
        w_int = torch.round(w_blocked / scales).clamp(-7, 7).to(torch.int8) + 8
        w_flat = w_int.view(N, -1)
        lo = w_flat[:, 0::2].to(torch.uint8)
        hi = w_flat[:, 1::2].to(torch.uint8)
        packed = (hi << 4) | lo
        return nn.Parameter(packed, requires_grad=False), nn.Parameter(scales.squeeze(-1).to(torch.float16), requires_grad=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_cuda
        orig_shape = x.shape
        x2d = x.reshape(-1, self.K).contiguous()
        M = x2d.shape[0]
        out = torch.empty((M, self.N), device=x.device, dtype=torch.float16)
        grid = lambda META: (triton.cdiv(M, META['BLOCK_M']) * triton.cdiv(self.N, META['BLOCK_N']),)
        fp4_blockscaled_gemm_kernel[grid](
            x2d, self.weight_packed, self.scales, out,
            M, self.N, self.K,
            x2d.stride(0), x2d.stride(1),
            self.weight_packed.stride(0),
            self.scales.stride(0),
            out.stride(0), out.stride(1),
            QUANT_BLOCK=BLOCK_SIZE,
        )
        return out.reshape(*orig_shape[:-1], self.N)


K = 4096
N = 4096
batch_size = 8
seq_len = 2048


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [K, N]
