import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__device__ __forceinline__ T tanh_activation(T x) {
    return tanhf(x);
}

// Optimized fused kernel: Tanh + Scaling + BiasAdd + MaxPool
// Uses better memory access patterns and loop unrolling hints
template <typename T>
__global__ void fused_conv_post_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T* __restrict__ bias,
    T scaling_factor,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int pool_kernel_size,
    int out_height,
    int out_width
) {
    // Use 256 threads per block, each thread handles one output
    int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_outputs = batch_size * channels * out_height * out_width;
    
    if (out_idx >= total_outputs) return;
    
    // Calculate output coordinates
    int tmp = out_idx;
    int b = tmp / (channels * out_height * out_width);
    tmp = tmp % (channels * out_height * out_width);
    int c = tmp / (out_height * out_width);
    tmp = tmp % (out_height * out_width);
    int out_y = tmp / out_width;
    int out_x = tmp % out_width;
    
    // Calculate corresponding input region for max pool
    int in_y_start = out_y * pool_kernel_size;
    int in_x_start = out_x * pool_kernel_size;
    
    // Find max in the pool region with fused operations
    T max_val = -INFINITY;
    
    // Manual unroll for common pool size of 2
    if (pool_kernel_size == 2) {
        // Load all 4 positions, apply ops, compare
        const int idx00 = ((b * channels + c) * in_height + in_y_start) * in_width + in_x_start;
        const int idx01 = idx00 + 1;
        const int idx10 = ((b * channels + c) * in_height + in_y_start + 1) * in_width + in_x_start;
        const int idx11 = idx10 + 1;
        
        T v00 = input[idx00];
        T v01 = input[idx01];
        T v10 = input[idx10];
        T v11 = input[idx11];
        
        // Apply tanh, scale, bias and find max
        v00 = tanh_activation(v00) * scaling_factor + bias[c];
        v01 = tanh_activation(v01) * scaling_factor + bias[c];
        v10 = tanh_activation(v10) * scaling_factor + bias[c];
        v11 = tanh_activation(v11) * scaling_factor + bias[c];
        
        max_val = v00;
        if (v01 > max_val) max_val = v01;
        if (v10 > max_val) max_val = v10;
        if (v11 > max_val) max_val = v11;
    } else {
        // Generic pool size
        for (int dy = 0; dy < pool_kernel_size; dy++) {
            for (int dx = 0; dx < pool_kernel_size; dx++) {
                int in_y = in_y_start + dy;
                int in_x = in_x_start + dx;
                
                if (in_y < in_height && in_x < in_width) {
                    int in_idx = ((b * channels + c) * in_height + in_y) * in_width + in_x;
                    T val = input[in_idx];
                    val = tanh_activation(val) * scaling_factor + bias[c];
                    
                    if (val > max_val) {
                        max_val = val;
                    }
                }
            }
        }
    }
    
    output[out_idx] = max_val;
}

// Grid-stride loop version for larger inputs
template <typename T>
__global__ void fused_conv_post_kernel_v2(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T* __restrict__ bias,
    T scaling_factor,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int pool_kernel_size,
    int out_height,
    int out_width
) {
    int total_outputs = batch_size * channels * out_height * out_width;
    
    // Grid-stride loop
    for (int out_idx = blockIdx.x * blockDim.x + threadIdx.x; 
         out_idx < total_outputs; 
         out_idx += blockDim.x * gridDim.x) {
    
        // Calculate output coordinates
        int tmp = out_idx;
        int b = tmp / (channels * out_height * out_width);
        tmp = tmp % (channels * out_height * out_width);
        int c = tmp / (out_height * out_width);
        tmp = tmp % (out_height * out_width);
        int out_y = tmp / out_width;
        int out_x = tmp % out_width;
        
        // Calculate corresponding input region for max pool
        int in_y_start = out_y * pool_kernel_size;
        int in_x_start = out_x * pool_kernel_size;
        
        // Find max in the pool region with fused operations
        T max_val = -INFINITY;
        
        // Manual unroll for pool size 2
        if (pool_kernel_size == 2) {
            const int idx00 = ((b * channels + c) * in_height + in_y_start) * in_width + in_x_start;
            const int idx01 = idx00 + 1;
            const int idx10 = ((b * channels + c) * in_height + in_y_start + 1) * in_width + in_x_start;
            const int idx11 = idx10 + 1;
            
            T v00 = input[idx00];
            T v01 = input[idx01];
            T v10 = input[idx10];
            T v11 = input[idx11];
            
            v00 = tanh_activation(v00) * scaling_factor + bias[c];
            v01 = tanh_activation(v01) * scaling_factor + bias[c];
            v10 = tanh_activation(v10) * scaling_factor + bias[c];
            v11 = tanh_activation(v11) * scaling_factor + bias[c];
            
            max_val = v00;
            if (v01 > max_val) max_val = v01;
            if (v10 > max_val) max_val = v10;
            if (v11 > max_val) max_val = v11;
        } else {
            for (int dy = 0; dy < pool_kernel_size; dy++) {
                for (int dx = 0; dx < pool_kernel_size; dx++) {
                    int in_y = in_y_start + dy;
                    int in_x = in_x_start + dx;
                    
                    if (in_y < in_height && in_x < in_width) {
                        int in_idx = ((b * channels + c) * in_height + in_y) * in_width + in_x;
                        T val = input[in_idx];
                        val = tanh_activation(val) * scaling_factor + bias[c];
                        
                        if (val > max_val) {
                            max_val = val;
                        }
                    }
                }
            }
        }
        
        output[out_idx] = max_val;
    }
}

torch::Tensor fused_conv_post_op(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_kernel_size
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int out_height = in_height / pool_kernel_size;
    int out_width = in_width / pool_kernel_size;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    int total_outputs = batch_size * channels * out_height * out_width;
    int block_size = 256;
    int grid_size = min(1024, (total_outputs + block_size - 1) / block_size);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_conv_post_kernel", ([&] {
        fused_conv_post_kernel_v2<scalar_t><<<grid_size, block_size>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            scaling_factor,
            batch_size,
            channels,
            in_height,
            in_width,
            pool_kernel_size,
            out_height,
            out_width
        );
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_conv_post_op(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_kernel_size
);
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_post_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Use PyTorch's optimized conv (cuDNN)
        x = self.conv(x)
        
        # Fused post-processing kernel: tanh + scaling + bias + maxpool
        x = ext.fused_conv_post_op(
            x, 
            self.bias, 
            self.scaling_factor, 
            self.pool_kernel_size
        )
        
        return x

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0, (16, 1, 1), 2]
