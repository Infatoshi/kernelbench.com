import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _gemv_bf16_kernel(
    x_ptr, w_ptr, y_ptr,
    B, K, N,
    stride_xb, stride_xk,
    stride_wn, stride_wk,
    stride_yb, stride_yn,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // tl.cdiv(N, BLOCK_N)
    n_block = pid % tl.cdiv(N, BLOCK_N)
    offs_n = n_block * BLOCK_N + tl.arange(0, BLOCK_N)
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)

    k = 0
    while k < K:
        offs_k = k + tl.arange(0, BLOCK_K)
        x_ptrs = x_ptr + b * stride_xb + offs_k * stride_xk
        x = tl.load(x_ptrs, mask=offs_k < K, other=0.0).to(tl.float32)

        w_ptrs = w_ptr + offs_n[:, None] * stride_wn + offs_k[None, :] * stride_wk
        w = tl.load(w_ptrs, mask=(offs_n[:, None] < N) & (offs_k[None, :] < K), other=0.0).to(tl.float32)
        acc += tl.sum(w * x[None, :], axis=1)
        k += BLOCK_K

    y_ptrs = y_ptr + b * stride_yb + offs_n * stride_yn
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=offs_n < N)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf16 = x.to(dtype=torch.bfloat16)
        assert x_bf16.is_cuda and self.weight.is_cuda
        B, K = x_bf16.shape
        N = self.weight.shape[0]
        y = torch.empty((B, N), device=x_bf16.device, dtype=torch.bfloat16)
        grid = (B * triton.cdiv(N, 128),)
        _gemv_bf16_kernel[grid](
            x_bf16, self.weight, y,
            B, K, N,
            x_bf16.stride(0), x_bf16.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            y.stride(0), y.stride(1),
            BLOCK_N=128,
            BLOCK_K=128,
            num_warps=4,
            num_stages=4,
        )
        return y


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX3090", "A100", "H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16)]


def get_init_inputs():
    return []
