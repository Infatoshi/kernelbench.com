import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def layernorm_kernel(
    X,  # pointer to input
    Y,  # pointer to output
    W,  # pointer to weights (gamma)
    B,  # pointer to bias (beta)
    stride,  # how much to increase the pointer when moving by 1 row
    N,  # number of columns in X
    eps,  # epsilon to avoid division by zero
    BLOCK_SIZE: tl.constexpr,
):
    # Map the program id to the row of X and Y it should compute.
    row = tl.program_id(0)
    X += row * stride
    Y += row * stride
    
    # Compute mean
    # NOTE: using float32 for accumulation to avoid precision issues
    mean = tl.zeros([1], dtype=tl.float32)
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        a = tl.load(X + cols, mask=cols < N, other=0.0).to(tl.float32)
        mean += tl.sum(a, axis=0)
    mean = mean / N
    
    # Compute variance
    _var = tl.zeros([1], dtype=tl.float32)
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        x = tl.load(X + cols, mask=cols < N, other=0.0).to(tl.float32)
        x = tl.where(cols < N, x - mean, 0.0)
        _var += tl.sum(x * x, axis=0)
    var = _var / N
    rstd = 1 / tl.sqrt(var + eps)
    
    # Normalize and apply linear transformation
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        w = tl.load(W + cols, mask=mask)
        b = tl.load(B + cols, mask=mask)
        x = tl.load(X + cols, mask=mask, other=0.0).to(tl.float32)
        x_hat = (x - mean) * rstd
        y = x_hat * w + b
        # Write output
        tl.store(Y + cols, y, mask=mask)


def layernorm_forward(x, weight, bias, eps=1e-5):
    # Reshape input to 2D: (batch_size * ..., N)
    # where N is the normalized dimension
    shape = x.shape
    x_2d = x.view(-1, shape[-1]) if len(shape) > 1 else x.view(1, -1)
    
    N = x_2d.shape[-1]
    M = x_2d.shape[0]
    
    # Allocate output
    y = torch.empty_like(x_2d)
    
    # Less than 64KB per feature: enqueue fused kernel
    MAX_FUSED_SIZE = 65536 // x.element_size()
    BLOCK_SIZE = min(MAX_FUSED_SIZE, triton.next_power_of_2(N))
    if N > BLOCK_SIZE:
        raise RuntimeError("This layer norm doesn't support feature dim >= 64KB.")
    
    # heuristics for number of warps
    num_warps = min(max(BLOCK_SIZE // 256, 1), 8)
    
    # enqueue kernel
    layernorm_kernel[(M,)](
        x_2d, y, weight, bias,
        x_2d.stride(0),
        N, eps,
        BLOCK_SIZE=BLOCK_SIZE,
        num_warps=num_warps,
    )
    
    return y.view(shape)


class Model(nn.Module):
    """
    Optimized Layer Normalization model using Triton kernels.
    """
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)
        self.normalized_shape = normalized_shape
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Flatten normalized dimensions
        if isinstance(self.normalized_shape, int):
            normalized_shape = (self.normalized_shape,)
        else:
            normalized_shape = self.normalized_shape
        
        # Calculate the normalized size
        normalized_size = 1
        for dim in normalized_shape:
            normalized_size *= dim
        
        # Get gamma and beta from the reference LayerNorm
        gamma = self.ln.weight
        beta = self.ln.bias
        
        # Reshape input for layer norm
        original_shape = x.shape
        x_reshaped = x.view(-1, normalized_size)
        
        # Apply custom layer norm
        y = layernorm_forward(x_reshaped, gamma, beta, self.ln.eps)
        
        return y.view(original_shape)
