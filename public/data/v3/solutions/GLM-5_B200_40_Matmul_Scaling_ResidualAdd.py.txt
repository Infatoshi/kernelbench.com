import torch
import torch.nn as nn
import triton
import triton.language as tl

# Fused Matmul + Scale + Residual Add kernel
# Computes: output = (x @ W^T + bias) * (1 + scaling_factor)
# Which is equivalent to: x * scaling_factor + x after matmul

@triton.jit
def fused_matmul_scale_residual_kernel(
    # Pointers to matrices
    x_ptr, w_ptr, b_ptr, output_ptr,
    # Matrix dimensions
    M, N, K,
    # Strides
    stride_x_m, stride_x_k,
    stride_w_n, stride_w_k,
    stride_b_n,
    stride_out_m, stride_out_n,
    # Scaling
    scale_factor,
    # Meta parameters
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """Fused kernel for matmul + scale + residual add."""
    # Block IDs
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Create pointers for X
    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    x_ptrs = x_ptr + (offs_am[:, None] * stride_x_m + offs_k[None, :] * stride_x_k)
    w_ptrs = w_ptr + (offs_k[:, None] * stride_w_k + offs_bn[None, :] * stride_w_n)

    # Accumulator
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    # Main loop over K
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Load X and W tiles
        x = tl.load(x_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        w = tl.load(w_ptrs, mask=offs_k[:, None] < K - k * BLOCK_SIZE_K, other=0.0)
        
        # Matrix multiply accumulate
        accumulator += tl.dot(x, w)
        
        # Advance pointers
        x_ptrs += BLOCK_SIZE_K * stride_x_k
        w_ptrs += BLOCK_SIZE_K * stride_w_k

    # Load bias if present (b_ptr is not null)
    if b_ptr is not None:
        offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
        bias = tl.load(b_ptr + offs_bn * stride_b_n, mask=offs_bn < N, other=0.0)
        accumulator += bias[None, :]

    # Apply scaling: output = accumulator * (1 + scale_factor)
    # This fuses: x * scale_factor + x = x * (1 + scale_factor)
    output = accumulator * (1.0 + scale_factor)

    # Store output
    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    output_ptrs = output_ptr + (offs_am[:, None] * stride_out_m + offs_bn[None, :] * stride_out_n)
    output_mask = (offs_am[:, None] < M) & (offs_bn[None, :] < N)
    tl.store(output_ptrs, output, mask=output_mask)


def fused_matmul_scale_residual(x, weight, bias, scale_factor):
    """Fused matmul + scale + residual add."""
    # x: (M, K), weight: (N, K), output: (M, N)
    M, K = x.shape
    N, K_w = weight.shape
    assert K == K_w, f"K mismatch: {K} vs {K_w}"
    
    # Allocate output
    output = torch.empty((M, N), device=x.device, dtype=torch.float32 if x.dtype == torch.float32 else torch.float16)
    
    # Tuning parameters
    BLOCK_SIZE_M = 64
    BLOCK_SIZE_N = 64
    BLOCK_SIZE_K = 32
    GROUP_SIZE_M = 8
    
    # Grid
    grid = lambda META: (
        triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
    )
    
    # Kernel launch
    fused_matmul_scale_residual_kernel[grid](
        x, weight, bias, output,
        M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        bias.stride(0) if bias is not None else 0,
        output.stride(0), output.stride(1),
        scale_factor,
        BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
    )
    
    return output


class Model(nn.Module):
    """
    A model that performs a matrix multiplication, scaling, and residual addition.
    Uses fused Triton kernel for optimal performance.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Use nn.Linear to match state_dict structure from reference
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        """
        Forward pass: fused matmul + scale + residual add.
        Equivalent to: (x @ W^T + bias) * (1 + scaling_factor)
        """
        # Convert to same dtype for computation
        input_dtype = x.dtype
        if x.dtype == torch.float32:
            x_compute = x
        else:
            x_compute = x.half()
        
        # Ensure weight is contiguous and transposed correctly
        weight = self.matmul.weight.half() if x_compute.dtype == torch.float16 else self.matmul.weight
        bias = self.matmul.bias.half() if x_compute.dtype == torch.float16 else self.matmul.bias
        
        # Run fused kernel
        output = fused_matmul_scale_residual(
            x_compute, weight, bias, self.scaling_factor
        )
        
        # Convert back to input dtype
        if input_dtype == torch.float32:
            return output.float() if output.dtype != torch.float32 else output
        else:
            return output.half() if output.dtype != torch.float16 else output


batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 0.5


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, scaling_factor]
