import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cstdio>

// Optimized RMSNorm kernel with fused computation
template<typename T>
__global__ void rmsnorm_kernel(
    const T* __restrict__ x,
    T* __restrict__ output,
    const int batch_size,
    const int num_features,
    const int dim1,
    const int dim2,
    const float eps) {
    
    // Shared memory for reduction
    extern __shared__ float sdata[];
    
    const int b = blockIdx.y;
    const int rest_idx = blockIdx.x;
    const int tid = threadIdx.x;
    
    // Compute d1, d2 from rest_idx
    const int d1 = rest_idx / dim2;
    const int d2 = rest_idx % dim2;
    
    // Compute base index for this (b, d1, d2) position
    // Tensor is stored in row-major (C) order: [b, f, d1, d2]
    // Index = b * num_features * dim1 * dim2 + f * dim1 * dim2 + d1 * dim2 + d2
    const int base = b * num_features * dim1 * dim2 + d1 * dim2 + d2;
    const int feat_stride = dim1 * dim2;
    
    // Debug output for first block
    if (b == 0 && rest_idx == 0 && tid == 0) {
        printf("batch_size=%d, features=%d, dim1=%d, dim2=%d\n", batch_size, num_features, dim1, dim2);
        printf("b=%d, d1=%d, d2=%d, base=%d, feat_stride=%d\n", b, d1, d2, base, feat_stride);
    }
    
    // First phase: compute sum of squares
    // Each thread handles multiple features
    float sum_sq = 0.0f;
    
    for (int f = tid; f < num_features; f += blockDim.x) {
        const int idx = base + f * feat_stride;
        const T val = x[idx];
        sum_sq += (float)val * (float)val;
    }
    
    // Store to shared memory
    sdata[tid] = sum_sq;
    __syncthreads();
    
    // Block reduction in shared memory
    for (unsigned int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    // Final warp reduction
    if (tid < 32) {
        float warp_sum = sdata[tid];
        if (blockDim.x > 32) warp_sum += sdata[tid + 32];
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            warp_sum += __shfl_down_sync(0xffffffff, warp_sum, offset);
        }
        if (tid == 0) sdata[0] = warp_sum;
    }
    
    __syncthreads();
    float rms = 0.0f;
    if (tid == 0) {
        rms = rsqrtf(sdata[0] / (float)num_features + eps);
        if (b == 0 && rest_idx == 0) {
            printf("b=%d, d1=%d, rms=%.6f, sum_sq=%.6f\n", b, d1, rms, sdata[0]);
        }
    }
    rms = __shfl_sync(0xffffffff, rms, 0);
    
    // Second phase: normalize
    for (int f = tid; f < num_features; f += blockDim.x) {
        const int idx = base + f * feat_stride;
        const T val = x[idx];
        output[idx] = (T)((float)val * rms);
    }
}

torch::Tensor rmsnorm_op(torch::Tensor x, float eps) {
    auto output = torch::empty_like(x);
    
    const int batch_size = x.size(0);
    const int num_features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    const int threads = 128;
    const int shared_mem = threads * sizeof(float);
    
    // Grid: (dim1 * dim2, batch_size)
    dim3 grid(dim1 * dim2, batch_size);
    
    if (x.dtype() == torch::kFloat32) {
        rmsnorm_kernel<float><<<grid, threads, shared_mem>>>(
            x.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size, num_features, dim1, dim2, eps
        );
    } else if (x.dtype() == torch::kFloat16) {
        rmsnorm_kernel<half><<<grid, threads, shared_mem>>>(
            x.data_ptr<half>(),
            output.data_ptr<half>(),
            batch_size, num_features, dim1, dim2, eps
        );
    } else if (x.dtype() == torch::kBFloat16) {
        rmsnorm_kernel<nv_bfloat16><<<grid, threads, shared_mem>>>(
            x.data_ptr<nv_bfloat16>(),
            output.data_ptr<nv_bfloat16>(),
            batch_size, num_features, dim1, dim2, eps
        );
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='rmsnorm_cuda',
    cpp_sources='torch::Tensor rmsnorm_op(torch::Tensor x, float eps);',
    cuda_sources=cuda_source,
    functions=['rmsnorm_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_op(x, self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    return [torch.randn(batch_size, features, dim1, dim2)]

def get_init_inputs():
    return [features]
