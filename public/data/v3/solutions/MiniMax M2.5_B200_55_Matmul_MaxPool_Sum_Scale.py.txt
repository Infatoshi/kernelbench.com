import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused CUDA kernel - fixed version
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__global__ void fused_kernel(
    const T* __restrict__ x,
    const T* __restrict__ w,
    const T* __restrict__ b,
    T* __restrict__ out,
    T scale,
    int batch,
    int M,
    int N
) {
    int bid = blockIdx.x;
    int tid = threadIdx.x;
    int nthreads = blockDim.x;
    
    extern __shared__ T sdata[];
    T* matmul_buf = sdata;
    T* pooled_buf = sdata + N;
    
    // Initialize bias
    for (int j = tid; j < N; j += nthreads) {
        matmul_buf[j] = b ? b[j] : T(0);
    }
    __syncthreads();
    
    // GEMM: y = x @ w^T
    for (int j = tid; j < N; j += nthreads) {
        T sum = matmul_buf[j];
        const T* x_row = x + bid * M;
        const T* w_row = w + j * M;
        #pragma unroll 8
        for (int i = 0; i < M; i++) {
            sum = sum + x_row[i] * w_row[i];
        }
        matmul_buf[j] = sum;
    }
    __syncthreads();
    
    // MaxPool1d with kernel_size=2
    int N2 = N / 2;
    for (int j = tid; j < N2; j += nthreads) {
        T a = matmul_buf[2*j];
        T b_val = matmul_buf[2*j + 1];
        pooled_buf[j] = (a > b_val) ? a : b_val;
    }
    __syncthreads();
    
    // Sum reduction
    for (int s = N2 / 2; s > 32; s >>= 1) {
        if (tid < s) {
            pooled_buf[tid] = pooled_buf[tid] + pooled_buf[tid + s];
        }
        __syncthreads();
    }
    
    // Warp-level reduction
    if (tid < 32) {
        volatile T* s = pooled_buf;
        if (N2 >= 64) s[tid] = s[tid] + s[tid + 32];
        if (N2 >= 32) s[tid] = s[tid] + s[tid + 16];
        s[tid] = s[tid] + s[tid + 8];
        s[tid] = s[tid] + s[tid + 4];
        s[tid] = s[tid] + s[tid + 2];
        s[tid] = s[tid] + s[tid + 1];
    }
    
    if (tid == 0) {
        out[bid] = pooled_buf[0] * scale;
    }
}

torch::Tensor fused_op_float32(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale) {
    auto output = torch::empty({x.size(0)}, x.options());
    
    const int batch = x.size(0);
    const int M = x.size(1);
    const int N = weight.size(0);
    
    const int threads = 256;
    const int blocks = batch;
    const int smem_size = (N + N/2) * sizeof(float);
    
    fused_kernel<float><<<blocks, threads, smem_size>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        scale,
        batch, M, N
    );
    
    cudaDeviceSynchronize();
    return output;
}

torch::Tensor fused_op_float16(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale) {
    // Convert to float32 for computation
    auto x_f = x.to(torch::kFloat32);
    auto w_f = weight.to(torch::kFloat32);
    auto b_f = bias.defined() ? bias.to(torch::kFloat32) : torch::Tensor();
    
    auto output_f = fused_op_float32(x_f, w_f, b_f, scale);
    
    return output_f.to(torch::kFloat16);
}

torch::Tensor fused_op_bfloat16(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale) {
    // Convert to float32 for computation
    auto x_f = x.to(torch::kFloat32);
    auto w_f = weight.to(torch::kFloat32);
    auto b_f = bias.defined() ? bias.to(torch::kFloat32) : torch::Tensor();
    
    auto output_f = fused_op_float32(x_f, w_f, b_f, scale);
    
    return output_f.to(torch::kBFloat16);
}
'''

cpp_source = r'''
torch::Tensor fused_op_float32(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale);
torch::Tensor fused_op_float16(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale);
torch::Tensor fused_op_bfloat16(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float scale);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_op_float32', 'fused_op_float16', 'fused_op_bfloat16'],
    verbose=False,
    extra_cuda_cflags=['-maxrregcount=128', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        
        self.linear = nn.Linear(in_features, out_features)
        
    def forward(self, x):
        if x.dtype == torch.float32:
            return ext.fused_op_float32(x, self.linear.weight, self.linear.bias, self.scale_factor)
        elif x.dtype == torch.float16:
            return ext.fused_op_float16(x, self.linear.weight, self.linear.bias, self.scale_factor)
        elif x.dtype == torch.bfloat16:
            return ext.fused_op_bfloat16(x, self.linear.weight, self.linear.bias, self.scale_factor)
        else:
            # Fallback
            x = self.linear(x)
            x = x.unsqueeze(1)
            x = torch.nn.functional.max_pool1d(x, self.kernel_size).squeeze(1)
            x = torch.sum(x, dim=1)
            x = x * self.scale_factor
            return x

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 2, 0.5]
