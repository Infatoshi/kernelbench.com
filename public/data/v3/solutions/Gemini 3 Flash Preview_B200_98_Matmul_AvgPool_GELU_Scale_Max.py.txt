import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(
    A_ptr, B_ptr, Bias_ptr, Out_ptr,
    M, N, K,
    pool_kernel_size, scale_factor,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """
    Compute Matmul + AvgPool + GELU + Scale + Max.
    Wait, doing AvgPool and Max inside one kernel requires careful handling.
    Since we need to Max over the entire row for each batch element, 
    but Matmul is tiled, we have a challenge.
    
    Strategy:
    1. Each kernel instance computes a portion of the row after matmul.
    2. But we need a full row available for AvgPool if we cross block boundaries.
    
    Actually, notice N=4096 and M=128.
    For each m in M:
        row = A[m, :] @ B.T + bias
        pooled_row = AvgPool(row)
        final[m] = Max(Scale(GELU(pooled_row)))
    
    Since M is small (128) and N is 4096, 
    we could have one block per row (one block per M).
    Inside each block, threads compute the whole row, then pool, then max.
    """
    
    m = tl.program_id(0)
    if m >= M: return

    # Offsets for A [M, K]
    a_row_ptr = A_ptr + m * stride_am
    
    # We want to compute the entire row [N] for this m.
    # N = 4096. We can loop over N in chunks.
    
    # For global max reduction
    row_max = -1.0e38

    # To do AvgPool(size=4), we need to handle boundaries. 
    # With pool_kernel_size=4, N_pooled = 4096 - 4 + 1 = 4093.
    # Let's compute Matmul in tiles of N.
    
    # We'll use a sliding window approach for AvgPool to avoid computing matmul twice.
    # Shared memory within Triton is managed by the compiler.
    
    # But for N_pooled, we need matmul_out[i:i+4].
    # Let's compute a tile of matmul values, then pool what we can.
    
    # Current matmul element
    # last_3_vals = [0, 0, 0]
    
    m_val = -1e38
    
    # Buffering for AvgPool: space to keep 'pool_kernel_size - 1' previous values
    prev_vals = tl.zeros((4,), dtype=tl.float32) # Enough for kernel_size 4
    
    # Let's simplify: compute all N matmul values for this row first.
    # Since N=4096 is small, we can probably just use a large tile or loop.
    
    # N is 4096. Let's use BLOCK_SIZE_N = 1024.
    # We store the results in a temporary buffer or just keep them in registers if possible.
    # But 4096 floats in registers is too many.
    
    # Let's use a local array in "shared memory" (tl.static_print doesn't exist, but we can use tl.zeros)
    # Actually, Triton doesn't allow dynamic indexing into large registers easily.
    
    # Better: Compute tiles of N. For each tile, compute AvgPool.
    # Crossing tiles: need to keep last (pool_kernel_size - 1) elements from previous tile.

    prev_0 = 0.0
    prev_1 = 0.0
    prev_2 = 0.0
    
    for n_offset in range(0, N, BLOCK_SIZE_N):
        cols = n_offset + tl.arange(0, BLOCK_SIZE_N)
        # Compute Matmul tile [1, BLOCK_SIZE_N]
        # acc = A[m, :] @ B[n_offset:n_offset+BLOCK_SIZE_N, :]^T + bias
        acc = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        for k_offset in range(0, K, BLOCK_SIZE_K):
            k_indices = k_offset + tl.arange(0, BLOCK_SIZE_K)
            a_tile = tl.load(a_row_ptr + k_indices * stride_ak) # [BLOCK_SIZE_K]
            b_tile = tl.load(B_ptr + cols[:, None] * stride_bn + k_indices[None, :] * stride_bk) # [BLOCK_SIZE_N, BLOCK_SIZE_K]
            acc += tl.sum(a_tile[None, :] * b_tile, axis=1)
        
        acc += tl.load(Bias_ptr + cols)
        
        # Now we have acc [BLOCK_SIZE_N].
        # We need to compute AvgPool.
        # For a 1D window of size 4:
        # result[i] = (vals[i] + vals[i+1] + vals[i+2] + vals[i+3]) / 4
        
        # To compute this efficiently in Triton for the whole tile:
        # We can shift 'acc' and sum.
        
        # For the very first elements of the tile (that use elements from the previous tile):
        # We handle them separately or by padding.
        
        # Let's create a padded version: [prev3, prev2, prev1, acc0, acc1, ..., accN-1]
        # size = 3 + BLOCK_SIZE_N
        
        # Actually, simpler: compute all pools that fall ENTIRELY within the tile.
        # And handle the overlapping ones.
        
        # Shifted versions of acc
        # v0: [p0, p1, p2, acc0, acc1, ..., accN-4]
        # v1: [p1, p2, acc0, acc1, ..., accN-3]
        # v2: [p2, acc0, acc1, ..., accN-2]
        # v3: [acc0, acc1, ..., accN-1]
        
        # Let's use tl.cat to prepend previous values
        # This is slightly complex in Triton.
        
        # Alternative: just compute the matmul and store to a temp global buffer.
        # Then have a second pass or do it in one kernel with a sync (if possible).
        # But Triton doesn't have cross-block sync.
        
        # Given M=128, N=4096, K=4096, the matmul is the bottleneck.
        # We can just store the full row in a scratchpad.
        # Scratchpad size: 128 * 4096 * 4 bytes = 2MB. Very small.
        
        # Let's just do it in two steps for correctness and then fuse.
        # Wait, if I use one block per row, I don't need sync. I have the whole row.
        # I can just store the row in temp global memory.
        pass

# Since I need to be fast, I should use a fused approach.
# Let's use the scratchpad strategy.

@triton.jit
def fused_all_in_one(
    A_ptr, B_ptr, Bias_ptr, Out_ptr,
    M, N, K,
    pool_kernel_size, scale_factor,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    m = tl.program_id(0)
    
    # We'll compute the entire row and keep it in a temp buffer in global memory
    # shared between the tiles of the same m.
    # Actually, we can just compute the max for each tile and then reduce.
    # But AvgPool needs access to neighboring tiles.
    
    # Let's use a simpler strategy:
    # Each m is handled by a single program. 
    # It loops through N in tiles, keeps the last 3 elements of the previous tile.
    
    a_row_ptr = A_ptr + m * stride_am
    
    prev_vals = tl.zeros((4,), dtype=tl.float32)
    max_val = -1e38

    for n_offset in range(0, N, BLOCK_SIZE_N):
        cols = n_offset + tl.arange(0, BLOCK_SIZE_N)
        
        # Matmul tile [1, BLOCK_SIZE_N]
        acc = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        for k_offset in range(0, K, BLOCK_SIZE_K):
            k_indices = k_offset + tl.arange(0, BLOCK_SIZE_K)
            a_tile = tl.load(a_row_ptr + k_indices * stride_ak) 
            b_tile = tl.load(B_ptr + cols[:, None] * stride_bn + k_indices[None, :] * stride_bk)
            acc += tl.sum(a_tile[None, :] * b_tile, axis=1)
        acc += tl.load(Bias_ptr + cols)
        
        # Now compute AvgPool for this tile. 
        # A pool window [i, i+1, i+2, i+3]
        # Elements in this tile: n=0..1023
        # Indices in row: n_offset + n
        
        # We'll use a local buffer of size BLOCK_SIZE_N + 4
        # to simplify. (Must be power of 2 for Triton?)
        # Let's use 1024 + 4 and pad to 2048 or just handle with concat.
        
        # Actually, let's just do it manually.
        # Combined view of current tile and previous 3 elements.
        
        # pools[0] = (prev[1] + prev[2] + prev[3] + acc[0]) / 4
        # pools[1] = (prev[2] + prev[3] + acc[0] + acc[1]) / 4
        # pools[2] = (prev[3] + acc[0] + acc[1] + acc[2]) / 4
        # pools[3:] = (acc[0:N-3] + acc[1:N-2] + acc[2:N-1] + acc[3:N]) / 4
        
        p0 = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        p1 = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        p2 = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        p3 = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
        
        # p0: prev[1], prev[2], prev[3], acc[0], ..., acc[N-4]
        p0 = tl.join(prev_vals[1:4], acc[0:BLOCK_SIZE_N-3])
        # p1: prev[2], prev[3], acc[0], acc[1], ..., acc[N-3]
        p1 = tl.join(prev_vals[2:4], acc[0:BLOCK_SIZE_N-2])
        # p2: prev[3], acc[0], acc[1], acc[2], ..., acc[N-2]
        p2 = tl.join(prev_vals[3:4], acc[0:BLOCK_SIZE_N-1])
        # p3: acc
        p3 = acc
        
        pools = (p0 + p1 + p2 + p3) / 4.0
        
        i_row = n_offset - 3 + tl.arange(0, BLOCK_SIZE_N)
        mask = (i_row >= 0) & (i_row <= N - pool_kernel_size)
        
        activated = pools * 0.5 * (1.0 + tl.extra.cuda.libdevice.erf(pools * 0.70710678118))
        scaled = activated * scale_factor
        
        tile_max = tl.max(tl.where(mask, scaled, -1e38), axis=0)
        max_val = tl.maximum(max_val, tile_max)
        
        # Update prev_vals - we need the last 3 elements of acc
        # prev_vals[1] = acc[N-3], prev_vals[2] = acc[N-2], prev_vals[3] = acc[N-1]
        # But we must use power-of-2 indices for slicing in some cases or use join.
        # Let's just store the whole acc and pick the last elements.
        new_prev = tl.zeros((4,), dtype=tl.float32)
        # Slicing acc[BLOCK_SIZE_N-3:BLOCK_SIZE_N] might not work if not power of 2
        # Let's try it.
        new_prev[1] = tl.load(Out_ptr + M) # dummy, just to initialize
        # Wait, let's just use indices.
        new_prev[1] = tl.sum(tl.where(tl.arange(0, BLOCK_SIZE_N) == BLOCK_SIZE_N-3, acc, 0.0), axis=0)
        new_prev[2] = tl.sum(tl.where(tl.arange(0, BLOCK_SIZE_N) == BLOCK_SIZE_N-2, acc, 0.0), axis=0)
        new_prev[3] = tl.sum(tl.where(tl.arange(0, BLOCK_SIZE_N) == BLOCK_SIZE_N-1, acc, 0.0), axis=0)
        prev_vals = new_prev

    tl.store(Out_ptr + m, max_val)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        out = torch.empty((M,), device=x.device, dtype=x.dtype)
        
        grid = (M,)
        fused_all_in_one[grid](
            x, self.matmul.weight, self.matmul.bias, out,
            M, N, K,
            self.pool_kernel_size, self.scale_factor,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            BLOCK_SIZE_N=1024, BLOCK_SIZE_K=32,
        )
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
