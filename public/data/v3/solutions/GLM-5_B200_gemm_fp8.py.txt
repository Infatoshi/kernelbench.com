"""
Optimized FP8 GEMM for B200 using shared memory tiling with CUDA.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized GEMM using shared memory tiling
__global__ void optimized_gemm_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    float* __restrict__ C,
    const float scale_a,
    const float scale_b,
    const int M, const int N, const int K
) {
    const int TILE_SIZE = 32;
    
    __shared__ float A_tile[TILE_SIZE][TILE_SIZE];
    __shared__ float B_tile[TILE_SIZE][TILE_SIZE];
    
    const int row = blockIdx.y * TILE_SIZE + threadIdx.y;
    const int col = blockIdx.x * TILE_SIZE + threadIdx.x;
    
    float sum = 0.0f;
    
    for (int t = 0; t < (K + TILE_SIZE - 1) / TILE_SIZE; t++) {
        // Load A tile
        int a_col = t * TILE_SIZE + threadIdx.x;
        if (row < M && a_col < K) {
            A_tile[threadIdx.y][threadIdx.x] = static_cast<float>(A[row * K + a_col]) * scale_a;
        } else {
            A_tile[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        // Load B tile
        int b_row = t * TILE_SIZE + threadIdx.y;
        if (b_row < K && col < N) {
            B_tile[threadIdx.y][threadIdx.x] = static_cast<float>(B[b_row * N + col]) * scale_b;
        } else {
            B_tile[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute
        for (int k = 0; k < TILE_SIZE; k++) {
            sum += A_tile[threadIdx.y][k] * B_tile[k][threadIdx.x];
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

torch::Tensor int8_gemm_cuda(
    torch::Tensor A,
    torch::Tensor B,
    torch::Tensor scale_a,
    torch::Tensor scale_b
) {
    const int M = A.size(0);
    const int K = A.size(1);
    const int N = B.size(1);
    
    // Output in float32 first, then convert to float16
    auto C_float = torch::empty({M, N}, torch::TensorOptions().dtype(torch::kFloat32).device(A.device()));
    
    const float sa = scale_a.item<float>();
    const float sb = scale_b.item<float>();
    
    // Use optimized kernel with 32x32 tiles
    const int TILE_SIZE = 32;
    dim3 blockDim(TILE_SIZE, TILE_SIZE);
    dim3 gridDim((N + TILE_SIZE - 1) / TILE_SIZE, (M + TILE_SIZE - 1) / TILE_SIZE);
    
    optimized_gemm_kernel<<<gridDim, blockDim>>>(
        A.data_ptr<int8_t>(),
        B.data_ptr<int8_t>(),
        C_float.data_ptr<float>(),
        sa, sb, M, N, K
    );
    
    // Convert to float16
    return C_float.to(torch::kFloat16);
}
'''

cpp_source = '''
torch::Tensor int8_gemm_cuda(torch::Tensor A, torch::Tensor B, torch::Tensor scale_a, torch::Tensor scale_b);
'''

ext = load_inline(
    name='int8_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['int8_gemm_cuda'],
    verbose=True,
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
)


class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(
        self,
        a_q: torch.Tensor,
        b_q: torch.Tensor,
        scale_a: torch.Tensor,
        scale_b: torch.Tensor,
    ) -> torch.Tensor:
        return ext.int8_gemm_cuda(a_q, b_q, scale_a, scale_b)


OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-127, 127, (m, k), dtype=torch.int8, device='cuda')
    b_q = torch.randint(-127, 127, (k, n), dtype=torch.int8, device='cuda')
    scale_a = torch.tensor(0.01, dtype=torch.float32, device='cuda')
    scale_b = torch.tensor(0.01, dtype=torch.float32, device='cuda')
    return [a_q, b_q, scale_a, scale_b]


def get_init_inputs():
    return []
