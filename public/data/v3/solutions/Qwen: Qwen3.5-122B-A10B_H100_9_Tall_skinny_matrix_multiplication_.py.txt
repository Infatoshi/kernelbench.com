import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized GEMM kernel for tall-skinny matrices on H100
// A: (M, K), B: (K, N) -> C: (M, N)
// K is small (16), M and N are large (16384)

template<typename T>
__global__ void gemm_kernel(
    const T* __restrict__ A,  // (M, K)
    const T* __restrict__ B,  // (K, N)
    T* __restrict__ C,        // (M, N)
    int M, int K, int N,
    int lda, int ldb, int ldc
) {
    // Each thread computes one element of C
    // Use 2D thread indexing within block
    constexpr int TILE_M = 32;
    constexpr int TILE_N = 32;
    
    int tid = threadIdx.x;
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    
    // Thread within tile
    int tx = tid % TILE_N;
    int ty = tid / TILE_N;
    
    int row = block_m * TILE_M + ty;
    int col = block_n * TILE_N + tx;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    // K is small (16), unroll loop
    #pragma unroll
    for (int k = 0; k < K; k++) {
        sum += static_cast<float>(A[row * lda + k]) * static_cast<float>(B[k * ldb + col]);
    }
    
    C[row * ldc + col] = static_cast<T>(sum);
}

// Specialized kernel for FP16 with better memory coalescing
__global__ void gemm_kernel_fp16(
    const at::Half* __restrict__ A,
    const at::Half* __restrict__ B,
    at::Half* __restrict__ C,
    int M, int K, int N,
    int lda, int ldb, int ldc
) {
    constexpr int TILE_M = 32;
    constexpr int TILE_N = 32;
    
    int tid = threadIdx.x;
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    
    int tx = tid % TILE_N;
    int ty = tid / TILE_N;
    
    int row = block_m * TILE_M + ty;
    int col = block_n * TILE_N + tx;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    #pragma unroll
    for (int k = 0; k < K; k++) {
        sum += __half2float(A[row * lda + k]) * __half2float(B[k * ldb + col]);
    }
    
    C[row * ldc + col] = __float2half(sum);
}

// Specialized kernel for BF16
__global__ void gemm_kernel_bf16(
    const at::BFloat16* __restrict__ A,
    const at::BFloat16* __restrict__ B,
    at::BFloat16* __restrict__ C,
    int M, int K, int N,
    int lda, int ldb, int ldc
) {
    constexpr int TILE_M = 32;
    constexpr int TILE_N = 32;
    
    int tid = threadIdx.x;
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    
    int tx = tid % TILE_N;
    int ty = tid / TILE_N;
    
    int row = block_m * TILE_M + ty;
    int col = block_n * TILE_N + tx;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    #pragma unroll
    for (int k = 0; k < K; k++) {
        // Access as uint16_t and convert
        uint16_t a_raw = *((uint16_t*)&A[row * lda + k]);
        uint16_t b_raw = *((uint16_t*)&B[k * ldb + col]);
        float a_val = __bfloat162float((__nv_bfloat16)a_raw);
        float b_val = __bfloat162float((__nv_bfloat16)b_raw);
        sum += a_val * b_val;
    }
    
    *((uint16_t*)&C[row * ldc + col]) = __bfloat162raw(__float2bfloat16(sum));
}

torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    int lda = A.stride(0);
    int ldb = B.stride(0);
    int ldc = C.stride(0);
    
    // Use 256 threads per block (32x8 or 16x16 tiles)
    dim3 block(256);
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    
    if (A.dtype() == torch::kFloat32) {
        gemm_kernel<float><<<grid, block>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K, N, lda, ldb, ldc
        );
    } else if (A.dtype() == torch::kBFloat16) {
        gemm_kernel_bf16<<<grid, block>>>(
            A.data_ptr<at::BFloat16>(),
            B.data_ptr<at::BFloat16>(),
            C.data_ptr<at::BFloat16>(),
            M, K, N, lda, ldb, ldc
        );
    } else if (A.dtype() == torch::kFloat16) {
        gemm_kernel_fp16<<<grid, block>>>(
            A.data_ptr<at::Half>(),
            B.data_ptr<at::Half>(),
            C.data_ptr<at::Half>(),
            M, K, N, lda, ldb, ldc
        );
    } else {
        // Default to float32
        gemm_kernel<float><<<grid, block>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K, N, lda, ldb, ldc
        );
    }
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='gemm_ext',
    cpp_sources='torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A, B):
        return ext.gemm_forward(A, B)

def get_inputs():
    return [torch.randn(16384, 16), torch.randn(16, 16384)]

def get_init_inputs():
    return []
