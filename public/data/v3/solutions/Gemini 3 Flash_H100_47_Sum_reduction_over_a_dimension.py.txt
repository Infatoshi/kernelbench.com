import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def sum_reduction_kernel(
    input_ptr, output_ptr,
    stride_pre, stride_reduce, stride_post,
    n_reduce, n_post,
    BLOCK_SIZE_REDUCE: tl.constexpr
):
    pid = tl.program_id(0)
    
    idx_pre = pid // n_post
    idx_post = pid % n_post
    
    # Let idx_pre and idx_post be handled as triton tensors
    idx_pre = idx_pre.to(tl.int64)
    idx_post = idx_post.to(tl.int64)
    
    # Input strides are passed as scalars, Triton treats them as int64 usually but let's be safe
    s_pre = stride_pre.to(tl.int64)
    s_reduce = stride_reduce.to(tl.int64)
    s_post = stride_post.to(tl.int64)
    
    curr_input_ptr = input_ptr + idx_pre * s_pre + idx_post * s_post
    
    acc = 0.0
    for i in range(0, n_reduce, BLOCK_SIZE_REDUCE):
        offsets = i + tl.arange(0, BLOCK_SIZE_REDUCE)
        mask = offsets < n_reduce
        vals = tl.load(curr_input_ptr + offsets.to(tl.int64) * s_reduce, mask=mask, other=0.0).to(tl.float32)
        acc += tl.sum(vals, axis=0)
    
    tl.store(output_ptr + pid, acc.to(output_ptr.dtype.element_ty))

class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        shape = x.shape
        dim = self.dim if self.dim >= 0 else len(shape) + self.dim
        
        n_pre = 1
        for i in range(dim):
            n_pre *= shape[i]
        n_reduce = shape[dim]
        n_post = 1
        for i in range(dim + 1, len(shape)):
            n_post *= shape[i]
            
        x_reshaped = x.view(n_pre, n_reduce, n_post)
        stride_pre, stride_reduce, stride_post = x_reshaped.stride()
        
        output_shape = list(shape)
        output_shape[dim] = 1
        y = torch.empty(output_shape, device=x.device, dtype=x.dtype)
        
        grid = (n_pre * n_post,)
        BLOCK_SIZE_REDUCE = min(1024, triton.next_power_of_2(n_reduce))
        
        sum_reduction_kernel[grid](
            x, y,
            stride_pre, stride_reduce, stride_post,
            n_reduce, n_post,
            BLOCK_SIZE_REDUCE=BLOCK_SIZE_REDUCE
        )
        return y

def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2, device='cuda')
    return [x]
