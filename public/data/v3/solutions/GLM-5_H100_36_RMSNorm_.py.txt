cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp-level reduction using shuffle instructions
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Float kernel with vectorization
__global__ void rmsnorm_kernel_float(
    const float* __restrict__ input,
    float* __restrict__ output,
    const int num_features,
    const float eps,
    const int num_elements
) {
    const int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / warpSize;
    const int lane_id = threadIdx.x % warpSize;
    
    const int element_idx = warp_id;
    
    if (element_idx >= num_elements) return;
    
    const float* input_ptr = input + element_idx * num_features;
    float* output_ptr = output + element_idx * num_features;
    
    float sum_sq = 0.0f;
    
    // Vectorized processing for float
    const int vec_count = num_features / 4;
    const float4* vec_ptr = reinterpret_cast<const float4*>(input_ptr);
    
    for (int i = lane_id; i < vec_count; i += warpSize) {
        float4 vec = vec_ptr[i];
        sum_sq += vec.x * vec.x + vec.y * vec.y + vec.z * vec.z + vec.w * vec.w;
    }
    
    // Handle remaining elements
    for (int i = vec_count * 4 + lane_id; i < num_features; i += warpSize) {
        float val = input_ptr[i];
        sum_sq += val * val;
    }
    
    sum_sq = warp_reduce_sum(sum_sq);
    
    const float rms = __fdividef(1.0f, sqrtf(sum_sq / num_features + eps));
    const float rms_broadcast = __shfl_sync(0xffffffff, rms, 0);
    
    // Apply normalization with vectorized stores
    float4* out_vec_ptr = reinterpret_cast<float4*>(output_ptr);
    
    for (int i = lane_id; i < vec_count; i += warpSize) {
        float4 vec;
        vec.x = input_ptr[i * 4] * rms_broadcast;
        vec.y = input_ptr[i * 4 + 1] * rms_broadcast;
        vec.z = input_ptr[i * 4 + 2] * rms_broadcast;
        vec.w = input_ptr[i * 4 + 3] * rms_broadcast;
        out_vec_ptr[i] = vec;
    }
    
    for (int i = vec_count * 4 + lane_id; i < num_features; i += warpSize) {
        output_ptr[i] = input_ptr[i] * rms_broadcast;
    }
}

// Float kernel with shared memory for larger features
__global__ void rmsnorm_kernel_float_shared(
    const float* __restrict__ input,
    float* __restrict__ output,
    const int num_features,
    const float eps,
    const int num_elements
) {
    extern __shared__ float shared_sum[];
    
    const int warp_id = threadIdx.x / warpSize;
    const int lane_id = threadIdx.x % warpSize;
    const int warps_per_block = blockDim.x / warpSize;
    
    const int element_idx = blockIdx.x;
    
    if (element_idx >= num_elements) return;
    
    const float* input_ptr = input + element_idx * num_features;
    float* output_ptr = output + element_idx * num_features;
    
    float sum_sq = 0.0f;
    for (int i = threadIdx.x; i < num_features; i += blockDim.x) {
        float val = input_ptr[i];
        sum_sq += val * val;
    }
    
    sum_sq = warp_reduce_sum(sum_sq);
    
    if (lane_id == 0) {
        shared_sum[warp_id] = sum_sq;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        sum_sq = (lane_id < warps_per_block) ? shared_sum[lane_id] : 0.0f;
        sum_sq = warp_reduce_sum(sum_sq);
        
        if (lane_id == 0) {
            shared_sum[0] = __fdividef(1.0f, sqrtf(sum_sq / num_features + eps));
        }
    }
    __syncthreads();
    
    const float rms = shared_sum[0];
    
    for (int i = threadIdx.x; i < num_features; i += blockDim.x) {
        output_ptr[i] = input_ptr[i] * rms;
    }
}

// Dispatcher function
torch::Tensor rmsnorm_forward(torch::Tensor x, int64_t num_features, double eps) {
    auto output = torch::empty_like(x);
    
    auto sizes = x.sizes();
    int num_elements = 1;
    for (int i = 0; i < static_cast<int>(sizes.size()); ++i) {
        if (i != 1) num_elements *= sizes[i];
    }
    
    const int threads_per_block = 256;
    const int warps_per_block = threads_per_block / 32;
    
    auto dtype = x.scalar_type();
    
    if (dtype == torch::kFloat32) {
        if (num_features <= 128) {
            int total_warps = num_elements;
            int blocks = (total_warps + warps_per_block - 1) / warps_per_block;
            
            rmsnorm_kernel_float<<<blocks, threads_per_block>>>(
                x.data_ptr<float>(),
                output.data_ptr<float>(),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        } else {
            int shared_mem = warps_per_block * sizeof(float);
            rmsnorm_kernel_float_shared<<<num_elements, threads_per_block, shared_mem>>>(
                x.data_ptr<float>(),
                output.data_ptr<float>(),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        }
    } else if (dtype == torch::kHalf) {
        if (num_features <= 128) {
            int total_warps = num_elements;
            int blocks = (total_warps + warps_per_block - 1) / warps_per_block;
            
            rmsnorm_kernel_half<<<blocks, threads_per_block>>>(
                reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
                reinterpret_cast<half*>(output.data_ptr<at::Half>()),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        } else {
            int shared_mem = warps_per_block * sizeof(float);
            rmsnorm_kernel_half_shared<<<num_elements, threads_per_block, shared_mem>>>(
                reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
                reinterpret_cast<half*>(output.data_ptr<at::Half>()),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        }
    } else if (dtype == torch::kBFloat16) {
        if (num_features <= 128) {
            int total_warps = num_elements;
            int blocks = (total_warps + warps_per_block - 1) / warps_per_block;
            
            rmsnorm_kernel_bfloat16<<<blocks, threads_per_block>>>(
                reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
                reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        } else {
            int shared_mem = warps_per_block * sizeof(float);
            rmsnorm_kernel_bfloat16_shared<<<num_elements, threads_per_block, shared_mem>>>(
                reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
                reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
                num_features,
                static_cast<float>(eps),
                num_elements
            );
        }
    } else {
        throw std::runtime_error("Unsupported dtype for RMSNorm");
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor rmsnorm_forward(torch::Tensor x, int64_t num_features, double eps);
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rmsnorm_forward'],
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs RMS Normalization.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        """
        Initializes the RMSNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
            eps (float, optional): A small value added to the denominator to avoid division by zero. Defaults to 1e-5.
        """
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies RMS Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, *).

        Returns:
            torch.Tensor: Output tensor with RMS Normalization applied, same shape as input.
        """
        return ext.rmsnorm_forward(x, self.num_features, self.eps)

def get_inputs():
    x = torch.randn(16, 64, 256, 256)
    return [x]

def get_init_inputs():
    return [64]
