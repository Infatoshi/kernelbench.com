import os
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename scalar_t>
__global__ void dwconv2d_forward_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size,
    int in_channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    bool has_bias)
{
    // Block shape: blockDim.x = 32, blockDim.y = 8 => 256 threads
    const int TILE_H = 8;
    const int TILE_W = 32;
    
    int ow = blockIdx.x * TILE_W + threadIdx.x;
    int oh = blockIdx.y * TILE_H + threadIdx.y;
    int cb = blockIdx.z;
    
    if (cb >= batch_size * in_channels) return;
    
    int c = cb % in_channels;
    int b = cb / in_channels;
    
    scalar_t sum = 0;
    
    if (ow < out_width && oh < out_height) {
        int in_y_origin = oh * stride - padding;
        int in_x_origin = ow * stride - padding;
        
        const scalar_t* weight_c = weight + c * kernel_size * kernel_size;
        
        for (int ky = 0; ky < kernel_size; ++ky) {
            int iy = in_y_origin + ky;
            if (iy >= 0 && iy < in_height) {
                #pragma unroll
                for (int kx = 0; kx < kernel_size; ++kx) {
                    int ix = in_x_origin + kx;
                    if (ix >= 0 && ix < in_width) {
                        scalar_t val = input[b * (in_channels * in_height * in_width) + c * (in_height * in_width) + iy * in_width + ix];
                        scalar_t w = weight_c[ky * kernel_size + kx];
                        sum += val * w;
                    }
                }
            }
        }
        
        if (has_bias) sum += bias[c];
        
        output[b * (in_channels * out_height * out_width) + c * (out_height * out_width) + oh * out_width + ow] = sum;
    }
}

torch::Tensor dwconv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias,
    int stride,
    int padding)
{
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int kernel_size = weight.size(2);
    
    int out_height = (in_height + 2 * padding - kernel_size) / stride + 1;
    int out_width = (in_width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, in_channels, out_height, out_width}, input.options());
    
    const int TILE_H = 8;
    const int TILE_W = 32;
    dim3 block(TILE_W, TILE_H);
    dim3 grid((out_width + block.x - 1) / block.x, (out_height + block.y - 1) / block.y, batch_size * in_channels);
    
    bool has_bias = bias.has_value();
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "dwconv2d_forward_kernel", ([&] {
        dwconv2d_forward_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            has_bias ? bias.value().data_ptr<scalar_t>() : nullptr,
            output.data_ptr<scalar_t>(),
            batch_size, in_channels, in_height, in_width, out_height, out_width, kernel_size, stride, padding, has_bias
        );
    }));
    
    return output;
}
"""

cpp_source = r"""
torch::Tensor dwconv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias,
    int stride,
    int padding);
"""

import os
# Compile extension
ext = load_inline(
    name='dwconv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['dwconv2d_forward'],
    verbose=False,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '-Xptxas=-O3', '-use_fast_math']
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
        self.stride = stride
        self.padding = padding
        self.bias_flag = bias
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.conv2d.bias if self.bias_flag else None
        return ext.dwconv2d_forward(x.contiguous(), self.conv2d.weight.contiguous(), 
                                    b.contiguous() if b is not None else None, 
                                    self.stride, self.padding)
