import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ['fp4', 'fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_M': 64, 'BLOCK_K': 256, 'num_warps': 4, 'num_stages': 4}),
        triton.Config({'BLOCK_M': 128, 'BLOCK_K': 256, 'num_warps': 8, 'num_stages': 4}),
        triton.Config({'BLOCK_M': 64, 'BLOCK_K': 512, 'num_warps': 8, 'num_stages': 4}),
        triton.Config({'BLOCK_M': 128, 'BLOCK_K': 512, 'num_warps': 8, 'num_stages': 5}),
    ],
    key=['M', 'K'],
)
@triton.jit
def matvec_kernel(
    A_ptr, B_ptr, C_ptr,
    M, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    BLOCK_M: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    offs_m = pid * BLOCK_M + tl.arange(0, BLOCK_M)
    acc = tl.zeros((BLOCK_M,), dtype=tl.float32)

    for k0 in range(0, K, BLOCK_K):
        offs_k = k0 + tl.arange(0, BLOCK_K)
        a_ptrs = A_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak
        b_ptrs = B_ptr + offs_k * stride_bk

        a_mask = (offs_m[:, None] < M) & (offs_k[None, :] < K)
        b_mask = offs_k < K

        a = tl.load(a_ptrs, mask=a_mask, other=0.0)
        b = tl.load(b_ptrs, mask=b_mask, other=0.0)
        acc += tl.sum(a * b[None, :], axis=1)

    c_ptrs = C_ptr + offs_m * stride_cm
    c_mask = offs_m < M
    tl.store(c_ptrs, acc, mask=c_mask)


class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        assert A.is_cuda and B.is_cuda
        assert A.dim() == 2 and B.dim() == 2 and B.shape[1] == 1
        M, K = A.shape
        out = torch.empty((M,), device=A.device, dtype=torch.float32)
        grid = (triton.cdiv(M, 64),)
        matvec_kernel[grid](
            A, B, out,
            M, K,
            A.stride(0), A.stride(1),
            B.stride(0), B.stride(1),
            out.stride(0), 1,
        )
        return out[:, None].to(A.dtype if A.dtype in (torch.float16, torch.bfloat16, torch.float32) else torch.float32)


M = 256
K = 131072


def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

    return []
