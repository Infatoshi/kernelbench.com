// Vectorized FP32 softmax using float4 loads
__global__ void softmax_fp32_vec_kernel(const float* __restrict__ input, float* __restrict__ output, int dim) {
    __shared__ float s_max, s_sum;
    __shared__ float s_partial[16];
    
    int row = blockIdx.x;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    const float4* row_input4 = reinterpret_cast<const float4*>(input + row * dim);
    float4* row_output4 = reinterpret_cast<float4*>(output + row * dim);
    int dim4 = dim / 4;
    
    int lane = tid & 31;
    int warpId = tid >> 5;
    
    // Step 1: Find max value in row using vectorized loads
    float local_max = -FLT_MAX;
    for (int i = tid; i < dim4; i += blockSize) {
        float4 val4 = row_input4[i];
        local_max = fmaxf(local_max, val4.x);
        local_max = fmaxf(local_max, val4.y);
        local_max = fmaxf(local_max, val4.z);
        local_max = fmaxf(local_max, val4.w);
    }
    
    // Warp reduction for max
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, mask, 32));
    }
    
    if (lane == 0) {
        s_partial[warpId] = local_max;
    }
    __syncthreads();
    
    float max_val;
    if (warpId == 0) {
        float val = (tid < (blockSize >> 5)) ? s_partial[lane] : -FLT_MAX;
        #pragma unroll
        for (int mask = 16; mask > 0; mask >>= 1) {
            val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, mask, 32));
        }
        max_val = val;
        if (tid == 0) {
            s_max = max_val;
        }
    }
    __syncthreads();
    max_val = s_max;
    
    // Step 2: Compute sum of exp(x - max)
    float local_sum = 0.0f;
    for (int i = tid; i < dim4; i += blockSize) {
        float4 val4 = row_input4[i];
        local_sum += expf(val4.x - max_val);
        local_sum += expf(val4.y - max_val);
        local_sum += expf(val4.z - max_val);
        local_sum += expf(val4.w - max_val);
    }
    
    // Warp reduction for sum
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        local_sum += __shfl_xor_sync(0xffffffff, local_sum, mask, 32);
    }
    
    if (lane == 0) {
        s_partial[warpId] = local_sum;
    }
    __syncthreads();
    
    float sum_val;
    if (warpId == 0) {
        float val = (tid < (blockSize >> 5)) ? s_partial[lane] : 0.0f;
        #pragma unroll
        for (int mask = 16; mask > 0; mask >>= 1) {
            val += __shfl_xor_sync(0xffffffff, val, mask, 32);
        }
        sum_val = val;
        if (tid == 0) {
            s_sum = sum_val;
        }
    }
    __syncthreads();
    sum_val = s_sum;
    
    // Step 3: Compute softmax output
    float inv_sum = 1.0f / sum_val;
    for (int i = tid; i < dim4; i += blockSize) {
        float4 val4 = row_input4[i];
        float4 out4;
        out4.x = expf(val4.x - max_val) * inv_sum;
        out4.y = expf(val4.y - max_val) * inv_sum;
        out4.z = expf(val4.z - max_val) * inv_sum;
        out4.w = expf(val4.w - max_val) * inv_sum;
        row_output4[i] = out4;
    }
}

// Vectorized FP16 softmax using half2 loads
__global__ void softmax_fp16_vec_kernel(const half* __restrict__ input, half* __restrict__ output, int dim) {
    __shared__ float s_max, s_sum;
    __shared__ float s_partial[16];
    
    int row = blockIdx.x;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    // Use uint4 for 128-bit loads (8 half values)
    const uint4* row_input_u4 = reinterpret_cast<const uint4*>(input + row * dim);
    uint4* row_output_u4 = reinterpret_cast<uint4*>(output + row * dim);
    int dim8 = dim / 8;
    
    int lane = tid & 31;
    int warpId = tid >> 5;
    
    // Step 1: Find max value in row using vectorized loads
    float local_max = -FLT_MAX;
    for (int i = tid; i < dim8; i += blockSize) {
        uint4 val_u4 = row_input_u4[i];
        half2* h2 = reinterpret_cast<half2*>(&val_u4);
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float v0 = __half2float(h2[j].x);
            float v1 = __half2float(h2[j].y);
            local_max = fmaxf(local_max, v0);
            local_max = fmaxf(local_max, v1);
        }
    }
    
    // Warp reduction for max
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, mask, 32));
    }
    
    if (lane == 0) {
        s_partial[warpId] = local_max;
    }
    __syncthreads();
    
    float max_val;
    if (warpId == 0) {
        float val = (tid < (blockSize >> 5)) ? s_partial[lane] : -FLT_MAX;
        #pragma unroll
        for (int mask = 16; mask > 0; mask >>= 1) {
            val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, mask, 32));
        }
        max_val = val;
        if (tid == 0) {
            s_max = max_val;
        }
    }
    __syncthreads();
    max_val = s_max;
    
    // Step 2: Compute sum of exp(x - max)
    float local_sum = 0.0f;
    for (int i = tid; i < dim8; i += blockSize) {
        uint4 val_u4 = row_input_u4[i];
        half2* h2 = reinterpret_cast<half2*>(&val_u4);
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float v0 = __half2float(h2[j].x) - max_val;
            float v1 = __half2float(h2[j].y) - max_val;
            local_sum += expf(v0);
            local_sum += expf(v1);
        }
    }
    
    // Warp reduction for sum
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        local_sum += __shfl_xor_sync(0xffffffff, local_sum, mask, 32);
    }
    
    if (lane == 0) {
        s_partial[warpId] = local_sum;
    }
    __syncthreads();
    
    float sum_val;
    if (warpId == 0) {
        float val = (tid < (blockSize >> 5)) ? s_partial[lane] : 0.0f;
        #pragma unroll
        for (int mask = 16; mask > 0; mask >>= 1) {
            val += __shfl_xor_sync(0xffffffff, val, mask, 32);
        }
        sum_val = val;
        if (tid == 0) {
            s_sum = sum_val;
        }
    }
    __syncthreads();
    sum_val = s_sum;
    
    // Step 3: Compute softmax output
    float inv_sum = 1.0f / sum_val;
    for (int i = tid; i < dim8; i += blockSize) {
        uint4 val_u4 = row_input_u4[i];
        half2* h2_in = reinterpret_cast<half2*>(&val_u4);
        uint4 out_u4;
        half2* h2_out = reinterpret_cast<half2*>(&out_u4);
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float v0 = expf(__half2float(h2_in[j].x) - max_val) * inv_sum;
            float v1 = expf(__half2float(h2_in[j].y) - max_val) * inv_sum;
            h2_out[j] = __floats2half2_rn(v0, v1);
        }
        row_output_u4[i] = out_u4;
    }
}

// Dispatch function
torch::Tensor softmax_cuda(torch::Tensor x) {
    auto output = torch::empty_like(x);
    int batch_size = x.size(0);
    int dim = x.size(1);
    
    const int BLOCK_SIZE = 512;
    int grid = batch_size;
    int block = BLOCK_SIZE;
    
    if (x.scalar_type() == torch::kFloat32) {
        softmax_fp32_vec_kernel<<<grid, block>>>(
            x.data_ptr<float>(), output.data_ptr<float>(), dim);
    } else if (x.scalar_type() == torch::kFloat16) {
        softmax_fp16_vec_kernel<<<grid, block>>>(
            reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
            reinterpret_cast<half*>(output.data_ptr<at::Half>()), dim);
    } else if (x.scalar_type() == torch::kBFloat16) {
        softmax_bf16_vec_kernel<<<grid, block>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()), dim);
    }
    
    return output;
}
