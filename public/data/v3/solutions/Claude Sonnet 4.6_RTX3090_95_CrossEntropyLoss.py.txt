import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def cross_entropy_kernel(
    logits_ptr,       # [N, C]
    targets_ptr,      # [N]
    losses_ptr,       # [N] output per-sample losses
    N: tl.constexpr,
    C: tl.constexpr,
    BLOCK_C: tl.constexpr,
):
    # Each program handles one row
    row = tl.program_id(0)
    
    # Load target for this row
    target = tl.load(targets_ptr + row)
    
    # Step 1: Find max for numerical stability
    max_val = -float('inf')
    for block_start in range(0, C, BLOCK_C):
        offsets = block_start + tl.arange(0, BLOCK_C)
        mask = offsets < C
        vals = tl.load(logits_ptr + row * C + offsets, mask=mask, other=-float('inf'))
        max_val = tl.maximum(max_val, tl.max(vals, axis=0))
    
    # Step 2: Compute sum of exp(x - max) and get target logit
    sum_exp = 0.0
    target_val = 0.0
    for block_start in range(0, C, BLOCK_C):
        offsets = block_start + tl.arange(0, BLOCK_C)
        mask = offsets < C
        vals = tl.load(logits_ptr + row * C + offsets, mask=mask, other=0.0)
        exp_vals = tl.exp(vals - max_val)
        exp_vals = tl.where(mask, exp_vals, 0.0)
        sum_exp += tl.sum(exp_vals, axis=0)
        # Get target value
        target_mask = mask & (offsets == target)
        target_val += tl.sum(tl.where(target_mask, vals, 0.0), axis=0)
    
    # CE loss = -target_val + log(sum_exp) + max_val
    loss = -target_val + tl.log(sum_exp) + max_val
    tl.store(losses_ptr + row, loss)


@triton.jit
def cross_entropy_kernel_fast(
    logits_ptr,       # [N, C]
    targets_ptr,      # [N]
    losses_ptr,       # [N] output per-sample losses
    N,
    C,
    stride_n,
    BLOCK_C: tl.constexpr,
):
    # Each program handles one row
    row = tl.program_id(0)
    
    # Load target for this row
    target = tl.load(targets_ptr + row)
    
    row_start = row * stride_n
    
    # Step 1: Find max for numerical stability
    max_val = -float('inf')
    for block_start in range(0, C, BLOCK_C):
        offsets = block_start + tl.arange(0, BLOCK_C)
        mask = offsets < C
        vals = tl.load(logits_ptr + row_start + offsets, mask=mask, other=-float('inf'))
        block_max = tl.max(vals, axis=0)
        max_val = tl.maximum(max_val, block_max)
    
    # Step 2: Compute sum of exp(x - max) and get target logit
    sum_exp = 0.0
    target_val = 0.0
    for block_start in range(0, C, BLOCK_C):
        offsets = block_start + tl.arange(0, BLOCK_C)
        mask = offsets < C
        vals = tl.load(logits_ptr + row_start + offsets, mask=mask, other=0.0)
        exp_vals = tl.exp(vals - max_val)
        exp_vals = tl.where(mask, exp_vals, 0.0)
        sum_exp += tl.sum(exp_vals, axis=0)
        # Get target value
        target_mask = mask & (offsets == target)
        target_val += tl.sum(tl.where(target_mask, vals, 0.0), axis=0)
    
    # CE loss = -target_val + log(sum_exp) + max_val
    loss = -target_val + tl.log(sum_exp) + max_val
    tl.store(losses_ptr + row, loss)


def cross_entropy_triton(logits: torch.Tensor, targets: torch.Tensor) -> torch.Tensor:
    N, C = logits.shape
    losses = torch.empty(N, device=logits.device, dtype=torch.float32)
    
    # Choose block size: next power of 2 >= C, capped at 1024
    BLOCK_C = min(triton.next_power_of_2(C), 1024)
    
    grid = (N,)
    cross_entropy_kernel_fast[grid](
        logits, targets, losses,
        N, C, logits.stride(0),
        BLOCK_C=BLOCK_C,
    )
    
    return losses.mean()


class Model(nn.Module):
    """
    A model that computes Cross Entropy Loss for multi-class classification tasks.
    Optimized Triton implementation.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        if predictions.dtype != torch.float32:
            predictions = predictions.float()
        predictions = predictions.contiguous()
        targets = targets.contiguous()
        return cross_entropy_triton(predictions, targets)
