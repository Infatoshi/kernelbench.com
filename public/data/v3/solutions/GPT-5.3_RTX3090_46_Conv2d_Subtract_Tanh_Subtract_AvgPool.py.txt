import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N,int C,int H,int W,
    int OC,int K,
    float sub1,float sub2,
    int OH,int OW,
    int PY,int PX
){
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*OC*PY*PX;
    if(idx>=total) return;

    int px = idx % PX; idx/=PX;
    int py = idx % PY; idx/=PY;
    int oc = idx % OC; idx/=OC;
    int n = idx;

    float acc_pool = 0.0f;

    #pragma unroll
    for(int iy=0; iy<2; iy++){
        #pragma unroll
        for(int ix=0; ix<2; ix++){
            int oy = py*2 + iy;
            int ox = px*2 + ix;

            float sum = b ? b[oc] : 0.0f;

            for(int ic=0; ic<C; ic++){
                for(int ky=0; ky<K; ky++){
                    for(int kx=0; kx<K; kx++){
                        int in_y = oy + ky;
                        int in_x = ox + kx;
                        int x_idx = ((n*C + ic)*H + in_y)*W + in_x;
                        int w_idx = ((oc*C + ic)*K + ky)*K + kx;
                        sum += x[x_idx] * w[w_idx];
                    }
                }
            }

            float v = sum - sub1;
            v = tanhf(v);
            v = v - sub2;
            acc_pool += v;
        }
    }

    out[((n*OC + oc)*PY + py)*PX + px] = acc_pool * 0.25f;
}

torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor b,
    float sub1,
    float sub2
){
    int N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3);
    int OC=w.size(0), K=w.size(2);

    int OH = H - K + 1;
    int OW = W - K + 1;
    int PY = OH/2;
    int PX = OW/2;

    auto out = torch::empty({N,OC,PY,PX}, x.options());

    int total = N*OC*PY*PX;
    int threads = 256;
    int blocks = (total + threads - 1)/threads;

    fused_kernel<<<blocks,threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.numel()? b.data_ptr<float>() : nullptr,
        out.data_ptr<float>(),
        N,C,H,W,OC,K,sub1,sub2,OH,OW,PY,PX
    );

    return out;
}
'''

ext = load_inline(
    name='fused_conv_tanh_pool',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, float sub1, float sub2);',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)

    def forward(self, x):
        w = self.conv.weight.contiguous()
        b = self.conv.bias.contiguous() if self.conv.bias is not None else torch.empty(0, device=x.device)
        return ext.fused_forward(x.contiguous(), w, b, self.subtract1_value, self.subtract2_value)

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width, device='cuda')]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
