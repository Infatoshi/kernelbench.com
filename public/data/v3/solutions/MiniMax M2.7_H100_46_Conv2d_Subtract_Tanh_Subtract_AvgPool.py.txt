import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Optimized fused kernel with proper unrolling and memory coalescing
template <typename T, int BLOCK_SIZE = 256>
__global__ void fused_subtract_tanh_subtract_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    float subtract1,
    float subtract2,
    int n
) {
    int idx = blockIdx.x * BLOCK_SIZE + threadIdx.x;
    int stride = BLOCK_SIZE * gridDim.x;
    
    // Process multiple elements per thread for better memory bandwidth utilization
    #pragma unroll
    for (int i = 0; i < 4 && idx + i * stride < n; ++i) {
        int local_idx = idx + i * stride;
        float x = static_cast<float>(input[local_idx]) - subtract1;
        float tanh_val = tanhf(x);
        output[local_idx] = static_cast<T>(tanh_val - subtract2);
    }
}

torch::Tensor fused_ops(torch::Tensor input, float subtract1, float subtract2) {
    auto output = torch::empty_like(input);
    
    int n = input.numel();
    int block_size = 256;
    int num_blocks = min((n + block_size * 4 - 1) / (block_size * 4), 65536);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_ops", ([&] {
        fused_subtract_tanh_subtract_kernel<scalar_t, 256><<<num_blocks, block_size>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            subtract1,
            subtract2,
            n
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_ops(torch::Tensor input, float subtract1, float subtract2);',
    cuda_sources=cuda_source,
    functions=['fused_ops'],
    verbose=False,
)

class Model(nn.Module):
    """
    Fused model: Conv2d -> subtract -> tanh -> subtract -> AvgPool2d
    Optimized with custom CUDA kernel for element-wise operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=0)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.avgpool = nn.AvgPool2d(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_ops(x, self.subtract1_value, self.subtract2_value)
        x = self.avgpool(x)
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    subtract1_value = 0.5
    subtract2_value = 0.2
    kernel_size_pool = 2
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
