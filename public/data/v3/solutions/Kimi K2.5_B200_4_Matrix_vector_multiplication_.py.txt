import torch
import torch.nn as nn
import triton
import triton.language as tl

# Matrix dimensions from reference
M = 256
K = 131072


@triton.jit
def gemv_kernel_fp16(
    A_ptr, B_ptr, C_ptr,
    M, K,
    stride_am, stride_ak,
    BLOCK_SIZE_K: tl.constexpr,
):
    """
    Matrix-vector multiplication: C = A @ B
    - A: (M, K) matrix
    - B: (K,) vector  
    - C: (M,) vector
    
    Each block computes one row of the output.
    Uses FP16 for computation to leverage B200 tcgen05.mma (2x Hopper throughput).
    """
    # Each block handles one row
    row_idx = tl.program_id(0)
    
    # Accumulator in FP32 for numerical stability
    acc = tl.zeros((), dtype=tl.float32)
    
    # Iterate over K dimension in tiles
    for k_start in range(0, K, BLOCK_SIZE_K):
        # Offsets for this tile
        offs_k = k_start + tl.arange(0, BLOCK_SIZE_K)
        mask = offs_k < K
        
        # Load A[row_idx, k_start:k_start+BLOCK_SIZE_K] as FP16
        # Address calculation: row start + column offset
        a_ptrs = A_ptr + row_idx * stride_am + offs_k
        a = tl.load(a_ptrs, mask=mask, other=0.0).to(tl.float16)
        
        # Load B[k_start:k_start+BLOCK_SIZE_K] as FP16
        b_ptrs = B_ptr + offs_k
        b = tl.load(b_ptrs, mask=mask, other=0.0).to(tl.float16)
        
        # Multiply and accumulate
        # Convert to FP32 before multiply to match reference precision closely
        prod = a.to(tl.float32) * b.to(tl.float32)
        acc += tl.sum(prod, axis=0)
    
    # Store output
    tl.store(C_ptr + row_idx, acc)


class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    Optimized for B200 Blackwell using FP16 Tensor Cores (tcgen05.mma).
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        M, K = A.shape
        
        # Reshape B to 1D
        if B.dim() == 2:
            B = B.squeeze(-1)
        
        # Prepare output tensor (FP32 as per reference expectation)
        C = torch.empty(M, dtype=torch.float32, device=A.device)
        
        # Convert to FP16 for B200 Tensor Core acceleration
        # B200 tcgen05.mma.kind::f16 provides 2x throughput over Hopper
        A_fp16 = A.to(torch.float16)
        B_fp16 = B.to(torch.float16)
        
        # Launch one block per row (M blocks total)
        # Each block reduces over K dimension
        grid = (M,)
        
        gemv_kernel_fp16[grid](
            A_fp16, B_fp16, C,
            M, K,
            A_fp16.stride(0), A_fp16.stride(1),
            BLOCK_SIZE_K=512,  # Process 512 elements per iteration to maximize bandwidth
        )
        
        # Reshape to (M, 1)
        return C.unsqueeze(-1)


def get_inputs():
    """Return sample inputs matching reference"""
    A = torch.randn(M, K, device='cuda')
    B = torch.randn(K, 1, device='cuda')
    return [A, B]


def get_init_inputs():
    """Return initialization inputs (empty for this model)"""
    return []
