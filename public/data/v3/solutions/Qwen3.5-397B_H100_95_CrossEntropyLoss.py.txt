import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// CrossEntropyLoss kernel for FP32
__global__ void cross_entropy_loss_fp32_kernel(
    const float* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const float* x = predictions + batch_idx * num_classes;
    int target = targets[batch_idx];
    
    // Find max for numerical stability
    float max_val = x[0];
    for (int i = 1; i < num_classes; i++) {
        max_val = fmaxf(max_val, x[i]);
    }
    
    // Compute sum of exp(x - max)
    float sum_exp = 0.0f;
    for (int i = 0; i < num_classes; i++) {
        sum_exp += expf(x[i] - max_val);
    }
    
    // Compute loss: -x[target] + max + log(sum_exp)
    float loss = -x[target] + max_val + logf(sum_exp);
    losses[batch_idx] = loss;
}

// Optimized kernel with shared memory reduction for larger num_classes
__global__ void cross_entropy_loss_fp32_optimized_kernel(
    const float* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    extern __shared__ float sdata[];
    float* max_vals = sdata;
    float* sum_vals = sdata + blockDim.x;
    
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const float* x = predictions + batch_idx * num_classes;
    int target = targets[batch_idx];
    int tid = threadIdx.x;
    
    // Parallel max reduction
    float local_max = -FLT_MAX;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        local_max = fmaxf(local_max, x[i]);
    }
    max_vals[tid] = local_max;
    __syncthreads();
    
    // Reduce max
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            max_vals[tid] = fmaxf(max_vals[tid], max_vals[tid + stride]);
        }
        __syncthreads();
    }
    float max_val = max_vals[0];
    __syncthreads();
    
    // Parallel exp sum reduction
    float local_sum = 0.0f;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        local_sum += expf(x[i] - max_val);
    }
    sum_vals[tid] = local_sum;
    __syncthreads();
    
    // Reduce sum
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            sum_vals[tid] += sum_vals[tid + stride];
        }
        __syncthreads();
    }
    float sum_exp = sum_vals[0];
    
    // Get target value
    float target_val = x[target];
    
    // Compute loss
    if (tid == 0) {
        losses[batch_idx] = -target_val + max_val + logf(sum_exp);
    }
}

// FP16 kernel
__global__ void cross_entropy_loss_fp16_kernel(
    const half* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const half* x = predictions + batch_idx * num_classes;
    int target = targets[batch_idx];
    
    // Find max for numerical stability
    half max_val = x[0];
    for (int i = 1; i < num_classes; i++) {
        max_val = __hmax(max_val, x[i]);
    }
    
    // Compute sum of exp(x - max)
    float sum_exp = 0.0f;
    for (int i = 0; i < num_classes; i++) {
        float diff = __half2float(x[i]) - __half2float(max_val);
        sum_exp += expf(diff);
    }
    
    // Compute loss
    float target_val = __half2float(x[target]);
    float loss = -target_val + __half2float(max_val) + logf(sum_exp);
    losses[batch_idx] = loss;
}

// BF16 kernel
__global__ void cross_entropy_loss_bf16_kernel(
    const __nv_bfloat16* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const __nv_bfloat16* x = predictions + batch_idx * num_classes;
    int target = targets[batch_idx];
    
    // Find max for numerical stability
    __nv_bfloat16 max_val = x[0];
    for (int i = 1; i < num_classes; i++) {
        max_val = __hmax(max_val, x[i]);
    }
    
    // Compute sum of exp(x - max)
    float sum_exp = 0.0f;
    for (int i = 0; i < num_classes; i++) {
        float diff = __bfloat162float(x[i]) - __bfloat162float(max_val);
        sum_exp += expf(diff);
    }
    
    // Compute loss
    float target_val = __bfloat162float(x[target]);
    float loss = -target_val + __bfloat162float(max_val) + logf(sum_exp);
    losses[batch_idx] = loss;
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    auto losses = torch::empty({batch_size}, predictions.options().dtype(torch::kFloat32));
    
    if (predictions.dtype() == torch::kFloat32) {
        // Use optimized kernel with shared memory
        int threads = 256;
        int shared_mem = 2 * threads * sizeof(float);
        cross_entropy_loss_fp32_optimized_kernel<<<batch_size, threads, shared_mem>>>(
            predictions.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kFloat16) {
        cross_entropy_loss_fp16_kernel<<<batch_size, 256>>>(
            reinterpret_cast<const half*>(predictions.data_ptr<c10::Half>()),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kBFloat16) {
        cross_entropy_loss_bf16_kernel<<<batch_size, 256>>>(
            reinterpret_cast<const __nv_bfloat16*>(predictions.data_ptr<c10::BFloat16>()),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }
    
    // Mean of losses to match F.cross_entropy behavior with reduction='mean'
    return losses.mean();
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_loss(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    input_shape = (num_classes,)
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

def get_init_inputs():
    return []
