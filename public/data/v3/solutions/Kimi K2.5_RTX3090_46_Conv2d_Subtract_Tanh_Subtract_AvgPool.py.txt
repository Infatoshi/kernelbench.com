import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

#define IN_C 3
#define OUT_C 16
#define KH 3
#define KW 3

using at::Half;

__global__ void __launch_bounds__(512) fused_kernel(
    const Half* __restrict__ input,
    const Half* __restrict__ weight,
    Half* __restrict__ output,
    int batch, int in_h, int in_w, int out_h, int out_w,
    float sub1, float sub2) {
    
    const int ox = blockIdx.x * 16 + threadIdx.x;
    const int oy = blockIdx.y * 8 + threadIdx.y;
    const int oc = (blockIdx.z % 4) * 4 + threadIdx.z;
    const int b = blockIdx.z / 4;
    
    if (ox >= out_w || oy >= out_h || oc >= OUT_C || b >= batch) return;
    
    __shared__ Half tile[IN_C][18][34];
    __shared__ Half wgt[4][IN_C][3][3];
    
    const int tid = threadIdx.z * 64 + threadIdx.y * 16 + threadIdx.x;
    const int nthreads = 512;
    
    int tile_y = blockIdx.y * 16;
    int tile_x = blockIdx.x * 32;
    
    for (int i = tid; i < IN_C * 18 * 34; i += nthreads) {
        int c = i / (18 * 34);
        int y = (i % (18 * 34)) / 34;
        int x = i % 34;
        int iy = tile_y + y;
        int ix = tile_x + x;
        
        Half val = Half(0.0f);
        if (iy < in_h && ix < in_w) {
            val = input[((b * IN_C + c) * in_h + iy) * in_w + ix];
        }
        tile[c][y][x] = val;
    }
    
    for (int i = tid; i < 4 * IN_C * 9; i += nthreads) {
        int oc_l = i / (IN_C * 9);
        int rem = i % (IN_C * 9);
        int ic = rem / 9;
        int kh = (rem % 9) / 3;
        int kw = rem % 3;
        int oc_g = (blockIdx.z % 4) * 4 + oc_l;
        wgt[oc_l][ic][kh][kw] = weight[((oc_g * IN_C + ic) * 3 + kh) * 3 + kw];
    }
    
    __syncthreads();
    
    int ly = threadIdx.y * 2;
    int lx = threadIdx.x * 2;
    int ocl = threadIdx.z;
    
    float c00 = 0, c01 = 0, c10 = 0, c11 = 0;
    
    #pragma unroll
    for (int ic = 0; ic < IN_C; ic++) {
        #pragma unroll
        for (int kh = 0; kh < 3; kh++) {
            #pragma unroll
            for (int kw = 0; kw < 3; kw++) {
                float w = float(wgt[ocl][ic][kh][kw]);
                c00 += float(tile[ic][ly + kh][lx + kw]) * w;
                c01 += float(tile[ic][ly + kh][lx + kw + 1]) * w;
                c10 += float(tile[ic][ly + kh + 1][lx + kw]) * w;
                c11 += float(tile[ic][ly + kh + 1][lx + kw + 1]) * w;
            }
        }
    }
    
    c00 = tanhf(c00 - sub1) - sub2;
    c01 = tanhf(c01 - sub1) - sub2;
    c10 = tanhf(c10 - sub1) - sub2;
    c11 = tanhf(c11 - sub1) - sub2;
    
    float avg = (c00 + c01 + c10 + c11) * 0.25f;
    
    output[((b * OUT_C + oc) * out_h + oy) * out_w + ox] = Half(avg);
}

torch::Tensor forward_cuda(torch::Tensor input, torch::Tensor weight, float s1, float s2) {
    int batch = input.size(0);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_h = (in_h - 2) / 2;
    int out_w = (in_w - 2) / 2;
    
    auto output = torch::empty({batch, OUT_C, out_h, out_w}, 
                               torch::TensorOptions().dtype(torch::kFloat16).device(input.device()));
    
    dim3 block(16, 8, 4);
    dim3 grid((out_w + 15) / 16, (out_h + 7) / 8, 4 * batch);
    
    fused_kernel<<<grid, block>>>(
        input.data_ptr<Half>(), weight.data_ptr<Half>(), output.data_ptr<Half>(),
        batch, in_h, in_w, out_h, out_w, s1, s2);
    
    return output;
}
'''

cpp_source = 'torch::Tensor forward_cuda(torch::Tensor input, torch::Tensor weight, float s1, float s2);'

ext = load_inline(
    name='cnv_46_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['forward_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        
        assert in_channels == 3, "in_channels must be 3"
        assert out_channels == 16, "out_channels must be 16"
        assert kernel_size == 3, "kernel_size must be 3"
        assert kernel_size_pool == 2, "kernel_size_pool must be 2"
        
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        self.sub1 = float(subtract1_value)
        self.sub2 = float(subtract2_value)
        
    def forward(self, x):
        x16 = x.half()
        w16 = self.weight.half()
        out16 = ext.forward_cuda(x16, w16, self.sub1, self.sub2)
        return out16.to(x.dtype)


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    subtract1_value = 0.5
    subtract2_value = 0.2
    kernel_size_pool = 2
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
