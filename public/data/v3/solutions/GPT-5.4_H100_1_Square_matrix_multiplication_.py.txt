import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# CUTLASS-based GEMM using Hopper tensor cores when available.
# We expose a custom op and keep the Model/get_inputs/get_init_inputs API
# compatible with reference.py.

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor gemm(torch::Tensor A, torch::Tensor B);
'''

cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <vector>

#include <cutlass/cutlass.h>
#include <cutlass/gemm/device/gemm.h>
#include <cutlass/half.h>
#include <cutlass/bfloat16.h>
#include <cutlass/layout/matrix.h>

namespace {

template <typename T>
struct CutlassType;

template <>
struct CutlassType<at::Half> { using type = cutlass::half_t; };

template <>
struct CutlassType<at::BFloat16> { using type = cutlass::bfloat16_t; };

template <>
struct CutlassType<float> { using type = float; };

template <typename scalar_t>
torch::Tensor run_gemm(torch::Tensor A, torch::Tensor B) {
    using ElementInput = typename CutlassType<scalar_t>::type;
    using ElementOutput = typename CutlassType<scalar_t>::type;
    using ElementAccumulator = typename std::conditional<std::is_same<scalar_t, float>::value, float, float>::type;

    auto A_c = A.contiguous();
    auto B_c = B.contiguous();

    int64_t M = A_c.size(0);
    int64_t K = A_c.size(1);
    int64_t N = B_c.size(1);

    auto C = torch::empty({M, N}, A_c.options());

    using LayoutInputA = cutlass::layout::RowMajor;
    using LayoutInputB = cutlass::layout::RowMajor;
    using LayoutOutput = cutlass::layout::RowMajor;

    using Gemm = cutlass::gemm::device::Gemm<
        ElementInput,
        LayoutInputA,
        ElementInput,
        LayoutInputB,
        ElementOutput,
        LayoutOutput,
        ElementAccumulator
    >;

    typename Gemm::Arguments args(
        {static_cast<int>(M), static_cast<int>(N), static_cast<int>(K)},
        {reinterpret_cast<ElementInput const*>(A_c.data_ptr<scalar_t>()), static_cast<int>(K)},
        {reinterpret_cast<ElementInput const*>(B_c.data_ptr<scalar_t>()), static_cast<int>(N)},
        {reinterpret_cast<ElementOutput const*>(C.data_ptr<scalar_t>()), static_cast<int>(N)},
        {reinterpret_cast<ElementOutput*>(C.data_ptr<scalar_t>()), static_cast<int>(N)},
        {ElementAccumulator(1), ElementAccumulator(0)}
    );

    Gemm gemm_op;
    cutlass::Status status = gemm_op(args, at::cuda::getDefaultCUDAStream());
    TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS GEMM failed");
    return C;
}

} // namespace

torch::Tensor gemm(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "A and B must be CUDA tensors");
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "A and B must be 2D");
    TORCH_CHECK(A.size(1) == B.size(0), "Inner dimensions must match");
    TORCH_CHECK(A.scalar_type() == B.scalar_type(), "A and B dtypes must match");

    switch (A.scalar_type()) {
        case at::kHalf:
            return run_gemm<at::Half>(A, B);
        case at::kBFloat16:
            return run_gemm<at::BFloat16>(A, B);
        case at::kFloat:
            return run_gemm<float>(A, B);
        default:
            TORCH_CHECK(false, "Unsupported dtype");
    }
}
'''

ext = load_inline(
    name='h100_cutlass_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm'],
    extra_cuda_cflags=['-O3', '-std=c++17', '-arch=sm_90', '-I/opt/cutlass/include'],
    extra_cflags=['-O3', '-std=c++17'],
    verbose=False,
)

HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']
N = 2048
OP_TYPE = 'gemm'
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemm(A, B)

def get_inputs():
    A = torch.randn(N, N)
    B = torch.randn(N, N)
    return [A, B]

def get_init_inputs():
    return []
