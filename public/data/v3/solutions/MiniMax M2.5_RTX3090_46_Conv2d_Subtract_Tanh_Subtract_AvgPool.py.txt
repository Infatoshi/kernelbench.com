import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Tanh activation function
__device__ __forceinline__ float tanhActivation(float x) {
    return tanhf(x);
}

// Fused kernel: subtract -> tanh -> subtract -> avgpool
// This fuses all element-wise operations after convolution into a single kernel
template <typename T>
__global__ void fused_ops_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const float subtract1_value,
    const float subtract2_value,
    const int batch_size,
    const int channels,
    const int in_height,
    const int in_width,
    const int out_height,
    const int out_width,
    const int pool_kernel_size
) {
    // Each thread handles one output element
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total_out = batch_size * channels * out_height * out_width;
    
    if (out_idx >= total_out) return;
    
    // Decode output index
    int remainder = out_idx;
    const int b = remainder / (channels * out_height * out_width);
    remainder %= (channels * out_height * out_width);
    const int c = remainder / (out_height * out_width);
    remainder %= (out_height * out_width);
    const int oh = remainder / out_width;
    const int ow = remainder % out_width;
    
    // Calculate corresponding input position for average pooling
    const int ih_start = oh * pool_kernel_size;
    const int iw_start = ow * pool_kernel_size;
    
    // Accumulate for average pooling
    float sum = 0.0f;
    
    // Unroll the pooling loop for better performance
    #pragma unroll
    for (int ph = 0; ph < pool_kernel_size; ph++) {
        #pragma unroll
        for (int pw = 0; pw < pool_kernel_size; pw++) {
            const int ih = ih_start + ph;
            const int iw = iw_start + pw;
            
            // Calculate input index
            const int in_idx = ((b * channels + c) * in_height + ih) * in_width + iw;
            
            // Fused operations: subtract1 -> tanh -> subtract2
            float val = static_cast<float>(input[in_idx]);
            val = val - subtract1_value;
            val = tanhActivation(val);
            val = val - subtract2_value;
            
            sum += val;
        }
    }
    
    // Average pooling
    const float pool_size = static_cast<float>(pool_kernel_size * pool_kernel_size);
    output[out_idx] = static_cast<T>(sum / pool_size);
}

// Optimized version using vectorized loads and shared memory
template <typename T>
__global__ void fused_ops_kernel_optimized(
    const T* __restrict__ input,
    T* __restrict__ output,
    const float subtract1_value,
    const float subtract2_value,
    const int batch_size,
    const int channels,
    const int in_height,
    const int in_width,
    const int out_height,
    const int out_width,
    const int pool_kernel_size
) {
    extern __shared__ float sdata[];
    
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    // Each block processes one output element, with threads pooling together
    const int b = blockIdx.z % batch_size;
    const int c = (blockIdx.z / batch_size) % channels;
    const int oh = blockIdx.y;
    const int ow = blockIdx.x;
    
    if (oh >= out_height || ow >= out_width) return;
    
    // Calculate corresponding input position for average pooling
    const int ih_start = oh * pool_kernel_size;
    const int iw_start = ow * pool_kernel_size;
    
    // Each thread loads one input element
    float val = 0.0f;
    const int ph = ty;
    const int pw = tx;
    
    if (ph < pool_kernel_size && pw < pool_kernel_size) {
        const int ih = ih_start + ph;
        const int iw = iw_start + pw;
        
        if (ih < in_height && iw < in_width) {
            const int in_idx = ((b * channels + c) * in_height + ih) * in_width + iw;
            val = static_cast<float>(input[in_idx]);
            val = val - subtract1_value;
            val = tanhActivation(val);
            val = val - subtract2_value;
        }
    }
    
    // Store in shared memory
    sdata[ty * pool_kernel_size + tx] = val;
    __syncthreads();
    
    // Parallel reduction for average
    if (pool_kernel_size == 2) {
        if (ph == 0 && pw == 0) {
            float sum = sdata[0] + sdata[1] + sdata[2] + sdata[3];
            const int out_idx = ((b * channels + c) * out_height + oh) * out_width + ow;
            output[out_idx] = static_cast<T>(sum / 4.0f);
        }
    }
}

torch::Tensor fused_ops(torch::Tensor input, float subtract1_value, float subtract2_value, 
                        int pool_kernel_size, int out_height, int out_width) {
    auto batch_size = input.size(0);
    auto channels = input.size(1);
    auto in_height = input.size(2);
    auto in_width = input.size(3);
    
    // Create output tensor
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    const int total_out = batch_size * channels * out_height * out_width;
    
    const int block_size = 256;
    const int grid_size = (total_out + block_size - 1) / block_size;
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_ops", ([&] {
        fused_ops_kernel<scalar_t><<<grid_size, block_size>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            subtract1_value,
            subtract2_value,
            batch_size,
            channels,
            in_height,
            in_width,
            out_height,
            out_width,
            pool_kernel_size
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='fused_ops_ext',
    cpp_sources='torch::Tensor fused_ops(torch::Tensor input, float subtract1_value, float subtract2_value, int pool_kernel_size, int out_height, int out_width);',
    cuda_sources=cuda_source,
    functions=['fused_ops'],
    verbose=False,
)

class Model(nn.Module):
    """
    Model that performs a convolution, subtraction, tanh activation, subtraction and average pooling.
    Uses a fused CUDA kernel for element-wise operations after convolution.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.avgpool = nn.AvgPool2d(kernel_size_pool)
        self.kernel_size_pool = kernel_size_pool
        
    def forward(self, x):
        # Use PyTorch's optimized conv (cuDNN)
        x = self.conv(x)
        
        # Get output size for avgpool
        batch_size = x.size(0)
        channels = x.size(1)
        in_height = x.size(2)
        in_width = x.size(3)
        
        out_height = in_height // self.kernel_size_pool
        out_width = in_width // self.kernel_size_pool
        
        # Use fused CUDA kernel for element-wise operations
        x = ext.fused_ops(x, self.subtract1_value, self.subtract2_value, 
                         self.kernel_size_pool, out_height, out_width)
        return x

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
