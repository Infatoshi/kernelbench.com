import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cutlass/cutlass.h>
#include <cutlass/gemm/device/gemm.h>
#include <cutlass/util/host_tensor.h>

// Define CUTLASS GEMM types for different precisions
using GemmF32 = cutlass::gemm::device::Gemm<
    float,                           // ElementA
    cutlass::layout::RowMajor,       // LayoutA
    float,                           // ElementB
    cutlass::layout::RowMajor,       // LayoutB
    float,                           // ElementC
    cutlass::layout::RowMajor,       // LayoutC
    float,                           // ElementAccumulator
    cutlass::arch::OpClassTensorOp,  // OpClass
    cutlass::arch::Sm90,             // ArchTag
    cutlass::gemm::GemmShape<128, 128, 32>,  // ThreadblockShape
    cutlass::gemm::GemmShape<64, 64, 32>,    // WarpShape
    cutlass::gemm::GemmShape<1, 1, 1>,       // InstructionShape
    cutlass::epilogue::thread::LinearCombination<
        float,                       
        1,                           
        float,                       
        float                        
    >,                               
    cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<>, 
    3                                 // Stages
>;

using GemmF16 = cutlass::gemm::device::Gemm<
    cutlass::half_t,
    cutlass::layout::RowMajor,
    cutlass::half_t,
    cutlass::layout::RowMajor,
    cutlass::half_t,
    cutlass::layout::RowMajor,
    float,
    cutlass::arch::OpClassTensorOp,
    cutlass::arch::Sm90,
    cutlass::gemm::GemmShape<128, 128, 64>,
    cutlass::gemm::GemmShape<64, 64, 64>,
    cutlass::gemm::GemmShape<16, 8, 16>,
    cutlass::epilogue::thread::LinearCombination<
        cutlass::half_t,
        128 / cutlass::sizeof_bits<cutlass::half_t>::value,
        float,
        float
    >,
    cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<>,
    3
>;

torch::Tensor gemm_f32(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    GemmF32 gemm_op;
    
    cutlass::Status status = gemm_op({
        {M, N, K},
        {A.data_ptr<float>(), K},
        {B.data_ptr<float>(), N},
        {C.data_ptr<float>(), N},
        {C.data_ptr<float>(), N},
        {alpha, beta}
    });
    
    TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS GEMM failed");
    
    return C;
}

torch::Tensor gemm_f16(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    GemmF16 gemm_op;
    
    cutlass::Status status = gemm_op({
        {M, N, K},
        {reinterpret_cast<cutlass::half_t*>(A.data_ptr<at::Half>()), K},
        {reinterpret_cast<cutlass::half_t*>(B.data_ptr<at::Half>()), N},
        {reinterpret_cast<cutlass::half_t*>(C.data_ptr<at::Half>()), N},
        {reinterpret_cast<cutlass::half_t*>(C.data_ptr<at::Half>()), N},
        {alpha, beta}
    });
    
    TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS GEMM failed");
    
    return C;
}

torch::Tensor matmul_cutlass(torch::Tensor A, torch::Tensor B) {
    // Ensure contiguous
    A = A.contiguous();
    B = B.contiguous();
    
    if (A.scalar_type() == torch::kFloat32) {
        return gemm_f32(A, B);
    } else if (A.scalar_type() == torch::kFloat16) {
        return gemm_f16(A, B);
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }
}
'''

ext = load_inline(
    name='cutlass_gemm',
    cpp_sources='torch::Tensor matmul_cutlass(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matmul_cutlass'],
    extra_cuda_cflags=['-arch=sm_90', '-I/opt/cutlass/include', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs a single square matrix multiplication (C = A * B)
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs the matrix multiplication.

        Args:
            A (torch.Tensor): Input matrix A of shape (N, N).
            B (torch.Tensor): Input matrix B of shape (N, N).

        Returns:
            torch.Tensor: Output matrix C of shape (N, N).
        """
        return ext.matmul_cutlass(A, B)


N = 2048
OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp16", "bf16", "fp32"]


def get_inputs():
    A = torch.randn(N, N)
    B = torch.randn(N, N)
    return [A, B]


def get_init_inputs():
    return []  # No special initialization inputs needed
