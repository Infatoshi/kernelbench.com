
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename scalar_t>
__global__ void dw_conv_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size, int channels, int in_h, int in_w,
    int kernel_size, int stride, int padding,
    int out_h, int out_w) {

    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int bc = blockIdx.z; // batch * channels

    if (ow < out_w && oh < out_h) {
        int b = bc / channels;
        int c = bc % channels;

        float sum = 0.0f;
        int in_base = (b * channels + c) * in_h * in_w;
        int weight_base = c * kernel_size * kernel_size;

        for (int kh = 0; kh < kernel_size; ++kh) {
            int ih = oh * stride - padding + kh;
            if (ih >= 0 && ih < in_h) {
                for (int kw = 0; kw < kernel_size; ++kw) {
                    int iw = ow * stride - padding + kw;
                    if (iw >= 0 && iw < in_w) {
                        sum += (float)input[in_base + ih * in_w + iw] * (float)weight[weight_base + kh * kernel_size + kw];
                    }
                }
            }
        }
        if (bias != nullptr) sum += (float)bias[c];
        output[bc * out_h * out_w + oh * out_w + ow] = (scalar_t)sum;
    }
}

torch::Tensor dw_conv_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride,
    int padding) {
    
    auto batch_size = input.size(0);
    auto channels = input.size(1);
    auto in_h = input.size(2);
    auto in_w = input.size(3);
    auto kernel_size = weight.size(1);
    
    auto out_h = (in_h + 2 * padding - kernel_size) / stride + 1;
    auto out_w = (in_w + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());
    
    dim3 block(16, 16);
    dim3 grid((out_w + block.x - 1) / block.x, (out_h + block.y - 1) / block.y, batch_size * channels);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "dw_conv_forward", ([&] {
        dw_conv_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.has_value() ? bias.value().data_ptr<scalar_t>() : nullptr,
            output.data_ptr<scalar_t>(),
            batch_size, channels, in_h, in_w,
            kernel_size, stride, padding,
            out_h, out_w);
    }));
    
    return output;
}
'''

ext = load_inline(
    name='dw_conv_ext',
    cpp_sources='torch::Tensor dw_conv_forward(torch::Tensor input, torch::Tensor weight, at::optional<torch::Tensor> bias, int stride, int padding);',
    cuda_sources=cuda_source,
    functions=['dw_conv_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.stride = stride
        self.padding = padding
        self.weight = nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.randn(in_channels))
        else:
            self.register_parameter('bias', None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.dw_conv_forward(x, self.weight.view(self.weight.size(0), self.weight.size(2), self.weight.size(3)), self.bias, self.stride, self.padding)

def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    return [torch.randn(batch_size, in_channels, height, width, device='cuda')]

def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
