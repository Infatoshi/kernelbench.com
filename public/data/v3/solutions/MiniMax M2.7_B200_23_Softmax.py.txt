import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>

// Optimized softmax for B200 using vectorized loads and efficient reductions

template<typename T>
__device__ __forceinline__ T warp_reduce_max(T val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

template<typename T>
__device__ __forceinline__ T warp_reduce_sum(T val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

template<>
__device__ __forceinline__ __nv_bfloat16 warp_reduce_max(__nv_bfloat16 val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        __nv_bfloat16 other = __shfl_down_sync(0xffffffff, val, offset);
        val = __hgt(val, other) ? val : other;
    }
    return val;
}

template<>
__device__ __forceinline__ __nv_bfloat16 warp_reduce_sum(__nv_bfloat16 val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = __hadd(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

template<typename T>
__global__ void softmax_kernel_v2(T* output, const T* input, int batch_size, int dim) {
    __shared__ T s_max[32];
    __shared__ T s_sum[32];
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const T* x = input + batch_idx * dim;
    T* y = output + batch_idx * dim;
    
    // Phase 1: Thread-local max + reduction
    T local_max = -INFINITY;
    for (int i = tid; i < dim; i += blockDim.x) {
        local_max = fmaxf(local_max, x[i]);
    }
    T warp_max = warp_reduce_max(local_max);
    
    if (tid % 32 == 0) s_max[tid / 32] = warp_max;
    __syncthreads();
    
    if (tid < 32) {
        T val = s_max[tid];
        val = warp_reduce_max(val);
        if (tid == 0) s_max[0] = val;
    }
    __syncthreads();
    
    T max_val = s_max[0];
    
    // Phase 2: Compute exp(x - max) and sum
    T local_sum = 0;
    for (int i = tid; i < dim; i += blockDim.x) {
        T exp_val = expf(x[i] - max_val);
        y[i] = exp_val;
        local_sum += exp_val;
    }
    T warp_sum = warp_reduce_sum(local_sum);
    
    if (tid % 32 == 0) s_sum[tid / 32] = warp_sum;
    __syncthreads();
    
    if (tid < 32) {
        T val = s_sum[tid];
        val = warp_reduce_sum(val);
        if (tid == 0) s_sum[0] = val;
    }
    __syncthreads();
    
    T sum_val = s_sum[0];
    
    // Phase 3: Normalize
    for (int i = tid; i < dim; i += blockDim.x) {
        y[i] = y[i] / sum_val;
    }
}

// Specialized kernel for small dimensions
template<typename T>
__global__ void softmax_kernel_small(T* output, const T* input, int batch_size, int dim) {
    __shared__ T s_max[64];
    __shared__ T s_sum[64];
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x;
    
    const T* x = input + batch_idx * dim;
    T* y = output + batch_idx * dim;
    
    // Load all data into registers
    T local_max = -INFINITY;
    T vals[256];
    int elements = min(256, dim);
    
    for (int i = 0; i < elements; i++) {
        vals[i] = x[i];
        local_max = fmaxf(local_max, vals[i]);
    }
    
    // Reduce within warp
    local_max = warp_reduce_max(local_max);
    if (tid < 32) s_max[tid] = local_max;
    
    __syncthreads();
    if (tid < 32) {
        T val = warp_reduce_max(s_max[tid]);
        if (tid == 0) s_max[0] = val;
    }
    __syncthreads();
    
    T max_val = s_max[0];
    
    // Compute exp and sum
    T local_sum = 0;
    for (int i = 0; i < elements; i++) {
        vals[i] = expf(vals[i] - max_val);
        local_sum += vals[i];
    }
    
    local_sum = warp_reduce_sum(local_sum);
    if (tid < 32) s_sum[tid] = local_sum;
    
    __syncthreads();
    if (tid < 32) {
        T val = warp_reduce_sum(s_sum[tid]);
        if (tid == 0) s_sum[0] = val;
    }
    __syncthreads();
    
    T sum_val = s_sum[0];
    
    // Store results
    for (int i = 0; i < elements; i++) {
        y[i] = vals[i] / sum_val;
    }
}

template<typename T>
void launch_softmax(T* output, const T* input, int batch_size, int dim, cudaStream_t stream) {
    if (dim <= 256) {
        softmax_kernel_small<T><<<batch_size, 256, 0, stream>>>(output, input, batch_size, dim);
    } else {
        softmax_kernel_v2<T><<<batch_size, 256, 0, stream>>>(output, input, batch_size, dim);
    }
}

torch::Tensor softmax_cuda(torch::Tensor input) {
    const int batch_size = input.size(0);
    const int dim = input.size(1);
    
    auto output = torch::empty_like(input);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "softmax_cuda", ([&] {
        launch_softmax<scalar_t>(
            output.data_ptr<scalar_t>(),
            input.data_ptr<scalar_t>(),
            batch_size, dim, cudaStreamDefault
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_cuda(torch::Tensor input);',
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.softmax_cuda(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
