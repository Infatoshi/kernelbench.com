import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp reduction helper
template <typename T>
__inline__ __device__ T warpReduceSum(T val) {
    for (int offset = warpSize/2; offset > 0; offset /= 2) 
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;
}

template <typename T>
__global__ void sum_reduce_mid_kernel(const T* __restrict__ x, T* __restrict__ out, int O, int R, int I) {
    int o_idx = blockIdx.x;
    int i_idx = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (i_idx < I) {
        float sum = 0.0f;
        int base_idx = o_idx * R * I + i_idx;
        for (int r = 0; r < R; ++r) {
            sum += static_cast<float>(x[base_idx + r * I]);
        }
        out[o_idx * I + i_idx] = static_cast<T>(sum);
    }
}

template <typename T>
__global__ void sum_reduce_inner_kernel(const T* __restrict__ x, T* __restrict__ out, int O, int R) {
    int o_idx = blockIdx.x;
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    for (int r = tid; r < R; r += blockDim.x) {
        sum += static_cast<float>(x[o_idx * R + r]);
    }
    
    sum = warpReduceSum(sum);
    
    extern __shared__ float sdata[];
    int wid = tid / warpSize;
    int lane = tid % warpSize;
    
    if (lane == 0) sdata[wid] = sum;
    __syncthreads();
    
    sum = (tid < blockDim.x / warpSize) ? sdata[lane] : 0.0f;
    if (wid == 0) sum = warpReduceSum(sum);
    
    if (tid == 0) out[o_idx] = static_cast<T>(sum);
}

// Special case for Inner dimension, when R is large
template <typename T>
__global__ void sum_reduce_inner_kernel_large(const T* __restrict__ x, T* __restrict__ out, int O, int R) {
    // Similar to above but handles many elements efficiently. Using CUB block reduce could be better but let's see.
    // For many small reductions it's best to have threads do multiple sums into a local register.
    // ... we will use ATen directly or ATen dispatch for this
}

torch::Tensor my_op(torch::Tensor x, int dim) {
    int ndim = x.dim();
    dim = dim < 0 ? dim + ndim : dim;
    
    auto out_shape = x.sizes().vec();
    out_shape[dim] = 1;
    auto out = torch::empty(out_shape, x.options());
    
    long long O = 1;
    for (int i = 0; i < dim; ++i) O *= x.size(i);
    long long R = x.size(dim);
    long long I = 1;
    for (int i = dim + 1; i < ndim; ++i) I *= x.size(i);
    
    // Choose kernel
    if (I == 1) {
        // Inner reduction
        int num_threads = 256;
        int num_blocks = O;
        int shared_mem = (num_threads / 32) * sizeof(float);
        
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "sum_reduce_inner", ([&] {
            sum_reduce_inner_kernel<scalar_t><<<num_blocks, num_threads, shared_mem>>>(
                x.data_ptr<scalar_t>(), out.data_ptr<scalar_t>(), O, R);
        }));
    } else {
        // Middle/Outer reduction
        int num_threads = 256;
        dim3 blocks(O, (I + num_threads - 1) / num_threads);
        
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "sum_reduce_mid", ([&] {
            sum_reduce_mid_kernel<scalar_t><<<blocks, num_threads>>>(
                x.data_ptr<scalar_t>(), out.data_ptr<scalar_t>(), O, R, I);
        }));
    }
    
    return out;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources='torch::Tensor my_op(torch::Tensor x, int dim);',
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim

    def forward(self, x):
        return ext.my_op(x, self.dim)
