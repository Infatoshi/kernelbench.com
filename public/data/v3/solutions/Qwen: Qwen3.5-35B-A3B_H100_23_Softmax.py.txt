import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Numerically stable softmax kernel for float32
__global__ void softmax_kernel_fp32(const float* __restrict__ input, 
                                     float* __restrict__ output, 
                                     int batch_size, int feature_dim) {
    int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const float* row_ptr = input + row_idx * feature_dim;
    float* out_ptr = output + row_idx * feature_dim;
    
    // Find max for numerical stability
    float max_val = -1e38f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        max_val = fmaxf(max_val, row_ptr[i]);
    }
    
    // Warp-level reduction for max
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }
    
    // Compute exp(x - max) and sum
    float sum_exp = 0.0f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float exp_val = expf(row_ptr[i] - max_val);
        sum_exp += exp_val;
    }
    
    // Warp-level reduction for sum
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        sum_exp += __shfl_down_sync(0xffffffff, sum_exp, offset);
    }
    
    // Write output
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        out_ptr[i] = expf(row_ptr[i] - max_val) / sum_exp;
    }
}

// Softmax kernel for float16
__global__ void softmax_kernel_fp16(const half* __restrict__ input, 
                                     half* __restrict__ output, 
                                     int batch_size, int feature_dim) {
    int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const half* row_ptr = input + row_idx * feature_dim;
    half* out_ptr = output + row_idx * feature_dim;
    
    // Find max for numerical stability (using float for accumulation)
    float max_val = -1e38f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __half2float(row_ptr[i]);
        max_val = fmaxf(max_val, val);
    }
    
    // Warp-level reduction for max
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }
    
    // Compute exp(x - max) and sum
    float sum_exp = 0.0f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __half2float(row_ptr[i]);
        float exp_val = expf(val - max_val);
        sum_exp += exp_val;
    }
    
    // Warp-level reduction for sum
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        sum_exp += __shfl_down_sync(0xffffffff, sum_exp, offset);
    }
    
    // Write output
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __half2float(row_ptr[i]);
        out_ptr[i] = __float2half(expf(val - max_val) / sum_exp);
    }
}

// Softmax kernel for bfloat16
__global__ void softmax_kernel_bf16(const nv_bfloat16* __restrict__ input, 
                                     nv_bfloat16* __restrict__ output, 
                                     int batch_size, int feature_dim) {
    int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const nv_bfloat16* row_ptr = input + row_idx * feature_dim;
    nv_bfloat16* out_ptr = output + row_idx * feature_dim;
    
    // Find max for numerical stability (using float for accumulation)
    float max_val = -1e38f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __bfloat162float(row_ptr[i]);
        max_val = fmaxf(max_val, val);
    }
    
    // Warp-level reduction for max
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }
    
    // Compute exp(x - max) and sum
    float sum_exp = 0.0f;
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __bfloat162float(row_ptr[i]);
        float exp_val = expf(val - max_val);
        sum_exp += exp_val;
    }
    
    // Warp-level reduction for sum
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        sum_exp += __shfl_down_sync(0xffffffff, sum_exp, offset);
    }
    
    // Write output
    for (int i = threadIdx.x; i < feature_dim; i += blockDim.x) {
        float val = __bfloat162float(row_ptr[i]);
        out_ptr[i] = __float2bfloat16(expf(val - max_val) / sum_exp);
    }
}

// Generic dispatch based on dtype
torch::Tensor softmax_cuda(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "Input must be CUDA tensor");
    TORCH_CHECK(x.dim() == 2, "Input must be 2D tensor");
    
    auto output = torch::empty_like(x);
    int batch_size = x.size(0);
    int feature_dim = x.size(1);
    
    int threads = 256;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "softmax_cuda", ([&] {
        if constexpr (std::is_same<scalar_t, half>::value) {
            softmax_kernel_fp16<<<batch_size, threads>>>(
                x.data_ptr<half>(), output.data_ptr<half>(), batch_size, feature_dim);
        } else if constexpr (std::is_same<scalar_t, nv_bfloat16>::value) {
            softmax_kernel_bf16<<<batch_size, threads>>>(
                x.data_ptr<nv_bfloat16>(), output.data_ptr<nv_bfloat16>(), batch_size, feature_dim);
        } else {
            softmax_kernel_fp32<<<batch_size, threads>>>(
                x.data_ptr<float>(), output.data_ptr<float>(), batch_size, feature_dim);
        }
    }));
    
    return output;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_cuda(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.softmax_cuda(x)

def get_inputs():
    x = torch.randn(256, 16384, device='cuda')
    return [x]

def get_init_inputs():
    return []
