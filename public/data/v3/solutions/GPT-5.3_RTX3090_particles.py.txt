import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void particles_kernel(
    const float* __restrict__ pos,
    const float* __restrict__ vel,
    float* __restrict__ out,
    int n,
    float dt,
    float gx,
    float gy,
    float damping)
{
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= n) return;

    int p = i * 2;
    float px = pos[p];
    float py = pos[p+1];
    float vx = vel[p];
    float vy = vel[p+1];

    float nvx = vx + gx * dt;
    float nvy = vy + gy * dt;

    float npx = px + nvx * dt;
    float npy = py + nvy * dt;

    if (fabsf(npx) > 1.0f) {
        nvx = -nvx * damping;
        npx = fminf(1.0f, fmaxf(-1.0f, npx));
    }
    if (fabsf(npy) > 1.0f) {
        nvy = -nvy * damping;
        npy = fminf(1.0f, fmaxf(-1.0f, npy));
    }

    int o = i * 4;
    out[o] = npx;
    out[o+1] = npy;
    out[o+2] = nvx;
    out[o+3] = nvy;
}

torch::Tensor particles_op(torch::Tensor pos, torch::Tensor vel, float dt, float gx, float gy, float damping){
    int n = pos.size(0);
    auto out = torch::empty({n,4}, pos.options().dtype(torch::kFloat32));

    int threads = 256;
    int blocks = (n + threads - 1) / threads;

    particles_kernel<<<blocks, threads>>>(
        pos.data_ptr<float>(),
        vel.data_ptr<float>(),
        out.data_ptr<float>(),
        n, dt, gx, gy, damping);

    return out;
}
'''

ext = load_inline(
    name='particles_ext',
    cpp_sources='torch::Tensor particles_op(torch::Tensor pos, torch::Tensor vel, float dt, float gx, float gy, float damping);',
    cuda_sources=cuda_source,
    functions=['particles_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = float(damping)

    def forward(self, positions: torch.Tensor, velocities: torch.Tensor, dt: torch.Tensor, gravity: torch.Tensor) -> torch.Tensor:
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        gx = float(gravity[0].item())
        gy = float(gravity[1].item())

        pos = positions.contiguous().to(torch.float32)
        vel = velocities.contiguous().to(torch.float32)

        out = ext.particles_op(pos, vel, float(dt_value), gx, gy, float(self.damping))
        return out


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2, device='cuda') * 2.0 - 1.0
    velocities = torch.randn(n, 2, device='cuda') * 0.05
    dt = torch.tensor(1.0 / 60.0, device='cuda')
    gravity = torch.tensor([0.0, -9.81], device='cuda')
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
