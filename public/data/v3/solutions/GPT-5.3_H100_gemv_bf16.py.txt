import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_bf16.h>

__global__ void gemv_bf16_kernel(
    const __nv_bfloat16* __restrict__ x,
    const __nv_bfloat16* __restrict__ w,
    float* __restrict__ y,
    int B, int IN, int OUT)
{
    int b = blockIdx.x;
    int o = blockIdx.y * blockDim.x + threadIdx.x;
    if (b >= B || o >= OUT) return;

    const __nv_bfloat16* x_ptr = x + b * IN;
    const __nv_bfloat16* w_ptr = w + o * IN;

    float acc = 0.0f;

    for (int i = 0; i < IN; i++) {
        float xv = __bfloat162float(x_ptr[i]);
        float wv = __bfloat162float(w_ptr[i]);
        acc += xv * wv;
    }

    y[b * OUT + o] = acc;
}

torch::Tensor gemv_bf16(torch::Tensor x, torch::Tensor w) {
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = w.size(0);

    auto y = torch::empty({B, OUT}, x.options().dtype(torch::kFloat32));

    dim3 block(256);
    dim3 grid(B, (OUT + 255) / 256);

    gemv_bf16_kernel<<<grid, block>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
        reinterpret_cast<const __nv_bfloat16*>(w.data_ptr<at::BFloat16>()),
        y.data_ptr<float>(),
        B, IN, OUT
    );

    return y.to(torch::kBFloat16);
}
'''

ext = load_inline(
    name='gemv_bf16_ext',
    cpp_sources='torch::Tensor gemv_bf16(torch::Tensor x, torch::Tensor w);',
    cuda_sources=cuda_source,
    functions=['gemv_bf16'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf16 = x.to(dtype=torch.bfloat16).contiguous()
        w = self.weight.contiguous()
        return ext.gemv_bf16(x_bf16, w)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16, device='cuda')]


def get_init_inputs():
    return []
