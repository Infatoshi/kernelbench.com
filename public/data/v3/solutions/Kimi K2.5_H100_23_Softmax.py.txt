import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <limits>

template <typename T>
__global__ void softmax_kernel(T* __restrict__ out, const T* __restrict__ in, int rows, int cols) {
  int row = blockIdx.x;
  if (row >= rows) return;
  
  const T* row_in = in + row * cols;
  T* row_out = out + row * cols;
  
  // First pass: find max for numerical stability
  float local_max = -std::numeric_limits<float>::infinity();
  
  for (int i = threadIdx.x; i < cols; i += blockDim.x) {
    float val = static_cast<float>(row_in[i]);
    local_max = fmaxf(local_max, val);
  }
  
  // Warp reduction for max
  #pragma unroll
  for (int offset = 16; offset > 0; offset /= 2) {
    local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, offset));
  }
  
  // Block reduction for max
  __shared__ float warp_max[32];
  int lane = threadIdx.x % 32;
  int warp_id = threadIdx.x / 32;
  
  if (lane == 0) warp_max[warp_id] = local_max;
  __syncthreads();
  
  if (warp_id == 0) {
    local_max = (lane * 32 < blockDim.x) ? warp_max[lane] : -std::numeric_limits<float>::infinity();
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
      local_max = fmaxf(local_max, __shfl_xor_sync(0xffffffff, local_max, offset));
    }
    if (lane == 0) warp_max[0] = local_max;
  }
  __syncthreads();
  
  float row_max = warp_max[0];
  
  // Second pass: compute sum of exp(x - max)
  float local_sum = 0.0f;
  for (int i = threadIdx.x; i < cols; i += blockDim.x) {
    float val = static_cast<float>(row_in[i]);
    local_sum += expf(val - row_max);
  }
  
  // Warp reduction for sum
  #pragma unroll
  for (int offset = 16; offset > 0; offset /= 2) {
    local_sum += __shfl_xor_sync(0xffffffff, local_sum, offset);
  }
  
  // Block reduction for sum
  __shared__ float warp_sum[32];
  if (lane == 0) warp_sum[warp_id] = local_sum;
  __syncthreads();
  
  if (warp_id == 0) {
    local_sum = (lane * 32 < blockDim.x) ? warp_sum[lane] : 0.0f;
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
      local_sum += __shfl_xor_sync(0xffffffff, local_sum, offset);
    }
    if (lane == 0) warp_sum[0] = local_sum;
  }
  __syncthreads();
  
  float row_sum = warp_sum[0];
  float inv_sum = 1.0f / row_sum;
  
  // Third pass: write normalized values
  for (int i = threadIdx.x; i < cols; i += blockDim.x) {
    float val = static_cast<float>(row_in[i]);
    float out_val = expf(val - row_max) * inv_sum;
    row_out[i] = static_cast<T>(out_val);
  }
}

torch::Tensor softmax_cuda(torch::Tensor input) {
  TORCH_CHECK(input.is_cuda(), "Input must be a CUDA tensor");
  TORCH_CHECK(input.dim() == 2, "Input must be 2-dimensional");
  
  auto output = torch::empty_like(input);
  int rows = input.size(0);
  int cols = input.size(1);
  
  const int threads = 512;  // Optimal for H100 occupancy
  dim3 blocks(rows);
  dim3 threads_per_block(threads);
  
  AT_DISPATCH_FLOATING_TYPES_AND2(
    at::ScalarType::Half, 
    at::ScalarType::BFloat16, 
    input.scalar_type(), 
    "softmax_cuda", 
    [&] {
      softmax_kernel<scalar_t><<<blocks, threads_per_block>>>(
        output.data_ptr<scalar_t>(),
        input.data_ptr<scalar_t>(),
        rows,
        cols
      );
    }
  );
  
  return output;
}
'''

cpp_source = r'''
torch::Tensor softmax_cuda(torch::Tensor input);
'''

# Compile the extension
ext = load_inline(
    name='softmax_opt_h100',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Softmax activation using custom CUDA kernel for H100.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features).

        Returns:
            torch.Tensor: Output tensor with Softmax applied, same shape as input.
        """
        return ext.softmax_cuda(x)
