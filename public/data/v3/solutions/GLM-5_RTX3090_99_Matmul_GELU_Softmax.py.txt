import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// GELU function
__device__ __forceinline__ float gelu_float(float x) {
    // 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    const float sqrt_2_over_pi = 0.7978845608028654f;
    const float coeff = 0.044715f;
    float x3 = x * x * x;
    float inner = sqrt_2_over_pi * (x + coeff * x3);
    return 0.5f * x * (1.0f + tanhf(inner));
}

// Main fused kernel - one row per block
__global__ void __launch_bounds__(256) fused_gemm_gelu_softmax_kernel(
    const at::Half* __restrict__ A,
    const at::Half* __restrict__ B,
    const float* __restrict__ bias,
    at::Half* __restrict__ C,
    int M, int N, int K
) {
    extern __shared__ float smem[];
    float* row_acc = smem;        // Per-row accumulator (N elements)
    float* row_exp = smem + N;    // Exp values (N elements)
    
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    // Initialize accumulator for all columns
    for (int col = tid; col < N; col += block_size) {
        row_acc[col] = 0.0f;
    }
    __syncthreads();
    
    // GEMM: C[row, col] = sum_k A[row, k] * B[col, k]
    // Process K in chunks to reduce register pressure
    for (int k = 0; k < K; k++) {
        float a_val = static_cast<float>(A[row * K + k]);
        
        for (int col = tid; col < N; col += block_size) {
            float b_val = static_cast<float>(B[col * K + k]);
            row_acc[col] += a_val * b_val;
        }
    }
    __syncthreads();
    
    // Add bias and GELU
    for (int col = tid; col < N; col += block_size) {
        float val = row_acc[col];
        if (bias != nullptr) {
            val += bias[col];
        }
        row_acc[col] = gelu_float(val);
    }
    __syncthreads();
    
    // Softmax
    // Find max
    float max_val = -INFINITY;
    for (int col = tid; col < N; col += block_size) {
        max_val = fmaxf(max_val, row_acc[col]);
    }
    
    for (int offset = block_size / 2; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xffffffff, max_val, offset);
        max_val = fmaxf(max_val, other);
    }
    __shared__ float global_max;
    if (tid == 0) global_max = max_val;
    __syncthreads();
    max_val = global_max;
    
    // Compute exp and sum
    float sum = 0.0f;
    for (int col = tid; col < N; col += block_size) {
        row_exp[col] = expf(row_acc[col] - max_val);
        sum += row_exp[col];
    }
    
    for (int offset = block_size / 2; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    __shared__ float global_sum;
    if (tid == 0) global_sum = sum;
    __syncthreads();
    sum = global_sum;
    
    float inv_sum = 1.0f / sum;
    
    // Normalize and store
    for (int col = tid; col < N; col += block_size) {
        C[row * N + col] = static_cast<at::Half>(row_exp[col] * inv_sum);
    }
}

// Main entry point
torch::Tensor fused_gemm_gelu_softmax(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias
) {
    int M = input.size(0);   // batch_size = 128
    int K = input.size(1);   // in_features = 4096
    int N = weight.size(0);  // out_features = 4096
    
    // Ensure FP16 inputs
    auto input_fp16 = input.to(torch::kHalf);
    auto weight_fp16 = weight.to(torch::kHalf);
    
    // Output
    auto output = torch::empty({M, N}, torch::TensorOptions().dtype(torch::kHalf).device(input.device()));
    
    const float* bias_ptr = bias.defined() && bias.numel() > 0 ? bias.data_ptr<float>() : nullptr;
    
    // Use fused kernel
    int block_size = 256;
    int shared_mem_size = N * 2 * sizeof(float);  // row_acc + row_exp
    
    fused_gemm_gelu_softmax_kernel<<<M, block_size, shared_mem_size>>>(
        input_fp16.data_ptr<at::Half>(),
        weight_fp16.data_ptr<at::Half>(),
        bias_ptr,
        output.data_ptr<at::Half>(),
        M, N, K
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_gemm_gelu_softmax(torch::Tensor input, torch::Tensor weight, torch::Tensor bias);
'''

ext = load_inline(
    name='fused_kernel',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_gemm_gelu_softmax'],
    extra_cuda_cflags=['-arch=sm_86', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_gemm_gelu_softmax(x, self.weight, self.bias)

batch_size = 128
in_features = 4096
out_features = 4096

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features]
