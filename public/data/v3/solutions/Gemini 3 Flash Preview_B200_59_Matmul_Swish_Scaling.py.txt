import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 256, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 256, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def swish_scaling_kernel(
    X_ptr, W_ptr, B_ptr, Y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_ym, stride_yn,
    scaling_factor,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    # L2 Cache-friendly grouping
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M))
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N))
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptr = X_ptr + (offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    b_ptr = W_ptr + (offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        rem_k = K - k * BLOCK_SIZE_K
        a = tl.load(a_ptr, mask=(offs_am[:, None] < M) & (offs_k[None, :] < rem_k), other=0.0)
        b = tl.load(b_ptr, mask=(offs_k[:, None] < rem_k) & (offs_bn[None, :] < N), other=0.0)
        accumulator += tl.dot(a, b)
        a_ptr += BLOCK_SIZE_K * stride_xk
        b_ptr += BLOCK_SIZE_K * stride_wk

    if B_ptr is not None:
        bias = tl.load(B_ptr + offs_bn, mask=offs_bn < N, other=0.0)
        accumulator += bias[None, :]

    # Swish activation: x * (1 / (1 + exp(-x)))
    # and Scaling
    # tl.sigmoid is usually faster/accurate enough
    out = accumulator * tl.sigmoid(accumulator) * scaling_factor
    
    y_ptr = Y_ptr + (offs_am[:, None] * stride_ym + offs_bn[None, :] * stride_yn)
    mask = (offs_am[:, None] < M) & (offs_bn[None, :] < N)
    tl.store(y_ptr, out.to(Y_ptr.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = float(scaling_factor)
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        M, K = x.shape
        N, _ = self.matmul.weight.shape
        y = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        
        swish_scaling_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, y,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            y.stride(0), y.stride(1),
            self.scaling_factor
        )
        return y

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 2.0]
