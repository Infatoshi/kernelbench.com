import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void particle_step_kernel(
    const float2* __restrict__ positions,
    const float2* __restrict__ velocities,
    float2* __restrict__ result,
    float dt,
    float gx,
    float gy,
    float damping,
    int n) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float2 pos = positions[idx];
        float2 vel = velocities[idx];

        // Integrate
        vel.x += gx * dt;
        vel.y += gy * dt;
        pos.x += vel.x * dt;
        pos.y += vel.y * dt;

        // Boundary check X
        if (pos.x > 1.0f) {
            pos.x = 1.0f;
            vel.x = -vel.x * damping;
        } else if (pos.x < -1.0f) {
            pos.x = -1.0f;
            vel.x = -vel.x * damping;
        }

        // Boundary check Y
        if (pos.y > 1.0f) {
            pos.y = 1.0f;
            vel.y = -vel.y * damping;
        } else if (pos.y < -1.0f) {
            pos.y = -1.0f;
            vel.y = -vel.y * damping;
        }

        // Write output: pos followed by vel in cat dim -1
        // Output tensor shape is (n, 4) if pos is (n, 2) and vel is (n, 2)
        // result[idx] = {pos.x, pos.y, vel.x, vel.y}
        float4 res;
        res.x = pos.x;
        res.y = pos.y;
        res.z = vel.x;
        res.w = vel.y;
        ((float4*)result)[idx] = res;
    }
}

torch::Tensor particle_step(
    torch::Tensor positions,
    torch::Tensor velocities,
    torch::Tensor dt_tensor,
    torch::Tensor gravity_tensor,
    float damping) 
{
    int n = positions.size(0);
    auto options = torch::TensorOptions().dtype(positions.dtype()).device(positions.device());
    auto result = torch::empty({n, 4}, options);

    float dt = dt_tensor.item<float>();
    auto gravity_cpu = gravity_tensor.to(torch::kCPU);
    float gx = gravity_cpu.data_ptr<float>()[0];
    float gy = gravity_cpu.data_ptr<float>()[1];

    const int threads = 256;
    const int blocks = (n + threads - 1) / threads;

    particle_step_kernel<<<blocks, threads>>>(
        (const float2*)positions.data_ptr<float>(),
        (const float2*)velocities.data_ptr<float>(),
        (float2*)result.data_ptr<float>(),
        dt, gx, gy, damping, n
    );

    return result;
}
'''

cpp_source = r'''
torch::Tensor particle_step(
    torch::Tensor positions,
    torch::Tensor velocities,
    torch::Tensor dt_tensor,
    torch::Tensor gravity_tensor,
    float damping);
'''

particle_ext = load_inline(
    name='particle_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_step'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(self, positions, velocities, dt, gravity):
        return particle_ext.particle_step(positions, velocities, dt, gravity, self.damping)

def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2, device='cuda') * 2.0 - 1.0
    velocities = torch.randn(n, 2, device='cuda') * 0.05
    dt = torch.tensor(1.0 / 60.0, device='cuda')
    gravity = torch.tensor([0.0, -9.81], device='cuda')
    return [positions, velocities, dt, gravity]

def get_init_inputs():
    return [0.9]

if __name__ == "__main__":
    import reference
    ref_m = reference.Model(*reference.get_init_inputs()).cuda().eval()
    sol_m = Model(*get_init_inputs()).cuda().eval()
    sol_m.load_state_dict(ref_m.state_dict(), strict=False)
    inputs = [x.cuda() if isinstance(x, torch.Tensor) else x for x in get_inputs()]
    ref_out = ref_m(*inputs)
    sol_out = sol_m(*inputs)
    diff = torch.max(torch.abs(ref_out - sol_out)).item()
    print(f'max_diff={diff:.6f}')
    print('PASS' if torch.allclose(ref_out, sol_out, atol=1e-2, rtol=1e-2) else 'FAIL')
