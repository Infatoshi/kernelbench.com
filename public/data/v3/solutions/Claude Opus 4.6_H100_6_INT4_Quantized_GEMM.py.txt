import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def int4_dequant_gemm_kernel(
    x_ptr, w_packed_ptr, scales_ptr, out_ptr,
    M, N, K,
    group_size,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_sn, stride_sg,
    stride_om, stride_on,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """Fused INT4 dequantization + GEMM kernel.
    
    Process BLOCK_K packed bytes per iteration (= 2*BLOCK_K actual K elements).
    Load x in two contiguous BLOCK_K-sized chunks per iteration.
    """
    pid = tl.program_id(0)
    
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m
    
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    mask_m = offs_m < M
    mask_n = offs_n < N
    
    # K_packed = K // 2
    # We iterate over packed dimension in steps of BLOCK_K
    # Each step processes 2*BLOCK_K actual K elements
    K_packed = K // 2
    
    for kp_start in range(0, K_packed, BLOCK_K):
        # Actual k start in the unpacked dimension
        k_actual = kp_start * 2
        
        # Load packed weights: (BLOCK_N, BLOCK_K) uint8
        w_ptrs = w_packed_ptr + offs_n[:, None] * stride_wn + (kp_start + offs_k[None, :]) * stride_wk
        mask_k = (kp_start + offs_k) < K_packed
        w_packed = tl.load(w_ptrs, mask=mask_n[:, None] & mask_k[None, :], other=0)
        
        # Unpack
        w_low = (w_packed & 0x0F).to(tl.float16)   # (BLOCK_N, BLOCK_K) - even positions
        w_high = ((w_packed >> 4) & 0x0F).to(tl.float16)  # (BLOCK_N, BLOCK_K) - odd positions
        
        # Load scale for even group
        g_even = k_actual // group_size
        s_ptrs = scales_ptr + offs_n * stride_sn + g_even * stride_sg
        scale_even = tl.load(s_ptrs, mask=mask_n, other=0.0)
        
        # Dequantize low nibble (even k positions)
        w_low = scale_even[:, None] * (w_low - 8.0)
        
        # Load x for even positions: (BLOCK_M, BLOCK_K) at k_actual, k_actual+2, ...
        # These are NOT contiguous! They are at positions k_actual, k_actual+2, k_actual+4, ...
        # This is the strided access problem.
        
        # Alternative: load x contiguously as two BLOCK_K chunks
        # Chunk 1: x[:, k_actual : k_actual + BLOCK_K] (contiguous)
        # Chunk 2: x[:, k_actual + BLOCK_K : k_actual + 2*BLOCK_K] (contiguous)
        # But w_low corresponds to even positions and w_high to odd positions...
        
        # Actually, the packed byte at position p encodes:
        # - low nibble -> unpacked position 2*p (even)
        # - high nibble -> unpacked position 2*p + 1 (odd)
        # So if we load x contiguously as two halves of BLOCK_K each,
        # we need to split each half into even/odd components.
        
        # Better approach: load two BLOCK_K chunks of x contiguously,
        # and split the weight accordingly.
        
        # Actually, let's just accept the reality:
        # w_low[n, j] = weight at unpacked position (n, 2*(kp_start+j))
        # w_high[n, j] = weight at unpacked position (n, 2*(kp_start+j)+1)
        # x_even[m, j] = x at position (m, 2*(kp_start+j)) 
        # x_odd[m, j] = x at position (m, 2*(kp_start+j)+1)
        
        # The contiguous x load approach:
        # Load x[:, k_actual : k_actual + 2*BLOCK_K] as a contiguous block
        # Then split into even and odd

        # Load first contiguous chunk of x
        x_ptrs_1 = x_ptr + offs_m[:, None] * stride_xm + (k_actual + offs_k[None, :]) * stride_xk
        mask_k1 = (k_actual + offs_k) < K
        x_chunk1 = tl.load(x_ptrs_1, mask=mask_m[:, None] & mask_k1[None, :], other=0.0)
        
        # Load second contiguous chunk of x
        x_ptrs_2 = x_ptr + offs_m[:, None] * stride_xm + (k_actual + BLOCK_K + offs_k[None, :]) * stride_xk
        mask_k2 = (k_actual + BLOCK_K + offs_k) < K
        x_chunk2 = tl.load(x_ptrs_2, mask=mask_m[:, None] & mask_k2[None, :], other=0.0)
        
        # Now x_chunk1 has x[:, k_actual : k_actual+BLOCK_K] 
        # and x_chunk2 has x[:, k_actual+BLOCK_K : k_actual+2*BLOCK_K]
        # 
        # But w_low[n, j] = weight(n, 2j+k_actual) -> x needed at column 2j+k_actual
        # For j=0..BLOCK_K-1, the x columns needed are: k_actual, k_actual+2, k_actual+4, ...
        # These span k_actual to k_actual+2*(BLOCK_K-1)
        # 
        # From x_chunk1 (cols k_actual..k_actual+BLOCK_K-1):
        #   x_even needs cols 0, 2, 4, ... BLOCK_K-2 of chunk1 (BLOCK_K//2 values)
        #   x_odd needs cols 1, 3, 5, ... BLOCK_K-1 of chunk1 (BLOCK_K//2 values)
        # From x_chunk2 (cols k_actual+BLOCK_K..k_actual+2*BLOCK_K-1):
        #   x_even needs cols 0, 2, 4, ... of chunk2 (BLOCK_K//2 values)
        #   x_odd needs cols 1, 3, 5, ... of chunk2 (BLOCK_K//2 values)
        
        # This splitting is also complex. Let me just do the simple dequant-then-matmul approach
        # with a different data layout.
        
        # SIMPLEST CORRECT APPROACH: 
        # Process BLOCK_K actual K elements per loop iteration (BLOCK_K//2 packed bytes)
        # This means the dot product inner dimension is BLOCK_K
        # We need to construct the full unpacked weight tile of shape (BLOCK_N, BLOCK_K)
        
        # Let me restart with this simpler approach below
        # For now, use the scale for odd positions too (same group since BLOCK_K <= group_size)
        w_high = scale_even[:, None] * (w_high - 8.0)
        
        # acc += x_even @ w_low.T + x_odd @ w_high.T
        acc += tl.dot(x_chunk1, tl.trans(w_low))  # WRONG - dimensions don't match
        acc += tl.dot(x_chunk2, tl.trans(w_high))  # WRONG
    
    out_ptrs = out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on
    mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(out_ptrs, acc.to(tl.float16), mask=mask)


class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0
        assert K % 2 == 0

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, K = x.shape
        M = batch_size * seq_len
        N = self.N
        
        x_2d = x.reshape(M, K)
        out = torch.empty((M, N), dtype=torch.float16, device=x.device)
        
        BLOCK_M = 128
        BLOCK_N = 128
        BLOCK_K = 64  # packed bytes per iteration, = 128 actual K elements
        GROUP_SIZE_M = 8
        
        grid = (triton.cdiv(M, BLOCK_M) * triton.cdiv(N, BLOCK_N),)
        
        int4_dequant_gemm_kernel[grid](
            x_2d, self.weight_packed, self.scales, out,
            M, N, K,
            self.group_size,
            x_2d.stride(0), x_2d.stride(1),
            self.weight_packed.stride(0), self.weight_packed.stride(1),
            self.scales.stride(0), self.scales.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
        )
        
        return out.view(batch_size, seq_len, N)


batch_size = 4
seq_len = 2048
K = 4096
N = 11008
group_size = 128


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [K, N, group_size]
