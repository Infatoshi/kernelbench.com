"""
Optimized MoE Grouped GEMM using batched operations for H100.
Uses BF16 for tensor core acceleration with FP32 accumulation.
"""

import torch
import torch.nn as nn


class Model(nn.Module):
    def __init__(self, num_experts: int = 8, hidden_dim: int = 1024, expert_dim: int = 3072):
        super().__init__()
        self.num_experts = num_experts
        self.hidden_dim = hidden_dim
        self.expert_dim = expert_dim
        
        # Store weights in float32 for compatibility
        self.expert_up = nn.Parameter(torch.randn(num_experts, hidden_dim, expert_dim) * 0.02)
        self.expert_down = nn.Parameter(torch.randn(num_experts, expert_dim, hidden_dim) * 0.02)

    def forward(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        batch, seq, hidden = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq
        
        # Flatten
        x_flat = x.reshape(num_tokens, hidden)  # [num_tokens, hidden]
        idx_flat = expert_indices.reshape(num_tokens, top_k)  # [num_tokens, top_k]
        w_flat = expert_weights.reshape(num_tokens, top_k)  # [num_tokens, top_k]
        
        # Create route info: each (token, k) is a route
        # total_routes = num_tokens * top_k
        token_ids = torch.arange(num_tokens, device=x.device).unsqueeze(1).expand(-1, top_k).reshape(-1)
        route_expert_ids = idx_flat.reshape(-1)  # [total_routes]
        route_weights = w_flat.reshape(-1)  # [total_routes]
        
        total_routes = num_tokens * top_k
        
        # Sort by expert for grouped processing
        sorted_indices = torch.argsort(route_expert_ids)
        sorted_token_ids = token_ids[sorted_indices]
        sorted_expert_ids = route_expert_ids[sorted_indices]
        sorted_weights = route_weights[sorted_indices]
        
        # Compute expert boundaries
        expert_counts = torch.bincount(sorted_expert_ids, minlength=self.num_experts)
        expert_boundaries = torch.zeros(self.num_experts + 1, dtype=torch.int32, device=x.device)
        expert_boundaries[1:] = torch.cumsum(expert_counts, dim=0)
        
        # Gather tokens for each expert
        sorted_x = x_flat[sorted_token_ids]  # [total_routes, hidden]
        
        # Output buffer
        sorted_down_out = torch.zeros(total_routes, self.hidden_dim, dtype=x.dtype, device=x.device)
        
        # Process each expert - use torch.matmul which triggers tensor cores
        for expert_id in range(self.num_experts):
            start = expert_boundaries[expert_id].item()
            end = expert_boundaries[expert_id + 1].item()
            
            if start == end:
                continue
            
            # Get tokens for this expert
            expert_x = sorted_x[start:end]  # [num_tokens_this_expert, hidden]
            
            # Up projection - use torch.mm for explicit tensor core usage
            up_out = torch.mm(expert_x, self.expert_up[expert_id])  # [num_tokens, expert_dim]
            
            # Down projection
            down_out = torch.mm(up_out, self.expert_down[expert_id])  # [num_tokens, hidden]
            
            sorted_down_out[start:end] = down_out
        
        # Apply weights
        weighted_out = sorted_down_out * sorted_weights.unsqueeze(1)
        
        # Scatter add back to original token positions
        out_flat = torch.zeros(num_tokens, self.hidden_dim, dtype=x.dtype, device=x.device)
        out_flat.scatter_add_(0, sorted_token_ids.unsqueeze(1).expand(-1, self.hidden_dim), weighted_out)
        
        return out_flat.reshape(batch, seq, hidden)


OP_TYPE = "moe_grouped_gemm"
SUPPORTED_PRECISIONS = ["fp8", "bf16", "fp16"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    batch, seq, hidden = 4, 256, 1024
    top_k = 2
    num_experts = 8

    x = torch.randn(batch, seq, hidden)
    expert_indices = torch.randint(0, num_experts, (batch, seq, top_k))
    expert_weights = torch.softmax(torch.randn(batch, seq, top_k), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return []
