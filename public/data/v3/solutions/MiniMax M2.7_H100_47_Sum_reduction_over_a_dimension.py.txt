cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>

// Helper macros for type conversion
#define TO_FLOAT(val) __half2float(val)
#define FROM_FLOAT(val) __float2half(val)
#define BF16_TO_FLOAT(val) __bfloat162float(val)
#define BF16_FROM_FLOAT(val) __float2bfloat16(val)

// Generic reduction kernel for FP32
template <typename T>
__global__ void sum_reduction_kernel_fp32(const T* __restrict__ x, T* __restrict__ y,
                                          int batch_size, int dim1, int dim2, int reduce_dim) {
    extern __shared__ float shared_data_float[];
    
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    // Calculate output index based on reduce_dim
    int out_idx;
    int reduce_size;
    
    if (reduce_dim == 0) {
        // Reduce over batch_size
        // Output shape: [1, dim1, dim2]
        int out_dim1 = blockIdx.x;
        int out_dim2 = blockIdx.y;
        out_idx = out_dim1 * dim2 + out_dim2;
        reduce_size = batch_size;
    } else if (reduce_dim == 1) {
        // Reduce over dim1
        // Output shape: [batch_size, 1, dim2]
        int out_batch = blockIdx.x;
        int out_dim2 = blockIdx.y;
        out_idx = out_batch * dim2 + out_dim2;
        reduce_size = dim1;
    } else {  // reduce_dim == 2
        // Reduce over dim2
        // Output shape: [batch_size, dim1, 1]
        int out_batch = blockIdx.x;
        int out_dim1 = blockIdx.y;
        out_idx = out_batch * dim1 + out_dim1;
        reduce_size = dim2;
    }
    
    // Each thread loads elements strided by block_size
    float sum_val = 0.0f;
    int num_elements = (reduce_size + block_size - 1) / block_size;
    
    if (reduce_dim == 0) {
        // Reduce over batch_size dimension
        // Input index: b * dim1 * dim2 + out_dim1 * dim2 + out_dim2
        int out_dim1 = blockIdx.x;
        int out_dim2 = blockIdx.y;
        int base_idx = out_dim1 * dim2 + out_dim2;
        int stride = dim1 * dim2;
        for (int i = 0; i < num_elements; i++) {
            int b = tid + i * block_size;
            if (b < reduce_size) {
                sum_val += static_cast<float>(x[b * stride + base_idx]);
            }
        }
    } else if (reduce_dim == 1) {
        // Reduce over dim1 dimension
        // Input index: out_batch * dim1 * dim2 + d1 * dim2 + out_dim2
        int out_batch = blockIdx.x;
        int out_dim2 = blockIdx.y;
        int base_idx = out_batch * dim1 * dim2 + out_dim2;
        for (int i = 0; i < num_elements; i++) {
            int d1 = tid + i * block_size;
            if (d1 < reduce_size) {
                sum_val += static_cast<float>(x[base_idx + d1 * dim2]);
            }
        }
    } else {  // reduce_dim == 2
        // Reduce over dim2 dimension
        // Input index: out_batch * dim1 * dim2 + out_dim1 * dim1 + d2
        int out_batch = blockIdx.x;
        int out_dim1 = blockIdx.y;
        int base_idx = out_batch * dim1 * dim2 + out_dim1;
        for (int i = 0; i < num_elements; i++) {
            int d2 = tid + i * block_size;
            if (d2 < reduce_size) {
                sum_val += static_cast<float>(x[base_idx + d2 * dim1]);
            }
        }
    }
    
    // Store to shared memory
    shared_data_float[tid] = sum_val;
    __syncthreads();
    
    // Block reduction in shared memory (tree reduction)
    for (int s = block_size / 2; s >= 32; s >>= 1) {
        if (tid < s) {
            shared_data_float[tid] += shared_data_float[tid + s];
        }
        __syncthreads();
    }
    
    // Warp-level reduction using shuffle
    float warp_sum = (tid < 32) ? shared_data_float[tid] : 0.0f;
    warp_sum += __shfl_down_sync(0xffffffff, warp_sum, 16);
    warp_sum += __shfl_down_sync(0xffffffff, warp_sum, 8);
    warp_sum += __shfl_down_sync(0xffffffff, warp_sum, 4);
    warp_sum += __shfl_down_sync(0xffffffff, warp_sum, 2);
    warp_sum += __shfl_down_sync(0xffffffff, warp_sum, 1);
    
    // First thread writes result
    if (tid == 0) {
        y[out_idx] = static_cast<T>(warp_sum);
    }
}

// Dispatch function
torch::Tensor sum_reduction_op(torch::Tensor x, int reduce_dim) {
    int batch_size = x.size(0);
    int dim1 = x.size(1);
    int dim2 = x.size(2);
    
    // Output shape - always 3D with keepdim=True
    std::vector<int64_t> out_shape = {batch_size, dim1, dim2};
    if (reduce_dim == 0) {
        out_shape[0] = 1;
    } else if (reduce_dim == 1) {
        out_shape[1] = 1;
    } else {
        out_shape[2] = 1;
    }
    
    torch::Tensor y = torch::empty(out_shape, x.options());
    
    // Launch configuration
    dim3 blocks;
    int threads = 256;
    int smem_size = threads * sizeof(float);
    
    if (reduce_dim == 0) {
        // Each block handles one output element (dim1 * dim2 blocks)
        blocks = dim3(dim1, dim2);
    } else if (reduce_dim == 1) {
        // Each block handles one output element (batch_size * dim2 blocks)
        blocks = dim3(batch_size, dim2);
    } else {
        // Each block handles one output element (batch_size * dim1 blocks)
        blocks = dim3(batch_size, dim1);
    }
    
    if (x.scalar_type() == torch::kFloat) {
        sum_reduction_kernel_fp32<float><<<blocks, threads, smem_size>>>(
            x.data_ptr<float>(), y.data_ptr<float>(),
            batch_size, dim1, dim2, reduce_dim);
    } else if (x.scalar_type() == torch::kHalf) {
        sum_reduction_kernel_fp16<<<blocks, threads, smem_size>>>(
            reinterpret_cast<__half*>(x.data_ptr<torch::Half>()),
            reinterpret_cast<__half*>(y.data_ptr<torch::Half>()),
            batch_size, dim1, dim2, reduce_dim);
    } else if (x.scalar_type() == torch::kBFloat16) {
        sum_reduction_kernel_bf16<<<blocks, threads, smem_size>>>(
            reinterpret_cast<__nv_bfloat16*>(x.data_ptr<torch::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(y.data_ptr<torch::BFloat16>()),
            batch_size, dim1, dim2, reduce_dim);
    }
    
    return y;
}

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources='''
torch::Tensor sum_reduction_op(torch::Tensor x, int reduce_dim);
''',
    cuda_sources=cuda_source,
    functions=['sum_reduction_op'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that performs sum reduction over a specified dimension.
    Uses CUDA kernel with warp-level reductions for high throughput on H100.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): Dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies sum reduction over the specified dimension.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor after sum reduction, shape (..., 1, ...).
        """
        return ext.sum_reduction_op(x, self.dim)

def get_inputs():
    return [torch.randn(64, 256, 256, device='cuda')]

def get_init_inputs():
    return [1]  # reduce_dim = 1
