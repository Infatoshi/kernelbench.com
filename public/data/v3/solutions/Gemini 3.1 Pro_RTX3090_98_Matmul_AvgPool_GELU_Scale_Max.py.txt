import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_fused_kernel(
    a_ptr, b_ptr, bias_ptr, out_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    stride_out_m, stride_out_n,
    scale_factor: tl.constexpr,
    BLOCK_SIZE_M: tl.constexpr, 
    BLOCK_SIZE_N: tl.constexpr, 
    BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
    POOL_KERNEL_SIZE: tl.constexpr,
    POOL_OUT_N: tl.constexpr
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = b_ptr + (offs_bn[None, :] * stride_bn + offs_k[:, None] * stride_bk)
    
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        mask_a = (offs_am[:, None] < M) & ((k * BLOCK_SIZE_K + offs_k)[None, :] < K)
        mask_b = (offs_bn[None, :] < N) & ((k * BLOCK_SIZE_K + offs_k)[:, None] < K)
        
        a = tl.load(a_ptrs, mask=mask_a, other=0.0)
        b = tl.load(b_ptrs, mask=mask_b, other=0.0)
        
        accumulator += tl.dot(a, b)
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_bk

    if bias_ptr is not None:
        bias = tl.load(bias_ptr + offs_bn, mask=offs_bn < N, other=0.0)
        accumulator += bias[None, :]

    # We need to reshape accumulator from (BLOCK_SIZE_M, BLOCK_SIZE_N) to (BLOCK_SIZE_M, POOL_OUT_N, POOL_KERNEL_SIZE)
    acc_reshaped = tl.reshape(accumulator, (BLOCK_SIZE_M, POOL_OUT_N, POOL_KERNEL_SIZE))
    pooled = tl.sum(acc_reshaped, axis=2) / POOL_KERNEL_SIZE
    
    # GELU
    # x * 0.5 * (1.0 + tl.math.erf(x * 0.707106781))
    gelu = pooled * 0.5 * (1.0 + tl.math.erf(pooled * 0.707106781))
    
    # Scale
    scaled = gelu * scale_factor
    
    # Handle mask for local_max out-of-bounds columns (where valid pool outputs end).
    valid_n_pools = tl.cdiv(min(BLOCK_SIZE_N, N - pid_n * BLOCK_SIZE_N), POOL_KERNEL_SIZE)
    # create mask mapping pools to range [0, POOL_OUT_N)
    pool_idx = tl.arange(0, POOL_OUT_N)
    pool_mask = pool_idx[None, :] < valid_n_pools
    
    # Mask out invalid pooled items with negative infinity
    scaled_masked = tl.where(pool_mask, scaled, -float('inf'))
    
    local_max = tl.max(scaled_masked, axis=1) # Shape: (BLOCK_SIZE_M,)
    
    out_ptrs = out_ptr + (offs_am * stride_out_m + pid_n * stride_out_n)
    tl.store(out_ptrs, local_max, mask=(offs_am < M) & (pid_n < num_pid_n))


def triton_fused_op(x, weight, bias, pool_kernel_size, scale_factor):
    # Determine autotune or block sizes
    if x.dtype == torch.float32:
        BLOCK_SIZE_M = 64
        BLOCK_SIZE_N = 64
        BLOCK_SIZE_K = 32
        GROUP_SIZE_M = 8
    else:
        BLOCK_SIZE_M = 128
        BLOCK_SIZE_N = 128
        BLOCK_SIZE_K = 32
        GROUP_SIZE_M = 8
        
    M, K = x.shape
    N, K2 = weight.shape
    
    # Adjust N so it's a multiple of pool_kernel_size in case it's not (it is 4096 here so it's fine)
    assert N % pool_kernel_size == 0
    POOL_OUT_N = BLOCK_SIZE_N // pool_kernel_size
    
    num_pid_n = triton.cdiv(N, BLOCK_SIZE_N)
    
    out_workspace = torch.empty((M, num_pid_n), device=x.device, dtype=torch.float32)
    
    grid = lambda META: (
        triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
    )
    
    matmul_fused_kernel[grid](
        x, weight, bias, out_workspace,
        M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        out_workspace.stride(0), out_workspace.stride(1),
        scale_factor=scale_factor,
        BLOCK_SIZE_M=BLOCK_SIZE_M,
        BLOCK_SIZE_N=BLOCK_SIZE_N,
        BLOCK_SIZE_K=BLOCK_SIZE_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
        POOL_KERNEL_SIZE=pool_kernel_size,
        POOL_OUT_N=POOL_OUT_N
    )
    
    return torch.max(out_workspace, dim=1).values

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.avg_pool = nn.AvgPool1d(kernel_size=pool_kernel_size)
        self.scale_factor = scale_factor
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        weight = self.matmul.weight
        bias = self.matmul.bias
        return triton_fused_op(x, weight, bias, self.pool_kernel_size, self.scale_factor)
