cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cuda_fp16.h>

// Optimized fused kernel: MatMul + Swish + Scale
// For B200 Blackwell architecture

template<typename T>
__device__ __forceinline__ T sigmoid(T x) {
    return T(1.0f) / (T(1.0f) + exp(-x));
}

template<typename T>
__device__ __forceinline__ T swish(T x) {
    return x * sigmoid(x);
}

// FP32 kernel with optimal thread block configuration
__global__ void fused_matmul_swish_scale_kernel_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    // Use shared memory for weight tiles
    extern __shared__ float shared_mem[];
    float* weight_tile = shared_mem;
    
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    // Each block processes one output tile for one batch element
    float sum = 0.0f;
    
    if (out_idx < out_features) {
        // Compute dot product
        const float* weight_row = weight + out_idx * in_features;
        const float* input_row = input + batch_idx * in_features;
        
        // Vectorized load and compute
        for (int k = 0; k < in_features; k++) {
            sum += input_row[k] * weight_row[k];
        }
        
        // Add bias
        sum += bias[out_idx];
        
        // Apply Swish: x * sigmoid(x)
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig;
        
        // Apply scaling
        sum = sum * scaling_factor;
        
        output[batch_idx * out_features + out_idx] = sum;
    }
}

// Optimized FP32 kernel using warp-level parallelism
__global__ void fused_matmul_swish_scale_kernel_fp32_opt(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x;
    const int warp_id = threadIdx.x / 32;
    const int lane_id = threadIdx.x % 32;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    // Use warp reduction for dot product
    float sum = 0.0f;
    
    const float* weight_row = weight + out_idx * in_features;
    const float* input_row = input + batch_idx * in_features;
    
    // Each thread processes a portion of the dot product
    for (int k = lane_id; k < in_features; k += 32) {
        sum += input_row[k] * weight_row[k];
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        // Add bias
        sum += bias[out_idx];
        
        // Apply Swish: x * sigmoid(x)
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig;
        
        // Apply scaling
        sum = sum * scaling_factor;
        
        output[batch_idx * out_features + out_idx] = sum;
    }
}

// FP16 kernel using half2 for vectorized operations
__global__ void fused_matmul_swish_scale_kernel_fp16(
    const half* __restrict__ input,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    half* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x;
    const int lane_id = threadIdx.x % 32;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    float sum = 0.0f;
    
    const half* weight_row = weight + out_idx * in_features;
    const half* input_row = input + batch_idx * in_features;
    
    // Vectorized half2 load
    const half2* weight_vec = reinterpret_cast<const half2*>(weight_row);
    const half2* input_vec = reinterpret_cast<const half2*>(input_row);
    const int vec_len = in_features / 2;
    
    float2 partial_sum = make_float2(0.0f, 0.0f);
    
    for (int k = lane_id; k < vec_len; k += 32) {
        half2 w = weight_vec[k];
        half2 inp = input_vec[k];
        partial_sum.x += __half2float(w.x) * __half2float(inp.x);
        partial_sum.y += __half2float(w.y) * __half2float(inp.y);
    }
    
    sum = partial_sum.x + partial_sum.y;
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        sum += __half2float(bias[out_idx]);
        
        // Swish activation
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig * scaling_factor;
        
        output[batch_idx * out_features + out_idx] = __float2half(sum);
    }
}

// BF16 kernel
__global__ void fused_matmul_swish_scale_kernel_bf16(
    const nv_bfloat16* __restrict__ input,
    const nv_bfloat16* __restrict__ weight,
    const nv_bfloat16* __restrict__ bias,
    nv_bfloat16* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x;
    const int lane_id = threadIdx.x % 32;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    float sum = 0.0f;
    
    const nv_bfloat16* weight_row = weight + out_idx * in_features;
    const nv_bfloat16* input_row = input + batch_idx * in_features;
    
    for (int k = lane_id; k < in_features; k += 32) {
        sum += __bfloat162float(input_row[k]) * __bfloat162float(weight_row[k]);
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        sum += __bfloat162float(bias[out_idx]);
        
        // Swish activation
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig * scaling_factor;
        
        output[batch_idx * out_features + out_idx] = __float2bfloat16(sum);
    }
}

// High performance kernel using shared memory and better memory access patterns
template<int TILE_M, int TILE_N, int TILE_K>
__global__ void fused_matmul_swish_scale_tiled(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    __shared__ float input_tile[TILE_M][TILE_K];
    __shared__ float weight_tile[TILE_N][TILE_K];
    
    const int batch_idx = blockIdx.y;
    const int out_block = blockIdx.x;
    const int tid = threadIdx.y * blockDim.x + threadIdx.x;
    
    const int out_row = threadIdx.y;
    const int out_col = threadIdx.x;
    
    const int global_out_idx = out_block * TILE_N + out_row;
    
    if (batch_idx >= batch_size) return;
    
    float sum = 0.0f;
    
    const int num_k_tiles = (in_features + TILE_K - 1) / TILE_K;
    
    for (int k_tile = 0; k_tile < num_k_tiles; k_tile++) {
        // Load input tile
        int k_idx = k_tile * TILE_K + out_col;
        if (k_idx < in_features && batch_idx < batch_size) {
            input_tile[out_row][out_col] = input[batch_idx * in_features + k_idx];
        } else {
            input_tile[out_row][out_col] = 0.0f;
        }
        
        // Load weight tile (transposed access)
        int weight_out_idx = out_block * TILE_N + out_row;
        if (k_idx < in_features && weight_out_idx < out_features) {
            weight_tile[out_row][out_col] = weight[weight_out_idx * in_features + k_idx];
        } else {
            weight_tile[out_row][out_col] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial dot product
        for (int k = 0; k < TILE_K; k++) {
            sum += input_tile[out_row][k] * weight_tile[out_col][k];
        }
        
        __syncthreads();
    }
    
    // Write result
    global_out_idx = out_block * TILE_N + out_col;
    if (global_out_idx < out_features && batch_idx < batch_size) {
        sum += bias[global_out_idx];
        
        // Swish activation
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig * scaling_factor;
        
        output[batch_idx * out_features + global_out_idx] = sum;
    }
}

// Optimized batched GEMM style kernel
__global__ void fused_matmul_swish_scale_batched(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_features, int out_features,
    float scaling_factor
) {
    // Each warp processes one output element
    const int warp_size = 32;
    const int warps_per_block = blockDim.x / warp_size;
    const int warp_id = threadIdx.x / warp_size;
    const int lane_id = threadIdx.x % warp_size;
    
    const int batch_warp = blockIdx.y * warps_per_block + warp_id;
    const int batch_idx = batch_warp / out_features;
    const int out_idx = batch_warp % out_features;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    float sum = 0.0f;
    
    const float* weight_row = weight + out_idx * in_features;
    const float* input_row = input + batch_idx * in_features;
    
    // Each lane processes part of the dot product
    for (int k = lane_id; k < in_features; k += warp_size) {
        sum += input_row[k] * weight_row[k];
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = warp_size / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        sum += bias[out_idx];
        
        // Swish activation
        float sig = 1.0f / (1.0f + expf(-sum));
        sum = sum * sig * scaling_factor;
        
        output[batch_idx * out_features + out_idx] = sum;
    }
}

// Main dispatch functions
torch::Tensor fused_matmul_swish_scale_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scaling_factor
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    auto output = torch::empty({batch_size, out_features}, input.options());
    
    // Use optimized batched kernel
    int threads = 256;
    int warps_per_block = threads / 32;
    int total_outputs = batch_size * out_features;
    int blocks_y = (total_outputs + warps_per_block - 1) / warps_per_block;
    
    // Limit blocks in y dimension
    blocks_y = min(blocks_y, 65535);
    
    dim3 grid(1, blocks_y);
    
    fused_matmul_swish_scale_batched<<<grid, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_features, out_features,
        scaling_factor
    );
    
    return output;
}

// Main entry point that handles dtype dispatch
torch::Tensor fused_matmul_swish_scale(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double scaling_factor
) {
    auto dtype = input.scalar_type();
    
    if (dtype == torch::kFloat32) {
        return fused_matmul_swish_scale_fp32(input, weight, bias, static_cast<float>(scaling_factor));
    } else if (dtype == torch::kHalf) {
        return fused_matmul_swish_scale_fp16(input, weight, bias, static_cast<float>(scaling_factor));
    } else if (dtype == torch::kBFloat16) {
        return fused_matmul_swish_scale_bf16(input, weight, bias, static_cast<float>(scaling_factor));
    } else {
        throw std::runtime_error("Unsupported dtype");
    }
}

class Model(nn.Module):
    """
    Optimized model that performs fused matrix multiplication, Swish activation, and scaling.
    Uses custom CUDA kernels for B200 Blackwell architecture.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Create weight and bias parameters (matching nn.Linear format)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize parameters
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_matmul_swish_scale(x, self.weight, self.bias, self.scaling_factor)
    
    def _load_from_state_dict(self, state_dict, prefix, local_metadata, strict, missing_keys, unexpected_keys, error_msgs):
        # Map 'matmul.weight' -> 'weight' and 'matmul.bias' -> 'bias'
        if 'matmul.weight' in state_dict:
            state_dict['weight'] = state_dict.pop('matmul.weight')
        if 'matmul.bias' in state_dict:
            state_dict['bias'] = state_dict.pop('matmul.bias')
        super()._load_from_state_dict(state_dict, prefix, local_metadata, strict, missing_keys, unexpected_keys, error_msgs)
