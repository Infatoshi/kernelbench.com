import torch

from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Warp-level max reduction
__device__ __forceinline__ float warpReduceMax(float val) {
    for (int offset = warpSize/2; offset > 0; offset /= 2) {
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, offset));
    }
    return val;
}

// Warp-level sum reduction
__device__ __forceinline__ float warpReduceSum(float val) {
    for (int offset = warpSize/2; offset > 0; offset /= 2) {
        val = val + __shfl_xor_sync(0xffffffff, val, offset);
    }
    return val;
}

// FP32 softmax kernel with better memory coalescing
template <int THREADS, int ELEMS_PER_THREAD = 8>
__global__ void softmax_forward_kernel_float(
    const float* __restrict__ x,
    float* __restrict__ y,
    int batch_size,
    int dim)
{
    int batch_idx = blockIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const float* x_row = x + batch_idx * dim;
    float* y_row = y + batch_idx * dim;
    
    int tid = threadIdx.x;
    int nWarps = THREADS / 32;
    
    extern __shared__ float smem[];
    float* max_smem = smem;
    float* sum_smem = smem + nWarps;
    
    // Step 1: Find max using vectorized loads
    float thread_max = -INFINITY;
    int base = tid * ELEMS_PER_THREAD;
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        thread_max = fmaxf(thread_max, x_row[base + i]);
    }
    
    float warp_max = warpReduceMax(thread_max);
    
    if (tid % 32 == 0) {
        max_smem[tid / 32] = warp_max;
    }
    __syncthreads();
    
    if (tid < 32) {
        float block_max = (tid < nWarps) ? max_smem[tid] : -INFINITY;
        block_max = warpReduceMax(block_max);
        if (tid == 0) {
            max_smem[0] = block_max;
        }
    }
    __syncthreads();
    
    float row_max = max_smem[0];
    
    // Step 2: Compute exp and sum
    float thread_sum = 0.0f;
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        float val = expf(x_row[base + i] - row_max);
        y_row[base + i] = val;
        thread_sum += val;
    }
    
    float warp_sum = warpReduceSum(thread_sum);
    
    if (tid % 32 == 0) {
        sum_smem[tid / 32] = warp_sum;
    }
    __syncthreads();
    
    if (tid < 32) {
        float block_sum = (tid < nWarps) ? sum_smem[tid] : 0.0f;
        block_sum = warpReduceSum(block_sum);
        if (tid == 0) {
            sum_smem[0] = block_sum;
        }
    }
    __syncthreads();
    
    float row_sum = sum_smem[0];
    float inv_sum = 1.0f / row_sum;
    
    // Step 3: Normalize
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        y_row[base + i] *= inv_sum;
    }
}

// BF16 helper functions
__device__ __forceinline__ float bf16_to_float(uint16_t val) {
    unsigned int u = ((unsigned int)val) << 16;
    return __uint_as_float(u);
}

__device__ __forceinline__ uint16_t float_to_bf16(float val) {
    unsigned int u = __float_as_uint(val);
    return (uint16_t)(u >> 16);
}

// BF16 softmax kernel
template <int THREADS, int ELEMS_PER_THREAD = 8>
__global__ void softmax_forward_kernel_bf16(
    const uint16_t* __restrict__ x,
    uint16_t* __restrict__ y,
    int batch_size,
    int dim)
{
    int batch_idx = blockIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const uint16_t* x_row = x + batch_idx * dim;
    uint16_t* y_row = y + batch_idx * dim;
    
    int tid = threadIdx.x;
    int nWarps = THREADS / 32;
    
    extern __shared__ float smem[];
    float* max_smem = smem;
    float* sum_smem = smem + nWarps;
    
    // Step 1: Find max
    float thread_max = -INFINITY;
    int base = tid * ELEMS_PER_THREAD;
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        thread_max = fmaxf(thread_max, bf16_to_float(x_row[base + i]));
    }
    
    float warp_max = warpReduceMax(thread_max);
    
    if (tid % 32 == 0) {
        max_smem[tid / 32] = warp_max;
    }
    __syncthreads();
    
    if (tid < 32) {
        float block_max = (tid < nWarps) ? max_smem[tid] : -INFINITY;
        block_max = warpReduceMax(block_max);
        if (tid == 0) {
            max_smem[0] = block_max;
        }
    }
    __syncthreads();
    
    float row_max = max_smem[0];
    
    // Step 2: Compute exp and sum
    float thread_sum = 0.0f;
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        float val = expf(bf16_to_float(x_row[base + i]) - row_max);
        y_row[base + i] = float_to_bf16(val);
        thread_sum += val;
    }
    
    float warp_sum = warpReduceSum(thread_sum);
    
    if (tid % 32 == 0) {
        sum_smem[tid / 32] = warp_sum;
    }
    __syncthreads();
    
    if (tid < 32) {
        float block_sum = (tid < nWarps) ? sum_smem[tid] : 0.0f;
        block_sum = warpReduceSum(block_sum);
        if (tid == 0) {
            sum_smem[0] = block_sum;
        }
    }
    __syncthreads();
    
    float row_sum = sum_smem[0];
    float inv_sum = 1.0f / row_sum;
    
    // Step 3: Normalize
    #pragma unroll
    for (int i = 0; i < ELEMS_PER_THREAD && base + i < dim; i++) {
        float val = bf16_to_float(y_row[base + i]) * inv_sum;
        y_row[base + i] = float_to_bf16(val);
    }
}

torch::Tensor softmax_forward_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    
    int batch_size = x.size(0);
    int dim = x.size(1);
    
    // Calculate optimal threads based on dimension size
    // For dim=16384, we want each thread to handle multiple elements
    int threads = (dim >= 8192) ? 512 : 256;
    int elems_per_thread = (dim + threads - 1) / threads;
    // Round up to power of 8 for better vectorization
    elems_per_thread = ((elems_per_thread + 7) / 8) * 8;
    
    const int smem_size = (threads / 32) * 2 * sizeof(float);
    
    if (x.scalar_type() == at::ScalarType::Float) {
        if (elems_per_thread >= 8) {
            softmax_forward_kernel_float<512, 32><<<batch_size, 512, smem_size>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), batch_size, dim);
        } else if (elems_per_thread >= 4) {
            softmax_forward_kernel_float<512, 4><<<batch_size, 512, smem_size>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), batch_size, dim);
        } else {
            softmax_forward_kernel_float<512, 2><<<batch_size, 512, smem_size>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), batch_size, dim);
        }
    } else if (x.scalar_type() == at::ScalarType::Half) {
        if (elems_per_thread >= 8) {
            softmax_forward_kernel_half<512, 8><<<batch_size, 512, smem_size>>>(
                (const __half*)x.data_ptr<at::Half>(),
                (__half*)y.data_ptr<at::Half>(),
                batch_size, dim);
        } else if (elems_per_thread >= 4) {
            softmax_forward_kernel_half<512, 4><<<batch_size, 512, smem_size>>>(
                (const __half*)x.data_ptr<at::Half>(),
                (__half*)y.data_ptr<at::Half>(),
                batch_size, dim);
        } else {
            softmax_forward_kernel_half<512, 2><<<batch_size, 512, smem_size>>>(
                (const __half*)x.data_ptr<at::Half>(),
                (__half*)y.data_ptr<at::Half>(),
                batch_size, dim);
        }
    } else if (x.scalar_type() == at::ScalarType::BFloat16) {
        if (elems_per_thread >= 8) {
            softmax_forward_kernel_bf16<512, 8><<<batch_size, 512, smem_size>>>(
                (const uint16_t*)x.data_ptr<at::BFloat16>(),
                (uint16_t*)y.data_ptr<at::BFloat16>(),
                batch_size, dim);
        } else if (elems_per_thread >= 4) {
            softmax_forward_kernel_bf16<512, 4><<<batch_size, 512, smem_size>>>(
                (const uint16_t*)x.data_ptr<at::BFloat16>(),
                (uint16_t*)y.data_ptr<at::BFloat16>(),
                batch_size, dim);
        } else {
            softmax_forward_kernel_bf16<512, 2><<<batch_size, 512, smem_size>>>(
                (const uint16_t*)x.data_ptr<at::BFloat16>(),
                (uint16_t*)y.data_ptr<at::BFloat16>(),
                batch_size, dim);
        }
    }
    
    return y;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_forward_cuda(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['softmax_forward_cuda'],
    extra_cuda_cflags=['-O3', '-use_fast_math', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Softmax model using CUDA kernel with warp-level reductions.
    """
    def __init__(self):
        super().__init__()
    
    def forward(self, x):
        return ext.softmax_forward_cuda(x)

    return [torch.randn(256, 16384, device='cuda')]
