import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>

// Warp-level reduction for max
__device__ __forceinline__ float warp_reduce_max(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val = max(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

// Warp-level reduction for sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Cross entropy loss kernel
// Each block processes one sample (row)
__global__ void cross_entropy_loss_kernel(
    const float* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    float* __restrict__ output,
    const int num_classes
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    const int lane_id = tid & 31;
    const int warp_id = tid >> 5;
    const int num_warps = blockDim.x >> 5;
    
    // Pointers for this batch item
    const float* row = predictions + batch_idx * num_classes;
    const int target = static_cast<int>(targets[batch_idx]);
    
    // Step 1: Find max value in the row
    float max_val = -FLT_MAX;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        max_val = max(max_val, row[i]);
    }
    
    // Warp-level reduction for max
    max_val = warp_reduce_max(max_val);
    
    // Shared memory for cross-warp reduction
    __shared__ float shared_max[32];
    if (lane_id == 0) {
        shared_max[warp_id] = max_val;
    }
    __syncthreads();
    
    // First warp reduces the warp maxes
    if (warp_id == 0) {
        max_val = (tid < num_warps) ? shared_max[tid] : -FLT_MAX;
        max_val = warp_reduce_max(max_val);
        if (tid == 0) {
            shared_max[0] = max_val;
        }
    }
    __syncthreads();
    max_val = shared_max[0];
    
    // Step 2: Compute exp(x - max) sum
    float sum_exp = 0.0f;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        sum_exp += expf(row[i] - max_val);
    }
    
    // Warp-level reduction for sum
    sum_exp = warp_reduce_sum(sum_exp);
    
    // Shared memory for cross-warp reduction
    __shared__ float shared_sum[32];
    if (lane_id == 0) {
        shared_sum[warp_id] = sum_exp;
    }
    __syncthreads();
    
    // First warp reduces the warp sums
    if (warp_id == 0) {
        sum_exp = (tid < num_warps) ? shared_sum[tid] : 0.0f;
        sum_exp = warp_reduce_sum(sum_exp);
        if (tid == 0) {
            shared_sum[0] = sum_exp;
        }
    }
    __syncthreads();
    sum_exp = shared_sum[0];
    
    // Step 3: Compute log_sum_exp
    float log_sum_exp = logf(sum_exp) + max_val;
    
    // Step 4: Get target value and compute loss
    // Only one thread needs to do this
    if (tid == 0) {
        float target_val = row[target];
        float loss = -(target_val - log_sum_exp);
        output[batch_idx] = loss;
    }
}

// Final reduction kernel to sum all losses
__global__ void sum_losses_kernel(
    const float* __restrict__ losses,
    float* __restrict__ output,
    const int n
) {
    float sum = 0.0f;
    for (int i = blockIdx.x * blockDim.x + threadIdx.x; i < n; i += blockDim.x * gridDim.x) {
        sum += losses[i];
    }
    
    // Warp-level reduction
    sum = warp_reduce_sum(sum);
    
    __shared__ float shared_sum[32];
    const int lane_id = threadIdx.x & 31;
    const int warp_id = threadIdx.x >> 5;
    
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    __syncthreads();
    
    const int num_warps = blockDim.x >> 5;
    if (warp_id == 0) {
        sum = (threadIdx.x < num_warps) ? shared_sum[threadIdx.x] : 0.0f;
        sum = warp_reduce_sum(sum);
        if (threadIdx.x == 0) {
            atomicAdd(output, sum);
        }
    }
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    // Ensure contiguous and correct type
    auto pred_contiguous = predictions.contiguous();
    auto targets_contiguous = targets.contiguous();
    
    const int batch_size = predictions.size(0);
    const int num_classes = predictions.size(1);
    
    // Allocate intermediate buffer for per-sample losses
    auto losses = torch::empty({batch_size}, predictions.options());
    auto output = torch::zeros({1}, predictions.options());
    
    const int threads_per_block = 256;
    const int blocks = batch_size;
    
    cross_entropy_loss_kernel<<<blocks, threads_per_block>>>(
        pred_contiguous.data_ptr<float>(),
        targets_contiguous.data_ptr<int64_t>(),
        losses.data_ptr<float>(),
        num_classes
    );
    
    // Sum all losses
    const int reduce_threads = 256;
    const int reduce_blocks = min((batch_size + reduce_threads - 1) / reduce_threads, 1024);
    
    sum_losses_kernel<<<reduce_blocks, reduce_threads>>>(
        losses.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size
    );
    
    // Divide by batch size for mean
    return output / batch_size;
}
'''

cpp_source = r'''
torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets);
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
    extra_cuda_cflags=['-O3'],
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_loss(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [torch.randn(batch_size, num_classes, device='cuda'), 
            torch.randint(0, num_classes, (batch_size,), device='cuda')]

def get_init_inputs():
    return []
