import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    dropout_p, seed,
    is_training: tl.constexpr,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    col_acc = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
    bias = tl.load(B_ptr + offs_bn, mask=offs_bn < N, other=0.0).to(tl.float32)

    for m_start in range(0, M, BLOCK_SIZE_M):
        offs_am = m_start + tl.arange(0, BLOCK_SIZE_M)
        acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
        
        for k_start in range(0, K, BLOCK_SIZE_K):
            offs_k = k_start + tl.arange(0, BLOCK_SIZE_K)
            x = tl.load(X_ptr + offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk, 
                        mask=(offs_am[:, None] < M) & (offs_k[None, :] < K), other=0.0)
            w = tl.load(W_ptr + offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk,
                        mask=(offs_bn[None, :] < N) & (offs_k[:, None] < K), other=0.0)
            acc += tl.dot(x, w)
            
        acc += bias[None, :]
        if is_training and dropout_p > 0.0:
            m_idx_global = offs_am[:, None]
            n_idx_global = offs_bn[None, :]
            random = tl.rand(seed, m_idx_global * N + n_idx_global)
            keep_mask = random > dropout_p
            acc = tl.where(keep_mask, acc / (1.0 - dropout_p), 0.0)
        
        col_acc += tl.sum(acc, axis=0)

    mean_val = col_acc / M
    tl.store(Out_ptr + offs_bn, mean_val, mask=offs_bn < N)

@triton.jit
def softmax_kernel(
    Out_ptr,
    N,
    BLOCK_SIZE_N: tl.constexpr,
):
    offs = tl.arange(0, BLOCK_SIZE_N)
    mask = offs < N
    x = tl.load(Out_ptr + offs, mask=mask, other=-float('inf'))
    x_max = tl.max(x, axis=0)
    safe_x = x - x_max
    exp_x = tl.exp(safe_x)
    sum_exp = tl.sum(exp_x, axis=0)
    res = exp_x / sum_exp
    tl.store(Out_ptr + offs, res.to(Out_ptr.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout_p = dropout_p
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        seed = torch.randint(0, 1000000, (1,)).item()
        out = torch.empty((1, N), device=x.device, dtype=torch.float32)
        
        grid = lambda META: (triton.cdiv(N, META['BLOCK_SIZE_N']), )
        fused_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            self.dropout_p, seed,
            is_training=self.training,
            BLOCK_SIZE_M=64, 
            BLOCK_SIZE_N=128, 
            BLOCK_SIZE_K=64,
            num_warps=8,
            num_stages=3
        )
        
        softmax_kernel[(1,)](
            out, N,
            BLOCK_SIZE_N=triton.next_power_of_2(N),
            num_warps=16
        )
        
        return out.to(x.dtype)

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 0.2]
