import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// SiLU/Swish activation: x * tanh(softplus(x)) = x * tanh(log(1 + exp(x)))

template<typename scalar_t>
__global__ void fused_conv_silu_bn_kernel(
    const scalar_t* input,      // [N, C, H, W]
    const scalar_t* weight,     // [out_ch, in_ch, kH, kW]
    const scalar_t* bias,       // [out_ch]
    const scalar_t* bn_weight,  // [out_ch]
    const scalar_t* bn_bias,    // [out_ch]
    const scalar_t* bn_running_mean,  // [out_ch]
    const scalar_t* bn_running_var,   // [out_ch]
    scalar_t* output,           // [N, out_ch, H_out, W_out]
    int N, int C, int H, int W,
    int out_ch, int kH, int kW,
    int H_out, int W_out,
    float eps
) {
    int n = blockIdx.z;
    int oc = blockIdx.y;
    int oh = blockIdx.x / W_out;
    int ow = blockIdx.x % W_out;
    
    float acc = 0.0f;
    
    // Direct 3x3 convolution (no padding in kernel, padding handled by input)
    for (int ic = 0; ic < C; ic++) {
        for (int kh = 0; kh < kH; kh++) {
            for (int kw = 0; kw < kW; kw++) {
                int ih = oh + kh;
                int iw = ow + kw;
                if (ih >= 0 && ih < H && iw >= 0 && iw < W) {
                    int in_idx = ((n * C + ic) * H + ih) * W + iw;
                    int w_idx = ((oc * C + ic) * kH + kh) * kW + kw;
                    acc += (float)input[in_idx] * (float)weight[w_idx];
                }
            }
        }
    }
    
    acc += (float)bias[oc];
    
    // Apply SiLU activation: x * tanh(softplus(x))
    float softplus_acc = acc > 20.0f ? acc : log1pf(expf(acc));
    float tanh_softplus = tanhf(softplus_acc);
    float activated = acc * tanh_softplus;
    
    // Apply BatchNorm: y = gamma * (x - mean) / sqrt(var + eps) + beta
    float mean = (float)bn_running_mean[oc];
    float var = (float)bn_running_var[oc];
    float bn_weight_val = (float)bn_weight[oc];
    float bn_bias_val = (float)bn_bias[oc];
    
    float normalized = (activated - mean) * rsqrtf(var + eps);
    float bn_out = normalized * bn_weight_val + bn_bias_val;
    
    int out_idx = ((n * out_ch + oc) * H_out + oh) * W_out + ow;
    output[out_idx] = (scalar_t)bn_out;
}

torch::Tensor fused_conv_silu_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    float eps
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    int out_ch = weight.size(0);
    int kH = weight.size(2);
    int kW = weight.size(3);
    
    int H_out = H - kH + 1;
    int W_out = W - kW + 1;
    
    auto output = torch::empty({N, out_ch, H_out, W_out}, input.options());
    
    dim3 block(256);
    dim3 grid(W_out * H_out, out_ch, N);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "fused_conv_silu_bn", ([&] {
        if (input.scalar_type() == torch::kHalf) {
            fused_conv_silu_bn_kernel<at::Half><<<grid, block>>>(
                input.data_ptr<at::Half>(),
                weight.data_ptr<at::Half>(),
                bias.data_ptr<at::Half>(),
                bn_weight.data_ptr<at::Half>(),
                bn_bias.data_ptr<at::Half>(),
                bn_running_mean.data_ptr<at::Half>(),
                bn_running_var.data_ptr<at::Half>(),
                output.data_ptr<at::Half>(),
                N, C, H, W, out_ch, kH, kW, H_out, W_out, eps
            );
        } else if (input.scalar_type() == torch::kBFloat16) {
            fused_conv_silu_bn_kernel<at::BFloat16><<<grid, block>>>(
                input.data_ptr<at::BFloat16>(),
                weight.data_ptr<at::BFloat16>(),
                bias.data_ptr<at::BFloat16>(),
                bn_weight.data_ptr<at::BFloat16>(),
                bn_bias.data_ptr<at::BFloat16>(),
                bn_running_mean.data_ptr<at::BFloat16>(),
                bn_running_var.data_ptr<at::BFloat16>(),
                output.data_ptr<at::BFloat16>(),
                N, C, H, W, out_ch, kH, kW, H_out, W_out, eps
            );
        } else {
            fused_conv_silu_bn_kernel<float><<<grid, block>>>(
                input.data_ptr<float>(),
                weight.data_ptr<float>(),
                bias.data_ptr<float>(),
                bn_weight.data_ptr<float>(),
                bn_bias.data_ptr<float>(),
                bn_running_mean.data_ptr<float>(),
                bn_running_var.data_ptr<float>(),
                output.data_ptr<float>(),
                N, C, H, W, out_ch, kH, kW, H_out, W_out, eps
            );
        }
    }));
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources='''
#include <torch/extension.h>
torch::Tensor fused_conv_silu_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    float eps
);
''',
    cuda_sources=cuda_source,
    functions=['fused_conv_silu_bn'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Fused Conv2d + SiLU activation + BatchNorm
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=kernel_size//2)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        # Get the running stats from batch norm
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        
        # Use fused kernel
        return ext.fused_conv_silu_bn(
            x,
            self.conv.weight,
            self.conv.bias,
            self.bn.weight,
            self.bn.bias,
            running_mean,
            running_var,
            self.eps
        )


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    return [in_channels, out_channels, kernel_size]
