import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void moe_kernel(
    const float* __restrict__ x,
    const int* __restrict__ expert_idx,
    const float* __restrict__ expert_w,
    const float* __restrict__ up,
    const float* __restrict__ down,
    float* __restrict__ out,
    int T, int H, int E, int topk, int expert_dim)
{
    int t = blockIdx.x;
    int h = threadIdx.x;
    if (t >= T || h >= H) return;

    float acc = 0.0f;

    for (int k = 0; k < topk; ++k) {
        int e = expert_idx[t * topk + k];
        float w = expert_w[t * topk + k];

        float tmp = 0.0f;
        for (int j = 0; j < expert_dim; ++j) {
            float up_acc = 0.0f;
            for (int i = 0; i < H; ++i) {
                float xv = x[t * H + i];
                float uv = up[e * H * expert_dim + i * expert_dim + j];
                up_acc += xv * uv;
            }
            float dv = down[e * expert_dim * H + j * H + h];
            tmp += up_acc * dv;
        }
        acc += w * tmp;
    }

    out[t * H + h] = acc;
}

torch::Tensor moe_forward(
    torch::Tensor x,
    torch::Tensor expert_idx,
    torch::Tensor expert_w,
    torch::Tensor up,
    torch::Tensor down)
{
    int B = x.size(0);
    int S = x.size(1);
    int H = x.size(2);
    int T = B * S;
    int topk = expert_idx.size(2);
    int E = up.size(0);
    int expert_dim = up.size(2);

    auto x_flat = x.reshape({T, H}).contiguous();
    auto idx_flat = expert_idx.reshape({T, topk}).contiguous();
    auto w_flat = expert_w.reshape({T, topk}).contiguous();

    auto out = torch::zeros_like(x_flat);

    dim3 grid(T);
    dim3 block(H);

    moe_kernel<<<grid, block>>>(
        x_flat.data_ptr<float>(),
        idx_flat.data_ptr<int>(),
        w_flat.data_ptr<float>(),
        up.data_ptr<float>(),
        down.data_ptr<float>(),
        out.data_ptr<float>(),
        T, H, E, topk, expert_dim
    );

    return out.reshape({B, S, H});
}
'''

ext = load_inline(
    name='moe_ext',
    cpp_sources='torch::Tensor moe_forward(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_src,
    functions=['moe_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_experts: int = 8, hidden_dim: int = 1024, expert_dim: int = 3072):
        super().__init__()
        self.num_experts = num_experts
        self.hidden_dim = hidden_dim
        self.expert_dim = expert_dim
        self.expert_up = nn.Parameter(torch.randn(num_experts, hidden_dim, expert_dim) * 0.02)
        self.expert_down = nn.Parameter(torch.randn(num_experts, expert_dim, hidden_dim) * 0.02)

    def forward(self, x, expert_indices, expert_weights):
        return ext.moe_forward(x, expert_indices.int(), expert_weights, self.expert_up, self.expert_down)


def get_inputs():
    batch, seq, hidden = 4, 256, 1024
    top_k = 2
    num_experts = 8

    x = torch.randn(batch, seq, hidden, device='cuda')
    expert_indices = torch.randint(0, num_experts, (batch, seq, top_k), device='cuda')
    expert_weights = torch.softmax(torch.randn(batch, seq, top_k, device='cuda'), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return []
