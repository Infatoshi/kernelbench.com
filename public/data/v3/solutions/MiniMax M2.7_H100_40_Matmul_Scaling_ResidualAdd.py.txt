import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

// cuBLAS handle
static cublasHandle_t cublas_handle = nullptr;

void init_cublas() {
    if (!cublas_handle) {
        cublasCreate(&cublas_handle);
    }
}

// Fused scale + residual add kernel for fp16
__global__ void fused_scale_residual_add_kernel_half(
    const __half* __restrict__ matmul_out,
    __half* __restrict__ output,
    int num_elements,
    __half scale_factor) {
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = gridDim.x * blockDim.x;
    
    __half2 scale_half2 = __half2half2(scale_factor);
    __half2 scale_plus_one = __hadd2(scale_half2, __half2half2(__float2half(1.0f)));
    
    for (int i = tid * 4; i < num_elements; i += total_threads * 4) {
        __half2 vals[2];
        
        if (i + 7 < num_elements) {
            vals[0] = *reinterpret_cast<const __half2*>(&matmul_out[i]);
            vals[1] = *reinterpret_cast<const __half2*>(&matmul_out[i + 4]);
        } else {
            for (int j = 0; j < 4 && i + j < num_elements; j++) {
                ((__half*)vals)[j] = matmul_out[i + j];
            }
        }
        
        #pragma unroll
        for (int j = 0; j < 2 && i + j * 4 < num_elements; j++) {
            *reinterpret_cast<__half2*>(&output[i + j * 4]) = __hmul2(vals[j], scale_plus_one);
        }
    }
}

// Kernel for fp32
__global__ void fused_scale_residual_add_kernel_float(
    const float* __restrict__ matmul_out,
    float* __restrict__ output,
    int num_elements,
    float scale_factor) {
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = gridDim.x * blockDim.x;
    float scale_plus_one = scale_factor + 1.0f;
    
    for (int i = tid * 4; i < num_elements; i += total_threads * 4) {
        float4 vals;
        
        if (i + 3 < num_elements) {
            vals = *reinterpret_cast<const float4*>(&matmul_out[i]);
            vals.x *= scale_plus_one;
            vals.y *= scale_plus_one;
            vals.z *= scale_plus_one;
            vals.w *= scale_plus_one;
            *reinterpret_cast<float4*>(&output[i]) = vals;
        } else {
            for (int j = 0; j < 4 && i + j < num_elements; j++) {
                output[i + j] = matmul_out[i + j] * scale_plus_one;
            }
        }
    }
}

// Bias add kernel for fp16
__global__ void add_bias_kernel_half(
    const __half* __restrict__ matmul_out,
    const __half* __restrict__ bias,
    __half* __restrict__ output,
    int num_elements) {
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = gridDim.x * blockDim.x;
    
    for (int i = tid * 2; i < num_elements; i += total_threads * 2) {
        __half2 mat_vals = *reinterpret_cast<const __half2*>(&matmul_out[i]);
        __half2 bias_vals = *reinterpret_cast<const __half2*>(&bias[i]);
        *reinterpret_cast<__half2*>(&output[i]) = __hadd2(mat_vals, bias_vals);
    }
}

// Bias add kernel for fp32
__global__ void add_bias_kernel_float(
    const float* __restrict__ matmul_out,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int num_elements) {
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_threads = gridDim.x * blockDim.x;
    
    for (int i = tid * 4; i < num_elements; i += total_threads * 4) {
        float4 mat_vals = *reinterpret_cast<const float4*>(&matmul_out[i]);
        float4 bias_vals = *reinterpret_cast<const float4*>(&bias[i]);
        float4 result = {mat_vals.x + bias_vals.x, mat_vals.y + bias_vals.y,
                         mat_vals.z + bias_vals.z, mat_vals.w + bias_vals.w};
        *reinterpret_cast<float4*>(&output[i]) = result;
    }
}

torch::Tensor fused_matmul_bias_scale_residual_fp16(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor) {
    
    const int M = input.size(0);
    const int K = input.size(1);
    const int N = weight.size(0);
    
    auto output = torch::empty({M, N}, input.options());
    
    const __half* input_ptr = reinterpret_cast<const __half*>(input.data_ptr<torch::Half>());
    const __half* weight_ptr = reinterpret_cast<const __half*>(weight.data_ptr<torch::Half>());
    const __half* bias_ptr = reinterpret_cast<const __half*>(bias.data_ptr<torch::Half>());
    __half* output_ptr = reinterpret_cast<__half*>(output.data_ptr<torch::Half>());
    
    init_cublas();
    
    __half alpha = __float2half(1.0f);
    __half beta = __float2half(0.0f);
    
    // GEMM: output = input @ weight^T
    cublasStatus_t status = cublasGemmEx(
        cublas_handle,
        CUBLAS_OP_T,  // weight is transposed
        CUBLAS_OP_N,
        N, M, K,
        &alpha,
        weight_ptr, CUDA_R_16F, N,
        input_ptr, CUDA_R_16F, K,
        &beta,
        output_ptr, CUDA_R_16F, N,
        CUDA_R_16F,
        CUBLAS_GEMM_DEFAULT_TENSOR_OP
    );
    
    // Add bias (broadcast along rows)
    const int total_elements = M * N;
    const int threads = 256;
    
    // Use kernel with half2 (2 elements per thread)
    int blocks_half = (total_elements + threads * 2 - 1) / (threads * 2);
    add_bias_kernel_half<<<blocks_half, threads>>>(
        output_ptr, bias_ptr, output_ptr, total_elements);
    
    // Fused scale + residual: output = output * (scale + 1)
    int blocks_scale = (total_elements + threads * 4 - 1) / (threads * 4);
    __half scale_half = __float2half(scale_factor);
    
    fused_scale_residual_add_kernel_half<<<blocks_scale, threads>>>(
        output_ptr, output_ptr, total_elements, scale_half);
    
    return output;
}

torch::Tensor fused_matmul_bias_scale_residual_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor) {
    
    const int M = input.size(0);
    const int K = input.size(1);
    const int N = weight.size(0);
    
    auto output = torch::empty({M, N}, input.options());
    
    const float* input_ptr = input.data_ptr<float>();
    const float* weight_ptr = weight.data_ptr<float>();
    const float* bias_ptr = bias.data_ptr<float>();
    float* output_ptr = output.data_ptr<float>();
    
    init_cublas();
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    // GEMM: output = input @ weight^T
    cublasStatus_t status = cublasSgemm(
        cublas_handle,
        CUBLAS_OP_T,
        CUBLAS_OP_N,
        N, M, K,
        &alpha,
        weight_ptr, K,
        input_ptr, K,
        &beta,
        output_ptr, N
    );
    
    // Add bias
    const int total_elements = M * N;
    const int threads = 256;
    
    int blocks_float = (total_elements + threads * 4 - 1) / (threads * 4);
    add_bias_kernel_float<<<blocks_float, threads>>>(
        output_ptr, bias_ptr, output_ptr, total_elements);
    
    // Fused scale + residual
    int blocks_scale = (total_elements + threads * 4 - 1) / (threads * 4);
    
    fused_scale_residual_add_kernel_float<<<blocks_scale, threads>>>(
        output_ptr, output_ptr, total_elements, scale_factor);
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_matmul_bias_scale_residual_fp16(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor);

torch::Tensor fused_matmul_bias_scale_residual_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor);
'''

ext = load_inline(
    name='matmul_scale_add_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=[
        'fused_matmul_bias_scale_residual_fp16',
        'fused_matmul_bias_scale_residual_fp32'
    ],
    extra_include_paths=['/usr/local/cuda/include'],
    extra_cuda_cflags=['-arch=sm_90', '-std=c++17'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))
        
        # Initialize with reference model's state
        torch.nn.init.kaiming_uniform_(self.weight, a=0)
        torch.nn.init.zeros_(self.bias)

    def forward(self, x):
        if x.dtype == torch.float16:
            return ext.fused_matmul_bias_scale_residual_fp16(
                x, self.weight, self.bias, self.scaling_factor
            )
        elif x.dtype == torch.bfloat16:
            # Convert to fp16 for computation, then back
            x_f16 = x.to(torch.float16)
            out_f16 = ext.fused_matmul_bias_scale_residual_fp16(
                x_f16, self.weight.to(torch.float16), self.bias.to(torch.float16), self.scaling_factor
            )
            return out_f16.to(torch.bfloat16)
        else:
            return ext.fused_matmul_bias_scale_residual_fp32(
                x, self.weight, self.bias, self.scaling_factor
            )

def get_inputs():
    return [torch.randn(128, 4096, device='cuda', dtype=torch.float16)]

def get_init_inputs():
    return [4096, 4096, 0.5]
