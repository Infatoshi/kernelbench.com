import torch
import torch.nn as nn
import triton
import triton.language as tl

# Enable TF32 for faster matmul
torch.backends.cuda.matmul.allow_tf32 = True
torch.backends.cudnn.allow_tf32 = True


@triton.jit
def fused_swish_bias_groupnorm_kernel(
    X_ptr,          # input: (B, C) after linear
    BIAS_ptr,       # bias to add: (C,)
    GN_WEIGHT_ptr,  # groupnorm weight: (C,)
    GN_BIAS_ptr,    # groupnorm bias: (C,)
    OUT_ptr,        # output: (B, C)
    B: tl.constexpr,
    C: tl.constexpr,
    num_groups: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    eps: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // num_groups
    group_idx = pid % num_groups
    
    group_start = group_idx * GROUP_SIZE
    offs = tl.arange(0, GROUP_SIZE)
    x_ptrs = X_ptr + batch_idx * C + group_start + offs
    
    x = tl.load(x_ptrs).to(tl.float32)
    
    # Swish: sigmoid(x) * x
    sigmoid_x = tl.sigmoid(x)
    x = sigmoid_x * x
    
    # Add bias
    bias = tl.load(BIAS_ptr + group_start + offs).to(tl.float32)
    x = x + bias
    
    # GroupNorm
    mean = tl.sum(x, axis=0) / GROUP_SIZE
    x_centered = x - mean
    var = tl.sum(x_centered * x_centered, axis=0) / GROUP_SIZE
    inv_std = tl.rsqrt(var + eps)
    x_norm = x_centered * inv_std
    
    # Apply affine
    gn_weight = tl.load(GN_WEIGHT_ptr + group_start + offs).to(tl.float32)
    gn_bias = tl.load(GN_BIAS_ptr + group_start + offs).to(tl.float32)
    
    out = x_norm * gn_weight + gn_bias
    
    out_ptrs = OUT_ptr + batch_idx * C + group_start + offs
    tl.store(out_ptrs, out)


class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.num_groups = num_groups
        self.out_features = out_features
        self.in_features = in_features

    def forward(self, x):
        B = x.shape[0]
        C = self.out_features
        group_size = C // self.num_groups
        
        # Linear using addmm (TF32 enabled for speed)
        x = torch.addmm(self.matmul.bias, x, self.matmul.weight.t())
        
        # Allocate output
        out = torch.empty_like(x)
        
        # Launch fused kernel
        grid = (B * self.num_groups,)
        fused_swish_bias_groupnorm_kernel[grid](
            x,
            self.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            out,
            B=B,
            C=C,
            num_groups=self.num_groups,
            GROUP_SIZE=group_size,
            eps=self.group_norm.eps,
        )
        
        return out


batch_size = 128
in_features = 4096
out_features = 4096
num_groups = 32
bias_shape = (out_features,)

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, num_groups, bias_shape]
