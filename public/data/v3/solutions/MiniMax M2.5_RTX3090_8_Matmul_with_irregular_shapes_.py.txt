import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# WMMA-based GEMM using Tensor Cores for RTX 3090 (Ampere)
# Uses m16n8k16 tile size for FP16 Tensor Core operations
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda;

// Tile sizes for Tensor Core operations
#define WMMA_M 16
#define WMMA_N 8  
#define WMMA_K 16

#define BLOCK_M 128
#define BLOCK_N 128
#define BLOCK_K 32

// Number of warps in a block
#define WARPS_M (BLOCK_M / 32)
#define WARPS_N (BLOCK_N / 32)

// Shared memory layout for double buffering
__global__ void wmma_gemm_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    float* __restrict__ C,
    int M, int K, int N
) {
    // Shared memory for double buffering
    __shared__ half As[2][BLOCK_K * BLOCK_M];
    __shared__ half Bs[2][BLOCK_K * BLOCK_N];
    
    // Warp and thread indices
    const int warpId = threadIdx.x / 32;
    const int laneId = threadIdx.x % 32;
    
    // Block indices
    const int blockM = blockIdx.y;
    const int blockN = blockIdx.x;
    
    // Global offset for this block
    const int globalM = blockM * BLOCK_M;
    const int globalN = blockN * BLOCK_N;
    
    // Allocate accumulator fragments
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> b_frag;
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> c_frag;
    
    // Initialize accumulator to zero
    wmma::fill_fragment(c_frag, 0.0f);
    
    // Number of k-loop iterations
    const int numKBlocks = (K + BLOCK_K - 1) / BLOCK_K;
    
    // Double buffering control
    int bufId = 0;
    
    // First load into shared memory
    for (int k = 0; k < numKBlocks; ++k) {
        // Load A tile into shared memory
        #pragma unroll
        for (int i = 0; i < WMMA_M; ++i) {
            int row = warpId * WMMA_M + i;
            int col = 0;
            
            // Each thread loads multiple elements
            #pragma unroll
            for (int j = 0; j < 2; ++j) {
                int colIdx = laneId / 16 + j * 2;
                if (colIdx < BLOCK_K / 8) {
                    int globalRow = globalM + row;
                    int globalCol = k * BLOCK_K + colIdx * 8;
                    if (globalRow < M && globalCol < K) {
                        As[bufId][row * BLOCK_K + colIdx * 8 + (laneId % 16)] = 
                            A[globalRow * K + globalCol + (laneId % 16)];
                    }
                }
            }
        }
        
        // Load B tile into shared memory
        #pragma unroll
        for (int i = 0; i < WMMA_N / 2; ++i) {
            int row = warpId * (WMMA_N / 2) + i;
            int col = 0;
            
            #pragma unroll
            for (int j = 0; j < 4; ++j) {
                int colIdx = laneId / 8 + j;
                if (colIdx < BLOCK_K / 8) {
                    int globalRow = globalN + row;
                    int globalCol = k * BLOCK_K + colIdx * 8;
                    if (globalRow < N && globalCol < K) {
                        Bs[bufId][row * BLOCK_K + colIdx * 8 + (laneId % 8)] = 
                            B[globalRow * K + globalCol + (laneId % 8)];
                    }
                }
            }
        }
        
        __syncthreads();
        
        // Compute for this k-tile
        // Each warp computes WMMA_M x WMMA_N output block
        int warpM = warpId / WARPS_N;
        int warpN = warpId % WARPS_N;
        
        for (int kk = 0; kk < BLOCK_K / WMMA_K; ++kk) {
            // Load A fragment
            wmma::load_matrix_sync(a_frag, 
                As[bufId] + (warpM * WMMA_M) * BLOCK_K + kk * WMMA_K, 
                BLOCK_K);
            
            // Load B fragment  
            wmma::load_matrix_sync(b_frag,
                Bs[bufId] + (warpN * WMMA_N) * BLOCK_K + kk * WMMA_K,
                BLOCK_K);
            
            // Matrix multiply
            wmma::mma_sync(c_frag, a_frag, b_frag, c_frag);
        }
        
        // Toggle buffer
        bufId = 1 - bufId;
    }
    
    // Store result
    int storeRow = warpM * WMMA_M;
    int storeCol = warpN * WMMA_N;
    
    wmma::store_matrix_sync(
        C + (globalM + storeRow) * N + globalN + storeCol,
        c_frag, N, wmma::mem_row_major);
}

// Wrapper function
torch::Tensor wmma_gemm(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(1);
    
    // Convert to half if needed
    auto A_half = A.to(torch::kHalf);
    auto B_half = B.to(torch::kHalf);
    
    // Output in float for accuracy
    auto C = torch::zeros({M, N}, A.options().dtype(torch::kFloat));
    
    // Calculate grid dimensions
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);
    dim3 block(WARPS_M * WARPS_N * 32);
    
    wmma_gemm_kernel<<<grid, block>>>(
        A_half.data_ptr<half>(),
        B_half.data_ptr<half>(),
        C.data_ptr<float>(),
        M, K, N
    );
    
    return C;
}
'''

ext = load_inline(
    name='wmma_gemm_ext',
    cpp_sources='torch::Tensor wmma_gemm(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['wmma_gemm'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix multiplication using WMMA Tensor Cores.
        """
        return ext.wmma_gemm(A, B)

M = 8205
K = 2949
N = 5921

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, N)
    return [A, B]

def get_init_inputs():
    return []
