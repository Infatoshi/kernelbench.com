import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_conv_kernel(
    x_ptr, weight_ptr, bias_ptr,
    y_ptr,
    BATCH, IN_C, IN_H, IN_W,
    OUT_C, OUT_H, OUT_W,
    K_H, K_W,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_yn, stride_yc, stride_yh, stride_yw,
    BLOCK_SIZE_H: tl.constexpr, BLOCK_SIZE_W: tl.constexpr
):
    """
    Fused Convolution + BatchNorm + Scaling Kernel.
    Operates on a tile (BLOCK_SIZE_H x BLOCK_SIZE_W) of the output per thread block.
    """
    c_out = tl.program_id(0)
    batch_idx = tl.program_id(1)
    hw_block = tl.program_id(2)
    
    num_h_blocks = tl.cdiv(OUT_H, BLOCK_SIZE_H)
    h_block = hw_block % num_h_blocks
    w_block = hw_block // num_h_blocks
    
    h_start = h_block * BLOCK_SIZE_H
    w_start = w_block * BLOCK_SIZE_W
    
    y_h_idx = h_start + tl.arange(0, BLOCK_SIZE_H)
    y_w_idx = w_start + tl.arange(0, BLOCK_SIZE_W)
    
    # Load precomputed fused bias once per output channel
    bias = tl.load(bias_ptr + c_out)

    # Temporary accumulator for the output tile
    acc = tl.zeros((BLOCK_SIZE_H, BLOCK_SIZE_W), dtype=tl.float32)
    
    # Direct convolution loop over input channels and kernel window
    for ci in range(IN_C):
        for kh in range(K_H):
            for kw in range(K_W):
                # Load weight value for the specific filter and spatial tap
                w_val = tl.load(weight_ptr + c_out * (IN_C * K_H * K_W) + ci * (K_H * K_W) + kh * K_W + kw)
                
                # Corresponding input spatial locations
                h_in = y_h_idx[:, None] + kh
                w_in = y_w_idx[None, :] + kw
                
                x_idx = (batch_idx * stride_xn + 
                         ci * stride_xc + 
                         h_in * stride_xh + 
                         w_in * stride_xw)
                
                mask_x = (h_in < IN_H) & (w_in < IN_W)
                x_val = tl.load(x_ptr + x_idx, mask=mask_x, other=0.0)
                acc += x_val * w_val
    
    # Add precomputed bias
    res = acc + bias
    
    # Define output indices and write with mask
    y_idx = (batch_idx * stride_yn + 
             c_out * stride_yc + 
             y_h_idx[:, None] * stride_yh + 
             y_w_idx[None, :] * stride_yw)
    
    mask_y = (y_h_idx[:, None] < OUT_H) & (y_w_idx[None, :] < OUT_W)
    tl.store(y_ptr + y_idx, res, mask=mask_y)

class Model(nn.Module):
    """
    Fused Conv2d + BatchNorm2d + Scaling factor implementation in Triton.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = float(scaling_factor)
        self.in_channels = in_channels
        self.out_channels = out_channels
        if isinstance(kernel_size, int):
            self.kernel_size = (kernel_size, kernel_size)
        else:
            self.kernel_size = kernel_size

    def forward(self, x):
        N, IC, IH, IW = x.shape
        OC, _, KH, KW = self.out_channels, self.in_channels, self.kernel_size[0], self.kernel_size[1]
        OH, OW = IH - KH + 1, IW - KW + 1

        # Fuse weights and biases only once for inference (eval mode)
        # In a real scenario, we might want to cache these.
        with torch.no_grad():
            w = self.conv.weight
            b = self.conv.bias if self.conv.bias is not None else torch.zeros(OC, device=x.device, dtype=x.dtype)
            mean = self.bn.running_mean
            var = self.bn.running_var
            gamma = self.bn.weight
            beta = self.bn.bias
            eps = self.bn.eps
            std = torch.sqrt(var + eps)
            
            # Fusing arithmetic:
            # Output = (((Input * W + b) - mean) / std * gamma + beta) * scale
            #        = Input * (W * gamma / std * scale) + ((b - mean) * gamma / std + beta) * scale
            f_w = w * (gamma / std * self.scaling_factor).view(-1, 1, 1, 1)
            f_b = (beta + (b - mean) * (gamma / std)) * self.scaling_factor

        y = torch.empty((N, OC, OH, OW), device=x.device, dtype=x.dtype)
        
        # Grid parameters: parallelize across Batch, Output Channels, and Spatial tiling
        BLOCK_SIZE_H = 16
        BLOCK_SIZE_W = 16
        
        num_h_blocks = (OH + BLOCK_SIZE_H - 1) // BLOCK_SIZE_H
        num_w_blocks = (OW + BLOCK_SIZE_W - 1) // BLOCK_SIZE_W
        grid = (OC, N, num_h_blocks * num_w_blocks)
        
        fused_conv_kernel[grid](
            x, f_w.contiguous(), f_b.contiguous(),
            y,
            N, IC, IH, IW, OC, OH, OW, KH, KW,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            y.stride(0), y.stride(1), y.stride(2), y.stride(3),
            BLOCK_SIZE_H, BLOCK_SIZE_W
        )
        return y
