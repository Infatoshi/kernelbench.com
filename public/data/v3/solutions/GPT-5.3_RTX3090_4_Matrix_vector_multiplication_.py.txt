import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void matvec_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C, int M, int K){
    int row = blockIdx.x;
    int tid = threadIdx.x;

    if(row >= M) return;

    const float* Arow = A + row * K;

    float sum = 0.0f;
    for(int k = tid; k < K; k += blockDim.x){
        float a = Arow[k];
        float b = __ldg(&B[k]);
        sum += a * b;
    }

    __shared__ float sdata[256];
    sdata[tid] = sum;
    __syncthreads();

    for(int s = blockDim.x/2; s > 0; s >>= 1){
        if(tid < s) sdata[tid] += sdata[tid + s];
        __syncthreads();
    }

    if(tid == 0){
        C[row] = sdata[0];
    }
}

torch::Tensor matvec(torch::Tensor A, torch::Tensor B){
    int M = A.size(0);
    int K = A.size(1);

    auto C = torch::empty({M,1}, A.options());

    const float* A_ptr = A.data_ptr<float>();
    const float* B_ptr = B.data_ptr<float>();
    float* C_ptr = C.data_ptr<float>();

    dim3 grid(M);
    dim3 block(256);

    matvec_kernel<<<grid, block>>>(A_ptr, B_ptr, C_ptr, M, K);

    return C;
}
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources='torch::Tensor matvec(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_src,
    functions=['matvec'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        A = A.contiguous()
        B = B.contiguous().view(-1)
        return ext.matvec(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K, device='cuda', dtype=torch.float32)
    B = torch.randn(K, 1, device='cuda', dtype=torch.float32)
    return [A, B]

def get_init_inputs():
    return []
