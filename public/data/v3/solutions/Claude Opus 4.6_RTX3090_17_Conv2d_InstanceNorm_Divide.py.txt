import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Warp-level reduction
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;
}

// Fused InstanceNorm + Divide kernel with float4 vectorization
// Each block handles one (N,C) slice. Uses 2-pass: compute stats, then normalize.
// Optimized with float4 reads and writes.
__global__ void fused_instance_norm_divide_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int HW,
    float divide_by,
    float eps
) {
    int nc = blockIdx.x;
    const float* in_ptr = input + nc * HW;
    float* out_ptr = output + nc * HW;
    
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    // Shared memory for warp reductions
    __shared__ float smem[32]; // max 32 warps
    __shared__ float s_mean, s_inv_std_div;
    
    int HW4 = HW >> 2;
    const float4* in_ptr4 = reinterpret_cast<const float4*>(in_ptr);
    float4* out_ptr4 = reinterpret_cast<float4*>(out_ptr);
    
    float sum = 0.0f;
    float sum_sq = 0.0f;
    
    // Vectorized accumulation
    for (int i = tid; i < HW4; i += blockSize) {
        float4 v = __ldg(in_ptr4 + i);
        sum += v.x + v.y + v.z + v.w;
        sum_sq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
    }
    // Remainder
    int HW_rem = HW4 << 2;
    for (int i = HW_rem + tid; i < HW; i += blockSize) {
        float val = __ldg(in_ptr + i);
        sum += val;
        sum_sq += val * val;
    }
    
    // Block reduce for sum
    int lane = tid & 31;
    int wid = tid >> 5;
    
    sum = warp_reduce_sum(sum);
    if (lane == 0) smem[wid] = sum;
    __syncthreads();
    
    int nwarps = blockSize >> 5;
    sum = (tid < nwarps) ? smem[tid] : 0.0f;
    if (wid == 0) sum = warp_reduce_sum(sum);
    
    // Block reduce for sum_sq
    if (lane == 0 && wid == 0) smem[0] = sum; // store sum temporarily
    __syncthreads();
    float total_sum = smem[0];
    __syncthreads();
    
    sum_sq = warp_reduce_sum(sum_sq);
    if (lane == 0) smem[wid] = sum_sq;
    __syncthreads();
    
    sum_sq = (tid < nwarps) ? smem[tid] : 0.0f;
    if (wid == 0) sum_sq = warp_reduce_sum(sum_sq);
    
    if (tid == 0) {
        float mean = total_sum / HW;
        float var = sum_sq / HW - mean * mean;
        s_mean = mean;
        s_inv_std_div = rsqrtf(var + eps) / divide_by;
    }
    __syncthreads();
    
    float mean = s_mean;
    float inv_std_div = s_inv_std_div;
    
    // Vectorized normalize + divide
    for (int i = tid; i < HW4; i += blockSize) {
        float4 v = __ldg(in_ptr4 + i);
        float4 out;
        out.x = (v.x - mean) * inv_std_div;
        out.y = (v.y - mean) * inv_std_div;
        out.z = (v.z - mean) * inv_std_div;
        out.w = (v.w - mean) * inv_std_div;
        out_ptr4[i] = out;
    }
    for (int i = HW_rem + tid; i < HW; i += blockSize) {
        float val = __ldg(in_ptr + i);
        out_ptr[i] = (val - mean) * inv_std_div;
    }
}

torch::Tensor fused_instance_norm_divide(torch::Tensor input, float divide_by, float eps) {
    auto output = torch::empty_like(input);
    
    int N = input.size(0);
    int C = input.size(1);
    int HW = input.size(2) * input.size(3);
    int NC = N * C;
    
    // Use 512 threads for large HW
    int threads = (HW >= 4096) ? 512 : 256;
    
    fused_instance_norm_divide_kernel<<<NC, threads>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        HW, divide_by, eps
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_instance_norm_divide(torch::Tensor input, float divide_by, float eps);
'''

ext = load_inline(
    name='fused_instnorm_div',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_instance_norm_divide'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = divide_by

    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_instance_norm_divide(x, self.divide_by, 1e-5)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
divide_by = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, divide_by]
