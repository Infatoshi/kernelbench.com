import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 256, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def fp4_matmul_kernel(
    a_ptr, b_ptr, s_ptr, c_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    stride_sn, stride_sk,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    pid_m = pid % num_pid_m
    pid_n = (pid // num_pid_m) % num_pid_n

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    # B is (N, K/2), each byte contains two 4-bit values.
    # K indices k_idx maps to byte k_idx // 2.
    b_ptrs = b_ptr + (offs_bn[None, :] * stride_bn + (offs_k[:, None] // 2) * stride_bk)
    
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)

    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        b_packed = tl.load(b_ptrs, mask=(offs_k[:, None] // 2) < (K // 2) - (k * BLOCK_SIZE_K // 2), other=0)
        
        # Dequantize B packed 4-bit to float16
        # even elements: bits 0-3, odd elements: bits 4-7
        # b_packed is uint8
        mask_even = (offs_k % 2 == 0)[:, None]
        # (val & 0xF) - 8 dequant
        b_vals = tl.where(mask_even, b_packed & 0x0F, (b_packed >> 4) & 0x0F).to(tl.int8)
        b_float = (b_vals - 8).to(tl.float16)

        # Scale loading: s_ptr is (N, K/32)
        # Each scalefactor covers 32 elements of K.
        # Scale for [k*BLOCK_SIZE_K, (k+1)*BLOCK_SIZE_K)
        # Note: BLOCK_SIZE_K must be a multiple of 32 for this simplified logic.
        s_offs_k = (k * BLOCK_SIZE_K + offs_k) // 32
        # s_ptrs shape (BLOCK_SIZE_K, BLOCK_SIZE_N)
        s_ptrs = s_ptr + (offs_bn[None, :] * stride_sn + s_offs_k[:, None] * stride_sk)
        scales = tl.load(s_ptrs, mask=s_offs_k[:, None] < (K // 32), other=1.0)
        
        b_scaled = b_float * scales
        accumulator += tl.dot(a.to(tl.float16), b_scaled.to(tl.float16), out_dtype=tl.float32)
        
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += (BLOCK_SIZE_K // 2) * stride_bk

    c = accumulator.to(tl.float16)
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, c, mask=c_mask)

class Model(nn.Module):
    def __init__(self, K: int, N: int):
        super().__init__()
        self.K = K
        self.N = N
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        w_fp16 = torch.randn(N, K) * 0.02
        torch.random.set_rng_state(rng_state)
        self.weight_packed, self.scales = self._quantize_fp4(w_fp16)

    def _quantize_fp4(self, w: torch.Tensor):
        BLOCK_SIZE = 32
        N, K = w.shape
        n_blocks = (K + BLOCK_SIZE - 1) // BLOCK_SIZE
        w_padded = torch.zeros(N, n_blocks * BLOCK_SIZE, dtype=w.dtype)
        w_padded[:, :K] = w
        w_blocked = w_padded.view(N, n_blocks, BLOCK_SIZE)
        amax = w_blocked.abs().amax(dim=-1, keepdim=True).clamp(min=1e-12)
        scales = amax / 7.0
        w_int = torch.round(w_blocked / scales).clamp(-7, 7).to(torch.int8) + 8
        w_flat = w_int.view(N, -1)
        lo = w_flat[:, 0::2].to(torch.uint8)
        hi = w_flat[:, 1::2].to(torch.uint8)
        packed = (hi << 4) | lo
        return nn.Parameter(packed, requires_grad=False), nn.Parameter(scales.squeeze(-1).to(torch.float16), requires_grad=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        shape = x.shape
        x_reshaped = x.view(-1, self.K)
        M, K = x_reshaped.shape
        N = self.weight_packed.shape[0]
        output = torch.empty((M, N), device=x.device, dtype=torch.float16)
        
        # Grid is handled by autotune
        fp4_matmul_kernel[(triton.cdiv(M, 128) * triton.cdiv(N, 128),)](
            x_reshaped, self.weight_packed, self.scales, output,
            M, N, K,
            x_reshaped.stride(0), x_reshaped.stride(1),
            self.weight_packed.stride(0), self.weight_packed.stride(1),
            self.scales.stride(0), self.scales.stride(1),
            output.stride(0), output.stride(1),
        )
        return output.view(*shape[:-1], N)

def get_inputs():
    return [torch.randn(8, 2048, 4096, device='cuda', dtype=torch.float16)]

def get_init_inputs():
    return [4096, 4096]
