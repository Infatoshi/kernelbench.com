import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Fused kernel: Conv2d(3x3) + Tanh + Scale + Bias + MaxPool2d(2x2)
// Input: (N, C_in, H, W) where C_in=3
// Output: (N, C_out, H_out, W_out) where H_out=(H-2)/2, W_out=(W-2)/2
// Conv kernel is 3x3, no padding, stride=1
// MaxPool is 2x2, stride=2

__global__ void fused_conv_tanh_scale_bias_maxpool_kernel(
    const float* input,      // (N, C_in, H, W)
    const float* weight,     // (C_out, C_in, 3, 3)
    const float* conv_bias,  // (C_out,)
    const float* post_bias,  // (C_out, 1, 1)
    float* output,           // (N, C_out, H_out, W_out)
    int N, int C_in, int H, int W, int C_out,
    float scaling_factor
) {
    // Output dimensions
    int H_out = (H - 2) / 2;
    int W_out = (W - 2) / 2;
    
    // Each thread handles one output element
    int n = blockIdx.z % N;
    int c_out = blockIdx.z / N;
    int h_out = blockIdx.x * blockDim.x + threadIdx.x;
    int w_out = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (n >= N || c_out >= C_out || h_out >= H_out || w_out >= W_out) return;
    
    float max_val = -1e30f;
    
    // Max pool region in conv output space
    int h_start = h_out * 2;
    int w_start = w_out * 2;
    
    // Iterate over 2x2 maxpool region
    for (int ph = 0; ph < 2; ph++) {
        for (int pw = 0; pw < 2; pw++) {
            int h_conv = h_start + ph;
            int w_conv = w_start + pw;
            
            if (h_conv < 0 || h_conv >= H - 2 || w_conv < 0 || w_conv >= W - 2) continue;
            
            // Compute conv output at (h_conv, w_conv) for channel c_out
            float conv_val = conv_bias[c_out];  // Start with conv bias
            
            // 3x3 convolution
            for (int c = 0; c < C_in; c++) {
                for (int kh = 0; kh < 3; kh++) {
                    for (int kw = 0; kw < 3; kw++) {
                        int h_in = h_conv + kh;
                        int w_in = w_conv + kw;
                        
                        if (h_in >= 0 && h_in < H && w_in >= 0 && w_in < W) {
                            int input_idx = ((n * C_in + c) * H + h_in) * W + w_in;
                            int weight_idx = ((c_out * C_in + c) * 3 + kh) * 3 + kw;
                            conv_val += input[input_idx] * weight[weight_idx];
                        }
                    }
                }
            }
            
            // Apply tanh
            float tanh_val = tanhf(conv_val);
            
            // Apply scaling
            float scaled_val = tanh_val * scaling_factor;
            
            // Apply post bias
            float final_val = scaled_val + post_bias[c_out];
            
            // Max reduction
            if (final_val > max_val) {
                max_val = final_val;
            }
        }
    }
    
    int out_idx = ((n * C_out + c_out) * H_out + h_out) * W_out + w_out;
    output[out_idx] = max_val;
}

torch::Tensor fused_conv_tanh_scale_bias_maxpool_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor conv_bias,
    torch::Tensor post_bias,
    float scaling_factor
) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int C_out = weight.size(0);
    
    int H_out = (H - 2) / 2;
    int W_out = (W - 2) / 2;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    dim3 block(16, 16);
    dim3 grid((H_out + block.x - 1) / block.x, (W_out + block.y - 1) / block.y, N * C_out);
    
    fused_conv_tanh_scale_bias_maxpool_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        conv_bias.data_ptr<float>(),
        post_bias.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C_in, H, W, C_out, scaling_factor
    );
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor fused_conv_tanh_scale_bias_maxpool_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor conv_bias,
    torch::Tensor post_bias,
    float scaling_factor
);
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_tanh_scale_bias_maxpool_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    A model that performs a convolution, applies tanh, scaling, adds a bias term, and then max-pools.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)

    def forward(self, x):
        weight = self.conv.weight
        conv_bias = self.conv.bias
        post_bias = self.bias  # (C_out, 1, 1)
        return ext.fused_conv_tanh_scale_bias_maxpool_cuda(
            x, weight, conv_bias, post_bias, self.scaling_factor
        )


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    scaling_factor = 2.0
    bias_shape = (out_channels, 1, 1)
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
