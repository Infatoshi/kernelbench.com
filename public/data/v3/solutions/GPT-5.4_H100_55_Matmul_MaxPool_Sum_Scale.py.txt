import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


def _acc_dtype(x: torch.Tensor):
    if x.dtype in (torch.float16, torch.bfloat16):
        return torch.float32
    return x.dtype


@triton.jit
def fused_mm_maxpool_sum_scale_kernel(
    x_ptr, w_ptr, b_ptr, out_ptr,
    B, K, N,
    stride_xb, stride_xk,
    stride_wn, stride_wk,
    stride_ob,
    scale,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    HAS_BIAS: tl.constexpr,
):
    pid = tl.program_id(0)
    offs_m = pid * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < B

    pair_idx = tl.arange(0, BLOCK_N)
    n0 = pair_idx * 2
    n1 = n0 + 1
    valid_pair = n1 < N

    acc0 = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc1 = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k0 in range(0, K, BLOCK_K):
        offs_k = k0 + tl.arange(0, BLOCK_K)
        mask_k = offs_k < K
        x = tl.load(x_ptr + offs_m[:, None] * stride_xb + offs_k[None, :] * stride_xk,
                    mask=mask_m[:, None] & mask_k[None, :], other=0.0)
        w0 = tl.load(w_ptr + n0[:, None] * stride_wn + offs_k[None, :] * stride_wk,
                     mask=valid_pair[:, None] & mask_k[None, :], other=0.0)
        w1 = tl.load(w_ptr + n1[:, None] * stride_wn + offs_k[None, :] * stride_wk,
                     mask=valid_pair[:, None] & mask_k[None, :], other=0.0)
        acc0 += tl.dot(x, tl.trans(w0))
        acc1 += tl.dot(x, tl.trans(w1))

    if HAS_BIAS:
        b0 = tl.load(b_ptr + n0, mask=valid_pair, other=0.0).to(tl.float32)
        b1 = tl.load(b_ptr + n1, mask=valid_pair, other=0.0).to(tl.float32)
        acc0 += b0[None, :]
        acc1 += b1[None, :]

    pooled = tl.maximum(acc0, acc1)
    summed = tl.sum(pooled, axis=0) * scale
    tl.store(out_ptr + offs_m * stride_ob, summed, mask=mask_m)


class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        assert kernel_size == 2, 'Optimized kernel assumes kernel_size=2'
        self.matmul = nn.Linear(in_features, out_features)
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        assert x.is_cuda, 'Input must be CUDA tensor'
        B, K = x.shape
        N = self.matmul.weight.shape[0]
        assert K == self.matmul.weight.shape[1]
        assert N % 2 == 0

        x_c = x.contiguous()
        w_c = self.matmul.weight.contiguous()
        b = self.matmul.bias

        out = torch.empty((B,), device=x.device, dtype=_acc_dtype(x_c))

        grid = (triton.cdiv(B, 128),)
        fused_mm_maxpool_sum_scale_kernel[grid](
            x_c, w_c, b, out,
            B, K, N,
            x_c.stride(0), x_c.stride(1),
            w_c.stride(0), w_c.stride(1),
            out.stride(0),
            self.scale_factor,
            BLOCK_M=128,
            BLOCK_N=128,
            BLOCK_K=32,
            HAS_BIAS=b is not None,
            num_warps=8,
            num_stages=4,
        )
        return out


batch_size = 128
in_features = 4096
out_features = 4096
kernel_size = 2
scale_factor = 0.5


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, kernel_size, scale_factor]
