import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAGuard.h>

// Optimized Max Pooling 2D kernel with support for dilation
template<typename scalar_t>
__global__ void max_pool2d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    // Each thread block handles one output channel slice for multiple batches
    // Each thread handles one or more output pixels
    
    int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_out_pixels = batch_size * channels * out_height * out_width;
    
    if (out_idx >= total_out_pixels) return;
    
    // Decode the output index
    int tmp = out_idx;
    int out_w = tmp % out_width;
    tmp /= out_width;
    int out_h = tmp % out_height;
    tmp /= out_height;
    int ch = tmp % channels;
    int bn = tmp / channels;
    
    // Calculate input start position with padding
    int in_h_start = out_h * stride - padding;
    int in_w_start = out_w * stride - padding;
    
    // Find max value over the dilated kernel
    scalar_t max_val = -std::numeric_limits<scalar_t>::infinity();
    
    for (int ki = 0; ki < kernel_size; ki++) {
        for (int kj = 0; kj < kernel_size; kj++) {
            int in_h = in_h_start + ki * dilation;
            int in_w = in_w_start + kj * dilation;
            
            // Check bounds
            if (in_h >= 0 && in_h < in_height && in_w >= 0 && in_w < in_width) {
                int in_idx = ((bn * channels + ch) * in_height + in_h) * in_width + in_w;
                scalar_t val = input[in_idx];
                if (val > max_val) {
                    max_val = val;
                }
            }
        }
    }
    
    output[out_idx] = max_val;
}

// Optimized version using shared memory and better thread organization
template<typename scalar_t>
__global__ void max_pool2d_optimized_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    // Block handles a tile of output pixels
    // Thread block: 16x16 threads for output tile
    
    extern __shared__ unsigned char smem_raw[];
    scalar_t* smem = reinterpret_cast<scalar_t*>(smem_raw);
    
    int tile_w = 16;
    int tile_h = 16;
    
    int batch_ch = blockIdx.z;
    int bn = batch_ch / channels;
    int ch = batch_ch % channels;
    
    int tile_start_w = blockIdx.x * tile_w;
    int tile_start_h = blockIdx.y * tile_h;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int out_w = tile_start_w + tx;
    int out_h = tile_start_h + ty;
    
    // Compute max pooling for this thread's output pixel
    if (out_h < out_height && out_w < out_width) {
        int in_h_start = out_h * stride - padding;
        int in_w_start = out_w * stride - padding;
        
        scalar_t max_val = -std::numeric_limits<scalar_t>::infinity();
        
        #pragma unroll
        for (int ki = 0; ki < kernel_size; ki++) {
            #pragma unroll
            for (int kj = 0; kj < kernel_size; kj++) {
                int in_h = in_h_start + ki * dilation;
                int in_w = in_w_start + kj * dilation;
                
                if (in_h >= 0 && in_h < in_height && in_w >= 0 && in_w < in_width) {
                    int in_idx = ((bn * channels + ch) * in_height + in_h) * in_width + in_w;
                    scalar_t val = input[in_idx];
                    if (val > max_val) {
                        max_val = val;
                    }
                }
            }
        }
        
        int out_idx = ((bn * channels + ch) * out_height + out_h) * out_width + out_w;
        output[out_idx] = max_val;
    }
}

torch::Tensor max_pool2d_forward(torch::Tensor input, 
                                  int kernel_size, 
                                  int stride, 
                                  int padding, 
                                  int dilation) {
    // Input: (batch, channels, height, width)
    auto sizes = input.sizes();
    int batch_size = sizes[0];
    int channels = sizes[1];
    int in_height = sizes[2];
    int in_width = sizes[3];
    
    // Calculate output dimensions
    int out_height = (in_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int out_width = (in_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    int threads_per_block = 256;
    int total_out_pixels = batch_size * channels * out_height * out_width;
    int blocks = (total_out_pixels + threads_per_block - 1) / threads_per_block;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "max_pool2d_forward", ([&] {
        max_pool2d_kernel<scalar_t><<<blocks, threads_per_block>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            channels,
            in_height,
            in_width,
            out_height,
            out_width,
            kernel_size,
            stride,
            padding,
            dilation
        );
    }));
    
    cudaDeviceSynchronize();
    return output;
}
'''

ext = load_inline(
    name='maxpool_ext',
    cpp_sources='''
#include <torch/extension.h>
torch::Tensor max_pool2d_forward(torch::Tensor input, int kernel_size, int stride, int padding, int dilation);
''',
    cuda_sources=cuda_source,
    functions=['max_pool2d_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    """
    Simple model that performs Max Pooling 2D.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.
        """
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Max Pooling 2D to the input tensor.
        """
        return ext.max_pool2d_forward(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    kernel_size = 2
    stride = 2
    padding = 1
    dilation = 3
    return [kernel_size, stride, padding, dilation]
