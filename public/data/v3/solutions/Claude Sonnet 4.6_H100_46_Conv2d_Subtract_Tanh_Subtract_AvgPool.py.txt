import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Fused kernel: subtract1 + tanh + subtract2 + avgpool
# Conv output: [N, C, H_out, W_out]  (float32, fp16, bf16)
# AvgPool with kernel_size_pool x kernel_size_pool, stride = kernel_size_pool (no padding)

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math.h>

// -------------------------------------------------------------------
// Type conversion helpers for half/bfloat16
// -------------------------------------------------------------------
template <typename T>
__device__ __forceinline__ float to_float(T x) { return (float)x; }

template <>
__device__ __forceinline__ float to_float<__half>(__half x) { return __half2float(x); }

template <>
__device__ __forceinline__ float to_float<__nv_bfloat16>(__nv_bfloat16 x) { return __bfloat162float(x); }

template <typename T>
__device__ __forceinline__ T from_float(float x) { return (T)x; }

template <>
__device__ __forceinline__ __half from_float<__half>(float x) { return __float2half(x); }

template <>
__device__ __forceinline__ __nv_bfloat16 from_float<__nv_bfloat16>(float x) { return __float2bfloat16(x); }

// -------------------------------------------------------------------
// Generic fused kernel: subtract1 + tanh + subtract2 + avgpool
// Works for float, half, bfloat16 via template + compile-time dispatch
// -------------------------------------------------------------------

template <typename T>
__global__ void fused_tanh_avgpool_kernel(
    const T* __restrict__ input,   // [N, C, H, W]
    T* __restrict__ output,        // [N, C, H/P, W/P]
    int N, int C, int H, int W,
    int pool_size,
    float sub1, float sub2,
    int H_out, int W_out
) {
    // Each thread computes one output element
    // output layout: [N, C, H_out, W_out]
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c     = (idx / (W_out * H_out)) % C;
    int n     = idx / (W_out * H_out * C);

    int h_start = h_out * pool_size;
    int w_start = w_out * pool_size;

    float acc = 0.0f;
    float inv_pool = 1.0f / (float)(pool_size * pool_size);

    for (int ph = 0; ph < pool_size; ph++) {
        for (int pw = 0; pw < pool_size; pw++) {
            int h_in = h_start + ph;
            int w_in = w_start + pw;
            float val = to_float(input[n * C * H * W + c * H * W + h_in * W + w_in]);
            val = tanhf(val - sub1) - sub2;
            acc += val;
        }
    }
    output[idx] = from_float<T>(acc * inv_pool);
}

// Vectorized version for float32 with pool_size=2 (common case)
__global__ void fused_tanh_avgpool_float_p2_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int N, int C, int H, int W,
    float sub1, float sub2,
    int H_out, int W_out
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c     = (idx / (W_out * H_out)) % C;
    int n     = idx / (W_out * H_out * C);

    int h_start = h_out * 2;
    int w_start = w_out * 2;

    // Load 4 values (2x2 pool), all in same channel/batch
    int base = n * C * H * W + c * H * W;
    float v0 = input[base + (h_start + 0) * W + (w_start + 0)];
    float v1 = input[base + (h_start + 0) * W + (w_start + 1)];
    float v2 = input[base + (h_start + 1) * W + (w_start + 0)];
    float v3 = input[base + (h_start + 1) * W + (w_start + 1)];

    v0 = tanhf(v0 - sub1) - sub2;
    v1 = tanhf(v1 - sub1) - sub2;
    v2 = tanhf(v2 - sub1) - sub2;
    v3 = tanhf(v3 - sub1) - sub2;

    output[idx] = (v0 + v1 + v2 + v3) * 0.25f;
}

// Half-precision pool_size=2 optimized version
__global__ void fused_tanh_avgpool_half_p2_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    int N, int C, int H, int W,
    float sub1, float sub2,
    int H_out, int W_out
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c     = (idx / (W_out * H_out)) % C;
    int n     = idx / (W_out * H_out * C);

    int h_start = h_out * 2;
    int w_start = w_out * 2;

    int base = n * C * H * W + c * H * W;
    float v0 = __half2float(input[base + (h_start + 0) * W + (w_start + 0)]);
    float v1 = __half2float(input[base + (h_start + 0) * W + (w_start + 1)]);
    float v2 = __half2float(input[base + (h_start + 1) * W + (w_start + 0)]);
    float v3 = __half2float(input[base + (h_start + 1) * W + (w_start + 1)]);

    v0 = tanhf(v0 - sub1) - sub2;
    v1 = tanhf(v1 - sub1) - sub2;
    v2 = tanhf(v2 - sub1) - sub2;
    v3 = tanhf(v3 - sub1) - sub2;

    output[idx] = __float2half((v0 + v1 + v2 + v3) * 0.25f);
}

// BF16 pool_size=2 optimized version
__global__ void fused_tanh_avgpool_bf16_p2_kernel(
    const __nv_bfloat16* __restrict__ input,
    __nv_bfloat16* __restrict__ output,
    int N, int C, int H, int W,
    float sub1, float sub2,
    int H_out, int W_out
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c     = (idx / (W_out * H_out)) % C;
    int n     = idx / (W_out * H_out * C);

    int h_start = h_out * 2;
    int w_start = w_out * 2;

    int base = n * C * H * W + c * H * W;
    float v0 = __bfloat162float(input[base + (h_start + 0) * W + (w_start + 0)]);
    float v1 = __bfloat162float(input[base + (h_start + 0) * W + (w_start + 1)]);
    float v2 = __bfloat162float(input[base + (h_start + 1) * W + (w_start + 0)]);
    float v3 = __bfloat162float(input[base + (h_start + 1) * W + (w_start + 1)]);

    v0 = tanhf(v0 - sub1) - sub2;
    v1 = tanhf(v1 - sub1) - sub2;
    v2 = tanhf(v2 - sub1) - sub2;
    v3 = tanhf(v3 - sub1) - sub2;

    output[idx] = __float2bfloat16((v0 + v1 + v2 + v3) * 0.25f);
}

torch::Tensor fused_post_conv(
    torch::Tensor input,   // conv output [N, C, H, W]
    float sub1, float sub2,
    int pool_size
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int H_out = H / pool_size;
    int W_out = W / pool_size;

    auto output = torch::empty({N, C, H_out, W_out}, input.options());

    int total = N * C * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    if (input.scalar_type() == torch::kFloat32) {
        if (pool_size == 2) {
            fused_tanh_avgpool_float_p2_kernel<<<blocks, threads>>>(
                input.data_ptr<float>(),
                output.data_ptr<float>(),
                N, C, H, W, sub1, sub2, H_out, W_out
            );
        } else {
            fused_tanh_avgpool_kernel<float><<<blocks, threads>>>(
                input.data_ptr<float>(),
                output.data_ptr<float>(),
                N, C, H, W, pool_size, sub1, sub2, H_out, W_out
            );
        }
    } else if (input.scalar_type() == torch::kFloat16) {
        if (pool_size == 2) {
            fused_tanh_avgpool_half_p2_kernel<<<blocks, threads>>>(
                (__half*)input.data_ptr(),
                (__half*)output.data_ptr(),
                N, C, H, W, sub1, sub2, H_out, W_out
            );
        } else {
            fused_tanh_avgpool_kernel<__half><<<blocks, threads>>>(
                (__half*)input.data_ptr(),
                (__half*)output.data_ptr(),
                N, C, H, W, pool_size, sub1, sub2, H_out, W_out
            );
        }
    } else if (input.scalar_type() == torch::kBFloat16) {
        if (pool_size == 2) {
            fused_tanh_avgpool_bf16_p2_kernel<<<blocks, threads>>>(
                (__nv_bfloat16*)input.data_ptr(),
                (__nv_bfloat16*)output.data_ptr(),
                N, C, H, W, sub1, sub2, H_out, W_out
            );
        } else {
            fused_tanh_avgpool_kernel<__nv_bfloat16><<<blocks, threads>>>(
                (__nv_bfloat16*)input.data_ptr(),
                (__nv_bfloat16*)output.data_ptr(),
                N, C, H, W, pool_size, sub1, sub2, H_out, W_out
            );
        }
    }

    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_post_conv(torch::Tensor input, float sub1, float sub2, int pool_size);
'''

ext = load_inline(
    name='fused_post_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_post_conv'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Fused: Conv2d → subtract1 → tanh → subtract2 → AvgPool2d
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        self.kernel_size_pool = int(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_post_conv(x, self.subtract1_value, self.subtract2_value, self.kernel_size_pool)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
