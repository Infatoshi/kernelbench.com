import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

template <typename scalar_t>
__global__ void cross_entropy_kernel(
    const scalar_t* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    float* __restrict__ output,
    int batch_size,
    int num_classes) {
    
    int bid = blockIdx.x;
    if (bid >= batch_size) return;
    
    int tid = threadIdx.x;
    int lane = tid % 32;
    int warp = tid / 32;
    
    // Load value and apply identity for max reduction
    float val = (tid < num_classes) ? static_cast<float>(predictions[bid * num_classes + tid]) : -FLT_MAX;
    
    // Warp-level reduction for max
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    
    // Store warp max to shared memory
    __shared__ float warp_max[32];
    if (lane == 0) warp_max[warp] = val;
    __syncthreads();
    
    // Inter-warp reduction for max (only first warp)
    float max_val = (tid < 32) ? warp_max[tid] : -FLT_MAX;
    if (tid < 32) {
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
        }
    }
    // Broadcast max to all threads in block
    max_val = __shfl_sync(0xffffffff, max_val, 0);
    __syncthreads();
    
    // Compute exp(x - max) for numerical stability
    val = (tid < num_classes) ? expf(static_cast<float>(predictions[bid * num_classes + tid]) - max_val) : 0.0f;
    
    // Warp-level reduction for sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    
    // Store warp sum to shared memory
    __shared__ float warp_sum[32];
    if (lane == 0) warp_sum[warp] = val;
    __syncthreads();
    
    // Inter-warp reduction for sum (only first warp)
    float sum_exp = (tid < 32) ? warp_sum[tid] : 0.0f;
    if (tid < 32) {
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum_exp += __shfl_down_sync(0xffffffff, sum_exp, offset);
        }
    }
    // Broadcast sum to all threads
    sum_exp = __shfl_sync(0xffffffff, sum_exp, 0);
    
    // Thread 0 computes final loss for this sample and accumulates to global output
    if (tid == 0) {
        int target_idx = targets[bid];
        float target_val = static_cast<float>(predictions[bid * num_classes + target_idx]);
        float loss = logf(sum_exp) + max_val - target_val;
        // Divide by batch_size here to compute mean directly
        atomicAdd(output, loss / static_cast<float>(batch_size));
    }
}

torch::Tensor cross_entropy_cuda(torch::Tensor predictions, torch::Tensor targets) {
    // Ensure contiguous memory layout
    predictions = predictions.contiguous();
    targets = targets.contiguous();
    
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    // Allocate output tensor (scalar)
    auto output = torch::zeros({1}, torch::dtype(torch::kFloat32).device(predictions.device()));
    
    // Launch configuration: one block per sample, 1024 threads per block
    const int threads = 1024;
    const int blocks = batch_size;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half,
        at::ScalarType::BFloat16,
        predictions.scalar_type(),
        "cross_entropy_cuda",
        [&] {
            cross_entropy_kernel<scalar_t><<<blocks, threads>>>(
                predictions.data_ptr<scalar_t>(),
                targets.data_ptr<int64_t>(),
                output.data_ptr<float>(),
                batch_size,
                num_classes
            );
        }
    );
    
    // Convert to scalar tensor (0-dimensional)
    return output.view({});
}
'''

cpp_source = 'torch::Tensor cross_entropy_cuda(torch::Tensor predictions, torch::Tensor targets);'

# Load the CUDA extension
cross_entropy_ext = load_inline(
    name='cross_entropy_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Cross Entropy Loss using custom CUDA kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, predictions, targets):
        return cross_entropy_ext.cross_entropy_cuda(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [
        torch.randn(batch_size, num_classes),
        torch.randint(0, num_classes, (batch_size,))
    ]
