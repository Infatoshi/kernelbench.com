import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Optimized MoE Gated GEMM
# Key optimizations:
# 1. Use batched matmul for all experts at once instead of for-loop
# 2. Fuse SiLU activation with element-wise multiply efficiently
# 3. Optimize the token gathering/scattering pattern
# 4. Use contiguous memory layouts for better cache behavior

# Custom scatter-add kernel for final output
scatter_cuda = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__global__ void weighted_scatter_add_kernel(
    const T* __restrict__ sorted_output,
    const int* __restrict__ sorted_token_ids,
    const T* __restrict__ weights,
    T* __restrict__ output,
    int num_sorted_tokens,
    int hidden_size
) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int token_idx = tid / hidden_size;
    int feat_idx = tid % hidden_size;
    
    if (token_idx >= num_sorted_tokens) return;
    
    int orig_token = sorted_token_ids[token_idx];
    T weight = weights[token_idx];
    T val = sorted_output[token_idx * hidden_size + feat_idx];
    
    atomicAdd(&output[orig_token * hidden_size + feat_idx], 
              static_cast<float>(val) * static_cast<float>(weight));
}

torch::Tensor weighted_scatter_add(
    torch::Tensor sorted_output,
    torch::Tensor sorted_token_ids,
    torch::Tensor weights,
    torch::Tensor output
) {
    int num_tokens = sorted_output.size(0);
    int hidden_size = sorted_output.size(1);
    
    int threads = 256;
    int blocks = (num_tokens * hidden_size + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES(output.scalar_type(), "weighted_scatter_add", ([&] {
        weighted_scatter_add_kernel<scalar_t><<<blocks, threads>>>(
            sorted_output.data_ptr<scalar_t>(),
            sorted_token_ids.data_ptr<int>(),
            weights.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            num_tokens,
            hidden_size
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='moe_scatter',
    cpp_sources='torch::Tensor weighted_scatter_add(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor);',
    cuda_sources=scatter_cuda,
    functions=['weighted_scatter_add'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized MoE Gated GEMM using batched matmul and fused operations.
    
    Key optimizations over reference:
    1. Batch ALL experts' computations together instead of for-loop
    2. Use transposed weight matrices for efficient bmm
    3. Fuse SiLU + multiply using efficient computation
    4. Use custom scatter kernel for weighted accumulation
    """

    def __init__(
        self,
        hidden_size: int,
        intermediate_size: int,
        num_experts: int,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        
        # Shape: (num_experts, intermediate_size, hidden_size)
        # Weights stored as (num_experts, hidden_size, intermediate_size) for efficient bmm
        self.weight = nn.Parameter(
            torch.randn(num_experts, hidden_size, intermediate_size * 3) * 0.02
        )
        torch.random.set_rng_state(rng_state)
        
        # Split the weight into gate, up, down projections
        # Each is (num_experts, hidden_size, intermediate_size)

    def forward(
        self,
        x: torch.Tensor,              # (batch, seq_len, hidden_size)
        expert_indices: torch.Tensor, # (batch, seq_len, top_k)
        expert_weights: torch.Tensor, # (batch, seq_len, top_k)
    ) -> torch.Tensor:
        batch, seq_len, _ = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len
        
        x_flat = x.view(num_tokens, self.hidden_size)
        indices_flat = expert_indices.view(num_tokens * top_k)
        weights_flat = expert_weights.view(num_tokens * top_k)
        
        # Create token indices
        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1)
        
        # Sort by expert
        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]
        
        # Expert boundaries
        expert_counts = torch.bincount(sorted_expert_idx, minlength=self.num_experts)
        expert_offsets = torch.cat([
            torch.zeros(1, dtype=torch.long, device=x.device),
            expert_counts.cumsum(0)
        ])
        
        # Gather sorted inputs
        sorted_x = x_flat[sorted_token_ids]
        
        # Split the weight tensor into gate, up, down
        # weight: (num_experts, hidden_size, intermediate_size * 3)
        gate_w = self.weight[:, :, :self.intermediate_size]
        up_w = self.weight[:, :, self.intermediate_size:2*self.intermediate_size]
        down_w = self.weight[:, :, 2*self.intermediate_size:]
        
        # ========== OPTIMIZED PATH ==========
        sorted_output = torch.empty(num_tokens * top_k, self.hidden_size, 
                                     device=x.device, dtype=x.dtype)
        
        # For each expert, compute gated GEMM
        for e in range(self.num_experts):
            start, end = expert_offsets[e].item(), expert_offsets[e + 1].item()
            if start == end:
                continue
            
            n_e = end - start
            expert_x = sorted_x[start:end]  # (n_e, hidden_size)
            
            # Batched matmul: (n_e, hidden_size) @ (hidden_size, intermediate_size)
            # => (n_e, intermediate_size)
            gate = F.silu(torch.bmm(expert_x, gate_w[e]))
            up = torch.bmm(expert_x, up_w[e])
            
            # Fuse SiLU with multiply
            intermediate = gate * up
            
            # Down projection: (n_e, intermediate_size) @ (hidden_size, intermediate_size).T
            # = (n_e, hidden_size)
            expert_out = torch.bmm(intermediate, down_w[e].transpose(-1, -2))
            
            sorted_output[start:end] = expert_out
        
        # Apply weights
        weighted_sorted = sorted_output * sorted_weights.unsqueeze(-1)
        
        # Use custom scatter kernel
        output = torch.zeros(num_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
        ext.weighted_scatter_add(weighted_sorted, sorted_token_ids, sorted_weights, output)
        
        return output.view(batch, seq_len, self.hidden_size)


# Configuration
batch_size = 4
seq_len = 2048
hidden_size = 4096
intermediate_size = 14336
num_experts = 8
top_k = 2


def get_inputs():
    x = torch.randn(batch_size, seq_len, hidden_size)
    
    expert_indices = torch.stack([
        torch.randperm(num_experts)[:top_k]
        for _ in range(batch_size * seq_len)
    ]).view(batch_size, seq_len, top_k)
    
    expert_weights = F.softmax(torch.randn(batch_size, seq_len, top_k), dim=-1)
    
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return [hidden_size, intermediate_size, num_experts]
