import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void layernorm_forward_kernel(
    const float* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ y,
    int rows,
    int cols,
    float eps) {

    int row = blockIdx.x;
    int tid = threadIdx.x;
    if (row >= rows) return;

    const float* row_x = x + (size_t)row * cols;
    float* row_y = y + (size_t)row * cols;

    float sum = 0.0f;
    for (int i = tid; i < cols; i += blockDim.x) {
        sum += row_x[i];
    }

    __shared__ float shm_sum[256];
    shm_sum[tid] = sum;
    __syncthreads();

    for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) {
        if (tid < offset) shm_sum[tid] += shm_sum[tid + offset];
        __syncthreads();
    }
    float mean = shm_sum[0] / cols;

    float var_sum = 0.0f;
    for (int i = tid; i < cols; i += blockDim.x) {
        float v = row_x[i] - mean;
        var_sum += v * v;
    }

    __shared__ float shm_var[256];
    shm_var[tid] = var_sum;
    __syncthreads();

    for (int offset = blockDim.x / 2; offset > 0; offset >>= 1) {
        if (tid < offset) shm_var[tid] += shm_var[tid + offset];
        __syncthreads();
    }
    float inv_std = rsqrtf(shm_var[0] / cols + eps);

    for (int i = tid; i < cols; i += blockDim.x) {
        float norm = (row_x[i] - mean) * inv_std;
        row_y[i] = norm * weight[i] + bias[i];
    }
}

torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, double eps) {
    auto x_contig = x.contiguous();
    auto weight_contig = weight.contiguous().view({-1});
    auto bias_contig = bias.contiguous().view({-1});

    const auto cols = weight_contig.numel();
    const auto rows = x_contig.numel() / cols;

    auto x_float = x_contig.view({rows, cols}).to(torch::kFloat);
    auto y_float = torch::empty_like(x_float);

    const int threads = 256;
    const dim3 blocks(rows);
    cudaStream_t stream = at::cuda::getDefaultCUDAStream();

    layernorm_forward_kernel<<<blocks, threads, 0, stream>>>(
        x_float.data_ptr<float>(),
        weight_contig.to(torch::kFloat).data_ptr<float>(),
        bias_contig.to(torch::kFloat).data_ptr<float>(),
        y_float.data_ptr<float>(),
        rows,
        cols,
        (float)eps
    );

    return y_float.view(x_contig.sizes()).to(x.scalar_type());
}
'''

ext = load_inline(
    name='layernorm_ext_40',
    cpp_sources='torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, double eps);',
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.normalized_shape = tuple(normalized_shape)
        self.weight = nn.Parameter(torch.ones(self.normalized_shape))
        self.bias = nn.Parameter(torch.zeros(self.normalized_shape))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(x, self.weight, self.bias, self.eps)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
