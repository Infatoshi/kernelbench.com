cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cublas_v2.h>
#include <algorithm>
#include <math.h>

// Constants
constexpr int THREADS_PER_BLOCK = 256;
constexpr int WARP_SIZE = 32;

// Error checking macro
#define CUDA_CHECK(call) \
    do { \
        auto err = call; \
        if (err != cudaSuccess) { \
            fprintf(stderr, "CUDA error at %s:%d: %s\\n", __FILE__, __LINE__, \
                    cudaGetErrorString(err)); \
            exit(EXIT_FAILURE); \
        } \
    } while(0)

#define CUBLAS_CHECK(call) \
    do { \
        auto status = call; \
        if (status != CUBLAS_STATUS_SUCCESS) { \
            fprintf(stderr, "CUBLAS error at %s:%d\\n", __FILE__, __LINE__); \
            exit(EXIT_FAILURE); \
        } \
    } while(0)

// Fast GELU approximation using Taylor series
__device__ __forceinline__ float fast_gelu(float x) {
    float cdf = 0.5f * (1.0f + tanhf(0.7978845608028654f * (x + 0.044715f * x * x * x)));
    return x * cdf;
}

__device__ __forceinline__ __half fast_gelu(__half x) {
    float fx = __half2float(x);
    float cdf = 0.5f * (1.0f + tanhf(0.7978845608028654f * (fx + 0.044715f * fx * fx * fx)));
    return __float2half(cdf * fx);
}

// Warp-level reduction for max
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

// Warp-level reduction for sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// GELU kernel for float
__global__ void gelu_kernel_float(const float* __restrict__ input, float* __restrict__ output, 
                                  int rows, int cols) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = rows * cols;
    
    while (idx < total) {
        output[idx] = fast_gelu(input[idx]);
        idx += gridDim.x * blockDim.x;
    }
}

// GELU kernel for half
__global__ void gelu_kernel_half(const __half* __restrict__ input, __half* __restrict__ output, 
                                 int rows, int cols) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = rows * cols;
    
    while (idx < total) {
        output[idx] = fast_gelu(input[idx]);
        idx += gridDim.x * blockDim.x;
    }
}

// Bias addition kernel
__global__ void add_bias_kernel_float(const float* __restrict__ input, float* __restrict__ output,
                                      const float* __restrict__ bias, int rows, int cols) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = rows * cols;
    while (idx < total) {
        int col = idx % cols;
        output[idx] = input[idx] + bias[col];
        idx += gridDim.x * blockDim.x;
    }
}

__global__ void add_bias_kernel_half(const __half* __restrict__ input, __half* __restrict__ output,
                                     const __half* __restrict__ bias, int rows, int cols) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = rows * cols;
    while (idx < total) {
        int col = idx % cols;
        output[idx] = __hadd(input[idx], bias[col]);
        idx += gridDim.x * blockDim.x;
    }
}

// Optimized softmax with warp-level primitives for float
__global__ void softmax_kernel_float(const float* __restrict__ input, float* __restrict__ output,
                                      int rows, int cols) {
    int row_idx = blockIdx.x;
    if (row_idx >= rows) return;
    
    const float* row_ptr = input + row_idx * cols;
    float* out_row_ptr = output + row_idx * cols;
    
    // Each warp handles a portion of the row
    int warp_id = threadIdx.x / WARP_SIZE;
    int lane_id = threadIdx.x % WARP_SIZE;
    int warps_per_block = blockDim.x / WARP_SIZE;
    int col_per_warp = (cols + warps_per_block - 1) / warps_per_block;
    int col_start = warp_id * col_per_warp;
    int col_end = min(col_start + col_per_warp, cols);
    
    // Step 1: Find local max in each warp
    float local_max = -INFINITY;
    for (int col = col_start + lane_id; col < col_end; col += WARP_SIZE) {
        local_max = fmaxf(local_max, row_ptr[col]);
    }
    
    // Warp-level reduction for max
    float warp_max = warp_reduce_max(local_max);
    
    // Block-level reduction
    __shared__ float shared_max;
    if (lane_id == 0) {
        shared_max = warp_max;
    }
    __syncthreads();
    
    float block_max = -INFINITY;
    if (threadIdx.x < warps_per_block) {
        block_max = shared_max;
    }
    if (warp_id == 0) {
        block_max = warp_reduce_max(block_max);
        if (lane_id == 0) {
            shared_max = block_max;
        }
    }
    __syncthreads();
    block_max = shared_max;
    
    // Step 2: Compute exp sum
    float local_sum = 0.0f;
    for (int col = col_start + lane_id; col < col_end; col += WARP_SIZE) {
        float val = expf(row_ptr[col] - block_max);
        local_sum += val;
    }
    
    float warp_sum = warp_reduce_sum(local_sum);
    
    __shared__ float shared_sum;
    if (lane_id == 0) {
        shared_sum = warp_sum;
    }
    __syncthreads();
    
    float block_sum = 0.0f;
    if (threadIdx.x < warps_per_block) {
        block_sum = shared_sum;
    }
    if (warp_id == 0) {
        block_sum = warp_reduce_sum(block_sum);
        if (lane_id == 0) {
            shared_sum = block_sum;
        }
    }
    __syncthreads();
    block_sum = shared_sum;
    
    // Step 3: Compute softmax output
    float inv_sum = 1.0f / (block_sum + 1e-7f);
    for (int col = col_start + lane_id; col < col_end; col += WARP_SIZE) {
        float val = expf(row_ptr[col] - block_max) * inv_sum;
        out_row_ptr[col] = val;
    }
}

        val = expf(val - block_max);
        local_sum += val;
    }

    if (lane_id == 0) {
        shared_sum = warp_sum;
    }
    __syncthreads();
    
    float block_sum = 0.0f;

    for (int col = col_start + lane_id; col < col_end; col += WARP_SIZE) {
        float val = __half2float(row_ptr[col]);
        val = expf(val - block_max) * inv_sum;
        out_row_ptr[col] = __float2half(val);
    }
}

// Cublas handle (singleton)
static cublasHandle_t g_cublas_handle;

void init_cublas() {
    static bool initialized = false;
    if (!initialized) {
        CUBLAS_CHECK(cublasCreate(&g_cublas_handle));
        initialized = true;
    }
}

// Matmul + GELU + Softmax for float32
torch::Tensor matmul_gelu_softmax_fp32(torch::Tensor input, torch::Tensor weight, 
                                        torch::Tensor bias, int rows, int cols) {
    init_cublas();
    
    auto output = torch::empty({rows, cols}, input.options());
    
    const float* input_ptr = input.data_ptr<float>();
    const float* weight_ptr = weight.data_ptr<float>();
    const float* bias_ptr = bias.data_ptr<float>();
    float* output_ptr = output.data_ptr<float>();
    
    // Cublas GEMM: output = input @ weight^T + bias
    const float alpha = 1.0f;
    const float beta = 0.0f;
    
    CUBLAS_CHECK(cublasSgemm(g_cublas_handle,
                             CUBLAS_OP_T,  // weight is transposed
                             CUBLAS_OP_N,  // input is not transposed
                             cols,         // n: output features
                             rows,         // m: batch size
                             input.size(1), // k: input features
                             &alpha,
                             weight_ptr, cols,      // weight: (k, n) -> (n, k) due to transpose
                             input_ptr, input.size(1), // input: (m, k)
                             &beta,
                             output_ptr, cols));
    
    // Add bias
    const int total_elements = rows * cols;
    const int num_blocks = (total_elements + THREADS_PER_BLOCK - 1) / THREADS_PER_BLOCK;
    add_bias_kernel_float<<<num_blocks, THREADS_PER_BLOCK>>>(
        output_ptr, output_ptr, bias_ptr, rows, cols);
    
    // Apply GELU
    gelu_kernel_float<<<num_blocks, THREADS_PER_BLOCK>>>(
        output_ptr, output_ptr, rows, cols);
    
    // Apply Softmax
    int softmax_blocks = rows;
    int threads_softmax = min(256, cols);
    softmax_kernel_float<<<softmax_blocks, threads_softmax>>>(
        output_ptr, output_ptr, rows, cols);
    
    CUDA_CHECK(cudaGetLastError());
    return output;
}

// Matmul + GELU + Softmax for float16
torch::Tensor matmul_gelu_softmax_fp16(torch::Tensor input, torch::Tensor weight,
                                       torch::Tensor bias, int rows, int cols) {
    init_cublas();
    
    auto output = torch::empty({rows, cols}, input.options());
    
    const __half* input_ptr = reinterpret_cast<const __half*>(input.data_ptr<torch::Half>());
    const __half* weight_ptr = reinterpret_cast<const __half*>(weight.data_ptr<torch::Half>());
    const __half* bias_ptr = reinterpret_cast<const __half*>(bias.data_ptr<torch::Half>());
    __half* output_ptr = reinterpret_cast<__half*>(output.data_ptr<torch::Half>());
    
    const float alpha = 1.0f;
    const float beta = 0.0f;
    
    CUBLAS_CHECK(cublasGemmEx(g_cublas_handle,
                              CUBLAS_OP_T,
                              CUBLAS_OP_N,
                              cols, rows, input.size(1),
                              &alpha,
                              weight_ptr, CUDA_R_16F, cols,
                              input_ptr, CUDA_R_16F, input.size(1),
                              &beta,
                              output_ptr, CUDA_R_16F, cols,
                              CUDA_R_32F,
                              CUBLAS_GEMM_DEFAULT_TENSOR_OP));
    
    // Add bias
    const int total_elements = rows * cols;
    const int num_blocks = (total_elements + THREADS_PER_BLOCK - 1) / THREADS_PER_BLOCK;
    add_bias_kernel_half<<<num_blocks, THREADS_PER_BLOCK>>>(
        output_ptr, output_ptr, bias_ptr, rows, cols);
    
    // Apply GELU
    gelu_kernel_half<<<num_blocks, THREADS_PER_BLOCK>>>(
        output_ptr, output_ptr, rows, cols);
    
    // Apply Softmax
    int softmax_blocks = rows;
    int threads_softmax = min(256, cols);
    softmax_kernel_half<<<softmax_blocks, threads_softmax>>>(
        output_ptr, output_ptr, rows, cols);
    
    CUDA_CHECK(cudaGetLastError());
    return output;
}

class Model(nn.Module):
    """
    Optimized model using fused CUDA kernels for matmul, GELU, and softmax.
    """
    def __init__(self, in_features, out_features, dtype=torch.float32):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.dtype = dtype
        
        self.linear = nn.Linear(in_features, out_features)
        
    def forward(self, x):
        x = x.to(self.dtype)
        
        weight = self.linear.weight.t()
        bias = self.linear.bias
        
        batch_size = x.size(0)
        
        if self.dtype == torch.float32:
            output = ext.matmul_gelu_softmax_fp32(
                x, weight, bias, batch_size, self.out_features)
        else:
            # For fp16 or bf16, use fp16 path
            x_f16 = x.half()
            weight_f16 = weight.half()
            bias_f16 = bias.half()
            output = ext.matmul_gelu_softmax_fp16(
                x_f16, weight_f16, bias_f16, batch_size, self.out_features)
            if self.dtype == torch.bfloat16:
                output = output.to(torch.bfloat16)
        
        return output
