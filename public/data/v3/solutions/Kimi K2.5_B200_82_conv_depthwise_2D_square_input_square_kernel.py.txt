import torch
import torch.nn as nn
import math
from torch.utils.cpp_extension import load_inline

# Global extension handle
_dwconv_ext = None

def _get_dwconv_ext():
    global _dwconv_ext
    if _dwconv_ext is None:
        cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename scalar_t>
__device__ __forceinline__ float to_float(scalar_t val);

template <>
__device__ __forceinline__ float to_float(c10::Half val) {
    return __half2float(*reinterpret_cast<const __half*>(&val));
}

template <>
__device__ __forceinline__ float to_float(c10::BFloat16 val) {
    return __bfloat162float(*reinterpret_cast<const nv_bfloat16*>(&val));
}

template <>
__device__ __forceinline__ float to_float(float val) {
    return val;
}

template <typename scalar_t>
__device__ __forceinline__ scalar_t from_float(float val);

template <>
__device__ __forceinline__ c10::Half from_float<c10::Half>(float val) {
    __half h = __float2half(val);
    return *reinterpret_cast<c10::Half*>(&h);
}

template <>
__device__ __forceinline__ c10::BFloat16 from_float<c10::BFloat16>(float val) {
    nv_bfloat16 b = __float2bfloat16(val);
    return *reinterpret_cast<c10::BFloat16*>(&b);
}

template <>
__device__ __forceinline__ float from_float<float>(float val) {
    return val;
}

template <typename scalar_t>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch_size,
    int in_channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    bool has_bias
) {
    int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    int channel = blockIdx.z % in_channels;
    int batch = blockIdx.z / in_channels;
    
    if (out_x >= out_width || out_y >= out_height || batch >= batch_size)
        return;
    
    int in_x_start = out_x * stride - padding;
    int in_y_start = out_y * stride - padding;
    
    float sum = 0.0f;
    
    const scalar_t* kernel_ptr = weight + channel * kernel_size * kernel_size;
    
    for (int ky = 0; ky < kernel_size; ++ky) {
        int in_y = in_y_start + ky;
        if (in_y >= 0 && in_y < in_height) {
            for (int kx = 0; kx < kernel_size; ++kx) {
                int in_x = in_x_start + kx;
                if (in_x >= 0 && in_x < in_width) {
                    int in_idx = ((batch * in_channels + channel) * in_height + in_y) * in_width + in_x;
                    float in_val = to_float(input[in_idx]);
                    float weight_val = to_float(kernel_ptr[ky * kernel_size + kx]);
                    sum += in_val * weight_val;
                }
            }
        }
    }
    
    if (has_bias) {
        sum += to_float(bias[channel]);
    }
    
    int out_idx = ((batch * in_channels + channel) * out_height + out_y) * out_width + out_x;
    output[out_idx] = from_float<scalar_t>(sum);
}

template <typename scalar_t>
torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    c10::optional<torch::Tensor> bias,
    int kernel_size,
    int stride,
    int padding
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int out_height = (in_height + 2 * padding - kernel_size) / stride + 1;
    int out_width = (in_width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, in_channels, out_height, out_width}, 
                               input.options());
    
    const int BLOCK_X = 16;
    const int BLOCK_Y = 16;
    
    dim3 block(BLOCK_X, BLOCK_Y);
    dim3 grid((out_width + BLOCK_X - 1) / BLOCK_X,
              (out_height + BLOCK_Y - 1) / BLOCK_Y,
              batch_size * in_channels);
    
    const scalar_t* bias_ptr = bias.has_value() ? bias.value().data_ptr<scalar_t>() : nullptr;
    
    depthwise_conv2d_kernel<scalar_t><<<grid, block>>>(
        input.data_ptr<scalar_t>(),
        weight.data_ptr<scalar_t>(),
        bias_ptr,
        output.data_ptr<scalar_t>(),
        batch_size,
        in_channels,
        in_height,
        in_width,
        out_height,
        out_width,
        kernel_size,
        stride,
        padding,
        bias.has_value()
    );
    
    return output;
}

torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, 
                               c10::optional<torch::Tensor> bias,
                               int kernel_size, int stride, int padding) {
    if (input.scalar_type() == at::ScalarType::Half) {
        return depthwise_conv2d_cuda<c10::Half>(input, weight, bias, kernel_size, stride, padding);
    } else if (input.scalar_type() == at::ScalarType::BFloat16) {
        return depthwise_conv2d_cuda<c10::BFloat16>(input, weight, bias, kernel_size, stride, padding);
    } else if (input.scalar_type() == at::ScalarType::Float) {
        return depthwise_conv2d_cuda<float>(input, weight, bias, kernel_size, stride, padding);
    } else {
        AT_ERROR("Unsupported dtype, only fp16, bf16, fp32 are supported");
    }
}
'''
        _dwconv_ext = load_inline(
            name='dwconv_ext_solution',
            cpp_sources='torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, c10::optional<torch::Tensor> bias, int kernel_size, int stride, int padding);',
            cuda_sources=cuda_source,
            functions=['depthwise_conv2d'],
            extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100', '-std=c++17'],
            verbose=False,
        )
    return _dwconv_ext

class Model(nn.Module):
    def __init__(self, in_channels, kernel_size, stride=1, padding=0, bias=False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
        self.weight = nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.randn(in_channels))
        else:
            self.bias = None
            
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if bias:
            fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
            bound = 1 / math.sqrt(fan_in)
            nn.init.uniform_(self.bias, -bound, bound)
        
    def forward(self, x):
        ext = _get_dwconv_ext()
        weight_reshaped = self.weight.squeeze(1)
        return ext.depthwise_conv2d(x, weight_reshaped, self.bias, 
                                     self.kernel_size, self.stride, self.padding)
