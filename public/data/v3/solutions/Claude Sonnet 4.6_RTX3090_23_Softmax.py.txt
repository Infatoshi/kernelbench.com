import torch
import torch.nn as nn
import triton
import triton.language as tl


# Triton softmax kernel: each program handles one row
# Uses online softmax with a single-pass approach (split into load+reduce, then normalize)
@triton.jit
def softmax_kernel(
    output_ptr,
    input_ptr,
    input_row_stride,
    output_row_stride,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one row
    row_idx = tl.program_id(0)
    row_start_ptr = input_ptr + row_idx * input_row_stride
    out_start_ptr = output_ptr + row_idx * output_row_stride

    # Load the row in blocks and compute max (for numerical stability)
    row_max = -float('inf')
    for block_start in range(0, n_cols, BLOCK_SIZE):
        col_offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = col_offsets < n_cols
        x = tl.load(row_start_ptr + col_offsets, mask=mask, other=-float('inf'))
        block_max = tl.max(x, axis=0)
        row_max = tl.maximum(row_max, block_max)

    # Compute sum of exp(x - max)
    row_sum = 0.0
    for block_start in range(0, n_cols, BLOCK_SIZE):
        col_offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = col_offsets < n_cols
        x = tl.load(row_start_ptr + col_offsets, mask=mask, other=-float('inf'))
        exp_x = tl.exp(x - row_max)
        # Zero out masked values for sum
        exp_x = tl.where(mask, exp_x, 0.0)
        row_sum += tl.sum(exp_x, axis=0)

    # Write normalized values
    for block_start in range(0, n_cols, BLOCK_SIZE):
        col_offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = col_offsets < n_cols
        x = tl.load(row_start_ptr + col_offsets, mask=mask, other=-float('inf'))
        exp_x = tl.exp(x - row_max)
        out = exp_x / row_sum
        tl.store(out_start_ptr + col_offsets, out, mask=mask)


# Optimized single-pass kernel when the entire row fits in BLOCK_SIZE
@triton.jit
def softmax_kernel_onepass(
    output_ptr,
    input_ptr,
    input_row_stride,
    output_row_stride,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0)
    row_start_ptr = input_ptr + row_idx * input_row_stride
    out_start_ptr = output_ptr + row_idx * output_row_stride

    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols

    x = tl.load(row_start_ptr + col_offsets, mask=mask, other=-float('inf'))
    x_max = tl.max(x, axis=0)
    exp_x = tl.exp(x - x_max)
    exp_x = tl.where(mask, exp_x, 0.0)
    sum_exp = tl.sum(exp_x, axis=0)
    out = exp_x / sum_exp

    tl.store(out_start_ptr + col_offsets, out, mask=mask)


def triton_softmax(x: torch.Tensor) -> torch.Tensor:
    assert x.ndim == 2, "Only 2D tensors supported"
    n_rows, n_cols = x.shape

    # Ensure output is contiguous float32
    output = torch.empty_like(x)

    # Choose block size
    # For n_cols=16384, we want BLOCK_SIZE >= 16384
    # Next power of 2 >= n_cols
    BLOCK_SIZE = triton.next_power_of_2(n_cols)

    # If fits in a single block, use the one-pass kernel
    # Max BLOCK_SIZE for Triton is typically 131072 for simple kernels
    if BLOCK_SIZE <= 131072:
        softmax_kernel_onepass[(n_rows,)](
            output,
            x,
            x.stride(0),
            output.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=min(32, max(4, BLOCK_SIZE // 256)),
        )
    else:
        # Multi-block approach: use a fixed BLOCK_SIZE of 1024 or 2048
        BLOCK_SIZE = 2048
        softmax_kernel[(n_rows,)](
            output,
            x,
            x.stride(0),
            output.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=8,
        )

    return output


class Model(nn.Module):
    """
    Optimized model that performs a Softmax activation using a Triton kernel.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_softmax(x)
