import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

#define M 32
#define N 14336
#define K 4096

__global__ void gemv_fp8_kernel(
    const signed char* __restrict__ A,
    const signed char* __restrict__ B,
    at::Half* __restrict__ C,
    float scale_a,
    float scale_b
) {
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (n >= N) return;
    
    float scale = scale_a * scale_b;
    
    // Compute dot product of A (M, K) with B^T (K,) for this output n
    // B is (N, K), so B[n, k] = B[n * K + k]
    
    #pragma unroll 4
    for (int m = 0; m < M; m++) {
        float sum = 0.0f;
        
        // Process K=4096 elements
        #pragma unroll 32
        for (int k = 0; k < K; k++) {
            float a_val = float(A[m * K + k]);
            float b_val = float(B[n * K + k]);
            sum += a_val * b_val;
        }
        
        C[m * N + n] = __float2half(sum * scale);
    }
}

torch::Tensor gemv_fp8_cuda(
    torch::Tensor x_q,
    torch::Tensor x_scale,
    torch::Tensor weight_q,
    torch::Tensor weight_scale
) {
    auto options = torch::TensorOptions().dtype(torch::kFloat16).device(x_q.device());
    auto output = torch::empty({M, N}, options);
    
    const int threads = 256;
    const int blocks = (N + threads - 1) / threads;
    
    float scale_a = x_scale.item<float>();
    float scale_b = weight_scale.item<float>();
    
    gemv_fp8_kernel<<<blocks, threads>>>(
        x_q.data_ptr<signed char>(),
        weight_q.data_ptr<signed char>(),
        output.data_ptr<at::Half>(),
        scale_a,
        scale_b
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor gemv_fp8_cuda(torch::Tensor x_q, torch::Tensor x_scale, torch::Tensor weight_q, torch::Tensor weight_scale);
'''

ext = load_inline(
    name='gemv_fp8_ext_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp8_cuda'],
    extra_cuda_cflags=['-O3'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp8_cuda(x_q, x_scale, self.weight_q, self.weight_scale)


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1

    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]
