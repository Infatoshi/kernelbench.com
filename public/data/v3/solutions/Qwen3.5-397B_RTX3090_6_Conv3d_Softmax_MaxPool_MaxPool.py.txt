cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <algorithm>
#include <cmath>

// Fused kernel: Conv3d + Softmax (dim=1) + 2x MaxPool3d
// Input: (N, C_in, D, H, W)
// Conv output: (N, C_out, D-K+1, H-K+1, W-K+1)
// After 2x maxpool with kernel_size=2: (N, C_out, (D-K+1)/4, (H-K+1)/4, (W-K+1)/4)

template<typename scalar_t>
__global__ void fused_conv3d_softmax_double_pool_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int N, int C_in, int D, int H, int W,
    int C_out, int K,
    int pool_K,
    int out_D, int out_H, int out_W
) {
    // Each block handles one output element (n, c_out, od, oh, ow)
    // We need to:
    // 1. Compute conv for all C_out channels at 4x4x4 spatial positions
    // 2. Apply softmax across channels at each position
    // 3. Apply max over the 4x4x4 region (double maxpool)
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_out = N * C_out * out_D * out_H * out_W;
    
    if (idx >= total_out) return;
    
    // Decode output index
    int tmp = idx;
    int ow = tmp % out_W; tmp /= out_W;
    int oh = tmp % out_H; tmp /= out_H;
    int od = tmp % out_D; tmp /= out_D;
    int c_out = tmp % C_out; tmp /= C_out;
    int n = tmp;
    
    // Base coordinates in conv output space (before pooling)
    // Each pool reduces by factor of pool_K, so 2 pools = factor of pool_K^2
    int base_cd = od * pool_K * pool_K;
    int base_ch = oh * pool_K * pool_K;
    int base_cw = ow * pool_K * pool_K;
    
    // Conv output dimensions
    int conv_D = D - K + 1;
    int conv_H = H - K + 1;
    int conv_W = W - K + 1;
    
    // We need to compute conv at all 4x4x4 = 64 positions in the pool region
    // Then apply softmax across channels, then take max
    
    // For numerical stability in softmax, find max across channels for each position
    float position_max[64];
    for (int region_idx = 0; region_idx < 64; region_idx++) {
        position_max[region_idx] = -__FLT_MAX__;
    }
    
    // First pass: compute conv and find max per position (for softmax stability)
    for (int region_idx = 0; region_idx < 64; region_idx++) {
        int rd = region_idx / 16;
        int rh = (region_idx % 16) / 4;
        int rw = region_idx % 4;
        
        int cd = base_cd + rd;
        int ch = base_ch + rh;
        int cw = base_cw + rw;
        
        if (cd >= conv_D || ch >= conv_H || cw >= conv_W) {
            continue;  // Out of bounds, keep -inf
        }
        
        float conv_val = bias ? (float)bias[c_out] : 0.0f;
        
        for (int kc = 0; kc < C_in; kc++) {
            for (int kd = 0; kd < K; kd++) {
                for (int kh = 0; kh < K; kh++) {
                    for (int kw = 0; kw < K; kw++) {
                        int id = cd + kd;
                        int ih = ch + kh;
                        int iw = cw + kw;
                        if (id < D && ih < H && iw < W) {
                            int input_idx = ((n * C_in + kc) * D + id) * (H * W) + ih * W + iw;
                            int weight_idx = ((c_out * C_in + kc) * K + kd) * (K * K) + kh * K + kw;
                            conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                        }
                    }
                }
            }
        }
        
        if (conv_val > position_max[region_idx]) {
            position_max[region_idx] = conv_val;
        }
    }
    
    // We need to share position_max across threads in the same block for softmax
    // But each thread is independent here, so we need a different approach
    // Let's use a simpler approach: each thread computes everything for its output element
    
    // Recompute conv values for all channels at all positions for softmax
    float softmax_val = 0.0f;
    float my_conv_val[64];
    
    for (int region_idx = 0; region_idx < 64; region_idx++) {
        int rd = region_idx / 16;
        int rh = (region_idx % 16) / 4;
        int rw = region_idx % 4;
        
        int cd = base_cd + rd;
        int ch = base_ch + rh;
        int cw = base_cw + rw;
        
        if (cd >= conv_D || ch >= conv_H || cw >= conv_W) {
            my_conv_val[region_idx] = -__FLT_MAX__;
            continue;
        }
        
        float conv_val = bias ? (float)bias[c_out] : 0.0f;
        
        for (int kc = 0; kc < C_in; kc++) {
            for (int kd = 0; kd < K; kd++) {
                for (int kh = 0; kh < K; kh++) {
                    for (int kw = 0; kw < K; kw++) {
                        int id = cd + kd;
                        int ih = ch + kh;
                        int iw = cw + kw;
                        if (id < D && ih < H && iw < W) {
                            int input_idx = ((n * C_in + kc) * D + id) * (H * W) + ih * W + iw;
                            int weight_idx = ((c_out * C_in + kc) * K + kd) * (K * K) + kh * K + kw;
                            conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                        }
                    }
                }
            }
        }
        
        my_conv_val[region_idx] = conv_val;
    }
    
    // Now compute softmax: we need sum of exp(val - max) across all channels at each position
    // This requires knowing all channel values, which we don't have in this thread
    // We need to restructure: use shared memory or a multi-pass approach
    
    // Simpler approach: since C_out is small (16), we can iterate over all channels
    // But we need to recompute conv for all channels - very inefficient but correct
    
    // Let's compute softmax denominator for each position
    float sum_exp[64];
    for (int region_idx = 0; region_idx < 64; region_idx++) {
        sum_exp[region_idx] = 0.0f;
    }
    
    for (int c = 0; c < C_out; c++) {
        for (int region_idx = 0; region_idx < 64; region_idx++) {
            int rd = region_idx / 16;
            int rh = (region_idx % 16) / 4;
            int rw = region_idx % 4;
            
            int cd = base_cd + rd;
            int ch = base_ch + rh;
            int cw = base_cw + rw;
            
            if (cd >= conv_D || ch >= conv_H || cw >= conv_W) {
                continue;
            }
            
            float conv_val = bias ? (float)bias[c] : 0.0f;
            
            for (int kc = 0; kc < C_in; kc++) {
                for (int kd = 0; kd < K; kd++) {
                    for (int kh = 0; kh < K; kh++) {
                        for (int kw = 0; kw < K; kw++) {
                            int id = cd + kd;
                            int ih = ch + kh;
                            int iw = cw + kw;
                            if (id < D && ih < H && iw < W) {
                                int input_idx = ((n * C_in + kc) * D + id) * (H * W) + ih * W + iw;
                                int weight_idx = ((c * C_in + kc) * K + kd) * (K * K) + kh * K + kw;
                                conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                            }
                        }
                    }
                }
            }
            
            // Update position max
            if (conv_val > position_max[region_idx]) {
                position_max[region_idx] = conv_val;
            }
        }
    }
    
    // Second pass: compute sum of exp(val - max) for each position
    for (int c = 0; c < C_out; c++) {
        for (int region_idx = 0; region_idx < 64; region_idx++) {
            int rd = region_idx / 16;
            int rh = (region_idx % 16) / 4;
            int rw = region_idx % 4;
            
            int cd = base_cd + rd;
            int ch = base_ch + rh;
            int cw = base_cw + rw;
            
            if (cd >= conv_D || ch >= conv_H || cw >= conv_W) {
                continue;
            }
            
            float conv_val = bias ? (float)bias[c] : 0.0f;
            
            for (int kc = 0; kc < C_in; kc++) {
                for (int kd = 0; kd < K; kd++) {
                    for (int kh = 0; kh < K; kh++) {
                        for (int kw = 0; kw < K; kw++) {
                            int id = cd + kd;
                            int ih = ch + kh;
                            int iw = cw + kw;
                            if (id < D && ih < H && iw < W) {
                                int input_idx = ((n * C_in + kc) * D + id) * (H * W) + ih * W + iw;
                                int weight_idx = ((c * C_in + kc) * K + kd) * (K * K) + kh * K + kw;
                                conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                            }
                        }
                    }
                }
            }
            
            sum_exp[region_idx] += expf(conv_val - position_max[region_idx]);
        }
    }
    
    // Third pass: compute softmax for our channel and find max over region
    float final_max = -__FLT_MAX__;
    
    for (int region_idx = 0; region_idx < 64; region_idx++) {
        int rd = region_idx / 16;
        int rh = (region_idx % 16) / 4;
        int rw = region_idx % 4;
        
        int cd = base_cd + rd;
        int ch = base_ch + rh;
        int cw = base_cw + rw;
        
        if (cd >= conv_D || ch >= conv_H || cw >= conv_W) {
            continue;
        }
        
        float conv_val = bias ? (float)bias[c_out] : 0.0f;
        
        for (int kc = 0; kc < C_in; kc++) {
            for (int kd = 0; kd < K; kd++) {
                for (int kh = 0; kh < K; kh++) {
                    for (int kw = 0; kw < K; kw++) {
                        int id = cd + kd;
                        int ih = ch + kh;
                        int iw = cw + kw;
                        if (id < D && ih < H && iw < W) {
                            int input_idx = ((n * C_in + kc) * D + id) * (H * W) + ih * W + iw;
                            int weight_idx = ((c_out * C_in + kc) * K + kd) * (K * K) + kh * K + kw;
                            conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                        }
                    }
                }
            }
        }
        
        float softmax = expf(conv_val - position_max[region_idx]) / sum_exp[region_idx];
        if (softmax > final_max) {
            final_max = softmax;
        }
    }
    
    output[idx] = (scalar_t)final_max;
}

torch::Tensor fused_conv3d_softmax_pool2(torch::Tensor input, torch::Tensor weight, torch::Tensor bias,
                                          int pool_kernel_size) {
    int N = input.size(0);
    int C_in = input.size(1);
    int D = input.size(2);
    int H = input.size(3);
    int W = input.size(4);

    int C_out = weight.size(0);
    int K = weight.size(2);

    int conv_D = D - K + 1;
    int conv_H = H - K + 1;
    int conv_W = W - K + 1;

    int out_D = conv_D / (pool_kernel_size * pool_kernel_size);
    int out_H = conv_H / (pool_kernel_size * pool_kernel_size);
    int out_W = conv_W / (pool_kernel_size * pool_kernel_size);

    auto output = torch::empty({N, C_out, out_D, out_H, out_W}, input.options());

    int total_out = N * C_out * out_D * out_H * out_W;
    int threads_per_block = 256;
    int num_blocks = (total_out + threads_per_block - 1) / threads_per_block;

    if (input.dtype() == torch::kHalf) {
        fused_conv3d_softmax_double_pool_kernel<at::Half><<<num_blocks, threads_per_block>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            bias.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            N, C_in, D, H, W,
            C_out, K,
            pool_kernel_size,
            out_D, out_H, out_W
        );
    } else if (input.dtype() == torch::kBFloat16) {
        fused_conv3d_softmax_double_pool_kernel<at::BFloat16><<<num_blocks, threads_per_block>>>(
            input.data_ptr<at::BFloat16>(),
            weight.data_ptr<at::BFloat16>(),
            bias.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            N, C_in, D, H, W,
            C_out, K,
            pool_kernel_size,
            out_D, out_H, out_W
        );
    } else {
        fused_conv3d_softmax_double_pool_kernel<float><<<num_blocks, threads_per_block>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            N, C_in, D, H, W,
            C_out, K,
            pool_kernel_size,
            out_D, out_H, out_W
        );
    }

    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_conv3d_softmax_pool2(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int pool_kernel_size);
"""

ext = load_inline(
    name='fused_conv3d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv3d_softmax_pool2'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies Softmax, and performs two max pooling operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        weight = self.conv.weight
        bias = self.conv.bias
        return ext.fused_conv3d_softmax_pool2(x, weight, bias, self.pool_kernel_size)
