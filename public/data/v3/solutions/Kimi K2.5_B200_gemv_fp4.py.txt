cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <stdint.h>

// Optimized FP4-simulated GEMV kernel for B200
// Uses int8 arithmetic with efficient memory access patterns

__global__ void gemv_fp4_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    float scale,
    half* __restrict__ out,
    int M, int N, int K
) {
    // Grid: (N/128, M) - parallel over both M and N
    // Block: 128 threads
    // Each thread computes one output element
    
    int tid = threadIdx.x;
    int n_idx = blockIdx.x * blockDim.x + tid;
    int m_idx = blockIdx.y;
    
    if (m_idx >= M || n_idx >= N) return;
    
    const int8_t* x_row = x_q + m_idx * K;
    const int8_t* w_row = weight_q + n_idx * K;
    
    int32_t acc = 0;
    
    #pragma unroll 32
    for (int k = 0; k < K; k++) {
        acc += int32_t(x_row[k]) * int32_t(w_row[k]);
    }
    
    out[m_idx * N + n_idx] = __float2half_rn(float(acc) * scale);
}

// Vectorized version with 4-element processing
__global__ void gemv_fp4_vec4_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    float scale,
    half* __restrict__ out,
    int M, int N, int K
) {
    int tid = threadIdx.x;
    int n_idx = blockIdx.x * blockDim.x + tid;
    int m_idx = blockIdx.y;
    
    if (m_idx >= M || n_idx >= N) return;
    
    const int8_t* x_row = x_q + m_idx * K;
    const int8_t* w_row = weight_q + n_idx * K;
    
    int32_t acc = 0;
    int k = 0;
    
    // Process 4 elements at a time
    #pragma unroll 32
    for (; k + 4 <= K; k += 4) {
        // Load and compute
        int8_t x0 = x_row[k];
        int8_t x1 = x_row[k+1];
        int8_t x2 = x_row[k+2];
        int8_t x3 = x_row[k+3];
        int8_t w0 = w_row[k];
        int8_t w1 = w_row[k+1];
        int8_t w2 = w_row[k+2];
        int8_t w3 = w_row[k+3];
        
        acc += int32_t(x0) * int32_t(w0);
        acc += int32_t(x1) * int32_t(w1);
        acc += int32_t(x2) * int32_t(w2);
        acc += int32_t(x3) * int32_t(w3);
    }
    
    // Handle remaining elements
    for (; k < K; k++) {
        acc += int32_t(x_row[k]) * int32_t(w_row[k]);
    }
    
    out[m_idx * N + n_idx] = __float2half_rn(float(acc) * scale);
}

tensorrt_llm_kernel(const int8_t* __restrict__ x_q, const int8_t* __restrict__ weight_q, float scale, half* __restrict__ out,
    int tid = threadIdx.x; int n_idx = blockIdx.x * blockDim.x + tid; int m_idx = blockIdx.y;
    
    if (m_idx >= M || n_idx >= N) return;
    
    const int8_t* x_row = x_q + m_idx * K;
    const int8_t* w_row = weight_q + n_idx * K;

        acc += float(x_row[k]) * float(w_row[k]);
    }

torch::Tensor gemv_fp4_cuda(
    torch::Tensor x_q,
    torch::Tensor weight_q,
    float x_scale,
    float weight_scale
) {
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = weight_q.size(0);
    
    auto out = torch::empty({M, N}, torch::dtype(torch::kFloat16).device(x_q.device()));
    
    float scale = x_scale * weight_scale;
    
    const int threads = 128;
    dim3 blocks((N + threads - 1) / threads, M);
    
    // Use vectorized kernel for better performance
    gemv_fp4_vec4_kernel<<<blocks, threads>>>(
        x_q.data_ptr<int8_t>(),
        weight_q.data_ptr<int8_t>(),
        scale,
        reinterpret_cast<half*>(out.data_ptr<at::Half>()),
        M, N, K
    );
    
    return out;
}

#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void gemv_simple_kernel(
    const int8_t* x_q,
    const int8_t* weight_q,
    float scale,
    half* out,
    int M, int N, int K
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = M * N;
    if (idx >= total) return;
    
    int m = idx / N;
    int n = idx % N;
    
    float acc = 0.0f;
    for (int k = 0; k < K; k++) {
        acc += float(x_q[m * K + k]) * float(weight_q[n * K + k]);
    }
    out[idx] = __float2half_rn(acc * scale);
}

torch::Tensor gemv_fp4_cuda(
    torch::Tensor x_q,
    torch::Tensor weight_q,
    float x_scale,
    float weight_scale
) {
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = weight_q.size(0);
    
    auto out = torch::empty({M, N}, torch::dtype(torch::kFloat16).device(x_q.device()));
    float scale = x_scale * weight_scale;
    
    const int threads = 256;
    const int blocks = (M * N + threads - 1) / threads;

        x_q.data_ptr<int8_t>(),
        weight_q.data_ptr<int8_t>(),
        scale,
        reinterpret_cast<half*>(out.data_ptr<at::Half>()),
        M, N, K
    );

}

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        # Simulate fp4 range in int8 storage.
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # Use optimized CUDA kernel instead of PyTorch operations
        return ext.gemv_fp4_cuda(
            x_q, 
            self.weight_q, 
            x_scale.item() if torch.is_tensor(x_scale) else float(x_scale), 
            self.weight_scale.item() if torch.is_tensor(self.weight_scale) else float(self.weight_scale)
        )
