"""Persistent GEMM using cuda.tile for B200 (Blackwell) optimization."""

import torch
import torch.nn as nn
import cuda.tile as ct

# Tile dimensions for optimal performance on B200
TM = 128
TN = 128
TK = 32

ConstInt = ct.Constant[int]

@ct.kernel
def persistent_gemm_kernel(
    A, B, C,
    tm: ConstInt, tn: ConstInt, tk: ConstInt,
    num_tiles: ConstInt
):
    """Persistent GEMM kernel that processes multiple output tiles per block."""
    tile_id = ct.bid(0)
    tile_stride = ct.num_blocks(0)
    
    # Get dimensions at compile time
    M = A.shape[0]
    K = A.shape[1]
    N = B.shape[1]
    
    # Number of tiles in M and N dimensions
    num_tiles_m = ct.cdiv(N, tn)
    
    # Persistent loop: each block processes multiple tiles
    while tile_id < num_tiles:
        # Compute tile coordinates
        pid_m = tile_id // num_tiles_m
        pid_n = tile_id % num_tiles_m
        
        # Accumulator for this tile
        acc = ct.full((tm, tn), 0.0, dtype=ct.float32)
        
        # Loop over K dimension
        num_k_tiles = ct.cdiv(K, tk)
        for k_tile in range(num_k_tiles):
            # Load tiles with zero padding for boundary tiles
            a = ct.load(A, (pid_m, k_tile), (tm, tk), padding_mode=ct.PaddingMode.ZERO)
            b = ct.load(B, (k_tile, pid_n), (tk, tn), padding_mode=ct.PaddingMode.ZERO)
            # Matrix multiply accumulate
            acc = ct.mma(a, b, acc)
        
        # Store result with type conversion
        ct.store(C, (pid_m, pid_n), ct.astype(acc, C.dtype))
        
        # Move to next tile
        tile_id += tile_stride


class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        M, K = a.shape
        K2, N = b.shape
        
        # Create output tensor
        c = torch.empty(M, N, dtype=a.dtype, device=a.device)
        
        # Calculate number of output tiles
        num_tiles_m = (M + TM - 1) // TM
        num_tiles_n = (N + TN - 1) // TN
        num_tiles = num_tiles_m * num_tiles_n
        
        # Number of SMs on B200 (use as number of persistent blocks)
        # B200 has 192 SMs, use all for maximum occupancy
        num_sms = 192
        
        # Launch the kernel
        grid = (min(num_tiles, num_sms),)
        ct.launch(
            torch.cuda.current_stream(),
            grid,
            persistent_gemm_kernel,
            (a, b, c, TM, TN, TK, num_tiles)
        )
        
        return c
