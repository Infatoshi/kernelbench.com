import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# CUDA source for fused post-GEMM operations:
# mean(dim=0) then softmax(dim=1)
# We parallelize across columns using multiple blocks.
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <math.h>
#include <float.h>

// =====================================================================
// Fused mean(dim=0) + softmax(dim=1) kernel
// Strategy:
//   Phase 1 (kernel 1): compute col means using many blocks (1 block per col group)
//   Phase 2 (kernel 2): softmax over the 1D mean vector
//
// OR single-kernel approach with 2 passes using atomics for the softmax.
//
// Actually, the cleanest fast approach:
// 1. Launch mean_kernel with many blocks -> col_means[4096]
// 2. Launch softmax_kernel with 1 block -> output[4096]
// 
// Mean kernel: each block handles COLS_PER_BLOCK columns, reduces over rows.
// With batch=128 rows and 4096 cols:
//   Use 4096 threads total in gridDim, each computes 1 column's mean.
// =====================================================================

// Fast mean kernel: 1 thread per output column
// Each thread sums over batch rows
__global__ void mean_cols_kernel_fp32(
    const float* __restrict__ X,
    float* __restrict__ col_means,
    int batch,
    int cols
) {
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (col >= cols) return;
    
    float sum = 0.0f;
    const float* col_ptr = X + col;
    for (int i = 0; i < batch; i++) {
        sum += col_ptr[i * cols];
    }
    col_means[col] = sum / (float)batch;
}

// Softmax over 1D vector of length n
// Using a single block with shared memory reduction
// For n=4096 and 1024 threads, each thread handles 4 elements
__global__ void softmax_1d_kernel_fp32(
    const float* __restrict__ in,
    float* __restrict__ out,
    int n
) {
    extern __shared__ float smem[];
    // smem[0..blockDim.x-1] for reductions
    
    int tid = threadIdx.x;
    int num_threads = blockDim.x;
    
    // Step 1: find max using reduction
    float local_max = -FLT_MAX;
    for (int i = tid; i < n; i += num_threads) {
        local_max = fmaxf(local_max, in[i]);
    }
    smem[tid] = local_max;
    __syncthreads();
    
    for (int stride = num_threads >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            smem[tid] = fmaxf(smem[tid], smem[tid + stride]);
        }
        __syncthreads();
    }
    float global_max = smem[0];
    __syncthreads();
    
    // Step 2: compute sum of exp(x - max)
    float local_sum = 0.0f;
    for (int i = tid; i < n; i += num_threads) {
        local_sum += __expf(in[i] - global_max);
    }
    smem[tid] = local_sum;
    __syncthreads();
    
    for (int stride = num_threads >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            smem[tid] += smem[tid + stride];
        }
        __syncthreads();
    }
    float inv_sum = 1.0f / smem[0];
    __syncthreads();
    
    // Step 3: write normalized output
    for (int i = tid; i < n; i += num_threads) {
        out[i] = __expf(in[i] - global_max) * inv_sum;
    }
}

// Fused: mean + softmax in 2 kernels dispatched back-to-back
// But we need an intermediate buffer.
// Instead, let's try a cooperative approach using a persistent kernel with atomics.

// Actually, the fastest approach for this problem:
// Use a column-parallel mean with warp-level reduction,
// then a fast softmax.

// Approach: 1 warp per column for reduction (batch=128)
// 128 warps = 4 thread blocks of 32 warps each (for 4096 cols)
// Each warp (32 threads) reduces 128 rows -> 4 elements per thread

__global__ void mean_cols_warp_kernel_fp32(
    const float* __restrict__ X,
    float* __restrict__ col_means,
    int batch,
    int cols
) {
    // Each warp handles one column
    int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / 32;
    int lane_id = threadIdx.x % 32;
    int col = warp_id;
    
    if (col >= cols) return;
    
    float sum = 0.0f;
    const float* col_ptr = X + col;
    for (int i = lane_id; i < batch; i += 32) {
        sum += col_ptr[i * cols];
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        col_means[col] = sum / (float)batch;
    }
}

// =====================================================================
// DROPOUT version
// =====================================================================
__global__ void mean_cols_dropout_warp_kernel_fp32(
    const float* __restrict__ X,
    const bool* __restrict__ mask,
    float* __restrict__ col_means,
    float scale,
    int batch,
    int cols
) {
    int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / 32;
    int lane_id = threadIdx.x % 32;
    int col = warp_id;
    
    if (col >= cols) return;
    
    float sum = 0.0f;
    const float* col_ptr = X + col;
    const bool* mask_ptr = mask + col;
    for (int i = lane_id; i < batch; i += 32) {
        float val = col_ptr[i * cols];
        sum += mask_ptr[i * cols] ? val * scale : 0.0f;
    }
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        col_means[col] = sum / (float)batch;
    }
}

// =====================================================================
// Transposed memory access version: load X transposed for coalescing
// X[batch, cols] -> for col reduction, we want to iterate over rows
// With 1 thread per col, access is strided -> bad
// With warp-per-col and 32 threads stepping by 32 over 128 rows -> coalesced
// =====================================================================

// =====================================================================
// Best single-kernel fused approach using shared memory:
// Load tiles of X into shared memory, accumulate into col_sums
// Then do softmax on col_sums
// =====================================================================

// Tile-based mean: each block handles TILE_COLS columns
// Threads: TILE_COLS threads in x, each accumulates all rows for its column
// Then we run softmax on the resulting means
template<int TILE_COLS, int THREADS>
__global__ void tiled_mean_kernel_fp32(
    const float* __restrict__ X,
    float* __restrict__ col_means,
    int batch,
    int cols
) {
    // blockIdx.x selects which tile of columns
    // Each thread handles one column
    int col = blockIdx.x * THREADS + threadIdx.x;
    if (col >= cols) return;
    
    float sum = 0.0f;
    const float* col_ptr = X + col;
    
    // Unroll for batch=128
    #pragma unroll 4
    for (int i = 0; i < batch; i++) {
        sum += col_ptr[i * cols];
    }
    col_means[col] = sum * (1.0f / batch);
}

// =====================================================================
// C++ interface functions
// =====================================================================

torch::Tensor fused_mean_softmax_fp32(torch::Tensor X) {
    int batch = X.size(0);
    int cols = X.size(1);
    
    auto col_means = torch::empty({cols}, X.options());
    auto out = torch::empty({1, cols}, X.options());
    
    // Step 1: compute column means
    // Use warp-per-column for good coalescing (batch=128, 4 elements per lane)
    // 4096 columns -> 4096 warps -> 4096*32 = 131072 threads = 512 blocks of 256 threads
    int warps_per_block = 8; // 256 threads / 32
    int blocks = (cols + warps_per_block - 1) / warps_per_block;
    
    mean_cols_warp_kernel_fp32<<<blocks, warps_per_block * 32>>>(
        X.data_ptr<float>(),
        col_means.data_ptr<float>(),
        batch, cols
    );
    
    // Step 2: softmax over col_means
    int sm_threads = 1024;
    int smem = sm_threads * sizeof(float);
    softmax_1d_kernel_fp32<<<1, sm_threads, smem>>>(
        col_means.data_ptr<float>(),
        out.data_ptr<float>(),
        cols
    );
    
    return out;
}

torch::Tensor fused_dropout_mean_softmax_fp32(
    torch::Tensor X,
    torch::Tensor mask,
    float scale
) {
    int batch = X.size(0);
    int cols = X.size(1);
    
    auto col_means = torch::empty({cols}, X.options());
    auto out = torch::empty({1, cols}, X.options());
    
    int warps_per_block = 8;
    int blocks = (cols + warps_per_block - 1) / warps_per_block;
    
    mean_cols_dropout_warp_kernel_fp32<<<blocks, warps_per_block * 32>>>(
        X.data_ptr<float>(),
        mask.data_ptr<bool>(),
        col_means.data_ptr<float>(),
        scale,
        batch, cols
    );
    
    int sm_threads = 1024;
    int smem = sm_threads * sizeof(float);
    softmax_1d_kernel_fp32<<<1, sm_threads, smem>>>(
        col_means.data_ptr<float>(),
        out.data_ptr<float>(),
        cols
    );
    
    return out;
}
'''

cpp_source = r'''
torch::Tensor fused_mean_softmax_fp32(torch::Tensor X);
torch::Tensor fused_dropout_mean_softmax_fp32(torch::Tensor X, torch::Tensor mask, float scale);
'''

ext = load_inline(
    name='fused_ops_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_mean_softmax_fp32', 'fused_dropout_mean_softmax_fp32'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Optimized: cuBLAS GEMM + custom fused mean+softmax kernel
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout_p = dropout_p

    def forward(self, x):
        weight = self.matmul.weight  # [out_features, in_features]
        bias = self.matmul.bias      # [out_features]
        
        # GEMM: cuBLAS SGEMM
        x_linear = torch.addmm(bias, x, weight.t())  # [batch, out_features]
        
        if self.training and self.dropout_p > 0.0:
            mask = torch.rand_like(x_linear) >= self.dropout_p
            scale = 1.0 / (1.0 - self.dropout_p)
            if x_linear.dtype == torch.float32:
                return ext.fused_dropout_mean_softmax_fp32(x_linear, mask, scale)
            else:
                x_linear = x_linear * mask.float() * scale
                x_mean = x_linear.mean(dim=0, keepdim=True)
                return torch.softmax(x_mean, dim=1)
        else:
            if x_linear.dtype == torch.float32:
                return ext.fused_mean_softmax_fp32(x_linear)
            else:
                x_mean = x_linear.mean(dim=0, keepdim=True)
                return torch.softmax(x_mean, dim=1)


batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, dropout_p]
