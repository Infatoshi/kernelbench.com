import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fused bloom kernel: threshold -> 5x5 avg pool -> blend -> clamp
// Optimized with shared memory tiling

template<int BLOCK_X, int BLOCK_Y, int KERNEL_R>
__global__ void bloom_fused_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    const int H, const int W,
    const float threshold, const float intensity, const float inv_kernel_area
) {
    constexpr int SMEM_X = BLOCK_X + 2 * KERNEL_R;
    constexpr int SMEM_Y = BLOCK_Y + 2 * KERNEL_R;
    
    __shared__ float smem[SMEM_Y * SMEM_X];
    
    const int nc = blockIdx.z;
    const int base_offset = nc * H * W;
    
    const int tile_x = blockIdx.x * BLOCK_X;
    const int tile_y = blockIdx.y * BLOCK_Y;
    
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int tid = ty * BLOCK_X + tx;
    const int total_threads = BLOCK_X * BLOCK_Y;
    
    // Load shared memory with halo - compute bright = relu(val - threshold) during load
    constexpr int smem_total = SMEM_X * SMEM_Y;
    #pragma unroll 4
    for (int i = tid; i < smem_total; i += total_threads) {
        int sy = i / SMEM_X;
        int sx = i % SMEM_X;
        int gy = tile_y + sy - KERNEL_R;
        int gx = tile_x + sx - KERNEL_R;
        
        float val = 0.0f;
        if ((unsigned)gy < (unsigned)H && (unsigned)gx < (unsigned)W) {
            val = __ldg(&input[base_offset + gy * W + gx]);
            val = fmaxf(val - threshold, 0.0f);
        }
        smem[i] = val;
    }
    
    __syncthreads();
    
    int gy = tile_y + ty;
    int gx = tile_x + tx;
    
    if (gy < H && gx < W) {
        // Compute 5x5 sum from shared memory - fully unrolled
        float sum = 0.0f;
        int base_sy = ty * SMEM_X + tx;  // offset for (ty+0, tx+0) in smem
        
        #pragma unroll
        for (int dy = 0; dy < 5; dy++) {
            int row_base = base_sy + dy * SMEM_X;
            #pragma unroll
            for (int dx = 0; dx < 5; dx++) {
                sum += smem[row_base + dx];
            }
        }
        
        float blurred = sum * inv_kernel_area;
        float orig = __ldg(&input[base_offset + gy * W + gx]);
        float result = orig + intensity * blurred;
        result = fminf(fmaxf(result, 0.0f), 10.0f);
        output[base_offset + gy * W + gx] = result;
    }
}

torch::Tensor bloom_forward(torch::Tensor input, float threshold, float intensity, int blur_kernel) {
    TORCH_CHECK(input.is_cuda(), "Input must be CUDA tensor");
    TORCH_CHECK(input.dim() == 4, "Input must be 4D (N,C,H,W)");
    TORCH_CHECK(blur_kernel == 5, "Only blur_kernel=5 is supported");
    
    auto output = torch::empty_like(input);
    
    const int N = input.size(0);
    const int C = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    
    constexpr int BLOCK_X = 32;
    constexpr int BLOCK_Y = 8;
    
    dim3 block(BLOCK_X, BLOCK_Y);
    dim3 grid((W + BLOCK_X - 1) / BLOCK_X, (H + BLOCK_Y - 1) / BLOCK_Y, N * C);
    
    float inv_kernel_area = 1.0f / (float)(blur_kernel * blur_kernel);
    
    bloom_fused_kernel<32, 8, 2><<<grid, block>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        H, W,
        threshold, intensity, inv_kernel_area
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor bloom_forward(torch::Tensor input, float threshold, float intensity, int blur_kernel);
'''

ext = load_inline(
    name='bloom_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bloom_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        return ext.bloom_forward(image, self.threshold, self.intensity, self.blur_kernel)


def get_inputs():
    return [torch.randn(2, 3, 1024, 1024) * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
