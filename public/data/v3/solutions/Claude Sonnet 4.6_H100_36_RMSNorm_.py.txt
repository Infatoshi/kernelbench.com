import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rmsnorm_kernel(
    X_ptr, Y_ptr,
    B, C, HW,
    eps,
    stride_b, stride_c, stride_hw,
    BLOCK_C: tl.constexpr,
    BLOCK_HW: tl.constexpr,
):
    # Each program handles BLOCK_HW spatial positions
    # pid_hw: which tile of HW
    # pid_b:  which batch
    pid_hw = tl.program_id(0)
    pid_b  = tl.program_id(1)

    hw_start = pid_hw * BLOCK_HW
    hw_offsets = hw_start + tl.arange(0, BLOCK_HW)
    hw_mask = hw_offsets < HW

    c_offsets = tl.arange(0, BLOCK_C)
    c_mask = c_offsets < C

    # Load all C features for BLOCK_HW positions
    # Shape: [BLOCK_C, BLOCK_HW]
    # X layout: [B, C, HW] with strides [stride_b, stride_c, stride_hw]
    base = pid_b * stride_b
    # ptrs[c, hw] = base + c * stride_c + hw * stride_hw
    ptrs = (base +
            c_offsets[:, None] * stride_c +
            hw_offsets[None, :] * stride_hw)  # [BLOCK_C, BLOCK_HW]

    mask_2d = c_mask[:, None] & hw_mask[None, :]
    x = tl.load(X_ptr + ptrs, mask=mask_2d, other=0.0).to(tl.float32)

    # Compute sum of squares along C dimension
    sum_sq = tl.sum(x * x, axis=0)  # [BLOCK_HW]

    # RMS inverse
    rms_inv = tl.rsqrt(sum_sq / C + eps)  # [BLOCK_HW]

    # Normalize
    y = x * rms_inv[None, :]  # [BLOCK_C, BLOCK_HW]

    # Store
    out_ptrs = (base +
                c_offsets[:, None] * stride_c +
                hw_offsets[None, :] * stride_hw)
    tl.store(Y_ptr + out_ptrs, y, mask=mask_2d)


@triton.jit
def rmsnorm_kernel_fp16(
    X_ptr, Y_ptr,
    B, C, HW,
    eps,
    stride_b, stride_c, stride_hw,
    BLOCK_C: tl.constexpr,
    BLOCK_HW: tl.constexpr,
):
    pid_hw = tl.program_id(0)
    pid_b  = tl.program_id(1)

    hw_start = pid_hw * BLOCK_HW
    hw_offsets = hw_start + tl.arange(0, BLOCK_HW)
    hw_mask = hw_offsets < HW

    c_offsets = tl.arange(0, BLOCK_C)
    c_mask = c_offsets < C

    base = pid_b * stride_b
    ptrs = (base +
            c_offsets[:, None] * stride_c +
            hw_offsets[None, :] * stride_hw)
    mask_2d = c_mask[:, None] & hw_mask[None, :]

    x = tl.load(X_ptr + ptrs, mask=mask_2d, other=0.0).to(tl.float32)
    sum_sq = tl.sum(x * x, axis=0)
    rms_inv = tl.rsqrt(sum_sq / C + eps)
    y = x * rms_inv[None, :]

    tl.store(Y_ptr + out_ptrs, y.to(tl.float16), mask=mask_2d)


def rmsnorm_triton(x: torch.Tensor, num_features: int, eps: float) -> torch.Tensor:
    output = torch.empty_like(x)

    B = x.shape[0]
    C = x.shape[1]
    HW = x.numel() // (B * C)

    # Strides (in elements)
    stride_b  = x.stride(0)
    stride_c  = x.stride(1)
    # For contiguous [B, C, H, W], stride_hw = 1
    # We flatten spatial dims
    stride_hw = 1  # x is contiguous in HW

    # Choose BLOCK_C to be next power of 2 >= C
    BLOCK_C = triton.next_power_of_2(C)
    # Choose BLOCK_HW based on BLOCK_C to target ~1024 elements per program
    if BLOCK_C <= 64:
        BLOCK_HW = 16
    elif BLOCK_C <= 128:
        BLOCK_HW = 8
    elif BLOCK_C <= 256:
        BLOCK_HW = 4
    else:
        BLOCK_HW = 2

    grid = (triton.cdiv(HW, BLOCK_HW), B)

    rmsnorm_kernel[grid](
        x, output,
        B, C, HW,
        eps,
        stride_b, stride_c, stride_hw,
        BLOCK_C=BLOCK_C,
        BLOCK_HW=BLOCK_HW,
    )

    return output


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return rmsnorm_triton(x, self.num_features, self.eps)
