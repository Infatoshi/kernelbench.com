cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <ATen/ATen.h>

// Block sizes
#define BLOCK_X 16
#define BLOCK_Y 16

// Fused convolution kernel - direct convolution with shared memory
__global__ void conv2d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int in_h, int in_w, int kernel_size, int out_h, int out_w
) {
    // Each thread computes one output element
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int oc = blockIdx.z % out_channels;
    int b = blockIdx.z / out_channels;
    
    if (ow >= out_w || oh >= out_h) return;
    
    float sum = bias ? bias[oc] : 0.0f;
    
    // Direct convolution
    for (int ic = 0; ic < in_channels; ic++) {
        for (int kh = 0; kh < kernel_size; kh++) {
            for (int kw = 0; kw < kernel_size; kw++) {
                int ih = oh + kh;
                int iw = ow + kw;
                float in_val = input[b * in_channels * in_h * in_w + ic * in_h * in_w + ih * in_w + iw];
                float w_val = weight[oc * in_channels * kernel_size * kernel_size + ic * kernel_size * kernel_size + kh * kernel_size + kw];
                sum += in_val * w_val;
            }
        }
    }
    
    output[b * out_channels * out_h * out_w + oc * out_h * out_w + oh * out_w + ow] = sum;
}

// Instance Norm + Divide fused kernel
// Computes mean and variance per channel per batch, then normalizes and divides
__global__ void instancenorm_divide_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size, int channels, int height, int width,
    float divide_by, float eps
) {
    // One block per (batch, channel)
    int b = blockIdx.x / channels;
    int c = blockIdx.x % channels;
    
    int spatial_size = height * width;
    int offset = b * channels * spatial_size + c * spatial_size;
    
    // Use shared memory for reduction
    __shared__ float s_sum[256];
    __shared__ float s_sum_sq[256];
    
    int tid = threadIdx.x;
    int num_threads = blockDim.x;
    
    // First pass: compute mean
    float sum = 0.0f;
    for (int i = tid; i < spatial_size; i += num_threads) {
        float val = input[offset + i];
        sum += val;
    }
    s_sum[tid] = sum;
    __syncthreads();
    
    // Reduce
    for (int s = num_threads / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_sum[tid] += s_sum[tid + s];
        }
        __syncthreads();
    }
    
    float mean = s_sum[0] / spatial_size;
    
    // Second pass: compute variance
    float sum_sq = 0.0f;
    for (int i = tid; i < spatial_size; i += num_threads) {
        float val = input[offset + i];
        float diff = val - mean;
        sum_sq += diff * diff;
    }
    s_sum_sq[tid] = sum_sq;
    __syncthreads();
    
    // Reduce
    for (int s = num_threads / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_sum_sq[tid] += s_sum_sq[tid + s];
        }
        __syncthreads();
    }
    
    // InstanceNorm uses unbiased variance (N-1 denominator)
    float var = s_sum_sq[0] / (spatial_size - 1);
    float inv_std = rsqrtf(var + eps);
    float scale = inv_std / divide_by;
    
    // Third pass: normalize and divide
    for (int i = tid; i < spatial_size; i += num_threads) {
        float val = input[offset + i];
        output[offset + i] = (val - mean) * scale;
    }
}

// Fused Conv + InstanceNorm + Divide kernel
// Processes one (batch, out_channel) at a time
__global__ void fused_conv_instancenorm_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int in_h, int in_w, int kernel_size, int out_h, int out_w,
    float divide_by, float eps
) {
    // One block per (batch, out_channel)
    int b = blockIdx.x / out_channels;
    int oc = blockIdx.x % out_channels;
    
    int spatial_size = out_h * out_w;
    
    extern __shared__ float shared_mem[];
    float* s_output = shared_mem;
    float* s_sum = shared_mem + spatial_size;
    float* s_sum_sq = s_sum + 256;
    
    int tid = threadIdx.x;
    int num_threads = blockDim.x;
    
    // Step 1: Compute convolution for this (batch, out_channel)
    for (int idx = tid; idx < spatial_size; idx += num_threads) {
        int oh = idx / out_w;
        int ow = idx % out_w;
        
        float sum = bias ? bias[oc] : 0.0f;
        
        for (int ic = 0; ic < in_channels; ic++) {
            for (int kh = 0; kh < kernel_size; kh++) {
                for (int kw = 0; kw < kernel_size; kw++) {
                    int ih = oh + kh;
                    int iw = ow + kw;
                    float in_val = input[b * in_channels * in_h * in_w + ic * in_h * in_w + ih * in_w + iw];
                    float w_val = weight[oc * in_channels * kernel_size * kernel_size + ic * kernel_size * kernel_size + kh * kernel_size + kw];
                    sum += in_val * w_val;
                }
            }
        }
        
        s_output[idx] = sum;
    }
    __syncthreads();
    
    // Step 2: Compute mean
    float sum = 0.0f;
    for (int i = tid; i < spatial_size; i += num_threads) {
        sum += s_output[i];
    }
    s_sum[tid] = sum;
    __syncthreads();
    
    for (int s = num_threads / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_sum[tid] += s_sum[tid + s];
        }
        __syncthreads();
    }
    
    float mean = s_sum[0] / spatial_size;
    
    // Step 3: Compute variance
    float sum_sq = 0.0f;
    for (int i = tid; i < spatial_size; i += num_threads) {
        float diff = s_output[i] - mean;
        sum_sq += diff * diff;
    }
    s_sum_sq[tid] = sum_sq;
    __syncthreads();
    
    for (int s = num_threads / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_sum_sq[tid] += s_sum_sq[tid + s];
        }
        __syncthreads();
    }
    
    // InstanceNorm uses unbiased variance (N-1 denominator)
    float var = s_sum_sq[0] / (spatial_size - 1);
    float inv_std = rsqrtf(var + eps);
    float scale = inv_std / divide_by;
    
    // Step 4: Normalize, divide, and write output
    for (int idx = tid; idx < spatial_size; idx += num_threads) {
        float val = s_output[idx];
        int oh = idx / out_w;
        int ow = idx % out_w;
        output[b * out_channels * spatial_size + oc * spatial_size + oh * out_w + ow] = (val - mean) * scale;
    }
}

    for (int ic = 0; ic < in_channels; ic++) {

                float in_val = static_cast<float>(input[b * in_channels * in_h * in_w + ic * in_h * in_w + ih * in_w + iw]);
                float w_val = static_cast<float>(weight[oc * in_channels * kernel_size * kernel_size + ic * kernel_size * kernel_size + kh * kernel_size + kw]);
                sum += in_val * w_val;
            }
        }
    }
    
    output[b * out_channels * out_h * out_w + oc * out_h * out_w + oh * out_w + ow] = c10::Half(sum);

// Main entry point
torch::Tensor fused_conv_instancenorm_divide(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double divide_by,
    int64_t out_channels,
    int64_t kernel_size
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_h = in_h - kernel_size + 1;
    int out_w = in_w - kernel_size + 1;
    float eps = 1e-5f;
    
    auto output = torch::empty({batch_size, out_channels, out_h, out_w}, input.options());
    
    bool is_half = input.scalar_type() == torch::kFloat16;
    
    int num_blocks = batch_size * out_channels;
    int threads = 256;
    int spatial_size = out_h * out_w;
    int shared_mem_size = (spatial_size + 256 + 256) * sizeof(float);
    
    if (is_half) {
        // Half precision path - use fused kernel
        fused_conv_instancenorm_half_kernel<<<num_blocks, threads, shared_mem_size>>>(
            input.data_ptr<c10::Half>(),
            weight.data_ptr<c10::Half>(),
            bias.defined() && bias.numel() > 0 ? bias.data_ptr<c10::Half>() : nullptr,
            output.data_ptr<c10::Half>(),
            batch_size, in_channels, out_channels,
            in_h, in_w, kernel_size, out_h, out_w,
            static_cast<float>(divide_by), eps
        );
    } else {
        // Float precision path - use fused kernel
        fused_conv_instancenorm_kernel<<<num_blocks, threads, shared_mem_size>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.defined() && bias.numel() > 0 ? bias.data_ptr<float>() : nullptr,
            output.data_ptr<float>(),
            batch_size, in_channels, out_channels,
            in_h, in_w, kernel_size, out_h, out_w,
            static_cast<float>(divide_by), eps
        );
    }
    
    return output;
}

OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

class Model(nn.Module):
    """
    Optimized model that performs fused Conv2d + InstanceNorm + Divide.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.divide_by = divide_by
        
        # Initialize weights and bias like Conv2d
        self.weight = nn.Parameter(torch.empty(out_channels, in_channels, kernel_size, kernel_size))
        self.bias = nn.Parameter(torch.empty(out_channels))
        
        # Initialize weights using kaiming uniform
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        if self.bias is not None:
            fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
            bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
            nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_conv_instancenorm_divide(
            x, 
            self.weight, 
            self.bias,
            self.divide_by,
            self.out_channels,
            self.kernel_size
        )
