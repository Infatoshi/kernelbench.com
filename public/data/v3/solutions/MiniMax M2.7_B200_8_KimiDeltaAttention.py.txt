import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Kimi Delta Attention (KDA): Linear Attention with Channel-wise Gating
# Optimized CUDA implementation
OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ['fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Kernel for Kimi Delta Attention - chunk-wise processing
// Algorithm:
//   For each position t: S_t = diag(a_t) @ S_{t-1} + beta_t * (k_t @ v_t^T - S_{t-1} @ k_t @ k_t^T)
//   Output: o_t = S_t @ q_t
//
// For efficiency, we use the approximation:
//   S_t = a_t * S_{t-1} + beta_t * k_t @ v_t^T

template <typename T>
__global__ void kimi_delta_attn_kernel(
    const T* __restrict__ q,     // (B, H, T, D_qk)
    const T* __restrict__ k,     // (B, H, T, D_qk)
    const T* __restrict__ v,     // (B, H, T, D_v)
    const T* __restrict__ a,     // (B, H, T, D_v) - channel-wise gates
    const T* __restrict__ beta,  // (B, H, T)
    const float scale,
    T* __restrict__ output,       // (B, H, T, D_v)
    int B, int H, int T_len, int D_qk, int D_v,
    int chunk_size) {
    
    int bid = blockIdx.z;  // batch * heads + chunk
    int b = bid / H;
    int h = bid % H;
    
    int chunk_id = blockIdx.y;
    int tid = threadIdx.x;
    
    int chunk_start = chunk_id * chunk_size;
    int chunk_end = min(chunk_start + chunk_size, T_len);
    int chunk_len = chunk_end - chunk_start;
    
    if (b >= B || h >= H) return;
    
    // Strides
    int stride_bt = H * T_len * D_qk;
    int stride_ht = T_len * D_qk;
    int stride_t = D_qk;
    
    int stride_bt_v = H * T_len * D_v;
    int stride_ht_v = T_len * D_v;
    int stride_t_v = D_v;
    
    // Base pointers for this batch and head
    const T* q_ptr = q + b * stride_bt + h * stride_ht;
    const T* k_ptr = k + b * stride_bt + h * stride_ht;
    const T* v_ptr = v + b * stride_bt_v + h * stride_ht_v;
    const T* a_ptr = a + b * stride_bt_v + h * stride_ht_v;
    const T* beta_ptr = beta + b * H * T_len + h * T_len;
    
    T* o_ptr = output + b * stride_bt_v + h * stride_ht_v;
    
    // Dynamic shared memory for state
    extern __shared__ char smem[];
    T* state = (T*)smem;  // (D_qk * D_v)
    
    // Initialize state to zero
    for (int i = tid; i < D_qk * D_v; i += blockDim.x) {
        state[i] = (T)0.0f;
    }
    __syncthreads();
    
    // Process each element in chunk
    for (int t = chunk_start; t < chunk_end; t++) {
        // Load q_t
        T q_local[128];
        #pragma unroll
        for (int i = 0; i < 128; i++) {
            q_local[i] = (i < D_qk) ? q_ptr[t * stride_t + i] : (T)0.0f;
        }
        
        // Compute o = S @ q * scale using thread-local accumulation
        T o_local[128] = {(T)0.0f};
        
        if (tid < D_v) {
            #pragma unroll
            for (int d_q = 0; d_q < 128; d_q++) {
                if (d_q < D_qk) {
                    T q_val = q_local[d_q];
                    T s_val = state[d_q * D_v + tid];
                    o_local[tid] += s_val * q_val;
                }
            }
            o_local[tid] *= (T)scale;
        }
        
        // Store output
        if (tid < D_v) {
            o_ptr[t * stride_t_v + tid] = o_local[tid];
        }
        __syncthreads();
        
        // Update state for next iteration
        // S_new = a * S + beta * k @ v^T
        if (tid < D_qk * D_v) {
            int d_q = tid / D_v;
            int d_v = tid % D_v;
            
            // Load k_dq and v_dv, a_dv, beta
            T k_val = k_ptr[t * stride_t + d_q];
            T v_val = v_ptr[t * stride_t_v + d_v];
            T a_val = a_ptr[t * stride_t_v + d_v];
            T beta_val = beta_ptr[t];
            
            T old_s = state[tid];
            T new_s = a_val * old_s + beta_val * k_val * v_val;
            state[tid] = new_s;
        }
        __syncthreads();
    }
}

// Wrapper for half precision
void kimi_delta_attn_cuda(
    torch::Tensor q,     // (B, H, T, D_qk)
    torch::Tensor k,     // (B, H, T, D_qk)
    torch::Tensor v,     // (B, H, T, D_v)
    torch::Tensor a,     // (B, H, T, D_v)
    torch::Tensor beta,  // (B, H, T)
    float scale,
    torch::Tensor output) {
    
    int B = q.size(0);
    int H = q.size(1);
    int T_len = q.size(2);
    int D_qk = q.size(3);
    int D_v = v.size(3);
    
    int chunk_size = 32;
    int num_chunks = (T_len + chunk_size - 1) / chunk_size;
    int num_blocks = B * H * num_chunks;
    
    int threads = 128;
    int smem_size = D_qk * D_v * sizeof(half);
    
    kimi_delta_attn_kernel<half><<<num_blocks, threads, smem_size>>>(
        reinterpret_cast<const half*>(q.data_ptr()),
        reinterpret_cast<const half*>(k.data_ptr()),
        reinterpret_cast<const half*>(v.data_ptr()),
        reinterpret_cast<const half*>(a.data_ptr()),
        reinterpret_cast<const half*>(beta.data_ptr()),
        scale,
        reinterpret_cast<half*>(output.data_ptr()),
        B, H, T_len, D_qk, D_v,
        chunk_size
    );
}

    torch::Tensor k,     // (B, H, T, D_qk)
    torch::Tensor v,     // (B, H, T, D_v)
    torch::Tensor a,     // (B, H, T, D_v)
    torch::Tensor beta,  // (B, H, T)
    float scale,
    torch::Tensor output) {
    
    int B = q.size(0);
    int H = q.size(1);
    int T_len = q.size(2);
    int D_qk = q.size(3);
    int D_v = v.size(3);
    
    int chunk_size = 32;
    int num_chunks = (T_len + chunk_size - 1) / chunk_size;
    int num_blocks = B * H * num_chunks;
    
    int threads = 128;
    int smem_size = D_qk * D_v * sizeof(float);
    
    kimi_delta_attn_kernel<float><<<num_blocks, threads, smem_size>>>(

        k.data_ptr<float>(),
        v.data_ptr<float>(),
        a.data_ptr<float>(),
        beta.data_ptr<float>(),
        scale,
        output.data_ptr<float>(),
        B, H, T_len, D_qk, D_v,
        chunk_size
    );
}
'''

cpp_source = r'''
void kimi_delta_attn_cuda(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v,
    torch::Tensor a,
    torch::Tensor beta,
    float scale,
    torch::Tensor output);

void kimi_delta_attn_cuda_float(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v,
    torch::Tensor a,
    torch::Tensor beta,
    float scale,
    torch::Tensor output);

torch::Tensor kimi_delta_attn(torch::Tensor q, torch::Tensor k, torch::Tensor v, 
                               torch::Tensor a, torch::Tensor beta, float scale) {
    auto output = torch::empty_like(v);
    if (q.dtype == torch::kFloat32) {
        kimi_delta_attn_cuda_float(q, k, v, a, beta, scale, output);
    } else {
        kimi_delta_attn_cuda(q, k, v, a, beta, scale, output);
    }
    return output;
}
'''

ext = load_inline(
    name='kimi_delta_attn_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['kimi_delta_attn'],
    verbose=False,
)


def kimi_delta_attention(
    q: torch.Tensor,      # (batch, heads, seq, d_qk)
    k: torch.Tensor,      # (batch, heads, seq, d_qk)
    v: torch.Tensor,      # (batch, heads, seq, d_v)
    a: torch.Tensor,      # (batch, heads, seq, d_v) - channel-wise gates
    beta: torch.Tensor,   # (batch, heads, seq) - update gate
    scale: float,
) -> torch.Tensor:
    """
    Kimi delta attention using custom CUDA kernel.
    """
    return ext.kimi_delta_attn(q, k, v, a, beta, scale)


class Model(nn.Module):
    """
    Kimi Delta Attention with channel-wise gating.
    Optimized version using fused CUDA kernel.
    """

    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_dplr: bool = False,
        dplr_rank: int = 4,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        # Channel-wise gating: d_v gates per head
        self.a_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=True)
        self.b_proj = nn.Linear(hidden_size, num_heads, bias=True)

        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)

        if use_short_conv:
            self.q_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.k_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.v_conv = nn.Conv1d(
                num_heads * head_dim_v, num_heads * head_dim_v,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_v,
                padding=conv_kernel_size - 1
            )

        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape

        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)

        if self.use_short_conv:
            q = self.q_conv(q.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            k = self.k_conv(k.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            v = self.v_conv(v.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            q = F.silu(q)
            k = F.silu(k)
            v = F.silu(v)

        # Reshape to (B, H, T, D)
        q = q.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        k = k.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        v = v.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        # Channel-wise gates (B, H, T, d_v)
        a = torch.sigmoid(self.a_proj(x))
        a = a.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        beta = torch.sigmoid(self.b_proj(x)).transpose(1, 2)  # (B, H, T)

        # Chunk-wise parallel attention (CUDA)
        o = kimi_delta_attention(q, k, v, a, beta, scale=self.scale)

        o = o.transpose(1, 2)  # (B, T, H, d_v)
        o = self.o_norm(o)

        g = torch.sigmoid(self.g_proj(x))
        g = g.view(batch_size, seq_len, self.num_heads, self.head_dim_v)
        o = o * g

        o = o.reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
        o = self.o_proj(o)

        return o


batch_size = 4
seq_len = 2048
hidden_size = 2048
num_heads = 16
head_dim_qk = 128
head_dim_v = 128


def get_inputs():
    return [torch.randn(batch_size, seq_len, hidden_size)]


def get_init_inputs():
    return [hidden_size, num_heads, head_dim_qk, head_dim_v]
