import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fast GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
__device__ __forceinline__ float gelu_fast(float x) {
    const float sqrt_2_over_pi = 0.7978845608028654f;
    const float coeff = 0.044715f;
    float x3 = x * x * x;
    return 0.5f * x * (1.0f + tanhf(sqrt_2_over_pi * (x + coeff * x3)));
}

__global__ void gelu_kernel_float(const float* input, float* output, int64_t n) {
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = blockDim.x * gridDim.x;
    for (int64_t i = idx; i < n; i += stride) {
        output[i] = gelu_fast(input[i]);
    }
}

__global__ void gelu_kernel_half(const c10::Half* input, c10::Half* output, int64_t n) {
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = blockDim.x * gridDim.x;
    for (int64_t i = idx; i < n; i += stride) {
        float x = __half2float(*reinterpret_cast<const __half*>(&input[i]));
        float y = gelu_fast(x);
        output[i] = *reinterpret_cast<c10::Half*>(&y);
    }
}

__global__ void gelu_kernel_bfloat16(const c10::BFloat16* input, c10::BFloat16* output, int64_t n) {
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t stride = blockDim.x * gridDim.x;
    for (int64_t i = idx; i < n; i += stride) {
        float x = static_cast<float>(input[i]);
        float y = gelu_fast(x);
        output[i] = static_cast<c10::BFloat16>(y);
    }
}

torch::Tensor gelu_cuda_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int64_t n = input.numel();
    if (n == 0) return output;
    
    const int threads = 256;
    const int blocks = 128;
    
    auto input_fp = input.data_ptr();
    auto output_fp = output.data_ptr();
    
    if (input.scalar_type() == at::kFloat) {
        gelu_kernel_float<<<blocks, threads>>>(
            static_cast<const float*>(input_fp), 
            static_cast<float*>(output_fp), n);
    } else if (input.scalar_type() == at::kHalf) {
        gelu_kernel_half<<<blocks, threads>>>(
            static_cast<const c10::Half*>(input_fp), 
            static_cast<c10::Half*>(output_fp), n);
    } else if (input.scalar_type() == at::kBFloat16) {
        gelu_kernel_bfloat16<<<blocks, threads>>>(
            static_cast<const c10::BFloat16*>(input_fp), 
            static_cast<c10::BFloat16*>(output_fp), n);
    } else {
        AT_ERROR("gelu_cuda_forward: unsupported dtype");
    }
    
    return output;
}
'''

cpp_source = "torch::Tensor gelu_cuda_forward(torch::Tensor input);"

gelu_cuda = load_inline(
    name="gelu_opt_v3",
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=["gelu_cuda_forward"],
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_contiguous():
            x = x.contiguous()
        return gelu_cuda.gelu_cuda_forward(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim)
    return [x]
