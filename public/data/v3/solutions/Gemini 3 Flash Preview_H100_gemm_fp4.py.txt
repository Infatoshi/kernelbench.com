
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Tiled General Matrix Multiply (GEMM) for INT8, with scaling.
// This version is optimized for H100 by using larger tiles and unrolling.

__global__ void __launch_bounds__(256) optimized_scaled_gemm_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    at::Half* __restrict__ C,
    int M, int N, int K,
    float scale) {

    const int BM = 128;
    const int BN = 128;
    const int BK = 32;

    __shared__ int8_t sA[BM][BK];
    __shared__ int8_t sB[BK][BN];

    int bx = blockIdx.x;
    int by = blockIdx.y;
    int tx = threadIdx.x;

    int tidx = tx % 16;
    int tidy = tx / 16;

    int32_t accum[8][8];
    #pragma unroll
    for (int i = 0; i < 8; i++) {
        #pragma unroll
        for (int j = 0; j < 8; j++) {
            accum[i][j] = 0;
        }
    }

    for (int k_offset = 0; k_offset < K; k_offset += BK) {
        // Shared memory loading
        // Each thread loads BM*BK / 256 = 16 elements.
        // We Use uint4 for vectorized 128-bit loads (16 int8 elements).
        {
            int load_idx = tx; // 0 to 255
            // Load sA: 128x32 total elements. Load 16 elements per thread.
            // 256 threads * 16 = 4096. Correct.
            // Each thread loads two 8-element segments? Or let's simplify.
            int load_row_A = (tx * 16) / BK;
            int load_col_A = (tx * 16) % BK;
            if (by * BM + load_row_A < M) {
                // Use vectorized loads if possible, but alignment is tricky. 
                // Let's stick to a robust way first.
                for(int i = 0; i < 16; i++) {
                    int r = load_row_A + (load_col_A + i) / BK;
                    int c = (load_col_A + i) % BK;
                    if (by * BM + r < M && k_offset + c < K)
                        sA[r][c] = A[(by * BM + r) * K + k_offset + c];
                    else
                        sA[r][c] = 0;
                }
            } else {
                for(int i = 0; i < 16; i++) {
                    int r = load_row_A + (load_col_A + i) / BK;
                    int c = (load_col_A + i) % BK;
                    sA[r][c] = 0;
                }
            }

            int load_row_B = (tx * 16) / BN;
            int load_col_B = (tx * 16) % BN;
            // Load sB: 32x128 total elements. Load 16 chars per thread.
            for(int i = 0; i < 16; i++) {
                int r = load_row_B + (load_col_B + i) / BN;
                int c = (load_col_B + i) % BN;
                if (k_offset + r < K && bx * BN + c < N)
                    sB[r][c] = B[(k_offset + r) * N + bx * BN + c];
                else
                    sB[r][c] = 0;
            }
        }
        __syncthreads();

        #pragma unroll
        for (int k = 0; k < BK; k++) {
            int8_t a_vals[8];
            int8_t b_vals[8];
            #pragma unroll
            for (int i = 0; i < 8; i++) a_vals[i] = sA[tidy * 8 + i][k];
            #pragma unroll
            for (int j = 0; j < 8; j++) b_vals[j] = sB[k][tidx * 8 + j];

            #pragma unroll
            for (int i = 0; i < 8; i++) {
                #pragma unroll
                for (int j = 0; j < 8; j++) {
                    accum[i][j] += (int32_t)a_vals[i] * (int32_t)b_vals[j];
                }
            }
        }
        __syncthreads();
    }

    int row_start = by * BM + tidy * 8;
    int col_start = bx * BN + tidx * 8;
    #pragma unroll
    for (int i = 0; i < 8; i++) {
        #pragma unroll
        for (int j = 0; j < 8; j++) {
            if (row_start + i < M && col_start + j < N) {
                C[(row_start + i) * N + (col_start + j)] = __float2half((float)accum[i][j] * scale);
            }
        }
    }
}

torch::Tensor scaled_gemm(torch::Tensor a_q, torch::Tensor b_q, torch::Tensor scale_a, torch::Tensor scale_b) {
    int M = a_q.size(0);
    int K = a_q.size(1);
    int N = b_q.size(1);
    
    auto C = torch::empty({M, N}, torch::dtype(torch::kHalf).device(a_q.device()));
    float total_scale = scale_a.item<float>() * scale_b.item<float>();
    
    dim3 block(256);
    dim3 grid((N + 127) / 128, (M + 127) / 128);
    
    optimized_scaled_gemm_kernel<<<grid, block>>>(
        (const int8_t*)a_q.data_ptr<int8_t>(),
        (const int8_t*)b_q.data_ptr<int8_t>(),
        (at::Half*)C.data_ptr<at::Half>(),
        M, N, K,
        total_scale
    );
    
    return C;
}
'''

cpp_source = r'''
torch::Tensor scaled_gemm(torch::Tensor a_q, torch::Tensor b_q, torch::Tensor scale_a, torch::Tensor scale_b);
'''

ext = load_inline(
    name='scaled_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['scaled_gemm'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(self, a_q, b_q, scale_a, scale_b):
        return ext.scaled_gemm(a_q, b_q, scale_a, scale_b)

def get_inputs():
    m, n, k = 2048, 2048, 2048
    a_q = torch.randint(-8, 8, (m, k), dtype=torch.int8, device='cuda')
    b_q = torch.randint(-8, 8, (k, n), dtype=torch.int8, device='cuda')
    scale_a = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    scale_b = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    return [a_q, b_q, scale_a, scale_b]

def get_init_inputs():
    return []
