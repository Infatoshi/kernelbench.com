import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Model(nn.Module):
    """
    Optimized version of causal multi-head self-attention.
    Uses PyTorch's scaled_dot_product_attention which dispatches to FlashAttention-2 
    or Memory Efficient Attention on RTX 3090, offering significant speedups 
    over manual attention implementations.
    """

    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        # Multi-head attention projections combined into a single linear layer
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        # Final output projection
        self.c_proj = nn.Linear(n_embd, n_embd)
        # Dropout probabilities
        self.attn_pdrop = attn_pdrop
        self.resid_pdrop = resid_pdrop
        self.n_head = n_head
        self.n_embd = n_embd

    def forward(self, x):
        B, T, C = x.size() # Batch size, sequence length, embedding dim

        # Combined projection: q, k, v for all heads in one shot
        qkv = self.c_attn(x)
        q, k, v = qkv.split(self.n_embd, dim=2)

        # Reshape to (B, n_head, T, head_size) for optimized attention kernels
        # Using .view followed by .transpose for proper layout
        head_size = C // self.n_head
        q = q.view(B, T, self.n_head, head_size).transpose(1, 2)
        k = k.view(B, T, self.n_head, head_size).transpose(1, 2)
        v = v.view(B, T, self.n_head, head_size).transpose(1, 2)

        # PyTorch's F.scaled_dot_product_attention (SDPA) is the fastest way to perform 
        # causal self-attention on Ampere architecture (RTX 3090). It leverages 
        # FlashAttention/Cutlass kernels which are faster and more memory-efficient 
        # than explicit softmax and matmul operations.
        y = F.scaled_dot_product_attention(
            q, k, v, 
            attn_mask=None, 
            dropout_p=self.attn_pdrop if self.training else 0.0, 
            is_causal=True
        )

        # Re-assemble all head outputs side by side
        # .transpose followed by .contiguous and .view recreates (B, T, C)
        y = y.transpose(1, 2).contiguous().view(B, T, C)

        # Final projection and residue dropout
        y = self.c_proj(y)
        if self.resid_pdrop > 0.0:
            y = F.dropout(y, p=self.resid_pdrop, training=self.training)
            
        return y

def get_inputs():
    return [torch.randn(64, 512, 768, device='cuda')]

def get_init_inputs():
    return [768, 8, 0.0, 0.0, 1024]
