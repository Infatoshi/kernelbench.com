import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized GEMM kernel for H100 using shared memory tiling

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// FP32 GEMM kernel with shared memory tiling
__global__ void gemm_fp32_kernel(float* C, const float* A, const float* B, int N) {
    constexpr int BM = 128;
    constexpr int BN = 128;
    constexpr int BK = 32;
    
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    int tid = threadIdx.x;
    
    extern __shared__ float smem_f32[];
    float* smem_A = smem_f32;
    float* smem_B = smem_f32 + BM * BK;
    
    int tx = tid % 32;
    int ty = tid / 32;
    
    int row = block_m * BM + ty;
    int col = block_n * BN + tx;
    
    float sum = 0.0f;
    
    for (int k = 0; k < N; k += BK) {
        // Load A tile
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (row < N && k + i < N) smem_A[ty * BK + i] = A[row * N + (k + i)];
            if (row < N && k + i + 1 < N) smem_A[ty * BK + i + 1] = A[row * N + (k + i + 1)];
            if (row < N && k + i + 2 < N) smem_A[ty * BK + i + 2] = A[row * N + (k + i + 2)];
            if (row < N && k + i + 3 < N) smem_A[ty * BK + i + 3] = A[row * N + (k + i + 3)];
        }
        
        // Load B tile
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (k + i < N && col < N) smem_B[i * BN + tx] = B[(k + i) * N + col];
            if (k + i + 1 < N && col < N) smem_B[(i + 1) * BN + tx] = B[(k + i + 1) * N + col];
            if (k + i + 2 < N && col < N) smem_B[(i + 2) * BN + tx] = B[(k + i + 2) * N + col];
            if (k + i + 3 < N && col < N) smem_B[(i + 3) * BN + tx] = B[(k + i + 3) * N + col];
        }
        __syncthreads();
        
        // Compute dot product
        #pragma unroll
        for (int kk = 0; kk < BK; kk++) {
            sum += smem_A[ty * BK + kk] * smem_B[kk * BN + tx];
        }
        __syncthreads();
    }
    
    if (row < N && col < N) {
        C[row * N + col] = sum;
    }
}

// FP16 GEMM kernel
__global__ void gemm_fp16_kernel(half* C, const half* A, const half* B, int N) {
    constexpr int BM = 128;
    constexpr int BN = 128;
    constexpr int BK = 32;
    
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    int tid = threadIdx.x;
    
    extern __shared__ half smem_f16[];
    half* smem_A = smem_f16;
    half* smem_B = smem_f16 + BM * BK;
    
    int tx = tid % 32;
    int ty = tid / 32;
    
    int row = block_m * BM + ty;
    int col = block_n * BN + tx;
    
    float sum = 0.0f;
    
    for (int k = 0; k < N; k += BK) {
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (row < N && k + i < N) smem_A[ty * BK + i] = A[row * N + (k + i)];
            if (row < N && k + i + 1 < N) smem_A[ty * BK + i + 1] = A[row * N + (k + i + 1)];
            if (row < N && k + i + 2 < N) smem_A[ty * BK + i + 2] = A[row * N + (k + i + 2)];
            if (row < N && k + i + 3 < N) smem_A[ty * BK + i + 3] = A[row * N + (k + i + 3)];
        }
        
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (k + i < N && col < N) smem_B[i * BN + tx] = B[(k + i) * N + col];
            if (k + i + 1 < N && col < N) smem_B[(i + 1) * BN + tx] = B[(k + i + 1) * N + col];
            if (k + i + 2 < N && col < N) smem_B[(i + 2) * BN + tx] = B[(k + i + 2) * N + col];
            if (k + i + 3 < N && col < N) smem_B[(i + 3) * BN + tx] = B[(k + i + 3) * N + col];
        }
        __syncthreads();
        
        #pragma unroll
        for (int kk = 0; kk < BK; kk++) {
            sum += __half2float(smem_A[ty * BK + kk]) * __half2float(smem_B[kk * BN + tx]);
        }
        __syncthreads();
    }
    
    if (row < N && col < N) {
        C[row * N + col] = __float2half(sum);
    }
}

// BF16 GEMM kernel

    constexpr int BM = 128;
    constexpr int BN = 128;
    constexpr int BK = 32;
    
    int block_m = blockIdx.y;

    int ty = tid / 32;
    
    int row = block_m * BM + ty;
    int col = block_n * BN + tx;
    
    float sum = 0.0f;
    
    for (int k = 0; k < N; k += BK) {
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (row < N && k + i < N) smem_A[ty * BK + i] = A[row * N + (k + i)];
            if (row < N && k + i + 1 < N) smem_A[ty * BK + i + 1] = A[row * N + (k + i + 1)];
            if (row < N && k + i + 2 < N) smem_A[ty * BK + i + 2] = A[row * N + (k + i + 2)];
            if (row < N && k + i + 3 < N) smem_A[ty * BK + i + 3] = A[row * N + (k + i + 3)];
        }
        
        #pragma unroll
        for (int i = 0; i < BK; i += 4) {
            if (k + i < N && col < N) smem_B[i * BN + tx] = B[(k + i) * N + col];
            if (k + i + 1 < N && col < N) smem_B[(i + 1) * BN + tx] = B[(k + i + 1) * N + col];
            if (k + i + 2 < N && col < N) smem_B[(i + 2) * BN + tx] = B[(k + i + 2) * N + col];
            if (k + i + 3 < N && col < N) smem_B[(i + 3) * BN + tx] = B[(k + i + 3) * N + col];
        }
        __syncthreads();
        
        #pragma unroll
        for (int kk = 0; kk < BK; kk++) {
            sum += __bfloat162float(smem_A[ty * BK + kk]) * __bfloat162float(smem_B[kk * BN + tx]);
        }
        __syncthreads();
    }
    
    if (row < N && col < N) {
        C[row * N + col] = __float2bfloat16(sum);
    }
}

torch::Tensor gemm_fp32(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    auto C = torch::empty_like(A);
    
    dim3 block(128);
    dim3 grid((N + 127) / 128, (N + 127) / 128);
    size_t smem_size = 128 * 32 * sizeof(float) * 2;
    
    gemm_fp32_kernel<<<grid, block, smem_size>>>(
        C.data_ptr<float>(),
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        N
    );
    
    cudaDeviceSynchronize();
    return C;
}

torch::Tensor gemm_fp16(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    auto C = torch::empty_like(A);
    
    dim3 block(128);
    dim3 grid((N + 127) / 128, (N + 127) / 128);
    size_t smem_size = 128 * 32 * sizeof(half) * 2;
    
    gemm_fp16_kernel<<<grid, block, smem_size>>>(
        (half*)C.data_ptr(),
        (half*)A.data_ptr(),
        (half*)B.data_ptr(),
        N
    );
    
    cudaDeviceSynchronize();
    return C;
}

torch::Tensor gemm_bf16(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    auto C = torch::empty_like(A);
    
    dim3 block(128);
    dim3 grid((N + 127) / 128, (N + 127) / 128);
    size_t smem_size = 128 * 32 * sizeof(nv_bfloat16) * 2;
    
    gemm_bf16_kernel<<<grid, block, smem_size>>>(
        (nv_bfloat16*)C.data_ptr(),
        (nv_bfloat16*)A.data_ptr(),
        (nv_bfloat16*)B.data_ptr(),
        N
    );
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='gemm_ext',
    cpp_sources='''
torch::Tensor gemm_fp32(torch::Tensor A, torch::Tensor B);
torch::Tensor gemm_fp16(torch::Tensor A, torch::Tensor B);
torch::Tensor gemm_bf16(torch::Tensor A, torch::Tensor B);
''',
    cuda_sources=cuda_source,
    functions=['gemm_fp32', 'gemm_fp16', 'gemm_bf16'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        if A.dtype == torch.float32:
            return ext.gemm_fp32(A, B)
        elif A.dtype == torch.float16:
            return ext.gemm_fp16(A, B)
        elif A.dtype == torch.bfloat16:
            return ext.gemm_bf16(A, B)
        else:
            raise ValueError(f"Unsupported dtype: {A.dtype}")

def get_inputs():
    N = 2048
    A = torch.randn(N, N, device='cuda')
    B = torch.randn(N, N, device='cuda')
    return [A, B]

def get_init_inputs():
    return []
