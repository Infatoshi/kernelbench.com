import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def instance_norm_divide_kernel(
    X_ptr,        # input: (N, C, H, W) after conv
    Y_ptr,        # output: same shape
    N,            # batch size
    C,            # channels
    HW: tl.constexpr,  # H * W (spatial size)
    inv_divide: tl.constexpr,  # 1.0 / divide_by
    BLOCK_SIZE: tl.constexpr,  # block size for spatial reduction
):
    # Each program handles one (n, c) pair
    pid = tl.program_id(0)

    # Base offset for this (n, c) slice
    base = pid * HW

    # Two-pass: first compute mean, then variance + normalize
    # Pass 1: compute mean
    mean = tl.zeros([], dtype=tl.float32)
    for start in tl.static_range(0, HW, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < HW
        vals = tl.load(X_ptr + base + offs, mask=mask, other=0.0).to(tl.float32)
        mean += tl.sum(vals, axis=0)
    mean = mean / HW

    # Pass 2: compute variance
    var = tl.zeros([], dtype=tl.float32)
    for start in tl.static_range(0, HW, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < HW
        vals = tl.load(X_ptr + base + offs, mask=mask, other=0.0).to(tl.float32)
        diff = vals - mean
        var += tl.sum(diff * diff, axis=0)
    var = var / HW

    # Compute inv_std
    inv_std = 1.0 / tl.sqrt(var + 1e-5)
    scale = inv_std * inv_divide

    # Pass 3: normalize and divide
    for start in tl.static_range(0, HW, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < HW
        vals = tl.load(X_ptr + base + offs, mask=mask, other=0.0).to(tl.float32)
        out = (vals - mean) * scale
        tl.store(Y_ptr + base + offs, out.to(tl.float32), mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = divide_by

    def forward(self, x):
        # Run convolution
        x = self.conv(x)
        
        N, C, H, W = x.shape
        HW = H * W
        
        y = torch.empty_like(x)
        
        # Launch one program per (n, c) pair
        grid = (N * C,)
        
        instance_norm_divide_kernel[grid](
            x, y,
            N, C, HW,
            1.0 / self.divide_by,
            BLOCK_SIZE=8192,
        )
        
        return y


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
divide_by = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, divide_by]
