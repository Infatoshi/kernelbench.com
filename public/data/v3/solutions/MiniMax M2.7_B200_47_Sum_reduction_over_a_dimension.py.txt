import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Warp-level reduction using shuffle
template <typename T>
__device__ T warp_reduce(T val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_xor_sync(0xffffffff, val, offset);
    }
    return val;
}

// Kernel for sum reduction over dimension 1
// Input: [batch, dim1, dim2], reduce over dim1 (256 elements)
// Output: [batch, 1, dim2]
template <typename T>
__global__ void sum_reduce_dim1_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int dim1,
    int dim2
) {
    // Grid: (batch_size * dim2) blocks
    int batch_idx = blockIdx.x / dim2;
    int dim2_idx = blockIdx.x % dim2;
    
    if (batch_idx >= batch_size || dim2_idx >= dim2) return;
    
    // Only use first warp (32 threads) for actual computation
    T sum_val = 0.0f;
    if (threadIdx.x < 32) {
        // Sequential load and reduction within thread
        int base_offset = batch_idx * dim1 * dim2 + dim2_idx;
        
        #pragma unroll
        for (int i = 0; i < dim1; i++) {
            sum_val += input[base_offset + i * dim2];
        }
        
        // Warp-level reduction
        sum_val = warp_reduce(sum_val);
        
        // Write result (only lane 0)
        if (threadIdx.x == 0) {
            output[batch_idx * dim2 + dim2_idx] = sum_val;
        }
    }
}

torch::Tensor my_sum_op(torch::Tensor x, int reduce_dim) {
    TORCH_CHECK(x.is_cuda(), "Input must be a CUDA tensor");
    TORCH_CHECK(reduce_dim == 1, "Optimized for reduce_dim=1");
    TORCH_CHECK(x.dim() == 3, "Expected 3D tensor");
    
    int batch_size = x.size(0);
    int dim1 = x.size(1);
    int dim2 = x.size(2);
    
    // Output: [batch, 1, dim2]
    auto output = torch::empty({batch_size, 1, dim2}, x.options());
    
    int num_blocks = batch_size * dim2;
    int threads = 256;  // 8 warps, but only first warp does work
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "sum_reduce", ([&] {
        sum_reduce_dim1_kernel<scalar_t><<<num_blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, dim1, dim2
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources='torch::Tensor my_sum_op(torch::Tensor x, int reduce_dim);',
    cuda_sources=cuda_source,
    functions=['my_sum_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x):
        return ext.my_sum_op(x, self.dim)

def get_inputs():
    return [torch.randn(64, 256, 256, device='cuda')]

def get_init_inputs():
    return [1]
