import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

# Simple fused kernel for Swish + Bias + GroupNorm
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fused kernel: Swish + Bias + GroupNorm
// Processes one group per block
// x: (batch, out_features) - after matmul
// bias: (out_features,)
// output: (batch, out_features)
// num_groups = 32, out_features = 4096 -> 128 channels per group

template <typename T>
__global__ void fused_kernel_v3(
    const T* __restrict__ x,
    const T* __restrict__ bias,
    T* __restrict__ output,
    const int batch_size,
    const int out_features,
    const int num_groups
) {
    // Each block processes one group
    const int gid = blockIdx.x;
    
    const int channels_per_group = out_features / num_groups;
    const int group_start = gid * channels_per_group;
    
    extern __shared__ float stats[];
    float* mean = stats;
    float* inv_std = stats + 1;
    
    const int tid = threadIdx.x;
    const int nthreads = blockDim.x;
    
    // First pass: compute mean and variance across all batch items
    // Each thread processes multiple channels
    float sum = 0;
    float sum_sq = 0;
    
    const int channels_per_thread = (channels_per_group + nthreads - 1) / nthreads;
    
    for (int iter = 0; iter < 2; iter++) {
        int channel_start = tid * channels_per_thread + iter * nthreads * channels_per_thread;
        
        for (int c = 0; c < channels_per_thread; c++) {
            int channel = group_start + channel_start + c;
            if (channel < group_start + channels_per_group) {
                // Accumulate across batch
                for (int b = 0; b < batch_size; b++) {
                    float val = x[b * out_features + channel];
                    sum += val;
                    sum_sq += val * val;
                }
            }
        }
    }
    
    // Block-level reduction
    __shared__ float reduce_smem[32];
    int warpId = tid / 32;
    int laneId = tid % 32;
    
    // Warp reduction
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
        sum_sq += __shfl_down_sync(0xffffffff, sum_sq, offset);
    }
    
    if (laneId == 0) {
        reduce_smem[warpId] = sum;
    }
    __syncthreads();
    
    if (warpId == 0) {
        if (laneId < blockDim.x / 32) {
            sum = reduce_smem[laneId];
        } else {
            sum = 0;
        }
        for (int offset = 16; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (laneId == 0) {
            reduce_smem[0] = sum;
        }
    }
    __syncthreads();
    
    if (tid == 0) {
        float total_sum = reduce_smem[0];
        float m = total_sum / (batch_size * channels_per_group);
        mean[0] = m;
        
        // Now compute variance
        sum = 0;
        for (int b = 0; b < batch_size; b++) {
            for (int c = 0; c < channels_per_group; c++) {
                float val = x[b * out_features + group_start + c] - m;
                sum += val * val;
            }
        }
        float v = sum / (batch_size * channels_per_group);
        inv_std[0] = rsqrtf(v + 1e-5f);
    }
    __syncthreads();
    
    // Second pass: apply activation, bias, and normalization
    for (int iter = 0; iter < 2; iter++) {
        int channel_start = tid * channels_per_thread + iter * nthreads * channels_per_thread;
        
        for (int c = 0; c < channels_per_thread; c++) {
            int channel = group_start + channel_start + c;
            if (channel < group_start + channels_per_group) {
                // Process all batch items for this channel
                for (int b = 0; b < batch_size; b++) {
                    int idx = b * out_features + channel;
                    float val = x[idx];
                    
                    // Swish
                    float sigmoid_val = 1.0f / (1.0f + expf(-val));
                    val = val * sigmoid_val;
                    
                    // Bias
                    val = val + bias[channel];
                    
                    // GroupNorm
                    val = (val - mean[0]) * inv_std[0];
                    
                    output[idx] = val;
                }
            }
        }
    }
}

// Host wrapper
torch::Tensor fused_op(torch::Tensor x, torch::Tensor bias, int num_groups) {
    auto output = torch::empty_like(x);
    
    const int batch_size = x.size(0);
    const int out_features = x.size(1);
    
    const int blocks = num_groups;  // One block per group
    const int threads = 256;
    const int smem_size = 64;
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "fused_op", ([&] {
        fused_kernel_v3<scalar_t><<<blocks, threads, smem_size>>>(
            x.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            out_features,
            num_groups
        );
    }));
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='''
torch::Tensor fused_op(torch::Tensor x, torch::Tensor bias, int num_groups);
''',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        
    def forward(self, x):
        # Matmul using PyTorch's optimized linear
        x = self.matmul(x)
        # Use fused custom kernel for Swish + Bias + GroupNorm
        x = ext.fused_op(x, self.bias, self.group_norm.num_groups)
        return x

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 32, (4096,)]
