import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

__global__ void conv_fused_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_h, int in_w,
    int out_channels, int out_h, int out_w,
    int k_h, int k_w,
    float sub1, float sub2,
    int pool_h, int pool_w,
    int final_h, int final_w) 
{
    int pw = blockIdx.x * blockDim.x + threadIdx.x;
    int ph = blockIdx.y * blockDim.y + threadIdx.y;
    int oc = blockIdx.z % out_channels;
    int b = blockIdx.z / out_channels;

    extern __shared__ float s_bias[];
    if (threadIdx.x == 0 && threadIdx.y == 0) {
        s_bias[0] = bias ? bias[oc] : 0.0f;
    }
    __syncthreads();

    if (pw < final_w && ph < final_h && b < batch_size) {
        float b_val = s_bias[0];
        float pool_sum = 0.0f;

        for (int i = 0; i < pool_h; ++i) {
            for (int j = 0; j < pool_w; ++j) {
                int oh = ph * pool_h + i;
                int ow = pw * pool_w + j;

                float conv_val = b_val;
                // Optimization: reorder loops or use more efficient memory access
                for (int ic = 0; ic < in_channels; ++ic) {
                    const float* input_ptr = &input[((b * in_channels + ic) * in_h + oh) * in_w + ow];
                    const float* weight_ptr = &weight[(oc * in_channels + ic) * k_h * k_w];
                    for (int kh = 0; kh < k_h; ++kh) {
                        for (int kw = 0; kw < k_w; ++kw) {
                            conv_val += input_ptr[kh * in_w + kw] * weight_ptr[kh * k_w + kw];
                        }
                    }
                }
                
                float val = tanhf(conv_val - sub1) - sub2;
                pool_sum += val;
            }
        }
        output[((b * out_channels + oc) * final_h + ph) * final_w + pw] = pool_sum / (float)(pool_h * pool_w);
    }
}

torch::Tensor forward(
    torch::Tensor input, 
    torch::Tensor weight, 
    at::optional<torch::Tensor> bias,
    float sub1, float sub2, int pool_size) 
{
    int b = input.size(0);
    int ic = input.size(1);
    int ih = input.size(2);
    int iw = input.size(3);
    int oc = weight.size(0);
    int kh = weight.size(2);
    int kw = weight.size(3);

    int oh = ih - kh + 1;
    int ow = iw - kw + 1;
    int fh = oh / pool_size;
    int fw = ow / pool_size;

    auto output = torch::empty({b, oc, fh, fw}, input.options());

    dim3 block(16, 16);
    dim3 grid((fw + block.x - 1) / block.x, (fh + block.y - 1) / block.y, b * oc);

    conv_fused_kernel<<<grid, block, sizeof(float)>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias.value().data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        b, ic, ih, iw, oc, oh, ow, kh, kw, sub1, sub2, pool_size, pool_size, fh, fw
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor forward(torch::Tensor input, torch::Tensor weight, at::optional<torch::Tensor> bias, float sub1, float sub2, int pool_size);
'''

fused_lib = load_inline(
    name='fused_conv',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        return fused_lib.forward(x.contiguous(), self.conv.weight.contiguous(), self.conv.bias, 
                                 float(self.subtract1_value), float(self.subtract2_value), int(self.kernel_size_pool))

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
