import torch
import torch.nn as nn
import triton
import triton.language as tl


# Optimized for the benchmark shape: x.shape = [64, 256, 256], reduce over dim=1, keepdim=True.
# We provide a fast Triton kernel for contiguous reduction over dim=1 and fall back to torch.sum
# only for unsupported layouts/shapes. The benchmark uses the optimized path.


@triton.jit
def _sum_dim1_kernel_fp32acc(
    x_ptr,
    y_ptr,
    B,
    M,
    N,
    stride_b,
    stride_m,
    stride_n,
    out_stride_b,
    out_stride_m,
    out_stride_n,
    BLOCK_N: tl.constexpr,
    BLOCK_M: tl.constexpr,
):
    pid0 = tl.program_id(0)  # tile over N
    pid1 = tl.program_id(1)  # batch index

    offs_n = pid0 * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = offs_n < N

    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)

    offs_m = tl.arange(0, BLOCK_M)
    for m0 in range(0, M, BLOCK_M):
        m = m0 + offs_m
        mask_m = m < M
        ptrs = x_ptr + pid1 * stride_b + m[:, None] * stride_m + offs_n[None, :] * stride_n
        vals = tl.load(ptrs, mask=mask_m[:, None] & mask_n[None, :], other=0.0)
        acc += tl.sum(vals, axis=0)

    out_ptrs = y_ptr + pid1 * out_stride_b + 0 * out_stride_m + offs_n * out_stride_n
    tl.store(out_ptrs, acc.to(tl.float32), mask=mask_n)


@triton.jit
def _sum_dim1_kernel_castback(
    x_ptr,
    y_ptr,
    B,
    M,
    N,
    stride_b,
    stride_m,
    stride_n,
    out_stride_b,
    out_stride_m,
    out_stride_n,
    BLOCK_N: tl.constexpr,
    BLOCK_M: tl.constexpr,
):
    pid0 = tl.program_id(0)
    pid1 = tl.program_id(1)

    offs_n = pid0 * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = offs_n < N

    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)

    offs_m = tl.arange(0, BLOCK_M)
    for m0 in range(0, M, BLOCK_M):
        m = m0 + offs_m
        mask_m = m < M
        ptrs = x_ptr + pid1 * stride_b + m[:, None] * stride_m + offs_n[None, :] * stride_n
        vals = tl.load(ptrs, mask=mask_m[:, None] & mask_n[None, :], other=0.0)
        acc += tl.sum(vals, axis=0)

    out_ptrs = y_ptr + pid1 * out_stride_b + 0 * out_stride_m + offs_n * out_stride_n
    out = acc.to(tl.float16) if y_ptr.dtype.element_ty == tl.float16 else acc.to(tl.bfloat16)
    tl.store(out_ptrs, out, mask=mask_n)


def _sum_dim1_fast(x: torch.Tensor) -> torch.Tensor:
    assert x.ndim == 3
    B, M, N = x.shape
    y = torch.empty((B, 1, N), device=x.device, dtype=x.dtype)
    grid = (triton.cdiv(N, 128), B)

    # Use fp32 accumulation for all supported dtypes.
    if x.dtype == torch.float32:
        _sum_dim1_kernel_fp32acc[grid](
            x, y,
            B, M, N,
            x.stride(0), x.stride(1), x.stride(2),
            y.stride(0), y.stride(1), y.stride(2),
            BLOCK_N=128,
            BLOCK_M=32,
            num_warps=4,
            num_stages=4,
        )
    else:
        # Store back in input dtype for exact API compatibility with torch.sum.
        _sum_dim1_kernel_castback[grid](
            x, y,
            B, M, N,
            x.stride(0), x.stride(1), x.stride(2),
            y.stride(0), y.stride(1), y.stride(2),
            BLOCK_N=128,
            BLOCK_M=32,
            num_warps=4,
            num_stages=4,
        )
    return y


class Model(nn.Module):
    """
    Simple model that performs sum reduction over a specified dimension.
    Optimized for dim=1 on 3D CUDA tensors used by the benchmark.
    """
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if (
            x.is_cuda
            and x.ndim == 3
            and self.dim == 1
            and x.is_contiguous()
            and x.dtype in (torch.float16, torch.bfloat16, torch.float32)
        ):
            return _sum_dim1_fast(x)
        return torch.sum(x, dim=self.dim, keepdim=True)


batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1


def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]


def get_init_inputs():
    return [reduce_dim]
