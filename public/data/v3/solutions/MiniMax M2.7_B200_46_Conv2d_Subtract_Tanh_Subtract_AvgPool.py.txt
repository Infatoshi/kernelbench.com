import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Kernel for fused element-wise operations: subtract, tanh, subtract
template <typename T>
__global__ void fused_elementwise_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int size,
    float subtract1,
    float subtract2
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < size) {
        float val = static_cast<float>(input[idx]);
        val = val - subtract1;
        val = tanhf(val);
        val = val - subtract2;
        output[idx] = static_cast<T>(val);
    }
}

// Optimized avgpool with thread coalescing
template <typename T>
__global__ void avgpool_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int channels,
    int pool_size
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = out_height * out_width * channels;
    
    if (idx < total) {
        int c = idx % channels;
        int ph = (idx / channels) % out_height;
        int pw = (idx / channels / out_height) % out_width;
        
        int h_start = ph * pool_size;
        int w_start = pw * pool_size;
        
        float sum = 0.0f;
        
        #pragma unroll 2
        for (int h = h_start; h < h_start + pool_size && h < in_height; h++) {
            #pragma unroll 2
            for (int w = w_start; w < w_start + pool_size && w < in_width; w++) {
                int in_idx = c * in_height * in_width + h * in_width + w;
                sum += static_cast<float>(input[in_idx]);
            }
        }
        
        int pool_elements = pool_size * pool_size;
        output[idx] = static_cast<T>(sum / static_cast<float>(pool_elements));
    }
}

template <typename T>
void fused_elementwise_cuda(const T* input, T* output, int size, float subtract1, float subtract2) {
    const int block = 256;
    const int grid = (size + block - 1) / block;
    fused_elementwise_kernel<T><<<grid, block>>>(input, output, size, subtract1, subtract2);
}

template <typename T>
void avgpool_cuda(const T* input, T* output, int in_height, int in_width, int out_height, int out_width, int channels, int pool_size) {
    int total = out_height * out_width * channels;
    const int block = 256;
    const int grid = (total + block - 1) / block;
    avgpool_kernel<T><<<grid, block>>>(input, output, in_height, in_width, out_height, out_width, channels, pool_size);
}

// Type dispatch
torch::Tensor fused_elementwise_forward(torch::Tensor input, float subtract1, float subtract2) {
    auto output = torch::empty_like(input);
    int size = input.numel();
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_elementwise", [&]() {
        fused_elementwise_cuda<scalar_t>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            size, subtract1, subtract2
        );
    });
    
    return output;
}

torch::Tensor avgpool_forward(torch::Tensor input, int pool_size) {
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    int out_height = in_height / pool_size;
    int out_width = in_width / pool_size;
    
    auto output = torch::empty({input.size(0), channels, out_height, out_width}, input.options());
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "avgpool", [&]() {
        avgpool_cuda<scalar_t>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            in_height, in_width, out_height, out_width, channels, pool_size
        );
    });
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_elementwise_forward(torch::Tensor input, float subtract1, float subtract2);
torch::Tensor avgpool_forward(torch::Tensor input, int pool_size);
'''

ext = load_inline(
    name='fused_ops_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_elementwise_forward', 'avgpool_forward'],
    extra_include_paths=['/usr/local/cuda/include'],
    extra_cuda_cflags=['-gencode=arch=compute_100,code=sm_100', '-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=kernel_size//2)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        self.conv.benchmark = True

    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_elementwise_forward(x, self.subtract1_value, self.subtract2_value)
        x = ext.avgpool_forward(x, self.kernel_size_pool)
        return x

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
