import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Two-pass LayerNorm: stats reduction + apply normalization
// Each instance has N=4194304 elements, batch=16
// Use cooperative block counting with atomics to fuse reduction

static constexpr int BLOCK_SIZE = 256;
static constexpr int VEC_PER_THREAD = 16;  // float4 per thread
// floats per block = 256 * 16 * 4 = 16384

__device__ __forceinline__ float warpReduceSum(float val) {
    val += __shfl_xor_sync(0xffffffff, val, 16);
    val += __shfl_xor_sync(0xffffffff, val, 8);
    val += __shfl_xor_sync(0xffffffff, val, 4);
    val += __shfl_xor_sync(0xffffffff, val, 2);
    val += __shfl_xor_sync(0xffffffff, val, 1);
    return val;
}

__device__ __forceinline__ float blockReduceSum(float val) {
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    val = warpReduceSum(val);
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    int nwarps = BLOCK_SIZE >> 5;  // 8
    val = (threadIdx.x < nwarps) ? shared[threadIdx.x] : 0.0f;
    if (wid == 0) val = warpReduceSum(val);
    return val;
}

// Fused stats kernel: each block computes partial sums, last block to finish does the final reduction
// Uses atomic counter per instance to determine the last block
__global__ void layernorm_fused_stats_kernel(
    const float4* __restrict__ input,
    float* __restrict__ partial_sum,
    float* __restrict__ partial_sum_sq,
    float* __restrict__ mean_out,
    float* __restrict__ invstd_out,
    int* __restrict__ block_counter,  // one per instance
    int N4,  // N/4
    int N,
    int num_blocks_per_instance,
    float eps
) {
    int instance_id = blockIdx.y;
    int block_id = blockIdx.x;
    
    const float4* inst_input = input + (size_t)instance_id * N4;
    
    int elems_per_block = BLOCK_SIZE * VEC_PER_THREAD;
    int base = block_id * elems_per_block + threadIdx.x;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    #pragma unroll
    for (int i = 0; i < VEC_PER_THREAD; i++) {
        int idx = base + i * BLOCK_SIZE;
        if (idx < N4) {
            float4 v = __ldg(&inst_input[idx]);
            local_sum += v.x + v.y + v.z + v.w;
            local_sum_sq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
        }
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    __shared__ bool is_last;
    
    if (threadIdx.x == 0) {
        int partial_idx = instance_id * num_blocks_per_instance + block_id;
        partial_sum[partial_idx] = local_sum;
        partial_sum_sq[partial_idx] = local_sum_sq;
        
        // Memory fence to ensure writes are visible
        __threadfence();
        
        // Increment counter
        int old = atomicAdd(&block_counter[instance_id], 1);
        is_last = (old == num_blocks_per_instance - 1);
    }
    __syncthreads();
    
    if (is_last) {
        // This block does the final reduction
        float sum = 0.0f;
        float sum_sq = 0.0f;
        int pbase = instance_id * num_blocks_per_instance;
        
        for (int i = threadIdx.x; i < num_blocks_per_instance; i += BLOCK_SIZE) {
            sum += partial_sum[pbase + i];
            sum_sq += partial_sum_sq[pbase + i];
        }
        
        sum = blockReduceSum(sum);
        sum_sq = blockReduceSum(sum_sq);
        
        if (threadIdx.x == 0) {
            float mean = sum / N;
            float var = fmaf(-mean, mean, sum_sq / N);
            mean_out[instance_id] = mean;
            invstd_out[instance_id] = rsqrtf(var + eps);
        }
    }
}

// Apply kernel: normalize and scale+shift
__global__ void layernorm_apply_kernel(
    const float4* __restrict__ input,
    const float4* __restrict__ weight,
    const float4* __restrict__ bias,
    const float* __restrict__ mean,
    const float* __restrict__ invstd,
    float4* __restrict__ output,
    int N4
) {
    int instance_id = blockIdx.y;
    int block_id = blockIdx.x;
    
    float m = __ldg(&mean[instance_id]);
    float s = __ldg(&invstd[instance_id]);
    
    const float4* inst_input = input + (size_t)instance_id * N4;
    float4* inst_output = output + (size_t)instance_id * N4;
    
    int elems_per_block = BLOCK_SIZE * VEC_PER_THREAD;
    int base = block_id * elems_per_block + threadIdx.x;
    
    #pragma unroll
    for (int i = 0; i < VEC_PER_THREAD; i++) {
        int idx = base + i * BLOCK_SIZE;
        if (idx < N4) {
            float4 val = __ldg(&inst_input[idx]);
            float4 wt = __ldg(&weight[idx]);
            float4 bi = __ldg(&bias[idx]);
            float4 out;
            out.x = fmaf((val.x - m) * s, wt.x, bi.x);
            out.y = fmaf((val.y - m) * s, wt.y, bi.y);
            out.z = fmaf((val.z - m) * s, wt.z, bi.z);
            out.w = fmaf((val.w - m) * s, wt.w, bi.w);
            inst_output[idx] = out;
        }
    }
}

torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float eps) {
    auto output = torch::empty_like(input);
    
    int batch_size = input.size(0);
    int N = input.numel() / batch_size;
    int N4 = N >> 2;
    
    int elems_per_block = BLOCK_SIZE * VEC_PER_THREAD;  // float4 per block
    int num_blocks = (N4 + elems_per_block - 1) / elems_per_block;
    
    auto options_f = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    auto options_i = torch::TensorOptions().dtype(torch::kInt32).device(input.device());
    
    auto partial_sum = torch::empty({batch_size * num_blocks}, options_f);
    auto partial_sum_sq = torch::empty({batch_size * num_blocks}, options_f);
    auto mean_t = torch::empty({batch_size}, options_f);
    auto invstd_t = torch::empty({batch_size}, options_f);
    auto block_counter = torch::zeros({batch_size}, options_i);
    
    // Phase 1: fused stats
    dim3 grid1(num_blocks, batch_size);
    layernorm_fused_stats_kernel<<<grid1, BLOCK_SIZE>>>(
        reinterpret_cast<const float4*>(input.data_ptr<float>()),
        partial_sum.data_ptr<float>(),
        partial_sum_sq.data_ptr<float>(),
        mean_t.data_ptr<float>(),
        invstd_t.data_ptr<float>(),
        block_counter.data_ptr<int>(),
        N4, N, num_blocks, eps
    );
    
    // Phase 2: apply
    dim3 grid2(num_blocks, batch_size);
    layernorm_apply_kernel<<<grid2, BLOCK_SIZE>>>(
        reinterpret_cast<const float4*>(input.data_ptr<float>()),
        reinterpret_cast<const float4*>(weight.data_ptr<float>()),
        reinterpret_cast<const float4*>(bias.data_ptr<float>()),
        mean_t.data_ptr<float>(),
        invstd_t.data_ptr<float>(),
        reinterpret_cast<float4*>(output.data_ptr<float>()),
        N4
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(
            x, self.ln.weight, self.ln.bias, self.ln.eps
        )

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
