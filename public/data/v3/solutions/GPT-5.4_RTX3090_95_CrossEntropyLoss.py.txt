import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "reduction"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _cross_entropy_fwd_kernel(
    logits_ptr,
    targets_ptr,
    out_ptr,
    stride_row,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    row_ptr = logits_ptr + pid * stride_row
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(row_ptr + offs, mask=mask, other=-float('inf')).to(tl.float32)
    t = tl.load(targets_ptr + pid)

    row_max = tl.max(x, axis=0)
    x_shifted = x - row_max
    exp_x = tl.exp(x_shifted)
    sum_exp = tl.sum(exp_x, axis=0)
    logsumexp = tl.log(sum_exp) + row_max
    target_logit = tl.load(row_ptr + t).to(tl.float32)
    loss = logsumexp - target_logit
    tl.store(out_ptr + pid, loss)


class Model(nn.Module):
    """
    Optimized Cross Entropy Loss for [batch, classes] logits and int64 targets.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        assert predictions.is_cuda and targets.is_cuda
        assert predictions.ndim == 2
        assert targets.ndim == 1
        batch, n_cols = predictions.shape
        out = torch.empty((batch,), device=predictions.device, dtype=torch.float32)
        block_size = triton.next_power_of_2(n_cols)
        num_warps = 4 if block_size <= 1024 else 8
        _cross_entropy_fwd_kernel[(batch,)](
            predictions,
            targets,
            out,
            predictions.stride(0),
            n_cols,
            BLOCK_SIZE=block_size,
            num_warps=num_warps,
        )
        return out.mean()


batch_size = 4096
num_classes = 1024
input_shape = (num_classes,)

    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]
