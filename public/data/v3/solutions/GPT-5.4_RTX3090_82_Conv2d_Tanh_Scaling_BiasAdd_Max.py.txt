import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def fused_scale_bias_maxpool2x2_kernel(
    x_ptr, bias_ptr, y_ptr,
    N, C, H, W,
    OH, OW,
    scale,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_biasc, stride_biash, stride_biasw,
    stride_yn, stride_yc, stride_yh, stride_yw,
    BLOCK_W: tl.constexpr,
):
    pid = tl.program_id(0)
    ow_block = pid * BLOCK_W
    offs = ow_block + tl.arange(0, BLOCK_W)
    mask = offs < OW

    n = tl.program_id(1)
    c = tl.program_id(2)

    bias = tl.load(bias_ptr + c * stride_biasc)

    h0 = 2 * tl.program_id(3)
    w0 = 2 * offs

    base = x_ptr + n * stride_xn + c * stride_xc + h0 * stride_xh

    v00 = tl.load(base + w0 * stride_xw, mask=mask, other=-float('inf'))
    v01 = tl.load(base + (w0 + 1) * stride_xw, mask=mask, other=-float('inf'))
    v10 = tl.load(base + stride_xh + w0 * stride_xw, mask=mask, other=-float('inf'))
    v11 = tl.load(base + stride_xh + (w0 + 1) * stride_xw, mask=mask, other=-float('inf'))

    v00 = v00 * scale + bias
    v01 = v01 * scale + bias
    v10 = v10 * scale + bias
    v11 = v11 * scale + bias

    m = tl.maximum(tl.maximum(v00, v01), tl.maximum(v10, v11))

    out_ptr = y_ptr + n * stride_yn + c * stride_yc + tl.program_id(3) * stride_yh + offs * stride_yw
    tl.store(out_ptr, m, mask=mask)


class Model(nn.Module):
    """
    Optimized model: use cuDNN conv, then fuse tanh+scale+bias+2x2 maxpool in Triton.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = float(scaling_factor)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        assert pool_kernel_size == 2, 'This optimized kernel assumes 2x2 max-pool.'
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        x = self.conv(x)
        x = torch.tanh(x)
        assert x.is_cuda, 'CUDA tensor required'
        N, C, H, W = x.shape
        OH, OW = H // 2, W // 2
        y = torch.empty((N, C, OH, OW), device=x.device, dtype=x.dtype)

        BLOCK_W = 128
        grid = (triton.cdiv(OW, BLOCK_W), N, C, OH)
        fused_scale_bias_maxpool2x2_kernel[grid](
            x, self.bias, y,
            N, C, H, W,
            OH, OW,
            self.scaling_factor,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            self.bias.stride(0), self.bias.stride(1), self.bias.stride(2),
            y.stride(0), y.stride(1), y.stride(2), y.stride(3),
            BLOCK_W=BLOCK_W,
        )
        return y


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
