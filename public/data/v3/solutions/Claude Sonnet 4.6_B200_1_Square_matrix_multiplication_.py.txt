#define CUTLASS_CHECK(status) \

    } \
  } while(0)

// ============================================================
// FP16 GEMM on Blackwell SM100
// ============================================================
namespace fp16_gemm {

using LayoutA            = cutlass::layout::RowMajor;

constexpr int AlignmentB = 128 / cutlass::sizeof_bits<ElementB>::value;

// ============================================================
// Dispatch function
// ============================================================
torch::Tensor cutlass_gemm(torch::Tensor A, torch::Tensor B) {
  TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "A and B must be 2D tensors");
  TORCH_CHECK(A.size(1) == B.size(0), "A.cols must equal B.rows");
  TORCH_CHECK(A.is_cuda() && B.is_cuda(), "tensors must be on CUDA");
  TORCH_CHECK(A.is_contiguous() && B.is_contiguous(), "tensors must be contiguous");
  TORCH_CHECK(A.scalar_type() == B.scalar_type(), "A and B must have the same dtype");

  int M = A.size(0);
  int N = B.size(1);
  int K = A.size(1);

  auto dtype = A.scalar_type();
  torch::Tensor D = torch::empty({M, N}, A.options());

  if (dtype == torch::kFloat16) {
    using namespace fp16_gemm;
    auto stride_A = cutlass::make_cute_packed_stride(StrideA{}, {M, K, 1});
    auto stride_B = cutlass::make_cute_packed_stride(StrideB{}, {K, N, 1});
    auto stride_C = cutlass::make_cute_packed_stride(StrideC{}, {M, N, 1});
    auto stride_D = cutlass::make_cute_packed_stride(StrideD{}, {M, N, 1});

    typename Gemm::Arguments arguments{
      cutlass::gemm::GemmUniversalMode::kGemm,
      {M, N, K, 1},
      {(ElementA*)A.data_ptr(), stride_A, (ElementB*)B.data_ptr(), stride_B},
      {{1.0f, 0.0f}, (ElementC*)nullptr, stride_C, (ElementC*)D.data_ptr(), stride_D}
    };

    Gemm gemm_op;
    size_t workspace_size = Gemm::get_workspace_size(arguments);
    auto workspace = torch::empty({(int64_t)workspace_size}, torch::TensorOptions().dtype(torch::kUInt8).device(A.device()));
    CUTLASS_CHECK(gemm_op.can_implement(arguments));
    CUTLASS_CHECK(gemm_op.initialize(arguments, workspace.data_ptr<uint8_t>()));
    CUTLASS_CHECK(gemm_op.run());

  } else if (dtype == torch::kBFloat16) {
    using namespace bf16_gemm;
    auto stride_A = cutlass::make_cute_packed_stride(StrideA{}, {M, K, 1});
    auto stride_B = cutlass::make_cute_packed_stride(StrideB{}, {K, N, 1});
    auto stride_C = cutlass::make_cute_packed_stride(StrideC{}, {M, N, 1});
    auto stride_D = cutlass::make_cute_packed_stride(StrideD{}, {M, N, 1});

    typename Gemm::Arguments arguments{
      cutlass::gemm::GemmUniversalMode::kGemm,
      {M, N, K, 1},
      {(ElementA*)A.data_ptr(), stride_A, (ElementB*)B.data_ptr(), stride_B},
      {{1.0f, 0.0f}, (ElementC*)nullptr, stride_C, (ElementC*)D.data_ptr(), stride_D}
    };

    Gemm gemm_op;
    size_t workspace_size = Gemm::get_workspace_size(arguments);
    auto workspace = torch::empty({(int64_t)workspace_size}, torch::TensorOptions().dtype(torch::kUInt8).device(A.device()));
    CUTLASS_CHECK(gemm_op.can_implement(arguments));
    CUTLASS_CHECK(gemm_op.initialize(arguments, workspace.data_ptr<uint8_t>()));
    CUTLASS_CHECK(gemm_op.run());

  } else if (dtype == torch::kFloat32) {
    using namespace fp32_gemm;
    auto stride_A = cutlass::make_cute_packed_stride(StrideA{}, {M, K, 1});
    auto stride_B = cutlass::make_cute_packed_stride(StrideB{}, {K, N, 1});
    auto stride_C = cutlass::make_cute_packed_stride(StrideC{}, {M, N, 1});
    auto stride_D = cutlass::make_cute_packed_stride(StrideD{}, {M, N, 1});

    typename Gemm::Arguments arguments{
      cutlass::gemm::GemmUniversalMode::kGemm,
      {M, N, K, 1},
      {(ElementA*)A.data_ptr(), stride_A, (ElementB*)B.data_ptr(), stride_B},
      {{1.0f, 0.0f}, (ElementC*)nullptr, stride_C, (ElementC*)D.data_ptr(), stride_D}
    };

    Gemm gemm_op;
    size_t workspace_size = Gemm::get_workspace_size(arguments);
    auto workspace = torch::empty({(int64_t)workspace_size}, torch::TensorOptions().dtype(torch::kUInt8).device(A.device()));
    CUTLASS_CHECK(gemm_op.can_implement(arguments));
    CUTLASS_CHECK(gemm_op.initialize(arguments, workspace.data_ptr<uint8_t>()));
    CUTLASS_CHECK(gemm_op.run());

  } else {
    throw std::runtime_error("Unsupported dtype: only float16, bfloat16, float32 are supported");
  }

  return D;
}
"""

cutlass_gemm_cpp = "torch::Tensor cutlass_gemm(torch::Tensor A, torch::Tensor B);"

ext = load_inline(
    name='cutlass_gemm_sm100',
    cpp_sources=cutlass_gemm_cpp,
    cuda_sources=cutlass_gemm_source,
    functions=['cutlass_gemm'],
    verbose=True,
    extra_cuda_cflags=[
        '-arch=sm_100a',
        '-std=c++17',
        '-O3',
        f'-I/opt/cutlass/include',
        f'-I/opt/cutlass/tools/util/include',
        '-DCUTLASS_ARCH_MMA_SM100_SUPPORTED=1',
        '--expt-relaxed-constexpr',
        '-U__CUDA_NO_HALF_OPERATORS__',
        '-U__CUDA_NO_HALF_CONVERSIONS__',
        '-U__CUDA_NO_BFLOAT16_OPERATORS__',
        '-U__CUDA_NO_BFLOAT16_CONVERSIONS__',
    ],
    extra_cflags=['-O3', '-std=c++17'],
)


class Model(nn.Module):
    """
    Optimized square matrix multiplication using CUTLASS Blackwell SM100 kernels.
    Supports fp16, bf16, fp32 precisions.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.cutlass_gemm(A, B)


N = 2048
OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp16", "bf16", "fp32"]


def get_inputs():
    A = torch.randn(N, N)
    B = torch.randn(N, N)
    return [A, B]


def get_init_inputs():
    return []
