import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <stdio.h>

// Warp reduction for float
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Optimized sum reduction kernel using shared memory and warp shuffles
template <int BLOCK_SIZE>
__global__ void sum_reduction_kernel_float(
    const float* __restrict__ input,
    float* __restrict__ output,
    const int reduce_size,
    const int outer_size,
    const int inner_size,
    const int64_t* input_strides) {
    
    const int outer_idx = blockIdx.x;
    const int inner_idx = blockIdx.y;
    
    if (outer_idx >= outer_size || inner_idx >= inner_size) return;
    
    __shared__ float sdata[BLOCK_SIZE];
    
    float thread_sum = 0.0f;
    
    // Input strides: [outer_stride, reduce_stride, inner_stride]
    const int64_t outer_stride = input_strides[0];
    const int64_t reduce_stride = input_strides[1];
    const int64_t inner_stride = input_strides[2];
    
    // Base offset for this thread block's work
    const int64_t base_offset = outer_idx * outer_stride + inner_idx * inner_stride;
    
    for (int i = threadIdx.x; i < reduce_size; i += BLOCK_SIZE) {
        thread_sum += input[base_offset + i * reduce_stride];
    }
    
    sdata[threadIdx.x] = thread_sum;
    __syncthreads();
    
    // Parallel reduction in shared memory
    #pragma unroll
    for (int s = BLOCK_SIZE / 2; s > 32; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }
    
    // Warp-level reduction
    if (threadIdx.x < 32) {
        float val = sdata[threadIdx.x];
        val = warp_reduce_sum(val);
        if (threadIdx.x == 0) {
            sdata[0] = val;
        }
    }
    
    if (threadIdx.x == 0) {
        const int out_offset = outer_idx * inner_size + inner_idx;
        output[out_offset] = sdata[0];
    }
}

// Half precision kernel
template <int BLOCK_SIZE>
__global__ void sum_reduction_kernel_half(
    const half* __restrict__ input,
    half* __restrict__ output,
    const int reduce_size,
    const int outer_size,
    const int inner_size,
    const int64_t* input_strides) {
    
    const int outer_idx = blockIdx.x;
    const int inner_idx = blockIdx.y;
    
    if (outer_idx >= outer_size || inner_idx >= inner_size) return;
    
    __shared__ float sdata[BLOCK_SIZE];
    
    float thread_sum = 0.0f;
    
    const int64_t outer_stride = input_strides[0];
    const int64_t reduce_stride = input_strides[1];
    const int64_t inner_stride = input_strides[2];
    
    const int64_t base_offset = outer_idx * outer_stride + inner_idx * inner_stride;
    
    for (int i = threadIdx.x; i < reduce_size; i += BLOCK_SIZE) {
        thread_sum += __half2float(input[base_offset + i * reduce_stride]);
    }
    
    sdata[threadIdx.x] = thread_sum;
    __syncthreads();
    
    #pragma unroll
    for (int s = BLOCK_SIZE / 2; s > 32; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }
    
    if (threadIdx.x < 32) {
        float val = sdata[threadIdx.x];
        val = warp_reduce_sum(val);

            sdata[0] = val;
        }
    }
    
    if (threadIdx.x == 0) {

    }
}

// BFloat16 kernel
template <int BLOCK_SIZE>
__global__ void sum_reduction_kernel_bfloat16(
    const __nv_bfloat16* __restrict__ input,
    __nv_bfloat16* __restrict__ output,
    const int reduce_size,
    const int outer_size,
    const int inner_size,
    const int64_t* input_strides) {
    
    const int outer_idx = blockIdx.x;
    const int inner_idx = blockIdx.y;
    
    if (outer_idx >= outer_size || inner_idx >= inner_size) return;
    
    __shared__ float sdata[BLOCK_SIZE];
    
    float thread_sum = 0.0f;
    
    const int64_t outer_stride = input_strides[0];
    const int64_t reduce_stride = input_strides[1];
    const int64_t inner_stride = input_strides[2];
    
    const int64_t base_offset = outer_idx * outer_stride + inner_idx * inner_stride;
    
    for (int i = threadIdx.x; i < reduce_size; i += BLOCK_SIZE) {
        thread_sum += __bfloat162float(input[base_offset + i * reduce_stride]);
    }
    
    sdata[threadIdx.x] = thread_sum;
    __syncthreads();
    
    #pragma unroll
    for (int s = BLOCK_SIZE / 2; s > 32; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }
    
    if (threadIdx.x < 32) {
        float val = sdata[threadIdx.x];
        val = warp_reduce_sum(val);
        if (threadIdx.x == 0) {
            sdata[0] = val;
        }
    }
    
    if (threadIdx.x == 0) {
        const int out_offset = outer_idx * inner_size + inner_idx;

    }
}

torch::Tensor sum_reduction(torch::Tensor input, int64_t dim) {
    // Handle negative dimension
    if (dim < 0) {
        dim += input.dim();
    }
    
    auto sizes = input.sizes();
    auto strides = input.strides();
    int64_t ndim = input.dim();
    
    // Calculate outer size, reduce size, and inner size
    int64_t outer_size = 1;
    int64_t outer_stride = 0;
    for (int64_t i = 0; i < dim; ++i) {
        outer_size *= sizes[i];
        if (i == dim - 1) outer_stride = strides[i];
    }
    if (dim == 0) outer_stride = strides[0];
    // Actually we need the stride of the first dimension after flattening
    outer_stride = (dim > 0) ? strides[0] : 0;
    
    int64_t reduce_size = sizes[dim];
    int64_t reduce_stride = strides[dim];
    
    int64_t inner_size = 1;
    int64_t inner_stride = 0;
    for (int64_t i = dim + 1; i < ndim; ++i) {
        inner_size *= sizes[i];
    }
    inner_stride = (dim + 1 < ndim) ? strides[dim + 1] : 1;
    
    // Create output shape
    std::vector<int64_t> out_sizes(sizes.begin(), sizes.end());
    out_sizes[dim] = 1;
    
    auto output = torch::empty(out_sizes, input.options());
    
    dim3 grid(outer_size, inner_size);
    const int BLOCK_SIZE = 256;
    
    // Create stride tensor
    auto strides_tensor = torch::tensor(
        {outer_stride, reduce_stride, inner_stride},
        torch::TensorOptions().dtype(torch::kInt64).device(torch::kCPU)
    );
    auto strides_dev = strides_tensor.to(input.device());
    
    if (input.scalar_type() == torch::kFloat32) {
        sum_reduction_kernel_float<BLOCK_SIZE><<<grid, BLOCK_SIZE>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            static_cast<int>(reduce_size),
            static_cast<int>(outer_size),
            static_cast<int>(inner_size),
            strides_dev.data_ptr<int64_t>()
        );
    } else if (input.scalar_type() == torch::kFloat16) {
        sum_reduction_kernel_half<BLOCK_SIZE><<<grid, BLOCK_SIZE>>>(
            reinterpret_cast<const half*>(input.data_ptr<c10::Half>()),
            reinterpret_cast<half*>(output.data_ptr<c10::Half>()),
            static_cast<int>(reduce_size),
            static_cast<int>(outer_size),
            static_cast<int>(inner_size),
            strides_dev.data_ptr<int64_t>()
        );
    } else if (input.scalar_type() == torch::kBFloat16) {
        sum_reduction_kernel_bfloat16<BLOCK_SIZE><<<grid, BLOCK_SIZE>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<c10::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<c10::BFloat16>()),
            static_cast<int>(reduce_size),
            static_cast<int>(outer_size),
            static_cast<int>(inner_size),
            strides_dev.data_ptr<int64_t>()
        );
    }
    
    return output;
}
'''

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources='torch::Tensor sum_reduction(torch::Tensor, int64_t);',
    cuda_sources=cuda_source,
    functions=['sum_reduction'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduction(x, self.dim)

batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
