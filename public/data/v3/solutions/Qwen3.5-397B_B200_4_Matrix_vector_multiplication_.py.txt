import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized matrix-vector multiplication kernel for B200
// A: M x K, B: K x 1, C: M x 1
// Each thread block handles one row of A
// Uses shared memory to cache B and warp-level reduction

__global__ void matvec_kernel(const float* __restrict__ A, 
                               const float* __restrict__ B,
                               float* __restrict__ C,
                               int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    extern __shared__ float B_shared[];
    
    int tid = threadIdx.x;
    int blockDim_x = blockDim.x;
    
    float sum = 0.0f;
    
    // Process B in tiles, caching in shared memory
    for (int k_base = 0; k_base < K; k_base += blockDim_x) {
        // Load B elements into shared memory (coalesced access)
        int k_idx = k_base + tid;
        B_shared[tid] = (k_idx < K) ? B[k_idx] : 0.0f;
        __syncthreads();
        
        // Compute partial dot product
        if (k_base + tid < K) {
            sum += A[row * K + k_base + tid] * B_shared[tid];
        }
        __syncthreads();
    }
    
    // Block-level reduction using shared memory
    B_shared[tid] = sum;
    __syncthreads();
    
    // Reduce in log2(blockDim) steps
    for (int stride = blockDim_x / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            B_shared[tid] += B_shared[tid + stride];
        }
        __syncthreads();
    }
    
    // Write result
    if (tid == 0) {
        C[row] = B_shared[0];
    }
}

torch::Tensor matvec(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, torch::dtype(A.dtype()).device(A.device()));
    
    // Launch configuration: one block per row, 256 threads per block
    int threads_per_block = 256;
    int num_blocks = M;
    int shared_mem_size = threads_per_block * sizeof(float);
    
    matvec_kernel<<<num_blocks, threads_per_block, shared_mem_size>>>(
        A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, K);
    
    return C;
}
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources='torch::Tensor matvec(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['matvec'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math'],
)

class Model(nn.Module):
    """
    Optimized model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.matvec(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
