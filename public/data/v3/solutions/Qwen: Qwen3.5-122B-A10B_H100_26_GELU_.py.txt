import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math.h>

// GELU exact: 0.5 * x * (1 + erf(x / sqrt(2)))
__device__ __forceinline__ float gelu_float(float x) {
    return 0.5f * x * (1.0f + erff(x * 0.7071067811865475f));
}

__device__ __forceinline__ half gelu_half(half x) {
    float fx = __half2float(x);
    return __float2half(gelu_float(fx));
}

__device__ __forceinline__ nv_bfloat16 gelu_bfloat(nv_bfloat16 x) {
    float fx = __bfloat162float(x);
    return __float2bfloat16(gelu_float(fx));
}

// FP32 kernel with vectorized access
__global__ void gelu_kernel_fp32_vec(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x * 4 + threadIdx.x;
    int stride = blockDim.x * gridDim.x * 4;
    
    for (int i = 0; i < 4 && idx + i < n; i++) {
        y[idx + i] = gelu_float(x[idx + i]);
    }
}

// FP16 kernel with half4 vectorized access
__global__ void gelu_kernel_fp16(const half* __restrict__ x, half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x * 4 + threadIdx.x;
    int stride = blockDim.x * gridDim.x * 4;
    
    for (int i = 0; i < 4 && idx + i < n; i++) {
        y[idx + i] = gelu_half(x[idx + i]);
    }
}

// BF16 kernel with bfloat16x4 vectorized access
__global__ void gelu_kernel_bf16(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x * 4 + threadIdx.x;
    int stride = blockDim.x * gridDim.x * 4;
    
    for (int i = 0; i < 4 && idx + i < n; i++) {
        y[idx + i] = gelu_bfloat(x[idx + i]);
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    if (n == 0) return y;
    
    int threads = 256;
    int blocks = (n + threads * 4 - 1) / (threads * 4);
    blocks = max(blocks, 1);
    
    if (x.dtype() == torch::kFloat32) {
        gelu_kernel_fp32_vec<<<blocks, threads>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            n
        );
    } else if (x.dtype() == torch::kFloat16) {
        gelu_kernel_fp16<<<blocks, threads>>>(
            reinterpret_cast<const half*>(x.data_ptr()),
            reinterpret_cast<half*>(y.data_ptr()),
            n
        );
    } else if (x.dtype() == torch::kBFloat16) {
        gelu_kernel_bf16<<<blocks, threads>>>(
            reinterpret_cast<const nv_bfloat16*>(x.data_ptr()),
            reinterpret_cast<nv_bfloat16*>(y.data_ptr()),
            n
        );
    } else {
        gelu_kernel_fp32_vec<<<blocks, threads>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            n
        );
    }
    
    cudaDeviceSynchronize();
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math', '-arch=sm_90'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_op(x)

def get_inputs():
    return [torch.randn(256, 16384)]

def get_init_inputs():
    return []
