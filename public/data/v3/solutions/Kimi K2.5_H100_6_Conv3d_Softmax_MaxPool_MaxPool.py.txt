import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import math

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/util/Half.h>
#include <c10/util/BFloat16.h>
#include <cmath>

namespace {

template <typename scalar_t>
__global__ void conv3d_softmax_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int B, int C_in, int D_in, int H_in, int W_in,
    int C_out, int D_out, int H_out, int W_out)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_spatial = B * D_out * H_out * W_out;
    
    if (idx >= total_spatial) return;
    
    int w = idx % W_out;
    int tmp = idx / W_out;
    int h = tmp % H_out;
    tmp /= H_out;
    int d = tmp % D_out;
    int b = tmp / D_out;
    
    float vals[16];
    
    #pragma unroll 16
    for (int co = 0; co < C_out; ++co) {
        float sum = static_cast<float>(bias[co]);
        
        for (int ci = 0; ci < C_in; ++ci) {
            #pragma unroll 3
            for (int kd = 0; kd < 3; ++kd) {
                #pragma unroll 3
                for (int kh = 0; kh < 3; ++kh) {
                    #pragma unroll 3
                    for (int kw = 0; kw < 3; ++kw) {
                        int in_d = d + kd;
                        int in_h = h + kh;
                        int in_w = w + kw;
                        
                        int in_idx = ((b * C_in + ci) * D_in + in_d) * (H_in * W_in) 
                                    + in_h * W_in + in_w;
                        int w_idx = ((co * C_in + ci) * 3 + kd) * 9 + kh * 3 + kw;
                        
                        sum += static_cast<float>(input[in_idx]) * static_cast<float>(weight[w_idx]);
                    }
                }
            }
        }
        vals[co] = sum;
    }
    
    float max_val = vals[0];
    #pragma unroll 16
    for (int i = 1; i < C_out; ++i) {
        max_val = fmaxf(max_val, vals[i]);
    }
    
    float exp_sum = 0.0f;
    #pragma unroll 16
    for (int i = 0; i < C_out; ++i) {
        vals[i] = expf(vals[i] - max_val);
        exp_sum += vals[i];
    }
    
    float inv_sum = 1.0f / exp_sum;
    
    #pragma unroll 16
    for (int co = 0; co < C_out; ++co) {
        int out_idx = ((b * C_out + co) * D_out + d) * (H_out * W_out) + h * W_out + w;
        output[out_idx] = static_cast<scalar_t>(vals[co] * inv_sum);
    }
}

template <typename scalar_t>
__global__ void dual_maxpool_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int B, int C, int D_in, int H_in, int W_in,
    int D_out, int H_out, int W_out)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * C * D_out * H_out * W_out;
    
    if (idx >= total) return;
    
    int w = idx % W_out;
    int tmp = idx / W_out;
    int h = tmp % H_out;
    tmp /= H_out;
    int d = tmp % D_out;
    tmp /= D_out;
    int ch = tmp % C;
    int b = tmp / C;
    
    float max_val = -1e20f;
    
    #pragma unroll 4
    for (int kd = 0; kd < 4; ++kd) {
        int in_d = d * 4 + kd;
        if (in_d >= D_in) continue;
        
        #pragma unroll 4
        for (int kh = 0; kh < 4; ++kh) {
            int in_h = h * 4 + kh;
            if (in_h >= H_in) continue;
            
            #pragma unroll 4
            for (int kw = 0; kw < 4; ++kw) {
                int in_w = w * 4 + kw;
                if (in_w >= W_in) continue;
                
                int in_idx = ((b * C + ch) * D_in + in_d) * (H_in * W_in) 
                           + in_h * W_in + in_w;
                float val = static_cast<float>(input[in_idx]);
                max_val = fmaxf(max_val, val);
            }
        }
    }
    
    int out_idx = ((b * C + ch) * D_out + d) * (H_out * W_out) + h * W_out + w;
    output[out_idx] = static_cast<scalar_t>(max_val);
}

} // namespace

torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int64_t D_conv, int64_t H_conv, int64_t W_conv,
    int64_t D_out, int64_t H_out, int64_t W_out)
{
    int64_t B = input.size(0);
    int64_t C_in = input.size(1);
    int64_t C_out = weight.size(0);
    
    input = input.contiguous();
    weight = weight.contiguous();
    bias = bias.contiguous();
    
    auto intermediate = torch::empty({B, C_out, D_conv, H_conv, W_conv}, input.options());
    auto output = torch::empty({B, C_out, D_out, H_out, W_out}, input.options());
    
    int total_spatial = B * D_conv * H_conv * W_conv;
    const int threads = 256;
    const int blocks = (total_spatial + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half,
        at::ScalarType::BFloat16, 
        input.scalar_type(), 
        "conv3d_softmax", 
        [&]() {
            conv3d_softmax_kernel<scalar_t><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                bias.data_ptr<scalar_t>(),
                intermediate.data_ptr<scalar_t>(),
                B, C_in, input.size(2), input.size(3), input.size(4),
                C_out, D_conv, H_conv, W_conv
            );
        }
    );
    
    int total_out = B * C_out * D_out * H_out * W_out;
    const int blocks2 = (total_out + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half,
        at::ScalarType::BFloat16, 
        input.scalar_type(), 
        "dual_maxpool", 
        [&]() {
            dual_maxpool_kernel<scalar_t><<<blocks2, threads>>>(
                intermediate.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                B, C_out, D_conv, H_conv, W_conv,
                D_out, H_out, W_out
            );
        }
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv3d_softmax_pool',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor, torch::Tensor, torch::Tensor, int64_t, int64_t, int64_t, int64_t, int64_t, int64_t);',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3', '-use_fast_math', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels=None, out_channels=None, kernel_size=None, pool_kernel_size=None):
        super(Model, self).__init__()
        
        # Default values from reference if not provided
        if in_channels is None:
            in_channels = 3
        if out_channels is None:
            out_channels = 16
        if kernel_size is None:
            kernel_size = 3
        if pool_kernel_size is None:
            pool_kernel_size = 2
            
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.pool_kernel_size = pool_kernel_size
        
        self.weight = nn.Parameter(torch.empty(out_channels, in_channels, kernel_size, kernel_size, kernel_size))
        self.bias = nn.Parameter(torch.empty(out_channels))
        
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in)
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        B, C, D, H, W = x.shape
        
        D_conv = D - self.kernel_size + 1
        H_conv = H - self.kernel_size + 1
        W_conv = W - self.kernel_size + 1
        
        D_out = D_conv // (self.pool_kernel_size * self.pool_kernel_size)
        H_out = H_conv // (self.pool_kernel_size * self.pool_kernel_size)
        W_out = W_conv // (self.pool_kernel_size * self.pool_kernel_size)
        
        return ext.fused_forward(
            x, self.weight, self.bias,
            D_conv, H_conv, W_conv,
            D_out, H_out, W_out
        )

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
