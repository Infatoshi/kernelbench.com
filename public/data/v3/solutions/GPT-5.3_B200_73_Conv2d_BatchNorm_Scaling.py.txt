import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void fused_conv_bn_scale(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ a,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int C_in, int H, int W,
    int C_out, int K, int H_out, int W_out)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c_out = (idx / (W_out * H_out)) % C_out;
    int n = idx / (W_out * H_out * C_out);

    float sum = 0.0f;

    for (int c = 0; c < C_in; ++c) {
        for (int kh = 0; kh < K; ++kh) {
            for (int kw = 0; kw < K; ++kw) {
                int h_in = h_out + kh;
                int w_in = w_out + kw;
                int x_idx = ((n * C_in + c) * H + h_in) * W + w_in;
                int w_idx = ((c_out * C_in + c) * K + kh) * K + kw;
                sum += x[x_idx] * w[w_idx];
            }
        }
    }

    float val = sum * a[c_out] + b[c_out];
    y[idx] = val;
}


torch::Tensor run(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor a,
    torch::Tensor b)
{
    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int C_out = w.size(0);
    int K = w.size(2);

    int H_out = H - K + 1;
    int W_out = W - K + 1;

    auto y = torch::empty({N, C_out, H_out, W_out}, x.options());

    int total = N * C_out * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_conv_bn_scale<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        a.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        N, C_in, H, W, C_out, K, H_out, W_out);

    return y;
}
'''

ext = load_inline(
    name='fused_conv_bn',
    cpp_sources='torch::Tensor run(torch::Tensor x, torch::Tensor w, torch::Tensor a, torch::Tensor b);',
    cuda_sources=cuda_src,
    functions=['run'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        w = self.conv.weight
        if self.conv.bias is not None:
            # fold bias into b later
            bias = self.conv.bias
        else:
            bias = torch.zeros(w.size(0), device=w.device, dtype=w.dtype)

        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        gamma = self.bn.weight
        beta = self.bn.bias
        eps = self.bn.eps

        # compute fused a, b
        denom = torch.rsqrt(running_var + eps)
        a = gamma * denom
        b = beta - running_mean * a

        # include conv bias
        b = b + bias * a

        # include scaling
        a = a * self.scaling_factor
        b = b * self.scaling_factor

        return ext.run(x.contiguous(), w.contiguous(), a.contiguous(), b.contiguous())


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda', dtype=torch.float32)]


def get_init_inputs():
    return [3, 16, 3, 2.0]
