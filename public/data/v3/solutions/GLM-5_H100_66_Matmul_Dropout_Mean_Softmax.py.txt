import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def matmul_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Meta-parameters
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """Kernel for computing C = A x B^T where A is MxK and B is NxK"""
    # Block indices
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Create pointers for the initial blocks
    rm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    rn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    ram = tl.max_contiguous(tl.multiple_of(rm, BLOCK_SIZE_M), BLOCK_SIZE_M)
    rbn = tl.max_contiguous(tl.multiple_of(rn, BLOCK_SIZE_N), BLOCK_SIZE_N)
    rk = tl.arange(0, BLOCK_SIZE_K)

    # Pointers
    a = a_ptr + (ram[:, None] * stride_am + rk[None, :] * stride_ak)
    b = b_ptr + (rbn[None, :] * stride_bn + rk[:, None] * stride_bk)

    # Accumulator
    acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    # K-loop
    for k in range(0, K, BLOCK_SIZE_K):
        k_remaining = K - k
        _0 = tl.zeros((1, 1), dtype=tl.float32)
        
        a_mask = (ram[:, None] < M) & (rk[None, :] < k_remaining)
        b_mask = (rbn[None, :] < N) & (rk[:, None] < k_remaining)
        
        a_block = tl.load(a, mask=a_mask, other=_0)
        b_block = tl.load(b, mask=b_mask, other=_0)
        
        acc += tl.dot(a_block, b_block)
        
        a += BLOCK_SIZE_K * stride_ak
        b += BLOCK_SIZE_K * stride_bk

    # Store result
    cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    
    c = c_ptr + (cm[:, None] * stride_cm + cn[None, :] * stride_cn)
    c_mask = (cm[:, None] < M) & (cn[None, :] < N)
    tl.store(c, acc, mask=c_mask)


@triton.jit
def fused_dropout_mean_softmax_kernel(
    # Input/output
    x_ptr, output_ptr, bias_ptr,
    # Dropout mask and scale
    dropout_mask_ptr, dropout_scale,
    # Dimensions  
    batch_size, out_features,
    # Strides
    stride_x_batch, stride_x_feat,
    # Block size
    BLOCK_SIZE: tl.constexpr,
):
    """Fused bias + dropout + mean + softmax kernel.
    
    x: (batch_size, out_features)
    output: (out_features,) - will be reshaped to (1, out_features) outside
    dropout_mask: (batch_size, out_features), 1 for keep, 0 for drop
    """
    pid = tl.program_id(0)
    
    # Feature offsets this block handles
    feat_start = pid * BLOCK_SIZE
    feat_offsets = feat_start + tl.arange(0, BLOCK_SIZE)
    feat_mask = feat_offsets < out_features
    
    # Load bias for this block
    bias = tl.load(bias_ptr + feat_offsets, mask=feat_mask, other=0.0)
    
    # First pass: compute mean across batch (with dropout and bias)
    # Sum all batch elements for each feature
    sum_val = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for b in range(batch_size):
        # Load x[b, feat_start:feat_start+BLOCK_SIZE]
        x_ptrs = x_ptr + b * stride_x_batch + feat_offsets * stride_x_feat
        x_val = tl.load(x_ptrs, mask=feat_mask, other=0.0)
        
        # Load dropout mask
        mask_ptrs = dropout_mask_ptr + b * out_features + feat_offsets
        mask_val = tl.load(mask_ptrs, mask=feat_mask, other=0.0)
        
        # Apply bias + dropout: (x + bias) * mask * scale
        x_val = (x_val + bias) * mask_val * dropout_scale
        
        sum_val += x_val
    
    # Compute mean
    mean_val = sum_val / batch_size
    
    # Store intermediate mean values
    tl.store(output_ptr + feat_offsets, mean_val, mask=feat_mask)


@triton.jit
def softmax_kernel(
    input_ptr, output_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    """Compute softmax on a 1D array."""
    # First pass: find max
    max_val = float('-inf')
    for block_start in range(0, n_elements, BLOCK_SIZE):
        offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < n_elements
        vals = tl.load(input_ptr + offsets, mask=mask, other=float('-inf'))
        block_max = tl.max(vals, axis=0)
        max_val = tl.maximum(max_val, block_max)
    
    # Second pass: compute exp and sum
    sum_exp = 0.0
    for block_start in range(0, n_elements, BLOCK_SIZE):
        offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < n_elements
        vals = tl.load(input_ptr + offsets, mask=mask, other=float('-inf'))
        exp_vals = tl.exp(vals - max_val)
        sum_exp += tl.sum(exp_vals, axis=0)
    
    # Third pass: compute softmax
    for block_start in range(0, n_elements, BLOCK_SIZE):
        offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < n_elements
        vals = tl.load(input_ptr + offsets, mask=mask, other=float('-inf'))
        exp_vals = tl.exp(vals - max_val)
        softmax_vals = exp_vals / sum_exp
        tl.store(output_ptr + offsets, softmax_vals, mask=mask)


def fused_matmul_dropout_mean_softmax(x, weight, bias, dropout_p):
    """
    Fused operation: matmul -> dropout -> mean -> softmax
    """
    batch_size, in_features = x.shape
    out_features = weight.shape[0]
    
    # Step 1: Matmul using Triton kernel
    matmul_out = torch.empty(batch_size, out_features, device=x.device, dtype=torch.float32)
    
    # Optimized block sizes for H100
    BLOCK_SIZE_M = 64
    BLOCK_SIZE_N = 128
    BLOCK_SIZE_K = 64
    GROUP_SIZE_M = 8
    
    grid = (triton.cdiv(batch_size, BLOCK_SIZE_M) * triton.cdiv(out_features, BLOCK_SIZE_N),)
    
    matmul_kernel[grid](
        x, weight, matmul_out,
        batch_size, out_features, in_features,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        matmul_out.stride(0), matmul_out.stride(1),
        BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
    )
    
    # Step 2: Generate dropout mask
    dropout_mask = (torch.rand(batch_size, out_features, device=x.device, dtype=x.dtype) > dropout_p).float()
    dropout_scale = 1.0 / (1.0 - dropout_p)
    
    # Step 3: Fused bias + dropout + mean
    mean_out = torch.empty(out_features, device=x.device, dtype=torch.float32)
    
    BLOCK_SIZE = 512
    grid_mean = (triton.cdiv(out_features, BLOCK_SIZE),)
    
    fused_dropout_mean_softmax_kernel[grid_mean](
        matmul_out, mean_out, bias,
        dropout_mask, dropout_scale,
        batch_size, out_features,
        matmul_out.stride(0), matmul_out.stride(1),
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    # Step 4: Softmax
    output = torch.empty_like(mean_out)
    softmax_kernel[(1,)](
        mean_out, output,
        out_features,
        BLOCK_SIZE=2048,
    )
    
    return output.unsqueeze(0)


class Model(nn.Module):
    """
    A model that performs matrix multiplication, applies dropout, calculates the mean, and then applies softmax.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.dropout_p = dropout_p
        
        # Create weight and bias directly (like nn.Linear)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize weights (same as nn.Linear)
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).

        Returns:
            torch.Tensor: Output tensor of shape (1, out_features).
        """
        return fused_matmul_dropout_mean_softmax(x, self.weight, self.bias, self.dropout_p)


batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
