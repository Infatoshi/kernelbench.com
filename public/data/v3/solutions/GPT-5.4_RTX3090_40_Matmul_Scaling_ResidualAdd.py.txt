import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _scale_residual_kernel(x_ptr, y_ptr, n_elements, alpha, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offsets = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask)
    y = x * alpha + x
    tl.store(y_ptr + offsets, y, mask=mask)


def _fused_scale_residual(x: torch.Tensor, alpha: float):
    y = torch.empty_like(x)
    n = x.numel()
    grid = lambda meta: (triton.cdiv(n, meta['BLOCK']),)
    _scale_residual_kernel[grid](x, y, n, alpha, BLOCK=1024)
    return y


class Model(nn.Module):
    """
    Optimized model: keep the Linear module for parameter/state_dict compatibility,
    but fuse the post-matmul scaling + residual add into a Triton kernel.
    Since original_x is clone().detach() immediately after the linear output, the
    numerics are exactly x * scaling_factor + x == x * (1 + scaling_factor).
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = torch.ops.aten.linear.default(x, self.matmul.weight, self.matmul.bias)
        return _fused_scale_residual(x, self.scaling_factor)


batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 0.5


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, scaling_factor]
