import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized CUDA GELU implementation using vectorized loads/stores
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Fast GELU approximation: 0.5 * x * (1 + tanh(√(2/π) * (x + 0.044715 * x³)))
__device__ __forceinline__ float gelu_fast(float x) {
    const float sqrt_2_over_pi = 0.7978845608028654f;
    const float coeff = 0.044715f;
    return 0.5f * x * (1.0f + tanhf(sqrt_2_over_pi * (x + coeff * x * x * x)));
}

// Vectorized kernel for float (float4)
__global__ void gelu_kernel_float4(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * 4;
    if (idx + 3 < n) {
        float4 x_vec = reinterpret_cast<const float4*>(x)[blockIdx.x * blockDim.x + threadIdx.x];
        float4 y_vec;
        y_vec.x = gelu_fast(x_vec.x);
        y_vec.y = gelu_fast(x_vec.y);
        y_vec.z = gelu_fast(x_vec.z);
        y_vec.w = gelu_fast(x_vec.w);
        reinterpret_cast<float4*>(y)[blockIdx.x * blockDim.x + threadIdx.x] = y_vec;
    } else {
        // Handle remaining elements
        for (int i = idx; i < n && i < idx + 4; i++) {
            y[i] = gelu_fast(x[i]);
        }
    }
}

// Kernel for half (fp16) - compute in float, store as half
__global__ void gelu_kernel_half(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = gelu_fast(x[idx]);
    }
}

// Kernel for bf16 - compute in float, store as float (PyTorch handles conversion)
__global__ void gelu_kernel_bfloat16(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = gelu_fast(x[idx]);
    }
}

torch::Tensor gelu_op_float(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int block_size = 256;
    // Use float4 for better memory bandwidth utilization
    int grid_size = (n + block_size * 4 - 1) / (block_size * 4);
    gelu_kernel_float4<<<grid_size, block_size>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    return y;
}

torch::Tensor gelu_op_half(torch::Tensor x) {
    // Convert half to float, process, convert back
    auto x_float = x.to(torch::kFloat);
    auto y = torch::empty_like(x_float);
    int n = x_float.numel();
    int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;
    gelu_kernel_half<<<grid_size, block_size>>>(x_float.data_ptr<float>(), y.data_ptr<float>(), n);
    return y.to(torch::kHalf);
}

torch::Tensor gelu_op_bfloat16(torch::Tensor x) {
    // Convert bf16 to float, process, convert back
    auto x_float = x.to(torch::kFloat);
    auto y = torch::empty_like(x_float);
    int n = x_float.numel();
    int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;
    gelu_kernel_bfloat16<<<grid_size, block_size>>>(x_float.data_ptr<float>(), y.data_ptr<float>(), n);
    return y.to(torch::kBFloat16);
}
'''

cpp_source = r'''
torch::Tensor gelu_op_float(torch::Tensor x);
torch::Tensor gelu_op_half(torch::Tensor x);
torch::Tensor gelu_op_bfloat16(torch::Tensor x);
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gelu_op_float', 'gelu_op_half', 'gelu_op_bfloat16'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        # Dispatch to appropriate kernel based on dtype
        if x.dtype == torch.float32:
            return ext.gelu_op_float(x)
        elif x.dtype == torch.float16:
            return ext.gelu_op_half(x)
        elif x.dtype == torch.bfloat16:
            return ext.gelu_op_bfloat16(x)
        else:
            # Fallback for unsupported dtypes
            return torch.nn.functional.gelu(x)

def get_inputs():
    return [torch.randn(256, 16384)]

def get_init_inputs():
    return []
