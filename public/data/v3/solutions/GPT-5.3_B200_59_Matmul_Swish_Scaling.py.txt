import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ inline float swish(float x) {
    return x / (1.0f + expf(-x));
}

__global__ void fused_kernel(
    const float* __restrict__ A,
    const float* __restrict__ W,
    const float* __restrict__ bias,
    float* __restrict__ C,
    int B, int K, int N,
    float scale
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < B && col < N) {
        float acc = 0.0f;
        for (int k = 0; k < K; ++k) {
            acc += A[row * K + k] * W[col * K + k];
        }
        acc += bias[col];
        acc = swish(acc);
        acc *= scale;
        C[row * N + col] = acc;
    }
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, float scale) {
    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);

    auto y = torch::empty({B, N}, x.options());

    dim3 block(16, 16);
    dim3 grid((N + 15)/16, (B + 15)/16);

    fused_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        B, K, N,
        scale
    );
    return y;
}
'''

ext = load_inline(
    name='fused_matmul_swish',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, float scale);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        w = self.matmul.weight
        b = self.matmul.bias
        return ext.fused_forward(x.contiguous(), w.contiguous(), b.contiguous(), float(self.scaling_factor))


def get_inputs():
    return [torch.randn(128, 4096, device='cuda', dtype=torch.float32)]


def get_init_inputs():
    return [4096, 4096, 2.0]
