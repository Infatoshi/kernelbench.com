import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# -----------------------------------------------------------------------
# Optimised depthwise conv2d CUDA kernels
#  - Uses tiled shared-memory loading
#  - float32, float16, bfloat16 specialisations
#  - Special-cased 3×3 kernel with compile-time unrolling
# -----------------------------------------------------------------------
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>

// ------------------------------------------------------------------
// Helper: depthwise conv, float32
//   - TH, TW: output tile height/width per block
//   - KH, KW: kernel size (compile-time for unrolling when K==3)
// ------------------------------------------------------------------
template<int KH, int KW, int TH, int TW>
__global__ void dw_conv_float_kt(
    const float* __restrict__ inp,
    const float* __restrict__ wgt,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int N, int C, int IH, int IW, int OH, int OW,
    int stride_h, int stride_w, int pad_h, int pad_w,
    bool has_bias)
{
    // shared memory holds input patch: (TH+KH-1) x (TW+KW-1)
    const int SH = TH + KH - 1;
    const int SW = TW + KW - 1;

    __shared__ float sm[( TH + KH - 1 ) * ( TW + KW - 1 )];

    int nc  = blockIdx.z;
    int n   = nc / C;
    int c   = nc % C;
    int oh0 = blockIdx.y * TH;
    int ow0 = blockIdx.x * TW;

    int ty = threadIdx.y;   // 0 .. TH-1
    int tx = threadIdx.x;   // 0 .. TW-1

    // --- load shared mem (collab, loop over tile) ---
    int total = SH * SW;
    int tid   = ty * TW + tx;
    const float* in_ptr = inp + ((long long)(n * C + c)) * IH * IW;

    for (int i = tid; i < total; i += TH * TW) {
        int si = i / SW;
        int sj = i % SW;
        int ih = oh0 * stride_h - pad_h + si;
        int iw = ow0 * stride_w - pad_w + sj;
        float v = 0.f;
        if ((unsigned)ih < (unsigned)IH && (unsigned)iw < (unsigned)IW)
            v = in_ptr[ih * IW + iw];
        sm[si * SW + sj] = v;
    }
    __syncthreads();

    int oh = oh0 + ty;
    int ow = ow0 + tx;
    if (oh >= OH || ow >= OW) return;

    const float* w = wgt + (long long)c * KH * KW;
    float acc = 0.f;
    #pragma unroll
    for (int kh = 0; kh < KH; ++kh) {
        #pragma unroll
        for (int kw = 0; kw < KW; ++kw) {
            acc += sm[(ty + kh) * SW + (tx + kw)] * w[kh * KW + kw];
        }
    }
    if (has_bias) acc += bias[c];

    out[((long long)(n * C + c)) * OH * OW + oh * OW + ow] = acc;
}

// ------------------------------------------------------------------
// Generic float32 kernel (dynamic K, shared mem via pointer cast)
// ------------------------------------------------------------------
__global__ void dw_conv_float_gen(
    const float* __restrict__ inp,
    const float* __restrict__ wgt,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int N, int C, int IH, int IW, int OH, int OW,
    int KH, int KW,
    int stride_h, int stride_w, int pad_h, int pad_w,
    int TH, int TW,
    bool has_bias)
{
    extern __shared__ char smem_raw[];
    float* sm = reinterpret_cast<float*>(smem_raw);

    int SH = TH + KH - 1;
    int SW = TW + KW - 1;

    int nc  = blockIdx.z;
    int n   = nc / C;
    int c   = nc % C;
    int oh0 = blockIdx.y * TH;
    int ow0 = blockIdx.x * TW;

    int ty = threadIdx.y;
    int tx = threadIdx.x;
    int total = SH * SW;
    int tid   = ty * TW + tx;

    const float* in_ptr = inp + (long long)(n * C + c) * IH * IW;
    for (int i = tid; i < total; i += TH * TW) {
        int si = i / SW;
        int sj = i % SW;
        int ih = oh0 * stride_h - pad_h + si;
        int iw = ow0 * stride_w - pad_w + sj;
        float v = 0.f;
        if ((unsigned)ih < (unsigned)IH && (unsigned)iw < (unsigned)IW)
            v = in_ptr[ih * IW + iw];
        sm[si * SW + sj] = v;
    }
    __syncthreads();

    int oh = oh0 + ty;
    int ow = ow0 + tx;
    if (oh >= OH || ow >= OW) return;

    const float* w = wgt + (long long)c * KH * KW;
    float acc = 0.f;
    for (int kh = 0; kh < KH; ++kh)
        for (int kw = 0; kw < KW; ++kw)
            acc += sm[(ty + kh) * SW + (tx + kw)] * w[kh * KW + kw];

    if (has_bias) acc += bias[c];
    out[(long long)(n * C + c) * OH * OW + oh * OW + ow] = acc;
}

// ------------------------------------------------------------------
// Generic float16 kernel
// ------------------------------------------------------------------
__global__ void dw_conv_half_gen(
    const __half* __restrict__ inp,
    const __half* __restrict__ wgt,
    const __half* __restrict__ bias,
    __half* __restrict__ out,
    int N, int C, int IH, int IW, int OH, int OW,
    int KH, int KW,
    int stride_h, int stride_w, int pad_h, int pad_w,
    int TH, int TW,
    bool has_bias)
{
    extern __shared__ char smem_raw[];
    __half* sm = reinterpret_cast<__half*>(smem_raw);

    int SH = TH + KH - 1;
    int SW = TW + KW - 1;

    int nc  = blockIdx.z;
    int n   = nc / C;
    int c   = nc % C;
    int oh0 = blockIdx.y * TH;
    int ow0 = blockIdx.x * TW;

    int ty = threadIdx.y;
    int tx = threadIdx.x;
    int total = SH * SW;
    int tid   = ty * TW + tx;

    const __half* in_ptr = inp + (long long)(n * C + c) * IH * IW;
    for (int i = tid; i < total; i += TH * TW) {
        int si = i / SW;
        int sj = i % SW;
        int ih = oh0 * stride_h - pad_h + si;
        int iw = ow0 * stride_w - pad_w + sj;
        __half v = __float2half(0.f);
        if ((unsigned)ih < (unsigned)IH && (unsigned)iw < (unsigned)IW)
            v = in_ptr[ih * IW + iw];
        sm[si * SW + sj] = v;
    }
    __syncthreads();

    int oh = oh0 + ty;
    int ow = ow0 + tx;
    if (oh >= OH || ow >= OW) return;

    const __half* w = wgt + (long long)c * KH * KW;
    float acc = 0.f;
    for (int kh = 0; kh < KH; ++kh)
        for (int kw = 0; kw < KW; ++kw)
            acc += __half2float(sm[(ty + kh) * SW + (tx + kw)]) * __half2float(w[kh * KW + kw]);

    if (has_bias) acc += __half2float(bias[c]);
    out[(long long)(n * C + c) * OH * OW + oh * OW + ow] = __float2half(acc);
}

    int SH = TH + KH - 1;

        __nv_bfloat16 v = __float2bfloat16(0.f);
        if ((unsigned)ih < (unsigned)IH && (unsigned)iw < (unsigned)IW)
            v = in_ptr[ih * IW + iw];

    }

    for (int kh = 0; kh < KH; ++kh)
        for (int kw = 0; kw < KW; ++kw)

    out[(long long)(n * C + c) * OH * OW + oh * OW + ow] = __float2bfloat16(acc);

// ------------------------------------------------------------------
// Dispatcher
// ------------------------------------------------------------------
torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor input,   // (N, C, IH, IW)
    torch::Tensor weight,  // (C, 1, KH, KW)  — depthwise
    torch::optional<torch::Tensor> bias_opt,
    int stride_h, int stride_w,
    int pad_h,    int pad_w)
{
    TORCH_CHECK(input.is_cuda(),  "input must be on CUDA");
    TORCH_CHECK(weight.is_cuda(), "weight must be on CUDA");
    TORCH_CHECK(input.is_contiguous(),  "input must be contiguous");
    TORCH_CHECK(weight.is_contiguous(), "weight must be contiguous");

    int N  = input.size(0);
    int C  = input.size(1);
    int IH = input.size(2);
    int IW = input.size(3);
    int KH = weight.size(2);
    int KW = weight.size(3);

    int OH = (IH + 2 * pad_h - KH) / stride_h + 1;
    int OW = (IW + 2 * pad_w - KW) / stride_w + 1;

    auto output = torch::empty({N, C, OH, OW}, input.options());

    bool has_bias = bias_opt.has_value() && bias_opt.value().defined();

    // Tile size
    const int TH = 16, TW = 16;

    dim3 block(TW, TH, 1);
    dim3 grid((OW + TW - 1) / TW,
              (OH + TH - 1) / TH,
              N * C);

    auto stream = at::cuda::getCurrentCUDAStream();

    if (input.scalar_type() == at::kFloat) {
        const float* b_ptr = has_bias ? bias_opt.value().data_ptr<float>() : nullptr;
        if (KH == 3 && KW == 3) {
            // compile-time unrolled 3x3
            dw_conv_float_kt<3,3,TH,TW><<<grid, block, 0, stream>>>(
                input.data_ptr<float>(),
                weight.data_ptr<float>(),
                b_ptr,
                output.data_ptr<float>(),
                N, C, IH, IW, OH, OW,
                stride_h, stride_w, pad_h, pad_w, has_bias);
        } else if (KH == 5 && KW == 5) {
            dw_conv_float_kt<5,5,TH,TW><<<grid, block, 0, stream>>>(
                input.data_ptr<float>(),
                weight.data_ptr<float>(),
                b_ptr,
                output.data_ptr<float>(),
                N, C, IH, IW, OH, OW,
                stride_h, stride_w, pad_h, pad_w, has_bias);
        } else {
            int SH = TH + KH - 1, SW = TW + KW - 1;
            size_t smem = (size_t)SH * SW * sizeof(float);
            dw_conv_float_gen<<<grid, block, smem, stream>>>(
                input.data_ptr<float>(),
                weight.data_ptr<float>(),
                b_ptr,
                output.data_ptr<float>(),
                N, C, IH, IW, OH, OW, KH, KW,
                stride_h, stride_w, pad_h, pad_w, TH, TW, has_bias);
        }
    } else if (input.scalar_type() == at::kHalf) {
        int SH = TH + KH - 1, SW = TW + KW - 1;
        size_t smem = (size_t)SH * SW * sizeof(__half);
        const __half* b_ptr = has_bias ?
            reinterpret_cast<const __half*>(bias_opt.value().data_ptr<at::Half>()) : nullptr;
        dw_conv_half_gen<<<grid, block, smem, stream>>>(
            reinterpret_cast<const __half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<const __half*>(weight.data_ptr<at::Half>()),
            b_ptr,
            reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
            N, C, IH, IW, OH, OW, KH, KW,
            stride_h, stride_w, pad_h, pad_w, TH, TW, has_bias);
    } else if (input.scalar_type() == at::kBFloat16) {
        int SH = TH + KH - 1, SW = TW + KW - 1;
        size_t smem = (size_t)SH * SW * sizeof(__nv_bfloat16);
        const __nv_bfloat16* b_ptr = has_bias ?
            reinterpret_cast<const __nv_bfloat16*>(bias_opt.value().data_ptr<at::BFloat16>()) : nullptr;
        dw_conv_bf16_gen<<<grid, block, smem, stream>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr<at::BFloat16>()),
            b_ptr,
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            N, C, IH, IW, OH, OW, KH, KW,
            stride_h, stride_w, pad_h, pad_w, TH, TW, has_bias);
    } else {
        TORCH_CHECK(false, "Unsupported dtype for depthwise_conv2d_cuda");
    }

    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias_opt,
    int stride_h, int stride_w,
    int pad_h,    int pad_w);
'''

ext = load_inline(
    name='depthwise_conv2d_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_cuda'],
    verbose=True,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
)


class Model(nn.Module):
    """
    Optimised depthwise 2D convolution (square input, square kernel).
    Compatible with reference.Model interface.
    """
    def __init__(self, in_channels: int, kernel_size: int,
                 stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        self.in_channels  = in_channels
        self.kernel_size  = kernel_size
        self.stride       = stride
        self.padding      = padding
        # Weight shape = (C, 1, K, K), same as nn.Conv2d with groups=C
        self.weight = nn.Parameter(
            torch.empty(in_channels, 1, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.zeros(in_channels))
        else:
            self.register_parameter('bias', None)
        nn.init.kaiming_uniform_(self.weight, a=0)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.contiguous()
        w = self.weight.contiguous()
        return ext.depthwise_conv2d_cuda(
            x, w,
            self.bias,
            self.stride, self.stride,
            self.padding, self.padding)

    # Override load_state_dict to map 'conv2d.weight' → 'weight'
    def load_state_dict(self, state_dict, strict=True):
        # Remap keys coming from reference Model (nn.Conv2d inside)
        new_sd = {}
        for k, v in state_dict.items():
            if k == 'conv2d.weight':
                new_sd['weight'] = v
            elif k == 'conv2d.bias':
                new_sd['bias'] = v
            else:
                new_sd[k] = v
        return super().load_state_dict(new_sd, strict=strict)
