import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _gemv_int8_dequant_kernel(
    x_ptr,
    w_ptr,
    y_ptr,
    x_scale_ptr,
    w_scale_ptr,
    K: tl.constexpr,
    N: tl.constexpr,
    stride_xb,
    stride_xk,
    stride_wn,
    stride_wk,
    stride_yb,
    stride_yn,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_b = tl.program_id(1)

    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)

    x_scale = tl.load(x_scale_ptr)
    w_scale = tl.load(w_scale_ptr)
    total_scale = x_scale * w_scale

    for k0 in range(0, K, BLOCK_K):
        k = k0 + offs_k
        x_mask = k < K
        x = tl.load(x_ptr + pid_b * stride_xb + k * stride_xk, mask=x_mask, other=0).to(tl.float32)

        w_ptrs = w_ptr + offs_n[:, None] * stride_wn + k[None, :] * stride_wk
        w_mask = (offs_n[:, None] < N) & (k[None, :] < K)
        w = tl.load(w_ptrs, mask=w_mask, other=0).to(tl.float32)

        acc += tl.sum(w * x[None, :], axis=1)

    acc *= total_scale
    y_ptrs = y_ptr + pid_b * stride_yb + offs_n * stride_yn
    tl.store(y_ptrs, acc.to(tl.float16), mask=offs_n < N)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        assert x_q.is_cuda and x_scale.is_cuda
        B, K = x_q.shape
        N = self.weight_q.shape[0]
        y = torch.empty((B, N), device=x_q.device, dtype=torch.float16)

        grid = (triton.cdiv(N, 128), B)
        _gemv_int8_dequant_kernel[grid](
            x_q,
            self.weight_q,
            y,
            x_scale,
            self.weight_scale,
            K=K,
            N=N,
            stride_xb=x_q.stride(0),
            stride_xk=x_q.stride(1),
            stride_wn=self.weight_q.stride(0),
            stride_wk=self.weight_q.stride(1),
            stride_yb=y.stride(0),
            stride_yn=y.stride(1),
            BLOCK_N=128,
            BLOCK_K=128,
            num_warps=4,
            num_stages=3,
        )
        return y


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
