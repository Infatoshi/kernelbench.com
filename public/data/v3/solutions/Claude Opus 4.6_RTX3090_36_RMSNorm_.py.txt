import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rms_norm_fused(
    X_ptr, Y_ptr,
    B, F: tl.constexpr, D1, D2,
    eps,
    BLOCK_D2: tl.constexpr,
):
    # Grid: (B * D1 * cdiv(D2, BLOCK_D2),)
    # Each program handles one (b, d1, d2_block) and loops over F features
    pid = tl.program_id(0)
    
    num_d2_blocks = tl.cdiv(D2, BLOCK_D2)
    bd1_idx = pid // num_d2_blocks
    d2_block = pid % num_d2_blocks
    
    b = bd1_idx // D1
    d1 = bd1_idx % D1
    
    d2_offs = d2_block * BLOCK_D2 + tl.arange(0, BLOCK_D2)
    d2_mask = d2_offs < D2
    
    stride_f = D1 * D2
    base = b * (F * stride_f) + d1 * D2 + d2_offs
    
    # Accumulate sum of squares
    acc = tl.zeros([BLOCK_D2], dtype=tl.float32)
    for f in range(F):
        ptr = base + f * stride_f
        x = tl.load(X_ptr + ptr, mask=d2_mask, other=0.0)
        x_f32 = x.to(tl.float32)
        acc += x_f32 * x_f32
    
    inv_rms = tl.rsqrt(acc / F + eps)
    
    for f in range(F):
        ptr = base + f * stride_f
        x = tl.load(X_ptr + ptr, mask=d2_mask, other=0.0)
        x_f32 = x.to(tl.float32)
        y = x_f32 * inv_rms
        tl.store(Y_ptr + ptr, y.to(x.dtype), mask=d2_mask)


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_contiguous()
        B, F, D1, D2 = x.shape
        y = torch.empty_like(x)
        
        BLOCK_D2 = min(D2, 1024)
        
        num_d2_blocks = triton.cdiv(D2, BLOCK_D2)
        grid = (B * D1 * num_d2_blocks,)
        
        rms_norm_fused[grid](
            x, y,
            B, F, D1, D2,
            self.eps,
            BLOCK_D2=BLOCK_D2,
        )
        
        return y


batch_size = 16
