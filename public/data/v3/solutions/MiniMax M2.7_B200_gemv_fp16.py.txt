"""
FP16 GEMV optimized for B200 using CUDA with warp-cooperative approach.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized warp-cooperative GEMV
template <const int WARP_ROWS>
__global__ void gemv_kernel_cooperative(
    const __half* __restrict__ x,
    const __half* __restrict__ w,
    __half* __restrict__ y,
    const int in_features,
    const int out_features
) {
    const int tid = threadIdx.x;
    const int lane = tid % 32;
    
    // Each block processes WARP_ROWS output rows
    const int block_row_start = blockIdx.x * WARP_ROWS;
    
    float results[WARP_ROWS] = {0.0f};
    
    // Process all elements with strided access
    for (int k = lane; k < in_features; k += 32) {
        __half x_val = __ldg(&x[k]);
        float x_f = __half2float(x_val);
        
        // Each warp computes WARP_ROWS output elements
        for (int r = 0; r < WARP_ROWS; r++) {
            int row = block_row_start + r;
            if (row < out_features) {
                __half w_val = __ldg(&w[row * in_features + k]);
                results[r] += x_f * __half2float(w_val);
            }
        }
    }
    
    // Warp reduction for each result
    for (int r = 0; r < WARP_ROWS; r++) {
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            results[r] += __shfl_down_sync(0xffffffff, results[r], offset);
        }
    }
    
    // Write results
    if (lane == 0) {
        for (int r = 0; r < WARP_ROWS; r++) {
            int row = block_row_start + r;
            if (row < out_features) {
                y[row] = __float2half(results[r]);
            }
        }
    }
}

torch::Tensor gemv_cuda(torch::Tensor x, torch::Tensor weight) {
    auto y = torch::empty({weight.size(0)}, torch::dtype(torch::kFloat16).device(torch::kCUDA));
    
    const int in_features = x.size(0);
    const int out_features = weight.size(0);
    
    const int BLOCK_SIZE = 256;
    const int WARP_SIZE = 32;
    const int NUM_WARPS = BLOCK_SIZE / WARP_SIZE;
    
    // Each warp processes WARP_ROWS output rows
    const int WARP_ROWS = 2;
    const int rows_per_block = NUM_WARPS * WARP_ROWS;
    const int num_blocks = (out_features + rows_per_block - 1) / rows_per_block;
    
    gemv_kernel_cooperative<WARP_ROWS><<<num_blocks, BLOCK_SIZE>>>(
        static_cast<const __half*>(x.data_ptr()),
        static_cast<const __half*>(weight.data_ptr()),
        static_cast<__half*>(y.data_ptr()),
        in_features,
        out_features
    );
    
    return y;
}

// Vectorized version for batch processing
template <const int WARP_ROWS>
__global__ void gemv_kernel_batch(
    const __half* __restrict__ x,     // [batch, in_features]
    const __half* __restrict__ w,     // [out_features, in_features]
    __half* __restrict__ y,           // [batch, out_features]
    const int in_features,
    const int out_features,
    const int batch
) {
    const int tid = threadIdx.x;
    const int lane = tid % 32;
    
    // Global thread maps to (batch_idx, output_row)
    int total_threads = gridDim.x * blockDim.x;
    int thread_id = blockIdx.x * blockDim.x + tid;
    
    // Each thread computes one output element
    for (int idx = thread_id; idx < batch * out_features; idx += total_threads) {
        int batch_idx = idx / out_features;
        int out_row = idx % out_features;
        
        float acc = 0.0f;
        
        for (int k = lane; k < in_features; k += 32) {
            __half x_val = __ldg(&x[batch_idx * in_features + k]);
            __half w_val = __ldg(&w[out_row * in_features + k]);
            acc += __half2float(x_val) * __half2float(w_val);
        }
        
        // Warp reduction
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            acc += __shfl_down_sync(0xffffffff, acc, offset);
        }
        
        if (lane == 0) {
            y[idx] = __float2half(acc);
        }
    }
}

torch::Tensor gemv_cuda_batch(torch::Tensor x, torch::Tensor weight) {
    // x: [batch, in_features]
    const int batch = x.size(0);
    const int in_features = x.size(1);
    const int out_features = weight.size(0);
    
    auto y = torch::empty({batch, out_features}, torch::dtype(torch::kFloat16).device(torch::kCUDA));
    
    const int BLOCK_SIZE = 256;
    const int total_elements = batch * out_features;
    const int num_blocks = (total_elements + BLOCK_SIZE - 1) / BLOCK_SIZE;
    
    gemv_kernel_batch<2><<<num_blocks, BLOCK_SIZE>>>(
        static_cast<const __half*>(x.data_ptr()),
        static_cast<const __half*>(weight.data_ptr()),
        static_cast<__half*>(y.data_ptr()),
        in_features,
        out_features,
        batch
    );
    
    return y;
}
'''

cpp_source = '''
torch::Tensor gemv_cuda(torch::Tensor x, torch::Tensor weight);
torch::Tensor gemv_cuda_batch(torch::Tensor x, torch::Tensor weight);
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_cuda', 'gemv_cuda_batch'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        
        if x_fp16.dim() == 1:
            return ext.gemv_cuda(x_fp16, self.weight)
        else:
            # Batch processing
            batch = x_fp16.size(0)
            if batch == 1:
                return ext.gemv_cuda(x_fp16[0], self.weight).unsqueeze(0)
            else:
                return ext.gemv_cuda_batch(x_fp16, self.weight)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
