"""
Optimized BF16 GEMV for H100 using Triton.
"""

import torch
import triton
import triton.language as tl


@triton.jit
def gemv_kernel(
    x_ptr,           # [batch_size, in_features]
    weight_ptr,      # [out_features, in_features]
    output_ptr,      # [batch_size, out_features]
    batch_size: tl.constexpr,
    in_features: tl.constexpr,
    out_features: tl.constexpr,
    stride_x: tl.constexpr,
    stride_w: tl.constexpr,
    stride_out: tl.constexpr,
    BLOCK_M: tl.constexpr,  # batch size block
    BLOCK_N: tl.constexpr,  # output block
):
    """
    Matrix-vector multiplication with batch processing.
    Each program instance handles BLOCK_M batches and BLOCK_N outputs.
    """
    # Program indices for batch and output
    batch_start = tl.program_id(0) * BLOCK_M
    output_start = tl.program_id(1) * BLOCK_N
    
    # Block-level indices
    batch_offsets = batch_start + tl.arange(0, BLOCK_M)
    output_offsets = output_start + tl.arange(0, BLOCK_N)
    
    # Create masks
    batch_mask = batch_offsets < batch_size
    output_mask = output_offsets < out_features
    
    # Accumulator for FP32 precision
    acc = tl.zeros([BLOCK_M, BLOCK_N], dtype=tl.float32)
    
    # Loop over input features with tiles
    for i in range(0, in_features, BLOCK_N):
        # Output offsets for this tile
        out_offs = output_offsets + i
        out_mask = output_mask & (out_offs < out_features)
        
        # Create weight tile: [BLOCK_M, BLOCK_N]
        # weight is [out_features, in_features], so we need to load transpose
        # weight[output_offsets, in_features_offsets]
        
        in_offs = i + tl.arange(0, BLOCK_N)
        in_mask = in_offs < in_features
        
        # Load x for this batch: [BLOCK_M, BLOCK_N]
        x_row_start = batch_offsets[:, None] * stride_x
        x_offs = x_row_start + in_offs[None, :]
        x = tl.load(x_ptr + x_offs, mask=batch_mask[:, None] & in_mask[None, :], other=0.0)
        x_fp32 = x.to(tl.float32)
        
        # Load weight tile: [BLOCK_N, BLOCK_N] -> transpose to [BLOCK_N, BLOCK_N]
        weight_col_start = output_offsets[:, None] * stride_w
        weight_offs = weight_col_start + in_offs[None, :]
        w = tl.load(weight_ptr + weight_offs, mask=output_mask[:, None] & in_mask[None, :], other=0.0)
        w_fp32 = w.to(tl.float32)
        
        # Matrix multiply: [BLOCK_M, BLOCK_N] = [BLOCK_M, BLOCK_N] * [BLOCK_N, BLOCK_N]
        acc = tl.dot(x_fp32, w_fp32, acc)
    
    # Convert back to bf16 and store
    acc_bf16 = acc.to(tl.bfloat16)
    
    # Store result
    out_row_start = batch_offsets[:, None] * stride_out
    store_offsets = out_row_start + output_offsets[None, :]
    tl.store(output_ptr + store_offsets, acc_bf16, mask=batch_mask[:, None] & output_mask[None, :])


def gemv_bf16(x: torch.Tensor, weight: torch.Tensor) -> torch.Tensor:
    """
    Optimized GEMV using Triton on H100.
    x: [batch_size, in_features], bf16
    weight: [out_features, in_features], bf16
    returns: [batch_size, out_features], bf16
    """
    batch_size, in_features = x.shape
    out_features = weight.shape[0]
    
    assert x.shape[1] == weight.shape[1], "Input and weight must have same in_features"
    
    # Output tensor
    output = torch.empty((batch_size, out_features), dtype=torch.bfloat16, device=x.device)
    
    # Determine block sizes based on shapes
    # For H100, use 64x64 blocks for good occupancy
    BLOCK_M = min(64, batch_size)
    BLOCK_N = 64
    if BLOCK_N > out_features:
        BLOCK_N = triton.next_power_of_2(out_features)
        if BLOCK_N > out_features:
            BLOCK_N = out_features
    
    # Grid dimensions
    grid_m = triton.cdiv(batch_size, BLOCK_M)
    grid_n = triton.cdiv(out_features, BLOCK_N)
    
    # Call kernel
    gemv_kernel[grid_m, grid_n](
        x, weight, output,
        batch_size, in_features, out_features,
        x.stride(0), weight.stride(0), output.stride(0),
        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,
    )
    
    return output


class Model(torch.nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = torch.nn.Parameter(
            torch.randn(out_features, in_features, dtype=torch.bfloat16, device='cuda') * 0.02
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Convert to bf16 if needed
        x_bf16 = x.to(dtype=torch.bfloat16)
        return gemv_bf16(x_bf16, self.weight)
