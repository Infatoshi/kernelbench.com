import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Swish activation using FP32 math for accuracy
__device__ __forceinline__ float sigmoidf(float x) {
    return 1.0f / (1.0f + expf(-x));
}

__device__ __forceinline__ float swishf(float x) {
    return x * sigmoidf(x);
}

// Optimized FP16 kernel with 2-column vectorization using half2
// weight is transposed: (K, N) instead of (N, K)
__global__ void optimized_fused_kernel_half2(
    const half* __restrict__ input,
    const half* __restrict__ weight_t,  // transposed: (K, N)
    half* __restrict__ output,
    float scale,
    int M, int N, int K) {
    
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    int col_base = (blockIdx.y * blockDim.y + threadIdx.y) * 2;
    
    if (row >= M || col_base >= N) return;
    
    // Process 2 elements at a time using half2
    half2 sum_h2 = __float2half2_rn(0.0f);
    
    // Unroll the K loop for better ILP
    int k = 0;
    
    // Process 8 half2 elements per iteration (16 half values)
    for (; k + 15 < K; k += 16) {
        // Load 8 pairs from input row
        half2 a0 = __half2half2(input[row * K + k]);
        half2 a1 = __half2half2(input[row * K + k + 2]);
        half2 a2 = __half2half2(input[row * K + k + 4]);
        half2 a3 = __half2half2(input[row * K + k + 6]);
        half2 a4 = __half2half2(input[row * K + k + 8]);
        half2 a5 = __half2half2(input[row * K + k + 10]);
        half2 a6 = __half2half2(input[row * K + k + 12]);
        half2 a7 = __half2half2(input[row * K + k + 14]);
        
        // Load 8 pairs from weight columns (transposed layout)
        half2 b0 = __halves2half2(weight_t[(k + 0) * N + col_base], weight_t[(k + 0) * N + col_base + 1]);
        half2 b1 = __halves2half2(weight_t[(k + 1) * N + col_base], weight_t[(k + 1) * N + col_base + 1]);
        half2 b2 = __halves2half2(weight_t[(k + 2) * N + col_base], weight_t[(k + 2) * N + col_base + 1]);
        half2 b3 = __halves2half2(weight_t[(k + 3) * N + col_base], weight_t[(k + 3) * N + col_base + 1]);
        half2 b4 = __halves2half2(weight_t[(k + 4) * N + col_base], weight_t[(k + 4) * N + col_base + 1]);
        half2 b5 = __halves2half2(weight_t[(k + 5) * N + col_base], weight_t[(k + 5) * N + col_base + 1]);
        half2 b6 = __halves2half2(weight_t[(k + 6) * N + col_base], weight_t[(k + 6) * N + col_base + 1]);
        half2 b7 = __halves2half2(weight_t[(k + 7) * N + col_base], weight_t[(k + 7) * N + col_base + 1]);
        
        // Multiply-add
        sum_h2 = __hfma2(a0, b0, sum_h2);
        sum_h2 = __hfma2(a1, b1, sum_h2);
        sum_h2 = __hfma2(a2, b2, sum_h2);
        sum_h2 = __hfma2(a3, b3, sum_h2);
        sum_h2 = __hfma2(a4, b4, sum_h2);
        sum_h2 = __hfma2(a5, b5, sum_h2);
        sum_h2 = __hfma2(a6, b6, sum_h2);
        sum_h2 = __hfma2(a7, b7, sum_h2);
    }
    
    // Handle remaining elements
    for (; k < K; k++) {
        half2 a_val = __half2half2(input[row * K + k]);
        half2 b_val = __halves2half2(weight_t[k * N + col_base], weight_t[k * N + col_base + 1]);
        sum_h2 = __hfma2(a_val, b_val, sum_h2);
    }
    
    // Apply Swish + Scale using FP32 for accuracy
    float2 sum_f = __half22float2(sum_h2);
    
    float swish0 = swishf(sum_f.x) * scale;
    float swish1 = swishf(sum_f.y) * scale;
    
    // Store results
    if (col_base + 1 < N) {
        output[row * N + col_base] = __float2half_rn(swish0);
        output[row * N + col_base + 1] = __float2half_rn(swish1);
    } else if (col_base < N) {
        output[row * N + col_base] = __float2half_rn(swish0);
    }
}

// Optimized FP32 kernel
// weight is transposed: (K, N) instead of (N, K)
__global__ void optimized_fused_kernel_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight_t,  // transposed: (K, N)
    float* __restrict__ output,
    float scale,
    int M, int N, int K) {
    
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    int col = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    // Unroll by 8 for better ILP
    int k = 0;
    for (; k + 7 < K; k += 8) {
        sum += input[row * K + k] * weight_t[(k + 0) * N + col];
        sum += input[row * K + k + 1] * weight_t[(k + 1) * N + col];
        sum += input[row * K + k + 2] * weight_t[(k + 2) * N + col];
        sum += input[row * K + k + 3] * weight_t[(k + 3) * N + col];
        sum += input[row * K + k + 4] * weight_t[(k + 4) * N + col];
        sum += input[row * K + k + 5] * weight_t[(k + 5) * N + col];
        sum += input[row * K + k + 6] * weight_t[(k + 6) * N + col];
        sum += input[row * K + k + 7] * weight_t[(k + 7) * N + col];
    }
    
    for (; k < K; k++) {
        sum += input[row * K + k] * weight_t[k * N + col];
    }
    
    // Apply Swish + Scale
    output[row * N + col] = swishf(sum) * scale;
}

// Launch wrapper
void launch_fused_matmul_swish_scale(
    torch::Tensor input,
    torch::Tensor weight_t,  // transposed weight (K, N)
    torch::Tensor output,
    float scale,
    int M, int N, int K) {
    
    const at::ScalarType dtype = input.scalar_type();
    
    if (dtype == at::kHalf) {
        // Use 16x8 blocks for half2 vectorization
        dim3 block(16, 8);
        dim3 grid((M + block.x - 1) / block.x, (N / 2 + block.y - 1) / block.y);
        optimized_fused_kernel_half2<<<grid, block>>>(
            reinterpret_cast<const half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(weight_t.data_ptr<at::Half>()),
            reinterpret_cast<half*>(output.data_ptr<at::Half>()),
            scale, M, N, K);
    } else if (dtype == at::kFloat) {
        dim3 block(16, 16);
        dim3 grid((M + block.x - 1) / block.x, (N + block.y - 1) / block.y);
        optimized_fused_kernel_fp32<<<grid, block>>>(
            input.data_ptr<float>(),
            weight_t.data_ptr<float>(),
            output.data_ptr<float>(),
            scale, M, N, K);
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\\n", cudaGetErrorString(err));
    }
}

torch::Tensor fused_matmul_swish_scale(
    torch::Tensor input,
    torch::Tensor weight_t,  // transposed: (K, N)
    float scale) {
    
    int M = input.size(0);
    int K = input.size(1);
    int N = weight_t.size(0);  // After transpose, first dim is K
    
    auto output = torch::empty({M, N}, input.options());
    
    launch_fused_matmul_swish_scale(input, weight_t, output, scale, M, N, K);
    cudaDeviceSynchronize();
    
    return output;
}
'''

ext = load_inline(
    name='fused_matmul_ext',
    cpp_sources='''
    torch::Tensor fused_matmul_swish_scale(
        torch::Tensor input,
        torch::Tensor weight_t,
        float scale);
    ''',
    cuda_sources=cuda_source,
    functions=['fused_matmul_swish_scale'],
    extra_cuda_cflags=['-gencode=arch=compute_86,code=sm_86', '-std=c++17'],
    verbose=False
)

class Model(nn.Module):
    """
    Optimized fused model: Matmul + Swish + Scale using custom CUDA kernel
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        # Use the same structure as nn.Linear for compatibility
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
        
    def forward(self, x):
        input_dtype = x.dtype
        
        # Get weight and bias
        weight = self.matmul.weight  # Shape: (out_features, in_features)
        bias = self.matmul.bias
        
        # Transpose weight to (in_features, out_features) for x @ W.T
        weight_t = weight.t()  # Shape: (in_features, out_features)
        
        # Work with FP16 or FP32 based on input type
        if input_dtype == torch.float16:
            x = x.half()
            weight_t = weight_t.half()
        elif input_dtype == torch.float32:
            weight_t = weight_t.float()
        else:
            # Default to FP32
            x = x.float()
            weight_t = weight_t.float()
        
        # Custom fused kernel for matmul
        output = ext.fused_matmul_swish_scale(x, weight_t, self.scaling_factor)
        
        # Add bias (PyTorch Linear adds bias before activation, but reference adds after)
        # Reference: x = self.matmul(x) -> x * sigmoid(x) -> x * scale
        # So bias should be added after matmul
        if bias is not None:
            output = output + bias.to(output.dtype)
        
        return output.to(input_dtype)


def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 2.0]
