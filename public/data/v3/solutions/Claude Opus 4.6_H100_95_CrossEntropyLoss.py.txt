import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def cross_entropy_kernel(
    logits_ptr,      # [batch_size, num_classes]
    targets_ptr,     # [batch_size]
    losses_ptr,      # [batch_size]
    num_classes: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0)
    row_start = logits_ptr + row_idx * num_classes
    target = tl.load(targets_ptr + row_idx)
    
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < num_classes
    x = tl.load(row_start + cols, mask=mask, other=float('-inf'))
    
    m = tl.max(x, axis=0)
    sum_exp = tl.sum(tl.exp(x - m), axis=0)
    log_sum_exp = m + tl.log(sum_exp)
    target_logit = tl.load(row_start + target)
    loss = log_sum_exp - target_logit
    
    tl.store(losses_ptr + row_idx, loss)


@triton.jit
def reduce_mean_kernel(
    input_ptr,
    output_ptr,
    n: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Single program reduces all elements
    acc = 0.0
    for offset in tl.static_range(0, n, BLOCK_SIZE):
        cols = offset + tl.arange(0, BLOCK_SIZE)
        mask = cols < n
        x = tl.load(input_ptr + cols, mask=mask, other=0.0)
        acc += tl.sum(x, axis=0)
    tl.store(output_ptr, acc / n)


# Pre-allocate buffers
_losses_buf = None
_output_buf = None


def cross_entropy_triton(predictions, targets):
    global _losses_buf, _output_buf
    batch_size, num_classes = predictions.shape
    
    if _losses_buf is None or _losses_buf.shape[0] != batch_size:
        _losses_buf = torch.empty(batch_size, device=predictions.device, dtype=predictions.dtype)
    if _output_buf is None:
        _output_buf = torch.empty(1, device=predictions.device, dtype=predictions.dtype)
    
    BLOCK_SIZE = 1024
    
    cross_entropy_kernel[(batch_size,)](
        predictions, targets, _losses_buf,
        num_classes=num_classes,
        BLOCK_SIZE=BLOCK_SIZE,
        num_warps=8,
    )
    
    reduce_mean_kernel[(1,)](
        _losses_buf, _output_buf,
        n=batch_size,
        BLOCK_SIZE=4096,
        num_warps=8,
    )
    
    return _output_buf[0]


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return cross_entropy_triton(predictions, targets)
