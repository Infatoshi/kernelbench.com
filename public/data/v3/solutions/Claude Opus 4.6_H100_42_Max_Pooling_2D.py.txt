import torch
import torch.nn as nn
import triton
import triton.language as tl

# Parameters from the problem
batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3


@triton.jit
def maxpool2d_kernel(
    input_ptr, output_ptr,
    batch_channels,
    H: tl.constexpr, W: tl.constexpr,
    OH: tl.constexpr, OW: tl.constexpr,
    kernel_size: tl.constexpr,
    stride: tl.constexpr,
    padding: tl.constexpr,
    dilation: tl.constexpr,
    HW: tl.constexpr,
    OHOW: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    total_out = batch_channels * OHOW
    mask = offsets < total_out
    
    # Decompose linear index into (bc, oh, ow)
    ow = offsets % OW
    temp = offsets // OW
    oh = temp % OH
    bc = temp // OH
    
    # Compute input base coordinates
    ih_base = oh * stride - padding
    iw_base = ow * stride - padding
    
    # Input base offset for this batch*channel
    input_base = bc * HW
    
    # Initialize with -inf
    max_val = tl.full([BLOCK_SIZE], value=float('-inf'), dtype=tl.float32)
    
    # Unrolled loop over kernel_size x kernel_size
    for kh in tl.static_range(kernel_size):
        for kw in tl.static_range(kernel_size):
            ih = ih_base + kh * dilation
            iw = iw_base + kw * dilation
            
            valid = (ih >= 0) & (ih < H) & (iw >= 0) & (iw < W) & mask
            
            idx = input_base + ih * W + iw
            val = tl.load(input_ptr + idx, mask=valid, other=float('-inf'))
            val_f32 = val.to(tl.float32)
            max_val = tl.maximum(max_val, val_f32)
    
    # Store output
    tl.store(output_ptr + offsets, max_val, mask=mask)


def maxpool2d_triton(x, kernel_size, stride, padding, dilation):
    B, C, H, W = x.shape
    OH = (H + 2 * padding - dilation * (kernel_size - 1) - 1) // stride + 1
    OW = (W + 2 * padding - dilation * (kernel_size - 1) - 1) // stride + 1
    
    output = torch.empty(B, C, OH, OW, device=x.device, dtype=x.dtype)
    
    batch_channels = B * C
    total_elements = batch_channels * OH * OW
    
    BLOCK_SIZE = 1024
    grid = ((total_elements + BLOCK_SIZE - 1) // BLOCK_SIZE,)
    
    maxpool2d_kernel[grid](
        x, output,
        batch_channels,
        H, W,
        OH, OW,
        kernel_size, stride, padding, dilation,
        H * W, OH * OW,
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return output


class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return maxpool2d_triton(x, self.kernel_size, self.stride, self.padding, self.dilation)


def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
