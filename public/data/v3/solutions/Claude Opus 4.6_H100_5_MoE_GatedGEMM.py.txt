import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def silu_mul_kernel(
    gate_ptr, up_ptr, out_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    offsets = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    gate = tl.load(gate_ptr + offsets, mask=mask).to(tl.float32)
    up = tl.load(up_ptr + offsets, mask=mask).to(tl.float32)
    silu_gate = gate * tl.sigmoid(gate)
    result = silu_gate * up
    tl.store(out_ptr + offsets, result, mask=mask)


def silu_mul(gate, up):
    out = torch.empty_like(gate)
    n = gate.numel()
    BLOCK = 1024
    grid = ((n + BLOCK - 1) // BLOCK,)
    silu_mul_kernel[grid](gate, up, out, n, BLOCK_SIZE=BLOCK)
    return out


class Model(nn.Module):
    def __init__(self, hidden_size, intermediate_size, num_experts):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.gate_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.up_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.down_proj = nn.Parameter(
            torch.randn(num_experts, hidden_size, intermediate_size) * 0.02
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x, expert_indices, expert_weights):
        batch, seq_len, _ = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len

        x_flat = x.view(num_tokens, self.hidden_size)
        indices_flat = expert_indices.view(num_tokens * top_k)
        weights_flat = expert_weights.view(num_tokens * top_k)

        # Create token indices
        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1)

        # Sort by expert
        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]

        # Expert boundaries
        expert_counts = torch.bincount(sorted_expert_idx, minlength=self.num_experts)
        max_tokens = expert_counts.max().item()
        expert_offsets = torch.zeros(self.num_experts + 1, dtype=torch.long, device=x.device)
        expert_offsets[1:] = expert_counts.cumsum(0)

        # Gather sorted inputs
        sorted_x = x_flat[sorted_token_ids]

        # Build padded index tensor for vectorized packing
        # Create a (num_experts, max_tokens) index array
        # For expert e, positions 0..count[e]-1 map to sorted_x[offset[e]..offset[e]+count[e]-1]
        # We need this to avoid Python loop
        
        # Use vectorized approach: create position within each expert group
        positions_in_expert = torch.arange(sorted_x.shape[0], device=x.device)
        positions_in_expert = positions_in_expert - expert_offsets[sorted_expert_idx]
        
        # Create padded tensor using advanced indexing
        padded_x = torch.zeros(self.num_experts, max_tokens, self.hidden_size,
                              device=x.device, dtype=x.dtype)
        padded_x[sorted_expert_idx, positions_in_expert] = sorted_x

        # Phase 1: gate+up projections via bmm
        # padded_x: (E, max_tok, H), gate_proj.T: (E, H, I)
        gate_result = torch.bmm(padded_x, self.gate_proj.transpose(1, 2))
        up_result = torch.bmm(padded_x, self.up_proj.transpose(1, 2))
        
        # Fused SiLU * up
        intermediate = silu_mul(gate_result, up_result)
        
        # Phase 2: down projection
        down_result = torch.bmm(intermediate, self.down_proj.transpose(1, 2))
        
        # Extract results back using vectorized indexing
        sorted_output = down_result[sorted_expert_idx, positions_in_expert]
        
        # Apply weights and scatter back
        weighted_sorted = sorted_output * sorted_weights.unsqueeze(-1)
        output = torch.zeros(num_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
        output.index_add_(0, sorted_token_ids, weighted_sorted)

        return output.view(batch, seq_len, self.hidden_size)


batch_size = 4
seq_len = 2048
hidden_size = 4096
intermediate_size = 14336
num_experts = 8
top_k = 2


def get_inputs():
    x = torch.randn(batch_size, seq_len, hidden_size)
    expert_indices = torch.stack([
        torch.randperm(num_experts)[:top_k]
        for _ in range(batch_size * seq_len)
    ]).view(batch_size, seq_len, top_k)
    expert_weights = F.softmax(torch.randn(batch_size, seq_len, top_k), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return [hidden_size, intermediate_size, num_experts]
