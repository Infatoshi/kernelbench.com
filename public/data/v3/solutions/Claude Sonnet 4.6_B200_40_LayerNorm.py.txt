import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized LayerNorm for large normalized_shape (C=4M)
# Problem: 16 rows, each with 4M float32 elements
# With 16 CTAs, GPU is severely underutilized
# Solution: use multiple CTAs per row + atomic reduction

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cooperative_groups.h>
#include <stdint.h>

namespace cg = cooperative_groups;

// Warp reduce sum
__device__ __forceinline__ float warpReduceSum(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, mask);
    return val;
}

// Block reduce sum
__device__ __forceinline__ float blockReduceSum(float val, float* smem, int tid, int blockSize) {
    int lane = tid & 31;
    int wid  = tid >> 5;
    val = warpReduceSum(val);
    if (lane == 0) smem[wid] = val;
    __syncthreads();
    int numWarps = (blockSize + 31) >> 5;
    float v = (tid < numWarps) ? smem[tid] : 0.0f;
    if (wid == 0) v = warpReduceSum(v);
    return v;
}

// Phase 1: each block computes partial sum and sum_sq for its chunk
// Stores into workspace array of size [N * blocksPerRow, 2]
__global__ void layernorm_partial_stats(
    const float* __restrict__ x,
    float* __restrict__ workspace,  // [N * blocksPerRow, 2]: [sum, sum_sq]
    int* __restrict__ counters,     // [N]: atomic counters for synchronization
    float* __restrict__ row_mean,   // [N]
    float* __restrict__ row_istd,   // [N]
    int N,
    int C,
    int blocksPerRow,
    float eps
) {
    int row = blockIdx.x / blocksPerRow;
    int chunk = blockIdx.x % blocksPerRow;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    if (row >= N) return;
    
    int chunkSize = (C + blocksPerRow - 1) / blocksPerRow;
    int start = chunk * chunkSize;
    int end = min(start + chunkSize, C);
    
    const float* x_row = x + (long)row * C;
    
    // Vectorized load
    float sum = 0.0f, sum_sq = 0.0f;
    int C_aligned = (end - start) & ~3;
    const float4* x4 = reinterpret_cast<const float4*>(x_row + start);
    int n4 = (end - start) >> 2;
    
    for (int i = tid; i < n4; i += blockSize) {
        float4 v = __ldg(x4 + i);
        sum += v.x + v.y + v.z + v.w;
        sum_sq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
    }
    for (int i = start + n4*4 + tid; i < end; i += blockSize) {
        float v = __ldg(x_row + i);
        sum += v;
        sum_sq += v * v;
    }
    
    extern __shared__ float smem[];
    float* smem1 = smem;
    float* smem2 = smem + ((blockSize + 31) >> 5);
    
    sum    = blockReduceSum(sum,    smem1, tid, blockSize);
    __syncthreads();
    sum_sq = blockReduceSum(sum_sq, smem2, tid, blockSize);
    
    if (tid == 0) {
        int ws_idx = row * blocksPerRow + chunk;
        workspace[ws_idx * 2 + 0] = sum;
        workspace[ws_idx * 2 + 1] = sum_sq;
        __threadfence();
        // Increment counter atomically; last block to finish computes mean/var
        int old = atomicAdd(&counters[row], 1);
        if (old == blocksPerRow - 1) {
            // I'm the last block for this row - compute mean and var
            float total_sum = 0.0f, total_sq = 0.0f;
            for (int b = 0; b < blocksPerRow; b++) {
                total_sum += workspace[(row * blocksPerRow + b) * 2 + 0];
                total_sq  += workspace[(row * blocksPerRow + b) * 2 + 1];
            }
            float mean = total_sum / C;
            float var  = total_sq  / C - mean * mean;
            row_mean[row] = mean;
            row_istd[row] = rsqrtf(var + eps);
        }
    }
}

// Phase 2: normalize using precomputed mean and istd
__global__ void layernorm_normalize(
    const float* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ out,
    const float* __restrict__ row_mean,
    const float* __restrict__ row_istd,
    int N,
    int C,
    int blocksPerRow
) {
    int row = blockIdx.x / blocksPerRow;
    int chunk = blockIdx.x % blocksPerRow;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    if (row >= N) return;
    
    float mean = row_mean[row];
    float istd = row_istd[row];
    
    int chunkSize = (C + blocksPerRow - 1) / blocksPerRow;
    int start = chunk * chunkSize;
    int end = min(start + chunkSize, C);
    
    const float* x_row = x + (long)row * C;
    float* out_row = out + (long)row * C;
    
    int n4 = (end - start) >> 2;
    const float4* x4 = reinterpret_cast<const float4*>(x_row + start);
    const float4* w4 = reinterpret_cast<const float4*>(weight + start);
    const float4* b4 = reinterpret_cast<const float4*>(bias + start);
    float4* o4 = reinterpret_cast<float4*>(out_row + start);
    
    for (int i = tid; i < n4; i += blockSize) {
        float4 v = __ldg(x4 + i);
        float4 w = __ldg(w4 + i);
        float4 b = __ldg(b4 + i);
        float4 r;
        r.x = (v.x - mean) * istd * w.x + b.x;
        r.y = (v.y - mean) * istd * w.y + b.y;
        r.z = (v.z - mean) * istd * w.z + b.z;
        r.w = (v.w - mean) * istd * w.w + b.w;
        o4[i] = r;
    }
    for (int i = start + n4*4 + tid; i < end; i += blockSize) {
        float v = __ldg(x_row + i);
        out_row[i] = (v - mean) * istd * weight[i] + bias[i];
    }
}

// Workspace buffers (persistent allocation)
static float* g_workspace = nullptr;
static int*   g_counters  = nullptr;
static float* g_mean      = nullptr;
static float* g_istd      = nullptr;
static int    g_workspace_N = 0;
static int    g_workspace_B = 0;

void ensure_workspace(int N, int blocksPerRow) {
    if (N > g_workspace_N || blocksPerRow > g_workspace_B) {
        if (g_workspace) cudaFree(g_workspace);
        if (g_counters)  cudaFree(g_counters);
        if (g_mean)      cudaFree(g_mean);
        if (g_istd)      cudaFree(g_istd);
        cudaMalloc(&g_workspace, (long)N * blocksPerRow * 2 * sizeof(float));
        cudaMalloc(&g_counters,  N * sizeof(int));
        cudaMalloc(&g_mean,      N * sizeof(float));
        cudaMalloc(&g_istd,      N * sizeof(float));
        g_workspace_N = N;
        g_workspace_B = blocksPerRow;
    }
    cudaMemset(g_counters, 0, N * sizeof(int));
}

torch::Tensor layernorm_forward(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float eps
) {
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "x must be float32");
    
    auto out = torch::empty_like(x);
    
    int C = weight.numel();
    int N = x.numel() / C;
    
    // Choose blocksPerRow to saturate GPU
    // B200 has ~160 SMs; with N=16 rows, we need blocksPerRow such that
    // N * blocksPerRow >= 160 * some_factor
    // Use 128 blocks per row -> 16*128 = 2048 total blocks
    int blocksPerRow = 128;
    // Make C divisible by 4 per chunk (round up chunkSize to multiple of 4)
    // Each chunk: ~32768 elements processed by 1024 threads = 32 iterations
    
    int blockSize = 1024;
    int numWarps = blockSize / 32;
    int smem_size = 2 * numWarps * sizeof(float);
    
    ensure_workspace(N, blocksPerRow);
    
    int totalBlocks = N * blocksPerRow;
    
    layernorm_partial_stats<<<totalBlocks, blockSize, smem_size>>>(
        x.data_ptr<float>(),
        g_workspace,
        g_counters,
        g_mean,
        g_istd,
        N, C, blocksPerRow, eps
    );
    
    layernorm_normalize<<<totalBlocks, blockSize>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        g_mean,
        g_istd,
        N, C, blocksPerRow
    );
    
    return out;
}
'''

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float eps);
'''

ext = load_inline(
    name='layernorm_ext2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        self.eps = 1e-5
        normalized_size = 1
        for s in normalized_shape:
            normalized_size *= s
        self.weight = nn.Parameter(torch.ones(normalized_size))
        self.bias = nn.Parameter(torch.zeros(normalized_size))
        self._normalized_size = normalized_size

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        orig_shape = x.shape
        N = x.numel() // self._normalized_size
        x_flat = x.contiguous().view(N, self._normalized_size)
        out = ext.layernorm_forward(x_flat, self.weight, self.bias, self.eps)
        return out.view(orig_shape)

    def load_state_dict(self, state_dict, strict=True):
        new_state_dict = {}
        for k, v in state_dict.items():
            if k == 'ln.weight':
                new_state_dict['weight'] = v.view(-1)
            elif k == 'ln.bias':
                new_state_dict['bias'] = v.view(-1)
            else:
                new_state_dict[k] = v
        return super().load_state_dict(new_state_dict, strict=strict)


def get_inputs():
    x = torch.randn(16, 64, 256, 256)
    return [x]


def get_init_inputs():
    return [(64, 256, 256)]
