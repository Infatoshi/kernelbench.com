import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cmath>

#define M_SQRT2_OVER_2 0.70710678118654752440

template <typename T>
__device__ __forceinline__ T gelu_impl(T x) {
    float fx = (float)x;
    return (T)(fx * 0.5f * (1.0f + erff(fx * (float)M_SQRT2_OVER_2)));
}

template <typename T, int VecSize, typename VecType>
__global__ void gelu_kernel_vec(const VecType* __restrict__ x, VecType* __restrict__ y, int n_vec) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n_vec) {
        VecType val = x[idx];
        T* ptr = (T*)&val;
        #pragma unroll
        for (int i = 0; i < VecSize; ++i) {
            ptr[i] = gelu_impl(ptr[i]);
        }
        y[idx] = val;
    }
}

template <typename T>
__global__ void gelu_tail_kernel(const T* __restrict__ x, T* __restrict__ y, int start, int end) {
    int idx = start + blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < end) {
        y[idx] = gelu_impl(x[idx]);
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int threads = 256;

    if (x.scalar_type() == torch::kFloat32) {
        int vec_size = 4;
        int n_vec = n / vec_size;
        if (n_vec > 0) {
            int blocks = (n_vec + threads - 1) / threads;
            gelu_kernel_vec<float, 4, float4><<<blocks, threads>>>(
                (const float4*)x.data_ptr(), (float4*)y.data_ptr(), n_vec
            );
        }
        int tail_start = n_vec * vec_size;
        if (tail_start < n) {
            int tail_n = n - tail_start;
            int blocks = (tail_n + threads - 1) / threads;
            gelu_tail_kernel<float><<<blocks, threads>>>(
                (const float*)x.data_ptr(), (float*)y.data_ptr(), tail_start, n
            );
        }
    } else if (x.scalar_type() == torch::kFloat16) {
        int vec_size = 8;
        int n_vec = n / vec_size;
        if (n_vec > 0) {
            int blocks = (n_vec + threads - 1) / threads;
            gelu_kernel_vec<at::Half, 8, uint4><<<blocks, threads>>>(
                (const uint4*)x.data_ptr(), (uint4*)y.data_ptr(), n_vec
            );
        }
        int tail_start = n_vec * vec_size;
        if (tail_start < n) {
            int tail_n = n - tail_start;
            int blocks = (tail_n + threads - 1) / threads;
            gelu_tail_kernel<at::Half><<<blocks, threads>>>(
                (const at::Half*)x.data_ptr(), (at::Half*)y.data_ptr(), tail_start, n
            );
        }
    } else if (x.scalar_type() == torch::kBFloat16) {
        int vec_size = 8;
        int n_vec = n / vec_size;
        if (n_vec > 0) {
            int blocks = (n_vec + threads - 1) / threads;
            gelu_kernel_vec<at::BFloat16, 8, uint4><<<blocks, threads>>>(
                (const uint4*)x.data_ptr(), (uint4*)y.data_ptr(), n_vec
            );
        }
        int tail_start = n_vec * vec_size;
        if (tail_start < n) {
            int tail_n = n - tail_start;
            int blocks = (tail_n + threads - 1) / threads;
            gelu_tail_kernel<at::BFloat16><<<blocks, threads>>>(
                (const at::BFloat16*)x.data_ptr(), (at::BFloat16*)y.data_ptr(), tail_start, n
            );
        }
    }
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    with_cuda=True,
    extra_cuda_cflags=['-O3', '-arch=sm_86']
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_op(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
