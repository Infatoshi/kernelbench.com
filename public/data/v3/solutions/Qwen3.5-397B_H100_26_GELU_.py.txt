import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// GELU approximation using tanh: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
// sqrt(2/pi) ≈ 0.7978845608

__global__ void gelu_kernel_fp32(const float* x, float* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float xi = x[idx];
        float x3 = xi * xi * xi;
        float inner = 0.7978845608f * (xi + 0.044715f * x3);
        float tanh_inner = tanhf(inner);
        y[idx] = 0.5f * xi * (1.0f + tanh_inner);
    }
}

__global__ void gelu_kernel_fp16(const c10::Half* x, c10::Half* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        c10::Half xi = x[idx];
        float xf = __half2float(static_cast<__half>(xi));
        float x3 = xf * xf * xf;
        float inner = 0.7978845608f * (xf + 0.044715f * x3);
        float tanh_inner = tanhf(inner);
        float result = 0.5f * xf * (1.0f + tanh_inner);
        y[idx] = c10::Half(__float2half(result));
    }
}

__global__ void gelu_kernel_bf16(const c10::BFloat16* x, c10::BFloat16* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        c10::BFloat16 xi = x[idx];
        float xf = __bfloat162float(static_cast<__nv_bfloat16>(xi));
        float x3 = xf * xf * xf;
        float inner = 0.7978845608f * (xf + 0.044715f * x3);
        float tanh_inner = tanhf(inner);
        float result = 0.5f * xf * (1.0f + tanh_inner);
        y[idx] = c10::BFloat16(__float2bfloat16(result));
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int threads = 512;
    int blocks = (n + threads - 1) / threads;
    
    if (x.dtype() == torch::kFloat32) {
        gelu_kernel_fp32<<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    } else if (x.dtype() == torch::kFloat16) {
        gelu_kernel_fp16<<<blocks, threads>>>(x.data_ptr<c10::Half>(), y.data_ptr<c10::Half>(), n);
    } else if (x.dtype() == torch::kBFloat16) {
        gelu_kernel_bf16<<<blocks, threads>>>(x.data_ptr<c10::BFloat16>(), y.data_ptr<c10::BFloat16>(), n);
    }
    
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_op(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
