import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized CUDA kernel for fused activation: x * tanh(softplus(x))
# Uses better memory access patterns
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fast sigmoid using expf
__device__ __forceinline__ float sigmoid_fast(float x) {
    return 1.0f / (1.0f + expf(-x));
}

// Softplus with fast math
__device__ __forceinline__ float softplus_fast(float x) {
    return log1pf(expf(x));
}

// Tanh using fast math
__device__ __forceinline__ float tanh_fast(float x) {
    return tanhf(x);
}

// Activation: x * tanh(softplus(x))
__device__ __forceinline__ float activation_fused(float x) {
    float sp = softplus_fast(x);
    return x * tanh_fast(sp);
}

// Float32 kernel
__global__ void activation_kernel_fp32(const float* __restrict__ input, 
                                         float* __restrict__ output, 
                                         int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        output[idx] = activation_fused(input[idx]);
    }
}

// Half precision kernel - convert to float, compute, convert back
__global__ void activation_kernel_fp16(const half* __restrict__ input, 
                                         half* __restrict__ output, 
                                         int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x = __half2float(input[idx]);
        float result = activation_fused(x);
        output[idx] = __float2half(result);
    }
}

// BFloat16 kernel - convert to float, compute, convert back
__global__ void activation_kernel_bf16(const at::BFloat16* __restrict__ input, 
                                         at::BFloat16* __restrict__ output, 
                                         int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x = float(input[idx]);
        float result = activation_fused(x);
        output[idx] = at::BFloat16(result);
    }
}

torch::Tensor activation_fp32(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n = input.numel();
    
    const int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;
    
    activation_kernel_fp32<<<grid_size, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        n
    );
    
    return output;
}

torch::Tensor activation_fp16(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n = input.numel();
    
    const int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;
    
    activation_kernel_fp16<<<grid_size, block_size>>>(
        input.data_ptr<at::Half>(),
        output.data_ptr<at::Half>(),
        n
    );
    
    return output;
}

torch::Tensor activation_bf16(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n = input.numel();
    
    const int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;
    
    activation_kernel_bf16<<<grid_size, block_size>>>(
        input.data_ptr<at::BFloat16>(),
        output.data_ptr<at::BFloat16>(),
        n
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_activation_v3',
    cpp_sources='''
torch::Tensor activation_fp32(torch::Tensor input);
torch::Tensor activation_fp16(torch::Tensor input);
torch::Tensor activation_bf16(torch::Tensor input);
''',
    cuda_sources=cuda_source,
    functions=['activation_fp32', 'activation_fp16', 'activation_bf16'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that performs a convolution, applies activation, and then applies Batch Normalization.
    Uses fused CUDA kernel for activation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        
        # Register dispatchers based on dtype
        self.activation_map = {
            torch.float32: ext.activation_fp32,
            torch.float16: ext.activation_fp16,
            torch.bfloat16: ext.activation_bf16,
        }

    def forward(self, x):
        x = self.conv(x)
        
        # Use fused activation kernel based on dtype
        dtype = x.dtype
        if dtype in self.activation_map:
            x = self.activation_map[dtype](x)
        else:
            # Fallback for other dtypes
            x = torch.multiply(torch.tanh(torch.nn.functional.softplus(x)), x)
        
        x = self.bn(x)
        return x

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
