cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <algorithm>

// Fused Conv2d + InstanceNorm + Divide kernel
// For RTX 3090 (SM86)

template<typename scalar_t>
__global__ void fused_conv_instnorm_divide_kernel(
    const scalar_t* __restrict__ x,      // Input: (N, C_in, H_in, W_in)
    const scalar_t* __restrict__ weight, // Weight: (C_out, C_in, kH, kW)
    const scalar_t* __restrict__ bias,   // Bias: (C_out,)
    scalar_t* __restrict__ y,            // Output: (N, C_out, H_out, W_out)
    int N, int C_in, int H_in, int W_in,
    int C_out, int kH, int kW,
    int stride, int padding,
    scalar_t eps, scalar_t divide_by) {
    
    // Each block handles one (n, c_out) pair
    int n = blockIdx.y;
    int c_out = blockIdx.x;
    
    int H_out = H_in - kH + 1;  // No padding output size
    int W_out = W_in - kW + 1;
    
    // Shared memory for caching
    extern __shared__ float shared_mem[];
    
    // Compute mean and variance for instance norm
    float sum = 0.0f;
    float sum_sq = 0.0f;
    int count = H_out * W_out;
    
    // First pass: compute sum for mean
    for (int h_out = threadIdx.x; h_out < H_out; h_out += blockDim.x) {
        for (int w_out = 0; w_out < W_out; ++w_out) {
            float acc = 0.0f;
            
            // Convolution
            for (int kh = 0; kh < kH; ++kh) {
                for (int kw = 0; kw < kW; ++kw) {
                    int h_in = h_out * stride - padding + kh;
                    int w_in = w_out * stride - padding + kw;
                    
                    if (h_in >= 0 && h_in < H_in && w_in >= 0 && w_in < W_in) {
                        for (int c_in = 0; c_in < C_in; ++c_in) {
                            int x_idx = ((n * C_in + c_in) * H_in + h_in) * W_in + w_in;
                            int w_idx = ((c_out * C_in + c_in) * kH + kh) * kW + kw;
                            acc += static_cast<float>(x[x_idx]) * static_cast<float>(weight[w_idx]);
                        }
                    }
                }
            }
            acc += static_cast<float>(bias[c_out]);
            
            sum += acc;
            sum_sq += acc * acc;
        }
    }
    
    // Block reduction for mean
    float block_sum = 0.0f;
    float block_sum_sq = 0.0f;
    for (int i = threadIdx.x; i < count; i += blockDim.x) {
        // This approach won't work - need different strategy
    }
}

// Simplified kernel - each thread handles one output element
template<typename scalar_t>
__global__ void fused_conv_instnorm_divide_kernel_v2(
    const scalar_t* __restrict__ x,      // Input: (N, C_in, H_in, W_in)
    const scalar_t* __restrict__ weight, // Weight: (C_out, C_in, kH, kW)
    const scalar_t* __restrict__ bias,   // Bias: (C_out,)
    scalar_t* __restrict__ y,            // Output: (N, C_out, H_out, W_out)
    int N, int C_in, int H_in, int W_in,
    int C_out, int kH, int kW,
    int stride, int padding,
    scalar_t eps, scalar_t divide_by) {
    
    int H_out = H_in - kH + 1;
    int W_out = W_in - kW + 1;
    
    // Each thread handles one output element (n, c_out, h_out, w_out)
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_out * W_out;
    
    if (idx >= total) return;
    
    // Decode indices
    int tmp = idx;
    int w_out = tmp % W_out;
    tmp /= W_out;
    int h_out = tmp % H_out;
    tmp /= H_out;
    int c_out = tmp % C_out;
    int n = tmp / C_out;
    
    // Compute convolution for this output element
    float acc = 0.0f;
    for (int kh = 0; kh < kH; ++kh) {
        for (int kw = 0; kw < kW; ++kw) {
            int h_in = h_out * stride - padding + kh;
            int w_in = w_out * stride - padding + kw;
            
            if (h_in >= 0 && h_in < H_in && w_in >= 0 && w_in < W_in) {
                for (int c_in = 0; c_in < C_in; ++c_in) {
                    int x_idx = ((n * C_in + c_in) * H_in + h_in) * W_in + w_in;
                    int w_idx = ((c_out * C_in + c_in) * kH + kh) * kW + kw;
                    acc += static_cast<float>(x[x_idx]) * static_cast<float>(weight[w_idx]);
                }
            }
        }
    }
    acc += static_cast<float>(bias[c_out]);
    
    // Store intermediate result
    y[idx] = static_cast<scalar_t>(acc);
}

// Instance norm + divide kernel (separate for now)
template<typename scalar_t>
__global__ void instnorm_divide_kernel(
    const scalar_t* __restrict__ x,  // Input: (N, C, H, W)
    scalar_t* __restrict__ y,        // Output: (N, C, H, W)
    int N, int C, int H, int W,
    scalar_t eps, scalar_t divide_by) {
    
    int HW = H * W;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C;
    
    if (idx >= total) return;
    
    int c = idx % C;
    int n = idx / C;
    
    int base = (n * C + c) * H * W;
    
    // Compute mean
    float sum = 0.0f;
    for (int i = 0; i < HW; ++i) {
        sum += static_cast<float>(x[base + i]);
    }
    float mean = sum / HW;
    
    // Compute variance
    float var = 0.0f;
    for (int i = 0; i < HW; ++i) {
        float diff = static_cast<float>(x[base + i]) - mean;
        var += diff * diff;
    }
    var /= HW;
    
    float inv_std = 1.0f / sqrtf(var + static_cast<float>(eps));
    float scale = inv_std / static_cast<float>(divide_by);
    
    // Normalize and divide
    for (int i = 0; i < HW; ++i) {
        float val = static_cast<float>(x[base + i]);
        y[base + i] = static_cast<scalar_t>((val - mean) * scale);
    }
}

torch::Tensor fused_conv2d_instnorm_divide(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float divide_by,
    float eps = 1e-5) {
    
    int N = x.size(0);
    int C_in = x.size(1);
    int H_in = x.size(2);
    int W_in = x.size(3);
    
    int C_out = weight.size(0);
    int kH = weight.size(2);
    int kW = weight.size(3);
    
    int stride = 1;
    int padding = 0;  // No padding like reference
    
    int H_out = H_in - kH + 1;
    int W_out = W_in - kW + 1;
    
    // Allocate intermediate buffer for conv output
    auto conv_out = torch::empty({N, C_out, H_out, W_out}, x.options());
    auto output = torch::empty({N, C_out, H_out, W_out}, x.options());
    
    int threads = 256;
    int total_elems = N * C_out * H_out * W_out;
    int blocks_conv = (total_elems + threads - 1) / threads;
    
    // Launch conv kernel
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "fused_conv_kernel", ([&] {
        fused_conv_instnorm_divide_kernel_v2<scalar_t><<<blocks_conv, threads>>>(
            x.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            conv_out.data_ptr<scalar_t>(),
            N, C_in, H_in, W_in,
            C_out, kH, kW,
            stride, padding,
            eps, divide_by);
    }));
    
    // Launch instance norm + divide kernel
    int total_nc = N * C_out;
    int blocks_norm = (total_nc + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(conv_out.scalar_type(), "instnorm_divide_kernel", ([&] {
        instnorm_divide_kernel<scalar_t><<<blocks_norm, threads>>>(
            conv_out.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            N, C_out, H_out, W_out,
            eps, divide_by);
    }));
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_conv2d_instnorm_divide(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float divide_by,
    float eps);
"""

# Compile the extension
ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv2d_instnorm_divide'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86', '-O3'],
)


class Model(nn.Module):
    """
    Optimized model that performs a convolution, applies Instance Normalization, and divides by a constant.
    Uses fused CUDA kernels for better performance.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels)
        self.divide_by = divide_by

    def forward(self, x):
        # Use fused kernel
        weight = self.conv.weight
        bias = self.conv.bias
        return ext.fused_conv2d_instnorm_divide(x, weight, bias, float(self.divide_by))


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    divide_by = 2.0
    return [in_channels, out_channels, kernel_size, divide_by]
