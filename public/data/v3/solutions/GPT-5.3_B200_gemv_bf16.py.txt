import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def gemv_kernel(X, W, Y, B, IN, OUT, stride_xb, stride_xk, stride_wo, stride_wk, stride_yb, stride_yo, BLOCK_K: tl.constexpr):
    pid = tl.program_id(0)
    b = pid // OUT
    o = pid % OUT

    if b >= B:
        return

    x_ptr = X + b * stride_xb
    w_ptr = W + o * stride_wo

    acc = tl.zeros((), dtype=tl.float32)

    for k in range(0, IN, BLOCK_K):
        k_offsets = k + tl.arange(0, BLOCK_K)
        mask = k_offsets < IN

        x = tl.load(x_ptr + k_offsets * stride_xk, mask=mask, other=0).to(tl.float32)
        w = tl.load(w_ptr + k_offsets * stride_wk, mask=mask, other=0).to(tl.float32)

        acc += tl.sum(x * w)

    y_ptr = Y + b * stride_yb + o * stride_yo
    tl.store(y_ptr, acc.to(tl.bfloat16))


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.contiguous()
        B, IN = x.shape
        OUT = self.out_features

        y = torch.empty((B, OUT), device=x.device, dtype=torch.bfloat16)

        grid = (B * OUT,)

        gemv_kernel[grid](
            x, self.weight, y,
            B, IN, OUT,
            x.stride(0), x.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            y.stride(0), y.stride(1),
            BLOCK_K=128,
        )

        return y


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16, device='cuda')]


def get_init_inputs():
