import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

#define CUDA_CHECK_ERROR(call) \
  do { \
    cudaError_t err = call; \
    if (err != cudaSuccess) { \
      printf("CUDA error at %s:%d: %s\n", __FILE__, __LINE__, cudaGetErrorString(err)); \
      exit(EXIT_FAILURE); \
    } \
  } while (0)

// Generic activation + BN kernel for float32
__global__ void fused_act_bn_kernel_float(
    const float* __restrict__ x,
    float* __restrict__ y,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const int n,
    const int c,
    const float eps
) {
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    int c_idx = idx % c;

    float w = weight[c_idx];
    float b = bias ? bias[c_idx] : 0.0f;
    float mean = running_mean[c_idx];
    float var = running_var[c_idx];

    float val = x[idx];

    // Swish activation: x * sigmoid(x) with numerical stability
    float sig;
    if (val > 0) {
        sig = 1.0f / (1.0f + expf(-val));
    } else {
        sig = expf(val) / (1.0f + expf(val));
    }
    float act = val * sig;

    // BatchNorm
    float inv_std = rsqrtf(var + eps);
    y[idx] = (act - mean) * inv_std * w + b;
}

// FP16 kernel
__global__ void fused_act_bn_kernel_fp16(
    const half* __restrict__ x,
    half* __restrict__ y,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const int n,
    const int c,
    const float eps
) {
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    int c_idx = idx % c;

    float w = __half2float(weight[c_idx]);
    float b = bias ? __half2float(bias[c_idx]) : 0.0f;
    float mean = running_mean[c_idx];
    float var = running_var[c_idx];

    float val = __half2float(x[idx]);

    // Sigmoid for Swish
    float sig;
    if (val > 0) {
        sig = 1.0f / (1.0f + expf(-val));
    } else {
        sig = expf(val) / (1.0f + expf(val));
    }
    float act = val * sig;

    float inv_std = rsqrtf(var + eps);
    y[idx] = __float2half((act - mean) * inv_std * w + b);
}

// BF16 kernel
__global__ void fused_act_bn_kernel_bf16(
    const nv_bfloat16* __restrict__ x,
    nv_bfloat16* __restrict__ y,
    const nv_bfloat16* __restrict__ weight,
    const nv_bfloat16* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const int n,
    const int c,
    const float eps
) {
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    int c_idx = idx % c;

    float w = __bfloat162float(weight[c_idx]);
    float b = bias ? __bfloat162float(bias[c_idx]) : 0.0f;
    float mean = running_mean[c_idx];
    float var = running_var[c_idx];

    float val = __bfloat162float(x[idx]);

    float sig;
    if (val > 0) {
        sig = 1.0f / (1.0f + expf(-val));
    } else {
        sig = expf(val) / (1.0f + expf(val));
    }
    float act = val * sig;

    float inv_std = rsqrtf(var + eps);
    y[idx] = __float2bfloat16((act - mean) * inv_std * w + b);
}

// Host wrapper for float32
torch::Tensor fused_act_bn_cuda_float(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps
) {
    auto y = torch::empty_like(x);

    int n = x.numel();
    int c = x.size(1);
    
    const int block_size = 256;
    int grid_size = (n + block_size - 1) / block_size;

    fused_act_bn_kernel_float<<<grid_size, block_size>>>(
        x.data_ptr<float>(),
        y.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        running_mean.data_ptr<float>(),
        running_var.data_ptr<float>(),
        n, c, (float)eps
    );

    CUDA_CHECK_ERROR(cudaGetLastError());
    return y;
}

// Polymorphic wrapper
torch::Tensor fused_act_bn_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps
) {
    auto scalar_type = x.scalar_type();
    if (scalar_type == at::ScalarType::Float) {
        return fused_act_bn_cuda_float(x, weight, bias, running_mean, running_var, eps);
    } else if (scalar_type == at::ScalarType::Half) {
        return fused_act_bn_cuda_fp16(x, weight, bias, running_mean, running_var, eps);
    } else if (scalar_type == at::ScalarType::BFloat16) {
        return fused_act_bn_cuda_bf16(x, weight, bias, running_mean, running_var, eps);
    } else {
        throw std::runtime_error("Unsupported dtype");
    }
}
'''

cpp_source = r'''
torch::Tensor fused_act_bn_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps);
'''

fused_conv_act_bn = load_inline(
    name='fused_conv_act_bn',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_act_bn_cuda'],
    verbose=False,
    extra_cuda_cflags=['-use_fast_math'],
)

class Model(nn.Module):
    """
    Fused model: Conv2d + Activation (Swish) + BatchNorm
    Uses cuDNN for Conv2d and a custom fused CUDA kernel for activation + batchnorm.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        # Use cuDNN convolution
        x = self.conv(x)
        
        # Fused activation (Swish) + BatchNorm
        x = fused_conv_act_bn.fused_act_bn_cuda(
            x,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            self.eps
        )
        return x

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3]
