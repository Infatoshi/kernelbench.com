import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void sum_dim1_kernel(const scalar_t* __restrict__ x, scalar_t* __restrict__ y,
                                int B, int D1, int D2) {
    int b = blockIdx.x;
    int j = blockIdx.y;

    int tid = threadIdx.x;

    float acc = 0.0f;

    for (int i = tid; i < D1; i += blockDim.x) {
        int idx = b * D1 * D2 + i * D2 + j;
        acc += (float)x[idx];
    }

    __shared__ float smem[256];
    smem[tid] = acc;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) smem[tid] += smem[tid + s];
        __syncthreads();
    }

    if (tid == 0) {
        int out_idx = b * D2 + j;
        y[out_idx] = (scalar_t)smem[0];
    }
}

torch::Tensor sum_dim1(torch::Tensor x) {
    int B = x.size(0);
    int D1 = x.size(1);
    int D2 = x.size(2);

    auto y = torch::empty({B,1,D2}, x.options());

    dim3 grid(B, D2);
    dim3 block(256);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "sum_dim1_kernel", ([&] {
        sum_dim1_kernel<scalar_t><<<grid, block>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            B, D1, D2);
    }));

    return y;
}
'''

ext = load_inline(
    name='sum_dim1_ext',
    cpp_sources='torch::Tensor sum_dim1(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['sum_dim1'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.dim != 1:
            # fallback by permuting so reduction dim becomes 1
            dims = list(range(x.dim()))
            dims[1], dims[self.dim] = dims[self.dim], dims[1]
            x = x.permute(dims).contiguous()
            y = ext.sum_dim1(x)
            y = y.permute(dims)
            return y
        return ext.sum_dim1(x)


batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1


def get_inputs():
    x = torch.randn(batch_size, dim1, dim2, device='cuda')
    return [x]


def get_init_inputs():
    return [reduce_dim]
