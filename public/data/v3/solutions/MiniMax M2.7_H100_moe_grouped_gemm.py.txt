"""
Optimized MoE Grouped GEMM using CUDA with fused kernel.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void moe_fused_kernel(
    const __half* __restrict__ x,
    const int* __restrict__ expert_idx,
    const __half* __restrict__ expert_w,
    const __half* __restrict__ expert_up,
    const __half* __restrict__ expert_down,
    __half* __restrict__ output,
    int num_tokens, int hidden, int expert_dim, int top_k, int num_experts
) {
    int tid = threadIdx.x;
    int block_id = blockIdx.x;
    
    int expert_id = block_id;
    if (expert_id >= num_experts) return;
    
    extern __shared__ __half smem[];
    __half* x_tokens = smem;
    __half* up_result = smem + 64 * 1024;
    int* token_ids = (int*)(smem + 64 * 1024 + 64 * 3072);
    
    __shared__ int num_active;
    if (tid == 0) num_active = 0;
    __syncthreads();
    
    for (int t = tid; t < num_tokens; t += blockDim.x) {
        for (int k = 0; k < top_k; k++) {
            if (expert_idx[t * top_k + k] == expert_id) {
                int slot = atomicAdd(&num_active, 1);
                if (slot < 64) {
                    token_ids[slot] = t;
                    for (int h = 0; h < hidden; h++) {
                        x_tokens[slot * hidden + h] = x[t * hidden + h];
                    }
                }
                break;
            }
        }
    }
    __syncthreads();
    
    int n_active = num_active;
    if (n_active == 0) return;
    
    for (int t = tid / expert_dim; t < n_active; t += blockDim.x / expert_dim) {
        int k_idx = tid % expert_dim;
        __half sum = __float2half(0.0f);
        
        for (int h = 0; h < 1024; h += 4) {
            __half x0 = x_tokens[t * hidden + h];
            __half x1 = x_tokens[t * hidden + h + 1];
            __half x2 = x_tokens[t * hidden + h + 2];
            __half x3 = x_tokens[t * hidden + h + 3];
            
            __half w0 = expert_up[(expert_id * 1024 + h) * 3072 + k_idx];
            __half w1 = expert_up[(expert_id * 1024 + h + 1) * 3072 + k_idx];
            __half w2 = expert_up[(expert_id * 1024 + h + 2) * 3072 + k_idx];
            __half w3 = expert_up[(expert_id * 1024 + h + 3) * 3072 + k_idx];
            
            sum = __hadd(sum, __hmul(x0, w0));
            sum = __hadd(sum, __hmul(x1, w1));
            sum = __hadd(sum, __hmul(x2, w2));
            sum = __hadd(sum, __hmul(x3, w3));
        }
        up_result[t * 3072 + k_idx] = sum;
    }
    __syncthreads();
    
    for (int t = tid / hidden; t < n_active; t += blockDim.x / hidden) {
        int h_idx = tid % hidden;
        __half sum = __float2half(0.0f);
        
        for (int k = 0; k < 3072; k += 4) {
            __half u0 = up_result[t * 3072 + k];
            __half u1 = up_result[t * 3072 + k + 1];
            __half u2 = up_result[t * 3072 + k + 2];
            __half u3 = up_result[t * 3072 + k + 3];
            
            __half w0 = expert_down[(expert_id * 3072 + k) * 1024 + h_idx];
            __half w1 = expert_down[(expert_id * 3072 + k + 1) * 1024 + h_idx];
            __half w2 = expert_down[(expert_id * 3072 + k + 2) * 1024 + h_idx];
            __half w3 = expert_down[(expert_id * 3072 + k + 3) * 1024 + h_idx];
            
            sum = __hadd(sum, __hmul(u0, w0));
            sum = __hadd(sum, __hmul(u1, w1));
            sum = __hadd(sum, __hmul(u2, w2));
            sum = __hadd(sum, __hmul(u3, w3));
        }
        
        int token = token_ids[t];
        __half weight = __float2half(0.0f);
        for (int k = 0; k < top_k; k++) {
            if (expert_idx[token * top_k + k] == expert_id) {
                weight = expert_w[token * top_k + k];
                break;
            }
        }
        
        sum = __hmul(sum, weight);
        atomicAdd((float*)&output[token * 1024 + h_idx], __half2float(sum));
    }
}

torch::Tensor moe_forward(
    torch::Tensor x,
    torch::Tensor expert_indices,
    torch::Tensor expert_weights,
    torch::Tensor expert_up,
    torch::Tensor expert_down
) {
    int num_tokens = x.size(0);
    int hidden = x.size(1);
    int expert_dim = expert_up.size(2);
    int top_k = expert_indices.size(1);
    int num_experts = expert_up.size(0);
    
    x = x.contiguous();
    expert_indices = expert_indices.contiguous();
    expert_weights = expert_weights.contiguous();
    expert_up = expert_up.contiguous();
    expert_down = expert_down.contiguous();
    
    auto x_h = x.to(torch::kHalf);
    auto w_h = expert_weights.to(torch::kHalf);
    auto up_h = expert_up.to(torch::kHalf);
    auto down_h = expert_down.to(torch::kHalf);
    auto output = torch::zeros({num_tokens, hidden}, x.options());
    auto out_h = output.to(torch::kHalf);
    
    int smem_size = (64 * 1024 + 64 * 3072) * sizeof(__half) + 64 * sizeof(int);
    
    dim3 grid(num_experts);
    dim3 block(128);
    
    moe_fused_kernel<<<grid, block, smem_size>>>(
        reinterpret_cast<const __half*>(x_h.data_ptr()),
        reinterpret_cast<const int*>(expert_indices.data_ptr()),
        reinterpret_cast<const __half*>(w_h.data_ptr()),
        reinterpret_cast<const __half*>(up_h.data_ptr()),
        reinterpret_cast<const __half*>(down_h.data_ptr()),
        reinterpret_cast<__half*>(out_h.data_ptr()),
        num_tokens, hidden, expert_dim, top_k, num_experts
    );
    
    cudaDeviceSynchronize();
    
    return out_h.to(x.dtype());
}
'''

cpp_source = r'''
torch::Tensor moe_forward(
    torch::Tensor x,
    torch::Tensor expert_indices,
    torch::Tensor expert_weights,
    torch::Tensor expert_up,
    torch::Tensor expert_down
);
'''

ext = load_inline(
    name='moe_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['moe_forward'],
    verbose=False,
    extra_cuda_cflags=['-gencode=arch=compute_90,code=sm_90', '-O3'],
)


class Model(nn.Module):
    def __init__(self, num_experts: int = 8, hidden_dim: int = 1024, expert_dim: int = 3072):
        super().__init__()
        self.num_experts = num_experts
        self.hidden_dim = hidden_dim
        self.expert_dim = expert_dim
        self.expert_up = nn.Parameter(torch.randn(num_experts, hidden_dim, expert_dim) * 0.02)
        self.expert_down = nn.Parameter(torch.randn(num_experts, expert_dim, hidden_dim) * 0.02)

    def forward(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        """
        x: [batch, seq, hidden]
        expert_indices: [batch, seq, top_k]
        expert_weights: [batch, seq, top_k]
        """
        batch, seq, hidden = x.shape
        top_k = expert_indices.shape[-1]
        
        x_flat = x.reshape(batch * seq, hidden)
        idx_flat = expert_indices.reshape(batch * seq, top_k)
        w_flat = expert_weights.reshape(batch * seq, top_k)
        
        return ext.moe_forward(
            x_flat,
            idx_flat,
            w_flat,
            self.expert_up,
            self.expert_down
        ).reshape(batch, seq, hidden)


def get_inputs():
    batch, seq, hidden = 4, 256, 1024
    top_k = 2
    num_experts = 8

    x = torch.randn(batch, seq, hidden)
    expert_indices = torch.randint(0, num_experts, (batch, seq, top_k))
    expert_weights = torch.softmax(torch.randn(batch, seq, top_k), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return []
