import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>

using namespace nvcuda;

// Simple tiled GEMM using WMMA
// Each block computes one tile of C
// Each warp in the block computes one 16x16 WMMA tile

__global__ void wmma_gemm_kernel(const half* __restrict__ A, const half* __restrict__ B, float* __restrict__ C,
                                  int M, int N, int K) {
    constexpr int WMMA_M = 16;
    constexpr int WMMA_N = 16;
    constexpr int WMMA_K = 16;

    // Each block handles a 64x64 tile (4 warps, 2x2 arrangement)
    constexpr int BLOCK_M = 64;
    constexpr int BLOCK_N = 64;
    constexpr int WARPS_M = 2;
    constexpr int WARPS_N = 2;

    int blockRow = blockIdx.y;
    int blockCol = blockIdx.x;
    int warpId = threadIdx.x / 32;
    int laneId = threadIdx.x % 32;

    int warpRow = warpId / WARPS_N;
    int warpCol = warpId % WARPS_N;

    // Starting position of this warp's output tile
    int row = blockRow * BLOCK_M + warpRow * WMMA_M;
    int col = blockCol * BLOCK_N + warpCol * WMMA_N;

    // Bounds check
    if (row >= M || col >= N) return;

    // WMMA fragments
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> b_frag;
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> c_frag;

    wmma::fill_fragment(c_frag, 0.0f);

    const half* a_ptr = A + row * K;
    const half* b_ptr = B + col;

    // Number of WMMA iterations along K dimension
    int k_tiles = K / WMMA_K;

    for (int t = 0; t < k_tiles; t++) {
        int k = t * WMMA_K;
        
        // Load A tile (16 elements from row)
        // wmma::load_matrix_sync expects the fragment to be loaded from memory
        // For row-major A, we load 16 consecutive elements from the current row
        wmma::load_matrix_sync(a_frag, a_ptr + k, K);
        
        // Load B tile (16 elements from row)
        // For row-major B, we need to load from the appropriate row
        wmma::load_matrix_sync(b_frag, b_ptr + k * N, N);
        
        // MMA operation
        wmma::mma_sync(c_frag, a_frag, b_frag, c_frag);
    }

    // Store result
    float* c_ptr = C + row * N + col;
    wmma::store_matrix_sync(c_ptr, c_frag, N, wmma::mem_row_major);
}

torch::Tensor wmma_gemm(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);

    // Convert to FP16
    torch::Tensor A_half = A.to(torch::kHalf);
    torch::Tensor B_half = B.to(torch::kHalf);

    // Output tensor (FP32)
    torch::Tensor C = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(A.device()));

    // Launch kernel
    constexpr int BLOCK_M = 64;
    constexpr int BLOCK_N = 64;
    constexpr int THREADS = 128;  // 4 warps

    dim3 blockDim(THREADS);
    dim3 gridDim((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);

    wmma_gemm_kernel<<<gridDim, blockDim>>>(
        (const half*)A_half.data_ptr<c10::Half>(),
        (const half*)B_half.data_ptr<c10::Half>(),
        C.data_ptr<float>(),
        M, N, K
    );

    cudaDeviceSynchronize();
    
    return C;
}
'''

ext = load_inline(
    name='wmma_gemm_ext',
    cpp_sources='torch::Tensor wmma_gemm(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['wmma_gemm'],
    extra_cuda_cflags=['-arch=sm_86', '-std=c++17', '-use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A, B):
        return ext.wmma_gemm(A, B)

def get_inputs():
    M = 1024
    K = 4096
    N = 2048
    A = torch.randn(M, K)
    B = torch.randn(K, N)
    return [A, B]

def get_init_inputs():
    return []
