import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

#define EPS 1e-5f

__global__ void conv_forward_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N, int C_in, int H, int W,
    int C_out, int K,
    int H_out, int W_out)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int c_out = (idx / (W_out * H_out)) % C_out;
    int n = idx / (W_out * H_out * C_out);

    float val = b ? b[c_out] : 0.0f;

    for (int c = 0; c < C_in; ++c) {
        for (int ky = 0; ky < K; ++ky) {
            for (int kx = 0; kx < K; ++kx) {
                int in_y = h_out + ky;
                int in_x = w_out + kx;
                int in_idx = ((n * C_in + c) * H + in_y) * W + in_x;
                int w_idx = ((c_out * C_in + c) * K + ky) * K + kx;
                val += x[in_idx] * w[w_idx];
            }
        }
    }

    out[idx] = val;
}

__global__ void instancenorm_reduce_kernel(
    const float* __restrict__ x,
    float* __restrict__ mean,
    float* __restrict__ var,
    int N, int C, int HW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= N * C) return;

    int n = idx / C;
    int c = idx % C;

    const float* ptr = x + (n * C + c) * HW;

    float m = 0.0f;
    for (int i = 0; i < HW; ++i) m += ptr[i];
    m /= HW;

    float v = 0.0f;
    for (int i = 0; i < HW; ++i) {
        float d = ptr[i] - m;
        v += d * d;
    }
    v /= HW;

    mean[idx] = m;
    var[idx] = v;
}

__global__ void instancenorm_apply_kernel(
    float* __restrict__ x,
    const float* __restrict__ mean,
    const float* __restrict__ var,
    float div,
    int N, int C, int HW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * HW;
    if (idx >= total) return;

    int hw = idx % HW;
    int c = (idx / HW) % C;
    int n = idx / (HW * C);

    int mc = n * C + c;

    float m = mean[mc];
    float v = var[mc];

    float val = x[idx];
    val = (val - m) * rsqrtf(v + EPS);
    val = val / div;

    x[idx] = val;
}


torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b, float div) {
    x = x.contiguous();
    w = w.contiguous();
    if (b.defined()) b = b.contiguous();

    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int C_out = w.size(0);
    int K = w.size(2);

    int H_out = H - K + 1;
    int W_out = W - K + 1;

    auto out = torch::empty({N, C_out, H_out, W_out}, x.options());

    int total = N * C_out * H_out * W_out;
    conv_forward_kernel<<<(total+255)/256, 256>>>(
        x.data_ptr<float>(), w.data_ptr<float>(),
        b.defined() ? b.data_ptr<float>() : nullptr,
        out.data_ptr<float>(),
        N, C_in, H, W, C_out, K, H_out, W_out);

    int NC = N * C_out;
    auto mean = torch::empty({NC}, x.options());
    auto var = torch::empty({NC}, x.options());

    instancenorm_reduce_kernel<<<(NC+255)/256, 256>>>(
        out.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(),
        N, C_out, H_out * W_out);

    int total2 = N * C_out * H_out * W_out;
    instancenorm_apply_kernel<<<(total2+255)/256, 256>>>(
        out.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), div,
        N, C_out, H_out * W_out);

    return out;
}
'''

ext = load_inline(
    name='fused_conv_instnorm',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b, float div);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = divide_by

    def forward(self, x):
        return ext.fused_op(x, self.conv.weight, self.conv.bias, self.divide_by)


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 2.0]
