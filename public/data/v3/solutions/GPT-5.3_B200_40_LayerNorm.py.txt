import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

template<typename scalar_t>
__global__ void layernorm_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ gamma,
    const scalar_t* __restrict__ beta,
    scalar_t* __restrict__ y,
    int64_t N, int64_t inner)
{
    int b = blockIdx.x;
    if (b >= N) return;

    const scalar_t* x_ptr = x + b * inner;
    scalar_t* y_ptr = y + b * inner;

    float sum = 0.f;
    float sumsq = 0.f;

    for (int i = threadIdx.x; i < inner; i += blockDim.x) {
        float v = (float)x_ptr[i];
        sum += v;
        sumsq += v * v;
    }

    __shared__ float s_sum[256];
    __shared__ float s_sumsq[256];

    int tid = threadIdx.x;
    s_sum[tid] = sum;
    s_sumsq[tid] = sumsq;
    __syncthreads();

    // reduction
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            s_sum[tid] += s_sum[tid + stride];
            s_sumsq[tid] += s_sumsq[tid + stride];
        }
        __syncthreads();
    }

    float mean = s_sum[0] / inner;
    float var = s_sumsq[0] / inner - mean * mean;
    float inv_std = rsqrtf(var + 1e-5f);

    for (int i = threadIdx.x; i < inner; i += blockDim.x) {
        float v = (float)x_ptr[i];
        float g = (float)gamma[i];
        float be = (float)beta[i];
        float out = (v - mean) * inv_std;
        out = out * g + be;
        y_ptr[i] = (scalar_t)out;
    }
}

torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta) {
    auto y = torch::empty_like(x);

    int64_t inner = 1;
    for (int i = 1; i < x.dim(); ++i) inner *= x.size(i);
    int64_t N = x.size(0);

    int threads = 256;

    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "layernorm_kernel", ([&] {
        layernorm_kernel<scalar_t><<<N, threads>>>(
            x.data_ptr<scalar_t>(),
            gamma.data_ptr<scalar_t>(),
            beta.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            N, inner);
    }));

    return y;
}
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources='torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta);',
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.normalized_shape = normalized_shape
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(x, self.weight, self.bias)


def get_inputs():
    return [torch.randn(16, 64, 256, 256, device='cuda')]


def get_init_inputs():
    return [(64, 256, 256)]
