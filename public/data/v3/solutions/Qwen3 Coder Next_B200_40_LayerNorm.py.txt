import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Convert any numeric type to float
template<typename T>
__device__ __forceinline__ float to_float(T val) {
    return static_cast<float>(val);
}

template<>
__device__ __forceinline__ float to_float<__half>(__half val) {
    return __half2float(val);
}

template<>
__device__ __forceinline__ float to_float<__nv_bfloat16>(__nv_bfloat16 val) {
    return __bfloat162float(val);
}

// LayerNorm kernel using 1D blocking with proper synchronization
template<typename scalar_t>
__global__ void layer_norm_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ gamma,
    const scalar_t* __restrict__ beta,
    scalar_t* __restrict__ y,
    int batch_size,
    int normalized_size,
    float eps
) {
    extern __shared__ float shared_mem[];
    float* mean_var = shared_mem;
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.y;
    
    constexpr int BLOCK_SIZE_X = 256;
    
    // Step 1: Compute partial sums for mean
    float sum = 0.0f;
    for (int i = tid; i < normalized_size; i += BLOCK_SIZE_X) {
        int idx = batch_idx * normalized_size + i;
        sum += to_float(x[idx]);
    }
    
    mean_var[tid] = sum;
    __syncthreads();
    
    // Parallel reduction for mean
    for (int stride = BLOCK_SIZE_X / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            mean_var[tid] += mean_var[tid + stride];
        }
        __syncthreads();
    }
    
    float mean = mean_var[0] / normalized_size;
    
    // Step 2: Compute partial sums for variance
    float var_sum = 0.0f;
    for (int i = tid; i < normalized_size; i += BLOCK_SIZE_X) {
        int idx = batch_idx * normalized_size + i;
        float diff = to_float(x[idx]) - mean;
        var_sum += diff * diff;
    }
    
    mean_var[tid] = var_sum;
    __syncthreads();
    
    // Parallel reduction for variance
    for (int stride = BLOCK_SIZE_X / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            mean_var[tid] += mean_var[tid + stride];
        }
        __syncthreads();
    }
    
    float var = mean_var[0] / normalized_size;
    float inv_std = rsqrtf(var + eps);
    
    // Step 3: Compute normalized output
    for (int i = tid; i < normalized_size; i += BLOCK_SIZE_X) {
        int idx = batch_idx * normalized_size + i;
        float norm_x = (to_float(x[idx]) - mean) * inv_std;
        float g = to_float(gamma[i]);
        float b = to_float(beta[i]);
        y[idx] = static_cast<scalar_t>(norm_x * g + b);
    }
}

torch::Tensor layer_norm_op(
    torch::Tensor x,
    torch::Tensor gamma,
    torch::Tensor beta,
    float eps
) {
    TORCH_CHECK(x.is_cuda(), "Input must be on CUDA");
    TORCH_CHECK(gamma.is_cuda(), "Gamma must be on CUDA");
    TORCH_CHECK(beta.is_cuda(), "Beta must be on CUDA");
    
    auto y = torch::empty_like(x);
    int batch_size = x.numel() / gamma.numel();
    int normalized_size = gamma.numel();
    
    dim3 threads(256);
    dim3 blocks(1, batch_size);
    size_t shared_mem = 2 * 256 * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "layer_norm_kernel", [&] {
        layer_norm_kernel<scalar_t><<<blocks, threads, shared_mem>>>(
            x.data_ptr<scalar_t>(),
            gamma.data_ptr<scalar_t>(),
            beta.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            batch_size,
            normalized_size,
            eps
        );
    });
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor layer_norm_op(
    torch::Tensor x,
    torch::Tensor gamma,
    torch::Tensor beta,
    float eps
);
'''

ext = load_inline(
    name='layer_norm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layer_norm_op'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized LayerNorm implementation for B200.
    """
    def __init__(self, normalized_shape):
        super().__init__()
        # Use same buffer names to match nn.LayerNorm state dict
        self.register_buffer('weight', torch.ones(normalized_shape))
        self.register_buffer('bias', torch.zeros(normalized_shape))
        self.normalized_shape = normalized_shape
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layer_norm_op(x, self.weight, self.bias, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(64, 256, 256)]
