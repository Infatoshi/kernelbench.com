import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Highly optimized kernel for max pool 2d
// Each thread handles one output position with fully unrolled window computation
template <typename T>
__global__ void max_pool2d_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    // Each block handles multiple output positions
    // Threads are arranged as (TILE_H, TILE_W) = (16, 16)
    const int TILE_H = 16;
    const int TILE_W = 16;
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int oy = by * TILE_H + ty;
    int ox = bx * TILE_W + tx;
    
    if (oy >= out_height || ox >= out_width) return;
    
    int batch_idx = bz / channels;
    int chan_idx = bz % channels;
    
    int iy_start = oy * stride - padding;
    int ix_start = ox * stride - padding;
    
    float max_val = -INFINITY;
    
    // Fully unrolled 2x2 window with dilation
    if (kernel_size == 2) {
        int iy0 = iy_start;
        int iy1 = iy_start + dilation;
        int ix0 = ix_start;
        int ix1 = ix_start + dilation;
        
        // Load all 4 positions
        float v0 = -INFINITY, v1 = -INFINITY, v2 = -INFINITY, v3 = -INFINITY;
        
        if (iy0 >= 0 && iy0 < in_height && ix0 >= 0 && ix0 < in_width)
            v0 = static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy0) * in_width + ix0]);
        if (iy0 >= 0 && iy0 < in_height && ix1 >= 0 && ix1 < in_width)
            v1 = static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy0) * in_width + ix1]);
        if (iy1 >= 0 && iy1 < in_height && ix0 >= 0 && ix0 < in_width)
            v2 = static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy1) * in_width + ix0]);
        if (iy1 >= 0 && iy1 < in_height && ix1 >= 0 && ix1 < in_width)
            v3 = static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy1) * in_width + ix1]);
        
        // Pairwise max for better ILP
        float m01 = fmaxf(v0, v1);
        float m23 = fmaxf(v2, v3);
        max_val = fmaxf(m01, m23);
    } else {
        // Generic kernel for other kernel sizes
        for (int ky = 0; ky < kernel_size; ++ky) {
            int iy = iy_start + ky * dilation;
            for (int kx = 0; kx < kernel_size; ++kx) {
                int ix = ix_start + kx * dilation;
                if (iy >= 0 && iy < in_height && ix >= 0 && ix < in_width) {
                    max_val = fmaxf(max_val, static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy) * in_width + ix]));
                }
            }
        }
    }
    
    output[((batch_idx * channels + chan_idx) * out_height + oy) * out_width + ox] = static_cast<T>(max_val);
}

// Highly optimized kernel with shared memory for better data locality
// Each block loads a tile of input data and computes partial max pooling
template <typename T>
__global__ void max_pool2d_smem_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    extern __shared__ float smem[];
    
    // Each block loads a (TILE_H + kernel - 1) * (TILE_W + kernel - 1) tile
    const int TILE_H = 16;
    const int TILE_W = 16;
    const int TILE_IN_H = TILE_H + 1;  // For dilation=1, but we handle dilation differently
    const int TILE_IN_W = TILE_W + 1;
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int batch_idx = bz / channels;
    int chan_idx = bz % channels;
    
    // Calculate output position for this block
    int oy_start = by * TILE_H;
    int ox_start = bx * TILE_W;
    
    // Calculate input tile start position
    int iy_start = oy_start * stride - padding;
    int ix_start = ox_start * stride - padding;
    
    // Load data into shared memory
    for (int i = ty; i < TILE_IN_H; i += TILE_H) {
        for (int j = tx; j < TILE_IN_W; j += TILE_W) {
            int iy = iy_start + i * dilation;
            int ix = ix_start + j * dilation;
            
            if (iy >= 0 && iy < in_height && ix >= 0 && ix < in_width) {
                smem[i * TILE_IN_W + j] = static_cast<float>(input[((batch_idx * channels + chan_idx) * in_height + iy) * in_width + ix]);
            } else {
                smem[i * TILE_IN_W + j] = -INFINITY;
            }
        }
    }
    
    __syncthreads();
    
    // Compute max pooling
    int oy = oy_start + ty;
    int ox = ox_start + tx;
    
    if (oy < out_height && ox < out_width) {
        int local_oy = ty * stride;
        int local_ox = tx * stride;
        
        float max_val = -INFINITY;
        for (int ky = 0; ky < kernel_size; ++ky) {
            for (int kx = 0; kx < kernel_size; ++kx) {
                max_val = fmaxf(max_val, smem[(local_oy + ky) * TILE_IN_W + (local_ox + kx)]);
            }
        }
        
        output[((batch_idx * channels + chan_idx) * out_height + oy) * out_width + ox] = static_cast<T>(max_val);
    }
}

torch::Tensor max_pool2d_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int out_height = (in_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int out_width = (in_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                                input.options());
    
    const int TILE_H = 16;
    const int TILE_W = 16;
    
    dim3 blocks(
        (out_width + TILE_W - 1) / TILE_W,
        (out_height + TILE_H - 1) / TILE_H,
        batch_size * channels
    );
    dim3 threads(TILE_W, TILE_H);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "max_pool2d_cuda", ([&] {
        max_pool2d_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels,
            in_height, in_width,
            out_height, out_width,
            kernel_size, stride, padding, dilation
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='max_pool2d_ext',
    cpp_sources='torch::Tensor max_pool2d_cuda(torch::Tensor input, int kernel_size, int stride, int padding, int dilation);',
    cuda_sources=cuda_source,
    functions=['max_pool2d_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.max_pool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width, device='cuda')
    return [x]

def get_init_inputs():
    kernel_size = 2
    stride = 2
    padding = 1
    dilation = 3
    return [kernel_size, stride, padding, dilation]
