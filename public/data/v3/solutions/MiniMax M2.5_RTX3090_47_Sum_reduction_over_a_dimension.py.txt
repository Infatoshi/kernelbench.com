import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Warp-level sum using shuffle
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Optimized sum reduction over dimension 1
// Input: (batch_size, dim1, dim2) = (64, 256, 256)
// Output: (batch_size, 1, dim2) = (64, 1, 256)
//
// Grid: batch_size * (dim2 / TILE_WIDTH) blocks
// Each block computes TILE_WIDTH outputs in parallel
__global__ void sum_reduce_dim1_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    const int batch_size,
    const int dim1,
    const int dim2,
    const int tile_width
) {
    const int tile_id = blockIdx.x;
    const int batch = tile_id / (dim2 / tile_width);
    const int tile_start = (tile_id % (dim2 / tile_width)) * tile_width;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    // Each warp handles one output element
    const int col = tile_start + warp_id;
    if (col >= dim2 || batch >= batch_size) return;
    
    // Warp-level reduction: each lane sums elements
    float sum = 0.0f;
    
    // Sequential loop over dim1 with stride 32 (warp size)
    for (int i = lane_id; i < dim1; i += 32) {
        const int idx = batch * dim1 * dim2 + i * dim2 + col;
        sum += x[idx];
    }
    
    // Warp reduction
    sum = warp_reduce_sum(sum);
    
    // First lane writes result
    if (lane_id == 0) {
        const int out_idx = batch * dim2 + col;
        y[out_idx] = sum;
    }
}

torch::Tensor sum_reduce_dim1(torch::Tensor x, int reduce_dim) {
    const int batch_size = x.size(0);
    const int dim1 = x.size(1);
    const int dim2 = x.size(2);
    
    auto options = x.options();
    auto y = torch::empty({batch_size, 1, dim2}, options);
    
    // tile_width = 8 outputs per block (8 warps)
    const int tile_width = 8;
    const int num_tiles = (dim2 + tile_width - 1) / tile_width;
    const int num_blocks = batch_size * num_tiles;
    const int threads = 256;  // 8 warps
    
    sum_reduce_dim1_kernel<<<num_blocks, threads>>>(
        x.data_ptr<float>(),
        y.data_ptr<float>(),
        batch_size, dim1, dim2, tile_width
    );
    
    return y;
}

// Half precision kernel
__global__ void sum_reduce_dim1_half_kernel(
    const half* __restrict__ x,
    half* __restrict__ y,
    const int batch_size,
    const int dim1,
    const int dim2,
    const int tile_width
) {
    const int tile_id = blockIdx.x;
    const int batch = tile_id / (dim2 / tile_width);
    const int tile_start = (tile_id % (dim2 / tile_width)) * tile_width;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    const int col = tile_start + warp_id;
    if (col >= dim2 || batch >= batch_size) return;
    
    // Use float for accumulation
    float sum = 0.0f;
    
    for (int i = lane_id; i < dim1; i += 32) {
        const int idx = batch * dim1 * dim2 + i * dim2 + col;
        sum += __half2float(x[idx]);
    }
    
    sum = warp_reduce_sum(sum);
    
    if (lane_id == 0) {
        const int out_idx = batch * dim2 + col;
        y[out_idx] = __float2half(sum);
    }
}

torch::Tensor sum_reduce_dim1_half(torch::Tensor x, int reduce_dim) {
    const int batch_size = x.size(0);
    const int dim1 = x.size(1);
    const int dim2 = x.size(2);
    
    auto options = x.options();
    auto y = torch::empty({batch_size, 1, dim2}, options);
    
    const int tile_width = 8;
    const int num_tiles = (dim2 + tile_width - 1) / tile_width;
    const int num_blocks = batch_size * num_tiles;
    const int threads = 256;
    
    sum_reduce_dim1_half_kernel<<<num_blocks, threads>>>(
        reinterpret_cast<half*>(x.data_ptr<at::Half>()),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        batch_size, dim1, dim2, tile_width
    );
    
    return y;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources=r'''
torch::Tensor sum_reduce_dim1(torch::Tensor x, int reduce_dim);
torch::Tensor sum_reduce_dim1_half(torch::Tensor x, int reduce_dim);
''',
    cuda_sources=cuda_source,
    functions=['sum_reduce_dim1', 'sum_reduce_dim1_half'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if x.dtype == torch.float32:
            return ext.sum_reduce_dim1(x, self.dim)
        elif x.dtype == torch.float16 or x.dtype == torch.bfloat16:
            return ext.sum_reduce_dim1_half(x.to(torch.float16), self.dim).to(x.dtype)
        else:
            return torch.sum(x, dim=self.dim, keepdim=True)

def get_inputs():
    return [torch.randn(64, 256, 256, device='cuda')]

def get_init_inputs():
    return [1]
