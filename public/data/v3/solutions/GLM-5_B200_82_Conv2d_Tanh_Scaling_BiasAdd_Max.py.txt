import torch
import torch.nn as nn
import triton
import triton.language as tl
import math

# Fused kernel: Tanh -> Scale -> Bias Add -> MaxPool2x2
@triton.jit
def fused_tanh_scale_bias_maxpool_kernel(
    # Pointers to matrices
    input_ptr, output_ptr, bias_ptr,
    # Matrix dimensions
    batch_size, channels, height, width,
    # Strides
    in_batch_stride, in_chan_stride, in_h_stride, in_w_stride,
    out_batch_stride, out_chan_stride, out_h_stride, out_w_stride,
    # Other parameters
    scaling_factor,
    # Meta-parameters
    BLOCK_H: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    """Fused kernel for tanh -> scale -> bias -> maxpool2x2"""
    # Output dimensions
    out_height = height // 2
    out_width = width // 2
    
    # Program IDs
    pid_batch = tl.program_id(0)
    pid_chan = tl.program_id(1)
    pid_h_block = tl.program_id(2)
    
    # Bounds check for batch and channel
    if pid_batch >= batch_size or pid_chan >= channels:
        return
    
    # Load bias for this channel
    bias = tl.load(bias_ptr + pid_chan)
    
    # Compute output block start
    out_h_start = pid_h_block * BLOCK_H
    
    # Process multiple rows of output
    for h_offset in range(BLOCK_H):
        out_h = out_h_start + h_offset
        if out_h >= out_height:
            break
        
        # Compute input h positions
        in_h0 = out_h * 2
        in_h1 = out_h * 2 + 1
        
        # Process multiple columns
        out_w = tl.arange(0, BLOCK_W)
        mask_w = out_w < out_width
        
        if not tl.any(mask_w):
            continue
        
        # Compute input w positions for 2x2 block
        in_w0 = out_w * 2
        in_w1 = out_w * 2 + 1
        
        # Masks for input positions
        mask_w0 = in_w0 < width
        mask_w1 = in_w1 < width
        
        # Base pointer for this batch and channel
        base_ptr = input_ptr + pid_batch * in_batch_stride + pid_chan * in_chan_stride
        
        # Load input values for 2x2 blocks
        val00 = tl.load(base_ptr + in_h0 * in_h_stride + in_w0 * in_w_stride, 
                       mask=mask_w0 & (in_h0 < height), other=-float('inf'))
        val01 = tl.load(base_ptr + in_h0 * in_h_stride + in_w1 * in_w_stride, 
                       mask=mask_w1 & (in_h0 < height), other=-float('inf'))
        val10 = tl.load(base_ptr + in_h1 * in_h_stride + in_w0 * in_w_stride, 
                       mask=mask_w0 & (in_h1 < height), other=-float('inf'))
        val11 = tl.load(base_ptr + in_h1 * in_h_stride + in_w1 * in_w_stride, 
                       mask=mask_w1 & (in_h1 < height), other=-float('inf'))
        
        # Apply tanh, scaling, and bias
        val00 = tl.tanh(val00) * scaling_factor + bias
        val01 = tl.tanh(val01) * scaling_factor + bias
        val10 = tl.tanh(val10) * scaling_factor + bias
        val11 = tl.tanh(val11) * scaling_factor + bias
        
        # Max pooling over 2x2 block
        max_val = tl.maximum(tl.maximum(val00, val01), tl.maximum(val10, val11))
        
        # Store result
        out_offsets = (pid_batch * out_batch_stride + 
                       pid_chan * out_chan_stride + 
                       out_h * out_h_stride + 
                       out_w * out_w_stride)
        tl.store(output_ptr + out_offsets, max_val, mask=mask_w)


def fused_tanh_scale_bias_maxpool(input_tensor, bias, scaling_factor):
    """Fused operation: tanh -> scale -> bias add -> maxpool2d(2)"""
    batch_size, channels, height, width = input_tensor.shape
    out_height = height // 2
    out_width = width // 2
    
    output = torch.empty((batch_size, channels, out_height, out_width), 
                         dtype=input_tensor.dtype, device=input_tensor.device)
    
    # Grid size
    BLOCK_H = 4
    BLOCK_W = 64  # Process multiple columns at once
    
    grid = (
        batch_size,
        channels,
        (out_height + BLOCK_H - 1) // BLOCK_H
    )
    
    # Get strides
    in_strides = input_tensor.stride()
    out_strides = output.stride()
    
    fused_tanh_scale_bias_maxpool_kernel[grid](
        input_tensor, output, bias,
        batch_size, channels, height, width,
        in_strides[0], in_strides[1], in_strides[2], in_strides[3],
        out_strides[0], out_strides[1], out_strides[2], out_strides[3],
        scaling_factor,
        BLOCK_H=BLOCK_H,
        BLOCK_W=BLOCK_W,
    )
    
    return output


# Alternative: fully vectorized kernel processing all elements
@triton.jit
def fused_tanh_scale_bias_maxpool_v2_kernel(
    input_ptr, output_ptr, bias_ptr,
    batch_size, channels, height, width,
    in_batch_stride, in_chan_stride, in_h_stride, in_w_stride,
    out_batch_stride, out_chan_stride, out_h_stride, out_w_stride,
    scaling_factor,
    BLOCK_SIZE: tl.constexpr,
):
    """Fused kernel processing output elements in blocks"""
    out_height = height // 2
    out_width = width // 2
    total_out = batch_size * channels * out_height * out_width
    
    pid = tl.program_id(0)
    block_start = pid * BLOCK_SIZE
    
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < total_out
    
    # Decode linear index to (batch, channel, out_h, out_w)
    out_w = offsets % out_width
    temp = offsets // out_width
    out_h = temp % out_height
    temp = temp // out_height
    oc = temp % channels
    batch = temp // channels
    
    # Input positions for 2x2 pooling
    in_h0 = out_h * 2
    in_h1 = out_h * 2 + 1
    in_w0 = out_w * 2
    in_w1 = out_w * 2 + 1
    
    # Base pointers
    in_base = input_ptr + batch * in_batch_stride + oc * in_chan_stride
    out_base = output_ptr + batch * out_batch_stride + oc * out_chan_stride
    
    # Load bias
    bias_val = tl.load(bias_ptr + oc, mask=mask, other=0.0)
    
    # Load 2x2 input blocks
    val00 = tl.load(in_base + in_h0 * in_h_stride + in_w0 * in_w_stride, mask=mask, other=-float('inf'))
    val01 = tl.load(in_base + in_h0 * in_h_stride + in_w1 * in_w_stride, mask=mask, other=-float('inf'))
    val10 = tl.load(in_base + in_h1 * in_h_stride + in_w0 * in_w_stride, mask=mask, other=-float('inf'))
    val11 = tl.load(in_base + in_h1 * in_h_stride + in_w1 * in_w_stride, mask=mask, other=-float('inf'))
    
    # Apply tanh, scale, bias
    val00 = tl.tanh(val00) * scaling_factor + bias_val
    val01 = tl.tanh(val01) * scaling_factor + bias_val
    val10 = tl.tanh(val10) * scaling_factor + bias_val
    val11 = tl.tanh(val11) * scaling_factor + bias_val
    
    # Max pool
    max_val = tl.maximum(tl.maximum(val00, val01), tl.maximum(val10, val11))
    
    # Store
    tl.store(out_base + out_h * out_h_stride + out_w * out_w_stride, max_val, mask=mask)


def fused_tanh_scale_bias_maxpool_v2(input_tensor, bias, scaling_factor):
    """Optimized fused operation using vectorized kernel"""
    batch_size, channels, height, width = input_tensor.shape
    out_height = height // 2
    out_width = width // 2
    
    output = torch.empty((batch_size, channels, out_height, out_width), 
                         dtype=input_tensor.dtype, device=input_tensor.device)
    
    total_out = batch_size * channels * out_height * out_width
    BLOCK_SIZE = 1024
    grid = ((total_out + BLOCK_SIZE - 1) // BLOCK_SIZE,)
    
    in_strides = input_tensor.stride()
    out_strides = output.stride()
    
    fused_tanh_scale_bias_maxpool_v2_kernel[grid](
        input_tensor, output, bias,
        batch_size, channels, height, width,
        in_strides[0], in_strides[1], in_strides[2], in_strides[3],
        out_strides[0], out_strides[1], out_strides[2], out_strides[3],
        scaling_factor,
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return output


class Model(nn.Module):
    """
    A model that performs a convolution, applies tanh, scaling, adds a bias term, and then max-pools.
    Optimized with fused kernels for post-conv operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)

    def forward(self, x):
        # Use optimized convolution (PyTorch's cuDNN) and fused post-conv operations
        x = self.conv(x)
        
        # Fused: tanh -> scale -> bias add -> maxpool
        if x.is_cuda and self.max_pool.kernel_size == 2:
            x = fused_tanh_scale_bias_maxpool_v2(x, self.bias.squeeze(), self.scaling_factor)
        else:
            x = torch.tanh(x)
            x = x * self.scaling_factor
            x = x + self.bias
            x = self.max_pool(x)
        
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
