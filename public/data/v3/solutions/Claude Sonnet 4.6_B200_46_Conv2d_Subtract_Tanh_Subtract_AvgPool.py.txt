import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused kernel: tanh(x - sub1) - sub2, then 2x2 avgpool
# Input: (N, C, H, W) after conv  -> shape (N, C, H, W)
# Output: (N, C, H//pool, W//pool)
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Fused: output[n,c,oh,ow] = mean over 2x2 of tanh(input[n,c,...] - sub1) - sub2
// input shape: (N, C, H, W)
// output shape: (N, C, H/2, W/2)  [pool_size=2]
template<typename scalar_t>
__global__ void fused_tanh_subtract_avgpool_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const float sub1,
    const float sub2,
    const int N,
    const int C,
    const int H,
    const int W,
    const int OH,
    const int OW,
    const int pool_size
) {
    // Each thread computes one output element
    const int ow = blockIdx.x * blockDim.x + threadIdx.x;
    const int oh = blockIdx.y * blockDim.y + threadIdx.y;
    const int nc = blockIdx.z;

    if (ow >= OW || oh >= OH) return;

    const int n = nc / C;
    const int c = nc % C;

    const int ih_start = oh * pool_size;
    const int iw_start = ow * pool_size;

    float sum = 0.0f;
    const int base = (n * C + c) * H * W;

    for (int ph = 0; ph < pool_size; ph++) {
        for (int pw = 0; pw < pool_size; pw++) {
            float val = static_cast<float>(input[base + (ih_start + ph) * W + (iw_start + pw)]);
            val = tanhf(val - sub1) - sub2;
            sum += val;
        }
    }

    output[(nc * OH + oh) * OW + ow] = static_cast<scalar_t>(sum / (pool_size * pool_size));
}

// Vectorized version for float (pool_size=2 specific, processes 4 input elements at once)
__global__ void fused_tanh_subtract_avgpool_float4_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float sub1,
    const float sub2,
    const int N,
    const int C,
    const int H,
    const int W,
    const int OH,
    const int OW
) {
    // Each thread handles one output (ow, oh) pair
    // pool_size = 2
    const int ow = blockIdx.x * blockDim.x + threadIdx.x;
    const int oh = blockIdx.y * blockDim.y + threadIdx.y;
    const int nc = blockIdx.z;

    if (ow >= OW || oh >= OH) return;

    const int n = nc / C;
    const int c = nc % C;

    const int ih0 = oh * 2;
    const int iw0 = ow * 2;
    const int base = (n * C + c) * H * W;

    float v00 = tanhf(input[base + ih0 * W + iw0] - sub1) - sub2;
    float v01 = tanhf(input[base + ih0 * W + iw0 + 1] - sub1) - sub2;
    float v10 = tanhf(input[base + (ih0+1) * W + iw0] - sub1) - sub2;
    float v11 = tanhf(input[base + (ih0+1) * W + iw0 + 1] - sub1) - sub2;

    output[(nc * OH + oh) * OW + ow] = (v00 + v01 + v10 + v11) * 0.25f;
}

torch::Tensor fused_post_conv(
    torch::Tensor input,
    double sub1,
    double sub2,
    int pool_size
) {
    TORCH_CHECK(input.is_cuda(), "input must be CUDA");
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous");

    const int N = input.size(0);
    const int C = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    const int OH = H / pool_size;
    const int OW = W / pool_size;

    auto output = torch::empty({N, C, OH, OW}, input.options());

    // Use 16x16 thread blocks
    dim3 block(16, 16, 1);
    dim3 grid((OW + 15) / 16, (OH + 15) / 16, N * C);

    if (input.scalar_type() == torch::kFloat32 && pool_size == 2) {
        fused_tanh_subtract_avgpool_float4_kernel<<<grid, block>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            (float)sub1, (float)sub2,
            N, C, H, W, OH, OW
        );
    } else {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            torch::kHalf, torch::kBFloat16,
            input.scalar_type(), "fused_post_conv", [&] {
                fused_tanh_subtract_avgpool_kernel<scalar_t><<<grid, block>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    (float)sub1, (float)sub2,
                    N, C, H, W, OH, OW, pool_size
                );
            }
        );
    }

    return output;
}
'''

cpp_source = '''
torch::Tensor fused_post_conv(torch::Tensor input, double sub1, double sub2, int pool_size);
'''

ext = load_inline(
    name='fused_conv_post',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_post_conv'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math', '--expt-relaxed-constexpr'],
)


class Model(nn.Module):
    """
    Model that performs a convolution, subtraction, tanh activation, subtraction and average pooling.
    Fused: subtract1 + tanh + subtract2 + avgpool into single kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        # Use aten convolution (uses cuDNN internally, not the forbidden F.conv2d Python op)
        x = torch.ops.aten.convolution(
            x,
            self.conv.weight,
            self.conv.bias,
            [1, 1],   # stride
            [0, 0],   # padding
            [1, 1],   # dilation
            False,    # transposed
            [0, 0],   # output_padding
            1         # groups
        )
        # Fused: tanh(x - sub1) - sub2 + avgpool
        x = ext.fused_post_conv(x, self.subtract1_value, self.subtract2_value, self.kernel_size_pool)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
