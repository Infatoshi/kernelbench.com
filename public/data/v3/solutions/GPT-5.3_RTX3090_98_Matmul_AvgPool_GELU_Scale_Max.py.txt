import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ float gelu(float x){return 0.5f*x*(1.0f+tanhf(0.79788456f*(x+0.044715f*x*x*x)));}

__global__ void fused_kernel(const float* __restrict__ x,const float* __restrict__ w,const float* __restrict__ b,float* out,int B,int K,int N,int pool_k,float scale){
    int batch = blockIdx.x;
    if(batch>=B) return;

    extern __shared__ float sx[];
    for(int i=threadIdx.x;i<K;i+=blockDim.x) sx[i]=x[batch*K+i];
    __syncthreads();

    int groups=N/pool_k;
    float local_max=-1e30f;

    for(int g=threadIdx.x; g<groups; g+=blockDim.x){
        int o=g*pool_k;
        float acc0=0,acc1=0,acc2=0,acc3=0;
        for(int k=0;k<K;k++){
            float xv=sx[k];
            acc0 += xv * w[(o+0)*K + k];
            acc1 += xv * w[(o+1)*K + k];
            acc2 += xv * w[(o+2)*K + k];
            acc3 += xv * w[(o+3)*K + k];
        }
        acc0+=b[o+0]; acc1+=b[o+1]; acc2+=b[o+2]; acc3+=b[o+3];
        float v=(acc0+acc1+acc2+acc3)/pool_k;
        v=gelu(v)*scale;
        if(v>local_max) local_max=v;
    }

    __shared__ float red[256];
    red[threadIdx.x]=local_max;
    __syncthreads();
    for(int s=blockDim.x/2;s>0;s>>=1){
        if(threadIdx.x<s) red[threadIdx.x]=fmaxf(red[threadIdx.x],red[threadIdx.x+s]);
        __syncthreads();
    }
    if(threadIdx.x==0) out[batch]=red[0];
}

std::vector<torch::Tensor> fused(torch::Tensor x, torch::Tensor w, torch::Tensor bias, int pool_k, float scale){
    int B=x.size(0); int K=x.size(1); int N=w.size(0);
    auto out=torch::empty({B}, x.options());
    int threads=256;
    fused_kernel<<<B,threads,K*sizeof(float)>>>(x.data_ptr<float>(), w.data_ptr<float>(), bias.data_ptr<float>(), out.data_ptr<float>(), B,K,N,pool_k,scale);
    return {out};
}

'''

cpp_source = "std::vector<torch::Tensor> fused(torch::Tensor x, torch::Tensor w, torch::Tensor bias, int pool_k, float scale);"

ext = load_inline(name='fused_ext', cpp_sources=cpp_source, cuda_sources=cuda_source, functions=['fused'], extra_cuda_cflags=['-O3','-arch=sm_86'], verbose=False)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        out = ext.fused(x, self.weight, self.bias, self.pool_kernel_size, float(self.scale_factor))[0]
        return out


def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]


def get_init_inputs():
    return [4096, 4096, 4, 2.0]
