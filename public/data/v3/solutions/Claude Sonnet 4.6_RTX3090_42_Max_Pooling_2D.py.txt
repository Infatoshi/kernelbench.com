cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

// -----------------------------------------------------------------------
// Generic max-pool 2D kernel (float32)
// Each thread computes one output element.
// Grid: (ceil(OW/BX)*ceil(OH/BY), C, N)  where BX=32, BY=8
// -----------------------------------------------------------------------
template <int KH, int KW>
__global__ void maxpool2d_float_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int N, int C, int IH, int IW,
    int OH, int OW,
    int stride_h, int stride_w,
    int pad_h,    int pad_w,
    int dil_h,    int dil_w)
{
    // output pixel
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int c  = blockIdx.z % C;
    int n  = blockIdx.z / C;

    if (ow >= OW || oh >= OH) return;

    const int in_base = n * C * IH * IW + c * IH * IW;
    float maxval = -FLT_MAX;

    int ih_start = oh * stride_h - pad_h;
    int iw_start = ow * stride_w - pad_w;

    if constexpr (KH > 0 && KW > 0) {
        #pragma unroll
        for (int kh = 0; kh < KH; ++kh) {
            int ih = ih_start + kh * dil_h;
            if (ih >= 0 && ih < IH) {
                #pragma unroll
                for (int kw = 0; kw < KW; ++kw) {
                    int iw = iw_start + kw * dil_w;
                    if (iw >= 0 && iw < IW) {
                        float v = input[in_base + ih * IW + iw];
                        maxval = fmaxf(maxval, v);
                    }
                }
            }
        }
    }

    output[n * C * OH * OW + c * OH * OW + oh * OW + ow] = maxval;
}

        }

}

// -----------------------------------------------------------------------
// Half-precision (fp16) max-pool kernel
// -----------------------------------------------------------------------
#include <cuda_fp16.h>

template <int KH, int KW>
__global__ void maxpool2d_half_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    int N, int C, int IH, int IW,
    int OH, int OW,
    int stride_h, int stride_w,
    int pad_h,    int pad_w,
    int dil_h,    int dil_w)
{
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int c  = blockIdx.z % C;
    int n  = blockIdx.z / C;

    if (ow >= OW || oh >= OH) return;

    const int in_base = n * C * IH * IW + c * IH * IW;
    __half maxval = __float2half(-65504.0f);

    int ih_start = oh * stride_h - pad_h;
    int iw_start = ow * stride_w - pad_w;

    if constexpr (KH > 0 && KW > 0) {
        #pragma unroll
        for (int kh = 0; kh < KH; ++kh) {
            int ih = ih_start + kh * dil_h;
            if (ih >= 0 && ih < IH) {
                #pragma unroll
                for (int kw = 0; kw < KW; ++kw) {
                    int iw = iw_start + kw * dil_w;
                    if (iw >= 0 && iw < IW) {
                        __half v = input[in_base + ih * IW + iw];
                        maxval = __hgt(v, maxval) ? v : maxval;
                    }
                }
            }
        }
    }

    output[n * C * OH * OW + c * OH * OW + oh * OW + ow] = maxval;
}

// -----------------------------------------------------------------------
// Dispatcher
// -----------------------------------------------------------------------
torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_size, int stride, int padding, int dilation)
{
    // Support square kernels only for now (kernel_size x kernel_size)
    const int N  = input.size(0);
    const int C  = input.size(1);
    const int IH = input.size(2);
    const int IW = input.size(3);

    const int KH = kernel_size;
    const int KW = kernel_size;
    const int SH = stride;
    const int SW = stride;
    const int PH = padding;
    const int PW = padding;
    const int DH = dilation;
    const int DW = dilation;

    // output dims
    int OH = (IH + 2 * PH - DH * (KH - 1) - 1) / SH + 1;
    int OW = (IW + 2 * PW - DW * (KW - 1) - 1) / SW + 1;

    auto output = torch::empty({N, C, OH, OW}, input.options());

    // 32x8 threads per block, map to OW x OH
    const int BX = 32;
    const int BY = 8;
    dim3 block(BX, BY);
    dim3 grid(
        (OW + BX - 1) / BX,
        (OH + BY - 1) / BY,
        N * C
    );

    auto dtype = input.scalar_type();

    if (dtype == torch::kFloat32) {
        const float* inp = input.data_ptr<float>();
        float* out = output.data_ptr<float>();
        if (KH == 2 && KW == 2) {
            maxpool2d_float_kernel<2,2><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else if (KH == 3 && KW == 3) {
            maxpool2d_float_kernel<3,3><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else if (KH == 4 && KW == 4) {
            maxpool2d_float_kernel<4,4><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else if (KH == 5 && KW == 5) {
            maxpool2d_float_kernel<5,5><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else {
            maxpool2d_float_generic_kernel<<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, KH, KW, SH, SW, PH, PW, DH, DW);
        }
    } else if (dtype == torch::kFloat16) {
        const __half* inp = reinterpret_cast<const __half*>(input.data_ptr<at::Half>());
        __half* out = reinterpret_cast<__half*>(output.data_ptr<at::Half>());
        if (KH == 2 && KW == 2) {
            maxpool2d_half_kernel<2,2><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else if (KH == 3 && KW == 3) {
            maxpool2d_half_kernel<3,3><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else {
            maxpool2d_half_generic_kernel<<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, KH, KW, SH, SW, PH, PW, DH, DW);
        }
    } else if (dtype == torch::kBFloat16) {
        const __nv_bfloat16* inp = reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>());
        __nv_bfloat16* out = reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>());
        if (KH == 2 && KW == 2) {
            maxpool2d_bf16_kernel<2,2><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else if (KH == 3 && KW == 3) {
            maxpool2d_bf16_kernel<3,3><<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, SH, SW, PH, PW, DH, DW);
        } else {
            maxpool2d_bf16_generic_kernel<<<grid, block>>>(inp, out, N, C, IH, IW, OH, OW, KH, KW, SH, SW, PH, PW, DH, DW);
        }
    } else {
        TORCH_CHECK(false, "maxpool2d_cuda: unsupported dtype");
    }

    return output;
}

cpp_source = r'''
torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_size, int stride, int padding, int dilation);
'''

ext = load_inline(
    name='maxpool2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['maxpool2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math', '-std=c++17'],
)


class Model(nn.Module):
    """
    Optimized Max Pooling 2D using a custom CUDA kernel.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)
