
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <cuda_fp16.h>
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void gemv_int8_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ w_q,
    float* __restrict__ y_fp,
    float* __restrict__ out_half,
    float x_scale,
    float w_scale,
    int M,
    int N,
    int K) {
    
    // For M=32, N=14336, K=4096, M is small (one row of X at a time)
    // We can assign multiple CTAs per row of output or just use 1 CTA per row if N is large.
    // Currently 1 block per (row_x, row_w) pair. M*N = 32 * 14336 = 458752 blocks.
    
    int row_x = blockIdx.x;
    int row_w = blockIdx.y;
    
    if (row_x >= M || row_w >= N) return;

    int local_sum = 0;
    const int* x_vec = (const int*)(x_q + row_x * K);
    const int* w_vec = (const int*)(w_q + row_w * K);
    
    int k_limit = K / 4;
    for (int k = threadIdx.x; k < k_limit; k += blockDim.x) {
        local_sum = __dp4a(x_vec[k], w_vec[k], local_sum);
    }

    extern __shared__ int s_mem[];
    s_mem[threadIdx.x] = local_sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            s_mem[threadIdx.x] += s_mem[threadIdx.x + s];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        float res = (float)s_mem[0] * x_scale * w_scale;
        ((__half*)out_half)[row_x * N + row_w] = __float2half(res);
    }
}

torch::Tensor gemv_int8(torch::Tensor x_q, torch::Tensor w_q, float x_scale, float w_scale) {
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = w_q.size(0);
    
    auto y = torch::empty({M, N}, torch::TensorOptions().dtype(torch::kHalf).device(x_q.device()));
    
    dim3 threads(256);
    dim3 blocks(M, N);
    
    gemv_int8_kernel<<<blocks, threads, threads.x * sizeof(int)>>>(
        x_q.data_ptr<int8_t>(),
        w_q.data_ptr<int8_t>(),
        nullptr,
        (float*)y.data_ptr<at::Half>(),
        x_scale,
        w_scale,
        M, N, K
    );
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor gemv_int8(torch::Tensor x_q, torch::Tensor w_q, float x_scale, float w_scale);
'''

ext = load_inline(
    name='gemv_fp8_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_int8'],
    verbose=False,
    extra_cuda_cflags=['-O3']
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # x_q: [32, 4096], weight_q: [14336, 4096]
        return ext.gemv_int8(x_q, self.weight_q, x_scale.item(), self.weight_scale.item())

def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]

def get_init_inputs():
    return []
