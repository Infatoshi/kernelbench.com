import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def silu(x):
    return x * tl.sigmoid(x)


@triton.jit
def moe_gate_up_fused_kernel(
    x_ptr,           # (num_tokens, hidden_size)
    gate_ptr,        # (num_experts, intermediate_size, hidden_size)
    up_ptr,          # (num_experts, intermediate_size, hidden_size)
    expert_indices_ptr,  # (num_slots) - sorted by expert
    sorted_token_ids_ptr,  # (num_slots) - maps slot to original token
    gate_up_out_ptr, # (num_slots, intermediate_size)
    num_slots: tl.constexpr,
    hidden_size: tl.constexpr,
    intermediate_size: tl.constexpr,
    num_experts: tl.constexpr,
    BLOCK_SIZE_H: tl.constexpr,
    BLOCK_SIZE_I: tl.constexpr,
):
    """
    Fused kernel: for each token_slot, compute gate_proj and up_proj, apply SiLU and multiply.
    Each program handles one token_slot and computes the full intermediate vector.
    Uses block-wise computation over hidden and intermediate dimensions.
    """
    slot_id = tl.program_id(0)
    
    # Get expert and original token for this slot
    expert_id = tl.load(expert_indices_ptr + slot_id)
    token_id = tl.load(sorted_token_ids_ptr + slot_id)
    
    # Pointers for this expert's weights
    gate_expert_ptr = gate_ptr + expert_id * intermediate_size * hidden_size
    up_expert_ptr = up_ptr + expert_id * intermediate_size * hidden_size
    
    # Output pointer for this slot
    out_ptr = gate_up_out_ptr + slot_id * intermediate_size
    
    # Process intermediate in blocks
    for i_start in range(0, intermediate_size, BLOCK_SIZE_I):
        i_offsets = i_start + tl.arange(0, BLOCK_SIZE_I)
        i_mask = i_offsets < intermediate_size
        
        # Accumulate dot product for this block of intermediate dims
        acc_gate = tl.zeros([BLOCK_SIZE_I], dtype=tl.float32)
        acc_up = tl.zeros([BLOCK_SIZE_I], dtype=tl.float32)
        
        # Process hidden in blocks
        for h_start in range(0, hidden_size, BLOCK_SIZE_H):
            h_offsets = h_start + tl.arange(0, BLOCK_SIZE_H)
            h_mask = h_offsets < hidden_size
            
            # Load x values for this token
            x_ptr_token = x_ptr + token_id * hidden_size
            x_block = tl.load(x_ptr_token + h_offsets, mask=h_mask, other=0.0)
            
            # Load gate weights for this block of intermediate rows and hidden cols
            # gate_ptr shape: (intermediate_size, hidden_size) for this expert
            gate_row_ptrs = gate_expert_ptr + (i_offsets[:, None] * hidden_size + h_offsets[None, :])
            gate_block = tl.load(gate_row_ptrs, 
                                 mask=(i_offsets[:, None] < intermediate_size) & (h_offsets[None, :] < hidden_size), 
                                 other=0.0)
            
            # Load up weights
            up_row_ptrs = up_expert_ptr + (i_offsets[:, None] * hidden_size + h_offsets[None, :])
            up_block = tl.load(up_row_ptrs, 
                               mask=(i_offsets[:, None] < intermediate_size) & (h_offsets[None, :] < hidden_size), 
                               other=0.0)
            
            # Accumulate: acc += x @ weights.T
            acc_gate += tl.sum(x_block[None, :] * gate_block, axis=1)
            acc_up += tl.sum(x_block[None, :] * up_block, axis=1)
        
        # Apply SiLU to gate and multiply with up
        gate_val = silu(acc_gate)
        result = gate_val * acc_up
        
        # Store result
        tl.store(out_ptr + i_offsets, result, mask=i_mask)


@triton.jit
def moe_down_scatter_kernel(
    gate_up_out_ptr,   # (num_slots, intermediate_size)
    down_ptr,          # (num_experts, hidden_size, intermediate_size)
    expert_indices_ptr,  # (num_slots)
    weights_ptr,       # (num_slots)
    sorted_token_ids_ptr,  # (num_slots)
    output_ptr,        # (num_tokens, hidden_size)
    num_slots: tl.constexpr,
    hidden_size: tl.constexpr,
    intermediate_size: tl.constexpr,
    BLOCK_SIZE_I: tl.constexpr,
    BLOCK_SIZE_H: tl.constexpr,
):
    """
    Compute down projection and scatter-add with weights.
    Each program handles one slot and computes full hidden output, then atomic-adds.
    """
    slot_id = tl.program_id(0)
    
    # Get expert, weight, and original token for this slot
    expert_id = tl.load(expert_indices_ptr + slot_id)
    weight = tl.load(weights_ptr + slot_id)
    token_id = tl.load(sorted_token_ids_ptr + slot_id)
    
    # Down weights pointer for this expert: shape (hidden_size, intermediate_size)
    down_expert_ptr = down_ptr + expert_id * hidden_size * intermediate_size
    
    # Input pointer (gate_up output)
    in_ptr = gate_up_out_ptr + slot_id * intermediate_size
    
    # Output pointer for atomic add
    out_ptr = output_ptr + token_id * hidden_size
    
    # Load intermediate values
    i_offsets = tl.arange(0, BLOCK_SIZE_I)
    i_mask = i_offsets < intermediate_size
    intermediate = tl.load(in_ptr + i_offsets, mask=i_mask, other=0.0).to(tl.float32)
    
    # Process hidden in blocks
    for h_start in range(0, hidden_size, BLOCK_SIZE_H):
        h_offsets = h_start + tl.arange(0, BLOCK_SIZE_H)
        h_mask = h_offsets < hidden_size
        
        # Load down weights for this block of hidden rows
        down_row_ptrs = down_expert_ptr + (h_offsets[:, None] * intermediate_size + i_offsets[None, :])
        down_block = tl.load(down_row_ptrs, 
                             mask=(h_offsets[:, None] < hidden_size) & (i_offsets[None, :] < intermediate_size), 
                             other=0.0)
        
        # Compute output: sum_i(intermediate[i] * down[h, i])
        result = tl.sum(intermediate[None, :] * down_block, axis=1) * weight
        
        # Atomic add to output
        tl.atomic_add(out_ptr + h_offsets, result, mask=h_mask)


class Model(nn.Module):
    def __init__(
        self,
        hidden_size: int,
        intermediate_size: int,
        num_experts: int,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.gate_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.up_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.down_proj = nn.Parameter(
            torch.randn(num_experts, hidden_size, intermediate_size) * 0.02
        )
        torch.random.set_rng_state(rng_state)

    def forward(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        batch, seq_len, _ = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len
        num_slots = num_tokens * top_k

        x_flat = x.view(num_tokens, self.hidden_size).contiguous()
        indices_flat = expert_indices.view(num_slots).contiguous()
        weights_flat = expert_weights.view(num_slots).contiguous()

        # Create token indices for each (token, slot) pair
        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids_ex = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1).contiguous()

        # Sort by expert to enable batched processing
        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids_ex[sort_order].contiguous()
        sorted_weights = weights_flat[sort_order].contiguous()
        sorted_indices = sorted_expert_idx.contiguous()

        # Allocate output for gate_up
        gate_up_out = torch.empty(num_slots, self.intermediate_size, device=x.device, dtype=x.dtype)

        # Block sizes - tune for RTX 3090
        BLOCK_H = 256
        BLOCK_I = 128
        
        # Launch gate_up kernel - one program per slot
        grid_gate_up = (num_slots,)
        moe_gate_up_fused_kernel[grid_gate_up](
            x_flat,
            self.gate_proj,
            self.up_proj,
            sorted_indices,
            sorted_token_ids,
            gate_up_out,
            num_slots,
            self.hidden_size,
            self.intermediate_size,
            self.num_experts,
            BLOCK_SIZE_H=BLOCK_H,
            BLOCK_SIZE_I=BLOCK_I,
        )

        # Allocate output buffer
        output = torch.zeros(num_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
        
        # Launch down kernel
        grid_down = (num_slots,)
        moe_down_scatter_kernel[grid_down](
            gate_up_out,
            self.down_proj,
            sorted_indices,
            sorted_weights,
            sorted_token_ids,
            output,
            num_slots,
            self.hidden_size,
            self.intermediate_size,
            BLOCK_SIZE_I=BLOCK_I,
            BLOCK_SIZE_H=BLOCK_H,
        )

        return output.view(batch, seq_len, self.hidden_size)


# Mixtral-style configuration
batch_size = 4
seq_len = 2048
hidden_size = 4096
intermediate_size = 14336
num_experts = 8
top_k = 2


def get_inputs():
    x = torch.randn(batch_size, seq_len, hidden_size)
    expert_indices = torch.stack([
        torch.randperm(num_experts)[:top_k]
        for _ in range(batch_size * seq_len)
    ]).view(batch_size, seq_len, top_k)
    expert_weights = F.softmax(torch.randn(batch_size, seq_len, top_k), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return [hidden_size, intermediate_size, num_experts]
