import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused CUDA kernel: Conv2d + Activation + BatchNorm

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__device__ __forceinline__ T fused_activation(T x) {
    T sp = x;
    if (x > (T)20.0) {
        sp = x;
    } else if (x < (T)-20.0) {
        sp = (T)0.0;
    } else {
        sp = log1p(exp(x));
    }
    T t = tanh(sp);
    return x * t;
}

// Grid: (out_width * out_height, out_channels, batch_size)
template <typename T>
__global__ void fused_conv_act_bn_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    const T* __restrict__ bn_w,
    const T* __restrict__ bn_b,
    const T* __restrict__ bn_m,
    const T* __restrict__ bn_v,
    T* __restrict__ output,
    int batch_size,
    int in_channels,
    int out_channels,
    int height,
    int width,
    int kernel_size,
    int out_height,
    int out_width,
    int stride,
    int padding,
    int groups,
    float eps
) {
    // Linearize output position
    int out_pos = blockIdx.x * blockDim.x + threadIdx.x;
    if (out_pos >= batch_size * out_channels * out_height * out_width) return;
    
    // Decode position
    int b = out_pos / (out_channels * out_height * out_width);
    int c_out = (out_pos / (out_height * out_width)) % out_channels;
    int out_y = (out_pos / out_width) % out_height;
    int out_x = out_pos % out_width;
    
    int out_idx = b * (out_channels * out_height * out_width) + 
                  c_out * (out_height * out_width) + out_y * out_width + out_x;
    
    int in_x_base = out_x * stride - padding;
    int in_y_base = out_y * stride - padding;
    
    T sum = (T)0.0;
    int in_c_per_group = in_channels / groups;
    int group_id = c_out / (out_channels / groups);
    int group_start = group_id * in_c_per_group;
    
    for (int c_in = 0; c_in < in_c_per_group; c_in++) {
        int actual_c_in = group_start + c_in;
        int weight_idx = c_out * (in_channels * kernel_size * kernel_size) + 
                        actual_c_in * (kernel_size * kernel_size);
        
        for (int ky = 0; ky < kernel_size; ky++) {
            int in_y = in_y_base + ky;
            if (in_y < 0 || in_y >= height) continue;
            
            for (int kx = 0; kx < kernel_size; kx++) {
                int in_x = in_x_base + kx;
                if (in_x < 0 || in_x >= width) continue;
                
                int in_idx = b * (in_channels * height * width) + 
                            actual_c_in * (height * width) + 
                            in_y * width + in_x;
                T w = weight[weight_idx + ky * kernel_size + kx];
                sum += input[in_idx] * w;
            }
        }
    }
    
    if (bias != nullptr) {
        sum += bias[c_out];
    }
    
    sum = fused_activation(sum);
    
    T m = bn_m[c_out];
    T v = bn_v[c_out];
    T std_val = sqrt(v + (T)eps);
    T wt = bn_w[c_out];
    T b = bn_b[c_out];
    
    T normalized = (sum - m) / std_val;
    sum = normalized * wt + b;
    
    output[out_idx] = sum;
}

torch::Tensor fused_conv_act_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    int stride,
    int padding,
    int groups,
    double eps
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_height = (height + 2 * padding - kernel_size) / stride + 1;
    int out_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_height, out_width}, input.options());
    
    int total_outputs = batch_size * out_channels * out_height * out_width;
    int block_size = 256;
    int grid_size = (total_outputs + block_size - 1) / block_size;
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_conv_act_bn", ([&] {
        fused_conv_act_bn_kernel<scalar_t><<<grid_size, block_size>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.numel() > 0 ? bias.data_ptr<scalar_t>() : nullptr,
            bn_weight.data_ptr<scalar_t>(),
            bn_bias.data_ptr<scalar_t>(),
            bn_mean.data_ptr<scalar_t>(),
            bn_var.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, in_channels, out_channels, height, width,
            kernel_size, out_height, out_width,
            stride, padding, groups, (float)eps
        );
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_conv_act_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    int stride,
    int padding,
    int groups,
    double eps
);
'''

ext = load_inline(
    name='fused_conv_act_bn',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_act_bn'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = 1
        self.padding = 1
        self.groups = 1
        self.eps = eps
        
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, 
                              stride=1, padding=1, bias=False)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        
    def forward(self, x):
        if not self.training:
            weight = self.conv.weight
            bias = self.conv.bias if self.conv.bias is not None else torch.zeros(
                self.out_channels, dtype=x.dtype, device=x.device)
            
            bn_weight = self.bn.weight
            bn_bias = self.bn.bias
            bn_mean = self.bn.running_mean
            bn_var = self.bn.running_var
            
            output = ext.fused_conv_act_bn(
                x, weight, bias,
                bn_weight, bn_bias, bn_mean, bn_var,
                self.stride, self.padding, self.groups, self.eps
            )
            
            return output
        else:
            x = self.conv(x)
            x = torch.multiply(torch.tanh(torch.nn.functional.softplus(x)), x)
            x = self.bn(x)
            return x


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width, dtype=torch.float32)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    return [in_channels, out_channels, kernel_size]
