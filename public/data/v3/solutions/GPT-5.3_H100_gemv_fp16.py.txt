import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_fp16.h>

#define TILE_K 256

__global__ void gemv_fp16_kernel(
    const half* __restrict__ x,   // [B, K]
    const half* __restrict__ w,   // [M, K]
    half* __restrict__ y,         // [B, M]
    int B, int M, int K)
{
    int b = blockIdx.x;
    int m = blockIdx.y * blockDim.x + threadIdx.x;

    if (b >= B || m >= M) return;

    extern __shared__ half x_tile[];

    float acc = 0.0f;

    for (int k0 = 0; k0 < K; k0 += TILE_K) {
        int k_idx = k0 + threadIdx.x;
        if (threadIdx.x < TILE_K && k_idx < K) {
            x_tile[threadIdx.x] = x[b * K + k_idx];
        }
        __syncthreads();

        int tile_size = min(TILE_K, K - k0);
        const half* w_row = w + m * K + k0;

        for (int kk = 0; kk < tile_size; ++kk) {
            acc += __half2float(x_tile[kk]) * __half2float(w_row[kk]);
        }
        __syncthreads();
    }

    y[b * M + m] = __float2half(acc);
}


torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w) {
    int B = x.size(0);
    int K = x.size(1);
    int M = w.size(0);

    auto y = torch::empty({B, M}, x.options());

    dim3 block(256);
    dim3 grid(B, (M + 255) / 256);

    size_t shm = TILE_K * sizeof(half);

    gemv_fp16_kernel<<<grid, block, shm>>>(
        reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
        reinterpret_cast<const half*>(w.data_ptr<at::Half>()),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        B, M, K);

    return y;
}
'''

ext = load_inline(
    name='gemv_fp16_ext',
    cpp_sources='torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w);',
    cuda_sources=cuda_source,
    functions=['gemv_fp16'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16).contiguous()
        w = self.weight.contiguous()
        return ext.gemv_fp16(x_fp16, w)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
