import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// RMSNorm: input shape (B, C, H*W)
// Key insight: x[b, c, hw] with hw varying fastest.
// For a given hw, we need x[b, 0, hw], x[b, 1, hw], ..., x[b, C-1, hw]
// These are strided in memory (stride = HW).
//
// Strategy: Use a 2D tile where:
//   - blockDim.x threads handle adjacent hw positions (coalesced reads)
//   - We loop over C channels
//   - Each thread accumulates its own sum_sq independently
//
// For float4: process 4 adjacent hw positions per thread -> even better BW utilization

// Main kernel: float4 vectorized, each thread handles 4 adjacent spatial positions
__global__ void rmsnorm_f32_vec4(
    const float* __restrict__ x,
    float* __restrict__ y,
    int B, int C, int HW,
    float eps
) {
    // hw4 index: each thread handles 4 consecutive hw positions
    int hw4 = blockIdx.x * blockDim.x + threadIdx.x;
    int b   = blockIdx.y;
    int hw_base = hw4 * 4;
    
    if (hw_base >= HW) return;
    
    const float* x_b = x + b * C * HW;
    float* y_b = y + b * C * HW;
    
    float4 sums = {0.f, 0.f, 0.f, 0.f};
    
    for (int c = 0; c < C; c++) {
        float4 v = *reinterpret_cast<const float4*>(x_b + c * HW + hw_base);
        sums.x += v.x * v.x;
        sums.y += v.y * v.y;
        sums.z += v.z * v.z;
        sums.w += v.w * v.w;
    }
    
    float inv_c = 1.0f / (float)C;
    float rinv_x = rsqrtf(sums.x * inv_c + eps);
    float rinv_y = rsqrtf(sums.y * inv_c + eps);
    float rinv_z = rsqrtf(sums.z * inv_c + eps);
    float rinv_w = rsqrtf(sums.w * inv_c + eps);
    
    for (int c = 0; c < C; c++) {
        int off = c * HW + hw_base;
        float4 v = *reinterpret_cast<const float4*>(x_b + off);
        float4 r;
        r.x = v.x * rinv_x;
        r.y = v.y * rinv_y;
        r.z = v.z * rinv_z;
        r.w = v.w * rinv_w;
        *reinterpret_cast<float4*>(y_b + off) = r;
    }
}

// FP16 version: half2 vectorized
__global__ void rmsnorm_f16_vec2(
    const __half* __restrict__ x,
    __half* __restrict__ y,
    int B, int C, int HW,
    float eps
) {
    int hw2 = blockIdx.x * blockDim.x + threadIdx.x;
    int b   = blockIdx.y;
    int hw_base = hw2 * 2;
    
    if (hw_base >= HW) return;
    
    const __half* x_b = x + b * C * HW;
    __half* y_b = y + b * C * HW;
    
    float2 sums = {0.f, 0.f};
    
    for (int c = 0; c < C; c++) {
        __half2 v = *reinterpret_cast<const __half2*>(x_b + c * HW + hw_base);
        float2 vf = __half22float2(v);
        sums.x += vf.x * vf.x;
        sums.y += vf.y * vf.y;
    }
    
    float inv_c = 1.0f / (float)C;
    float rinv_x = rsqrtf(sums.x * inv_c + eps);
    float rinv_y = rsqrtf(sums.y * inv_c + eps);
    
    for (int c = 0; c < C; c++) {
        int off = c * HW + hw_base;
        __half2 v = *reinterpret_cast<const __half2*>(x_b + off);
        float2 vf = __half22float2(v);
        vf.x *= rinv_x;
        vf.y *= rinv_y;
        *reinterpret_cast<__half2*>(y_b + off) = __float22half2_rn(vf);
    }
}

// BF16 version using float
__global__ void rmsnorm_bf16_vec2(
    const __nv_bfloat16* __restrict__ x,
    __nv_bfloat16* __restrict__ y,
    int B, int C, int HW,
    float eps
) {
    int hw2 = blockIdx.x * blockDim.x + threadIdx.x;
    int b   = blockIdx.y;
    int hw_base = hw2 * 2;
    
    if (hw_base >= HW) return;
    
    const __nv_bfloat16* x_b = x + b * C * HW;
    __nv_bfloat16* y_b = y + b * C * HW;
    
    float2 sums = {0.f, 0.f};
    
    for (int c = 0; c < C; c++) {
        __nv_bfloat162 v = *reinterpret_cast<const __nv_bfloat162*>(x_b + c * HW + hw_base);
        float2 vf = __bfloat1622float2(v);
        sums.x += vf.x * vf.x;
        sums.y += vf.y * vf.y;
    }
    
    float inv_c = 1.0f / (float)C;
    float rinv_x = rsqrtf(sums.x * inv_c + eps);
    float rinv_y = rsqrtf(sums.y * inv_c + eps);
    
    for (int c = 0; c < C; c++) {
        int off = c * HW + hw_base;
        __nv_bfloat162 v = *reinterpret_cast<const __nv_bfloat162*>(x_b + off);
        float2 vf = __bfloat1622float2(v);
        vf.x *= rinv_x;
        vf.y *= rinv_y;
        *reinterpret_cast<__nv_bfloat162*>(y_b + off) = __float22bfloat162_rn(vf);
    }
}

// Generic scalar fallback
template<typename scalar_t>
__global__ void rmsnorm_generic(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int B, int C, int HW,
    float eps
) {
    int hw = blockIdx.x * blockDim.x + threadIdx.x;
    int b  = blockIdx.y;
    
    if (hw >= HW) return;
    
    const scalar_t* x_b = x + b * C * HW;
    scalar_t* y_b = y + b * C * HW;
    
    float sum_sq = 0.f;
    for (int c = 0; c < C; c++) {
        float v = (float)x_b[c * HW + hw];
        sum_sq += v * v;
    }
    
    float rinv = rsqrtf(sum_sq / (float)C + eps);
    
    for (int c = 0; c < C; c++) {
        int off = c * HW + hw;
        y_b[off] = (scalar_t)((float)x_b[off] * rinv);
    }
}


torch::Tensor rmsnorm_cuda(torch::Tensor x, int num_features, float eps) {
    TORCH_CHECK(x.is_contiguous(), "Input must be contiguous");
    auto y = torch::empty_like(x);
    
    int B = x.size(0);
    int C = x.size(1);
    int HW = x.numel() / (B * C);
    
    const int threads = 256;
    
    if (x.scalar_type() == at::ScalarType::Float && HW % 4 == 0) {
        int hw4 = HW / 4;
        dim3 grid((hw4 + threads - 1) / threads, B);
        rmsnorm_f32_vec4<<<grid, threads>>>(
            x.data_ptr<float>(), y.data_ptr<float>(),
            B, C, HW, eps
        );
    } else if (x.scalar_type() == at::ScalarType::Half && HW % 2 == 0) {
        int hw2 = HW / 2;
        dim3 grid((hw2 + threads - 1) / threads, B);
        rmsnorm_f16_vec2<<<grid, threads>>>(
            reinterpret_cast<const __half*>(x.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(y.data_ptr<at::Half>()),
            B, C, HW, eps
        );
    } else if (x.scalar_type() == at::ScalarType::BFloat16 && HW % 2 == 0) {
        int hw2 = HW / 2;
        dim3 grid((hw2 + threads - 1) / threads, B);
        rmsnorm_bf16_vec2<<<grid, threads>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()),
            B, C, HW, eps
        );
    } else {
        dim3 grid((HW + threads - 1) / threads, B);
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half, at::ScalarType::BFloat16,
            x.scalar_type(), "rmsnorm_generic", [&]() {
                rmsnorm_generic<scalar_t><<<grid, threads>>>(
                    x.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(),
                    B, C, HW, eps
                );
            }
        );
    }
    
    return y;
}
'''

cpp_source = '''
torch::Tensor rmsnorm_cuda(torch::Tensor x, int num_features, float eps);
'''

ext = load_inline(
    name='rmsnorm_ext_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rmsnorm_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_cuda(x, self.num_features, self.eps)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256


def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [features]
