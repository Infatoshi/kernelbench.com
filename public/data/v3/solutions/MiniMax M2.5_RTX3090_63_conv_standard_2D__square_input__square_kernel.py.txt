import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Optimized im2col CUDA kernel
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void im2col_kernel(
    const float* input,
    float* output,
    int batch_size,
    int channels,
    int height,
    int width,
    int kernel_size,
    int output_height,
    int output_width,
    int stride,
    int padding,
    int dilation
) {
    const int k = kernel_size;
    const int col_size = k * k * channels;
    const int num_threads = batch_size * output_height * output_width;
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    if (tid >= num_threads) return;
    
    int b = tid / (output_height * output_width);
    int idx = tid % (output_height * output_width);
    int out_h = idx / output_width;
    int out_w = idx % output_width;
    
    int in_h_base = out_h * stride - padding;
    int in_w_base = out_w * stride - padding;
    
    float* col_ptr = output + tid * col_size;
    
    for (int c = 0; c < channels; ++c) {
        for (int kh = 0; kh < k; ++kh) {
            for (int kw = 0; kw < k; ++kw) {
                int in_h = in_h_base + kh * dilation;
                int in_w = in_w_base + kw * dilation;
                
                if (in_h >= 0 && in_h < height && in_w >= 0 && in_w < width) {
                    int input_idx = ((b * channels + c) * height + in_h) * width + in_w;
                    *col_ptr = input[input_idx];
                } else {
                    *col_ptr = 0.0f;
                }
                col_ptr++;
            }
        }
    }
}

torch::Tensor im2col_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    
    int output_height = (height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int output_width = (width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    int col_size = kernel_size * kernel_size * channels;
    int num_cols = batch_size * output_height * output_width;
    
    auto output = torch::zeros({num_cols, col_size}, input.options());
    
    int num_threads = batch_size * output_height * output_width;
    int block_size = 256;
    int grid_size = (num_threads + block_size - 1) / block_size;
    
    im2col_kernel<<<grid_size, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, channels, height, width,
        kernel_size, output_height, output_width,
        stride, padding, dilation
    );
    
    return output.view({batch_size, output_height, output_width, col_size});
}
'''

cpp_source = r'''
torch::Tensor im2col_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
);
'''

# Load the inline extension
ext = load_inline(
    name='im2col_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['im2col_cuda'],
    verbose=False,
)


class OptimizedConv2d(nn.Module):
    """
    Optimized 2D convolution using im2col + GEMM approach.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1, bias=False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        
        # Weight shape: (out_channels, in_channels, kernel_size, kernel_size)
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_parameter('bias', None)
        
        # Pre-allocate column matrix
        self.col_buffer = None
        
    def forward(self, x):
        batch_size, channels, height, width = x.shape
        
        # Compute output dimensions
        out_h = (height + 2 * self.padding - self.dilation * (self.kernel_size - 1) - 1) // self.stride + 1
        out_w = (width + 2 * self.padding - self.dilation * (self.kernel_size - 1) - 1) // self.stride + 1
        
        # Apply im2col using custom CUDA kernel
        col = ext.im2col_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)
        
        # Reshape for matrix multiplication
        # col: (batch_size, out_h, out_w, kernel_size * kernel_size * in_channels)
        # weight: (out_channels, in_channels, kernel_size, kernel_size)
        
        col = col.contiguous().view(batch_size * out_h * out_w, -1)  # (N*H_out*W_out, K*K*C_in)
        weight_flat = self.weight.view(self.out_channels, -1)  # (C_out, K*K*C_in)
        
        # Matrix multiplication using bmm
        out = torch.mm(weight_flat, col.t()).t()  # (N*H_out*W_out, C_out)
        
        # Reshape output
        out = out.view(batch_size, out_h, out_w, self.out_channels)
        out = out.permute(0, 3, 1, 2).contiguous()  # (N, C_out, H_out, W_out)
        
        if self.bias is not None:
            out = out + self.bias.view(1, -1, 1, 1)
        
        return out


# Create a simpler but efficient fallback using im2col with torch operations
class FastConv2d(nn.Module):
    """
    Fast 2D convolution using PyTorch's unfold (im2col) + bmm.
    This uses optimized BLAS kernels underneath.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1, bias=False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels // groups, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_parameter('bias', None)
    
    def forward(self, x):
        # Use PyTorch's unfold for efficient im2col
        col = F.unfold(x, kernel_size=self.kernel_size, stride=self.stride, 
                       padding=self.padding, dilation=self.dilation)  # (N, C_in*k*k, H_out*W_out)
        
        # Reshape for grouped convolution
        # col: (N, C_in*k*k, H_out*W_out) -> (N, groups, C_in/groups*k*k, H_out*W_out)
        # weight: (C_out, C_in/groups, k, k) -> (groups, C_out/groups, C_in/groups*k*k)
        
        col = col.view(x.size(0), self.groups, self.in_channels // self.groups, 
                       self.kernel_size * self.kernel_size, -1)
        col = col.permute(1, 0, 4, 2, 3).contiguous()  # (groups, N, H_out*W_out, C_in/groups, k*k)
        col = col.view(self.groups, x.size(0) * col.size(2), -1)  # (groups, N*H_out*W_out, C_in/groups*k*k)
        
        weight = self.weight.view(self.groups, self.out_channels // self.groups, -1)  # (groups, C_out/groups, C_in/groups*k*k)
        
        # Batch matrix multiplication
        out = torch.bmm(weight, col.permute(0, 2, 1))  # (groups, C_out/groups, N*H_out*W_out)
        
        # Reshape output
        out = out.view(self.groups, self.out_channels // self.groups, x.size(0), -1)
        out = out.permute(2, 0, 1, 3).contiguous()  # (N, groups, C_out/groups, H_out*W_out)
        out = out.view(x.size(0), self.out_channels, -1)
        
        # Reshape to (N, C_out, H_out, W_out)
        out_h = (x.size(2) + 2 * self.padding - self.dilation * (self.kernel_size - 1) - 1) // self.stride + 1
        out_w = (x.size(3) + 2 * self.padding - self.dilation * (self.kernel_size - 1) - 1) // self.stride + 1
        out = out.view(x.size(0), self.out_channels, out_h, out_w)
        
        if self.bias is not None:
            out = out + self.bias.view(1, -1, 1, 1)
        
        return out


# Use the FastConv2d implementation
class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, 
                 padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = FastConv2d(in_channels, out_channels, kernel_size, stride, padding, dilation, groups, bias)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.conv2d(x)
