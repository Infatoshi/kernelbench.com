
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def softmax_kernel(
    output_ptr, input_ptr,
    input_row_stride, output_row_stride,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    # The rows of the softmax are independent, so we parallelize over the rows
    row_idx = tl.program_id(0)
    # The pointer to the beginning of this row
    row_start_ptr = input_ptr + row_idx * input_row_stride
    # The offsets for the columns
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_start_ptr + col_offsets
    
    # Load the row and mask out-of-bounds elements
    mask = col_offsets < n_cols
    row = tl.load(input_ptrs, mask=mask, other=-float('inf'))
    
    # Subtract max for numerical stability
    row_max = tl.max(row, axis=0)
    row = row - row_max
    
    # Exponentiate
    numerator = tl.exp(row)
    
    # Normalize
    denominator = tl.sum(numerator, axis=0)
    softmax_output = numerator / denominator
    
    # Write back
    output_row_start_ptr = output_ptr + row_idx * output_row_stride
    output_ptrs = output_row_start_ptr + col_offsets
    tl.store(output_ptrs, softmax_output, mask=mask)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        n_rows, n_cols = x.shape
        BLOCK_SIZE = triton.next_power_of_2(n_cols)
        
        # We use a single block per row if n_cols is small enough.
        # For very large n_cols, we'd need a multi-pass approach,
        # but for dim=16384, we can fit it in one BLOCK_SIZE if the GPU allows.
        # H100 allows large shared memory, but Triton's maximum BLOCK_SIZE is typically 65536 or 131072.
        
        y = torch.empty_like(x)
        
        # Grid is (n_rows,)
        softmax_kernel[(n_rows,)](
            y, x,
            x.stride(0), y.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=32 if BLOCK_SIZE >= 16384 else 8
        )
        return y

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
