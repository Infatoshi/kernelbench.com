import torch
import torch.nn as nn
import torch.nn.functional as F

# Module level constants matching the reference
in_channels = 3
out_channels = 16
kernel_size = 3
height = 256
width = 256

class Model(nn.Module):
    """
    Optimized model that fuses Conv2d, Mish activation, and BatchNorm.
    Uses standard PyTorch ops first, can be replaced with custom CUDA kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps
        
    def forward(self, x):
        # Conv2d
        x = self.conv(x)
        # Mish activation: x * tanh(softplus(x))
        x = x * torch.tanh(F.softplus(x))
        # BatchNorm
        x = self.bn(x)
        return x

def get_inputs():
    batch_size = 16
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
