import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import math

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Exact GELU using erf: x * 0.5 * (1 + erf(x/sqrt(2)))
__device__ __forceinline__ float gelu_exact(float x) {
    return x * 0.5f * (1.0f + erff(x * 0.7071067811865476f));
}

// Exact GELU for half
__device__ __forceinline__ half gelu_exact(half x) {
    float fx = __half2float(x);
    float result = fx * 0.5f * (1.0f + erff(fx * 0.7071067811865476f));
    return __float2half(result);
}

// Matmul kernel
template <typename T, int BLOCK_SIZE=256>
__global__ void matmul_kernel(
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    const T* __restrict__ input,
    T* __restrict__ output,
    const int in_features,
    const int out_features,
    const int batch_size
) {
    const int bid = blockIdx.x;
    const int out_feat = blockIdx.y * BLOCK_SIZE + threadIdx.x;
    
    if (out_feat >= out_features) return;
    
    const T* w_row = weight + out_feat * in_features;
    const T* in_ptr = input + bid * in_features;
    
    T sum_val = 0;
    
    const int chunk_size = 8;
    const int num_chunks = in_features / chunk_size;
    
    for (int c = 0; c < num_chunks; c++) {
        #pragma unroll
        for (int i = 0; i < chunk_size; i++) {
            sum_val = sum_val + in_ptr[c * chunk_size + i] * w_row[c * chunk_size + i];
        }
    }
    
    for (int i = num_chunks * chunk_size; i < in_features; i++) {
        sum_val = sum_val + in_ptr[i] * w_row[i];
    }
    
    if (bias != nullptr) {
        sum_val = sum_val + bias[out_feat];
    }
    
    output[bid * out_features + out_feat] = sum_val;
}

// Pool + GELU + Scale kernel
template <typename T>
__global__ void pool_gelu_scale_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T scale_factor,
    const int out_features,
    const int pool_kernel_size,
    const int pooled_out_features,
    const int batch_size
) {
    const int bid = blockIdx.x;
    const int pool_idx = blockIdx.y;
    
    if (bid >= batch_size || pool_idx >= pooled_out_features) return;
    
    const int region_start = pool_idx * pool_kernel_size;
    const T* in_ptr = input + bid * out_features;
    
    T sum_val = 0;
    
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        int idx = region_start + i;
        sum_val = sum_val + in_ptr[idx];
    }
    
    // Compute average using float for precision
    float avg_val = (float)sum_val / (float)pool_kernel_size;
    
    // Exact GELU: x * 0.5 * (1 + erf(x/sqrt(2)))
    float gelu_val = avg_val * 0.5f * (1.0f + erff(avg_val * 0.7071067811865476f));
    
    // Scale
    float scaled_val = gelu_val * (float)scale_factor;
    
    output[bid * pooled_out_features + pool_idx] = (T)scaled_val;
}

// Max reduction kernel
template <typename T>
__global__ void max_reduction_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int pooled_out_features
) {
    const int bid = blockIdx.x;
    
    if (bid >= batch_size) return;
    
    const T* in_ptr = input + bid * pooled_out_features;
    
    float max_val = -1e38;
    
    for (int i = 0; i < pooled_out_features; i++) {
        float val = (float)in_ptr[i];
        if (val > max_val) max_val = val;
    }
    
    output[bid] = (T)max_val;
}

// Wrapper
torch::Tensor optimized_forward(
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor input,
    float scale_factor,
    int pool_kernel_size
) {
    auto in_features = input.size(1);
    auto batch_size = input.size(0);
    auto out_features = weight.size(0);
    auto pooled_out_features = out_features / pool_kernel_size;
    
    auto dtype = input.dtype();
    auto device = input.device();
    
    auto matmul_output = torch::empty({batch_size, out_features}, torch::dtype(dtype).device(device));
    auto pooled_output = torch::empty({batch_size, pooled_out_features}, torch::dtype(dtype).device(device));
    auto final_output = torch::empty({batch_size}, torch::dtype(dtype).device(device));
    
    // Kernel 1: Matmul
    {
        const int BLOCK_SIZE = 256;
        dim3 grid(batch_size, (out_features + BLOCK_SIZE - 1) / BLOCK_SIZE);
        dim3 block(BLOCK_SIZE);
        
        AT_DISPATCH_FLOATING_TYPES(weight.scalar_type(), "matmul_kernel", ([&] {
            matmul_kernel<scalar_t, BLOCK_SIZE><<<grid, block>>>(
                weight.data_ptr<scalar_t>(),
                bias.defined() ? bias.data_ptr<scalar_t>() : nullptr,
                input.data_ptr<scalar_t>(),
                matmul_output.data_ptr<scalar_t>(),
                in_features,
                out_features,
                batch_size
            );
        }));
    }
    
    // Kernel 2: Pool + GELU + Scale
    {
        dim3 grid(batch_size, pooled_out_features);
        dim3 block(1);
        
        AT_DISPATCH_FLOATING_TYPES(weight.scalar_type(), "pool_gelu_scale_kernel", ([&] {
            pool_gelu_scale_kernel<scalar_t><<<grid, block>>>(
                matmul_output.data_ptr<scalar_t>(),
                pooled_output.data_ptr<scalar_t>(),
                (scalar_t)scale_factor,
                out_features,
                pool_kernel_size,
                pooled_out_features,
                batch_size
            );
        }));
    }
    
    // Kernel 3: Max reduction
    {
        dim3 grid(batch_size);
        dim3 block(1);
        
        AT_DISPATCH_FLOATING_TYPES(weight.scalar_type(), "max_reduction_kernel", ([&] {
            max_reduction_kernel<scalar_t><<<grid, block>>>(
                pooled_output.data_ptr<scalar_t>(),
                final_output.data_ptr<scalar_t>(),
                batch_size,
                pooled_out_features
            );
        }));
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return final_output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor optimized_forward(torch::Tensor weight, torch::Tensor bias, torch::Tensor input, float scale_factor, int pool_kernel_size);',
    cuda_sources=cuda_source,
    functions=['optimized_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        
    def forward(self, x):
        return ext.optimized_forward(
            self.linear.weight,
            self.linear.bias,
            x,
            2.0,
            4
        )


def get_inputs():
    return [torch.randn(128, 4096, device='cuda', dtype=torch.float16)]


def get_init_inputs():
    return [4096, 4096, 4, 2.0]
