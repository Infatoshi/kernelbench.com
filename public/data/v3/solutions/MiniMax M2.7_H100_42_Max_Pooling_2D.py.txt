import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized Max Pooling 2D kernel
// Each thread handles one output element

template <typename scalar_t>
__device__ __forceinline__ scalar_t load_val(const scalar_t* ptr, bool& valid) {
    return *ptr;
}

template <>
__device__ __forceinline__ float load_val<float>(const float* ptr, bool& valid) {
    return *ptr;
}

template <typename scalar_t>
__global__ void max_pool2d_unrolled_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation,
    int total_elements
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total_elements) return;

    // Decode flat index
    int batch_idx = idx / (channels * out_height * out_width);
    int c_out = (idx % (channels * out_height * out_width)) / (out_height * out_width);
    int out_pos = idx % (out_height * out_width);
    int out_h = out_pos / out_width;
    int out_w = out_pos % out_width;

    // Compute start position in input
    int ih_start = out_h * stride - padding;
    int iw_start = out_w * stride - padding;

    // Use registers for max computation
    scalar_t max_val;
    bool first = true;

    // General case - iterate over kernel
    for (int ki = 0; ki < kernel_size; ++ki) {
        for (int kj = 0; kj < kernel_size; ++kj) {
            int ih = ih_start + ki * dilation;
            int iw = iw_start + kj * dilation;

            if (ih >= 0 && ih < in_height && iw >= 0 && iw < in_width) {
                int in_idx = ((batch_idx * channels + c_out) * in_height + ih) * in_width + iw;
                scalar_t val = input[in_idx];
                if (first || val > max_val) {
                    max_val = val;
                }
                first = false;
            }
        }
    }

    int out_idx = ((batch_idx * channels + c_out) * out_height + out_h) * out_width + out_w;
    output[out_idx] = max_val;
}

// Kernel optimized for the specific task: kernel_size=2, stride=2, padding=1, dilation=3
template <typename scalar_t>
__global__ void max_pool2d_specialized_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int total_elements
) {
    // Fixed: kernel_size=2, stride=2, padding=1, dilation=3
    const int stride = 2;
    const int padding = 1;
    const int dilation = 3;

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total_elements) return;

    // Decode flat index
    int batch_idx = idx / (channels * out_height * out_width);
    int c_out = (idx % (channels * out_height * out_width)) / (out_height * out_width);
    int out_pos = idx % (out_height * out_width);
    int out_h = out_pos / out_width;
    int out_w = out_pos % out_width;

    // Compute start position in input with dilation
    int ih_start = out_h * stride - padding;
    int iw_start = out_w * stride - padding;

    scalar_t v00, v01, v10, v11;

    int ih0 = ih_start;
    int ih1 = ih_start + 3;
    int iw0 = iw_start;
    int iw1 = iw_start + 3;

    // Load all 4 values with bounds checking
    if (ih0 >= 0 && ih0 < in_height && iw0 >= 0 && iw0 < in_width) {
        v00 = input[((batch_idx * channels + c_out) * in_height + ih0) * in_width + iw0];
    } else {
        v00 = (scalar_t)-3.4e38;
    }

    if (ih0 >= 0 && ih0 < in_height && iw1 >= 0 && iw1 < in_width) {
        v01 = input[((batch_idx * channels + c_out) * in_height + ih0) * in_width + iw1];
    } else {
        v01 = (scalar_t)-3.4e38;
    }

    if (ih1 >= 0 && ih1 < in_height && iw0 >= 0 && iw0 < in_width) {
        v10 = input[((batch_idx * channels + c_out) * in_height + ih1) * in_width + iw0];
    } else {
        v10 = (scalar_t)-3.4e38;
    }

    if (ih1 >= 0 && ih1 < in_height && iw1 >= 0 && iw1 < in_width) {
        v11 = input[((batch_idx * channels + c_out) * in_height + ih1) * in_width + iw1];
    } else {
        v11 = (scalar_t)-3.4e38;
    }

    // Find max of 4 values
    scalar_t max_val = v00;
    if (v01 > max_val) max_val = v01;
    if (v10 > max_val) max_val = v10;
    if (v11 > max_val) max_val = v11;

    int out_idx = ((batch_idx * channels + c_out) * out_height + out_h) * out_width + out_w;
    output[out_idx] = max_val;
}

torch::Tensor max_pool2d_op(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);

    // Calculate output size
    int out_height = (in_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int out_width = (in_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;

    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());

    int total_elements = batch_size * channels * out_height * out_width;

    const int threads = 256;
    const int blocks = (total_elements + threads - 1) / threads;

    // Use specialized kernel for kernel_size=2, stride=2, padding=1, dilation=3
    if (kernel_size == 2 && stride == 2 && padding == 1 && dilation == 3) {
        AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "max_pool2d_cuda", ([&] {
            max_pool2d_specialized_kernel<scalar_t><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, in_height, in_width,
                out_height, out_width,
                total_elements
            );
        }));
    } else {
        // General case
        AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "max_pool2d_cuda", ([&] {
            max_pool2d_unrolled_kernel<scalar_t><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, in_height, in_width,
                out_height, out_width,
                kernel_size, stride, padding, dilation,
                total_elements
            );
        }));
    }

    return output;
}
'''

ext = load_inline(
    name='max_pool2d_ext',
    cpp_sources='''
torch::Tensor max_pool2d_op(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
);
''',
    cuda_sources=cuda_source,
    functions=['max_pool2d_op'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that performs Max Pooling 2D using custom CUDA kernel.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.

        Args:
            kernel_size (int): Size of the pooling window.
            stride (int): Stride of the pooling window.
            padding (int): Padding to be applied before pooling.
            dilation (int): Spacing between kernel elements.
        """
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Max Pooling 2D to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, channels, height, width).

        Returns:
            torch.Tensor: Output tensor after Max Pooling 2D.
        """
        return ext.max_pool2d_op(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    return [torch.randn(16, 32, 512, 512, device='cuda')]

def get_init_inputs():
    return [2, 2, 1, 3]
