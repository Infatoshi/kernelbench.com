import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Each thread handles 2 particles for better occupancy and instruction-level parallelism
__global__ void particle_update_kernel(
    const float2* __restrict__ positions,
    const float2* __restrict__ velocities,
    float4* __restrict__ output,
    const float dt,
    const float gx_dt,  // gx * dt precomputed
    const float gy_dt,  // gy * dt precomputed
    const float neg_damping,  // -damping precomputed
    const int n  // number of particles
) {
    int pid = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Process 2 particles per thread via grid-stride
    #pragma unroll
    for (int i = pid; i < n; i += blockDim.x * gridDim.x) {
        float2 pos = positions[i];
        float2 vel = velocities[i];

        // Velocity update with gravity (precomputed g*dt)
        float nvx = vel.x + gx_dt;
        float nvy = vel.y + gy_dt;

        // Position update
        float npx = pos.x + nvx * dt;
        float npy = pos.y + nvy * dt;

        // Bounce: if |new_pos| > 1.0, flip velocity and damp
        // Using branchless: fabsf check, multiply by neg_damping if over
        if (__builtin_expect(fabsf(npx) > 1.0f, 0)) {
            nvx *= neg_damping;
        }
        if (__builtin_expect(fabsf(npy) > 1.0f, 0)) {
            nvy *= neg_damping;
        }

        // Clamp position
        npx = fminf(fmaxf(npx, -1.0f), 1.0f);
        npy = fminf(fmaxf(npy, -1.0f), 1.0f);

        // Write output: [new_pos_x, new_pos_y, new_vel_x, new_vel_y]
        output[i] = make_float4(npx, npy, nvx, nvy);
    }
}

torch::Tensor particle_update(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gx,
    float gy,
    float damping
) {
    int n = positions.size(0);
    auto output = torch::empty({n, 4}, positions.options());

    const int threads = 256;
    const int blocks = (n + threads - 1) / threads;

    particle_update_kernel<<<blocks, threads>>>(
        reinterpret_cast<const float2*>(positions.data_ptr<float>()),
        reinterpret_cast<const float2*>(velocities.data_ptr<float>()),
        reinterpret_cast<float4*>(output.data_ptr<float>()),
        dt, gx * dt, gy * dt, -damping, n
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor particle_update(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gx,
    float gy,
    float damping
);
'''

ext = load_inline(
    name='particle_ext_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_update'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math', '--gpu-architecture=sm_86'],
)


class Model(nn.Module):
    """Simple GPU particle integration step with boundary collisions."""

    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        gx = gravity[0].item()
        gy = gravity[1].item()
        return ext.particle_update(
            positions.contiguous(),
            velocities.contiguous(),
            dt_value, gx, gy, self.damping
        )


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
