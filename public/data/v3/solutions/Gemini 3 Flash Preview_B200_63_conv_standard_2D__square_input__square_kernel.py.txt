
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel_simple(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_h, int in_w,
    int out_channels, int out_h, int out_w,
    int kernel_size, int stride, int padding, int dilation,
    int groups, int in_channels_per_group, int out_channels_per_group, bool has_bias) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_channels * out_h * out_w;
    
    if (idx < total_elements) {
        int ow = idx % out_w;
        int oh = (idx / out_w) % out_h;
        int oc = (idx / (out_w * out_h)) % out_channels;
        int b = idx / (out_w * out_h * out_channels);
        
        int g = oc / out_channels_per_group;
        int ic_start = g * in_channels_per_group;
        
        float val = 0.0f;
        for (int ic_offset = 0; ic_offset < in_channels_per_group; ++ic_offset) {
            int ic = ic_start + ic_offset;
            for (int kh = 0; kh < kernel_size; ++kh) {
                int ih = oh * stride - padding + kh * dilation;
                if (ih >= 0 && ih < in_h) {
                    for (int kw = 0; kw < kernel_size; ++kw) {
                        int iw = ow * stride - padding + kw * dilation;
                        if (iw >= 0 && iw < in_w) {
                            int input_idx = ((b * in_channels + ic) * in_h + ih) * in_w + iw;
                            int weight_idx = ((oc * in_channels_per_group + ic_offset) * kernel_size + kh) * kernel_size + kw;
                            val += input[input_idx] * weight[weight_idx];
                        }
                    }
                }
            }
        }
        if (has_bias) {
            val += bias[oc];
        }
        output[idx] = val;
    }
}

torch::Tensor conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding, int dilation, int groups) 
{
    auto batch_size = input.size(0);
    auto in_channels = input.size(1);
    auto in_h = input.size(2);
    auto in_w = input.size(3);
    
    auto out_channels = weight.size(0);
    auto kernel_size = weight.size(2);
    
    auto out_h = (in_h + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    auto out_w = (in_w + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_h, out_w}, input.options().dtype(at::kFloat));
    
    int total_elements = (int)(batch_size * out_channels * out_h * out_w);
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;
    
    bool has_bias = bias.has_value();
    float* bias_ptr = has_bias ? bias.value().data_ptr<float>() : nullptr;
    
    conv2d_kernel_simple<<<blocks, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias_ptr,
        output.data_ptr<float>(),
        (int)batch_size, (int)in_channels, (int)in_h, (int)in_w,
        (int)out_channels, (int)out_h, (int)out_w,
        (int)kernel_size, stride, padding, dilation,
        groups, (int)(in_channels / groups), (int)(out_channels / groups), has_bias
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding, int dilation, int groups);
'''

ext = load_inline(
    name='conv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['conv2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(Model, self).__init__()
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels // groups, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.randn(out_channels))
        else:
            self.register_parameter('bias', None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        orig_dtype = x.dtype
        x_f = x.float().contiguous()
        w_f = self.weight.float().contiguous()
        b_f = self.bias.float().contiguous() if self.bias is not None else None
        
        out = ext.conv2d_forward(x_f, w_f, b_f, self.stride, self.padding, self.dilation, self.groups)
        return out.to(orig_dtype)

def get_inputs():
    x = torch.randn(16, 3, 256, 256)
    return [x]

def get_init_inputs():
    return [3, 64, 3]
