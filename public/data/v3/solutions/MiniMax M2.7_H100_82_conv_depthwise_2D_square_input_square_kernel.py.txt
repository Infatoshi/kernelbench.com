import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__global__ void depthwise_conv_nchw_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int height,
    int width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding
) {
    int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    int batch_ch = blockIdx.z;
    
    int batch_idx = batch_ch / channels;
    int ch = batch_ch % channels;
    
    if (out_x >= out_width || out_y >= out_height) return;
    if (batch_idx >= batch_size || ch >= channels) return;
    
    // Compute input position
    int in_x = out_x * stride - padding;
    int in_y = out_y * stride - padding;
    
    T sum = static_cast<T>(0.0f);
    
    // Convolution
    for (int ky = 0; ky < kernel_size; ky++) {
        int in_y_pos = in_y + ky;
        if (in_y_pos >= 0 && in_y_pos < height) {
            for (int kx = 0; kx < kernel_size; kx++) {
                int in_x_pos = in_x + kx;
                if (in_x_pos >= 0 && in_x_pos < width) {
                    int input_idx = ((batch_idx * channels + ch) * height + in_y_pos) * width + in_x_pos;
                    int weight_idx = ch * kernel_size * kernel_size + ky * kernel_size + kx;
                    sum += input[input_idx] * weight[weight_idx];
                }
            }
        }
    }
    
    // Store result
    int out_idx = ((batch_idx * channels + ch) * out_height + out_y) * out_width + out_x;
    output[out_idx] = sum;
}

torch::Tensor depthwise_conv2d(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    
    int out_height = (height + 2 * padding - 3) / stride + 1;
    int out_width = (width + 2 * padding - 3) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                               input.options());
    
    const dim3 block(16, 16);
    const dim3 grid((out_width + 15) / 16, (out_height + 15) / 16, batch_size * channels);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "depthwise_conv2d", ([&] {
        depthwise_conv_nchw_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels, height, width,
            out_height, out_width,
            3, stride, padding
        );
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding
);
'''

ext = load_inline(
    name='depthwise_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d'],
    verbose=False,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '-arch=sm_90']
)

class Model(nn.Module):
    def __init__(self, in_channels, kernel_size, stride=1, padding=0, bias=False):
        super().__init__()
        self.stride = stride
        self.padding = padding
        # Create the conv2d module to get proper weight initialization
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, 
                                stride=stride, padding=padding, 
                                groups=in_channels, bias=bias)
        
    def forward(self, x):
        return ext.depthwise_conv2d(x, self.conv2d.weight, self.stride, self.padding)

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 3, 1, 0]
