import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Unified fused kernel: Conv -> Tanh -> Scale -> Bias -> MaxPool for FP32
template <typename T>
__global__ void fused_all_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ extra_bias,
    T* __restrict__ output,
    float scaling_factor,
    int batch_size,
    int in_channels,
    int out_channels,
    int in_height,
    int in_width,
    int out_height_after_conv,
    int out_height_after_pool,
    int out_width_after_conv,
    int out_width_after_pool,
    int kH,
    int kW,
    int conv_stride,
    int conv_padding,
    int pool_size,
    int pool_stride
) {
    // Grid: (out_channels, out_height_after_pool, out_width_after_pool)
    int oc = blockIdx.x;
    int oh_pool = blockIdx.y;
    int ow_pool = blockIdx.z;
    
    int threads_per_block = blockDim.x;
    int tid = threadIdx.x;
    
    // Convert pool output position to conv output position
    int oh_conv_start = oh_pool * pool_stride;
    int ow_conv_start = ow_pool * pool_stride;
    
    for (int b = tid; b < batch_size; b += threads_per_block) {
        float max_val = -1e30f;
        
        // Iterate over the pool window in conv output space
        for (int ph = 0; ph < pool_size; ++ph) {
            for (int pw = 0; pw < pool_size; ++pw) {
                int oh_conv = oh_conv_start + ph;
                int ow_conv = ow_conv_start + pw;
                
                if (oh_conv < out_height_after_conv && ow_conv < out_width_after_conv) {
                    // Compute convolution at this position
                    float conv_sum = 0.0f;
                    
                    for (int ic = 0; ic < in_channels; ++ic) {
                        for (int kh = 0; kh < kH; ++kh) {
                            for (int kw = 0; kw < kW; ++kw) {
                                int in_h = oh_conv * conv_stride - conv_padding + kh;
                                int in_w = ow_conv * conv_stride - conv_padding + kw;
                                
                                if (in_h >= 0 && in_h < in_height && in_w >= 0 && in_w < in_width) {
                                    int in_idx = ((b * in_channels + ic) * in_height + in_h) * in_width + in_w;
                                    int w_idx = ((oc * in_channels + ic) * kH + kh) * kW + kw;
                                    conv_sum += (float)input[in_idx] * (float)weight[w_idx];
                                }
                            }
                        }
                    }
                    
                    // Tanh + Scale + Bias
                    conv_sum = tanhf(conv_sum) * scaling_factor + (float)extra_bias[oc];
                    
                    // Max pooling
                    max_val = fmaxf(max_val, conv_sum);
                }
            }
        }
        
        int out_idx = (b * out_channels + oc) * out_height_after_pool * out_width_after_pool + oh_pool * out_width_after_pool + ow_pool;
        output[out_idx] = (T)max_val;
    }
}

// Half precision optimized fused kernel
__global__ void fused_all_half_kernel(
    const __half* __restrict__ input,
    const __half* __restrict__ weight,
    const __half* __restrict__ extra_bias,
    __half* __restrict__ output,
    float scaling_factor,
    int batch_size,
    int in_channels,
    int out_channels,
    int in_height,
    int in_width,
    int out_height_after_conv,
    int out_height_after_pool,
    int out_width_after_conv,
    int out_width_after_pool,
    int kH,
    int kW,
    int conv_stride,
    int conv_padding,
    int pool_size,
    int pool_stride
) {
    int oc = blockIdx.x;
    int oh_pool = blockIdx.y;
    int ow_pool = blockIdx.z;
    
    int tid = threadIdx.x;
    
    int oh_conv_start = oh_pool * pool_stride;
    int ow_conv_start = ow_pool * pool_stride;
    
    for (int b = tid; b < batch_size; b += blockDim.x) {
        float max_val = -1e30f;
        
        for (int ph = 0; ph < pool_size; ++ph) {
            for (int pw = 0; pw < pool_size; ++pw) {
                int oh_conv = oh_conv_start + ph;
                int ow_conv = ow_conv_start + pw;
                
                if (oh_conv < out_height_after_conv && ow_conv < out_width_after_conv) {
                    float conv_sum = 0.0f;
                    
                    for (int ic = 0; ic < in_channels; ++ic) {
                        for (int kh = 0; kh < kH; ++kh) {
                            for (int kw = 0; kw < kW; ++kw) {
                                int in_h = oh_conv * conv_stride - conv_padding + kh;
                                int in_w = ow_conv * conv_stride - conv_padding + kw;
                                
                                if (in_h >= 0 && in_h < in_height && in_w >= 0 && in_w < in_width) {
                                    int in_idx = ((b * in_channels + ic) * in_height + in_h) * in_width + in_w;
                                    int w_idx = ((oc * in_channels + ic) * kH + kh) * kW + kw;
                                    float a = __half2float(input[in_idx]);
                                    float w = __half2float(weight[w_idx]);
                                    conv_sum += a * w;
                                }
                            }
                        }
                    }
                    
                    conv_sum = tanhf(conv_sum) * scaling_factor + __half2float(extra_bias[oc]);
                    max_val = fmaxf(max_val, conv_sum);
                }
            }
        }
        
        int out_idx = (b * out_channels + oc) * out_height_after_pool * out_width_after_pool + oh_pool * out_width_after_pool + ow_pool;
        output[out_idx] = __float2half(max_val);
    }
}

// Host function dispatch based on dtype
torch::Tensor fused_conv_tanh_scale_bias_maxpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor extra_bias,
    float scaling_factor,
    int conv_padding,
    int conv_stride,
    int pool_size,
    int pool_stride
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int out_channels = weight.size(0);
    int kH = weight.size(2);
    int kW = weight.size(3);
    
    // Conv output size
    int out_h_conv = (in_height + 2 * conv_padding - kH) / conv_stride + 1;
    int out_w_conv = (in_width + 2 * conv_padding - kW) / conv_stride + 1;
    
    // Pool output size
    int out_h = (out_h_conv - pool_size) / pool_stride + 1;
    int out_w = (out_w_conv - pool_size) / pool_stride + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_h, out_w}, input.options());
    
    if (input.dtype() == torch::kFloat32) {
        dim3 grid(out_channels, out_h, out_w);
        int threads = 256;
        fused_all_kernel<float><<<grid, threads>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            extra_bias.data_ptr<float>(),
            output.data_ptr<float>(),
            scaling_factor,
            batch_size, in_channels, out_channels,
            in_height, in_width,
            out_h_conv, out_h, out_w_conv, out_w,
            kH, kW,
            conv_stride, conv_padding,
            pool_size, pool_stride
        );
    } else if (input.dtype() == torch::kHalf) {
        dim3 grid(out_channels, out_h, out_w);
        int threads = 256;
        fused_all_half_kernel<<<grid, threads>>>(
            (__half*)input.data_ptr(),
            (__half*)weight.data_ptr(),
            (__half*)extra_bias.data_ptr(),
            (__half*)output.data_ptr(),
            scaling_factor,
            batch_size, in_channels, out_channels,
            in_height, in_width,
            out_h_conv, out_h, out_w_conv, out_w,
            kH, kW,
            conv_stride, conv_padding,
            pool_size, pool_stride
        );
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources='''
torch::Tensor fused_conv_tanh_scale_bias_maxpool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor extra_bias,
    float scaling_factor,
    int conv_padding,
    int conv_stride,
    int pool_size,
    int pool_stride
);
''',
    cuda_sources=cuda_source,
    functions=['fused_conv_tanh_scale_bias_maxpool'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        # Conv2d with bias=False since we use a separate bias
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=False)
        self.scaling_factor = scaling_factor
        # Separate bias parameter (matches reference)
        self.extra_bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size
        self.pool_stride = pool_kernel_size
        
    def forward(self, x):
        return ext.fused_conv_tanh_scale_bias_maxpool(
            x,
            self.conv.weight,
            self.extra_bias,
            self.scaling_factor,
            self.conv.padding[0],
            self.conv.stride[0],
            self.pool_kernel_size,
            self.pool_stride
        )

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 2.0, (16, 1, 1), 2]
