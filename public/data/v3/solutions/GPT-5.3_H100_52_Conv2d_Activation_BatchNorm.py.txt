import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(
    x_ptr, w_ptr, b_ptr,
    gamma_ptr, beta_ptr, mean_ptr, var_ptr,
    y_ptr,
    N, Cin, H, W, Cout, Kh, Kw, Hout, Wout,
    eps,
    BLOCK: tl.constexpr
):
    pid = tl.program_id(0)
    idx = pid * BLOCK + tl.arange(0, BLOCK)

    total = N * Cout * Hout * Wout
    mask = idx < total

    wout = idx % Wout
    htmp = idx // Wout
    hout = htmp % Hout
    ctmp = htmp // Hout
    cout = ctmp % Cout
    n = ctmp // Cout

    acc = tl.zeros([BLOCK], dtype=tl.float32)

    for ci in range(0, Cin):
        for kh in range(0, Kh):
            for kw in range(0, Kw):
                h_in = hout + kh
                w_in = wout + kw
                x_idx = ((n * Cin + ci) * H + h_in) * W + w_in
                w_idx = ((cout * Cin + ci) * Kh + kh) * Kw + kw
                x = tl.load(x_ptr + x_idx, mask=mask, other=0.0).to(tl.float32)
                w = tl.load(w_ptr + w_idx, mask=mask, other=0.0).to(tl.float32)
                acc += x * w

    if b_ptr is not None:
        acc += tl.load(b_ptr + cout, mask=mask, other=0.0).to(tl.float32)

    # activation: x * tanh(softplus(x))
    sp = tl.log(1 + tl.exp(acc))
    t = (tl.exp(sp) - tl.exp(-sp)) / (tl.exp(sp) + tl.exp(-sp))
    act = acc * t

    mean = tl.load(mean_ptr + cout, mask=mask, other=0.0)
    var = tl.load(var_ptr + cout, mask=mask, other=0.0)
    gamma = tl.load(gamma_ptr + cout, mask=mask, other=1.0)
    beta = tl.load(beta_ptr + cout, mask=mask, other=0.0)

    norm = (act - mean) / tl.sqrt(var + eps)
    out = norm * gamma + beta

    tl.store(y_ptr + idx, out, mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.kernel_size = kernel_size
        self.eps = eps

    def forward(self, x):
        N, Cin, H, W = x.shape
        Cout = self.conv.weight.shape[0]
        Kh = Kw = self.kernel_size
        Hout = H - Kh + 1
        Wout = W - Kw + 1

        y = torch.empty((N, Cout, Hout, Wout), device=x.device, dtype=torch.float32)

        total = N * Cout * Hout * Wout
        BLOCK = 256
        grid = ( (total + BLOCK - 1) // BLOCK, )

        fused_kernel[grid](
            x, self.conv.weight, self.conv.bias,
            self.bn.weight, self.bn.bias,
            self.bn.running_mean, self.bn.running_var,
            y,
            N, Cin, H, W, Cout, Kh, Kw, Hout, Wout,
            self.bn.eps,
            BLOCK=BLOCK
        )

        return y


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
