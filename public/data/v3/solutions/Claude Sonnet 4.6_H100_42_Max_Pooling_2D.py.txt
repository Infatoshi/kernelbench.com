import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>

// Optimized specialized kernel for kernel_size=2, stride=2, padding=1, dilation=3
// Each thread processes one output element
// Uses 128 threads per block (warp-aligned), large grid for H100

template<typename scalar_t>
__global__ void maxpool2d_k2s2p1d3_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int total_out   // batch*channels*out_h*out_w
) {
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total_out) return;

    // Decode index
    const int ow  = idx % out_w;
    const int oh  = (idx / out_w) % out_h;
    const int bc  = idx / (out_w * out_h);

    const scalar_t* in_ptr = input + bc * in_h * in_w;

    // stride=2, pad=1, dil=3
    // ih_start = oh*2 - 1, iw_start = ow*2 - 1
    const int ih0 = oh * 2 - 1;
    const int ih1 = ih0 + 3;  // ih0 + dilation
    const int iw0 = ow * 2 - 1;
    const int iw1 = iw0 + 3;

    float maxval = -FLT_MAX;

    // Fully unrolled 2x2 with boundary checks
    bool h0_ok = (ih0 >= 0 && ih0 < in_h);
    bool h1_ok = (ih1 >= 0 && ih1 < in_h);
    bool w0_ok = (iw0 >= 0 && iw0 < in_w);
    bool w1_ok = (iw1 >= 0 && iw1 < in_w);

    if (h0_ok) {
        const scalar_t* row = in_ptr + ih0 * in_w;
        if (w0_ok) { float v = (float)row[iw0]; if (v > maxval) maxval = v; }
        if (w1_ok) { float v = (float)row[iw1]; if (v > maxval) maxval = v; }
    }
    if (h1_ok) {
        const scalar_t* row = in_ptr + ih1 * in_w;
        if (w0_ok) { float v = (float)row[iw0]; if (v > maxval) maxval = v; }
        if (w1_ok) { float v = (float)row[iw1]; if (v > maxval) maxval = v; }
    }

    output[idx] = (scalar_t)maxval;
}

// Generic max pooling kernel - flat indexing for coalesced access
template<typename scalar_t>
__global__ void maxpool2d_generic_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int kernel_h,
    const int kernel_w,
    const int stride_h,
    const int stride_w,
    const int pad_h,
    const int pad_w,
    const int dil_h,
    const int dil_w,
    const int total_out
) {
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total_out) return;

    const int ow  = idx % out_w;
    const int oh  = (idx / out_w) % out_h;
    const int bc  = idx / (out_w * out_h);

    const scalar_t* in_ptr = input + bc * in_h * in_w;

    float maxval = -FLT_MAX;

    const int ih_start = oh * stride_h - pad_h;
    const int iw_start = ow * stride_w - pad_w;

    for (int kh = 0; kh < kernel_h; kh++) {
        const int ih = ih_start + kh * dil_h;
        if (ih < 0 || ih >= in_h) continue;
        for (int kw = 0; kw < kernel_w; kw++) {
            const int iw = iw_start + kw * dil_w;
            if (iw < 0 || iw >= in_w) continue;
            float v = (float)in_ptr[ih * in_w + iw];
            if (v > maxval) maxval = v;
        }
    }

    output[idx] = (scalar_t)maxval;
}


torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_h, int kernel_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w
) {
    TORCH_CHECK(input.is_cuda(), "input must be CUDA tensor");
    TORCH_CHECK(input.dim() == 4, "input must be 4D");
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous");

    const int batch_size = input.size(0);
    const int channels   = input.size(1);
    const int in_h       = input.size(2);
    const int in_w       = input.size(3);

    const int out_h = (in_h + 2 * pad_h - dil_h * (kernel_h - 1) - 1) / stride_h + 1;
    const int out_w = (in_w + 2 * pad_w - dil_w * (kernel_w - 1) - 1) / stride_w + 1;

    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());

    const int total_out = batch_size * channels * out_h * out_w;
    const int block_size = 256;
    const int grid_size = (total_out + block_size - 1) / block_size;

    bool use_special = (kernel_h == 2 && kernel_w == 2 &&
                        stride_h == 2 && stride_w == 2 &&
                        pad_h == 1 && pad_w == 1 &&
                        dil_h == 3 && dil_w == 3);

    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half, at::ScalarType::BFloat16,
        input.scalar_type(), "maxpool2d_cuda", [&] {
            if (use_special) {
                maxpool2d_k2s2p1d3_kernel<scalar_t><<<grid_size, block_size>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    batch_size, channels,
                    in_h, in_w, out_h, out_w,
                    total_out
                );
            } else {
                maxpool2d_generic_kernel<scalar_t><<<grid_size, block_size>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    in_h, in_w, out_h, out_w,
                    kernel_h, kernel_w,
                    stride_h, stride_w,
                    pad_h, pad_w,
                    dil_h, dil_w,
                    total_out
                );
            }
        }
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_h, int kernel_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w
);
'''

ext = load_inline(
    name='maxpool2d_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['maxpool2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    """
    Optimized Max Pooling 2D using custom CUDA kernel.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d_cuda(
            x,
            self.kernel_size, self.kernel_size,
            self.stride, self.stride,
            self.padding, self.padding,
            self.dilation, self.dilation
        )


batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
