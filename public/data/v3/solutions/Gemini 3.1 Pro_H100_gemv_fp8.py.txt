import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__device__ __forceinline__ int dp4a(int a, int b, int c) {
    int res;
    asm("dp4a.s32.s32 %0, %1, %2, %3;" : "=r"(res) : "r"(a), "r"(b), "r"(c));
    return res;
}

// Optimization: 
// 1. Load x into shared memory since it's very small (32 x 4096 bytes = 128KB). 
// Wait, 128KB requires dynamic shared memory and cudaFuncSetAttribute, let's keep it simple.
// Since x is reused across all N=14336 columns, we should tile N and load x once per block.
// Or just let L2 cache handle x! 128KB fits entirely in L2.
// The issue is memory bandwidth for w. W is 14336 x 4096 = 58MB. 
// We are reading W and doing only M=32 multiply-adds per element of W.
// 289ms -> this is terrible performance. The kernel should take < 1 ms!
// The problem is probably memory access pattern.
// w is stored row-major: tensor shape is (14336, 4096)
// This means w[n, k] is contiguous in k.
// But in my kernel, threads in a warp are doing different Ns!
// const int* w_ptr = (const int*)(w + n * K);
// This means thread 0 reads w[0, k], thread 1 reads w[1, k], etc.
// This is strided memory access! Stride is K=4096 bytes!
// This causes massive uncoalesced memory reads.

// To fix this, threads in a warp must read contiguous memory.
// Meaning threads in a warp should iterate over K together!
// We can assign one block to compute ONE output element (or a small tile).
// Or 1 warp computes ONE output element (n, m) by reducing over K.

__global__ void my_int8_gemm_kernel_warp_reduce(
    const int8_t* __restrict__ x,
    const int8_t* __restrict__ w,
    half* __restrict__ out,
    const float* __restrict__ x_scale_ptr,
    const float* __restrict__ w_scale_ptr,
    int M, int N, int K
) {
    // 1 warp handles 1 pair of (m, n)
    int n = blockIdx.x * (blockDim.y) + threadIdx.y;
    int m = blockIdx.y;
    
    if (m >= M || n >= N) return;
    
    // warp lane
    int lane = threadIdx.x;
    
    // Each thread in warp handles K_vec / 32 elements.
    int K_vec = K / 4; // K_vec = 1024
    
    const int* x_ptr = (const int*)(x + m * K);
    const int* w_ptr = (const int*)(w + n * K);
    
    int acc = 0;
    for (int k = lane; k < K_vec; k += 32) {
        int x_val = __ldg(x_ptr + k);
        int w_val = __ldg(w_ptr + k);
        acc = dp4a(x_val, w_val, acc);
    }
    
    // Warp reduce sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        acc += __shfl_down_sync(0xffffffff, acc, offset);
    }
    
    if (lane == 0) {
        float x_scale = *x_scale_ptr;
        float w_scale = *w_scale_ptr;
        float out_f = (float)acc * x_scale * w_scale;
        out[m * N + n] = __float2half(out_f);
    }
}

torch::Tensor gemm_int8(torch::Tensor x, torch::Tensor w, torch::Tensor x_scale, torch::Tensor w_scale) {
    int M = x.size(0);
    int K = x.size(1);
    int N = w.size(0);
    
    auto out = torch::empty({M, N}, torch::dtype(torch::kFloat16).device(x.device()));
    
    // Block size: 32 threads in x (1 warp), 4 warps in y
    dim3 block(32, 4); 
    dim3 grid((N + 3) / 4, M); 
    
    my_int8_gemm_kernel_warp_reduce<<<grid, block>>>(
        x.data_ptr<int8_t>(),
        w.data_ptr<int8_t>(),
        reinterpret_cast<half*>(out.data_ptr<at::Half>()),
        x_scale.data_ptr<float>(),
        w_scale.data_ptr<float>(),
        M, N, K
    );
    
    return out;
}
"""

cpp_source = """
torch::Tensor gemm_int8(torch::Tensor x, torch::Tensor w, torch::Tensor x_scale, torch::Tensor w_scale);
"""

ext = load_inline(
    name='my_gemm',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_int8'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemm_int8(x_q, self.weight_q, x_scale, self.weight_scale)

def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), device='cuda', dtype=torch.int8)
    x_scale = torch.tensor(0.02, device='cuda', dtype=torch.float32)
    return [x_q, x_scale]

def get_init_inputs():
    return []
