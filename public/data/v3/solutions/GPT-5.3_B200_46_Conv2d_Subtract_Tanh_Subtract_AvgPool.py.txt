import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N, int C, int H, int W,
    int OC, int K,
    float sub1, float sub2,
    int Hp, int Wp
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * Hp * Wp;
    if (idx >= total) return;

    int ow = idx % Wp;
    int oh = (idx / Wp) % Hp;
    int oc = (idx / (Wp * Hp)) % OC;
    int n = idx / (Wp * Hp * OC);

    int h_base = oh * 2;
    int w_base = ow * 2;

    float acc_pool = 0.0f;

    for (int ph = 0; ph < 2; ++ph) {
        for (int pw = 0; pw < 2; ++pw) {
            int h = h_base + ph;
            int w0 = w_base + pw;

            float val = b[oc];

            for (int ic = 0; ic < C; ++ic) {
                for (int kh = 0; kh < K; ++kh) {
                    for (int kw = 0; kw < K; ++kw) {
                        int ih = h + kh;
                        int iw = w0 + kw;
                        float xv = x[((n*C + ic)*H + ih)*W + iw];
                        float wv = w[(((oc*C + ic)*K + kh)*K + kw)];
                        val += xv * wv;
                    }
                }
            }

            val = val - sub1;
            val = tanhf(val);
            val = val - sub2;

            acc_pool += val;
        }
    }

    out[idx] = acc_pool * 0.25f;
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                           float sub1, float sub2) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int K = w.size(2);

    int Hc = H - K + 1;
    int Wc = W - K + 1;

    int Hp = Hc / 2;
    int Wp = Wc / 2;

    auto out = torch::empty({N, OC, Hp, Wp}, x.options());

    int total = N * OC * Hp * Wp;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, H, W, OC, K,
        sub1, sub2,
        Hp, Wp
    );

    return out;
}
'''

ext = load_inline(
    name='fused_conv_pool',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, float sub1, float sub2);',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.sub1 = subtract1_value
        self.sub2 = subtract2_value

    def forward(self, x):
        return ext.fused_forward(x, self.conv.weight, self.conv.bias, self.sub1, self.sub2)


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
