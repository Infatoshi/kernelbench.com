import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def rmsnorm_kernel(x_ptr, y_ptr, B, F, H, W, eps, stride_b, stride_f, stride_h, stride_w, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    HW = H * W
    total = B * HW
    if pid >= total:
        return

    b = pid // HW
    rem = pid % HW
    h = rem // W
    w = rem % W

    base = b * stride_b + h * stride_h + w * stride_w

    offs = tl.arange(0, BLOCK)
    mask = offs < F

    ptrs = x_ptr + base + offs * stride_f
    x = tl.load(ptrs, mask=mask, other=0.0)
    x_f32 = x.to(tl.float32)

    sq = x_f32 * x_f32
    sum_sq = tl.sum(sq, axis=0)
    mean_sq = sum_sq / F
    inv_rms = tl.rsqrt(mean_sq + eps)

    y = x_f32 * inv_rms
    y = y.to(x.dtype)

    out_ptrs = y_ptr + base + offs * stride_f
    tl.store(out_ptrs, y, mask=mask)


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, F, H, W = x.shape
        y = torch.empty_like(x)

        grid = (B * H * W,)

        rmsnorm_kernel[grid](
            x, y,
            B, F, H, W,
            self.eps,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            BLOCK=64
        )
        return y
