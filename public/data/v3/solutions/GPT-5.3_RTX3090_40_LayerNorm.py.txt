import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

using namespace at;

template <typename scalar_t>
__device__ inline float to_float(scalar_t v){ return (float)v; }

template <>
__device__ inline float to_float<half>(half v){ return __half2float(v); }

template <>
__device__ inline float to_float<nv_bfloat16>(nv_bfloat16 v){ return __bfloat162float(v); }

template <typename scalar_t>
__device__ inline scalar_t from_float(float v){ return (scalar_t)v; }

template <>
__device__ inline half from_float<half>(float v){ return __float2half(v); }

template <>
__device__ inline nv_bfloat16 from_float<nv_bfloat16>(float v){ return __float2bfloat16(v); }

template <typename scalar_t>
__global__ void layernorm_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ gamma,
    const scalar_t* __restrict__ beta,
    scalar_t* __restrict__ y,
    int M,
    float eps)
{
    int b = blockIdx.x;
    int tid = threadIdx.x;

    const scalar_t* xb = x + (size_t)b * M;
    scalar_t* yb = y + (size_t)b * M;

    float sum = 0.f;
    float sumsq = 0.f;

    for (int i = tid; i < M; i += blockDim.x){
        float v = to_float(xb[i]);
        sum += v;
        sumsq += v * v;
    }

    __shared__ float s_sum[256];
    __shared__ float s_sumsq[256];

    s_sum[tid] = sum;
    s_sumsq[tid] = sumsq;
    __syncthreads();

    for (int s = blockDim.x/2; s > 0; s >>= 1){
        if (tid < s){
            s_sum[tid] += s_sum[tid + s];
            s_sumsq[tid] += s_sumsq[tid + s];
        }
        __syncthreads();
    }

    float mean = s_sum[0] / M;
    float var = s_sumsq[0] / M - mean * mean;
    float inv_std = rsqrtf(var + eps);

    for (int i = tid; i < M; i += blockDim.x){
        float v = to_float(xb[i]);
        float g = to_float(gamma[i]);
        float be = to_float(beta[i]);
        float out = (v - mean) * inv_std;
        out = out * g + be;
        yb[i] = from_float<scalar_t>(out);
    }
}

torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps){
    int B = x.size(0);
    int64_t M = 1;
    for (int i = 1; i < x.dim(); ++i) M *= x.size(i);

    auto y = torch::empty_like(x);

    int threads = 256;

    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "layernorm_kernel", ([&] {
        layernorm_kernel<<<B, threads>>>(
            (scalar_t*)x.data_ptr<scalar_t>(),
            (scalar_t*)gamma.data_ptr<scalar_t>(),
            (scalar_t*)beta.data_ptr<scalar_t>(),
            (scalar_t*)y.data_ptr<scalar_t>(),
            M,
            eps
        );
    }));

    return y;
}
'''

cpp_src = "torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps);"

ext = load_inline(
    name='fast_layernorm',
    cpp_sources=cpp_src,
    cuda_sources=cuda_src,
    functions=['layernorm_forward'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.normalized_shape = normalized_shape
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        gamma = self.weight.view(-1)
        beta = self.bias.view(-1)
        return ext.layernorm_forward(x.contiguous(), gamma.contiguous(), beta.contiguous(), self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
