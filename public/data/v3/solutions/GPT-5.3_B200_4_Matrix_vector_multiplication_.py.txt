import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void matvec_kernel(const float* __restrict__ A,
                              const float* __restrict__ B,
                              float* __restrict__ C,
                              int M, int K) {
    int row = blockIdx.x;
    int tid = threadIdx.x;

    float sum = 0.0f;

    int stride = blockDim.x * 4;
    int start = tid * 4;

    const float* A_row = A + row * K;

    for (int k = start; k < K; k += stride) {
        if (k + 3 < K) {
            float4 a = *(const float4*)(A_row + k);
            float4 b = *(const float4*)(B + k);
            sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
        } else {
            for (int kk = k; kk < K; ++kk) {
                sum += A_row[kk] * B[kk];
            }
        }
    }

    __shared__ float sdata[256];
    sdata[tid] = sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) {
        C[row] = sdata[0];
    }
}

torch::Tensor matvec(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);

    auto C = torch::empty({M, 1}, A.options());

    int threads = 256;
    matvec_kernel<<<M, threads>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, K);

    return C;
}
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources='torch::Tensor matvec(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matvec'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.matvec(A, B)


def get_inputs():
    M = 256
    K = 131072
    A = torch.randn(M, K, device='cuda', dtype=torch.float32)
    B = torch.randn(K, 1, device='cuda', dtype=torch.float32)
    return [A, B]


def get_init_inputs():
    return []
