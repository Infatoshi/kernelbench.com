import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>

using namespace nvcuda;

// Tile sizes for WMMA GEMM
#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

#define BLOCK_M 128
#define BLOCK_N 128
#define BLOCK_K 32

#define WARPS_M 2
#define WARPS_N 2
#define NUM_WARPS (WARPS_M * WARPS_N)

__global__ void wmma_gemm_kernel(const half* A, const half* B, half* C, int N) {
    int bx = blockIdx.x;
    int by = blockIdx.y;
    
    int warpId = threadIdx.x / 32;
    int warpM = warpId / WARPS_N;
    int warpN = warpId % WARPS_N;
    int laneId = threadIdx.x % 32;
    
    // WMMA fragments
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::col_major> b_frag;
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, half> c_frag;
    
    wmma::fill_fragment(c_frag, __float2half(0.0f));
    
    // Shared memory
    __shared__ half As[BLOCK_M * BLOCK_K];
    __shared__ half Bs[BLOCK_K * BLOCK_N];
    
    int numKTiles = (N + BLOCK_K - 1) / BLOCK_K;
    
    for (int tk = 0; tk < numKTiles; tk++) {
        // Load A tile - each thread loads 8 elements
        int aRow = by * BLOCK_M + threadIdx.x / 4;
        int aColBase = tk * BLOCK_K + (threadIdx.x % 4) * 8;
        
        #pragma unroll
        for (int i = 0; i < 8; i++) {
            int col = aColBase + i;
            if (aRow < N && col < N) {
                As[(threadIdx.x / 4) * BLOCK_K + (threadIdx.x % 4) * 8 + i] = A[aRow * N + col];
            } else {
                As[(threadIdx.x / 4) * BLOCK_K + (threadIdx.x % 4) * 8 + i] = __float2half(0.0f);
            }
        }
        
        // Load B tile - transposed layout for coalesced access
        int bRowBase = tk * BLOCK_K + threadIdx.x / 4;
        int bCol = bx * BLOCK_N + (threadIdx.x % 4) * 8;
        
        #pragma unroll
        for (int i = 0; i < 8; i++) {
            int row = bRowBase + i;
            if (row < N && bCol < N) {
                Bs[(threadIdx.x % 4) * 8 * BLOCK_N + (threadIdx.x / 4) + i * BLOCK_N] = B[row * N + bCol];
            } else {
                Bs[(threadIdx.x % 4) * 8 * BLOCK_N + (threadIdx.x / 4) + i * BLOCK_N] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // WMMA compute
        for (int k = 0; k < BLOCK_K; k += WMMA_K) {
            int aOffset = warpM * WMMA_M * BLOCK_K + k;
            int bOffset = k * BLOCK_N + warpN * WMMA_N;
            
            wmma::load_matrix_sync(a_frag, As + aOffset, BLOCK_K);
            wmma::load_matrix_sync(b_frag, Bs + bOffset, BLOCK_N);
            wmma::mma_sync(c_frag, a_frag, b_frag, c_frag);
        }
        
        __syncthreads();
    }
    
    // Store result
    int outRow = by * BLOCK_M + warpM * WMMA_M + laneId / 4;
    int outCol = bx * BLOCK_N + warpN * WMMA_N + (laneId % 4) * 2;
    
    if (outRow < N && outCol < N) {
        wmma::store_matrix_sync(C + outRow * N + outCol, c_frag, N, wmma::mem_row_major);
    }
}

torch::Tensor wmma_matmul(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    
    // Get device and dtype
    auto device = A.device();
    auto dtype = A.dtype();
    
    // Create output tensor in FP16
    auto options = torch::TensorOptions().dtype(at::kHalf).device(device);
    auto C = torch::empty({N, N}, options);
    
    // Convert to FP16 for tensor cores
    torch::Tensor A_half = A.dtype() == at::kHalf ? A : A.to(at::kHalf);
    torch::Tensor B_half = B.dtype() == at::kHalf ? B : B.to(at::kHalf);
    
    // Grid and block dimensions
    dim3 blockDim(128);  // 4 warps (2x2)
    dim3 gridDim((N + BLOCK_N - 1) / BLOCK_N, (N + BLOCK_M - 1) / BLOCK_M);
    
    // Launch kernel
    wmma_gemm_kernel<<<gridDim, blockDim>>>(
        (const half*)A_half.data_ptr<at::Half>(),
        (const half*)B_half.data_ptr<at::Half>(),
        (half*)C.data_ptr<at::Half>(),
        N
    );
    
    // Convert back to original dtype if needed
    if (dtype != at::kHalf) {
        C = C.to(dtype);
    }
    
    return C;
}
'''

ext = load_inline(
    name='wmma_gemm_ext',
    cpp_sources='torch::Tensor wmma_matmul(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['wmma_matmul'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86', '-use_fast_math', '-O3'],
)

class Model(nn.Module):
    """
    Optimized matrix multiplication using WMMA tensor cores
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.wmma_matmul(A, B)

N = 2048
OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp16", "bf16", "fp32"]

def get_inputs():
    A = torch.randn(N, N)
    B = torch.randn(N, N)
    return [A, B]

def get_init_inputs():
    return []
