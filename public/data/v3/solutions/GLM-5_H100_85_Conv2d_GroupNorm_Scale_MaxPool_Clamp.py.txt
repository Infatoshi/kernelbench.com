import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def compute_group_stats_kernel(
    x_ptr,
    mean_ptr,
    rstd_ptr,
    N, C, H, W,
    num_groups,
    eps,
    x_stride_n, x_stride_c, x_stride_h, x_stride_w,
):
    """Compute mean and rstd for each group."""
    pid_n = tl.program_id(0)
    pid_g = tl.program_id(1)
    
    channels_per_group = C // num_groups
    group_size = channels_per_group * H * W
    
    # Compute mean and variance for this (n, g) pair
    sum_val = 0.0
    sum_sq_val = 0.0
    
    for cg in range(channels_per_group):
        c = pid_g * channels_per_group + cg
        for h in range(H):
            for w in range(W):
                offset = (pid_n * x_stride_n + c * x_stride_c + 
                          h * x_stride_h + w * x_stride_w)
                val = tl.load(x_ptr + offset).to(tl.float32)
                sum_val += val
                sum_sq_val += val * val
    
    mean = sum_val / group_size
    var = sum_sq_val / group_size - mean * mean
    rstd = tl.rsqrt(var + eps)
    
    # Store mean and rstd
    tl.store(mean_ptr + pid_n * num_groups + pid_g, mean)
    tl.store(rstd_ptr + pid_n * num_groups + pid_g, rstd)


@triton.jit
def fused_norm_pool_clamp_kernel(
    x_ptr,
    gn_weight_ptr,
    gn_bias_ptr,
    scale_ptr,
    mean_ptr,
    rstd_ptr,
    out_ptr,
    N, C, H, W,
    num_groups,
    clamp_min, clamp_max,
    pool_kernel: tl.constexpr,
    x_stride_n, x_stride_c, x_stride_h, x_stride_w,
):
    """Fused GroupNorm + Scale + MaxPool + Clamp using precomputed stats."""
    pid_n = tl.program_id(0)
    pid_c = tl.program_id(1)
    pid_hw = tl.program_id(2)
    
    out_h = H // pool_kernel
    out_w = W // pool_kernel
    out_hw = out_h * out_w
    
    pid_oh = pid_hw // out_w
    pid_ow = pid_hw % out_w
    
    if pid_hw >= out_hw:
        return
    
    # Get group index and load precomputed stats
    channels_per_group = C // num_groups
    group_idx = pid_c // channels_per_group
    
    mean = tl.load(mean_ptr + pid_n * num_groups + group_idx)
    rstd = tl.load(rstd_ptr + pid_n * num_groups + group_idx)
    
    # Load GroupNorm weight and bias, and scale
    gn_w = tl.load(gn_weight_ptr + pid_c).to(tl.float32)
    gn_b = tl.load(gn_bias_ptr + pid_c).to(tl.float32)
    scale_val = tl.load(scale_ptr + pid_c).to(tl.float32)
    
    # MaxPool: find max in the pool window
    max_val = -1e30
    
    for ph in range(pool_kernel):
        for pw in range(pool_kernel):
            ih = pid_oh * pool_kernel + ph
            iw = pid_ow * pool_kernel + pw
            
            # Load conv output
            offset = (pid_n * x_stride_n + pid_c * x_stride_c + 
                      ih * x_stride_h + iw * x_stride_w)
            val = tl.load(x_ptr + offset).to(tl.float32)
            
            # GroupNorm
            normalized = (val - mean) * rstd
            normalized = normalized * gn_w + gn_b
            
            # Scale
            normalized = normalized * scale_val
            
            # MaxPool
            max_val = tl.maximum(max_val, normalized)
    
    # Clamp
    result = tl.maximum(clamp_min, tl.minimum(clamp_max, max_val))
    
    # Store output
    out_offset = pid_n * C * out_hw + pid_c * out_hw + pid_hw
    tl.store(out_ptr + out_offset, result)


def compute_group_stats(x, num_groups, eps):
    """Compute mean and rstd for each group."""
    N, C, H, W = x.shape
    
    mean = torch.empty((N, num_groups), device=x.device, dtype=torch.float32)
    rstd = torch.empty((N, num_groups), device=x.device, dtype=torch.float32)
    
    grid = (N, num_groups)
    
    x_stride = x.stride()
    
    compute_group_stats_kernel[grid](
        x, mean, rstd,
        N, C, H, W,
        num_groups, eps,
        x_stride[0], x_stride[1], x_stride[2], x_stride[3],
    )
    
    return mean, rstd


def fused_norm_pool_clamp(x, gn_weight, gn_bias, scale, mean, rstd, num_groups, clamp_min, clamp_max, pool_kernel):
    """Fused GroupNorm + Scale + MaxPool + Clamp."""
    N, C, H, W = x.shape
    out_h = H // pool_kernel
    out_w = W // pool_kernel
    out_hw = out_h * out_w
    
    output = torch.empty((N, C, out_h, out_w), device=x.device, dtype=x.dtype)
    
    grid = (N, C, out_hw)
    
    x_stride = x.stride()
    
    fused_norm_pool_clamp_kernel[grid](
        x, gn_weight, gn_bias, scale, mean, rstd, output,
        N, C, H, W,
        num_groups, clamp_min, clamp_max,
        pool_kernel,
        x_stride[0], x_stride[1], x_stride[2], x_stride[3],
    )
    
    return output


class Model(nn.Module):
    """
    Model that performs convolution, group normalization, scaling, max pooling, and clamping.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool = nn.MaxPool2d(kernel_size=maxpool_kernel_size)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        
        self.num_groups = num_groups
        self.eps = 1e-5

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, height, width).
        Returns:
            Output tensor of shape (batch_size, out_channels, height', width').
        """
        batch_size, in_channels, in_h, in_w = x.shape
        out_channels = self.conv.out_channels
        kernel_size = self.conv.kernel_size[0]
        padding = self.conv.padding[0]
        stride = self.conv.stride[0]
        
        out_h = (in_h + 2 * padding - kernel_size) // stride + 1
        out_w = (in_w + 2 * padding - kernel_size) // stride + 1
        
        # Conv2d via unfold + reduce
        x_unfold = torch.nn.functional.unfold(x, kernel_size, padding=padding, stride=stride)
        weight_flat = self.conv.weight.view(out_channels, -1)
        
        # Efficient conv using einsum-style broadcasting
        x_unfold_4d = x_unfold.unsqueeze(1)  
        weight_4d = weight_flat.unsqueeze(0).unsqueeze(-1)  
        conv_out_flat = (x_unfold_4d * weight_4d).sum(dim=2)
        
        if self.conv.bias is not None:
            conv_out_flat = conv_out_flat + self.conv.bias.view(1, -1, 1)
        
        conv_out = conv_out_flat.view(batch_size, out_channels, out_h, out_w)
        
        # Compute group stats
        mean, rstd = compute_group_stats(conv_out, self.num_groups, self.eps)
        
        # Fused GroupNorm + Scale + MaxPool + Clamp
        result = fused_norm_pool_clamp(
            conv_out, 
            self.group_norm.weight, 
            self.group_norm.bias,
            self.scale.squeeze(),
            mean, rstd,
            self.num_groups,
            self.clamp_min,
            self.clamp_max,
            self.maxpool.kernel_size
        )
        
        return result


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
