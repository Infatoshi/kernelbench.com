import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def cross_entropy_kernel(x_ptr, t_ptr, out_ptr, B, C, stride_xb, stride_xc, BLOCK_C: tl.constexpr):
    b = tl.program_id(0)
    if b >= B:
        return

    offs = tl.arange(0, BLOCK_C)
    x_row_ptr = x_ptr + b * stride_xb

    # compute max
    max_val = -1e30
    for c in range(0, C, BLOCK_C):
        idx = c + offs
        mask = idx < C
        vals = tl.load(x_row_ptr + idx * stride_xc, mask=mask, other=-1e30)
        max_val = tl.maximum(max_val, tl.max(vals, axis=0))

    # compute sum exp
    sum_exp = 0.0
    for c in range(0, C, BLOCK_C):
        idx = c + offs
        mask = idx < C
        vals = tl.load(x_row_ptr + idx * stride_xc, mask=mask, other=-1e30)
        vals = vals - max_val
        sum_exp += tl.sum(tl.exp(vals), axis=0)

    # target
    t = tl.load(t_ptr + b)
    target_val = tl.load(x_row_ptr + t * stride_xc)

    loss = -(target_val - max_val - tl.log(sum_exp))
    tl.store(out_ptr + b, loss)


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        B, C = predictions.shape
        x = predictions
        if x.dtype != torch.float32:
            x = x.float()

        out = torch.empty(B, device=x.device, dtype=torch.float32)

        grid = (B,)
        cross_entropy_kernel[grid](
            x, targets, out,
            B, C,
            x.stride(0), x.stride(1),
            BLOCK_C=128
        )

        return out.mean()
