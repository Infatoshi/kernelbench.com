import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused BN + Scale kernel: applies BN inference and scaling in one pass
# BN inference: y = (x - mean) / sqrt(var + eps) * gamma + beta
# then scale: y = y * scaling_factor
# Combined: y = x * (gamma * scaling_factor / sqrt(var + eps))
#             + (beta - mean * gamma / sqrt(var + eps)) * scaling_factor
# = x * A + B  per channel

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fused BN-inference + scale kernel
// Input: conv output [N, C, H, W] in NCHW layout
// bn_weight = gamma, bn_bias = beta, bn_mean = running_mean, bn_var = running_var
// Computes: out[n,c,h,w] = (in[n,c,h,w] - mean[c]) / sqrt(var[c] + eps) * gamma[c] + beta[c]) * scale
// Simplified to: out = in * A[c] + B[c]
// where A[c] = gamma[c] * scale / sqrt(var[c] + eps)
//       B[c] = (beta[c] - mean[c] * gamma[c] / sqrt(var[c] + eps)) * scale

__global__ void bn_scale_fused_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ A,  // per-channel multiplier
    const float* __restrict__ B,  // per-channel bias
    int N, int C, int HW
) {
    // Grid: (N*HW / blockDim.x, C) - each block handles one channel slice
    int c = blockIdx.y;
    int nhw_base = blockIdx.x * blockDim.x + threadIdx.x;
    
    float a = A[c];
    float b = B[c];
    
    // Each thread processes one element in the spatial+batch dimension
    if (nhw_base < N * HW) {
        // Compute the linear index: for NCHW, element [n,c,h,w] is at n*C*HW + c*HW + hw
        // We iterate over all n
        int stride = C * HW;
        // nhw_base encodes (n * HW + hw) but we need to spread across N
        // Actually let's use: thread processes index = n*HW + hw for fixed c
        // So global index = n*C*HW + c*HW + hw
        int n = nhw_base / HW;
        int hw = nhw_base % HW;
        int idx = n * C * HW + c * HW + hw;
        output[idx] = input[idx] * a + b;
    }
}

// Vectorized version using float4 for better memory throughput
__global__ void bn_scale_fused_kernel_v4(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ A,
    const float* __restrict__ B,
    int N, int C, int HW
) {
    int c = blockIdx.y;
    float a = A[c];
    float b = B[c];
    
    int nhw4 = (N * HW) / 4;
    int idx4 = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx4 < nhw4) {
        // Find which n this idx4 belongs to
        // base pointer for channel c
        const float4* in4 = reinterpret_cast<const float4*>(input + c * HW);
        float4* out4 = reinterpret_cast<float4*>(output + c * HW);
        
        // But we have N batches, each separated by C*HW
        // This approach works only if we treat it differently
        // Let's use per-element approach
        int base = idx4 * 4;
        int n = base / HW;
        int hw = base % HW;
        
        // Check we don't cross N boundary in this float4
        if (n < N && hw + 3 < HW) {
            int global_idx = n * C * HW + c * HW + hw;
            float4 val = *reinterpret_cast<const float4*>(input + global_idx);
            float4 res;
            res.x = val.x * a + b;
            res.y = val.y * a + b;
            res.z = val.z * a + b;
            res.w = val.w * a + b;
            *reinterpret_cast<float4*>(output + global_idx) = res;
        } else {
            // Scalar fallback for boundary
            for (int i = 0; i < 4; i++) {
                int bi = base + i;
                if (bi < N * HW) {
                    int ni = bi / HW;
                    int hwi = bi % HW;
                    int global_idx = ni * C * HW + c * HW + hwi;
                    output[global_idx] = input[global_idx] * a + b;
                }
            }
        }
    }
}

// Better vectorized: process spatially contiguous float4s
// For NCHW, within a channel slice [n,c,:,:] elements are contiguous
// So for each (n,c) pair, we have HW contiguous floats
__global__ void bn_scale_fused_kernel_nc(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ A,
    const float* __restrict__ B,
    int N, int C, int HW
) {
    // blockIdx.x = (n*C + c) chunk index
    // threadIdx.x = spatial offset within that chunk
    int nc = blockIdx.x;
    int c = nc % C;
    float a = A[c];
    float b = B[c];
    
    const float* in_ptr = input + nc * HW;
    float* out_ptr = output + nc * HW;
    
    // Process HW elements with stride blockDim.x, using float4
    int tid = threadIdx.x;
    int hw4 = HW / 4;
    
    for (int i = tid; i < hw4; i += blockDim.x) {
        float4 val = reinterpret_cast<const float4*>(in_ptr)[i];
        float4 res;
        res.x = val.x * a + b;
        res.y = val.y * a + b;
        res.z = val.z * a + b;
        res.w = val.w * a + b;
        reinterpret_cast<float4*>(out_ptr)[i] = res;
    }
    // Handle remainder
    for (int i = (hw4 * 4) + tid; i < HW; i += blockDim.x) {
        out_ptr[i] = in_ptr[i] * a + b;
    }
}

torch::Tensor bn_scale_fused(
    torch::Tensor input,       // [N, C, H, W]
    torch::Tensor bn_weight,   // [C] gamma
    torch::Tensor bn_bias,     // [C] beta
    torch::Tensor bn_mean,     // [C] running_mean
    torch::Tensor bn_var,      // [C] running_var
    float eps,
    float scale
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int HW = H * W;
    
    auto output = torch::empty_like(input);
    
    // Precompute A[c] and B[c] on GPU
    // A[c] = gamma[c] * scale / sqrt(var[c] + eps)
    // B[c] = (beta[c] - mean[c] * gamma[c] / sqrt(var[c] + eps)) * scale
    auto inv_std = torch::rsqrt(bn_var + eps);
    auto A = bn_weight * inv_std * scale;
    auto B = (bn_bias - bn_mean * bn_weight * inv_std) * scale;
    
    // Launch kernel: one block per (n,c) pair, threads handle spatial
    // NC blocks, each processing HW elements
    int block_size = 256;
    dim3 grid(N * C);
    dim3 block(block_size);
    
    bn_scale_fused_kernel_nc<<<grid, block>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        N, C, HW
    );
    
    return output;
}
'''

cpp_source = '''
torch::Tensor bn_scale_fused(
    torch::Tensor input,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    float eps,
    float scale
);
'''

ext = load_inline(
    name='bn_scale_fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bn_scale_fused'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    Optimized model: Conv2d + fused BatchNorm + Scaling
    - Conv2d uses cuDNN (fastest on RTX 3090)
    - BatchNorm inference + scaling fused into a single custom CUDA kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = float(scaling_factor)

    def forward(self, x):
        # Step 1: Convolution via cuDNN (fastest available)
        x = self.conv(x)
        
        # Step 2: Fused BN inference + scaling
        if self.training:
            # During training, use standard BN (updates running stats)
            x = self.bn(x)
            x = x * self.scaling_factor
        else:
            # During inference, use fused kernel
            x = ext.bn_scale_fused(
                x,
                self.bn.weight,
                self.bn.bias,
                self.bn.running_mean,
                self.bn.running_var,
                self.bn.eps,
                self.scaling_factor
            )
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor]
