import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fused particle integration kernel
// Processes 2D particles: position, velocity update with gravity, boundary bounce, and clamp
__global__ void particle_integrate_kernel(
    const float* __restrict__ positions,
    const float* __restrict__ velocities,
    float* __restrict__ output,
    int n,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping
) {
    // Each thread processes one particle (2 floats for pos, 2 for vel)
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx < n) {
        // Load position and velocity (each is 2D)
        float pos_x = positions[idx * 2];
        float pos_y = positions[idx * 2 + 1];
        float vel_x = velocities[idx * 2];
        float vel_y = velocities[idx * 2 + 1];
        
        // Update velocity with gravity
        float new_vel_x = vel_x + gravity_x * dt;
        float new_vel_y = vel_y + gravity_y * dt;
        
        // Update position with new velocity
        float new_pos_x = pos_x + new_vel_x * dt;
        float new_pos_y = pos_y + new_vel_y * dt;
        
        // Boundary collision detection and response
        // Bounce from normalized viewport bounds [-1, 1]
        if (fabsf(new_pos_x) > 1.0f) {
            new_vel_x = -new_vel_x * damping;
            new_pos_x = fmaxf(-1.0f, fminf(1.0f, new_pos_x));
        }
        if (fabsf(new_pos_y) > 1.0f) {
            new_vel_y = -new_vel_y * damping;
            new_pos_y = fmaxf(-1.0f, fminf(1.0f, new_pos_y));
        }
        
        // Store output: [new_pos, new_vel] concatenated
        output[idx * 4] = new_pos_x;
        output[idx * 4 + 1] = new_pos_y;
        output[idx * 4 + 2] = new_vel_x;
        output[idx * 4 + 3] = new_vel_y;
    }
}

// Vectorized version using float4 for better memory throughput
__global__ void particle_integrate_kernel_vec4(
    const float4* __restrict__ pos_vel,  // [pos_x, pos_y, vel_x, vel_y] per particle
    float4* __restrict__ output,
    int n,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx < n) {
        // Load position and velocity as float4
        float4 pv = pos_vel[idx];
        float pos_x = pv.x;
        float pos_y = pv.y;
        float vel_x = pv.z;
        float vel_y = pv.w;
        
        // Update velocity with gravity
        float new_vel_x = vel_x + gravity_x * dt;
        float new_vel_y = vel_y + gravity_y * dt;
        
        // Update position with new velocity
        float new_pos_x = pos_x + new_vel_x * dt;
        float new_pos_y = pos_y + new_vel_y * dt;
        
        // Boundary collision detection and response
        if (fabsf(new_pos_x) > 1.0f) {
            new_vel_x = -new_vel_x * damping;
            new_pos_x = fmaxf(-1.0f, fminf(1.0f, new_pos_x));
        }
        if (fabsf(new_pos_y) > 1.0f) {
            new_vel_y = -new_vel_y * damping;
            new_pos_y = fmaxf(-1.0f, fminf(1.0f, new_pos_y));
        }
        
        // Store output
        output[idx] = make_float4(new_pos_x, new_pos_y, new_vel_x, new_vel_y);
    }
}

torch::Tensor particle_integrate(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping
) {
    // positions: [n, 2], velocities: [n, 2]
    int n = positions.size(0);
    
    // Output: [n, 4] - concatenated position and velocity
    auto output = torch::empty({n, 4}, positions.options());
    
    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    
    particle_integrate_kernel<<<blocks, threads>>>(
        positions.data_ptr<float>(),
        velocities.data_ptr<float>(),
        output.data_ptr<float>(),
        n,
        dt,
        gravity_x,
        gravity_y,
        damping
    );
    
    return output;
}

torch::Tensor particle_integrate_vec4(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping
) {
    int n = positions.size(0);
    
    // Create interleaved input
    auto pv = torch::stack({positions.select(1, 0), positions.select(1, 1), 
                            velocities.select(1, 0), velocities.select(1, 1)}, 1);
    auto pv_contiguous = pv.contiguous().view({n, 4});
    
    auto output = torch::empty_like(pv_contiguous);
    
    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    
    particle_integrate_kernel_vec4<<<blocks, threads>>>(
        reinterpret_cast<const float4*>(pv_contiguous.data_ptr<float>()),
        reinterpret_cast<float4*>(output.data_ptr<float>()),
        n,
        dt,
        gravity_x,
        gravity_y,
        damping
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor particle_integrate(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping
);
'''

ext = load_inline(
    name='particle_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_integrate'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)

class Model(nn.Module):
    """Simple GPU particle integration step with boundary collisions."""

    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        gravity_vals = gravity.tolist() if isinstance(gravity, torch.Tensor) else gravity
        
        return ext.particle_integrate(
            positions,
            velocities,
            dt_value,
            gravity_vals[0],
            gravity_vals[1],
            self.damping
        )


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
