import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <limits>

template <typename scalar_t>
__global__ void maxpool2d_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    int OH, int OW,
    int k, int stride, int pad, int dil
){
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*C*OH*OW;
    for(int i = idx; i < total; i += blockDim.x * gridDim.x){
        int ow = i % OW;
        int oh = (i / OW) % OH;
        int c = (i / (OW*OH)) % C;
        int n = i / (OW*OH*C);

        int hstart = oh * stride - pad;
        int wstart = ow * stride - pad;

        scalar_t maxv = (scalar_t)(-1e20f);

        for(int kh=0; kh<k; ++kh){
            int ih = hstart + kh * dil;
            if(ih < 0 || ih >= H) continue;
            for(int kw=0; kw<k; ++kw){
                int iw = wstart + kw * dil;
                if(iw < 0 || iw >= W) continue;
                int in_idx = ((n*C + c)*H + ih)*W + iw;
                scalar_t v = x[in_idx];
                if(v > maxv) maxv = v;
            }
        }
        y[i] = maxv;
    }
}

torch::Tensor maxpool2d_forward(torch::Tensor x, int k, int stride, int pad, int dil){
    x = x.contiguous();

    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OH = (H + 2*pad - dil*(k-1) - 1)/stride + 1;
    int OW = (W + 2*pad - dil*(k-1) - 1)/stride + 1;

    auto y = torch::empty({N, C, OH, OW}, x.options());

    int total = N*C*OH*OW;
    int threads = 256;
    int blocks = (total + threads - 1)/threads;
    blocks = blocks > 4096 ? 4096 : blocks;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "maxpool2d_cuda", ([&] {
        maxpool2d_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            N,C,H,W,OH,OW,k,stride,pad,dil
        );
    }));

    return y;
}
'''

ext = load_inline(
    name='maxpool2d_ext',
    cpp_sources='torch::Tensor maxpool2d_forward(torch::Tensor x, int k, int stride, int pad, int dil);',
    cuda_sources=cuda_source,
    functions=['maxpool2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super().__init__()
        self.k = kernel_size
        self.s = stride
        self.p = padding
        self.d = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d_forward(x, self.k, self.s, self.p, self.d)

batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
