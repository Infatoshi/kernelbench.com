import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math_constants.h>

__device__ __forceinline__ float to_float(float x) { return x; }
__device__ __forceinline__ float to_float(__half x) { return __half2float(x); }
__device__ __forceinline__ float to_float(__nv_bfloat16 x) { return __bfloat162float(x); }

template <typename T>
__device__ __forceinline__ T from_float(float x);

template <>
__device__ __forceinline__ float from_float(float x) { return x; }

template <>
__device__ __forceinline__ __half from_float(float x) { return __float2half(x); }

template <>
__device__ __forceinline__ __nv_bfloat16 from_float(float x) { return __float2bfloat16(x); }

template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, T* __restrict__ output, int rows, int cols) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    const int block_size = blockDim.x;
    
    if (row >= rows) return;
    
    const T* row_input = input + row * cols;
    T* row_output = output + row * cols;
    
    extern __shared__ float shared_mem[];
    
    // Step 1: Find max value in the row
    float local_max = -CUDART_INF_F;
    for (int i = tid; i < cols; i += block_size) {
        float val = to_float(row_input[i]);
        local_max = fmaxf(local_max, val);
    }
    
    // Warp reduction for max
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        local_max = fmaxf(local_max, __shfl_down_sync(0xffffffff, local_max, offset));
    }
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    int num_warps = (block_size + 31) / 32;
    
    if (lane_id == 0) {
        shared_mem[warp_id] = local_max;
    }
    __syncthreads();
    
    // Block reduction for max
    if (warp_id == 0) {
        local_max = (lane_id < num_warps) ? shared_mem[lane_id] : -CUDART_INF_F;
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            local_max = fmaxf(local_max, __shfl_down_sync(0xffffffff, local_max, offset));
        }
        if (lane_id == 0) {
            shared_mem[0] = local_max;
        }
    }
    __syncthreads();
    
    float row_max = shared_mem[0];
    
    // Step 2: Compute exp(x - max) and sum
    float local_sum = 0.0f;
    for (int i = tid; i < cols; i += block_size) {
        float val = expf(to_float(row_input[i]) - row_max);
        local_sum += val;
        row_output[i] = from_float<T>(val);
    }
    
    // Warp reduction for sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    
    if (lane_id == 0) {
        shared_mem[warp_id] = local_sum;
    }
    __syncthreads();
    
    // Block reduction for sum
    if (warp_id == 0) {
        local_sum = (lane_id < num_warps) ? shared_mem[lane_id] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
        }
        if (lane_id == 0) {
            shared_mem[0] = local_sum;
        }
    }
    __syncthreads();
    
    float row_sum = shared_mem[0];
    float inv_sum = 1.0f / row_sum;
    
    // Step 3: Normalize
    for (int i = tid; i < cols; i += block_size) {
        float val = to_float(row_output[i]);
        row_output[i] = from_float<T>(val * inv_sum);
    }
}

torch::Tensor softmax_op(torch::Tensor input) {
    TORCH_CHECK(input.is_cuda(), "Input must be a CUDA tensor");
    TORCH_CHECK(input.dim() == 2, "Input must be 2D");
    
    int rows = input.size(0);
    int cols = input.size(1);
    
    auto output = torch::empty_like(input);
    
    const int block_size = 256;
    const int num_warps = (block_size + 31) / 32;
    size_t shared_mem_size = num_warps * sizeof(float);
    
    if (input.scalar_type() == at::ScalarType::Float) {
        softmax_kernel<float><<<rows, block_size, shared_mem_size>>>(
            input.data_ptr<float>(), output.data_ptr<float>(), rows, cols);
    } else if (input.scalar_type() == at::ScalarType::Half) {
        softmax_kernel<__half><<<rows, block_size, shared_mem_size>>>(
            reinterpret_cast<const __half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
            rows, cols);
    } else if (input.scalar_type() == at::ScalarType::BFloat16) {
        softmax_kernel<__nv_bfloat16><<<rows, block_size, shared_mem_size>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            rows, cols);
    } else {
        AT_ERROR("Unsupported dtype");
    }
    
    return output;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_op(torch::Tensor input);',
    cuda_sources=cuda_source,
    functions=['softmax_op'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, x):
        return ext.softmax_op(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    return [torch.randn(batch_size, dim)]

def get_init_inputs():
    return []
