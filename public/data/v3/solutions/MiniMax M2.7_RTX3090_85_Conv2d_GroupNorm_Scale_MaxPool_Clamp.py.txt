import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fused Scale + MaxPool + Clamp kernel with efficient tiling
// Each block processes one output element, but uses warp-level parallelism for pooling window
template <typename T>
__global__ void fused_scale_maxpool_clamp_kernel(
    const T* __restrict__ gn_out,
    const T* __restrict__ scale,
    T* __restrict__ output,
    int batch, int channels, int height, int width,
    int pool_kernel_size,
    T clamp_min, T clamp_max,
    int out_height, int out_width
) {
    int b = blockIdx.x;
    int c = blockIdx.y;
    int h_out = blockIdx.z / out_width;
    int w_out = blockIdx.z % out_width;
    
    int tid = threadIdx.x;
    int pool_size = pool_kernel_size * pool_kernel_size;
    int lane_id = tid % 32;
    int warp_id = tid / 32;
    
    // Compute pool region
    int h_start = h_out * pool_kernel_size;
    int w_start = w_out * pool_kernel_size;
    
    // Load data for this channel
    T local_vals[4];  // Max 2x2 pool, load 4 elements per thread
    int num_elements = 0;
    
    for (int ph = 0; ph < pool_kernel_size; ph++) {
        for (int pw = 0; pw < pool_kernel_size; pw++) {
            int h = h_start + ph;
            int w = w_start + pw;
            if (h < height && w < width) {
                if (lane_id < 4 && num_elements < 4) {
                    int in_idx = ((b * channels + c) * height + h) * width + w;
                    local_vals[num_elements++] = gn_out[in_idx];
                }
            }
        }
    }
    
    // Get scale for this channel
    T scale_val = scale[c];
    
    // Apply scale to local values
    T scaled_vals[4];
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        if (i < num_elements) {
            scaled_vals[i] = local_vals[i] * scale_val;
        }
    }
    
    // Warp-level reduction for max
    T max_val = scaled_vals[0];
    #pragma unroll
    for (int i = 1; i < 4; i++) {
        max_val = max(max_val, scaled_vals[i]);
    }
    
    // Warp reduce
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        max_val = max(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }
    
    // Only lane 0 writes result
    if (lane_id == 0) {
        // Apply clamp
        max_val = max(min(max_val, clamp_max), clamp_min);
        
        int out_idx = ((b * channels + c) * out_height + h_out) * out_width + w_out;
        output[out_idx] = max_val;
    }
}

// Alternative: simple efficient kernel for small pool sizes
template <typename T>
__global__ void fused_scale_maxpool_clamp_simple_kernel(
    const T* __restrict__ gn_out,
    const T* __restrict__ scale,
    T* __restrict__ output,
    int batch, int channels, int height, int width,
    int pool_kernel_size,
    T clamp_min, T clamp_max,
    int out_height, int out_width
) {
    int b = blockIdx.x;
    int c = blockIdx.y;
    int h_out = blockIdx.z / out_width;
    int w_out = blockIdx.z % out_width;
    
    int h_start = h_out * pool_kernel_size;
    int w_start = w_out * pool_kernel_size;
    
    T max_val = static_cast<T>(-1e10f);
    for (int ph = 0; ph < pool_kernel_size; ph++) {
        for (int pw = 0; pw < pool_kernel_size; pw++) {
            int h = h_start + ph;
            int w = w_start + pw;
            if (h < height && w < width) {
                int in_idx = ((b * channels + c) * height + h) * width + w;
                T val = gn_out[in_idx];
                val = val * scale[c];
                max_val = max(max_val, val);
            }
        }
    }
    
    max_val = max(min(max_val, clamp_max), clamp_min);
    
    int out_idx = ((b * channels + c) * out_height + h_out) * out_width + w_out;
    output[out_idx] = max_val;
}

torch::Tensor fused_scale_maxpool_clamp(
    torch::Tensor gn_out,
    torch::Tensor scale,
    int pool_kernel_size,
    float clamp_min,
    float clamp_max
) {
    int batch = gn_out.size(0);
    int channels = gn_out.size(1);
    int height = gn_out.size(2);
    int width = gn_out.size(3);
    
    int out_height = height / pool_kernel_size;
    int out_width = width / pool_kernel_size;
    
    auto output = torch::empty({batch, channels, out_height, out_width}, 
                               gn_out.options());
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(gn_out.scalar_type(), "fused_scale_maxpool_clamp", ([&] {
        dim3 grid(batch, channels, out_height * out_width);
        
        fused_scale_maxpool_clamp_simple_kernel<scalar_t><<<grid, 1>>>(
            gn_out.data_ptr<scalar_t>(),
            scale.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch, channels, height, width,
            pool_kernel_size,
            static_cast<scalar_t>(clamp_min),
            static_cast<scalar_t>(clamp_max),
            out_height, out_width
        );
    }));
    
    return output;
}
'''

cpp_source = '''
torch::Tensor fused_scale_maxpool_clamp(
    torch::Tensor gn_out,
    torch::Tensor scale,
    int pool_kernel_size,
    float clamp_min,
    float clamp_max
);
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_scale_maxpool_clamp'],
    extra_cflags=['-O3'],
    verbose=False,
)

class Model(nn.Module):
    """
    Fused model: Conv2d + GroupNorm + Scale + MaxPool + Clamp
    Uses PyTorch's cuDNN-optimized Conv2d and GroupNorm, 
    with a custom fused kernel for Scale + MaxPool + Clamp.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        
    def forward(self, x):
        # Conv2d (cuDNN optimized)
        conv_out = self.conv(x)
        
        # GroupNorm (cuDNN optimized)
        gn_out = self.group_norm(conv_out)
        
        # Fused Scale + MaxPool + Clamp
        out = ext.fused_scale_maxpool_clamp(
            gn_out,
            self.scale,
            self.maxpool_kernel_size,
            self.clamp_min,
            self.clamp_max
        )
        return out

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 8, (16, 1, 1), 2, 0.0, 1.0]
