import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAGuard.h>

// Fused kernel: Conv2d -> SiLU (x * tanh(softplus(x))) -> BatchNorm
template<typename scalar_t>
__global__ void fused_conv_bn_silu_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    const scalar_t* __restrict__ bn_weight,
    const scalar_t* __restrict__ bn_bias,
    const scalar_t* __restrict__ bn_running_mean,
    const scalar_t* __restrict__ bn_running_var,
    scalar_t* __restrict__ output,
    int N, int C_in, int H_in, int W_in,
    int C_out, int H_out, int W_out, int K,
    scalar_t eps
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elems = N * C_out * H_out * W_out;
    
    if (idx >= total_elems) return;
    
    // Decode indices
    int tmp = idx;
    int ow = tmp % W_out;
    tmp /= W_out;
    int oh = tmp % H_out;
    tmp /= H_out;
    int oc = tmp % C_out;
    int n = tmp / C_out;
    
    // Compute convolution for this output position
    float conv_val = 0.0f;
    
    for (int ci = 0; ci < C_in; ci++) {
        for (int kh = 0; kh < K; kh++) {
            for (int kw = 0; kw < K; kw++) {
                int ih = oh + kh;
                int iw = ow + kw;
                
                if (ih < H_in && iw < W_in) {
                    int in_idx = n * C_in * H_in * W_in + ci * H_in * W_in + ih * W_in + iw;
                    int w_idx = oc * C_in * K * K + ci * K * K + kh * K + kw;
                    
                    conv_val += (float)input[in_idx] * (float)weight[w_idx];
                }
            }
        }
    }
    
    // Add bias
    conv_val += (float)bias[oc];
    
    // Apply SiLU activation: x * tanh(softplus(x))
    // softplus(x) = log(1 + exp(x))
    float softplus_val;
    if (conv_val > 0) {
        softplus_val = conv_val + logf(1.0f + expf(-conv_val));
    } else {
        softplus_val = logf(1.0f + expf(conv_val));
    }
    
    // tanh(softplus_val)
    float exp_2sp = expf(2.0f * softplus_val);
    float tanh_sp = (exp_2sp - 1.0f) / (exp_2sp + 1.0f);
    
    // activated = conv_val * tanh(softplus(conv_val))
    float activated = conv_val * tanh_sp;
    
    // Apply BatchNorm: (x - mean) / sqrt(var + eps) * weight + bias
    float mean = (float)bn_running_mean[oc];
    float var = (float)bn_running_var[oc];
    float bn_w = (float)bn_weight[oc];
    float bn_b = (float)bn_bias[oc];
    
    float normalized = (activated - mean) / sqrtf(var + (float)eps);
    float bn_out = normalized * bn_w + bn_b;
    
    // Store output
    int out_idx = n * C_out * H_out * W_out + oc * H_out * W_out + oh * W_out + ow;
    output[out_idx] = (scalar_t)bn_out;
}

torch::Tensor fused_conv_bn_silu(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    double eps
) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H_in = input.size(2);
    int W_in = input.size(3);
    
    int C_out = weight.size(0);
    int K = weight.size(2);
    
    int H_out = H_in - K + 1;
    int W_out = W_in - K + 1;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    int total_elems = N * C_out * H_out * W_out;
    int threads = 256;
    int blocks = (total_elems + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "fused_conv_bn_silu_kernel", ([&] {
        fused_conv_bn_silu_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            bn_weight.data_ptr<scalar_t>(),
            bn_bias.data_ptr<scalar_t>(),
            bn_running_mean.data_ptr<scalar_t>(),
            bn_running_var.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            N, C_in, H_in, W_in, C_out, H_out, W_out, K,
            (scalar_t)eps
        );
    }));
    
    return output;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("fused_conv_bn_silu", &fused_conv_bn_silu, "Fused Conv2d + SiLU + BatchNorm");
}
'''

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor fused_conv_bn_silu(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    double eps
);
'''

ext = load_inline(
    name='fused_conv_bn_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_bn_silu'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    extra_cflags=['-O3'],
    verbose=False,
)


class Model(nn.Module):
    """
    Optimized fused Conv2d + SiLU activation + BatchNorm
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        return ext.fused_conv_bn_silu(
            x,
            self.conv.weight,
            self.conv.bias,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            self.eps
        )


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
