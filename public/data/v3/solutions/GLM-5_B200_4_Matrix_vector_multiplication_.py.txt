import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// High-performance GEMV using warp-level operations
// Each warp handles one output element (row)
template <typename scalar_t, int WARP_SIZE=32>
__global__ void gemv_kernel_warp(
    const scalar_t* __restrict__ A,
    const scalar_t* __restrict__ B,
    scalar_t* __restrict__ C,
    const int M,
    const int K
) {
    // One warp per row
    const int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / WARP_SIZE;
    const int lane_id = threadIdx.x % WARP_SIZE;
    
    if (warp_id >= M) return;
    
    const scalar_t* A_row = A + warp_id * K;
    
    // Use float for accumulation to avoid precision issues
    float sum = 0.0f;
    
    // Strided access to K - each lane handles K/WARP_SIZE elements
    for (int k = lane_id; k < K; k += WARP_SIZE) {
        sum += static_cast<float>(A_row[k]) * static_cast<float>(B[k]);
    }
    
    // Warp reduction using shuffle
    #pragma unroll
    for (int offset = WARP_SIZE / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        C[warp_id] = static_cast<scalar_t>(sum);
    }
}

torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B) {
    // A: (M, K), B: (K, 1) or (K,), C: (M, 1)
    const int M = A.size(0);
    const int K = A.size(1);
    
    // Squeeze B if needed
    auto B_flat = B.squeeze();
    
    // Output
    auto C = torch::empty({M, 1}, A.options());
    
    // Choose kernel based on problem size
    const int THREADS_PER_BLOCK = 256;
    const int WARP_SIZE = 32;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(A.scalar_type(), "gemv_forward", ([&] {
        // Use warp-based kernel: each warp handles one row
        // M = 256 rows, so we need 256 warps = 8 blocks of 32 warps each
        const int warps_per_block = THREADS_PER_BLOCK / WARP_SIZE;
        const int blocks = (M + warps_per_block - 1) / warps_per_block;
        
        gemv_kernel_warp<scalar_t><<<blocks, THREADS_PER_BLOCK>>>(
            A.data_ptr<scalar_t>(),
            B_flat.data_ptr<scalar_t>(),
            C.data_ptr<scalar_t>(),
            M,
            K
        );
    }));
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
)

class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.gemv_forward(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
