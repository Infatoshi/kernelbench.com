import torch
import torch.nn as nn
import triton
import triton.language as tl

# Parameters from reference
batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

@triton.jit
def maxpool2d_kernel_fused(
    X_ptr, Y_ptr,
    BC,
    H: tl.constexpr, W: tl.constexpr,
    OH: tl.constexpr, OW: tl.constexpr,
    KS: tl.constexpr,
    STRIDE: tl.constexpr,
    PAD: tl.constexpr,
    DIL: tl.constexpr,
    BLOCK_OW: tl.constexpr,
    BLOCK_OH: tl.constexpr,
):
    # 3D grid: (bc, ow_blocks, oh_blocks)
    pid_bc = tl.program_id(0)
    pid_ow = tl.program_id(1)
    pid_oh = tl.program_id(2)
    
    ow_start = pid_ow * BLOCK_OW
    oh_start = pid_oh * BLOCK_OH
    
    ow_offs = ow_start + tl.arange(0, BLOCK_OW)  # [BLOCK_OW]
    oh_offs = oh_start + tl.arange(0, BLOCK_OH)  # [BLOCK_OH]
    
    ow_mask = ow_offs < OW  # [BLOCK_OW]
    oh_mask = oh_offs < OH  # [BLOCK_OH]
    
    # Base h and w positions (before kernel offsets)
    base_h = oh_offs * STRIDE - PAD  # [BLOCK_OH]
    base_w = ow_offs * STRIDE - PAD  # [BLOCK_OW]
    
    x_base = pid_bc * H * W
    y_base = pid_bc * OH * OW
    
    max_val = tl.full([BLOCK_OH, BLOCK_OW], float('-inf'), dtype=tl.float32)
    
    for kh in tl.static_range(KS):
        ih = base_h + kh * DIL  # [BLOCK_OH]
        h_valid = (ih >= 0) & (ih < H)  # [BLOCK_OH]
        
        for kw in tl.static_range(KS):
            iw = base_w + kw * DIL  # [BLOCK_OW]
            w_valid = (iw >= 0) & (iw < W)  # [BLOCK_OW]
            
            # Compute 2D mask and indices
            valid = (h_valid[:, None] & w_valid[None, :]) & (oh_mask[:, None] & ow_mask[None, :])
            x_idx = x_base + ih[:, None] * W + iw[None, :]
            
            val = tl.load(X_ptr + x_idx, mask=valid, other=float('-inf'))
            val_f32 = val.to(tl.float32)
            max_val = tl.maximum(max_val, val_f32)
    
    # Store
    y_idx = y_base + oh_offs[:, None] * OW + ow_offs[None, :]
    out_mask = oh_mask[:, None] & ow_mask[None, :]
    tl.store(Y_ptr + y_idx, max_val, mask=out_mask)


def maxpool2d_triton(x, kernel_size, stride_val, padding, dilation):
    B, C, H, W = x.shape
    OH = (H + 2 * padding - dilation * (kernel_size - 1) - 1) // stride_val + 1
    OW = (W + 2 * padding - dilation * (kernel_size - 1) - 1) // stride_val + 1
    
    BC = B * C
    x_flat = x.reshape(BC, H, W)
    y_flat = torch.empty(BC, OH, OW, device=x.device, dtype=torch.float32)
    
    BLOCK_OW = 128
    BLOCK_OH = 4
    
    grid = (BC, triton.cdiv(OW, BLOCK_OW), triton.cdiv(OH, BLOCK_OH))
    
    maxpool2d_kernel_fused[grid](
        x_flat, y_flat,
        BC,
        H, W,
        OH, OW,
        kernel_size, stride_val, padding, dilation,
        BLOCK_OW, BLOCK_OH,
    )
    
    return y_flat.reshape(B, C, OH, OW).to(x.dtype)


class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return maxpool2d_triton(x, self.kernel_size, self.stride, self.padding, self.dilation)


def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
