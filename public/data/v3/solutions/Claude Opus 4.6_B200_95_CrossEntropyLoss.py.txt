import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

// Warp-level reduce sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;
}

// Warp-level reduce max
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    return val;
}

// Each block processes one row, uses all threads for the row reduction
// C=1024, use 256 threads, each thread handles 4 elements
__global__ void cross_entropy_kernel(
    const float* __restrict__ logits,   // [B, C]
    const long* __restrict__ targets,   // [B]
    float* __restrict__ output,         // [1]
    int B, int C
) {
    int row = blockIdx.x;
    if (row >= B) return;

    const float* row_logits = logits + row * C;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;

    // Step 1: Find max for numerical stability
    float local_max = -FLT_MAX;
    for (int i = tid; i < C; i += blockSize) {
        local_max = fmaxf(local_max, row_logits[i]);
    }

    // Warp reduce max
    local_max = warp_reduce_max(local_max);

    // Block reduce max using shared memory
    __shared__ float smem[32];  // max 32 warps
    int warp_id = tid / 32;
    int lane_id = tid % 32;

    if (lane_id == 0) smem[warp_id] = local_max;
    __syncthreads();

    int num_warps = blockSize / 32;
    if (tid < 32) {
        local_max = (tid < num_warps) ? smem[tid] : -FLT_MAX;
        local_max = warp_reduce_max(local_max);
        if (tid == 0) smem[0] = local_max;
    }
    __syncthreads();
    float row_max = smem[0];

    // Step 2: Compute sum of exp(logits - max)
    float local_sum = 0.0f;
    for (int i = tid; i < C; i += blockSize) {
        local_sum += expf(row_logits[i] - row_max);
    }

    local_sum = warp_reduce_sum(local_sum);
    if (lane_id == 0) smem[warp_id] = local_sum;
    __syncthreads();

    if (tid < 32) {
        local_sum = (tid < num_warps) ? smem[tid] : 0.0f;
        local_sum = warp_reduce_sum(local_sum);
        if (tid == 0) smem[0] = local_sum;
    }
    __syncthreads();
    float log_sum_exp = logf(smem[0]);

    // Step 3: Compute per-sample loss and atomic add
    if (tid == 0) {
        long target = targets[row];
        float target_logit = row_logits[target];
        float loss = (log_sum_exp + row_max - target_logit) / (float)B;
        atomicAdd(output, loss);
    }
}

torch::Tensor cross_entropy_cuda(torch::Tensor logits, torch::Tensor targets) {
    int B = logits.size(0);
    int C = logits.size(1);

    auto output = torch::zeros({1}, logits.options().dtype(torch::kFloat32));

    int threads = 256;
    cross_entropy_kernel<<<B, threads>>>(
        logits.data_ptr<float>(),
        targets.data_ptr<long>(),
        output.data_ptr<float>(),
        B, C
    );

    return output.squeeze();
}
'''

cpp_source = r'''
torch::Tensor cross_entropy_cuda(torch::Tensor logits, torch::Tensor targets);
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_cuda(predictions, targets)


batch_size = 4096
num_classes = 1024
input_shape = (num_classes,)
dim = 1


def get_inputs():
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

    return []
