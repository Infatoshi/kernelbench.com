import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Simple CrossEntropyLoss kernel - one thread per sample
// Computes per-sample loss: loss[i] = -log(softmax(predictions[i, targets[i]]))
// Which simplifies to: -predictions[i, targets[i]] + log(sum(exp(predictions[i, :])))

template <typename scalar_t>
__global__ void cross_entropy_loss_kernel(
    const scalar_t* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    scalar_t* __restrict__ output,
    int64_t batch_size,
    int64_t num_classes) {
    
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (idx >= batch_size) return;
    
    const scalar_t* pred_row = predictions + idx * num_classes;
    int64_t target = targets[idx];
    
    // Find max for numerical stability
    scalar_t max_val = pred_row[0];
    for (int64_t j = 1; j < num_classes; j++) {
        scalar_t val = pred_row[j];
        if (val > max_val) max_val = val;
    }
    
    // Compute sum of exp(x - max)
    scalar_t sum_exp = 0.0;
    for (int64_t j = 0; j < num_classes; j++) {
        sum_exp += expf((float)(pred_row[j] - max_val));
    }
    
    // Compute log(sum_exp)
    scalar_t log_sum_exp = logf((float)sum_exp);
    
    // Final loss: -target_logit + log_sum_exp
    // = -target_logit + max + log(sum(exp(x-max)))
    scalar_t target_logit = pred_row[target];
    output[idx] = -target_logit + max_val + log_sum_exp;
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    int64_t batch_size = predictions.size(0);
    int64_t num_classes = predictions.size(1);
    
    // Output per-sample losses
    auto output = torch::empty({batch_size}, predictions.options());
    
    int threads = 256;
    int blocks = (batch_size + threads - 1) / threads;
    
    if (predictions.dtype() == torch::kFloat32) {
        cross_entropy_loss_kernel<float><<<blocks, threads>>>(
            predictions.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            output.data_ptr<float>(),
            batch_size,
            num_classes);
    } else if (predictions.dtype() == torch::kFloat16) {
        cross_entropy_loss_kernel<at::Half><<<blocks, threads>>>(
            predictions.data_ptr<at::Half>(),
            targets.data_ptr<int64_t>(),
            output.data_ptr<at::Half>(),
            batch_size,
            num_classes);
    } else if (predictions.dtype() == torch::kBFloat16) {
        cross_entropy_loss_kernel<at::BFloat16><<<blocks, threads>>>(
            predictions.data_ptr<at::BFloat16>(),
            targets.data_ptr<int64_t>(),
            output.data_ptr<at::BFloat16>(),
            batch_size,
            num_classes);
    }
    
    return output;
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='torch::Tensor cross_entropy_loss(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        # Compute per-sample losses and return mean (like torch.nn.functional.cross_entropy default)
        per_sample_losses = ext.cross_entropy_loss(predictions, targets)
        return per_sample_losses.mean()

def get_inputs():
    return [torch.randn(4096, 1024, device='cuda'), torch.randint(0, 1024, (4096,), device='cuda')]

def get_init_inputs():
    return []
