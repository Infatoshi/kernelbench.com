import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

# Fused Scale + MaxPool + Clamp CUDA kernel
# This fuses the last 3 operations: scale multiply, maxpool, and clamp
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename T>
__device__ __forceinline__ T load_val(const T* ptr, int idx) {
    return ptr[idx];
}

__global__ void fused_scale_maxpool_clamp_fp32(
    const float* __restrict__ x,
    float* __restrict__ y,
    const float* __restrict__ scale,
    const float* __restrict__ bias,
    int batch_size,
    int num_channels,
    int in_height,
    int in_width,
    int pool_kernel_size,
    float clamp_min,
    float clamp_max
) {
    int out_height = in_height / pool_kernel_size;
    int out_width = in_width / pool_kernel_size;
    
    // Each block processes one channel in one batch
    int batch_idx = blockIdx.x;
    int channel_idx = blockIdx.y;
    int out_idx = blockIdx.z;
    
    int out_y = out_idx / out_width;
    int out_x = out_idx % out_width;
    
    if (batch_idx >= batch_size || channel_idx >= num_channels) return;
    
    // Compute the max over the pool window
    float max_val = -INFINITY;
    
    #pragma unroll 4
    for (int ky = 0; ky < pool_kernel_size; ky++) {
        int in_y = out_y * pool_kernel_size + ky;
        #pragma unroll 4
        for (int kx = 0; kx < pool_kernel_size; kx++) {
            int in_x = out_x * pool_kernel_size + kx;
            int in_idx = batch_idx * num_channels * in_height * in_width 
                       + channel_idx * in_height * in_width 
                       + in_y * in_width + in_x;
            float val = x[in_idx];
            if (val > max_val) max_val = val;
        }
    }
    
    // Apply scale, bias, and clamp
    float scaled = max_val * scale[channel_idx] + bias[channel_idx];
    float clamped = fmaxf(fminf(scaled, clamp_max), clamp_min);
    
    // Write output
    int out_idx_final = batch_idx * num_channels * out_height * out_width 
                      + channel_idx * out_height * out_width 
                      + out_y * out_width + out_x;
    y[out_idx_final] = clamped;
}

torch::Tensor fused_scale_maxpool_clamp_cuda(
    torch::Tensor x,
    torch::Tensor scale,
    torch::Tensor bias,
    int pool_kernel_size,
    float clamp_min,
    float clamp_max
) {
    const int batch_size = x.size(0);
    const int num_channels = x.size(1);
    const int in_height = x.size(2);
    const int in_width = x.size(3);
    const int out_height = in_height / pool_kernel_size;
    const int out_width = in_width / pool_kernel_size;
    
    auto y = torch::empty({batch_size, num_channels, out_height, out_width}, 
                         x.options());
    
    dim3 blocks(batch_size, num_channels, out_height * out_width);
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "fused_scale_maxpool_clamp_cuda", ([&] {
        if (x.scalar_type() == at::kFloat) {
            fused_scale_maxpool_clamp_fp32<<<blocks, 1>>>(
                x.data_ptr<float>(),
                y.data_ptr<float>(),
                scale.data_ptr<float>(),
                bias.data_ptr<float>(),
                batch_size, num_channels, in_height, in_width,
                pool_kernel_size, clamp_min, clamp_max
            );
        } else if (x.scalar_type() == at::kHalf) {
            fused_scale_maxpool_clamp_fp16<<<blocks, 1>>>(
                x.data_ptr<at::Half>(),
                y.data_ptr<at::Half>(),
                scale.data_ptr<at::Half>(),
                bias.data_ptr<at::Half>(),
                batch_size, num_channels, in_height, in_width,
                pool_kernel_size, clamp_min, clamp_max
            );
        } else if (x.scalar_type() == at::kBFloat16) {
            fused_scale_maxpool_clamp_bf16<<<blocks, 1>>>(
                x.data_ptr<at::BFloat16>(),
                y.data_ptr<at::BFloat16>(),
                scale.data_ptr<at::BFloat16>(),
                bias.data_ptr<at::BFloat16>(),
                batch_size, num_channels, in_height, in_width,
                pool_kernel_size, clamp_min, clamp_max
            );
        }
    }));
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor fused_scale_maxpool_clamp_cuda(
    torch::Tensor x,
    torch::Tensor scale,
    torch::Tensor bias,
    int pool_kernel_size,
    float clamp_min,
    float clamp_max
);
'''

ext = load_inline(
    name='fused_scale_maxpool_clamp',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_scale_maxpool_clamp_cuda'],
    verbose=False,
    extra_cuda_cflags=['-maxrregcount=128'],
)


class Model(nn.Module):
    """
    Optimized model that uses fused GroupNorm output + Scale + MaxPool + Clamp kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.bias = nn.Parameter(torch.zeros(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, height, width).
        Returns:
            Output tensor of shape (batch_size, out_channels, height', width').
        """
        x = self.conv(x)
        x = self.group_norm(x)
        
        # Use fused kernel for Scale + Bias + MaxPool + Clamp
        scale = self.scale.squeeze().contiguous()
        bias = self.bias.squeeze().contiguous()
        x = ext.fused_scale_maxpool_clamp_cuda(
            x, scale, bias, 
            self.maxpool_kernel_size, 
            self.clamp_min, self.clamp_max
        )
        
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
