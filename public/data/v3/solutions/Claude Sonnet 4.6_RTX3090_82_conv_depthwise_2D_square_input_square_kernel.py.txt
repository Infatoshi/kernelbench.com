import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Highly optimized depthwise conv2d CUDA kernel
# For k=3, s=1: uses large tiles with each thread computing multiple outputs
# Thread block: 256 threads (16x16), each thread computes 2x2 outputs
# Tile: 32x32 outputs, shared memory: 34x34 inputs

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

#define CHECK_CUDA(x) TORCH_CHECK(x.is_cuda(), #x " must be a CUDA tensor")

// ============================================================
// k=3, s=1: each thread computes 2x2 outputs
// Block: 16x16 = 256 threads
// Tile: 32x32 outputs => smem 34x34 inputs
// ============================================================
#define BX 16
#define BY 16
#define OX 2
#define OY 2
#define TILE_X (BX * OX)  // 32
#define TILE_Y (BY * OY)  // 32
#define SMEM_X (TILE_X + 2)  // 34
#define SMEM_Y (TILE_Y + 2)  // 34

__global__ void __launch_bounds__(256, 4)
depthwise_k3s1_2x2_f32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int OH, int OW,
    int pad_h, int pad_w,
    bool has_bias
) {
    int nc = blockIdx.z;
    int n = nc / C;
    int c = nc % C;

    int ow_start = blockIdx.x * TILE_X;
    int oh_start = blockIdx.y * TILE_Y;

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int tid = ty * BX + tx;

    // Load 3x3 kernel into registers
    float w00 = weight[c*9 + 0], w01 = weight[c*9 + 1], w02 = weight[c*9 + 2];
    float w10 = weight[c*9 + 3], w11 = weight[c*9 + 4], w12 = weight[c*9 + 5];
    float w20 = weight[c*9 + 6], w21 = weight[c*9 + 7], w22 = weight[c*9 + 8];

    __shared__ float smem[SMEM_Y * SMEM_X];

    // Load (TILE_Y+2) x (TILE_X+2) = 34x34 = 1156 elements
    // 256 threads, so we need ceil(1156/256) = 5 iterations
    const int total = SMEM_Y * SMEM_X;
    const float* in_base = input + (n * C + c) * H * W;

    for (int i = tid; i < total; i += BX * BY) {
        int sy = i / SMEM_X;
        int sx = i % SMEM_X;
        int ih = oh_start + sy - pad_h;
        int iw = ow_start + sx - pad_w;
        float val = 0.0f;
        if ((unsigned)ih < (unsigned)H && (unsigned)iw < (unsigned)W) {
            val = in_base[ih * W + iw];
        }
        smem[i] = val;
    }

    __syncthreads();

    // Each thread computes 2x2 outputs at positions:
    // (oh_start + ty*OY + oy, ow_start + tx*OX + ox) for oy in [0,OY), ox in [0,OX)
    float bias_val = has_bias ? bias[c] : 0.0f;

    #pragma unroll
    for (int oy = 0; oy < OY; oy++) {
        int oh = oh_start + ty * OY + oy;
        if (oh >= OH) continue;
        int ly = ty * OY + oy;  // local y in smem

        #pragma unroll
        for (int ox = 0; ox < OX; ox++) {
            int ow = ow_start + tx * OX + ox;
            if (ow >= OW) continue;
            int lx = tx * OX + ox;  // local x in smem

            // Compute 3x3 conv using smem
            float sum = 
                smem[(ly+0)*SMEM_X + (lx+0)] * w00 +
                smem[(ly+0)*SMEM_X + (lx+1)] * w01 +
                smem[(ly+0)*SMEM_X + (lx+2)] * w02 +
                smem[(ly+1)*SMEM_X + (lx+0)] * w10 +
                smem[(ly+1)*SMEM_X + (lx+1)] * w11 +
                smem[(ly+1)*SMEM_X + (lx+2)] * w12 +
                smem[(ly+2)*SMEM_X + (lx+0)] * w20 +
                smem[(ly+2)*SMEM_X + (lx+1)] * w21 +
                smem[(ly+2)*SMEM_X + (lx+2)] * w22;

            sum += bias_val;
            output[((n * C + c) * OH + oh) * OW + ow] = sum;
        }
    }
}


// ============================================================
// k=3, s=1: each thread computes 4x1 outputs (4 consecutive in x)
// Block: 64x4 = 256 threads
// Tile: 256x4 outputs => smem 258x6 inputs
// ============================================================
#define B2X 64
#define B2Y 4
#define O2X 4
#define T2X (B2X * O2X)  // 256
#define T2Y B2Y          // 4
#define S2X (T2X + 2)    // 258
#define S2Y (T2Y + 2)    // 6

__global__ void __launch_bounds__(256, 3)
depthwise_k3s1_4x1_f32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int OH, int OW,
    int pad_h, int pad_w,
    bool has_bias
) {
    int nc = blockIdx.z;
    int n = nc / C;
    int c = nc % C;

    int ow_start = blockIdx.x * T2X;
    int oh_start = blockIdx.y * T2Y;

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int tid = ty * B2X + tx;

    float w[3][3];
    #pragma unroll
    for (int i = 0; i < 3; i++)
        #pragma unroll
        for (int j = 0; j < 3; j++)
            w[i][j] = weight[c * 9 + i * 3 + j];

    __shared__ float smem[S2Y * S2X];

    const int total = S2Y * S2X;
    const float* in_base = input + (n * C + c) * H * W;

    for (int i = tid; i < total; i += B2X * B2Y) {
        int sy = i / S2X;
        int sx = i % S2X;
        int ih = oh_start + sy - pad_h;
        int iw = ow_start + sx - pad_w;
        float val = 0.0f;
        if ((unsigned)ih < (unsigned)H && (unsigned)iw < (unsigned)W) {
            val = in_base[ih * W + iw];
        }
        smem[i] = val;
    }

    __syncthreads();

    int oh = oh_start + ty;
    if (oh >= OH) return;

    float bias_val = has_bias ? bias[c] : 0.0f;

    #pragma unroll
    for (int ox = 0; ox < O2X; ox++) {
        int ow = ow_start + tx * O2X + ox;
        if (ow >= OW) continue;
        int lx = tx * O2X + ox;

        float sum = 0.0f;
        #pragma unroll
        for (int kh = 0; kh < 3; kh++) {
            const float* row = smem + (ty + kh) * S2X + lx;
            sum += row[0] * w[kh][0] + row[1] * w[kh][1] + row[2] * w[kh][2];
        }
        output[((n * C + c) * OH + oh) * OW + ow] = sum + bias_val;
    }
}


// ============================================================
// General kernel for arbitrary k, stride, padding
// ============================================================
#define GEN_BX 32
#define GEN_BY 8

__global__ void __launch_bounds__(256, 4)
depthwise_general_f32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int KH, int KW,
    int OH, int OW,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    bool has_bias
) {
    int nc = blockIdx.z;
    int n = nc / C;
    int c = nc % C;

    int ow = blockIdx.x * GEN_BX + threadIdx.x;
    int oh = blockIdx.y * GEN_BY + threadIdx.y;

    if (oh >= OH || ow >= OW) return;

    extern __shared__ float ksmem[];
    int tid = threadIdx.y * GEN_BX + threadIdx.x;
    if (tid < KH * KW) ksmem[tid] = weight[c * KH * KW + tid];
    __syncthreads();

    float sum = 0.0f;
    int ih_base = oh * stride_h - pad_h;
    int iw_base = ow * stride_w - pad_w;

    for (int kh = 0; kh < KH; kh++) {
        int ih = ih_base + kh;
        if ((unsigned)ih >= (unsigned)H) continue;
        for (int kw = 0; kw < KW; kw++) {
            int iw = iw_base + kw;
            if ((unsigned)iw >= (unsigned)W) continue;
            sum += input[((n * C + c) * H + ih) * W + iw] * ksmem[kh * KW + kw];
        }
    }

    if (has_bias) sum += bias[c];
    output[((n * C + c) * OH + oh) * OW + ow] = sum;
}


torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias_opt,
    int stride_h, int stride_w,
    int pad_h, int pad_w
) {
    CHECK_CUDA(input);
    CHECK_CUDA(weight);
    
    input = input.contiguous();
    weight = weight.contiguous();
    
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int KH = weight.size(2);
    int KW = weight.size(3);
    
    int OH = (H + 2 * pad_h - KH) / stride_h + 1;
    int OW = (W + 2 * pad_w - KW) / stride_w + 1;
    
    auto output = torch::empty({N, C, OH, OW}, input.options());
    
    bool has_bias = bias_opt.has_value() && bias_opt.value().defined();
    const float* bias_ptr = nullptr;
    if (has_bias) {
        bias_ptr = bias_opt.value().contiguous().data_ptr<float>();
    }
    
    if (KH == 3 && KW == 3 && stride_h == 1 && stride_w == 1) {
        // 2x2 outputs per thread, 16x16 block, 32x32 tile
        dim3 threads(BX, BY);
        dim3 blocks(
            (OW + TILE_X - 1) / TILE_X,
            (OH + TILE_Y - 1) / TILE_Y,
            N * C
        );
        depthwise_k3s1_2x2_f32<<<blocks, threads>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias_ptr,
            output.data_ptr<float>(),
            N, C, H, W, OH, OW,
            pad_h, pad_w,
            has_bias
        );
    } else {
        dim3 threads(GEN_BX, GEN_BY);
        dim3 blocks(
            (OW + GEN_BX - 1) / GEN_BX,
            (OH + GEN_BY - 1) / GEN_BY,
            N * C
        );
        int smem_size = KH * KW * sizeof(float);
        depthwise_general_f32<<<blocks, threads, smem_size>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias_ptr,
            output.data_ptr<float>(),
            N, C, H, W,
            KH, KW, OH, OW,
            stride_h, stride_w,
            pad_h, pad_w,
            has_bias
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias_opt,
    int stride_h, int stride_w,
    int pad_h, int pad_w
);
'''

ext = load_inline(
    name='depthwise_conv2d_ext_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    """
    Optimized depthwise 2D convolution with square input and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.use_bias = bias
        
        self.weight = nn.Parameter(
            torch.empty(in_channels, 1, kernel_size, kernel_size)
        )
        if bias:
            self.bias = nn.Parameter(torch.zeros(in_channels))
        else:
            self.bias = None
        
        nn.init.kaiming_uniform_(self.weight, a=0)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias = self.bias if self.bias is not None else None
        return ext.depthwise_conv2d_cuda(
            x, self.weight, bias,
            self.stride, self.stride,
            self.padding, self.padding
        )

    def load_state_dict(self, state_dict, strict=True):
        new_state_dict = {}
        for k, v in state_dict.items():
            if k.startswith('conv2d.'):
                new_key = k[len('conv2d.'):]
                new_state_dict[new_key] = v
            else:
                new_state_dict[k] = v
        super().load_state_dict(new_state_dict, strict=strict)
