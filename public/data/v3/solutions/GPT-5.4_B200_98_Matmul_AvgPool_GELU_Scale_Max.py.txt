import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def fused_post_kernel(
    x_ptr,
    y_ptr,
    B,
    N,
    stride_xb,
    stride_xn,
    stride_y,
    scale,
    BLOCK_OUT: tl.constexpr,
    KERNEL: tl.constexpr,
):
    pid_b = tl.program_id(0)
    offs = tl.arange(0, BLOCK_OUT)
    n_blocks = (N + BLOCK_OUT * KERNEL - 1) // (BLOCK_OUT * KERNEL)

    max_val = tl.full((BLOCK_OUT,), -float('inf'), tl.float32)

    base_x = x_ptr + pid_b * stride_xb
    for blk in range(0, n_blocks):
        base = blk * BLOCK_OUT * KERNEL + offs * KERNEL
        acc = tl.zeros((BLOCK_OUT,), tl.float32)
        valid_group = base < N
        for k in range(0, KERNEL):
            idx = base + k
            mask = idx < N
            v = tl.load(base_x + idx * stride_xn, mask=mask, other=0.0)
            acc += v.to(tl.float32)
        avg = acc / KERNEL
        gelu = 0.5 * avg * (1.0 + tl.erf(avg * 0.7071067811865475))
        gelu = gelu * scale
        gelu = tl.where(valid_group, gelu, -float('inf'))
        max_val = tl.maximum(max_val, gelu)

    out = tl.max(max_val, axis=0)
    tl.store(y_ptr + pid_b * stride_y, out)


class FusedMatmulAvgPoolGELUScaleMaxFn(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, scale_factor, pool_kernel_size):
        # Use addmm for fast GEMM, then fuse remaining ops in Triton.
        mm = torch.addmm(bias, x, weight.t())
        B, N = mm.shape
        out_len = (N - pool_kernel_size) // pool_kernel_size + 1
        y = torch.empty((B,), device=mm.device, dtype=torch.float32)
        BLOCK_OUT = 256
        fused_post_kernel[(B,)](
            mm, y,
            B, out_len * pool_kernel_size,
            mm.stride(0), mm.stride(1), y.stride(0),
            float(scale_factor),
            BLOCK_OUT=BLOCK_OUT,
            KERNEL=pool_kernel_size,
        )
        return y.to(mm.dtype)


class Model(nn.Module):
    """
    A model implementing the pattern "Matmul_AvgPool_GELU_Scale_Max".
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scale_factor = scale_factor
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        return FusedMatmulAvgPoolGELUScaleMaxFn.apply(
            x, self.matmul.weight, self.matmul.bias, self.scale_factor, self.pool_kernel_size
        )


batch_size = 128
in_features = 4096
out_features = 4096
pool_kernel_size = 4
scale_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, pool_kernel_size, scale_factor]
