import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized GEMV kernel for RTX 3090 (Ampere)
// A: (M, K), B: (K, 1), C: (M, 1)
// Each block computes one row of output

template<typename T, int BLOCK_SIZE>
__global__ void gemv_kernel(const T* __restrict__ A, const T* __restrict__ B, T* __restrict__ C, int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    extern __shared__ float shared_sum[];
    int tid = threadIdx.x;
    int warp_id = tid / warpSize;
    int lane_id = tid % warpSize;
    int num_warps = (BLOCK_SIZE + warpSize - 1) / warpSize;
    
    float sum = 0.0f;
    
    // Each thread accumulates partial sum
    for (int k = tid; k < K; k += BLOCK_SIZE) {
        float a_val = static_cast<float>(A[row * K + k]);
        float b_val = static_cast<float>(B[k]);
        sum += a_val * b_val;
    }
    
    // Warp-level reduction first
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Lane 0 of each warp writes to shared memory
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    __syncthreads();
    
    // First warp does final reduction
    if (warp_id == 0) {
        sum = (lane_id < num_warps) ? shared_sum[lane_id] : 0.0f;
        for (int offset = warpSize / 2; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (lane_id == 0) {
            C[row] = static_cast<T>(sum);
        }
    }
}

// Specialized kernel using vectorized loads for FP16
__global__ void gemv_fp16_kernel(const __half* __restrict__ A, const __half* __restrict__ B, __half* __restrict__ C, int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    extern __shared__ float shared_sum[];
    int tid = threadIdx.x;
    int warp_id = tid / warpSize;
    int lane_id = tid % warpSize;
    int num_warps = (256 + warpSize - 1) / warpSize;
    
    float sum = 0.0f;
    const int BLOCK_SIZE = 256;
    
    // Process 2 elements per thread per iteration (vectorized)
    for (int k = tid * 2; k < K; k += BLOCK_SIZE * 2) {
        if (k + 1 < K) {
            half2 a_vec = reinterpret_cast<const half2*>(&A[row * K + k])[0];
            half2 b_vec = reinterpret_cast<const half2*>(&B[k])[0];
            float2 a_f = __half22float2(a_vec);
            float2 b_f = __half22float2(b_vec);
            sum += a_f.x * b_f.x + a_f.y * b_f.y;
        } else if (k < K) {
            sum += __half2float(A[row * K + k]) * __half2float(B[k]);
        }
    }
    
    // Warp-level reduction
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Lane 0 of each warp writes to shared memory
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    __syncthreads();
    
    // First warp does final reduction
    if (warp_id == 0) {
        sum = (lane_id < num_warps) ? shared_sum[lane_id] : 0.0f;
        for (int offset = warpSize / 2; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (lane_id == 0) {
            C[row] = __float2half(sum);
        }
    }
}

torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    if (A.dtype() == torch::kHalf) {
        // FP16 kernel with vectorized loads
        gemv_fp16_kernel<<<M, 256, 256 * sizeof(float)>>>(
            reinterpret_cast<const __half*>(A.data_ptr<at::Half>()),
            reinterpret_cast<const __half*>(B.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(C.data_ptr<at::Half>()),
            M, K
        );
    } else if (A.dtype() == torch::kFloat) {
        // FP32 kernel
        gemv_kernel<float, 256><<<M, 256, 256 * sizeof(float)>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    } else {
        // Fallback for other types - use generic kernel
        auto A_f32 = A.to(torch::kFloat32);
        auto B_f32 = B.to(torch::kFloat32);
        auto C_f32 = torch::empty({M, 1}, torch::kFloat32);
        gemv_kernel<float, 256><<<M, 256, 256 * sizeof(float)>>>(
            A_f32.data_ptr<float>(),
            B_f32.data_ptr<float>(),
            C_f32.data_ptr<float>(),
            M, K
        );
        C = C_f32.to(A.dtype());
    }
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources='torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemv_op'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
)

class Model(nn.Module):
    """
    Optimized matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.gemv_op(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
