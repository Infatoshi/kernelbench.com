import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fast approximate GELU using tanh formula (float32)
// GELU(x) ≈ 0.5 * x * (1 + tanh(√(2/π) * (x + 0.044715 * x³)))
__device__ __forceinline__ float gelu_fast(float x) {
    const float c = 0.7978845608028654f;  // √(2/π)
    const float d = 0.044715f;
    float x3 = x * x * x;
    float tanh_arg = c * (x + d * x3);
    float tanh_val = tanhf(tanh_arg);
    return 0.5f * x * (1.0f + tanh_val);
}

// GELU kernel with vectorized float4 load/store for float32
__global__ void gelu_kernel_float4(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * 4;
    if (idx + 3 < n) {
        float4 x4 = reinterpret_cast<const float4*>(x)[idx / 4];
        float4 y4;
        y4.x = gelu_fast(x4.x);
        y4.y = gelu_fast(x4.y);
        y4.z = gelu_fast(x4.z);
        y4.w = gelu_fast(x4.w);
        reinterpret_cast<float4*>(y)[idx / 4] = y4;
    } else {
        // Handle remaining elements
        for (int i = idx; i < n && i < idx + 4; i++) {
            y[i] = gelu_fast(x[i]);
        }
    }
}

// Half precision GELU using __half intrinsics
__device__ __forceinline__ __half gelu_fast_half(__half x) {
    const float c = 0.7978845608028654f;
    const float d = 0.044715f;
    
    float fx = __half2float(x);
    float x3 = fx * fx * fx;
    float tanh_arg = c * (fx + d * x3);
    float tanh_val = tanhf(tanh_arg);
    float result = 0.5f * fx * (1.0f + tanh_val);
    
    return __float2half(result);
}

// GELU kernel for half precision (fp16/bf16 via conversion)
__global__ void gelu_kernel_half(const __half* __restrict__ x, __half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = gelu_fast_half(x[idx]);
    }
}

// Vectorized half GELU using half2
__device__ __forceinline__ half2 gelu_fast_half2(half2 x) {
    const float c = 0.7978845608028654f;
    const float d = 0.044715f;
    
    float2 fx = __half22float2(x);
    float2 x3;
    x3.x = fx.x * fx.x * fx.x;
    x3.y = fx.y * fx.y * fx.y;
    
    float2 tanh_arg;
    tanh_arg.x = c * (fx.x + d * x3.x);
    tanh_arg.y = c * (fx.y + d * x3.y);
    
    float2 tanh_val;
    tanh_val.x = tanhf(tanh_arg.x);
    tanh_val.y = tanhf(tanh_arg.y);
    
    float2 result;
    result.x = 0.5f * fx.x * (1.0f + tanh_val.x);
    result.y = 0.5f * fx.y * (1.0f + tanh_val.y);
    
    return __float22half2_rn(result);
}

// Vectorized half2 GELU kernel
__global__ void gelu_kernel_half2(const __half* __restrict__ x, __half* __restrict__ y, int n) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * 2;
    if (idx + 1 < n) {
        half2 x2 = reinterpret_cast<const half2*>(x)[idx / 2];
        half2 y2 = gelu_fast_half2(x2);
        reinterpret_cast<half2*>(y)[idx / 2] = y2;
    } else if (idx < n) {
        // Handle single element
        y[idx] = gelu_fast_half(x[idx]);
    }
}

torch::Tensor gelu_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int block_size = 256;
    
    if (x.dtype() == torch::kFloat32) {
        // Use vectorized float4 kernel for float32
        int grid_size = (n + 3) / 4;
        gelu_kernel_float4<<<grid_size, block_size>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            n
        );
    } else if (x.dtype() == torch::kFloat16) {
        // Use vectorized half2 kernel for float16
        int grid_size = (n + 1) / 2;
        gelu_kernel_half2<<<grid_size, block_size>>>(
            reinterpret_cast<const __half*>(x.data_ptr<c10::Half>()),
            reinterpret_cast<__half*>(y.data_ptr<c10::Half>()),
            n
        );
    } else if (x.dtype() == torch::kBFloat16) {
        // Convert to float32, apply GELU, convert back
        auto x_float = x.to(torch::kFloat32);
        auto y_float = torch::empty_like(x_float);
        int grid_size = (n + 3) / 4;
        gelu_kernel_float4<<<grid_size, block_size>>>(
            x_float.data_ptr<float>(),
            y_float.data_ptr<float>(),
            n
        );
        y = y_float.to(torch::kBFloat16);
    }
    
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_cuda(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['gelu_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_cuda(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
