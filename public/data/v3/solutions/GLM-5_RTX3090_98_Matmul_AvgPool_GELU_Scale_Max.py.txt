import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// GELU constants for tanh approximation
static constexpr float GELU_COEF = 0.7978845608028654f;  // sqrt(2/pi)
static constexpr float GELU_COEF2 = 0.044715f;

// Fast GELU using tanh approximation (matching PyTorch's default)
__device__ __forceinline__ float gelu_tanh(float x) {
    float x3 = x * x * x;
    float inner = GELU_COEF * (x + GELU_COEF2 * x3);
    return 0.5f * x * (1.0f + tanhf(inner));
}

// Fused kernel: AvgPool -> GELU -> Scale -> Max
// Each block handles one batch element
__global__ void fused_pool_gelu_scale_max_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch_size,
    int out_features,
    int pool_kernel_size,
    float scale_factor
) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    int blockDim = gridDim.y * blockDim.x;  // Total threads for this batch
    
    int pooled_features = out_features / pool_kernel_size;
    
    // Shared memory for reduction
    extern __shared__ float s_max[];
    
    float local_max = -INFINITY;
    float inv_pool = 1.0f / pool_kernel_size;
    
    // Each thread handles multiple pooled features
    for (int pf = tid; pf < pooled_features; pf += blockDim.x) {
        int base_idx = b * out_features + pf * pool_kernel_size;
        
        // AvgPool: sum pool_kernel_size elements
        float sum = 0.0f;
        for (int k = 0; k < pool_kernel_size; k++) {
            sum += input[base_idx + k];
        }
        float pooled = sum * inv_pool;
        
        // GELU
        float g = gelu_tanh(pooled);
        
        // Scale
        float scaled = g * scale_factor;
        
        // Update local max
        local_max = fmaxf(local_max, scaled);
    }
    
    // Store in shared memory for reduction
    s_max[tid] = local_max;
    __syncthreads();
    
    // Block reduction for max
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            s_max[tid] = fmaxf(s_max[tid], s_max[tid + s]);
        }
        __syncthreads();
    }
    
    // Thread 0 writes result
    if (tid == 0) {
        output[b] = s_max[0];
    }
}

// Optimized matmul kernel using shared memory tiling
#define TILE_M 32
#define TILE_N 32
#define TILE_K 32

__global__ void matmul_kernel(
    const float* __restrict__ A,
    const float* __restrict__ W,
    const float* __restrict__ bias,
    float* __restrict__ C,
    int M, int N, int K
) {
    // Block and thread indices
    int bm = blockIdx.y * TILE_M;
    int bn = blockIdx.x * TILE_N;
    
    int tm = threadIdx.y;
    int tn = threadIdx.x;
    
    int row = bm + tm;
    int col = bn + tn;
    
    // Shared memory tiles
    __shared__ float As[TILE_M][TILE_K];
    __shared__ float Ws[TILE_N][TILE_K];  // Note: transposed for coalesced access
    
    float acc = 0.0f;
    
    // Loop over K dimension in tiles
    for (int bk = 0; bk < (K + TILE_K - 1) / TILE_K; bk++) {
        int k_start = bk * TILE_K;
        
        // Load A tile (M x K)
        int a_col = k_start + tn;
        if (row < M && a_col < K) {
            As[tm][tn] = A[row * K + a_col];
        } else {
            As[tm][tn] = 0.0f;
        }
        
        // Load W tile (N x K) - transposed, so we load W^T
        // W is stored as (N, K), we want W^T which is (K, N)
        // So W[col, k] = W[col * K + k]
        int w_col = k_start + tm;
        if (col < N && w_col < K) {
            Ws[tn][tm] = W[col * K + w_col];  // Transpose in shared memory
        } else {
            Ws[tn][tm] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial product
        #pragma unroll
        for (int k = 0; k < TILE_K; k++) {
            acc += As[tm][k] * Ws[tn][k];
        }
        
        __syncthreads();
    }
    
    // Add bias and store result
    if (row < M && col < N) {
        if (bias != nullptr) {
            acc += bias[col];
        }
        C[row * N + col] = acc;
    }
}

// Entry function for fused operations
torch::Tensor fused_forward_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int64_t pool_kernel_size,
    double scale_factor
) {
    // Get dimensions
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    // Convert to float32 if needed
    auto input_f32 = input.to(torch::kFloat32);
    auto weight_f32 = weight.to(torch::kFloat32);
    auto bias_f32 = bias.to(torch::kFloat32);
    
    // Allocate output for matmul
    auto matmul_out = torch::empty({batch_size, out_features}, 
                                   torch::dtype(torch::kFloat32).device(input.device()));
    
    // Launch matmul kernel
    dim3 block(TILE_N, TILE_M);
    dim3 grid((out_features + TILE_N - 1) / TILE_N, 
              (batch_size + TILE_M - 1) / TILE_M);
    
    matmul_kernel<<<grid, block>>>(
        input_f32.data_ptr<float>(),
        weight_f32.data_ptr<float>(),
        bias_f32.data_ptr<float>(),
        matmul_out.data_ptr<float>(),
        batch_size, out_features, in_features
    );
    
    // Allocate final output
    auto output = torch::empty({batch_size}, 
                               torch::dtype(torch::kFloat32).device(input.device()));
    
    // Launch fused kernel
    int threads_per_block = 256;
    int shared_mem = threads_per_block * sizeof(float);
    
    fused_pool_gelu_scale_max_kernel<<<batch_size, threads_per_block, shared_mem>>>(
        matmul_out.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size,
        out_features,
        static_cast<int>(pool_kernel_size),
        static_cast<float>(scale_factor)
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_forward_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int64_t pool_kernel_size,
    double scale_factor
);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '-use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor
        
        # Parameters matching nn.Linear
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize parameters
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        return ext.fused_forward_cuda(x, self.weight, self.bias, 
                                       self.pool_kernel_size, self.scale_factor)

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
