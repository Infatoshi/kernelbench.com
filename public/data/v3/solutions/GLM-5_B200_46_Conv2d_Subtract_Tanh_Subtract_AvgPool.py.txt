import torch
import torch.nn as nn
import triton
import triton.language as tl

# Fused kernel: subtract1 -> tanh -> subtract2 -> avgpool2d
@triton.jit
def fused_subtract_tanh_subtract_avgpool_kernel(
    # Pointers to matrices
    input_ptr, output_ptr,
    # Input dimensions
    N, C, H, W,
    # Pool dimensions  
    pool_size,
    # Constants
    subtract1_val, subtract2_val,
    # Strides
    in_n_stride, in_c_stride, in_h_stride, in_w_stride,
    out_n_stride, out_c_stride, out_h_stride, out_w_stride,
    # Dimensions for indexing
    out_h, out_w,
):
    """Fused kernel for subtract -> tanh -> subtract -> avgpool"""
    
    # Get linear program ID and compute 4D indices
    pid = tl.program_id(0)
    
    # Total output elements per batch/channel
    out_hw = out_h * out_w
    out_nc = N * C
    
    # Decode indices: pid = ((n * C + c) * out_h + oh) * out_w + ow
    total_elements = out_nc * out_hw
    
    if pid >= total_elements:
        return
    
    # Compute 4D indices from linear pid
    pid_w = pid % out_w
    pid = pid // out_w
    pid_h = pid % out_h
    pid = pid // out_h
    pid_c = pid % C
    pid_n = pid // C
    
    # Compute pool region
    h_start = pid_h * pool_size
    w_start = pid_w * pool_size
    
    # Accumulate for average pooling
    acc = 0.0
    
    # Loop over pool window
    for ph in range(pool_size):
        for pw in range(pool_size):
            h_idx = h_start + ph
            w_idx = w_start + pw
            
            # Load input value
            in_offset = pid_n * in_n_stride + pid_c * in_c_stride + h_idx * in_h_stride + w_idx * in_w_stride
            val = tl.load(input_ptr + in_offset).to(tl.float32)
            
            # Apply fused operations: subtract1 -> tanh -> subtract2
            val = val - subtract1_val
            # tanh(x) = (exp(2x) - 1) / (exp(2x) + 1)
            exp_2x = tl.exp(2.0 * val)
            val = (exp_2x - 1.0) / (exp_2x + 1.0)
            val = val - subtract2_val
            
            # Accumulate for average pooling
            acc = acc + val
    
    # Compute average
    pool_elements = pool_size * pool_size
    result = acc / pool_elements
    
    # Store output
    out_offset = pid_n * out_n_stride + pid_c * out_c_stride + pid_h * out_h_stride + pid_w * out_w_stride
    tl.store(output_ptr + out_offset, result)


def fused_subtract_tanh_subtract_avgpool(input_tensor, subtract1_val, subtract2_val, pool_size):
    """Launch the fused kernel"""
    N, C, H, W = input_tensor.shape
    
    # Output dimensions after pooling
    out_h = H // pool_size
    out_w = W // pool_size
    
    # Create output tensor
    output = torch.empty(N, C, out_h, out_w, device=input_tensor.device, dtype=input_tensor.dtype)
    
    # Get strides
    in_n_stride, in_c_stride, in_h_stride, in_w_stride = input_tensor.stride()
    out_n_stride, out_c_stride, out_h_stride, out_w_stride = output.stride()
    
    # Total number of output elements
    total_elements = N * C * out_h * out_w
    
    # Launch kernel with 1D grid
    grid = (total_elements,)
    
    # Convert subtract values to same dtype
    sub1 = float(subtract1_val)
    sub2 = float(subtract2_val)
    
    fused_subtract_tanh_subtract_avgpool_kernel[grid](
        input_tensor, output,
        N, C, H, W,
        pool_size,
        sub1, sub2,
        in_n_stride, in_c_stride, in_h_stride, in_w_stride,
        out_n_stride, out_c_stride, out_h_stride, out_w_stride,
        out_h, out_w,
    )
    
    return output


class Model(nn.Module):
    """
    Model that performs a convolution, subtraction, tanh activation, subtraction and average pooling.
    Optimized with fused post-conv operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        # Use optimized convolution (cuDNN with Tensor Cores)
        x = self.conv(x)
        
        # Fused kernel: subtract -> tanh -> subtract -> avgpool
        x = fused_subtract_tanh_subtract_avgpool(
            x, 
            self.subtract1_value, 
            self.subtract2_value, 
            self.kernel_size_pool
        )
        
        return x


# Model parameters
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
