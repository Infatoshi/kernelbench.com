import torch
import torch.nn as nn
import torch.nn.functional as F
import math

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ['fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


class Model(nn.Module):
    """
    Optimized causal attention using efficient memory-access patterns.
    """

    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        # key, query, value projections for all heads, but in a batch
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        # output projection
        self.c_proj = nn.Linear(n_embd, n_embd)
        # regularization
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.n_head = n_head
        self.n_embd = n_embd
        self.hs = n_embd // n_head

    def forward(self, x):
        B, T, C = x.size()
        scale = 1.0 / math.sqrt(self.hs)
        
        # calculate query, key, values for all heads in batch and move head forward to be the batch dim
        q, k, v = self.c_attn(x).split(self.n_embd, dim=2)
        k = k.view(B, T, self.n_head, self.hs).transpose(1, 2)  # (B, nh, T, hs)
        q = q.view(B, T, self.n_head, self.hs).transpose(1, 2)  # (B, nh, T, hs)
        v = v.view(B, T, self.n_head, self.hs).transpose(1, 2)  # (B, nh, T, hs)
        
        # Optimized Flash Attention-style implementation
        y = self._flash_attention_cuda(q, k, v, scale, B, T)
        
        y = y.transpose(1, 2).contiguous().view(B, T, C)  # re-assemble all head outputs side by side
        
        # output projection
        y = self.resid_dropout(self.c_proj(y))
        return y
    
    def _flash_attention_cuda(self, q, k, v, scale, B, T):
        """
        Memory-efficient Flash Attention-style implementation using online softmax.
        Processes in blocks to avoid materializing the full NxN attention matrix.
        """
        _, nh, _, hs = q.shape
        device = q.device
        dtype = q.dtype
        
        # Output accumulator - use float32 for numerical stability
        out = torch.zeros_like(q, dtype=torch.float32)
        q = q.to(torch.float32)
        k = k.to(torch.float32)
        v = v.to(torch.float32)
        
        # Block size for processing
        BLOCK = 64
        
        # Each block processes BLOCK queries at a time
        for start_i in range(0, T, BLOCK):
            end_i = min(start_i + BLOCK, T)
            block_size = end_i - start_i
            
            # Load Q block: (B, nh, block, hs)
            q_block = q[:, :, start_i:end_i, :]
            
            # Initialize for online softmax
            # m_i: running max, shape (B, nh, block)
            m_i = torch.full((B, nh, block_size), -float('inf'), device=device, dtype=torch.float32)
            # l_i: running sum of exp, shape (B, nh, block)
            l_i = torch.zeros((B, nh, block_size), device=device, dtype=torch.float32)
            # o_i: output accumulator, shape (B, nh, block, hs)
            o_i = torch.zeros((B, nh, block_size, hs), device=device, dtype=torch.float32)
            
            # Process K, V in blocks
            for start_j in range(0, end_i, BLOCK):
                end_j = min(start_j + BLOCK, end_i)
                kj_block_size = end_j - start_j
                
                # Load K, V blocks
                k_block = k[:, :, start_j:end_j, :]  # (B, nh, kj_block, hs)
                v_block = v[:, :, start_j:end_j, :]  # (B, nh, kj_block, hs)
                
                # Compute S = Q @ K^T: (B, nh, block_i, kj_block)
                s = torch.einsum('bnid,bnjd->bnij', q_block, k_block) * scale
                
                # Causal mask: for each query at position i, only attend to keys at positions j <= i
                # Create indices for this block
                q_indices = torch.arange(start_i, end_i, device=device).unsqueeze(1)  # (block, 1)
                k_indices = torch.arange(start_j, end_j, device=device).unsqueeze(0)    # (1, kj_block)
                causal_mask = q_indices >= k_indices  # (block, kj_block)
                
                # Apply causal mask
                s = s.masked_fill(~causal_mask.unsqueeze(0).unsqueeze(0), float('-inf'))
                
                # Online softmax computation
                # Compute block max along key dimension
                m_j = s.amax(dim=-1)  # (B, nh, block)
                
                # Update running max
                m_i_new = torch.maximum(m_i, m_j)  # (B, nh, block)
                
                # Compute correction factors for the softmax sum
                # When updating: l_i_new = l_i * exp(m_i - m_i_new) + sum_j(exp(s_ij - m_i_new))
                correction_i = (m_i - m_i_new).exp()  # (B, nh, block)
                correction_j = (m_j - m_i_new).exp()  # (B, nh, block)
                
                # Compute softmax probabilities
                p = (s - m_i_new.unsqueeze(-1)).exp()  # (B, nh, block, kj_block)
                
                # Update sum of exponentials
                l_i_new = l_i * correction_i + p.sum(dim=-1) * correction_j
                
                # Update output accumulator: o_i_new = o_i * exp(m_i - m_i_new) + p @ V * exp(m_j - m_i_new)
                # p @ V: (B, nh, block, hs)
                weighted_v = torch.einsum('bnij,bnjd->bnid', p, v_block)
                o_i_new = o_i * correction_i.unsqueeze(-1) + weighted_v * correction_j.unsqueeze(-1)
                
                # Update states
                m_i = m_i_new
                l_i = l_i_new
                o_i = o_i_new
            
            # Normalize
            out[:, :, start_i:end_i, :] = o_i / (l_i.unsqueeze(-1) + 1e-8)
        
        return out.to(dtype)
