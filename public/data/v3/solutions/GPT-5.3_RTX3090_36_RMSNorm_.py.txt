import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void rmsnorm_kernel(const scalar_t* __restrict__ x, scalar_t* __restrict__ y,
                               int B,int C,int H,int W,float eps){
    int bh = blockIdx.x;
    int b = bh / H;
    int h = bh % H;
    int w = threadIdx.x;
    if (w>=W) return;

    int HW = H*W;
    float sumsq = 0.f;

    for(int c=0;c<C;c++){
        int idx = ((b*C + c)*H + h)*W + w;
        float v = (float)x[idx];
        sumsq += v*v;
    }

    float rms = sqrtf(sumsq / C + eps);

    for(int c=0;c<C;c++){
        int idx = ((b*C + c)*H + h)*W + w;
        float v = (float)x[idx];
        y[idx] = (scalar_t)(v / rms);
    }
}

torch::Tensor rmsnorm(torch::Tensor x, float eps){
    auto y = torch::empty_like(x);

    int B = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    dim3 grid(B*H);
    dim3 block(W);

    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "rmsnorm_kernel", ([&] {
        rmsnorm_kernel<scalar_t><<<grid, block>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            B,C,H,W,eps);
    }));

    return y;
}
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources='torch::Tensor rmsnorm(torch::Tensor x, float eps);',
    cuda_sources=cuda_src,
    functions=['rmsnorm'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm(x, self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]

def get_init_inputs():
    return [features]
