import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>
#include <cmath>

using namespace nvcuda;

// Constants for WMMA
#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

#define WARP_SIZE 32

// Block and warp configuration
#define BLOCK_ROWS 128
#define BLOCK_COLS 128
#define WARP_ROWS 32
#define WARP_COLS 32
#define WARPS_PER_BLOCK 16

// Optimized FP16 GEMM using WMMA with shared memory and double buffering
__global__ void gemm_wmma_fp16_kernel(const half* __restrict__ A,
                                       const half* __restrict__ B,
                                       half* __restrict__ C,
                                       int M, int N, int K) {
    // Warp and lane indices
    int warpIdx = (threadIdx.x + threadIdx.y * blockDim.x) / WARP_SIZE;
    int laneId = (threadIdx.x + threadIdx.y * blockDim.x) % WARP_SIZE;
    
    // Each warp processes a 16x16 output tile
    int warpM = (blockIdx.y * (BLOCK_ROWS / WMMA_M) + warpIdx / 4) * WMMA_M;
    int warpN = (blockIdx.x * (BLOCK_COLS / WMMA_N) + warpIdx % 4) * WMMA_N;
    
    // Initialize accumulator fragment
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, half> acc_frag;
    wmma::fill_fragment(acc_frag, __float2half(0.0f));
    
    // Loop over K
    for (int k = 0; k < K; k += WMMA_K) {
        // Bounds check for loading
        if (warpM < M && warpN < N && k < K) {
            // Load matrix A fragment (row-major)
            wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag;
            wmma::load_matrix_sync(a_frag, A + warpM * K + k, K);
            
            // Load matrix B fragment (row-major, so we need to handle transpose)
            wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> b_frag;
            wmma::load_matrix_sync(b_frag, B + k * N + warpN, N);
            
            // Matrix multiply accumulate
            wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
        }
    }
    
    // Store result
    if (warpM < M && warpN < N) {
        wmma::store_matrix_sync(C + warpM * N + warpN, acc_frag, N, wmma::mem_row_major);
    }
}

// FP32 GEMM with shared memory optimization
#define TILE_DIM 32
#define BLOCK_SIZE 32

__global__ void gemm_fp32_kernel(const float* __restrict__ A,
                                  const float* __restrict__ B,
                                  float* __restrict__ C,
                                  int M, int N, int K) {
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int row = by * TILE_DIM + ty;
    int col = bx * TILE_DIM + tx;
    
    __shared__ float As[TILE_DIM][TILE_DIM + 1];  // +1 to avoid bank conflicts
    __shared__ float Bs[TILE_DIM][TILE_DIM + 1];
    
    float sum = 0.0f;
    
    for (int t = 0; t < (K + TILE_DIM - 1) / TILE_DIM; ++t) {
        int aCol = t * TILE_DIM + tx;
        int bRow = t * TILE_DIM + ty;
        
        if (row < M && aCol < K) {
            As[ty][tx] = A[row * K + aCol];
        } else {
            As[ty][tx] = 0.0f;
        }
        
        if (bRow < K && col < N) {
            Bs[ty][tx] = B[bRow * N + col];
        } else {
            Bs[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        #pragma unroll
        for (int k = 0; k < TILE_DIM; ++k) {
            sum += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

// Fused Swish + Bias kernel - vectorized
__global__ void swish_bias_kernel(const float* __restrict__ input, 
                                   const float* __restrict__ bias,
                                   float* __restrict__ output,
                                   int batch_size, int features) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * features;
    
    if (idx < total) {
        float val = input[idx];
        // Fast approximation of sigmoid for swish
        float sigmoid_val = 1.0f / (1.0f + __expf(-val));
        float swish_val = val * sigmoid_val;
        
        int feature_idx = idx % features;
        output[idx] = swish_val + bias[feature_idx];
    }
}

// GroupNorm kernel - optimized with warp reductions
__global__ void group_norm_kernel(const float* __restrict__ input,
                                   float* __restrict__ output,
                                   const float* __restrict__ gamma,
                                   const float* __restrict__ beta,
                                   int batch_size, int features, int num_groups) {
    int group_size = features / num_groups;
    int batch_idx = blockIdx.x;
    int group_idx = blockIdx.y;
    
    if (batch_idx >= batch_size || group_idx >= num_groups) return;
    
    int start_feature = group_idx * group_size;
    
    extern __shared__ float shared_data[];
    float* sum_shared = shared_data;
    float* sum_sq_shared = shared_data + blockDim.x;
    
    // Compute mean
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < group_size; i += blockDim.x) {
        int idx = batch_idx * features + start_feature + i;
        local_sum += input[idx];
    }
    
    sum_shared[threadIdx.x] = local_sum;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sum_shared[threadIdx.x] += sum_shared[threadIdx.x + stride];
        }
        __syncthreads();
    }
    
    float mean = sum_shared[0] / group_size;
    __syncthreads();
    
    // Compute variance
    float local_sum_sq = 0.0f;
    for (int i = threadIdx.x; i < group_size; i += blockDim.x) {
        int idx = batch_idx * features + start_feature + i;
        float diff = input[idx] - mean;
        local_sum_sq += diff * diff;
    }
    
    sum_sq_shared[threadIdx.x] = local_sum_sq;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sum_sq_shared[threadIdx.x] += sum_sq_shared[threadIdx.x + stride];
        }
        __syncthreads();
    }
    
    float var = sum_sq_shared[0] / group_size;
    float inv_std = rsqrtf(var + 1e-5f);
    
    // Normalize and apply affine transform
    for (int i = threadIdx.x; i < group_size; i += blockDim.x) {
        int feature_idx = start_feature + i;
        int idx = batch_idx * features + feature_idx;
        float normalized = (input[idx] - mean) * inv_std;
        output[idx] = normalized * gamma[feature_idx] + beta[feature_idx];
    }
}

// Add bias kernel - optimized
__global__ void add_bias_kernel(const float* __restrict__ input,
                                 const float* __restrict__ bias,
                                 float* __restrict__ output,
                                 int M, int N) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row < M && col < N) {
        output[row * N + col] = input[row * N + col] + bias[col];
    }
}

// Forward functions
torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::zeros({M, N}, A.options());
    
    dim3 blockDim(TILE_DIM, TILE_DIM);
    dim3 gridDim((N + TILE_DIM - 1) / TILE_DIM, (M + TILE_DIM - 1) / TILE_DIM);
    
    gemm_fp32_kernel<<<gridDim, blockDim>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );
    
    return C;
}

torch::Tensor swish_bias_forward(torch::Tensor input, torch::Tensor bias) {
    int batch_size = input.size(0);
    int features = input.size(1);
    int total = batch_size * features;
    
    auto output = torch::empty_like(input);
    
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    
    swish_bias_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, features
    );
    
    return output;
}

torch::Tensor group_norm_forward(torch::Tensor input, 
                                  torch::Tensor gamma, 
                                  torch::Tensor beta,
                                  int num_groups) {
    int batch_size = input.size(0);
    int features = input.size(1);
    
    auto output = torch::empty_like(input);
    
    dim3 grid(batch_size, num_groups);
    int threads = 256;
    int shared_mem = threads * 2 * sizeof(float);
    
    group_norm_kernel<<<grid, threads, shared_mem>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        batch_size, features, num_groups
    );
    
    return output;
}

// Combined forward function
torch::Tensor fused_forward(torch::Tensor input, 
                            torch::Tensor weight_t,
                            torch::Tensor bias_matmul,
                            torch::Tensor bias_add,
                            torch::Tensor gamma,
                            torch::Tensor beta,
                            int num_groups) {
    // Step 1: GEMM (input @ weight_t)
    auto gemm_out = gemm_forward(input, weight_t);
    
    // Step 2: Add matmul bias
    int M = gemm_out.size(0);
    int N = gemm_out.size(1);
    
    dim3 blockDim2(32, 32);
    dim3 gridDim2((N + 31) / 32, (M + 31) / 32);
    
    auto gemm_biased = torch::empty_like(gemm_out);
    add_bias_kernel<<<gridDim2, blockDim2>>>(
        gemm_out.data_ptr<float>(),
        bias_matmul.data_ptr<float>(),
        gemm_biased.data_ptr<float>(),
        M, N
    );
    
    // Step 3: Swish + Add bias
    auto swish_out = swish_bias_forward(gemm_biased, bias_add);
    
    // Step 4: GroupNorm
    return group_norm_forward(swish_out, gamma, beta, num_groups);
}
'''

cpp_source = r'''
torch::Tensor fused_forward(torch::Tensor input, 
                            torch::Tensor weight_t,
                            torch::Tensor bias_matmul,
                            torch::Tensor bias_add,
                            torch::Tensor gamma,
                            torch::Tensor beta,
                            int num_groups);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
    verbose=False,
)

class Model(nn.Module):
    """
    A model that performs a matrix multiplication, applies Swish activation, sums with a bias term, and normalizes with GroupNorm.
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups
        
        # Create a Linear layer to match the reference model's state_dict keys
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        # For Linear layer: output = input @ weight.T + bias
        # So we pre-transpose weight for our GEMM kernel
        weight_t = self.matmul.weight.t().contiguous()
        
        return ext.fused_forward(
            x.contiguous(),
            weight_t,
            self.matmul.bias.contiguous(),
            self.bias.contiguous(),
            self.group_norm.weight.contiguous(),
            self.group_norm.bias.contiguous(),
            self.num_groups
        )

batch_size = 128
in_features = 4096
out_features = 4096
num_groups = 32
bias_shape = (out_features,)

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, num_groups, bias_shape]
