import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized GEMV kernel for M=256, K=131072
# Memory bandwidth is the bottleneck (~80% peak DRAM BW)
# Strategy: multiple blocks per row to maximize bandwidth utilization
# Use float4 vectorized loads + warp shuffles + 2-pass reduction

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// GEMV with multiple blocks per row for better parallelism
// Phase 1: partial reductions stored in temp buffer
// Phase 2: final reduction
// But for K=131072, M=256, a single-pass approach with enough threads works well

// Kernel: assign multiple blocks per row, each block processes a chunk of K
// This maximizes memory bandwidth by having many concurrent memory requests

#define BLOCK_SIZE 256
#define BLOCKS_PER_ROW 4  // 4 blocks per row * 256 rows = 1024 blocks total

__global__ void gemv_partial(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ partial,  // [M, BLOCKS_PER_ROW]
    int M,
    int K,
    int blocks_per_row
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    int tid = threadIdx.x;
    
    // Each block processes K/blocks_per_row elements
    int chunk = K / blocks_per_row;
    int k_start = block_in_row * chunk;
    int k_end = (block_in_row == blocks_per_row - 1) ? K : k_start + chunk;
    
    float sum = 0.0f;
    
    // float4 vectorized loads
    // Align to float4 boundary
    int k4_start = k_start / 4;
    int k4_end = k_end / 4;
    
    const float4* A4 = reinterpret_cast<const float4*>(A + row * K);
    const float4* B4 = reinterpret_cast<const float4*>(B);
    
    for (int i = k4_start + tid; i < k4_end; i += BLOCK_SIZE) {
        float4 a = __ldg(&A4[i]);
        float4 b = __ldg(&B4[i]);
        sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
    }
    
    // Warp reduction
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    __shared__ float warp_sums[BLOCK_SIZE / 32];
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        int num_warps = BLOCK_SIZE / 32;
        float val = (lane_id < num_warps) ? warp_sums[lane_id] : 0.0f;
        for (int offset = 16; offset > 0; offset >>= 1) {
            val += __shfl_down_sync(0xffffffff, val, offset);
        }
        if (lane_id == 0) {
            partial[row * blocks_per_row + block_in_row] = val;
        }
    }
}

__global__ void reduce_partial(
    const float* __restrict__ partial,
    float* __restrict__ C,
    int M,
    int blocks_per_row
) {
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    if (row < M) {
        float sum = 0.0f;
        for (int b = 0; b < blocks_per_row; b++) {
            sum += partial[row * blocks_per_row + b];
        }
        C[row] = sum;
    }
}

// Single-pass GEMV: each row uses one block with 1024 threads
// For K=131072: 1024 threads, each loads K/1024/4 = 32 float4 = 128 floats
__global__ void gemv_single(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M,
    int K
) {
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int block_size = blockDim.x;  // 1024
    
    float sum = 0.0f;
    
    const float4* A4 = reinterpret_cast<const float4*>(A + row * K);
    const float4* B4 = reinterpret_cast<const float4*>(B);
    
    int k4 = K >> 2;  // K/4
    
    // Unrolled loop for better ILP
    int i = tid;
    for (; i + block_size * 3 < k4; i += block_size * 4) {
        float4 a0 = __ldg(&A4[i]);
        float4 b0 = __ldg(&B4[i]);
        float4 a1 = __ldg(&A4[i + block_size]);
        float4 b1 = __ldg(&B4[i + block_size]);
        float4 a2 = __ldg(&A4[i + block_size*2]);
        float4 b2 = __ldg(&B4[i + block_size*2]);
        float4 a3 = __ldg(&A4[i + block_size*3]);
        float4 b3 = __ldg(&B4[i + block_size*3]);
        
        sum += a0.x*b0.x + a0.y*b0.y + a0.z*b0.z + a0.w*b0.w;
        sum += a1.x*b1.x + a1.y*b1.y + a1.z*b1.z + a1.w*b1.w;
        sum += a2.x*b2.x + a2.y*b2.y + a2.z*b2.z + a2.w*b2.w;
        sum += a3.x*b3.x + a3.y*b3.y + a3.z*b3.z + a3.w*b3.w;
    }
    for (; i < k4; i += block_size) {
        float4 a = __ldg(&A4[i]);
        float4 b = __ldg(&B4[i]);
        sum += a.x*b.x + a.y*b.y + a.z*b.z + a.w*b.w;
    }
    
    // Warp reduction
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // 1024 threads = 32 warps
    __shared__ float warp_sums[32];
    
    int warp_id = tid >> 5;
    int lane_id = tid & 31;
    
    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        float val = (lane_id < 32) ? warp_sums[lane_id] : 0.0f;
        for (int offset = 16; offset > 0; offset >>= 1) {
            val += __shfl_down_sync(0xffffffff, val, offset);
        }
        if (lane_id == 0) {
            C[row] = val;
        }
    }
}

// Two-pass: many blocks per row for maximum BW
torch::Tensor gemv(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "Inputs must be CUDA tensors");
    TORCH_CHECK(A.dtype() == torch::kFloat32 && B.dtype() == torch::kFloat32, "Must be float32");
    
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    // Use 1024-thread single-pass kernel
    // For K=131072: each thread does 131072/1024/4 = 32 float4 loads
    dim3 block(1024);
    dim3 grid(M);
    
    gemv_single<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K
    );
    
    return C;
}

// Two-pass version using more blocks
torch::Tensor gemv_twopass(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "Inputs must be CUDA tensors");
    TORCH_CHECK(A.dtype() == torch::kFloat32 && B.dtype() == torch::kFloat32, "Must be float32");
    
    int M = A.size(0);
    int K = A.size(1);
    int bpr = BLOCKS_PER_ROW;
    
    auto partial = torch::empty({M, bpr}, A.options());
    auto C = torch::empty({M, 1}, A.options());
    
    gemv_partial<<<M * bpr, BLOCK_SIZE>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        partial.data_ptr<float>(),
        M, K, bpr
    );
    
    reduce_partial<<<(M + 31)/32, 32>>>(
        partial.data_ptr<float>(),
        C.data_ptr<float>(),
        M, bpr
    );
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemv(torch::Tensor A, torch::Tensor B);
torch::Tensor gemv_twopass(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv', 'gemv_twopass'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-Xptxas=-v,-O3'],
)


# Benchmark at load time to pick best kernel
def _pick_best_kernel():
    M = 256
    K = 131072
    A = torch.randn(M, K, device='cuda')
    B = torch.randn(K, 1, device='cuda')
    
    # warmup
    for _ in range(10):
        ext.gemv(A, B)
        ext.gemv_twopass(A, B)
    torch.cuda.synchronize()
    
    N = 200
    
    start = torch.cuda.Event(enable_timing=True)
    end = torch.cuda.Event(enable_timing=True)
    
    start.record()
    for _ in range(N): ext.gemv(A, B)
    end.record()
    torch.cuda.synchronize()
    t1 = start.elapsed_time(end) / N
    
    start.record()
    for _ in range(N): ext.gemv_twopass(A, B)
    end.record()
    torch.cuda.synchronize()
    t2 = start.elapsed_time(end) / N
    
    return 'gemv' if t1 <= t2 else 'gemv_twopass', t1, t2

_best_kernel, _t1, _t2 = _pick_best_kernel()


class Model(nn.Module):
    """
    Optimized matrix-vector multiplication using custom CUDA GEMV kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication C = A * B.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        if _best_kernel == 'gemv':
            return ext.gemv(A, B)
        else:
            return ext.gemv_twopass(A, B)
