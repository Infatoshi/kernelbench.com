import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rmsnorm_kernel(
    X_ptr, Y_ptr,
    num_features: tl.constexpr,
    spatial_size,
    batch_stride,  # F * D1 * D2
    feature_stride,  # D1 * D2
    eps,
    BLOCK_S: tl.constexpr,
):
    # pid_x: spatial block, pid_y: batch
    pid_s = tl.program_id(0)
    pid_b = tl.program_id(1)
    
    s_offsets = pid_s * BLOCK_S + tl.arange(0, BLOCK_S)
    s_mask = s_offsets < spatial_size
    
    base = pid_b * batch_stride + s_offsets  # base pointer for spatial locations in this batch
    
    # Accumulate sum of squares
    acc = tl.zeros([BLOCK_S], dtype=tl.float32)
    
    for f in range(num_features):
        x_ptrs = X_ptr + base + f * feature_stride
        x = tl.load(x_ptrs, mask=s_mask, other=0.0).to(tl.float32)
        acc += x * x
    
    # Compute inverse RMS
    inv_rms = 1.0 / tl.sqrt(acc / num_features + eps)
    
    # Normalize and store
    for f in range(num_features):
        x_ptrs = X_ptr + base + f * feature_stride
        x = tl.load(x_ptrs, mask=s_mask, other=0.0)
        y = x * inv_rms.to(x.dtype)
        y_ptrs = Y_ptr + base + f * feature_stride
        tl.store(y_ptrs, y, mask=s_mask)


def rmsnorm_triton(x: torch.Tensor, num_features: int, eps: float) -> torch.Tensor:
    batch_size = x.shape[0]
    dim1 = x.shape[2]
    dim2 = x.shape[3]
    
    y = torch.empty_like(x)
    
    spatial_size = dim1 * dim2
    feature_stride = spatial_size  # stride along feature dim
    batch_stride = num_features * spatial_size
    
    BLOCK_S = 1024
    
    grid = ((spatial_size + BLOCK_S - 1) // BLOCK_S, batch_size)
    
    rmsnorm_kernel[grid](
        x, y,
        num_features, spatial_size,
        batch_stride, feature_stride,
        eps,
        BLOCK_S=BLOCK_S,
    )
    
    return y


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return rmsnorm_triton(x, self.num_features, self.eps)


batch_size = 16

dim1 = 256
dim2 = 256
