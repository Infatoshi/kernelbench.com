import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define MAX_D 128

__global__ void gated_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    const float* __restrict__ alpha,
    const float* __restrict__ beta,
    float* __restrict__ out,
    int B, int H, int T, int Dqk, int Dv
) {
    int bh = blockIdx.x;
    int b = bh / H;
    int h = bh % H;

    extern __shared__ float S[]; // Dv * Dqk

    // init S=0
    for (int i = threadIdx.x; i < Dqk*Dv; i += blockDim.x)
        S[i] = 0.f;
    __syncthreads();

    for (int t = 0; t < T; ++t) {
        const float* qt = q + (((b*H+h)*T + t)*Dqk);
        const float* kt = k + (((b*H+h)*T + t)*Dqk);
        const float* vt = v + (((b*H+h)*T + t)*Dv);

        float a = alpha[(b*H+h)*T + t];
        float bta = beta[(b*H+h)*T + t];

        // compute u = S @ k_t  (Dv)
        __shared__ float u[MAX_D];
        for (int j = threadIdx.x; j < Dv; j += blockDim.x) {
            float sum = 0.f;
            for (int i = 0; i < Dqk; ++i) {
                sum += S[j*Dqk + i] * kt[i];
            }
            u[j] = sum;
        }
        __syncthreads();

        // update S
        for (int i = threadIdx.x; i < Dqk*Dv; i += blockDim.x) {
            int col = i / Dqk;
            int row = i % Dqk;
            float val = S[col*Dqk + row];
            float diff = (u[col] - vt[col]) * kt[row];
            S[col*Dqk + row] = a * val - bta * diff;
        }
        __syncthreads();

        // compute out = S @ q_t
        for (int j = threadIdx.x; j < Dv; j += blockDim.x) {
            float sum = 0.f;
            for (int i = 0; i < Dqk; ++i) {
                sum += S[j*Dqk + i] * (qt[i] * 0.08838834764831845f);
            }
            out[(((b*H+h)*T + t)*Dv) + j] = sum;
        }
        __syncthreads();
    }
}

torch::Tensor gated_forward(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v,
    torch::Tensor alpha,
    torch::Tensor beta
) {
    int B = q.size(0);
    int H = q.size(1);
    int T = q.size(2);
    int Dqk = q.size(3);
    int Dv = v.size(3);

    auto out = torch::zeros_like(v);

    int blocks = B * H;
    int threads = 256;
    size_t shmem = Dqk * Dv * sizeof(float);

    gated_kernel<<<blocks, threads, shmem>>>(
        q.data_ptr<float>(), k.data_ptr<float>(), v.data_ptr<float>(),
        alpha.data_ptr<float>(), beta.data_ptr<float>(),
        out.data_ptr<float>(), B, H, T, Dqk, Dv
    );

    return out;
}
'''

ext = load_inline(
    name='gated_delta_ext',
    cpp_sources='torch::Tensor gated_forward(torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor);',
    cuda_sources=cuda_src,
    functions=['gated_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, hidden_size, num_heads, head_dim_qk, head_dim_v, use_short_conv=True, conv_kernel_size=4):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v

        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        self.a_proj = nn.Linear(hidden_size, num_heads)
        self.b_proj = nn.Linear(hidden_size, num_heads)

        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        self.scale = head_dim_qk ** -0.5

    def forward(self, x):
        B, T, _ = x.shape

        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)

        q = q.view(B, T, self.num_heads, self.head_dim_qk).transpose(1,2).contiguous()
        k = k.view(B, T, self.num_heads, self.head_dim_qk).transpose(1,2).contiguous()
        v = v.view(B, T, self.num_heads, self.head_dim_v).transpose(1,2).contiguous()

        alpha = torch.sigmoid(self.a_proj(x)).transpose(1,2).contiguous()
        beta = torch.sigmoid(self.b_proj(x)).transpose(1,2).contiguous()

        o = ext.gated_forward(q.float(), k.float(), v.float(), alpha.float(), beta.float())

        o = o.transpose(1,2)
        o = self.o_norm(o)

        g = torch.sigmoid(self.g_proj(x))
        g = g.view(B, T, self.num_heads, self.head_dim_v)

        o = o * g
        o = o.reshape(B, T, self.num_heads * self.head_dim_v)
        o = self.o_proj(o)

        return o
