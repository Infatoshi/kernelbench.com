import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math_constants.h>

// Vectorized GELU kernel using float4 for coalesced memory access
__global__ void gelu_kernel_f32_vec4(const float4* __restrict__ x, float4* __restrict__ y, int n4) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n4) {
        float4 val = x[idx];
        const float c = 0.7978845608028654f; // sqrt(2/pi)
        const float k = 0.044715f;
        
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float v = ((float*)&val)[i];
            float inner = c * (v + k * v * v * v);
            float t = tanhf(inner);
            ((float*)&val)[i] = 0.5f * v * (1.0f + t);
        }
        y[idx] = val;
    }
}

// Scalar fallback for remaining elements
__global__ void gelu_kernel_f32(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = x[idx];
        const float c = 0.7978845608028654f;
        const float k = 0.044715f;
        float inner = c * (v + k * v * v * v);
        float t = tanhf(inner);
        y[idx] = 0.5f * v * (1.0f + t);
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    if (x.scalar_type() == torch::kFloat32) {
        int n4 = n / 4;
        int remainder = n % 4;
        
        if (n4 > 0) {
            int threads = 256;
            int blocks = (n4 + threads - 1) / threads;
            gelu_kernel_f32_vec4<<<blocks, threads>>>(
                reinterpret_cast<const float4*>(x.data_ptr<float>()),
                reinterpret_cast<float4*>(y.data_ptr<float>()),
                n4
            );
        }
        if (remainder > 0) {
            int offset = n4 * 4;
            int threads = 256;
            int blocks = (remainder + threads - 1) / threads;
            gelu_kernel_f32<<<blocks, threads>>>(
                x.data_ptr<float>() + offset,
                y.data_ptr<float>() + offset,
                remainder
            );
        }
    }
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor gelu_forward(torch::Tensor x);
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_forward(x)

batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []
