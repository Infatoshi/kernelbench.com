import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Fusion of InstanceNorm + Divide for all supported precisions
// Uses two-pass approach: 1) compute mean+var, 2) apply normalization+divide

template <typename scalar_t>
__global__ void instance_norm_stats_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ mean_out,
    scalar_t* __restrict__ var_out,
    int N, int C, int H, int W,
    int spatial_size) {
    
    int n = blockIdx.y;
    int c = blockIdx.x;
    int tid = threadIdx.x;
    
    int instance_idx = n * C + c;
    int elements_per_block = spatial_size / blockDim.x;
    int start = tid * elements_per_block;
    int end = start + elements_per_block;
    
    // First pass: compute sum
    scalar_t sum = static_cast<scalar_t>(0);
    for (int i = start; i < end; ++i) {
        int h = i / W;
        int w = i % W;
        int idx = ((n * C + c) * H + h) * W + w;
        sum += x[idx];
    }
    
    // Use shared memory for reduction
    __shared__ scalar_t shared[256];
    shared[tid] = sum;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared[tid] += shared[tid + s];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        mean_out[instance_idx] = shared[0] / static_cast<scalar_t>(spatial_size);
    }
    
    // Second pass: compute variance
    __syncthreads();
    if (tid < blockDim.x) {
        scalar_t mean_val = mean_out[instance_idx];
        scalar_t sum_sq = static_cast<scalar_t>(0);
        for (int i = start; i < end; ++i) {
            int h = i / W;
            int w = i % W;
            int idx = ((n * C + c) * H + h) * W + w;
            scalar_t diff = x[idx] - mean_val;
            sum_sq += diff * diff;
        }
        
        shared[tid] = sum_sq;
        __syncthreads();
        
        for (int s = blockDim.x / 2; s > 0; s >>= 1) {
            if (tid < s) {
                shared[tid] += shared[tid + s];
            }
            __syncthreads();
        }
        
        if (tid == 0) {
            var_out[instance_idx] = shared[0] / static_cast<scalar_t>(spatial_size);
        }
    }
}

template <typename scalar_t>
__global__ void instance_norm_divide_apply_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    const scalar_t* __restrict__ mean,
    const scalar_t* __restrict__ var,
    const scalar_t inv_divide,
    const float eps,
    int N, int C, int H, int W) {
    
    int n = blockIdx.z;
    int c = blockIdx.y;
    int h = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (h >= H) return;
    
    int instance_idx = n * C + c;
    scalar_t m = mean[instance_idx];
    scalar_t v = var[instance_idx];
    scalar_t inv_std = rsqrtf(v + static_cast<scalar_t>(eps));
    scalar_t factor = inv_std * inv_divide;
    
    for (int w = 0; w < W; ++w) {
        int idx = ((n * C + c) * H + h) * W + w;
        y[idx] = (x[idx] - m) * factor;
    }
}

template <typename scalar_t>
torch::Tensor instance_norm_divide_fwd_impl(
    torch::Tensor x,
    double divide_by,
    double eps) {
    
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int spatial_size = H * W;
    
    auto y = torch::empty_like(x);
    auto mean = torch::empty({N, C}, x.options());
    auto var = torch::empty({N, C}, x.options());
    
    const int threads = 256;
    dim3 stat_blocks(C, N);
    
    instance_norm_stats_kernel<scalar_t><<<stat_blocks, threads>>>(
        x.data_ptr<scalar_t>(),
        mean.data_ptr<scalar_t>(),
        var.data_ptr<scalar_t>(),
        N, C, H, W, spatial_size);
    
    dim3 apply_blocks((H + threads - 1) / threads, C, N);
    
    scalar_t inv_divide = static_cast<scalar_t>(1.0f / divide_by);
    
    instance_norm_divide_apply_kernel<scalar_t><<<apply_blocks, threads>>>(
        x.data_ptr<scalar_t>(),
        y.data_ptr<scalar_t>(),
        mean.data_ptr<scalar_t>(),
        var.data_ptr<scalar_t>(),
        inv_divide,
        static_cast<float>(eps),
        N, C, H, W);
    
    return y;
}

torch::Tensor instance_norm_divide_fwd(
    torch::Tensor x,
    double divide_by,
    double eps) {
    
    return AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "instance_norm_divide_fwd", ([&] {
        return instance_norm_divide_fwd_impl<scalar_t>(x, divide_by, eps);
    }));
}
'''

ext = load_inline(
    name='instance_norm_divide_ext',
    cpp_sources='torch::Tensor instance_norm_divide_fwd(torch::Tensor x, double divide_by, double eps);',
    cuda_sources=cuda_source,
    functions=['instance_norm_divide_fwd'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Optimized model that uses cuDNN Conv2d + fused CUDA InstanceNorm + Divide
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=kernel_size//2)
        self.divide_by = divide_by

    def forward(self, x):
        # Conv2d uses cuDNN internally (highly optimized)
        x = self.conv(x)
        # Fused InstanceNorm + Divide using custom CUDA kernel
        x = ext.instance_norm_divide_fwd(x, self.divide_by, 1e-5)
        return x


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
