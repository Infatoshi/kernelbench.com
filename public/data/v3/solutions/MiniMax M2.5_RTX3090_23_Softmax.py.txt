import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

template <typename scalar_t>
__device__ __forceinline__ scalar_t warp_reduce_max(scalar_t val) {
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        scalar_t other = __shfl_down_sync(0xffffffff, val, offset);
        val = (val > other) ? val : other;
    }
    return val;
}

template <typename scalar_t>
__device__ __forceinline__ scalar_t warp_reduce_sum(scalar_t val) {
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

template <typename scalar_t>
__global__ void softmax_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    const int batch_size,
    const int dim) {
    // Each block handles one row
    const int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    const int num_warps = blockDim.x / 32;
    const int elements_per_warp = (dim + num_warps - 1) / num_warps;
    const int warp_start = warp_id * elements_per_warp;
    const int warp_end = min(warp_start + elements_per_warp, dim);
    
    // Phase 1: Find max in each warp
    scalar_t warp_max = -FLT_MAX;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        warp_max = (x[row_idx * dim + i] > warp_max) ? x[row_idx * dim + i] : warp_max;
    }
    warp_max = warp_reduce_max(warp_max);
    
    // Store warp max to shared memory
    extern __shared__ scalar_t shared[];
    if (lane_id == 0 && warp_id < num_warps) {
        shared[warp_id] = warp_max;
    }
    __syncthreads();
    
    // Get block max from shared memory
    scalar_t block_max = -FLT_MAX;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_max = (shared[i] > block_max) ? shared[i] : block_max;
        }
    }
    __syncthreads();
    
    // Phase 2: Compute exp and sum
    scalar_t warp_sum = 0;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        scalar_t val = __expf(x[row_idx * dim + i] - block_max);
        warp_sum += val;
        y[row_idx * dim + i] = val;
    }
    warp_sum = warp_reduce_sum(warp_sum);
    
    if (lane_id == 0 && warp_id < num_warps) {
        shared[warp_id] = warp_sum;
    }
    __syncthreads();
    
    // Get block sum from shared memory
    scalar_t block_sum = 0;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_sum += shared[i];
        }
    }
    __syncthreads();
    
    // Phase 3: Divide by sum
    scalar_t inv_sum = 1.0f / block_sum;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        y[row_idx * dim + i] = y[row_idx * dim + i] * inv_sum;
    }
}

// Specialized kernel for half precision using float arithmetic
__global__ void softmax_kernel_half(
    const __half* __restrict__ x,
    __half* __restrict__ y,
    const int batch_size,
    const int dim) {
    const int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    const int num_warps = blockDim.x / 32;
    const int elements_per_warp = (dim + num_warps - 1) / num_warps;
    const int warp_start = warp_id * elements_per_warp;
    const int warp_end = min(warp_start + elements_per_warp, dim);
    
    // Phase 1: Find max in each warp (use float for comparisons)
    float warp_max = -65504.0f;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __half2float(x[row_idx * dim + i]);
        warp_max = (val > warp_max) ? val : warp_max;
    }
    
    // Warp reduction for max
    for (int offset = 16; offset > 0; offset /= 2) {
        float other = __shfl_down_sync(0xffffffff, warp_max, offset);
        warp_max = (warp_max > other) ? warp_max : other;
    }
    
    extern __shared__ float shared_half[];
    if (lane_id == 0 && warp_id < num_warps) {
        shared_half[warp_id] = warp_max;
    }
    __syncthreads();
    
    float block_max = -65504.0f;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_max = (shared_half[i] > block_max) ? shared_half[i] : block_max;
        }
    }
    __syncthreads();
    
    // Phase 2: Compute exp and sum (in float)
    float warp_sum = 0.0f;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __half2float(x[row_idx * dim + i]);
        val = expf(val - block_max);
        warp_sum += val;
        y[row_idx * dim + i] = __float2half(val);
    }
    
    // Warp reduction for sum
    for (int offset = 16; offset > 0; offset /= 2) {
        warp_sum += __shfl_down_sync(0xffffffff, warp_sum, offset);
    }
    
    if (lane_id == 0 && warp_id < num_warps) {
        shared_half[warp_id] = warp_sum;
    }
    __syncthreads();
    
    float block_sum = 0.0f;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_sum += shared_half[i];
        }
    }
    __syncthreads();
    
    // Phase 3: Divide by sum
    float inv_sum = 1.0f / block_sum;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __half2float(y[row_idx * dim + i]);
        y[row_idx * dim + i] = __float2half(val * inv_sum);
    }
}

// Specialized kernel for bfloat16 using float arithmetic
__global__ void softmax_kernel_bfloat16(
    const __nv_bfloat16* __restrict__ x,
    __nv_bfloat16* __restrict__ y,
    const int batch_size,
    const int dim) {
    const int row_idx = blockIdx.x;
    if (row_idx >= batch_size) return;
    
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    const int num_warps = blockDim.x / 32;
    const int elements_per_warp = (dim + num_warps - 1) / num_warps;
    const int warp_start = warp_id * elements_per_warp;
    const int warp_end = min(warp_start + elements_per_warp, dim);
    
    // Phase 1: Find max in each warp (use float for comparisons)
    float warp_max = -3.38953189e38f;  // Very small bf16 value
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __bfloat162float(x[row_idx * dim + i]);
        warp_max = (val > warp_max) ? val : warp_max;
    }
    
    // Warp reduction for max
    for (int offset = 16; offset > 0; offset /= 2) {
        float other = __shfl_down_sync(0xffffffff, warp_max, offset);
        warp_max = (warp_max > other) ? warp_max : other;
    }
    
    extern __shared__ float shared_bf16[];
    if (lane_id == 0 && warp_id < num_warps) {
        shared_bf16[warp_id] = warp_max;
    }
    __syncthreads();
    
    float block_max = -3.38953189e38f;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_max = (shared_bf16[i] > block_max) ? shared_bf16[i] : block_max;
        }
    }
    __syncthreads();
    
    // Phase 2: Compute exp and sum (in float)
    float warp_sum = 0.0f;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __bfloat162float(x[row_idx * dim + i]);
        val = expf(val - block_max);
        warp_sum += val;
        y[row_idx * dim + i] = __float2bfloat16(val);
    }
    
    // Warp reduction for sum
    for (int offset = 16; offset > 0; offset /= 2) {
        warp_sum += __shfl_down_sync(0xffffffff, warp_sum, offset);
    }
    
    if (lane_id == 0 && warp_id < num_warps) {
        shared_bf16[warp_id] = warp_sum;
    }
    __syncthreads();
    
    float block_sum = 0.0f;
    if (tid == 0) {
        for (int i = 0; i < num_warps && warp_start + i * elements_per_warp < dim; i++) {
            block_sum += shared_bf16[i];
        }
    }
    __syncthreads();
    
    // Phase 3: Divide by sum
    float inv_sum = 1.0f / block_sum;
    for (int i = warp_start + lane_id; i < warp_end; i += 32) {
        float val = __bfloat162float(y[row_idx * dim + i]);
        y[row_idx * dim + i] = __float2bfloat16(val * inv_sum);
    }
}

torch::Tensor softmax_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    
    const int batch_size = x.size(0);
    const int dim = x.size(1);
    
    const int threads = 256;
    const int blocks = batch_size;
    const int shared_mem = (threads / 32) * sizeof(float);
    
    if (x.scalar_type() == at::ScalarType::Half) {
        softmax_kernel_half<<<blocks, threads, shared_mem>>>(
            reinterpret_cast<const __half*>(x.data_ptr<c10::Half>()),
            reinterpret_cast<__half*>(y.data_ptr<c10::Half>()),
            batch_size,
            dim
        );
    } else if (x.scalar_type() == at::ScalarType::BFloat16) {
        softmax_kernel_bfloat16<<<blocks, threads, shared_mem>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<c10::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(y.data_ptr<c10::BFloat16>()),
            batch_size,
            dim
        );
    } else {
        AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "softmax_cuda", ([&] {
            softmax_kernel<scalar_t><<<blocks, threads, shared_mem>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                batch_size,
                dim
            );
        }));
    }
    
    return y;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_cuda(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.softmax_cuda(x)

batch_size = 256
dim = 16384

def get_inputs():
    return [torch.randn(batch_size, dim, device='cuda')]

def get_init_inputs():
    return []
