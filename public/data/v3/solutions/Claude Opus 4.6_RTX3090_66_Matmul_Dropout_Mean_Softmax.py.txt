import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def mean_kernel(X, Out, M, K, stride_xm, stride_xk, BLOCK_M: tl.constexpr, BLOCK_K: tl.constexpr):
    """Compute mean over dim=0: Out[k] = sum(X[:, k]) / M for each k"""
    pid = tl.program_id(0)
    k_offset = pid * BLOCK_K + tl.arange(0, BLOCK_K)
    k_mask = k_offset < K

    acc = tl.zeros([BLOCK_K], dtype=tl.float32)
    inv_M = 1.0 / M
    for m in range(0, M, BLOCK_M):
        m_indices = m + tl.arange(0, BLOCK_M)
        m_mask = m_indices < M
        ptrs = X + m_indices[:, None] * stride_xm + k_offset[None, :] * stride_xk
        mask = m_mask[:, None] & k_mask[None, :]
        block = tl.load(ptrs, mask=mask, other=0.0).to(tl.float32)
        acc += tl.sum(block, axis=0)

    acc = acc * inv_M
    tl.store(Out + k_offset, acc, mask=k_mask)


@triton.jit
def vecmat_bias_kernel(
    X, W, bias, Out,
    K, N,
    stride_wn, stride_wk,
    BLOCK_K: tl.constexpr, BLOCK_N: tl.constexpr,
):
    """Compute Out[n] = dot(X, W[n, :]) + bias[n] for a block of n's"""
    pid = tl.program_id(0)
    n_offset = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offset < N

    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    for k in range(0, K, BLOCK_K):
        k_indices = k + tl.arange(0, BLOCK_K)
        k_mask = k_indices < K
        x_block = tl.load(X + k_indices, mask=k_mask, other=0.0).to(tl.float32)
        w_ptrs = W + n_offset[:, None] * stride_wn + k_indices[None, :] * stride_wk
        w_block = tl.load(w_ptrs, mask=n_mask[:, None] & k_mask[None, :], other=0.0).to(tl.float32)
        acc += tl.sum(w_block * x_block[None, :], axis=1)

    b = tl.load(bias + n_offset, mask=n_mask, other=0.0).to(tl.float32)
    acc += b
    tl.store(Out + n_offset, acc, mask=n_mask)


@triton.jit
def softmax_kernel(X, Out, N, BLOCK_N: tl.constexpr):
    offsets = tl.arange(0, BLOCK_N)
    mask = offsets < N
    x = tl.load(X + offsets, mask=mask, other=float('-inf')).to(tl.float32)
    x_max = tl.max(x, axis=0)
    x = x - x_max
    x_exp = tl.exp(x)
    x_sum = tl.sum(x_exp, axis=0)
    out = x_exp / x_sum
    tl.store(Out + offsets, out, mask=mask)


class FusedModel(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, dropout_p, training):
        M, K = x.shape
        N = weight.shape[0]

        if not training:
            # mean(X @ W^T + bias, dim=0) = mean(X, dim=0) @ W^T + bias
            
            # Step 1: mean of X over dim=0
            x_mean = torch.empty(K, device=x.device, dtype=torch.float32)
            BLOCK_K = 256
            grid_mean = ((K + BLOCK_K - 1) // BLOCK_K,)
            mean_kernel[grid_mean](
                x, x_mean, M, K,
                x.stride(0), x.stride(1),
                BLOCK_M=64, BLOCK_K=BLOCK_K,
            )

            # Step 2: vector-matrix multiply + bias
            result = torch.empty(N, device=x.device, dtype=torch.float32)
            BLOCK_N = 64
            BLOCK_K_VM = 256
            grid_vm = ((N + BLOCK_N - 1) // BLOCK_N,)
            vecmat_bias_kernel[grid_vm](
                x_mean, weight, bias, result,
                K, N,
                weight.stride(0), weight.stride(1),
                BLOCK_K=BLOCK_K_VM, BLOCK_N=BLOCK_N,
            )

            # Step 3: softmax
            BLOCK_N_SF = triton.next_power_of_2(N)
            out = torch.empty(N, device=x.device, dtype=torch.float32)
            softmax_kernel[(1,)](result, out, N, BLOCK_N=BLOCK_N_SF)

            return out.unsqueeze(0)
        else:
            y = x @ weight.t() + bias
            if dropout_p > 0:
                mask_t = torch.bernoulli(torch.full_like(y, 1 - dropout_p)).to(y.dtype)
                y = y * mask_t / (1 - dropout_p)
            y = y.mean(dim=0, keepdim=True)
            y = torch.softmax(y, dim=1)
            return y


class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout_p = dropout_p

    def forward(self, x):
        return FusedModel.apply(x, self.matmul.weight, self.matmul.bias, self.dropout_p, self.training)


batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
