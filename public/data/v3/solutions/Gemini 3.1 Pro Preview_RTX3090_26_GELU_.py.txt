import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math_constants.h>
#include <cmath>

__global__ void gelu_kernel_fp32(const float4* x, float4* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float4 in = x[idx];
        float4 out;
        out.x = in.x * 0.5f * (1.0f + erff(in.x * 0.7071067811865475f));
        out.y = in.y * 0.5f * (1.0f + erff(in.y * 0.7071067811865475f));
        out.z = in.z * 0.5f * (1.0f + erff(in.z * 0.7071067811865475f));
        out.w = in.w * 0.5f * (1.0f + erff(in.w * 0.7071067811865475f));
        y[idx] = out;
    }
}

__global__ void gelu_kernel_fp16(const float4* x, float4* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float4 in = x[idx];
        half2* in_h2 = reinterpret_cast<half2*>(&in);
        float4 out;
        half2* out_h2 = reinterpret_cast<half2*>(&out);
        
        #pragma unroll
        for (int i = 0; i < 4; ++i) {
            float2 f = __half22float2(in_h2[i]);
            f.x = f.x * 0.5f * (1.0f + erff(f.x * 0.7071067811865475f));
            f.y = f.y * 0.5f * (1.0f + erff(f.y * 0.7071067811865475f));
            out_h2[i] = __float22half2_rn(f);
        }
        y[idx] = out;
    }
}

__global__ void gelu_kernel_bf16(const float4* x, float4* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float4 in = x[idx];
        nv_bfloat162* in_bf2 = reinterpret_cast<nv_bfloat162*>(&in);
        float4 out;
        nv_bfloat162* out_bf2 = reinterpret_cast<nv_bfloat162*>(&out);
        
        #pragma unroll
        for (int i = 0; i < 4; ++i) {
            #if __CUDA_ARCH__ >= 800
            float2 f = __bfloat1622float2(in_bf2[i]);
            f.x = f.x * 0.5f * (1.0f + erff(f.x * 0.7071067811865475f));
            f.y = f.y * 0.5f * (1.0f + erff(f.y * 0.7071067811865475f));
            out_bf2[i] = __float22bfloat162_rn(f);
            #endif
        }
        y[idx] = out;
    }
}

template<typename T>
__global__ void gelu_kernel_generic(const T* x, T* y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float in = static_cast<float>(x[idx]);
        y[idx] = static_cast<T>(in * 0.5f * (1.0f + erff(in * 0.7071067811865475f)));
    }
}

torch::Tensor my_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int threads = 256;
    
    bool is_fp32 = x.scalar_type() == torch::kFloat32;
    bool is_fp16 = x.scalar_type() == torch::kFloat16;
    bool is_bf16 = x.scalar_type() == torch::kBFloat16;
    
    int vec_size = 4;
    if (is_fp16 || is_bf16) vec_size = 8;
    
    // Check alignment
    bool aligned = (n % vec_size == 0) && (reinterpret_cast<uintptr_t>(x.data_ptr()) % 16 == 0) && (reinterpret_cast<uintptr_t>(y.data_ptr()) % 16 == 0);
    
    if (aligned) {
        int n_vec = n / vec_size;
        int blocks = (n_vec + threads - 1) / threads;
        if (is_fp32) {
            gelu_kernel_fp32<<<blocks, threads>>>(reinterpret_cast<const float4*>(x.data_ptr()), reinterpret_cast<float4*>(y.data_ptr()), n_vec);
        } else if (is_fp16) {
            gelu_kernel_fp16<<<blocks, threads>>>(reinterpret_cast<const float4*>(x.data_ptr()), reinterpret_cast<float4*>(y.data_ptr()), n_vec);
        } else if (is_bf16) {
            gelu_kernel_bf16<<<blocks, threads>>>(reinterpret_cast<const float4*>(x.data_ptr()), reinterpret_cast<float4*>(y.data_ptr()), n_vec);
        }
    } else {
        int blocks = (n + threads - 1) / threads;
        if (is_fp32) {
            gelu_kernel_generic<float><<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
        } else if (is_fp16) {
            gelu_kernel_generic<at::Half><<<blocks, threads>>>(x.data_ptr<at::Half>(), y.data_ptr<at::Half>(), n);
        } else if (is_bf16) {
            gelu_kernel_generic<at::BFloat16><<<blocks, threads>>>(x.data_ptr<at::BFloat16>(), y.data_ptr<at::BFloat16>(), n);
        }
    }
    return y;
}
'''

ext = load_inline(
    name='my_ext',
    cpp_sources='torch::Tensor my_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=False,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3']
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.my_op(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
