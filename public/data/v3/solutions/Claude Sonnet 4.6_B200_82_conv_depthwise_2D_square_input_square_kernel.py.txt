# Optimized depthwise 2D convolution CUDA kernel
# Key optimizations:
# 1. Template kernel size for full unrolling
# 2. Shared memory tiling to reduce global memory loads
# 3. Float accumulation for accuracy
# 4. Support fp32, fp16, bf16

#include <torch/extension.h>
#include <cuda_runtime.h>

// Helper to convert various types to float
template<typename T> __device__ __forceinline__ float to_float(T x);
template<> __device__ __forceinline__ float to_float<float>(float x) { return x; }
template<> __device__ __forceinline__ float to_float<__half>(__half x) { return __half2float(x); }
template<> __device__ __forceinline__ float to_float<__nv_bfloat16>(__nv_bfloat16 x) { return __bfloat162float(x); }

// Helper to convert float to various types
template<typename T> __device__ __forceinline__ T from_float(float x);
template<> __device__ __forceinline__ float from_float<float>(float x) { return x; }
template<> __device__ __forceinline__ __half from_float<__half>(float x) { return __float2half(x); }
template<> __device__ __forceinline__ __nv_bfloat16 from_float<__nv_bfloat16>(float x) { return __float2bfloat16(x); }

// Helper for zero value
template<typename T> __device__ __forceinline__ T zero_val();
template<> __device__ __forceinline__ float zero_val<float>() { return 0.0f; }
template<> __device__ __forceinline__ __half zero_val<__half>() { return __float2half(0.0f); }
template<> __device__ __forceinline__ __nv_bfloat16 zero_val<__nv_bfloat16>() { return __float2bfloat16(0.0f); }

// Depthwise conv2d kernel with shared memory tiling
// TILE_W=32, TILE_H=8 per block
// Each block handles one (batch, channel) pair and TILE_H x TILE_W output elements
template<typename scalar_t, int KS>
__global__ void depthwise_conv2d_smem_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch, int channels,
    int in_h, int in_w,
    int out_h, int out_w,
    int stride, int padding
) {
    const int TILE_W = 32;
    const int TILE_H = 8;
    
    const int bc = blockIdx.z;
    const int n = bc / channels;
    const int c = bc % channels;
    
    // Output tile start
    const int out_x0 = blockIdx.x * TILE_W;
    const int out_y0 = blockIdx.y * TILE_H;
    
    // Thread position within tile
    const int tx = threadIdx.x;  // 0..TILE_W-1
    const int ty = threadIdx.y;  // 0..TILE_H-1
    
    // Shared memory dimensions for stride=1
    // For stride=1: smem needs (TILE_H + KS - 1) x (TILE_W + KS - 1) elements
    const int smem_w = TILE_W + KS - 1;
    const int smem_h = TILE_H + KS - 1;
    
    extern __shared__ char smem_buf[];
    scalar_t* smem = reinterpret_cast<scalar_t*>(smem_buf);
    
    // Input tile start (top-left corner in padded input space)
    const int in_x0 = out_x0 * stride - padding;
    const int in_y0 = out_y0 * stride - padding;
    
    // Load input tile into shared memory
    const int smem_size = smem_w * smem_h;
    const int tid = ty * TILE_W + tx;
    const int block_size = TILE_W * TILE_H;
    
    const scalar_t* input_nc = input + (n * channels + c) * in_h * in_w;
    
    for (int i = tid; i < smem_size; i += block_size) {
        int sy = i / smem_w;
        int sx = i % smem_w;
        int iy = in_y0 + sy;
        int ix = in_x0 + sx;
        
        if (iy >= 0 && iy < in_h && ix >= 0 && ix < in_w) {
            smem[i] = input_nc[iy * in_w + ix];
        } else {
            smem[i] = zero_val<scalar_t>();
        }
    }
    
    __syncthreads();
    
    const int out_x = out_x0 + tx;
    const int out_y = out_y0 + ty;
    
    if (out_x >= out_w || out_y >= out_h) return;
    
    // Load kernel weights for this channel (in registers)
    const scalar_t* w = weight + c * KS * KS;
    
    // Compute convolution
    float sum = 0.0f;
    
    const int smem_y = ty * stride;
    const int smem_x = tx * stride;
    
    #pragma unroll
    for (int ky = 0; ky < KS; ky++) {
        #pragma unroll
        for (int kx = 0; kx < KS; kx++) {
            float in_val = to_float(smem[(smem_y + ky) * smem_w + smem_x + kx]);
            float w_val = to_float(w[ky * KS + kx]);
            sum += in_val * w_val;
        }
    }
    
    if (bias != nullptr) {
        sum += to_float(bias[c]);
    }
    
    output[(n * channels + c) * out_h * out_w + out_y * out_w + out_x] = from_float<scalar_t>(sum);
}

// General strided kernel (no shared memory) for arbitrary stride/KS
template<typename scalar_t, int KS>
__global__ void depthwise_conv2d_direct_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch, int channels,
    int in_h, int in_w,
    int out_h, int out_w,
    int stride, int padding
) {
    const int bc = blockIdx.z;
    const int n = bc / channels;
    const int c = bc % channels;
    
    const int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    const int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (out_x >= out_w || out_y >= out_h) return;
    
    const scalar_t* input_nc = input + (n * channels + c) * in_h * in_w;
    const scalar_t* w = weight + c * KS * KS;
    
    const int in_y0 = out_y * stride - padding;
    const int in_x0 = out_x * stride - padding;
    
    float sum = 0.0f;
    
    #pragma unroll
    for (int ky = 0; ky < KS; ky++) {
        int iy = in_y0 + ky;
        #pragma unroll
        for (int kx = 0; kx < KS; kx++) {
            int ix = in_x0 + kx;
            if (iy >= 0 && iy < in_h && ix >= 0 && ix < in_w) {
                sum += to_float(input_nc[iy * in_w + ix]) * to_float(w[ky * KS + kx]);
            }
        }
    }
    
    if (bias != nullptr) {
        sum += to_float(bias[c]);
    }
    
    output[(n * channels + c) * out_h * out_w + out_y * out_w + out_x] = from_float<scalar_t>(sum);
}

// Generic kernel for runtime kernel size
template<typename scalar_t>
__global__ void depthwise_conv2d_generic_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int batch, int channels,
    int in_h, int in_w,
    int out_h, int out_w,
    int kernel_size, int stride, int padding
) {
    const int bc = blockIdx.z;
    const int n = bc / channels;
    const int c = bc % channels;
    
    const int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    const int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (out_x >= out_w || out_y >= out_h) return;
    
    const scalar_t* input_nc = input + (n * channels + c) * in_h * in_w;
    const scalar_t* w = weight + c * kernel_size * kernel_size;
    
    const int in_y0 = out_y * stride - padding;
    const int in_x0 = out_x * stride - padding;
    
    float sum = 0.0f;
    
    for (int ky = 0; ky < kernel_size; ky++) {
        int iy = in_y0 + ky;
        if (iy >= 0 && iy < in_h) {
            for (int kx = 0; kx < kernel_size; kx++) {
                int ix = in_x0 + kx;
                if (ix >= 0 && ix < in_w) {
                    sum += to_float(input_nc[iy * in_w + ix]) * to_float(w[ky * kernel_size + kx]);
                }
            }
        }
    }
    
    if (bias != nullptr) {
        sum += to_float(bias[c]);
    }
    
    output[(n * channels + c) * out_h * out_w + out_y * out_w + out_x] = from_float<scalar_t>(sum);
}

torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::optional<torch::Tensor> bias_opt,
    int stride,
    int padding
) {
    TORCH_CHECK(input.is_cuda(), "input must be CUDA tensor");
    TORCH_CHECK(weight.is_cuda(), "weight must be CUDA tensor");
    input = input.contiguous();
    weight = weight.contiguous();
    
    const int batch = input.size(0);
    const int channels = input.size(1);
    const int in_h = input.size(2);
    const int in_w = input.size(3);
    const int kernel_size = weight.size(1);  // weight shape: (C, KH, KW)
    
    const int out_h = (in_h + 2 * padding - kernel_size) / stride + 1;
    const int out_w = (in_w + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch, channels, out_h, out_w}, input.options());
    
    torch::Tensor bias;
    void* bias_ptr = nullptr;
    if (bias_opt.has_value() && bias_opt.value().defined()) {
        bias = bias_opt.value().contiguous();
        bias_ptr = bias.data_ptr();
    }
    
    auto stream = at::cuda::getCurrentCUDAStream();
    auto dtype = input.scalar_type();
    
    const int TILE_W = 32;
    const int TILE_H = 8;
    
    // For stride=1, use shared memory tiling kernel
    if (stride == 1) {
        dim3 block(TILE_W, TILE_H, 1);
        dim3 grid(
            (out_w + TILE_W - 1) / TILE_W,
            (out_h + TILE_H - 1) / TILE_H,
            batch * channels
        );
        
        int elem_size = (dtype == torch::kFloat32) ? 4 : 2;
        
        auto get_data_ptr = [](torch::Tensor& t) { return t.data_ptr(); };
        
        if (kernel_size == 3) {
            int smem_bytes = (TILE_W + 2) * (TILE_H + 2) * elem_size;
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_smem_kernel<float, 3><<<grid, block, smem_bytes, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_smem_kernel<__half, 3><<<grid, block, smem_bytes, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else {
                depthwise_conv2d_smem_kernel<__nv_bfloat16, 3><<<grid, block, smem_bytes, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            }
        } else if (kernel_size == 5) {
            int smem_bytes = (TILE_W + 4) * (TILE_H + 4) * elem_size;
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_smem_kernel<float, 5><<<grid, block, smem_bytes, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_smem_kernel<__half, 5><<<grid, block, smem_bytes, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else {
                depthwise_conv2d_smem_kernel<__nv_bfloat16, 5><<<grid, block, smem_bytes, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            }
        } else if (kernel_size == 7) {
            int smem_bytes = (TILE_W + 6) * (TILE_H + 6) * elem_size;
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_smem_kernel<float, 7><<<grid, block, smem_bytes, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_smem_kernel<__half, 7><<<grid, block, smem_bytes, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else {
                depthwise_conv2d_smem_kernel<__nv_bfloat16, 7><<<grid, block, smem_bytes, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            }
        } else {
            // Fallback generic for other kernel sizes
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_generic_kernel<float><<<grid, block, 0, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, kernel_size, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_generic_kernel<__half><<<grid, block, 0, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, kernel_size, stride, padding);
            } else {
                depthwise_conv2d_generic_kernel<__nv_bfloat16><<<grid, block, 0, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, kernel_size, stride, padding);
            }
        }
    } else {
        // Strided: use direct kernel with unrolled inner loops
        dim3 block(TILE_W, TILE_H, 1);
        dim3 grid(
            (out_w + TILE_W - 1) / TILE_W,
            (out_h + TILE_H - 1) / TILE_H,
            batch * channels
        );
        
        if (kernel_size == 3) {
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_direct_kernel<float, 3><<<grid, block, 0, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_direct_kernel<__half, 3><<<grid, block, 0, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else {
                depthwise_conv2d_direct_kernel<__nv_bfloat16, 3><<<grid, block, 0, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            }
        } else if (kernel_size == 5) {
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_direct_kernel<float, 5><<<grid, block, 0, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_direct_kernel<__half, 5><<<grid, block, 0, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else {
                depthwise_conv2d_direct_kernel<__nv_bfloat16, 5><<<grid, block, 0, stream>>>(
                    (__nv_bfloat16*)input.data_ptr(), (__nv_bfloat16*)weight.data_ptr(),
                    (__nv_bfloat16*)bias_ptr, (__nv_bfloat16*)output.data_ptr(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            }
        } else if (kernel_size == 7) {
            if (dtype == torch::kFloat32) {
                depthwise_conv2d_direct_kernel<float, 7><<<grid, block, 0, stream>>>(
                    input.data_ptr<float>(), weight.data_ptr<float>(),
                    (float*)bias_ptr, output.data_ptr<float>(),
                    batch, channels, in_h, in_w, out_h, out_w, stride, padding);
            } else if (dtype == torch::kFloat16) {
                depthwise_conv2d_direct_kernel<__half, 7><<<grid, block, 0, stream>>>(
                    (__half*)input.data_ptr(), (__half*)weight.data_ptr(),
                    (__half*)bias_ptr, (__half*)output.data_ptr(),

            } else if (dtype == torch::kFloat16) {

            }
        }
    }
    
    return output;
}

class Model(nn.Module):
    """
    Optimized depthwise 2D convolution with square input and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.use_bias = bias
        
        # Weight shape matches nn.Conv2d with groups=in_channels: (in_channels, 1, kH, kW)
        self.weight = nn.Parameter(
            torch.empty(in_channels, 1, kernel_size, kernel_size)
        )
        if bias:
            self.bias = nn.Parameter(torch.zeros(in_channels))
        else:
            self.register_parameter('bias', None)
        
        # Initialize weights
        nn.init.kaiming_uniform_(self.weight, a=0)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Weight shape: (in_channels, 1, kH, kW) -> reshape to (in_channels, kH, kW)
        w = self.weight.view(self.in_channels, self.kernel_size, self.kernel_size)
        return ext.depthwise_conv2d_forward(
            x, w, self.bias, self.stride, self.padding
        )
    
    def load_state_dict(self, state_dict, strict=True):
        # Handle mapping from reference model's conv2d.weight/bias to our weight/bias
        new_state_dict = {}
        for key, val in state_dict.items():
            if key == 'conv2d.weight':
                new_state_dict['weight'] = val
            elif key == 'conv2d.bias':
                new_state_dict['bias'] = val
            else:
                new_state_dict[key] = val
        return super().load_state_dict(new_state_dict, strict=strict)
