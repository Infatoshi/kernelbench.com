import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _post_kernel(
    z_ptr, out_ptr,
    B, O, P, scale,
    stride_zb, stride_zo,
    stride_outb,
    BLOCK: tl.constexpr,
):
    pid_b = tl.program_id(0)
    offs = tl.arange(0, BLOCK)

    max_val = tl.full((BLOCK,), -float('inf'), tl.float32)
    base0 = z_ptr + pid_b * stride_zb + offs * stride_zo
    base1 = z_ptr + pid_b * stride_zb + (offs + 1) * stride_zo
    base2 = z_ptr + pid_b * stride_zb + (offs + 2) * stride_zo
    base3 = z_ptr + pid_b * stride_zb + (offs + 3) * stride_zo

    m0 = offs < P
    x0 = tl.load(base0, mask=m0, other=0.0).to(tl.float32)
    x1 = tl.load(base1, mask=m0, other=0.0).to(tl.float32)
    x2 = tl.load(base2, mask=m0, other=0.0).to(tl.float32)
    x3 = tl.load(base3, mask=m0, other=0.0).to(tl.float32)
    y = 0.25 * (x0 + x1 + x2 + x3)
    cdf = 0.5 * (1.0 + tl.math.erf(y * 0.7071067811865475))
    gelu = y * cdf
    scaled = gelu * scale
    max_val = tl.where(m0, scaled, max_val)

    red = tl.max(max_val, axis=0)
    tl.store(out_ptr + pid_b * stride_outb, red)


class Model(nn.Module):
    """
    Optimized model: GEMM via cuBLAS-backed nn.Linear, fused tail via Triton.
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        assert pool_kernel_size == 4
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        z = self.matmul(x)
        B, O = z.shape
        P = O - self.pool_kernel_size + 1
        out = torch.empty((B,), device=z.device, dtype=torch.float32)
        BLOCK = triton.next_power_of_2(P)
        BLOCK = min(BLOCK, 2048)
        _post_kernel[(B,)](
            z, out,
            B, O, P, self.scale_factor,
            z.stride(0), z.stride(1),
            out.stride(0),
            BLOCK=BLOCK,
        )
        return out.to(z.dtype)


batch_size = 128
in_features = 4096
out_features = 4096
pool_kernel_size = 4
scale_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, pool_kernel_size, scale_factor]
