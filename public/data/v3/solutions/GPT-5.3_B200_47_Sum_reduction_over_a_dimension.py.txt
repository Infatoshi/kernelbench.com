import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

template <typename scalar_t>
__global__ void sum_reduce_kernel(const scalar_t* __restrict__ x,
                                  scalar_t* __restrict__ y,
                                  int outer, int reduce, int inner) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = outer * inner;
    if (idx >= total) return;

    int o = idx / inner;
    int j = idx % inner;

    int base = o * reduce * inner + j;

    float acc = 0.0f;
    for (int k = 0; k < reduce; ++k) {
        acc += (float)x[base + k * inner];
    }

    y[idx] = (scalar_t)acc;
}

torch::Tensor sum_reduce(torch::Tensor x, int dim) {
    x = x.contiguous();

    auto sizes = x.sizes();
    int ndim = x.dim();

    int64_t outer = 1, inner = 1, reduce = sizes[dim];

    for (int i = 0; i < dim; ++i) outer *= sizes[i];
    for (int i = dim + 1; i < ndim; ++i) inner *= sizes[i];

    std::vector<int64_t> out_sizes(sizes.begin(), sizes.end());
    out_sizes[dim] = 1;

    auto y = torch::empty(out_sizes, x.options());
    auto y_flat = y.view({outer * inner});

    int total = outer * inner;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "sum_reduce", ([&] {
        sum_reduce_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            y_flat.data_ptr<scalar_t>(),
            outer, reduce, inner);
    }));

    return y;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources='torch::Tensor sum_reduce(torch::Tensor x, int dim);',
    cuda_sources=cuda_src,
    functions=['sum_reduce'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce(x, self.dim)


def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2, device='cuda')
    return [x]


def get_init_inputs():
    return [1]
