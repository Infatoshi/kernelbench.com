import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

template <typename scalar_t>
__device__ __forceinline__ float to_float_device(scalar_t x) {
    return static_cast<float>(x);
}

template <>
__device__ __forceinline__ float to_float_device<at::Half>(at::Half x) {
    return __half2float(*reinterpret_cast<__half*>(&x));
}

template <>
__device__ __forceinline__ float to_float_device<at::BFloat16>(at::BFloat16 x) {
    return __bfloat162float(*reinterpret_cast<__nv_bfloat16*>(&x));
}

template <typename scalar_t>
__device__ __forceinline__ scalar_t from_float_device(float x) {
    return static_cast<scalar_t>(x);
}

template <>
__device__ __forceinline__ at::Half from_float_device<at::Half>(float x) {
    at::Half out;
    *reinterpret_cast<__half*>(&out) = __float2half_rn(x);
    return out;
}

template <>
__device__ __forceinline__ at::BFloat16 from_float_device<at::BFloat16>(float x) {
    at::BFloat16 out;
    *reinterpret_cast<__nv_bfloat16*>(&out) = __float2bfloat16(x);
    return out;
}

template <typename scalar_t>
__global__ void fused_groupnorm_scale_pool_clamp_kernel(
    const scalar_t* __restrict__ x,
    const float* __restrict__ scale,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    int groups, float eps,
    float clamp_min, float clamp_max)
{
    int w2 = blockIdx.x * blockDim.x + threadIdx.x;
    int h2 = blockIdx.y;
    int nc = blockIdx.z;
    if (w2 >= W / 2) return;

    int n = nc / C;
    int c = nc - n * C;
    int group_size = C / groups;
    int g = c / group_size;
    int HW = H * W;
    int base_nc = (n * C + c) * HW;
    int group_base_c = g * group_size;

    float sum = 0.0f;
    float sqsum = 0.0f;
    int count = group_size * HW;
    for (int gc = 0; gc < group_size; ++gc) {
        int cc = group_base_c + gc;
        int off = (n * C + cc) * HW;
        for (int idx = 0; idx < HW; ++idx) {
            float v = to_float_device<scalar_t>(x[off + idx]);
            sum += v;
            sqsum += v * v;
        }
    }
    float mean = sum / count;
    float var = fmaxf(sqsum / count - mean * mean, 0.0f);
    float inv_std = rsqrtf(var + eps);
    float s = scale[c];

    int in_h = h2 * 2;
    int in_w = w2 * 2;
    float m = -1.0e30f;
    #pragma unroll
    for (int kh = 0; kh < 2; ++kh) {
        #pragma unroll
        for (int kw = 0; kw < 2; ++kw) {
            int idx = (in_h + kh) * W + (in_w + kw);
            float v = to_float_device<scalar_t>(x[base_nc + idx]);
            float z = (v - mean) * inv_std;
            z *= s;
            m = fmaxf(m, z);
        }
    }
    m = fminf(fmaxf(m, clamp_min), clamp_max);
    int outH = H / 2;
    int outW = W / 2;
    y[((n * C + c) * outH + h2) * outW + w2] = from_float_device<scalar_t>(m);
}

torch::Tensor fused_post_op(torch::Tensor x, torch::Tensor scale, int groups, double eps, double clamp_min, double clamp_max) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(scale.is_cuda(), "scale must be CUDA");
    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    auto x_contig = x.contiguous();
    auto scale_f = scale.contiguous().view({x.size(1)}).to(torch::kFloat);
    int64_t N = x_contig.size(0), C = x_contig.size(1), H = x_contig.size(2), W = x_contig.size(3);
    TORCH_CHECK(H % 2 == 0 && W % 2 == 0, "H and W must be even");
    auto y = torch::empty({N, C, H / 2, W / 2}, x_contig.options());
    dim3 block(256);
    dim3 grid((W / 2 + block.x - 1) / block.x, H / 2, N * C);
    auto stream = at::cuda::getDefaultCUDAStream();
    AT_DISPATCH_FLOATING_TYPES_AND2(torch::kHalf, torch::kBFloat16, x_contig.scalar_type(), "fused_post_op", [&] {
        fused_groupnorm_scale_pool_clamp_kernel<scalar_t><<<grid, block, 0, stream>>>(
            x_contig.data_ptr<scalar_t>(),
            scale_f.data_ptr<float>(),
            y.data_ptr<scalar_t>(),
            (int)N, (int)C, (int)H, (int)W,
            groups, (float)eps, (float)clamp_min, (float)clamp_max
        );
    });
    return y;
}
'''

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor fused_post_op(torch::Tensor x, torch::Tensor scale, int groups, double eps, double clamp_min, double clamp_max);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("fused_post_op", &fused_post_op, "fused_post_op");
}
'''

ext = load_inline(
    name='b200_fused_post_gn_pool_clamp_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
    extra_cflags=['-O3'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool = nn.MaxPool2d(kernel_size=maxpool_kernel_size)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)
        return ext.fused_post_op(
            x,
            self.scale,
            self.group_norm.num_groups,
            self.group_norm.eps,
            self.clamp_min,
            self.clamp_max,
        )


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
