import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# FP4 Block-Scaled Matrix Multiplication - Optimized for B200 Blackwell
# This implementation uses:
# 1. Fused dequantization during the GEMM kernel
# 2. Tensor Core optimized HMMA instructions
# 3. Efficient shared memory and register usage

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

#define BLOCK_SIZE 32
#define TILE_M 128
#define TILE_N 128
#define TILE_K 64

// Unpack FP4 values from packed uint8 and dequantize to FP16
__device__ __forceinline__ void unpack_fp4_block(
    const uint8_t* packed_ptr,
    const half* scales_ptr,
    half* output,
    int N, int K, int block_idx, int row
) {
    int n_blocks = (K + BLOCK_SIZE - 1) / BLOCK_SIZE;
    
    for (int b = 0; b < n_blocks; b++) {
        int k_start = b * BLOCK_SIZE;
        int k_end = min(k_start + BLOCK_SIZE, K);
        half scale = scales_ptr[row * n_blocks + b];
        
        for (int k = k_start; k < k_end; k += 2) {
            uint8_t packed = packed_ptr[row * ((K + 1) / 2) + k / 2];
            int8_t lo = (packed & 0x0F) - 8;
            int8_t hi = ((packed >> 4) & 0x0F) - 8;
            
            // Convert to FP16 with scale
            output[k] = __short2half_rn(lo) * scale;
            if (k + 1 < K) {
                output[k + 1] = __short2half_rn(hi) * scale;
            }
        }
    }
}

// Optimized GEMM kernel using WMMA
__global__ void fp4_gemm_kernel(
    const half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const half* __restrict__ scales,
    half* __restrict__ output,
    int M, int N, int K, int n_blocks
) {
    // Each block handles a TILE_M x TILE_N tile of output
    int block_m = blockIdx.y * TILE_M;
    int block_n = blockIdx.x * TILE_N;
    
    if (block_m >= M || block_n >= N) return;
    
    // Shared memory for weight tile (dequantized on-the-fly)
    __shared__ half weight_tile[TILE_N][TILE_K + 8];  // +8 for bank conflict avoidance
    __shared__ half x_tile[TILE_M][TILE_K + 8];
    
    // Accumulator in registers (FP32 for precision)
    float accum[TILE_M / 32][TILE_N / 32][16][16] = {0};
    
    int tid = threadIdx.x + threadIdx.y * blockDim.x;
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    // Number of warps per block
    int warps_m = 4;  // TILE_M / 32
    int warps_n = 4;  // TILE_N / 32
    
    int warp_m = warp_id / warps_n;
    int warp_n = warp_id % warps_n;
    
    // Loop over K dimension in tiles
    for (int k_tile = 0; k_tile < K; k_tile += TILE_K) {
        int k_size = min(TILE_K, K - k_tile);
        
        // Load X tile into shared memory
        for (int i = tid; i < TILE_M * k_size; i += blockDim.x * blockDim.y) {
            int m_idx = i / k_size;
            int k_idx = i % k_size;
            int global_m = block_m + m_idx;
            int global_k = k_tile + k_idx;
            if (global_m < M && global_k < K) {
                x_tile[m_idx][k_idx] = x[global_m * K + global_k];
            } else {
                x_tile[m_idx][k_idx] = __float2half(0.0f);
            }
        }
        
        // Load and dequantize weight tile into shared memory
        // Each thread handles a portion of the weight tile
        int weight_rows_per_thread = max(1, TILE_N / (blockDim.x * blockDim.y / 4));
        
        for (int thread_row = 0; thread_row < weight_rows_per_thread; thread_row++) {
            int row_idx = tid / 4 + thread_row * (blockDim.x * blockDim.y / 4);
            
            if (row_idx < TILE_N) {
                int global_n = block_n + row_idx;
                if (global_n < N) {
                    // Dequantize this row for the current k_tile
                    int block_idx = k_tile / BLOCK_SIZE;
                    int blocks_in_tile = (k_size + BLOCK_SIZE - 1) / BLOCK_SIZE;
                    
                    for (int kb = 0; kb < blocks_in_tile; kb++) {
                        int k_block_start = kb * BLOCK_SIZE;
                        int k_block_end = min(k_block_start + BLOCK_SIZE, k_size);
                        int global_k_block = block_idx + kb;
                        
                        if (global_k_block < n_blocks) {
                            half scale = scales[global_n * n_blocks + global_k_block];
                            
                            for (int k = k_block_start; k < k_block_end; k += 2) {
                                int global_k_packed = (k_tile + k) / 2;
                                int packed_k = (K + 1) / 2;
                                
                                uint8_t packed = weight_packed[global_n * packed_k + global_k_packed];
                                int8_t lo = (packed & 0x0F) - 8;
                                int8_t hi = ((packed >> 4) & 0x0F) - 8;
                                
                                weight_tile[row_idx][k] = __short2half_rn(lo) * scale;
                                if (k + 1 < k_block_end) {
                                    weight_tile[row_idx][k + 1] = __short2half_rn(hi) * scale;
                                }
                            }
                        } else {
                            for (int k = k_block_start; k < k_block_end; k++) {
                                weight_tile[row_idx][k] = __float2half(0.0f);
                            }
                        }
                    }
                }
            }
        }
        
        __syncthreads();
        
        // Compute partial products using HMMA
        // Each warp computes a 32x32 output tile
        for (int k_step = 0; k_step < k_size; k_step += 16) {
            // Load 16x16 fragments from shared memory
            half a_frag[16], b_frag[16];
            
            // A fragment: 32x16 from x_tile (rows for this warp)
            // B fragment: 16x32 from weight_tile (cols for this warp)
            
            int a_row_base = warp_m * 32;
            int b_col_base = warp_n * 32;
            
            // Use WMMA or simple computation
            for (int i = 0; i < 16; i++) {
                int row = a_row_base + (lane_id / 16) * 16 + i;
                int col = k_step + (lane_id % 16);
                if (row < TILE_M && col < k_size) {
                    a_frag[i] = x_tile[row][col];
                } else {
                    a_frag[i] = __float2half(0.0f);
                }
            }
            
            for (int i = 0; i < 16; i++) {
                int row = k_step + (lane_id / 16);
                int col = b_col_base + (lane_id % 16) * 16 + i;
                col = col % 32;
                if (row < k_size && col < TILE_N) {
                    b_frag[i] = weight_tile[col][row];
                } else {
                    b_frag[i] = __float2half(0.0f);
                }
            }
            
            // Simple dot product accumulation (HMMA would be more efficient)
            for (int m = 0; m < 16; m++) {
                for (int n = 0; n < 16; n++) {
                    float sum = 0.0f;
                    for (int kk = 0; kk < 16; kk++) {
                        sum += __half2float(a_frag[m]) * __half2float(b_frag[n]);
                    }
                    accum[warp_m][warp_n][m][n] += sum * (k_step < k_size ? 1.0f : 0.0f);
                }
            }
        }
    }
    
    // Write output
    for (int wm = 0; wm < TILE_M / 32; wm++) {
        for (int wn = 0; wn < TILE_N / 32; wn++) {
            for (int m = 0; m < 16; m++) {
                for (int n = 0; n < 16; n++) {
                    int out_m = block_m + wm * 32 + m;
                    int out_n = block_n + wn * 32 + n;
                    if (out_m < M && out_n < N) {
                        output[out_m * N + out_n] = __float2half(accum[wm][wn][m][n]);
                    }
                }
            }
        }
    }
}

// Simpler, more efficient kernel using fused dequant + GEMM
__global__ void fp4_gemm_fused_kernel(
    const half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const half* __restrict__ scales,
    half* __restrict__ output,
    int batch_seq, int N, int K, int n_blocks
) {
    // Output tile: each block handles 64x64 output
    const int BLOCK_M = 64;
    const int BLOCK_N = 64;
    const int BLOCK_K = 32;  // Process K in chunks
    
    int block_m = blockIdx.y * BLOCK_M;
    int block_n = blockIdx.x * BLOCK_N;
    
    if (block_m >= batch_seq || block_n >= N) return;
    
    int tid = threadIdx.x;
    int wid = tid / 32;
    int lane = tid % 32;
    
    // Shared memory
    __shared__ half s_x[BLOCK_M][BLOCK_K];
    __shared__ half s_w[BLOCK_N][BLOCK_K];
    
    // Accumulator
    float accum[8][8] = {0.0f};
    
    // Initialize output to zero
    for (int i = 0; i < 8; i++) {
        for (int j = 0; j < 8; j++) {
            accum[i][j] = 0.0f;
        }
    }
    
    // Loop over K
    for (int k0 = 0; k0 < K; k0 += BLOCK_K) {
        int k_size = min(BLOCK_K, K - k0);
        
        // Load X tile - cooperative loading
        for (int idx = tid; idx < BLOCK_M * k_size; idx += 128) {
            int m = idx / k_size;
            int k = idx % k_size;
            int gm = block_m + m;
            int gk = k0 + k;
            if (gm < batch_seq && gk < K) {
                s_x[m][k] = x[gm * K + gk];
            } else {
                s_x[m][k] = __float2half(0.0f);
            }
        }
        
        // Load and dequantize weight tile
        // Each thread loads and dequantizes a portion
        for (int idx = tid; idx < BLOCK_N * k_size; idx += 128) {
            int wn = idx / k_size;
            int wk = idx % k_size;
            int gn = block_n + wn;
            int gk = k0 + wk;
            
            if (gn < N && gk < K) {
                int block_idx = gk / BLOCK_SIZE;
                int k_in_block = gk % BLOCK_SIZE;
                
                // Load packed weight
                int packed_k = (K + 1) / 2;
                uint8_t packed = weight_packed[gn * packed_k + gk / 2];
                
                int8_t w_val;
                if (gk % 2 == 0) {
                    w_val = (packed & 0x0F) - 8;
                } else {
                    w_val = ((packed >> 4) & 0x0F) - 8;
                }
                
                half scale = scales[gn * n_blocks + block_idx];
                s_w[wn][wk] = __short2half_rn(w_val) * scale;
            } else {
                s_w[wn][wk] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Matrix multiply accumulation
        // Each warp handles 16x16 output
        int warp_m = wid / 4;  // 0-3
        int warp_n = wid % 4;  // 0-3
        
        int m_start = warp_m * 16;
        int n_start = warp_n * 16;
        
        // Compute 16x16 = 16xk_size * k_sizex16
        for (int kk = 0; kk < k_size; kk++) {
            half x_vals[4];
            half w_vals[4];
            
            // Load values - distributed across lanes
            for (int i = 0; i < 4; i++) {
                int m = m_start + lane / 4 + i * 4;
                if (m < BLOCK_M) {
                    x_vals[i] = s_x[m][kk];
                } else {
                    x_vals[i] = __float2half(0.0f);
                }
            }
            
            for (int j = 0; j < 4; j++) {
                int n = n_start + lane % 4 + j * 4;
                if (n < BLOCK_N) {
                    w_vals[j] = s_w[n][kk];
                } else {
                    w_vals[j] = __float2half(0.0f);
                }
            }
            
            // Accumulate
            for (int i = 0; i < 4; i++) {
                for (int j = 0; j < 4; j++) {
                    accum[i][j] += __half2float(x_vals[i]) * __half2float(w_vals[j]);
                }
            }
        }
        
        __syncthreads();
    }
    
    // Write output
    int warp_m = wid / 4;
    int warp_n = wid % 4;
    int m_start = warp_m * 16;
    int n_start = warp_n * 16;
    
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 4; j++) {
            int m = m_start + lane / 4 + i * 4;
            int n = n_start + lane % 4 + j * 4;
            int gm = block_m + m;
            int gn = block_n + n;
            
            if (gm < batch_seq && gn < N) {
                // Warp reduction for this element
                float val = accum[i][j];
                output[gm * N + gn] = __float2half(val);
            }
        }
    }
}

// Use Triton-style block-wise GEMM with fused dequantization
// Highly optimized for large batch/sequence lengths
__global__ void fp4_gemm_optimized(
    const half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const half* __restrict__ scales,
    half* __restrict__ output,
    int M, int N, int K, int n_blocks
) {
    // Each CTA handles a 128x128 output tile
    const int TM = 128;
    const int TN = 128;
    const int TK = 32;
    
    int cta_m = blockIdx.y * TM;
    int cta_n = blockIdx.x * TN;
    
    if (cta_m >= M || cta_n >= N) return;
    
    int tid = threadIdx.x;
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    // Shared memory tiles with padding for bank conflicts
    __shared__ half s_a[TM][TK + 8];
    __shared__ half s_b[TN][TK + 8];
    
    // Register file accumulators (each warp handles 32x32)
    float reg_c[32][32] = {0};
    
    // Loop over K dimension
    for (int k0 = 0; k0 < K; k0 += TK) {
        int k_size = min(TK, K - k0);
        
        // Cooperative load of A tile (x input)
        // 128 threads x 4 elements each = 512 elements per iteration
        // Need to load TM x k_size elements
        int total_a = TM * k_size;
        for (int idx = tid; idx < total_a; idx += 128) {
            int row = idx / k_size;
            int col = idx % k_size;
            int global_m = cta_m + row;
            int global_k = k0 + col;
            if (global_m < M && global_k < K) {
                s_a[row][col] = x[global_m * K + global_k];
            } else {
                s_a[row][col] = __float2half(0.0f);
            }
        }
        
        // Cooperative load and dequantization of B tile (weights)
        int total_b = TN * k_size;
        int packed_k = (K + 1) / 2;
        
        for (int idx = tid; idx < total_b; idx += 128) {
            int col = idx / k_size;  // weight row (N dimension)
            int k = idx % k_size;    // K dimension
            int global_n = cta_n + col;
            int global_k = k0 + k;
            
            if (global_n < N && global_k < K) {
                // Determine which block and scale
                int block_idx = global_k / BLOCK_SIZE;
                
                // Load packed weight and dequantize
                uint8_t packed = weight_packed[global_n * packed_k + global_k / 2];
                int8_t w_val;
                if (global_k % 2 == 0) {
                    w_val = (packed & 0x0F) - 8;
                } else {
                    w_val = ((packed >> 4) & 0x0F) - 8;
                }
                
                half scale = scales[global_n * n_blocks + block_idx];
                s_b[col][k] = __short2half_rn(w_val) * scale;
            } else {
                s_b[col][k] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute using tensor cores or warp-level matrix multiply
        // Simplified: each warp computes a portion
        int warp_row = (warp_id / 4) * 32;  // 4 warps in M dimension
        int warp_col = (warp_id % 4) * 32;  // 4 warps in N dimension
        
        // Manual matmul
        for (int kk = 0; kk < k_size; kk++) {
            for (int m = 0; m < 32; m++) {
                half a_val = s_a[warp_row + m][kk];
                for (int n = 0; n < 32; n++) {
                    half b_val = s_b[warp_col + n][kk];
                    reg_c[m][n] += __half2float(a_val) * __half2float(b_val);
                }
            }
        }
        
        __syncthreads();
    }
    
    // Write output
    int warp_row = (warp_id / 4) * 32;
    int warp_col = (warp_id % 4) * 32;
    
    for (int m = 0; m < 32; m++) {
        for (int n = 0; n < 32; n++) {
            int global_m = cta_m + warp_row + m;
            int global_n = cta_n + warp_col + n;
            if (global_m < M && global_n < N) {
                output[global_m * N + global_n] = __float2half(reg_c[m][n]);
            }
        }
    }
}

// Simpler kernel that splits work more evenly
__global__ void simple_fp4_gemm(
    const half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const half* __restrict__ scales,
    half* __restrict__ output,
    int M, int N, int K, int n_blocks
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    int packed_k = (K + 1) / 2;
    
    for (int k = 0; k < K; k++) {
        // Load x value
        half x_val = x[row * K + k];
        
        // Load and dequantize weight
        int block_idx = k / BLOCK_SIZE;
        uint8_t packed = weight_packed[col * packed_k + k / 2];
        int8_t w_val;
        if (k % 2 == 0) {
            w_val = (packed & 0x0F) - 8;
        } else {
            w_val = ((packed >> 4) & 0x0F) - 8;
        }
        half scale = scales[col * n_blocks + block_idx];
        half w_deq = __short2half_rn(w_val) * scale;
        
        sum += __half2float(x_val) * __half2float(w_deq);
    }
    
    output[row * N + col] = __float2half(sum);
}

// WMMA-based kernel for better tensor core utilization
#include <mma.h>
using namespace nvcuda;

__global__ void wmma_fp4_gemm(
    const half* __restrict__ x,
    const uint8_t* __restrict__ weight_packed,
    const half* __restrict__ scales,
    half* __restrict__ output,
    int M, int N, int K, int n_blocks
) {
    // Tile sizes for WMMA
    const int WMMA_M = 16;
    const int WMMA_N = 16;
    const int WMMA_K = 16;
    
    const int TILE_M = 128;
    const int TILE_N = 128;
    const int TILE_K = 32;
    
    int cta_m = blockIdx.y * TILE_M;
    int cta_n = blockIdx.x * TILE_N;
    
    if (cta_m >= M || cta_n >= N) return;
    
    int tid = threadIdx.x;
    int warp_id = tid / 32;
    
    // Shared memory
    __shared__ half s_a[TILE_M][TILE_K + 8];
    __shared__ half s_b[TILE_N][TILE_K + 8];
    
    // Accumulator fragments (each warp has multiple)
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> frag_c[4];
    
    // Initialize accumulators to zero
    for (int i = 0; i < 4; i++) {
        wmma::fill_fragment(frag_c[i], 0.0f);
    }
    
    int packed_k = (K + 1) / 2;
    
    // Loop over K
    for (int k0 = 0; k0 < K; k0 += TILE_K) {
        int k_size = min(TILE_K, K - k0);
        
        // Load tiles cooperatively
        int total_a = TILE_M * k_size;
        for (int idx = tid; idx < total_a; idx += 128) {
            int row = idx / k_size;
            int col = idx % k_size;
            int gm = cta_m + row;
            int gk = k0 + col;
            if (gm < M && gk < K) {
                s_a[row][col] = x[gm * K + gk];
            } else {
                s_a[row][col] = __float2half(0.0f);
            }
        }
        
        int total_b = TILE_N * k_size;
        for (int idx = tid; idx < total_b; idx += 128) {
            int row = idx / k_size;
            int col = idx % k_size;
            int gn = cta_n + row;
            int gk = k0 + col;
            
            if (gn < N && gk < K) {
                int block_idx = gk / BLOCK_SIZE;
                uint8_t packed = weight_packed[gn * packed_k + gk / 2];
                int8_t w_val;
                if (gk % 2 == 0) {
                    w_val = (packed & 0x0F) - 8;
                } else {
                    w_val = ((packed >> 4) & 0x0F) - 8;
                }
                half scale = scales[gn * n_blocks + block_idx];
                s_b[row][col] = __short2half_rn(w_val) * scale;
            } else {
                s_b[row][col] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // WMMA computation
        int warp_m = (warp_id / 4) * 64;
        int warp_n = (warp_id % 4) * 64;
        
        for (int kk = 0; kk < k_size; kk += WMMA_K) {
            if (kk + WMMA_K > k_size) break;
            
            for (int m = 0; m < 4; m++) {
                for (int n = 0; n < 4; n++) {
                    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> frag_a;
                    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::col_major> frag_b;
                    
                    int a_row = warp_m + m * WMMA_M;
                    int b_col = warp_n + n * WMMA_N;
                    
                    // Load fragments
                    wmma::load_matrix_sync(frag_a, &s_a[a_row][kk], TILE_K + 8);
                    wmma::load_matrix_sync(frag_b, &s_b[b_col][kk], TILE_K + 8);
                    
                    // MMA
                    wmma::mma_sync(frag_c[m * 4 + n], frag_a, frag_b, frag_c[m * 4 + n]);
                }
            }
        }
        
        __syncthreads();
    }
    
    // Store results
    int warp_m = (warp_id / 4) * 64;
    int warp_n = (warp_id % 4) * 64;
    
    for (int m = 0; m < 4; m++) {
        for (int n = 0; n < 4; n++) {
            int out_row = warp_m + m * WMMA_M;
            int out_col = warp_n + n * WMMA_N;
            
            if (out_row < M && out_col < N) {
                wmma::store_matrix_sync(&output[out_row * N + out_col], frag_c[m * 4 + n], N, wmma::mem_row_major);
            }
        }
    }
}

torch::Tensor fp4_matmul(
    torch::Tensor x,
    torch::Tensor weight_packed,
    torch::Tensor scales,
    int K
) {
    int M = x.size(0);  // batch * seq_len
    int N = weight_packed.size(0);  // output features
    int n_blocks = scales.size(1);
    
    auto output = torch::empty({M, N}, x.options());
    
    // Use WMMA kernel for better performance
    dim3 block(128);
    dim3 grid((N + 127) / 128, (M + 127) / 128);
    
    wmma_fp4_gemm<<<grid, block>>>(
        x.data_ptr<at::Half>(),
        weight_packed.data_ptr<uint8_t>(),
        scales.data_ptr<at::Half>(),
        output.data_ptr<at::Half>(),
        M, N, K, n_blocks
    );
    
    return output;
}

    cuda_sources=cuda_source,
    functions=['fp4_matmul'],
    extra_cuda_cflags=['-arch=sm_100', '-O3', '--use_fast_math'],
    verbose=True,

BLOCK_SIZE = 32

class Model(nn.Module):
    """
    Optimized FP4 block-scaled GEMM using fused dequantization and tensor cores.
    """

    def __init__(self, K: int, N: int):
        super().__init__()
        self.K = K
        self.N = N

        # Fixed-seed weight generation so ref and sol models get identical weights
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        w_fp16 = torch.randn(N, K) * 0.02
        torch.random.set_rng_state(rng_state)
        self.weight_packed, self.scales = self._quantize_fp4(w_fp16)

    def _quantize_fp4(self, w: torch.Tensor):
        """Symmetric FP4 quantization with block-wise scales."""
        N, K = w.shape
        n_blocks = (K + BLOCK_SIZE - 1) // BLOCK_SIZE
        w_padded = torch.zeros(N, n_blocks * BLOCK_SIZE, dtype=w.dtype)
        w_padded[:, :K] = w

        w_blocked = w_padded.view(N, n_blocks, BLOCK_SIZE)
        amax = w_blocked.abs().amax(dim=-1, keepdim=True).clamp(min=1e-12)
        scales = amax / 7.0  # FP4 signed range: [-7, 7] with zero-point 8
        w_int = torch.round(w_blocked / scales).clamp(-7, 7).to(torch.int8) + 8  # shift to [1, 15]

        # Pack 2x4-bit values into uint8
        w_flat = w_int.view(N, -1)
        assert w_flat.shape[1] % 2 == 0
        lo = w_flat[:, 0::2].to(torch.uint8)
        hi = w_flat[:, 1::2].to(torch.uint8)
        packed = (hi << 4) | lo

        return nn.Parameter(packed, requires_grad=False), nn.Parameter(scales.squeeze(-1).to(torch.float16), requires_grad=False)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Optimized FP4 GEMM using fused dequantization.
        
        Input x: (batch, seq_len, K) in FP16
        Output: (batch, seq_len, N) in FP16
        """
        batch_size, seq_len, K = x.shape
        x_2d = x.view(batch_size * seq_len, K)
        
        # Use our optimized kernel
        out_2d = ext.fp4_matmul(
            x_2d.contiguous(),
            self.weight_packed,
            self.scales,
            self.K
        )
        
        return out_2d.view(batch_size, seq_len, self.N)


K = 4096
N = 4096
batch_size = 8
seq_len = 2048


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [K, N]
