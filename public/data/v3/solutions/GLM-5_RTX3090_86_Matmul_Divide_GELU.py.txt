import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>
#include <cmath>

// Constants for GELU approximation
constexpr float GELU_COEF = 0.044715f;
constexpr float SQRT_2_OVER_PI = 0.7978845608028654f;

__device__ __forceinline__ float gelu_float(float x) {
    float x3 = x * x * x;
    float inner = SQRT_2_OVER_PI * (x + GELU_COEF * x3);
    float tanh_inner = tanhf(inner);
    return 0.5f * x * (1.0f + tanh_inner);
}

// Simple FP16 kernel with vectorized operations
__global__ void fused_matmul_divide_gelu_fp16_simple(
    const at::Half* __restrict__ A_ptr,
    const at::Half* __restrict__ B_ptr,
    const at::Half* __restrict__ bias_ptr,
    at::Half* __restrict__ C_ptr,
    int M, int N, int K,
    float inv_divisor
) {
    const half* A = reinterpret_cast<const half*>(A_ptr);
    const half* B = reinterpret_cast<const half*>(B_ptr);
    const half* bias = reinterpret_cast<const half*>(bias_ptr);
    half* C = reinterpret_cast<half*>(C_ptr);
    
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row < M && col < N) {
        float sum = 0.0f;
        
        for (int k = 0; k < K; k++) {
            sum += __half2float(A[row * K + k]) * __half2float(B[k * N + col]);
        }
        
        sum += __half2float(bias[col]);
        sum *= inv_divisor;
        C[row * N + col] = __float2half(gelu_float(sum));
    }
}

// FP32 kernel
__global__ void fused_matmul_divide_gelu_fp32_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    const float* __restrict__ bias,
    float* __restrict__ C,
    int M, int N, int K,
    float inv_divisor
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row < M && col < N) {
        float sum = 0.0f;
        
        for (int k = 0; k < K; k++) {
            sum += A[row * K + k] * B[k * N + col];
        }
        
        sum += bias[col];
        sum *= inv_divisor;
        C[row * N + col] = gelu_float(sum);
    }
}

// FP16 Tensor Core kernel using WMMA with proper tile handling
// Tile sizes
#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

#define BLOCK_M 128
#define BLOCK_N 128
#define BLOCK_K 16

__global__ void fused_matmul_divide_gelu_fp16_wmma(
    const at::Half* __restrict__ A_ptr,
    const at::Half* __restrict__ __restrict__ B_ptr,
    const at::Half* __restrict__ bias_ptr,
    at::Half* __restrict__ C_ptr,
    int M, int N, int K,
    float inv_divisor
) {
    using namespace nvcuda::wmma;
    
    const half* A = reinterpret_cast<const half*>(A_ptr);
    const half* B = reinterpret_cast<const half*>(B_ptr);
    const half* bias = reinterpret_cast<const half*>(bias_ptr);
    half* C = reinterpret_cast<half*>(C_ptr);
    
    // Shared memory tiles
    __shared__ half sA[BLOCK_K][BLOCK_M];  // K x M (transposed for col-major load)
    __shared__ half sB[BLOCK_K][BLOCK_N];  // K x N
    
    // Block coordinates
    int block_m = blockIdx.y * BLOCK_M;  // Starting row of this block
    int block_n = blockIdx.x * BLOCK_N;  // Starting column of this block
    
    // Warp ID (0-15 for 16 warps in a 128-thread block with 8 warps)
    int warp_id = threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    
    // Each warp handles 32x32 output (2x2 tiles of 16x16)
    int warp_m = (warp_id % 4) * 32;  // 4 warps in M direction
    int warp_n = (warp_id / 4) * 32;  // 4 warps in N direction
    
    // Accumulator fragments (2x2 tiles per warp)
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc[2][2];
    
    #pragma unroll
    for (int i = 0; i < 2; i++) {
        #pragma unroll
        for (int j = 0; j < 2; j++) {
            fill_fragment(acc[i][j], 0.0f);
        }
    }
    
    // Loop over K dimension
    for (int k_block = 0; k_block < K; k_block += BLOCK_K) {
        // Load tiles into shared memory cooperatively
        // Each thread loads multiple elements
        
        // Load A tile: A is M x K, we need to load BLOCK_M x BLOCK_K
        // A[row, k] is at A[(block_m + local_m) * K + k_block + local_k]
        for (int i = threadIdx.x; i < BLOCK_M * BLOCK_K; i += blockDim.x) {
            int local_m = i / BLOCK_K;
            int local_k = i % BLOCK_K;
            int global_m = block_m + local_m;
            int global_k = k_block + local_k;
            
            if (global_m < M && global_k < K) {
                // Store transposed for easier col-major load
                sA[local_k][local_m] = A[global_m * K + global_k];
            } else {
                sA[local_k][local_m] = __float2half(0.0f);
            }
        }
        
        // Load B tile: B is K x N, we need to load BLOCK_K x BLOCK_N
        // B[k, col] is at B[(k_block + local_k) * N + block_n + local_n]
        for (int i = threadIdx.x; i < BLOCK_K * BLOCK_N; i += blockDim.x) {
            int local_k = i / BLOCK_N;
            int local_n = i % BLOCK_N;
            int global_k = k_block + local_k;
            int global_n = block_n + local_n;
            
            if (global_k < K && global_n < N) {
                sB[local_k][local_n] = B[global_k * N + global_n];
            } else {
                sB[local_k][local_n] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute using tensor cores
        // For each k step within the block
        fragment<matrix_a, WMMA_M, WMMA_N, WMMA_K, half, row_major> frag_a;
        fragment<matrix_b, WMMA_M, WMMA_N, WMMA_K, half, col_major> frag_b;
        
        // Load fragments and compute
        // frag_a loads from sA[k_step][warp_m]: row-major, 16 rows, 16 cols = 16x16 from sA
        // frag_b loads from sB[k_step][warp_n]: col-major, 16 rows, 16 cols = 16x16 from sB
        
        int k_step = 0;  // BLOCK_K is 16, so only one step
        
        #pragma unroll
        for (int i = 0; i < 2; i++) {
            #pragma unroll
            for (int j = 0; j < 2; j++) {
                // Load A fragment (row-major from shared memory)
                // sA is [K][M] = [16][128], we want sA[k_step:k_step+16, warp_m+i*16:warp_m+i*16+16]
                load_matrix_sync(frag_a, &sA[k_step][warp_m + i * WMMA_M], BLOCK_M);
                
                // Load B fragment (col-major from shared memory)
                // sB is [K][N] = [16][128], we want sB[k_step:k_step+16, warp_n+j*16:warp_n+j*16+16]
                load_matrix_sync(frag_b, &sB[k_step][warp_n + j * WMMA_N], BLOCK_N);
                
                mma_sync(acc[i][j], frag_a, frag_b, acc[i][j]);
            }
        }
        
        __syncthreads();
    }
    
    // Apply division and GELU, then store results
    #pragma unroll
    for (int i = 0; i < 2; i++) {
        #pragma unroll
        for (int j = 0; j < 2; j++) {
            // Apply division and GELU
            #pragma unroll
            for (int e = 0; e < acc[i][j].num_elements; e++) {
                float val = acc[i][j].x[e] * inv_divisor;
                acc[i][j].x[e] = gelu_float(val);
            }
            
            // Convert to half for storage
            fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, half> acc_half;
            #pragma unroll
            for (int e = 0; e < acc[i][j].num_elements; e++) {
                acc_half.x[e] = __float2half(acc[i][j].x[e]);
            }
            
            // Store to global memory
            int global_m = block_m + warp_m + i * WMMA_M;
            int global_n = block_n + warp_n + j * WMMA_N;
            
            if (global_m < M && global_n < N) {
                store_matrix_sync(&C[global_m * N + global_n], acc_half, N, mem_row_major);
            }
        }
    }
}

torch::Tensor fused_matmul_divide_gelu(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    double divisor
) {
    // x: (M, K), weight: (N, K) stored as Linear weight
    // We compute: output = x @ weight.T = (M, K) @ (K, N) = (M, N)
    
    int M = x.size(0);
    int K = x.size(1);
    int N = weight.size(0);
    
    auto output_dtype = x.scalar_type();
    auto output = torch::empty({M, N}, x.options().dtype(output_dtype));
    
    float inv_divisor = 1.0f / static_cast<float>(divisor);
    
    // Transpose weight for computation (weight is (N, K), we need (K, N))
    auto weight_t = weight.transpose(0, 1).contiguous();
    
    if (output_dtype == torch::kFloat16) {
        dim3 block(128);  // 4 warps
        dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);
        
        fused_matmul_divide_gelu_fp16_wmma<<<grid, block>>>(
            x.data_ptr<at::Half>(),
            weight_t.data_ptr<at::Half>(),
            bias.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            M, N, K, inv_divisor
        );
    } else if (output_dtype == torch::kFloat32) {
        dim3 block(16, 16);
        dim3 grid((N + 15) / 16, (M + 15) / 16);
        
        fused_matmul_divide_gelu_fp32_kernel<<<grid, block>>>(
            x.data_ptr<float>(),
            weight_t.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            M, N, K, inv_divisor
        );
    } else if (output_dtype == torch::kBFloat16) {
        // Convert to FP32, compute, convert back
        auto x_fp32 = x.to(torch::kFloat32);
        auto weight_fp32 = weight.to(torch::kFloat32);
        auto bias_fp32 = bias.to(torch::kFloat32);
        auto weight_t_fp32 = weight_fp32.transpose(0, 1).contiguous();
        auto output_fp32 = torch::empty({M, N}, x_fp32.options());
        
        dim3 block(16, 16);
        dim3 grid((N + 15) / 16, (M + 15) / 16);
        
        fused_matmul_divide_gelu_fp32_kernel<<<grid, block>>>(
            x_fp32.data_ptr<float>(),
            weight_t_fp32.data_ptr<float>(),
            bias_fp32.data_ptr<float>(),
            output_fp32.data_ptr<float>(),
            M, N, K, inv_divisor
        );
        
        output = output_fp32.to(torch::kBFloat16);
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_matmul_divide_gelu(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    double divisor
);
'''

ext = load_inline(
    name='fused_matmul_divide_gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_divide_gelu'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, input_size, output_size, divisor):
        super(Model, self).__init__()
        self.input_size = input_size
        self.output_size = output_size
        self.divisor = divisor
        
        # Store weight and bias as parameters
        self.weight = nn.Parameter(torch.empty(output_size, input_size))
        self.bias = nn.Parameter(torch.empty(output_size))
        
        # Initialize same as nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_matmul_divide_gelu(x, self.weight, self.bias, self.divisor)

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 10.0]
