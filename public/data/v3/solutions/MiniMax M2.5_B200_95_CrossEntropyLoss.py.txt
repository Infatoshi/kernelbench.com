import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def softmax_kernel(
    output_ptr, input_ptr, input_row_stride, output_row_stride, 
    n_cols, BLOCK_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0)
    row_start_ptr = input_ptr + row_idx * input_row_stride
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_start_ptr + col_offsets
    
    mask = col_offsets < n_cols
    row = tl.load(input_ptrs, mask=mask, other=float('-inf'))
    
    row_minus_max = row - tl.max(row, axis=0)
    numerator = tl.exp(row_minus_max)
    denominator = tl.sum(numerator, axis=0)
    softmax_output = numerator / denominator
    
    output_row_start_ptr = output_ptr + row_idx * output_row_stride
    output_ptrs = output_row_start_ptr + col_offsets
    tl.store(output_ptrs, softmax_output, mask=mask)


@triton.jit
def cross_entropy_kernel(
    loss_ptr, log_probs_ptr, targets_ptr,
    n_rows, n_classes, BLOCK_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    target = tl.load(targets_ptr + row_idx)
    
    log_probs_row_start = log_probs_ptr + row_idx * n_classes
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_classes
    
    log_probs = tl.load(log_probs_row_start + col_offsets, mask=mask)
    
    neg_log_prob = -tl.load(log_probs_row_start + target)
    
    tl.store(loss_ptr + row_idx, neg_log_prob)


def triton_cross_entropy(predictions, targets):
    """Triton-based cross entropy loss with softmax."""
    batch_size, num_classes = predictions.shape
    
    log_probs = torch.empty_like(predictions)
    BLOCK_SIZE = triton.next_power_of_2(num_classes)
    
    softmax_kernel[(batch_size,)](
        log_probs, predictions, predictions.stride(0),
        log_probs.stride(0), num_classes, BLOCK_SIZE=BLOCK_SIZE
    )
    
    loss = torch.empty(batch_size, device=predictions.device, dtype=predictions.dtype)
    
    cross_entropy_kernel[(batch_size,)](
        loss, log_probs, targets,
        batch_size, num_classes, BLOCK_SIZE=min(BLOCK_SIZE, 1024)
    )
    
    return loss.mean()


class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, predictions, targets):
        return torch.nn.functional.cross_entropy(predictions, targets)


class TritonModel(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, predictions, targets):
        return triton_cross_entropy(predictions, targets)
