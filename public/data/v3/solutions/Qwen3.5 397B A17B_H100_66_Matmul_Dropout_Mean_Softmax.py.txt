import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def reduce_sum_batches_kernel(
    x_ptr,
    sum_ptr,
    batch_size: tl.constexpr,
    in_features: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Reduce sum across batch dimension."""
    pid = tl.program_id(0)
    
    # Each program handles a block of input features
    block_start = pid * BLOCK_SIZE
    block_idx = block_start + tl.arange(0, BLOCK_SIZE)
    mask = block_idx < in_features
    
    # Sum across all batches
    acc = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    for batch_idx in range(batch_size):
        x_block = tl.load(x_ptr + batch_idx * in_features + block_idx,
                          mask=mask, other=0.0).to(tl.float32)
        acc += x_block
    
    tl.store(sum_ptr + block_idx, acc, mask=mask)


@triton.jit
def matmul_mean_dropout_kernel(
    sum_ptr, weight_ptr, bias_ptr,
    out_ptr,
    in_features: tl.constexpr,
    out_features: tl.constexpr,
    dropout_scale: tl.constexpr,
    batch_size: tl.constexpr,
    BLOCK_IN: tl.constexpr,
    BLOCK_OUT: tl.constexpr,
):
    """Compute matmul, mean, and dropout for each output element."""
    pid_out = tl.program_id(0)
    
    # Load bias
    bias = tl.load(bias_ptr + pid_out).to(tl.float32)
    
    acc = 0.0
    
    # Process input in blocks
    for block_start in range(0, in_features, BLOCK_IN):
        block_in = block_start + tl.arange(0, BLOCK_IN)
        mask_in = block_in < in_features
        
        # Load weights for this output column
        w_col = tl.load(weight_ptr + pid_out * in_features + block_in,
                        mask=mask_in, other=0.0).to(tl.float32)
        
        # Load sum across batches
        x_sum = tl.load(sum_ptr + block_in, mask=mask_in, other=0.0).to(tl.float32)
        
        # Accumulate dot product
        acc += tl.sum(x_sum * w_col)
    
    # Add bias, apply dropout scale, and compute mean
    acc += bias
    acc *= dropout_scale
    mean_val = acc / batch_size
    
    tl.store(out_ptr + pid_out, mean_val)


@triton.jit
def softmax_kernel(
    in_ptr,
    out_ptr,
    n: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Apply softmax to a 1D tensor."""
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < n
    
    x = tl.load(in_ptr + offs, mask=mask, other=-float('inf')).to(tl.float32)
    
    # Compute max for numerical stability
    x_max = tl.max(x, axis=0)
    
    # Compute exp(x - max)
    exp_x = tl.exp(x - x_max)
    
    # Compute sum
    sum_exp = tl.sum(exp_x, axis=0)
    
    # Normalize
    softmax_x = exp_x / sum_exp
    
    # Store result
    tl.store(out_ptr + offs, softmax_x, mask=mask)


class FusedMatmulDropoutMeanSoftmax(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, dropout_p):
        batch_size, in_features = x.shape
        out_features = weight.shape[0]
        
        dropout_scale = 1.0 / (1.0 - dropout_p)
        
        # Step 1: Sum across batches
        sum_x = torch.empty(in_features, dtype=torch.float32, device=x.device)
        BLOCK_SIZE = triton.next_power_of_2(in_features)
        if BLOCK_SIZE > 1024:
            BLOCK_SIZE = 1024
        grid = (triton.cdiv(in_features, BLOCK_SIZE),)
        reduce_sum_batches_kernel[grid](
            x, sum_x, batch_size, in_features, BLOCK_SIZE
        )
        
        # Step 2: Matmul with sum, apply mean and dropout
        mean_vals = torch.empty(out_features, dtype=torch.float32, device=x.device)
        BLOCK_IN = 1024
        BLOCK_OUT = 1
        grid = (out_features,)
        matmul_mean_dropout_kernel[grid](
            sum_x, weight, bias,
            mean_vals,
            in_features, out_features,
            dropout_scale, batch_size,
            BLOCK_IN, BLOCK_OUT,
        )
        
        # Step 3: Softmax
        output = torch.empty(out_features, dtype=torch.float32, device=x.device)
        BLOCK_SIZE = triton.next_power_of_2(out_features)
        if BLOCK_SIZE > 4096:
            BLOCK_SIZE = 4096
        softmax_kernel[(1,)](mean_vals, output, out_features, BLOCK_SIZE)
        
        # Reshape to (1, out_features) to match reference
        output = output.view(1, out_features)
        
        return output


def fused_matmul_dropout_mean_softmax(x, weight, bias, dropout_p):
    return FusedMatmulDropoutMeanSoftmax.apply(x, weight, bias, dropout_p)


class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.dropout_p = dropout_p
        
        # Initialize weight and bias
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        self.reset_parameters()
    
    def reset_parameters(self):
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        return fused_matmul_dropout_mean_softmax(x, self.weight, self.bias, self.dropout_p)


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [4096, 4096, 0.2]
