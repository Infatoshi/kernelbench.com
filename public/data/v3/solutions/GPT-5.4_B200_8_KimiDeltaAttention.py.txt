import torch
import torch.nn as nn
import torch.nn.functional as F
from fla.ops import chunk_kda

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ['fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


def kimi_delta_attention(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    a: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
) -> torch.Tensor:
    q_t = q.transpose(1, 2).contiguous()
    k_t = k.transpose(1, 2).contiguous()
    v_t = v.transpose(1, 2).contiguous()
    a_t = a.transpose(1, 2).contiguous()
    beta_t = beta.transpose(1, 2).contiguous()
    g = a_t.clamp(min=1e-6).log()
    output, _ = chunk_kda(q_t, k_t, v_t, g, beta_t, scale=scale)
    return output.transpose(1, 2)


class Model(nn.Module):
    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_dplr: bool = False,
        dplr_rank: int = 4,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.a_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=True)
        self.b_proj = nn.Linear(hidden_size, num_heads, bias=True)
        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)

        if use_short_conv:
            channels_qk = num_heads * head_dim_qk
            channels_v = num_heads * head_dim_v
            self.q_conv = nn.Conv1d(channels_qk, channels_qk, kernel_size=conv_kernel_size, groups=channels_qk, padding=conv_kernel_size - 1)
            self.k_conv = nn.Conv1d(channels_qk, channels_qk, kernel_size=conv_kernel_size, groups=channels_qk, padding=conv_kernel_size - 1)
            self.v_conv = nn.Conv1d(channels_v, channels_v, kernel_size=conv_kernel_size, groups=channels_v, padding=conv_kernel_size - 1)

        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, T, _ = x.shape
        H = self.num_heads
        Dq = self.head_dim_qk
        Dv = self.head_dim_v

        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)

        if self.use_short_conv:
            qt = q.transpose(1, 2)
            kt = k.transpose(1, 2)
            vt = v.transpose(1, 2)
            q = F.silu(self.q_conv(qt)[:, :, :T].transpose(1, 2))
            k = F.silu(self.k_conv(kt)[:, :, :T].transpose(1, 2))
            v = F.silu(self.v_conv(vt)[:, :, :T].transpose(1, 2))

        q = q.view(B, T, H, Dq).transpose(1, 2).contiguous()
        k = k.view(B, T, H, Dq).transpose(1, 2).contiguous()
        v = v.view(B, T, H, Dv).transpose(1, 2).contiguous()

        a = torch.sigmoid(self.a_proj(x)).view(B, T, H, Dv).transpose(1, 2).contiguous()
        beta = torch.sigmoid(self.b_proj(x)).transpose(1, 2).contiguous()

        o = kimi_delta_attention(q, k, v, a, beta, scale=self.scale)
        o = self.o_norm(o.transpose(1, 2))

        g = torch.sigmoid(self.g_proj(x)).view(B, T, H, Dv)
        o = (o * g).reshape(B, T, H * Dv)
        return self.o_proj(o)


batch_size = 4
seq_len = 2048
hidden_size = 2048
num_heads = 16
head_dim_qk = 128
head_dim_v = 128


def get_inputs():
    return [torch.randn(batch_size, seq_len, hidden_size)]


def get_init_inputs():
    return [hidden_size, num_heads, head_dim_qk, head_dim_v]
