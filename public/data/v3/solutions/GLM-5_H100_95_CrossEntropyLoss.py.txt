import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>

// Optimized cross entropy loss kernel
// Each thread block handles one sample (1024 classes = 32 warps)
// Uses warp reduction for finding max and sum of exponentials

template <typename scalar_t>
__global__ void cross_entropy_loss_kernel(
    const scalar_t* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    float* __restrict__ loss,
    int batch_size,
    int num_classes) {
    
    extern __shared__ float shared_data[];
    
    int sample_idx = blockIdx.x;
    if (sample_idx >= batch_size) return;
    
    const scalar_t* sample_preds = predictions + sample_idx * num_classes;
    int target = static_cast<int>(targets[sample_idx]);
    
    // Find max for numerical stability (parallel reduction)
    float local_max = -FLT_MAX;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        float val = static_cast<float>(sample_preds[i]);
        if (val > local_max) local_max = val;
    }
    
    // Warp-level reduction for max
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xffffffff, local_max, offset);
        if (other > local_max) local_max = other;
    }
    
    // Broadcast max from lane 0 to all threads in warp
    float warp_max = __shfl_sync(0xffffffff, local_max, 0);
    
    // Block-level reduction for max using shared memory
    if (threadIdx.x % 32 == 0) {
        shared_data[threadIdx.x / 32] = warp_max;
    }
    __syncthreads();
    
    // First warp reduces the per-warp maxes
    float block_max;
    if (threadIdx.x < (blockDim.x + 31) / 32) {
        block_max = shared_data[threadIdx.x];
    } else {
        block_max = -FLT_MAX;
    }
    
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xffffffff, block_max, offset);
        if (other > block_max) block_max = other;
    }
    
    block_max = __shfl_sync(0xffffffff, block_max, 0);
    
    // Compute sum of exp(logits - max)
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        float val = static_cast<float>(sample_preds[i]);
        local_sum += expf(val - block_max);
    }
    
    // Warp-level reduction for sum
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    
    // Store warp sums to shared memory
    if (threadIdx.x % 32 == 0) {
        shared_data[threadIdx.x / 32] = local_sum;
    }
    __syncthreads();
    
    // First warp reduces the per-warp sums
    float block_sum = 0.0f;
    if (threadIdx.x < (blockDim.x + 31) / 32) {
        block_sum = shared_data[threadIdx.x];
    }
    
    for (int offset = 16; offset > 0; offset >>= 1) {
        block_sum += __shfl_down_sync(0xffffffff, block_sum, offset);
    }
    
    block_sum = __shfl_sync(0xffffffff, block_sum, 0);
    
    // Thread 0 computes the final loss for this sample
    if (threadIdx.x == 0) {
        float target_logit = static_cast<float>(sample_preds[target]);
        float log_sum_exp = logf(block_sum);
        loss[sample_idx] = -target_logit + block_max + log_sum_exp;
    }
}

// Final reduction kernel to compute mean loss
__global__ void reduce_mean_kernel(const float* __restrict__ loss, float* __restrict__ output, int n) {
    float sum = 0.0f;
    for (int i = threadIdx.x; i < n; i += blockDim.x) {
        sum += loss[i];
    }
    
    // Warp reduction
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Single block - store result
    if (threadIdx.x == 0) {
        output[0] = sum / n;
    }
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    // Output tensor for per-sample losses
    auto loss = torch::empty({batch_size}, predictions.options().dtype(torch::kFloat32));
    auto output = torch::empty({}, predictions.options().dtype(torch::kFloat32));
    
    int threads_per_block = 256;  // 8 warps per block
    int shared_mem_size = ((threads_per_block + 31) / 32) * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(predictions.scalar_type(), "cross_entropy_loss", ([&] {
        cross_entropy_loss_kernel<scalar_t><<<batch_size, threads_per_block, shared_mem_size>>>(
            predictions.data_ptr<scalar_t>(),
            targets.data_ptr<int64_t>(),
            loss.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }));
    
    // Reduce to mean
    reduce_mean_kernel<<<1, 256>>>(
        loss.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size
    );
    
    return output;
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_loss(predictions, targets)

batch_size = 4096
num_classes = 1024
input_shape = (num_classes, )
dim = 1

def get_inputs():
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

def get_init_inputs():
    return []
