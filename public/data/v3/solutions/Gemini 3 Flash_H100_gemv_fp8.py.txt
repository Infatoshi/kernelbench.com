
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void gemv_int8_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    float x_scale,
    float weight_scale,
    half* __restrict__ output,
    int M,
    int N,
    int K) 
{
    // x_q: [M, K]
    // weight_q: [N, K]
    // output: [M, N]
    
    int row = blockIdx.y * blockDim.y + threadIdx.y; // batch index (M)
    int col = blockIdx.x * blockDim.x + threadIdx.x; // output feature index (N)

    if (row < M && col < N) {
        float sum = 0.0f;
        for (int k = 0; k < K; ++k) {
            sum += (float)x_q[row * K + k] * (float)weight_q[col * K + k];
        }
        output[row * N + col] = __float2half(sum * x_scale * weight_scale);
    }
}

// Optimized version with vectorized loads and partial reductions
__global__ void gemv_int8_vec_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    float x_scale,
    float weight_scale,
    half* __restrict__ output,
    int M,
    int N,
    int K) 
{
    // K is 4096. 32 threads per warp.
    // Each thread in a warp can handle a different column (N).
    // Or we can have threads in a warp collaborate on one column.
    // For N=14336 and K=4096, N is large.
    
    int row = blockIdx.y; // Batch index (M = 32)
    int col = blockIdx.x * blockDim.x + threadIdx.x; // Output feature index (N = 14336)

    if (row < M && col < N) {
        int32_t sum = 0;
        const int4* x_vec = (const int4*)(x_q + row * K);
        const int4* w_vec = (const int4*)(weight_q + col * K);
        
        // Cache x_vec in shared memory or just rely on L1/L2
        // Since many threads share the same x_q row, shared memory might help.
        // However, with M=32, each row is used by many threads (N=14336).
        
        #pragma unroll 8
        for (int i = 0; i < K / 16; ++i) {
            int4 xv = x_vec[i];
            int4 wv = w_vec[i];
            
            #pragma unroll
            for (int j = 0; j < 4; ++j) {
                int x_sub = ((int*)&xv)[j];
                int w_sub = ((int*)&wv)[j];
                // Using dp4a for int8 dot product to accelerate
                asm volatile("dp4a.s32.s32 %0, %1, %2, %0;" : "+r"(sum) : "r"(x_sub), "r"(w_sub));
            }
        }
        output[row * N + col] = __float2half((float)sum * x_scale * weight_scale);
    }
}

torch::Tensor gemv_int8(torch::Tensor x_q, torch::Tensor weight_q, float x_scale, float weight_scale) {
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = weight_q.size(0);
    
    auto output = torch::empty({M, N}, torch::dtype(torch::kFloat16).device(x_q.device()));
    
    dim3 block(256);
    dim3 grid((N + block.x - 1) / block.x, M);
    
    gemv_int8_vec_kernel<<<grid, block>>>(
        x_q.data_ptr<int8_t>(),
        weight_q.data_ptr<int8_t>(),
        x_scale,
        weight_scale,
        (half*)output.data_ptr<at::Half>(),
        M, N, K
    );
    
    return output;
}
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources='torch::Tensor gemv_int8(torch::Tensor x_q, torch::Tensor weight_q, float x_scale, float weight_scale);',
    cuda_sources=cuda_source,
    functions=['gemv_int8'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # x_q shape: [32, 4096]
        # x_scale: float
        return ext.gemv_int8(x_q, self.weight_q, float(x_scale), float(self.weight_scale))

def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8).cuda()
    x_scale = torch.tensor(0.02, dtype=torch.float32).cuda()
    return [x_q, x_scale]

def get_init_inputs():
    return []
