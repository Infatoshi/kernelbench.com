
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cmath>

template <typename T>
__device__ __forceinline__ T gelu_func(T x) {
    float fx = (float)x;
    return (T)(0.5f * fx * (1.0f + erff(fx * 0.7071067811865475f)));
}

template <typename T, int VEC_SIZE>
__global__ void gelu_kernel_vec(const T* __restrict__ x, T* __restrict__ y, int n) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * VEC_SIZE;
    if (idx + VEC_SIZE <= n) {
        alignas(sizeof(T) * VEC_SIZE) T vals[VEC_SIZE];
        #pragma unroll
        for (int i = 0; i < VEC_SIZE; i++) {
            vals[i] = x[idx + i];
        }
        #pragma unroll
        for (int i = 0; i < VEC_SIZE; i++) {
            vals[i] = gelu_func<T>(vals[i]);
        }
        #pragma unroll
        for (int i = 0; i < VEC_SIZE; i++) {
            y[idx + i] = vals[i];
        }
    } else {
        for (int i = idx; i < n; ++i) {
            y[i] = gelu_func<T>(x[i]);
        }
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int threads = 256;
    
    // Choose vector size based on type and alignment
    int vec_size = 1;
    if (n % 4 == 0) vec_size = 4;
    else if (n % 2 == 0) vec_size = 2;

    int blocks = (n / vec_size + threads - 1) / threads;

    if (x.scalar_type() == at::ScalarType::Float) {
        if (vec_size == 4) gelu_kernel_vec<float, 4><<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
        else gelu_kernel_vec<float, 1><<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    } else if (x.scalar_type() == at::ScalarType::Half) {
        if (vec_size == 4) gelu_kernel_vec<at::Half, 4><<<blocks, threads>>>(x.data_ptr<at::Half>(), y.data_ptr<at::Half>(), n);
        else gelu_kernel_vec<at::Half, 1><<<blocks, threads>>>(x.data_ptr<at::Half>(), y.data_ptr<at::Half>(), n);
    } else if (x.scalar_type() == at::ScalarType::BFloat16) {
        if (vec_size == 4) gelu_kernel_vec<at::BFloat16, 4><<<blocks, threads>>>(x.data_ptr<at::BFloat16>(), y.data_ptr<at::BFloat16>(), n);
        else gelu_kernel_vec<at::BFloat16, 1><<<blocks, threads>>>(x.data_ptr<at::BFloat16>(), y.data_ptr<at::BFloat16>(), n);
    }
    return y;
}
'''

cpp_source = r'''
torch::Tensor gelu_op(torch::Tensor x);
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    verbose=False,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '--use_fast_math']
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_op(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
