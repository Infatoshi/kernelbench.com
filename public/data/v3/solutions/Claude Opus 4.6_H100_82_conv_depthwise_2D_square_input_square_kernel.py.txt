import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Tiled depthwise conv2d for 3x3, stride=1, no padding
// Uses shared memory tiling: each block loads a tile of input and computes a tile of output
__global__ void depthwise_conv2d_3x3_tiled(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    int batch_size, int channels,
    int H, int W,
    int out_H, int out_W
) {
    // Block handles a TILE_H x TILE_W output region for one (batch, channel)
    const int TILE_W = 32;
    const int TILE_H = 8;
    
    int bc = blockIdx.z;
    int b = bc / channels;
    int c = bc - b * channels;
    
    int tile_row = blockIdx.y * TILE_H;
    int tile_col = blockIdx.x * TILE_W;
    
    int tx = threadIdx.x; // 0..31
    int ty = threadIdx.y; // 0..7
    
    // Load weights into registers
    float w[9];
    // Only 9 weights, load them collaboratively
    int tid = ty * 32 + tx;
    __shared__ float sw[9];
    if (tid < 9) {
        sw[tid] = weight[c * 9 + tid];
    }
    __syncthreads();
    
    #pragma unroll
    for (int i = 0; i < 9; i++) {
        w[i] = sw[i];
    }
    
    const float* in_ptr = input + (b * channels + c) * H * W;
    float* out_ptr = output + (b * channels + c) * out_H * out_W;
    
    int oh = tile_row + ty;
    int ow = tile_col + tx;
    
    if (oh < out_H && ow < out_W) {
        float acc = 0.0f;
        
        // stride=1, no padding: ih = oh + kh, iw = ow + kw
        int base_h = oh;
        int base_w = ow;
        
        #pragma unroll
        for (int kh = 0; kh < 3; kh++) {
            int ih = base_h + kh;
            #pragma unroll
            for (int kw = 0; kw < 3; kw++) {
                int iw = base_w + kw;
                acc += in_ptr[ih * W + iw] * w[kh * 3 + kw];
            }
        }
        
        out_ptr[oh * out_W + ow] = acc;
    }
}

// Vectorized version using float4 loads where possible
__global__ void depthwise_conv2d_3x3_vec(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    int batch_size, int channels,
    int H, int W,
    int out_H, int out_W
) {
    int bc = blockIdx.y;
    int b = bc / channels;
    int c = bc - b * channels;
    
    // Load weights
    __shared__ float sw[9];
    int tid = threadIdx.x;
    if (tid < 9) {
        sw[tid] = weight[c * 9 + tid];
    }
    __syncthreads();
    
    float w0 = sw[0], w1 = sw[1], w2 = sw[2];
    float w3 = sw[3], w4 = sw[4], w5 = sw[5];
    float w6 = sw[6], w7 = sw[7], w8 = sw[8];
    
    const float* in_ptr = input + (b * channels + c) * H * W;
    float* out_ptr = output + (b * channels + c) * out_H * out_W;
    
    int total_out = out_H * out_W;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Process 4 output pixels at a time along the width dimension
    // But we need to be careful about row boundaries
    for (int i = idx; i < total_out; i += gridDim.x * blockDim.x) {
        int oh = i / out_W;
        int ow = i - oh * out_W;
        
        float acc = 0.0f;
        
        int r0 = oh * W + ow;
        int r1 = r0 + W;
        int r2 = r1 + W;
        
        acc += in_ptr[r0] * w0 + in_ptr[r0 + 1] * w1 + in_ptr[r0 + 2] * w2;
        acc += in_ptr[r1] * w3 + in_ptr[r1 + 1] * w4 + in_ptr[r1 + 2] * w5;
        acc += in_ptr[r2] * w6 + in_ptr[r2 + 1] * w7 + in_ptr[r2 + 2] * w8;
        
        out_ptr[i] = acc;
    }
}


torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, int stride, int padding) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int KH = weight.size(2);
    int KW = weight.size(3);
    int out_H = (H + 2 * padding - KH) / stride + 1;
    int out_W = (W + 2 * padding - KW) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_H, out_W}, input.options());
    
    const float* in_ptr = input.data_ptr<float>();
    const float* w_ptr = weight.data_ptr<float>();
    float* out_ptr = output.data_ptr<float>();
    
    if (KH == 3 && KW == 3 && stride == 1 && padding == 0) {
        int total_out = out_H * out_W;
        int threads = 256;
        int blocks_x = (total_out + threads - 1) / threads;
        int blocks_y = batch_size * channels;
        
        dim3 grid(blocks_x, blocks_y);
        dim3 block(threads);
        
        depthwise_conv2d_3x3_vec<<<grid, block>>>(
            in_ptr, w_ptr, out_ptr, batch_size, channels, H, W, out_H, out_W);
    } else {
        // Tiled version
        const int TILE_W = 32;
        const int TILE_H = 8;
        dim3 block(32, 8);
        dim3 grid(
            (out_W + TILE_W - 1) / TILE_W,
            (out_H + TILE_H - 1) / TILE_H,
            batch_size * channels
        );
        depthwise_conv2d_3x3_tiled<<<grid, block>>>(
            in_ptr, w_ptr, out_ptr, batch_size, channels, H, W, out_H, out_W);
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, int stride, int padding);
'''

ext = load_inline(
    name='depthwise_conv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.depthwise_conv2d(x, self.conv2d.weight, self.conv2d.stride[0], self.conv2d.padding[0])


batch_size = 16
in_channels = 3
kernel_size = 3
width = 256
height = 256
stride = 1
padding = 0

def get_inputs():
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]

def get_init_inputs():
    return [in_channels, kernel_size, stride, padding]
