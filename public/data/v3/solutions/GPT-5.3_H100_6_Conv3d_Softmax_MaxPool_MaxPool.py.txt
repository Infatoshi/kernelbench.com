import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void conv3d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N,int Cin,int Cout,
    int D,int H,int W,
    int k)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*Cout*(D-k+1)*(H-k+1)*(W-k+1);
    if(idx>=total) return;

    int w_out = (W-k+1);
    int h_out = (H-k+1);
    int d_out = (D-k+1);

    int t = idx;
    int xw = t % w_out; t/=w_out;
    int xh = t % h_out; t/=h_out;
    int xd = t % d_out; t/=d_out;
    int co = t % Cout; t/=Cout;
    int n = t;

    float val = b ? b[co] : 0.f;

    for(int ci=0;ci<Cin;++ci){
        for(int kd=0;kd<k;++kd)
        for(int kh=0;kh<k;++kh)
        for(int kw=0;kw<k;++kw){
            int id = xd+kd;
            int ih = xh+kh;
            int iw = xw+kw;
            int in_idx = (((n*Cin+ci)*D+id)*H+ih)*W+iw;
            int w_idx = ((((co*Cin+ci)*k+kd)*k+kh)*k+kw);
            val += x[in_idx]*w[w_idx];
        }
    }

    int out_idx = (((n*Cout+co)*d_out+xd)*h_out+xh)*w_out+xw;
    out[out_idx] = val;
}

__global__ void softmax_pool_kernel(
    const float* __restrict__ x,
    float* __restrict__ out,
    int N,int C,int D,int H,int W)
{
    int D2 = D/4;
    int H2 = H/4;
    int W2 = W/4;

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*C*D2*H2*W2;
    if(idx>=total) return;

    int t=idx;
    int w2=t%W2; t/=W2;
    int h2=t%H2; t/=H2;
    int d2=t%D2; t/=D2;
    int c=t%C; t/=C;
    int n=t;

    float maxv = -1e20f;

    for(int dd=0;dd<4;++dd)
    for(int hh=0;hh<4;++hh)
    for(int ww=0;ww<4;++ww){
        int d = d2*4+dd;
        int h = h2*4+hh;
        int w = w2*4+ww;

        // stable softmax at this voxel
        float max_c = -1e20f;
        for(int cc=0;cc<C;++cc){
            int idx2 = (((n*C+cc)*D+d)*H+h)*W+w;
            float v = x[idx2];
            if(v > max_c) max_c = v;
        }
        float denom=0.f;
        for(int cc=0;cc<C;++cc){
            int idx2 = (((n*C+cc)*D+d)*H+h)*W+w;
            denom += expf(x[idx2] - max_c);
        }
        int idx_self = (((n*C+c)*D+d)*H+h)*W+w;
        float val = expf(x[idx_self] - max_c) / denom;

        if(val>maxv) maxv = val;
    }

    int out_idx = (((n*C+c)*D2+d2)*H2+h2)*W2+w2;
    out[out_idx]=maxv;
}

torch::Tensor fused(torch::Tensor x, torch::Tensor w, torch::Tensor b){
    int N=x.size(0), Cin=x.size(1), D=x.size(2), H=x.size(3), W=x.size(4);
    int Cout=w.size(0);
    int k=w.size(2);

    int D1=D-k+1, H1=H-k+1, W1=W-k+1;

    auto conv_out = torch::empty({N,Cout,D1,H1,W1}, x.options());

    int total = N*Cout*D1*H1*W1;
    conv3d_kernel<<<(total+255)/256,256>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), b.data_ptr<float>(), conv_out.data_ptr<float>(),
        N,Cin,Cout,D,H,W,k);

    int D2=D1/4, H2=H1/4, W2=W1/4;
    auto out = torch::empty({N,Cout,D2,H2,W2}, x.options());

    int total2 = N*Cout*D2*H2*W2;
    softmax_pool_kernel<<<(total2+255)/256,256>>>(
        conv_out.data_ptr<float>(), out.data_ptr<float>(),
        N,Cout,D1,H1,W1);

    return out;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused(torch::Tensor x, torch::Tensor w, torch::Tensor b);',
    cuda_sources=cuda_src,
    functions=['fused'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        w = self.conv.weight
        b = self.conv.bias
        return ext.fused(x.contiguous(), w.contiguous(), b.contiguous())


def get_inputs():
    return [torch.randn(16,3,16,128,128, device='cuda')]


def get_init_inputs():
    return [3,16,3,2]
