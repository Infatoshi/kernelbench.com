import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8, 'num_stages': 3, 'num_warps': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8, 'num_stages': 4, 'num_warps': 4}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 64, 'GROUP_SIZE_M': 8, 'num_stages': 4, 'num_warps': 4}, num_stages=4, num_warps=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def matmul_gelu_kernel(
    a_ptr, b_ptr, bias_ptr, out_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_om, stride_on,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = b_ptr + (offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        b = tl.load(b_ptrs, mask=offs_k[:, None] < K - k * BLOCK_SIZE_K, other=0.0)
        accumulator += tl.dot(a, b)
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_bk

    if bias_ptr is not None:
        bias_ptrs = bias_ptr + offs_bn[None, :]
        bias = tl.load(bias_ptrs, mask=offs_bn[None, :] < N, other=0.0)
        accumulator += bias

    # GELU
    accumulator = 0.5 * accumulator * (1.0 + tl.erf(accumulator * 0.7071067811865476))

    offs_om = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_on = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    out_ptrs = out_ptr + (offs_om[:, None] * stride_om + offs_on[None, :] * stride_on)
    tl.store(out_ptrs, accumulator, mask=(offs_om[:, None] < M) & (offs_on[None, :] < N))

@triton.jit
def softmax_kernel(
    output_ptr,
    M, N,
    stride_om, stride_on,
    BLOCK_SIZE_N: tl.constexpr,
):
    row_idx = tl.program_id(0)
    if row_idx >= M:
        return
    
    offsets = tl.arange(0, BLOCK_SIZE_N)
    mask = offsets < N
    
    row_ptr = output_ptr + row_idx * stride_om + offsets * stride_on
    row = tl.load(row_ptr, mask=mask, other=-float('inf')).to(tl.float32)
    
    row_max = tl.max(row, axis=0)
    numerator = tl.exp(row - row_max)
    denominator = tl.sum(numerator, axis=0)
    softmax_output = numerator / denominator
    
    tl.store(row_ptr, softmax_output, mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)

    def forward(self, x):
        M, K = x.shape
        N, K_linear = self.linear.weight.shape
        # Weight in nn.Linear is [N, K]
        # Reference: x @ weight.T + bias
        output = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        
        matmul_gelu_kernel[grid](
            x, self.linear.weight, self.linear.bias, output,
            M, N, K,
            x.stride(0), x.stride(1),
            self.linear.weight.stride(1), self.linear.weight.stride(0),
            output.stride(0), output.stride(1),
        )
        
        BLOCK_SIZE_N = 1
        while BLOCK_SIZE_N < N:
            BLOCK_SIZE_N *= 2
            
        softmax_grid = (M,)
        softmax_kernel[softmax_grid](
            output, M, N,
            output.stride(0), output.stride(1),
            BLOCK_SIZE_N=BLOCK_SIZE_N,
            num_warps=16 if BLOCK_SIZE_N >= 4096 else 8,
        )
        return output

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096]
