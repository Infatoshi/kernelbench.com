import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized matrix-vector multiplication kernel for H100
// A: (M, K), B: (K, 1), Output: (M, 1)
// Each thread block computes one row of the output

template<typename scalar_t>
__global__ void matvec_kernel(const scalar_t* __restrict__ A, 
                               const scalar_t* __restrict__ B,
                               scalar_t* __restrict__ C,
                               int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    // Use shared memory for B (the vector)
    extern __shared__ char smem[];
    scalar_t* B_shared = reinterpret_cast<scalar_t*>(smem);
    
    // Load B into shared memory in a coalesced manner
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    for (int i = tid; i < K; i += blockSize) {
        B_shared[i] = B[i];
    }
    __syncthreads();
    
    // Compute dot product for this row
    scalar_t sum = 0;
    const scalar_t* A_row = A + row * K;
    
    // Unrolled loop for better performance
    int vecSize = blockSize * 4;
    int i = tid;
    while (i + vecSize <= K) {
        sum += A_row[i] * B_shared[i];
        sum += A_row[i + blockSize] * B_shared[i + blockSize];
        sum += A_row[i + 2 * blockSize] * B_shared[i + 2 * blockSize];
        sum += A_row[i + 3 * blockSize] * B_shared[i + 3 * blockSize];
        i += vecSize;
    }
    
    // Handle remaining elements
    while (i < K) {
        sum += A_row[i] * B_shared[i];
        i += blockSize;
    }
    
    // Block reduction
    for (int stride = blockSize / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            sum += B_shared[tid + stride];  // Reuse B_shared as reduction buffer
        }
        __syncthreads();
        B_shared[tid] = sum;
        __syncthreads();
        sum = B_shared[tid];
    }
    
    if (tid == 0) {
        C[row] = sum;
    }
}

// Better version: use warp-level reduction and avoid shared memory overhead
template<typename scalar_t>
__global__ void matvec_kernel_v2(const scalar_t* __restrict__ A, 
                                  const scalar_t* __restrict__ B,
                                  scalar_t* __restrict__ C,
                                  int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int warpId = tid / 32;
    int laneId = tid % 32;
    
    // Each warp processes a portion of the row
    const scalar_t* A_row = A + row * K;
    
    scalar_t sum = 0;
    int elementsPerWarp = (K + 31) / 32;
    int startIdx = warpId * elementsPerWarp + laneId;
    
    for (int i = startIdx; i < K; i += 32) {
        sum += A_row[i] * B[i];
    }
    
    // Warp-level reduction using shuffle
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // First thread of each warp has the partial sum
    if (laneId == 0) {
        C[row + warpId * M] = sum;  // Store partial sums
    }
    __syncthreads();
    
    // Final reduction by first warp
    if (warpId == 0) {
        scalar_t finalSum = 0;
        int numWarps = (blockDim.x + 31) / 32;
        for (int w = 0; w < numWarps; w++) {
            finalSum += C[row + w * M];
        }
        if (laneId == 0) {
            C[row] = finalSum;
        }
    }
}

// Simple optimized version with better memory access pattern
__global__ void matvec_kernel_fp32(const float* __restrict__ A, 
                                    const float* __restrict__ B,
                                    float* __restrict__ C,
                                    int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    const float* A_row = A + row * K;
    
    float sum = 0;
    // Process 4 elements per iteration for better ILP
    int i = tid * 4;
    while (i + 3 < K) {
        sum += A_row[i] * B[i];
        sum += A_row[i + 1] * B[i + 1];
        sum += A_row[i + 2] * B[i + 2];
        sum += A_row[i + 3] * B[i + 3];
        i += blockDim.x * 4;
    }
    
    // Handle remaining elements
    while (i < K) {
        sum += A_row[i] * B[i];
        i += blockDim.x;
    }
    
    // Block reduction using shared memory
    extern __shared__ float sdata[];
    sdata[tid] = sum;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        C[row] = sdata[0];
    }
}

torch::Tensor matvec_op(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    int blockSize = 256;
    int gridSize = M;
    int sharedMemSize = blockSize * sizeof(float);
    
    if (A.dtype() == torch::kFloat32) {
        matvec_kernel_fp32<<<gridSize, blockSize, sharedMemSize>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    } else if (A.dtype() == torch::kFloat16) {
        // Launch fp32 kernel with cast (for simplicity)
        matvec_kernel_fp32<<<gridSize, blockSize, sharedMemSize>>>(
            (float*)A.data_ptr<at::Half>(),
            (float*)B.data_ptr<at::Half>(),
            (float*)C.data_ptr<at::Half>(),
            M, K
        );
    } else if (A.dtype() == torch::kBFloat16) {
        matvec_kernel_fp32<<<gridSize, blockSize, sharedMemSize>>>(
            (float*)A.data_ptr<at::BFloat16>(),
            (float*)B.data_ptr<at::BFloat16>(),
            (float*)C.data_ptr<at::BFloat16>(),
            M, K
        );
    }
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources='torch::Tensor matvec_op(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matvec_op'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_90', '-O3', '--use_fast_math'],
)

class Model(nn.Module):
    """
    Optimized model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.matvec_op(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
