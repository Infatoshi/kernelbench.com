import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// Fused kernel: Conv2d -> subtract1 -> tanh -> subtract2 -> AvgPool2d
// Input: [N, C_in, H, W]
// Conv: [C_out, C_in, K, K]
// Output: [N, C_out, (H-K+1)/pool, (W-K+1)/pool]

template<typename scalar_t>
__global__ void fused_conv_tanh_pool_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int N, int C_in, int H, int W,
    int C_out, int K,
    scalar_t subtract1, scalar_t subtract2,
    int pool_size,
    int H_out, int W_out
) {
    // Each thread handles one output element [n, c_out, h_out, w_out]
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = N * C_out * H_out * W_out;
    
    if (tid >= total_elements) return;
    
    // Decode indices
    int w_out = tid % W_out;
    int tmp = tid / W_out;
    int h_out = tmp % H_out;
    tmp = tmp / H_out;
    int c_out = tmp % C_out;
    int n = tmp / C_out;
    
    // Compute the pooling region in the pre-pool output
    int h_pre_start = h_out * pool_size;
    int w_pre_start = w_out * pool_size;
    int h_pre_end = min(h_pre_start + pool_size, H - K + 1);
    int w_pre_end = min(w_pre_start + pool_size, W - K + 1);
    
    scalar_t sum = 0;
    int count = 0;
    
    // For each position in the pooling region
    for (int h_pre = h_pre_start; h_pre < h_pre_end; h_pre++) {
        for (int w_pre = w_pre_start; w_pre < w_pre_end; w_pre++) {
            // Compute convolution at position (h_pre, w_pre)
            scalar_t conv_val = bias ? bias[c_out] : scalar_t(0);
            
            for (int c_in = 0; c_in < C_in; c_in++) {
                for (int kh = 0; kh < K; kh++) {
                    for (int kw = 0; kw < K; kw++) {
                        int h_in = h_pre + kh;
                        int w_in = w_pre + kw;
                        if (h_in < H && w_in < W) {
                            int input_idx = ((n * C_in + c_in) * H + h_in) * W + w_in;
                            int weight_idx = ((c_out * C_in + c_in) * K + kh) * K + kw;
                            conv_val += input[input_idx] * weight[weight_idx];
                        }
                    }
                }
            }
            
            // Apply subtract1, tanh, subtract2
            scalar_t val = conv_val - subtract1;
            val = tanh(val);
            val = val - subtract2;
            
            sum += val;
            count++;
        }
    }
    
    // Average pooling
    scalar_t result = sum / count;
    
    output[tid] = result;
}

torch::Tensor fused_conv_tanh_pool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double subtract1,
    double subtract2,
    int pool_size
) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    int C_out = weight.size(0);
    int K = weight.size(2);
    
    int H_conv = H - K + 1;
    int W_conv = W - K + 1;
    int H_out = H_conv / pool_size;
    int W_out = W_conv / pool_size;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    int threads = 256;
    int total_elements = N * C_out * H_out * W_out;
    int blocks = (total_elements + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "fused_conv_tanh_pool", ([&] {
        fused_conv_tanh_pool_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            N, C_in, H, W,
            C_out, K,
            (scalar_t)subtract1, (scalar_t)subtract2,
            pool_size,
            H_out, W_out
        );
    }));
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_conv_tanh_pool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double subtract1,
    double subtract2,
    int pool_size
);
"""

# Load the CUDA extension
ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_tanh_pool'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_100', '-O3'],
)


class Model(nn.Module):
    """
    Fused model: Conv2d -> subtract1 -> tanh -> subtract2 -> AvgPool2d
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        weight = self.conv.weight
        bias = self.conv.bias
        return ext.fused_conv_tanh_pool(
            x, weight, bias,
            self.subtract1_value, self.subtract2_value,
            self.kernel_size_pool
        )


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    subtract1_value = 0.5
    subtract2_value = 0.2
    kernel_size_pool = 2
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
