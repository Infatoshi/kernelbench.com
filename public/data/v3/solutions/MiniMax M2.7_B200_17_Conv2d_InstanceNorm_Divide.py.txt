import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// CUDA kernel for fused InstanceNorm forward with division
template <typename scalar_t>
__global__ void fused_instancenorm_divide_kernel(
    const scalar_t* __restrict__ input,
    const float* __restrict__ mean,
    const float* __restrict__ rstd,
    scalar_t* __restrict__ output,
    int N, int C, int H, int W,
    float divide_by) {
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * H * W;
    
    if (idx < total) {
        int tmp = idx;
        int w = tmp % W;
        tmp /= W;
        int h = tmp % H;
        tmp /= H;
        int c = tmp % C;
        int n = tmp / C;
        
        int offset = n * C + c;
        output[idx] = (input[idx] - mean[offset]) * rstd[offset] / divide_by;
    }
}

// Optimized kernel with vectorized access
template <typename scalar_t>
__global__ void optimized_instancenorm_divide_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const float* __restrict__ mean,
    const float* __restrict__ rstd,
    int N, int C, int H, int W,
    float divide_by) {
    
    int n = blockIdx.z;
    int c = blockIdx.y;
    int h = blockIdx.x * blockDim.y + threadIdx.y;
    int w = threadIdx.x;
    
    if (h < H && w < W) {
        int idx = ((n * C + c) * H + h) * W + w;
        int offset = n * C + c;
        output[idx] = (input[idx] - mean[offset]) * rstd[offset] / divide_by;
    }
}

torch::Tensor fast_fused_instancenorm_divide(
    torch::Tensor input,
    torch::Tensor mean,
    torch::Tensor var,
    float eps,
    float divide_by) {
    
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    auto output = torch::empty_like(input);
    
    // Precompute rstd: 1 / sqrt(var + eps)
    auto rstd = var.add_(eps).rsqrt_();
    
    dim3 block(16, 8);
    dim3 grid((H + block.y - 1) / block.y, C, N);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fast_fused_instancenorm_divide", ([&] {
        optimized_instancenorm_divide_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            mean.data_ptr<float>(),
            rstd.data_ptr<float>(),
            N, C, H, W, divide_by);
    }));
    
    return output;
}
'''

cpp_source = '''
torch::Tensor fast_fused_instancenorm_divide(
    torch::Tensor input,
    torch::Tensor mean,
    torch::Tensor var,
    float eps,
    float divide_by);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fast_fused_instancenorm_divide'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.out_channels = out_channels
        self.divide_by = divide_by
        
    def forward(self, x):
        # Conv2d with cuDNN acceleration (PyTorch uses cuDNN internally)
        x = self.conv(x)
        
        # Fused InstanceNorm + Divide
        # Compute mean and variance efficiently
        N, C, H, W = x.shape
        
        # Reshape for efficient batched computation
        x_flat = x.view(N, C, -1)  # (N, C, H*W)
        
        # Compute mean across spatial dimensions
        mean = x_flat.mean(dim=2, keepdim=True)  # (N, C, 1)
        
        # Compute variance (unbiased=False for InstanceNorm)
        var = x_flat.var(dim=2, unbiased=False, keepdim=True)  # (N, C, 1)
        
        # Fused normalization and divide using optimized CUDA kernel
        x_norm = ext.fast_fused_instancenorm_divide(x, mean.squeeze(2), var.squeeze(2), 1e-5, float(self.divide_by))
        
        return x_norm

def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    divide_by = 2.0
    return [in_channels, out_channels, kernel_size, divide_by]
