import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <float.h>
#include <math.h>

// FP32 Cross Entropy Loss kernel
__global__ void cross_entropy_loss_fp32_kernel(
    const float* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const float* pred_row = predictions + batch_idx * num_classes;
    int target = targets[batch_idx];
    
    // Step 1: Find max for numerical stability
    float max_val = -FLT_MAX;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        max_val = fmaxf(max_val, pred_row[i]);
    }
    
    // Reduce within block
    extern __shared__ float shared_mem[];
    float* max_shared = shared_mem;
    
    max_shared[threadIdx.x] = max_val;
    __syncthreads();
    
    for (int offset = blockDim.x / 2; offset > 0; offset /= 2) {
        if (threadIdx.x < offset) {
            max_shared[threadIdx.x] = fmaxf(max_shared[threadIdx.x], max_shared[threadIdx.x + offset]);
        }
        __syncthreads();
    }
    
    float row_max = max_shared[0];
    
    // Step 2: Compute sum of exp(x - max)
    float sum_exp = 0.0f;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        sum_exp += expf(pred_row[i] - row_max);
    }
    
    float* sum_shared = max_shared;
    sum_shared[threadIdx.x] = sum_exp;
    __syncthreads();
    
    for (int offset = blockDim.x / 2; offset > 0; offset /= 2) {
        if (threadIdx.x < offset) {
            sum_shared[threadIdx.x] += sum_shared[threadIdx.x + offset];
        }
        __syncthreads();
    }
    
    float log_sum_exp = row_max + logf(sum_shared[0]);
    
    // Step 3: Get the prediction for target class and compute loss
    float loss = 0.0f;
    if (threadIdx.x == 0) {
        float pred_target = pred_row[target];
        loss = log_sum_exp - pred_target;
        losses[batch_idx] = loss;
    }
}

// BF16 Cross Entropy Loss kernel
__global__ void cross_entropy_loss_bf16_kernel(
    const __nv_bfloat16* predictions,
    const int64_t* targets,
    float* losses,
    int batch_size,
    int num_classes

    if (batch_idx >= batch_size) return;

    float max_val = -FLT_MAX;
    for (int i = threadIdx.x; i < num_classes; i += blockDim.x) {
        float val = __bfloat162float(pred_row[i]);
        max_val = fmaxf(max_val, val);

        losses[batch_idx] = log_sum_exp - pred_target;
    }

// Final reduction kernel to compute mean
__global__ void reduce_mean_kernel(
    const float* losses,
    float* output,
    int n
) {
    extern __shared__ float shared[];
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    for (int i = tid; i < n; i += blockDim.x) {
        sum += losses[i];
    }
    
    shared[tid] = sum;
    __syncthreads();
    
    for (int offset = blockDim.x / 2; offset > 0; offset /= 2) {
        if (tid < offset) {
            shared[tid] += shared[tid + offset];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        *output = shared[0] / n;
    }
}

torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    auto losses = torch::empty(batch_size, predictions.options().dtype(torch::kFloat32).device(predictions.device()));
    auto output = torch::empty({}, predictions.options().dtype(torch::kFloat32).device(predictions.device()));
    
    int threads = 256;
    int shared_mem = threads * sizeof(float);
    
    if (predictions.dtype() == torch::kFloat32) {
        cross_entropy_loss_fp32_kernel<<<batch_size, threads, shared_mem>>>(
            predictions.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kFloat16) {
        cross_entropy_loss_fp16_kernel<<<batch_size, threads, shared_mem>>>(
            reinterpret_cast<const __half*>(predictions.data_ptr<c10::Half>()),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (predictions.dtype() == torch::kBFloat16) {
        cross_entropy_loss_bf16_kernel<<<batch_size, threads, shared_mem>>>(
            reinterpret_cast<const __nv_bfloat16*>(predictions.data_ptr<c10::BFloat16>()),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }
    
    // Reduce to get mean
    int reduce_threads = 256;
    int reduce_shared = reduce_threads * sizeof(float);
    reduce_mean_kernel<<<1, reduce_threads, reduce_shared>>>(
        losses.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size
    );
    
    return output;
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='torch::Tensor cross_entropy_loss(torch::Tensor predictions, torch::Tensor targets);',
    cuda_sources=cuda_source,
    functions=['cross_entropy_loss'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_loss(predictions, targets)

batch_size = 4096
num_classes = 1024
input_shape = (num_classes, )

def get_inputs():
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

def get_init_inputs():
    return []
