import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAStream.h>

// Fused activation + BatchNorm kernel
// activation: x * tanh(softplus(x))
// bn: (x - mean) / sqrt(var + eps) * weight + bias

template <typename T>
__device__ __forceinline__ T softplus(T x) {
    // softplus(x) = log(1 + exp(x))
    // Use log1p for numerical stability
    return log1pf(expf(x));
}

template <typename T>
__device__ __forceinline__ T tanh_softplus(T x) {
    // tanh(softplus(x))
    T sp = softplus(x);
    return tanhf(sp);
}

template <typename T>
__global__ void fused_activation_bn_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    const T* __restrict__ running_mean,
    const T* __restrict__ running_var,
    const T eps,
    int N, int C, int H, int W
) {
    // Each block processes one channel
    int c = blockIdx.x;
    if (c >= C) return;
    
    int HW = H * W;
    int num_elements = N * HW;
    
    // Load BN parameters for this channel
    T mean = running_mean[c];
    T inv_std = rsqrtf(running_var[c] + eps);
    T w = weight[c];
    T b = bias[c];
    
    // Process all elements in this channel
    for (int idx = threadIdx.x; idx < num_elements; idx += blockDim.x) {
        int n = idx / HW;
        int hw = idx % HW;
        
        // Load and compute activation
        T x = input[(n * C + c) * HW + hw];
        
        // Swish-like activation: x * tanh(softplus(x))
        T ts = tanh_softplus(x);
        T act = x * ts;
        
        // BatchNorm: (x - mean) * inv_std * w + b
        T normalized = (act - mean) * inv_std;
        T y = normalized * w + b;
        
        output[(n * C + c) * HW + hw] = y;
    }
}

torch::Tensor fused_activation_bn_op(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    auto output = torch::empty_like(input);
    
    int num_threads = 256;
    int num_blocks = C;
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_activation_bn", ([&] {
        fused_activation_bn_kernel<scalar_t><<<num_blocks, num_threads, 0, at::cuda::getCurrentCUDAStream()>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            running_mean.data_ptr<scalar_t>(),
            running_var.data_ptr<scalar_t>(),
            static_cast<scalar_t>(eps),
            N, C, H, W
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv_act_bn',
    cpp_sources=r'''
torch::Tensor fused_activation_bn_op(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps
);
''',
    cuda_sources=cuda_source,
    functions=['fused_activation_bn_op'],
    verbose=False,
)


class Model(nn.Module):
    """
    Optimized model that fuses Conv2d (cuDNN) + activation + BatchNorm
    using a custom CUDA kernel for activation + BN fusion.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps
        
        # Pre-allocate tensors for fused operation
        self.weight = None
        self.bias = None
        self.running_mean = None
        self.running_var = None
        
    def forward(self, x):
        # Use cuDNN-optimized Conv2d
        x = self.conv(x)
        
        # Use fused CUDA kernel for activation + BatchNorm
        # First, ensure BN buffers are on the same device
        if self.weight is None or self.weight.device != x.device:
            self.weight = self.bn.weight.data
            self.bias = self.bn.bias.data
            self.running_mean = self.bn.running_mean.data
            self.running_var = self.bn.running_var.data
        
        x = ext.fused_activation_bn_op(
            x,
            self.weight,
            self.bias,
            self.running_mean,
            self.running_var,
            self.eps
        )
        return x


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    return [in_channels, out_channels, kernel_size]
