import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def sum_reduction_kernel(
    x_ptr,
    y_ptr,
    reduce_dim_size,
    outer_size,
    inner_size,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    """
    Optimized sum reduction kernel.
    Each program handles BLOCK_M rows, reducing along dimension of size reduce_dim_size.
    """
    # Program IDs
    pid_m = tl.program_id(0)  # which group of outer_size rows
    pid_n = tl.program_id(1)  # which group of inner_size cols
    
    # Compute the row and column indices we're responsible for
    outer_idx = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    inner_idx = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    
    # Bounds check masks
    outer_mask = outer_idx < outer_size
    inner_mask = inner_idx < inner_size
    
    # Accumulator for sum
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Loop over reduction dimension
    for k in range(reduce_dim_size):
        # Compute input pointer offsets
        # Input layout: [outer_size, reduce_dim_size, inner_size]
        x_offsets = outer_idx[:, None] * reduce_dim_size * inner_size + k * inner_size + inner_idx[None, :]
        
        # Load and accumulate
        x_vals = tl.load(x_ptr + x_offsets, mask=outer_mask[:, None] & inner_mask[None, :], other=0.0)
        acc += x_vals.to(tl.float32)
    
    # Store results
    # Output layout: [outer_size, 1, inner_size]
    y_offsets = outer_idx[:, None] * inner_size + inner_idx[None, :]
    tl.store(y_ptr + y_offsets, acc, mask=outer_mask[:, None] & inner_mask[None, :])


def sum_reduction_triton(x: torch.Tensor, dim: int) -> torch.Tensor:
    """
    Perform sum reduction using Triton kernel.
    """
    # Handle negative dimension
    if dim < 0:
        dim = x.dim() + dim
    
    # Get dimensions
    shape = x.shape
    reduce_dim_size = shape[dim]
    
    # Compute outer and inner sizes
    outer_size = 1
    for i in range(dim):
        outer_size *= shape[i]
    
    inner_size = 1
    for i in range(dim + 1, len(shape)):
        inner_size *= shape[i]
    
    # Output shape
    out_shape = list(shape)
    out_shape[dim] = 1
    y = torch.empty(out_shape, dtype=x.dtype, device=x.device)
    
    # Ensure contiguous
    x = x.contiguous()
    
    # Block sizes for tuning
    BLOCK_M = 8
    BLOCK_N = 64
    
    # Grid dimensions
    grid_m = (outer_size + BLOCK_M - 1) // BLOCK_M
    grid_n = (inner_size + BLOCK_N - 1) // BLOCK_N
    
    # Launch kernel
    sum_reduction_kernel[grid_m, grid_n](
        x, y,
        reduce_dim_size,
        outer_size,
        inner_size,
        BLOCK_M,
        BLOCK_N,
    )
    
    return y


class Model(nn.Module):
    """
    Simple model that performs sum reduction over a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): Dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies sum reduction over the specified dimension.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor after sum reduction, shape (..., 1, ...).
        """
        return sum_reduction_triton(x, self.dim)
