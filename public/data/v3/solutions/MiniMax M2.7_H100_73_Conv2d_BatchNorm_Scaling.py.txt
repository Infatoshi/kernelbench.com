cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// CUDA kernel for fused Conv2d + BatchNorm + Scale
// Each thread handles one output element
template <typename scalar_t>
__global__ void conv_bn_scale_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    const scalar_t* __restrict__ bn_mean,
    const scalar_t* __restrict__ bn_var,
    const scalar_t* __restrict__ bn_gamma,
    const scalar_t* __restrict__ bn_beta,
    scalar_t* __restrict__ output,
    const float scaling_factor,
    const float eps,
    const int batch_size,
    const int out_channels,
    const int out_height,
    const int out_width,
    const int in_channels,
    const int kernel_size,
    const int in_height,
    const int in_width,
    const int stride_h,
    const int stride_w,
    const int pad_h,
    const int pad_w,
    const int dilation_h,
    const int dilation_w) {
    
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total_elements = batch_size * out_channels * out_height * out_width;
    
    if (out_idx >= total_elements) return;
    
    // Compute output indices
    const int n = out_idx / (out_channels * out_height * out_width);
    const int c = (out_idx / (out_height * out_width)) % out_channels;
    const int h = (out_idx / out_width) % out_height;
    const int w = out_idx % out_width;
    
    // Compute convolution
    scalar_t sum = static_cast<scalar_t>(0);
    
    const int in_start_h = h * stride_h - pad_h;
    const int in_start_w = w * stride_w - pad_w;
    
    for (int kc = 0; kc < in_channels; ++kc) {
        for (int kh = 0; kh < kernel_size; ++kh) {
            const int in_h = in_start_h + kh * dilation_h;
            if (in_h < 0 || in_h >= in_height) continue;
            
            for (int kw = 0; kw < kernel_size; ++kw) {
                const int in_w = in_start_w + kw * dilation_w;
                if (in_w < 0 || in_w >= in_width) continue;
                
                const int in_idx = ((n * in_channels + kc) * in_height + in_h) * in_width + in_w;
                const int weight_idx = ((c * in_channels + kc) * kernel_size + kh) * kernel_size + kw;
                
                sum += input[in_idx] * weight[weight_idx];
            }
        }
    }
    
    // Add bias
    if (bias != nullptr) {
        sum += bias[c];
    }
    
    // BatchNorm: (x - mean) / sqrt(var + eps) * gamma + beta
    const scalar_t mean = bn_mean[c];
    const scalar_t var = bn_var[c];
    const scalar_t gamma = bn_gamma[c];
    const scalar_t beta = bn_beta[c];
    
    scalar_t bn_val = (sum - mean) / sqrtf(var + eps) * gamma + beta;
    
    // Apply scaling factor
    bn_val = bn_val * static_cast<scalar_t>(scaling_factor);
    
    output[out_idx] = bn_val;
}

torch::Tensor conv_bn_scale_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    torch::Tensor bn_gamma,
    torch::Tensor bn_beta,
    double scaling_factor,
    double eps,
    int stride_h,
    int stride_w,
    int pad_h,
    int pad_w,
    int dilation_h,
    int dilation_w) {
    
    const int batch_size = input.size(0);
    const int in_channels = input.size(1);
    const int in_height = input.size(2);
    const int in_width = input.size(3);
    const int out_channels = weight.size(0);
    const int kernel_size = weight.size(2);
    
    // Compute output spatial dimensions
    const int out_height = (in_height + 2 * pad_h - dilation_h * (kernel_size - 1) - 1) / stride_h + 1;
    const int out_width = (in_width + 2 * pad_w - dilation_w * (kernel_size - 1) - 1) / stride_w + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_height, out_width}, input.options());
    
    const int total_elements = batch_size * out_channels * out_height * out_width;
    const int block_size = 256;
    const int num_blocks = (total_elements + block_size - 1) / block_size;
    
    const float scale = static_cast<float>(scaling_factor);
    const float epsilon = static_cast<float>(eps);
    
    if (input.scalar_type() == at::ScalarType::Float) {
        conv_bn_scale_float_kernel<<<num_blocks, block_size>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.defined() ? bias.data_ptr<float>() : nullptr,
            bn_mean.data_ptr<float>(),
            bn_var.data_ptr<float>(),
            bn_gamma.data_ptr<float>(),
            bn_beta.data_ptr<float>(),
            output.data_ptr<float>(),
            scale, epsilon,
            batch_size, out_channels, out_height, out_width,
            in_channels, kernel_size, in_height, in_width,
            stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w);
    } else if (input.scalar_type() == at::ScalarType::Half) {
        conv_bn_scale_half_kernel<<<num_blocks, block_size>>>(
            reinterpret_cast<const half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(weight.data_ptr<at::Half>()),
            bias.defined() ? reinterpret_cast<const half*>(bias.data_ptr<at::Half>()) : nullptr,
            bn_mean.data_ptr<float>(),
            bn_var.data_ptr<float>(),
            bn_gamma.data_ptr<float>(),
            bn_beta.data_ptr<float>(),
            reinterpret_cast<half*>(output.data_ptr<at::Half>()),
            scale, epsilon,
            batch_size, out_channels, out_height, out_width,
            in_channels, kernel_size, in_height, in_width,
            stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w);
    } else if (input.scalar_type() == at::ScalarType::BFloat16) {
        conv_bn_scale_bf16_kernel<<<num_blocks, block_size>>>(
            reinterpret_cast<const nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<const nv_bfloat16*>(weight.data_ptr<at::BFloat16>()),
            bias.defined() ? reinterpret_cast<const nv_bfloat16*>(bias.data_ptr<at::BFloat16>()) : nullptr,
            bn_mean.data_ptr<float>(),
            bn_var.data_ptr<float>(),
            bn_gamma.data_ptr<float>(),
            bn_beta.data_ptr<float>(),
            reinterpret_cast<nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            scale, epsilon,
            batch_size, out_channels, out_height, out_width,
            in_channels, kernel_size, in_height, in_width,
            stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w);
    } else {
        AT_ERROR("Unsupported dtype");
    }
    
    return output;
}

ext = load_inline(
    name='conv_bn_scale_ext',
    cpp_sources='''
torch::Tensor conv_bn_scale_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    torch::Tensor bn_gamma,
    torch::Tensor bn_beta,

    double eps,
    int stride_h,
    int stride_w,
    int pad_h,
    int pad_w,
    int dilation_h,
    int dilation_w);

    cuda_sources=cuda_source,
    functions=['conv_bn_scale_forward'],
    verbose=False,

class Model(nn.Module):
    """
    Fused Conv2d + BatchNorm + Scale model using custom CUDA kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        
        # Store BN parameters
        self.register_buffer('bn_mean', self.bn.running_mean.clone())
        self.register_buffer('bn_var', self.bn.running_var.clone())
        self.register_buffer('bn_gamma', self.bn.weight.data.clone() if self.bn.weight is not None else torch.ones(out_channels))
        self.register_buffer('bn_beta', self.bn.bias.data.clone() if self.bn.bias is not None else torch.zeros(out_channels))
        
        # Precompute fused parameters for inference
        self._compute_fused_params()
    
    def _compute_fused_params(self):
        """Precompute parameters for fused batchnorm."""
        eps = self.bn.eps
        var = self.bn.running_var
        mean = self.bn.running_mean
        gamma = self.bn.weight.data if self.bn.weight is not None else torch.ones_like(var)
        beta = self.bn.bias.data if self.bn.bias is not None else torch.zeros_like(var)
        
        # Compute std
        std = torch.sqrt(var + eps)
        
        # Store scale and bias for fused operation: y = (x - mean) / std * gamma + beta
        self.register_buffer('bn_scale', gamma / std)
        self.register_buffer('bn_bias', -mean / std * gamma + beta)
    
    def forward(self, x):
        weight = self.conv.weight.to(x.dtype)
        bias = self.conv.bias
        if bias is not None:
            bias = bias.to(x.dtype)
        else:
            bias = torch.zeros(x.size(1), device=x.device, dtype=x.dtype)
        
        # Convert BN parameters to match input dtype
        bn_scale = self.bn_scale.to(x.dtype)
        bn_bias = self.bn_bias.to(x.dtype)
        bn_mean = self.bn_mean.to(x.dtype)
        bn_var = self.bn_var.to(x.dtype)
        bn_gamma = self.bn_gamma.to(x.dtype)
        bn_beta = self.bn_beta.to(x.dtype)
        
        output = ext.conv_bn_scale_forward(
            x, weight, bias,
            bn_mean, bn_var, bn_gamma, bn_beta,
            self.scaling_factor, self.bn.eps,
            self.conv.stride[0], self.conv.stride[1],
            self.conv.padding[0], self.conv.padding[1],
            self.conv.dilation[0], self.conv.dilation[1]
        )
        
        return output


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 2.0]
