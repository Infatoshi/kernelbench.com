import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "conv"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _dwconv3x3_kernel(
    x_ptr, w_ptr, b_ptr, y_ptr,
    N, C, H, W, OH, OW,
    stride_h, stride_w, pad_h, pad_w,
    x_sN, x_sC, x_sH, x_sW,
    w_sC, w_sKH, w_sKW,
    b_sC,
    y_sN, y_sC, y_sH, y_sW,
    HAS_BIAS: tl.constexpr,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    offs = pid * BLOCK + tl.arange(0, BLOCK)
    total = N * C * OH * OW
    mask = offs < total

    ow = offs % OW
    t = offs // OW
    oh = t % OH
    t = t // OH
    c = t % C
    n = t // C

    acc = tl.zeros([BLOCK], dtype=tl.float32)

    base_h = oh * stride_h - pad_h
    base_w = ow * stride_w - pad_w

    for kh in range(3):
        ih = base_h + kh
        hmask = (ih >= 0) & (ih < H)
        for kw in range(3):
            iw = base_w + kw
            valid = mask & hmask & (iw >= 0) & (iw < W)
            x_idx = n * x_sN + c * x_sC + ih * x_sH + iw * x_sW
            w_idx = c * w_sC + kh * w_sKH + kw * w_sKW
            x = tl.load(x_ptr + x_idx, mask=valid, other=0.0)
            w = tl.load(w_ptr + w_idx, mask=mask, other=0.0)
            acc += x * w

    if HAS_BIAS:
        acc += tl.load(b_ptr + c * b_sC, mask=mask, other=0.0)

    y_idx = n * y_sN + c * y_sC + oh * y_sH + ow * y_sW
    tl.store(y_ptr + y_idx, acc, mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        conv = self.conv2d
        if (
            x.is_cuda and conv.weight.is_cuda and
            conv.groups == conv.in_channels == conv.out_channels and
            conv.kernel_size == (3, 3) and
            conv.stride == (1, 1) and
            conv.padding == (0, 0) and
            x.ndim == 4 and
            x.is_contiguous(memory_format=torch.contiguous_format) and
            conv.weight.is_contiguous() and
            (conv.bias is None or conv.bias.is_contiguous()) and
            x.dtype in (torch.float16, torch.bfloat16, torch.float32)
        ):
            N, C, H, W = x.shape
            OH = H - 2
            OW = W - 2
            y = torch.empty((N, C, OH, OW), device=x.device, dtype=x.dtype)
            total = N * C * OH * OW
            BLOCK = 256
            grid = (triton.cdiv(total, BLOCK),)
            _dwconv3x3_kernel[grid](
                x, conv.weight, conv.bias if conv.bias is not None else x,
                y,
                N, C, H, W, OH, OW,
                1, 1, 0, 0,
                x.stride(0), x.stride(1), x.stride(2), x.stride(3),
                conv.weight.stride(0), conv.weight.stride(2), conv.weight.stride(3),
                conv.bias.stride(0) if conv.bias is not None else 0,
                y.stride(0), y.stride(1), y.stride(2), y.stride(3),
                HAS_BIAS=conv.bias is not None,
                BLOCK=BLOCK,
            )
            return y
        return conv(x)


batch_size = 16
in_channels = 3
kernel_size = 3
width = 256
height = 256
stride = 1
padding = 0


def get_inputs():
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    return [in_channels, kernel_size, stride, padding]
