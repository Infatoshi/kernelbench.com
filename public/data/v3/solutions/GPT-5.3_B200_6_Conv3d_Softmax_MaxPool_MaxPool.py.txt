import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

#define MAX_C 32

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N, int Cin, int Cout,
    int D, int H, int W,
    int k,
    int D1, int H1, int W1,
    int D2, int H2, int W2
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * D2 * H2 * W2;
    if (idx >= total) return;

    int w2 = idx % W2; idx /= W2;
    int h2 = idx % H2; idx /= H2;
    int d2 = idx % D2; idx /= D2;
    int c = idx % Cout; idx /= Cout;
    int n = idx;

    float max_val = -1e30f;

    // region size = 4x4x4
    for (int rd = 0; rd < 4; ++rd) {
        for (int rh = 0; rh < 4; ++rh) {
            for (int rw = 0; rw < 4; ++rw) {
                int d1 = d2 * 4 + rd;
                int h1 = h2 * 4 + rh;
                int w1 = w2 * 4 + rw;

                if (d1 >= D1 || h1 >= H1 || w1 >= W1) continue;

                float vals[MAX_C];
                float max_c = -1e30f;

                // compute conv + find max
                for (int co = 0; co < Cout; ++co) {
                    float acc = b[co];
                    for (int ci = 0; ci < Cin; ++ci) {
                        for (int kd = 0; kd < k; ++kd) {
                            for (int kh = 0; kh < k; ++kh) {
                                for (int kw = 0; kw < k; ++kw) {
                                    int id = d1 + kd;
                                    int ih = h1 + kh;
                                    int iw = w1 + kw;
                                    float xv = x[((((n*Cin+ci)*D+id)*H+ih)*W+iw)];
                                    float wv = w[((((co*Cin+ci)*k+kd)*k+kh)*k+kw)];
                                    acc += xv * wv;
                                }
                            }
                        }
                    }
                    vals[co] = acc;
                    if (acc > max_c) max_c = acc;
                }

                float sum = 0.0f;
                for (int co = 0; co < Cout; ++co) {
                    float e = expf(vals[co] - max_c);
                    vals[co] = e;
                    sum += e;
                }

                float soft = vals[c] / sum;
                if (soft > max_val) max_val = soft;
            }
        }
    }

    out[((((n*Cout+c)*D2+d2)*H2+h2)*W2+w2)] = max_val;
}

torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b) {
    int N = x.size(0);
    int Cin = x.size(1);
    int D = x.size(2);
    int H = x.size(3);
    int W = x.size(4);

    int Cout = w.size(0);
    int k = w.size(2);

    int D1 = D - k + 1;
    int H1 = H - k + 1;
    int W1 = W - k + 1;

    int D2 = D1 / 4;
    int H2 = H1 / 4;
    int W2 = W1 / 4;

    auto out = torch::empty({N, Cout, D2, H2, W2}, x.options());

    int total = N * Cout * D2 * H2 * W2;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        N, Cin, Cout,
        D, H, W,
        k,
        D1, H1, W1,
        D2, H2, W2
    );

    return out;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b);',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        return ext.fused_forward(x.float(), self.conv.weight.float(), self.conv.bias.float())


def get_inputs():
    return [torch.randn(16, 3, 16, 128, 128, device='cuda')]


def get_init_inputs():
