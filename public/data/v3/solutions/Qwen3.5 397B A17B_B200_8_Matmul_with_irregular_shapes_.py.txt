import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Simple but correct tiled GEMM kernel
// Block size: 16x16 threads
// Tile size: 16 elements in K dimension
#define BLOCK_SIZE 16
#define TILE_SIZE 16

__global__ void tiled_gemm_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K) {
    
    // Shared memory for tiles - statically sized
    __shared__ float As[BLOCK_SIZE][TILE_SIZE];
    __shared__ float Bs[TILE_SIZE][BLOCK_SIZE];
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Global row and column this thread computes
    int row = by * BLOCK_SIZE + ty;
    int col = bx * BLOCK_SIZE + tx;
    
    float acc = 0.0f;
    
    // Number of tiles along K dimension
    int num_tiles = (K + TILE_SIZE - 1) / TILE_SIZE;
    
    for (int t = 0; t < num_tiles; ++t) {
        int k_base = t * TILE_SIZE;
        
        // Load tile from A (row-major)
        // Each thread loads one element
        int a_col = k_base + tx;
        if (row < M && a_col < K) {
            As[ty][tx] = A[row * K + a_col];
        } else {
            As[ty][tx] = 0.0f;
        }
        
        // Load tile from B (row-major)
        // Each thread loads one element
        int b_row = k_base + ty;
        if (b_row < K && col < N) {
            Bs[ty][tx] = B[b_row * N + col];
        } else {
            Bs[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute dot product for this tile
        for (int k = 0; k < TILE_SIZE; ++k) {
            acc += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    // Store result
    if (row < M && col < N) {
        C[row * N + col] = acc;
    }
}

torch::Tensor gemm_op(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.dim() == 2, "A must be 2D");
    TORCH_CHECK(B.dim() == 2, "B must be 2D");
    TORCH_CHECK(A.size(1) == B.size(0), "Dimension mismatch");
    TORCH_CHECK(A.is_cuda(), "A must be on CUDA");
    TORCH_CHECK(B.is_cuda(), "B must be on CUDA");
    
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(A.device()));
    
    // Block configuration: 16x16 threads
    dim3 block(BLOCK_SIZE, BLOCK_SIZE);
    dim3 grid((N + BLOCK_SIZE - 1) / BLOCK_SIZE, (M + BLOCK_SIZE - 1) / BLOCK_SIZE);
    
    tiled_gemm_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );
    
    return C;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor gemm_op(torch::Tensor A, torch::Tensor B);
'''

# Compile with Sm100 architecture
ext = load_inline(
    name='gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_op'],
    extra_cuda_cflags=[
        '-O3',
        '-arch=sm_100',
        '-std=c++17',
        '-use_fast_math',
    ],
    extra_cflags=['-O3', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemm_op(A, B)

M = 8205
K = 2949
N = 5921

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, N)
    return [A, B]

def get_init_inputs():
    return []
