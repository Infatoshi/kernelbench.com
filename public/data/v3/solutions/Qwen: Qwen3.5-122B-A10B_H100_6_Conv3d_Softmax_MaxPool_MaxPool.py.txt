import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <algorithm>

// 3D Convolution kernel (with bias)
// Input: (N, C_in, D, H, W)
// Weight: (C_out, C_in, kD, kH, kW)
// Bias: (C_out)
// Output: (N, C_out, D, H, W) - same spatial dims with padding=0, stride=1
template<typename scalar_t>
__global__ void conv3d_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int N, int C_in, int C_out,
    int D, int H, int W,
    int kD, int kH, int kW,
    int outD, int outH, int outW)
{
    // Each thread handles one output element
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * outD * outH * outW;
    
    if (idx >= total) return;
    
    // Decode indices
    int tmp = idx;
    int w_out = tmp % outW; tmp /= outW;
    int h_out = tmp % outH; tmp /= outH;
    int d_out = tmp % outD; tmp /= outD;
    int c_out = tmp % C_out;
    int n = tmp / C_out;
    
    scalar_t sum = 0;
    
    // Convolution computation
    for (int c_in = 0; c_in < C_in; c_in++) {
        for (int kd = 0; kd < kD; kd++) {
            for (int kh = 0; kh < kH; kh++) {
                for (int kw = 0; kw < kW; kw++) {
                    int d_in = d_out + kd;
                    int h_in = h_out + kh;
                    int w_in = w_out + kw;
                    
                    if (d_in < D && h_in < H && w_in < W) {
                        int in_idx = ((n * C_in + c_in) * D + d_in) * H * W + h_in * W + w_in;
                        int wt_idx = ((c_out * C_in + c_in) * kD + kd) * kH * kW + kh * kW + kw;
                        sum += input[in_idx] * weight[wt_idx];
                    }
                }
            }
        }
    }
    
    if (bias != nullptr) {
        sum += bias[c_out];
    }
    
    output[idx] = sum;
}

// Softmax kernel along dimension 1 (channel dimension)
// Input shape: (N, C, D, H, W)
template<typename scalar_t>
__global__ void softmax_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int N, int C, int D, int H, int W)
{
    // Each thread block handles one (n, d, h, w) position across all channels
    int pos = blockIdx.x;  // position in N*D*H*W
    int tid = threadIdx.x;
    
    int total_positions = N * D * H * W;
    if (pos >= total_positions) return;
    
    int spatial_size = C;
    int stride = N * D * H * W;
    
    // Find max for this position
    scalar_t max_val = -FLT_MAX;
    for (int c = tid; c < C; c += blockDim.x) {
        scalar_t val = input[pos + c * stride];
        if (val > max_val) max_val = val;
    }
    
    // Reduce max across threads
    __shared__ scalar_t shared_max[256];
    shared_max[tid] = max_val;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_max[tid] = fmaxf(shared_max[tid], shared_max[tid + s]);
        }
        __syncthreads();
    }
    max_val = shared_max[0];
    
    // Compute exp and sum
    scalar_t sum_exp = 0;
    scalar_t* local_output = output + pos;
    const scalar_t* local_input = input + pos;
    
    for (int c = tid; c < C; c += blockDim.x) {
        scalar_t exp_val = expf(local_input[c * stride] - max_val);
        local_output[c * stride] = exp_val;
        sum_exp += exp_val;
    }
    
    // Reduce sum across threads
    __shared__ scalar_t shared_sum[256];
    shared_sum[tid] = sum_exp;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_sum[tid] += shared_sum[tid + s];
        }
        __syncthreads();
    }
    sum_exp = shared_sum[0];
    
    // Normalize
    scalar_t inv_sum = 1.0f / sum_exp;
    for (int c = tid; c < C; c += blockDim.x) {
        local_output[c * stride] *= inv_sum;
    }
}

// MaxPool3D kernel
// Input: (N, C, D, H, W)
// Output: (N, C, outD, outH, outW)
template<typename scalar_t>
__global__ void maxpool3d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int N, int C, int D, int H, int W,
    int kD, int kH, int kW,
    int outD, int outH, int outW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * outD * outH * outW;
    
    if (idx >= total) return;
    
    // Decode indices
    int tmp = idx;
    int w_out = tmp % outW; tmp /= outW;
    int h_out = tmp % outH; tmp /= outH;
    int d_out = tmp % outD; tmp /= outD;
    int c = tmp % C;
    int n = tmp / C;
    
    scalar_t max_val = -FLT_MAX;
    
    for (int kd = 0; kd < kD; kd++) {
        for (int kh = 0; kh < kH; kh++) {
            for (int kw = 0; kw < kW; kw++) {
                int d_in = d_out * 2 + kd;  // stride = 2 for pool_kernel_size=2
                int h_in = h_out * 2 + kh;
                int w_in = w_out * 2 + kw;
                
                if (d_in < D && h_in < H && w_in < W) {
                    int in_idx = ((n * C + c) * D + d_in) * H * W + h_in * W + w_in;
                    if (input[in_idx] > max_val) {
                        max_val = input[in_idx];
                    }
                }
            }
        }
    }
    
    output[idx] = max_val;
}

// Fused forward pass: conv3d -> softmax -> maxpool -> maxpool
template<typename scalar_t>
torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int kD, int kH, int kW,
    int pool_kernel_size)
{
    int N = input.size(0);
    int C_in = input.size(1);
    int D = input.size(2);
    int H = input.size(3);
    int W = input.size(4);
    
    int C_out = weight.size(0);
    
    // Conv output dimensions (no padding, stride=1)
    int outD_conv = D - kD + 1;
    int outH_conv = H - kH + 1;
    int outW_conv = W - kW + 1;
    
    // After conv
    int conv_total = N * C_out * outD_conv * outH_conv * outW_conv;
    
    // Allocate conv output
    auto conv_out = torch::empty({N, C_out, outD_conv, outH_conv, outW_conv}, 
                                  input.options().dtype(torch::kFloat32));
    
    // Launch conv kernel
    int threads = 256;
    int blocks = (conv_total + threads - 1) / threads;
    conv3d_kernel<scalar_t><<<blocks, threads>>>(
        input.data_ptr<scalar_t>(),
        weight.data_ptr<scalar_t>(),
        bias.data_ptr<scalar_t>(),
        conv_out.data_ptr<scalar_t>(),
        N, C_in, C_out, D, H, W, kD, kH, kW,
        outD_conv, outH_conv, outW_conv);
    
    // Softmax output dimensions (same as conv output)
    int softmax_total = conv_total;
    auto softmax_out = torch::empty({N, C_out, outD_conv, outH_conv, outW_conv},
                                     input.options().dtype(torch::kFloat32));
    
    int softmax_blocks = N * outD_conv * outH_conv * outW_conv;
    softmax_kernel<scalar_t><<<softmax_blocks, threads>>>(
        conv_out.data_ptr<scalar_t>(),
        softmax_out.data_ptr<scalar_t>(),
        N, C_out, outD_conv, outH_conv, outW_conv);
    
    // First maxpool
    int outD_p1 = outD_conv / pool_kernel_size;
    int outH_p1 = outH_conv / pool_kernel_size;
    int outW_p1 = outW_conv / pool_kernel_size;
    int pool1_total = N * C_out * outD_p1 * outH_p1 * outW_p1;
    
    auto pool1_out = torch::empty({N, C_out, outD_p1, outH_p1, outW_p1},
                                   input.options().dtype(torch::kFloat32));
    
    blocks = (pool1_total + threads - 1) / threads;
    maxpool3d_kernel<scalar_t><<<blocks, threads>>>(
        softmax_out.data_ptr<scalar_t>(),
        pool1_out.data_ptr<scalar_t>(),
        N, C_out, outD_conv, outH_conv, outW_conv,
        pool_kernel_size, pool_kernel_size, pool_kernel_size,
        outD_p1, outH_p1, outW_p1);
    
    // Second maxpool
    int outD_p2 = outD_p1 / pool_kernel_size;
    int outH_p2 = outH_p1 / pool_kernel_size;
    int outW_p2 = outW_p1 / pool_kernel_size;
    int pool2_total = N * C_out * outD_p2 * outH_p2 * outW_p2;
    
    auto pool2_out = torch::empty({N, C_out, outD_p2, outH_p2, outW_p2},
                                   input.options().dtype(torch::kFloat32));
    
    blocks = (pool2_total + threads - 1) / threads;
    maxpool3d_kernel<scalar_t><<<blocks, threads>>>(
        pool1_out.data_ptr<scalar_t>(),
        pool2_out.data_ptr<scalar_t>(),
        N, C_out, outD_p1, outH_p1, outW_p1,
        pool_kernel_size, pool_kernel_size, pool_kernel_size,
        outD_p2, outH_p2, outW_p2);
    
    return pool2_out;
}

torch::Tensor fused_forward_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int kD, int kH, int kW,
    int pool_kernel_size)
{
    return fused_forward<float>(input, weight, bias, kD, kH, kW, pool_kernel_size);
}

torch::Tensor fused_forward_fp16(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int kD, int kH, int kW,
    int pool_kernel_size)
{
    // Convert to float32 for computation
    auto input_f32 = input.to(torch::kFloat32);
    auto weight_f32 = weight.to(torch::kFloat32);
    auto bias_f32 = bias.to(torch::kFloat32);
    auto result = fused_forward<float>(input_f32, weight_f32, bias_f32, kD, kH, kW, pool_kernel_size);
    return result.to(torch::kFloat16);
}

torch::Tensor fused_forward_bf16(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int kD, int kH, int kW,
    int pool_kernel_size)
{
    // Convert to float32 for computation
    auto input_f32 = input.to(torch::kFloat32);
    auto weight_f32 = weight.to(torch::kFloat32);
    auto bias_f32 = bias.to(torch::kFloat32);
    auto result = fused_forward<float>(input_f32, weight_f32, bias_f32, kD, kH, kW, pool_kernel_size);
    return result.to(torch::kBFloat16);
}
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources='''
torch::Tensor fused_forward_fp32(torch::Tensor, torch::Tensor, torch::Tensor, int, int, int, int);
torch::Tensor fused_forward_fp16(torch::Tensor, torch::Tensor, torch::Tensor, int, int, int, int);
torch::Tensor fused_forward_bf16(torch::Tensor, torch::Tensor, torch::Tensor, int, int, int, int);
''',
    cuda_sources=cuda_source,
    functions=['fused_forward_fp32', 'fused_forward_fp16', 'fused_forward_bf16'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool1 = nn.MaxPool3d(pool_kernel_size)
        self.pool2 = nn.MaxPool3d(pool_kernel_size)
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Get kernel dimensions
        kD, kH, kW = self.kernel_size, self.kernel_size, self.kernel_size
        pool_size = self.pool_kernel_size
        
        # Determine precision
        dtype = x.dtype
        
        if dtype == torch.float32:
            result = ext.fused_forward_fp32(
                x, self.conv.weight, self.conv.bias,
                kD, kH, kW, pool_size
            )
        elif dtype == torch.float16:
            result = ext.fused_forward_fp16(
                x, self.conv.weight, self.conv.bias,
                kD, kH, kW, pool_size
            )
        elif dtype == torch.bfloat16:
            result = ext.fused_forward_bf16(
                x, self.conv.weight, self.conv.bias,
                kD, kH, kW, pool_size
            )
        else:
            # Fallback to PyTorch for other dtypes
            x = self.conv(x)
            x = torch.softmax(x, dim=1)
            x = self.pool1(x)
            x = self.pool2(x)
            return x
        
        return result

def get_inputs():
    return [torch.randn(16, 3, 16, 128, 128)]

def get_init_inputs():
    return [3, 16, 3, 2]
