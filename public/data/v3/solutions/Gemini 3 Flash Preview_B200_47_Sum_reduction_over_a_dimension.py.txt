import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename scalar_t>
__global__ void sum_reduction_kernel(const scalar_t* __restrict__ input, scalar_t* __restrict__ output,
                                    int dim_pre, int dim_reduce, int dim_post) {
    int post_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int pre_idx = blockIdx.y;

    if (post_idx < dim_post) {
        float sum = 0.0f;
        const scalar_t* in_ptr = input + pre_idx * dim_reduce * dim_post + post_idx;
        for (int i = 0; i < dim_reduce; ++i) {
            sum += (float)(*in_ptr);
            in_ptr += dim_post;
        }
        output[pre_idx * dim_post + post_idx] = (scalar_t)sum;
    }
}

template <typename scalar_t>
__global__ void sum_reduction_kernel_large_dim_reduce(const scalar_t* __restrict__ input, scalar_t* __restrict__ output,
                                                     int dim_pre, int dim_reduce, int dim_post) {
    int idx = blockIdx.x;
    int pre_idx = idx / dim_post;
    int post_idx = idx % dim_post;

    extern __shared__ float sdata[];

    float sum = 0.0f;
    const scalar_t* in_ptr = input + pre_idx * dim_reduce * dim_post + post_idx;
    for (int i = threadIdx.x; i < dim_reduce; i += blockDim.x) {
        sum += (float)in_ptr[i * dim_post];
    }
    sdata[threadIdx.x] = sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }
    
    if (threadIdx.x < 32) {
        volatile float* v_sdata = sdata;
        if (blockDim.x >= 64) v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 32];
        v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 16];
        v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 8];
        v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 4];
        v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 2];
        v_sdata[threadIdx.x] += v_sdata[threadIdx.x + 1];
    }

    if (threadIdx.x == 0) {
        output[pre_idx * dim_post + post_idx] = (scalar_t)sdata[0];
    }
}

torch::Tensor sum_reduction_cuda(torch::Tensor x, int64_t dim) {
    auto shape = x.sizes().vec();
    int64_t dim_reduce = shape[dim];
    int64_t dim_pre = 1;
    for (int i = 0; i < dim; ++i) dim_pre *= shape[i];
    int64_t dim_post = 1;
    for (int i = dim + 1; i < (int)shape.size(); ++i) dim_post *= shape[i];

    auto output_shape = shape;
    output_shape[dim] = 1;
    auto y = torch::empty(output_shape, x.options());

    if (dim_reduce > 512 && dim_post < 64) {
        int threads = 256;
        int blocks = dim_pre * dim_post;
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "sum_reduction_cuda", ([&] {
            sum_reduction_kernel_large_dim_reduce<scalar_t><<<blocks, threads, threads * sizeof(float)>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                dim_pre, (int)dim_reduce, (int)dim_post);
        }));
    } else {
        dim3 block(256);
        dim3 grid((dim_post + 255) / 256, dim_pre);
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "sum_reduction_cuda", ([&] {
            sum_reduction_kernel<scalar_t><<<grid, block>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                (int)dim_pre, (int)dim_reduce, (int)dim_post);
        }));
    }

    return y;
}
'''

cpp_source = r'''
torch::Tensor sum_reduction_cuda(torch::Tensor x, int64_t dim);
'''

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduction_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduction_cuda(x, self.dim)

def get_inputs():
    return [torch.randn(64, 256, 256, device='cuda')]

def get_init_inputs():
    return [1]
