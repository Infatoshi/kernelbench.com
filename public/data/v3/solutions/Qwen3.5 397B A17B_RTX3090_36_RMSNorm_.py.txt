import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

template<typename T>
__global__ void rmsnorm_kernel(const T* __restrict__ x, T* __restrict__ y, 
                                int total_rows, int features, float eps) {
    int row_idx = blockIdx.x;
    if (row_idx >= total_rows) return;
    
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    extern __shared__ float shared_mem[];
    
    // Compute sum of squares over features
    float sum_sq = 0.0f;
    int row_start = row_idx * features;
    
    for (int f = tid; f < features; f += stride) {
        float val = static_cast<float>(x[row_start + f]);
        sum_sq += val * val;
    }
    
    shared_mem[tid] = sum_sq;
    __syncthreads();
    
    // Parallel reduction
    for (int s = stride / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_mem[tid] += shared_mem[tid + s];
        }
        __syncthreads();
    }
    
    float rms = sqrtf(shared_mem[0] / features + eps);
    float inv_rms = 1.0f / rms;
    
    // Apply normalization
    for (int f = tid; f < features; f += stride) {
        float val = static_cast<float>(x[row_start + f]);
        y[row_start + f] = static_cast<T>(val * inv_rms);
    }
}

torch::Tensor rmsnorm_forward(torch::Tensor x, float eps) {
    auto batch = x.size(0);
    auto features = x.size(1);
    auto dim1 = x.size(2);
    auto dim2 = x.size(3);
    
    // Reshape to (batch * dim1 * dim2, features) for processing
    auto total_rows = batch * dim1 * dim2;
    
    auto y = torch::empty_like(x);
    
    // Flatten x and y for kernel
    auto x_flat = x.view({-1, features});
    auto y_flat = y.view({-1, features});
    
    int threads = 256;
    if (features < threads) threads = features;
    int blocks = total_rows;
    
    int shared_mem_size = threads * sizeof(float);
    
    if (x.dtype() == torch::kFloat32) {
        rmsnorm_kernel<float><<<blocks, threads, shared_mem_size>>>(
            x_flat.data_ptr<float>(), y_flat.data_ptr<float>(), total_rows, features, eps);
    } else if (x.dtype() == torch::kFloat16) {
        rmsnorm_kernel<at::Half><<<blocks, threads, shared_mem_size>>>(
            x_flat.data_ptr<at::Half>(), y_flat.data_ptr<at::Half>(), total_rows, features, eps);
    } else if (x.dtype() == torch::kBFloat16) {
        rmsnorm_kernel<at::BFloat16><<<blocks, threads, shared_mem_size>>>(
            x_flat.data_ptr<at::BFloat16>(), y_flat.data_ptr<at::BFloat16>(), total_rows, features, eps);
    }
    
    return y;
}
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources='''
#include <torch/extension.h>
torch::Tensor rmsnorm_forward(torch::Tensor x, float eps);
''',
    cuda_sources=cuda_source,
    functions=['rmsnorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_forward(x, self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [features]
