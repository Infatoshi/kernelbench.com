import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized sum reduction kernel with shared memory
// Each block handles one row: reduces dim1 elements to produce one output
// Grid: (batch_size * dim2) blocks
// Block: 256 threads
template<typename T>
__global__ void sum_reduce_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    const int batch_size,
    const int dim1,
    const int dim2
) {
    extern __shared__ T sdata[];
    
    const int tid = threadIdx.x;
    
    // Each block reduces over dim1 dimension for one (batch, dim2) position
    const int batch_idx = blockIdx.x / dim2;
    const int dim2_idx = blockIdx.x % dim2;
    
    // Memory layout: x[b,i,j] at offset b*dim1*dim2 + i*dim2 + j
    const int stride = blockDim.x;
    const int base_offset = batch_idx * dim1 * dim2 + dim2_idx;
    
    // Thread-local reduction: each thread sums its portion
    T sum_val = static_cast<T>(0);
    for (int i = tid; i < dim1; i += stride) {
        sum_val += x[base_offset + i * dim2];
    }
    
    // Store to shared memory
    sdata[tid] = sum_val;
    __syncthreads();
    
    // Block-level reduction in shared memory (tree reduction)
    for (unsigned int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    // Warp-level reduction (no sync needed within a warp)
    if (tid < 32) {
        volatile T* smem = sdata;
        if (blockDim.x >= 64) smem[tid] += smem[tid + 32];
        smem[tid] += smem[tid + 16];
        smem[tid] += smem[tid + 8];
        smem[tid] += smem[tid + 4];
        smem[tid] += smem[tid + 2];
        smem[tid] += smem[tid + 1];
    }
    
    // Store result (only thread 0 writes)
    if (tid == 0) {
        y[blockIdx.x] = sdata[0];
    }
}

// Type-dispatched wrapper
torch::Tensor sum_reduce_op(torch::Tensor x, int reduce_dim) {
    TORCH_CHECK(x.is_cuda(), "Input must be a CUDA tensor");
    TORCH_CHECK(reduce_dim == 1, "Only reduction over dim=1 supported in this optimized kernel");
    
    const int batch_size = x.size(0);
    const int dim1 = x.size(1);
    const int dim2 = x.size(2);
    
    // Output shape: (batch_size, 1, dim2)
    std::vector<int64_t> output_size = {batch_size, 1, dim2};
    torch::Tensor y = torch::empty(output_size, x.options());
    
    const int blocks = batch_size * dim2;
    const int threads = 256;
    const int smem_size = threads * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "sum_reduce_cuda", ([&] {
        sum_reduce_kernel<scalar_t><<<blocks, threads, smem_size>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            batch_size, dim1, dim2
        );
    }));
    
    cudaDeviceSynchronize();
    return y;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources='torch::Tensor sum_reduce_op(torch::Tensor x, int reduce_dim);',
    cuda_sources=cuda_source,
    functions=['sum_reduce_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce_op(x, self.dim)

batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    return [torch.randn(batch_size, dim1, dim2, device='cuda')]

def get_init_inputs():
    return [reduce_dim]
