import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <ATen/cuda/CUDAContext.h>

__global__ void scaled_int8_gemm_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    half* __restrict__ C,
    int M,
    int N,
    int K,
    float scale,
    int lda,
    int ldb,
    int ldc) {

    constexpr int BM = 64;
    constexpr int BN = 64;
    constexpr int BK = 32;
    constexpr int TM = 4;
    constexpr int TN = 4;

    __shared__ int8_t As[BM][BK];
    __shared__ int8_t Bs[BK][BN];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int block_m = blockIdx.y * BM;
    int block_n = blockIdx.x * BN;

    int row_base = block_m + ty * TM;
    int col_base = block_n + tx * TN;

    int acc[TM][TN];
    #pragma unroll
    for (int i = 0; i < TM; ++i)
        #pragma unroll
        for (int j = 0; j < TN; ++j)
            acc[i][j] = 0;

    for (int k0 = 0; k0 < K; k0 += BK) {
        int linear = ty * blockDim.x + tx;
        #pragma unroll
        for (int idx = linear; idx < BM * BK; idx += blockDim.x * blockDim.y) {
            int r = idx / BK;
            int c = idx % BK;
            int gm = block_m + r;
            int gk = k0 + c;
            As[r][c] = (gm < M && gk < K) ? A[gm * lda + gk] : 0;
        }
        #pragma unroll
        for (int idx = linear; idx < BK * BN; idx += blockDim.x * blockDim.y) {
            int r = idx / BN;
            int c = idx % BN;
            int gk = k0 + r;
            int gn = block_n + c;
            Bs[r][c] = (gk < K && gn < N) ? B[gk * ldb + gn] : 0;
        }
        __syncthreads();

        #pragma unroll
        for (int kk = 0; kk < BK; ++kk) {
            int a_frag[TM];
            int b_frag[TN];
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                a_frag[i] = static_cast<int>(As[ty * TM + i][kk]);
            }
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                b_frag[j] = static_cast<int>(Bs[kk][tx * TN + j]);
            }
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                #pragma unroll
                for (int j = 0; j < TN; ++j) {
                    acc[i][j] += a_frag[i] * b_frag[j];
                }
            }
        }
        __syncthreads();
    }

    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        int row = row_base + i;
        if (row < M) {
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                int col = col_base + j;
                if (col < N) {
                    C[row * ldc + col] = __float2half_rn(static_cast<float>(acc[i][j]) * scale);
                }
            }
        }
    }
}

torch::Tensor scaled_int8_gemm(torch::Tensor a, torch::Tensor b, torch::Tensor scale_a, torch::Tensor scale_b) {
    TORCH_CHECK(a.is_cuda(), "a must be CUDA");
    TORCH_CHECK(b.is_cuda(), "b must be CUDA");
    TORCH_CHECK(a.scalar_type() == torch::kInt8, "a must be int8");
    TORCH_CHECK(b.scalar_type() == torch::kInt8, "b must be int8");
    TORCH_CHECK(a.dim() == 2 && b.dim() == 2, "a and b must be 2D");
    TORCH_CHECK(a.size(1) == b.size(0), "shape mismatch");
    TORCH_CHECK(a.is_contiguous(), "a must be contiguous");
    TORCH_CHECK(b.is_contiguous(), "b must be contiguous");

    auto M = (int)a.size(0);
    auto K = (int)a.size(1);
    auto N = (int)b.size(1);

    auto c = torch::empty({M, N}, a.options().dtype(torch::kFloat16));
    float scale = scale_a.item<float>() * scale_b.item<float>();

    dim3 block(16, 16);
    dim3 grid((N + 63) / 64, (M + 63) / 64);
    cudaStream_t stream = at::cuda::getDefaultCUDAStream();

    scaled_int8_gemm_kernel<<<grid, block, 0, stream>>>(
        a.data_ptr<int8_t>(),
        b.data_ptr<int8_t>(),
        reinterpret_cast<half*>(c.data_ptr<at::Half>()),
        M,
        N,
        K,
        scale,
        (int)a.stride(0),
        (int)b.stride(0),
        (int)c.stride(0));

    return c;
}
'''

ext = load_inline(
    name='scaled_int8_gemm_ext',
    cpp_sources='torch::Tensor scaled_int8_gemm(torch::Tensor a, torch::Tensor b, torch::Tensor scale_a, torch::Tensor scale_b);',
    cuda_sources=cuda_source,
    functions=['scaled_int8_gemm'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(
        self,
        a_q: torch.Tensor,
        b_q: torch.Tensor,
        scale_a: torch.Tensor,
        scale_b: torch.Tensor,
    ) -> torch.Tensor:
        if not a_q.is_cuda:
            a_q = a_q.cuda()
        if not b_q.is_cuda:
            b_q = b_q.cuda()
        if not scale_a.is_cuda:
            scale_a = scale_a.cuda()
        if not scale_b.is_cuda:
            scale_b = scale_b.cuda()
        return ext.scaled_int8_gemm(a_q.contiguous(), b_q.contiguous(), scale_a, scale_b)


OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-127, 127, (m, k), dtype=torch.int8)
    b_q = torch.randint(-127, 127, (k, n), dtype=torch.int8)
    scale_a = torch.tensor(0.01, dtype=torch.float32)
    scale_b = torch.tensor(0.01, dtype=torch.float32)
    return [a_q, b_q, scale_a, scale_b]


def get_init_inputs():
    return []
