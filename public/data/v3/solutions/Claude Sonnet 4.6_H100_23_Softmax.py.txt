import torch
import torch.nn as nn
import triton
import triton.language as tl


# Online softmax kernel using Triton
# Each program handles one row. We use a single pass (online softmax) for efficiency.
@triton.jit
def softmax_kernel(
    input_ptr,
    output_ptr,
    input_row_stride,
    output_row_stride,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program processes one row
    row_idx = tl.program_id(0)
    row_start_in = row_idx * input_row_stride
    row_start_out = row_idx * output_row_stride

    # ---- Pass 1: compute max for numerical stability ----
    m = tl.full([BLOCK_SIZE], float('-inf'), dtype=tl.float32)
    
    for col_start in range(0, n_cols, BLOCK_SIZE):
        cols = col_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < n_cols
        x = tl.load(input_ptr + row_start_in + cols, mask=mask, other=float('-inf')).to(tl.float32)
        m = tl.maximum(m, x)
    
    row_max = tl.max(m, axis=0)

    # ---- Pass 2: compute sum of exp(x - max) ----
    s = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for col_start in range(0, n_cols, BLOCK_SIZE):
        cols = col_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < n_cols
        x = tl.load(input_ptr + row_start_in + cols, mask=mask, other=float('-inf')).to(tl.float32)
        s += tl.exp(x - row_max) * mask

    row_sum = tl.sum(s, axis=0)

    # ---- Pass 3: compute output ----
    for col_start in range(0, n_cols, BLOCK_SIZE):
        cols = col_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < n_cols
        x = tl.load(input_ptr + row_start_in + cols, mask=mask, other=float('-inf')).to(tl.float32)
        out = tl.exp(x - row_max) / row_sum
        tl.store(output_ptr + row_start_out + cols, out, mask=mask)


# Single-pass online softmax using the Numerically Stable Online Softmax algorithm
# Uses tl.associative_scan for a true single-pass, but the multi-pass approach above
# is simpler. Let's use a truly optimized version with large BLOCK_SIZE.

@triton.jit
def softmax_kernel_v2(
    input_ptr,
    output_ptr,
    input_row_stride,
    output_row_stride,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Optimized kernel: load the entire row in one block if BLOCK_SIZE >= n_cols.
    This is the best case for n_cols=16384.
    """
    row_idx = tl.program_id(0)
    row_start_in = row_idx * input_row_stride
    row_start_out = row_idx * output_row_stride

    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < n_cols

    # Load entire row
    x = tl.load(input_ptr + row_start_in + cols, mask=mask, other=float('-inf')).to(tl.float32)

    # Compute max
    row_max = tl.max(x, axis=0)

    # Compute softmax
    x = x - row_max
    x = tl.exp(x)
    x = x * mask  # zero out padding
    row_sum = tl.sum(x, axis=0)
    x = x / row_sum

    tl.store(output_ptr + row_start_out + cols, x, mask=mask)


def softmax_triton(x: torch.Tensor) -> torch.Tensor:
    assert x.ndim == 2, "Input must be 2D"
    n_rows, n_cols = x.shape

    # Allocate output with float32 to match PyTorch behavior
    y = torch.empty_like(x, dtype=torch.float32)

    # For n_cols=16384, use BLOCK_SIZE=16384 for single-pass
    # triton requires BLOCK_SIZE to be power of 2
    BLOCK_SIZE = triton.next_power_of_2(n_cols)

    # If BLOCK_SIZE is too large for SRAM, split into smaller blocks
    # H100 has 228KB SMEM per SM; float32 * 16384 = 64KB which fits fine
    if BLOCK_SIZE <= 65536:
        softmax_kernel_v2[(n_rows,)](
            x,
            y,
            x.stride(0),
            y.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=max(1, min(32, BLOCK_SIZE // 256)),
        )
    else:
        # Fall back to multi-pass kernel for very large rows
        BLOCK_SIZE = 4096
        softmax_kernel[(n_rows,)](
            x,
            y,
            x.stride(0),
            y.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=16,
        )

    return y


class Model(nn.Module):
    """
    Optimized softmax using Triton kernel.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return softmax_triton(x)
