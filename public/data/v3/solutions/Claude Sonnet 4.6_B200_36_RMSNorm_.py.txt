import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// RMSNorm: normalize along dim=1 (feature dimension)
// Input shape: (batch, features, spatial...)
// Each "row" in the reduction is across features for a given (batch, spatial) position
// We think of it as: N = batch * spatial_elements, C = features
// For each of the N positions, compute rms over C features, then divide

// Strategy: each block handles one spatial position
// threads in block each handle a subset of features, then do block reduction
// For C=64, we can use 64 threads, one per feature

template <typename scalar_t, typename accum_t>
__global__ void rmsnorm_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int C,       // number of features
    const int spatial, // product of spatial dims (dim1 * dim2 * ...)
    const float eps
) {
    // blockIdx.x = batch index
    // blockIdx.y = spatial position index
    const int b = blockIdx.x;
    const int s = blockIdx.y;
    const int tid = threadIdx.x;

    // Base pointer for this (batch, spatial) position
    // input layout: [batch, C, spatial] (channels-first)
    // input[b, c, s] = input[b * C * spatial + c * spatial + s]

    const int base_in = b * C * spatial;

    // Each thread loads one feature value
    accum_t val = (tid < C) ? (accum_t)input[base_in + tid * spatial + s] : (accum_t)0;

    // Compute sum of squares using warp shuffle reduction
    accum_t sum_sq = val * val;

    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset >= 1; offset >>= 1) {
        sum_sq += __shfl_down_sync(0xffffffff, sum_sq, offset);
    }

    // If C > 32, we need to handle multiple warps
    __shared__ accum_t shared_sum[32]; // max 32 warps
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    const int num_warps = (blockDim.x + 31) / 32;

    if (lane_id == 0) {
        shared_sum[warp_id] = sum_sq;
    }
    __syncthreads();

    // Final reduction across warps (done by first warp)
    accum_t total_sum_sq = (accum_t)0;
    if (warp_id == 0) {
        accum_t warp_val = (lane_id < num_warps) ? shared_sum[lane_id] : (accum_t)0;
        #pragma unroll
        for (int offset = 16; offset >= 1; offset >>= 1) {
            warp_val += __shfl_down_sync(0xffffffff, warp_val, offset);
        }
        if (lane_id == 0) {
            shared_sum[0] = warp_val;
        }
    }
    __syncthreads();
    total_sum_sq = shared_sum[0];

    // Compute RMS inverse
    accum_t rms_inv = (accum_t)rsqrtf((float)(total_sum_sq / (accum_t)C + (accum_t)eps));

    // Write normalized output
    if (tid < C) {
        output[base_in + tid * spatial + s] = (scalar_t)(val * rms_inv);
    }
}

// Vectorized version for float with vec4
// For large spatial dims, we process multiple spatial elements per block
template <typename scalar_t, typename accum_t, int BLOCK_C>
__global__ void rmsnorm_kernel_vec(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int C,
    const int spatial,
    const float eps,
    const int spatial_per_block  // number of spatial elements per block
) {
    const int b = blockIdx.x;
    const int s_start = blockIdx.y * spatial_per_block;
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;

    // Each "work item" within the block handles one spatial element
    // tid maps to: spatial_offset = tid / BLOCK_C, feature = tid % BLOCK_C
    // But we need to process spatial_per_block spatial elements
    // Layout: blockDim.x = BLOCK_C threads handle feature reduction
    // We iterate over spatial elements

    const int base_in = b * C * spatial;

    __shared__ accum_t shared_sum[32];

    for (int s_off = 0; s_off < spatial_per_block; s_off++) {
        int s = s_start + s_off;
        if (s >= spatial) break;

        // Load feature value for this thread
        accum_t val = (tid < C) ? (accum_t)input[base_in + tid * spatial + s] : (accum_t)0;

        // Compute sum of squares
        accum_t sum_sq = val * val;

        // Warp reduction
        #pragma unroll
        for (int offset = 16; offset >= 1; offset >>= 1) {
            sum_sq += __shfl_down_sync(0xffffffff, sum_sq, offset);
        }

        const int num_warps = (blockDim.x + 31) / 32;
        if (lane_id == 0) {
            shared_sum[warp_id] = sum_sq;
        }
        __syncthreads();

        accum_t total_sum_sq = (accum_t)0;
        if (warp_id == 0) {
            accum_t warp_val = (lane_id < num_warps) ? shared_sum[lane_id] : (accum_t)0;
            #pragma unroll
            for (int offset = 16; offset >= 1; offset >>= 1) {
                warp_val += __shfl_down_sync(0xffffffff, warp_val, offset);
            }
            if (lane_id == 0) {
                shared_sum[0] = warp_val;
            }
        }
        __syncthreads();
        total_sum_sq = shared_sum[0];

        accum_t rms_inv = (accum_t)rsqrtf((float)(total_sum_sq / (accum_t)C + (accum_t)eps));

        if (tid < C) {
            output[base_in + tid * spatial + s] = (scalar_t)(val * rms_inv);
        }
        __syncthreads();
    }
}

torch::Tensor rmsnorm_cuda(torch::Tensor input, int num_features, float eps) {
    auto output = torch::empty_like(input);
    
    const int batch = input.size(0);
    const int C = input.size(1);
    
    // Compute spatial size
    int spatial = 1;
    for (int i = 2; i < input.dim(); i++) {
        spatial *= input.size(i);
    }
    
    // Choose block size: must be >= C, power of 2
    int block_size = 32;
    while (block_size < C) block_size *= 2;
    block_size = min(block_size, 1024);

    // For C=64, use 64 threads, process multiple spatial per block to improve occupancy
    // spatial = 256*256 = 65536
    // We batch spatial elements per block to reduce kernel launch overhead
    int spatial_per_block = 1;
    // If block_size >> C, we could pack more, but let's keep it simple
    // Pack multiple spatial per block for better memory coalescing
    // Actually, for contiguous memory, each warp should handle multiple spatial
    
    // For the case where C fits in one warp (C <= 32), we can pack spatial_per_block > 1
    if (C <= 32) {
        spatial_per_block = 32 / C; // multiple spatial per warp
        block_size = 32; // one warp
    }
    
    // Actually let's use a different approach:
    // For C=64, we have 64 threads doing reduction over 64 features
    // Use grid: (batch, spatial)
    
    dim3 grid(batch, spatial);
    dim3 block(block_size);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16,
        input.scalar_type(), "rmsnorm_cuda", [&] {
            using accum_t = float; // always accumulate in float
            rmsnorm_kernel<scalar_t, accum_t><<<grid, block>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                C,
                spatial,
                eps
            );
        });
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor rmsnorm_cuda(torch::Tensor input, int num_features, float eps);
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rmsnorm_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math', '-arch=sm_100'],
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_cuda(x, self.num_features, self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [features]
