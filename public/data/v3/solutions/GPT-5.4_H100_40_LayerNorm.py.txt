import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "layernorm"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

# One program handles one sample row of length N = features*dim1*dim2.
# For this benchmark N=64*256*256=4,194,304, so a bandwidth-oriented fused kernel
# avoids PyTorch LayerNorm overhead and reads the row three times total.

@triton.jit
def _layernorm_stats_kernel(
    x_ptr, mean_ptr, rstd_ptr,
    N,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    row_start = row * N

    acc = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    for start in range(0, N, BLOCK_SIZE):
        idx = start + offs
        mask = idx < N
        x = tl.load(x_ptr + row_start + idx, mask=mask, other=0.0).to(tl.float32)
        acc += x
    mean = tl.sum(acc, axis=0) / N

    acc2 = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    for start in range(0, N, BLOCK_SIZE):
        idx = start + offs
        mask = idx < N
        x = tl.load(x_ptr + row_start + idx, mask=mask, other=0.0).to(tl.float32)
        d = x - mean
        acc2 += d * d
    var = tl.sum(acc2, axis=0) / N
    rstd = tl.rsqrt(var + 1e-5)

    tl.store(mean_ptr + row, mean)
    tl.store(rstd_ptr + row, rstd)


@triton.jit
def _layernorm_apply_kernel(
    x_ptr, y_ptr, weight_ptr, bias_ptr, mean_ptr, rstd_ptr,
    N,
    HAS_WEIGHT: tl.constexpr,
    HAS_BIAS: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    row_start = row * N
    mean = tl.load(mean_ptr + row)
    rstd = tl.load(rstd_ptr + row)

    for start in range(0, N, BLOCK_SIZE):
        idx = start + offs
        mask = idx < N
        x = tl.load(x_ptr + row_start + idx, mask=mask, other=0.0).to(tl.float32)
        y = (x - mean) * rstd
        if HAS_WEIGHT:
            w = tl.load(weight_ptr + idx, mask=mask, other=1.0).to(tl.float32)
            y *= w
        if HAS_BIAS:
            b = tl.load(bias_ptr + idx, mask=mask, other=0.0).to(tl.float32)
            y += b
        tl.store(y_ptr + row_start + idx, y, mask=mask)


class FastLayerNorm(nn.Module):
    def __init__(self, normalized_shape, eps=1e-5, elementwise_affine=True, bias=True, device=None, dtype=None):
        super().__init__()
        self.normalized_shape = tuple(normalized_shape) if isinstance(normalized_shape, (tuple, list)) else (normalized_shape,)
        self.eps = eps
        if elementwise_affine:
            self.weight = nn.Parameter(torch.ones(self.normalized_shape, device=device, dtype=dtype))
            self.bias = nn.Parameter(torch.zeros(self.normalized_shape, device=device, dtype=dtype)) if bias else None
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_cuda
        # Benchmark/reference uses shape [16,64,256,256] and normalizes over last 3 dims,
        # so flatten outer dims into rows.
        N = 1
        for d in self.normalized_shape:
            N *= d
        x_contig = x.contiguous()
        rows = x_contig.numel() // N
        x2d = x_contig.view(rows, N)
        y = torch.empty_like(x2d)
        mean = torch.empty((rows,), device=x.device, dtype=torch.float32)
        rstd = torch.empty((rows,), device=x.device, dtype=torch.float32)

        BLOCK = 1024
        _layernorm_stats_kernel[(rows,)](x2d, mean, rstd, N, BLOCK_SIZE=BLOCK)
        _layernorm_apply_kernel[(rows,)](
            x2d, y,
            self.weight.view(-1) if self.weight is not None else x2d,
            self.bias.view(-1) if self.bias is not None else x2d,
            mean, rstd, N,
            HAS_WEIGHT=self.weight is not None,
            HAS_BIAS=self.bias is not None,
            BLOCK_SIZE=BLOCK,
        )
        return y.view_as(x_contig)


class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = FastLayerNorm(normalized_shape=normalized_shape)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.ln(x)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
