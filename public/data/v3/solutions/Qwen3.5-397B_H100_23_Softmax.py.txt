import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def softmax_kernel(
    X_ptr,
    Y_ptr,
    stride_x_row,
    stride_y_row,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Triton softmax kernel optimized for H100.
    Each program handles one row of the input.
    """
    row_idx = tl.program_id(0)
    
    # Pointers to the start of the row
    X_row_ptr = X_ptr + row_idx * stride_x_row
    Y_row_ptr = Y_ptr + row_idx * stride_y_row
    
    # Compute row max for numerical stability
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols
    
    # Load the row
    x_row = tl.load(X_row_ptr + col_offsets, mask=mask, other=-float('inf'))
    
    # Compute max
    row_max = tl.max(x_row, axis=0)
    
    # Compute exp(x - max) for numerical stability
    x_exp = tl.exp(x_row - row_max)
    
    # Compute sum of exp
    row_sum = tl.sum(x_exp, axis=0)
    
    # Normalize
    y_row = x_exp / row_sum
    
    # Store result
    tl.store(Y_row_ptr + col_offsets, y_row, mask=mask)


class SoftmaxFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, dim):
        ctx.dim = dim
        ctx.save_for_backward(x)
        
        # Ensure contiguous
        x = x.contiguous()
        
        # Move dim to last if needed, but for dim=1 with 2D tensor it's fine
        if dim != 1 or x.dim() != 2:
            # Permute to make softmax dim the last dimension
            x = x.transpose(dim, -1).contiguous()
            ctx.needs_transpose = True
        else:
            ctx.needs_transpose = False
        
        batch_size, n_cols = x.shape
        
        # Choose block size - power of 2, at least 1024 for H100
        BLOCK_SIZE = triton.next_power_of_2(n_cols)
        if BLOCK_SIZE > 65536:
            BLOCK_SIZE = 65536  # Max reasonable block size
        
        y = torch.empty_like(x)
        
        # Launch kernel - one program per row
        grid = (batch_size,)
        softmax_kernel[grid](
            x, y,
            x.stride(0), y.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        if ctx.needs_transpose:
            y = y.transpose(-1, dim).contiguous()
        
        return y
    
    @staticmethod
    def backward(ctx, grad_output):
        x = ctx.saved_tensors[0]
        
        if ctx.needs_transpose:
            x = x.transpose(ctx.dim, -1).contiguous()
            grad_output = grad_output.transpose(ctx.dim, -1).contiguous()
        
        batch_size, n_cols = x.shape
        
        BLOCK_SIZE = triton.next_power_of_2(n_cols)
        if BLOCK_SIZE > 65536:
            BLOCK_SIZE = 65536
        
        # Recompute softmax for backward
        grad_input = torch.empty_like(x)
        
        grid = (batch_size,)
        softmax_backward_kernel[grid](
            x, grad_output, grad_input,
            x.stride(0), grad_output.stride(0), grad_input.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        if ctx.needs_transpose:
            grad_input = grad_input.transpose(-1, ctx.dim).contiguous()
        
        return grad_input, None


@triton.jit
def softmax_backward_kernel(
    X_ptr,
    GradOutput_ptr,
    GradInput_ptr,
    stride_x_row,
    stride_go_row,
    stride_gi_row,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Backward pass for softmax: grad_input = softmax_output * (grad_output - sum(softmax_output * grad_output))
    """
    row_idx = tl.program_id(0)
    
    X_row_ptr = X_ptr + row_idx * stride_x_row
    GO_row_ptr = GradOutput_ptr + row_idx * stride_go_row
    GI_row_ptr = GradInput_ptr + row_idx * stride_gi_row
    
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols
    
    # Load row and compute softmax
    x_row = tl.load(X_row_ptr + col_offsets, mask=mask, other=-float('inf'))
    row_max = tl.max(x_row, axis=0)
    x_exp = tl.exp(x_row - row_max)
    row_sum = tl.sum(x_exp, axis=0)
    softmax_out = x_exp / row_sum
    
    # Load grad_output
    go_row = tl.load(GO_row_ptr + col_offsets, mask=mask, other=0.0)
    
    # Compute sum(softmax * grad_output)
    sum_s_g = tl.sum(softmax_out * go_row, axis=0)
    
    # grad_input = softmax * (grad_output - sum_s_g)
    gi_row = softmax_out * (go_row - sum_s_g)
    
    tl.store(GI_row_ptr + col_offsets, gi_row, mask=mask)


class Model(nn.Module):
    """
    Optimized Softmax model using Triton kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return SoftmaxFunction.apply(x, 1)
