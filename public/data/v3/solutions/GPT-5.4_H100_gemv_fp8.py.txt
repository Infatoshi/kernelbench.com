import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _gemv_int8_scales_kernel(
    x_ptr,
    w_ptr,
    y_ptr,
    x_scale_ptr,
    w_scale_ptr,
    stride_xb,
    stride_wm,
    stride_wk,
    stride_yb,
    K,
    M,
    BLOCK_M: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_m = tl.program_id(0)
    pid_b = tl.program_id(1)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < M

    acc = tl.zeros((BLOCK_M,), dtype=tl.int32)

    for k0 in range(0, K, BLOCK_K):
        offs_k = k0 + tl.arange(0, BLOCK_K)
        mask_k = offs_k < K

        x = tl.load(x_ptr + pid_b * stride_xb + offs_k, mask=mask_k, other=0).to(tl.int32)
        w = tl.load(w_ptr + offs_m[:, None] * stride_wm + offs_k[None, :] * stride_wk,
                    mask=mask_m[:, None] & mask_k[None, :], other=0).to(tl.int32)
        acc += tl.sum(w * x[None, :], axis=1)

    scale = tl.load(x_scale_ptr) * tl.load(w_scale_ptr)
    out = acc.to(tl.float32) * scale
    tl.store(y_ptr + pid_b * stride_yb + offs_m, out, mask=mask_m)


def gemv_int8_scaled(x_q: torch.Tensor, weight_q: torch.Tensor, x_scale: torch.Tensor, weight_scale: torch.Tensor) -> torch.Tensor:
    assert x_q.is_cuda and weight_q.is_cuda
    assert x_q.dtype == torch.int8 and weight_q.dtype == torch.int8
    B, K = x_q.shape
    M, K2 = weight_q.shape
    assert K == K2
    y = torch.empty((B, M), device=x_q.device, dtype=torch.float32)
    grid = (triton.cdiv(M, 128), B)
    _gemv_int8_scales_kernel[grid](
        x_q,
        weight_q,
        y,
        x_scale,
        weight_scale,
        x_q.stride(0),
        weight_q.stride(0),
        weight_q.stride(1),
        y.stride(0),
        K,
        M,
        BLOCK_M=128,
        BLOCK_K=256,
        num_warps=8,
        num_stages=4,
    )
    return y


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return gemv_int8_scaled(x_q, self.weight_q, x_scale, self.weight_scale).to(torch.float16)
