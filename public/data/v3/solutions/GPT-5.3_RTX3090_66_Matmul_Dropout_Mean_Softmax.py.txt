import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void mean_kernel(const float* __restrict__ x, float* __restrict__ mean, int B, int F){
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if(j>=F) return;
    float s = 0.f;
    for(int b=0;b<B;++b) s += x[b*F + j];
    mean[j] = s / B;
}

__global__ void gemv_kernel(const float* __restrict__ mean, const float* __restrict__ W, const float* __restrict__ b, float* __restrict__ y, int IN, int OUT){
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if(j>=OUT) return;
    const float* wrow = W + j*IN;
    float acc = 0.f;
    for(int k=0;k<IN;++k) acc += mean[k]*wrow[k];
    y[j] = acc + b[j];
}

__global__ void softmax_kernel(float* y, int N){
    __shared__ float smax;
    __shared__ float ssum;

    float v = -1e30f;
    int tid = threadIdx.x;
    for(int i=tid;i<N;i+=blockDim.x) v = fmaxf(v, y[i]);

    __shared__ float buf[4096];
    buf[tid] = v;
    __syncthreads();

    for(int stride=blockDim.x/2; stride>0; stride/=2){
        if(tid<stride) buf[tid] = fmaxf(buf[tid], buf[tid+stride]);
        __syncthreads();
    }
    if(tid==0) smax = buf[0];
    __syncthreads();

    float sum = 0.f;
    for(int i=tid;i<N;i+=blockDim.x){
        float e = expf(y[i]-smax);
        y[i] = e;
        sum += e;
    }
    buf[tid] = sum;
    __syncthreads();

    for(int stride=blockDim.x/2; stride>0; stride/=2){
        if(tid<stride) buf[tid] += buf[tid+stride];
        __syncthreads();
    }
    if(tid==0) ssum = buf[0];
    __syncthreads();

    for(int i=tid;i<N;i+=blockDim.x) y[i] /= ssum;
}


torch::Tensor fused_forward(torch::Tensor x, torch::Tensor W, torch::Tensor b){
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = W.size(0);

    auto opts = x.options();
    auto mean = torch::empty({IN}, opts);
    auto y = torch::empty({OUT}, opts);

    mean_kernel<<<(IN+255)/256,256>>>(x.data_ptr<float>(), mean.data_ptr<float>(), B, IN);
    gemv_kernel<<<(OUT+255)/256,256>>>(mean.data_ptr<float>(), W.data_ptr<float>(), b.data_ptr<float>(), y.data_ptr<float>(), IN, OUT);
    softmax_kernel<<<1,1024>>>(y.data_ptr<float>(), OUT);

    return y.view({1, OUT});
}
'''

ext = load_inline(
    name='fused_mean_gemv_softmax',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor W, torch::Tensor b);',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        if x.dtype != torch.float32:
            x = x.float()
        W = self.matmul.weight
        b = self.matmul.bias
        y = ext.fused_forward(x.contiguous(), W.contiguous(), b.contiguous())
        return y

batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features, device='cuda', dtype=torch.float32)]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
