"""
Optimized FP16 GEMV for H100 — batched matrix-vector multiply.
Input: [B, K], Weight: [N, K], Output: [B, N]
i.e. out = x @ W.T  where W is [N, K]

Strategy:
- One warp (32 threads) handles one (batch, output-row) pair
- Each thread loads K/32 elements using float4 (8 fp16 values) vectorized loads
- Warp-level reduction via __shfl_down_sync
- For large N and B, we tile with multiple warps per block
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Warp-level dot product for FP16 vectors
// Each warp computes one output element
// gridDim.x = batch_size, gridDim.y = ceil(N / WARPS_PER_BLOCK)
// blockDim = (32 * WARPS_PER_BLOCK, 1, 1)

#define WARPS_PER_BLOCK 8

__global__ void gemv_fp16_kernel(
    const __half* __restrict__ x,      // [B, K]
    const __half* __restrict__ w,      // [N, K]
    __half* __restrict__ out,          // [B, N]
    int B, int N, int K
) {
    // warp index within block
    int warp_id = threadIdx.x >> 5;        // threadIdx.x / 32
    int lane    = threadIdx.x & 31;        // threadIdx.x % 32

    int batch_idx = blockIdx.x;
    int out_row   = blockIdx.y * WARPS_PER_BLOCK + warp_id;

    if (batch_idx >= B || out_row >= N) return;

    const __half* x_row = x + batch_idx * K;
    const __half* w_row = w + out_row   * K;

    // Use float2 loads (4 bytes = 2 fp16) for vectorized access
    // We'll accumulate in float for precision
    float acc = 0.0f;

    // Each lane processes K/32 elements, stepping by 32
    // Vectorized: process 8 fp16 at a time using float4
    int vec_K = K / 8;   // number of float4 chunks (8 fp16 per float4)

    const float4* x_vec = reinterpret_cast<const float4*>(x_row);
    const float4* w_vec = reinterpret_cast<const float4*>(w_row);

    for (int i = lane; i < vec_K; i += 32) {
        float4 xv = x_vec[i];
        float4 wv = w_vec[i];

        // Each float4 holds 8 fp16 values
        __half2* xh = reinterpret_cast<__half2*>(&xv);
        __half2* wh = reinterpret_cast<__half2*>(&wv);

        // 4 pairs of half2
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            float2 xf = __half22float2(xh[j]);
            float2 wf = __half22float2(wh[j]);
            acc += xf.x * wf.x + xf.y * wf.y;
        }
    }

    // Handle remainder (when K % 8 != 0)
    int rem_start = vec_K * 8;
    for (int i = rem_start + lane; i < K; i += 32) {
        acc += __half2float(x_row[i]) * __half2float(w_row[i]);
    }

    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        acc += __shfl_down_sync(0xFFFFFFFF, acc, offset);
    }

    if (lane == 0) {
        out[batch_idx * N + out_row] = __float2half(acc);
    }
}


// Alternative: use __half2 dot products natively for speed
__global__ void gemv_fp16_h2_kernel(
    const __half* __restrict__ x,      // [B, K]
    const __half* __restrict__ w,      // [N, K]
    __half* __restrict__ out,          // [B, N]
    int B, int N, int K
) {
    int warp_id = threadIdx.x >> 5;
    int lane    = threadIdx.x & 31;

    int batch_idx = blockIdx.x;
    int out_row   = blockIdx.y * WARPS_PER_BLOCK + warp_id;

    if (batch_idx >= B || out_row >= N) return;

    const __half* x_row = x + batch_idx * K;
    const __half* w_row = w + out_row   * K;

    float acc = 0.0f;

    // Process 8 fp16 at once (float4 = 128 bits)
    int vec_K = K >> 3;  // K / 8

    const float4* x_vec = reinterpret_cast<const float4*>(x_row);
    const float4* w_vec = reinterpret_cast<const float4*>(w_row);

    for (int i = lane; i < vec_K; i += 32) {
        float4 xv = __ldg(x_vec + i);
        float4 wv = __ldg(w_vec + i);

        __half2 xh[4], wh[4];
        // Unpack float4 -> 4x half2
        *reinterpret_cast<float4*>(xh) = xv;
        *reinterpret_cast<float4*>(wh) = wv;

        #pragma unroll
        for (int j = 0; j < 4; j++) {
            // __half2 dot via __half2 multiply-add
            __half2 prod = __hmul2(xh[j], wh[j]);
            acc += __half2float(prod.x) + __half2float(prod.y);
        }
    }

    // Remainder
    int rem_start = vec_K << 3;
    for (int i = rem_start + lane; i < K; i += 32) {
        acc += __half2float(__hmul(x_row[i], w_row[i]));
    }

    // Warp reduce
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        acc += __shfl_down_sync(0xFFFFFFFF, acc, offset);
    }

    if (lane == 0) {
        out[batch_idx * N + out_row] = __float2half(acc);
    }
}


torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w) {
    // x: [B, K], w: [N, K]
    TORCH_CHECK(x.dtype() == torch::kFloat16, "x must be float16");
    TORCH_CHECK(w.dtype() == torch::kFloat16, "w must be float16");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");
    TORCH_CHECK(w.is_contiguous(), "w must be contiguous");

    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);
    TORCH_CHECK(w.size(1) == K, "weight K dimension mismatch");

    auto out = torch::empty({B, N}, x.options());

    // Grid: (B, ceil(N / WARPS_PER_BLOCK))
    dim3 grid(B, (N + WARPS_PER_BLOCK - 1) / WARPS_PER_BLOCK);
    dim3 block(32 * WARPS_PER_BLOCK);

    gemv_fp16_h2_kernel<<<grid, block>>>(
        reinterpret_cast<const __half*>(x.data_ptr<at::Half>()),
        reinterpret_cast<const __half*>(w.data_ptr<at::Half>()),
        reinterpret_cast<__half*>(out.data_ptr<at::Half>()),
        B, N, K
    );

    return out;
}
'''

cpp_source = '''
torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w);
'''

ext = load_inline(
    name='gemv_fp16_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp16'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math',
                       '-Xptxas', '-v'],
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(
            torch.randn(out_features, in_features, dtype=torch.float16) * 0.02
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        # Ensure weight is contiguous
        w = self.weight.contiguous()
        return ext.gemv_fp16(x_fp16.contiguous(), w)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
