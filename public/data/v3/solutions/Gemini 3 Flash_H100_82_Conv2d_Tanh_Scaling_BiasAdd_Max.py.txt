import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_conv_pool_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ conv_bias,
    const float* __restrict__ bias_add,
    float* __restrict__ output,
    int N, int C_in, int C_out, int H_in, int W_in,
    int H_out, int W_out, int K,
    int H_pool, int W_pool, int pool_size,
    float scaling_factor)
{
    int ow_p = blockIdx.x * blockDim.x + threadIdx.x;
    int oh_p = blockIdx.y * blockDim.y + threadIdx.y;
    int n_oc = blockIdx.z;
    int n = n_oc / C_out;
    int oc = n_oc % C_out;

    if (oh_p >= H_pool || ow_p >= W_pool || n >= N) return;

    float local_w[27];
    if (C_in == 3 && K == 3) {
        #pragma unroll
        for (int i = 0; i < 27; ++i) local_w[i] = weight[oc * 27 + i];
    }

    float b_c = conv_bias[oc];
    float b_a = bias_add[oc];

    float max_val = -1e38f;
    for (int ph = 0; ph < pool_size; ++ph) {
        for (int pw = 0; pw < pool_size; ++pw) {
            int oh = oh_p * pool_size + ph;
            int ow = ow_p * pool_size + pw;
            
            float sum = b_c;
            if (C_in == 3 && K == 3) {
                #pragma unroll
                for (int ic = 0; ic < 3; ++ic) {
                    const float* in_ptr = &input[((n * 3 + ic) * H_in + oh) * W_in + ow];
                    sum += in_ptr[0] * local_w[ic * 9 + 0];
                    sum += in_ptr[1] * local_w[ic * 9 + 1];
                    sum += in_ptr[2] * local_w[ic * 9 + 2];
                    sum += in_ptr[W_in] * local_w[ic * 9 + 3];
                    sum += in_ptr[W_in + 1] * local_w[ic * 9 + 4];
                    sum += in_ptr[W_in + 2] * local_w[ic * 9 + 5];
                    sum += in_ptr[2 * W_in] * local_w[ic * 9 + 6];
                    sum += in_ptr[2 * W_in + 1] * local_w[ic * 9 + 7];
                    sum += in_ptr[2 * W_in + 2] * local_w[ic * 9 + 8];
                }
            } else {
                for (int ic = 0; ic < C_in; ++ic) {
                    for (int kh = 0; kh < K; ++kh) {
                        for (int kw = 0; kw < K; ++kw) {
                            sum += input[((n * C_in + ic) * H_in + (oh + kh)) * W_in + (ow + kw)] * weight[((oc * C_in + ic) * K + kh) * K + kw];
                        }
                    }
                }
            }
            float res = tanhf(sum) * scaling_factor + b_a;
            if (res > max_val) max_val = res;
        }
    }
    output[((n * C_out + oc) * H_pool + oh_p) * W_pool + ow_p] = max_val;
}

torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, torch::Tensor conv_bias, 
                      float scaling_factor, torch::Tensor bias_add, int pool_size) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H_in = input.size(2);
    int W_in = input.size(3);
    int C_out = weight.size(0);
    int K = weight.size(2);
    
    int H_out = H_in - K + 1;
    int W_out = W_in - K + 1;
    int H_pool = H_out / pool_size;
    int W_pool = W_out / pool_size;
    
    auto output = torch::empty({N, C_out, H_pool, W_pool}, input.options());
    
    dim3 block(32, 8);
    dim3 grid((W_pool + 31) / 32, (H_pool + 7) / 8, N * C_out);
    
    fused_conv_pool_kernel<<<grid, block>>>(
        input.data_ptr<float>(), weight.data_ptr<float>(), conv_bias.data_ptr<float>(),
        bias_add.data_ptr<float>(), output.data_ptr<float>(),
        N, C_in, C_out, H_in, W_in, H_out, W_out, K, H_pool, W_pool, pool_size, scaling_factor
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_ext_v3',
    cpp_sources='torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, torch::Tensor conv_bias, float scaling_factor, torch::Tensor bias_add, int pool_size);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        return ext.fused_op(x, self.conv.weight, self.conv.bias, float(self.scaling_factor), self.bias, int(self.pool_kernel_size))

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0, (16, 1, 1), 2]
