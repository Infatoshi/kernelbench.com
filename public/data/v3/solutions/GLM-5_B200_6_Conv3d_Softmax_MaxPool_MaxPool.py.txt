    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,

    const int channels,
    const int depth,
    const int height,
    const int width,
    const int out_depth,
    const int out_height,
    const int out_width
) {
    // Each thread block handles one (batch, spatial) position
    // Threads within block cooperate on channel dimension
    
    const int n = blockIdx.x;
    const int out_d = blockIdx.y;
    const int out_h = blockIdx.z / ((out_height * out_width - 1) / gridDim.z + 1);

    const int rem = spatial_idx % (out_height * out_width);

    const int ow = rem % out_width;
    
    const int tid = threadIdx.x;
    const int block_dim = blockDim.x;

    const int in_d_start = od * 4;
    const int in_h_start = oh * 4;
    const int in_w_start = ow * 4;

    __shared__ scalar_t shared_sum[256];

    scalar_t final_max = -INFINITY;
    
    // Iterate over spatial positions in 4x4x4 region

                const int in_d = in_d_start + pd;
                const int in_h = in_h_start + ph;
                const int in_w = in_w_start + pw;

                // Parallel max reduction across channels

                                  + in_d * height * width 
                                  + in_h * width + in_w;

                }
                shared_sum[tid] = local_sum;
                __syncthreads();

                    if (tid < s) {
                        shared_sum[tid] += shared_sum[tid + s];
                    }
                    __syncthreads();

// Correct kernel: each block handles one (batch, channel) pair
// Grid: (batch * channels) x (output spatial positions)
template<typename scalar_t>
__global__ void fused_softmax_double_maxpool_v2(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int depth,
    const int height,
    const int width,
    const int out_depth,
    const int out_height,
    const int out_width
) {
    // blockIdx.x = batch * channels combined
    // blockIdx.y = output spatial position
    
    const int bc = blockIdx.x;
    const int n = bc / channels;
    const int c = bc % channels;
    
    const int spatial_idx = blockIdx.y;
    const int od = spatial_idx / (out_height * out_width);
    const int rem = spatial_idx % (out_height * out_width);
    const int oh = rem / out_width;
    const int ow = rem % out_width;
    
    const int tid = threadIdx.x;
    const int block_dim = blockDim.x;
    
    // Input starting position (4x4x4 region for double 2x2x2 pool)
    const int in_d_start = od * 4;
    const int in_h_start = oh * 4;
    const int in_w_start = ow * 4;
    
    // Shared memory for reduction across channels (for softmax)
    __shared__ scalar_t shared_data[256];
    
    // For each spatial position in 4x4x4 region, compute softmax value and track max
    scalar_t max_pool_val = -INFINITY;
    
    // Iterate over spatial positions in 4x4x4 region
    for (int pd = 0; pd < 4; pd++) {
        for (int ph = 0; ph < 4; ph++) {
            for (int pw = 0; pw < 4; pw++) {
                const int in_d = in_d_start + pd;
                const int in_h = in_h_start + ph;
                const int in_w = in_w_start + pw;
                
                // Bounds check
                if (in_d >= depth || in_h >= height || in_w >= width) continue;
                
                // Compute softmax for this position
                // Need max and sum across all channels
                
                // Parallel max across channels
                scalar_t local_max = -INFINITY;
                for (int ch = tid; ch < channels; ch += block_dim) {
                    const int idx = n * channels * depth * height * width 
                                  + ch * depth * height * width 
                                  + in_d * height * width 
                                  + in_h * width + in_w;
                    local_max = max(local_max, input[idx]);
                }
                shared_data[tid] = local_max;
                __syncthreads();
                
                for (int s = block_dim / 2; s > 0; s >>= 1) {
                    if (tid < s) {
                        shared_data[tid] = max(shared_data[tid], shared_data[tid + s]);
                    }
                    __syncthreads();
                }
                const scalar_t max_val = shared_data[0];
                
                // Parallel sum of exp
                scalar_t local_sum = 0;
                for (int ch = tid; ch < channels; ch += block_dim) {
                    const int idx = n * channels * depth * height * width 
                                  + ch * depth * height * width 
                                  + in_d * height * width 
                                  + in_h * width + in_w;
                    local_sum += expf(input[idx] - max_val);
                }
                shared_data[tid] = local_sum;
                __syncthreads();
                
                for (int s = block_dim / 2; s > 0; s >>= 1) {
                    if (tid < s) {
                        shared_data[tid] += shared_data[tid + s];
                    }
                    __syncthreads();
                }
                const scalar_t sum_val = shared_data[0];
                
                // Compute softmax value for this channel at this spatial position
                const int idx = n * channels * depth * height * width 
                              + c * depth * height * width 
                              + in_d * height * width 
                              + in_h * width + in_w;
                const scalar_t softmax_val = expf(input[idx] - max_val) / sum_val;

                           + od * out_height * out_width 
                           + oh * out_width + ow;
        output[out_idx] = max_pool_val;
    }

// Optimized version using warp-level primitives
template<typename scalar_t>
__global__ void fused_softmax_double_maxpool_optimized(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int depth,
    const int height,
    const int width,
    const int out_depth,
    const int out_height,
    const int out_width
) {
    const int bc = blockIdx.x;
    const int n = bc / channels;
    const int c = bc % channels;
    
    const int spatial_idx = blockIdx.y;
    const int od = spatial_idx / (out_height * out_width);
    const int rem = spatial_idx % (out_height * out_width);
    const int oh = rem / out_width;
    const int ow = rem % out_width;
    
    const int tid = threadIdx.x;
    const int lane_id = tid & 31;
    const int warp_id = tid >> 5;
    const int num_warps = blockDim.x >> 5;
    
    const int in_d_start = od * 4;
    const int in_h_start = oh * 4;
    const int in_w_start = ow * 4;
    
    scalar_t max_pool_val = -INFINITY;
    
    // Iterate over 4x4x4 spatial region
    for (int pd = 0; pd < 4; pd++) {
        for (int ph = 0; ph < 4; ph++) {
            for (int pw = 0; pw < 4; pw++) {
                const int in_d = in_d_start + pd;
                const int in_h = in_h_start + ph;
                const int in_w = in_w_start + pw;
                
                if (in_d >= depth || in_h >= height || in_w >= width) continue;
                
                // Warp-level max reduction across channels
                scalar_t local_max = -INFINITY;
                for (int ch = tid; ch < channels; ch += blockDim.x) {
                    const int idx = n * channels * depth * height * width 
                                  + ch * depth * height * width 
                                  + in_d * height * width 
                                  + in_h * width + in_w;
                    local_max = max(local_max, input[idx]);
                }
                
                // Warp shuffle reduction
                for (int offset = 16; offset > 0; offset >>= 1) {
                    local_max = max(local_max, __shfl_down_sync(0xffffffff, local_max, offset));
                }
                
                // Broadcast max from lane 0
                const scalar_t max_val = __shfl_sync(0xffffffff, local_max, 0);
                
                // Warp-level sum of exp
                scalar_t local_sum = 0;
                for (int ch = tid; ch < channels; ch += blockDim.x) {
                    const int idx = n * channels * depth * height * width 
                                  + ch * depth * height * width 
                                  + in_d * height * width 
                                  + in_h * width + in_w;
                    local_sum += expf(input[idx] - max_val);
                }
                
                for (int offset = 16; offset > 0; offset >>= 1) {
                    local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
                }
                
                const scalar_t sum_val = __shfl_sync(0xffffffff, local_sum, 0);
                
                // Compute softmax value for this channel
                const int idx = n * channels * depth * height * width 
                              + c * depth * height * width 
                              + in_d * height * width 
                              + in_h * width + in_w;
                const scalar_t softmax_val = expf(input[idx] - max_val) / sum_val;
                max_pool_val = max(max_pool_val, softmax_val);
            }
        }
    }
    
    if (tid == 0) {
        const int out_idx = n * channels * out_depth * out_height * out_width 
                           + c * out_depth * out_height * out_width 
                           + od * out_height * out_width 
                           + oh * out_width + ow;
        output[out_idx] = max_pool_val;
    }
}

torch::Tensor fused_softmax_double_maxpool_cuda(torch::Tensor input) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int depth = input.size(2);
    const int height = input.size(3);
    const int width = input.size(4);
    
    // Double pool with kernel=2 twice = pool with kernel=4, stride=4
    const int out_depth = (depth + 3) / 4;
    const int out_height = (height + 3) / 4;
    const int out_width = (width + 3) / 4;
    
    auto output = torch::empty({batch_size, channels, out_depth, out_height, out_width},
                               input.options().dtype(torch::kFloat32));
    
    const int total_bc = batch_size * channels;
    const int total_spatial = out_depth * out_height * out_width;
    
    dim3 grid(total_bc, total_spatial);
    dim3 block(std::min(channels, 128));  // Use up to 128 threads per block
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "fused_softmax_double_maxpool", ([&]{
        fused_softmax_double_maxpool_optimized<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<float>(),
            batch_size, channels, depth, height, width,
            out_depth, out_height, out_width
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources='torch::Tensor fused_softmax_double_maxpool_cuda(torch::Tensor input);',
    cuda_sources=cuda_source,
    functions=['fused_softmax_double_maxpool_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Optimized Model with fused Softmax + MaxPool operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size
        
        # Enable cuDNN benchmark for optimal convolution algorithms
        torch.backends.cudnn.benchmark = True

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (batch_size, in_channels, depth, height, width)
        Returns:
            Output tensor after Conv3d -> Softmax -> MaxPool -> MaxPool
        """
        # Convolution with mixed precision
        with torch.amp.autocast('cuda'):
            x = self.conv(x)
        
        # Convert to float32 for numerical stability
        x = x.float()
        
        # Fused softmax + double maxpool
        x = ext.fused_softmax_double_maxpool_cuda(x)
        
        return x
