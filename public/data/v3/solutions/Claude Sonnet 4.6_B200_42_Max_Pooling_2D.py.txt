import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>

// ============================================================
// Generic Max Pool 2D - handles all cases
// ============================================================
template <typename scalar_t>
__global__ void maxpool2d_generic_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int kernel_h,
    const int kernel_w,
    const int stride_h,
    const int stride_w,
    const int pad_h,
    const int pad_w,
    const int dilation_h,
    const int dilation_w
) {
    const int ow = blockIdx.x * blockDim.x + threadIdx.x;
    const int oh = blockIdx.y * blockDim.y + threadIdx.y;
    const int bc = blockIdx.z;
    
    if (ow >= out_w || oh >= out_h) return;
    
    const int input_base = bc * in_h * in_w;
    float max_val = -FLT_MAX;
    
    const int ih_start = oh * stride_h - pad_h;
    const int iw_start = ow * stride_w - pad_w;
    
    #pragma unroll
    for (int kh = 0; kh < kernel_h; kh++) {
        const int ih = ih_start + kh * dilation_h;
        if (ih >= 0 && ih < in_h) {
            #pragma unroll
            for (int kw = 0; kw < kernel_w; kw++) {
                const int iw = iw_start + kw * dilation_w;
                if (iw >= 0 && iw < in_w) {
                    float val = (float)input[input_base + ih * in_w + iw];
                    if (val > max_val) max_val = val;
                }
            }
        }
    }
    
    output[(bc * out_h + oh) * out_w + ow] = (scalar_t)max_val;
}

// ============================================================
// Specialized kernel: kernel_size=2, processes 4 channels at once
// Uses shared memory to cache input data
// ============================================================
template <typename scalar_t>
__global__ void maxpool2d_k2_smem_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int stride_h,
    const int stride_w,
    const int pad_h,
    const int pad_w,
    const int dilation_h,
    const int dilation_w
) {
    // Block covers TILE_OW x TILE_OH output positions
    // Each block processes 4 channels
    const int TILE_OW = 32;
    const int TILE_OH = 8;
    
    // Shared memory for input: (TILE_OH*stride + dil) x (TILE_OW*stride + dil) per channel, 4 channels
    // For stride=2, dil=3, ks=2: effective dil span = (ks-1)*dil = 3
    // smem_h = TILE_OH * stride_h + dilation_h  (for ks=2)
    // smem_w = TILE_OW * stride_w + dilation_w  (for ks=2)
    // = 8*2+3=19, 32*2+3=67 -> 19*67*4*sizeof(float) = 20312 bytes - fits in 48KB
    
    extern __shared__ float smem[];
    
    const int smem_h = TILE_OH * stride_h + dilation_h; // 19 for default params
    const int smem_w = TILE_OW * stride_w + dilation_w; // 67 for default params
    const int smem_area = smem_h * smem_w;
    
    // blockIdx.x = output tile x index
    // blockIdx.y = output tile y index  
    // blockIdx.z = b * (channels/4) + c_group
    
    const int out_tile_x = blockIdx.x;
    const int out_tile_y = blockIdx.y;
    const int bc4 = blockIdx.z;
    
    const int b = bc4 / (channels / 4);
    const int c_base = (bc4 % (channels / 4)) * 4;
    
    const int ow_base = out_tile_x * TILE_OW;
    const int oh_base = out_tile_y * TILE_OH;
    
    // Input tile start (with padding consideration)
    const int ih_base = oh_base * stride_h - pad_h;
    const int iw_base = ow_base * stride_w - pad_w;
    
    const int tid = threadIdx.y * blockDim.x + threadIdx.x; // linear thread ID
    const int num_threads = blockDim.x * blockDim.y; // 256
    
    // Load input into shared memory for all 4 channels
    // Total elements to load: smem_area * 4
    const int in_base = (b * channels + c_base) * in_h * in_w;
    const int in_hw = in_h * in_w;
    
    for (int ci = 0; ci < 4; ci++) {
        float* smem_c = smem + ci * smem_area;
        for (int idx = tid; idx < smem_area; idx += num_threads) {
            int si = idx / smem_w;
            int sj = idx % smem_w;
            int ih = ih_base + si;
            int iw = iw_base + sj;
            float val = -FLT_MAX;
            if (ih >= 0 && ih < in_h && iw >= 0 && iw < in_w) {
                val = (float)input[in_base + ci * in_hw + ih * in_w + iw];
            }
            smem_c[idx] = val;
        }
    }
    __syncthreads();
    
    // Now compute outputs
    const int lx = threadIdx.x;
    const int ly = threadIdx.y;
    const int ow = ow_base + lx;
    const int oh = oh_base + ly;
    
    if (ow >= out_w || oh >= out_h) return;
    
    // Local positions in shared memory
    const int lih0 = ly * stride_h;
    const int liw0 = lx * stride_w;
    const int lih1 = lih0 + dilation_h;
    const int liw1 = liw0 + dilation_w;
    
    const int out_hw = out_h * out_w;
    const int out_base = (b * channels + c_base) * out_hw + oh * out_w + ow;
    
    #pragma unroll
    for (int ci = 0; ci < 4; ci++) {
        float* smem_c = smem + ci * smem_area;
        // 2x2 kernel, fully unrolled
        float v00 = smem_c[lih0 * smem_w + liw0];
        float v01 = smem_c[lih0 * smem_w + liw1];
        float v10 = smem_c[lih1 * smem_w + liw0];
        float v11 = smem_c[lih1 * smem_w + liw1];
        float mx = fmaxf(fmaxf(v00, v01), fmaxf(v10, v11));
        output[out_base + ci * out_hw] = (scalar_t)mx;
    }
}

// ============================================================
// Vec4 kernel - processes 4 channels at once, no shared memory
// Better for large kernels where smem would be too large
// ============================================================
template <typename scalar_t>
__global__ void maxpool2d_vec4_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int kernel_h,
    const int kernel_w,
    const int stride_h,
    const int stride_w,
    const int pad_h,
    const int pad_w,
    const int dilation_h,
    const int dilation_w
) {
    const int ow = blockIdx.x * blockDim.x + threadIdx.x;
    const int oh = blockIdx.y * blockDim.y + threadIdx.y;
    const int bc4 = blockIdx.z;
    
    if (ow >= out_w || oh >= out_h) return;
    
    const int b = bc4 / (channels / 4);
    const int c = (bc4 % (channels / 4)) * 4;
    
    float max0 = -FLT_MAX, max1 = -FLT_MAX, max2 = -FLT_MAX, max3 = -FLT_MAX;
    
    const int in_hw = in_h * in_w;
    const int base0 = (b * channels + c) * in_hw;
    
    const int ih_start = oh * stride_h - pad_h;
    const int iw_start = ow * stride_w - pad_w;
    
    #pragma unroll
    for (int kh = 0; kh < kernel_h; kh++) {
        const int ih = ih_start + kh * dilation_h;
        if (ih >= 0 && ih < in_h) {
            #pragma unroll
            for (int kw = 0; kw < kernel_w; kw++) {
                const int iw = iw_start + kw * dilation_w;
                if (iw >= 0 && iw < in_w) {
                    int off = ih * in_w + iw;
                    float v0 = (float)input[base0 + off];
                    float v1 = (float)input[base0 + in_hw + off];
                    float v2 = (float)input[base0 + 2*in_hw + off];
                    float v3 = (float)input[base0 + 3*in_hw + off];
                    max0 = fmaxf(max0, v0);
                    max1 = fmaxf(max1, v1);
                    max2 = fmaxf(max2, v2);
                    max3 = fmaxf(max3, v3);
                }
            }
        }
    }
    
    const int out_hw = out_h * out_w;
    const int out_base = (b * channels + c) * out_hw + oh * out_w + ow;
    output[out_base] = (scalar_t)max0;
    output[out_base + out_hw] = (scalar_t)max1;
    output[out_base + 2*out_hw] = (scalar_t)max2;
    output[out_base + 3*out_hw] = (scalar_t)max3;
}

torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_h, int kernel_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dilation_h, int dilation_w
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_h = input.size(2);
    const int in_w = input.size(3);
    
    const int out_h = (in_h + 2 * pad_h - dilation_h * (kernel_h - 1) - 1) / stride_h + 1;
    const int out_w = (in_w + 2 * pad_w - dilation_w * (kernel_w - 1) - 1) / stride_w + 1;
    
    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());
    
    const int TILE_X = 32;
    const int TILE_Y = 8;
    
    // Use shared memory kernel for k=2 case with channels divisible by 4
    bool use_smem = (kernel_h == 2 && kernel_w == 2 && channels % 4 == 0);
    bool use_vec4 = (channels % 4 == 0 && !use_smem);
    
    if (use_smem) {
        const int smem_h = TILE_Y * stride_h + dilation_h;
        const int smem_w = TILE_X * stride_w + dilation_w;
        int smem_size = smem_h * smem_w * 4 * sizeof(float);
        
        dim3 block(TILE_X, TILE_Y, 1);
        dim3 grid(
            (out_w + TILE_X - 1) / TILE_X,
            (out_h + TILE_Y - 1) / TILE_Y,
            batch_size * (channels / 4)
        );
        
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16,
            input.scalar_type(), "maxpool2d_k2_smem", [&] {
                maxpool2d_k2_smem_kernel<scalar_t><<<grid, block, smem_size>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    batch_size, channels, in_h, in_w, out_h, out_w,
                    stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w
                );
            });
    } else if (use_vec4) {
        dim3 block(TILE_X, TILE_Y, 1);
        dim3 grid(
            (out_w + TILE_X - 1) / TILE_X,
            (out_h + TILE_Y - 1) / TILE_Y,
            batch_size * (channels / 4)
        );
        
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16,
            input.scalar_type(), "maxpool2d_vec4", [&] {
                maxpool2d_vec4_kernel<scalar_t><<<grid, block>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    batch_size, channels, in_h, in_w, out_h, out_w,
                    kernel_h, kernel_w, stride_h, stride_w,
                    pad_h, pad_w, dilation_h, dilation_w
                );
            });
    } else {
        dim3 block(TILE_X, TILE_Y, 1);
        dim3 grid(
            (out_w + TILE_X - 1) / TILE_X,
            (out_h + TILE_Y - 1) / TILE_Y,
            batch_size * channels
        );
        
        AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16,
            input.scalar_type(), "maxpool2d_generic", [&] {
                maxpool2d_generic_kernel<scalar_t><<<grid, block>>>(
                    input.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    batch_size, channels, in_h, in_w, out_h, out_w,
                    kernel_h, kernel_w, stride_h, stride_w,
                    pad_h, pad_w, dilation_h, dilation_w
                );
            });
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_h, int kernel_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dilation_h, int dilation_w
);
'''

ext = load_inline(
    name='maxpool2d_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['maxpool2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    """
    Optimized Max Pooling 2D using custom CUDA kernel.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d_cuda(
            x,
            self.kernel_size, self.kernel_size,
            self.stride, self.stride,
            self.padding, self.padding,
            self.dilation, self.dilation
        )


batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3


def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]


def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
