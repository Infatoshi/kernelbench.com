cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Highly optimized depthwise conv2d for 3x3 kernel, stride=1
// Each block: 32x4 threads, each thread processes 8 output rows
// Tile: 32 wide x 32 tall output elements per block
__global__ void depthwise_conv2d_3x3_s1_v3(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int padding
) {
    constexpr int TILE_W = 32;
    constexpr int ROWS_PER_THREAD = 8;
    constexpr int TILE_H = 4 * ROWS_PER_THREAD; // 32
    
    const int bc = blockIdx.z;
    const int b = bc / channels;
    const int c = bc % channels;
    
    const int out_col_start = blockIdx.x * TILE_W;
    const int out_row_start = blockIdx.y * TILE_H;
    
    const int tx = threadIdx.x; // 0..31
    const int ty = threadIdx.y; // 0..3
    
    // Load weights into registers
    float w0, w1, w2, w3, w4, w5, w6, w7, w8;
    {
        const float* wptr = weight + c * 9;
        w0 = wptr[0]; w1 = wptr[1]; w2 = wptr[2];
        w3 = wptr[3]; w4 = wptr[4]; w5 = wptr[5];
        w6 = wptr[6]; w7 = wptr[7]; w8 = wptr[8];
    }
    
    // Shared memory: (TILE_H + 2) x (TILE_W + 4) for bank-conflict-free access
    constexpr int SMEM_H = TILE_H + 2;  // 34
    constexpr int SMEM_W = TILE_W + 4;  // 36 (padded)
    __shared__ float smem[SMEM_H * SMEM_W];
    
    const int in_row_start = out_row_start - padding;
    const int in_col_start = out_col_start - padding;
    
    const float* input_bc = input + (b * channels + c) * in_h * in_w;
    
    // Load shared memory cooperatively: 128 threads, need 34 * 34 = 1156 elements
    const int tid = ty * 32 + tx;
    const int actual_cols = TILE_W + 2; // 34
    const int num_elements = SMEM_H * actual_cols; // 34 * 34 = 1156
    
    for (int i = tid; i < num_elements; i += 128) {
        int sr = i / actual_cols;
        int sc = i % actual_cols;
        int ir = in_row_start + sr;
        int ic = in_col_start + sc;
        float val = 0.0f;
        if (ir >= 0 && ir < in_h && ic >= 0 && ic < in_w) {
            val = input_bc[ir * in_w + ic];
        }
        smem[sr * SMEM_W + sc] = val;
    }
    __syncthreads();
    
    float* output_bc = output + (b * channels + c) * out_h * out_w;
    const int out_col = out_col_start + tx;
    
    if (out_col < out_w) {
        #pragma unroll
        for (int r = 0; r < ROWS_PER_THREAD; r++) {
            int out_row = out_row_start + ty * ROWS_PER_THREAD + r;
            
            if (out_row < out_h) {
                int sr = ty * ROWS_PER_THREAD + r;
                int sc = tx;
                
                float sum = 0.0f;
                sum += smem[(sr + 0) * SMEM_W + sc + 0] * w0;
                sum += smem[(sr + 0) * SMEM_W + sc + 1] * w1;
                sum += smem[(sr + 0) * SMEM_W + sc + 2] * w2;
                sum += smem[(sr + 1) * SMEM_W + sc + 0] * w3;
                sum += smem[(sr + 1) * SMEM_W + sc + 1] * w4;
                sum += smem[(sr + 1) * SMEM_W + sc + 2] * w5;
                sum += smem[(sr + 2) * SMEM_W + sc + 0] * w6;
                sum += smem[(sr + 2) * SMEM_W + sc + 1] * w7;
                sum += smem[(sr + 2) * SMEM_W + sc + 2] * w8;
                
                output_bc[out_row * out_w + out_col] = sum;
            }
        }
    }
}

// Even larger tiles: 64 wide
__global__ void depthwise_conv2d_3x3_s1_v4(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_h,
    const int in_w,
    const int out_h,
    const int out_w,
    const int padding
) {
    // blockDim = (32, 8), each thread processes 4 output rows
    // Tile: 32 wide x 32 tall
    constexpr int TILE_W = 32;
    constexpr int ROWS_PER_THREAD = 4;
    constexpr int BLOCK_ROWS = 8;
    constexpr int TILE_H = BLOCK_ROWS * ROWS_PER_THREAD; // 32
    
    const int bc = blockIdx.z;
    const int b = bc / channels;
    const int c = bc % channels;
    
    const int out_col_start = blockIdx.x * TILE_W;
    const int out_row_start = blockIdx.y * TILE_H;
    
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    float w0, w1, w2, w3, w4, w5, w6, w7, w8;
    {
        const float* wptr = weight + c * 9;
        w0 = wptr[0]; w1 = wptr[1]; w2 = wptr[2];
        w3 = wptr[3]; w4 = wptr[4]; w5 = wptr[5];
        w6 = wptr[6]; w7 = wptr[7]; w8 = wptr[8];
    }
    
    constexpr int SMEM_H = TILE_H + 2;  // 34
    constexpr int SMEM_W = TILE_W + 4;  // 36
    __shared__ float smem[SMEM_H * SMEM_W];
    
    const int in_row_start = out_row_start - padding;
    const int in_col_start = out_col_start - padding;
    
    const float* input_bc = input + (b * channels + c) * in_h * in_w;
    
    const int tid = ty * 32 + tx;
    const int actual_cols = TILE_W + 2;
    const int num_elements = SMEM_H * actual_cols;
    const int num_threads = 256;
    
    for (int i = tid; i < num_elements; i += num_threads) {
        int sr = i / actual_cols;
        int sc = i % actual_cols;
        int ir = in_row_start + sr;
        int ic = in_col_start + sc;
        float val = 0.0f;
        if (ir >= 0 && ir < in_h && ic >= 0 && ic < in_w) {
            val = input_bc[ir * in_w + ic];
        }
        smem[sr * SMEM_W + sc] = val;
    }
    __syncthreads();
    
    float* output_bc = output + (b * channels + c) * out_h * out_w;
    const int out_col = out_col_start + tx;
    
    if (out_col < out_w) {
        #pragma unroll
        for (int r = 0; r < ROWS_PER_THREAD; r++) {
            int out_row = out_row_start + ty * ROWS_PER_THREAD + r;
            
            if (out_row < out_h) {
                int sr = ty * ROWS_PER_THREAD + r;
                int sc = tx;
                
                float sum = 0.0f;
                sum += smem[(sr + 0) * SMEM_W + sc + 0] * w0;
                sum += smem[(sr + 0) * SMEM_W + sc + 1] * w1;
                sum += smem[(sr + 0) * SMEM_W + sc + 2] * w2;
                sum += smem[(sr + 1) * SMEM_W + sc + 0] * w3;
                sum += smem[(sr + 1) * SMEM_W + sc + 1] * w4;
                sum += smem[(sr + 1) * SMEM_W + sc + 2] * w5;
                sum += smem[(sr + 2) * SMEM_W + sc + 0] * w6;
                sum += smem[(sr + 2) * SMEM_W + sc + 1] * w7;
                sum += smem[(sr + 2) * SMEM_W + sc + 2] * w8;
                
                output_bc[out_row * out_w + out_col] = sum;
            }
        }
    }
}


torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_h = input.size(2);
    const int in_w = input.size(3);
    const int kernel_size = weight.size(2);
    
    const int out_h = (in_h + 2 * padding - kernel_size) / stride + 1;
    const int out_w = (in_w + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_h, out_w}, input.options());
    
    // Use v4 with 256 threads per block for better occupancy
    {
        const int TILE_W = 32;
        const int TILE_H = 32;
        dim3 block(32, 8);
        dim3 grid(
            (out_w + TILE_W - 1) / TILE_W,
            (out_h + TILE_H - 1) / TILE_H,
            batch_size * channels
        );
        depthwise_conv2d_3x3_s1_v4<<<grid, block>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size, channels, in_h, in_w, out_h, out_w,
            padding
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d_forward(torch::Tensor input, torch::Tensor weight, int stride, int padding);
'''

ext = load_inline(
    name='depthwise_conv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.stride = stride
        self.padding = padding
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.depthwise_conv2d_forward(
            x, 
            self.conv2d.weight,
            self.stride,
            self.padding
        )
