import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized reduction kernel using shared memory
template<typename T>
__global__ void sum_reduction_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int64_t batch_size,
    int64_t reduce_dim,
    int64_t output_dim)
{
    constexpr int BLOCK_SIZE = 256;
    
    int tid = threadIdx.x;
    
    // Each block handles one output element (one (batch, out_idx) pair)
    int64_t total_outputs = batch_size * output_dim;
    int64_t output_idx = blockIdx.x;
    
    if (output_idx >= total_outputs) {
        return;
    }
    
    int64_t batch_idx = output_idx / output_dim;
    int64_t out_idx = output_idx % output_dim;
    
    extern __shared__ unsigned char smem_raw[];
    T* smem = reinterpret_cast<T*>(smem_raw);
    
    // Each thread sums a strided portion of the reduce dimension
    T sum = 0;
    for (int64_t i = tid; i < reduce_dim; i += BLOCK_SIZE) {
        int64_t global_idx = batch_idx * reduce_dim * output_dim + i * output_dim + out_idx;
        sum += input[global_idx];
    }
    
    // Store partial sum in shared memory
    smem[tid] = sum;
    __syncthreads();
    
    // Parallel reduction in shared memory
    for (int stride = BLOCK_SIZE / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            smem[tid] += smem[tid + stride];
        }
        __syncthreads();
    }
    
    // Write result
    if (tid == 0) {
        output[output_idx] = smem[0];
    }
}

// Templated wrapper function
template<typename T>
torch::Tensor sum_reduction(torch::Tensor x, int dim) {
    int ndim = x.dim();
    
    // Normalize dimension
    if (dim < 0) dim += ndim;
    
    // Calculate sizes
    int64_t before_size = 1;
    for (int i = 0; i < dim; i++) before_size *= x.size(i);
    int64_t reduce_size = x.size(dim);
    int64_t after_size = 1;
    for (int i = dim + 1; i < ndim; i++) after_size *= x.size(i);
    
    int64_t batch_size = before_size;
    int64_t output_dim = after_size;
    
    // Create output tensor with reduced dimension
    std::vector<int64_t> out_shape;
    for (int i = 0; i < ndim; i++) {
        if (i == dim) out_shape.push_back(1);
        else out_shape.push_back(x.size(i));
    }
    auto output = torch::empty(out_shape, x.options());
    
    // Total number of output elements
    int64_t num_outputs = batch_size * output_dim;
    
    // Choose kernel based on reduce dimension size
    constexpr int BLOCK_SIZE = 256;
    
    // Launch kernel
    int num_blocks = static_cast<int>(num_outputs);
    size_t smem_size = BLOCK_SIZE * sizeof(T);
    sum_reduction_kernel<T><<<num_blocks, BLOCK_SIZE, smem_size>>>(
        x.data_ptr<T>(),
        output.data_ptr<T>(),
        batch_size,
        reduce_size,
        output_dim
    );
    
    return output;
}

// Main dispatch function
torch::Tensor my_op(torch::Tensor x, int dim) {
    if (x.dtype() == torch::kFloat32) {
        return sum_reduction<float>(x, dim);
    } else if (x.dtype() == torch::kFloat16) {
        return sum_reduction<at::Half>(x, dim);
    } else if (x.dtype() == torch::kBFloat16) {
        return sum_reduction<at::BFloat16>(x, dim);
    } else {
        throw std::runtime_error("Unsupported dtype");
    }
}
'''

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources='torch::Tensor my_op(torch::Tensor x, int dim);',
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.my_op(x, self.dim)

def get_inputs():
    return [torch.randn(64, 256, 256, device='cuda')]

def get_init_inputs():
    return [1]
