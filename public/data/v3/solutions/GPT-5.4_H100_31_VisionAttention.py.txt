import math
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ['fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


def _next_power_of_2(x: int) -> int:
    return 1 if x <= 1 else 1 << (x - 1).bit_length()


@triton.jit
def _layernorm_residual_kernel(
    x_ptr, r_ptr, y_ptr,
    stride_xm, stride_xn,
    stride_rm, stride_rn,
    stride_ym, stride_yn,
    N,
    eps,
    BLOCK_N: tl.constexpr,
):
    row = tl.program_id(0)
    cols = tl.arange(0, BLOCK_N)
    mask = cols < N

    x = tl.load(x_ptr + row * stride_xm + cols * stride_xn, mask=mask, other=0.0)
    r = tl.load(r_ptr + row * stride_rm + cols * stride_rn, mask=mask, other=0.0)
    z = x + r
    zf = z.to(tl.float32)
    mean = tl.sum(zf, axis=0) / N
    centered = zf - mean
    var = tl.sum(centered * centered, axis=0) / N
    inv_std = tl.rsqrt(var + eps)
    y = centered * inv_std
    tl.store(y_ptr + row * stride_ym + cols * stride_yn, y.to(z.dtype), mask=mask)


class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        B, C, H, W = x.shape
        seq = H * W
        inp = x.view(B, C, seq).permute(2, 0, 1).contiguous()

        qkv = inp
        with torch.backends.cuda.sdp_kernel(enable_flash=True, enable_math=False, enable_mem_efficient=True, enable_cudnn=True):
            attn_output = torch.nn.functional.scaled_dot_product_attention(
                qkv.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3),
                qkv.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3),
                qkv.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3),
                dropout_p=0.0,
                is_causal=False,
            )
        attn_output = attn_output.permute(2, 0, 1, 3).reshape(seq, B, C)

        # apply packed in_proj and out_proj to match nn.MultiheadAttention weights exactly
        w = self.attn.in_proj_weight
        b = self.attn.in_proj_bias
        q = torch.addmm(b[:C], inp.reshape(-1, C), w[:C].t()).view(seq, B, C)
        k = torch.addmm(b[C:2*C], inp.reshape(-1, C), w[C:2*C].t()).view(seq, B, C)
        v = torch.addmm(b[2*C:], inp.reshape(-1, C), w[2*C:].t()).view(seq, B, C)
        q = q.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3)
        k = k.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3)
        v = v.view(seq, B, self.attn.num_heads, C // self.attn.num_heads).permute(1, 2, 0, 3)
        with torch.backends.cuda.sdp_kernel(enable_flash=True, enable_math=False, enable_mem_efficient=True, enable_cudnn=True):
            attn = torch.nn.functional.scaled_dot_product_attention(q, k, v, dropout_p=0.0, is_causal=False)
        attn = attn.permute(2, 0, 1, 3).reshape(seq, B, C)
        out = torch.addmm(self.attn.out_proj.bias, attn.reshape(-1, C), self.attn.out_proj.weight.t()).view(seq, B, C)

        rows = seq * B
        out2d = out.reshape(rows, C)
        inp2d = inp.reshape(rows, C)
        y = torch.empty_like(out2d)
        BLOCK_N = _next_power_of_2(C)
        _layernorm_residual_kernel[(rows,)](
            out2d, inp2d, y,
            out2d.stride(0), out2d.stride(1),
            inp2d.stride(0), inp2d.stride(1),
            y.stride(0), y.stride(1),
            C,
            self.norm.eps,
            BLOCK_N=BLOCK_N,
        )
        return y.view(seq, B, C).permute(1, 2, 0).reshape(B, C, H, W)


embed_dim = 128
num_heads = 4
batch_size = 2
num_channels = embed_dim
image_height = 128
image_width = 128


def get_inputs():
    return [torch.randn(batch_size, num_channels, image_height, image_width)]


def get_init_inputs():
    return [embed_dim, num_heads]
