import torch
import torch.nn as nn
import torch.nn.functional as F
from fla.ops import chunk_kda

class Model(nn.Module):
    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_dplr: bool = False,
        dplr_rank: int = 4,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        num_qk = num_heads * head_dim_qk
        num_v = num_heads * head_dim_v
        
        # We fuse Q, K, V, a, b, g projections into a single Linear
        self.in_proj = nn.Linear(hidden_size, 2*num_qk + 2*num_v + num_heads + num_v, bias=True)
        
        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)

        if use_short_conv:
            self.qkv_conv = nn.Conv1d(
                2 * num_qk + num_v, 2 * num_qk + num_v,
                kernel_size=conv_kernel_size, groups=2 * num_qk + num_v,
                padding=conv_kernel_size - 1
            )

        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape

        proj = self.in_proj(x)
        num_qk = self.num_heads * self.head_dim_qk
        num_v = self.num_heads * self.head_dim_v
        
        qkv = proj[:, :, :2*num_qk + num_v]
        a = proj[:, :, 2*num_qk + num_v : 2*num_qk + 2*num_v]
        b = proj[:, :, 2*num_qk + 2*num_v : 2*num_qk + 2*num_v + self.num_heads]
        g = proj[:, :, 2*num_qk + 2*num_v + self.num_heads :]

        if self.use_short_conv:
            qkv = self.qkv_conv(qkv.mT)[:, :, :seq_len].mT
            qkv = F.silu(qkv)

        # q, k: (B, T, H, Dqk) -> (B, H, T, Dqk)
        # v: (B, T, H, Dv) -> (B, H, T, Dv)
        qkv = qkv.view(batch_size, seq_len, self.num_heads, 2 * self.head_dim_qk + self.head_dim_v)
        q, k, v = qkv.split([self.head_dim_qk, self.head_dim_qk, self.head_dim_v], dim=-1)
        q = q.transpose(1, 2)
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)

        a = torch.sigmoid(a).view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)
        beta = torch.sigmoid(b).transpose(1, 2)  # (B, H, T)

        # fla expects gate in log-space for numerical stability
        a_log = a.clamp_min(1e-6).log_()

        output, _ = chunk_kda(q, k, v, a_log, beta, scale=self.scale)

        output = output.transpose(1, 2)  # (B, T, H, d_v)
        output = self.o_norm(output)

        g = torch.sigmoid_(g)
        g = g.view(batch_size, seq_len, self.num_heads, self.head_dim_v)
        output = output * g

        output = output.reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
        output = self.o_proj(output)

        return output

    def load_state_dict(self, state_dict, strict=True):
        # Intercept load_state_dict to map reference weights to fused weights
        my_state_dict = self.state_dict()
        
        if 'q_proj.weight' in state_dict:
            q_w = state_dict['q_proj.weight']
            k_w = state_dict['k_proj.weight']
            v_w = state_dict['v_proj.weight']
            a_w = state_dict['a_proj.weight']
            b_w = state_dict['b_proj.weight']
            g_w = state_dict['g_proj.weight']
            
            in_w = torch.cat([q_w, k_w, v_w, a_w, b_w, g_w], dim=0)
            
            a_b = state_dict['a_proj.bias']
            b_b = state_dict['b_proj.bias']
            
            z_qkv = torch.zeros(q_w.shape[0] + k_w.shape[0] + v_w.shape[0], dtype=a_b.dtype, device=a_b.device)
            z_g = torch.zeros(g_w.shape[0], dtype=a_b.dtype, device=a_b.device)
            
            in_b = torch.cat([z_qkv, a_b, b_b, z_g], dim=0)
            
            my_state_dict['in_proj.weight'].copy_(in_w)
            my_state_dict['in_proj.bias'].copy_(in_b)
            
            my_state_dict['o_proj.weight'].copy_(state_dict['o_proj.weight'])
            my_state_dict['o_norm.weight'].copy_(state_dict['o_norm.weight'])
            my_state_dict['o_norm.bias'].copy_(state_dict['o_norm.bias'])
            
            if self.use_short_conv:
                q_cw = state_dict['q_conv.weight']
                k_cw = state_dict['k_conv.weight']
                v_cw = state_dict['v_conv.weight']
                my_state_dict['qkv_conv.weight'].copy_(torch.cat([q_cw, k_cw, v_cw], dim=0))
                
                q_cb = state_dict['q_conv.bias']
                k_cb = state_dict['k_conv.bias']
                v_cb = state_dict['v_conv.bias']
                my_state_dict['qkv_conv.bias'].copy_(torch.cat([q_cb, k_cb, v_cb], dim=0))
        else:
            super().load_state_dict(state_dict, strict=strict)

def get_inputs():
    batch_size = 4
    seq_len = 2048
    hidden_size = 2048
    return [torch.randn(batch_size, seq_len, hidden_size)]

def get_init_inputs():
    hidden_size = 2048
    num_heads = 16
    head_dim_qk = 128
    head_dim_v = 128
    return [hidden_size, num_heads, head_dim_qk, head_dim_v]
