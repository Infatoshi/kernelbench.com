
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <float.h>

template <typename T>
struct MaxOp {
    __device__ static inline T initial() { return -static_cast<T>(1e38); }
    __device__ static inline T apply(T a, T b) { return (a > b) ? a : b; }
};

template <>
struct MaxOp<half> {
    __device__ static inline half initial() { return __float2half(-65504.0f); }
    __device__ static inline half apply(half a, half b) { 
#if __CUDA_ARCH__ >= 800
        return __hmax(a, b);
#else
        return __hgt(a, b) ? a : b; 
#endif
    }
};

template <>
struct MaxOp<nv_bfloat16> {
    __device__ static inline nv_bfloat16 initial() { return __float2bfloat16(-3.38e38f); }
    __device__ static inline nv_bfloat16 apply(nv_bfloat16 a, nv_bfloat16 b) { 
#if __CUDA_ARCH__ >= 800
        return __hmax(a, b);
#else
        return __hgt(a, b) ? a : b; 
#endif
    }
};

template <typename T>
__global__ void max_pool2d_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size, int channels, int in_h, int in_w,
    int out_h, int out_w,
    int kernel_h, int kernel_w,
    int stride_h, int stride_w,
    int padding_h, int padding_w,
    int dilation_h, int dilation_w)
{
    int total_elements = batch_size * channels * out_h * out_w;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= total_elements) return;

    int ow = idx % out_w;
    int oh = (idx / out_w) % out_h;
    int c = (idx / (out_w * out_h)) % channels;
    int b = idx / (out_w * out_h * channels);

    int ih_start = oh * stride_h - padding_h;
    int iw_start = ow * stride_w - padding_w;

    T max_val = MaxOp<T>::initial();
    
    const T* input_ptr = input + (b * channels + c) * (in_h * in_w);

    for (int kh = 0; kh < kernel_h; ++kh) {
        int ih = ih_start + kh * dilation_h;
        if (ih >= 0 && ih < in_h) {
            for (int kw = 0; kw < kernel_w; ++kw) {
                int iw = iw_start + kw * dilation_w;
                if (iw >= 0 && iw < in_w) {
                    T val = input_ptr[ih * in_w + iw];
                    max_val = MaxOp<T>::apply(max_val, val);
                }
            }
        }
    }
    output[idx] = max_val;
}

torch::Tensor max_pool2d_cuda(
    torch::Tensor input,
    int kernel_size, int stride, int padding, int dilation) 
{
    auto b = input.size(0);
    auto c = input.size(1);
    auto h = input.size(2);
    auto w = input.size(3);

    int kh = kernel_size, kw = kernel_size;
    int sh = stride, sw = stride;
    int ph = padding, pw = padding;
    int dh = dilation, dw = dilation;

    int oh = (h + 2 * ph - dh * (kh - 1) - 1) / sh + 1;
    int ow = (w + 2 * pw - dw * (kw - 1) - 1) / sw + 1;

    auto output = torch::empty({b, c, oh, ow}, input.options());
    int total_elements = b * c * oh * ow;
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "max_pool2d_cuda", ([&] {
        max_pool2d_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            b, c, h, w, oh, ow, kh, kw, sh, sw, ph, pw, dh, dw
        );
    }));

    return output;
}
'''

cpp_source = r'''
torch::Tensor max_pool2d_cuda(torch::Tensor input, int kernel_size, int stride, int padding, int dilation);
'''

ext = load_inline(
    name='max_pool2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['max_pool2d_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.max_pool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    kernel_size = 2
    stride = 2
    padding = 1
    dilation = 3
    return [kernel_size, stride, padding, dilation]
