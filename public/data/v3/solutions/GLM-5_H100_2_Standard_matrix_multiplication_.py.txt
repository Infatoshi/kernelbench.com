import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cutlass/cutlass.h>
#include <cutlass/gemm/dispatch_policy.hpp>
#include <cutlass/gemm/collective/collective_builder.hpp>
#include <cutlass/gemm/device/gemm_universal_adapter.h>
#include <cutlass/epilogue/collective/default_epilogue.hpp>
#include <cutlass/epilogue/thread/linear_combination.h>

using ElementA = cutlass::half_t;
using ElementB = cutlass::half_t;
using ElementC = cutlass::half_t;
using ElementAccumulator = float;

using LayoutA = cutlass::layout::RowMajor;
using LayoutB = cutlass::layout::ColumnMajor;
using LayoutC = cutlass::layout::RowMajor;

using KernelSchedule = cutlass::gemm::KernelTmaWarpSpecialized;
using EpilogueSchedule = cutlass::epilogue::TmaWarpSpecialized;

using CollectiveOp = typename cutlass::gemm::collective::CollectiveBuilder<
    cutlass::arch::Sm90,
    cutlass::arch::OpClassTensorOp,
    ElementA, LayoutA, 8,
    ElementB, LayoutB, 8,
    ElementAccumulator,
    cutlass::gemm::GemmShape<128, 128, 64>,
    cutlass::gemm::collective::StageCountAuto,
    cutlass::gemm::collective::KernelScheduleAuto
>::CollectiveOp;

using EpilogueOp = cutlass::epilogue::collective::DefaultEpilogue<
    cutlass::gemm::TagToStrideC<LayoutC>::Stride,
    cutlass::gemm::TagToStrideC<LayoutC>::Stride,
    cutlass::epilogue::collective::EpilogueScheduleAuto
>;

using GemmKernel = cutlass::gemm::kernel::GemmUniversal<
    cutlass::gemm::collective::CollectiveOp,
    cutlass::epilogue::collective::DefaultEpilogue
>;

using Gemm = cutlass::gemm::device::GemmUniversalAdapter<GemmKernel>;

torch::Tensor gemm_op(torch::Tensor A, torch::Tensor B) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    typename Gemm::Arguments arguments{
        cutlass::gemm::GemmUniversalMode::kGemm,
        {M, N, K},
        {A.data_ptr<cutlass::half_t>(), LayoutA::Stride{K}},
        {B.data_ptr<cutlass::half_t>(), LayoutB::Stride{K}},
        {C.data_ptr<cutlass::half_t>(), LayoutC::Stride{N}},
        {C.data_ptr<cutlass::half_t>(), LayoutC::Stride{N}},
        {alpha, beta}
    };
    
    Gemm gemm_op;
    size_t workspace_size = gemm_op.get_workspace_size(arguments);
    auto workspace = torch::empty({static_cast<int64_t>(workspace_size)}, 
                                   torch::dtype(torch::kUInt8).device(torch::kCUDA));
    
    cutlass::Status status = gemm_op.can_implement(arguments);
    if (status != cutlass::Status::kSuccess) {
        throw std::runtime_error("CUTLASS cannot implement this GEMM configuration");
    }
    
    status = gemm_op.run(arguments, workspace.data_ptr());
    if (status != cutlass::Status::kSuccess) {
        throw std::runtime_error("CUTLASS GEMM execution failed");
    }
    
    return C;
}
'''

cpp_source = r'''
torch::Tensor gemm_op(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_op'],
    extra_cuda_cflags=['-arch=sm_90', '-I/opt/cutlass/include', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        A_half = A.half() if A.dtype != torch.float16 else A
        B_half = B.half() if B.dtype != torch.float16 else B
        C = ext.gemm_op(A_half, B_half)
        return C.float() if A.dtype == torch.float32 else C

M = 1024
K = 4096
N = 2048

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, N)
    return [A, B]

def get_init_inputs():
    return []
