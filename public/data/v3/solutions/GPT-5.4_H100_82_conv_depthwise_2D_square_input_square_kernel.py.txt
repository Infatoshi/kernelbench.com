import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

template <typename scalar_t, bool HasBias>
__global__ void depthwise_conv2d_k3s1_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ w,
    const scalar_t* __restrict__ b,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W, int Ho, int Wo) {

    int linear = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * Ho * Wo;
    if (linear >= total) return;

    int ow = linear % Wo;
    linear /= Wo;
    int oh = linear % Ho;
    linear /= Ho;
    int c = linear % C;
    int n = linear / C;

    int in_base = ((n * C + c) * H + oh) * W + ow;
    int w_base = c * 9;

    float acc = 0.f;
    acc += (float)x[in_base + 0 * W + 0] * (float)w[w_base + 0];
    acc += (float)x[in_base + 0 * W + 1] * (float)w[w_base + 1];
    acc += (float)x[in_base + 0 * W + 2] * (float)w[w_base + 2];
    acc += (float)x[in_base + 1 * W + 0] * (float)w[w_base + 3];
    acc += (float)x[in_base + 1 * W + 1] * (float)w[w_base + 4];
    acc += (float)x[in_base + 1 * W + 2] * (float)w[w_base + 5];
    acc += (float)x[in_base + 2 * W + 0] * (float)w[w_base + 6];
    acc += (float)x[in_base + 2 * W + 1] * (float)w[w_base + 7];
    acc += (float)x[in_base + 2 * W + 2] * (float)w[w_base + 8];
    if constexpr (HasBias) acc += (float)b[c];

    y[((n * C + c) * Ho + oh) * Wo + ow] = (scalar_t)acc;
}

torch::Tensor depthwise_conv2d_forward(torch::Tensor x, torch::Tensor w, c10::optional<torch::Tensor> bias_opt) {
    TORCH_CHECK(x.is_cuda(), "x must be CUDA");
    TORCH_CHECK(w.is_cuda(), "w must be CUDA");
    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(w.dim() == 4, "w must be [C,1,3,3]");
    TORCH_CHECK(x.is_contiguous(), "x must be contiguous");
    TORCH_CHECK(w.is_contiguous(), "w must be contiguous");
    TORCH_CHECK(w.size(1) == 1 && w.size(2) == 3 && w.size(3) == 3, "only kernel 3x3 depthwise supported");
    TORCH_CHECK(x.size(1) == w.size(0), "channel mismatch");
    TORCH_CHECK(x.scalar_type() == w.scalar_type(), "dtype mismatch");

    const int64_t N = x.size(0);
    const int64_t C = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Ho = H - 2;
    const int64_t Wo = W - 2;
    TORCH_CHECK(Ho > 0 && Wo > 0, "invalid output size");

    auto y = torch::empty({N, C, Ho, Wo}, x.options());
    const int threads = 256;
    const int total = (int)(N * C * Ho * Wo);
    const int blocks = (total + threads - 1) / threads;
    auto stream = at::cuda::getDefaultCUDAStream();

    bool has_bias = bias_opt.has_value() && bias_opt.value().defined();
    torch::Tensor bias;
    if (has_bias) {
        bias = bias_opt.value();
        TORCH_CHECK(bias.is_cuda(), "bias must be CUDA");
        TORCH_CHECK(bias.is_contiguous(), "bias must be contiguous");
        TORCH_CHECK(bias.dim() == 1 && bias.size(0) == C, "bias shape mismatch");
        TORCH_CHECK(bias.scalar_type() == x.scalar_type(), "bias dtype mismatch");
    }

    if (x.scalar_type() == at::ScalarType::Float) {
        using scalar_t = float;
        if (has_bias) {
            depthwise_conv2d_k3s1_kernel<scalar_t, true><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), bias.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        } else {
            depthwise_conv2d_k3s1_kernel<scalar_t, false><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), nullptr, y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        }
    } else if (x.scalar_type() == at::ScalarType::Half) {
        using scalar_t = at::Half;
        if (has_bias) {
            depthwise_conv2d_k3s1_kernel<scalar_t, true><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), bias.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        } else {
            depthwise_conv2d_k3s1_kernel<scalar_t, false><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), nullptr, y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        }
    } else if (x.scalar_type() == at::ScalarType::BFloat16) {
        using scalar_t = at::BFloat16;
        if (has_bias) {
            depthwise_conv2d_k3s1_kernel<scalar_t, true><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), bias.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        } else {
            depthwise_conv2d_k3s1_kernel<scalar_t, false><<<blocks, threads, 0, stream>>>(
                x.data_ptr<scalar_t>(), w.data_ptr<scalar_t>(), nullptr, y.data_ptr<scalar_t>(),
                (int)N, (int)C, (int)H, (int)W, (int)Ho, (int)Wo);
        }
    } else {
        TORCH_CHECK(false, "unsupported dtype");
    }

    return y;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor depthwise_conv2d_forward(torch::Tensor x, torch::Tensor w, c10::optional<torch::Tensor> bias_opt);
'''

ext = load_inline(
    name='dwconv2d_k3s1_ext_h100_v1',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo', '-std=c++17', '-U__CUDA_NO_HALF_OPERATORS__', '-U__CUDA_NO_BFLOAT16_OPERATORS__'],
    verbose=False,
)


OP_TYPE = "conv"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        assert kernel_size == 3 and stride == 1 and padding == 0, 'optimized kernel supports reference shape only'
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_cuda:
            return self.conv2d(x)
        return ext.depthwise_conv2d_forward(x.contiguous(), self.conv2d.weight.contiguous(), self.conv2d.bias.contiguous() if self.conv2d.bias is not None else None)


def get_inputs():
    batch_size = 16
    in_channels = 3
    width = 256
    height = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    batch_size = 16
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
