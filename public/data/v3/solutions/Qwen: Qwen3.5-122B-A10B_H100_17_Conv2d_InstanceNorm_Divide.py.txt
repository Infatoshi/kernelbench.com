import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math.h>

// Fused InstanceNorm + Divide kernel for FP32
template<typename scalar_t>
__global__ void fused_instance_norm_divide_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    int batch_size,
    int channels,
    int height,
    int width,
    scalar_t eps,
    scalar_t divide_by,
    bool use_weight,
    bool use_bias
) {
    int channel_idx = blockIdx.x;
    int tid = threadIdx.x;
    int threads_per_block = blockDim.x;
    
    int hw_size = height * width;
    
    __shared__ scalar_t shared_mean[32];
    
    // Compute mean for each sample in this channel
    for (int s = tid; s < batch_size; s += threads_per_block) {
        scalar_t sum = 0;
        int base_idx = (s * channels + channel_idx) * hw_size;
        for (int hw = 0; hw < hw_size; hw++) {
            sum += input[base_idx + hw];
        }
        shared_mean[s] = sum / (scalar_t)hw_size;
    }
    __syncthreads();
    
    // Compute variance and normalize
    for (int s = tid; s < batch_size; s += threads_per_block) {
        scalar_t mean = shared_mean[s];
        scalar_t var_sum = 0;
        int base_idx = (s * channels + channel_idx) * hw_size;
        
        for (int hw = 0; hw < hw_size; hw++) {
            scalar_t diff = input[base_idx + hw] - mean;
            var_sum += diff * diff;
        }
        
        scalar_t var = var_sum / (scalar_t)hw_size + eps;
        scalar_t inv_std = scalar_t(1.0) / sqrt(var);
        
        scalar_t w = use_weight ? weight[channel_idx] : scalar_t(1);
        scalar_t b = use_bias ? bias[channel_idx] : scalar_t(0);
        
        for (int hw = tid; hw < hw_size; hw += threads_per_block) {
            scalar_t val = input[base_idx + hw];
            val = (val - mean) * inv_std * w + b;
            val = val / divide_by;
            output[base_idx + hw] = val;
        }
    }
}

torch::Tensor fused_instance_norm_divide(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float eps,
    float divide_by,
    bool use_weight,
    bool use_bias
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    
    auto output = torch::empty_like(input);
    
    int hw_size = height * width;
    int threads_per_block = 256;
    int num_blocks = channels;
    
    if (input.scalar_type() == torch::kFloat32) {
        fused_instance_norm_divide_kernel<float><<<num_blocks, threads_per_block>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            use_weight ? weight.data_ptr<float>() : nullptr,
            use_bias ? bias.data_ptr<float>() : nullptr,
            batch_size,
            channels,
            height,
            width,
            eps,
            divide_by,
            use_weight,
            use_bias
        );
    } else if (input.scalar_type() == torch::kFloat16) {
        fused_instance_norm_divide_kernel_fp16<<<num_blocks, threads_per_block>>>(
            reinterpret_cast<const __half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
            use_weight ? reinterpret_cast<const __half*>(weight.data_ptr<at::Half>()) : nullptr,
            use_bias ? reinterpret_cast<const __half*>(bias.data_ptr<at::Half>()) : nullptr,
            batch_size,
            channels,
            height,
            width,
            eps,
            divide_by,
            use_weight,
            use_bias
        );
    } else if (input.scalar_type() == torch::kBFloat16) {
        fused_instance_norm_divide_kernel_bf16<<<num_blocks, threads_per_block>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            use_weight ? reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr<at::BFloat16>()) : nullptr,
            use_bias ? reinterpret_cast<const __nv_bfloat16*>(bias.data_ptr<at::BFloat16>()) : nullptr,
            batch_size,
            channels,
            height,
            width,
            eps,
            divide_by,
            use_weight,
            use_bias
        );
    }
    
    cudaDeviceSynchronize();
    return output;
}

# Load the CUDA extension
ext = load_inline(
    name='fused_instance_norm',
    cpp_sources='''
    torch::Tensor fused_instance_norm_divide(
        torch::Tensor input,
        torch::Tensor weight,
        torch::Tensor bias,
        float eps,
        float divide_by,
        bool use_weight,
        bool use_bias
    );
    ''',
    cuda_sources=cuda_source,
    functions=['fused_instance_norm_divide'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model using fused CUDA kernel for InstanceNorm + Divide.
    Conv2d uses PyTorch's optimized implementation (cuDNN/CUTLASS).
    InstanceNorm + Divide are fused into a single kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels)
        self.divide_by = divide_by
        self.eps = self.instance_norm.eps

    def forward(self, x):
        # Conv2d - uses PyTorch's optimized cuDNN/CUTLASS
        x = self.conv(x)
        
        # Fused InstanceNorm + Divide
        weight = self.instance_norm.weight if self.instance_norm.affine else None
        bias = self.instance_norm.bias if self.instance_norm.affine else None
        x = ext.fused_instance_norm_divide(
            x, 
            weight if weight is not None else x,  # placeholder
            bias if bias is not None else x,  # placeholder
            self.eps,
            self.divide_by,
            self.instance_norm.affine,
            self.instance_norm.affine
        )
        return x


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]


def get_init_inputs():
    return [3, 16, 3, 2.0]
