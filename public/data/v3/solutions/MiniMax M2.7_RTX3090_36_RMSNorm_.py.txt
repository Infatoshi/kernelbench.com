import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Each block handles multiple spatial positions in parallel
// blockIdx.x = batch_idx * num_blocks_per_batch + block_in_batch
template <typename T>
__global__ void rmsnorm_kernel(const T* __restrict__ x, T* __restrict__ y,
                               int batch_size, int features, int spatial_size,
                               int num_blocks_per_batch,
                               float eps) {
    extern __shared__ float shared_rms[];
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x / num_blocks_per_batch;
    int block_offset = (blockIdx.x % num_blocks_per_batch) * blockDim.x;
    
    // Process TILE_SIZE spatial positions per block
    const int TILE_SIZE = 4;
    
    for (int tile = 0; tile < TILE_SIZE; tile++) {
        int spatial_idx = block_offset + tid + tile * blockDim.x;
        
        if (spatial_idx >= spatial_size) continue;
        if (batch_idx >= batch_size) continue;
        
        // Compute sum of squares for this (batch, spatial) position
        float sum_sq = 0.0f;
        
        // Each thread loads its assigned feature
        #pragma unroll
        for (int feat = tid; feat < features; feat += blockDim.x) {
            int flat_idx = ((batch_idx * features + feat) * spatial_size + spatial_idx);
            T val = x[flat_idx];
            sum_sq += static_cast<float>(val) * static_cast<float>(val);
        }
        
        // Warp reduction
        sum_sq = warp_reduce_sum(sum_sq);
        
        // Thread 0 computes RMS
        if (tid == 0) {
            shared_rms[0] = sqrtf(sum_sq / static_cast<float>(features) + eps);
        }
        __syncthreads();
        float rms = shared_rms[0];
        
        // Normalize
        #pragma unroll
        for (int feat = tid; feat < features; feat += blockDim.x) {
            int flat_idx = ((batch_idx * features + feat) * spatial_size + spatial_idx);
            T val = x[flat_idx];
            y[flat_idx] = static_cast<T>(static_cast<float>(val) / rms);
        }
        
        __syncthreads();
    }
}

template <typename T>
void launch_rmsnorm_kernel(const torch::Tensor& x, torch::Tensor& y,
                           int batch_size, int features, int spatial_size,
                           float eps) {
    const int threads = 64;  // Match feature dimension
    const int TILE_SIZE = 4;
    const int elements_per_block = threads * TILE_SIZE;
    
    int num_blocks_per_batch = (spatial_size + elements_per_block - 1) / elements_per_block;
    int grid_size = batch_size * num_blocks_per_batch;
    
    rmsnorm_kernel<T><<<grid_size, threads, sizeof(float)>>>(
        x.data_ptr<T>(), y.data_ptr<T>(),
        batch_size, features, spatial_size,
        num_blocks_per_batch, eps
    );
}

torch::Tensor rmsnorm_op(torch::Tensor x, float eps = 1e-5) {
    auto x_cont = x.contiguous();
    int batch_size = x.size(0);
    int features = x.size(1);
    int spatial_size = 1;
    for (int i = 2; i < x.dim(); i++) {
        spatial_size *= x.size(i);
    }
    
    auto y = torch::empty_like(x);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "rmsnorm_cuda", ([&] {
        launch_rmsnorm_kernel<scalar_t>(x_cont, y, batch_size, features, spatial_size, eps);
    }));
    
    return y;
}
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources='''
torch::Tensor rmsnorm_op(torch::Tensor x, float eps = 1e-5);
''',
    cuda_sources=cuda_source,
    functions=['rmsnorm_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_op(x, self.eps)

def get_inputs():
    return [torch.randn(16, 64, 256, 256, device='cuda')]

def get_init_inputs():
    return [64]
