import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cmath>

#define M_SQRT2_OVER_2 0.70710678118654752440

template <typename T>
__device__ __forceinline__ T gelu_op(T x);

template <>
__device__ __forceinline__ float gelu_op<float>(float x) {
    return 0.5f * x * (1.0f + erf(x * M_SQRT2_OVER_2));
}

template <>
__device__ __forceinline__ __half gelu_op<__half>(__half x) {
    float fx = __half2float(x);
    return __float2half(0.5f * fx * (1.0f + erf(fx * M_SQRT2_OVER_2)));
}

template <>
__device__ __forceinline__ __nv_bfloat16 gelu_op<__nv_bfloat16>(__nv_bfloat16 x) {
    float fx = __bfloat162float(x);
    return __float2bfloat16(0.5f * fx * (1.0f + erf(fx * M_SQRT2_OVER_2)));
}

template <typename T>
__global__ void gelu_kernel(const T* __restrict__ x, T* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = gelu_op<T>(x[idx]);
    }
}

// Vectorized kernels for better throughput
template <typename T, int VecSize>
__global__ void gelu_vector_kernel(const T* __restrict__ x, T* __restrict__ y, int n) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * VecSize;
    if (idx + VecSize <= n) {
        using VecT = typename std::conditional<
            std::is_same<T, float>::value, float4,
            typename std::conditional<std::is_same<T, __half>::value, ushort4, ulonglong1>::type
        >::type;

        if constexpr (std::is_same<T, float>::value && VecSize == 4) {
            float4 val = reinterpret_cast<const float4*>(x + idx)[0];
            val.x = gelu_op<float>(val.x);
            val.y = gelu_op<float>(val.y);
            val.z = gelu_op<float>(val.z);
            val.w = gelu_op<float>(val.w);
            reinterpret_cast<float4*>(y + idx)[0] = val;
        } else if constexpr (std::is_same<T, __half>::value && VecSize == 8) {
            uint4 val = reinterpret_cast<const uint4*>(x + idx)[0];
            __half2* h2 = (__half2*)&val;
            for(int i=0; i<4; ++i) {
                float2 f2 = __half22float2(h2[i]);
                f2.x = 0.5f * f2.x * (1.0f + erff(f2.x * 0.70710678118654752440f));
                f2.y = 0.5f * f2.y * (1.0f + erff(f2.y * 0.70710678118654752440f));
                h2[i] = __float22half2_rn(f2);
            }
            reinterpret_cast<uint4*>(y + idx)[0] = val;
        } else if constexpr (std::is_same<T, __nv_bfloat16>::value && VecSize == 8) {
            uint4 val = reinterpret_cast<const uint4*>(x + idx)[0];
            __nv_bfloat16* hv = reinterpret_cast<__nv_bfloat16*>(&val);
            for(int i=0; i<8; ++i) hv[i] = gelu_op<__nv_bfloat16>(hv[i]);
            reinterpret_cast<uint4*>(y + idx)[0] = val;
        } else {
            // Fallback
            for(int i=0; i<VecSize; ++i) y[idx+i] = gelu_op<T>(x[idx+i]);
        }
    } else {
        for (int i = idx; i < n; ++i) {
            y[i] = gelu_op<T>(x[i]);
        }
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int threads = 256;
    
    at::ScalarType type = x.scalar_type();
    if (type == at::kFloat) {
        if (n % 4 == 0) {
            int blocks = (n / 4 + threads - 1) / threads;
            gelu_vector_kernel<float, 4><<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
        } else {
            int blocks = (n + threads - 1) / threads;
            gelu_kernel<float><<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
        }
    } else if (type == at::kHalf) {
        if (n % 8 == 0) {
            int blocks = (n / 8 + threads - 1) / threads;
            gelu_vector_kernel<__half, 8><<<blocks, threads>>>(
                reinterpret_cast<const __half*>(x.data_ptr<at::Half>()), 
                reinterpret_cast<__half*>(y.data_ptr<at::Half>()), n);
        } else {
            int blocks = (n + threads - 1) / threads;
            gelu_kernel<__half><<<blocks, threads>>>(
                reinterpret_cast<const __half*>(x.data_ptr<at::Half>()), 
                reinterpret_cast<__half*>(y.data_ptr<at::Half>()), n);
        }
    } else if (type == at::kBFloat16) {
        if (n % 8 == 0) {
            int blocks = (n / 8 + threads - 1) / threads;
            gelu_vector_kernel<__nv_bfloat16, 8><<<blocks, threads>>>(
                reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()), 
                reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
        } else {
            int blocks = (n + threads - 1) / threads;
            gelu_kernel<__nv_bfloat16><<<blocks, threads>>>(
                reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()), 
                reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
        }
    }
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_forward(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_forward(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    return [torch.randn(batch_size, dim, device='cuda')]

def get_init_inputs():
    return []
