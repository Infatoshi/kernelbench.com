import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def swish_bias_kernel(
    X_ptr, W_ptr, Lin_B_ptr, Bias_ptr,
    Y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_ym, stride_yn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptrs = X_ptr + (offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    b_ptrs = W_ptr + (offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        mask_k = (k * BLOCK_SIZE_K + offs_k) < K
        a = tl.load(a_ptrs, mask=mask_k[None, :], other=0.0)
        b = tl.load(b_ptrs, mask=mask_k[:, None], other=0.0)
        accumulator += tl.dot(a, b)
        a_ptrs += BLOCK_SIZE_K * stride_xk
        b_ptrs += BLOCK_SIZE_K * stride_wk

    offs_m = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_n = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    
    if Lin_B_ptr is not None:
        lin_b = tl.load(Lin_B_ptr + offs_n, mask=offs_n < N, other=0.0)
        accumulator += lin_b[None, :]
    
    accumulator = accumulator * tl.sigmoid(accumulator)
    
    bias_val = tl.load(Bias_ptr + offs_n, mask=offs_n < N, other=0.0)
    accumulator += bias_val[None, :]
    
    y_ptrs = Y_ptr + (offs_m[:, None] * stride_ym + offs_n[None, :] * stride_yn)
    mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(y_ptrs, accumulator, mask=mask)

@triton.jit
def group_norm_kernel(
    Y_ptr, GN_weight_ptr, GN_bias_ptr,
    Out_ptr,
    M, N, num_groups,
    stride_ym, stride_yn,
    stride_om, stride_on,
    BLOCK_SIZE_N: tl.constexpr,
):
    row_idx = tl.program_id(0)
    group_idx = tl.program_id(1)
    group_size = N // num_groups
    
    offs_n = group_idx * group_size + tl.arange(0, BLOCK_SIZE_N)
    mask = (offs_n < (group_idx + 1) * group_size)
    
    y_ptrs = Y_ptr + row_idx * stride_ym + offs_n * stride_yn
    vals = tl.load(y_ptrs, mask=mask, other=0.0).to(tl.float32)
    
    # Accurate Welford's could be used, but for GN simple mean/var is usually enough if N is not too large
    mean = tl.sum(vals, axis=0) / group_size
    diff = tl.where(mask, vals - mean, 0.0)
    var = tl.sum(diff * diff, axis=0) / group_size
    rstd = 1.0 / tl.sqrt(var + 1e-5)
    
    gn_w = tl.load(GN_weight_ptr + offs_n, mask=mask, other=1.0).to(tl.float32)
    gn_b = tl.load(GN_bias_ptr + offs_n, mask=mask, other=0.0).to(tl.float32)
    
    out = (vals - mean) * rstd * gn_w + gn_b
    
    out_ptrs = Out_ptr + row_idx * stride_om + offs_n * stride_on
    tl.store(out_ptrs, out, mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        
        mid_res = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        swish_bias_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, self.bias,
            mid_res,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            mid_res.stride(0), mid_res.stride(1),
        )
        
        output = torch.empty_like(mid_res)
        group_size = N // self.num_groups
        BLOCK_SIZE_N_GN = triton.next_power_of_2(group_size)
        
        gn_grid = (M, self.num_groups)
        group_norm_kernel[gn_grid](
            mid_res, self.group_norm.weight, self.group_norm.bias,
            output,
            M, N, self.num_groups,
            mid_res.stride(0), mid_res.stride(1),
            output.stride(0), output.stride(1),
            BLOCK_SIZE_N=BLOCK_SIZE_N_GN
        )
        
        return output

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 32, (4096,)]
