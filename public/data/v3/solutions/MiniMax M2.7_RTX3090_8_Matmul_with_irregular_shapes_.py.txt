import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized FP32 tiled GEMM with correct memory access
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

constexpr int BLOCK_SIZE = 32;

// Single precision GEMM with shared memory tiling
__global__ void tiled_gemm_fp32_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K) {
    
    __shared__ float As[BLOCK_SIZE][BLOCK_SIZE];
    __shared__ float Bs[BLOCK_SIZE][BLOCK_SIZE];
    
    int tx = threadIdx.x % BLOCK_SIZE;
    int ty = threadIdx.x / BLOCK_SIZE;
    int bx = blockIdx.x;
    int by = blockIdx.y;
    
    int row = by * BLOCK_SIZE + ty;
    int col = bx * BLOCK_SIZE + tx;
    
    float sum = 0.0f;
    
    // Loop over K dimension in blocks
    for (int kk = 0; kk < K; kk += BLOCK_SIZE) {
        // Load A tile - each thread loads one element
        // A is row-major, so A[row, kk+tx]
        if (row < M && kk + tx < K) {
            As[ty][tx] = A[row * K + kk + tx];
        } else {
            As[ty][ty] = 0.0f;
        }
        
        // Load B tile - each thread loads one element
        // B is column-major (from matmul semantics), so B[kk+ty, col]
        if (col < N && kk + ty < K) {
            Bs[ty][tx] = B[(kk + ty) * N + col];
        } else {
            Bs[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute dot product for this tile
        for (int k = 0; k < BLOCK_SIZE; k++) {
            sum += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    // Store result
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

// Optimized version with double buffering
constexpr int BLOCK_DB = 32;

__global__ void tiled_gemm_double_buffer_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K) {
    
    __shared__ float As[2][BLOCK_DB][BLOCK_DB];
    __shared__ float Bs[2][BLOCK_DB][BLOCK_DB];
    
    int tx = threadIdx.x % BLOCK_DB;
    int ty = threadIdx.x / BLOCK_DB;
    int bx = blockIdx.x;
    int by = blockIdx.y;
    
    int row = by * BLOCK_DB + ty;
    int col = bx * BLOCK_DB + tx;
    
    float sum = 0.0f;
    int kk = 0;
    
    // Load first tile
    if (row < M && tx < K) {
        As[0][ty][tx] = A[row * K + tx];
    } else {
        As[0][ty][tx] = 0.0f;
    }
    
    if (col < N && ty < K) {
        Bs[0][ty][tx] = B[ty * N + col];
    } else {
        Bs[0][ty][tx] = 0.0f;
    }
    
    __syncthreads();
    
    // Process tiles with double buffering
    for (kk = BLOCK_DB; kk < K; kk += BLOCK_DB) {
        int buf = (kk / BLOCK_DB) % 2;
        int next_buf = 1 - buf;
        
        // Load next tile
        if (row < M && kk + tx < K) {
            As[next_buf][ty][tx] = A[row * K + kk + tx];
        } else {
            As[next_buf][ty][tx] = 0.0f;
        }
        
        if (col < N && kk + ty < K) {
            Bs[next_buf][ty][tx] = B[(kk + ty) * N + col];
        } else {
            Bs[next_buf][ty][tx] = 0.0f;
        }
        
        // Compute with previous buffer
        int cur_buf = 1 - next_buf;
        for (int k = 0; k < BLOCK_DB; k++) {
            sum += As[cur_buf][ty][k] * Bs[cur_buf][k][tx];
        }
        
        __syncthreads();
    }
    
    // Final tiles with last buffer
    int last_buf = ((K / BLOCK_DB) % 2);
    for (int k = 0; k < BLOCK_DB; k++) {
        sum += As[last_buf][ty][k] * Bs[last_buf][k][tx];
    }
    
    // Store result
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

// Vectorized loading version
__global__ void tiled_gemm_vec4_kernel(
    const float4* __restrict__ A,
    const float4* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K) {
    
    __shared__ float As[32][32];
    __shared__ float Bs[32][32];
    
    int tx = threadIdx.x % 32;
    int ty = threadIdx.x / 32;
    int bx = blockIdx.x;
    int by = blockIdx.y;
    
    int row = by * 32 + ty;
    int col = bx * 32 + tx;
    
    float sum = 0.0f;
    
    // K is multiple of 4, iterate in steps of 4 floats
    for (int kk = 0; kk < K; kk += 32) {
        // Load A tile
        if (row < M) {
            int k_idx = kk + tx * 4;
            if (k_idx < K) {
                float4 v = A[row * (K/4) + k_idx/4];
                As[ty][tx*4] = v.x;
                As[ty][tx*4+1] = v.y;
                As[ty][tx*4+2] = v.z;
                As[ty][tx*4+3] = v.w;
            }
        }
        
        // Load B tile (column-major)
        if (col < N) {
            int k_idx = kk + ty * 4;
            if (k_idx < K) {
                float4 v = B[k_idx/4 * (N/4) + col/4];
                Bs[ty*4][tx] = v.x;
                Bs[ty*4+1][tx] = v.y;
                Bs[ty*4+2][tx] = v.z;
                Bs[ty*4+3][tx] = v.w;
            }
        }
        
        __syncthreads();
        
        // Compute
        for (int k = 0; k < 32; k++) {
            sum += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

torch::Tensor tiled_gemm_fp32(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    torch::Tensor C = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(A.device()));
    
    dim3 block(32, 32);
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    
    tiled_gemm_fp32_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );
    
    cudaDeviceSynchronize();
    return C;
}

torch::Tensor tiled_gemm_db(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    torch::Tensor C = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(A.device()));
    
    dim3 block(32, 32);
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    
    tiled_gemm_double_buffer_kernel<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );
    
    cudaDeviceSynchronize();
    return C;
}
'''

ext = load_inline(
    name='tiled_gemm_fp32_ext',
    cpp_sources=r'''
torch::Tensor tiled_gemm_fp32(torch::Tensor A, torch::Tensor B);
torch::Tensor tiled_gemm_db(torch::Tensor A, torch::Tensor B);
''',
    cuda_sources=cuda_source,
    functions=['tiled_gemm_fp32', 'tiled_gemm_db'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.tiled_gemm_fp32(A, B)
