import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// Fused kernel: tanh(x) * scale + bias, then max-pool with pool_size=2
// Input:  [N, C, H, W]  (after conv)
// Output: [N, C, H/2, W/2]  (after max-pool with stride=2, kernel=2)
// Bias:   [C] (broadcast over N, H, W)

template <typename scalar_t>
__global__ void fused_tanh_scale_bias_maxpool_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    float scaling_factor,
    int N, int C, int H, int W,
    int out_H, int out_W,
    int pool_size
) {
    // Each thread computes one output pixel
    // output index: [n, c, oh, ow]
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int nc = blockIdx.z;  // combined n*C + c
    
    int n = nc / C;
    int c = nc % C;
    
    if (ow >= out_W || oh >= out_H || n >= N) return;
    
    float bias_val = (float)bias[c];
    float max_val = -1e38f;
    
    int h_start = oh * pool_size;
    int w_start = ow * pool_size;
    
    // Load the pool_size x pool_size region, apply tanh+scale+bias, find max
    for (int ph = 0; ph < pool_size; ph++) {
        for (int pw = 0; pw < pool_size; pw++) {
            int ih = h_start + ph;
            int iw = w_start + pw;
            if (ih < H && iw < W) {
                float v = (float)input[((n * C + c) * H + ih) * W + iw];
                v = tanhf(v) * scaling_factor + bias_val;
                if (v > max_val) max_val = v;
            }
        }
    }
    
    output[((n * C + c) * out_H + oh) * out_W + ow] = (scalar_t)max_val;
}

// Optimized version for pool_size=2 with explicit unrolling
template <typename scalar_t>
__global__ void fused_tanh_scale_bias_maxpool2_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    float scaling_factor,
    int N, int C, int H, int W,
    int out_H, int out_W
) {
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int nc = blockIdx.z;
    
    int n = nc / C;
    int c = nc % C;
    
    if (ow >= out_W || oh >= out_H || n >= N) return;
    
    float bias_val = (float)bias[c];
    int base = (n * C + c) * H;
    int ih0 = oh * 2;
    int ih1 = ih0 + 1;
    int iw0 = ow * 2;
    int iw1 = iw0 + 1;
    
    float v00 = tanhf((float)input[(base + ih0) * W + iw0]) * scaling_factor + bias_val;
    float v01 = tanhf((float)input[(base + ih0) * W + iw1]) * scaling_factor + bias_val;
    float v10 = tanhf((float)input[(base + ih1) * W + iw0]) * scaling_factor + bias_val;
    float v11 = tanhf((float)input[(base + ih1) * W + iw1]) * scaling_factor + bias_val;
    
    float max_val = fmaxf(fmaxf(v00, v01), fmaxf(v10, v11));
    
    output[((n * C + c) * out_H + oh) * out_W + ow] = (scalar_t)max_val;
}

// Vectorized version loading float2/float4 for better memory throughput
// This processes 2 output pixels per thread (2 columns at once) when W is even
template <typename scalar_t>
__global__ void fused_tanh_scale_bias_maxpool2_vec_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    float scaling_factor,
    int N, int C, int H, int W,
    int out_H, int out_W
) {
    // Process 2 consecutive output columns per thread
    int ow2 = (blockIdx.x * blockDim.x + threadIdx.x) * 2; // output col * 2
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int nc = blockIdx.z;
    
    int n = nc / C;
    int c = nc % C;
    
    if (oh >= out_H || n >= N) return;
    
    float bias_val = (float)bias[c];
    int ih0 = oh * 2;
    int ih1 = ih0 + 1;
    
    // Process output pixel (oh, ow2) and (oh, ow2+1) if in bounds
    if (ow2 < out_W) {
        int iw0 = ow2 * 2;
        int iw1 = iw0 + 1;
        int base = (n * C + c) * H;
        
        float v00 = tanhf((float)input[(base + ih0) * W + iw0]) * scaling_factor + bias_val;
        float v01 = tanhf((float)input[(base + ih0) * W + iw1]) * scaling_factor + bias_val;
        float v10 = tanhf((float)input[(base + ih1) * W + iw0]) * scaling_factor + bias_val;
        float v11 = tanhf((float)input[(base + ih1) * W + iw1]) * scaling_factor + bias_val;
        float max_val = fmaxf(fmaxf(v00, v01), fmaxf(v10, v11));
        output[((n * C + c) * out_H + oh) * out_W + ow2] = (scalar_t)max_val;
    }
    
    if (ow2 + 1 < out_W) {
        int iw0 = (ow2 + 1) * 2;
        int iw1 = iw0 + 1;
        int base = (n * C + c) * H;
        
        float v00 = tanhf((float)input[(base + ih0) * W + iw0]) * scaling_factor + bias_val;
        float v01 = tanhf((float)input[(base + ih0) * W + iw1]) * scaling_factor + bias_val;
        float v10 = tanhf((float)input[(base + ih1) * W + iw0]) * scaling_factor + bias_val;
        float v11 = tanhf((float)input[(base + ih1) * W + iw1]) * scaling_factor + bias_val;
        float max_val = fmaxf(fmaxf(v00, v01), fmaxf(v10, v11));
        output[((n * C + c) * out_H + oh) * out_W + ow2 + 1] = (scalar_t)max_val;
    }
}

torch::Tensor fused_tanh_scale_bias_maxpool(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_size
) {
    TORCH_CHECK(input.is_cuda(), "input must be on CUDA");
    TORCH_CHECK(bias.is_cuda(), "bias must be on CUDA");
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous");
    
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    int out_H = H / pool_size;
    int out_W = W / pool_size;
    
    auto output = torch::empty({N, C, out_H, out_W}, input.options());
    
    // Use 16x16 thread block for good occupancy
    dim3 block(16, 16);
    dim3 grid(
        (out_W + block.x - 1) / block.x,
        (out_H + block.y - 1) / block.y,
        N * C
    );
    
    if (pool_size == 2 && H % 2 == 0 && W % 2 == 0) {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half, at::ScalarType::BFloat16,
            input.scalar_type(), "fused_tanh_scale_bias_maxpool2", [&] {
                fused_tanh_scale_bias_maxpool2_kernel<scalar_t><<<grid, block>>>(
                    input.data_ptr<scalar_t>(),
                    bias.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    scaling_factor,
                    N, C, H, W, out_H, out_W
                );
            }
        );
    } else {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half, at::ScalarType::BFloat16,
            input.scalar_type(), "fused_tanh_scale_bias_maxpool", [&] {
                fused_tanh_scale_bias_maxpool_kernel<scalar_t><<<grid, block>>>(
                    input.data_ptr<scalar_t>(),
                    bias.data_ptr<scalar_t>(),
                    output.data_ptr<scalar_t>(),
                    scaling_factor,
                    N, C, H, W, out_H, out_W,
                    pool_size
                );
            }
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_tanh_scale_bias_maxpool(
    torch::Tensor input,
    torch::Tensor bias,
    float scaling_factor,
    int pool_size
);
'''

ext = load_inline(
    name='fused_conv_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_tanh_scale_bias_maxpool'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
)


class Model(nn.Module):
    """
    Optimized model: Conv2d + fused(tanh + scale + bias + maxpool)
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Convolution (uses cuDNN)
        x = self.conv(x)
        # Fused: tanh + scale + bias + max-pool
        # bias shape is (C, 1, 1), flatten to (C,)
        bias_flat = self.bias.view(-1)
        x = x.contiguous()
        x = ext.fused_tanh_scale_bias_maxpool(
            x, bias_flat, float(self.scaling_factor), self.pool_kernel_size
        )
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
