import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ W,
    const float* __restrict__ b,
    float* __restrict__ out,
    int B, int IN, int OUT, float scale)
{
    int row = blockIdx.x;
    if (row >= B) return;

    extern __shared__ float sx[];

    // load x[row] into shared memory
    for (int i = threadIdx.x; i < IN; i += blockDim.x) {
        sx[i] = x[row * IN + i];
    }
    __syncthreads();

    int pairs = OUT / 2;
    float local_sum = 0.0f;

    for (int p = threadIdx.x; p < pairs; p += blockDim.x) {
        int o0 = 2 * p;
        int o1 = o0 + 1;

        float acc0 = 0.0f;
        float acc1 = 0.0f;

        const float* w0 = W + o0 * IN;
        const float* w1 = W + o1 * IN;

        for (int k = 0; k < IN; ++k) {
            float v = sx[k];
            acc0 += v * w0[k];
            acc1 += v * w1[k];
        }

        acc0 += b[o0];
        acc1 += b[o1];

        float m = acc0 > acc1 ? acc0 : acc1;
        local_sum += m;
    }

    // reduction
    __shared__ float ssum[256];
    ssum[threadIdx.x] = local_sum;
    __syncthreads();

    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            ssum[threadIdx.x] += ssum[threadIdx.x + stride];
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        out[row] = ssum[0] * scale;
    }
}

torch::Tensor fused_op(torch::Tensor x, torch::Tensor W, torch::Tensor b, float scale) {
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = W.size(0);

    auto out = torch::empty({B}, x.options());

    int threads = 256;
    size_t smem = IN * sizeof(float);

    fused_kernel<<<B, threads, smem>>>(
        x.data_ptr<float>(),
        W.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        B, IN, OUT, scale
    );

    return out;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor W, torch::Tensor b, float scale);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scale = scale_factor

    def forward(self, x):
        return ext.fused_op(x.contiguous(), self.matmul.weight.contiguous(), self.matmul.bias.contiguous(), self.scale)


def get_inputs():
    return [torch.randn(128, 4096, device='cuda', dtype=torch.float32)]


def get_init_inputs():
    return [4096, 4096, 2, 0.5]
