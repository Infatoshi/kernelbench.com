"""
Optimized Grouped GEMM for Mixture of Experts using Triton.
Separate up and down kernels with efficient tiling.
"""

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def moe_up_kernel(
    x_ptr,           # [batch*seq, hidden]
    expert_up_ptr,   # [num_experts, hidden, expert_dim]
    idx_flat_ptr,    # [batch*seq, top_k]
    up_out_ptr,      # [batch*seq, top_k, expert_dim]
    batch_seq: tl.constexpr,
    hidden: tl.constexpr,
    expert_dim: tl.constexpr,
    num_experts: tl.constexpr,
    top_k: tl.constexpr,
    stride_x_batch: tl.constexpr,
    stride_x_hidden: tl.constexpr,
    stride_up_expert: tl.constexpr,
    stride_up_hidden: tl.constexpr,
    stride_up_expert_dim: tl.constexpr,
    stride_idx_batch: tl.constexpr,
    stride_out_batch: tl.constexpr,
    stride_out_topk: tl.constexpr,
    stride_out_expert: tl.constexpr,
    BLOCK_M: tl.constexpr,  # token block
    BLOCK_K: tl.constexpr,  # hidden block
    BLOCK_N: tl.constexpr,  # expert_dim block
):
    """
    Compute up_out[token, route, :] = x[token, :] @ expert_up[expert_id, :, :]
    One program per (token, route) pair.
    """
    pid = tl.program_id(0)
    token_id = pid // top_k
    route_id = pid % top_k
    
    if token_id >= batch_seq:
        return
    
    expert_id = tl.load(idx_flat_ptr + token_id * stride_idx_batch + route_id).to(tl.int32)
    
    # Compute up_out for this (token, route) - output is [expert_dim]
    # We need to compute: sum over k of x[token, k] * up[expert_id, k, n]
    
    for n_start in range(0, expert_dim, BLOCK_N):
        n_offs = n_start + tl.arange(0, BLOCK_N)
        n_mask = n_offs < expert_dim
        
        acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
        
        for k_start in range(0, hidden, BLOCK_K):
            k_offs = k_start + tl.arange(0, BLOCK_K)
            k_mask = k_offs < hidden
            
            # Load x[token, k_offs]
            x_vals = tl.load(
                x_ptr + token_id * stride_x_batch + k_offs * stride_x_hidden,
                mask=k_mask, other=0.0
            ).to(tl.float32)
            
            # Load up[expert_id, k_offs, n_offs]
            up_w = tl.load(
                expert_up_ptr +
                expert_id * stride_up_expert +
                k_offs[:, None] * stride_up_hidden +
                n_offs[None, :] * stride_up_expert_dim,
                mask=(k_mask[:, None] & n_mask[None, :]),
                other=0.0
            ).to(tl.float32)
            
            # acc[n] += sum_k x[k] * up[k, n]
            acc += tl.sum(x_vals[:, None] * up_w, axis=0)
        
        # Store result
        out_offs = token_id * stride_out_batch + route_id * stride_out_topk + n_offs * stride_out_expert
        tl.store(up_out_ptr + out_offs, acc, mask=n_mask)


@triton.jit
def moe_down_kernel(
    up_out_ptr,      # [batch*seq, top_k, expert_dim]
    expert_down_ptr, # [num_experts, expert_dim, hidden]
    idx_flat_ptr,    # [batch*seq, top_k]
    w_flat_ptr,      # [batch*seq, top_k]
    out_ptr,         # [batch*seq, hidden]
    batch_seq: tl.constexpr,
    expert_dim: tl.constexpr,
    hidden: tl.constexpr,
    num_experts: tl.constexpr,
    top_k: tl.constexpr,
    stride_up_batch: tl.constexpr,
    stride_up_topk: tl.constexpr,
    stride_up_expert: tl.constexpr,
    stride_down_expert: tl.constexpr,
    stride_down_expert_dim: tl.constexpr,
    stride_down_hidden: tl.constexpr,
    stride_idx_batch: tl.constexpr,
    stride_w_batch: tl.constexpr,
    stride_out_batch: tl.constexpr,
    stride_out_hidden: tl.constexpr,
    BLOCK_M: tl.constexpr,  # token block
    BLOCK_K: tl.constexpr,  # expert_dim block
    BLOCK_N: tl.constexpr,  # hidden block
):
    """
    Compute out[token, :] += up_out[token, route, :] @ expert_down[expert_id, :, :] * weight
    One program per (token, route) pair, atomic add to output.
    """
    pid = tl.program_id(0)
    token_id = pid // top_k
    route_id = pid % top_k
    
    if token_id >= batch_seq:
        return
    
    expert_id = tl.load(idx_flat_ptr + token_id * stride_idx_batch + route_id).to(tl.int32)
    weight = tl.load(w_flat_ptr + token_id * stride_w_batch + route_id)
    
    # Load up_out for this (token, route)
    k_offs_full = tl.arange(0, BLOCK_K)
    k_mask_full = k_offs_full < expert_dim
    up_vals = tl.load(
        up_out_ptr + token_id * stride_up_batch + route_id * stride_up_topk + k_offs_full * stride_up_expert,
        mask=k_mask_full, other=0.0
    ).to(tl.float32)
    
    # Compute down projection: up_vals @ down[expert_id, :, :]
    for n_start in range(0, hidden, BLOCK_N):
        n_offs = n_start + tl.arange(0, BLOCK_N)
        n_mask = n_offs < hidden
        
        acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
        
        for k_start in range(0, expert_dim, BLOCK_K):
            k_offs = k_start + tl.arange(0, BLOCK_K)
            k_mask = k_offs < expert_dim
            
            # Load relevant portion of up_vals
            up_slice = tl.load(
                up_out_ptr + token_id * stride_up_batch + route_id * stride_up_topk + k_offs * stride_up_expert,
                mask=k_mask, other=0.0
            ).to(tl.float32)
            
            # Load down[expert_id, k_offs, n_offs]
            down_w = tl.load(
                expert_down_ptr +
                expert_id * stride_down_expert +
                k_offs[:, None] * stride_down_expert_dim +
                n_offs[None, :] * stride_down_hidden,
                mask=(k_mask[:, None] & n_mask[None, :]),
                other=0.0
            ).to(tl.float32)
            
            acc += tl.sum(up_slice[:, None] * down_w, axis=0)
        
        # Apply weight and atomic add
        out_offs = token_id * stride_out_batch + n_offs * stride_out_hidden
        tl.atomic_add(out_ptr + out_offs, acc * weight, mask=n_mask)


class Model(nn.Module):
    def __init__(self, num_experts: int = 8, hidden_dim: int = 1024, expert_dim: int = 3072):
        super().__init__()
        self.num_experts = num_experts
        self.hidden_dim = hidden_dim
        self.expert_dim = expert_dim
        self.expert_up = nn.Parameter(torch.randn(num_experts, hidden_dim, expert_dim) * 0.02)
        self.expert_down = nn.Parameter(torch.randn(num_experts, expert_dim, hidden_dim) * 0.02)

    def forward(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        batch, seq, hidden = x.shape
        top_k = expert_indices.shape[-1]
        expert_dim = self.expert_dim
        
        x_flat = x.reshape(batch * seq, hidden).contiguous()
        idx_flat = expert_indices.reshape(batch * seq, top_k).contiguous()
        w_flat = expert_weights.reshape(batch * seq, top_k).contiguous()
        out_flat = torch.zeros_like(x_flat)
        
        # Allocate intermediate
        up_out = torch.zeros(batch * seq, top_k, expert_dim, dtype=x.dtype, device=x.device)
        
        # Block sizes tuned for H100
        BLOCK_M = 1  # one token per program
        BLOCK_K = 128  # reduction dimension
        BLOCK_N = 128  # output dimension
        
        grid = (batch * seq * top_k,)
        
        # Up projection
        moe_up_kernel[grid](
            x_flat,
            self.expert_up,
            idx_flat,
            up_out,
            batch * seq,
            hidden,
            expert_dim,
            self.num_experts,
            top_k,
            x_flat.stride(0),
            x_flat.stride(1),
            self.expert_up.stride(0),
            self.expert_up.stride(1),
            self.expert_up.stride(2),
            idx_flat.stride(0),
            up_out.stride(0),
            up_out.stride(1),
            up_out.stride(2),
            BLOCK_M=BLOCK_M,
            BLOCK_K=BLOCK_K,
            BLOCK_N=BLOCK_N,
            num_warps=4,
        )
        
        # Down projection
        moe_down_kernel[grid](
            up_out,
            self.expert_down,
            idx_flat,
            w_flat,
            out_flat,
            batch * seq,
            expert_dim,
            hidden,
            self.num_experts,
            top_k,
            up_out.stride(0),
            up_out.stride(1),
            up_out.stride(2),
            self.expert_down.stride(0),
            self.expert_down.stride(1),
            self.expert_down.stride(2),
            idx_flat.stride(0),
            w_flat.stride(0),
            out_flat.stride(0),
            out_flat.stride(1),
            BLOCK_M=BLOCK_M,
            BLOCK_K=BLOCK_K,
            BLOCK_N=BLOCK_N,
            num_warps=4,
        )
        
        return out_flat.reshape(batch, seq, hidden)


def get_inputs():
    batch, seq, hidden = 4, 256, 1024
    top_k = 2
    num_experts = 8

    x = torch.randn(batch, seq, hidden)
    expert_indices = torch.randint(0, num_experts, (batch, seq, top_k))
    expert_weights = torch.softmax(torch.randn(batch, seq, top_k), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return []
