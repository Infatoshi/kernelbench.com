import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ inline float act(float x, float scale, float bias){
    float v = tanhf(x);
    v = v * scale;
    v = v + bias;
    return v;
}

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ cbias,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int N,int C,int H,int W,
    int OC,int KH,int KW,
    int OH,int OW,
    int PH,int PW,
    float scale)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*OC*PH*PW;
    if(idx>=total) return;

    int pw = idx % PW; idx/=PW;
    int ph = idx % PH; idx/=PH;
    int oc = idx % OC; idx/=OC;
    int n = idx;

    int h0 = ph*2;
    int w0 = pw*2;

    float m = -1e30f;

    #pragma unroll
    for(int dh=0; dh<2; ++dh){
        #pragma unroll
        for(int dw=0; dw<2; ++dw){
            int oh = h0 + dh;
            int ow = w0 + dw;

            float sum = cbias ? cbias[oc] : 0.0f;

            for(int c=0;c<C;++c){
                for(int kh=0;kh<KH;++kh){
                    for(int kw=0;kw<KW;++kw){
                        int ih = oh + kh;
                        int iw = ow + kw;
                        int x_idx = ((n*C + c)*H + ih)*W + iw;
                        int w_idx = ((oc*C + c)*KH + kh)*KW + kw;
                        sum += x[x_idx] * w[w_idx];
                    }
                }
            }

            float v = act(sum, scale, bias[oc]);
            m = fmaxf(m, v);
        }
    }

    int oidx = ((n*OC + oc)*PH + ph)*PW + pw;
    out[oidx] = m;
}


torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor cbias,
    torch::Tensor bias,
    float scale)
{
    int N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3);
    int OC=w.size(0), KH=w.size(2), KW=w.size(3);

    int OH = H-KH+1;
    int OW = W-KW+1;

    int PH = (OH-2)/2 + 1;
    int PW = (OW-2)/2 + 1;

    auto out = torch::empty({N,OC,PH,PW}, x.options());

    int total = N*OC*PH*PW;
    int threads=256;
    int blocks=(total+threads-1)/threads;

    fused_kernel<<<blocks,threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        cbias.numel()?cbias.data_ptr<float>():nullptr,
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        N,C,H,W,OC,KH,KW,OH,OW,PH,PW,scale);

    return out;
}
'''

ext = load_inline(
    name='fused_conv_tanh_pool',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor cbias, torch::Tensor bias, float scale);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3','-arch=sm_86'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        w = self.conv.weight.contiguous()
        cb = self.conv.bias.contiguous()
        b = self.bias.view(-1).contiguous()
        return ext.fused_forward(x.contiguous(), w, cb, b, float(self.scaling_factor))


def get_inputs():
    return [torch.randn(16,3,256,256, device='cuda')]


def get_init_inputs():
    return [3,16,3,2.0,(16,1,1),2]
