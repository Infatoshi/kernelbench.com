"""
FP8-like GEMV optimized for B200 using CUDA kernel with int8 inputs.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMV kernel optimized for B200 with int8 inputs and fp32 accumulation
// x_q: [batch, in_features], x_scale: scalar
// weight_q: [out_features, in_features], weight_scale: scalar
// output: [batch, out_features] in float16

__global__ void gemv_fp8_vectorized(
    const int8_t* __restrict__ x_q,
    const float x_scale,
    const int8_t* __restrict__ weight_q,
    const float weight_scale,
    at::Half* __restrict__ output,
    int batch,
    int in_features,
    int out_features
) {
    int out_idx = blockIdx.x;
    int batch_idx = blockIdx.y;
    int tid = threadIdx.x;
    
    if (out_idx >= out_features || batch_idx >= batch) return;
    
    const float combined_scale = x_scale * weight_scale;
    
    const int8_t* x_ptr = x_q + batch_idx * in_features;
    const int8_t* w_ptr = weight_q + out_idx * in_features;
    
    // Use 4 threads per output element for parallel reduction
    float accum = 0.0f;
    int elements_per_thread = (in_features + 3) / 4;
    int start = tid * elements_per_thread;
    int end = min(start + elements_per_thread, in_features);
    
    for (int k = start; k < end; k++) {
        accum += static_cast<float>(x_ptr[k]) * static_cast<float>(w_ptr[k]);
    }
    
    // Shared memory reduction
    extern __shared__ float sdata[256];
    sdata[tid] = accum;
    __syncthreads();
    
    // Parallel reduction
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[batch_idx * out_features + out_idx] = static_cast<at::Half>(sdata[0] * combined_scale);
    }
}

torch::Tensor gemv_fp8_forward(
    torch::Tensor x_q,
    torch::Tensor x_scale,
    torch::Tensor weight_q,
    torch::Tensor weight_scale
) {
    int batch = x_q.size(0);
    int in_features = x_q.size(1);
    int out_features = weight_q.size(0);
    
    auto output = torch::empty({batch, out_features}, 
        torch::dtype(torch::kFloat16).device(x_q.device()));
    
    float x_scale_val = x_scale.item<float>();
    float weight_scale_val = weight_scale.item<float>();
    
    // Launch configuration: one block per output element, multiple threads for reduction
    dim3 block(128);
    dim3 grid(out_features, batch);
    
    gemv_fp8_vectorized<<<grid, block>>>(
        x_q.data_ptr<int8_t>(),
        x_scale_val,
        weight_q.data_ptr<int8_t>(),
        weight_scale_val,
        output.data_ptr<at::Half>(),
        batch,
        in_features,
        out_features
    );
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor gemv_fp8_forward(
    torch::Tensor x_q,
    torch::Tensor x_scale,
    torch::Tensor weight_q,
    torch::Tensor weight_scale
);
'''

ext = load_inline(
    name='gemv_fp8_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp8_forward'],
    extra_cuda_cflags=['-arch=sm_100', '-O3', '--use_fast_math'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp8_forward(x_q, x_scale, self.weight_q, self.weight_scale)


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]
