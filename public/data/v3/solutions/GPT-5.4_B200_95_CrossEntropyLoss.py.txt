import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "reduction"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

# One-program-per-row fused cross entropy.
# Computes: loss_i = logsumexp(row_i) - row_i[target_i]
# Then reduces over batch on device and returns mean loss.

@triton.jit
def _cross_entropy_row_kernel(
    logits_ptr,
    targets_ptr,
    losses_ptr,
    stride_n,
    n_cols,
    BLOCK: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK)
    row_ptr = logits_ptr + row * stride_n

    # Pass 1: row max
    max_val = tl.full((), -float('inf'), tl.float32)
    for start in range(0, n_cols, BLOCK):
        cols = start + offs
        mask = cols < n_cols
        vals = tl.load(row_ptr + cols, mask=mask, other=-float('inf')).to(tl.float32)
        max_val = tl.maximum(max_val, tl.max(vals, axis=0))

    # Pass 2: sum exp
    sum_exp = tl.zeros((), tl.float32)
    for start in range(0, n_cols, BLOCK):
        cols = start + offs
        mask = cols < n_cols
        vals = tl.load(row_ptr + cols, mask=mask, other=-float('inf')).to(tl.float32)
        sum_exp += tl.sum(tl.exp(vals - max_val), axis=0)

    target = tl.load(targets_ptr + row)
    target_logit = tl.load(row_ptr + target).to(tl.float32)
    loss = tl.log(sum_exp) + max_val - target_logit
    tl.store(losses_ptr + row, loss)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        assert predictions.is_cuda and targets.is_cuda
        assert predictions.ndim == 2
        assert targets.ndim == 1
        B, C = predictions.shape
        losses = torch.empty((B,), device=predictions.device, dtype=torch.float32)
        BLOCK = 1024 if C <= 1024 else 2048
        _cross_entropy_row_kernel[(B,)](
            predictions,
            targets,
            losses,
            predictions.stride(0),
            C,
            BLOCK=BLOCK,
            num_warps=8,
        )
        return losses.mean()
