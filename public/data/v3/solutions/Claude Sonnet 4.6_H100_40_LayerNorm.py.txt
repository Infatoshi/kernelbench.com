import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# LayerNorm with normalized_shape=(64, 256, 256)
# Input shape: (16, 64, 256, 256) = 16 rows, each of size 64*256*256 = 4,194,304 elements
# We need to normalize each of the 16 rows independently.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp reduce sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, mask);
    return val;
}

// Welford online mean/variance update
struct WelfordState {
    float mean;
    float m2;
    float count;
};

__device__ __forceinline__ WelfordState welford_combine(WelfordState a, WelfordState b) {
    if (a.count == 0) return b;
    if (b.count == 0) return a;
    float count = a.count + b.count;
    float delta = b.mean - a.mean;
    float mean = a.mean + delta * (b.count / count);
    float m2 = a.m2 + b.m2 + delta * delta * a.count * b.count / count;
    return {mean, m2, count};
}

__device__ __forceinline__ WelfordState warp_reduce_welford(WelfordState state) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1) {
        WelfordState other;
        other.mean = __shfl_xor_sync(0xffffffff, state.mean, mask);
        other.m2 = __shfl_xor_sync(0xffffffff, state.m2, mask);
        other.count = __shfl_xor_sync(0xffffffff, state.count, mask);
        state = welford_combine(state, other);
    }
    return state;
}

// Block reduce using shared memory for Welford
__device__ __forceinline__ void block_reduce_welford(
    WelfordState& state,
    float* smem_mean, float* smem_m2, float* smem_count,
    int tid, int nthreads
) {
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    int num_warps = nthreads / 32;
    
    state = warp_reduce_welford(state);
    
    if (lane_id == 0) {
        smem_mean[warp_id] = state.mean;
        smem_m2[warp_id] = state.m2;
        smem_count[warp_id] = state.count;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        WelfordState wstate;
        if (lane_id < num_warps) {
            wstate.mean = smem_mean[lane_id];
            wstate.m2 = smem_m2[lane_id];
            wstate.count = smem_count[lane_id];
        } else {
            wstate = {0.0f, 0.0f, 0.0f};
        }
        wstate = warp_reduce_welford(wstate);
        if (lane_id == 0) {
            smem_mean[0] = wstate.mean;
            smem_m2[0] = wstate.m2;
        }
    }
    __syncthreads();
    state.mean = smem_mean[0];
    state.m2 = smem_m2[0];
}

// LayerNorm kernel using Welford single-pass + vectorized loads
// Each block handles one row
// Uses float4 vectorized loads for 4x throughput
template<typename scalar_t>
__global__ void layernorm_kernel(
    const scalar_t* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    scalar_t* __restrict__ out,
    int N,
    float eps
) {
    extern __shared__ float smem[];
    float* smem_mean = smem;
    float* smem_m2 = smem + blockDim.x/32;
    float* smem_count = smem + 2*(blockDim.x/32);
    
    int row = blockIdx.x;
    int tid = threadIdx.x;
    int nthreads = blockDim.x;
    
    const scalar_t* x_row = x + row * N;
    scalar_t* out_row = out + row * N;
    
    // Single-pass Welford for mean and variance
    WelfordState state = {0.0f, 0.0f, 0.0f};
    for (int i = tid; i < N; i += nthreads) {
        float xi = (float)x_row[i];
        state.count += 1.0f;
        float delta = xi - state.mean;
        state.mean += delta / state.count;
        float delta2 = xi - state.mean;
        state.m2 += delta * delta2;
    }
    
    block_reduce_welford(state, smem_mean, smem_m2, smem_count, tid, nthreads);
    
    float mean = state.mean;
    float inv_std = rsqrtf(state.m2 / N + eps);
    
    // Pass 2: normalize
    for (int i = tid; i < N; i += nthreads) {
        float x_val = (float)x_row[i];
        float w = weight[i];
        float b = bias[i];
        out_row[i] = (scalar_t)((x_val - mean) * inv_std * w + b);
    }
}

// Specialized float kernel with float4 vectorized memory access
__global__ void layernorm_kernel_float4(
    const float* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int N,
    float eps
) {
    extern __shared__ float smem[];
    float* smem_mean = smem;
    float* smem_m2 = smem + blockDim.x/32;
    float* smem_count = smem + 2*(blockDim.x/32);
    
    int row = blockIdx.x;
    int tid = threadIdx.x;
    int nthreads = blockDim.x;
    
    const float* x_row = x + row * N;
    float* out_row = out + row * N;
    
    int N4 = N / 4;
    const float4* x4 = reinterpret_cast<const float4*>(x_row);
    
    // Single-pass Welford using float4 loads
    WelfordState state = {0.0f, 0.0f, 0.0f};
    for (int i = tid; i < N4; i += nthreads) {
        float4 xi = x4[i];
        // Process 4 elements
        float vals[4] = {xi.x, xi.y, xi.z, xi.w};
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            state.count += 1.0f;
            float delta = vals[j] - state.mean;
            state.mean += delta / state.count;
            float delta2 = vals[j] - state.mean;
            state.m2 += delta * delta2;
        }
    }
    // Handle remainder
    for (int i = N4*4 + tid; i < N; i += nthreads) {
        float xi = x_row[i];
        state.count += 1.0f;
        float delta = xi - state.mean;
        state.mean += delta / state.count;
        float delta2 = xi - state.mean;
        state.m2 += delta * delta2;
    }
    
    block_reduce_welford(state, smem_mean, smem_m2, smem_count, tid, nthreads);
    
    float mean = state.mean;
    float inv_std = rsqrtf(state.m2 / N + eps);
    
    // Pass 2: normalize using float4
    const float4* w4 = reinterpret_cast<const float4*>(weight);
    const float4* b4 = reinterpret_cast<const float4*>(bias);
    float4* out4 = reinterpret_cast<float4*>(out_row);
    
    for (int i = tid; i < N4; i += nthreads) {
        float4 xv = x4[i];
        float4 wv = w4[i];
        float4 bv = b4[i];
        float4 ov;
        ov.x = (xv.x - mean) * inv_std * wv.x + bv.x;
        ov.y = (xv.y - mean) * inv_std * wv.y + bv.y;
        ov.z = (xv.z - mean) * inv_std * wv.z + bv.z;
        ov.w = (xv.w - mean) * inv_std * wv.w + bv.w;
        out4[i] = ov;
    }
    for (int i = N4*4 + tid; i < N; i += nthreads) {
        out_row[i] = (x_row[i] - mean) * inv_std * weight[i] + bias[i];
    }
}

torch::Tensor layernorm_forward(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float eps
) {
    auto out = torch::empty_like(x);
    
    int batch = x.size(0);
    int N = x.numel() / batch;
    
    // 1024 threads - maximizes parallelism for large N
    int nthreads = 1024;
    int num_warps = nthreads / 32;
    int smem_size = 3 * num_warps * sizeof(float);
    
    auto w = weight.to(torch::kFloat32).contiguous();
    auto b = bias.to(torch::kFloat32).contiguous();
    
    if (x.scalar_type() == torch::kFloat32 && (N % 4 == 0)) {
        layernorm_kernel_float4<<<batch, nthreads, smem_size>>>(
            x.data_ptr<float>(),
            w.data_ptr<float>(),
            b.data_ptr<float>(),
            out.data_ptr<float>(),
            N,
            eps
        );
    } else {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half,
            at::ScalarType::BFloat16,
            x.scalar_type(),
            "layernorm_forward",
            [&]() {
                layernorm_kernel<scalar_t><<<batch, nthreads, smem_size>>>(
                    x.data_ptr<scalar_t>(),
                    w.data_ptr<float>(),
                    b.data_ptr<float>(),
                    out.data_ptr<scalar_t>(),
                    N,
                    eps
                );
            }
        );
    }
    
    return out;
}
'''

cpp_source = '''
torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, float eps);
'''

ext = load_inline(
    name='layernorm_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    """
    Optimized LayerNorm using custom CUDA kernel with Welford algorithm and vectorized loads.
    """
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        self.eps = 1e-5
        norm_size = 1
        for s in normalized_shape:
            norm_size *= s
        self.weight = nn.Parameter(torch.ones(norm_size))
        self.bias = nn.Parameter(torch.zeros(norm_size))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        orig_shape = x.shape
        ndim_norm = len(self.normalized_shape)
        batch_dims = orig_shape[:-ndim_norm]
        batch_size = 1
        for d in batch_dims:
            batch_size *= d
        
        x_2d = x.contiguous().view(batch_size, -1)
        out_2d = ext.layernorm_forward(x_2d, self.weight, self.bias, self.eps)
        return out_2d.view(orig_shape)

    def load_state_dict(self, state_dict, strict=True):
        new_sd = {}
        for k, v in state_dict.items():
            if k == 'ln.weight':
                new_sd['weight'] = v.view(-1)
            elif k == 'ln.bias':
                new_sd['bias'] = v.view(-1)
            else:
                new_sd[k] = v
        return super().load_state_dict(new_sd, strict=False)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
