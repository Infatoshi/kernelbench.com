import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void moe_kernel(
    const float* x,
    const int64_t* expert_idx,
    const float* weights,
    const float* expert_up,
    const float* expert_down,
    float* out,
    int N, int H, int D, int topk, int E
) {
    int rid = blockIdx.x; // route id
    if (rid >= N * topk) return;

    int token = rid / topk;
    int k = rid % topk;

    int expert = expert_idx[token * topk + k];
    float w = weights[token * topk + k];

    const float* x_ptr = x + token * H;
    const float* up = expert_up + expert * H * D;
    const float* down = expert_down + expert * D * H;

    extern __shared__ float tmp[];

    // compute up: tmp[d]
    for (int d = threadIdx.x; d < D; d += blockDim.x) {
        float acc = 0.0f;
        for (int h = 0; h < H; ++h) {
            acc += x_ptr[h] * up[h * D + d];
        }
        tmp[d] = acc;
    }
    __syncthreads();

    // compute down and atomic add
    for (int h = threadIdx.x; h < H; h += blockDim.x) {
        float acc = 0.0f;
        for (int d = 0; d < D; ++d) {
            acc += tmp[d] * down[d * H + h];
        }
        atomicAdd(&out[token * H + h], acc * w);
    }
}

void launch(
    torch::Tensor x,
    torch::Tensor expert_idx,
    torch::Tensor weights,
    torch::Tensor expert_up,
    torch::Tensor expert_down,
    torch::Tensor out
) {
    int N = x.size(0);
    int H = x.size(1);
    int topk = expert_idx.size(1);
    int E = expert_up.size(0);
    int D = expert_up.size(2);

    int blocks = N * topk;
    int threads = 256;
    size_t shm = D * sizeof(float);

    moe_kernel<<<blocks, threads, shm>>>(
        x.data_ptr<float>(),
        expert_idx.data_ptr<int64_t>(),
        weights.data_ptr<float>(),
        expert_up.data_ptr<float>(),
        expert_down.data_ptr<float>(),
        out.data_ptr<float>(),
        N, H, D, topk, E
    );
}
'''

ext = load_inline(
    name='moe_ext',
    cpp_sources='void launch(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['launch'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_experts: int = 8, hidden_dim: int = 1024, expert_dim: int = 3072):
        super().__init__()
        self.num_experts = num_experts
        self.hidden_dim = hidden_dim
        self.expert_dim = expert_dim
        self.expert_up = nn.Parameter(torch.randn(num_experts, hidden_dim, expert_dim) * 0.02)
        self.expert_down = nn.Parameter(torch.randn(num_experts, expert_dim, hidden_dim) * 0.02)

    def forward(self, x, expert_indices, expert_weights):
        batch, seq, hidden = x.shape
        top_k = expert_indices.shape[-1]

        x_flat = x.reshape(batch * seq, hidden).contiguous()
        idx_flat = expert_indices.reshape(batch * seq, top_k).contiguous()
        w_flat = expert_weights.reshape(batch * seq, top_k).contiguous()

        out = torch.zeros_like(x_flat)

        ext.launch(x_flat, idx_flat, w_flat, self.expert_up, self.expert_down, out)

        return out.reshape(batch, seq, hidden)


def get_inputs():
    batch, seq, hidden = 4, 256, 1024
    top_k = 2
    num_experts = 8

    x = torch.randn(batch, seq, hidden, device='cuda')
    expert_indices = torch.randint(0, num_experts, (batch, seq, top_k), device='cuda')
    expert_weights = torch.softmax(torch.randn(batch, seq, top_k, device='cuda'), dim=-1)
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return []
