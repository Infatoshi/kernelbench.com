import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <float.h>

template <typename T>
__device__ __forceinline__ float to_float(T x) { return (float)x; }
template <> __device__ __forceinline__ float to_float(half x) { return __half2float(x); }
template <> __device__ __forceinline__ float to_float(nv_bfloat16 x) { return __bfloat162float(x); }

template <typename T>
__device__ __forceinline__ T from_float(float x) { return (T)x; }
template <> __device__ __forceinline__ half from_float(float x) { return __float2half(x); }
template <> __device__ __forceinline__ nv_bfloat16 from_float(float x) { return __float2bfloat16(x); }

template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, T* __restrict__ output, int batch_size, int dim) {
    int row = blockIdx.x;
    if (row >= batch_size) return;

    const T* row_input = input + row * dim;
    T* row_output = output + row * dim;

    int tid = threadIdx.x;
    int wid = tid / 32;
    int lane = tid % 32;

    float max_val = -FLT_MAX;
    for (int i = tid; i < dim; i += blockDim.x) {
        max_val = fmaxf(max_val, to_float(row_input[i]));
    }

    // Full reduction for max
    for (int offset = 16; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xFFFFFFFF, max_val, offset));
    }
    
    __shared__ float shared_max;
    __shared__ float shared_sum;
    if (tid == 0) shared_max = max_val;
    __syncthreads();
    
    // Let's use a more robust reduction.
    __shared__ float sdata[32]; 
    
    // Partial max in each warp
    float warp_max = max_val; // max_val already contains result of shfl_down_sync for lane 0
    
    if (lane == 0) sdata[wid] = warp_max;
    __syncthreads();
    
    float block_max = -FLT_MAX;
    if (tid < (blockDim.x / 32)) block_max = sdata[tid];
    else if (tid < 32) block_max = -FLT_MAX;
    
    for (int offset = 16; offset > 0; offset /= 2) {
        block_max = fmaxf(block_max, __shfl_down_sync(0xFFFFFFFF, block_max, offset));
    }
    
    if (tid == 0) shared_max = block_max;
    __syncthreads();
    max_val = shared_max;

    float sum = 0.0f;
    for (int i = tid; i < dim; i += blockDim.x) {
        float val = expf(to_float(row_input[i]) - max_val);
        sum += val;
    }

    // Full reduction for sum
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xFFFFFFFF, sum, offset);
    }
    
    if (lane == 0) sdata[wid] = sum;
    __syncthreads();
    
    float block_sum = 0.0f;
    if (tid < (blockDim.x / 32)) block_sum = sdata[tid];
    else if (tid < 32) block_sum = 0.0f;
    
    for (int offset = 16; offset > 0; offset /= 2) {
        block_sum += __shfl_down_sync(0xFFFFFFFF, block_sum, offset);
    }
    
    if (tid == 0) shared_sum = block_sum;
    __syncthreads();
    sum = shared_sum;

    float inv_sum = 1.0f / sum;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        row_output[i] = from_float<T>(expf(to_float(row_input[i]) - max_val) * inv_sum);
    }
}

torch::Tensor softmax_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int batch_size = input.size(0);
    int dim = input.size(1);
    
    int threads = 1024;
    if (dim < 1024) threads = ((dim + 31) / 32) * 32;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "softmax_forward", ([&] {
        softmax_kernel<scalar_t><<<batch_size, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            dim
        );
    }));

    return output;
}
'''

cpp_source = r'''
torch::Tensor softmax_forward(torch::Tensor input);
'''

ext = load_inline(
    name='softmax_extension',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_forward'],
    with_cuda=True,
    extra_cuda_cflags=['-O3', '--use_fast_math']
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x):
        return ext.softmax_forward(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
