import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAException.h>

// Optimized sum reduction kernel using warp shuffles
// Each warp handles one output element
template<typename T>
__global__ void sum_reduce_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int reduce_size,
    int num_outputs,
    int stride_before,
    int stride_after
) {
    // Each warp handles one output element
    int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / 32;
    int lane_id = threadIdx.x % 32;
    
    if (warp_id >= num_outputs) return;
    
    // Calculate output position
    int out_idx = warp_id;
    int i = out_idx / stride_after;        // index before reduce dim
    int k = out_idx % stride_after;        // index after reduce dim
    
    // Input start position for this reduction
    int input_base = i * reduce_size * stride_after + k;
    
    // Each thread sums multiple elements
    float sum = 0.0f;
    for (int j = lane_id; j < reduce_size; j += 32) {
        sum += static_cast<float>(input[input_base + j * stride_after]);
    }
    
    // Warp reduction using shuffles
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Lane 0 writes the result
    if (lane_id == 0) {
        output[out_idx] = static_cast<T>(sum);
    }
}

// Alternative: block-level reduction for larger reduce dimensions
template<typename T, int BLOCK_SIZE>
__global__ void sum_reduce_block_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int reduce_size,
    int num_outputs,
    int stride_before,
    int stride_after
) {
    __shared__ float sdata[BLOCK_SIZE];
    
    int output_idx = blockIdx.x;
    if (output_idx >= num_outputs) return;
    
    int i = output_idx / stride_after;
    int k = output_idx % stride_after;
    int input_base = i * reduce_size * stride_after + k;
    
    // Each thread sums multiple elements
    float sum = 0.0f;
    for (int j = threadIdx.x; j < reduce_size; j += BLOCK_SIZE) {
        sum += static_cast<float>(input[input_base + j * stride_after]);
    }
    sdata[threadIdx.x] = sum;
    __syncthreads();
    
    // Block reduction
    for (int s = BLOCK_SIZE / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        output[output_idx] = static_cast<T>(sdata[0]);
    }
}

at::Tensor sum_reduce_dim(at::Tensor input, int64_t dim) {
    // Ensure input is contiguous
    auto input_contig = input.contiguous();
    
    // Get dimensions
    auto sizes = input_contig.sizes();
    int ndim = sizes.size();
    
    // Handle negative dimension
    if (dim < 0) dim += ndim;
    
    // Calculate dimensions
    int64_t reduce_size = sizes[dim];
    int64_t stride_before = 1;
    int64_t stride_after = 1;
    
    for (int i = 0; i < dim; i++) {
        stride_before *= sizes[i];
    }
    for (int i = dim + 1; i < ndim; i++) {
        stride_after *= sizes[i];
    }
    
    int64_t num_outputs = stride_before * stride_after;
    
    // Create output shape
    std::vector<int64_t> out_sizes;
    for (int i = 0; i < ndim; i++) {
        if (i == dim) {
            out_sizes.push_back(1);
        } else {
            out_sizes.push_back(sizes[i]);
        }
    }
    
    auto output_options = input_contig.options();
    at::Tensor output;
    
    if (input_contig.scalar_type() == at::kFloat) {
        output = at::empty(out_sizes, output_options);
        int threads = 256;
        int warps_per_block = threads / 32;
        int blocks = (num_outputs + warps_per_block - 1) / warps_per_block;
        
        sum_reduce_kernel<float><<<blocks, threads>>>(
            input_contig.data_ptr<float>(),
            output.data_ptr<float>(),
            reduce_size, num_outputs, stride_before, stride_after
        );
    } else if (input_contig.scalar_type() == at::kHalf) {
        output = at::empty(out_sizes, output_options);
        int threads = 256;
        int warps_per_block = threads / 32;
        int blocks = (num_outputs + warps_per_block - 1) / warps_per_block;
        
        sum_reduce_kernel<at::Half><<<blocks, threads>>>(
            reinterpret_cast<const at::Half*>(input_contig.data_ptr<at::Half>()),
            reinterpret_cast<at::Half*>(output.data_ptr<at::Half>()),
            reduce_size, num_outputs, stride_before, stride_after
        );
    } else if (input_contig.scalar_type() == at::kBFloat16) {
        output = at::empty(out_sizes, output_options);
        int threads = 256;
        int warps_per_block = threads / 32;
        int blocks = (num_outputs + warps_per_block - 1) / warps_per_block;
        
        sum_reduce_kernel<at::BFloat16><<<blocks, threads>>>(
            reinterpret_cast<const at::BFloat16*>(input_contig.data_ptr<at::BFloat16>()),
            reinterpret_cast<at::BFloat16*>(output.data_ptr<at::BFloat16>()),
            reduce_size, num_outputs, stride_before, stride_after
        );
    } else {
        // Fallback for other types
        output = at::empty(out_sizes, output_options);
        int threads = 256;
        int warps_per_block = threads / 32;
        int blocks = (num_outputs + warps_per_block - 1) / warps_per_block;
        
        AT_DISPATCH_FLOATING_TYPES(input_contig.scalar_type(), "sum_reduce_kernel", [&] {
            sum_reduce_kernel<scalar_t><<<blocks, threads>>>(
                input_contig.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                reduce_size, num_outputs, stride_before, stride_after
            );
        });
    }
    
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
at::Tensor sum_reduce_dim(at::Tensor input, int64_t dim);
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduce_dim'],
    verbose=False,
    extra_cuda_cflags=['-O3'],
)

class Model(nn.Module):
    """
    Simple model that performs sum reduction over a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): Dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies sum reduction over the specified dimension.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor after sum reduction, shape (..., 1, ...).
        """
        return ext.sum_reduce_dim(x, self.dim)

batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
