import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

template <typename scalar_t>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ w,
    const scalar_t* __restrict__ b,
    scalar_t* __restrict__ y,
    int N, int C, int H, int W,
    int K, int stride, int padding,
    int Hout, int Wout,
    bool has_bias) {

    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * Hout * Wout;
    if (idx >= total) return;

    int ow = idx % Wout;
    idx /= Wout;
    int oh = idx % Hout;
    idx /= Hout;
    int c = idx % C;
    int n = idx / C;

    float acc = has_bias ? static_cast<float>(b[c]) : 0.0f;

    const int in_h0 = oh * stride - padding;
    const int in_w0 = ow * stride - padding;

    int x_base = ((n * C + c) * H) * W;
    int w_base = c * K * K;

    #pragma unroll
    for (int kh = 0; kh < 8; ++kh) {
        if (kh >= K) break;
        int ih = in_h0 + kh;
        if ((unsigned)ih >= (unsigned)H) continue;
        #pragma unroll
        for (int kw = 0; kw < 8; ++kw) {
            if (kw >= K) break;
            int iw = in_w0 + kw;
            if ((unsigned)iw >= (unsigned)W) continue;
            acc += static_cast<float>(x[x_base + ih * W + iw]) * static_cast<float>(w[w_base + kh * K + kw]);
        }
    }

    y[((n * C + c) * Hout + oh) * Wout + ow] = static_cast<scalar_t>(acc);
}

std::vector<torch::Tensor> depthwise_conv2d_cuda(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor b,
    int64_t stride,
    int64_t padding) {

    const auto N = (int)x.size(0);
    const auto C = (int)x.size(1);
    const auto H = (int)x.size(2);
    const auto W = (int)x.size(3);
    const auto K = (int)w.size(2);
    const auto Hout = (H + 2 * (int)padding - K) / (int)stride + 1;
    const auto Wout = (W + 2 * (int)padding - K) / (int)stride + 1;

    auto y = torch::empty({N, C, Hout, Wout}, x.options());
    const int total = N * C * Hout * Wout;
    const int threads = 256;
    const int blocks = (total + threads - 1) / threads;
    cudaStream_t stream = at::cuda::getDefaultCUDAStream();
    bool has_bias = b.defined() && b.numel() > 0;

    AT_DISPATCH_FLOATING_TYPES_AND2(torch::kHalf, torch::kBFloat16, x.scalar_type(), "depthwise_conv2d_cuda", [&] {
        const scalar_t* bptr = has_bias ? b.data_ptr<scalar_t>() : nullptr;
        depthwise_conv2d_kernel<scalar_t><<<blocks, threads, 0, stream>>>(
            x.data_ptr<scalar_t>(),
            w.data_ptr<scalar_t>(),
            bptr,
            y.data_ptr<scalar_t>(),
            N, C, H, W, K, (int)stride, (int)padding, Hout, Wout, has_bias);
    });

    return {y};
}
'''

cpp_source = r'''
#include <torch/extension.h>
#include <vector>
std::vector<torch::Tensor> depthwise_conv2d_cuda(torch::Tensor x, torch::Tensor w, torch::Tensor b, int64_t stride, int64_t padding);
'''

ext = load_inline(
    name='dwconv2d_ext_fast_v1',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo', '-std=c++17', '-arch=sm_86'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.stride = stride
        self.padding = padding
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        weight = self.conv2d.weight.contiguous()
        bias = self.conv2d.bias.contiguous() if self.conv2d.bias is not None else torch.empty(0, device=x.device, dtype=x.dtype)
        return ext.depthwise_conv2d_cuda(x.contiguous(), weight, bias, self.stride, self.padding)[0]


def get_inputs():
    batch_size = 16
    in_channels = 3
    kernel_size = 3
    width = 256
    height = 256
    stride = 1
    padding = 0
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    batch_size = 16
    in_channels = 3
    kernel_size = 3
    width = 256
    height = 256
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
