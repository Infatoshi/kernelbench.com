import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fused bloom kernel: threshold -> blur -> blend
// Uses shared memory tiling with halo regions for efficient 5x5 average pooling

template<typename scalar_t>
__global__ void bloom_kernel(
    const scalar_t* __restrict__ image,
    scalar_t* __restrict__ output,
    int batch,
    int channels,
    int height,
    int width,
    scalar_t threshold,
    scalar_t intensity,
    int kernel_size,
    int padding
) {
    const int TILE_SIZE = 16;
    const int HALO = padding;  // padding = kernel_size // 2 = 2 for 5x5 kernel
    const int SHARED_SIZE = TILE_SIZE + 2 * HALO;  // 16 + 4 = 20
    
    extern __shared__ float shared_mem[];
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;  // batch * channels
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Each thread is responsible for loading one element in the extended tile
    // Thread (tx, ty) loads the element at shared_mem[ty][tx] which corresponds to
    // global position: start_x + tx - HALO, start_y + ty - HALO
    int start_x = bx * TILE_SIZE;
    int start_y = by * TILE_SIZE;
    
    int load_x = start_x + tx - HALO;
    int load_y = start_y + ty - HALO;
    
    // Load into shared memory
    if (load_x >= 0 && load_x < width && load_y >= 0 && load_y < height) {
        shared_mem[ty * SHARED_SIZE + tx] = float(image[bz * height * width + load_y * width + load_x]);
    } else {
        shared_mem[ty * SHARED_SIZE + tx] = 0.0f;
    }
    
    __syncthreads();
    
    // Now each thread computes output for position (bx*TILE_SIZE + tx, by*TILE_SIZE + ty)
    // But we need threads where tx, ty are in [0, TILE_SIZE) range
    if (tx < TILE_SIZE && ty < TILE_SIZE) {
        int pixel_x = start_x + tx;
        int pixel_y = start_y + ty;
        
        if (pixel_x < width && pixel_y < height) {
            // The value at this pixel is at shared_mem[ty + HALO][tx + HALO]
            float val = shared_mem[(ty + HALO) * SHARED_SIZE + (tx + HALO)];
            
            // Apply 5x5 average pooling (box blur)
            // The kernel window for this pixel spans shared_mem[ty..ty+kernel_size][tx..tx+kernel_size]
            // ky is the row offset, kx is the column offset
            float sum = 0.0f;
            for (int ky = 0; ky < kernel_size; ky++) {
                for (int kx = 0; kx < kernel_size; kx++) {
                    sum += shared_mem[(ty + ky) * SHARED_SIZE + (tx + kx)];
                }
            }
            float blurred = sum / float(kernel_size * kernel_size);
            
            // Blend: output = clamp(image + intensity * blurred, 0, 10)
            float result = val + float(intensity) * blurred;
            result = fminf(fmaxf(result, 0.0f), 10.0f);
            
            output[bz * height * width + pixel_y * width + pixel_x] = scalar_t(result);
        }
    }
}

torch::Tensor bloom_forward(torch::Tensor image, float threshold, float intensity, int kernel_size) {
    int batch = image.size(0);
    int channels = image.size(1);
    int height = image.size(2);
    int width = image.size(3);
    int padding = kernel_size / 2;
    
    auto output = torch::empty_like(image);
    
    const int TILE_SIZE = 16;
    const int SHARED_SIZE = TILE_SIZE + 2 * padding;
    dim3 block(SHARED_SIZE, SHARED_SIZE);
    dim3 grid((width + TILE_SIZE - 1) / TILE_SIZE, 
              (height + TILE_SIZE - 1) / TILE_SIZE,
              batch * channels);
    
    int shared_mem_size = SHARED_SIZE * SHARED_SIZE * sizeof(float);
    
    if (image.dtype() == torch::kFloat32) {
        bloom_kernel<float><<<grid, block, shared_mem_size>>>(
            image.data_ptr<float>(),
            output.data_ptr<float>(),
            batch, channels, height, width,
            threshold, intensity, kernel_size, padding
        );
    } else if (image.dtype() == torch::kFloat16) {
        bloom_kernel<at::Half><<<grid, block, shared_mem_size>>>(
            image.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            batch, channels, height, width,
            threshold, intensity, kernel_size, padding
        );
    } else if (image.dtype() == torch::kBFloat16) {
        bloom_kernel<at::BFloat16><<<grid, block, shared_mem_size>>>(
            image.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            batch, channels, height, width,
            threshold, intensity, kernel_size, padding
        );
    }
    
    return output;
}
'''

ext = load_inline(
    name='bloom_ext',
    cpp_sources='torch::Tensor bloom_forward(torch::Tensor, float, float, int);',
    cuda_sources=cuda_source,
    functions=['bloom_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    """Bloom effect approximation: threshold -> blur -> additive blend."""

    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        return ext.bloom_forward(image, self.threshold, self.intensity, self.blur_kernel)


def get_inputs():
    # HDR-like image input.
    return [torch.randn(2, 3, 1024, 1024) * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
