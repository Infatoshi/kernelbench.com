import torch
import torch.nn as nn
import torch.nn.functional as F
import math

try:
    from fla.ops import chunk_gated_delta_rule
    HAS_FLA = True
except ImportError:
    HAS_FLA = False


class EinsumLinear(nn.Module):
    def __init__(self, in_features, out_features, bias=True):
        super().__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        if bias:
            self.bias = nn.Parameter(torch.empty(out_features))
        else:
            self.register_parameter('bias', None)
        self.reset_parameters()
    
    def reset_parameters(self):
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        if self.bias is not None:
            fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
            bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
            nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        out = torch.einsum('...i,oi->...o', x, self.weight)
        if self.bias is not None:
            out = out + self.bias
        return out


class DepthwiseConv1dSiLU(nn.Module):
    def __init__(self, channels, kernel_size):
        super().__init__()
        self.channels = channels
        self.kernel_size = kernel_size
        self.padding = kernel_size - 1
        self.weight = nn.Parameter(torch.empty(channels, 1, kernel_size))
        self.bias = nn.Parameter(torch.empty(channels))
        self.reset_parameters()
    
    def reset_parameters(self):
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        bound = 1 / math.sqrt(self.channels * self.kernel_size)
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        out = F.conv1d(x, self.weight, self.bias, groups=self.channels, padding=self.padding)
        out = out[:, :, :x.size(2)]
        return F.silu(out)


def gated_delta_attention_fallback(q, k, v, alpha, beta, scale):
    B, H, T, D_qk = q.shape
    D_v = v.shape[-1]
    q_scaled = q * scale
    o = torch.empty_like(v)
    
    for b in range(B):
        for h in range(H):
            S = torch.zeros(D_v, D_qk, device=q.device, dtype=q.dtype)
            for t in range(T):
                k_t = k[b, h, t]
                a_t = alpha[b, h, t]
                b_t = beta[b, h, t]
                v_t = v[b, h, t]
                q_t = q_scaled[b, h, t]
                
                Sk = torch.einsum('ij,j->i', S, k_t)
                delta = Sk - v_t
                S = a_t * S - b_t * torch.einsum('i,j->ij', delta, k_t)
                o[b, h, t] = torch.einsum('ij,j->i', S, q_t)
    
    return o


def gated_delta_attention(q, k, v, alpha, beta, scale):
    if HAS_FLA:
        g = alpha.clamp(min=1e-6).log()
        output, _ = chunk_gated_delta_rule(q, k, v, g, beta, scale=scale)
        return output
    else:
        return gated_delta_attention_fallback(q, k, v, alpha, beta, scale)


class Model(nn.Module):
    def __init__(
        self,
        hidden_size,
        num_heads,
        head_dim_qk,
        head_dim_v,
        use_short_conv=True,
        conv_kernel_size=4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv
        
        self.q_proj = EinsumLinear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = EinsumLinear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = EinsumLinear(hidden_size, num_heads * head_dim_v, bias=False)
        self.a_proj = EinsumLinear(hidden_size, num_heads, bias=True)
        self.b_proj = EinsumLinear(hidden_size, num_heads, bias=True)
        self.o_proj = EinsumLinear(num_heads * head_dim_v, hidden_size, bias=False)
        
        if use_short_conv:
            self.q_conv = DepthwiseConv1dSiLU(num_heads * head_dim_qk, conv_kernel_size)
            self.k_conv = DepthwiseConv1dSiLU(num_heads * head_dim_qk, conv_kernel_size)
            self.v_conv = DepthwiseConv1dSiLU(num_heads * head_dim_v, conv_kernel_size)
        
        self.g_proj = EinsumLinear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5
    
    def forward(self, x):
        B, T, _ = x.shape
        
        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)
        
        if self.use_short_conv:
            q = self.q_conv(q.transpose(1, 2)).transpose(1, 2)
            k = self.k_conv(k.transpose(1, 2)).transpose(1, 2)
            v = self.v_conv(v.transpose(1, 2)).transpose(1, 2)
        
        q = q.view(B, T, self.num_heads, self.head_dim_qk).transpose(1, 2)
        k = k.view(B, T, self.num_heads, self.head_dim_qk).transpose(1, 2)
        v = v.view(B, T, self.num_heads, self.head_dim_v).transpose(1, 2)
        
        alpha = torch.sigmoid(self.a_proj(x)).transpose(1, 2)
        beta = torch.sigmoid(self.b_proj(x)).transpose(1, 2)
        
        o = gated_delta_attention(q, k, v, alpha, beta, self.scale)
        
        o = o.transpose(1, 2)
        o = self.o_norm(o)
        
        g = torch.sigmoid(self.g_proj(x))
        g = g.view(B, T, self.num_heads, self.head_dim_v)
        o = o * g
        
        o = o.reshape(B, T, self.num_heads * self.head_dim_v)
        o = self.o_proj(o)
        
        return o
