import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cmath>

__device__ __forceinline__ float gelu_exact(float x) {
    return 0.5f * x * erfcf(-x * 0.7071067811865475f);
}

__device__ __forceinline__ half gelu_exact_half(half x) {
    float xf = __half2float(x);
    return __float2half(gelu_exact(xf));
}

__device__ __forceinline__ nv_bfloat16 gelu_exact_bf16(nv_bfloat16 x) {
    float xf = __bfloat162float(x);
    return __float2bfloat16(gelu_exact(xf));
}

__global__ void gelu_f32_kernel(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        y[i] = gelu_exact(x[i]);
    }
}

__global__ void gelu_f16_kernel(const half* __restrict__ x, half* __restrict__ y, int n_half) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    const half2* x2 = reinterpret_cast<const half2*>(x);
    half2* y2 = reinterpret_cast<half2*>(y);
    int n2 = n_half >> 1;
    for (int i = idx; i < n2; i += stride) {
        half2 v = x2[i];
        float2 f = __half22float2(v);
        y2[i] = __halves2half2(__float2half(gelu_exact(f.x)), __float2half(gelu_exact(f.y)));
    }
    if ((n_half & 1) && idx == 0) {
        y[n_half - 1] = gelu_exact_half(x[n_half - 1]);
    }
}

__global__ void gelu_bf16_kernel(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        y[i] = gelu_exact_bf16(x[i]);
    }
}

torch::Tensor gelu_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    const int n = x.numel();
    const int threads = 256;
    int blocks = (n + threads - 1) / threads;
    blocks = blocks > 4096 ? 4096 : blocks;
    auto stream = at::cuda::getDefaultCUDAStream();

    if (x.scalar_type() == torch::kFloat32) {
        gelu_f32_kernel<<<blocks, threads, 0, stream>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    } else if (x.scalar_type() == torch::kFloat16) {
        gelu_f16_kernel<<<blocks, threads, 0, stream>>>(reinterpret_cast<half*>(x.data_ptr<at::Half>()), reinterpret_cast<half*>(y.data_ptr<at::Half>()), n);
    } else if (x.scalar_type() == torch::kBFloat16) {
        gelu_bf16_kernel<<<blocks, threads, 0, stream>>>(reinterpret_cast<nv_bfloat16*>(x.data_ptr<at::BFloat16>()), reinterpret_cast<nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
    } else {
        AT_ERROR("Unsupported dtype");
    }
    return y;
}
'''

ext = load_inline(
    name='gelu_ext_opt',
    cpp_sources='torch::Tensor gelu_cuda(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['gelu_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_cuda:
            return torch.nn.functional.gelu(x)
        return ext.gelu_cuda(x)


batch_size = 256
dim = 16384


def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]


def get_init_inputs():
    return []
