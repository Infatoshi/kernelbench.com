import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>

using namespace nvcuda;

#define BLOCK_SIZE_M 128
#define BLOCK_SIZE_N 128
#define BLOCK_SIZE_K 32

__global__ void wmma_gemm_kernel(const half *A, const half *B, float *C, int N) {
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int warp_id = threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    
    // 2x4 warp grid: 2 warps along M, 4 along N
    const int warps_m = 2;
    const int warps_n = 4;
    int warp_m = warp_id / warps_n;
    int warp_n = warp_id % warps_n;
    
    // Shared memory
    __shared__ half s_a[BLOCK_SIZE_M * BLOCK_SIZE_K];
    __shared__ half s_b[BLOCK_SIZE_K * BLOCK_SIZE_N];
    
    // WMMA fragments
    wmma::fragment<wmma::accumulator, 16, 16, 16, float> acc[4][2];
    wmma::fragment<wmma::matrix_a, 16, 16, 16, half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, 16, 16, 16, half, wmma::col_major> b_frag;
    
    // Initialize accumulators
    #pragma unroll
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 2; j++) {
            wmma::fill_fragment(acc[i][j], 0.0f);
        }
    }
    
    // Loop over K
    for (int k = 0; k < N; k += BLOCK_SIZE_K) {
        // Load A to shared memory
        for (int i = threadIdx.x; i < BLOCK_SIZE_M * BLOCK_SIZE_K; i += blockDim.x) {
            int row = i / BLOCK_SIZE_K;
            int col = i % BLOCK_SIZE_K;
            int g_row = by * BLOCK_SIZE_M + row;
            int g_col = k + col;
            if (g_row < N && g_col < N) {
                s_a[row * BLOCK_SIZE_K + col] = A[g_row * N + g_col];
            } else {
                s_a[row * BLOCK_SIZE_K + col] = __float2half(0.0f);
            }
        }
        
        // Load B to shared memory  
        for (int i = threadIdx.x; i < BLOCK_SIZE_K * BLOCK_SIZE_N; i += blockDim.x) {
            int row = i / BLOCK_SIZE_N;
            int col = i % BLOCK_SIZE_N;
            int g_row = k + row;
            int g_col = bx * BLOCK_SIZE_N + col;
            if (g_row < N && g_col < N) {
                s_b[row * BLOCK_SIZE_N + col] = B[g_row * N + g_col];
            } else {
                s_b[row * BLOCK_SIZE_N + col] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute - each warp handles 64x64 output tile using 16x16x16 WMMA
        // Warp layout within block: 2x4 warps, each warp handles 64x32 output
        for (int k_step = 0; k_step < BLOCK_SIZE_K; k_step += 16) {
            for (int i = 0; i < 4; i++) {  // 4 tiles along M (64 / 16)
                for (int j = 0; j < 2; j++) {  // 2 tiles along N (32 / 16)
                    int row_offset = warp_m * 64 + i * 16;
                    int col_offset = warp_n * 32 + j * 16;
                    
                    wmma::load_matrix_sync(a_frag, s_a + row_offset * BLOCK_SIZE_K + k_step, BLOCK_SIZE_K);
                    wmma::load_matrix_sync(b_frag, s_b + k_step * BLOCK_SIZE_N + col_offset, BLOCK_SIZE_N);
                    wmma::mma_sync(acc[i][j], a_frag, b_frag, acc[i][j]);
                }
            }
        }
        
        __syncthreads();
    }
    
    // Store to global memory
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 2; j++) {
            int row = by * BLOCK_SIZE_M + warp_m * 64 + i * 16;
            int col = bx * BLOCK_SIZE_N + warp_n * 32 + j * 16;
            if (row < N && col < N) {
                wmma::store_matrix_sync(C + row * N + col, acc[i][j], N, wmma::mem_row_major);
            }
        }
    }
}

torch::Tensor wmma_gemm(torch::Tensor A, torch::Tensor B) {
    int N = A.size(0);
    auto C = torch::zeros({N, N}, torch::TensorOptions().dtype(torch::kFloat32).device(A.device()));
    
    dim3 block(256);
    dim3 grid((N + BLOCK_SIZE_N - 1) / BLOCK_SIZE_N, (N + BLOCK_SIZE_M - 1) / BLOCK_SIZE_M);
    
    wmma_gemm_kernel<<<grid, block>>>(
        reinterpret_cast<const half*>(A.data_ptr<at::Half>()),
        reinterpret_cast<const half*>(B.data_ptr<at::Half>()),
        C.data_ptr<float>(),
        N
    );
    
    return C;
}
'''

cpp_source = '''
torch::Tensor wmma_gemm(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='wmma_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['wmma_gemm'],
    extra_cuda_cflags=['-arch=sm_86', '-std=c++17', '-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.wmma_gemm(A.half(), B.half())

def get_inputs():
    A = torch.randn(2048, 2048)
    B = torch.randn(2048, 2048)
    return [A, B]

def get_init_inputs():
    return []
