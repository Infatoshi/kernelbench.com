import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define BM 64
#define BN 64
#define BK 16

__global__ void gemm_kernel(const float* __restrict__ A, const float* __restrict__ B, float* __restrict__ C,
                            int M, int N, int K) {
    __shared__ float As[BM][BK];
    __shared__ float Bs[BK][BN];

    int tx = threadIdx.x; // 0..15
    int ty = threadIdx.y; // 0..15

    int block_row = blockIdx.y * BM;
    int block_col = blockIdx.x * BN;

    // load A (64x16) and B (16x64)
    for (int i = ty * blockDim.x + tx; i < BM*BK; i += blockDim.x*blockDim.y) {
        int r = i / BK;
        int c = i % BK;
        As[r][c] = A[(block_row + r) * K + c];
    }

    for (int i = ty * blockDim.x + tx; i < BK*BN; i += blockDim.x*blockDim.y) {
        int r = i / BN;
        int c = i % BN;
        Bs[r][c] = B[r * N + (block_col + c)];
    }

    __syncthreads();

    float acc[4][4] = {0};

    int row_base = block_row + ty * 4;
    int col_base = block_col + tx * 4;

    #pragma unroll
    for (int k = 0; k < BK; ++k) {
        float a0 = As[ty*4+0][k];
        float a1 = As[ty*4+1][k];
        float a2 = As[ty*4+2][k];
        float a3 = As[ty*4+3][k];

        float b0 = Bs[k][tx*4+0];
        float b1 = Bs[k][tx*4+1];
        float b2 = Bs[k][tx*4+2];
        float b3 = Bs[k][tx*4+3];

        acc[0][0] += a0*b0; acc[0][1] += a0*b1; acc[0][2] += a0*b2; acc[0][3] += a0*b3;
        acc[1][0] += a1*b0; acc[1][1] += a1*b1; acc[1][2] += a1*b2; acc[1][3] += a1*b3;
        acc[2][0] += a2*b0; acc[2][1] += a2*b1; acc[2][2] += a2*b2; acc[2][3] += a2*b3;
        acc[3][0] += a3*b0; acc[3][1] += a3*b1; acc[3][2] += a3*b2; acc[3][3] += a3*b3;
    }

    #pragma unroll
    for (int i = 0; i < 4; ++i) {
        #pragma unroll
        for (int j = 0; j < 4; ++j) {
            int r = row_base + i;
            int c = col_base + j;
            C[r * N + c] = acc[i][j];
        }
    }
}

torch::Tensor gemm(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);

    auto C = torch::empty({M, N}, A.options());

    dim3 block(16,16);
    dim3 grid(N/BN, M/BM);

    gemm_kernel<<<grid, block>>>(A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(), M, N, K);

    return C;
}
'''

ext = load_inline(
    name='ts_gemm',
    cpp_sources='torch::Tensor gemm(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemm'],
    extra_cuda_cflags=['-O3','-arch=sm_86'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A, B):
        A = A.contiguous()
        B = B.contiguous()
        return ext.gemm(A, B)

M = 16384
N = 16

def get_inputs():
    A = torch.randn(M, N, device='cuda', dtype=torch.float32)
    B = torch.randn(N, M, device='cuda', dtype=torch.float32)
    return [A, B]

def get_init_inputs():
    return []
