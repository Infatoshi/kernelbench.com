import torch
import torch.nn as nn
import torch.nn.functional as F
import math

OP_TYPE = "model"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


class OptimizedGELU(nn.Module):
    """Optimized GELU using PyTorch's native gelu"""
    def __init__(self):
        super().__init__()
    
    def forward(self, x):
        # Use PyTorch's optimized GELU (approximate gelu is faster)
        return F.gelu(x)


class CausalSelfAttention(nn.Module):
    """
    Optimized causal self-attention using SDPA
    """

    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        # key, query, value projections for all heads, but in a batch
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        # output projection
        self.c_proj = nn.Linear(n_embd, n_embd)
        # regularization - using dropout during eval doesn't matter
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.n_head = n_head
        self.n_embd = n_embd
        self.head_dim = n_embd // n_head

    def forward(self, x):
        B, T, C = x.size()

        # calculate query, key, values for all heads in batch and move head forward to be the batch dim
        qkv = self.c_attn(x)
        q, k, v = qkv.split(self.n_embd, dim=2)
        
        # Reshape for attention: (B, T, nh, hs) -> (B, nh, T, hs)
        q = q.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        k = k.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        v = v.view(B, T, self.n_head, self.head_dim).transpose(1, 2)

        # causal mask - create on the fly for the actual sequence length
        # UsingSDPA with is_causal=True handles this efficiently
        # The scale is 1/sqrt(head_dim)
        scale = 1.0 / math.sqrt(self.head_dim)
        
        # Use scaled_dot_product_attention (Flash Attention / memory-efficient attention)
        # This is much faster than manual softmax implementation
        attn = F.scaled_dot_product_attention(
            q, k, v,
            attn_mask=None,
            is_causal=True,
            scale=scale,
            dropout_p=self.attn_dropout.p if self.training else 0.0
        )
        
        # Apply dropout during training
        if self.training:
            attn = self.attn_dropout(attn)
        
        # Reshape back: (B, nh, T, hs) -> (B, T, C)
        attn = attn.transpose(1, 2).contiguous().view(B, T, C)

        # output projection
        attn = self.c_proj(attn)
        
        # Apply dropout during training
        if self.training:
            attn = self.resid_dropout(attn)
        
        return attn


class Model(nn.Module):
    """ an unassuming Transformer block """

    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        self.ln_1 = nn.LayerNorm(n_embd)
        self.attn = CausalSelfAttention(n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen)
        self.ln_2 = nn.LayerNorm(n_embd)
        
        # Use separate modules for clarity
        self.c_fc = nn.Linear(n_embd, 4 * n_embd)
        self.c_proj = nn.Linear(4 * n_embd, n_embd)
        self.act = OptimizedGELU()
        self.dropout = nn.Dropout(resid_pdrop)

    def forward(self, x):
        # Self-attention with residual
        x = x + self.attn(self.ln_1(x))
        # MLP with residual
        x = x + self.dropout(self.c_proj(self.act(self.c_fc(self.ln_2(x)))))
        return x


batch_size = 128
max_seqlen = 1024
seq_len = 512
n_embd = 768
n_head = 8
attn_pdrop = 0.0
resid_pdrop = 0.0


def get_inputs():
    return [torch.randn(batch_size, seq_len, n_embd)]


def get_init_inputs():
    return [n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen]
