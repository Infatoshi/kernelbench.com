import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Warp-level max reduction
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, offset));
    }
    return val;
}

// Warp-level sum reduction  
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        val = val + __shfl_xor_sync(0xffffffff, val, offset);
    }
    return val;
}

// Float/double kernel
template <typename T>
__global__ void online_softmax_kernel(const T* __restrict__ x, T* __restrict__ y,
                                       const int rows, const int cols) {
    const int tid = threadIdx.x;
    const int row = blockIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    if (row >= rows) return;
    
    const T* x_row = x + row * cols;
    T* y_row = y + row * cols;
    
    // Phase 1: Find row max using warp reductions
    float row_max = -INFINITY;
    
    // Each warp handles 32 elements
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        row_max = fmaxf(row_max, (float)x_row[col]);
    }
    
    // Reduce across warps in the block
    row_max = warp_reduce_max(row_max);
    __shared__ float shared_max[32];
    if (lane_id == 0) {
        shared_max[warp_id] = row_max;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        row_max = (lane_id < blockDim.x / 32) ? shared_max[lane_id] : -INFINITY;
        row_max = warp_reduce_max(row_max);
    }
    
    // Broadcast the max to all warps
    if (warp_id == 0) {
        row_max = __shfl_sync(0xffffffff, row_max, 0);
    }
    row_max = __shfl_sync(0xffffffff, row_max, warp_id);
    
    // Phase 2: Compute exp(x - max) and sum
    float row_sum = 0.0f;
    
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        float exp_val = expf((float)x_row[col] - row_max);
        y_row[col] = (T)exp_val;
        row_sum += exp_val;
    }
    
    // Reduce sums across warps
    row_sum = warp_reduce_sum(row_sum);
    if (lane_id == 0) {
        shared_max[warp_id] = row_sum;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        row_sum = (lane_id < blockDim.x / 32) ? shared_max[lane_id] : 0.0f;
        row_sum = warp_reduce_sum(row_sum);
    }
    
    // Broadcast sum to all warps and normalize
    row_sum = __shfl_sync(0xffffffff, row_sum, 0);
    row_sum = __shfl_sync(0xffffffff, row_sum, warp_id);
    
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        y_row[col] = (T)((float)y_row[col] / row_sum);
    }
}

// Half precision version
__global__ void online_softmax_kernel_half(const __half* __restrict__ x, __half* __restrict__ y,
                                            const int rows, const int cols) {
    const int tid = threadIdx.x;
    const int row = blockIdx.x;
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    if (row >= rows) return;
    
    const __half* x_row = x + row * cols;
    __half* y_row = y + row * cols;
    
    // Phase 1: Find row max
    float row_max = -INFINITY;
    
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        row_max = fmaxf(row_max, __half2float(x_row[col]));
    }
    
    row_max = warp_reduce_max(row_max);
    __shared__ float shared_max[32];
    if (lane_id == 0) {
        shared_max[warp_id] = row_max;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        row_max = (lane_id < blockDim.x / 32) ? shared_max[lane_id] : -INFINITY;
        row_max = warp_reduce_max(row_max);
    }
    
    row_max = __shfl_sync(0xffffffff, row_max, 0);
    row_max = __shfl_sync(0xffffffff, row_max, warp_id);
    
    // Phase 2: Compute exp and sum
    float row_sum = 0.0f;
    
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        float exp_val = expf(__half2float(x_row[col]) - row_max);
        y_row[col] = __float2half(exp_val);
        row_sum += exp_val;
    }
    
    row_sum = warp_reduce_sum(row_sum);
    if (lane_id == 0) {
        shared_max[warp_id] = row_sum;
    }
    __syncthreads();
    
    if (warp_id == 0) {
        row_sum = (lane_id < blockDim.x / 32) ? shared_max[lane_id] : 0.0f;
        row_sum = warp_reduce_sum(row_sum);
    }
    
    row_sum = __shfl_sync(0xffffffff, row_sum, 0);
    row_sum = __shfl_sync(0xffffffff, row_sum, warp_id);
    
    for (int col = warp_id * 32 + lane_id; col < cols; col += (blockDim.x / 32) * 32) {
        float val = __half2float(y_row[col]) / row_sum;
        y_row[col] = __float2half(val);
    }
}

torch::Tensor online_softmax_cuda(torch::Tensor x) {
    auto y = torch::empty_like(x);
    
    const int rows = x.size(0);
    const int cols = x.size(1);
    
    const int threads = 256;
    const int blocks = rows;
    
    if (x.scalar_type() == c10::ScalarType::Half) {
        online_softmax_kernel_half<<<blocks, threads>>>(
            (const __half*)x.data_ptr<c10::Half>(),
            (__half*)y.data_ptr<c10::Half>(),
            rows, cols
        );
    } else if (x.scalar_type() == c10::ScalarType::Float) {
        online_softmax_kernel<float><<<blocks, threads>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            rows, cols
        );
    } else if (x.scalar_type() == c10::ScalarType::Double) {
        online_softmax_kernel<double><<<blocks, threads>>>(
            x.data_ptr<double>(),
            y.data_ptr<double>(),
            rows, cols
        );
    }
    
    return y;
}
'''

ext = load_inline(
    name='online_softmax_cuda',
    cpp_sources='torch::Tensor online_softmax_cuda(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['online_softmax_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.online_softmax_cuda(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
