import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size
        self.fused_pool = nn.MaxPool3d(kernel_size=pool_kernel_size**2, stride=pool_kernel_size**2)

    def forward(self, x):
        x = self.conv(x)
        # Using the actual logic: softmax then maxpool.
        # Since the softmax sum is constant for all channels at a pixel, 
        # and exp(x) is monotonic, we can potentially optimize but 
        # only if we go into custom kernels.
        # Let's keep it clean.
        x = torch.softmax(x, dim=1)
        x = self.fused_pool(x)
        return x
