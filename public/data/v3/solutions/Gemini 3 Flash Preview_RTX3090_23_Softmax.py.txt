import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <float.h>

template <typename T>
__device__ __forceinline__ T inf();

template <>
__device__ __forceinline__ float inf<float>() { return 1e38f; }

template <>
__device__ __forceinline__ half inf<half>() { return 65504.0; }

template <>
__device__ __forceinline__ nv_bfloat16 inf<nv_bfloat16>() { return 3.38953138925153547590470800371487866880e+38; }

template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, T* __restrict__ output, int batch_size, int dim) {
    int row = blockIdx.x;
    if (row >= batch_size) return;

    const T* row_input = input + row * dim;
    T* row_output = output + row * dim;

    float max_val = -1e38f;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        max_val = fmaxf(max_val, (float)row_input[i]);
    }

    // Warp reduction for max
    for (int offset = 16; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }

    static __shared__ float shared_max;
    if (threadIdx.x == 0) shared_max = max_val;
    __syncthreads();
    max_val = shared_max;

    float sum = 0.0f;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        float val = expf((float)row_input[i] - max_val);
        sum += val;
    }

    // Warp reduction for sum
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }

    static __shared__ float shared_sum;
    if (threadIdx.x == 0) shared_sum = sum;
    __syncthreads();
    sum = shared_sum;

    float inv_sum = 1.0f / sum;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        row_output[i] = (T)(expf((float)row_input[i] - max_val) * inv_sum);
    }
}

torch::Tensor softmax_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int batch_size = input.size(0);
    int dim = input.size(1);

    int threads = 256;
    if (dim < 256) threads = 32;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "softmax_forward", ([&] {
        softmax_kernel<scalar_t><<<batch_size, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            dim
        );
    }));

    return output;
}

''',
ext = load_inline(
    name='softmax_ext',
    cpp_sources=['torch::Tensor softmax_forward(torch::Tensor input);'],
    cuda_sources=[cuda_source],
    functions=['softmax_forward'],
    with_cuda=True,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x):
        return ext.softmax_forward(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
