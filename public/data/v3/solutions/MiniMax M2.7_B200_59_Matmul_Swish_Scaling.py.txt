import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

// FP32 swish + scale kernel
__global__ void swish_scale_fp32_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    float scale_factor,
    int num_elements) {
    
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = gridDim.x * blockDim.x;
    
    for (int i = idx; i < num_elements; i += stride) {
        float x = input[i];
        // Swish: x * sigmoid(x)
        float sigmoid_x = 1.0f / (1.0f + expf(-x));
        output[i] = x * sigmoid_x * scale_factor;
    }
}

//cublas handle - created once and reused
static cublasHandle_t get_cublas_handle() {
    static bool initialized = false;
    static cublasHandle_t handle;
    if (!initialized) {
        cublasCreate(&handle);
        initialized = true;
    }
    return handle;
}

// FP32 implementation - fully optimized
torch::Tensor fused_matmul_swish_scale_fp32(
    torch::Tensor input, 
    torch::Tensor weight, 
    torch::Tensor bias,
    float scale_factor) {
    
    int m = input.size(0);
    int k = input.size(1);
    int n = weight.size(0);
    
    auto gemm_output = torch::empty({m, n}, input.options());
    
    // cuBLAS matmul: C = alpha * A^T * B + beta * C
    // weight is (n, k), input is (m, k), output is (m, n)
    float alpha = 1.0f;
    float beta = 0.0f;
    
    cublasHandle_t handle = get_cublas_handle();
    
    cublasSgemm(
        handle,
        CUBLAS_OP_T,  // Transpose weight
        CUBLAS_OP_N,  // No transpose input
        n, m, k,      // (n,k) x (k,m) = (n,m), result transposed to (m,n)
        &alpha,
        weight.data_ptr<float>(), k,
        input.data_ptr<float>(), k,
        &beta,
        gemm_output.data_ptr<float>(), n
    );
    
    // Add bias
    if (bias.numel() > 0) {
        gemm_output = gemm_output + bias.view({1, n});
    }
    
    // Swish + scale
    auto output = torch::empty_like(gemm_output);
    int num_elements = gemm_output.numel();
    int block_size = 256;
    int num_blocks = (num_elements + block_size - 1) / block_size;
    
    swish_scale_fp32_kernel<<<num_blocks, block_size>>>(
        gemm_output.data_ptr<float>(),
        output.data_ptr<float>(),
        scale_factor,
        num_elements
    );
    
    return output;
}

        handle,
        CUBLAS_OP_T,
        CUBLAS_OP_N,
        n, m, k,
        &alpha,
        weight_f32.data_ptr<float>(), k,
        input_f32.data_ptr<float>(), k,
        &beta,
        gemm_output.data_ptr<float>(), n
    );
    
    // Add bias

        gemm_output = gemm_output + bias_f32.view({1, n});
    }
    
    // Swish + scale
    auto output_f32 = torch::empty_like(gemm_output);
    int num_elements = gemm_output.numel();

}

'''

cpp_source = r'''
torch::Tensor fused_matmul_swish_scale_fp32(
    torch::Tensor input, 
    torch::Tensor weight, 
    torch::Tensor bias,
    float scale_factor);

torch::Tensor fused_matmul_swish_scale_fp16(
    torch::Tensor input, 
    torch::Tensor weight, 
    torch::Tensor bias,
    float scale_factor);

torch::Tensor fused_matmul_swish_scale_bf16(
    torch::Tensor input, 
    torch::Tensor weight, 
    torch::Tensor bias,
    float scale_factor);
'''

ext = load_inline(
    name='matmul_swish_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=[
        'fused_matmul_swish_scale_fp32',
        'fused_matmul_swish_scale_fp16',
        'fused_matmul_swish_scale_bf16',
    ],
    verbose=False,
)

class Model(nn.Module):
    """
    Fused model: matmul + swish activation + scaling
    Optimized with cuBLAS for GEMM and custom CUDA kernel for activation.
    """
    def __init__(self, in_features, out_features, scaling_factor, precision='fp32'):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
        self.precision = precision
        
    def forward(self, x):
        weight = self.linear.weight
        bias = self.linear.bias if self.linear.bias is not None else torch.zeros(weight.size(0), device=x.device, dtype=x.dtype)
        
        if x.dtype == torch.float16:
            return ext.fused_matmul_swish_scale_fp16(x, weight, bias, self.scaling_factor)
        elif x.dtype == torch.bfloat16:
            return ext.fused_matmul_swish_scale_bf16(x, weight, bias, self.scaling_factor)
        else:
            return ext.fused_matmul_swish_scale_fp32(x, weight, bias, self.scaling_factor)


def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 2.0]
