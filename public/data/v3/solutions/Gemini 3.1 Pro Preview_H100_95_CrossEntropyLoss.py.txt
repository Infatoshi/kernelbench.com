import torch
from torch.utils.cpp_extension import load_inline
import torch.nn as nn

cpp_src = """
#include <torch/extension.h>
torch::Tensor cross_entropy_forward(torch::Tensor preds, torch::Tensor targets);
"""

cuda_src = """
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename T>
__global__ void cross_entropy_kernel(const T* __restrict__ preds, const int64_t* __restrict__ targets, float* __restrict__ loss, int batch_size, int num_classes) {
    int row = blockIdx.x;
    int tid = threadIdx.x;

    if (row >= batch_size) return;

    extern __shared__ float s_row[]; // size = num_classes * float

    const T* row_preds = &preds[row * num_classes];
    int target = targets[row];

    float local_max = -1e38f;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        float val = static_cast<float>(row_preds[i]);
        s_row[i] = val;
        if (val > local_max) local_max = val;
    }

    // Warp reduce max
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        local_max = fmaxf(local_max, __shfl_down_sync(0xffffffff, local_max, offset));
    }
    
    __shared__ float s_reduce[32];
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    if (lane_id == 0) s_reduce[warp_id] = local_max;
    __syncthreads();

    float block_max = -1e38f;
    if (warp_id == 0) {
        block_max = (tid < (blockDim.x + 31) / 32) ? s_reduce[tid] : -1e38f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            block_max = fmaxf(block_max, __shfl_down_sync(0xffffffff, block_max, offset));
        }
        if (lane_id == 0) s_reduce[0] = block_max; 
    }
    __syncthreads();
    block_max = s_reduce[0];

    float local_sum = 0.0f;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        float val = s_row[i];
        local_sum += expf(val - block_max);
    }

    // Warp reduce sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    
    if (lane_id == 0) s_reduce[warp_id] = local_sum;
    __syncthreads();

    if (warp_id == 0) {
        float block_sum = (tid < (blockDim.x + 31) / 32) ? s_reduce[tid] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            block_sum += __shfl_down_sync(0xffffffff, block_sum, offset);
        }
        if (lane_id == 0) {
            float target_val = s_row[target];
            float row_loss = -target_val + block_max + logf(block_sum);
            atomicAdd(loss, row_loss / batch_size);
        }
    }
}

torch::Tensor cross_entropy_forward(torch::Tensor preds, torch::Tensor targets) {
    int batch_size = preds.size(0);
    int num_classes = preds.size(1);
    
    auto opts = preds.options().dtype(torch::kFloat32);
    auto loss = torch::zeros({1}, opts);
    
    int threads = 256;
    int blocks = batch_size;
    int shared_mem = num_classes * sizeof(float);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, preds.scalar_type(), "cross_entropy_kernel", ([&] {
        cross_entropy_kernel<scalar_t><<<blocks, threads, shared_mem>>>(
            preds.data_ptr<scalar_t>(),
            targets.data_ptr<int64_t>(),
            loss.data_ptr<float>(),
            batch_size,
            num_classes
        );
    }));
    
    return loss[0].to(preds.dtype());
}
"""

module = load_inline(
    name='cross_entropy',
    cpp_sources=cpp_src,
    cuda_sources=cuda_src,
    functions=['cross_entropy_forward'],
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '--use_fast_math'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return module.cross_entropy_forward(predictions, targets)

def get_inputs():
    return [torch.randn(4096, 1024, device='cuda', dtype=torch.float32), torch.randint(0, 1024, (4096,), device='cuda')]

def get_init_inputs():
    return []
