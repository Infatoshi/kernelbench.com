import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv_instancenorm_divide_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_c, int out_c, int in_h, int in_w,
    int out_h, int out_w, int k_h, int k_w,
    float divide_by, float eps)
{
    int b = blockIdx.x;
    int oc = blockIdx.y;
    int num_pixels = out_h * out_w;

    double sum = 0;
    double sum_sq = 0;

    for (int i = threadIdx.x; i < num_pixels; i += blockDim.x) {
        int oh = i / out_w;
        int ow = i % out_w;
        
        float val = bias[oc];
        for (int ic = 0; ic < in_c; ++ic) {
            const float* in_ptr = &input[((b * in_c + ic) * in_h + oh) * in_w + ow];
            const float* w_ptr = &weight[((oc * in_c + ic) * k_h) * k_w];
            
            if (k_h == 3 && k_w == 3) {
                val += in_ptr[0] * w_ptr[0];
                val += in_ptr[1] * w_ptr[1];
                val += in_ptr[2] * w_ptr[2];
                val += in_ptr[in_w] * w_ptr[3];
                val += in_ptr[in_w + 1] * w_ptr[4];
                val += in_ptr[in_w + 2] * w_ptr[5];
                val += in_ptr[2 * in_w] * w_ptr[6];
                val += in_ptr[2 * in_w + 1] * w_ptr[7];
                val += in_ptr[2 * in_w + 2] * w_ptr[8];
            } else {
                for (int kh = 0; kh < k_h; ++kh) {
                    for (int kw = 0; kw < k_w; ++kw) {
                        val += in_ptr[kh * in_w + kw] * w_ptr[kh * k_w + kw];
                    }
                }
            }
        }
        output[((b * out_c + oc) * out_h + oh) * out_w + ow] = val;
        double val_double = (double)val;
        sum += val_double;
        sum_sq += val_double * val_double;
    }

    // Warp-level reduction
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
        sum_sq += __shfl_down_sync(0xffffffff, sum_sq, offset);
    }
    
    __shared__ double s_sum[8];
    __shared__ double s_sum_sq[8];
    if ((threadIdx.x & 31) == 0) {
        s_sum[threadIdx.x / 32] = sum;
        s_sum_sq[threadIdx.x / 32] = sum_sq;
    }
    __syncthreads();
    
    if (threadIdx.x < 32) {
        double final_sum = (threadIdx.x < 8) ? s_sum[threadIdx.x] : 0.0;
        double final_sum_sq = (threadIdx.x < 8) ? s_sum_sq[threadIdx.x] : 0.0;
        for (int offset = 16; offset > 0; offset /= 2) {
            final_sum += __shfl_down_sync(0xffffffff, final_sum, offset);
            final_sum_sq += __shfl_down_sync(0xffffffff, final_sum_sq, offset);
        }
        if (threadIdx.x == 0) {
            s_sum[0] = final_sum;
            s_sum_sq[0] = final_sum_sq;
        }
    }
    __syncthreads();

    double mean = s_sum[0] / num_pixels;
    double var = s_sum_sq[0] / num_pixels - mean * mean;
    if (var < 0) var = 0;
    float factor = (1.0f / sqrtf((float)var + eps)) / divide_by;

    for (int i = threadIdx.x; i < num_pixels; i += blockDim.x) {
        int oh = i / out_w;
        int ow = i % out_w;
        int idx = ((b * out_c + oc) * out_h + oh) * out_w + ow;
        output[idx] = (output[idx] - (float)mean) * factor;
    }
}

torch::Tensor forward_fused(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float divide_by) {
    int b = input.size(0);
    int in_c = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_c = weight.size(0);
    int k_h = weight.size(2);
    int k_w = weight.size(3);
    int out_h = in_h - k_h + 1;
    int out_w = in_w - k_w + 1;

    auto output = torch::empty({b, out_c, out_h, out_w}, input.options());
    
    dim3 grid(b, out_c);
    dim3 block(256);
    
    conv_instancenorm_divide_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        b, in_c, out_c, in_h, in_w, out_h, out_w, k_h, k_w,
        divide_by, 1e-5f
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor forward_fused(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float divide_by);
'''

ext = load_inline(
    name='fused_op_v5',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['forward_fused'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = float(divide_by)

    def forward(self, x):
        return ext.forward_fused(
            x.contiguous(), 
            self.conv.weight.contiguous(), 
            self.conv.bias.contiguous() if self.conv.bias is not None else torch.zeros(self.conv.out_channels, device=x.device), 
            self.divide_by
        )

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0]
