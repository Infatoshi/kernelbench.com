import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_conv_bn_scale_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    batch_size, in_channels, out_channels,
    in_h, in_w, out_h, out_w,
    kernel_size,
    stride_in_n, stride_in_c, stride_in_h, stride_in_w,
    stride_w_oc, stride_w_ic, stride_w_kh, stride_w_kw,
    stride_out_n, stride_out_oc, stride_out_oh, stride_out_ow,
    BLOCK_SIZE_OH: tl.constexpr, BLOCK_SIZE_OW: tl.constexpr,
    BLOCK_SIZE_IC: tl.constexpr
):
    pid_oh = tl.program_id(0)
    pid_ow = tl.program_id(1)
    pid_oc_n = tl.program_id(2)
    
    pid_oc = pid_oc_n // batch_size
    pid_n = pid_oc_n % batch_size

    oh_start = pid_oh * BLOCK_SIZE_OH
    ow_start = pid_ow * BLOCK_SIZE_OW
    
    oh_offsets = oh_start + tl.arange(0, BLOCK_SIZE_OH)
    ow_offsets = ow_start + tl.arange(0, BLOCK_SIZE_OW)

    acc = tl.zeros((BLOCK_SIZE_OH, BLOCK_SIZE_OW), dtype=tl.float32)

    w_base_ptr = weight_ptr + pid_oc * stride_w_oc
    in_base_ptr = input_ptr + pid_n * stride_in_n

    for kh in range(kernel_size):
        for kw in range(kernel_size):
            # Inner loop over input channels
            for ic_start in range(0, in_channels, BLOCK_SIZE_IC):
                ic_offsets = ic_start + tl.arange(0, BLOCK_SIZE_IC)
                ic_mask = ic_offsets < in_channels
                
                # Load weight tile: [BLOCK_SIZE_IC]
                w_ptrs = w_base_ptr + ic_offsets * stride_w_ic + kh * stride_w_kh + kw * stride_w_kw
                w_vals = tl.load(w_ptrs, mask=ic_mask, other=0.0)
                
                # Load input tile
                in_h_idx = oh_offsets[:, None, None] + kh
                in_w_idx = ow_offsets[None, :, None] + kw
                ic_idx = ic_offsets[None, None, :]
                
                in_ptrs = in_base_ptr + ic_idx * stride_in_c + in_h_idx * stride_in_h + in_w_idx * stride_in_w
                
                # Since in_h and in_w are likely same and blocks are aligned we can optimize mask
                in_mask = (in_h_idx < in_h) & (in_w_idx < in_w) & (ic_idx < in_channels)
                x_vals = tl.load(in_ptrs, mask=in_mask, other=0.0)

                # Vectorized multiply and sum
                acc += tl.sum(x_vals * w_vals[None, None, :], axis=2)

    # Add bias
    bias = tl.load(bias_ptr + pid_oc)
    acc += bias

    # Store output tile
    out_mask = (oh_offsets[:, None] < out_h) & (ow_offsets[None, :] < out_w)
    out_ptr = output_ptr + pid_n * stride_out_n + pid_oc * stride_out_oc + oh_offsets[:, None] * stride_out_oh + ow_offsets[None, :] * stride_out_ow
    tl.store(out_ptr, acc, mask=out_mask)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        self.kernel_size = kernel_size
        self.out_channels = out_channels
        self.in_channels = in_channels

    def forward(self, x):
        # Weight folding logic remains the same
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        eps = self.bn.eps
        bn_weight = self.bn.weight
        bn_bias = self.bn.bias
        
        scale_factors = (bn_weight * self.scaling_factor) / torch.sqrt(running_var + eps)
        fused_weight = self.conv.weight * scale_factors.reshape(-1, 1, 1, 1)
        fused_bias = (self.conv.bias - running_mean) * scale_factors + (bn_bias * self.scaling_factor)

        batch_size, _, in_h, in_w = x.shape
        out_h = in_h - self.kernel_size + 1
        out_w = in_w - self.kernel_size + 1
        
        output = torch.empty((batch_size, self.out_channels, out_h, out_w), device=x.device, dtype=x.dtype)
        
        BLOCK_SIZE_OH = 32
        BLOCK_SIZE_OW = 32
        # Power of 2 nearest to in_channels
        BLOCK_SIZE_IC = 8 if self.in_channels <= 8 else 16
        
        grid = (
            (out_h + BLOCK_SIZE_OH - 1) // BLOCK_SIZE_OH,
            (out_w + BLOCK_SIZE_OW - 1) // BLOCK_SIZE_OW,
            self.out_channels * batch_size
        )

        fused_conv_bn_scale_kernel[grid](
            x, fused_weight, fused_bias, output,
            batch_size, self.in_channels, self.out_channels,
            in_h, in_w, out_h, out_w,
            self.kernel_size,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            fused_weight.stride(0), fused_weight.stride(1), fused_weight.stride(2), fused_weight.stride(3),
            output.stride(0), output.stride(1), output.stride(2), output.stride(3),
            BLOCK_SIZE_OH=BLOCK_SIZE_OH, BLOCK_SIZE_OW=BLOCK_SIZE_OW,
            BLOCK_SIZE_IC=BLOCK_SIZE_IC,
            num_warps=4,
            num_stages=2
        )
        
        return output

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0]
