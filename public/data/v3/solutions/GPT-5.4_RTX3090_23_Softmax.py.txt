import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _softmax_kernel_f32(
    x_ptr,
    y_ptr,
    stride_xm,
    stride_xn,
    stride_ym,
    stride_yn,
    n_cols,
    BLOCK: tl.constexpr,
):
    row = tl.program_id(0)
    cols = tl.arange(0, BLOCK)
    mask = cols < n_cols
    x_row = x_ptr + row * stride_xm + cols * stride_xn
    x = tl.load(x_row, mask=mask, other=-float('inf'))
    x = x.to(tl.float32)
    x_max = tl.max(x, axis=0)
    x = x - x_max
    num = tl.exp(x)
    denom = tl.sum(num, axis=0)
    y = num / denom
    y_row = y_ptr + row * stride_ym + cols * stride_yn
    tl.store(y_row, y, mask=mask)

def _softmax_kernel_fp16(

    y_ptr,
    stride_xm,
    stride_xn,

    stride_yn,
    n_cols,
    BLOCK: tl.constexpr,

def softmax_triton(x: torch.Tensor) -> torch.Tensor:
    assert x.is_cuda
    assert x.dim() == 2
    m, n = x.shape
    y = torch.empty_like(x)
    BLOCK = triton.next_power_of_2(n)
    num_warps = 8 if BLOCK >= 4096 else 4
    grid = (m,)
    if x.dtype == torch.float32:
        _softmax_kernel_f32[grid](x, y, x.stride(0), x.stride(1), y.stride(0), y.stride(1), n, BLOCK=BLOCK, num_warps=num_warps)
    elif x.dtype == torch.float16:
        _softmax_kernel_fp16[grid](x, y, x.stride(0), x.stride(1), y.stride(0), y.stride(1), n, BLOCK=BLOCK, num_warps=num_warps)
    elif x.dtype == torch.bfloat16:
        _softmax_kernel_bf16[grid](x, y, x.stride(0), x.stride(1), y.stride(0), y.stride(1), n, BLOCK=BLOCK, num_warps=num_warps)
    else:
        raise ValueError(f'Unsupported dtype: {x.dtype}')
    return y


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return softmax_triton(x)


batch_size = 256
dim = 16384


def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]


def get_init_inputs():
    return []
