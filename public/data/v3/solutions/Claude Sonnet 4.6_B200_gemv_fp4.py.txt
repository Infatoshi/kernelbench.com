"""
Optimized FP4-like GEMV using int8 tensor cores (WMMA) with fused scaling.
M=32, K=4096, N=14336
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>
#include <stdint.h>

using namespace nvcuda;

#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

// V4: Large tiles with shared memory for B (weight) reuse
// TILE_M = 16 (fits M=32 in 2 blocks along M)
// TILE_N = 128 (8 warps x 16 = 128 cols per block)
// TILE_K = 64 (load 64 K elements at a time)
// 8 warps = 256 threads

#define TILE_M 16
#define TILE_N 128
#define TILE_K 64
#define WARPS_N 8   // 8 warps, each handles 16 cols

// Shared memory layout for A: [TILE_M, TILE_K] = [16, 64]
// Shared memory layout for B: [TILE_N, TILE_K] = [128, 64]

__global__ void __launch_bounds__(256, 4) gemv_fp4_v4(
    const int8_t* __restrict__ x_q,       // [M, K]
    const int8_t* __restrict__ weight_q,  // [N, K]
    const float x_scale,
    const float w_scale,
    __half* __restrict__ output,          // [M, N]
    int M, int N, int K
) {
    const int block_m = blockIdx.y * TILE_M;
    const int block_n = blockIdx.x * TILE_N;
    
    const int warp_id = threadIdx.x / 32;
    // Each warp handles 1 WMMA tile: 16 rows x 16 cols
    // Only 1 warp in M, 8 warps in N
    const int warp_n = warp_id;  // 0..7
    
    const int warp_m_global = block_m;
    const int warp_n_global = block_n + warp_n * WMMA_N;
    
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, int32_t> acc_frag;
    wmma::fill_fragment(acc_frag, 0);
    
    // Shared memory for A and B tiles
    __shared__ int8_t smem_a[TILE_M * TILE_K];   // 16 * 64 = 1024 bytes
    __shared__ int8_t smem_b[TILE_N * TILE_K];   // 128 * 64 = 8192 bytes
    
    const int tid = threadIdx.x;
    const int nthreads = blockDim.x;  // 256
    
    for (int k_base = 0; k_base < K; k_base += TILE_K) {
        // Load A tile: [TILE_M, TILE_K] = [16, 64] = 1024 bytes
        // stride along K = K
        {
            const int a_elems = TILE_M * TILE_K;  // 1024
            for (int i = tid; i < a_elems; i += nthreads) {
                int row = i / TILE_K;
                int col = i % TILE_K;
                int src_row = block_m + row;
                int src_col = k_base + col;
                smem_a[i] = (src_row < M && src_col < K) ? 
                    x_q[src_row * K + src_col] : (int8_t)0;
            }
        }
        
        // Load B tile: [TILE_N, TILE_K] = [128, 64] = 8192 bytes
        {
            const int b_elems = TILE_N * TILE_K;  // 8192
            for (int i = tid; i < b_elems; i += nthreads) {
                int row = i / TILE_K;
                int col = i % TILE_K;
                int src_row = block_n + row;
                int src_col = k_base + col;
                smem_b[i] = (src_row < N && src_col < K) ? 
                    weight_q[src_row * K + src_col] : (int8_t)0;
            }
        }
        
        __syncthreads();
        
        // Each warp processes TILE_K / WMMA_K = 4 tiles
        if (warp_n_global < N && warp_m_global < M) {
            for (int k_inner = 0; k_inner < TILE_K; k_inner += WMMA_K) {
                wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::row_major> a_frag;
                wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::col_major> b_frag;
                
                // A in smem: [TILE_M, TILE_K], row 0, col k_inner
                wmma::load_matrix_sync(a_frag, smem_a + k_inner, TILE_K);
                
                // B in smem: [TILE_N, TILE_K], row warp_n*16, col k_inner
                wmma::load_matrix_sync(b_frag, smem_b + warp_n * WMMA_N * TILE_K + k_inner, TILE_K);
                
                wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
            }
        }
        
        __syncthreads();
    }
    
    // Write out scaled fp16 values
    // Each warp writes a 16x16 tile
    // Use temporary smem (reuse smem_a area for int32, then convert)
    // Actually directly write with scaling
    
    // Store accumulator to output
    __shared__ int32_t smem_out[TILE_M * TILE_N];  // 16*128=2048 ints = 8KB
    
    if (warp_n_global < N && warp_m_global < M) {
        wmma::store_matrix_sync(
            smem_out + warp_n * WMMA_N,
            acc_frag, TILE_N, wmma::mem_row_major);
    }
    
    __syncthreads();
    
    float combined_scale = x_scale * w_scale;
    const int total = TILE_M * TILE_N;  // 2048
    
    for (int i = tid; i < total; i += nthreads) {
        int row = i / TILE_N;
        int col = i % TILE_N;
        int out_row = block_m + row;
        int out_col = block_n + col;
        if (out_row < M && out_col < N) {
            output[out_row * N + out_col] = __float2half((float)smem_out[i] * combined_scale);
        }
    }
}


// V5: Most optimized - direct global memory WMMA loads (avoid smem for weight since it's huge)
// But for M=32 (small), reuse of A is key
// TILE_N=256, 16 warps=512 threads, each warp handles 16 cols
// No shared memory for B (direct global load)

#define V5_TILE_N 128
#define V5_WARPS  8

__global__ void __launch_bounds__(256, 4) gemv_fp4_v5(
    const int8_t* __restrict__ x_q,       // [M, K]
    const int8_t* __restrict__ weight_q,  // [N, K]
    const float x_scale,
    const float w_scale,
    __half* __restrict__ output,          // [M, N]
    int M, int N, int K
) {
    // 1D grid in N, 1D grid in M
    const int block_n = blockIdx.x * V5_TILE_N;
    const int block_m = blockIdx.y * WMMA_M;  // Only 1 warp in M per block
    
    const int warp_id = threadIdx.x / 32;
    const int warp_n_global = block_n + warp_id * WMMA_N;
    const int warp_m_global = block_m;
    
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, int32_t> acc_frag;
    wmma::fill_fragment(acc_frag, 0);
    
    if (warp_m_global < M && warp_n_global < N) {
        for (int k = 0; k < K; k += WMMA_K) {
            wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::row_major> a_frag;
            wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::col_major> b_frag;
            
            wmma::load_matrix_sync(a_frag, x_q + warp_m_global * K + k, K);
            wmma::load_matrix_sync(b_frag, weight_q + warp_n_global * K + k, K);
            wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
        }
    }
    
    // Write directly to output with scaling
    __shared__ int32_t smem[WMMA_M * V5_TILE_N];  // 16*128=2048 ints
    
    if (warp_m_global < M && warp_n_global < N) {
        wmma::store_matrix_sync(smem + warp_id * WMMA_N, acc_frag, V5_TILE_N, wmma::mem_row_major);
    }
    __syncthreads();
    
    float combined_scale = x_scale * w_scale;
    const int total = WMMA_M * V5_TILE_N;
    const int nthreads = blockDim.x;
    
    for (int i = threadIdx.x; i < total; i += nthreads) {
        int row = i / V5_TILE_N;
        int col = i % V5_TILE_N;
        int out_row = block_m + row;
        int out_col = block_n + col;
        if (out_row < M && out_col < N) {
            output[out_row * N + out_col] = __float2half((float)smem[i] * combined_scale);
        }
    }
}


// V6: Use A in shared mem (small, only 32*4096 = 128KB), each block loads full A row
// This avoids re-reading x_q from global memory for every N block
// But 128KB is too large for smem. Let's tile K.
// 
// Better approach: stream the weight (large, N*K = 14336*4096 ~58MB)
// and reuse A across N blocks using persistent threads or L2 cache
//
// Actually the bottleneck is likely weight loading (58MB bandwidth)
// For M=32, K=4096, N=14336:
// Weight data: 14336 * 4096 = 58.7 MB
// x data:      32 * 4096 = 131 KB (fits in L2 cache)
// 
// So x will be cached, and we're bandwidth-limited by weight reads
// Optimal kernel: maximize weight read throughput
// Use large warps with direct global loads

// V6: CTA tile M=32 (full), N=64, 8 warps (2M x 4N)
// This processes all M rows at once, maximizing reuse of B per block

#define V6_TILE_M 32
#define V6_TILE_N 64
#define V6_WARPS_M 2
#define V6_WARPS_N 4

__global__ void __launch_bounds__(256, 4) gemv_fp4_v6(
    const int8_t* __restrict__ x_q,       // [M, K]
    const int8_t* __restrict__ weight_q,  // [N, K]
    const float x_scale,
    const float w_scale,
    __half* __restrict__ output,          // [M, N]
    int M, int N, int K
) {
    const int block_m = blockIdx.y * V6_TILE_M;
    const int block_n = blockIdx.x * V6_TILE_N;
    
    const int warp_id = threadIdx.x / 32;
    const int warp_m = warp_id / V6_WARPS_N;   // 0 or 1
    const int warp_n = warp_id % V6_WARPS_N;   // 0..3
    
    const int warp_m_global = block_m + warp_m * WMMA_M;
    const int warp_n_global = block_n + warp_n * WMMA_N;
    
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, int32_t> acc_frag;
    wmma::fill_fragment(acc_frag, 0);
    
    if (warp_m_global < M && warp_n_global < N) {
        for (int k = 0; k < K; k += WMMA_K) {
            wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::row_major> a_frag;
            wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::col_major> b_frag;
            
            wmma::load_matrix_sync(a_frag, x_q + warp_m_global * K + k, K);
            wmma::load_matrix_sync(b_frag, weight_q + warp_n_global * K + k, K);
            wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
        }
    }
    
    __shared__ int32_t smem[V6_TILE_M * V6_TILE_N];  // 32*64=2048
    
    int smem_off = (warp_m * WMMA_M) * V6_TILE_N + (warp_n * WMMA_N);
    if (warp_m_global < M && warp_n_global < N) {
        wmma::store_matrix_sync(smem + smem_off, acc_frag, V6_TILE_N, wmma::mem_row_major);
    }
    __syncthreads();
    
    float combined_scale = x_scale * w_scale;
    const int total = V6_TILE_M * V6_TILE_N;
    
    for (int i = threadIdx.x; i < total; i += blockDim.x) {
        int row = i / V6_TILE_N;
        int col = i % V6_TILE_N;
        int out_row = block_m + row;
        int out_col = block_n + col;
        if (out_row < M && out_col < N) {
            output[out_row * N + out_col] = __float2half((float)smem[i] * combined_scale);
        }
    }
}


// V7: Use shared memory for A, loading full K-slice per CTA
// and stream B (weight) through - maximizes B bandwidth
// Each block handles one group of N columns and ALL M rows
// M=32 rows, so each block processes all 32 rows for TILE_N cols

#define V7_TILE_N 64    // N cols per block  
#define V7_TILE_K 128   // K chunk
#define V7_WARPS  8     // 8 warps per block (256 threads)
// Each warp handles M=32 → 2 tiles in M, 1 tile in N  
// Warp layout: 4 warps in N (4*16=64), 2 warps in M (2*16=32)

__global__ void __launch_bounds__(256, 4) gemv_fp4_v7(
    const int8_t* __restrict__ x_q,       // [M, K]
    const int8_t* __restrict__ weight_q,  // [N, K]
    const float x_scale,
    const float w_scale,
    __half* __restrict__ output,          // [M, N]
    int M, int N, int K
) {
    // Each block handles M rows x TILE_N cols
    const int block_n = blockIdx.x * V7_TILE_N;
    // blockIdx.y = 0 always (M=32 → 2 warps)
    
    const int warp_id = threadIdx.x / 32;
    const int warp_m = warp_id / 4;  // 0 or 1
    const int warp_n = warp_id % 4;  // 0..3
    
    const int warp_m_global = warp_m * WMMA_M;
    const int warp_n_global = block_n + warp_n * WMMA_N;
    
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, int32_t> acc_frag;
    wmma::fill_fragment(acc_frag, 0);
    
    if (warp_m_global < M && warp_n_global < N) {
        for (int k = 0; k < K; k += WMMA_K) {
            wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::row_major> a_frag;
            wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, int8_t, wmma::col_major> b_frag;
            
            wmma::load_matrix_sync(a_frag, x_q + warp_m_global * K + k, K);
            wmma::load_matrix_sync(b_frag, weight_q + warp_n_global * K + k, K);
            wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);
        }
    }
    
    __shared__ int32_t smem[32 * V7_TILE_N];  // 32*64=2048
    
    if (warp_m_global < M && warp_n_global < N) {
        wmma::store_matrix_sync(
            smem + warp_m * WMMA_M * V7_TILE_N + warp_n * WMMA_N,
            acc_frag, V7_TILE_N, wmma::mem_row_major);
    }
    __syncthreads();
    
    float combined_scale = x_scale * w_scale;
    const int total = 32 * V7_TILE_N;
    
    for (int i = threadIdx.x; i < total; i += blockDim.x) {
        int row = i / V7_TILE_N;
        int col = i % V7_TILE_N;
        int out_col = block_n + col;
        if (row < M && out_col < N) {
            output[row * N + out_col] = __float2half((float)smem[i] * combined_scale);
        }
    }
}


torch::Tensor gemv_fp4_forward(
    torch::Tensor x_q,
    torch::Tensor x_scale,
    torch::Tensor weight_q,
    torch::Tensor weight_scale
) {
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = weight_q.size(0);
    
    auto output = torch::empty({M, N}, 
        torch::TensorOptions().dtype(torch::kFloat16).device(x_q.device()));
    
    float xs = x_scale.item<float>();
    float ws = weight_scale.item<float>();
    
    // V7: each block handles all M rows x TILE_N cols
    // Grid: (N/64, 1)
    dim3 grid7((N + V7_TILE_N - 1) / V7_TILE_N, 1);
    dim3 block7(256);
    
    gemv_fp4_v7<<<grid7, block7>>>(
        x_q.data_ptr<int8_t>(),
        weight_q.data_ptr<int8_t>(),
        xs, ws,
        reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
        M, N, K
    );
    
    return output;
}

cpp_source = '''

ext = load_inline(
    name='gemv_fp4_ext2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp4_forward'],
    extra_cuda_cflags=['-arch=sm_100', '-O3', '--use_fast_math', '-std=c++17'],
    verbose=True,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp4_forward(x_q, x_scale, self.weight_q, self.weight_scale)


def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.08, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
