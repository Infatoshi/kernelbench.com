import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void gemv_int8_kernel(
    const int8_t* __restrict__ x,
    const int8_t* __restrict__ w,
    float* __restrict__ out,
    int B, int K, int N,
    float scale
) {
    int b = blockIdx.x;
    int n0 = blockIdx.y * blockDim.x + threadIdx.x;

    if (b >= B || n0 >= N) return;

    const int8_t* x_row = x + b * K;
    const int8_t* w_row = w + n0 * K;

    int32_t acc = 0;

    for (int k = 0; k < K; ++k) {
        acc += (int32_t)x_row[k] * (int32_t)w_row[k];
    }

    out[b * N + n0] = acc * scale;
}

torch::Tensor gemv_int8(torch::Tensor x, torch::Tensor w, float scale) {
    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);

    auto out = torch::empty({B, N}, torch::dtype(torch::kFloat32).device(x.device()));

    dim3 block(256);
    dim3 grid(B, (N + 255) / 256);

    gemv_int8_kernel<<<grid, block>>>(
        x.data_ptr<int8_t>(),
        w.data_ptr<int8_t>(),
        out.data_ptr<float>(),
        B, K, N,
        scale
    );

    return out;
}
'''

ext = load_inline(
    name='gemv_int8_ext',
    cpp_sources='torch::Tensor gemv_int8(torch::Tensor x, torch::Tensor w, float scale);',
    cuda_sources=cuda_source,
    functions=['gemv_int8'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        scale = (x_scale * self.weight_scale).item()
        out = ext.gemv_int8(x_q, self.weight_q, scale)
        return out.to(torch.float16)


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8, device='cuda')
    x_scale = torch.tensor(0.02, dtype=torch.float32, device='cuda')
    return [x_q, x_scale]


def get_init_inputs():
    return []
