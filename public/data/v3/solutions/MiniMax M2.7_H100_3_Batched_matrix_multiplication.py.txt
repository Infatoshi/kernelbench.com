import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

cublasHandle_t get_cublas_handle() {
    static cublasHandle_t handle = [] {
        cublasHandle_t h;
        cublasCreate(&h);
        return h;
    }();
    return handle;
}

torch::Tensor batched_gemm_cublas(torch::Tensor A, torch::Tensor B) {
    auto handle = get_cublas_handle();
    
    int batch = A.size(0);
    int m = A.size(1);
    int k = A.size(2);
    int n = B.size(2);
    
    auto C = torch::empty({batch, m, n}, A.options());
    
    long long strideA = m * k;
    long long strideB = k * n;
    long long strideC = m * n;
    
    // Use Tensor Op math mode for better performance on H100
    cublasSetMathMode(handle, CUBLAS_TENSOR_OP_MATH);
    
    if (A.dtype() == at::kFloat) {
        float alpha = 1.0f;
        float beta = 0.0f;
        
        cublasSgemmStridedBatched(
            handle,
            CUBLAS_OP_N, CUBLAS_OP_N,
            n, m, k,
            &alpha,
            B.data_ptr<float>(), n, strideB,
            A.data_ptr<float>(), k, strideA,
            &beta,
            C.data_ptr<float>(), n, strideC,
            batch
        );
    } else {
        // For non-float types, convert to float, compute, and cast back
        auto A_f = A.to(torch::kFloat32);
        auto B_f = B.to(torch::kFloat32);
        auto C_f = torch::empty({batch, m, n}, A_f.options());
        
        float alpha = 1.0f;
        float beta = 0.0f;
        
        long long strideA_f = m * k;
        long long strideB_f = k * n;
        long long strideC_f = m * n;
        
        cublasSgemmStridedBatched(
            handle,
            CUBLAS_OP_N, CUBLAS_OP_N,
            n, m, k,
            &alpha,
            B_f.data_ptr<float>(), n, strideB_f,
            A_f.data_ptr<float>(), k, strideA_f,
            &beta,
            C_f.data_ptr<float>(), n, strideC_f,
            batch
        );
        
        C = C_f.to(A.dtype());
    }
    
    return C;
}
'''

ext = load_inline(
    name='batched_gemm_ext',
    cpp_sources='''
torch::Tensor batched_gemm_cublas(torch::Tensor A, torch::Tensor B);
''',
    cuda_sources=cuda_source,
    functions=['batched_gemm_cublas'],
    extra_include_paths=['/usr/local/cuda/include'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.batched_gemm_cublas(A, B)

def get_inputs():
    batch_size = 128
    m = 128
    k = 256
    n = 512
    A = torch.randn(batch_size, m, k, device='cuda')
    B = torch.randn(batch_size, k, n, device='cuda')
    return [A, B]

def get_init_inputs():
    return []
