import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Fast GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/π) * (x + 0.044715 * x³)))
// Constants for fast GELU
constexpr float GELU_COEF = 0.7978845608028654f;  // sqrt(2/π)
constexpr float GELU_CUBIC = 0.044715f;

// FP32 kernel with vectorized loads
__global__ void gelu_kernel_fp32(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Vectorized processing (4 elements at a time)
    int vec_idx = idx * 4;
    
    if (vec_idx + 3 < n) {
        // Load 4 floats
        float4 x_vec = *reinterpret_cast<const float4*>(x + vec_idx);
        float4 y_vec;
        
        // Process each element
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float val = (&x_vec.x)[i];
            float val3 = val * val * val;
            float inner = GELU_COEF * (val + GELU_CUBIC * val3);
            float tanh_val = tanhf(inner);
            (&y_vec.x)[i] = 0.5f * val * (1.0f + tanh_val);
        }
        
        // Store 4 floats
        *reinterpret_cast<float4*>(y + vec_idx) = y_vec;
    } else {
        // Handle remaining elements
        for (int i = 0; i < 4; i++) {
            int cur_idx = vec_idx + i;
            if (cur_idx < n) {
                float val = x[cur_idx];
                float val3 = val * val * val;
                float inner = GELU_COEF * (val + GELU_CUBIC * val3);
                float tanh_val = tanhf(inner);
                y[cur_idx] = 0.5f * val * (1.0f + tanh_val);
            }
        }
    }
}

// FP16 kernel with vectorized loads
__global__ void gelu_kernel_fp16(const half* __restrict__ x, half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Vectorized processing (8 elements at a time for FP16)
    int vec_idx = idx * 8;
    
    if (vec_idx + 7 < n) {
        // Load 8 halfs as 2 float4
        float4 x_vec1 = *reinterpret_cast<const float4*>(x + vec_idx);
        float4 x_vec2 = *reinterpret_cast<const float4*>(x + vec_idx + 4);
        float4 y_vec1, y_vec2;
        
        half* x_half1 = reinterpret_cast<half*>(&x_vec1);
        half* x_half2 = reinterpret_cast<half*>(&x_vec2);
        half* y_half1 = reinterpret_cast<half*>(&y_vec1);
        half* y_half2 = reinterpret_cast<half*>(&y_vec2);
        
        // Process each element
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float val = __half2float(x_half1[i]);
            float val3 = val * val * val;
            float inner = GELU_COEF * (val + GELU_CUBIC * val3);
            float tanh_val = tanhf(inner);
            y_half1[i] = __float2half(0.5f * val * (1.0f + tanh_val));
        }
        
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float val = __half2float(x_half2[i]);
            float val3 = val * val * val;
            float inner = GELU_COEF * (val + GELU_CUBIC * val3);
            float tanh_val = tanhf(inner);
            y_half2[i] = __float2half(0.5f * val * (1.0f + tanh_val));
        }
        
        // Store 8 halfs
        *reinterpret_cast<float4*>(y + vec_idx) = y_vec1;
        *reinterpret_cast<float4*>(y + vec_idx + 4) = y_vec2;
    } else {
        // Handle remaining elements
        for (int i = 0; i < 8; i++) {
            int cur_idx = vec_idx + i;
            if (cur_idx < n) {
                float val = __half2float(x[cur_idx]);
                float val3 = val * val * val;
                float inner = GELU_COEF * (val + GELU_CUBIC * val3);
                float tanh_val = tanhf(inner);
                y[cur_idx] = __float2half(0.5f * val * (1.0f + tanh_val));
            }
        }
    }
}

// BF16 kernel with vectorized loads
__global__ void gelu_kernel_bf16(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Vectorized processing (8 elements at a time for BF16)
    int vec_idx = idx * 8;
    
    if (vec_idx + 7 < n) {
        // Load 8 bf16s as 2 float4
        float4 x_vec1 = *reinterpret_cast<const float4*>(x + vec_idx);
        float4 x_vec2 = *reinterpret_cast<const float4*>(x + vec_idx + 4);
        float4 y_vec1, y_vec2;
        
        nv_bfloat16* x_bf1 = reinterpret_cast<nv_bfloat16*>(&x_vec1);
        nv_bfloat16* x_bf2 = reinterpret_cast<nv_bfloat16*>(&x_vec2);
        nv_bfloat16* y_bf1 = reinterpret_cast<nv_bfloat16*>(&y_vec1);
        nv_bfloat16* y_bf2 = reinterpret_cast<nv_bfloat16*>(&y_vec2);
        
        // Process each element
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float val = __bfloat162float(x_bf1[i]);
            float val3 = val * val * val;
            float inner = GELU_COEF * (val + GELU_CUBIC * val3);
            float tanh_val = tanhf(inner);
            y_bf1[i] = __float2bfloat16(0.5f * val * (1.0f + tanh_val));
        }
        
        #pragma unroll
        for (int i = 0; i < 4; i++) {
            float val = __bfloat162float(x_bf2[i]);
            float val3 = val * val * val;
            float inner = GELU_COEF * (val + GELU_CUBIC * val3);
            float tanh_val = tanhf(inner);
            y_bf2[i] = __float2bfloat16(0.5f * val * (1.0f + tanh_val));
        }
        
        // Store 8 bf16s
        *reinterpret_cast<float4*>(y + vec_idx) = y_vec1;
        *reinterpret_cast<float4*>(y + vec_idx + 4) = y_vec2;
    } else {
        // Handle remaining elements
        for (int i = 0; i < 8; i++) {
            int cur_idx = vec_idx + i;
            if (cur_idx < n) {
                float val = __bfloat162float(x[cur_idx]);
                float val3 = val * val * val;
                float inner = GELU_COEF * (val + GELU_CUBIC * val3);
                float tanh_val = tanhf(inner);
                y[cur_idx] = __float2bfloat16(0.5f * val * (1.0f + tanh_val));
            }
        }
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    // Choose block size for optimal occupancy
    const int block_size = 256;
    
    if (x.scalar_type() == torch::kFloat32) {
        // FP32: process 4 elements per thread
        int grid_size = (n + block_size * 4 - 1) / (block_size * 4);
        gelu_kernel_fp32<<<grid_size, block_size>>>(
            x.data_ptr<float>(), y.data_ptr<float>(), n
        );
    } else if (x.scalar_type() == torch::kHalf) {
        // FP16: process 8 elements per thread
        int grid_size = (n + block_size * 8 - 1) / (block_size * 8);
        gelu_kernel_fp16<<<grid_size, block_size>>>(
            reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
            reinterpret_cast<half*>(y.data_ptr<at::Half>()),
            n
        );
    } else if (x.scalar_type() == torch::kBFloat16) {
        // BF16: process 8 elements per thread
        int grid_size = (n + block_size * 8 - 1) / (block_size * 8);
        gelu_kernel_bf16<<<grid_size, block_size>>>(
            reinterpret_cast<const nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
            reinterpret_cast<nv_bfloat16*>(y.data_ptr<at::BFloat16>()),
            n
        );
    } else {
        AT_ERROR("Unsupported dtype for GELU");
    }
    
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_forward(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86'],
)

class Model(nn.Module):
    """
    Simple model that performs a GELU activation with optimized CUDA kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies GELU activation to the input tensor using optimized CUDA kernel.

        Args:
            x (torch.Tensor): Input tensor of any shape.

        Returns:
            torch.Tensor: Output tensor with GELU applied, same shape as input.
        """
        return ext.gelu_forward(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []  # No special initialization inputs needed
