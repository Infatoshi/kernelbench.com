// Simple GEMM kernel using CUTLASS 3.x for Hopper
template<typename ElementA, typename ElementB, typename ElementC, typename ElementAccumulator>
void gemm_cutlass(
    const ElementA* a_ptr,
    const ElementB* b_ptr,
    ElementC* c_ptr,
    int M, int N, int K,
    ElementAccumulator alpha = 1.0f,
    ElementAccumulator beta = 0.0f,

) {

    using LayoutB = cutlass::layout::ColumnMajor;
    using LayoutC = cutlass::layout::RowMajor;
    
    using CollectiveOp = typename cutlass::gemm::collective::CollectiveBuilder<
        cutlass::arch::Sm90, cutlass::arch::OpClassTensorOp,
        ElementA, LayoutA, 16,
        ElementB, LayoutB, 16,
        ElementAccumulator,
        cutlass::gemm::GemmShape<128, 128, 64>,
        cutlass::gemm::collective::StageCountAuto,
        cutlass::gemm::collective::KernelScheduleAuto
    >::CollectiveOp;
    
    using EpilogueOp = typename cutlass::epilogue::collective::CollectiveBuilder<
        cutlass::arch::Sm90, cutlass::gemm::GemmShape<128, 128, 64>,
        ElementAccumulator, ElementAccumulator,
        ElementC, LayoutC, 16,
        cutlass::epilogue::collective::EpilogueScheduleAuto
    >::CollectiveOp;

    };
    
    Gemm gemm_op;
    cutlass::Status status = gemm_op.can_implement(args);

    }

    cutlass::device_memory::allocation<uint8_t> workspace(workspace_size);
    
    gemm_op.run(args, workspace.get(), stream);
}

// Fused SiLU activation kernel
__global__ void silu_kernel(float* __restrict__ out, const float* __restrict__ in, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x = in[idx];
        out[idx] = x * (1.0f / (1.0f + expf(-x)));
    }
}

__global__ void silu_kernel_half(__half* __restrict__ out, const __half* __restrict__ in, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x = __half2float(in[idx]);
        out[idx] = __float2half(x * (1.0f / (1.0f + expf(-x))));
    }
}

// Sigmoid kernel
__global__ void sigmoid_kernel(float* __restrict__ out, const float* __restrict__ in, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        out[idx] = 1.0f / (1.0f + expf(-in[idx]));
    }
}

__global__ void sigmoid_kernel_half(__half* __restrict__ out, const __half* __restrict__ in, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float x = __half2float(in[idx]);
        out[idx] = __float2half(1.0f / (1.0f + expf(-x)));
    }
}

// Vectorized copy kernel for transposes
__global__ void transpose_2d_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int rows, int cols
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row < rows && col < cols) {
        output[col * rows + row] = input[row * cols + col];
    }
}

// Depthwise conv1d kernel (optimized for kernel_size=4)
__global__ void depthwise_conv1d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int channels,
    int seq_len,
    int kernel_size,
    int padding
) {
    int b = blockIdx.z;
    int c = blockIdx.y;
    int t = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (b >= batch_size || c >= channels || t >= seq_len) return;
    
    float sum = bias ? bias[c] : 0.0f;
    
    for (int k = 0; k < kernel_size; k++) {
        int input_t = t + k - padding;
        if (input_t >= 0 && input_t < seq_len + padding - kernel_size + 1) {
            if (input_t < seq_len) {
                int input_idx = b * channels * seq_len + c * seq_len + input_t;
                int weight_idx = c * kernel_size + k;
                sum += input[input_idx] * weight[weight_idx];
            }
        }
    }
    
    int output_idx = b * channels * seq_len + c * seq_len + t;
    output[output_idx] = sum;
}

// Optimized GEMM using cuBLAS-style approach
__global__ void simple_gemm_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K,
    float alpha, float beta
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row < M && col < N) {
        float sum = 0.0f;
        for (int k = 0; k < K; k++) {
            sum += A[row * K + k] * B[k * N + col];
        }
        C[row * N + col] = alpha * sum + beta * C[row * N + col];
    }
}

// Optimized GEMM with shared memory tiling
#define TILE_SIZE 32

__global__ void tiled_gemm_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K
) {
    __shared__ float As[TILE_SIZE][TILE_SIZE];
    __shared__ float Bs[TILE_SIZE][TILE_SIZE];
    
    int row = blockIdx.y * TILE_SIZE + threadIdx.y;
    int col = blockIdx.x * TILE_SIZE + threadIdx.x;
    
    float sum = 0.0f;
    
    for (int t = 0; t < (K + TILE_SIZE - 1) / TILE_SIZE; t++) {
        // Load tiles into shared memory
        int a_row = row;
        int a_col = t * TILE_SIZE + threadIdx.x;
        if (a_row < M && a_col < K) {
            As[threadIdx.y][threadIdx.x] = A[a_row * K + a_col];
        } else {
            As[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        int b_row = t * TILE_SIZE + threadIdx.y;
        int b_col = col;
        if (b_row < K && b_col < N) {
            Bs[threadIdx.y][threadIdx.x] = B[b_row * N + b_col];
        } else {
            Bs[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        __syncthreads();
        
        for (int k = 0; k < TILE_SIZE; k++) {
            sum += As[threadIdx.y][k] * Bs[k][threadIdx.x];
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = sum;
    }
}

// Half precision GEMM
__global__ void tiled_gemm_kernel_half(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    __half* __restrict__ C,
    int M, int N, int K
) {
    __shared__ half As[TILE_SIZE][TILE_SIZE];
    __shared__ half Bs[TILE_SIZE][TILE_SIZE];
    
    int row = blockIdx.y * TILE_SIZE + threadIdx.y;
    int col = blockIdx.x * TILE_SIZE + threadIdx.x;
    
    float sum = 0.0f;
    
    for (int t = 0; t < (K + TILE_SIZE - 1) / TILE_SIZE; t++) {
        int a_row = row;
        int a_col = t * TILE_SIZE + threadIdx.x;
        if (a_row < M && a_col < K) {
            As[threadIdx.y][threadIdx.x] = A[a_row * K + a_col];
        } else {
            As[threadIdx.y][threadIdx.x] = __half(0.0f);
        }
        
        int b_row = t * TILE_SIZE + threadIdx.y;
        int b_col = col;
        if (b_row < K && b_col < N) {
            Bs[threadIdx.y][threadIdx.x] = B[b_row * N + b_col];
        } else {
            Bs[threadIdx.y][threadIdx.x] = __half(0.0f);
        }
        
        __syncthreads();
        
        for (int k = 0; k < TILE_SIZE; k++) {
            sum += __half2float(As[threadIdx.y][k]) * __half2float(Bs[k][threadIdx.x]);
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = __float2half(sum);
    }
}

// Element-wise multiply kernel
__global__ void mul_kernel(float* __restrict__ out, const float* __restrict__ a, const float* __restrict__ b, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        out[idx] = a[idx] * b[idx];
    }
}

__global__ void mul_kernel_half(__half* __restrict__ out, const __half* __restrict__ a, const __half* __restrict__ b, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        out[idx] = __float2half(__half2float(a[idx]) * __half2float(b[idx]));
    }
}

// LayerNorm kernel
__global__ void layernorm_kernel(
    float* __restrict__ out,
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    int batch_size, int hidden_size
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const float* input_ptr = input + batch_idx * hidden_size;
    float* output_ptr = out + batch_idx * hidden_size;
    
    // Compute mean
    float mean = 0.0f;
    for (int i = 0; i < hidden_size; i++) {
        mean += input_ptr[i];
    }
    mean /= hidden_size;
    
    // Compute variance
    float var = 0.0f;
    for (int i = 0; i < hidden_size; i++) {
        float diff = input_ptr[i] - mean;
        var += diff * diff;
    }
    var /= hidden_size;
    
    // Normalize and apply weight/bias
    float inv_std = 1.0f / sqrtf(var + 1e-5f);
    for (int i = 0; i < hidden_size; i++) {
        output_ptr[i] = (input_ptr[i] - mean) * inv_std * weight[i] + bias[i];
    }
}

// Add bias kernel
__global__ void add_bias_kernel(float* __restrict__ out, const float* __restrict__ bias, int batch_size, int hidden_size) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = batch_size * hidden_size;
    if (idx < total) {
        out[idx] += bias[idx % hidden_size];
    }
}

// PyTorch bindings
torch::Tensor linear_forward(torch::Tensor input, torch::Tensor weight, c10::optional<torch::Tensor> bias) {
    int M = input.size(0) * input.size(1);
    int K = input.size(2);
    int N = weight.size(0);
    
    auto output = torch::empty({input.size(0), input.size(1), N}, input.options());
    
    bool is_half = input.scalar_type() == torch::kFloat16;
    
    dim3 blockDim(TILE_SIZE, TILE_SIZE);
    dim3 gridDim((N + TILE_SIZE - 1) / TILE_SIZE, (M + TILE_SIZE - 1) / TILE_SIZE);
    
    if (is_half) {
        tiled_gemm_kernel_half<<<gridDim, blockDim>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            M, N, K
        );
    } else {
        tiled_gemm_kernel<<<gridDim, blockDim>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            output.data_ptr<float>(),
            M, N, K
        );
    }
    
    if (bias.has_value()) {
        int total = M * N;
        int threads = 256;
        int blocks = (total + threads - 1) / threads;
        if (is_half) {
            add_bias_kernel<<<blocks, threads>>>(
                reinterpret_cast<float*>(output.data_ptr<at::Half>()),
                reinterpret_cast<const float*>(bias->data_ptr<at::Half>()),
                M, N
            );
        } else {
            add_bias_kernel<<<blocks, threads>>>(
                output.data_ptr<float>(),
                bias->data_ptr<float>(),
                M, N
            );
        }
    }
    
    return output;
}

torch::Tensor silu_forward(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n = input.numel();
    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    
    if (input.scalar_type() == torch::kFloat16) {
        silu_kernel_half<<<blocks, threads>>>(
            output.data_ptr<at::Half>(),
            input.data_ptr<at::Half>(),
            n
        );
    } else {
        silu_kernel<<<blocks, threads>>>(
            output.data_ptr<float>(),
            input.data_ptr<float>(),
            n
        );
    }
    
    return output;
}

torch::Tensor depthwise_conv1d_forward(
    torch::Tensor input,  // (B, C, T)
    torch::Tensor weight,  // (C, kernel_size)
    c10::optional<torch::Tensor> bias,  // (C,)
    int padding
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int seq_len = input.size(2);
    int kernel_size = weight.size(1);
    
    auto output = torch::empty({batch_size, channels, seq_len}, input.options());
    
    dim3 grid((seq_len + 255) / 256, channels, batch_size);
    
    depthwise_conv1d_kernel<<<grid, 256>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias->data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size, channels, seq_len, kernel_size, padding
    );
    
    return output;
}

        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        batch_size, hidden_size
    );
    
    return output;
}

            b.data_ptr<float>(),
            n
        );
    }
    
    return output;

# Load the extension
ext = load_inline(
    name='gated_delta_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['linear_forward', 'silu_forward', 'sigmoid_forward', 'depthwise_conv1d_forward', 'layernorm_forward', 'mul_forward'],
    extra_cuda_cflags=['-I/opt/cutlass/include', '-arch=sm_90', '-std=c++17'],
    verbose=False,
)


def gated_delta_attention(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    alpha: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
) -> torch.Tensor:
    """Use fla's optimized kernel for attention."""
    g = alpha.clamp(min=1e-6).log()
    output, _ = chunk_gated_delta_rule(q, k, v, g, beta, scale=scale)
    return output


class Model(nn.Module):
    """
    Gated DeltaNet with custom CUDA kernels for projections.
    """

    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv
        self.conv_kernel_size = conv_kernel_size

        # Store weights for custom linear
        self.q_proj_weight = nn.Parameter(torch.empty(num_heads * head_dim_qk, hidden_size))
        self.k_proj_weight = nn.Parameter(torch.empty(num_heads * head_dim_qk, hidden_size))
        self.v_proj_weight = nn.Parameter(torch.empty(num_heads * head_dim_v, hidden_size))
        
        self.a_proj_weight = nn.Parameter(torch.empty(num_heads, hidden_size))
        self.a_proj_bias = nn.Parameter(torch.zeros(num_heads))
        self.b_proj_weight = nn.Parameter(torch.empty(num_heads, hidden_size))
        self.b_proj_bias = nn.Parameter(torch.zeros(num_heads))
        
        self.o_proj_weight = nn.Parameter(torch.empty(hidden_size, num_heads * head_dim_v))
        self.g_proj_weight = nn.Parameter(torch.empty(num_heads * head_dim_v, hidden_size))

        if use_short_conv:
            self.q_conv_weight = nn.Parameter(torch.empty(num_heads * head_dim_qk, 1, conv_kernel_size))
            self.k_conv_weight = nn.Parameter(torch.empty(num_heads * head_dim_qk, 1, conv_kernel_size))
            self.v_conv_weight = nn.Parameter(torch.empty(num_heads * head_dim_v, 1, conv_kernel_size))

        self.o_norm_weight = nn.Parameter(torch.ones(head_dim_v))
        self.o_norm_bias = nn.Parameter(torch.zeros(head_dim_v))
        self.scale = head_dim_qk ** -0.5
        
        self._init_weights()

    def _init_weights(self):
        nn.init.kaiming_uniform_(self.q_proj_weight)
        nn.init.kaiming_uniform_(self.k_proj_weight)
        nn.init.kaiming_uniform_(self.v_proj_weight)
        nn.init.kaiming_uniform_(self.a_proj_weight)
        nn.init.kaiming_uniform_(self.b_proj_weight)
        nn.init.kaiming_uniform_(self.o_proj_weight)
        nn.init.kaiming_uniform_(self.g_proj_weight)
        if self.use_short_conv:
            nn.init.kaiming_uniform_(self.q_conv_weight)
            nn.init.kaiming_uniform_(self.k_conv_weight)
            nn.init.kaiming_uniform_(self.v_conv_weight)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        orig_dtype = x.dtype
        
        # Projections using custom kernel
        q = ext.linear_forward(x, self.q_proj_weight, None)
        k = ext.linear_forward(x, self.k_proj_weight, None)
        v = ext.linear_forward(x, self.v_proj_weight, None)

        if self.use_short_conv:
            # Transpose for conv1d: (B, T, C) -> (B, C, T)
            q_t = q.transpose(1, 2).contiguous()
            k_t = k.transpose(1, 2).contiguous()
            v_t = v.transpose(1, 2).contiguous()
            
            # Apply depthwise conv1d
            q_conv_w = self.q_conv_weight.squeeze(1)  # (C, kernel_size)
            k_conv_w = self.k_conv_weight.squeeze(1)
            v_conv_w = self.v_conv_weight.squeeze(1)
            
            q_t = ext.depthwise_conv1d_forward(q_t, q_conv_w, None, self.conv_kernel_size - 1)
            k_t = ext.depthwise_conv1d_forward(k_t, k_conv_w, None, self.conv_kernel_size - 1)
            v_t = ext.depthwise_conv1d_forward(v_t, v_conv_w, None, self.conv_kernel_size - 1)
            
            # Slice to original seq_len and transpose back
            q = q_t[:, :, :seq_len].transpose(1, 2)
            k = k_t[:, :, :seq_len].transpose(1, 2)
            v = v_t[:, :, :seq_len].transpose(1, 2)
            
            # SiLU activation
            q = ext.silu_forward(q)
            k = ext.silu_forward(k)
            v = ext.silu_forward(v)

        # Reshape to (B, H, T, D)
        q = q.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        k = k.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        v = v.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        # Alpha and beta projections
        a_logits = ext.linear_forward(x, self.a_proj_weight, self.a_proj_bias)
        b_logits = ext.linear_forward(x, self.b_proj_weight, self.b_proj_bias)
        
        alpha = ext.sigmoid_forward(a_logits).transpose(1, 2)
        beta = ext.sigmoid_forward(b_logits).transpose(1, 2)

        # Chunk-wise parallel attention
        o = gated_delta_attention(q, k, v, alpha, beta, scale=self.scale)

        # (B, H, T, d_v) -> (B, T, H, d_v)
        o = o.transpose(1, 2)

        # LayerNorm
        o = ext.layernorm_forward(o, self.o_norm_weight, self.o_norm_bias)

        # Gate projection
        g_logits = ext.linear_forward(x, self.g_proj_weight, None)
        g = ext.sigmoid_forward(g_logits)
        g = g.view(batch_size, seq_len, self.num_heads, self.head_dim_v)
        
        # Multiply with gate
        o = ext.mul_forward(o, g)

        # Output projection
        o = o.reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
        o = ext.linear_forward(o, self.o_proj_weight, None)

        return o
