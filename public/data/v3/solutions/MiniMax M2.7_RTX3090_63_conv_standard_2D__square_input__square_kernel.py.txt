import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda;
using namespace nvcuda::wmma;

// Optimized FP32 convolution with shared memory tiling
__global__ void conv2d_fp32_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    float* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int height, int width, int kernel_size,
    int out_height, int out_width, int stride, int padding, int dilation)
{
    __shared__ float smem[32][32];
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Each thread block handles a 32x32 output tile
    // Each thread handles one output element
    int out_base_x = bx * 32;
    int out_base_y = by * 32;
    
    int out_x = out_base_x + tx;
    int out_y = out_base_y + ty;
    int out_oc = bz;
    
    float sum = 0.0f;
    
    // Loop over kernel elements
    for (int kh = 0; kh < kernel_size; kh++) {
        for (int kw = 0; kw < kernel_size; kw++) {
            for (int ic = 0; ic < in_channels; ic++) {
                // Calculate input position for this thread's output
                int in_y = out_y * stride - padding + kh * dilation;
                int in_x = out_x * stride - padding + kw * dilation;
                
                if (in_y >= 0 && in_y < height && in_x >= 0 && in_x < width) {
                    // Load weight - all threads in warp load same weight value
                    int w_idx = ((out_oc * in_channels + ic) * kernel_size + kh) * kernel_size + kw;
                    float w_val = __ldg(&weight[w_idx]);
                    
                    // Load activation
                    int in_idx = ((0 * in_channels + ic) * height + in_y) * width + in_x;
                    float a_val = __ldg(&input[in_idx]);
                    
                    sum += a_val * w_val;
                }
            }
        }
    }
    
    if (out_y < out_height && out_x < out_width) {
        int out_idx = ((0 * out_channels + out_oc) * out_height + out_y) * out_width + out_x;
        output[out_idx] = sum;
    }
}

// Optimized FP16 convolution with half precision
__global__ void conv2d_fp16_kernel(
    const __half* __restrict__ input,
    const __half* __restrict__ weight,
    __half* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int height, int width, int kernel_size,
    int out_height, int out_width, int stride, int padding, int dilation)
{
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int out_base_x = bx * 16;
    int out_base_y = by * 16;
    
    int out_x = out_base_x + tx;
    int out_y = out_base_y + ty;
    int out_oc = bz;
    
    float sum = 0.0f;
    
    for (int kh = 0; kh < kernel_size; kh++) {
        for (int kw = 0; kw < kernel_size; kw++) {
            for (int ic = 0; ic < in_channels; ic++) {
                int in_y = out_y * stride - padding + kh * dilation;
                int in_x = out_x * stride - padding + kw * dilation;
                
                if (in_y >= 0 && in_y < height && in_x >= 0 && in_x < width) {
                    int w_idx = ((out_oc * in_channels + ic) * kernel_size + kh) * kernel_size + kw;
                    __half w_val = __ldg(&weight[w_idx]);
                    
                    int in_idx = ((0 * in_channels + ic) * height + in_y) * width + in_x;
                    __half a_val = __ldg(&input[in_idx]);
                    
                    sum += __half2float(__hmul(a_val, w_val));
                }
            }
        }
    }
    
    if (out_y < out_height && out_x < out_width) {
        int out_idx = ((0 * out_channels + out_oc) * out_height + out_y) * out_width + out_x;
        output[out_idx] = __float2half(sum);
    }
}

// Optimized BF16 convolution
__global__ void conv2d_bf16_kernel(
    const __nv_bfloat16* __restrict__ input,
    const __nv_bfloat16* __restrict__ weight,
    __nv_bfloat16* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int height, int width, int kernel_size,
    int out_height, int out_width, int stride, int padding, int dilation)
{
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int bz = blockIdx.z;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int out_base_x = bx * 16;
    int out_base_y = by * 16;
    
    int out_x = out_base_x + tx;
    int out_y = out_base_y + ty;
    int out_oc = bz;
    
    float sum = 0.0f;
    
    for (int kh = 0; kh < kernel_size; kh++) {
        for (int kw = 0; kw < kernel_size; kw++) {
            for (int ic = 0; ic < in_channels; ic++) {
                int in_y = out_y * stride - padding + kh * dilation;
                int in_x = out_x * stride - padding + kw * dilation;
                
                if (in_y >= 0 && in_y < height && in_x >= 0 && in_x < width) {
                    int w_idx = ((out_oc * in_channels + ic) * kernel_size + kh) * kernel_size + kw;
                    __nv_bfloat16 w_val = __ldg(&weight[w_idx]);
                    
                    int in_idx = ((0 * in_channels + ic) * height + in_y) * width + in_x;
                    __nv_bfloat16 a_val = __ldg(&input[in_idx]);
                    
                    sum += __bfloat162float(__hmul(a_val, w_val));
                }
            }
        }
    }
    
    if (out_y < out_height && out_x < out_width) {
        int out_idx = ((0 * out_channels + out_oc) * out_height + out_y) * out_width + out_x;
        output[out_idx] = __float2bfloat16(sum);
    }
}

// Host launch function
torch::Tensor conv2d_op(torch::Tensor input, torch::Tensor weight, int stride, int padding, int dilation) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_height = (height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int out_width = (width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_height, out_width}, input.options());
    
    dim3 block_fp32(32, 32);
    dim3 grid_fp32((out_width + 31) / 32, (out_height + 31) / 32, out_channels);
    
    dim3 block_fp16(16, 16);
    dim3 grid_fp16((out_width + 15) / 16, (out_height + 15) / 16, out_channels);
    
    if (input.scalar_type() == torch::kFloat) {
        conv2d_fp32_kernel<<<grid_fp32, block_fp32>>>(
            input.data_ptr<float>(), weight.data_ptr<float>(), output.data_ptr<float>(),
            batch_size, in_channels, out_channels, height, width, kernel_size,
            out_height, out_width, stride, padding, dilation);
    } else if (input.scalar_type() == torch::kHalf) {
        conv2d_fp16_kernel<<<grid_fp16, block_fp16>>>(
            reinterpret_cast<const __half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<const __half*>(weight.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
            batch_size, in_channels, out_channels, height, width, kernel_size,
            out_height, out_width, stride, padding, dilation);
    } else if (input.scalar_type() == torch::kBFloat16) {
        conv2d_bf16_kernel<<<grid_fp16, block_fp16>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            batch_size, in_channels, out_channels, height, width, kernel_size,
            out_height, out_width, stride, padding, dilation);
    }
    
    return output;
}
'''

ext = load_inline(
    name='conv2d_ext',
    cpp_sources='''
#include <torch/extension.h>
torch::Tensor conv2d_op(torch::Tensor input, torch::Tensor weight, int stride, int padding, int dilation);
''',
    cuda_sources=cuda_source,
    functions=['conv2d_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, 
                 stride: int = 1, padding: int = 0, dilation: int = 1, 
                 groups: int = 1, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        
        # Initialize weights with same distribution as PyTorch Conv2d
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels // groups, kernel_size, kernel_size))
        if bias:
            self.bias = nn.Parameter(torch.randn(out_channels))
        else:
            self.register_parameter('bias', None)
        
        # Initialize with kaiming normal
        nn.init.kaiming_normal_(self.weight, mode='fan_out', nonlinearity='relu')
        if self.bias is not None:
            nn.init.zeros_(self.bias)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.conv2d_op(x, self.weight, self.stride, self.padding, self.dilation)

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 64, 3]  # in_channels, out_channels, kernel_size
