import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_conv_bn_scale_kernel(
    # Pointers to inputs and output
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    # Dimensions
    N, in_ch, H, W,
    out_ch, kH, kW,
    out_h, out_w,
    # Stride and padding
    stride: tl.constexpr,
    padding: tl.constexpr,
    scaling_factor,
    # Strides for input tensor [N, C, H, W]
    stride_in_n, stride_in_c, stride_in_h, stride_in_w,
    # Strides for weight tensor [out_ch, in_ch, kH, kW]
    stride_w_oc, stride_w_ic, stride_w_kh, stride_w_kw,
    # Strides for output tensor [N, out_ch, out_h, out_w]
    stride_out_n, stride_out_oc, stride_out_h, stride_out_w,
    # Block sizes
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one (n, oc) pair
    pid = tl.program_id(0)
    n = pid // out_ch
    oc = pid % out_ch
    
    if n >= N or oc >= out_ch:
        return
    
    # Process output positions in blocks
    total_positions = out_h * out_w
    
    for start_pos in range(0, total_positions, BLOCK_SIZE):
        # Create offsets for this block
        pos_offsets = start_pos + tl.arange(0, BLOCK_SIZE)
        
        # Filter valid positions
        mask = pos_offsets < total_positions
        
        # Decode h_out, w_out from position
        h_out = pos_offsets // out_w
        w_out = pos_offsets % out_w
        
        # Compute input starting position
        h_in = h_out * stride - padding
        w_in = w_out * stride - padding
        
        # Accumulate convolution result
        acc = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
        
        # Loop over input channels and kernel positions
        for ic in range(in_ch):
            for kh in range(kH):
                for kw in range(kW):
                    # Compute input coordinates
                    ih = h_in + kh
                    iw = w_in + kw
                    
                    # Check bounds
                    valid = (ih >= 0) & (ih < H) & (iw >= 0) & (iw < W)
                    
                    # Compute input index
                    in_idx = ((n * in_ch + ic) * H + ih) * W + iw
                    
                    # Compute weight index
                    w_idx = ((oc * in_ch + ic) * kH + kh) * kW + kw
                    
                    # Load values
                    in_val = tl.load(input_ptr + in_idx, mask=mask & valid, other=0.0).to(tl.float32)
                    w_val = tl.load(weight_ptr + w_idx).to(tl.float32)
                    
                    acc += in_val * w_val
        
        # Add bias and apply scaling
        bias_val = tl.load(bias_ptr + oc).to(tl.float32)
        result = (acc + bias_val) * scaling_factor
        
        # Compute output index
        out_idx = ((n * out_ch + oc) * out_h + h_out) * out_w + w_out
        
        # Store result
        tl.store(output_ptr + out_idx, result, mask=mask)


class FusedConvBNScale(torch.autograd.Function):
    @staticmethod
    def forward(ctx, input, weight, bias, scaling_factor, stride, padding):
        N, in_ch, H, W = input.shape
        out_ch, _, kH, kW = weight.shape
        out_h = (H + 2 * padding - kH) // stride + 1
        out_w = (W + 2 * padding - kW) // stride + 1
        
        output = torch.empty((N, out_ch, out_h, out_w), dtype=input.dtype, device=input.device)
        
        # Launch kernel
        grid = (N * out_ch,)
        BLOCK_SIZE = 256
        
        # Flatten tensors for Triton
        input_flat = input.contiguous().flatten()
        weight_flat = weight.contiguous().flatten()
        bias_flat = bias.contiguous().flatten()
        output_flat = output.flatten()
        
        fused_conv_bn_scale_kernel[grid](
            input_flat, weight_flat, bias_flat, output_flat,
            N, in_ch, H, W,
            out_ch, kH, kW,
            out_h, out_w,
            stride, padding,
            scaling_factor,
            in_ch * H * W, H * W, W, 1,  # input strides
            in_ch * kH * kW, kH * kW, kW, 1,  # weight strides
            out_ch * out_h * out_w, out_h * out_w, out_w, 1,  # output strides
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        return output.reshape(N, out_ch, out_h, out_w)


def fused_conv_bn_scale(input, weight, bias, scaling_factor, stride=1, padding=0):
    return FusedConvBNScale.apply(input, weight, bias, scaling_factor, stride, padding)


class Model(nn.Module):
    """
    Fused Conv2d + BatchNorm + Scaling model.
    At inference time, we fold BatchNorm into Conv weights and bias.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=1)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        self.kernel_size = kernel_size
        self.stride = 1
        self.padding = 1
        
        # Folded parameters (will be computed in forward)
        self.register_buffer('folded_weight', None)
        self.register_buffer('folded_bias', None)
        
    def _fold_bn(self):
        """Fold BatchNorm into Conv weights and bias."""
        if self.folded_weight is not None:
            return
            
        conv_weight = self.conv.weight.data  # [out_ch, in_ch, kH, kW]
        conv_bias = self.conv.bias.data if self.conv.bias is not None else torch.zeros_like(self.bn.running_mean)
        
        bn_gamma = self.bn.weight.data
        bn_beta = self.bn.bias.data
        bn_mean = self.bn.running_mean
        bn_var = self.bn.running_var
        bn_eps = self.bn.eps
        
        # Compute BN scaling factor: gamma / sqrt(var + eps)
        bn_scale = bn_gamma / torch.sqrt(bn_var + bn_eps)
        
        # Folded weight: bn_scale * conv_weight
        self.folded_weight = (bn_scale.view(-1, 1, 1, 1) * conv_weight).contiguous()
        
        # Folded bias: bn_scale * (conv_bias - mean) + beta
        self.folded_bias = (bn_scale * (conv_bias - bn_mean) + bn_beta).contiguous()
        
    def forward(self, x):
        self._fold_bn()
        
        # Apply scaling factor to folded bias
        scaled_bias = self.folded_bias * self.scaling_factor
        
        # Call Triton kernel
        output = fused_conv_bn_scale(
            x,
            self.folded_weight,
            scaled_bias,
            1.0,  # scaling already applied to bias
            self.stride,
            self.padding
        )
        
        return output


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    scaling_factor = 2.0
    return [in_channels, out_channels, kernel_size, scaling_factor]
