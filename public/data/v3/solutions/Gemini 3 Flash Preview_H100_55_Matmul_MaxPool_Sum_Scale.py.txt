import torch
import torch.nn as nn
import triton
import triton.language as tl
import math

@triton.jit
def matmul_maxpool_sum_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    batch_size, in_features, out_features, kernel_size, scale_factor,
    X_stride_m, X_stride_k,
    W_stride_n, W_stride_k,
    B_stride_n,
    Out_stride_m,
    BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    POOL_K: tl.constexpr
):
    # m is batch index, n_block is chunk of out_features
    pid = tl.program_id(0)
    num_n_blocks = tl.cdiv(out_features, BLOCK_SIZE_N)
    m = pid // num_n_blocks
    n_block_idx = pid % num_n_blocks
    
    if m >= batch_size:
        return

    n_start = n_block_idx * BLOCK_SIZE_N
    n_offsets = n_start + tl.arange(0, BLOCK_SIZE_N)
    mask_n = n_offsets < out_features
    
    X_ptr_row = X_ptr + m * X_stride_m
    acc = tl.zeros([BLOCK_SIZE_N], dtype=tl.float32)
    
    # Dot product
    for k_start in range(0, in_features, BLOCK_SIZE_K):
        k_offsets = k_start + tl.arange(0, BLOCK_SIZE_K)
        mask_k = k_offsets < in_features
        x_vals = tl.load(X_ptr_row + k_offsets, mask=mask_k, other=0.0)
        w_ptr_block = W_ptr + n_offsets[:, None] * W_stride_n + k_offsets[None, :] * W_stride_k
        w_vals = tl.load(w_ptr_block, mask=mask_n[:, None] & mask_k[None, :], other=0.0)
        acc += tl.sum(x_vals[None, :] * w_vals, axis=1)

    # Bias and pooling
    bias = tl.load(B_ptr + n_offsets, mask=mask_n, other=0.0)
    acc += bias
    acc = tl.where(mask_n, acc, -1e20)
    
    y_part = tl.reshape(acc, (BLOCK_SIZE_N // POOL_K, POOL_K))
    z_part = tl.max(y_part, axis=1)
    
    pooled_offsets = (n_start // POOL_K) + tl.arange(0, BLOCK_SIZE_N // POOL_K)
    mask_pooled = pooled_offsets < (out_features // POOL_K)
    block_sum = tl.sum(tl.where(mask_pooled, z_part, 0.0))
    
    # Atomic sum to final output
    tl.atomic_add(Out_ptr + m, block_sum * scale_factor)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        batch_size = x.shape[0]
        # Important: initialize output with zeros for atomic_add
        out = torch.zeros((batch_size,), device=x.device, dtype=x.dtype)
        
        # Adjust BLOCK_SIZE_N to be a multiple of kernel_size
        BS_N = 256
        if BS_N % self.kernel_size != 0:
            BS_N = ((BS_N + self.kernel_size - 1) // self.kernel_size) * self.kernel_size
            
        num_n_blocks = (self.out_features + BS_N - 1) // BS_N
        grid = (batch_size * num_n_blocks,)
        
        matmul_maxpool_sum_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, out,
            batch_size, self.in_features, self.out_features, self.kernel_size, self.scale_factor,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            self.matmul.bias.stride(0),
            out.stride(0),
            BLOCK_SIZE_N=BS_N, BLOCK_SIZE_K=64,
            POOL_K=self.kernel_size,
            num_warps=4
        )
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 2, 0.5]
