"""
Optimized BF16 GEMV using Triton for H100.
x @ weight.T where x is (B, K) and weight is (N, K), output is (B, N).
B=32, K=4096, N=14336. This is memory-bandwidth bound.

Key insight: Store weight transposed as (K, N) for coalesced reads along N dimension.
Each program tile reads a chunk of K and a chunk of N with coalesced access.
"""

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_N': 128, 'BLOCK_K': 128, 'BLOCK_B': 32}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_N': 128, 'BLOCK_K': 256, 'BLOCK_B': 32}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_N': 256, 'BLOCK_K': 128, 'BLOCK_B': 32}, num_warps=8, num_stages=4),
        triton.Config({'BLOCK_N': 256, 'BLOCK_K': 256, 'BLOCK_B': 32}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_N': 64, 'BLOCK_K': 256, 'BLOCK_B': 32}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_N': 128, 'BLOCK_K': 128, 'BLOCK_B': 32}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_N': 128, 'BLOCK_K': 64, 'BLOCK_B': 32}, num_warps=4, num_stages=4),
        triton.Config({'BLOCK_N': 64, 'BLOCK_K': 128, 'BLOCK_B': 32}, num_warps=4, num_stages=4),
    ],
    key=['K', 'N'],
)
@triton.jit
def gemv_kernel_transposed(
    X_ptr, WT_ptr, Out_ptr,
    K: tl.constexpr, N: tl.constexpr,
    BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    BLOCK_B: tl.constexpr,
):
    """
    X: (B, K), WT: (K, N), Out: (B, N)
    Out = X @ WT
    """
    pid_n = tl.program_id(0)
    pid_b = tl.program_id(1)
    
    n_offset = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    b_offset = pid_b * BLOCK_B + tl.arange(0, BLOCK_B)
    n_mask = n_offset < N
    
    acc = tl.zeros([BLOCK_B, BLOCK_N], dtype=tl.float32)
    
    for k_start in range(0, K, BLOCK_K):
        k_offset = k_start + tl.arange(0, BLOCK_K)
        
        # Load x[b_offset, k_offset] -> [BLOCK_B, BLOCK_K]
        x_offsets = b_offset[:, None] * K + k_offset[None, :]
        x_vals = tl.load(X_ptr + x_offsets)
        
        # Load wt[k_offset, n_offset] -> [BLOCK_K, BLOCK_N]  (coalesced along N!)
        wt_offsets = k_offset[:, None] * N + n_offset[None, :]
        wt_mask = n_mask[None, :]
        wt_vals = tl.load(WT_ptr + wt_offsets, mask=wt_mask, other=0.0)
        
        acc += tl.dot(x_vals, wt_vals, input_precision="tf32")
    
    out_offsets = b_offset[:, None] * N + n_offset[None, :]
    out_mask = n_mask[None, :]
    tl.store(Out_ptr + out_offsets, acc.to(tl.bfloat16), mask=out_mask)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)
        # We'll lazily create the transposed weight
        self._weight_t = None

    def _get_weight_t(self):
        if self._weight_t is None or self._weight_t.data_ptr() == 0:
            self._weight_t = self.weight.data.t().contiguous()
        return self._weight_t
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf16 = x.to(dtype=torch.bfloat16)
        B = x_bf16.shape[0]
        K = self.in_features
        N = self.out_features
        
        wt = self._get_weight_t()
        
        out = torch.empty(B, N, device=x.device, dtype=torch.bfloat16)
        
        grid = lambda META: (triton.cdiv(N, META['BLOCK_N']), triton.cdiv(B, META['BLOCK_B']))
        
        gemv_kernel_transposed[grid](
            x_bf16, wt, out,
            K, N,
        )
        
        return out


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16)]


def get_init_inputs():
    return []
