import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Direct kernel implementation for B200
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Simple kernel for GEMM + Bias + SiLU using __half2 vectorized operations
__global__ void gemm_bias_silu_kernel(
    const half* A,
    const half* B,
    const half* bias,
    half* C,
    int M,
    int N,
    int K
) {
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    
    int base_m = block_m * 32;
    int base_n = block_n * 32;
    
    if (base_m >= M || base_n >= N) return;
    
    // Each thread computes one output element
    half acc = __float2half(0.0f);
    
    for (int k = 0; k < K; ++k) {
        half a_val = A[(base_m + tx) * K + k];
        half b_val = B[k * N + (base_n + ty)];
        acc = __hadd(acc, __hmul(a_val, b_val));
    }
    
    // Add bias
    int out_n = base_n + ty;
    if (out_n < N) {
        half val = __hadd(acc, bias[out_n]);
        // SiLU: x * sigmoid(x) = x / (1 + exp(-x))
        float x_val = __half2float(val);
        float sigmoid_val = 1.0f / (1.0f + __expf(-x_val));
        C[(base_m + tx) * N + out_n] = __float2half(x_val * sigmoid_val);
    }
}

torch::Tensor gemm_bias_silu(torch::Tensor A, torch::Tensor B, torch::Tensor bias) {
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    dim3 grid((N + 31) / 32, (M + 31) / 32);
    dim3 block(32, 32);
    
    gemm_bias_silu_kernel<<<grid, block>>>(
        (half*)A.data_ptr<half>(),
        (half*)B.data_ptr<half>(),
        (half*)bias.data_ptr<half>(),
        (half*)C.data_ptr<half>(),
        M, N, K
    );
    
    return C;
}
'''

cpp_sources = 'torch::Tensor gemm_bias_silu(torch::Tensor A, torch::Tensor B, torch::Tensor bias);'

try:
    ext = load_inline(
        name='gemm_bias_silu_ext',
        cpp_sources=cpp_sources,
        cuda_sources=cuda_source,
        functions=['gemm_bias_silu'],
        verbose=False,
    )
    print('Extension loaded successfully!')
except Exception as e:
    print(f'Warning: Extension loading failed with error: {e}')
    print('Will use PyTorch fallback in forward pass')
    ext = None

class Model(nn.Module):
    def __init__(self, n: int = 4096):
        super().__init__()
        self.bias = nn.Parameter(torch.randn(n, dtype=torch.float16) * 0.02)

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        if ext:
            return ext.gemm_bias_silu(a.to(torch.float16), b.to(torch.float16), self.bias)
        else:
            # Fallback to PyTorch (only if extension failed to load)
            c = a.to(torch.float16) @ b.to(torch.float16)
            return torch.nn.functional.silu(c + self.bias.to(torch.float16))


def get_inputs():
    m = 2048
    n = 4096
    k = 2048
    return [torch.randn(m, k, dtype=torch.float16), torch.randn(k, n, dtype=torch.float16)]


def get_init_inputs():
    return []
