import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Strategy: Use CUTLASS GEMM with custom epilogue (divide + GELU)
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// CUTLASS headers
#include <cutlass/cutlass.h>
#include <cutlass/gemm/device/gemm.h>
#include <cutlass/epilogue/thread/linear_combination.h>
#include <cutlass/epilogue/thread/activation.h>

#include <math.h>

// ─────────────────────────────────────────────────────────────────────────────
// Custom epilogue: scale by (1/divisor) then GELU
// ─────────────────────────────────────────────────────────────────────────────

template <typename ElementOutput_, int Count>
struct EpilogueOpDivGelu {
    using ElementOutput = ElementOutput_;
    using ElementAccumulator = float;
    using ElementCompute = float;
    static const int kCount = Count;

    using FragmentOutput = cutlass::Array<ElementOutput, kCount>;
    using FragmentAccumulator = cutlass::Array<ElementAccumulator, kCount>;
    using FragmentCompute = cutlass::Array<ElementCompute, kCount>;

    struct Params {
        ElementCompute inv_divisor;
        CUTLASS_HOST_DEVICE Params(ElementCompute inv_d = 1.0f) : inv_divisor(inv_d) {}
    };

    CUTLASS_HOST_DEVICE EpilogueOpDivGelu(Params const &params) : inv_divisor_(params.inv_divisor) {}

    CUTLASS_HOST_DEVICE bool is_source_needed() const { return true; }
    CUTLASS_HOST_DEVICE bool is_source_tile_needed() const { return true; }

    CUTLASS_DEVICE FragmentOutput operator()(
        FragmentAccumulator const &accum,
        FragmentOutput const &source) const {

        FragmentOutput result;
        CUTLASS_PRAGMA_UNROLL
        for (int i = 0; i < kCount; ++i) {
            float v = float(accum[i]) + float(source[i]);
            v *= inv_divisor_;
            // GELU tanh approximation
            const float k0 = 0.7978845608028654f;
            const float k1 = 0.044715f;
            float c = k0 * (v + k1 * v * v * v);
            v = 0.5f * v * (1.0f + tanhf(c));
            result[i] = ElementOutput(v);
        }
        return result;
    }

    CUTLASS_DEVICE FragmentOutput operator()(
        FragmentAccumulator const &accum) const {

        FragmentOutput result;
        CUTLASS_PRAGMA_UNROLL
        for (int i = 0; i < kCount; ++i) {
            float v = float(accum[i]);
            v *= inv_divisor_;
            const float k0 = 0.7978845608028654f;
            const float k1 = 0.044715f;
            float c = k0 * (v + k1 * v * v * v);
            v = 0.5f * v * (1.0f + tanhf(c));
            result[i] = ElementOutput(v);
        }
        return result;
    }

private:
    ElementCompute inv_divisor_;
};

// ─────────────────────────────────────────────────────────────────────────────
// CUTLASS GEMM with standard LinearCombination epilogue (we'll fuse div+GELU
// separately with a fast kernel after since custom epilogue in CUTLASS 2.x
// requires more boilerplate)
// ─────────────────────────────────────────────────────────────────────────────

// Use FP16 Tensor Core GEMM
using ElementA = cutlass::half_t;
using ElementB = cutlass::half_t;
using ElementC = float;
using ElementAccumulator = float;

using LayoutA = cutlass::layout::RowMajor;
using LayoutB = cutlass::layout::ColumnMajor;
using LayoutC = cutlass::layout::RowMajor;

using EpilogueOp = cutlass::epilogue::thread::LinearCombination<
    ElementC, 128 / cutlass::sizeof_bits<ElementC>::value,
    ElementAccumulator, ElementAccumulator>;

using Gemm = cutlass::gemm::device::Gemm<
    ElementA, LayoutA,
    ElementB, LayoutB,
    ElementC, LayoutC,
    ElementAccumulator,
    cutlass::arch::OpClassTensorOp,
    cutlass::arch::Sm80,
    cutlass::gemm::GemmShape<128, 256, 32>,
    cutlass::gemm::GemmShape<64, 64, 32>,
    cutlass::gemm::GemmShape<16, 8, 16>,
    EpilogueOp,
    cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<8>,
    3
>;

// ─────────────────────────────────────────────────────────────────────────────
// Fused div + GELU kernel (fast, vectorized)
// ─────────────────────────────────────────────────────────────────────────────
__device__ __forceinline__ float fast_gelu(float x) {
    const float k0 = 0.7978845608028654f;
    const float k1 = 0.044715f;
    float c = k0 * (x + k1 * x * x * x);
    return 0.5f * x * (1.0f + tanhf(c));
}

__global__ void fused_div_gelu_kernel(float* __restrict__ C,
                                       int n, float inv_divisor) {
    int idx = (blockIdx.x * blockDim.x + threadIdx.x) * 4;
    if (idx + 3 < n) {
        float4 val = *reinterpret_cast<float4*>(C + idx);
        val.x = fast_gelu(val.x * inv_divisor);
        val.y = fast_gelu(val.y * inv_divisor);
        val.z = fast_gelu(val.z * inv_divisor);
        val.w = fast_gelu(val.w * inv_divisor);
        *reinterpret_cast<float4*>(C + idx) = val;
    } else {
        for (int i = idx; i < n; i++) {
            C[i] = fast_gelu(C[i] * inv_divisor);
        }
    }
}

// ─────────────────────────────────────────────────────────────────────────────
// FP16 intermediate: GEMM in fp16, output fp16, fuse div+gelu in fp16
// ─────────────────────────────────────────────────────────────────────────────

// GEMM C = A * B^T, then bias add, then div, then GELU
// A: [M,K] fp16, B: [N,K] fp16, bias: [N] float, C: [M,N] float
torch::Tensor linear_div_gelu(
    torch::Tensor x,       // [M, K] float32
    torch::Tensor weight,  // [N, K] float32
    torch::Tensor bias,    // [N]    float32
    float divisor) {

    int M = x.size(0);
    int K = x.size(1);
    int N = weight.size(0);

    // Convert to fp16 for tensor core
    auto x_fp16 = x.to(torch::kFloat16);
    auto w_fp16 = weight.to(torch::kFloat16);

    // Output buffer in float32
    auto C = torch::empty({M, N}, x.options());

    // Bias in float32
    auto bias_f32 = bias.to(torch::kFloat32);

    // CUTLASS expects: C = alpha * (A * B) + beta * C_in
    // We want: C = A * W^T + bias
    // A: [M,K] row-major, W: [N,K] row-major -> W^T is col-major as B
    // So B layout = ColumnMajor of W, which is same as RowMajor of W^T

    typename Gemm::Arguments args(
        {M, N, K},
        {(cutlass::half_t*)x_fp16.data_ptr(), K},
        {(cutlass::half_t*)w_fp16.data_ptr(), K},   // B in col-major = W row-major transposed
        {C.data_ptr<float>(), N},                    // C_in = bias broadcast (we handle separately)
        {C.data_ptr<float>(), N},
        {1.0f, 0.0f}  // alpha=1, beta=0 (no source)
    );

    Gemm gemm_op;
    auto status = gemm_op(args);
    if (status != cutlass::Status::kSuccess) {
        throw std::runtime_error("CUTLASS GEMM failed");
    }

    // Add bias
    if (bias.defined() && bias.numel() > 0) {
        // C: [M, N], bias: [N] -> broadcast add
        C.add_(bias_f32.unsqueeze(0));
    }

    // Fused div + GELU
    int n = M * N;
    float inv_divisor = 1.0f / divisor;
    int threads = 256;
    int blocks = (n / 4 + threads - 1) / threads;
    fused_div_gelu_kernel<<<blocks, threads>>>(C.data_ptr<float>(), n, inv_divisor);

    return C;
}
'''

cpp_source = '''
torch::Tensor linear_div_gelu(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float divisor);
'''

ext = load_inline(
    name='fused_linear_div_gelu_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['linear_div_gelu'],
    verbose=True,
    extra_cuda_cflags=[
        '-O3', '--use_fast_math', '-std=c++17',
        '--expt-relaxed-constexpr',
        f'-I/opt/cutlass/include',
        f'-I/opt/cutlass/tools/util/include',
    ],
    extra_ldflags=[],
)


class Model(nn.Module):
    def __init__(self, input_size, output_size, divisor):
        super().__init__()
        self.linear = nn.Linear(input_size, output_size)
        self.divisor = divisor

    def forward(self, x):
        return ext.linear_div_gelu(
            x,
            self.linear.weight,
            self.linear.bias,
            self.divisor
        )


batch_size = 128
input_size = 4096
output_size = 4096
divisor = 10.0


def get_inputs():
    return [torch.randn(batch_size, input_size)]


def get_init_inputs():
    return [input_size, output_size, divisor]
