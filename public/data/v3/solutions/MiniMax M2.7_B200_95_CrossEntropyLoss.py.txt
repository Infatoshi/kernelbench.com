import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fused Cross Entropy Loss kernel for B200 (SM100)
// Computes: -log(exp(logit[t]) / sum(exp(logit[i]))) for each sample

template <typename scalar_t>
__global__ void cross_entropy_forward_kernel(
    const scalar_t* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    scalar_t* __restrict__ output,
    int batch_size,
    int num_classes
) {
    int batch_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int tid = threadIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const scalar_t* pred_row = predictions + batch_idx * num_classes;
    int target_class = targets[batch_idx];
    
    // Thread-level max
    scalar_t local_max = (scalar_t)(-1e20f);
    for (int i = tid; i < num_classes; i += blockDim.x) {
        scalar_t val = pred_row[i];
        local_max = val > local_max ? val : local_max;
    }
    
    // Warp reduction for max using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        scalar_t other = __shfl_down_sync(0xffffffff, local_max, offset);
        local_max = other > local_max ? other : local_max;
    }
    scalar_t row_max = __shfl_sync(0xffffffff, local_max, 0);
    
    // Thread-level exp sum
    scalar_t local_sum = 0;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        scalar_t val = expf(pred_row[i] - row_max);
        local_sum += val;
    }
    
    // Warp reduction for sum using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    scalar_t row_sum = __shfl_sync(0xffffffff, local_sum, 0);
    
    // Compute cross entropy only from thread 0
    if (tid == 0) {
        scalar_t target_logit = pred_row[target_class];
        output[batch_idx] = logf(row_sum) + row_max - target_logit;
    }
}

torch::Tensor cross_entropy_forward(torch::Tensor predictions, torch::Tensor targets) {
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    auto output = torch::empty({batch_size}, predictions.options());
    
    const int threads = 256;
    const int blocks = (batch_size + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES(predictions.scalar_type(), "cross_entropy_forward", ([&] {
        cross_entropy_forward_kernel<scalar_t><<<blocks, threads>>>(
            predictions.data_ptr<scalar_t>(),
            targets.data_ptr<int64_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            num_classes
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources='''
torch::Tensor cross_entropy_forward(torch::Tensor predictions, torch::Tensor targets);
''',
    cuda_sources=cuda_source,
    functions=['cross_entropy_forward'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Cross Entropy Loss for B200 using fused CUDA kernels.
    """
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [
        torch.randn(batch_size, num_classes, device='cuda'),
        torch.randint(0, num_classes, (batch_size,), device='cuda')
    ]

def get_init_inputs():
    return []
