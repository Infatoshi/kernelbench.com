import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _sum_dim1_kernel_fp32acc(X, Y, B: tl.constexpr, D1: tl.constexpr, D2: tl.constexpr,
                             BLOCK_D2: tl.constexpr, BLOCK_R: tl.constexpr):
    pid0 = tl.program_id(0)
    pid1 = tl.program_id(1)

    b = pid0
    d2_block = pid1
    offs_d2 = d2_block * BLOCK_D2 + tl.arange(0, BLOCK_D2)
    mask_d2 = offs_d2 < D2

    acc = tl.zeros([BLOCK_D2], dtype=tl.float32)
    for r in range(0, D1, BLOCK_R):
        offs_r = r + tl.arange(0, BLOCK_R)
        mask_r = offs_r < D1
        ptrs = X + b * D1 * D2 + offs_r[:, None] * D2 + offs_d2[None, :]
        x = tl.load(ptrs, mask=mask_r[:, None] & mask_d2[None, :], other=0.0)
        acc += tl.sum(x, axis=0)

    y_ptrs = Y + b * D2 + offs_d2
    tl.store(y_ptrs, acc, mask=mask_d2)


class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.dim != 1:
            return torch.sum(x, dim=self.dim, keepdim=True)
        if not x.is_cuda:
            return torch.sum(x, dim=1, keepdim=True)
        if x.ndim != 3:
            return torch.sum(x, dim=1, keepdim=True)
        if not x.is_contiguous():
            x = x.contiguous()

        B, D1, D2 = x.shape
        y = torch.empty((B, D2), device=x.device, dtype=torch.float32)
        grid = (B, triton.cdiv(D2, 128))
        _sum_dim1_kernel_fp32acc[grid](x, y, B, D1, D2, BLOCK_D2=128, BLOCK_R=32)
        return y.to(dtype=x.dtype).unsqueeze(1)


batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1


def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

    return [reduce_dim]
