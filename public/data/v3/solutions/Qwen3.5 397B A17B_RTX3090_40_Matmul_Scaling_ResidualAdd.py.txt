cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Tiled GEMM kernel with shared memory and fused scale+residual
// output = (input @ weight.T + bias) * scale_multiplier
template<typename scalar_t, int TILE_M, int TILE_N, int TILE_K>
__global__ void tiled_fused_kernel_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features,
    float scale_multiplier
) {
    // Shared memory tiles
    __shared__ float input_tile[TILE_M][TILE_K];
    __shared__ float weight_tile[TILE_N][TILE_K];
    
    // Output tile position
    int row_start = blockIdx.y * TILE_M;
    int col_start = blockIdx.x * TILE_N;
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Accumulator for this thread
    float acc = 0.0f;
    
    // Loop over K tiles
    for (int k_tile = 0; k_tile < in_features; k_tile += TILE_K) {
        // Load input tile (coalesced)
        int input_row = row_start + ty;
        int input_col = k_tile + tx;
        if (input_row < batch_size && input_col < in_features) {
            input_tile[ty][tx] = input[input_row * in_features + input_col];
        } else {
            input_tile[ty][tx] = 0.0f;
        }
        
        // Load weight tile (transposed for coalesced access)
        int weight_row = col_start + ty;
        int weight_col = k_tile + tx;
        if (weight_row < out_features && weight_col < in_features) {
            weight_tile[ty][tx] = weight[weight_row * in_features + weight_col];
        } else {
            weight_tile[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial dot product
        for (int k = 0; k < TILE_K; k++) {
            float a = input_tile[ty][k];
            float b = weight_tile[tx][k];
            acc += a * b;
        }
        
        __syncthreads();
    }
    
    // Add bias and apply scale
    int out_row = row_start + ty;
    int out_col = col_start + tx;
    if (out_row < batch_size && out_col < out_features) {
        float result = acc + bias[out_col];
        result *= scale_multiplier;
        output[out_row * out_features + out_col] = result;
    }
}

            __nv_bfloat16 a = input_tile[ty][k];

            acc += __bfloat162float(a) * __bfloat162float(b);
        }

    }
    
    // Add bias and apply scale
    int out_row = row_start + ty;

    if (out_row < batch_size && out_col < out_features) {

torch::Tensor fused_matmul_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scaling_factor
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    float scale_multiplier = 1.0f + scaling_factor;
    
    auto output = torch::empty({batch_size, out_features}, input.options());
    
    // Use 16x16 thread block with 32 K-dimension tile
    dim3 threads(16, 16);
    dim3 blocks((out_features + 15) / 16, (batch_size + 15) / 16);
    
    if (input.dtype() == torch::kFloat16) {
        tiled_fused_kernel_fp16<16, 16, 32><<<blocks, threads>>>(
            (half*)input.data_ptr(),
            (half*)weight.data_ptr(),
            (half*)bias.data_ptr(),
            (half*)output.data_ptr(),
            batch_size, in_features, out_features, scale_multiplier
        );
    } else if (input.dtype() == torch::kBFloat16) {
        tiled_fused_kernel_bf16<16, 16, 32><<<blocks, threads>>>(
            (__nv_bfloat16*)input.data_ptr(),
            (__nv_bfloat16*)weight.data_ptr(),
            (__nv_bfloat16*)bias.data_ptr(),
            (__nv_bfloat16*)output.data_ptr(),
            batch_size, in_features, out_features, scale_multiplier
        );
    } else {
        // FP32
        tiled_fused_kernel_fp32<float, 16, 16, 32><<<blocks, threads>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size, in_features, out_features, scale_multiplier
        );
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(cudaGetErrorString(err));
    }
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_matmul_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scaling_factor
);
"""

# Compile the extension
ext = load_inline(
    name='fused_matmul_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_scale_residual'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '-use_fast_math'],
    extra_cflags=['-O3'],
    verbose=False,
)


class Model(nn.Module):
    """
    A model that performs a matrix multiplication, scaling, and residual addition.
    Uses fused CUDA kernel for optimization.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Forward pass: output = (x @ weight.T + bias) * (1 + scaling_factor)
        """
        weight = self.matmul.weight
        bias = self.matmul.bias
        return ext.fused_matmul_scale_residual(x, weight, bias, self.scaling_factor)


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    scaling_factor = 0.5
    return [in_features, out_features, scaling_factor]
