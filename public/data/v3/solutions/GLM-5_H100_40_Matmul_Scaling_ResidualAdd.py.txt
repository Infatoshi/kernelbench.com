import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cuda_fp16.h>
#include <cublas_v2.h>

// Fused scale and residual add kernel: Y = X * scale + X = X * (scale + 1)
template <typename scalar_t>
__global__ void scale_residual_kernel(
    scalar_t* __restrict__ Y,
    const float scaling_factor,
    const int N) {
    
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        float val = static_cast<float>(Y[idx]);
        val *= (scaling_factor + 1.0f);
        Y[idx] = static_cast<scalar_t>(val);
    }
}

// Create a global cuBLAS handle
static cublasHandle_t get_cublas_handle() {
    static cublasHandle_t handle = nullptr;
    if (handle == nullptr) {
        cublasCreate(&handle);
    }
    return handle;
}

torch::Tensor fused_matmul_scale_residual(
    torch::Tensor X,
    torch::Tensor W,
    torch::Tensor bias,
    double scaling_factor) {
    
    const int M = X.size(0);
    const int K = X.size(1);
    const int N = W.size(0);
    
    // Create output tensor
    auto Y = torch::empty({M, N}, X.options());
    
    // Get cuBLAS handle
    cublasHandle_t handle = get_cublas_handle();
    
    // cuBLAS GEMM: Y = alpha * op(A) * op(B) + beta * C
    // For X @ W^T: A = X (MxK), B = W (NxK), op(B) = B^T
    // Y = X @ W^T + bias (broadcasted)
    
    const float alpha = 1.0f;
    const float beta = 0.0f;
    
    // Add bias by initializing Y with bias values (broadcasted)
    // We'll do this by first computing Y = X @ W^T, then adding bias
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, X.scalar_type(), "fused_matmul", [&] {
        using scalar_t = scalar_t;
        
        // Y = X @ W^T
        if constexpr (std::is_same<scalar_t, at::Half>::value) {
            // Use half-precision GEMM
            half alpha_h = __float2half(1.0f);
            half beta_h = __float2half(0.0f);
            
            // Y = alpha * X * W^T + beta * Y
            // X is MxK, W is NxK, Y is MxN
            cublasGemmEx(handle,
                CUBLAS_OP_N, CUBLAS_OP_T,  // op(B) = B^T
                N, M, K,  // dimensions: N x M x K (note: cuBLAS uses column-major)
                &alpha_h,
                W.data_ptr<scalar_t>(), K,  // B (N rows, K cols in row-major = KxN in col-major)
                X.data_ptr<scalar_t>(), K,  // A (M rows, K cols in row-major = KxM in col-major)
                &beta_h,
                Y.data_ptr<scalar_t>(), N,  // C (M rows, N cols in row-major = NxM in col-major)
                CUDA_R_16F, CUBLAS_GEMM_DEFAULT_TENSOR_OP);
        } else {
            // BFloat16 or float32
            cublasGemmEx(handle,
                CUBLAS_OP_N, CUBLAS_OP_T,
                N, M, K,
                &alpha,
                W.data_ptr<scalar_t>(), K,
                X.data_ptr<scalar_t>(), K,
                &beta,
                Y.data_ptr<scalar_t>(), N,
                std::is_same<scalar_t, at::BFloat16>::value ? CUDA_R_16BF : CUDA_R_32F,
                CUBLAS_GEMM_DEFAULT_TENSOR_OP);
        }
        
        // Add bias to each row
        // This is a bit tricky - we need to add bias[n] to Y[m, n] for all m
        // We can do this with a simple kernel or use cuBLAS ger
        // For simplicity, let's add bias and do scaling in one kernel
    });
    
    // Fused kernel: add bias and apply scale + residual
    // Y[m,n] = (Y[m,n] + bias[n]) * (scale + 1)
    const int total_elements = M * N;
    const int threads = 256;
    const int blocks = (total_elements + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, X.scalar_type(), "add_bias_scale", [&] {
        using scalar_t = scalar_t;
        
        // Launch kernel that adds bias and scales
        auto bias_ptr = bias.data_ptr<scalar_t>();
        auto y_ptr = Y.data_ptr<scalar_t>();
        
        // We need a custom kernel here for the bias addition + scaling
        // Let me launch a separate kernel for this
    });
    
    // Actually, let's just do this in PyTorch for now to verify the GEMM is correct
    // Then we'll optimize
    
    return Y;
}

// Combined kernel: add bias and apply fused scaling + residual
template <typename scalar_t>
__global__ void add_bias_scale_residual_kernel(
    scalar_t* __restrict__ Y,
    const scalar_t* __restrict__ bias,
    const float scaling_factor,
    const int M, const int N) {
    
    const int idx = blockIdx.x * blockDim.x + threadIdx.x;
    const int total = M * N;
    
    if (idx < total) {
        const int row = idx / N;
        const int col = idx % N;
        
        float val = static_cast<float>(Y[idx]);
        val += static_cast<float>(bias[col]);  // Add bias
        val *= (scaling_factor + 1.0f);  // Scale + residual
        
        Y[idx] = static_cast<scalar_t>(val);
    }
}

torch::Tensor matmul_bias_scale_residual(
    torch::Tensor X,
    torch::Tensor W,
    torch::Tensor bias,
    double scaling_factor) {
    
    const int M = X.size(0);
    const int K = X.size(1);
    const int N = W.size(0);
    
    // Create output tensor
    auto Y = torch::empty({M, N}, X.options());
    
    // Get cuBLAS handle
    cublasHandle_t handle = get_cublas_handle();
    
    const float alpha = 1.0f;
    const float beta = 0.0f;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, X.scalar_type(), "fused_matmul", [&] {
        using scalar_t = scalar_t;
        
        // Y = X @ W^T
        cublasGemmEx(handle,
            CUBLAS_OP_N, CUBLAS_OP_T,
            N, M, K,
            &alpha,
            W.data_ptr<scalar_t>(), K,
            X.data_ptr<scalar_t>(), K,
            &beta,
            Y.data_ptr<scalar_t>(), N,
            std::is_same<scalar_t, at::Half>::value ? CUDA_R_16F : 
            (std::is_same<scalar_t, at::BFloat16>::value ? CUDA_R_16BF : CUDA_R_32F),
            CUBLAS_GEMM_DEFAULT_TENSOR_OP);
        
        // Add bias and apply scale + residual
        const int total = M * N;
        const int threads = 256;
        const int blocks = (total + threads - 1) / threads;
        
        add_bias_scale_residual_kernel<<<blocks, threads>>>(
            Y.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            static_cast<float>(scaling_factor),
            M, N
        );
    });
    
    return Y;
}

cpp_source = r'''
torch::Tensor matmul_bias_scale_residual(torch::Tensor X, torch::Tensor W, torch::Tensor bias, double scaling_factor);
'''

ext = load_inline(
    name='fused_matmul',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['matmul_bias_scale_residual'],
    verbose=True,
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    extra_ldflags=['-lcublas'],
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Create weight and bias parameters (matching nn.Linear format)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize parameters
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        # Ensure input is on the same device and dtype
        x = x.to(self.weight.dtype)
        
        # Use fused kernel
        return ext.matmul_bias_scale_residual(x, self.weight, self.bias, self.scaling_factor)

batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 0.5

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, scaling_factor]
