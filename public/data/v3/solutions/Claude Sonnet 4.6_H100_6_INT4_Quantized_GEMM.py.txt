import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized INT4 Weight-Only Quantized GEMM
# Strategy: Triton kernel with fused INT4 unpack + dequant + GEMM
# Uses tensor cores via tl.dot with efficient tiling

import triton
import triton.language as tl


@triton.jit
def int4_gemm_kernel(
    # Pointers
    x_ptr,           # (M, K) fp16
    w_packed_ptr,    # (N, K//2) uint8 - INT4 packed weights
    scales_ptr,      # (N, num_groups) fp16
    out_ptr,         # (M, N) fp16
    # Dimensions
    M, N, K,
    num_groups,
    group_size,
    # Strides
    stride_xm, stride_xk,
    stride_wn, stride_wk,   # wk is in packed (K//2) dimension
    stride_sn, stride_sg,
    stride_om, stride_on,
    # Tile sizes
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """
    INT4 Weight-Only Quantized GEMM:
    - x: (M, K) fp16 activations
    - w_packed: (N, K//2) uint8 - 2 INT4 weights per byte
    - scales: (N, num_groups) fp16 - group quantization scales
    - out: (M, N) fp16
    
    Dequant formula: w_fp16[n,k] = scales[n, k//group_size] * (w_int4[n,k] - 8)
    """
    # Program ID
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    
    # L2 cache-friendly swizzling
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % num_pid_in_group) % group_size_m
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Offsets for this tile
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    
    # Accumulator
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Loop over K dimension
    # BLOCK_K must be divisible by 2 for INT4 packing
    # Also must align with group_size for clean scale loading
    for k_start in tl.range(0, K, BLOCK_K):
        offs_k = k_start + tl.arange(0, BLOCK_K)
        
        # Load activation tile: (BLOCK_M, BLOCK_K)
        x_ptrs = x_ptr + offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk
        x_mask = (offs_m[:, None] < M) & (offs_k[None, :] < K)
        x_tile = tl.load(x_ptrs, mask=x_mask, other=0.0).to(tl.float16)
        
        # Load packed weight tile: (BLOCK_N, BLOCK_K//2)
        # Each byte contains 2 INT4 values
        offs_k_packed = k_start // 2 + tl.arange(0, BLOCK_K // 2)
        w_ptrs = w_packed_ptr + offs_n[:, None] * stride_wn + offs_k_packed[None, :] * stride_wk
        w_mask = (offs_n[:, None] < N) & (offs_k_packed[None, :] < (K // 2))
        w_packed = tl.load(w_ptrs, mask=w_mask, other=0).to(tl.uint8)
        
        # Unpack INT4: extract low and high nibbles
        w_low = (w_packed & 0x0F).to(tl.float16)   # (BLOCK_N, BLOCK_K//2)
        w_high = ((w_packed >> 4) & 0x0F).to(tl.float16)  # (BLOCK_N, BLOCK_K//2)
        
        # Interleave to get (BLOCK_N, BLOCK_K)
        # w_unpacked[:, 0::2] = w_low, w_unpacked[:, 1::2] = w_high
        # Use reshape/interleave trick
        # Stack along last dim and reshape: (BLOCK_N, BLOCK_K//2, 2) -> (BLOCK_N, BLOCK_K)
        w_unpacked = tl.interleave(w_low, w_high)  # (BLOCK_N, BLOCK_K)
        
        # Apply dequantization: (w - 8) * scale
        # Load scales for this k range
        # For the k range [k_start, k_start + BLOCK_K), groups are k // group_size
        # We process BLOCK_K elements, spanning ceil(BLOCK_K/group_size) groups
        # When BLOCK_K == group_size, exactly 1 group per k-tile
        
        # Get group index for this k-block
        g = k_start // group_size
        
        # Load scale for this (n, g) pair: (BLOCK_N,)
        scale_ptrs = scales_ptr + offs_n * stride_sn + g * stride_sg
        scale_mask = offs_n < N
        scales_tile = tl.load(scale_ptrs, mask=scale_mask, other=0.0).to(tl.float16)  # (BLOCK_N,)
        
        # Dequantize: (w_int - 8) * scale
        w_dequant = (w_unpacked - 8.0) * scales_tile[:, None]  # (BLOCK_N, BLOCK_K)
        
        # Matrix multiply: x_tile (BLOCK_M, BLOCK_K) @ w_dequant.T (BLOCK_K, BLOCK_N)
        acc += tl.dot(x_tile, tl.trans(w_dequant), allow_tf32=False)
    
    # Store output
    out_ptrs = out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on
    out_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(out_ptrs, acc.to(tl.float16), mask=out_mask)


def int4_gemm_triton(x, weight_packed, scales, K, N, group_size):
    """
    INT4 Weight-Only GEMM with fused dequantization.
    
    x: (M, K) fp16
    weight_packed: (N, K//2) uint8
    scales: (N, num_groups) fp16
    Returns: (M, N) fp16
    """
    M = x.shape[0]
    num_groups = K // group_size
    
    out = torch.empty((M, N), dtype=torch.float16, device=x.device)
    
    # Tile configuration - BLOCK_K must equal group_size for clean scale loading
    # group_size = 128, so BLOCK_K = 128
    BLOCK_M = 64
    BLOCK_N = 64
    BLOCK_K = group_size  # = 128, aligns with group boundaries
    GROUP_SIZE_M = 8
    
    grid = lambda meta: (
        triton.cdiv(M, meta['BLOCK_M']) * triton.cdiv(N, meta['BLOCK_N']),
    )
    
    int4_gemm_kernel[grid](
        x, weight_packed, scales, out,
        M, N, K,
        num_groups, group_size,
        x.stride(0), x.stride(1),
        weight_packed.stride(0), weight_packed.stride(1),
        scales.stride(0), scales.stride(1),
        out.stride(0), out.stride(1),
        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
    )
    
    return out


class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even for INT4 packing"

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        
        # Reshape to 2D
        x_2d = x.view(-1, self.K)  # (M, K)
        
        # Fused INT4 unpack + dequant + GEMM
        out = int4_gemm_triton(
            x_2d, self.weight_packed, self.scales,
            self.K, self.N, self.group_size
        )
        
        return out.view(batch_size, seq_len, self.N)


# Configuration sized for LLM inference workloads
batch_size = 4
seq_len = 2048
K = 4096
N = 11008
group_size = 128


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [K, N, group_size]
