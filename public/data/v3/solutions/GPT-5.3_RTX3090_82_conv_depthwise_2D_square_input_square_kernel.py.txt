import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void dwconv2d_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ w,
    const scalar_t* __restrict__ b,
    scalar_t* __restrict__ y,
    int N,int C,int H,int W,
    int K,int stride,int pad,
    int H_out,int W_out)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*C*H_out*W_out;
    if(idx>=total) return;

    int ow = idx % W_out;
    idx /= W_out;
    int oh = idx % H_out;
    idx /= H_out;
    int c = idx % C;
    int n = idx / C;

    float acc = 0.0f;

    for(int kh=0; kh<K; ++kh){
        for(int kw=0; kw<K; ++kw){
            int ih = oh*stride + kh - pad;
            int iw = ow*stride + kw - pad;
            if(ih>=0 && ih<H && iw>=0 && iw<W){
                int x_idx = ((n*C + c)*H + ih)*W + iw;
                int w_idx = (c*K + kh)*K + kw;
                acc += (float)x[x_idx] * (float)w[w_idx];
            }
        }
    }

    if(b!=nullptr) acc += (float)b[c];

    int y_idx = ((n*C + c)*H_out + oh)*W_out + ow;
    y[y_idx] = (scalar_t)acc;
}

torch::Tensor dwconv_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, int stride, int pad){
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int K = w.size(2);

    int H_out = (H + 2*pad - K)/stride + 1;
    int W_out = (W + 2*pad - K)/stride + 1;

    auto y = torch::empty({N,C,H_out,W_out}, x.options());

    int total = N*C*H_out*W_out;
    int threads = 256;
    int blocks = (total + threads - 1)/threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "dwconv_forward", ([&] {
        const scalar_t* b_ptr = nullptr;
        if(b.defined()) b_ptr = b.data_ptr<scalar_t>();
        dwconv2d_kernel<scalar_t><<<blocks,threads>>>(
            x.data_ptr<scalar_t>(),
            w.data_ptr<scalar_t>(),
            b_ptr,
            y.data_ptr<scalar_t>(),
            N,C,H,W,K,stride,pad,H_out,W_out);
    }));

    return y;
}
'''

cpp_src = "torch::Tensor dwconv_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, int stride, int pad);"

ext = load_inline(
    name='dwconv_ext',
    cpp_sources=cpp_src,
    cuda_sources=cuda_src,
    functions=['dwconv_forward'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.conv2d.weight
        b = self.conv2d.bias if self.conv2d.bias is not None else torch.Tensor()
        return ext.dwconv_forward(x, w, b, self.stride, self.padding)


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width, device='cuda')
    return [x]


def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
