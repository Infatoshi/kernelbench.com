import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def gelu_kernel(
    x_ptr,
    y_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # GELU exact: x * 0.5 * (1 + erf(x / sqrt(2)))
    # Use tanh approximation for speed:
    # x * 0.5 * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    
    # Exact GELU using erf
    inv_sqrt2 = 0.7071067811865476
    x_fp32 = x.to(tl.float32)
    cdf = 0.5 * (1.0 + tl.math.erf(x_fp32 * inv_sqrt2))
    result = x_fp32 * cdf
    
    y = result.to(x.dtype)
    tl.store(y_ptr + offsets, y, mask=mask)


def gelu_triton(x: torch.Tensor) -> torch.Tensor:
    y = torch.empty_like(x)
    n = x.numel()
    
    # Use large block size for memory bandwidth efficiency
    BLOCK_SIZE = 2048
    grid = (triton.cdiv(n, BLOCK_SIZE),)
    
    gelu_kernel[grid](x, y, n, BLOCK_SIZE=BLOCK_SIZE)
    return y


class Model(nn.Module):
    """
    Optimized GELU activation using Triton kernel.
    Supports fp16, bf16, fp32.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return gelu_triton(x)
