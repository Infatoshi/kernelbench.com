import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def _cross_entropy_kernel(
    logits_ptr, targets_ptr, loss_ptr,
    N, C,
    stride_n, stride_c,
    BLOCK_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0)
    # Offset the pointers to the current row
    row_logits_ptr = logits_ptr + row_idx * stride_n
    
    # Load logits with masking if C is not a power of 2
    offsets = tl.arange(0, BLOCK_SIZE)
    mask = offsets < C
    logits = tl.load(row_logits_ptr + offsets * stride_c, mask=mask, other=-float('inf'))
    
    # Numerical stability: subtract max(row)
    m = tl.max(logits, axis=0)
    
    # log_sum_exp = m + log(sum(exp(x - m)))
    exp_logits = tl.exp(logits - m)
    sum_exp = tl.sum(exp_logits, axis=0)
    log_sum_exp = tl.log(sum_exp) + m
    
    # Index of current target
    target_idx = tl.load(targets_ptr + row_idx).to(tl.int32)
    # The logit of the correct class
    target_logit_ptr = row_logits_ptr + target_idx * stride_c
    target_logit = tl.load(target_logit_ptr)
    
    # Loss for current row
    loss = log_sum_exp - target_logit
    tl.store(loss_ptr + row_idx, loss)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        # predictions (N, C), targets (N,)
        # Note: torch.nn.functional.cross_entropy does mean reduction by default.
        N, C = predictions.shape
        loss_per_row = torch.empty(N, device=predictions.device, dtype=torch.float32)
        
        # Determine BLOCK_SIZE
        # For larger C, we need to handle it properly. For now we assume C fits in memory.
        # Ampere supports up to 64KB shared memory per block, so we might need to split if C is very large.
        BLOCK_SIZE = triton.next_power_of_2(C)
        
        grid = (N,)
        _cross_entropy_kernel[grid](
            predictions, targets, loss_per_row,
            N, C,
            predictions.stride(0), predictions.stride(1),
            BLOCK_SIZE=BLOCK_SIZE
        )
        
        return loss_per_row.mean()
