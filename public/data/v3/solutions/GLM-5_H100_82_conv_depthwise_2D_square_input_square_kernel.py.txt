import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized depthwise convolution kernel for H100
// Each thread block processes one channel across multiple output pixels
// Uses shared memory for kernel weights and input tiles

template<typename scalar_t, int KERNEL_SIZE, int TILE_SIZE>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int stride,
    int padding
) {
    // Shared memory for kernel weights (small, fits easily)
    __shared__ scalar_t kernel_shared[KERNEL_SIZE * KERNEL_SIZE];
    
    // Load kernel weights into shared memory
    int tid = threadIdx.x;
    int kernel_elems = KERNEL_SIZE * KERNEL_SIZE;
    if (tid < kernel_elems) {
        kernel_shared[tid] = weight[tid];
    }
    __syncthreads();
    
    // Determine which batch and channel this block processes
    int batch = blockIdx.z / channels;
    int channel = blockIdx.z % channels;
    
    // Base pointers for this batch and channel
    const scalar_t* input_ptr = input + (batch * channels + channel) * in_height * in_width;
    scalar_t* output_ptr = output + (batch * channels + channel) * out_height * out_width;
    
    // Each thread processes multiple output pixels along x dimension
    int out_y = blockIdx.y;
    int out_x_base = blockIdx.x * TILE_SIZE;
    
    // Process TILE_SIZE output pixels
    for (int t = 0; t < TILE_SIZE; t++) {
        int out_x = out_x_base + threadIdx.x + t * blockDim.x;
        
        if (out_y < out_height && out_x < out_width) {
            scalar_t sum = static_cast<scalar_t>(0.0f);
            
            // Compute convolution at this output position
            for (int ky = 0; ky < KERNEL_SIZE; ky++) {
                for (int kx = 0; kx < KERNEL_SIZE; kx++) {
                    int in_y = out_y * stride - padding + ky;
                    int in_x = out_x * stride - padding + kx;
                    
                    // Bounds check (for padding)
                    if (in_y >= 0 && in_y < in_height && in_x >= 0 && in_x < in_width) {
                        sum += input_ptr[in_y * in_width + in_x] * kernel_shared[ky * KERNEL_SIZE + kx];
                    }
                }
            }
            
            output_ptr[out_y * out_width + out_x] = sum;
        }
    }
}

// Optimized version using vectorized loads for contiguous memory access
template<typename scalar_t, int KERNEL_SIZE>
__global__ void depthwise_conv2d_optimized(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int stride,
    int padding
) {
    // Shared memory for kernel weights
    __shared__ scalar_t kernel_shared[KERNEL_SIZE * KERNEL_SIZE];
    
    // Load kernel weights into shared memory
    if (threadIdx.x < KERNEL_SIZE * KERNEL_SIZE) {
        kernel_shared[threadIdx.x] = weight[threadIdx.x];
    }
    __syncthreads();
    
    // Each block processes one (batch, channel) pair
    int batch = blockIdx.z / channels;
    int channel = blockIdx.z % channels;
    
    const scalar_t* input_ptr = input + (batch * channels + channel) * in_height * in_width;
    scalar_t* output_ptr = output + (batch * channels + channel) * out_height * out_width;
    
    // 2D thread block: threadIdx.y for output rows, threadIdx.x for output cols
    int out_y = blockIdx.y * blockDim.y + threadIdx.y;
    int out_x = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (out_y < out_height && out_x < out_width) {
        scalar_t sum = static_cast<scalar_t>(0.0f);
        
        // Unrolled convolution loop for kernel_size=3
        #pragma unroll
        for (int ky = 0; ky < KERNEL_SIZE; ky++) {
            int in_y = out_y * stride - padding + ky;
            if (in_y >= 0 && in_y < in_height) {
                #pragma unroll
                for (int kx = 0; kx < KERNEL_SIZE; kx++) {
                    int in_x = out_x * stride - padding + kx;
                    if (in_x >= 0 && in_x < in_width) {
                        sum += input_ptr[in_y * in_width + in_x] * kernel_shared[ky * KERNEL_SIZE + kx];
                    }
                }
            }
        }
        
        output_ptr[out_y * out_width + out_x] = sum;
    }
}

// Launch configuration optimized for H100
torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int64_t stride,
    int64_t padding
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    int kernel_size = weight.size(2);  // Square kernel
    
    int out_height = (in_height + 2 * padding - kernel_size) / stride + 1;
    int out_width = (in_width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    // Configure kernel launch
    int threads_x = 16;
    int threads_y = 16;
    dim3 threads(threads_x, threads_y);
    dim3 blocks(
        (out_width + threads_x - 1) / threads_x,
        (out_height + threads_y - 1) / threads_y,
        batch_size * channels
    );
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "depthwise_conv2d", ([&] {
        if (kernel_size == 3) {
            depthwise_conv2d_optimized<scalar_t, 3><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                in_height, in_width,
                out_height, out_width,
                stride, padding
            );
        } else if (kernel_size == 5) {
            depthwise_conv2d_optimized<scalar_t, 5><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                in_height, in_width,
                out_height, out_width,
                stride, padding
            );
        } else if (kernel_size == 7) {
            depthwise_conv2d_optimized<scalar_t, 7><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                in_height, in_width,
                out_height, out_width,
                stride, padding
            );
        } else {
            // Fallback for other kernel sizes
            depthwise_conv2d_optimized<scalar_t, 3><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                in_height, in_width,
                out_height, out_width,
                stride, padding
            );
        }
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int64_t stride,
    int64_t padding
);
'''

ext = load_inline(
    name='depthwise_conv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_forward'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.

    Args:
        in_channels (int): Number of channels in the input tensor.
        kernel_size (int): Size of the convolution kernel.
        stride (int, optional): Stride of the convolution. Defaults to 1.
        padding (int, optional): Padding applied to the input. Defaults to 0.
        bias (bool, optional): If `True`, adds a learnable bias to the output. Defaults to `False`.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
        # Create conv2d wrapper to match reference state_dict structure
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the depthwise 2D convolution.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, height, width).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, in_channels, height_out, width_out).
        """
        # Use conv2d.weight which has shape (in_channels, 1, kernel_size, kernel_size)
        # Squeeze to get (in_channels, kernel_size, kernel_size) for our custom kernel
        weight = self.conv2d.weight.squeeze(1)
        
        return ext.depthwise_conv2d_forward(x, weight, self.stride, self.padding)
