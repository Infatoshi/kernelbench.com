import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

__global__ void fused_conv_pool_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int B, int IN_C, int IN_H, int IN_W,
    int OUT_C, int OUT_H, int OUT_W,
    int K, int POOL_K,
    float sub1, float sub2) {

    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int oc_b = blockIdx.z;

    if (ow >= OUT_W || oh >= OUT_H) return;

    int b = oc_b / OUT_C;
    int oc = oc_b % OUT_C;

    float pool_sum = 0.0f;
    float b_val = bias[oc];

    // Load weight for this output channel once.
    // Small K and IN_C (e.g. 3x3x3=27) fit in registers.
    float local_w[27]; // Sufficient for IN_C=3, K=3
    if (IN_C == 3 && K == 3) {
        for (int i = 0; i < 27; ++i) {
            local_w[i] = weight[oc * 27 + i];
        }
    }

    for (int ph = 0; ph < POOL_K; ++ph) {
        for (int pw = 0; pw < POOL_K; ++pw) {
            int ch = oh * POOL_K + ph;
            int cw = ow * POOL_K + pw;

            float conv_acc = b_val;
            if (IN_C == 3 && K == 3) {
                for (int ic = 0; ic < 3; ++ic) {
                    int in_base = (b * 3 + ic) * IN_H * IN_W + ch * IN_W + cw;
                    conv_acc += input[in_base] * local_w[ic * 9 + 0];
                    conv_acc += input[in_base + 1] * local_w[ic * 9 + 1];
                    conv_acc += input[in_base + 2] * local_w[ic * 9 + 2];
                    conv_acc += input[in_base + IN_W] * local_w[ic * 9 + 3];
                    conv_acc += input[in_base + IN_W + 1] * local_w[ic * 9 + 4];
                    conv_acc += input[in_base + IN_W + 2] * local_w[ic * 9 + 5];
                    conv_acc += input[in_base + 2 * IN_W] * local_w[ic * 9 + 6];
                    conv_acc += input[in_base + 2 * IN_W + 1] * local_w[ic * 9 + 7];
                    conv_acc += input[in_base + 2 * IN_W + 2] * local_w[ic * 9 + 8];
                }
            } else {
                for (int ic = 0; ic < IN_C; ++ic) {
                    int in_c_offset = (b * IN_C + ic) * IN_H * IN_W;
                    int weight_oc_ic_offset = (oc * IN_C + ic) * K * K;
                    for (int kh = 0; kh < K; ++kh) {
                        int ih = ch + kh;
                        int in_row_offset = in_c_offset + ih * IN_W;
                        int weight_rows = weight_oc_ic_offset + kh * K;
                        for (int kw = 0; kw < K; ++kw) {
                            int iw = cw + kw;
                            conv_acc += input[in_row_offset + iw] * weight[weight_rows + kw];
                        }
                    }
                }
            }
            pool_sum += tanhf(conv_acc - sub1) - sub2;
        }
    }

    output[((b * OUT_C + oc) * OUT_H + oh) * OUT_W + ow] = pool_sum / (float)(POOL_K * POOL_K);
}

torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, torch::Tensor bias,
                      int kernel_size, int pool_size, float sub1, float sub2) {
    int B = input.size(0);
    int IC = input.size(1);
    int IH = input.size(2);
    int IW = input.size(3);
    int OC = weight.size(0);
    int OH_conv = IH - kernel_size + 1;
    int OW_conv = IW - kernel_size + 1;
    int OH = OH_conv / pool_size;
    int OW = OW_conv / pool_size;

    auto output = torch::empty({B, OC, OH, OW}, input.options());

    dim3 block(16, 16);
    dim3 grid((OW + 15) / 16, (OH + 15) / 16, B * OC);

    fused_conv_pool_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        B, IC, IH, IW, OC, OH, OW, kernel_size, pool_size, sub1, sub2);

    return output;
}
'''

cpp_source = "torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int kernel_size, int pool_size, float sub1, float sub2);"

ext = load_inline(
    name='fused_ext_v3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        self.kernel_size_pool = kernel_size_pool
        self.kernel_size = kernel_size

    def forward(self, x):
        return ext.fused_op(x, self.conv.weight, self.conv.bias, self.kernel_size, self.kernel_size_pool, self.subtract1_value, self.subtract2_value)

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
