import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <algorithm>

// Generic max using standard comparison
template<typename T>
__device__ __forceinline__ T generic_max(T a, T b) {
    return (a > b) ? a : b;
}

// Specializations using CUDA intrinsics
template<>
__device__ __forceinline__ half generic_max(half a, half b) {
    return __hmax(a, b);
}

template<>
__device__ __forceinline__ nv_bfloat16 generic_max(nv_bfloat16 a, nv_bfloat16 b) {
    return __hmax(a, b);
}

// Create negative infinity for different types
template<typename T>
__device__ __forceinline__ T neg_inf();

template<>
__device__ __forceinline__ float neg_inf<float>() {
    return -INFINITY;
}

template<>
__device__ __forceinline__ half neg_inf<half>() {
    return __float2half(-INFINITY);
}

template<>
__device__ __forceinline__ nv_bfloat16 neg_inf<nv_bfloat16>() {
    return __float2bfloat16(-INFINITY);
}

// Version optimized for the specific case: kernel=2, stride=2, padding=1, dilation=3
// Output is 256x256, input is 512x512
template <typename T>
__global__ void max_pool2d_specialized(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_height,
    const int in_width
) {
    const int batch_idx = blockIdx.z;
    const int channel_idx = blockIdx.y;
    const int out_row = blockIdx.x;
    
    if (batch_idx >= batch_size || channel_idx >= channels || out_row >= 256) return;
    
    const int tid = threadIdx.x;
    const int num_threads = blockDim.x;
    
    const T* input_ptr = input + 
        batch_idx * channels * in_height * in_width + 
        channel_idx * in_height * in_width;
    
    T* output_ptr = output + 
        batch_idx * channels * 256 * 256 + 
        channel_idx * 256 * 256;
    
    const int in_row_start = out_row * 2 - 1;
    
    for (int out_col = tid; out_col < 256; out_col += num_threads) {
        const int in_col_start = out_col * 2 - 1;
        
        T max_val = neg_inf<T>();
        
        int in_row, in_col;
        
        // Kernel positions with dilation=3: (0,0), (0,3), (3,0), (3,3)
        in_row = in_row_start;
        in_col = in_col_start;
        if (in_row >= 0 && in_row < in_height && in_col >= 0 && in_col < in_width) {
            max_val = generic_max(max_val, input_ptr[in_row * in_width + in_col]);
        }
        
        in_col = in_col_start + 3;
        if (in_row >= 0 && in_row < in_height && in_col >= 0 && in_col < in_width) {
            max_val = generic_max(max_val, input_ptr[in_row * in_width + in_col]);
        }
        
        in_row = in_row_start + 3;
        in_col = in_col_start;
        if (in_row >= 0 && in_row < in_height && in_col >= 0 && in_col < in_width) {
            max_val = generic_max(max_val, input_ptr[in_row * in_width + in_col]);
        }
        
        in_col = in_col_start + 3;
        if (in_row >= 0 && in_row < in_height && in_col >= 0 && in_col < in_width) {
            max_val = generic_max(max_val, input_ptr[in_row * in_width + in_col]);
        }
        
        output_ptr[out_row * 256 + out_col] = max_val;
    }
}

// General version for arbitrary kernel sizes
template <typename T>
__global__ void max_pool2d_general(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_height,
    const int in_width,
    const int out_height,
    const int out_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation
) {
    const int batch_idx = blockIdx.z;
    const int channel_idx = blockIdx.y;
    const int out_row = blockIdx.x;
    
    if (batch_idx >= batch_size || channel_idx >= channels || out_row >= out_height) return;
    
    const int tid = threadIdx.x;
    const int num_threads = blockDim.x;
    
    const T* input_ptr = input + 
        batch_idx * channels * in_height * in_width + 
        channel_idx * in_height * in_width;
    
    T* output_ptr = output + 
        batch_idx * channels * out_height * out_width + 
        channel_idx * out_height * out_width;
    
    const int in_row_start = out_row * stride - padding;
    
    for (int out_col = tid; out_col < out_width; out_col += num_threads) {
        const int in_col_start = out_col * stride - padding;
        
        T max_val = neg_inf<T>();
        
        for (int ki = 0; ki < kernel_size; ki++) {
            const int in_row = in_row_start + ki * dilation;
            if (in_row >= 0 && in_row < in_height) {
                for (int kj = 0; kj < kernel_size; kj++) {
                    const int in_col = in_col_start + kj * dilation;
                    if (in_col >= 0 && in_col < in_width) {
                        max_val = generic_max(max_val, input_ptr[in_row * in_width + in_col]);
                    }
                }
            }
        }
        
        output_ptr[out_row * out_width + out_col] = max_val;
    }
}

torch::Tensor max_pool2d_forward(
    torch::Tensor input,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_height = input.size(2);
    const int in_width = input.size(3);
    
    // Calculate output size
    const int out_height = (in_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    const int out_width = (in_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    const int threads_per_block = std::min(256, out_width);
    
    // Launch kernel based on dtype
    if (input.dtype() == torch::kFloat32) {
        if (kernel_size == 2 && stride == 2 && padding == 1 && dilation == 3) {
            max_pool2d_specialized<float><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<float>(),
                output.data_ptr<float>(),
                batch_size, channels, in_height, in_width
            );
        } else {
            max_pool2d_general<float><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<float>(),
                output.data_ptr<float>(),
                batch_size, channels, in_height, in_width, out_height, out_width,
                kernel_size, stride, padding, dilation
            );
        }
    } else if (input.dtype() == torch::kFloat16) {
        if (kernel_size == 2 && stride == 2 && padding == 1 && dilation == 3) {
            max_pool2d_specialized<half><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<half>(),
                output.data_ptr<half>(),
                batch_size, channels, in_height, in_width
            );
        } else {
            max_pool2d_general<half><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<half>(),
                output.data_ptr<half>(),
                batch_size, channels, in_height, in_width, out_height, out_width,
                kernel_size, stride, padding, dilation
            );
        }
    } else if (input.dtype() == torch::kBFloat16) {
        if (kernel_size == 2 && stride == 2 && padding == 1 && dilation == 3) {
            max_pool2d_specialized<nv_bfloat16><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<nv_bfloat16>(),
                output.data_ptr<nv_bfloat16>(),
                batch_size, channels, in_height, in_width
            );
        } else {
            max_pool2d_general<nv_bfloat16><<<dim3(batch_size, channels, out_height), threads_per_block>>>(
                input.data_ptr<nv_bfloat16>(),
                output.data_ptr<nv_bfloat16>(),
                batch_size, channels, in_height, in_width, out_height, out_width,
                kernel_size, stride, padding, dilation
            );
        }
    }
    
    cudaDeviceSynchronize();
    return output;
}
'''

ext = load_inline(
    name='max_pool2d_ext',
    cpp_sources='torch::Tensor max_pool2d_forward(torch::Tensor input, const int kernel_size, const int stride, const int padding, const int dilation);',
    cuda_sources=cuda_source,
    functions=['max_pool2d_forward'],
    verbose=False,
    extra_cuda_cflags=['-std=c++17', '-res-usage', '--maxrregcount=128', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.max_pool2d_forward(x, self.kernel_size, self.stride, self.padding, self.dilation)

# Constants from reference
batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

def get_inputs():
    return [torch.randn(batch_size, channels, height, width, device='cuda')]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
