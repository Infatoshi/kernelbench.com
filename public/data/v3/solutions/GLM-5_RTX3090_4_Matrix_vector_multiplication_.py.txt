import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized FP16 GEMV kernel - one warp per row
__global__ void gemv_fp16_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    half* __restrict__ C,
    const int M,
    const int K
) {
    extern __shared__ char smem[];
    half* B_shared = reinterpret_cast<half*>(smem);
    
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    
    float sum = 0.0f;
    
    // Process K in chunks of 32 (one warp width)
    const int K_aligned = K & ~31;
    
    // Main loop - process 32 elements at a time
    for (int k = 0; k < K_aligned; k += 32) {
        // Load B into shared memory
        B_shared[lane] = B[k + lane];
        __syncwarp();
        
        // Load A and compute
        half a_val = A[row * K + k + lane];
        half b_val = B_shared[lane];
        sum += __half2float(a_val) * __half2float(b_val);
    }
    
    // Handle remaining elements
    if (K_aligned < K) {
        if (lane < (K - K_aligned)) {
            B_shared[lane] = B[K_aligned + lane];
        }
        __syncwarp();
        
        if (lane < (K - K_aligned)) {
            half a_val = A[row * K + K_aligned + lane];
            half b_val = B_shared[lane];
            sum += __half2float(a_val) * __half2float(b_val);
        }
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane == 0) {
        C[row] = __float2half(sum);
    }
}

// Optimized FP32 GEMV kernel - one warp per row
__global__ void gemv_fp32_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    const int M,
    const int K
) {
    extern __shared__ char smem[];
    float* B_shared = reinterpret_cast<float*>(smem);
    
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    
    float sum = 0.0f;
    
    const int K_aligned = K & ~31;
    
    for (int k = 0; k < K_aligned; k += 32) {
        B_shared[lane] = B[k + lane];
        __syncwarp();
        
        float a_val = A[row * K + k + lane];
        float b_val = B_shared[lane];
        sum += a_val * b_val;
    }
    
    if (K_aligned < K) {
        if (lane < (K - K_aligned)) {
            B_shared[lane] = B[K_aligned + lane];
        }
        __syncwarp();
        
        if (lane < (K - K_aligned)) {
            float a_val = A[row * K + K_aligned + lane];
            float b_val = B_shared[lane];
            sum += a_val * b_val;
        }
    }
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane == 0) {
        C[row] = sum;
    }
}

torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B) {
    // A: (M, K), B: (K, 1) or (K,), C: (M, 1)
    const int M = A.size(0);
    const int K = A.size(1);
    
    // Squeeze B if needed
    auto B_flat = B.sizes().size() > 1 && B.size(1) == 1 ? B.squeeze(1) : B;
    
    // Output shape
    auto C = torch::empty({M, 1}, A.options());
    
    const int threads = 32;  // One warp per row
    const int blocks = M;
    const int shared_mem = 32 * A.element_size();  // 32 elements for B tile
    
    if (A.dtype() == torch::kFloat32) {
        gemv_fp32_kernel<<<blocks, threads, shared_mem>>>(
            A.data_ptr<float>(),
            B_flat.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    } else if (A.dtype() == torch::kFloat16) {
        gemv_fp16_kernel<<<blocks, threads, shared_mem>>>(
            reinterpret_cast<const half*>(A.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(B_flat.data_ptr<at::Half>()),
            reinterpret_cast<half*>(C.data_ptr<at::Half>()),
            M, K
        );
    } else {
        // Fallback to generic implementation for other dtypes
        AT_ERROR("Unsupported dtype for GEMV kernel");
    }
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemv_op(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_op'],
    verbose=True,
    extra_cuda_cflags=['-arch=sm_86', '-O3'],
)

class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.gemv_op(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
