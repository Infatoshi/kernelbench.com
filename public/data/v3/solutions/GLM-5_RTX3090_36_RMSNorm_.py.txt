import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Block reduction using shared memory
__device__ __forceinline__ float block_reduce(float val, float* shared_mem, int tid, int block_size) {
    shared_mem[tid] = val;
    __syncthreads();
    
    for (int s = block_size / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_mem[tid] += shared_mem[tid + s];
        }
        __syncthreads();
    }
    return shared_mem[0];
}

// Optimized RMSNorm kernel for NCHW layout
template<typename T, int BLOCK_SIZE>
__global__ void rmsnorm_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    int batch_size,
    int num_features,
    int dim1,
    int dim2,
    float eps
) {
    int total_spatial = batch_size * dim1 * dim2;
    
    // Grid-stride loop over spatial positions
    for (int spatial_idx = blockIdx.x; spatial_idx < total_spatial; spatial_idx += gridDim.x) {
        int hw_size = dim1 * dim2;
        int b = spatial_idx / hw_size;
        int hw_rem = spatial_idx % hw_size;
        int h = hw_rem / dim2;
        int w = hw_rem % dim2;
        
        int64_t feature_stride = static_cast<int64_t>(dim1) * dim2;
        int64_t batch_stride = static_cast<int64_t>(num_features) * feature_stride;
        int64_t base_idx = static_cast<int64_t>(b) * batch_stride + h * dim2 + w;
        
        int tid = threadIdx.x;
        
        // Compute partial sum of squares
        float sum_sq = 0.0f;
        for (int f = tid; f < num_features; f += BLOCK_SIZE) {
            int64_t idx = base_idx + f * feature_stride;
            float val = static_cast<float>(x[idx]);
            sum_sq += val * val;
        }
        
        // Block reduction
        __shared__ float s_sum[BLOCK_SIZE];
        sum_sq = block_reduce(sum_sq, s_sum, tid, BLOCK_SIZE);
        
        // Compute inv_rms (only thread 0)
        __shared__ float inv_rms_shared;
        if (tid == 0) {
            float rms = sqrtf(sum_sq / num_features + eps);
            inv_rms_shared = 1.0f / rms;
        }
        __syncthreads();
        
        // Normalize
        float inv_rms = inv_rms_shared;
        for (int f = tid; f < num_features; f += BLOCK_SIZE) {
            int64_t idx = base_idx + f * feature_stride;
            float val = static_cast<float>(x[idx]);
            y[idx] = static_cast<T>(val * inv_rms);
        }
    }
}

torch::Tensor rmsnorm_cuda(torch::Tensor x, int64_t num_features_arg, double eps) {
    auto y = torch::empty_like(x);
    
    int batch_size = x.size(0);
    int num_features = x.size(1);
    int dim1 = x.size(2);
    int dim2 = x.size(3);
    
    int total_spatial = batch_size * dim1 * dim2;
    
    // Choose block size based on features
    const int BLOCK_SIZE = 64;
    int num_blocks = min(total_spatial, 2048);  // Limit for better occupancy
    
    if (x.scalar_type() == torch::kFloat16) {
        rmsnorm_kernel<at::Half, BLOCK_SIZE><<<num_blocks, BLOCK_SIZE>>>(
            reinterpret_cast<const at::Half*>(x.data_ptr<at::Half>()),
            reinterpret_cast<at::Half*>(y.data_ptr<at::Half>()),
            batch_size, num_features, dim1, dim2, static_cast<float>(eps)
        );
    } else if (x.scalar_type() == torch::kBFloat16) {
        rmsnorm_kernel<at::BFloat16, BLOCK_SIZE><<<num_blocks, BLOCK_SIZE>>>(
            reinterpret_cast<const at::BFloat16*>(x.data_ptr<at::BFloat16>()),
            reinterpret_cast<at::BFloat16*>(y.data_ptr<at::BFloat16>()),
            batch_size, num_features, dim1, dim2, static_cast<float>(eps)
        );
    } else {
        rmsnorm_kernel<float, BLOCK_SIZE><<<num_blocks, BLOCK_SIZE>>>(
            x.data_ptr<float>(), y.data_ptr<float>(),
            batch_size, num_features, dim1, dim2, static_cast<float>(eps)
        );
    }
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor rmsnorm_cuda(torch::Tensor x, int64_t num_features, double eps);
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rmsnorm_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)

class Model(nn.Module):
    """
    Simple model that performs RMS Normalization.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm_cuda(x, self.num_features, self.eps)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [features]
