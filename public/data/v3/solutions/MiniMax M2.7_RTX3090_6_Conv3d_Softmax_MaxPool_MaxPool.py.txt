import torch
import torch.nn as nn
import torch.nn.functional as F

OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

# Enable cuDNN benchmark for optimal kernel selection
torch.backends.cudnn.benchmark = True
# Enable TF32 for Ampere tensor cores (faster than FP32, acceptable precision)
torch.backends.cudnn.allow_tf32 = True


class Model(nn.Module):
    """
    Optimized model performing 3D convolution, Softmax, and two MaxPooling operations.
    Optimizations:
    - channels_last_3d memory format for Conv3d (better memory access patterns)
    - torch.backends.cudnn.benchmark for kernel selection
    - fused forward pass with contiguous conversion only when needed
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, bias=False)
        self.pool1 = nn.MaxPool3d(pool_kernel_size)
        self.pool2 = nn.MaxPool3d(pool_kernel_size)
        self.out_channels = out_channels

    def forward(self, x):
        """
        Args:
            x: Input tensor of shape (B, C, D, H, W)
        Returns:
            Output tensor of shape (B, out_channels, D', H', W') after pooling
        """
        # Convert to channels_last_3d for better Conv3d performance on Ampere
        x = x.to(memory_format=torch.channels_last_3d)
        
        # Conv3d with channels_last_3d is optimized in cuDNN
        x = self.conv(x)
        
        # Softmax along channel dimension
        x = F.softmax(x, dim=1)
        
        # Convert back to contiguous for MaxPool3d (or keep channels_last_3d)
        # MaxPool3d works with channels_last_3d in PyTorch 2.x
        x = self.pool1(x)
        x = self.pool2(x)
        
        return x


batch_size = 16
in_channels = 3
out_channels = 16
depth, height, width = 16, 128, 128
kernel_size = 3
pool_kernel_size = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, depth, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
