import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cstdint>

#define BLOCK_M 32
#define BLOCK_N 32
#define BLOCK_K 128

__global__ void int4_gemm_kernel(
    const half* __restrict__ X,
    const uint8_t* __restrict__ W_packed,
    const half* __restrict__ scales,
    half* __restrict__ Y,
    int M, int N, int K, int group_size, int num_groups
) {
    __shared__ half X_s[BLOCK_M][BLOCK_K + 4];
    __shared__ half W_s[BLOCK_N][BLOCK_K + 4];
    __shared__ half scales_s[BLOCK_N];
    
    int m_block = blockIdx.x * BLOCK_M;
    int n_block = blockIdx.y * BLOCK_N;
    
    int tid = threadIdx.x;
    int threads = blockDim.x;
    
    half sum = __float2half(0.0f);
    
    for (int k_base = 0; k_base < K; k_base += BLOCK_K) {
        int group_idx = k_base / group_size;
        
        if (tid < BLOCK_N) {
            int n = n_block + tid;
            scales_s[tid] = (n < N) ? scales[n * num_groups + group_idx] : __float2half(0.0f);
        }
        __syncthreads();
        
        int total_w = BLOCK_N * BLOCK_K;
        for (int idx = tid; idx < total_w; idx += threads) {
            int n_l = idx / BLOCK_K;
            int k_l = idx % BLOCK_K;
            int n_g = n_block + n_l;
            int k_g = k_base + k_l;
            
            if (n_g < N && k_g < K) {
                int byte_offset = n_g * (K / 2) + (k_g / 2);
                uint8_t packed = W_packed[byte_offset];
                int w_int = (k_g % 2 == 0) ? (packed & 0x0F) : (packed >> 4);
                half scale = scales_s[n_l];
                W_s[n_l][k_l] = __hmul(scale, __int2half_rn(w_int - 8));
            } else {
                W_s[n_l][k_l] = __float2half(0.0f);
            }
        }
        
        int total_x = BLOCK_M * BLOCK_K;
        for (int idx = tid; idx < total_x; idx += threads) {
            int m_l = idx / BLOCK_K;
            int k_l = idx % BLOCK_K;
            int m_g = m_block + m_l;
            int k_g = k_base + k_l;
            
            if (m_g < M && k_g < K) {
                X_s[m_l][k_l] = X[m_g * K + k_g];
            } else {
                X_s[m_l][k_l] = __float2half(0.0f);
            }
        }
        __syncthreads();
        
        int m_local = tid / BLOCK_N;
        int n_local = tid % BLOCK_N;
        
        if (m_local < BLOCK_M && n_local < BLOCK_N) {
            half local_sum = __float2half(0.0f);
            #pragma unroll
            for (int k = 0; k < BLOCK_K; k++) {
                local_sum = __hfma(X_s[m_local][k], W_s[n_local][k], local_sum);
            }
            sum = __hadd(sum, local_sum);
        }
        __syncthreads();
    }
    
    int m_local = tid / BLOCK_N;
    int n_local = tid % BLOCK_N;
    int m_g = m_block + m_local;
    int n_g = n_block + n_local;
    
    if (m_local < BLOCK_M && n_local < BLOCK_N && m_g < M && n_g < N) {
        Y[m_g * N + n_g] = sum;
    }
}

torch::Tensor int4_gemm_forward(
    torch::Tensor X,
    torch::Tensor W_packed,
    torch::Tensor scales,
    int64_t group_size
) {
    int M = X.size(0) * X.size(1);
    int K = X.size(2);
    int N = W_packed.size(0);
    int num_groups = scales.size(1);
    
    auto X_2d = X.reshape({M, K});
    auto Y = torch::empty({M, N}, X.options());
    
    dim3 block(1024);
    dim3 grid((M + BLOCK_M - 1) / BLOCK_M, (N + BLOCK_N - 1) / BLOCK_N);
    
    int4_gemm_kernel<<<grid, block>>>(
        reinterpret_cast<half*>(X_2d.data_ptr<at::Half>()),
        W_packed.data_ptr<uint8_t>(),
        reinterpret_cast<half*>(scales.data_ptr<at::Half>()),
        reinterpret_cast<half*>(Y.data_ptr<at::Half>()),
        M, N, K, static_cast<int>(group_size), num_groups
    );
    
    return Y.view({X.size(0), X.size(1), N});
}
'''

cpp_source = r'''
torch::Tensor int4_gemm_forward(
    torch::Tensor X,
    torch::Tensor W_packed,
    torch::Tensor scales,
    int64_t group_size
);
'''

ext = load_inline(
    name='int4_gemm_fused_v4',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['int4_gemm_forward'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math', '-std=c++17'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, K=None, N=None, group_size=128):
        super().__init__()
        # Handle case where K and N are not provided (use defaults from reference)
        if K is None:
            K = 4096
        if N is None:
            N = 11008
            
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even for INT4 packing"

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x):
        return ext.int4_gemm_forward(x, self.weight_packed, self.scales, self.group_size)


def get_inputs():
    return [torch.randn(4, 2048, 4096, dtype=torch.float16, device='cuda')]


def get_init_inputs():
    return [4096, 11008, 128]
