import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cublasLt.h>
#include <cublas_v2.h>

#define BLOCK_SIZE 256
#define WARP_SIZE 32

// GELU approximation for fp16
__device__ __half gelu_approx(__half x) {
    float fx = __half2float(x);
    float t = tanhf(0.7978845608f * (fx + 0.044715f * fx * fx * fx));
    return __float2half(0.5f * fx * (1.0f + t));
}

// GELU approximation for float
__device__ float gelu_approx_float(float x) {
    float t = tanhf(0.7978845608f * (x + 0.044715f * x * x * x));
    return 0.5f * x * (1.0f + t);
}

// Fused AvgPool + GELU + Scale + Max (fp16)
__global__ void fused_ops_fp16_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    float scale_factor,
    int batch_size,
    int out_features,
    int pool_kernel_size,
    int pooled_len
) {
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    extern __shared__ __half smem[];
    __half* shared_data = smem;
    
    const __half* batch_ptr = input + batch_idx * out_features;
    
    // Step 1: Average pooling
    if (tid < pooled_len) {
        float sum = 0.0f;
        for (int i = 0; i < pool_kernel_size; i++) {
            int idx = tid * pool_kernel_size + i;
            sum += __half2float(batch_ptr[idx]);
        }
        shared_data[tid] = __float2half(sum / pool_kernel_size);
    }
    __syncthreads();
    
    // Step 2: GELU + Scale
    if (tid < pooled_len) {
        __half val = gelu_approx(shared_data[tid]);
        shared_data[tid] = __float2half(__half2float(val) * scale_factor);
    }
    __syncthreads();
    
    // Step 3: Parallel max reduction
    for (int stride = BLOCK_SIZE / 2; stride >= 1; stride >>= 1) {
        if (tid < stride && tid < pooled_len && tid + stride < pooled_len) {
            __half a = shared_data[tid];
            __half b = shared_data[tid + stride];
            shared_data[tid] = a > b ? a : b;
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[batch_idx] = shared_data[0];
    }
}

// Fused AvgPool + GELU + Scale + Max (bf16)  
__global__ void fused_ops_bf16_kernel(
    const nv_bfloat16* __restrict__ input,
    nv_bfloat16* __restrict__ output,
    float scale_factor,
    int batch_size,
    int out_features,

    float* shared_data = smem;
    
    const nv_bfloat16* batch_ptr = input + batch_idx * out_features;

    if (tid < pooled_len) {
        float sum = 0.0f;
        for (int i = 0; i < pool_kernel_size; i++) {
            int idx = tid * pool_kernel_size + i;
            sum += __bfloat162float(batch_ptr[idx]);
        }
        shared_data[tid] = sum / pool_kernel_size;
    }
    __syncthreads();
    
    // Step 2: GELU + Scale
    if (tid < pooled_len) {
        float val = gelu_approx_float(shared_data[tid]);
        shared_data[tid] = val * scale_factor;
    }
    __syncthreads();
    
    // Step 3: Parallel max reduction
    for (int stride = BLOCK_SIZE / 2; stride >= 1; stride >>= 1) {
        if (tid < stride && tid < pooled_len && tid + stride < pooled_len) {
            float a = shared_data[tid];
            float b = shared_data[tid + stride];
            shared_data[tid] = a > b ? a : b;
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[batch_idx] = __float2bfloat16(shared_data[0]);
    }
}

// Fused AvgPool + GELU + Scale + Max (fp32)
__global__ void fused_ops_fp32_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    float scale_factor,
    int batch_size,
    int out_features,
    int pool_kernel_size,

    }

    for (int stride = BLOCK_SIZE / 2; stride >= 1; stride >>= 1) {
        if (tid < stride && tid < pooled_len && tid + stride < pooled_len) {

torch::Tensor fused_ops_launcher(
    torch::Tensor input,
    float scale_factor,
    int pool_kernel_size,
    int pooled_len,
    int out_features
) {
    int batch_size = input.size(0);
    auto output = torch::empty({batch_size}, input.options());
    
    int shared_mem_size = pooled_len * sizeof(float);
    
    if (input.scalar_type() == at::ScalarType::Half) {
        fused_ops_fp16_kernel<<<batch_size, BLOCK_SIZE, shared_mem_size>>>(
            reinterpret_cast<__half*>(input.data_ptr()),
            reinterpret_cast<__half*>(output.data_ptr()),
            scale_factor,
            batch_size,
            out_features,
            pool_kernel_size,
            pooled_len
        );
    } else if (input.scalar_type() == at::ScalarType::BFloat16) {
        fused_ops_bf16_kernel<<<batch_size, BLOCK_SIZE, shared_mem_size>>>(
            reinterpret_cast<nv_bfloat16*>(input.data_ptr()),
            reinterpret_cast<nv_bfloat16*>(output.data_ptr()),
            scale_factor,
            batch_size,
            out_features,
            pool_kernel_size,
            pooled_len
        );
    } else {
        fused_ops_fp32_kernel<<<batch_size, BLOCK_SIZE, shared_mem_size>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            scale_factor,
            batch_size,
            out_features,
            pool_kernel_size,
            pooled_len
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_ops_launcher(
    torch::Tensor input,
    float scale_factor,
    int pool_kernel_size,
    int pooled_len,
    int out_features
);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_ops_launcher'],
    extra_cuda_cflags=['-arch=sm_90', '-std=c++17'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor
        self.pooled_len = out_features // pool_kernel_size
        self.linear = nn.Linear(in_features, out_features)
        self.ext = ext
    
    def forward(self, x):
        # Step 1: Linear matmul
        x = self.linear(x)
        
        # Step 2: Fused AvgPool + GELU + Scale + Max
        x = self.ext.fused_ops_launcher(
            x, self.scale_factor, self.pool_kernel_size, 
            self.pooled_len, self.out_features
        )
        
        return x

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
