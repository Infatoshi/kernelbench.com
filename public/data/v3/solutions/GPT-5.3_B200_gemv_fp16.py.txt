import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda_fp16.h>

__inline__ __device__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2)
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;
}

__inline__ __device__ float block_reduce_sum(float val) {
    static __shared__ float shared[32];
    int lane = threadIdx.x % 32;
    int wid = threadIdx.x / 32;

    val = warp_reduce_sum(val);
    if (lane == 0) shared[wid] = val;
    __syncthreads();

    val = (threadIdx.x < blockDim.x / 32) ? shared[lane] : 0.0f;
    if (wid == 0) val = warp_reduce_sum(val);
    return val;
}

__global__ void gemv_fp16_kernel(
    const half* __restrict__ x,
    const half* __restrict__ w,
    half* __restrict__ y,
    int B, int N, int K)
{
    int b = blockIdx.y;
    int n = blockIdx.x;

    const half* x_ptr = x + b * K;
    const half* w_ptr = w + n * K;

    float sum = 0.0f;

    for (int k = threadIdx.x * 2; k < K; k += blockDim.x * 2) {
        half2 xv = *reinterpret_cast<const half2*>(x_ptr + k);
        half2 wv = *reinterpret_cast<const half2*>(w_ptr + k);

        float2 xf = __half22float2(xv);
        float2 wf = __half22float2(wv);

        sum += xf.x * wf.x + xf.y * wf.y;
    }

    sum = block_reduce_sum(sum);

    if (threadIdx.x == 0) {
        y[b * N + n] = __float2half(sum);
    }
}


torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w) {
    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);

    auto y = torch::empty({B, N}, x.options());

    dim3 grid(N, B);
    dim3 block(128);

    gemv_fp16_kernel<<<grid, block>>>(
        reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
        reinterpret_cast<const half*>(w.data_ptr<at::Half>()),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        B, N, K
    );

    return y;
}
'''

ext = load_inline(
    name='gemv_fp16_ext',
    cpp_sources='torch::Tensor gemv_fp16(torch::Tensor x, torch::Tensor w);',
    cuda_sources=cuda_src,
    functions=['gemv_fp16'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16).contiguous()
        w = self.weight.contiguous()
        return ext.gemv_fp16(x_fp16, w)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
