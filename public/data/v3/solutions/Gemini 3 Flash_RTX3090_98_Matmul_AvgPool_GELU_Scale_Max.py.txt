import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_kernel(
    X_ptr, W_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n
    
    offs_m = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M))
    offs_n = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N))
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    x_ptr = X_ptr + (offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    w_ptr = W_ptr + (offs_n[None, :] * stride_wn + offs_k[:, None] * stride_wk)
    
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        mask_k = (k * BLOCK_SIZE_K + offs_k) < K
        x = tl.load(x_ptr, mask=(offs_m[:, None] < M) & (mask_k[None, :]), other=0.0)
        w = tl.load(w_ptr, mask=(offs_n[None, :] < N) & (mask_k[:, None]), other=0.0)
        accumulator += tl.dot(x, w)
        x_ptr += BLOCK_SIZE_K * stride_xk
        w_ptr += BLOCK_SIZE_K * stride_wk
    
    out_ptr = Out_ptr + (offs_m[:, None] * stride_om + offs_n[None, :] * stride_on)
    mask_m = (offs_m < M)
    mask_n = (offs_n < N)
    tl.store(out_ptr, accumulator, mask=mask_m[:, None] & mask_n[None, :])

@triton.jit
def post_process_fused(
    Mat_ptr, Bias_ptr, Out_ptr,
    M, N, pool_kernel_size, scale_factor,
    stride_mm, stride_mn,
    BLOCK_N: tl.constexpr,
):
    row_idx = tl.program_id(0)
    if row_idx >= M:
        return
    
    num_pools = N // pool_kernel_size
    max_val = -1e20
    
    for pool_start in range(0, num_pools, BLOCK_N):
        pool_offs = pool_start + tl.arange(0, BLOCK_N)
        mask = pool_offs < num_pools
        
        pool_sum = tl.zeros((BLOCK_N,), dtype=tl.float32)
        for k in range(pool_kernel_size):
            col_idx = pool_offs * pool_kernel_size + k
            val = tl.load(Mat_ptr + row_idx * stride_mm + col_idx * stride_mn, mask=mask, other=0.0)
            if Bias_ptr is not None:
                bias = tl.load(Bias_ptr + col_idx, mask=mask, other=0.0)
                val += bias
            pool_sum += val
        
        avg_val = pool_sum / pool_kernel_size
        gelu_val = 0.5 * avg_val * (1.0 + tl.extra.cuda.libdevice.erf(avg_val * 0.7071067811865476))
        res = gelu_val * scale_factor
        tile_max = tl.max(tl.where(mask, res, -1e20), axis=0)
        if tile_max > max_val:
            max_val = tile_max
            
    tl.store(Out_ptr + row_idx, max_val)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = float(scale_factor)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        
        # Match weight/bias dtype to input x
        self.weight.data = self.weight.data.to(x.dtype)
        self.bias.data = self.bias.data.to(x.dtype)
        
        # Always use float32 for matmul internal accumulator and intermediate
        matmul_out = torch.empty((M, N), device=x.device, dtype=torch.float32)
        grid_matmul = lambda META: (triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),)
        matmul_kernel[grid_matmul](
            x, self.weight, matmul_out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            matmul_out.stride(0), matmul_out.stride(1),
            BLOCK_SIZE_M=64, BLOCK_SIZE_N=64, BLOCK_SIZE_K=32,
            num_warps=8, num_stages=3
        )
        
        out = torch.empty((M,), device=x.device, dtype=x.dtype)
        grid_post = (M,)
        post_process_fused[grid_post](
            matmul_out, self.bias, out,
            M, N, self.pool_kernel_size, self.scale_factor,
            matmul_out.stride(0), matmul_out.stride(1),
            BLOCK_N=1024,
            num_warps=8,
        )
        
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
