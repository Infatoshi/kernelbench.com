import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    # Swizzling for better L2 cache hit rate
    width = GROUP_SIZE_M * num_pid_n
    group_id = pid // width
    group_size = tl.minimum(num_pid_m - group_id * GROUP_SIZE_M, GROUP_SIZE_M)
    pid_m = group_id * GROUP_SIZE_M + (pid % group_size)
    pid_n = (pid % width) // group_size

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M))
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N))
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptr = X_ptr + (offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    b_ptr = W_ptr + (offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        k_remaining = K - k * BLOCK_SIZE_K
        a = tl.load(a_ptr, mask=offs_k[None, :] < k_remaining, other=0.0)
        b = tl.load(b_ptr, mask=offs_k[:, None] < k_remaining, other=0.0)
        accumulator = tl.dot(a, b, accumulator)
        a_ptr += BLOCK_SIZE_K * stride_xk
        b_ptr += BLOCK_SIZE_K * stride_wk

    if B_ptr is not None:
        bias = tl.load(B_ptr + offs_bn)[None, :]
        accumulator += bias

    # GELU
    accumulator = 0.5 * accumulator * (1.0 + tl.extra.cuda.libdevice.erf(accumulator * 0.7071067812))
    
    out_ptr = Out_ptr + (offs_am[:, None] * stride_om + offs_bn[None, :] * stride_on)
    mask = (offs_am[:, None] < M) & (offs_bn[None, :] < N)
    tl.store(out_ptr, accumulator.to(Out_ptr.dtype.element_ty), mask=mask)

@triton.jit
def softmax_kernel(
    Out_ptr,
    M, N,
    stride_om, stride_on,
    BLOCK_SIZE_N: tl.constexpr,
):
    pid_m = tl.program_id(0)
    offs_n = tl.arange(0, BLOCK_SIZE_N)
    row_ptr = Out_ptr + pid_m * stride_om + offs_n * stride_on
    
    row = tl.load(row_ptr, mask=offs_n < N, other=-float('inf')).to(tl.float32)
    row_max = tl.max(row, axis=0)
    # Using float32 for stable softmax
    row = tl.exp(row - row_max)
    row_sum = tl.sum(row, axis=0)
    out = row / row_sum
    tl.store(row_ptr, out.to(Out_ptr.dtype.element_ty), mask=offs_n < N)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        # weights should be in (out_features, in_features) according to reference Linear
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        out = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        # Grid settings
        BLOCK_SIZE_M = 128
        BLOCK_SIZE_N = 128
        BLOCK_SIZE_K = 64
        GROUP_SIZE_M = 8
        
        # We need to compute X @ W.T
        # In fused_kernel: accumulator = X (M, K) @ W.T (K, N)
        # Weight in torch.nn.Linear is stored as (N, K). So weight.T is (K, N).
        # Our fused_kernel expects W with stride_wn and stride_wk.
        # So it loads [offs_bn, offs_k]. If W is (N, K), this accesses the correct elements.
        
        grid = (triton.cdiv(M, BLOCK_SIZE_M) * triton.cdiv(N, BLOCK_SIZE_N),)
        
        fused_kernel[grid](
            x, self.weight, self.bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_SIZE_M, BLOCK_SIZE_N, BLOCK_SIZE_K,
            GROUP_SIZE_M,
            num_warps=8,
            num_stages=3
        )
        
        softmax_kernel[(M,)](
            out,
            M, N,
            out.stride(0), out.stride(1),
            BLOCK_SIZE_N=triton.next_power_of_2(N),
            num_warps=16 # Higher occupancy for small rows
        )
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096]
