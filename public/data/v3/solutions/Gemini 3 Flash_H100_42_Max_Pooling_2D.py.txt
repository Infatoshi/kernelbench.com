
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <algorithm>

template <typename T>
__device__ __forceinline__ T get_negative_infinity();

template <>
__device__ __forceinline__ float get_negative_infinity<float>() {
    return -1e38f;
}

template <>
__device__ __forceinline__ double get_negative_infinity<double>() {
    return -1e308;
}

template <>
__device__ __forceinline__ at::Half get_negative_infinity<at::Half>() {
    unsigned short bits = 0xFBFF; // -65504
    at::Half res;
    std::memcpy(&res, &bits, sizeof(bits));
    return res;
}

template <>
__device__ __forceinline__ at::BFloat16 get_negative_infinity<at::BFloat16>() {
    unsigned short bits = 0xFF7F;
    at::BFloat16 res;
    std::memcpy(&res, &bits, sizeof(bits));
    return res;
}


template <typename T>
__global__ void max_pool2d_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size, int channels, int in_h, int in_w,
    int out_h, int out_w,
    int k_h, int k_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w) 
{
    int index = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * channels * out_h * out_w;
    if (index >= total_elements) return;

    int pw = index % out_w;
    int ph = (index / out_w) % out_h;
    int c = (index / (out_w * out_h)) % channels;
    int b = index / (out_w * out_h * channels);

    int h_start = ph * stride_h - pad_h;
    int w_start = pw * stride_w - pad_w;
    
    T max_val = get_negative_infinity<T>();

    const T* input_ptr = input + (b * channels + c) * (in_h * in_w);

    for (int i = 0; i < k_h; ++i) {
        int h = h_start + i * dil_h;
        if (h >= 0 && h < in_h) {
            for (int j = 0; j < k_w; ++j) {
                int w = w_start + j * dil_w;
                if (w >= 0 && w < in_w) {
                    T val = input_ptr[h * in_w + w];
                    if (val > max_val) {
                        max_val = val;
                    }
                }
            }
        }
    }
    output[index] = max_val;
}

torch::Tensor max_pool2d_forward(
    torch::Tensor x,
    int k_h, int k_w,
    int stride_h, int stride_w,
    int pad_h, int pad_w,
    int dil_h, int dil_w) 
{
    const int batch_size = x.size(0);
    const int channels = x.size(1);
    const int in_h = x.size(2);
    const int in_w = x.size(3);

    const int out_h = (in_h + 2 * pad_h - dil_h * (k_h - 1) - 1) / stride_h + 1;
    const int out_w = (in_w + 2 * pad_w - dil_w * (k_w - 1) - 1) / stride_w + 1;

    auto y = torch::empty({batch_size, channels, out_h, out_w}, x.options());
    int total_elements = batch_size * channels * out_h * out_w;
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "max_pool2d_forward", ([&] {
        max_pool2d_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            batch_size, channels, in_h, in_w,
            out_h, out_w,
            k_h, k_w,
            stride_h, stride_w,
            pad_h, pad_w,
            dil_h, dil_w);
    }));

    return y;
}
'''

ext = load_inline(
    name='max_pool2d_ext',
    cpp_sources='torch::Tensor max_pool2d_forward(torch::Tensor x, int k_h, int k_w, int stride_h, int stride_w, int pad_h, int pad_w, int dil_h, int dil_w);',
    cuda_sources=cuda_source,
    functions=['max_pool2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size if isinstance(kernel_size, (list, tuple)) else (kernel_size, kernel_size)
        self.stride = stride if isinstance(stride, (list, tuple)) else (stride, stride)
        self.padding = padding if isinstance(padding, (list, tuple)) else (padding, padding)
        self.dilation = dilation if isinstance(dilation, (list, tuple)) else (dilation, dilation)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.max_pool2d_forward(
            x, 
            self.kernel_size[0], self.kernel_size[1],
            self.stride[0], self.stride[1],
            self.padding[0], self.padding[1],
            self.dilation[0], self.dilation[1]
        )

def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width, device='cuda')
    return [x]

def get_init_inputs():
    kernel_size = 2
    stride = 2
    padding = 1
    dilation = 3
    return [kernel_size, stride, padding, dilation]
