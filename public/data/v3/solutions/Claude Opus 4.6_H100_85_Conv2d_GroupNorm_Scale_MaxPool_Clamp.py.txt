import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl

# ============================================================================
# Fused GroupNorm + Scale + MaxPool2d + Clamp kernel
# ============================================================================

@triton.jit
def group_norm_stats_kernel(
    X_ptr,       # input: (N, C, H, W) 
    Mean_ptr,    # output: (N, num_groups)
    Var_ptr,     # output: (N, num_groups)
    N: tl.constexpr, C: tl.constexpr, H: tl.constexpr, W: tl.constexpr,
    num_groups: tl.constexpr, channels_per_group: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // num_groups
    group_idx = pid % num_groups
    
    group_size = channels_per_group * H * W
    base = batch_idx * C * H * W + group_idx * channels_per_group * H * W
    
    sum_val = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    sum_sq = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for start in range(0, group_size, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < group_size
        x = tl.load(X_ptr + base + offs, mask=mask, other=0.0).to(tl.float32)
        sum_val += tl.where(mask, x, 0.0)
        sum_sq += tl.where(mask, x * x, 0.0)
    
    total_sum = tl.sum(sum_val, axis=0)
    total_sq = tl.sum(sum_sq, axis=0)
    
    mean = total_sum / group_size
    var = total_sq / group_size - mean * mean
    
    tl.store(Mean_ptr + pid, mean)
    tl.store(Var_ptr + pid, var)


@triton.jit  
def fused_norm_scale_maxpool_clamp_kernel(
    X_ptr,        # input after conv: (N, C, H, W)
    Mean_ptr,     # (N, num_groups)
    Var_ptr,      # (N, num_groups)
    Gamma_ptr,    # GroupNorm weight: (C,)
    Beta_ptr,     # GroupNorm bias: (C,)
    Scale_ptr,    # Scale parameter: (C,)
    Out_ptr,      # output: (N, C, H_out, W_out)
    N: tl.constexpr, C: tl.constexpr, H: tl.constexpr, W: tl.constexpr,
    H_out: tl.constexpr, W_out: tl.constexpr,
    num_groups: tl.constexpr, channels_per_group: tl.constexpr,
    pool_size: tl.constexpr,
    clamp_min: tl.constexpr, clamp_max: tl.constexpr,
    eps: tl.constexpr,
    BLOCK_W: tl.constexpr,
):
    pid = tl.program_id(0)
    out_row = pid % H_out
    tmp = pid // H_out
    channel = tmp % C
    batch = tmp // C
    
    group_idx = channel // channels_per_group
    
    stat_idx = batch * num_groups + group_idx
    mean = tl.load(Mean_ptr + stat_idx)
    var = tl.load(Var_ptr + stat_idx)
    inv_std = 1.0 / tl.sqrt(var + eps)
    
    gamma = tl.load(Gamma_ptr + channel).to(tl.float32)
    beta = tl.load(Beta_ptr + channel).to(tl.float32)
    scale = tl.load(Scale_ptr + channel).to(tl.float32)
    
    # Combine gamma*inv_std*scale and beta*scale
    combined_scale = gamma * inv_std * scale
    combined_bias = beta * scale
    
    in_row0 = out_row * pool_size
    in_row1 = in_row0 + 1
    
    base_in = batch * C * H * W + channel * H * W
    base_out = batch * C * H_out * W_out + channel * H_out * W_out + out_row * W_out
    
    for col_start in range(0, W_out, BLOCK_W):
        col_offs = col_start + tl.arange(0, BLOCK_W)
        mask = col_offs < W_out
        
        in_col0 = col_offs * pool_size
        in_col1 = in_col0 + 1
        
        idx00 = base_in + in_row0 * W + in_col0
        idx01 = base_in + in_row0 * W + in_col1
        idx10 = base_in + in_row1 * W + in_col0
        idx11 = base_in + in_row1 * W + in_col1
        
        x00 = tl.load(X_ptr + idx00, mask=mask, other=0.0).to(tl.float32)
        x01 = tl.load(X_ptr + idx01, mask=mask, other=0.0).to(tl.float32)
        x10 = tl.load(X_ptr + idx10, mask=mask, other=0.0).to(tl.float32)
        x11 = tl.load(X_ptr + idx11, mask=mask, other=0.0).to(tl.float32)
        
        # Apply combined normalization to each
        x00 = (x00 - mean) * combined_scale + combined_bias
        x01 = (x01 - mean) * combined_scale + combined_bias
        x10 = (x10 - mean) * combined_scale + combined_bias
        x11 = (x11 - mean) * combined_scale + combined_bias
        
        # MaxPool
        mx = tl.maximum(tl.maximum(x00, x01), tl.maximum(x10, x11))
        
        # Clamp
        mx = tl.minimum(tl.maximum(mx, clamp_min), clamp_max)
        
        tl.store(Out_ptr + base_out + col_offs, mx.to(tl.float32), mask=mask)


def fused_gn_scale_maxpool_clamp(x, gn_weight, gn_bias, scale, num_groups, pool_size, clamp_min, clamp_max, eps=1e-5):
    N, C, H, W = x.shape
    channels_per_group = C // num_groups
    H_out = H // pool_size
    W_out = W // pool_size
    
    out = torch.empty((N, C, H_out, W_out), device=x.device, dtype=x.dtype)
    
    mean = torch.empty((N, num_groups), device=x.device, dtype=torch.float32)
    var = torch.empty((N, num_groups), device=x.device, dtype=torch.float32)
    
    group_size = channels_per_group * H * W
    BLOCK_SIZE = triton.next_power_of_2(min(group_size, 8192))
    
    grid_stats = (N * num_groups,)
    group_norm_stats_kernel[grid_stats](
        x, mean, var,
        N, C, H, W,
        num_groups, channels_per_group,
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    BLOCK_W = triton.next_power_of_2(min(W_out, 128))
    grid_main = (N * C * H_out,)
    fused_norm_scale_maxpool_clamp_kernel[grid_main](
        x, mean, var, gn_weight, gn_bias, scale, out,
        N, C, H, W, H_out, W_out,
        num_groups, channels_per_group,
        pool_size,
        clamp_min, clamp_max,
        eps,
        BLOCK_W=BLOCK_W,
    )
    
    return out


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.num_groups = num_groups
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        # Use conv2d functional directly (still cuDNN) 
        x = F.conv2d(x, self.conv.weight, self.conv.bias)
        # Fused GroupNorm + Scale + MaxPool + Clamp
        x = fused_gn_scale_maxpool_clamp(
            x, 
            self.group_norm.weight, 
            self.group_norm.bias,
            self.scale.view(-1),
            self.num_groups,
            self.maxpool_kernel_size,
            self.clamp_min,
            self.clamp_max,
        )
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
