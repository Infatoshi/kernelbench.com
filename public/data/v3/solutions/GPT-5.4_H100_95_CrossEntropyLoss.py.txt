import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "reduction"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def cross_entropy_forward_kernel(
    logits_ptr, targets_ptr, out_ptr,
    stride_row, stride_col,
    N, C,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < C
    row_ptr = logits_ptr + row * stride_row + offs * stride_col
    logits = tl.load(row_ptr, mask=mask, other=-float('inf')).to(tl.float32)
    t = tl.load(targets_ptr + row)

    max_val = tl.max(logits, axis=0)
    shifted = logits - max_val
    sum_exp = tl.sum(tl.exp(shifted), axis=0)
    log_denom = tl.log(sum_exp) + max_val
    target_logit = tl.load(logits_ptr + row * stride_row + t * stride_col).to(tl.float32)
    loss = log_denom - target_logit
    tl.store(out_ptr + row, loss)


def cross_entropy_loss(predictions: torch.Tensor, targets: torch.Tensor) -> torch.Tensor:
    assert predictions.is_cuda and targets.is_cuda
    assert predictions.ndim == 2 and targets.ndim == 1
    N, C = predictions.shape
    out = torch.empty((N,), device=predictions.device, dtype=torch.float32)
    block = triton.next_power_of_2(C)
    if block > 2048:
        block = 2048
    grid = (N,)
    cross_entropy_forward_kernel[grid](
        predictions, targets, out,
        predictions.stride(0), predictions.stride(1),
        N, C,
        BLOCK_SIZE=block,
        num_warps=8,
        num_stages=2,
    )
    return out.mean()


class Model(nn.Module):
    """
    A model that computes Cross Entropy Loss for multi-class classification tasks.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return cross_entropy_loss(predictions, targets)
