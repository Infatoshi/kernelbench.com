import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# For [64, 256, 256] summed over dim=1:
#   outer=64, reduce=256, inner=256
# The data layout is [outer, reduce, inner].
# To coalesce memory reads, we want consecutive threads to read
# consecutive memory addresses. Since the fastest-varying axis is `inner`,
# we assign consecutive threads to consecutive `inner` indices.
#
# Strategy: Each warp handles one (outer, reduce-chunk) row cooperatively
# across the inner dimension.  We tile the inner dimension with BLOCK_INNER
# threads so adjacent threads read adjacent elements.
#
# Better alternative for this specific shape: use warp-level parallelism.
# Each CTA handles a tile of `inner` with BLOCK_INNER threads and one `outer`.
# Loop over reduce dimension, accumulate into registers/shared mem, write out.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// -----------------------------------------------------------------------
// Coalesced sum-reduction kernel for [outer, reduce_size, inner] -> [outer, 1, inner]
//
// Grid:  (ceil(inner/BLOCK_X), outer)
// Block: (BLOCK_X,)   -- threads cover inner dimension (coalesced)
//
// Each thread loops over the reduce dimension and accumulates.
// -----------------------------------------------------------------------

template <typename scalar_t>
__global__ void sum_reduce_inner_kernel(
        const scalar_t* __restrict__ input,
        scalar_t*       __restrict__ output,
        int outer,
        int reduce_size,
        int inner)
{
    int inner_idx = (int)blockIdx.x * blockDim.x + (int)threadIdx.x;
    int outer_idx = (int)blockIdx.y;

    if (outer_idx >= outer || inner_idx >= inner) return;

    float acc = 0.0f;
    // input[outer_idx, r, inner_idx] = input[outer_idx*reduce_size*inner + r*inner + inner_idx]
    int base = outer_idx * reduce_size * inner + inner_idx;
    for (int r = 0; r < reduce_size; r++) {
        acc += (float)__ldg(input + base + r * inner);
    }

    // output[outer_idx, 0, inner_idx]
    output[outer_idx * inner + inner_idx] = (scalar_t)acc;
}

// -----------------------------------------------------------------------
// Reduction over the LAST dimension [outer, reduce_size]:
// Each thread handles one outer element, loops over reduce_size.
// -----------------------------------------------------------------------
template <typename scalar_t>
__global__ void sum_reduce_last_kernel(
        const scalar_t* __restrict__ input,
        scalar_t*       __restrict__ output,
        int outer,
        int reduce_size)
{
    int outer_idx = (int)blockIdx.x * blockDim.x + (int)threadIdx.x;
    if (outer_idx >= outer) return;

    float acc = 0.0f;
    int base = outer_idx * reduce_size;
    for (int r = 0; r < reduce_size; r++) {
        acc += (float)__ldg(input + base + r);
    }
    output[outer_idx] = (scalar_t)acc;
}

// -----------------------------------------------------------------------
// Reduction over FIRST dimension [reduce_size, inner] -> [1, inner]
// Each thread handles one inner element.
// -----------------------------------------------------------------------
template <typename scalar_t>
__global__ void sum_reduce_first_kernel(
        const scalar_t* __restrict__ input,
        scalar_t*       __restrict__ output,
        int reduce_size,
        int inner)
{
    int inner_idx = (int)blockIdx.x * blockDim.x + (int)threadIdx.x;
    if (inner_idx >= inner) return;

    float acc = 0.0f;
    for (int r = 0; r < reduce_size; r++) {
        acc += (float)__ldg(input + r * inner + inner_idx);
    }
    output[inner_idx] = (scalar_t)acc;
}

// -----------------------------------------------------------------------
// General shared-memory reduction for large reduce dims
// when inner is small (e.g., inner=1)
// -----------------------------------------------------------------------
template <typename scalar_t, int BLOCK>
__global__ void sum_reduce_large_kernel(
        const scalar_t* __restrict__ input,
        scalar_t*       __restrict__ output,
        int outer,
        int reduce_size,
        int inner)
{
    int outer_idx = blockIdx.x;
    int inner_idx = blockIdx.y;

    if (outer_idx >= outer || inner_idx >= inner) return;

    __shared__ float smem[BLOCK];

    float acc = 0.0f;
    for (int r = threadIdx.x; r < reduce_size; r += BLOCK) {
        int idx = outer_idx * reduce_size * inner + r * inner + inner_idx;
        acc += (float)__ldg(input + idx);
    }
    smem[threadIdx.x] = acc;
    __syncthreads();

    #pragma unroll
    for (int s = BLOCK/2; s > 32; s >>= 1) {
        if (threadIdx.x < s) smem[threadIdx.x] += smem[threadIdx.x + s];
        __syncthreads();
    }
    if (threadIdx.x < 32) {
        volatile float* vs = smem;
        if (BLOCK >= 64) vs[threadIdx.x] += vs[threadIdx.x + 32];
        vs[threadIdx.x] += vs[threadIdx.x + 16];
        vs[threadIdx.x] += vs[threadIdx.x +  8];
        vs[threadIdx.x] += vs[threadIdx.x +  4];
        vs[threadIdx.x] += vs[threadIdx.x +  2];
        vs[threadIdx.x] += vs[threadIdx.x +  1];
    }
    if (threadIdx.x == 0) {
        output[outer_idx * inner + inner_idx] = (scalar_t)smem[0];
    }
}

// -----------------------------------------------------------------------
// Launcher
// -----------------------------------------------------------------------
torch::Tensor sum_reduce(torch::Tensor x, int64_t dim) {
    int64_t ndim = x.dim();
    if (dim < 0) dim += ndim;

    auto xi = x.contiguous();

    // Compute outer, reduce, inner
    int64_t outer = 1, reduce_size = xi.size(dim), inner = 1;
    for (int d = 0;     d < dim;  ++d) outer *= xi.size(d);
    for (int d = dim+1; d < ndim; ++d) inner  *= xi.size(d);

    // Output shape
    auto out_sizes = xi.sizes().vec();
    out_sizes[dim] = 1;
    auto output = torch::empty(out_sizes, xi.options());

    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half, at::ScalarType::BFloat16,
        xi.scalar_type(), "sum_reduce", [&]() {

        if (inner >= 64) {
            // Coalesced: threads span inner dim
            // Heuristic: use 256 threads per block for good occupancy
            constexpr int BX = 256;
            dim3 grid((unsigned)((inner + BX - 1) / BX), (unsigned)outer);
            sum_reduce_inner_kernel<scalar_t><<<grid, BX>>>(
                xi.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                (int)outer, (int)reduce_size, (int)inner);
        } else if (inner == 1) {
            // Pure reduction per outer element; use shared memory
            int B = 256;
            dim3 grid((unsigned)outer, 1);
            sum_reduce_large_kernel<scalar_t, 256><<<grid, B>>>(
                xi.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                (int)outer, (int)reduce_size, (int)inner);
        } else {
            // Small inner: shared-memory reduction with 2D grid
            int B = 256;
            dim3 grid((unsigned)outer, (unsigned)inner);
            sum_reduce_large_kernel<scalar_t, 256><<<grid, B>>>(
                xi.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                (int)outer, (int)reduce_size, (int)inner);
        }
    });

    return output;
}
'''

cpp_source = 'torch::Tensor sum_reduce(torch::Tensor x, int64_t dim);'

ext = load_inline(
    name='sum_reduce_ext2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduce'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
    verbose=False,
)


class Model(nn.Module):
    """
    Sum reduction over a specified dimension (keepdim=True).
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce(x, self.dim)


# Keep identical to reference
batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
