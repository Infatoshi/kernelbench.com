import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template <typename scalar_t>
__global__ void particle_kernel(
    const scalar_t* __restrict__ positions,
    const scalar_t* __restrict__ velocities,
    const scalar_t dt_value,
    const scalar_t gravity_x,
    const scalar_t gravity_y,
    const scalar_t damping,
    scalar_t* __restrict__ output,
    const int64_t n
) {
    const int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    // Each particle has 2 position components and 2 velocity components
    // positions: [x, y] at idx*2, idx*2+1
    // velocities: [vx, vy] at idx*2, idx*2+1
    // output: [new_x, new_y, new_vx, new_vy] at idx*4, idx*4+1, idx*4+2, idx*4+3
    
    const scalar_t px = positions[idx * 2];
    const scalar_t py = positions[idx * 2 + 1];
    const scalar_t vx = velocities[idx * 2];
    const scalar_t vy = velocities[idx * 2 + 1];
    
    // Update velocity: new_vel = velocities + gravity * dt
    const scalar_t new_vx = vx + gravity_x * dt_value;
    const scalar_t new_vy = vy + gravity_y * dt_value;
    
    // Update position: new_pos = positions + new_vel * dt
    scalar_t new_px = px + new_vx * dt_value;
    scalar_t new_py = py + new_vy * dt_value;
    
    // Boundary collision detection and response
    const scalar_t abs_px = fabsf(new_px);
    const scalar_t abs_py = fabsf(new_py);
    
    // Apply damping where out of bounds and clamp position
    if (abs_px > 1.0f) {
        new_px = (new_px > 0.0f) ? 1.0f : -1.0f;
    }
    if (abs_py > 1.0f) {
        new_py = (new_py > 0.0f) ? 1.0f : -1.0f;
    }
    
    // Reflect and dampen velocity where collision occurred
    scalar_t final_vx = new_vx;
    scalar_t final_vy = new_vy;
    
    if (abs_px > 1.0f) {
        final_vx = -new_vx * damping;
    }
    if (abs_py > 1.0f) {
        final_vy = -new_vy * damping;
    }
    
    // Store output: [new_pos, new_vel] = [x, y, vx, vy]
    output[idx * 4] = new_px;
    output[idx * 4 + 1] = new_py;
    output[idx * 4 + 2] = final_vx;
    output[idx * 4 + 3] = final_vy;
}

torch::Tensor particle_forward(
    torch::Tensor positions,
    torch::Tensor velocities,
    torch::Tensor dt,
    torch::Tensor gravity,
    double damping
) {
    const int64_t n = positions.size(0);  // number of particles
    
    // Create output tensor: [n, 4] = [positions(2) + velocities(2)]
    auto output = torch::empty({n, 4}, positions.options());
    
    const int64_t block_size = 256;
    const int64_t num_blocks = (n + block_size - 1) / block_size;
    
    const auto dt_value = dt.item<double>();
    const auto gx = gravity[0].item<double>();
    const auto gy = gravity[1].item<double>();
    
    AT_DISPATCH_FLOATING_TYPES(positions.scalar_type(), "particle_kernel", ([&] {
        particle_kernel<scalar_t><<<num_blocks, block_size>>>(
            positions.data_ptr<scalar_t>(),
            velocities.data_ptr<scalar_t>(),
            static_cast<scalar_t>(dt_value),
            static_cast<scalar_t>(gx),
            static_cast<scalar_t>(gy),
            static_cast<scalar_t>(damping),
            output.data_ptr<scalar_t>(),
            n
        );
    }));
    
    return output;
}
'''

ext = load_inline(
    name='particle_ext',
    cpp_sources='torch::Tensor particle_forward(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, double);',
    cuda_sources=cuda_source,
    functions=['particle_forward'],
    verbose=False,
)


class Model(nn.Module):
    """Optimized GPU particle integration with fused CUDA kernel."""

    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        return ext.particle_forward(positions, velocities, dt, gravity, self.damping)


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
