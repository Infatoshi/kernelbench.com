import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized batched GEMM using WMMA tensor cores with FP16
# Shape: A(batch, m, k) * B(batch, k, n) = C(batch, m, n)
# batch=128, m=128, k=256, n=512
#
# Key design:
# - Use WMMA 16x16x16 tiles with FP16 inputs and FP32 accumulators
# - Each block handles one (batch, tile_m, tile_n)
# - Large block tiles to maximize tensor core utilization
# - Double-buffered shared memory for latency hiding

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>

using namespace nvcuda;

#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

// Large tile: 128x128 block, 32 k-step
// 4 warps in 2x2, each warp does 4x4 = 16 WMMA tiles
#define BM 128
#define BN 128
#define BK 32
#define WARP_M_TILES 4
#define WARP_N_TILES 4
#define WARPS_M 2
#define WARPS_N 2
#define TOTAL_WARPS (WARPS_M * WARPS_N)  // 4 warps, 128 threads

// sA: BM x (BK + pad) = 128 x 40
// sB: BK x (BN + pad) = 32 x 136

__global__ void __launch_bounds__(128, 4)
bmm_wmma_large_kernel(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    float* __restrict__ C,
    int batch, int m, int k, int n
) {
    int batch_idx = blockIdx.x;
    int block_m = blockIdx.y;
    int block_n = blockIdx.z;

    const __half* a_base = A + (long)batch_idx * m * k;
    const __half* b_base = B + (long)batch_idx * k * n;
    float*        c_base = C + (long)batch_idx * m * n;

    int warp_id = threadIdx.x / 32;
    int warp_row = warp_id / WARPS_N;
    int warp_col = warp_id % WARPS_N;

    int tid = threadIdx.x;
    int num_threads = blockDim.x;  // 128

    // Double-buffered shared memory
    __shared__ __half sA[2][BM][BK + 8];   // [2][128][40]
    __shared__ __half sB[2][BK][BN + 8];   // [2][32][136]

    // Accumulator fragments
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc[WARP_M_TILES][WARP_N_TILES];
    #pragma unroll
    for (int i = 0; i < WARP_M_TILES; i++)
        #pragma unroll
        for (int j = 0; j < WARP_N_TILES; j++)
            wmma::fill_fragment(acc[i][j], 0.0f);

    int k_tiles = (k + BK - 1) / BK;
    int buf = 0;

    // Preload first tile into buf=0
    {
        int k_start = 0;
        #pragma unroll 4
        for (int idx = tid; idx < BM * BK; idx += num_threads) {
            int r = idx / BK, c = idx % BK;
            int gr = block_m * BM + r;
            int gc = k_start + c;
            sA[0][r][c] = (gr < m && gc < k) ? a_base[gr * k + gc] : __float2half(0.0f);
        }
        #pragma unroll 4
        for (int idx = tid; idx < BK * BN; idx += num_threads) {
            int r = idx / BN, c = idx % BN;
            int gr = k_start + r;
            int gc = block_n * BN + c;
            sB[0][r][c] = (gr < k && gc < n) ? b_base[gr * n + gc] : __float2half(0.0f);
        }
        __syncthreads();
    }

    for (int kt = 0; kt < k_tiles; kt++) {
        int next_buf = 1 - buf;

        // Prefetch next tile
        if (kt + 1 < k_tiles) {
            int k_start = (kt + 1) * BK;
            #pragma unroll 4
            for (int idx = tid; idx < BM * BK; idx += num_threads) {
                int r = idx / BK, c = idx % BK;
                int gr = block_m * BM + r;
                int gc = k_start + c;
                sA[next_buf][r][c] = (gr < m && gc < k) ? a_base[gr * k + gc] : __float2half(0.0f);
            }
            #pragma unroll 4
            for (int idx = tid; idx < BK * BN; idx += num_threads) {
                int r = idx / BN, c = idx % BN;
                int gr = k_start + r;
                int gc = block_n * BN + c;
                sB[next_buf][r][c] = (gr < k && gc < n) ? b_base[gr * n + gc] : __float2half(0.0f);
            }
        }

        // Compute using current buffer
        // BK / WMMA_K = 2 inner k steps
        const int K_INNER = BK / WMMA_K;  // 2

        wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> a_frag[WARP_M_TILES][K_INNER];
        wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> b_frag[WARP_N_TILES][K_INNER];

        // Load A frags
        #pragma unroll
        for (int ki = 0; ki < K_INNER; ki++) {
            #pragma unroll
            for (int mi = 0; mi < WARP_M_TILES; mi++) {
                int m_offset = (warp_row * WARP_M_TILES + mi) * WMMA_M;
                wmma::load_matrix_sync(a_frag[mi][ki], &sA[buf][m_offset][ki * WMMA_K], BK + 8);
            }
        }

        // Load B frags
        #pragma unroll
        for (int ki = 0; ki < K_INNER; ki++) {
            #pragma unroll
            for (int ni = 0; ni < WARP_N_TILES; ni++) {
                int n_offset = (warp_col * WARP_N_TILES + ni) * WMMA_N;
                wmma::load_matrix_sync(b_frag[ni][ki], &sB[buf][ki * WMMA_K][n_offset], BN + 8);
            }
        }

        // Compute
        #pragma unroll
        for (int ki = 0; ki < K_INNER; ki++) {
            #pragma unroll
            for (int mi = 0; mi < WARP_M_TILES; mi++) {
                #pragma unroll
                for (int ni = 0; ni < WARP_N_TILES; ni++) {
                    wmma::mma_sync(acc[mi][ni], a_frag[mi][ki], b_frag[ni][ki], acc[mi][ni]);
                }
            }
        }

        buf = next_buf;
        __syncthreads();
    }

    // Store results
    int base_m = block_m * BM + warp_row * WARP_M_TILES * WMMA_M;
    int base_n = block_n * BN + warp_col * WARP_N_TILES * WMMA_N;
    #pragma unroll
    for (int mi = 0; mi < WARP_M_TILES; mi++) {
        #pragma unroll
        for (int ni = 0; ni < WARP_N_TILES; ni++) {
            int out_m = base_m + mi * WMMA_M;
            int out_n = base_n + ni * WMMA_N;
            if (out_m < m && out_n < n) {
                wmma::store_matrix_sync(
                    &c_base[out_m * n + out_n],
                    acc[mi][ni],
                    n,
                    wmma::mem_row_major
                );
            }
        }
    }
}


torch::Tensor bmm_wmma(torch::Tensor A, torch::Tensor B) {
    int batch = A.size(0);
    int m = A.size(1);
    int k = A.size(2);
    int n = B.size(2);

    auto C = torch::empty({batch, m, n},
        torch::TensorOptions().dtype(torch::kFloat32).device(A.device()));

    auto A_fp16 = A.to(torch::kFloat16);
    auto B_fp16 = B.to(torch::kFloat16);

    const __half* a_ptr = reinterpret_cast<const __half*>(A_fp16.data_ptr<at::Half>());
    const __half* b_ptr = reinterpret_cast<const __half*>(B_fp16.data_ptr<at::Half>());
    float* c_ptr = C.data_ptr<float>();

    dim3 grid(batch, (m + BM - 1) / BM, (n + BN - 1) / BN);
    dim3 block(TOTAL_WARPS * 32);  // 128 threads

    bmm_wmma_large_kernel<<<grid, block>>>(a_ptr, b_ptr, c_ptr, batch, m, k, n);
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(std::string("CUDA error: ") + cudaGetErrorString(err));
    }

    return C;
}
'''

cpp_source = '''
torch::Tensor bmm_wmma(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='bmm_wmma_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bmm_wmma'],
    verbose=True,
    extra_cuda_cflags=[
        '-arch=sm_86',
        '-std=c++17',
        '-O3',
        '--use_fast_math',
        '-DNDEBUG',
    ],
)


class Model(nn.Module):
    """
    Performs batched matrix multiplication using WMMA tensor cores.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.bmm_wmma(A, B)


batch_size = 128
m = 128
k = 256
n = 512


def get_inputs():
    A = torch.randn(batch_size, m, k)
    B = torch.randn(batch_size, k, n)
    return [A, B]


def get_init_inputs():
    return []
