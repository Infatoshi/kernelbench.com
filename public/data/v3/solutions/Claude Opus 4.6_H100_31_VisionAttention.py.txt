import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def fused_residual_layernorm_kernel(
    OUT_ptr, RESIDUAL_ptr, X_ptr,
    WEIGHT_ptr, BIAS_ptr,
    N: tl.constexpr,
    eps: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < N

    x_offset = row * N + cols
    x_val = tl.load(X_ptr + x_offset, mask=mask, other=0.0).to(tl.float32)
    res_val = tl.load(RESIDUAL_ptr + x_offset, mask=mask, other=0.0).to(tl.float32)

    val = x_val + res_val

    mean = tl.sum(val, axis=0) / N
    val_centered = val - mean
    var = tl.sum(val_centered * val_centered, axis=0) / N
    inv_std = 1.0 / tl.sqrt(var + eps)
    normed = val_centered * inv_std

    w = tl.load(WEIGHT_ptr + cols, mask=mask, other=1.0).to(tl.float32)
    b = tl.load(BIAS_ptr + cols, mask=mask, other=0.0).to(tl.float32)
    out = normed * w + b

    tl.store(OUT_ptr + x_offset, out.to(tl.float16), mask=mask)


@triton.jit
def fused_residual_layernorm_to_bchw_kernel(
    OUT_ptr,           # (B, C, H*W) output
    RESIDUAL_ptr,      # (B, S, C) input residual
    X_ptr,             # (B, S, C) attention output
    WEIGHT_ptr, BIAS_ptr,
    B: tl.constexpr,
    S: tl.constexpr,
    C: tl.constexpr,
    eps: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one row of (B*S, C) -> layernorm -> write transposed to (B, C, S)
    row = tl.program_id(0)
    b_idx = row // S
    s_idx = row % S
    
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < C

    # Read from (B, S, C) layout
    in_offset = row * C + cols
    x_val = tl.load(X_ptr + in_offset, mask=mask, other=0.0).to(tl.float32)
    res_val = tl.load(RESIDUAL_ptr + in_offset, mask=mask, other=0.0).to(tl.float32)

    val = x_val + res_val

    mean = tl.sum(val, axis=0) / C
    val_centered = val - mean
    var = tl.sum(val_centered * val_centered, axis=0) / C
    inv_std = 1.0 / tl.sqrt(var + eps)
    normed = val_centered * inv_std

    w = tl.load(WEIGHT_ptr + cols, mask=mask, other=1.0).to(tl.float32)
    b_val = tl.load(BIAS_ptr + cols, mask=mask, other=0.0).to(tl.float32)
    out = normed * w + b_val

    # Write to (B, C, S) layout -> offset = b_idx * C * S + cols * S + s_idx
    out_offset = b_idx * C * S + cols * S + s_idx
    tl.store(OUT_ptr + out_offset, out, mask=mask)


class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(Model, self).__init__()
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads

        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)
        
        self._cached_weights = False

    def _cache_weights(self):
        """Pre-transpose and convert weights to fp16 for efficiency."""
        with torch.no_grad():
            self._in_proj_weight_t = self.attn.in_proj_weight.data.half().t().contiguous()
            self._in_proj_bias = self.attn.in_proj_bias.data.half()
            self._out_proj_weight_t = self.attn.out_proj.weight.data.half().t().contiguous()
            self._out_proj_bias = self.attn.out_proj.bias.data.half()
            self._norm_weight = self.norm.weight.data.half()
            self._norm_bias = self.norm.bias.data.half()
        self._cached_weights = True

    def forward(self, x):
        if not self._cached_weights:
            self._cache_weights()
            
        B, C, H, W = x.shape
        seq_len = H * W

        # (B, C, H*W) -> (B, H*W, C) via permute, then contiguous for matmul
        x_half = x.view(B, C, seq_len).permute(0, 2, 1).contiguous().half()
        
        # QKV projection: (B*seq_len, C) @ (C, 3C) -> (B*seq_len, 3C)
        x_flat = x_half.view(B * seq_len, C)
        qkv = torch.addmm(self._in_proj_bias, x_flat, self._in_proj_weight_t)
        
        # Reshape to (B, seq_len, 3, num_heads, head_dim) and split
        qkv = qkv.view(B, seq_len, 3, self.num_heads, self.head_dim)
        q = qkv[:, :, 0].transpose(1, 2)  # (B, num_heads, seq_len, head_dim)
        k = qkv[:, :, 1].transpose(1, 2)
        v = qkv[:, :, 2].transpose(1, 2)

        # Flash attention
        attn_output = F.scaled_dot_product_attention(q, k, v, dropout_p=0.0, is_causal=False)

        # (B, num_heads, seq_len, head_dim) -> (B*seq_len, C)
        attn_output = attn_output.transpose(1, 2).contiguous().view(B * seq_len, C)

        # Output projection
        attn_output = torch.addmm(self._out_proj_bias, attn_output, self._out_proj_weight_t)
        attn_output = attn_output.view(B, seq_len, C)

        # Fused residual + layernorm + transpose to (B, C, S)
        out = torch.empty(B, C, seq_len, device=x.device, dtype=torch.float32)
        num_rows = B * seq_len
        BLOCK_SIZE = triton.next_power_of_2(C)
        
        fused_residual_layernorm_to_bchw_kernel[(num_rows,)](
            out, x_half, attn_output,
            self._norm_weight, self._norm_bias,
            B, seq_len, C,
            self.norm.eps,
            BLOCK_SIZE,
        )

        return out.view(B, C, H, W)
