import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import math

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <c10/cuda/CUDAException.h>
#include <cmath>

// Optimized kernel using shared memory tiling for matmul + mean + dropout
template<typename T, int BLOCK_SIZE>
__global__ void fused_matmul_mean_kernel(
    const T* input,      // (batch, in_features)
    const T* weight,     // (out_features, in_features)
    const T* bias,       // (out_features,)
    T* output,           // (1, out_features)
    int batch,
    int in_features,
    int out_features,
    float dropout_scale
) {
    int out_idx = blockIdx.x;
    
    if (out_idx >= out_features) return;
    
    // Shared memory for tiles
    extern __shared__ float shared_mem[];
    float* tile_input = shared_mem;
    float* tile_weight = shared_mem + BLOCK_SIZE;
    
    float accumulator = 0.0f;
    
    // Tile through K dimension
    for (int k_tile = 0; k_tile < in_features; k_tile += BLOCK_SIZE) {
        // Load input tile (sum across batch for each k)
        int k = k_tile + threadIdx.x;
        float input_sum = 0.0f;
        if (k < in_features) {
            for (int b = 0; b < batch; b++) {
                input_sum += static_cast<float>(input[b * in_features + k]);
            }
        }
        tile_input[threadIdx.x] = input_sum;
        
        // Load weight tile
        float w = 0.0f;
        if (k < in_features) {
            w = static_cast<float>(weight[out_idx * in_features + k]);
        }
        tile_weight[threadIdx.x] = w;
        
        __syncthreads();
        
        // Compute partial dot product
        for (int i = 0; i < BLOCK_SIZE; i++) {
            accumulator += tile_input[i] * tile_weight[i];
        }
        
        __syncthreads();
    }
    
    // Add bias (once per batch element, so multiply by batch)
    accumulator += static_cast<float>(bias[out_idx]) * batch;
    
    // Apply dropout scale and divide by batch for mean
    float result = accumulator * dropout_scale / batch;
    
    // Store intermediate result (before softmax)
    output[out_idx] = static_cast<T>(result);
}

// Softmax kernel
template<typename T>
__global__ void softmax_kernel(
    T* data,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    extern __shared__ float sdata[];
    float* max_vals = sdata;
    float* sum_vals = sdata + blockDim.x;
    
    float local_max = -FLT_MAX;
    float local_sum = 0.0f;
    
    if (idx < n) {
        float val = static_cast<float>(data[idx]);
        local_max = val;
    }
    
    // Find max
    max_vals[threadIdx.x] = local_max;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            max_vals[threadIdx.x] = fmaxf(max_vals[threadIdx.x], max_vals[threadIdx.x + stride]);
        }
        __syncthreads();
    }
    
    float max_val = max_vals[0];
    
    // Compute exp and sum
    if (idx < n) {
        float val = static_cast<float>(data[idx]);
        float exp_val = expf(val - max_val);
        data[idx] = static_cast<T>(exp_val);
        local_sum = exp_val;
    } else {
        local_sum = 0.0f;
    }
    
    sum_vals[threadIdx.x] = local_sum;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sum_vals[threadIdx.x] += sum_vals[threadIdx.x + stride];
        }
        __syncthreads();
    }
    
    float sum_val = sum_vals[0];
    
    // Normalize
    if (idx < n) {
        data[idx] = static_cast<T>(static_cast<float>(data[idx]) / sum_val);
    }
}

torch::Tensor fused_matmul_dropout_mean_softmax(
    torch::Tensor input,   // (batch, in_features)
    torch::Tensor weight,  // (out_features, in_features)
    torch::Tensor bias,    // (out_features,)
    float dropout_scale
) {
    int batch = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    auto output = torch::empty({1, out_features}, input.options());
    
    // Launch matmul + mean kernel
    int block_size = 256;
    int grid_size = out_features;
    size_t shared_mem_size = block_size * 2 * sizeof(float);
    
    if (input.dtype() == torch::kFloat16) {
        fused_matmul_mean_kernel<at::Half, 256><<<grid_size, block_size, shared_mem_size>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            bias.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            batch, in_features, out_features, dropout_scale
        );
    } else if (input.dtype() == torch::kBFloat16) {
        fused_matmul_mean_kernel<at::BFloat16, 256><<<grid_size, block_size, shared_mem_size>>>(
            input.data_ptr<at::BFloat16>(),
            weight.data_ptr<at::BFloat16>(),
            bias.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            batch, in_features, out_features, dropout_scale
        );
    } else {
        fused_matmul_mean_kernel<float, 256><<<grid_size, block_size, shared_mem_size>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            batch, in_features, out_features, dropout_scale
        );
    }
    
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    
    // Apply softmax
    softmax_kernel<<<1, 256, 256 * 2 * sizeof(float)>>>(
        output.data_ptr<float>(),
        out_features
    );
    C10_CUDA_KERNEL_LAUNCH_CHECK();
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor fused_matmul_dropout_mean_softmax(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float dropout_scale
);
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    """
    Optimized fused model: matmul + dropout + mean + softmax
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        self.dropout_scale = 1.0 / (1.0 - dropout_p)
        self.in_features = in_features
        self.out_features = out_features
        self._init_weights()

    def _init_weights(self):
        # Use same initialization as nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_op(x, self.weight, self.bias, self.dropout_scale)

def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    in_features = 4096
    out_features = 4096
    dropout_p = 0.2
    return [in_features, out_features, dropout_p]
