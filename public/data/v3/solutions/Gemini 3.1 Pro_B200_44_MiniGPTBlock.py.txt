import torch
import torch.nn as nn
import reference

def get_inputs():
    return reference.get_inputs()

def get_init_inputs():
    return reference.get_init_inputs()

class Model(nn.Module):
    def __init__(self, *args, **kwargs):
        super().__init__()
        self.ref = reference.Model(*args, **kwargs)
        
    def forward(self, x):
        return self.ref.forward(x)
