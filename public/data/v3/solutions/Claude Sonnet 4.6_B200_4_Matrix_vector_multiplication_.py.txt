import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized GEMV kernel for (M, K) x (K, 1) -> (M, 1)
# Strategy: each warp computes one output element via parallel dot product + warp reduction
# Use float4 vectorized loads for maximum memory bandwidth

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMV kernel: C[m] = sum_k A[m,k] * B[k]
// Grid: (M / ROWS_PER_BLOCK), 1
// Block: (WARP_SIZE * WARPS_PER_BLOCK), 1
// Each warp handles one row of A

#define WARP_SIZE 32

__inline__ __device__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = WARP_SIZE/2; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Each warp computes one dot product: one row of A times B
// Uses float4 vectorized loads (128-bit)
__global__ void gemv_kernel(
    const float* __restrict__ A,   // [M, K]
    const float* __restrict__ B,   // [K]
    float* __restrict__ C,         // [M]
    int M, int K
) {
    int warp_id = threadIdx.x / WARP_SIZE;
    int lane    = threadIdx.x % WARP_SIZE;
    int warps_per_block = blockDim.x / WARP_SIZE;

    int row = blockIdx.x * warps_per_block + warp_id;
    if (row >= M) return;

    const float* a_row = A + row * K;

    // Vectorized load: each thread loads 4 floats at a time
    float sum = 0.0f;
    int K4 = K / 4;
    const float4* a4 = reinterpret_cast<const float4*>(a_row);
    const float4* b4 = reinterpret_cast<const float4*>(B);

    for (int i = lane; i < K4; i += WARP_SIZE) {
        float4 av = a4[i];
        float4 bv = b4[i];
        sum += av.x * bv.x + av.y * bv.y + av.z * bv.z + av.w * bv.w;
    }

    // Handle remainder (K not divisible by 4)
    int rem_start = K4 * 4 + lane;
    for (int i = rem_start; i < K; i += WARP_SIZE) {
        sum += a_row[i] * B[i];
    }

    // Warp-level reduction
    sum = warp_reduce_sum(sum);

    // Lane 0 writes result
    if (lane == 0) {
        C[row] = sum;
    }
}

// Optimized version with shared memory staging and multiple rows per block
// Uses 8 warps per block = 256 threads, each warp handles 1 row
// With large K=131072, the bottleneck is reading A (M*K floats)
__global__ void gemv_kernel_v2(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int K
) {
    extern __shared__ float smem_b[];  // shared B cache

    int warp_id = threadIdx.x / WARP_SIZE;
    int lane    = threadIdx.x % WARP_SIZE;
    int warps_per_block = blockDim.x / WARP_SIZE;

    int row = blockIdx.x * warps_per_block + warp_id;

    // Pre-load a tile of B into shared memory cooperatively
    // Then each warp reads B from smem
    // With K=131072 floats = 512KB, too large for smem
    // So we tile: process TILE_K elements of K at a time from smem

    const int TILE_K = 512;  // floats per tile (2KB)
    float sum = 0.0f;

    for (int k_tile = 0; k_tile < K; k_tile += TILE_K) {
        int tile_size = min(TILE_K, K - k_tile);

        // Cooperatively load B tile into shared memory
        for (int i = threadIdx.x; i < tile_size; i += blockDim.x) {
            smem_b[i] = B[k_tile + i];
        }
        __syncthreads();

        if (row < M) {
            const float* a_row = A + row * K + k_tile;
            int ts4 = tile_size / 4;
            const float4* a4 = reinterpret_cast<const float4*>(a_row);
            const float4* b4 = reinterpret_cast<const float4*>(smem_b);

            for (int i = lane; i < ts4; i += WARP_SIZE) {
                float4 av = a4[i];
                float4 bv = b4[i];
                sum += av.x * bv.x + av.y * bv.y + av.z * bv.z + av.w * bv.w;
            }
            // handle remainder
            for (int i = ts4*4 + lane; i < tile_size; i += WARP_SIZE) {
                sum += a_row[i] * smem_b[i];
            }
        }
        __syncthreads();
    }

    if (row < M) {
        sum = warp_reduce_sum(sum);
        if (lane == 0) {
            C[row] = sum;
        }
    }
}

torch::Tensor gemv(torch::Tensor A, torch::Tensor B) {
    // A: [M, K], B: [K, 1] or [K]
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "Inputs must be CUDA tensors");
    TORCH_CHECK(A.dtype() == torch::kFloat32, "A must be float32");
    TORCH_CHECK(B.dtype() == torch::kFloat32, "B must be float32");
    TORCH_CHECK(A.is_contiguous(), "A must be contiguous");

    int M = A.size(0);
    int K = A.size(1);

    auto B_flat = B.reshape({K});
    if (!B_flat.is_contiguous()) B_flat = B_flat.contiguous();

    auto C = torch::empty({M, 1}, A.options());

    // Use kernel v1: simple warp-per-row with vectorized loads
    int warps_per_block = 8;
    int threads = warps_per_block * WARP_SIZE;
    int blocks = (M + warps_per_block - 1) / warps_per_block;

    gemv_kernel<<<blocks, threads>>>(
        A.data_ptr<float>(),
        B_flat.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K
    );

    return C;
}
'''

cpp_source = 'torch::Tensor gemv(torch::Tensor A, torch::Tensor B);'

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math', '-Xptxas=-v'],
)


class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.gemv(A, B)


M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
