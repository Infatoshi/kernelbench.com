import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Fused kernel: Conv2d + Subtract + Tanh + Subtract + AvgPool2d
// Conv2d parameters (hardcoded for this specific case)
const int KH = 3;  // kernel_height
const int KW = 3;  // kernel_width
const int IC = 3;  // input_channels
const int OC = 16; // output_channels
const int IH = 256; // input_height
const int IW = 256; // input_width
const int OH = 254; // output_height = IH - KH + 1
const int OW = 254; // output_width = IW - KW + 1

// Pooling parameters
const int POOL_K = 2;
const int POOL_OH = OH / POOL_K; // 127
const int POOL_OW = OW / POOL_K; // 127

// Main fused kernel - each thread computes one output value after pooling
__global__ void fused_kernel_pool_output(
    const float* __restrict__ input,
    const float* __restrict__ conv_weight,
    const float* __restrict__ conv_bias,
    float* __restrict__ output,
    float subtract1_value,
    float subtract2_value,
    int batch_size
) {
    const int batch_idx = blockIdx.z;
    const int oc = blockIdx.y;
    const int pool_h = blockIdx.x;
    const int pool_w = threadIdx.x;  // Fixed: threadIdx.x directly is the pool_w
    
    if (batch_idx >= batch_size || oc >= OC || pool_h >= POOL_OH || pool_w >= POOL_OW) return;
    
    // Compute the corresponding convolution output position (top-left of pool window)
    int conv_h = pool_h * POOL_K;
    int conv_w = pool_w * POOL_K;
    
    float pool_sum = 0.0f;
    int pool_count = 0;
    
    for (int ph = 0; ph < POOL_K; ++ph) {
        for (int pw = 0; pw < POOL_K; ++pw) {
            int conv_h = pool_h * POOL_K + ph;
            int conv_w = pool_w * POOL_K + pw;
            
            if (conv_h >= OH || conv_w >= OW) continue;
            
            float sum_val = conv_bias[oc];
            
            for (int ic = 0; ic < IC; ++ic) {
                const float* weight_oc_ic = conv_weight + ((oc * IC + ic) * KH * KW);
                const float* input_batch_ic = input + (batch_idx * IC + ic) * IH * IW;
                
                for (int kh = 0; kh < KH; ++kh) {
                    int iy = conv_h - kh;
                    if (iy < 0 || iy >= IH) continue;
                    
                    for (int kw = 0; kw < KW; ++kw) {
                        int ix = conv_w - kw;
                        if (ix < 0 || ix >= IW) continue;
                        
                        int weight_idx = kh * KW + kw;
                        int input_idx = iy * IW + ix;
                        
                        sum_val += weight_oc_ic[weight_idx] * input_batch_ic[input_idx];
                    }
                }
            }
            
            // Apply activations
            sum_val = tanhf(sum_val - subtract1_value) - subtract2_value;
            pool_sum += sum_val;
            pool_count++;
        }
    }
    
    // Average
    float pooled = pool_sum / static_cast<float>(pool_count);
    
    // Store output - correct indexing for [batch, oc, pool_h, pool_w]
    int out_idx = ((batch_idx * OC + oc) * POOL_OH + pool_h) * POOL_OW + pool_w;
    output[out_idx] = pooled;
}

// Wrapper to run the fused kernel
torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor conv_weight,
    torch::Tensor conv_bias,
    float subtract1_value,
    float subtract2_value
) {
    const int batch_size = input.size(0);
    
    // Allocate output
    auto output = torch::empty({batch_size, OC, POOL_OH, POOL_OW}, input.options());
    
    // Calculate grid dimensions
    dim3 grid(POOL_OH, OC, batch_size);
    
    // Launch kernel with POOL_OW threads per block
    fused_kernel_pool_output<<<grid, POOL_OW>>>(
        input.data_ptr<float>(),
        conv_weight.data_ptr<float>(),
        conv_bias.data_ptr<float>(),
        output.data_ptr<float>(),
        subtract1_value,
        subtract2_value,
        batch_size
    );
    
    cudaGetLastError();  // Check for errors
    return output;
}
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources='''
torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor conv_weight,
    torch::Tensor conv_bias,
    float subtract1_value,
    float subtract2_value
);
''',
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False,
    extra_include_paths=[],
)

class Model(nn.Module):
    """
    Optimized fused kernel for Conv2d + Subtract + Tanh + Subtract + AvgPool2d
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        
        # Ensure contiguous weights
        self.register_buffer('conv_weight', self.conv.weight.detach().contiguous())
        self.register_buffer('conv_bias', self.conv.bias.detach().contiguous())
        
    def forward(self, x):
        # Ensure contiguous memory
        x = x.contiguous()
        
        # Call fused CUDA kernel
        return ext.fused_forward(
            x,
            self.conv_weight,
            self.conv_bias,
            self.subtract1_value,
            self.subtract2_value
        )

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
