import torch
import torch.nn as nn
import torch.nn.functional as F
import math

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ['fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


class FusedQKVLinear(nn.Module):
    """Fused QKV projection - more efficient than 3 separate linear layers"""
    def __init__(self, in_features, out_features):
        super().__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        return F.linear(x, self.weight, self.bias)


class Model(nn.Module):
    """
    Optimized multi-head causal self-attention using Flash Attention.
    Uses F.scaled_dot_product_attention which dispatches to Flash Attention 2 on H100.
    """

    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        
        self.n_head = n_head
        self.n_embd = n_embd
        self.head_dim = n_embd // n_head
        self.scale = 1.0 / math.sqrt(self.head_dim)
        
        # Fused QKV projection for efficiency
        self.c_attn = FusedQKVLinear(n_embd, 3 * n_embd)
        # Output projection
        self.c_proj = nn.Linear(n_embd, n_embd)
        
        # Regularization (dropout rates stored but not used in inference when p=0)
        self.attn_pdrop = attn_pdrop
        self.resid_pdrop = resid_pdrop
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        
        # Causal mask for SDPA
        self.register_buffer(
            "causal_mask",
            torch.tril(torch.ones(max_seqlen, max_seqlen, dtype=torch.bool))
                .view(1, 1, max_seqlen, max_seqlen)
        )
        
        self.max_seqlen = max_seqlen

    def forward(self, x):
        B, T, C = x.size()
        
        # Compute Q, K, V in one fused projection
        qkv = self.c_attn(x)  # (B, T, 3*C)
        
        # Split and reshape for multi-head attention
        q, k, v = qkv.split(self.n_embd, dim=2)
        
        # Reshape to (B, n_head, T, head_dim) - optimized memory layout
        q = q.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        k = k.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        v = v.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        
        # Use Flash Attention via scaled_dot_product_attention
        # is_causal=True enables causal masking efficiently
        # This dispatches to Flash Attention 2 on H100 with SM90
        y = F.scaled_dot_product_attention(
            q, k, v,
            attn_mask=None,
            dropout_p=self.attn_pdrop if self.training else 0.0,
            is_causal=True,
            scale=self.scale
        )
        
        # Reassemble heads
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        
        # Output projection
        y = self.resid_dropout(self.c_proj(y))
        
        return y


# Same parameters as reference
batch_size = 64
max_seqlen = 1024
seq_len = 512
n_embd = 768
n_head = 8
attn_pdrop = 0.0
resid_pdrop = 0.0


def get_inputs():
    return [torch.randn(batch_size, seq_len, n_embd)]


def get_init_inputs():
    return [n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen]
