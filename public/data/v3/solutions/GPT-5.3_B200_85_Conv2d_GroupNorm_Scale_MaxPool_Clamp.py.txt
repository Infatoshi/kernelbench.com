import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel(const float* x, const float* w, const float* b, float* y,
                              int N, int C, int H, int W,
                              int OC, int K, int OH, int OW) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int oc = (idx / (OW*OH)) % OC;
    int n = idx / (OW*OH*OC);

    float val = b ? b[oc] : 0.0f;
    for (int ic = 0; ic < C; ic++) {
        for (int kh = 0; kh < K; kh++) {
            for (int kw = 0; kw < K; kw++) {
                int ih = oh + kh;
                int iw = ow + kw;
                int x_idx = ((n*C + ic)*H + ih)*W + iw;
                int w_idx = ((oc*C + ic)*K + kh)*K + kw;
                val += x[x_idx] * w[w_idx];
            }
        }
    }
    y[idx] = val;
}

__global__ void groupnorm_scale_kernel(float* x, const float* scale,
                                       int N, int C, int H, int W, int G) {
    int n = blockIdx.x;
    int g = blockIdx.y;

    int channels_per_group = C / G;
    int c_start = g * channels_per_group;

    float mean = 0.0f;
    float var = 0.0f;
    int count = channels_per_group * H * W;

    for (int c = 0; c < channels_per_group; c++) {
        for (int h = 0; h < H; h++) {
            for (int w = 0; w < W; w++) {
                int idx = ((n*C + (c_start+c))*H + h)*W + w;
                mean += x[idx];
            }
        }
    }
    mean /= count;

    for (int c = 0; c < channels_per_group; c++) {
        for (int h = 0; h < H; h++) {
            for (int w = 0; w < W; w++) {
                int idx = ((n*C + (c_start+c))*H + h)*W + w;
                float v = x[idx] - mean;
                var += v * v;
            }
        }
    }
    var = rsqrtf(var / count + 1e-5f);

    for (int c = 0; c < channels_per_group; c++) {
        float s = scale[c_start + c];
        for (int h = 0; h < H; h++) {
            for (int w = 0; w < W; w++) {
                int idx = ((n*C + (c_start+c))*H + h)*W + w;
                float v = (x[idx] - mean) * var;
                x[idx] = v * s;
            }
        }
    }
}

__global__ void maxpool_clamp_kernel(const float* x, float* y,
                                     int N, int C, int H, int W,
                                     int OH, int OW,
                                     float cmin, float cmax) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int c = (idx / (OW*OH)) % C;
    int n = idx / (OW*OH*C);

    float m = -1e30f;
    for (int kh = 0; kh < 2; kh++) {
        for (int kw = 0; kw < 2; kw++) {
            int ih = oh*2 + kh;
            int iw = ow*2 + kw;
            int in_idx = ((n*C + c)*H + ih)*W + iw;
            m = fmaxf(m, x[in_idx]);
        }
    }
    m = fminf(fmaxf(m, cmin), cmax);
    y[idx] = m;
}

torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b,
                       torch::Tensor scale, int G, float cmin, float cmax) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int K = w.size(2);
    int OH = H - K + 1;
    int OW = W - K + 1;

    auto y = torch::empty({N, OC, OH, OW}, x.options());

    int threads = 256;
    int total = N*OC*OH*OW;
    conv2d_kernel<<<(total+threads-1)/threads, threads>>>(
        x.data_ptr<float>(), w.data_ptr<float>(), b.data_ptr<float>(), y.data_ptr<float>(),
        N,C,H,W,OC,K,OH,OW);

    dim3 grid(N, G);
    groupnorm_scale_kernel<<<grid,1>>>(y.data_ptr<float>(), scale.data_ptr<float>(), N, OC, OH, OW, G);

    int PH = OH/2;
    int PW = OW/2;
    auto out = torch::empty({N, OC, PH, PW}, x.options());

    int total2 = N*OC*PH*PW;
    maxpool_clamp_kernel<<<(total2+threads-1)/threads, threads>>>(
        y.data_ptr<float>(), out.data_ptr<float>(), N, OC, OH, OW, PH, PW, cmin, cmax);

    return out;
}
'''

ext = load_inline(name='fused_ext',
                  cpp_sources='torch::Tensor fused_op(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, int, float, float);',
                  cuda_sources=cuda_source,
                  functions=['fused_op'],
                  verbose=False)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.num_groups = num_groups

    def forward(self, x):
        w = self.conv.weight
        b = self.conv.bias
        return ext.fused_op(x.contiguous(), w.contiguous(), b.contiguous(),
                            self.scale.contiguous().view(-1),
                            self.num_groups,
                            self.clamp_min,
                            self.clamp_max)


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 8, (16,1,1), 2, 0.0, 1.0]
