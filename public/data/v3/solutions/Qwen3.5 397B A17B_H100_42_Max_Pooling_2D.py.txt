import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <algorithm>

// Max pooling kernel optimized for H100
// Uses shared memory tiling and warp-level reduction

template<typename T>
__global__ void max_pool2d_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int height,
    int width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    // Each block handles one channel of one batch item
    int b = blockIdx.z;
    int c = blockIdx.y;
    int out_h_start = blockIdx.x * blockDim.y;
    int out_w_start = threadIdx.x * blockDim.x;
    
    int effective_kernel = kernel_size + (kernel_size - 1) * (dilation - 1);
    
    extern __shared__ float shared_mem[];
    
    // Load tile into shared memory with padding
    int tile_h = blockDim.y + effective_kernel - 1;
    int tile_w = blockDim.x * blockDim.z + effective_kernel - 1;
    
    for (int th = threadIdx.y; th < tile_h; th += blockDim.y) {
        int in_h = out_h_start + th - padding;
        for (int tw = threadIdx.x; tw < tile_w; tw += blockDim.x) {
            int in_w = out_w_start + tw - padding;
            float val = -__FLT_MAX__;
            if (in_h >= 0 && in_h < height && in_w >= 0 && in_w < width) {
                val = static_cast<float>(input[((b * channels + c) * height + in_h) * width + in_w]);
            }
            shared_mem[th * tile_w + tw] = val;
        }
    }
    __syncthreads();
    
    // Compute max pooling
    for (int oh = threadIdx.y; oh < blockDim.y; oh++) {
        int out_h = out_h_start + oh;
        if (out_h >= out_height) continue;
        
        for (int ow = 0; ow < blockDim.x; ow++) {
            int out_w = out_w_start + ow;
            if (out_w >= out_width) continue;
            
            float max_val = -__FLT_MAX__;
            for (int kh = 0; kh < kernel_size; kh++) {
                for (int kw = 0; kw < kernel_size; kw++) {
                    int in_h = oh + kh * dilation;
                    int in_w = ow + kw * dilation;
                    float val = shared_mem[in_h * tile_w + in_w];
                    if (val > max_val) max_val = val;
                }
            }
            output[((b * channels + c) * out_height + out_h) * out_width + out_w] = static_cast<T>(max_val);
        }
    }
}

// Optimized version: each thread block handles multiple output elements
template<typename T>
__global__ void max_pool2d_optimized(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int height,
    int width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int out_h = blockIdx.y * blockDim.y + threadIdx.y;
    int out_w = blockIdx.x * blockDim.x + threadIdx.x;
    int c = blockIdx.z % channels;
    int b = blockIdx.z / channels;
    
    if (out_h >= out_height || out_w >= out_width || b >= batch_size) return;
    
    int in_h_start = out_h * stride - padding;
    int in_w_start = out_w * stride - padding;
    
    int effective_kernel = kernel_size + (kernel_size - 1) * (dilation - 1);
    
    T max_val;
    bool first = true;
    
    #pragma unroll
    for (int kh = 0; kh < kernel_size; kh++) {
        #pragma unroll
        for (int kw = 0; kw < kernel_size; kw++) {
            int in_h = in_h_start + kh * dilation;
            int in_w = in_w_start + kw * dilation;
            
            if (in_h >= 0 && in_h < height && in_w >= 0 && in_w < width) {
                T val = input[((b * channels + c) * height + in_h) * width + in_w];
                if (first || val > max_val) {
                    max_val = val;
                    first = false;
                }
            }
        }
    }
    
    if (!first) {
        output[((b * channels + c) * out_height + out_h) * out_width + out_w] = max_val;
    }
}

// Highly optimized version using 2D block and better memory access
template<typename T>
__global__ void max_pool2d_tiled(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int channels,
    int height,
    int width,
    int out_height,
    int out_width,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    // Block handles a tile of output
    const int TILE_H = 16;
    const int TILE_W = 16;
    
    int tile_h_idx = blockIdx.y;
    int tile_w_idx = blockIdx.x;
    int bc = blockIdx.z;
    int c = bc % channels;
    int b = bc / channels;
    
    if (b >= batch_size) return;
    
    int out_h_base = tile_h_idx * TILE_H;
    int out_w_base = tile_w_idx * TILE_W;
    
    // Shared memory for input tile
    extern __shared__ float sdata[];
    float (*shared_tile)[TILE_W + 32] = (float (*)[TILE_W + 32])sdata;
    
    int effective_kernel = kernel_size + (kernel_size - 1) * (dilation - 1);
    int load_h = TILE_H + effective_kernel;
    int load_w = TILE_W + effective_kernel;
    
    // Load input tile
    int ty = threadIdx.y;
    int tx = threadIdx.x;
    
    for (int lh = ty; lh < load_h; lh += TILE_H) {
        int in_h = out_h_base + lh - padding;
        for (int lw = tx; lw < load_w; lw += TILE_W) {
            int in_w = out_w_base + lw - padding;
            float val = -__FLT_MAX__;
            if (in_h >= 0 && in_h < height && in_w >= 0 && in_w < width) {
                val = static_cast<float>(input[((b * channels + c) * height + in_h) * width + in_w]);
            }
            shared_tile[lh][lw] = val;
        }
    }
    __syncthreads();
    
    // Compute max pooling for this tile
    for (int oh = ty; oh < TILE_H; oh += TILE_H) {
        int out_h = out_h_base + oh;
        if (out_h >= out_height) continue;
        
        for (int ow = tx; ow < TILE_W; ow += TILE_W) {
            int out_w = out_w_base + ow;
            if (out_w >= out_width) continue;
            
            float max_val = -__FLT_MAX__;
            for (int kh = 0; kh < kernel_size; kh++) {
                for (int kw = 0; kw < kernel_size; kw++) {
                    int sh = oh + kh * dilation;
                    int sw = ow + kw * dilation;
                    float val = shared_tile[sh][sw];
                    if (val > max_val) max_val = val;
                }
            }
            output[((b * channels + c) * out_height + out_h) * out_width + out_w] = static_cast<T>(max_val);
        }
    }
}

torch::Tensor max_pool2d_cuda(torch::Tensor x, int kernel_size, int stride, int padding, int dilation) {
    int batch_size = x.size(0);
    int channels = x.size(1);
    int height = x.size(2);
    int width = x.size(3);
    
    int out_height = (height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int out_width = (width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, x.options());
    
    dim3 block(16, 16);
    dim3 grid((out_width + 15) / 16, (out_height + 15) / 16, batch_size * channels);
    
    int shared_mem_size = (16 + 32) * (16 + 32) * sizeof(float);
    
    if (x.dtype() == torch::kFloat32) {
        max_pool2d_tiled<float><<<grid, block, shared_mem_size>>>(
            x.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size, channels, height, width, out_height, out_width,
            kernel_size, stride, padding, dilation
        );
    } else if (x.dtype() == torch::kFloat16) {
        max_pool2d_tiled<at::Half><<<grid, block, shared_mem_size>>>(
            x.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            batch_size, channels, height, width, out_height, out_width,
            kernel_size, stride, padding, dilation
        );
    } else if (x.dtype() == torch::kBFloat16) {
        max_pool2d_tiled<at::BFloat16><<<grid, block, shared_mem_size>>>(
            x.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            batch_size, channels, height, width, out_height, out_width,
            kernel_size, stride, padding, dilation
        );
    }
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = """
torch::Tensor max_pool2d_cuda(torch::Tensor x, int kernel_size, int stride, int padding, int dilation);
"""

ext = load_inline(
    name='maxpool_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['max_pool2d_cuda'],
    extra_cuda_cflags=['-arch=sm_90', '-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs Max Pooling 2D.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.
        """
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Max Pooling 2D to the input tensor.
        """
        return ext.max_pool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)

batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
