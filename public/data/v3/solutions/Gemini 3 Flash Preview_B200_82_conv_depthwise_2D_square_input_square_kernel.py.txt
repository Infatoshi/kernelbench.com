
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename T>
__global__ void depthwise_conv2d_optimized_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    int batch_size, int channels, int in_h, int in_w,
    int out_h, int out_w,
    int kernel_size, int stride, int padding,
    bool has_bias) 
{
    // Each thread handles one output element
    int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_out_elements = batch_size * channels * out_h * out_w;

    if (out_idx < total_out_elements) {
        int temp_idx = out_idx;
        int w_out = temp_idx % out_w;
        temp_idx /= out_w;
        int h_out = temp_idx % out_h;
        temp_idx /= out_h;
        int c = temp_idx % channels;
        int b = temp_idx / channels;

        float sum = 0.0f;
        int h_in_start = h_out * stride - padding;
        int w_in_start = w_out * stride - padding;

        const T* weight_start = weight + c * kernel_size * kernel_size;

        #pragma unroll
        for (int kh = 0; kh < 3; ++kh) { 
            // We'll keep it general but unroll for small K
            int h_in = h_in_start + kh;
            if (h_in >= 0 && h_in < in_h) {
                const T* input_row = input + ((b * channels + c) * in_h + h_in) * in_w;
                for (int kw = 0; kw < 3; ++kw) {
                    int w_in = w_in_start + kw;
                    if (w_in >= 0 && w_in < in_w) {
                        sum += (float)input_row[w_in] * (float)weight_start[kh * kernel_size + kw];
                    }
                }
            }
        }
        
        // Handle generic kernel size if not 3x3
        if (kernel_size != 3) {
            // This part wouldn't be reached in 3x3 case if we hardcoded above, 
            // but for correctness we should handle it. 
            // Re-implementing logic for K!=3.
            sum = 0.0f; 
            for (int kh = 0; kh < kernel_size; ++kh) {
                int h_in = h_in_start + kh;
                if (h_in >= 0 && h_in < in_h) {
                    const T* input_row = input + ((b * channels + c) * in_h + h_in) * in_w;
                    for (int kw = 0; kw < kernel_size; ++kw) {
                        int w_in = w_in_start + kw;
                        if (w_in >= 0 && w_in < in_w) {
                            sum += (float)input_row[w_in] * (float)weight_start[kh * kernel_size + kw];
                        }
                    }
                }
            }
        }

        if (has_bias) {
            sum += (float)bias[c];
        }

        output[out_idx] = (T)sum;
    }
}

// Improved version for Blackwell
template <typename T>
__global__ void depthwise_conv2d_fast(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    int batch_size, int channels, int in_h, int in_w,
    int out_h, int out_w,
    int kernel_size, int stride, int padding,
    bool has_bias) 
{
    // Use tiles or other strategies if needed. 
    // Given the task is lvl 1, a simple thread-per-output with good memory access might be enough.
    int w_out = blockIdx.x * blockDim.x + threadIdx.x;
    int h_out = blockIdx.y * blockDim.y + threadIdx.y;
    int bc = blockIdx.z; // batch * channels
    
    if (w_out < out_w && h_out < out_h) {
        int b = bc / channels;
        int c = bc % channels;
        
        float sum = 0.0f;
        int h_in_base = h_out * stride - padding;
        int w_in_base = w_out * stride - padding;
        
        const T* weight_ptr = weight + c * (kernel_size * kernel_size);
        const T* input_ptr = input + (bc * in_h * in_w);
        
        for (int kh = 0; kh < kernel_size; ++kh) {
            int h_in = h_in_base + kh;
            if (h_in >= 0 && h_in < in_h) {
                for (int kw = 0; kw < kernel_size; ++kw) {
                    int w_in = w_in_base + kw;
                    if (w_in >= 0 && w_in < in_w) {
                        sum += (float)input_ptr[h_in * in_w + w_in] * (float)weight_ptr[kh * kernel_size + kw];
                    }
                }
            }
        }
        
        if (has_bias) sum += (float)bias[c];
        output[((b * channels + c) * out_h + h_out) * out_w + w_out] = (T)sum;
    }
}


torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding) 
{
    const int batch_size = x.size(0);
    const int channels = x.size(1);
    const int in_h = x.size(2);
    const int in_w = x.size(3);
    const int kernel_size = weight.size(2);
    const int out_h = (in_h + 2 * padding - kernel_size) / stride + 1;
    const int out_w = (in_w + 2 * padding - kernel_size) / stride + 1;

    auto output = torch::empty({batch_size, channels, out_h, out_w}, x.options());

    bool has_bias = bias.has_value();

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "depthwise_conv2d_cuda", ([&] {
        const scalar_t* bias_ptr = has_bias ? bias.value().data_ptr<scalar_t>() : nullptr;
        
        dim3 block(16, 16);
        dim3 grid((out_w + 15) / 16, (out_h + 15) / 16, batch_size * channels);
        
        depthwise_conv2d_fast<scalar_t><<<grid, block>>>(
            x.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias_ptr,
            output.data_ptr<scalar_t>(),
            batch_size, channels, in_h, in_w,
            out_h, out_w,
            kernel_size, stride, padding,
            has_bias);
    }));

    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d_cuda(
    torch::Tensor x,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    int stride, int padding);
'''

ext = load_inline(
    name='depthwise_conv_ext_opt',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3']
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.stride = stride
        self.padding = padding
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.depthwise_conv2d_cuda(x, self.conv2d.weight.squeeze(1), self.conv2d.bias, self.stride, self.padding)

def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]

def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
