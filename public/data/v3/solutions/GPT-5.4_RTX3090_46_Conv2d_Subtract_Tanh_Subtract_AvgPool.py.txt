import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_sub_tanh_sub_avgpool2x2_kernel(
    x_ptr, y_ptr,
    N, C, H, W,
    subtract1, subtract2,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_yn, stride_yc, stride_yh, stride_yw,
    BLOCK_W: tl.constexpr,
):
    pid = tl.program_id(0)
    W_OUT = W // 2
    H_OUT = H // 2
    total = N * C * H_OUT

    row = pid
    if row >= total:
        return

    nc = row // H_OUT
    oh = row % H_OUT
    n = nc // C
    c = nc % C

    ow = tl.arange(0, BLOCK_W)
    mask = ow < W_OUT

    ih0 = oh * 2
    iw0 = ow * 2

    base = x_ptr + n * stride_xn + c * stride_xc + ih0 * stride_xh

    x00 = tl.load(base + iw0 * stride_xw, mask=mask, other=0.0)
    x01 = tl.load(base + (iw0 + 1) * stride_xw, mask=mask, other=0.0)
    x10 = tl.load(base + stride_xh + iw0 * stride_xw, mask=mask, other=0.0)
    x11 = tl.load(base + stride_xh + (iw0 + 1) * stride_xw, mask=mask, other=0.0)

    s1 = tl.full([BLOCK_W], subtract1, tl.float32)
    s2 = tl.full([BLOCK_W], subtract2, tl.float32)

    y00 = tl.extra.cuda.libdevice.tanh(x00 - s1) - s2
    y01 = tl.extra.cuda.libdevice.tanh(x01 - s1) - s2
    y10 = tl.extra.cuda.libdevice.tanh(x10 - s1) - s2
    y11 = tl.extra.cuda.libdevice.tanh(x11 - s1) - s2

    out = (y00 + y01 + y10 + y11) * 0.25
    out_ptr = y_ptr + n * stride_yn + c * stride_yc + oh * stride_yh + ow * stride_yw
    tl.store(out_ptr, out, mask=mask)


class FusedPostOpAvgPool2x2(nn.Module):
    def __init__(self, subtract1_value, subtract2_value):
        super().__init__()
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)

    def forward(self, x):
        assert x.is_cuda
        N, C, H, W = x.shape
        assert H % 2 == 0 and W % 2 == 0
        y = torch.empty((N, C, H // 2, W // 2), device=x.device, dtype=x.dtype)
        grid = (N * C * (H // 2),)
        fused_sub_tanh_sub_avgpool2x2_kernel[grid](
            x, y,
            N, C, H, W,
            self.subtract1_value, self.subtract2_value,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            y.stride(0), y.stride(1), y.stride(2), y.stride(3),
            BLOCK_W=128,
        )
        return y


class Model(nn.Module):
    """
    Model that performs a convolution, subtraction, tanh activation, subtraction and average pooling.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        if kernel_size_pool == 2:
            self.fused_pool = FusedPostOpAvgPool2x2(subtract1_value, subtract2_value)
        else:
            self.avgpool = nn.AvgPool2d(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        if self.kernel_size_pool == 2:
            return self.fused_pool(x)
        x = x - self.subtract1_value
        x = torch.tanh(x)
        x = x - self.subtract2_value
        x = self.avgpool(x)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
