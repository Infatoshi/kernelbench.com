import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def max_pool2d_kernel(
    input_ptr, output_ptr,
    N, C, H, W, OH, OW,
    kH, kW, stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w,
    BLOCK_SIZE: tl.constexpr,
    DTYPE_ENUM: tl.constexpr,
):
    """
    Triton kernel for 2D Max Pooling.
    Each block processes BLOCK_SIZE output elements (flattened NCHW).
    """
    pid = tl.program_id(0)
    num_outputs = N * C * OH * OW
    
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < num_outputs
    
    # Decompose linear index into (n, c, oh, ow)
    ow = offs % OW
    tmp = offs // OW
    oh = tmp % OH
    nc = tmp // OH
    c = nc % C
    n = nc // C
    
    # Initialize max accumulator
    max_val = tl.full([BLOCK_SIZE], float('-inf'), dtype=tl.float32)
    
    # Loop over kernel dimensions
    for kh in range(kH):
        for kw in range(kW):
            # Compute input spatial coordinates
            ih = oh * stride_h + kh * dilation_h - pad_h
            iw = ow * stride_w + kw * dilation_w - pad_w
            
            # Check bounds
            in_bounds = (ih >= 0) & (ih < H) & (iw >= 0) & (iw < W)
            
            # Compute input memory offset
            input_idx = ((n * C + c) * H + ih) * W + iw
            
            # Load value with appropriate dtype handling
            if DTYPE_ENUM == 0:  # fp32
                val = tl.load(input_ptr + input_idx, mask=(mask & in_bounds), other=float('-inf'))
                val_f32 = val
            elif DTYPE_ENUM == 1:  # fp16
                val = tl.load(input_ptr + input_idx, mask=(mask & in_bounds), other=float('-inf'))
                val_f32 = val.to(tl.float32)
            else:  # bf16 (2)
                val = tl.load(input_ptr + input_idx, mask=(mask & in_bounds), other=float('-inf'))
                val_f32 = val.to(tl.float32)
            
            # Update maximum
            max_val = tl.maximum(max_val, val_f32)
    
    # Compute output memory offset
    out_idx = ((n * C + c) * OH + oh) * OW + ow
    
    # Store with appropriate dtype
    if DTYPE_ENUM == 0:  # fp32
        tl.store(output_ptr + out_idx, max_val, mask=mask)
    elif DTYPE_ENUM == 1:  # fp16
        tl.store(output_ptr + out_idx, max_val.to(tl.float16), mask=mask)
    else:  # bf16
        tl.store(output_ptr + out_idx, max_val.to(tl.bfloat16), mask=mask)


class MaxPool2dTriton(nn.Module):
    def __init__(self, kernel_size, stride, padding, dilation):
        super().__init__()
        self.kH = kernel_size
        self.kW = kernel_size
        self.stride_h = stride
        self.stride_w = stride
        self.pad_h = padding
        self.pad_w = padding
        self.dilation_h = dilation
        self.dilation_w = dilation
        
    def forward(self, x):
        N, C, H, W = x.shape
        
        # Calculate output dimensions
        OH = (H + 2 * self.pad_h - self.dilation_h * (self.kH - 1) - 1) // self.stride_h + 1
        OW = (W + 2 * self.pad_w - self.dilation_w * (self.kW - 1) - 1) // self.stride_w + 1
        
        # Determine dtype enum
        if x.dtype == torch.float32:
            dtype_enum = 0
        elif x.dtype == torch.float16:
            dtype_enum = 1
        elif x.dtype == torch.bfloat16:
            dtype_enum = 2
        else:
            raise ValueError(f"Unsupported dtype {x.dtype}")
        
        output = torch.empty(N, C, OH, OW, device=x.device, dtype=x.dtype)
        
        BLOCK_SIZE = 256
        num_outputs = N * C * OH * OW
        grid = ((num_outputs + BLOCK_SIZE - 1) // BLOCK_SIZE,)
        
        max_pool2d_kernel[grid](
            x, output,
            N, C, H, W, OH, OW,
            self.kH, self.kW, self.stride_h, self.stride_w,
            self.pad_h, self.pad_w, self.dilation_h, self.dilation_w,
            BLOCK_SIZE=BLOCK_SIZE,
            DTYPE_ENUM=dtype_enum
        )
        
        return output


class Model(nn.Module):
    """
    Model that performs Max Pooling 2D using optimized Triton kernel.
    """
    def __init__(self, kernel_size: int = 2, stride: int = 2, padding: int = 1, dilation: int = 3):
        super(Model, self).__init__()
        self.maxpool = MaxPool2dTriton(kernel_size, stride, padding, dilation)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.maxpool(x)
