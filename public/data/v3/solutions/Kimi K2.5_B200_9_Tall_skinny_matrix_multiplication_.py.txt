import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <c10/util/BFloat16.h>

using c10::BFloat16;

// Optimized tall-skinny GEMM for B200 (Blackwell SM100)
// A: [M, K], B: [K, N], C: [M, N]
// Optimized for small K using shared memory tiling and vectorized access

template <typename T>
__device__ __forceinline__ float to_float(T x) {
    return static_cast<float>(x);
}

template <>
__device__ __forceinline__ float to_float<half>(half x) {
    return __half2float(x);
}

template <>
__device__ __forceinline__ float to_float<BFloat16>(BFloat16 x) {
    return static_cast<float>(x);
}

template <typename T>
__device__ __forceinline__ T from_float(float x) {
    return static_cast<T>(x);
}

template <>
__device__ __forceinline__ half from_float<half>(float x) {
    return __float2half(x);
}

template <>
__device__ __forceinline__ BFloat16 from_float<BFloat16>(float x) {
    return BFloat16(x);
}

// Main kernel optimized for K <= 32
template <typename T>
__global__ void gemm_tall_skinny_kernel(
    const T* __restrict__ A,
    const T* __restrict__ B,
    T* __restrict__ C,
    int M, int N, int K) {
    
    const int BM = 64;
    const int BN = 64;
    
    int bx = blockIdx.x;
    int by = blockIdx.y;
    int tid = threadIdx.x;
    
    int m_start = by * BM;
    int n_start = bx * BN;
    
    // Shared memory for tiles
    __shared__ T sA[64 * 32];  // Max K=32 for safety
    __shared__ T sB[32 * 64];
    
    // Load A tile [BM, K]
    for (int i = tid; i < BM * K; i += blockDim.x) {
        int row = i / K;
        int col = i % K;
        int m_idx = m_start + row;
        if (m_idx < M) {
            sA[row * K + col] = A[m_idx * K + col];
        } else {
            sA[row * K + col] = T(0);
        }
    }
    
    // Load B tile [K, BN]
    for (int i = tid; i < K * BN; i += blockDim.x) {
        int k_idx = i / BN;
        int n_idx = i % BN;
        int n_global = n_start + n_idx;
        if (n_global < N) {
            sB[k_idx * BN + n_idx] = B[k_idx * N + n_global];
        } else {
            sB[k_idx * BN + n_idx] = T(0);
        }
    }
    
    __syncthreads();
    
    // Compute output elements
    // Each thread computes multiple elements to amortize syncthreads cost
    int total_elems = BM * BN;
    for (int idx = tid; idx < total_elems; idx += blockDim.x) {
        int i = idx / BN;
        int j = idx % BN;
        
        float sum = 0.0f;
        #pragma unroll
        for (int k = 0; k < K; k++) {
            sum += to_float(sA[i * K + k]) * to_float(sB[k * BN + j]);
        }
        
        int m_global = m_start + i;
        int n_global = n_start + j;
        if (m_global < M && n_global < N) {
            C[m_global * N + n_global] = from_float<T>(sum);
        }
    }
}

torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    TORCH_CHECK(K == B.size(0), "Inner dimensions must match");
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "Inputs must be CUDA tensors");
    
    auto C = torch::empty({M, N}, A.options());
    
    const int BM = 64;
    const int BN = 64;
    
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);
    dim3 block(256);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, A.scalar_type(), "gemm_forward", [&] {
        gemm_tall_skinny_kernel<scalar_t><<<grid, block>>>(
            A.data_ptr<scalar_t>(),
            B.data_ptr<scalar_t>(),
            C.data_ptr<scalar_t>(),
            M, N, K
        );
    });
    
    return C;
}
'''

ext = load_inline(
    name='tall_skinny_gemm',
    cpp_sources='torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemm_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A, B):
        # Ensure inputs are on CUDA and contiguous for optimal access
        if not A.is_cuda:
            A = A.cuda()
        if not B.is_cuda:
            B = B.cuda()
        A = A.contiguous()
        B = B.contiguous()
        return ext.gemm_forward(A, B)

def get_inputs():
    M = 16384
    N = 16
    A = torch.randn(M, N)
    B = torch.randn(N, M)
    return [A, B]

def get_init_inputs():
    return []
