import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_kernel(
    const float* __restrict__ input,
    const float* __restrict__ conv_w,
    const float* __restrict__ conv_b,
    const float* __restrict__ gn_w,
    const float* __restrict__ gn_b,
    const float* __restrict__ extra_scale,
    float* __restrict__ output,
    int N, int C_IN, int C_OUT, int IN_H, int IN_W, 
    int OUT_H, int OUT_W, int POOL_H, int POOL_W, int K, 
    int NUM_GROUPS, int CHAN_PER_GROUP,
    float eps, float clamp_min, float clamp_max
) {
    int n = blockIdx.x;
    int g = blockIdx.y;
    int tid = threadIdx.x;
    int blockSize = blockDim.x;
    
    int c0 = g * CHAN_PER_GROUP;
    
    __shared__ float s_sum[2][256];
    __shared__ float s_sum_sq[2][256];
    __shared__ float s_mean[2];
    __shared__ float s_inv_std[2];
    
    float lane_sum[2] = {0.0f, 0.0f};
    float lane_sum_sq[2] = {0.0f, 0.0f};
    
    int total_spatial = OUT_H * OUT_W;
    
    // First pass: compute statistics
    for (int idx = tid; idx < total_spatial; idx += blockSize) {
        int h = idx / OUT_W;
        int w = idx % OUT_W;
        
        for (int c_local = 0; c_local < 2; ++c_local) {
            int c_out = c0 + c_local;
            
            float acc = 0.0f;
            for (int c_in = 0; c_in < C_IN; ++c_in) {
                for (int kh = 0; kh < K; ++kh) {
                    for (int kw = 0; kw < K; ++kw) {
                        int h_in = h + kh;
                        int w_in = w + kw;
                        int in_idx = ((n * C_IN + c_in) * IN_H + h_in) * IN_W + w_in;
                        int w_idx = ((c_out * C_IN + c_in) * K + kh) * K + kw;
                        acc += input[in_idx] * conv_w[w_idx];
                    }
                }
            }
            acc += conv_b[c_out];
            lane_sum[c_local] += acc;
            lane_sum_sq[c_local] += acc * acc;
        }
    }
    
    for (int c_local = 0; c_local < 2; ++c_local) {
        s_sum[c_local][tid] = lane_sum[c_local];
        s_sum_sq[c_local][tid] = lane_sum_sq[c_local];
    }
    __syncthreads();
    
    for (int stride = blockSize / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            for (int c_local = 0; c_local < 2; ++c_local) {
                s_sum[c_local][tid] += s_sum[c_local][tid + stride];
                s_sum_sq[c_local][tid] += s_sum_sq[c_local][tid + stride];
            }
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        for (int c_local = 0; c_local < 2; ++c_local) {
            float mean = s_sum[c_local][0] / (float)total_spatial;
            float mean_sq = s_sum_sq[c_local][0] / (float)total_spatial;
            float var = mean_sq - mean * mean;
            s_mean[c_local] = mean;
            s_inv_std[c_local] = 1.0f / sqrtf(var + eps);
        }
    }
    __syncthreads();
    
    // Second pass: compute output
    int total_out_spatial = POOL_H * POOL_W * CHAN_PER_GROUP;
    
    for (int idx = tid; idx < total_out_spatial; idx += blockSize) {
        int c_local = idx / (POOL_H * POOL_W);
        int spatial = idx % (POOL_H * POOL_W);
        int h_p = spatial / POOL_W;
        int w_p = spatial % POOL_W;
        int c_out = c0 + c_local;
        
        int h = h_p * 2;
        int w = w_p * 2;
        
        float mean = s_mean[c_local];
        float inv_std = s_inv_std[c_local];
        float gn_weight = gn_w[c_out];
        float gn_bias = gn_b[c_out];
        float scale = extra_scale[c_out];
        
        float max_val = -1e9f;
        
        for (int ph = 0; ph < 2; ++ph) {
            for (int pw = 0; pw < 2; ++pw) {
                int hh = h + ph;
                int ww = w + pw;
                
                float acc = 0.0f;
                for (int c_in = 0; c_in < C_IN; ++c_in) {
                    for (int kh = 0; kh < K; ++kh) {
                        for (int kw = 0; kw < K; ++kw) {
                            int h_in = hh + kh;
                            int w_in = ww + kw;
                            int in_idx = ((n * C_IN + c_in) * IN_H + h_in) * IN_W + w_in;
                            int w_idx = ((c_out * C_IN + c_in) * K + kh) * K + kw;
                            acc += input[in_idx] * conv_w[w_idx];
                        }
                    }
                }
                acc += conv_b[c_out];
                
                acc = (acc - mean) * inv_std;
                acc = acc * gn_weight + gn_bias;
                acc *= scale;
                
                if (acc > max_val) max_val = acc;
            }
        }
        
        max_val = fminf(fmaxf(max_val, clamp_min), clamp_max);
        
        int out_idx = ((n * C_OUT + c_out) * POOL_H + h_p) * POOL_W + w_p;
        output[out_idx] = max_val;
    }
}

torch::Tensor fused_forward_cuda(
    torch::Tensor input,
    torch::Tensor conv_w,
    torch::Tensor conv_b,
    torch::Tensor gn_w,
    torch::Tensor gn_b,
    torch::Tensor extra_scale,
    float eps, float clamp_min, float clamp_max
) {
    const int N = 16;
    const int C_IN = 3;
    const int C_OUT = 16;
    const int IN_H = 256;
    const int IN_W = 256;
    const int K = 3;
    const int OUT_H = 254;
    const int OUT_W = 254;
    const int POOL_H = 127;
    const int POOL_W = 127;
    const int NUM_GROUPS = 8;
    const int CHAN_PER_GROUP = 2;
    
    auto output = torch::empty({N, C_OUT, POOL_H, POOL_W}, input.options());
    
    dim3 grid(N, NUM_GROUPS);
    dim3 block(256);
    
    fused_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        conv_w.data_ptr<float>(),
        conv_b.data_ptr<float>(),
        gn_w.data_ptr<float>(),
        gn_b.data_ptr<float>(),
        extra_scale.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C_IN, C_OUT, IN_H, IN_W, OUT_H, OUT_W, POOL_H, POOL_W, K, NUM_GROUPS, CHAN_PER_GROUP,
        eps, clamp_min, clamp_max
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_forward_cuda(
    torch::Tensor input,
    torch::Tensor conv_w,
    torch::Tensor conv_b,
    torch::Tensor gn_w,
    torch::Tensor gn_b,
    torch::Tensor extra_scale,
    float eps, float clamp_min, float clamp_max);
'''

ext = load_inline(
    name='fused_convgn',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '-std=c++17'],
    verbose=False
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.group_norm = nn.GroupNorm(num_groups, out_channels, affine=True)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        
    def forward(self, x):
        x = x.contiguous()
        scale_flat = self.scale.view(-1)
        return ext.fused_forward_cuda(
            x,
            self.conv.weight,
            self.conv.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            scale_flat,
            self.group_norm.eps,
            float(self.clamp_min),
            float(self.clamp_max)
        )


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]


def get_init_inputs():
    return [3, 16, 3, 8, (16, 1, 1), 2, 0.0, 1.0]
