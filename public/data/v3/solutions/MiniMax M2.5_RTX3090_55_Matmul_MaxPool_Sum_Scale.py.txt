import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fused kernel: Linear -> MaxPool1d(kernel_size=2) -> Sum -> Scale
// Each block handles one batch element
// Each thread computes multiple output elements of the GEMM
// Then we do max pooling, warp reduction for sum, and scale

__device__ __forceinline__ float warpReduceSum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

__global__ void fused_kernel(
    const float* __restrict__ input,   // (batch_size, in_features)
    const float* __restrict__ weight,  // (out_features, in_features)
    float* __restrict__ output,        // (batch_size)
    const float scale_factor,
    const int batch_size,
    const int in_features,
    const int out_features
) {
    // Shared memory for partial results
    extern __shared__ float smem[];
    
    const int tid = threadIdx.x;
    const int bid = blockIdx.x;
    
    if (bid >= batch_size) return;
    
    const int n_threads = blockDim.x;
    const int n_outputs_per_thread = (out_features + n_threads - 1) / n_threads;
    
    // Pointer to input for this batch
    const float* x_batch = input + bid * in_features;
    // Pointer to output for this batch
    float* smem_partial = smem;
    
    // Phase 1: GEMM - each thread computes multiple output elements
    #pragma unroll 4
    for (int i = 0; i < n_outputs_per_thread; i++) {
        int col = tid + i * n_threads;
        if (col >= out_features) break;
        
        // Compute dot product: y[col] = sum(x[j] * weight[col,j])
        const float* w_row = weight + col * in_features;
        
        float sum = 0.0f;
        
        // Vectorized load and compute (4 elements at a time)
        int j = 0;
        for (; j + 3 < in_features; j += 4) {
            float4 x_vec = reinterpret_cast<const float4*>(x_batch)[j >> 2];
            float4 w_vec = reinterpret_cast<const float4*>(w_row)[j >> 2];
            sum = fma(x_vec.x, w_vec.x, sum);
            sum = fma(x_vec.y, w_vec.y, sum);
            sum = fma(x_vec.z, w_vec.z, sum);
            sum = fma(x_vec.w, w_vec.w, sum);
        }
        // Handle remaining
        for (; j < in_features; j++) {
            sum += x_batch[j] * w_row[j];
        }
        
        smem_partial[col] = sum;
    }
    __syncthreads();
    
    // Phase 2: Max pooling with kernel_size=2, then sum
    // Each thread computes sum of pooled values for its assigned outputs
    
    // Thread 0 gathers all results via warp reduction
    // Actually, let's do a tree reduction across all threads
    
    // First, each thread sums its portion of the pooled data
    float local_sum = 0.0f;
    
    for (int i = 0; i < n_outputs_per_thread; i++) {
        int col = tid + i * n_threads;
        if (col >= out_features) break;
        
        // Get the value
        float val = smem_partial[col];
        
        // Max pooling with neighbor (kernel_size=2)
        // For even indices, take max with next; for odd, contribution is already included
        if (col % 2 == 0 && col + 1 < out_features) {
            float neighbor = smem_partial[col + 1];
            val = fmaxf(val, neighbor);
        }
        
        // Add to local sum
        local_sum += val;
    }
    __syncthreads();
    
    // Warp reduction for the sum
    float warp_sum = warpReduceSum(local_sum);
    
    // Only one thread writes the result
    if (tid == 0) {
        // The warp reduction gives us sum across warps if we do it right
        // But we need to handle multiple warps
        // Actually let's use a simple approach: store to smem and do block reduction
        
        // Store to shared memory
        smem[0] = warp_sum;
    }
    __syncthreads();
    
    // Block-level reduction (across warps)
    if (tid < 32) {
        float val = (tid == 0) ? smem[0] : 0.0f;
        val = warpReduceSum(val);
        if (tid == 0) {
            output[bid] = val * scale_factor;
        }
    }
}

torch::Tensor fused_op(
    torch::Tensor input,
    torch::Tensor weight,
    float scale_factor
) {
    const int batch_size = input.size(0);
    const int in_features = input.size(1);
    const int out_features = weight.size(0);
    
    auto output = torch::empty({batch_size}, input.options());
    
    const int n_threads = 256;
    const int n_blocks = batch_size;
    const int smem_size = out_features * sizeof(float);
    
    fused_kernel<<<n_blocks, n_threads, smem_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        output.data_ptr<float>(),
        scale_factor,
        batch_size,
        in_features,
        out_features
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_matmul_maxpool_sum_scale',
    cpp_sources='torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, float scale_factor);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        # Initialize weight like nn.Linear
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.zeros(out_features))
    
    def forward(self, x):
        # Use fused kernel
        out = ext.fused_op(x, self.weight, self.scale_factor)
        return out

def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features, device='cuda')]

def get_init_inputs():
    in_features = 4096
    out_features = 4096
    kernel_size = 2
    scale_factor = 0.5
    return [in_features, out_features, kernel_size, scale_factor]
