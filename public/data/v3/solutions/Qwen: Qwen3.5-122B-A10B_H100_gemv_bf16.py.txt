"""
Optimized BF16 GEMV for H100 using raw CUDA with BF16 support.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>

// Use __nv_bfloat16 for CUDA BF16 operations
using bf16 = __nv_bfloat16;

// Optimized GEMV kernel for BF16 on H100
// Uses shared memory for input broadcast and computes dot products efficiently

__device__ __forceinline__ float bf16_to_float(bf16 x) {
    return __bfloat162float(x);
}

__device__ __forceinline__ bf16 float_to_bf16(float x) {
    return __float2bfloat16(x);
}

// Warp-level optimized GEMV - each warp computes 32 consecutive output elements
__global__ void kernel_gemv_bf16_warp(
    bf16* output,
    const bf16* input,
    const bf16* weight,
    int m, int n, int k
) {
    constexpr int kWarpSize = 32;
    
    int row = blockIdx.x;
    if (row >= m) return;
    
    int tid = threadIdx.x;
    int warp_id = tid / kWarpSize;
    int lane_id = tid % kWarpSize;
    
    // Shared memory for input
    extern __shared__ bf16 s_input[];
    
    // Load input
    for (int i = tid; i < k; i += blockDim.x) {
        s_input[i] = input[row * k + i];
    }
    __syncthreads();
    
    // Each warp computes 32 consecutive output elements
    int col_start = warp_id * 32;
    int col = col_start + lane_id;
    
    if (col < n) {
        float sum = 0.0f;
        // Unrolled loop for better performance
        int k_unroll = 16;
        for (int i = 0; i < k; i += k_unroll) {
            #pragma unroll
            for (int j = 0; j < k_unroll && (i + j) < k; j++) {
                sum += bf16_to_float(weight[col * k + i + j]) * 
                       bf16_to_float(s_input[i + j]);
            }
        }
        output[row * n + col] = float_to_bf16(sum);
    }
}

// Vectorized version for better memory throughput
__global__ void kernel_gemv_bf16_vectorized(
    bf16* output,
    const bf16* input,
    const bf16* weight,
    int m, int n, int k
) {
    constexpr int kWarpSize = 32;
    constexpr int kVecSize = 4; // Load 4 BF16 at a time
    
    int row = blockIdx.x;
    if (row >= m) return;
    
    int tid = threadIdx.x;
    int warp_id = tid / kWarpSize;
    int lane_id = tid % kWarpSize;
    
    // Shared memory for input as float for faster compute
    extern __shared__ float s_input_f[];
    
    // Load input vectorized and convert to float
    for (int i = tid * kVecSize; i < k; i += blockDim.x * kVecSize) {
        #pragma unroll
        for (int j = 0; j < kVecSize && (i + j) < k; j++) {
            s_input_f[i + j] = bf16_to_float(input[row * k + i + j]);
        }
    }
    __syncthreads();
    
    // Each warp computes 32 consecutive output elements
    int col_start = warp_id * 32;
    int col = col_start + lane_id;
    
    if (col < n) {
        float sum = 0.0f;
        // Process weight row in chunks for better cache usage
        int chunk_size = 32;
        for (int chunk = 0; chunk < k; chunk += chunk_size) {
            float partial = 0.0f;
            #pragma unroll
            for (int i = 0; i < chunk_size && (chunk + i) < k; i++) {
                partial += bf16_to_float(weight[col * k + chunk + i]) * s_input_f[chunk + i];
            }
            sum += partial;
        }
        output[row * n + col] = float_to_bf16(sum);
    }
}

torch::Tensor gemv_bf16(torch::Tensor x, torch::Tensor weight) {
    // x: [m, k], weight: [n, k]
    // output: [m, n]
    
    TORCH_CHECK(x.dim() == 2, "x must be 2D");
    TORCH_CHECK(weight.dim() == 2, "weight must be 2D");
    TORCH_CHECK(x.dtype() == torch::kBFloat16, "x must be BF16");
    TORCH_CHECK(weight.dtype() == torch::kBFloat16, "weight must be BF16");
    
    int m = x.size(0);
    int k = x.size(1);
    int n = weight.size(0);
    
    TORCH_CHECK(x.size(1) == weight.size(1), "Inner dimensions must match");
    
    auto output = torch::empty({m, n}, torch::TensorOptions().dtype(torch::kBFloat16).device(x.device()));
    
    // Choose kernel and configuration based on problem size
    int threads_per_block = 256;
    int smem_size = k * sizeof(float); // Use float in shared memory for faster compute
    
    // Use warp-level vectorized kernel
    kernel_gemv_bf16_vectorized<<<m, threads_per_block, smem_size, 0>>>(
        reinterpret_cast<bf16*>(output.data_ptr()),
        reinterpret_cast<const bf16*>(x.data_ptr()),
        reinterpret_cast<const bf16*>(weight.data_ptr()),
        m, n, k
    );
    
    return output;
}
'''

ext = load_inline(
    name='gemv_bf16_ext',
    cpp_sources='torch::Tensor gemv_bf16(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gemv_bf16'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-std=c++17', '-use_fast_math', '-arch=sm_90'],
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        # Store weight in transposed form [out_features, in_features] for efficient access
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf16 = x.to(dtype=torch.bfloat16)
        # Use our optimized GEMV kernel
        # x_bf16: [m, k], weight: [n, k]
        # We need x @ weight.t() which computes dot products of x rows with weight rows
        return ext.gemv_bf16(x_bf16, self.weight)

def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16)]

def get_init_inputs():
    return []
