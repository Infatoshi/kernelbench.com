import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Bloom effect: threshold -> avg_pool2d (5x5, stride=1, pad=2) -> additive blend -> clamp
# We fuse all operations into a single kernel using shared memory for the blur.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fused bloom kernel for float32
// Computes: bright = relu(x - threshold)
// blurred = avg_pool2d(bright, 5x5, stride=1, pad=2)  -> each output pixel = mean of 5x5 neighborhood
// output = clamp(x + intensity * blurred, 0, 10)
//
// We tile the output with TILE_W x TILE_H output pixels per block, loading a (TILE_W+4) x (TILE_H+4) patch
// of the bright image into shared memory.

#define TILE_W 32
#define TILE_H 8
#define PAD 2
#define KERNEL_SIZE 5
#define INV_KK (1.0f / 25.0f)

__global__ void bloom_fused_f32(
    const float* __restrict__ input,
    float* __restrict__ output,
    int N, int C, int H, int W,
    float threshold, float intensity
) {
    // Shared memory tile: (TILE_H + 2*PAD) x (TILE_W + 2*PAD)
    __shared__ float smem[(TILE_H + 2*PAD) * (TILE_W + 2*PAD)];

    const int sh = TILE_H + 2*PAD;
    const int sw = TILE_W + 2*PAD;

    int tx = threadIdx.x; // 0..TILE_W-1
    int ty = threadIdx.y; // 0..TILE_H-1
    int out_x = blockIdx.x * TILE_W + tx;
    int out_y = blockIdx.y * TILE_H + ty;
    int nc = blockIdx.z; // combined N*C index
    int n = nc / C;
    int c = nc % C;

    // Base pointer for this (n, c) channel
    const float* ch_in = input + (n * C + c) * H * W;
    float* ch_out = output + (n * C + c) * H * W;

    // Load (TILE_H + 2*PAD) x (TILE_W + 2*PAD) into shared memory
    // using all threads in the block (TILE_W * TILE_H threads)
    int total_smem = sh * sw;
    int tid = ty * TILE_W + tx;
    int num_threads = TILE_W * TILE_H;

    int base_y = blockIdx.y * TILE_H - PAD;
    int base_x = blockIdx.x * TILE_W - PAD;

    for (int i = tid; i < total_smem; i += num_threads) {
        int sy = i / sw;
        int sx = i % sw;
        int gy = base_y + sy;
        int gx = base_x + sx;
        float val;
        if (gy >= 0 && gy < H && gx >= 0 && gx < W) {
            val = ch_in[gy * W + gx];
        } else {
            val = 0.0f;
        }
        // Apply threshold relu
        val = val - threshold;
        smem[i] = val > 0.0f ? val : 0.0f;
    }
    __syncthreads();

    if (out_x < W && out_y < H) {
        // Sum 5x5 neighborhood in shared memory
        // Top-left of this output pixel in smem is (ty, tx) (since PAD=2)
        float sum = 0.0f;
        int smem_base_y = ty; // ty + 0 (the -PAD offset is absorbed: smem[ty+PAD-PAD..ty+PAD+PAD])
        int smem_base_x = tx;
        
        #pragma unroll
        for (int ky = 0; ky < KERNEL_SIZE; ky++) {
            #pragma unroll
            for (int kx = 0; kx < KERNEL_SIZE; kx++) {
                sum += smem[(smem_base_y + ky) * sw + (smem_base_x + kx)];
            }
        }
        float blurred = sum * INV_KK;
        float orig = ch_in[out_y * W + out_x];
        float result = orig + intensity * blurred;
        // clamp(result, 0, 10)
        result = result < 0.0f ? 0.0f : (result > 10.0f ? 10.0f : result);
        ch_out[out_y * W + out_x] = result;
    }
}

torch::Tensor bloom_forward(
    torch::Tensor image,
    double threshold,
    double intensity,
    int blur_kernel
) {
    TORCH_CHECK(blur_kernel == 5, "Only blur_kernel=5 is supported");
    TORCH_CHECK(image.dim() == 4, "Input must be 4D (N,C,H,W)");
    TORCH_CHECK(image.is_cuda(), "Input must be on CUDA");

    auto output = torch::empty_like(image);
    int N = image.size(0);
    int C = image.size(1);
    int H = image.size(2);
    int W = image.size(3);

    dim3 block(TILE_W, TILE_H);
    dim3 grid(
        (W + TILE_W - 1) / TILE_W,
        (H + TILE_H - 1) / TILE_H,
        N * C
    );

    float thresh_f = (float)threshold;
    float intens_f = (float)intensity;

    if (image.scalar_type() == torch::kFloat32) {
        bloom_fused_f32<<<grid, block>>>(
            image.data_ptr<float>(),
            output.data_ptr<float>(),
            N, C, H, W, thresh_f, intens_f
        );
    } else if (image.scalar_type() == torch::kFloat16) {
        bloom_fused_f16<<<grid, block>>>(
            (__half*)image.data_ptr(),
            (__half*)output.data_ptr(),
            N, C, H, W, thresh_f, intens_f
        );
    } else if (image.scalar_type() == torch::kBFloat16) {
        bloom_fused_bf16<<<grid, block>>>(
            (__nv_bfloat16*)image.data_ptr(),
            (__nv_bfloat16*)output.data_ptr(),
            N, C, H, W, thresh_f, intens_f
        );
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }

    return output;
}
'''

cpp_source = r'''
torch::Tensor bloom_forward(torch::Tensor image, double threshold, double intensity, int blur_kernel);
'''

ext = load_inline(
    name='bloom_fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bloom_forward'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
    verbose=False,
)


class Model(nn.Module):
    """Bloom effect approximation: threshold -> blur -> additive blend."""

    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        return ext.bloom_forward(image, self.threshold, self.intensity, self.blur_kernel)


def get_inputs():
    # HDR-like image input.
    return [torch.randn(2, 3, 1024, 1024) * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
