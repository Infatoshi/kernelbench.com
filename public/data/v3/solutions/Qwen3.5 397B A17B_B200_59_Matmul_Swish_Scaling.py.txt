import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _fused_matmul_swish_scale_kernel(
    a_ptr, b_ptr, bias_ptr,
    out_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_outm, stride_outn,
    scaling_factor,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    HAS_BIAS: tl.constexpr,
):
    """
    Compute: out = (a @ b.T + bias) * sigmoid(a @ b.T + bias) * scaling_factor
    
    a: [M, K] - input
    b: [N, K] - weight (not transposed in memory)
    output: [M, N]
    
    We compute C = A @ B.T, so C[m,n] = sum_k A[m,k] * B[n,k]
    """
    pid = tl.program_id(axis=0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n
    
    m_start = pid_m * BLOCK_M
    n_start = pid_n * BLOCK_N
    
    m_range = m_start + tl.arange(0, BLOCK_M)
    n_range = n_start + tl.arange(0, BLOCK_N)
    
    m_mask = m_range < M
    n_mask = n_range < N
    
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Matrix multiplication: C = A @ B.T
    # For each k, we load A[m,k] and B[n,k] and accumulate
    for k_start in range(0, K, BLOCK_K):
        k_range = k_start + tl.arange(0, BLOCK_K)
        k_mask = k_range < K
        
        # Load A: shape [BLOCK_M, BLOCK_K]
        # A[m,k] where m in m_range, k in k_range
        a_idx = m_range[:, None] * stride_am + k_range[None, :] * stride_ak
        a = tl.load(a_ptr + a_idx, mask=m_mask[:, None] & k_mask[None, :], other=0.0)
        
        # Load B: shape [BLOCK_K, BLOCK_N] - but B is stored as [N, K]
        # We need B[k,n] = B[n,k] in memory, so load B[n,k] and transpose
        # Actually, for B.T, we want B.T[k,n] = B[n,k]
        # So we load B[n_range, k_range] which gives [BLOCK_N, BLOCK_K]
        # Then we need to transpose to get [BLOCK_K, BLOCK_N] for tl.dot
        b_idx = n_range[:, None] * stride_bk + k_range[None, :] * stride_bn
        b = tl.load(b_ptr + b_idx, mask=n_mask[:, None] & k_mask[None, :], other=0.0)
        
        # b is [BLOCK_N, BLOCK_K], we need to transpose for dot product
        # tl.dot(a, b.T) where a is [M,K] and b.T is [K,N]
        # So we do tl.dot(a, trans(b)) 
        acc += tl.dot(a, tl.trans(b))
    
    # Add bias if present
    if HAS_BIAS:
        bias = tl.load(bias_ptr + n_range, mask=n_mask, other=0.0)
        acc += bias[None, :]
    
    # Swish: x * sigmoid(x)
    swish = acc * tl.sigmoid(acc)
    out = swish * scaling_factor
    
    # Store result
    out_idx = m_range[:, None] * stride_outm + n_range[None, :] * stride_outn
    out_mask = m_mask[:, None] & n_mask[None, :]
    tl.store(out_ptr + out_idx, out, mask=out_mask)


class FusedMatmulSwishScale(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, scaling_factor):
        M, K = x.shape
        N, K_weight = weight.shape
        assert K == K_weight, f"Input dim {K} must match weight dim {K_weight}"
        
        # Ensure contiguous
        x = x.contiguous()
        weight = weight.contiguous()
        if bias is not None:
            bias = bias.contiguous()
        
        out = torch.empty((M, N), dtype=x.dtype, device=x.device)
        
        # Block sizes - tuned for B200
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_K = 32
        
        grid = (triton.cdiv(M, BLOCK_M) * triton.cdiv(N, BLOCK_N),)
        
        _fused_matmul_swish_scale_kernel[grid](
            x, weight, bias if bias is not None else x,  # dummy if no bias
            out,
            M, N, K,
            x.stride(0), x.stride(1),
            weight.stride(0), weight.stride(1),
            out.stride(0), out.stride(1),
            scaling_factor,
            BLOCK_M, BLOCK_N, BLOCK_K,
            HAS_BIAS=bias is not None,
        )
        
        return out


class Model(nn.Module):
    """
    Optimized model that performs a matrix multiplication, applies Swish activation, and scales the result.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        self.scaling_factor = scaling_factor
        self.reset_parameters()

    def reset_parameters(self):
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return FusedMatmulSwishScale.apply(x, self.weight, self.bias, self.scaling_factor)


batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, scaling_factor]
