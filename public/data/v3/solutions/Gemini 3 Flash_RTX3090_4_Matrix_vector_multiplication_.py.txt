import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def mv_kernel(
    A_ptr, B_ptr, C_ptr,
    M, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
):
    """
    Computes C = A @ B, where A is (M, K) and B is (K, 1).
    Each program computes BLOCK_SIZE_M rows of C.
    """
    pid = tl.program_id(0)
    m_offsets = pid * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    m_mask = m_offsets < M

    # Accumulator
    acc = tl.zeros([BLOCK_SIZE_M], dtype=tl.float32)

    # Pre-calculate B pointer
    b_base_ptr = B_ptr + tl.arange(0, BLOCK_SIZE_K) * stride_bk

    # Loop over K in blocks
    for k in range(0, K, BLOCK_SIZE_K):
        b_ptr = b_base_ptr + k * stride_bk
        b_vals = tl.load(b_ptr) 

        # A is (M, K)
        a_ptr = A_ptr + (m_offsets[:, None] * stride_am + (k + tl.arange(0, BLOCK_SIZE_K))[None, :] * stride_ak)
        a_vals = tl.load(a_ptr, mask=m_mask[:, None])

        # Multiply and accumulate
        acc += tl.sum(a_vals * b_vals[None, :], axis=1)

    # Store the result
    c_ptr = C_ptr + m_offsets * stride_cm
    tl.store(c_ptr, acc, mask=m_mask)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        M, K = A.shape
        C = torch.empty((M, 1), device=A.device, dtype=A.dtype)
        
        BLOCK_SIZE_M = 16
        BLOCK_SIZE_K = 256
        
        grid = (triton.cdiv(M, BLOCK_SIZE_M),)
        
        mv_kernel[grid](
            A, B, C,
            M, K,
            A.stride(0), A.stride(1),
            B.stride(0), B.stride(1),
            C.stride(0), C.stride(1),
            BLOCK_SIZE_M=BLOCK_SIZE_M,
            BLOCK_SIZE_K=BLOCK_SIZE_K,
            num_warps=8,
            num_stages=2
        )
        return C
