import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "fused"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def fused_pool_gelu_scale_max_kernel(
    x_ptr, out_ptr,
    M, N,
    stride_xm, stride_xn,
    stride_om,
    scale,
    BLOCK_N: tl.constexpr,
):
    pid = tl.program_id(0)
    m = pid
    offs = tl.arange(0, BLOCK_N)
    base = offs * 4

    max_val = -float('inf')

    n_blocks = tl.cdiv(N, BLOCK_N * 4)
    for b in range(0, n_blocks):
        idx = b * BLOCK_N * 4 + base
        mask = idx + 3 < N

        x0 = tl.load(x_ptr + m * stride_xm + (idx + 0) * stride_xn, mask=mask, other=0.0)
        x1 = tl.load(x_ptr + m * stride_xm + (idx + 1) * stride_xn, mask=mask, other=0.0)
        x2 = tl.load(x_ptr + m * stride_xm + (idx + 2) * stride_xn, mask=mask, other=0.0)
        x3 = tl.load(x_ptr + m * stride_xm + (idx + 3) * stride_xn, mask=mask, other=0.0)

        avg = (x0 + x1 + x2 + x3) * 0.25
        avg_f = avg.to(tl.float32)
        gelu = avg_f * 0.5 * (1.0 + tl.erf(avg_f * 0.7071067811865475))
        val = gelu * scale
        block_max = tl.max(val, axis=0)
        max_val = tl.maximum(max_val, block_max)

    tl.store(out_ptr + m * stride_om, max_val)


class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        assert pool_kernel_size == 4, "This optimized kernel expects pool_kernel_size=4"
        self.matmul = nn.Linear(in_features, out_features)
        self.scale_factor = float(scale_factor)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        w = self.matmul.weight.t()
        with torch.no_grad():
            y = torch.empty((x.shape[0], w.shape[1]), device=x.device, dtype=x.dtype)
            torch.ops.aten.mm.out(x, w, out=y)
            if self.matmul.bias is not None:
                y = y + self.matmul.bias

        out = torch.empty((y.shape[0],), device=y.device, dtype=torch.float32)
        grid = (y.shape[0],)
        fused_pool_gelu_scale_max_kernel[grid](
            y, out,
            y.shape[0], y.shape[1],
            y.stride(0), y.stride(1),
            out.stride(0),
            self.scale_factor,
            BLOCK_N=256,
        )
        return out.to(y.dtype)


batch_size = 128
in_features = 4096
out_features = 4096
pool_kernel_size = 4
scale_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, pool_kernel_size, scale_factor]
