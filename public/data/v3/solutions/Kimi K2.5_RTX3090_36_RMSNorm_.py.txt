import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

template <typename scalar_t>
__global__ void rms_norm_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int B, int C, int H, int W,
    float eps
) {
    int b = blockIdx.x;
    int h = (blockIdx.y * blockDim.y + threadIdx.y);
    int w = (blockIdx.z * blockDim.x + threadIdx.x);
    
    if (h >= H || w >= W) return;
    
    int stride_n = C * H * W;
    int stride_c = H * W;
    int stride_h = W;
    
    int base_idx = b * stride_n + h * stride_h + w;
    
    float sum_sq = 0.0f;
    for (int c = 0; c < C; c++) {
        int idx = base_idx + c * stride_c;
        float val = static_cast<float>(x[idx]);
        sum_sq += val * val;
    }
    
    float rms = sqrtf(sum_sq / static_cast<float>(C) + eps);
    
    for (int c = 0; c < C; c++) {
        int idx = base_idx + c * stride_c;
        float val = static_cast<float>(x[idx]);
        y[idx] = static_cast<scalar_t>(val / rms);
    }
}

torch::Tensor rms_norm_forward(torch::Tensor x, double eps) {
    int B = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    
    auto y = torch::empty_like(x);
    
    const int TILE_SIZE = 16;
    dim3 block(TILE_SIZE, TILE_SIZE);
    dim3 grid(B, (H + TILE_SIZE - 1) / TILE_SIZE, (W + TILE_SIZE - 1) / TILE_SIZE);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "rms_norm_kernel", [&] {
        rms_norm_kernel<scalar_t><<<grid, block>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            B, C, H, W,
            static_cast<float>(eps)
        );
    });
    
    return y;
}
'''

cpp_source = '''
torch::Tensor rms_norm_forward(torch::Tensor x, double eps);
'''

ext = load_inline(
    name='rms_norm_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rms_norm_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_features=64, eps=1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rms_norm_forward(x, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [64]
