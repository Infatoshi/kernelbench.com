import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// GELU exact: 0.5 * x * (1 + erf(x / sqrt(2)))
// With --use_fast_math, erff becomes a fast approximation

static constexpr float INV_SQRT2 = 0.7071067811865476f;

__device__ __forceinline__ float gelu_float(float x) {
    return 0.5f * x * (1.0f + erff(x * INV_SQRT2));
}

// Process 4 floats at a time using float4 vectorized loads
__global__ void gelu_kernel_float4(const float4* __restrict__ input,
                                    float4* __restrict__ output,
                                    int n4,  // number of float4 elements
                                    int n_rem) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n4) {
        float4 in = input[idx];
        float4 out;
        out.x = gelu_float(in.x);
        out.y = gelu_float(in.y);
        out.z = gelu_float(in.z);
        out.w = gelu_float(in.w);
        output[idx] = out;
    }
    // Handle remainder
    if (idx == n4 && n_rem > 0) {
        const float* fin = reinterpret_cast<const float*>(input);
        float* fout = reinterpret_cast<float*>(output);
        int base = n4 * 4;
        for (int i = 0; i < n_rem; i++) {
            fout[base + i] = gelu_float(fin[base + i]);
        }
    }
}

// Process 8 fp16 values at a time (float4 = 8 x fp16)
__global__ void gelu_kernel_half8(const float4* __restrict__ input,
                                   float4* __restrict__ output,
                                   int n8,
                                   int n_rem) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n8) {
        float4 in = input[idx];
        __half* h = reinterpret_cast<__half*>(&in);
        float4 out_val;
        __half* oh = reinterpret_cast<__half*>(&out_val);
        #pragma unroll
        for (int i = 0; i < 8; i++) {
            float fx = __half2float(h[i]);
            oh[i] = __float2half(gelu_float(fx));
        }
        output[idx] = out_val;
    }
    if (idx == n8 && n_rem > 0) {
        const __half* fin = reinterpret_cast<const __half*>(input);
        __half* fout = reinterpret_cast<__half*>(output);
        int base = n8 * 8;
        for (int i = 0; i < n_rem; i++) {
            float fx = __half2float(fin[base + i]);
            fout[base + i] = __float2half(gelu_float(fx));
        }
    }
}

// Process 8 bf16 values at a time
__global__ void gelu_kernel_bf16_8(const float4* __restrict__ input,
                                    float4* __restrict__ output,
                                    int n8,
                                    int n_rem) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n8) {
        float4 in = input[idx];
        __nv_bfloat16* h = reinterpret_cast<__nv_bfloat16*>(&in);
        float4 out_val;
        __nv_bfloat16* oh = reinterpret_cast<__nv_bfloat16*>(&out_val);
        #pragma unroll
        for (int i = 0; i < 8; i++) {
            float fx = __bfloat162float(h[i]);
            oh[i] = __float2bfloat16(gelu_float(fx));
        }
        output[idx] = out_val;
    }
    if (idx == n8 && n_rem > 0) {
        const __nv_bfloat16* fin = reinterpret_cast<const __nv_bfloat16*>(input);
        __nv_bfloat16* fout = reinterpret_cast<__nv_bfloat16*>(output);
        int base = n8 * 8;
        for (int i = 0; i < n_rem; i++) {
            float fx = __bfloat162float(fin[base + i]);
            fout[base + i] = __float2bfloat16(gelu_float(fx));
        }
    }
}

torch::Tensor gelu_cuda(torch::Tensor input) {
    TORCH_CHECK(input.is_cuda(), "Input must be on CUDA");
    auto output = torch::empty_like(input);
    int n = input.numel();
    
    const int threads = 512;
    
    if (input.scalar_type() == torch::kFloat32) {
        int n4 = n / 4;
        int n_rem = n % 4;
        int blocks = (n4 + threads - 1) / threads;
        if (blocks == 0) blocks = 1;
        gelu_kernel_float4<<<blocks, threads>>>(
            reinterpret_cast<const float4*>(input.data_ptr<float>()),
            reinterpret_cast<float4*>(output.data_ptr<float>()),
            n4, n_rem
        );
    } else if (input.scalar_type() == torch::kFloat16) {
        int n8 = n / 8;
        int n_rem = n % 8;
        int blocks = (n8 + threads - 1) / threads;
        if (blocks == 0) blocks = 1;
        gelu_kernel_half8<<<blocks, threads>>>(
            reinterpret_cast<const float4*>(input.data_ptr<at::Half>()),
            reinterpret_cast<float4*>(output.data_ptr<at::Half>()),
            n8, n_rem
        );
    } else if (input.scalar_type() == torch::kBFloat16) {
        int n8 = n / 8;
        int n_rem = n % 8;
        int blocks = (n8 + threads - 1) / threads;
        if (blocks == 0) blocks = 1;
        gelu_kernel_bf16_8<<<blocks, threads>>>(
            reinterpret_cast<const float4*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<float4*>(output.data_ptr<at::BFloat16>()),
            n8, n_rem
        );
    } else {
        AT_ERROR("Unsupported dtype for GELU kernel");
    }
    
    return output;
}
'''

cpp_source = 'torch::Tensor gelu_cuda(torch::Tensor input);'

ext = load_inline(
    name='gelu_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gelu_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
)


class Model(nn.Module):
    """
    Optimized GELU activation using a custom CUDA kernel with vectorized loads.
    Supports fp16, bf16, fp32.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_cuda(x)


batch_size = 256
dim = 16384


def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]


def get_init_inputs():
    return []
