import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_conv_pool_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    N, C_in, H_in, W_in, C_out, K, pool_size,
    sub1, sub2,
    H_out, W_out,
    BLOCK_SIZE: tl.constexpr
):
    pid = tl.program_id(0)
    
    # Total output elements
    total_out = N * C_out * H_out * W_out
    
    # Each thread block handles BLOCK_SIZE elements
    start = pid * BLOCK_SIZE
    offs = start + tl.arange(0, BLOCK_SIZE)
    mask = offs < total_out
    
    # Decode linear index to n, c, h, w
    w_out = offs % W_out
    tmp = offs // W_out
    h_out = tmp % H_out
    tmp = tmp // H_out
    c_out = tmp % C_out
    n = tmp // C_out
    
    # Load weight for this output channel into SRAM
    # Weight shape: [C_out, C_in, K, K]
    w_local = tl.zeros((C_in * K * K,), dtype=tl.float32)
    for i in range(C_in * K * K):
        w_idx = c_out * (C_in * K * K) + i
        w_local[i] = tl.load(weight_ptr + w_idx).to(tl.float32)
    
    b = tl.load(bias_ptr + c_out).to(tl.float32)
    
    pool_sum = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    
    # Pooling loops
    for ph in range(pool_size):
        for pw in range(pool_size):
            h_conv = h_out * pool_size + ph
            w_conv = w_out * pool_size + pw
            
            conv_val = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
            
            # Convolution
            for c_in in range(C_in):
                for kh in range(K):
                    for kw in range(K):
                        h_in = h_conv + kh
                        w_in = w_conv + kw
                        
                        # Compute input index
                        in_idx = ((n * C_in + c_in) * H_in + h_in) * W_in + w_in
                        in_val = tl.load(input_ptr + in_idx, mask=mask, other=0.0).to(tl.float32)
                        
                        w_idx = c_in * K * K + kh * K + kw
                        conv_val += in_val * w_local[w_idx]
            
            conv_val += b
            conv_val = tl.libdevice.tanh(conv_val - sub1) - sub2
            pool_sum += conv_val
    
    # Average pooling
    pool_sum /= (pool_size * pool_size)
    
    # Store output
    out_idx = ((n * C_out + c_out) * H_out + h_out) * W_out + w_out
    tl.store(output_ptr + out_idx, pool_sum, mask=mask)


def fused_conv_pool_triton(input, weight, bias, sub1, sub2, pool_size):
    N, C_in, H_in, W_in = input.shape
    C_out, _, K, _ = weight.shape
    
    # Valid conv then pool
    H_conv = H_in - K + 1
    W_conv = W_in - K + 1
    H_out = H_conv // pool_size
    W_out = W_conv // pool_size
    
    output = torch.empty(N, C_out, H_out, W_out, device=input.device, dtype=input.dtype)
    
    total = N * C_out * H_out * W_out
    BLOCK_SIZE = 128
    
    grid = (triton.cdiv(total, BLOCK_SIZE),)
    
    fused_conv_pool_kernel[grid](
        input, weight, bias, output,
        N, C_in, H_in, W_in, C_out, K, pool_size,
        sub1, sub2,
        H_out, W_out,
        BLOCK_SIZE=BLOCK_SIZE
    )
    
    return output


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        
    def forward(self, x):
        return fused_conv_pool_triton(
            x,
            self.conv.weight,
            self.conv.bias,
            self.subtract1_value,
            self.subtract2_value,
            self.kernel_size_pool
        )


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    subtract1_value = 0.5
    subtract2_value = 0.2
    kernel_size_pool = 2
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
