import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def sum_reduction_kernel(
    input_ptr, output_ptr,
    stride_pre, stride_reduce, stride_post,
    n_pre, n_reduce, n_post,
    BLOCK_REDUCE: tl.constexpr,
    BLOCK_POST: tl.constexpr
):
    pre_idx = tl.program_id(0)
    post_pid = tl.program_id(1)
    post_offsets = post_pid * BLOCK_POST + tl.arange(0, BLOCK_POST)
    post_mask = post_offsets < n_post

    acc = tl.zeros([BLOCK_POST], dtype=tl.float32)
    
    for r_offset in range(0, n_reduce, BLOCK_REDUCE):
        r_idx = r_offset + tl.arange(0, BLOCK_REDUCE)
        r_mask = r_idx < n_reduce
        
        ptr = input_ptr + pre_idx * stride_pre + r_idx[:, None] * stride_reduce + post_offsets[None, :] * stride_post
        mask = r_mask[:, None] & post_mask[None, :]
        
        data = tl.load(ptr, mask=mask, other=0.0).to(tl.float32)
        acc += tl.sum(data, axis=0)

    out_ptr = output_ptr + pre_idx * n_post + post_offsets
    tl.store(out_ptr, acc.to(output_ptr.dtype.element_ty), mask=post_mask)

class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        shape = x.shape
        dim = self.dim if self.dim >= 0 else len(shape) + self.dim
        
        n_pre = 1
        for i in range(dim):
            n_pre *= shape[i]
        n_reduce = shape[dim]
        n_post = 1
        for i in range(dim + 1, len(shape)):
            n_post *= shape[i]
            
        x_reshaped = x.view(n_pre, n_reduce, n_post)
        s_pre, s_reduce, s_post = x_reshaped.stride()
        
        output_shape = list(shape)
        output_shape[dim] = 1
        output = torch.empty(output_shape, device=x.device, dtype=x.dtype)
        
        BLOCK_POST = 256
        BLOCK_REDUCE = 256
        
        grid = (n_pre, triton.cdiv(n_post, BLOCK_POST))
        
        sum_reduction_kernel[grid](
            x, output,
            s_pre, s_reduce, s_post,
            n_pre, n_reduce, n_post,
            BLOCK_REDUCE=BLOCK_REDUCE,
            BLOCK_POST=BLOCK_POST
        )
        
        return output

def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2, device='cuda')
    return [x]
