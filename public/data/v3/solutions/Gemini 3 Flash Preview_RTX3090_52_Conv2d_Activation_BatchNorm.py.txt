
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import triton
import triton.language as tl

# The model performs: x = conv(x); x = x * tanh(softplus(x)) (Mish activation); x = BN(x)
# Mish(x) = x * tanh(softplus(x))

@triton.jit
def mish_bn_kernel(
    x_ptr, weight_ptr, bias_ptr, mean_ptr, var_ptr, out_ptr,
    N, C, H, W,
    eps,
    stride_n, stride_c, stride_h, stride_w,
    BLOCK_SIZE: tl.constexpr
):
    # This kernel handles Mish + BN together.
    # BN is: (x - mean) / sqrt(var + eps) * weight + bias
    
    c = tl.program_id(0)
    idx = tl.program_id(1) * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    
    # Load BN params for this channel
    w = tl.load(weight_ptr + c)
    b = tl.load(bias_ptr + c)
    m = tl.load(mean_ptr + c)
    v = tl.load(var_ptr + c)
    
    inv_std = 1.0 / tl.sqrt(v + eps)
    
    # Map linear idx back to n, h, w
    # idx = n * H * W + h * W + w_
    n = idx // (H * W)
    rem = idx % (H * W)
    h = rem // W
    w_ = rem % W
    
    mask = n < N
    
    offs = n * stride_n + c * stride_c + h * stride_h + w_ * stride_w
    val = tl.load(x_ptr + offs, mask=mask)
    
    # Mish activation: x * tanh(softplus(x))
    # softplus(x) = log(1 + exp(x))
    # For numerical stability: softplus(x) = x if x > 20 else log(1 + exp(x))
    sp = tl.where(val > 20.0, val, tl.log(1.0 + tl.exp(val)))
    mish_val = val * tl.extra.cuda.libdevice.tanh(sp)
    
    # BN
    out_val = (mish_val - m) * inv_std * w + b
    
    tl.store(out_ptr + offs, out_val, mask=mask)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.eps = eps

    def forward(self, x):
        # We can optimize the convolution using torch's backend, but the task says 
        # "No PyTorch operator fallbacks (torch.matmul, F.linear, etc.)"
        # However, for Conv2d, implementing a custom one that beats CuDNN is hard.
        # But wait, the prompt says "FORBIDDEN ... torch.matmul, torch.mm, torch.conv2d, F.linear, F.conv2d".
        # So I MUST implement the convolution manually or use a trick.
        
        # Let's implement a manual Conv2d using Triton.
        # Then fuse Mish and BN.
        
        # BN parameters:
        # self.bn.running_mean, self.bn.running_var, self.bn.weight, self.bn.bias
        
        # Actually, let's use a simpler approach.
        # Manual Conv2d + Manual Mish + Manual BN.
        
        x = self.manual_conv2d(x)
        x = self.apply_mish_bn(x)
        return x

    def manual_conv2d(self, x):
        # x: (N, Cin, Hin, Win)
        # weight: (Cout, Cin, K, K)
        # bias: (Cout)
        N, Cin, Hin, Win = x.shape
        Cout, _, K, K_ = self.conv.weight.shape
        Hout = Hin - K + 1
        Wout = Win - K + 1
        
        # Im2col style or direct conv
        # For simplicity and speed in Triton, let's do a direct convolution kernel.
        output = torch.empty((N, Cout, Hout, Wout), device=x.device, dtype=x.dtype)
        
        grid = lambda META: (Cout, triton.cdiv(N * Hout * Wout, META['BLOCK_SIZE']))
        
        _conv_kernel[grid](
            x, self.conv.weight, self.conv.bias, output,
            N, Cin, Hin, Win, Cout, Hout, Wout, K,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            self.conv.weight.stride(0), self.conv.weight.stride(1), self.conv.weight.stride(2), self.conv.weight.stride(3),
            output.stride(0), output.stride(1), output.stride(2), output.stride(3),
            BLOCK_SIZE=256
        )
        return output

    def apply_mish_bn(self, x):
        # x is (N, Cout, Hout, Wout)
        N, C, H, W = x.shape
        out = torch.empty_like(x)
        
        grid = lambda META: (C, triton.cdiv(N * H * W, META['BLOCK_SIZE']))
        mish_bn_kernel[grid](
            x, self.bn.weight, self.bn.bias, self.bn.running_mean, self.bn.running_var, out,
            N, C, H, W,
            self.eps,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            BLOCK_SIZE=256
        )
        return out

@triton.jit
def _conv_kernel(
    x_ptr, weight_ptr, bias_ptr, out_ptr,
    N, Cin, Hin, Win, Cout, Hout, Wout, K,
    stride_xn, stride_xc, stride_xh, stride_xw,
    stride_wn, stride_wc, stride_wh, stride_ww,
    stride_on, stride_oc, stride_oh, stride_ow,
    BLOCK_SIZE: tl.constexpr
):
    c_out = tl.program_id(0)
    idx = tl.program_id(1) * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    
    n = idx // (Hout * Wout)
    rem = idx % (Hout * Wout)
    h_out = rem // Wout
    w_out = rem % Wout
    
    mask = n < N
    
    # Initialize accumulator with bias
    acc = tl.load(bias_ptr + c_out) + tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for c_in in range(0, Cin):
        for kh in range(0, K):
            for kw in range(0, K):
                h_in = h_out + kh
                w_in = w_out + kw
                
                x_val = tl.load(
                    x_ptr + n * stride_xn + c_in * stride_xc + h_in * stride_xh + w_in * stride_xw,
                    mask=mask, other=0.0
                )
                w_val = tl.load(
                    weight_ptr + c_out * stride_wn + c_in * stride_wc + kh * stride_wh + kw * stride_ww
                )
                acc += x_val * w_val
                
    tl.store(out_ptr + n * stride_on + c_out * stride_oc + h_out * stride_oh + w_out * stride_ow, acc, mask=mask)
