import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def int4_gemm_kernel(
    # Pointers to inputs
    x_ptr,        # (M, K) FP16 activations
    w_ptr,        # (N, K//2) UINT8 packed weights
    s_ptr,        # (N, num_groups) FP16 scales
    out_ptr,      # (M, N) FP16 output
    # Dimensions
    M: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    group_size: tl.constexpr,
    num_groups: tl.constexpr,
    # Strides
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_sn, stride_sg,
    stride_outm, stride_outn,
    # Block sizes
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    # Program IDs
    pid_m = tl.program_id(axis=0)
    pid_n = tl.program_id(axis=1)
    
    # Starting positions
    m_start = pid_m * BLOCK_M
    n_start = pid_n * BLOCK_N
    
    # Create range for M and N dimensions
    m_range = m_start + tl.arange(0, BLOCK_M)
    n_range = n_start + tl.arange(0, BLOCK_N)
    
    # Mask for valid elements
    m_mask = m_range < M
    n_mask = n_range < N
    
    # Accumulator in FP32 for precision
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Iterate over K dimension in blocks
    for k_start in range(0, K, BLOCK_K):
        # Load activation block: (BLOCK_M, BLOCK_K)
        k_range = k_start + tl.arange(0, BLOCK_K)
        k_mask = k_range < K
        
        # Create 2D mask for x
        x_mask = m_mask[:, None] & k_mask[None, :]
        
        # Load x values
        x_ptrs = x_ptr + m_range[:, None] * stride_xm + k_range[None, :] * stride_xk
        x = tl.load(x_ptrs, mask=x_mask, other=0.0).to(tl.float32)
        
        # For weights, we need to load BLOCK_K INT4 values
        # These are stored in BLOCK_K//2 bytes
        # k_pair = k // 2, so k_start//2 to k_start//2 + BLOCK_K//2 - 1
        k_pair_start = k_start // 2
        k_pair_range = k_pair_start + tl.arange(0, BLOCK_K // 2)
        k_pair_mask = k_pair_range < (K // 2)
        
        # Load packed weights: (BLOCK_N, BLOCK_K//2)
        w_mask = n_mask[:, None] & k_pair_mask[None, :]
        w_ptrs = w_ptr + n_range[:, None] * stride_wn + k_pair_range[None, :] * stride_wk
        w_packed = tl.load(w_ptrs, mask=w_mask, other=0)
        
        # Unpack INT4: extract low and high nibbles
        w_low = w_packed & 0x0F  # First weight in each pair (even k indices)
        w_high = (w_packed >> 4) & 0x0F  # Second weight (odd k indices)
        
        # Now we need to interleave low and high to get (BLOCK_N, BLOCK_K)
        # For k_start even: byte 0 has k=0 (low), k=1 (high)
        # For k_start odd: we'd need to handle offset, but we always start at even k_start
        
        # Create the interleaved weight matrix
        # w_low[:, i] corresponds to k = k_start + 2*i
        # w_high[:, i] corresponds to k = k_start + 2*i + 1
        
        # Reshape for interleaving
        # We'll use a different approach: compute weights for even and odd k separately
        
        # For even k positions in this block (k_start, k_start+2, ...)
        k_even = k_start + 2 * tl.arange(0, BLOCK_K // 2)
        k_odd = k_start + 1 + 2 * tl.arange(0, BLOCK_K // 2)
        
        # Compute group indices for even and odd positions
        group_even = k_even // group_size
        group_odd = k_odd // group_size
        
        # Load scales for each group
        # For simplicity, load all scales needed for this block
        # First, determine which groups are in this block
        first_group = k_start // group_size
        last_group = (k_start + BLOCK_K - 1) // group_size
        
        # Load scales for each group in this range
        # We'll load them into a 2D array and select based on k
        
        # For each n, load scales for all groups in this block
        # Since num_groups_in_block is small (1 or 2 typically), we can unroll
        
        # Load scale for first group
        s_first = tl.load(s_ptr + n_range * stride_sn + tl.minimum(first_group, num_groups - 1) * stride_sg,
                          mask=n_mask, other=1.0)
        
        if last_group > first_group:
            s_last = tl.load(s_ptr + n_range * stride_sn + tl.minimum(last_group, num_groups - 1) * stride_sg,
                            mask=n_mask, other=1.0)
        
        # For each position in BLOCK_K, determine which scale to use
        k_local = tl.arange(0, BLOCK_K)
        k_global = k_start + k_local
        group_for_k = k_global // group_size
        
        # Build scale matrix (BLOCK_N, BLOCK_K)
        # Start with first group's scale
        scales_block = tl.broadcast_to(s_first[:, None], (BLOCK_N, BLOCK_K))
        
        if last_group > first_group:
            # For k positions in last_group, use s_last
            is_last_group = group_for_k == last_group
            s_last_col = tl.broadcast_to(s_last[:, None], (BLOCK_N, BLOCK_K))
            scales_block = tl.where(is_last_group[None, :], s_last_col, scales_block)
        
        # Now build the weight matrix (BLOCK_N, BLOCK_K)
        # Interleave w_low and w_high
        # w_low[:, i] -> position 2*i, w_high[:, i] -> position 2*i+1
        
        # Create weight matrix by selecting from w_low or w_high based on k position
        is_even = (k_local % 2) == 0
        pair_idx = k_local // 2  # Index into w_low/w_high
        
        # Load from w_low for even k, w_high for odd k
        w_selected = tl.where(
            is_even[None, :],
            tl.load(w_ptr + n_range[:, None] * stride_wn + (k_pair_start + pair_idx)[None, :] * stride_wk,
                    mask=n_mask[:, None] & (k_pair_start + pair_idx)[None, :] < (K // 2), other=0) & 0x0F,
            (tl.load(w_ptr + n_range[:, None] * stride_wn + (k_pair_start + pair_idx)[None, :] * stride_wk,
                     mask=n_mask[:, None] & (k_pair_start + pair_idx)[None, :] < (K // 2), other=0) >> 4) & 0x0F
        ).to(tl.int32)
        
        # Dequantize: w_dequant = scale * (w_int - 8)
        w_dequant = scales_block * (w_selected.to(tl.float16) - 8.0)
        w_fp32 = w_dequant.to(tl.float32)
        
        # Matrix multiply accumulate
        acc += tl.dot(x, tl.trans(w_fp32))
    
    # Store result
    out_mask = m_mask[:, None] & n_mask[None, :]
    out_ptrs = out_ptr + m_range[:, None] * stride_outm + n_range[None, :] * stride_outn
    tl.store(out_ptrs, acc.to(tl.float16), mask=out_mask)


class Model(nn.Module):
    """
    INT4 Weight-Only Quantized Linear Layer with Symmetric Quantization.
    Optimized implementation using Triton.
    """
    
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size
        
        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even for INT4 packing"
        
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        M = batch_size * seq_len
        
        # Reshape input
        x_2d = x.view(-1, self.K).contiguous()
        out = torch.empty((M, self.N), dtype=torch.float16, device=x.device)
        
        # Block sizes tuned for RTX 3090
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_K = 128
        
        grid = (triton.cdiv(M, BLOCK_M), triton.cdiv(self.N, BLOCK_N))
        
        int4_gemm_kernel[grid](
            x_2d,
            self.weight_packed,
            self.scales,
            out,
            M, self.N, self.K,
            self.group_size,
            self.num_groups,
            x_2d.stride(0), x_2d.stride(1),
            self.weight_packed.stride(0), self.weight_packed.stride(1),
            self.scales.stride(0), self.scales.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_M, BLOCK_N, BLOCK_K,
            num_warps=4,
            num_stages=3,
        )
        
        return out.view(batch_size, seq_len, self.N)


def get_inputs():
    batch_size = 4
    seq_len = 2048
    K = 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    K = 4096
    N = 11008
    group_size = 128
    return [K, N, group_size]
