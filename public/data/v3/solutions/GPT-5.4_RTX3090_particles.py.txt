import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <vector>

template <typename scalar_t>
__global__ void particles_kernel(
    const scalar_t* __restrict__ positions,
    const scalar_t* __restrict__ velocities,
    const float dt,
    const float gx,
    const float gy,
    const float damping,
    scalar_t* __restrict__ out,
    const int64_t n)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;

    int base2 = idx * 2;
    int base4 = idx * 4;

    float px = static_cast<float>(positions[base2 + 0]);
    float py = static_cast<float>(positions[base2 + 1]);
    float vx = static_cast<float>(velocities[base2 + 0]);
    float vy = static_cast<float>(velocities[base2 + 1]);

    float nvx = vx + gx * dt;
    float nvy = vy + gy * dt;
    float npx = px + nvx * dt;
    float npy = py + nvy * dt;

    bool overx = fabsf(npx) > 1.0f;
    bool overy = fabsf(npy) > 1.0f;

    if (overx) nvx = -nvx * damping;
    if (overy) nvy = -nvy * damping;

    npx = fminf(fmaxf(npx, -1.0f), 1.0f);
    npy = fminf(fmaxf(npy, -1.0f), 1.0f);

    out[base4 + 0] = static_cast<scalar_t>(npx);
    out[base4 + 1] = static_cast<scalar_t>(npy);
    out[base4 + 2] = static_cast<scalar_t>(nvx);
    out[base4 + 3] = static_cast<scalar_t>(nvy);
}

torch::Tensor particles_forward(
    torch::Tensor positions,
    torch::Tensor velocities,
    double dt,
    torch::Tensor gravity,
    double damping)
{
    TORCH_CHECK(positions.is_cuda(), "positions must be CUDA");
    TORCH_CHECK(velocities.is_cuda(), "velocities must be CUDA");
    TORCH_CHECK(gravity.is_cuda(), "gravity must be CUDA");
    TORCH_CHECK(positions.is_contiguous(), "positions must be contiguous");
    TORCH_CHECK(velocities.is_contiguous(), "velocities must be contiguous");
    TORCH_CHECK(gravity.is_contiguous(), "gravity must be contiguous");
    TORCH_CHECK(positions.dim() == 2 && positions.size(1) == 2, "positions shape must be [N,2]");
    TORCH_CHECK(velocities.sizes() == positions.sizes(), "velocities shape mismatch");
    TORCH_CHECK(gravity.numel() == 2, "gravity must have 2 elements");
    TORCH_CHECK(positions.scalar_type() == velocities.scalar_type(), "dtype mismatch");

    auto n = positions.size(0);
    auto out = torch::empty({n, 4}, positions.options());

    auto gravity_f = gravity.to(torch::kFloat32).contiguous();
    float hgrav[2];
    cudaMemcpy(hgrav, gravity_f.data_ptr<float>(), 2 * sizeof(float), cudaMemcpyDeviceToHost);

    constexpr int threads = 256;
    int blocks = (n + threads - 1) / threads;
    auto stream = at::cuda::getDefaultCUDAStream();

    AT_DISPATCH_FLOATING_TYPES_AND2(torch::kHalf, torch::kBFloat16, positions.scalar_type(), "particles_forward", [&] {
        particles_kernel<scalar_t><<<blocks, threads, 0, stream>>>(
            positions.data_ptr<scalar_t>(),
            velocities.data_ptr<scalar_t>(),
            static_cast<float>(dt),
            hgrav[0],
            hgrav[1],
            static_cast<float>(damping),
            out.data_ptr<scalar_t>(),
            n);
    });

    return out;
}
'''

cpp_source = r'''
torch::Tensor particles_forward(
    torch::Tensor positions,
    torch::Tensor velocities,
    double dt,
    torch::Tensor gravity,
    double damping);
'''

ext = load_inline(
    name='particles_ext_v1',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particles_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo', '-std=c++17', '-arch=sm_86'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(self, positions: torch.Tensor, velocities: torch.Tensor, dt: torch.Tensor, gravity: torch.Tensor) -> torch.Tensor:
        if not positions.is_cuda:
            positions = positions.cuda()
        if not velocities.is_cuda:
            velocities = velocities.cuda()
        if not gravity.is_cuda:
            gravity = gravity.cuda()
        positions = positions.contiguous()
        velocities = velocities.contiguous()
        gravity = gravity.contiguous()
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        return ext.particles_forward(positions, velocities, dt_value, gravity, float(self.damping))


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
