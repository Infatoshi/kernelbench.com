import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ inline float my_tanh(float x){ return tanhf(x); }

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    const float* __restrict__ conv_bias,
    float* __restrict__ out,
    int N, int C_in, int H, int W,
    int C_out, int K,
    int H_out, int W_out,
    float scale
){
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * C_out * H_out * W_out;
    if(idx >= total) return;

    int pw = idx % W_out;
    int ph = (idx / W_out) % H_out;
    int co = (idx / (W_out*H_out)) % C_out;
    int n = idx / (W_out*H_out*C_out);

    float max_val = -1e20f;

    for(int dy=0; dy<2; ++dy){
        for(int dx=0; dx<2; ++dx){
            int h0 = ph*2 + dy;
            int w0 = pw*2 + dx;

            float sum = 0.0f;
            for(int ci=0; ci<C_in; ++ci){
                for(int ky=0; ky<K; ++ky){
                    for(int kx=0; kx<K; ++kx){
                        int ih = h0 + ky;
                        int iw = w0 + kx;
                        int x_idx = ((n*C_in + ci)*H + ih)*W + iw;
                        int w_idx = ((co*C_in + ci)*K + ky)*K + kx;
                        sum += x[x_idx] * w[w_idx];
                    }
                }
            }
            sum += conv_bias[co];
            sum = my_tanh(sum);
            sum = sum * scale;
            sum = sum + bias[co];
            if(sum > max_val) max_val = sum;
        }
    }

    out[idx] = max_val;
}

torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor bias, torch::Tensor conv_bias, float scale){
    int N = x.size(0);
    int C_in = x.size(1);
    int H = x.size(2);
    int W = x.size(3);
    int C_out = w.size(0);
    int K = w.size(2);

    int Hc = H - K + 1;
    int Wc = W - K + 1;
    int H_out = Hc / 2;
    int W_out = Wc / 2;

    auto out = torch::empty({N, C_out, H_out, W_out}, x.options());

    int total = N * C_out * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C_in, H, W,
        C_out, K,
        H_out, W_out,
        scale
    );

    return out;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor bias, float scale);',
    cuda_sources=cuda_src,
    functions=['fused_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        self.conv_bias = nn.Parameter(torch.randn(out_channels))
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        if x.dtype != torch.float32:
            x = x.float()
        w = self.weight.float()
        b = self.bias.view(-1).float()
        cb = self.conv_bias.float()
        y = ext.fused_op(x, w, b, float(self.scaling_factor))
        return y


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 2.0, (16,1,1), 2]
