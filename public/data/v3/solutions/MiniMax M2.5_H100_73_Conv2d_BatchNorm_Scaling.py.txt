#include <torch/extension.h>
#include <cuda_runtime.h>

// Helper function to convert bfloat16 to float
__device__ __forceinline__ float bf16_to_float(const at::BFloat16& x) {
    return float(x);
}

// Fused Conv2d + BatchNorm + Scale kernel (FP32)
__global__ void fused_conv_bn_scale_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ mean,
    const float* __restrict__ var,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    const float scaling_factor,
    const int B, const int IC, const int OC,
    const int H, const int W,
    float* __restrict__ output
) {
    const int oc = blockIdx.z;
    const int tid = threadIdx.x;
    const int num_threads = blockDim.x;
    
    const int out_h = H - 2;
    const int out_w = W - 2;
    const int total_out = B * out_h * out_w;
    
    // Load BN params
    __shared__ float bn_mean, bn_var, bn_gamma, bn_beta;
    if (tid == 0) {
        bn_mean = mean[oc];
        bn_var = var[oc];
        bn_gamma = gamma[oc];
        bn_beta = beta[oc];
    }
    __syncthreads();
    
    const float inv_std = rsqrtf(bn_var + 1e-5f);
    
    // Each thread processes multiple output elements
    for (int idx = tid; idx < total_out; idx += num_threads) {
        const int b = idx / (out_h * out_w);
        const int rem = idx % (out_h * out_w);
        const int oh = rem / out_w;
        const int ow = rem % out_w;
        
        // Compute convolution
        float conv_result = 0.0f;
        const int weight_base = oc * IC * 9;
        
        for (int ic = 0; ic < IC; ic++) {
            const float* input_patch = input + b * IC * H * W + ic * H * W;
            const int w_offset = weight_base + ic * 9;
            
            float sum = 0.0f;
            sum += input_patch[oh * W + ow] * weight[w_offset + 4];
            sum += input_patch[oh * W + (ow-1)] * weight[w_offset + 3];
            sum += input_patch[oh * W + (ow+1)] * weight[w_offset + 5];
            sum += input_patch[(oh-1) * W + ow] * weight[w_offset + 1];
            sum += input_patch[(oh-1) * W + (ow-1)] * weight[w_offset + 0];
            sum += input_patch[(oh-1) * W + (ow+1)] * weight[w_offset + 2];
            sum += input_patch[(oh+1) * W + ow] * weight[w_offset + 7];
            sum += input_patch[(oh+1) * W + (ow-1)] * weight[w_offset + 6];
            sum += input_patch[(oh+1) * W + (ow+1)] * weight[w_offset + 8];
            
            conv_result += sum;
        }
        
        // Apply BatchNorm and scaling
        float bn_result = (conv_result - bn_mean) * inv_std * bn_gamma + bn_beta;
        float final_result = bn_result * scaling_factor;
        
        output[b * OC * out_h * out_w + oc * out_h * out_w + oh * out_w + ow] = final_result;
    }
}

torch::Tensor fused_conv_bn_scale_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor mean,
    torch::Tensor var,
    torch::Tensor gamma,
    torch::Tensor beta,
    float scaling_factor) {
    
    const int B = input.size(0);
    const int IC = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    const int OC = weight.size(0);
    
    const int out_h = H - 2;
    const int out_w = W - 2;
    
    auto output = torch::empty({B, OC, out_h, out_w}, input.options());
    
    const int total_out = B * OC * out_h * out_w;
    const int threads = 256;
    const int blocks_x = (total_out + threads - 1) / (threads * 4);
    const int blocks_z = OC;
    
    const dim3 blocks(blocks_x, 1, blocks_z);
    
    fused_conv_bn_scale_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        mean.data_ptr<float>(),
        var.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        scaling_factor,
        B, IC, OC, H, W,
        output.data_ptr<float>()
    );
    
    return output;
}

cpp_source = r'''
torch::Tensor fused_conv_bn_scale_fp32(
    torch::Tensor input, torch::Tensor weight,
    torch::Tensor mean, torch::Tensor var,
    torch::Tensor gamma, torch::Tensor beta,
    float scaling_factor);

torch::Tensor fused_conv_bn_scale_fp16(
    torch::Tensor input, torch::Tensor weight,
    torch::Tensor mean, torch::Tensor var,
    torch::Tensor gamma, torch::Tensor beta,
    float scaling_factor);

torch::Tensor fused_conv_bn_scale_bf16(
    torch::Tensor input, torch::Tensor weight,
    torch::Tensor mean, torch::Tensor var,
    torch::Tensor gamma, torch::Tensor beta,
    float scaling_factor);

ext = load_inline(
    name='fused_conv_bn',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_bn_scale_fp32', 'fused_conv_bn_scale_fp16', 'fused_conv_bn_scale_bf16'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        
    def forward(self, x):
        dtype = x.dtype
        
        weight = self.conv.weight
        mean = self.bn.running_mean
        var = self.bn.running_var
        gamma = self.bn.weight
        beta = self.bn.bias
        
        if dtype == torch.float16:
            weight_fp = weight.half()
            mean_fp = mean.float()
            var_fp = var.float()
            gamma_fp = gamma.float()
            beta_fp = beta.float()
            return ext.fused_conv_bn_scale_fp16(
                x.half(), weight_fp, mean_fp, var_fp, gamma_fp, beta_fp, self.scaling_factor
            ).float()
        elif dtype == torch.bfloat16:
            weight_fp = weight.bfloat16()
            mean_fp = mean.float()
            var_fp = var.float()
            gamma_fp = gamma.float()
            beta_fp = beta.float()
            return ext.fused_conv_bn_scale_bf16(
                x.bfloat16(), weight_fp, mean_fp, var_fp, gamma_fp, beta_fp, self.scaling_factor
            ).float()
        else:
            return ext.fused_conv_bn_scale_fp32(
                x, weight, mean, var, gamma, beta, self.scaling_factor
            )

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 2.0]
