import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Optimized sum reduction kernel using shared memory
// For reducing over dimension 1 of a (N, D1, D2) tensor

template<typename T>
__global__ void sum_reduction_kernel(const T* input, T* output, 
                                      int N, int D1, int D2) {
    // Each block handles one (n, d2) pair, reducing over d1
    int n = blockIdx.x;
    int d2 = blockIdx.y;
    
    extern __shared__ float sdata[];
    
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    // Load and accumulate input values
    float sum = 0.0f;
    for (int d1 = tid; d1 < D1; d1 += stride) {
        int idx = n * D1 * D2 + d1 * D2 + d2;
        sum += static_cast<float>(input[idx]);
    }
    
    sdata[tid] = sum;
    __syncthreads();
    
    // Parallel reduction in shared memory
    for (int s = stride / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    // Write result
    if (tid == 0) {
        int out_idx = n * D2 + d2;
        output[out_idx] = static_cast<T>(sdata[0]);
    }
}

// Specialized kernel for FP16
__global__ void sum_reduction_fp16_kernel(const half* input, half* output, 
                                           int N, int D1, int D2) {
    int n = blockIdx.x;
    int d2 = blockIdx.y;
    
    extern __shared__ float sdata[];
    
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    for (int d1 = tid; d1 < D1; d1 += blockDim.x) {
        int idx = n * D1 * D2 + d1 * D2 + d2;
        sum += __half2float(input[idx]);
    }
    
    sdata[tid] = sum;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        int out_idx = n * D2 + d2;
        output[out_idx] = __float2half(sdata[0]);
    }
}

// Specialized kernel for BF16
__global__ void sum_reduction_bf16_kernel(const __nv_bfloat16* input, __nv_bfloat16* output, 
                                           int N, int D1, int D2) {
    int n = blockIdx.x;
    int d2 = blockIdx.y;
    
    extern __shared__ float sdata[];
    
    int tid = threadIdx.x;
    
    float sum = 0.0f;
    for (int d1 = tid; d1 < D1; d1 += blockDim.x) {
        int idx = n * D1 * D2 + d1 * D2 + d2;
        sum += __bfloat162float(input[idx]);
    }
    
    sdata[tid] = sum;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        int out_idx = n * D2 + d2;
        output[out_idx] = __float2bfloat16(sdata[0]);
    }
}

torch::Tensor sum_reduction(torch::Tensor x, int dim) {
    auto input = x.contiguous();
    auto dtype = input.dtype();
    auto sizes = input.sizes();
    
    int ndim = sizes.size();
    TORCH_CHECK(dim >= 0 && dim < ndim, "Invalid dimension");
    
    // Calculate N, D1, D2 where we reduce over D1 (dimension dim)
    int N = 1, D1 = sizes[dim], D2 = 1;
    
    for (int i = 0; i < dim; i++) N *= sizes[i];
    for (int i = dim + 1; i < ndim; i++) D2 *= sizes[i];
    
    // Output shape: same as input but with dim size = 1
    std::vector<int64_t> output_sizes(sizes.begin(), sizes.end());
    output_sizes[dim] = 1;
    auto output = torch::empty(output_sizes, dtype);
    
    // Make output contiguous in the reduced dimensions
    auto output_flat = output.view({N * D2});
    
    int threads = 256;
    dim3 grid(N, D2);
    
    if (dtype == torch::kFloat32) {
        sum_reduction_kernel<float><<<grid, threads, threads * sizeof(float)>>>(
            input.data_ptr<float>(), output_flat.data_ptr<float>(), N, D1, D2);
    } else if (dtype == torch::kFloat16) {
        sum_reduction_fp16_kernel<<<grid, threads, threads * sizeof(float)>>>(
            reinterpret_cast<const half*>(input.data_ptr<c10::Half>()), 
            reinterpret_cast<half*>(output_flat.data_ptr<c10::Half>()), N, D1, D2);
    } else if (dtype == torch::kBFloat16) {
        sum_reduction_bf16_kernel<<<grid, threads, threads * sizeof(float)>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<c10::BFloat16>()), 
            reinterpret_cast<__nv_bfloat16*>(output_flat.data_ptr<c10::BFloat16>()), N, D1, D2);
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }
    
    // Reshape back to original shape with reduced dim = 1
    return output.view(output_sizes);
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor sum_reduction(torch::Tensor x, int dim);
"""

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduction'],
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs sum reduction over a specified dimension.
    """
    def __init__(self, dim: int):
        """
        Initializes the model with the dimension to reduce over.

        Args:
            dim (int): Dimension to reduce over.
        """
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies sum reduction over the specified dimension.

        Args:
            x (torch.Tensor): Input tensor of shape (..., dim, ...).

        Returns:
            torch.Tensor: Output tensor after sum reduction, shape (..., 1, ...).
        """
        return ext.sum_reduction(x, self.dim)

def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    reduce_dim = 1
    return [reduce_dim]
