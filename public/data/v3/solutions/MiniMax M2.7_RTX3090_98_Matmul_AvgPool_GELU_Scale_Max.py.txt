import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Tanh-based GELU approximation 
__device__ __forceinline__ float gelu_tanh(float x) {
    float cdf = 0.5f * (1.0f + tanhf(0.7978845608f * x * (1.0f + 0.044715f * x * x)));
    return x * cdf;
}

// Kernel 1: Matmul + Bias
__global__ void matmul_bias_kernel_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features
) {
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (out_idx >= out_features) return;
    
    const int batch_idx = blockIdx.y;
    
    float sum = 0.0f;
    const float* w_row = weight + out_idx * in_features;
    const float* in_batch = input + batch_idx * in_features;
    
    #pragma unroll 8
    for (int k = 0; k + 7 < in_features; k += 8) {
        sum += w_row[k] * in_batch[k];
        sum += w_row[k+1] * in_batch[k+1];
        sum += w_row[k+2] * in_batch[k+2];
        sum += w_row[k+3] * in_batch[k+3];
        sum += w_row[k+4] * in_batch[k+4];
        sum += w_row[k+5] * in_batch[k+5];
        sum += w_row[k+6] * in_batch[k+6];
        sum += w_row[k+7] * in_batch[k+7];
    }
    for (int k = (in_features / 8) * 8; k < in_features; ++k) {
        sum += w_row[k] * in_batch[k];
    }
    
    if (bias != nullptr) {
        sum += bias[out_idx];
    }
    
    output[batch_idx * out_features + out_idx] = sum;
}

// Kernel 2: AvgPool + GELU + Scale + Max
__global__ void pool_gelu_scale_max_kernel_fp32(
    const float* __restrict__ matmul_out,
    float* __restrict__ output,
    int batch_size,
    int out_features,
    int pool_kernel_size,
    float scale_factor
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    const int pooled_size = out_features / pool_kernel_size;
    
    extern __shared__ float shmem[];
    float* pooled_vals = shmem;
    float* gelu_vals = &shmem[pooled_size];
    
    // Thread 0-1023: compute pooled values and GELU+scale
    if (tid < pooled_size) {
        int start = tid * pool_kernel_size;
        float sum = matmul_out[batch_idx * out_features + start];
        sum += matmul_out[batch_idx * out_features + start + 1];
        sum += matmul_out[batch_idx * out_features + start + 2];
        sum += matmul_out[batch_idx * out_features + start + 3];
        float avg_val = sum * 0.25f;
        pooled_vals[tid] = gelu_tanh(avg_val) * scale_factor;
    }
    __syncthreads();
    
    // Warp-level max reduction within first warp
    if (tid < 32) {
        float val = pooled_vals[tid];
        #pragma unroll 5
        for (int offset = 16; offset > 0; offset /= 2) {
            float other = __shfl_down_sync(0xffffffff, val, offset);
            val = val > other ? val : other;
        }
        if (tid == 0) {
            output[batch_idx] = val;
        }
    }
}

// FP16 versions
__global__ void matmul_bias_kernel_fp16(
    const __half* __restrict__ input,
    const __half* __restrict__ weight,
    const __half* __restrict__ bias,
    __half* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features
) {
    const int out_idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (out_idx >= out_features) return;
    
    const int batch_idx = blockIdx.y;
    
    float sum = 0.0f;
    const __half* w_row = weight + out_idx * in_features;
    const __half* in_batch = input + batch_idx * in_features;
    
    int k = 0;
    for (; k + 3 < in_features; k += 4) {
        __half2 inp01 = *reinterpret_cast<const __half2*>(&in_batch[k]);
        __half2 inp23 = *reinterpret_cast<const __half2*>(&in_batch[k + 2]);
        __half2 w01 = *reinterpret_cast<const __half2*>(&w_row[k]);
        __half2 w23 = *reinterpret_cast<const __half2*>(&w_row[k + 2]);
        
        float2 inp_f1 = __half22float2(inp01);
        float2 inp_f2 = __half22float2(inp23);
        float2 w_f1 = __half22float2(w01);
        float2 w_f2 = __half22float2(w23);
        
        sum += inp_f1.x * w_f1.x + inp_f1.y * w_f1.y;
        sum += inp_f2.x * w_f2.x + inp_f2.y * w_f2.y;
    }
    for (; k < in_features; ++k) {
        sum += __half2float(in_batch[k]) * __half2float(w_row[k]);
    }
    
    if (bias != nullptr) {
        sum += __half2float(bias[out_idx]);
    }
    
    output[batch_idx * out_features + out_idx] = __float2half(sum);
}

__global__ void pool_gelu_scale_max_kernel_fp16(
    const __half* __restrict__ matmul_out,
    __half* __restrict__ output,
    int batch_size,
    int out_features,
    int pool_kernel_size,
    float scale_factor
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    const int pooled_size = out_features / pool_kernel_size;
    
    extern __shared__ __half shmem[];
    __half* pooled_vals = shmem;
    
    // Thread 0-1023: compute pooled values and GELU+scale
    if (tid < pooled_size) {
        int start = tid * pool_kernel_size;
        float r0 = __half2float(matmul_out[batch_idx * out_features + start]);
        float r1 = __half2float(matmul_out[batch_idx * out_features + start + 1]);
        float r2 = __half2float(matmul_out[batch_idx * out_features + start + 2]);
        float r3 = __half2float(matmul_out[batch_idx * out_features + start + 3]);
        float avg_val = (r0 + r1 + r2 + r3) * 0.25f;
        pooled_vals[tid] = __float2half(gelu_tanh(avg_val) * scale_factor);
    }
    __syncthreads();
    
    // Warp-level max reduction
    if (tid < 32) {
        __half val = pooled_vals[tid];
        for (int offset = 16; offset > 0; offset /= 2) {
            __half other = __shfl_down_sync(0xffffffff, val, offset);
            val = __hgt(val, other) ? val : other;
        }
        if (tid == 0) {
            output[batch_idx] = val;
        }
    }
}

// Combined fused kernel (single pass)
__global__ void fused_single_kernel_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features,
    int pool_kernel_size,
    float scale_factor
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    const int pooled_size = out_features / pool_kernel_size;
    
    // Each block handles one batch element with 1024 threads
    // Threads 0-4095: matmul
    // Threads 0-1023: pool+gelu+max
    
    extern __shared__ float shmem[];
    float* matmul_out = shmem;
    float* pooled_vals = &shmem[out_features];
    
    // Phase 1: Matmul
    if (tid < out_features) {
        float sum = 0.0f;
        const float* w_row = weight + tid * in_features;
        const float* in_batch = input + batch_idx * in_features;
        
        #pragma unroll 8
        for (int k = 0; k + 7 < in_features; k += 8) {
            sum += w_row[k] * in_batch[k];
            sum += w_row[k+1] * in_batch[k+1];
            sum += w_row[k+2] * in_batch[k+2];
            sum += w_row[k+3] * in_batch[k+3];
            sum += w_row[k+4] * in_batch[k+4];
            sum += w_row[k+5] * in_batch[k+5];
            sum += w_row[k+6] * in_batch[k+6];
            sum += w_row[k+7] * in_batch[k+7];
        }
        for (int k = (in_features / 8) * 8; k < in_features; ++k) {
            sum += w_row[k] * in_batch[k];
        }
        
        if (bias != nullptr) {
            sum += bias[tid];
        }
        
        matmul_out[tid] = sum;
    }
    __syncthreads();
    
    // Phase 2: Pool + GELU + Scale
    if (tid < pooled_size) {
        int start = tid * pool_kernel_size;
        float sum = matmul_out[start];
        sum += matmul_out[start + 1];
        sum += matmul_out[start + 2];
        sum += matmul_out[start + 3];
        float avg_val = sum * 0.25f;
        pooled_vals[tid] = gelu_tanh(avg_val) * scale_factor;
    }
    __syncthreads();
    
    // Phase 3: Max reduction
    if (tid < 32) {
        float val = pooled_vals[tid];
        #pragma unroll 5
        for (int offset = 16; offset > 0; offset /= 2) {
            float other = __shfl_down_sync(0xffffffff, val, offset);
            val = val > other ? val : other;
        }
        if (tid == 0) {
            output[batch_idx] = val;
        }
    }
}

// Host wrappers
torch::Tensor fused_op_fp32(torch::Tensor input, torch::Tensor weight, 
                              torch::Tensor bias, int pool_kernel_size, float scale_factor) {
    auto output = torch::empty({input.size(0)}, input.options());
    
    const int batch_size = input.size(0);
    const int in_features = input.size(1);
    const int out_features = weight.size(0);
    const int pooled_size = out_features / pool_kernel_size;
    
    // Use single kernel with 4096 threads (out_features) per batch
    const int threads = 1024;  // More threads for better parallelism
    dim3 blocks(batch_size);
    int shared_mem = (out_features + pooled_size) * sizeof(float);
    
    fused_single_kernel_fp32<<<blocks, threads, shared_mem>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size, in_features, out_features, pool_kernel_size, scale_factor
    );
    
    return output;
}

torch::Tensor fused_op_fp16(torch::Tensor input, torch::Tensor weight,
                              torch::Tensor bias, int pool_kernel_size, float scale_factor) {
    auto output = torch::empty({input.size(0)}, input.options());
    
    const int batch_size = input.size(0);
    const int in_features = input.size(1);
    const int out_features = weight.size(0);
    const int pooled_size = out_features / pool_kernel_size;
    
    const int threads = 1024;
    dim3 blocks(batch_size);
    int shared_mem = (out_features + pooled_size) * sizeof(__half);
    
    fused_single_kernel_fp16<<<blocks, threads, shared_mem>>>(
        reinterpret_cast<const __half*>(input.data_ptr<at::Half>()),
        reinterpret_cast<const __half*>(weight.data_ptr<at::Half>()),
        bias.defined() ? reinterpret_cast<const __half*>(bias.data_ptr<at::Half>()) : nullptr,
        reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
        batch_size, in_features, out_features, pool_kernel_size, scale_factor
    );
    
    return output;
}

ext = load_inline(
    name='fused_ext',
    cpp_sources='''
torch::Tensor fused_op_fp32(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int pool_kernel_size, float scale_factor);
torch::Tensor fused_op_fp16(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int pool_kernel_size, float scale_factor);
''',
    cuda_sources=cuda_source,
    functions=['fused_op_fp32', 'fused_op_fp16'],
    verbose=False,
)

class Model(nn.Module):
    """
    A fused model implementing Matmul_AvgPool_GELU_Scale_Max with custom CUDA kernels.
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.avg_pool = nn.AvgPool1d(kernel_size=pool_kernel_size)
        self.scale_factor = scale_factor
        self.pool_kernel_size = pool_kernel_size
        self.in_features = in_features
        self.out_features = out_features
        
    def forward(self, x):
        if x.dtype == torch.float16:
            weight_half = self.matmul.weight.data.half()
            bias_half = self.matmul.bias.data.half() if self.matmul.bias is not None else torch.tensor([], dtype=torch.float16, device=x.device)
            return ext.fused_op_fp16(
                x, 
                weight_half, 
                bias_half,
                self.pool_kernel_size, 
                self.scale_factor
            )
        else:
            return ext.fused_op_fp32(
                x, 
                self.matmul.weight.data.float(), 
                self.matmul.bias.data.float() if self.matmul.bias is not None else torch.tensor([], dtype=torch.float32, device=x.device),
                self.pool_kernel_size, 
                self.scale_factor
            )
