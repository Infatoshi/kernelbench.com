import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Simplified fused Conv2d + BatchNorm + Scale kernel
// Each thread handles one output element (element-wise parallel)

__global__ void fused_conv_bn_scale_kernel(
    const float* __restrict__ input,      // (N, C_in, H, W)
    const float* __restrict__ weight,     // (C_out, C_in, K, K)
    const float* __restrict__ bias,       // (C_out) - can be nullptr
    const float* __restrict__ bn_weight,  // (C_out) - can be nullptr
    const float* __restrict__ bn_bias,    // (C_out) - can be nullptr
    const float* __restrict__ bn_mean,    // (C_out)
    const float* __restrict__ bn_var,     // (C_out)
    float eps,
    float scale,
    float* __restrict__ output,
    int N, int C_in, int H, int W,
    int C_out, int K, int H_out, int W_out,
    int pad_h, int pad_w, int stride_h, int stride_w
) {
    // Each thread handles one output element
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_out = N * C_out * H_out * W_out;
    
    if (idx >= total_out) return;
    
    // Decode output index
    int n = idx / (C_out * H_out * W_out);
    int rem = idx % (C_out * H_out * W_out);
    int c_out = rem / (H_out * W_out);
    rem = rem % (H_out * W_out);
    int out_h = rem / W_out;
    int out_w = rem % W_out;
    
    // Compute input position range
    int in_h_start = out_h * stride_h - pad_h;
    int in_w_start = out_w * stride_w - pad_w;
    
    // BN parameters
    float gamma = bn_weight ? bn_weight[c_out] : 1.0f;
    float beta = bn_bias ? bn_bias[c_out] : 0.0f;
    float mean = bn_mean ? bn_mean[c_out] : 0.0f;
    float var = bn_var ? bn_var[c_out] : 1.0f;
    float inv_std = rsqrtf(var + eps);
    
    // Folded bias
    float b = bias ? bias[c_out] : 0.0f;
    float folded_bias = ((b - mean) * gamma * inv_std + beta) * scale;
    
    // Accumulate
    float sum = 0.0f;
    
    int k_half = K / 2;
    
    for (int c_in = 0; c_in < C_in; c_in++) {
        for (int ki = 0; ki < K; ki++) {
            int in_h = in_h_start + ki;
            if (in_h < 0 || in_h >= H) continue;
            
            for (int kj = 0; kj < K; kj++) {
                int in_w = in_w_start + kj;
                if (in_w < 0 || in_w >= W) continue;
                
                // Weight index
                int w_idx = c_out * C_in * K * K + c_in * K * K + ki * K + kj;
                float w_val = weight[w_idx] * gamma * inv_std;
                
                // Input index
                int in_idx = n * C_in * H * W + c_in * H * W + in_h * W + in_w;
                sum += w_val * input[in_idx];
            }
        }
    }
    
    // Store
    int out_idx = n * C_out * H_out * W_out + c_out * H_out * W_out + out_h * W_out + out_w;
    output[out_idx] = sum + folded_bias;
}

torch::Tensor fused_conv_bn_scale(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    double eps,
    double scale,
    int pad_h,
    int pad_w,
    int stride_h,
    int stride_w
) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    
    int C_out = weight.size(0);
    int K = weight.size(2);
    
    int H_out = (H + 2 * pad_h - K) / stride_h + 1;
    int W_out = (W + 2 * pad_w - K) / stride_w + 1;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    int total_out = N * C_out * H_out * W_out;
    int block = 256;
    int grid = (total_out + block - 1) / block;
    
    fused_conv_bn_scale_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.numel() > 0 ? bias.data_ptr<float>() : nullptr,
        bn_weight.numel() > 0 ? bn_weight.data_ptr<float>() : nullptr,
        bn_bias.numel() > 0 ? bn_bias.data_ptr<float>() : nullptr,
        bn_mean.data_ptr<float>(),
        bn_var.data_ptr<float>(),
        (float)eps,
        (float)scale,
        output.data_ptr<float>(),
        N, C_in, H, W,
        C_out, K, H_out, W_out,
        pad_h, pad_w, stride_h, stride_w
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv_bn',
    cpp_sources='''
torch::Tensor fused_conv_bn_scale(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    double eps,
    double scale,
    int pad_h,
    int pad_w,
    int stride_h,
    int stride_w
);
''',
    cuda_sources=cuda_source,
    functions=['fused_conv_bn_scale'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        self.eps = 1e-5
        
    def forward(self, x):
        weight = self.conv.weight
        bias = self.conv.bias if self.conv.bias is not None else torch.zeros(self.conv.out_channels, device='cuda')
        bn_weight = self.bn.weight
        bn_bias = self.bn.bias
        bn_mean = self.bn.running_mean
        bn_var = self.bn.running_var
        
        return ext.fused_conv_bn_scale(
            x, weight, bias, 
            bn_weight, bn_bias, bn_mean, bn_var,
            self.eps, self.scaling_factor,
            self.conv.padding[0], self.conv.padding[1],
            self.conv.stride[0], self.conv.stride[1]
        )


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]


def get_init_inputs():
    return [3, 16, 3, 2.0]
