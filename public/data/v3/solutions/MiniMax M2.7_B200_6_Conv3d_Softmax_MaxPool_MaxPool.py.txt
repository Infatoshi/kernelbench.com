import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Softmax over channel dim (C=16), then two 2x2x2 max pools
// Output: (B, C, D/4, H/4, W/4)

__launch_bounds__(256, 4)
__global__ void softmax_maxpool_fp32_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int B, int C, int D, int H, int W,
    int outD, int outH, int outW
) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = B * C * outD * outH * outW;
    
    if (tid >= total_elements) return;
    
    // Decompose back to (b, c, od, oh, ow)
    int rem = tid;
    int b = rem / (C * outD * outH * outW);
    rem %= C * outD * outH * outW;
    int c = rem / (outD * outH * outW);
    rem %= outD * outH * outW;
    int od = rem / (outH * outW);
    rem %= outH * outW;
    int oh = rem / outW;
    int ow = rem % outW;
    
    // Input window for two 2x2x2 pools (stride 4)
    int id_start = od * 4;
    int ih_start = oh * 4;
    int iw_start = ow * 4;
    
    // Find local max for this channel across spatial window
    float local_max = -1e20f;
    
    for (int dd = 0; dd < 4 && id_start + dd < D; dd++) {
        int id = id_start + dd;
        for (int hh = 0; hh < 4 && ih_start + hh < H; hh++) {
            int ih = ih_start + hh;
            for (int ww = 0; ww < 4 && iw_start + ww < W; ww++) {
                int iw = iw_start + ww;
                int idx = ((b * C + c) * D + id) * H + ih;
                idx = idx * W + iw;
                float val = input[idx];
                if (val > local_max) local_max = val;
            }
        }
    }
    
    // Compute sum_exp at local max position
    float local_sum_exp = 0.0f;
    for (int dd = 0; dd < 4 && id_start + dd < D; dd++) {
        int id = id_start + dd;
        for (int hh = 0; hh < 4 && ih_start + hh < H; hh++) {
            int ih = ih_start + hh;
            for (int ww = 0; ww < 4 && iw_start + ww < W; ww++) {
                int iw = iw_start + ww;
                int idx = ((b * C + c) * D + id) * H + ih;
                idx = idx * W + iw;
                local_sum_exp += expf(input[idx] - local_max);
            }
        }
    }
    
    float result = 1.0f / local_sum_exp;
    output[tid] = result;
}

__launch_bounds__(256, 4)
__global__ void softmax_maxpool_bf16_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    int B, int C, int D, int H, int W,
    int outD, int outH, int outW
) {
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = B * C * outD * outH * outW;
    
    if (tid >= total_elements) return;
    
    // Decompose back to (b, c, od, oh, ow)
    int rem = tid;
    int b = rem / (C * outD * outH * outW);
    rem %= C * outD * outH * outW;
    int c = rem / (outD * outH * outW);
    rem %= outD * outH * outW;
    int od = rem / (outH * outW);
    rem %= outH * outW;
    int oh = rem / outW;
    int ow = rem % outW;
    
    // Input window for two 2x2x2 pools (stride 4)
    int id_start = od * 4;
    int ih_start = oh * 4;
    int iw_start = ow * 4;
    
    // Find local max for this channel across spatial window
    float local_max = -1e20f;
    
    for (int dd = 0; dd < 4 && id_start + dd < D; dd++) {
        int id = id_start + dd;
        for (int hh = 0; hh < 4 && ih_start + hh < H; hh++) {
            int ih = ih_start + hh;
            for (int ww = 0; ww < 4 && iw_start + ww < W; ww++) {
                int iw = iw_start + ww;
                int idx = ((b * C + c) * D + id) * H + ih;
                idx = idx * W + iw;
                float val = __half2float(input[idx]);
                if (val > local_max) local_max = val;
            }
        }
    }
    
    // Compute sum_exp at local max position
    float local_sum_exp = 0.0f;
    for (int dd = 0; dd < 4 && id_start + dd < D; dd++) {
        int id = id_start + dd;
        for (int hh = 0; hh < 4 && ih_start + hh < H; hh++) {
            int ih = ih_start + hh;
            for (int ww = 0; ww < 4 && iw_start + ww < W; ww++) {
                int iw = iw_start + ww;
                int idx = ((b * C + c) * D + id) * H + ih;
                idx = idx * W + iw;
                local_sum_exp += expf(__half2float(input[idx]) - local_max);
            }
        }
    }
    
    float result = 1.0f / local_sum_exp;
    output[tid] = __float2half(result);
}

torch::Tensor fused_softmax_maxpool(torch::Tensor input, int pool_kernel_size) {
    int B = input.size(0);
    int C = input.size(1);
    int D = input.size(2);
    int H = input.size(3);
    int W = input.size(4);
    
    // Two pools with kernel 2, stride 2 = equivalent to one pool with kernel 4, stride 4
    int outD = D / 4;
    int outH = H / 4;
    int outW = W / 4;
    
    auto output = torch::empty({B, C, outD, outH, outW}, input.options());
    
    int total_elements = B * C * outD * outH * outW;
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;
    
    if (input.dtype() == torch::kFloat32) {
        softmax_maxpool_fp32_kernel<<<blocks, threads>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            B, C, D, H, W, outD, outH, outW
        );
    } else {
        softmax_maxpool_bf16_kernel<<<blocks, threads>>>(
            input.data_ptr<__half>(),
            output.data_ptr<__half>(),
            B, C, D, H, W, outD, outH, outW
        );
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv3d_softmax_maxpool',
    cpp_sources='torch::Tensor fused_softmax_maxpool(torch::Tensor input, int pool_kernel_size);',
    cuda_sources=cuda_source,
    functions=['fused_softmax_maxpool'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size
        
    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_softmax_maxpool(x, self.pool_kernel_size)
        return x

def get_inputs():
    return [torch.randn(16, 3, 16, 128, 128)]

def get_init_inputs():
    return [3, 16, 3, 2]
