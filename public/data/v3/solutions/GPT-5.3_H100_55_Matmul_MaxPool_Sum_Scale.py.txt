import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ W,
    const float* __restrict__ bias,
    float* __restrict__ out,
    int B, int IN, int OUT, float scale)
{
    int b = blockIdx.x;
    if (b >= B) return;

    extern __shared__ float x_sh[];

    // load x[b]
    for (int i = threadIdx.x; i < IN; i += blockDim.x) {
        x_sh[i] = x[b * IN + i];
    }
    __syncthreads();

    int tid = threadIdx.x;
    int num_pairs = OUT / 2;

    float local_sum = 0.0f;

    for (int p = tid; p < num_pairs; p += blockDim.x) {
        int j0 = 2 * p;
        int j1 = j0 + 1;

        float s0 = 0.0f;
        float s1 = 0.0f;

        const float* w0 = W + j0 * IN;
        const float* w1 = W + j1 * IN;

        for (int k = 0; k < IN; ++k) {
            float xv = x_sh[k];
            s0 += xv * w0[k];
            s1 += xv * w1[k];
        }

        s0 += bias[j0];
        s1 += bias[j1];

        float m = s0 > s1 ? s0 : s1;
        local_sum += m;
    }

    // reduction
    __shared__ float red[256];
    red[tid] = local_sum;
    __syncthreads();

    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            red[tid] += red[tid + stride];
        }
        __syncthreads();
    }

    if (tid == 0) {
        out[b] = red[0] * scale;
    }
}


torch::Tensor fused_forward(torch::Tensor x, torch::Tensor W, torch::Tensor bias, float scale) {
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = W.size(0);

    auto out = torch::empty({B}, x.options());

    int threads = 256;
    size_t shmem = IN * sizeof(float);

    fused_kernel<<<B, threads, shmem>>>(
        x.data_ptr<float>(),
        W.data_ptr<float>(),
        bias.data_ptr<float>(),
        out.data_ptr<float>(),
        B, IN, OUT, scale
    );

    return out;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor W, torch::Tensor bias, float scale);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scale_factor = scale_factor

    def forward(self, x):
        return ext.fused_forward(x, self.matmul.weight, self.matmul.bias, self.scale_factor)


def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]


def get_init_inputs():
    return [4096, 4096, 2, 0.5]
