import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>
#include <float.h>

#define POOL_SIZE 2

__global__ void fused_conv_tanh_scale_bias_pool_fp32(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int OC, int K,
    float scale,
    int pool_size,
    int conv_H, int conv_W,
    int out_H, int out_W
) {
    int n = blockIdx.z;
    int oc = blockIdx.y;
    int out_y = blockIdx.x;
    
    if (n >= N || oc >= OC || out_y >= out_H) return;
    
    // Process output x positions
    for (int out_x = threadIdx.x; out_x < out_W; out_x += blockDim.x) {
        // Max pool over POOL_SIZE x POOL_SIZE region in conv output space
        float max_val = -FLT_MAX;
        
        for (int py = 0; py < pool_size; py++) {
            for (int px = 0; px < pool_size; px++) {
                int y_conv = out_y * pool_size + py;
                int x_conv = out_x * pool_size + px;
                
                if (y_conv >= conv_H || x_conv >= conv_W) continue;
                
                // Convolution at position (y_conv, x_conv)
                float conv_sum = 0.0f;
                #pragma unroll
                for (int ky = 0; ky < K; ky++) {
                    #pragma unroll
                    for (int kx = 0; kx < K; kx++) {
                        int y_in = y_conv + ky;
                        int x_in = x_conv + kx;
                        if (y_in < H && x_in < W) {
                            #pragma unroll
                            for (int c = 0; c < C; c++) {
                                int in_idx = ((n * C + c) * H + y_in) * W + x_in;
                                int w_idx = ((oc * C + c) * K + ky) * K + kx;
                                conv_sum += input[in_idx] * weight[w_idx];
                            }
                        }
                    }
                }
                
                // Tanh + Scale + Bias
                float val = tanhf(conv_sum) * scale + bias[oc];
                if (val > max_val) max_val = val;
            }
        }
        
        int out_idx = ((n * OC + oc) * out_H + out_y) * out_W + out_x;
        output[out_idx] = max_val;
    }
}

torch::Tensor fused_conv_tanh_scale_bias_pool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale,
    int pool_size
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int OC = weight.size(0);
    int K = weight.size(2);
    
    int conv_H = H - K + 1;
    int conv_W = W - K + 1;
    int out_H = conv_H / pool_size;
    int out_W = conv_W / pool_size;
    
    auto output = torch::empty({N, OC, out_H, out_W}, input.options());
    
    dim3 block(256);
    dim3 grid(out_H, OC, N);
    
    fused_conv_tanh_scale_bias_pool_fp32<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C, H, W, OC, K, scale, pool_size, conv_H, conv_W, out_H, out_W
    );
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_conv_tanh_scale_bias_pool(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale,
    int pool_size
);
"""

# Compile the extension
ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_tanh_scale_bias_pool'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
)


class Model(nn.Module):
    """
    Optimized fused Conv2d + Tanh + Scale + Bias + MaxPool model
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, padding=0)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size
        self.out_channels = out_channels
        self.bias_shape = bias_shape

    def forward(self, x):
        # Get conv weight and bias
        weight = self.conv.weight
        conv_bias = self.conv.bias
        
        # Apply convolution bias first if exists
        if conv_bias is not None:
            # Add conv bias to our bias parameter
            # Our bias is (out_channels, 1, 1), conv_bias is (out_channels,)
            fused_bias = self.bias.squeeze() + conv_bias
        else:
            fused_bias = self.bias.squeeze()
        
        # Call fused kernel
        result = ext.fused_conv_tanh_scale_bias_pool(
            x, weight, fused_bias, self.scaling_factor, self.pool_kernel_size
        )
        return result


def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    in_channels = 3
    out_channels = 16
    height, width = 256, 256
    kernel_size = 3
    scaling_factor = 2.0
    bias_shape = (out_channels, 1, 1)
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
