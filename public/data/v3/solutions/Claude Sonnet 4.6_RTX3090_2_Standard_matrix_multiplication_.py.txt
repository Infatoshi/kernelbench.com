import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>
#include <cuda_fp16.h>

using namespace nvcuda;

// Tile dimensions
#define BM 128
#define BN 128
#define BK 32
#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16
#define PAD 8

// Warp tile: each warp handles 64x64
#define WM 64
#define WN 64
#define WARPS_M (BM / WM)         // = 2
#define WARPS_N (BN / WN)         // = 2
#define WARP_TILES_M (WM / WMMA_M)  // = 4
#define WARP_TILES_N (WN / WMMA_N)  // = 4

__global__ void wmma_gemm_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K)
{
    // Double-buffered shared memory
    __shared__ half sA[2][BM][BK + PAD];
    __shared__ half sB[2][BK][BN + PAD];

    int tid = threadIdx.x;
    int warp_id = tid / 32;
    int warp_m = warp_id / WARPS_N;   // 0 or 1
    int warp_n = warp_id % WARPS_N;   // 0 or 1

    int block_row = blockIdx.y * BM;
    int block_col = blockIdx.x * BN;

    // Accumulator fragments: 4 x 4 WMMA tiles per warp
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc[WARP_TILES_M][WARP_TILES_N];
    #pragma unroll
    for (int i = 0; i < WARP_TILES_M; i++)
        #pragma unroll
        for (int j = 0; j < WARP_TILES_N; j++)
            wmma::fill_fragment(acc[i][j], 0.0f);

    int num_k_tiles = (K + BK - 1) / BK;

    for (int ktile = 0; ktile < num_k_tiles; ktile++) {
        int cur = ktile & 1;
        int k_start = ktile * BK;

        // Load current tile into cur buffer
        #pragma unroll
        for (int i = 0; i < (BM * BK) / 128; i++) {
            int flat = tid + i * 128;
            int r = flat / BK;
            int c = flat % BK;
            int gr = block_row + r;
            int gc = k_start + c;
            sA[cur][r][c] = (gr < M && gc < K) ? __float2half(A[gr * K + gc]) : __float2half(0.0f);
        }
        #pragma unroll
        for (int i = 0; i < (BK * BN) / 128; i++) {
            int flat = tid + i * 128;
            int r = flat / BN;
            int c = flat % BN;
            int gr = k_start + r;
            int gc = block_col + c;
            sB[cur][r][c] = (gr < K && gc < N) ? __float2half(B[gr * N + gc]) : __float2half(0.0f);
        }
        __syncthreads();

        // Compute on cur buffer
        #pragma unroll
        for (int k = 0; k < BK / WMMA_K; k++) {
            wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> a_frag[WARP_TILES_M];
            wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, half, wmma::row_major> b_frag[WARP_TILES_N];

            #pragma unroll
            for (int wi = 0; wi < WARP_TILES_M; wi++) {
                int row_off = warp_m * WM + wi * WMMA_M;
                wmma::load_matrix_sync(a_frag[wi],
                    (const half*)sA[cur][row_off] + k * WMMA_K,
                    BK + PAD);
            }
            #pragma unroll
            for (int wj = 0; wj < WARP_TILES_N; wj++) {
                int col_off = warp_n * WN + wj * WMMA_N;
                wmma::load_matrix_sync(b_frag[wj],
                    (const half*)sB[cur][k * WMMA_K] + col_off,
                    BN + PAD);
            }
            #pragma unroll
            for (int wi = 0; wi < WARP_TILES_M; wi++)
                #pragma unroll
                for (int wj = 0; wj < WARP_TILES_N; wj++)
                    wmma::mma_sync(acc[wi][wj], a_frag[wi], b_frag[wj], acc[wi][wj]);
        }

        __syncthreads();
    }

    // Store results to C
    #pragma unroll
    for (int wi = 0; wi < WARP_TILES_M; wi++) {
        #pragma unroll
        for (int wj = 0; wj < WARP_TILES_N; wj++) {
            int c_row = block_row + warp_m * WM + wi * WMMA_M;
            int c_col = block_col + warp_n * WN + wj * WMMA_N;
            if (c_row < M && c_col < N) {
                wmma::store_matrix_sync(
                    C + c_row * N + c_col,
                    acc[wi][wj],
                    N,
                    wmma::mem_row_major);
            }
        }
    }
}

torch::Tensor gemm_wmma(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.is_cuda() && B.is_cuda(), "Inputs must be CUDA tensors");
    TORCH_CHECK(A.dtype() == torch::kFloat32 && B.dtype() == torch::kFloat32,
                "Expected float32 inputs");
    int M = A.size(0), K = A.size(1), N = B.size(1);

    auto C = torch::zeros({M, N}, A.options());

    dim3 block(128);  // 4 warps
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);

    wmma_gemm_kernel<<<grid, block>>>(
        A.data_ptr<float>(), B.data_ptr<float>(), C.data_ptr<float>(),
        M, N, K);

    cudaError_t err = cudaGetLastError();
    TORCH_CHECK(err == cudaSuccess, "CUDA error: ", cudaGetErrorString(err));

    return C;
}
'''

ext = load_inline(
    name='wmma_gemm_ext2',
    cpp_sources='torch::Tensor gemm_wmma(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['gemm_wmma'],
    verbose=True,
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math', '-std=c++17'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemm_wmma(A, B)

M = 1024
K = 4096
N = 2048

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, N)
    return [A, B]

def get_init_inputs():
    return []
