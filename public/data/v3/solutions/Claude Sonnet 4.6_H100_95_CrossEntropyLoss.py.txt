cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>

// Warp-level primitives
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        float other = __shfl_down_sync(0xffffffff, val, offset);
        val = val > other ? val : other;
    }
    return val;
}

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Optimized cross entropy kernel:
// - One block per row
// - float4 vectorized loads
// - warp-shuffle reduction (no shared memory bank conflicts)
// - Returns per-row loss, then we do a single mean() on N elements
// Block size 256 -> 8 warps
// For C=1024: each thread handles 4 elements (1024/256=4), each thread also does float4 reads

template <int BLOCK_SIZE>
__global__ __launch_bounds__(BLOCK_SIZE, 4)
void cross_entropy_kernel_fp32(
    const float* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ loss_per_row,
    int N,
    int C
) {
    const int WARPS = BLOCK_SIZE / 32;
    int row = blockIdx.x;
    if (row >= N) return;
    
    const float* row_logits = logits + (long)row * C;
    int target = (int)targets[row];
    
    __shared__ float smem[WARPS];
    
    int warp_id = threadIdx.x >> 5;
    int lane_id = threadIdx.x & 31;
    int tid = threadIdx.x;
    
    // Step 1: Find max with float4 vectorized loads
    float local_max = -FLT_MAX;
    int C4 = C >> 2;
    const float4* row_logits4 = (const float4*)row_logits;
    
    #pragma unroll 2
    for (int j = tid; j < C4; j += BLOCK_SIZE) {
        float4 v = __ldg(&row_logits4[j]);
        float mx = fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w));
        local_max = fmaxf(local_max, mx);
    }
    // Handle remainder (C not divisible by 4)
    for (int j = (C4 << 2) + tid; j < C; j += BLOCK_SIZE) {
        local_max = fmaxf(local_max, __ldg(&row_logits[j]));
    }
    
    // Warp reduce max
    local_max = warp_reduce_max(local_max);
    if (lane_id == 0) smem[warp_id] = local_max;
    __syncthreads();
    
    // Final reduce across warps (in first warp)
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : -FLT_MAX;
        v = warp_reduce_max(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    float row_max = smem[0];
    __syncthreads();
    
    // Step 2: sum of exp(x - max) with float4
    float local_sum = 0.0f;
    
    #pragma unroll 2
    for (int j = tid; j < C4; j += BLOCK_SIZE) {
        float4 v = __ldg(&row_logits4[j]);
        local_sum += expf(v.x - row_max) + expf(v.y - row_max) 
                   + expf(v.z - row_max) + expf(v.w - row_max);
    }
    for (int j = (C4 << 2) + tid; j < C; j += BLOCK_SIZE) {
        local_sum += expf(__ldg(&row_logits[j]) - row_max);
    }
    
    // Warp reduce sum
    local_sum = warp_reduce_sum(local_sum);
    if (lane_id == 0) smem[warp_id] = local_sum;
    __syncthreads();
    
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : 0.0f;
        v = warp_reduce_sum(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    
    if (threadIdx.x == 0) {
        float log_sum_exp = logf(smem[0]) + row_max;
        loss_per_row[row] = log_sum_exp - __ldg(&row_logits[target]);
    }
}

// FP16 version with half2 vectorized loads
template <int BLOCK_SIZE>
__global__ __launch_bounds__(BLOCK_SIZE, 4)
void cross_entropy_kernel_fp16(
    const __half* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ loss_per_row,
    int N,
    int C
) {
    const int WARPS = BLOCK_SIZE / 32;
    int row = blockIdx.x;
    if (row >= N) return;
    
    const __half* row_logits = logits + (long)row * C;
    int target = (int)targets[row];
    
    __shared__ float smem[WARPS];
    
    int warp_id = threadIdx.x >> 5;
    int lane_id = threadIdx.x & 31;
    int tid = threadIdx.x;
    
    float local_max = -FLT_MAX;
    // Use half2 reads (2 elements at a time)
    int C2 = C >> 1;
    const __half2* row_logits2 = (const __half2*)row_logits;
    
    #pragma unroll 2
    for (int j = tid; j < C2; j += BLOCK_SIZE) {
        __half2 v = __ldg(&row_logits2[j]);
        float a = __half2float(v.x);
        float b = __half2float(v.y);
        local_max = fmaxf(local_max, fmaxf(a, b));
    }
    for (int j = (C2 << 1) + tid; j < C; j += BLOCK_SIZE) {
        local_max = fmaxf(local_max, __half2float(__ldg(&row_logits[j])));
    }
    
    local_max = warp_reduce_max(local_max);
    if (lane_id == 0) smem[warp_id] = local_max;
    __syncthreads();
    
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : -FLT_MAX;
        v = warp_reduce_max(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    float row_max = smem[0];
    __syncthreads();
    
    float local_sum = 0.0f;
    #pragma unroll 2
    for (int j = tid; j < C2; j += BLOCK_SIZE) {
        __half2 v = __ldg(&row_logits2[j]);
        local_sum += expf(__half2float(v.x) - row_max) + expf(__half2float(v.y) - row_max);
    }
    for (int j = (C2 << 1) + tid; j < C; j += BLOCK_SIZE) {
        local_sum += expf(__half2float(__ldg(&row_logits[j])) - row_max);
    }
    
    local_sum = warp_reduce_sum(local_sum);
    if (lane_id == 0) smem[warp_id] = local_sum;
    __syncthreads();
    
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : 0.0f;
        v = warp_reduce_sum(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    
    if (threadIdx.x == 0) {
        float log_sum_exp = logf(smem[0]) + row_max;
        loss_per_row[row] = log_sum_exp - __half2float(__ldg(&row_logits[target]));
    }
}

        local_sum += expf(__bfloat162float(v.x) - row_max) + expf(__bfloat162float(v.y) - row_max);
    }
    for (int j = (C2 << 1) + tid; j < C; j += BLOCK_SIZE) {

    if (threadIdx.x == 0) {

    }
}

// Fully fused: compute per-row losses AND their mean in a single kernel
// Uses two-level reduction: per-block loss -> atomicAdd into output
template <int BLOCK_SIZE>
__global__ __launch_bounds__(BLOCK_SIZE, 4)
void cross_entropy_fused_mean_fp32(
    const float* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ output,   // scalar output, pre-zeroed
    int N,
    int C,
    float inv_N
) {
    const int WARPS = BLOCK_SIZE / 32;
    int row = blockIdx.x;
    if (row >= N) return;
    
    const float* row_logits = logits + (long)row * C;
    int target = (int)targets[row];
    
    __shared__ float smem[WARPS];
    
    int warp_id = threadIdx.x >> 5;
    int lane_id = threadIdx.x & 31;
    int tid = threadIdx.x;
    
    float local_max = -FLT_MAX;
    int C4 = C >> 2;
    const float4* row_logits4 = (const float4*)row_logits;
    
    #pragma unroll 2
    for (int j = tid; j < C4; j += BLOCK_SIZE) {
        float4 v = __ldg(&row_logits4[j]);
        float mx = fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w));
        local_max = fmaxf(local_max, mx);
    }
    for (int j = (C4 << 2) + tid; j < C; j += BLOCK_SIZE) {
        local_max = fmaxf(local_max, __ldg(&row_logits[j]));
    }
    
    local_max = warp_reduce_max(local_max);
    if (lane_id == 0) smem[warp_id] = local_max;
    __syncthreads();
    
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : -FLT_MAX;
        v = warp_reduce_max(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    float row_max = smem[0];
    __syncthreads();
    
    float local_sum = 0.0f;
    #pragma unroll 2
    for (int j = tid; j < C4; j += BLOCK_SIZE) {
        float4 v = __ldg(&row_logits4[j]);
        local_sum += expf(v.x - row_max) + expf(v.y - row_max) 
                   + expf(v.z - row_max) + expf(v.w - row_max);
    }
    for (int j = (C4 << 2) + tid; j < C; j += BLOCK_SIZE) {
        local_sum += expf(__ldg(&row_logits[j]) - row_max);
    }
    
    local_sum = warp_reduce_sum(local_sum);
    if (lane_id == 0) smem[warp_id] = local_sum;
    __syncthreads();
    
    if (warp_id == 0) {
        float v = (lane_id < WARPS) ? smem[lane_id] : 0.0f;
        v = warp_reduce_sum(v);
        if (lane_id == 0) smem[0] = v;
    }
    __syncthreads();
    
    if (threadIdx.x == 0) {
        float row_loss = (logf(smem[0]) + row_max - __ldg(&row_logits[target])) * inv_N;
        atomicAdd(output, row_loss);
    }
}

torch::Tensor cross_entropy_forward(torch::Tensor logits, torch::Tensor targets) {
    int N = logits.size(0);
    int C = logits.size(1);
    
    const int BLOCK_SIZE = 256;
    dim3 grid(N);
    dim3 block(BLOCK_SIZE);
    
    // For fp32 with compatible C, use fused kernel
    if (logits.scalar_type() == torch::kFloat32 && (C % 4 == 0)) {
        auto output = torch::zeros({1}, logits.options().dtype(torch::kFloat32));
        float inv_N = 1.0f / (float)N;
        cross_entropy_fused_mean_fp32<BLOCK_SIZE><<<grid, block>>>(
            logits.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            output.data_ptr<float>(),
            N, C, inv_N
        );
        return output.squeeze(0);
    }
    
    auto loss_per_row = torch::empty({N}, logits.options().dtype(torch::kFloat32));
    
    if (logits.scalar_type() == torch::kFloat32) {
        cross_entropy_kernel_fp32<BLOCK_SIZE><<<grid, block>>>(
            logits.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            loss_per_row.data_ptr<float>(),
            N, C
        );
    } else if (logits.scalar_type() == torch::kFloat16) {
        cross_entropy_kernel_fp16<BLOCK_SIZE><<<grid, block>>>(
            (__half*)logits.data_ptr(),
            targets.data_ptr<int64_t>(),
            loss_per_row.data_ptr<float>(),
            N, C
        );
    } else if (logits.scalar_type() == torch::kBFloat16) {
        cross_entropy_kernel_bf16<BLOCK_SIZE><<<grid, block>>>(
            (__nv_bfloat16*)logits.data_ptr(),
            targets.data_ptr<int64_t>(),
            loss_per_row.data_ptr<float>(),
            N, C
        );
    } else {
        AT_ERROR("Unsupported dtype");
    }
    
    return loss_per_row.mean();
}

class Model(nn.Module):
    """
    A model that computes Cross Entropy Loss for multi-class classification tasks.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets)
