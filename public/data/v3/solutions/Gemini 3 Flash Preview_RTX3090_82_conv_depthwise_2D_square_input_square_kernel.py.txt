import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def depthwise_conv2d_kernel(
    x_ptr, weight_ptr, bias_ptr, y_ptr,
    batch_size, channels, in_h, in_w,
    kernel_size_v, stride_s, padding_p,
    out_h, out_w,
    stride_bn, stride_cn, stride_hn, stride_wn,
    stride_weight_c, stride_weight_h, stride_weight_w,
    stride_bout, stride_cout, stride_hout, stride_wout,
    HAS_BIAS: tl.constexpr,
    BLOCK_H: tl.constexpr, BLOCK_W: tl.constexpr
):
    pid = tl.program_id(0)
    num_h_blocks = tl.cdiv(out_h, BLOCK_H)
    num_w_blocks = tl.cdiv(out_w, BLOCK_W)
    
    # Grid is (batch * channels * num_h_blocks * num_w_blocks)
    pid_batch = pid // (channels * num_h_blocks * num_w_blocks)
    rem = pid % (channels * num_h_blocks * num_w_blocks)
    pid_channel = rem // (num_h_blocks * num_w_blocks)
    rem_hw = rem % (num_h_blocks * num_w_blocks)
    pid_h = rem_hw // num_w_blocks
    pid_w = rem_hw % num_w_blocks

    off_h = pid_h * BLOCK_H + tl.arange(0, BLOCK_H)
    off_w = pid_w * BLOCK_W + tl.arange(0, BLOCK_W)
    
    acc = tl.zeros((BLOCK_H, BLOCK_W), dtype=tl.float32)
    
    weight_base_ptr = weight_ptr + pid_channel * stride_weight_c
    x_base_ptr = x_ptr + pid_batch * stride_bn + pid_channel * stride_cn

    for kh in range(kernel_size_v):
        for kw in range(kernel_size_v):
            ih = off_h * stride_s + kh - padding_p
            iw = off_w * stride_s + kw - padding_p
            
            mask_in = (off_h < out_h)[:, None] & (off_w < out_w)[None, :] & \
                      (ih >= 0)[:, None] & (ih < in_h)[:, None] & \
                      (iw >= 0)[None, :] & (iw < in_w)[None, :]
            
            weight_val = tl.load(weight_base_ptr + kh * stride_weight_h + kw * stride_weight_w)
            
            in_idx = ih[:, None] * stride_hn + iw[None, :] * stride_wn
            curr_x = tl.load(x_base_ptr + in_idx, mask=mask_in, other=0.0)
            
            acc += curr_x * weight_val

    if HAS_BIAS:
        bias_val = tl.load(bias_ptr + pid_channel)
        acc += bias_val
    
    mask_out = (off_h < out_h)[:, None] & (off_w < out_w)[None, :]
    out_idx = (pid_batch * stride_bout + 
               pid_channel * stride_cout + 
               off_h[:, None] * stride_hout + 
               off_w[None, :] * stride_wout)
    tl.store(y_ptr + out_idx, acc.to(y_ptr.dtype.element_ty), mask=mask_out)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Use groups=in_channels for depthwise
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, channels, in_h, in_w = x.shape
        out_h = (in_h + 2 * self.padding - self.kernel_size) // self.stride + 1
        out_w = (in_w + 2 * self.padding - self.kernel_size) // self.stride + 1
        
        y = torch.empty((batch_size, channels, out_h, out_w), device=x.device, dtype=x.dtype)
        
        BLOCK_H = 16
        BLOCK_W = 16
        grid = (batch_size * channels * triton.cdiv(out_h, BLOCK_H) * triton.cdiv(out_w, BLOCK_W),)
        
        # In nn.Conv2d with groups=in_channels, weight shape is (in_channels, 1, k, k)
        weight = self.conv2d.weight
        bias = self.conv2d.bias
        
        depthwise_conv2d_kernel[grid](
            x, weight, bias, y,
            batch_size, channels, in_h, in_w,
            self.kernel_size, self.stride, self.padding,
            out_h, out_w,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            weight.stride(0), weight.stride(2), weight.stride(3),
            y.stride(0), y.stride(1), y.stride(2), y.stride(3),
            HAS_BIAS=(bias is not None),
            BLOCK_H=BLOCK_H, BLOCK_W=BLOCK_W
        )
        return y

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 3, 1, 0]
