import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <cmath>

// Warp-level reduce max
__device__ __forceinline__ float warp_reduce_max(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

// Warp-level reduce sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Kernel for computing cross entropy loss
// Each block processes one sample (row)
__global__ void cross_entropy_kernel(
    const float* __restrict__ predictions,
    const int64_t* __restrict__ targets,
    float* __restrict__ loss,
    int num_classes
) {
    int batch_idx = blockIdx.x;
    int tid = threadIdx.x;
    int lane_id = tid & 31;
    int warp_id = tid >> 5;
    int num_warps = blockDim.x >> 5;
    
    const float* row = predictions + batch_idx * num_classes;
    int target_class = static_cast<int>(targets[batch_idx]);
    
    // Step 1: Find max for numerical stability
    float max_val = -FLT_MAX;
    for (int i = tid; i < num_classes; i += blockDim.x) {
        max_val = fmaxf(max_val, row[i]);
    }
    
    // Warp-level reduce max
    max_val = warp_reduce_max(max_val);
    
    // Broadcast max_val from lane 0 to all threads in warp
    max_val = __shfl_sync(0xffffffff, max_val, 0);
    
    // Cross-warp reduce max using shared memory
    __shared__ float shared_max[32];
    if (lane_id == 0) {
        shared_max[warp_id] = max_val;
    }
    __syncthreads();
    
    // First warp does final reduction
    if (warp_id == 0) {
        max_val = (lane_id < num_warps) ? shared_max[lane_id] : -FLT_MAX;
        max_val = warp_reduce_max(max_val);
        if (lane_id == 0) {
            shared_max[0] = max_val;
        }
    }
    __syncthreads();
    max_val = shared_max[0];
    
    // Step 2: Compute sum of exp(val - max) and get target prediction
    float sum_exp = 0.0f;
    float target_pred = -FLT_MAX;  // Initialize to invalid value
    
    for (int i = tid; i < num_classes; i += blockDim.x) {
        float val = row[i];
        float exp_val = expf(val - max_val);
        sum_exp += exp_val;
        if (i == target_class) {
            target_pred = val;
        }
    }
    
    // Warp-level reduce sum
    sum_exp = warp_reduce_sum(sum_exp);
    
    // Warp-level reduce max for target_pred (only one thread has valid value)
    target_pred = warp_reduce_max(target_pred);
    
    // Broadcast from lane 0
    sum_exp = __shfl_sync(0xffffffff, sum_exp, 0);
    target_pred = __shfl_sync(0xffffffff, target_pred, 0);
    
    // Cross-warp reduce using shared memory
    __shared__ float shared_sum[32];
    __shared__ float shared_target[32];
    if (lane_id == 0) {
        shared_sum[warp_id] = sum_exp;
        shared_target[warp_id] = target_pred;
    }
    __syncthreads();
    
    // First warp does final reduction
    if (warp_id == 0) {
        sum_exp = (lane_id < num_warps) ? shared_sum[lane_id] : 0.0f;
        target_pred = (lane_id < num_warps) ? shared_target[lane_id] : -FLT_MAX;
        
        sum_exp = warp_reduce_sum(sum_exp);
        target_pred = warp_reduce_max(target_pred);
        
        if (lane_id == 0) {
            // Compute loss: -log(softmax[target])
            // = -target_pred + max_val + log(sum_exp)
            float log_sum_exp = logf(sum_exp);
            loss[batch_idx] = log_sum_exp + max_val - target_pred;
        }
    }
}

// Kernel to compute mean of losses
__global__ void reduce_mean_kernel(
    const float* __restrict__ losses,
    float* __restrict__ output,
    int n
) {
    float sum = 0.0f;
    for (int i = blockIdx.x * blockDim.x + threadIdx.x; i < n; i += blockDim.x * gridDim.x) {
        sum += losses[i];
    }
    
    // Block-level reduction
    __shared__ float shared_sum[256];
    shared_sum[threadIdx.x] = sum;
    __syncthreads();
    
    for (int offset = blockDim.x / 2; offset > 0; offset /= 2) {
        if (threadIdx.x < offset) {
            shared_sum[threadIdx.x] += shared_sum[threadIdx.x + offset];
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        atomicAdd(output, shared_sum[0] / n);
    }
}

torch::Tensor cross_entropy_op(torch::Tensor predictions, torch::Tensor targets) {
    // predictions: [batch_size, num_classes]
    // targets: [batch_size]
    
    int batch_size = predictions.size(0);
    int num_classes = predictions.size(1);
    
    // Create output tensors
    auto losses = torch::empty({batch_size}, predictions.options().dtype(torch::kFloat32));
    auto output = torch::zeros({1}, predictions.options().dtype(torch::kFloat32));
    
    // Launch kernel for computing per-sample losses
    int threads = min(1024, ((num_classes + 31) / 32) * 32);
    int blocks = batch_size;
    
    cross_entropy_kernel<<<blocks, threads>>>(
        predictions.data_ptr<float>(),
        targets.data_ptr<int64_t>(),
        losses.data_ptr<float>(),
        num_classes
    );
    
    // Launch kernel for computing mean
    int reduce_threads = 256;
    int reduce_blocks = min(256, (batch_size + reduce_threads - 1) / reduce_threads);
    
    reduce_mean_kernel<<<reduce_blocks, reduce_threads>>>(
        losses.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor cross_entropy_op(torch::Tensor predictions, torch::Tensor targets);
'''

ext = load_inline(
    name='cross_entropy_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_op'],
    verbose=True,
    extra_cuda_cflags=['-O3', '-arch=sm_100'],
)

class Model(nn.Module):
    """
    A model that computes Cross Entropy Loss for multi-class classification tasks.
    Optimized CUDA implementation for B200.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        # Ensure float32 for computation (handles fp16/bf16 inputs)
        predictions_float = predictions.float()
        return ext.cross_entropy_op(predictions_float, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [torch.randn(batch_size, num_classes, device='cuda'), 
            torch.randint(0, num_classes, (batch_size,), device='cuda')]

def get_init_inputs():
    return []
