"""
Optimized BF16 GEMM using CUTLASS 3.x with WGMMA for H100.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cutlass/cutlass.h>
#include <cutlass/gemm/dispatch.hpp>
#include <cutlass/gemm/device/gemm_universal_adapter.h>
#include <cutlass/gemm/kernel/default_gemm_universal.hpp>
#include <cutlass/gemm/collective/collective_builder.hpp>
#include <cutlass/epilogue/collective/default_epilogue.hpp>
#include <cutlass/epilogue/thread/linear_combination.h>
#include <cutlass/util/packed_stride.hpp>

using namespace cute;

// Define BF16 GEMM types
using ElementA = cutlass::bfloat16_t;
using ElementB = cutlass::bfloat16_t;
using ElementC = cutlass::bfloat16_t;
using ElementAccumulator = float;
using ElementCompute = float;

// Define layouts
using LayoutA = cutlass::layout::RowMajor;
using LayoutB = cutlass::layout::ColumnMajor;
using LayoutC = cutlass::layout::RowMajor;

// Use the collective builder for Hopper architecture
using CollectiveOp = typename cutlass::gemm::collective::CollectiveBuilder<
    cutlass::arch::Sm90,                      // Architecture
    cutlass::arch::OpClassTensorOp,           // Operator class
    ElementA, LayoutA, 16,                    // ElementA, LayoutA, AlignmentA
    ElementB, LayoutB, 16,                    // ElementB, LayoutB, AlignmentB
    ElementAccumulator,                       // ElementAccumulator
    cutlass::gemm::KernelTmaWarpSpecializedPingpong,  // Kernel schedule
    128, 128, 64,                             // Tile shape M, N, K
    2                                         // Pipeline stages
>::CollectiveOp;

using EpilogueOp = cutlass::epilogue::collective::DefaultEpilogue<
    cutlass::gemm::TagToStrideC<LayoutC>::StrideType,
    cutlass::gemm::TagToStrideC<LayoutC>::StrideType,
    cutlass::epilogue::thread::LinearCombination<
        ElementC, 1, ElementCompute, ElementCompute>
>;

using GemmKernel = cutlass::gemm::kernel::GemmUniversal<
    cutlass::gemm::GemmIdentity<cutlass::gemm::GemmShape<128, 128, 64>>,
    CollectiveOp,
    EpilogueOp
>;

using Gemm = cutlass::gemm::device::GemmUniversalAdapter<GemmKernel>;

torch::Tensor gemm_bf16(torch::Tensor a, torch::Tensor b) {
    int M = a.size(0);
    int K = a.size(1);
    int N = b.size(1);
    
    auto c = torch::empty({M, N}, torch::TensorOptions()
        .dtype(torch::kBFloat16).device(torch::kCUDA));
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    typename Gemm::Arguments args{
        cutlass::gemm::GemmUniversalMode::kGemm,
        {M, N, K, 1},
        {a.data_ptr<cutlass::bfloat16_t>(), a.stride(0)},
        {b.data_ptr<cutlass::bfloat16_t>(), b.stride(0)},
        {c.data_ptr<cutlass::bfloat16_t>(), c.stride(0)},
        {c.data_ptr<cutlass::bfloat16_t>(), c.stride(0)},
        {alpha, beta}
    };
    
    Gemm gemm_op;
    size_t workspace_size = gemm_op.get_workspace_size(args);
    auto workspace = torch::empty({(int64_t)workspace_size}, 
        torch::TensorOptions().dtype(torch::kUInt8).device(torch::kCUDA));
    
    auto status = gemm_op(args, workspace.data_ptr());
    
    if (status != cutlass::Status::kSuccess) {
        AT_ERROR("CUTLASS GEMM failed with status ", (int)status);
    }
    
    return c;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor gemm_bf16(torch::Tensor a, torch::Tensor b);
'''

ext = load_inline(
    name='gemm_bf16_cutlass',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_bf16'],
    extra_cuda_cflags=[
        '-arch=sm_90',
        '-I/opt/cutlass/include',
        '-std=c++17',
        '-DCUTLASS_ENABLE_DEBUG_ASSERTIONS=0',
    ],
    verbose=True,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        return ext.gemm_bf16(a, b)

def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a = torch.randn(m, k, dtype=torch.bfloat16)
    b = torch.randn(k, n, dtype=torch.bfloat16)
    return [a, b]

def get_init_inputs():
    return []
