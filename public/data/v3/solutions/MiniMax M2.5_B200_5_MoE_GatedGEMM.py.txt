OP_TYPE = "fused"

# CUDA kernel source for MoE Gated GEMM with fused operations
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <algorithm>

// SiLU activation: x * sigmoid(x)
__device__ __forceinline__ float silu(float x) {
    return x * (1.0f / (1.0f + expf(-x)));
}

__device__ __forceinline__ float2 silu2(float2 x) {
    float2 sig;
    sig.x = 1.0f / (1.0f + expf(-x.x));
    sig.y = 1.0f / (1.0f + expf(-x.y));
    return {x.x * sig.x, x.y * sig.y};
}

__device__ __forceinline__ float4 silu4(float4 x) {
    float4 sig;
    sig.x = 1.0f / (1.0f + expf(-x.x));
    sig.y = 1.0f / (1.0f + expf(-x.y));
    sig.z = 1.0f / (1.0f + expf(-x.z));
    sig.w = 1.0f / (1.0f + expf(-x.w));
    return {x.x * sig.x, x.y * sig.y, x.z * sig.z, x.w * sig.w};
}

template<typename T>
__device__ __forceinline__ T relu(T x) {
    return x > T(0) ? x : T(0);
}

// Vectorized load for memory coalescing
template<typename T>
__device__ __forceinline__ void load_vector(T* dst, const T* src, int n) {
    int idx = threadIdx.x;
    int stride = blockDim.x;
    for (int i = idx; i < n; i += stride) {
        dst[i] = src[i];
    }
}

// Fused MoE kernel: computes gate_proj, up_proj, SiLU, elementwise mul, down_proj
// Each block processes tokens assigned to one expert
template<int BLOCK_M, int BLOCK_K, int BLOCK_N>
__global__ void moe_fused_kernel(
    const float* __restrict__ x,           // (num_tokens, hidden_size)
    const float* __restrict__ gate_weight, // (intermediate_size, hidden_size)
    const float* __restrict__ up_weight,   // (intermediate_size, hidden_size)  
    const float* __restrict__ down_weight, // (hidden_size, intermediate_size)
    const int* __restrict__ token_ids,     // (num_selected_tokens)
    const float* __restrict__ weights,     // (num_selected_tokens)
    float* __restrict__ output,            // (num_tokens, hidden_size)
    int num_selected_tokens,
    int hidden_size,
    int intermediate_size
) {
    // Shared memory for tiles
    __shared__ float a_tile[BLOCK_M * BLOCK_K];   // gate/up input
    __shared__ float b_gate[BLOCK_K * BLOCK_N];  // gate weight
    __shared__ float b_up[BLOCK_K * BLOCK_N];    // up weight
    __shared__ float c_tile[BLOCK_M * BLOCK_N];   // output
    __shared__ float d_tile[BLOCK_M * BLOCK_N];  // intermediate
    
    // Each block handles one expert's tokens
    const int expert_start = blockIdx.x;
    const int num_experts = gridDim.x;
    
    // Find this expert's token range (simple stride pattern)
    const int expert_id = expert_start;
    const int tokens_per_expert = (num_selected_tokens + num_experts - 1) / num_experts;
    const int expert_token_start = expert_id * tokens_per_expert;
    const int expert_token_end = min(expert_token_start + tokens_per_expert, num_selected_tokens);
    const int num_expert_tokens = expert_token_end - expert_token_start;
    
    if (num_expert_tokens == 0) return;
    
    const int M = num_expert_tokens;
    const int K = hidden_size;
    const int N = intermediate_size;
    
    // Compute gate: gate = SiLU(x @ gate_weight.T)
    // Output shape: (M, N)
    for (int mo = 0; mo < M; mo += BLOCK_M) {
        int mo_eff = min(BLOCK_M, M - mo);
        
        // Load input tile
        for (int i = threadIdx.x; i < mo_eff * K; i += blockDim.x) {
            int row = i / K;
            int col = i % K;
            int token_idx = token_ids[expert_token_start + mo + row];
            a_tile[row * K + col] = x[token_idx * K + col];
        }
        __syncthreads();
        
        // Gemm: gate = a_tile @ gate_weight
        for (int no = 0; no < N; no += BLOCK_N) {
            int no_eff = min(BLOCK_N, N - no);
            
            // Load weight tile
            for (int i = threadIdx.x; i < K * no_eff; i += blockDim.x) {
                int row = i / no_eff;
                int col = i % no_eff;
                b_gate[col * K + row] = gate_weight[(no + col) * K + row];
            }
            __syncthreads();
            
            // Initialize output
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                c_tile[i] = 0.0f;
            }
            __syncthreads();
            
            // GEMM
            for (int ko = 0; ko < K; ko += BLOCK_K) {
                // Load a tile
                for (int i = threadIdx.x; i < mo_eff * BLOCK_K; i += blockDim.x) {
                    int row = i / BLOCK_K;
                    int col = i % BLOCK_K;
                    if (ko + col < K) {
                        a_tile[row * BLOCK_K + col] = a_tile[row * K + (ko + col)];
                    }
                }
                __syncthreads();
                
                // Load b tile  
                for (int i = threadIdx.x; i < BLOCK_K * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    if (ko + row < K) {
                        b_gate[col * BLOCK_K + row] = b_gate[(ko + row) * no_eff + col];
                    }
                }
                __syncthreads();
                
                // Compute
                for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    float sum = 0.0f;
                    #pragma unroll
                    for (int k = 0; k < BLOCK_K; k++) {
                        sum += a_tile[row * BLOCK_K + k] * b_gate[col * BLOCK_K + k];
                    }
                    c_tile[row * no_eff + col] += sum;
                }
                __syncthreads();
            }
            
            // Apply SiLU activation
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                c_tile[i] = silu(c_tile[i]);
            }
            __syncthreads();
            
            // Store intermediate for up projection later
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                d_tile[i] = c_tile[i];
            }
            __syncthreads();
        }
        
        // Compute up: up = x @ up_weight.T, then mul with gate
        for (int no = 0; no < N; no += BLOCK_N) {
            int no_eff = min(BLOCK_N, N - no);
            
            // Load weight tile
            for (int i = threadIdx.x; i < K * no_eff; i += blockDim.x) {
                int row = i / no_eff;
                int col = i % no_eff;
                b_up[col * K + row] = up_weight[(no + col) * K + row];
            }
            __syncthreads();
            
            // Initialize output
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                c_tile[i] = 0.0f;
            }
            __syncthreads();
            
            // GEMM for up projection
            for (int ko = 0; ko < K; ko += BLOCK_K) {
                // Load a tile
                for (int i = threadIdx.x; i < mo_eff * BLOCK_K; i += blockDim.x) {
                    int row = i / BLOCK_K;
                    int col = i % BLOCK_K;
                    if (ko + col < K) {
                        a_tile[row * BLOCK_K + col] = a_tile[row * K + (ko + col)];
                    }
                }
                __syncthreads();
                
                // Load b tile
                for (int i = threadIdx.x; i < BLOCK_K * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    if (ko + row < K) {
                        b_up[col * BLOCK_K + row] = b_up[(ko + row) * no_eff + col];
                    }
                }
                __syncthreads();
                
                // Compute
                for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    float sum = 0.0f;
                    #pragma unroll
                    for (int k = 0; k < BLOCK_K; k++) {
                        sum += a_tile[row * BLOCK_K + k] * b_up[col * BLOCK_K + k];
                    }
                    c_tile[row * no_eff + col] += sum;
                }
                __syncthreads();
            }
            
            // Elementwise multiply with gate (stored in d_tile)
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                c_tile[i] = c_tile[i] * d_tile[i];
            }
            __syncthreads();
            
            // Store intermediate result
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                d_tile[i] = c_tile[i];
            }
            __syncthreads();
        }
        
        // Down projection: output = intermediate @ down_weight
        for (int no = 0; no < hidden_size; no += BLOCK_N) {
            int no_eff = min(BLOCK_N, hidden_size - no);
            
            // Load weight tile
            for (int i = threadIdx.x; i < N * no_eff; i += blockDim.x) {
                int row = i / no_eff;
                int col = i % no_eff;
                b_gate[col * N + row] = down_weight[(no + col) * N + row];
            }
            __syncthreads();
            
            // Initialize output
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                c_tile[i] = 0.0f;
            }
            __syncthreads();
            
            // GEMM
            for (int ko = 0; ko < N; ko += BLOCK_K) {
                // Load a tile
                for (int i = threadIdx.x; i < mo_eff * BLOCK_K; i += blockDim.x) {
                    int row = i / BLOCK_K;
                    int col = i % BLOCK_K;
                    if (ko + col < N) {
                        a_tile[row * BLOCK_K + col] = d_tile[row * N + (ko + col)];
                    }
                }
                __syncthreads();
                
                // Load b tile
                for (int i = threadIdx.x; i < BLOCK_K * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    if (ko + row < N) {
                        b_gate[col * BLOCK_K + row] = b_gate[(ko + row) * no_eff + col];
                    }
                }
                __syncthreads();
                
                // Compute
                for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                    int row = i / no_eff;
                    int col = i % no_eff;
                    float sum = 0.0f;
                    #pragma unroll
                    for (int k = 0; k < BLOCK_K; k++) {
                        sum += a_tile[row * BLOCK_K + k] * b_gate[col * BLOCK_K + k];
                    }
                    c_tile[row * no_eff + col] += sum;
                }
                __syncthreads();
            }
            
            // Apply weights and write output
            for (int i = threadIdx.x; i < mo_eff * no_eff; i += blockDim.x) {
                int row = i / no_eff;
                int col = i % no_eff;
                int token_idx = token_ids[expert_token_start + mo + row];
                float weight = weights[expert_token_start + mo + row];
                atomicAdd(&output[token_idx * hidden_size + no + col], c_tile[row * no_eff + col] * weight);
            }
            __syncthreads();
        }
    }
}

// Simpler and more robust kernel using cublas
__global__ void moe_fused_simple_kernel(
    const float* __restrict__ x,
    const float* __restrict__ gate_weight,
    const float* __restrict__ up_weight,
    const float* __restrict__ down_weight,
    const int* __restrict__ token_ids,
    const float* __restrict__ weights,
    float* __restrict__ output,
    int num_selected_tokens,
    int hidden_size,
    int intermediate_size,
    int num_experts
) {
    // Each expert gets a contiguous block of tokens
    const int expert_id = blockIdx.x;
    const int tokens_per_expert = (num_selected_tokens + num_experts - 1) / num_experts;
    const int expert_token_start = expert_id * tokens_per_expert;
    const int expert_token_end = min(expert_token_start + tokens_per_expert, num_selected_tokens);
    
    if (expert_token_start >= num_selected_tokens) return;
    
    const int num_tokens = expert_token_end - expert_token_start;
    if (num_tokens == 0) return;
    
    extern __shared__ float shared_mem[];
    float* gate_out = shared_mem;
    float* up_out = shared_mem + num_tokens * intermediate_size;
    
    // Shared memory for intermediate
    const int max_intermediate = intermediate_size > hidden_size ? intermediate_size : hidden_size;
    
    // Process tokens for this expert using global memory (simplified for correctness)
    for (int local_idx = threadIdx.x; local_idx < num_tokens; local_idx += blockDim.x) {
        const int global_idx = expert_token_start + local_idx;
        const int token_id = token_ids[global_idx];
        const float weight = weights[global_idx];
        
        // Gate projection: gate = SiLU(x @ gate_weight.T)
        // Each thread computes one element of gate output
        for (int j = 0; j < intermediate_size; j++) {
            float sum = 0.0f;
            for (int k = 0; k < hidden_size; k++) {
                sum += x[token_id * hidden_size + k] * gate_weight[j * hidden_size + k];
            }
            gate_out[local_idx * intermediate_size + j] = sum;
        }
        
        // SiLU activation
        for (int j = 0; j < intermediate_size; j++) {
            float val = gate_out[local_idx * intermediate_size + j];
            gate_out[local_idx * intermediate_size + j] = val / (1.0f + expf(-val));
        }
        
        // Up projection: up = x @ up_weight.T
        for (int j = 0; j < intermediate_size; j++) {
            float sum = 0.0f;
            for (int k = 0; k < hidden_size; k++) {
                sum += x[token_id * hidden_size + k] * up_weight[j * hidden_size + k];
            }
            up_out[local_idx * intermediate_size + j] = sum;
        }
        
        // Elementwise multiply
        for (int j = 0; j < intermediate_size; j++) {
            up_out[local_idx * intermediate_size + j] *= gate_out[local_idx * intermediate_size + j];
        }
        
        // Down projection: output = intermediate @ down_weight
        for (int j = 0; j < hidden_size; j++) {
            float sum = 0.0f;
            for (int k = 0; k < intermediate_size; k++) {
                sum += up_out[local_idx * intermediate_size + k] * down_weight[j * intermediate_size + k];
            }
            // Atomic add for weight accumulation
            atomicAdd(&output[token_id * hidden_size + j], sum * weight);
        }
    }
}

// Optimized kernel with better memory access patterns
__global__ void moe_fused_optimized_kernel(
    const float* __restrict__ x,
    const float* __restrict__ gate_weight,
    const float* __restrict__ up_weight,
    const float* __restrict__ down_weight,
    const int* __restrict__ token_ids,
    const float* __restrict__ weights,
    float* __restrict__ output,
    int num_selected_tokens,
    int hidden_size,
    int intermediate_size,
    int num_experts
) {
    const int expert_id = blockIdx.x;
    const int tokens_per_expert = (num_selected_tokens + num_experts - 1) / num_experts;
    const int expert_token_start = expert_id * tokens_per_expert;
    const int expert_token_end = min(expert_token_start + tokens_per_expert, num_selected_tokens);
    
    if (expert_token_start >= num_selected_tokens) return;
    
    const int num_tokens = expert_token_end - expert_token_start;
    if (num_tokens == 0) return;
    
    // Each warp processes one token
    const int warp_id = threadIdx.x / 32;
    const int lane_id = threadIdx.x % 32;
    
    if (warp_id >= num_tokens) return;
    
    const int local_idx = warp_id;
    const int global_idx = expert_token_start + local_idx;
    const int token_id = token_ids[global_idx];
    const float weight = weights[global_idx];
    
    // Registers for intermediate results
    float gate_reg[16];   // Process 16 outputs at a time
    float up_reg[16];
    
    const int vec_size = 16;
    const int num_iters = (intermediate_size + vec_size - 1) / vec_size;
    
    // Gate projection with vectorized loads
    for (int iter = 0; iter < num_iters; iter++) {
        int j_start = iter * vec_size;
        int j_end = min(j_start + vec_size, intermediate_size);
        
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            gate_reg[j] = 0.0f;
        }
        
        // Reduction over K dimension
        for (int k = 0; k < hidden_size; k++) {
            float x_val = x[token_id * hidden_size + k];
            #pragma unroll
            for (int j = 0; j < vec_size; j++) {
                int col = j_start + j;
                if (col < intermediate_size) {
                    gate_reg[j] += x_val * gate_weight[col * hidden_size + k];
                }
            }
        }
        
        // SiLU activation
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            int col = j_start + j;
            if (col < intermediate_size) {
                float val = gate_reg[j];
                gate_reg[j] = val / (1.0f + expf(-val));
            }
        }
        
        // Store gate result
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            int col = j_start + j;
            if (col < intermediate_size) {
                gate_reg[j] = gate_reg[j];  // Keep in registers
            }
        }
    }
    
    // Up projection
    for (int iter = 0; iter < num_iters; iter++) {
        int j_start = iter * vec_size;
        int j_end = min(j_start + vec_size, intermediate_size);
        
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            up_reg[j] = 0.0f;
        }
        
        // Reduction over K dimension
        for (int k = 0; k < hidden_size; k++) {
            float x_val = x[token_id * hidden_size + k];
            #pragma unroll
            for (int j = 0; j < vec_size; j++) {
                int col = j_start + j;
                if (col < intermediate_size) {
                    up_reg[j] += x_val * up_weight[col * hidden_size + k];
                }
            }
        }
        
        // Multiply with gate
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            int col = j_start + j;
            if (col < intermediate_size) {
                up_reg[j] *= gate_reg[j];
            }
        }
    }
    
    // Down projection - output hidden_size
    const int num_down_iters = (hidden_size + vec_size - 1) / vec_size;
    
    for (int iter = 0; iter < num_down_iters; iter++) {
        int j_start = iter * vec_size;
        int j_end = min(j_start + vec_size, hidden_size);
        
        float result[16];
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            result[j] = 0.0f;
        }
        
        // Reduction over intermediate_size dimension
        for (int k = 0; k < intermediate_size; k++) {
            #pragma unroll
            for (int j = 0; j < vec_size; j++) {
                int col = j_start + j;
                if (col < hidden_size) {
                    result[j] += up_reg[k] * down_weight[col * intermediate_size + k];
                }
            }
        }
        
        // Write output with atomic add
        #pragma unroll
        for (int j = 0; j < vec_size; j++) {
            int col = j_start + j;
            if (col < hidden_size) {
                atomicAdd(&output[token_id * hidden_size + col], result[j] * weight);
            }
        }
    }
}

// Wrapper function
torch::Tensor moe_fused_op(
    torch::Tensor x,                    // (batch*seq, hidden)
    torch::Tensor gate_weight,          // (num_experts, inter, hidden)
    torch::Tensor up_weight,            // (num_experts, inter, hidden)
    torch::Tensor down_weight,          // (num_experts, hidden, inter)
    torch::Tensor token_ids,            // (num_selected)
    torch::Tensor weights,              // (num_selected)
    int num_experts
) {
    const int hidden_size = x.size(1);
    const int intermediate_size = gate_weight.size(1);
    const int num_selected = token_ids.size(0);
    
    auto output = torch::zeros_like(x);
    
    const int block_dim = 256;  // Multiple warps per expert
    const int grid_dim = num_experts;
    
    moe_fused_optimized_kernel<<<grid_dim, block_dim>>>(
        x.data_ptr<float>(),
        gate_weight.data_ptr<float>(),
        up_weight.data_ptr<float>(),
        down_weight.data_ptr<float>(),
        token_ids.data_ptr<int>(),
        weights.data_ptr<float>(),
        output.data_ptr<float>(),
        num_selected,
        hidden_size,
        intermediate_size,
        num_experts
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}

class Model(nn.Module):
    """
    MoE Expert with Gated GEMM (SiLU-gated FFN) - Optimized version.
    
    Uses a fused CUDA kernel that:
    1. Reads input x once for both gate and up projections
    2. Fuses SiLU activation with elementwise multiply
    3. Processes all experts in parallel using one block per expert
    4. Uses atomic adds for weight accumulation
    """

    def __init__(
        self,
        hidden_size: int,
        intermediate_size: int,
        num_experts: int,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.gate_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.up_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.down_proj = nn.Parameter(
            torch.randn(num_experts, hidden_size, intermediate_size) * 0.02
        )
        torch.random.set_rng_state(rng_state)

    def forward(
        self,
        x: torch.Tensor,              # (batch, seq_len, hidden_size)
        expert_indices: torch.Tensor, # (batch, seq_len, top_k) - selected expert indices
        expert_weights: torch.Tensor, # (batch, seq_len, top_k) - routing weights
    ) -> torch.Tensor:
        """
        Optimized MoE forward using fused CUDA kernel.
        """
        if ext is None:
            # Fallback to PyTorch if extension fails to load
            return self.forward_pytorch(x, expert_indices, expert_weights)
            
        batch, seq_len, _ = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len

        x_flat = x.view(num_tokens, self.hidden_size)
        indices_flat = expert_indices.view(num_tokens * top_k)
        weights_flat = expert_weights.view(num_tokens * top_k)

        # Create token indices for each (token, slot) pair
        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1)

        # Sort by expert to group tokens per expert
        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]

        # Initialize output
        output = torch.zeros(num_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
        
        # Process each expert with its tokens
        for e in range(self.num_experts):
            # Find tokens assigned to this expert
            mask = sorted_expert_idx == e
            expert_token_ids = sorted_token_ids[mask]
            expert_weights_local = sorted_weights[mask]
            
            if expert_token_ids.numel() == 0:
                continue
            
            # Call fused kernel for this expert (convert to int32 for CUDA)
            expert_output = ext.moe_fused_op(
                x_flat,
                self.gate_proj[e],
                self.up_proj[e],
                self.down_proj[e],
                expert_token_ids.int(),
                expert_weights_local,
                1  # num_experts = 1 for single expert call
            )
            
            # Accumulate output
            output = output + expert_output
        
        return output.view(batch, seq_len, self.hidden_size)

    def forward_pytorch(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        """PyTorch fallback implementation."""
        batch, seq_len, _ = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len

        x_flat = x.view(num_tokens, self.hidden_size)
        indices_flat = expert_indices.view(num_tokens * top_k)
        weights_flat = expert_weights.view(num_tokens * top_k)

        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1)

        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]

        expert_counts = torch.bincount(sorted_expert_idx, minlength=self.num_experts)
        expert_offsets = torch.cat([
            torch.zeros(1, dtype=torch.long, device=x.device),
            expert_counts.cumsum(0)
        ])

        sorted_x = x_flat[sorted_token_ids]
        sorted_output = torch.empty_like(sorted_x)

        for e in range(self.num_experts):
            start, end = expert_offsets[e].item(), expert_offsets[e + 1].item()
            if start == end:
                continue

            expert_x = sorted_x[start:end]

            gate = F.silu(F.linear(expert_x, self.gate_proj[e]))
            up = F.linear(expert_x, self.up_proj[e])
            intermediate = gate * up
            sorted_output[start:end] = F.linear(intermediate, self.down_proj[e])

        weighted_sorted = sorted_output * sorted_weights.unsqueeze(-1)

        output = torch.zeros(num_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
        output.index_add_(0, sorted_token_ids, weighted_sorted)

        return output.view(batch, seq_len, self.hidden_size)


# Mixtral-style configuration
batch_size = 4
seq_len = 2048
hidden_size = 4096
intermediate_size = 14336
num_experts = 8
top_k = 2


def get_inputs():
    x = torch.randn(batch_size, seq_len, hidden_size)

    expert_indices = torch.stack([
        torch.randperm(num_experts)[:top_k]
        for _ in range(batch_size * seq_len)
    ]).view(batch_size, seq_len, top_k)

    expert_weights = F.softmax(torch.randn(batch_size, seq_len, top_k), dim=-1)

    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return [hidden_size, intermediate_size, num_experts]
