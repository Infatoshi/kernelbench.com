import torch

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp-level reduction for sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// LayerNorm kernel for float
__global__ void layernorm_fp32_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    int batch_size,
    int norm_size,
    float eps
) {
    extern __shared__ float shared_mem[];
    
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const float* x = input + batch_idx * norm_size;
    float* y = output + batch_idx * norm_size;
    
    int tid = threadIdx.x;
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    int num_warps = blockDim.x / 32;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float val = x[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = warp_reduce_sum(local_sum);
    local_sum_sq = warp_reduce_sum(local_sum_sq);
    
    if (lane_id == 0) {
        shared_mem[warp_id] = local_sum;
        shared_mem[num_warps + warp_id] = local_sum_sq;
    }
    __syncthreads();
    
    float mean = 0.0f;
    float inv_std = 0.0f;
    if (warp_id == 0) {
        if (lane_id < num_warps) {
            local_sum = shared_mem[lane_id];
            local_sum_sq = shared_mem[num_warps + lane_id];
        } else {
            local_sum = 0.0f;
            local_sum_sq = 0.0f;
        }
        local_sum = warp_reduce_sum(local_sum);
        local_sum_sq = warp_reduce_sum(local_sum_sq);
        
        if (lane_id == 0) {
            mean = local_sum / norm_size;
            float variance = (local_sum_sq / norm_size) - (mean * mean);
            variance = fmaxf(variance, 0.0f);
            inv_std = 1.0f / sqrtf(variance + eps);
            shared_mem[0] = mean;
            shared_mem[1] = inv_std;
        }
    }
    __syncthreads();
    
    mean = shared_mem[0];
    inv_std = shared_mem[1];
    
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float val = x[i];
        float normalized = (val - mean) * inv_std;
        if (gamma != nullptr) {
            normalized *= gamma[i];
        }
        if (beta != nullptr) {
            normalized += beta[i];
        }
        y[i] = normalized;
    }
}

torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps) {
    auto input_shape = x.sizes();
    int batch_size = input_shape[0];
    int norm_size = 1;
    for (int i = 1; i < input_shape.size(); i++) {
        norm_size *= input_shape[i];
    }
    
    auto output = torch::empty_like(x);
    
    int threads_per_block = 512;
    int blocks = batch_size;
    int shared_mem_size = 2 * (threads_per_block / 32) * sizeof(float);
    
    if (x.dtype() == torch::kFloat32) {
        layernorm_fp32_kernel<<<blocks, threads_per_block, shared_mem_size>>>(
            x.data_ptr<float>(),
            output.data_ptr<float>(),
            gamma.defined() ? gamma.data_ptr<float>() : nullptr,
            beta.defined() ? beta.data_ptr<float>() : nullptr,
            batch_size,
            norm_size,
            eps
        );
    } else if (x.dtype() == torch::kFloat16) {
        layernorm_fp16_kernel<<<blocks, threads_per_block, shared_mem_size>>>(
            reinterpret_cast<const half*>(x.data_ptr<c10::Half>()),
            reinterpret_cast<half*>(output.data_ptr<c10::Half>()),
            gamma.defined() ? reinterpret_cast<const half*>(gamma.data_ptr<c10::Half>()) : nullptr,
            beta.defined() ? reinterpret_cast<const half*>(beta.data_ptr<c10::Half>()) : nullptr,
            batch_size,
            norm_size,
            eps
        );
    } else if (x.dtype() == torch::kBFloat16) {
        layernorm_bf16_kernel<<<blocks, threads_per_block, shared_mem_size>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<c10::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<c10::BFloat16>()),
            gamma.defined() ? reinterpret_cast<const __nv_bfloat16*>(gamma.data_ptr<c10::BFloat16>()) : nullptr,
            beta.defined() ? reinterpret_cast<const __nv_bfloat16*>(beta.data_ptr<c10::BFloat16>()) : nullptr,
            batch_size,
            norm_size,
            eps
        );
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        throw std::runtime_error(cudaGetErrorString(err));
    }
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_100', '-O3'],
)


class Model(nn.Module):
    """
    Simple model that performs Layer Normalization.
    """
    def __init__(self, normalized_shape: tuple):
        """
        Initializes the LayerNorm layer.

        Args:
            normalized_shape (tuple): Shape of the input tensor to be normalized.
        """
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        # Register parameters
        self.gamma = nn.Parameter(torch.ones(normalized_shape))
        self.beta = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Layer Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, normalized_shape).

        Returns:
            torch.Tensor: Output tensor with Layer Normalization applied, same shape as input.
        """
        return ext.layernorm_forward(x, self.gamma, self.beta, self.eps)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
