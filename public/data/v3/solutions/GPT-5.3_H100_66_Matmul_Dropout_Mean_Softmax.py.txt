import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void gemv_kernel(const float* __restrict__ m,
                            const float* __restrict__ W,
                            const float* __restrict__ b,
                            float* __restrict__ y,
                            int in_f, int out_f) {
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if (j >= out_f) return;

    const float* w_row = W + j * in_f;
    float acc = 0.0f;

    for (int k = 0; k < in_f; ++k) {
        acc += m[k] * w_row[k];
    }

    acc += b[j];
    y[j] = acc;
}

torch::Tensor gemv(torch::Tensor m, torch::Tensor W, torch::Tensor b) {
    int in_f = m.size(0);
    int out_f = W.size(0);

    auto y = torch::empty({out_f}, m.options());

    int threads = 256;
    int blocks = (out_f + threads - 1) / threads;

    gemv_kernel<<<blocks, threads>>>(
        m.data_ptr<float>(),
        W.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        in_f, out_f);

    return y;
}
'''

ext = load_inline(
    name='fused_gemv_ext',
    cpp_sources='torch::Tensor gemv(torch::Tensor m, torch::Tensor W, torch::Tensor b);',
    cuda_sources=cuda_source,
    functions=['gemv'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        m = x.mean(dim=0)
        W = self.matmul.weight
        b = self.matmul.bias
        y = ext.gemv(m.contiguous(), W.contiguous(), b.contiguous())
        y = y.unsqueeze(0)
        y = torch.softmax(y, dim=1)
        return y

batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features, device='cuda')]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
