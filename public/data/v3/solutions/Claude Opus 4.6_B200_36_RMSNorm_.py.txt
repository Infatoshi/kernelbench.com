import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rms_norm_kernel_v3(
    X_ptr,
    Y_ptr,
    C: tl.constexpr,
    spatial_size,
    eps,
    BLOCK_C: tl.constexpr,
    BLOCK_S: tl.constexpr,
):
    """
    Each program handles one batch element and a block of BLOCK_S spatial positions.
    We load all C features for each spatial position.
    Memory layout: (B, C, H*W) with strides (C*HW, HW, 1)
    """
    pid = tl.program_id(0)
    num_spatial_blocks = tl.cdiv(spatial_size, BLOCK_S)
    batch_idx = pid // num_spatial_blocks
    spatial_block_idx = pid % num_spatial_blocks
    
    s_base = spatial_block_idx * BLOCK_S
    s_offsets = s_base + tl.arange(0, BLOCK_S)  # [BLOCK_S]
    s_mask = s_offsets < spatial_size
    
    batch_offset = batch_idx * C * spatial_size
    
    c_offsets = tl.arange(0, BLOCK_C)  # [BLOCK_C]
    
    # Accumulate sum of squares: [BLOCK_S]
    sum_sq = tl.zeros([BLOCK_S], dtype=tl.float32)
    
    for c_start in range(0, C, BLOCK_C):
        c_idx = c_start + c_offsets  # [BLOCK_C]
        c_mask = c_idx < C           # [BLOCK_C]
        
        # offsets: [BLOCK_C, BLOCK_S]
        offsets = batch_offset + c_idx[:, None] * spatial_size + s_offsets[None, :]
        mask = c_mask[:, None] & s_mask[None, :]
        
        x = tl.load(X_ptr + offsets, mask=mask, other=0.0).to(tl.float32)
        sum_sq += tl.sum(x * x, axis=0)  # reduce over C -> [BLOCK_S]
    
    # inv_rms: [BLOCK_S]
    inv_rms = tl.rsqrt(sum_sq / C + eps)
    
    # Second pass: normalize
    for c_start in range(0, C, BLOCK_C):
        c_idx = c_start + c_offsets
        c_mask = c_idx < C
        
        offsets = batch_offset + c_idx[:, None] * spatial_size + s_offsets[None, :]
        mask = c_mask[:, None] & s_mask[None, :]
        
        x = tl.load(X_ptr + offsets, mask=mask, other=0.0).to(tl.float32)
        y = x * inv_rms[None, :]
        tl.store(Y_ptr + offsets, y, mask=mask)


def rms_norm_triton(x: torch.Tensor, num_features: int, eps: float) -> torch.Tensor:
    B, C, H, W = x.shape
    spatial_size = H * W
    
    y = torch.empty_like(x)
    
    # Since C=64 and spatial is large, use BLOCK_C=64 (one shot) and large BLOCK_S
    BLOCK_C = 64
    BLOCK_S = 2048
    
    num_spatial_blocks = (spatial_size + BLOCK_S - 1) // BLOCK_S
    grid = (B * num_spatial_blocks,)
    
    rms_norm_kernel_v3[grid](
        x, y,
        C,
        spatial_size,
        eps,
        BLOCK_C=BLOCK_C,
        BLOCK_S=BLOCK_S,
    )
    
    return y


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return rms_norm_triton(x, self.num_features, self.eps)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256

    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]
