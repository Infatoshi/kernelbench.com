import torch
import torch.nn as nn
import torch.nn.functional as F
from fla.ops import chunk_kda
from einops import rearrange

# Kimi Delta Attention (KDA): Linear Attention with Channel-wise Gating
# Optimized implementation for H100 with:
# 1. Fused QKV+AB projections
# 2. Fused depthwise convolutions with SiLU
# 3. Fused sigmoid operations
# 4. Optimized memory layout to reduce transposes


class FusedQKVABProj(nn.Module):
    """Fused projection for Q, K, V, A, B to reduce kernel launches."""
    
    def __init__(self, hidden_size: int, num_heads: int, head_dim_qk: int, head_dim_v: int):
        super().__init__()
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        
        # Fused: q (num_heads * head_dim_qk) + k (num_heads * head_dim_qk) + 
        #        v (num_heads * head_dim_v) + a (num_heads * head_dim_v) + b (num_heads)
        self.out_features = (num_heads * (2 * head_dim_qk + 2 * head_dim_v) + num_heads)
        self.proj = nn.Linear(hidden_size, self.out_features, bias=True)
        # Disable bias for q, k, v, a parts (only b needs bias)
        with torch.no_grad():
            self.proj.bias[:-num_heads].zero_()
        
    def forward(self, x):
        # x: (B, T, H_in)
        out = self.proj(x)  # (B, T, out_features)
        
        # Split outputs
        q, k, v, a, b = out.split([
            self.num_heads * self.head_dim_qk,
            self.num_heads * self.head_dim_qk,
            self.num_heads * self.head_dim_v,
            self.num_heads * self.head_dim_v,
            self.num_heads
        ], dim=-1)
        
        return q, k, v, a, b


class FusedDepthwiseConvSiLU(nn.Module):
    """Fused depthwise conv + SiLU for efficiency."""
    
    def __init__(self, channels: int, kernel_size: int = 4):
        super().__init__()
        self.conv = nn.Conv1d(
            channels, channels, kernel_size=kernel_size,
            groups=channels, padding=kernel_size - 1, bias=True
        )
        
    def forward(self, x, seq_len):
        # x: (B, T, C) -> (B, C, T) -> conv -> (B, T, C)
        x = x.transpose(1, 2)
        x = self.conv(x)[:, :, :seq_len]
        x = F.silu(x)
        x = x.transpose(1, 2)
        return x


class OptimizedKDA(nn.Module):
    """Optimized KDA with fused operations."""
    
    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv
        self.scale = head_dim_qk ** -0.5
        
        # Fused projections
        self.qkvab_proj = FusedQKVABProj(hidden_size, num_heads, head_dim_qk, head_dim_v)
        
        # Fused convolutions
        if use_short_conv:
            self.q_conv = FusedDepthwiseConvSiLU(num_heads * head_dim_qk, conv_kernel_size)
            self.k_conv = FusedDepthwiseConvSiLU(num_heads * head_dim_qk, conv_kernel_size)
            self.v_conv = FusedDepthwiseConvSiLU(num_heads * head_dim_v, conv_kernel_size)
        
        # Output
        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        H = self.num_heads
        D_qk = self.head_dim_qk
        D_v = self.head_dim_v
        
        # Fused projection
        q, k, v, a, b = self.qkvab_proj(x)
        
        # Fused conv + silu
        if self.use_short_conv:
            q = self.q_conv(q, seq_len)
            k = self.k_conv(k, seq_len)
            v = self.v_conv(v, seq_len)
        
        # Reshape to (B, H, T, D) for chunk_kda
        q = q.view(batch_size, seq_len, H, D_qk).transpose(1, 2)
        k = k.view(batch_size, seq_len, H, D_qk).transpose(1, 2)
        v = v.view(batch_size, seq_len, H, D_v).transpose(1, 2)
        
        # Channel-wise gates with fused sigmoid
        a = torch.sigmoid(a)
        a = a.view(batch_size, seq_len, H, D_v).transpose(1, 2)
        beta = torch.sigmoid(b).transpose(1, 2)  # (B, H, T)
        
        # Convert a to log space for chunk_kda
        g = a.clamp(min=1e-6).log()
        
        # chunk_kda expects (B, T, H, D), we have (B, H, T, D)
        q = q.transpose(1, 2)
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)
        g = g.transpose(1, 2)
        beta = beta.transpose(1, 2)
        
        # Core attention
        o, _ = chunk_kda(q, k, v, g, beta, scale=self.scale)
        
        # Convert back to (B, T, H, D_v)
        o = o  # Already (B, T, H, D_v)
        
        # Output norm
        o = self.o_norm(o)
        
        # Gating
        g = torch.sigmoid(self.g_proj(x))
        g = g.view(batch_size, seq_len, H, D_v)
        o = o * g
        
        # Output projection
        o = o.reshape(batch_size, seq_len, H * D_v)
        o = self.o_proj(o)
        
        return o


# Keep the same Model interface for compatibility
class Model(nn.Module):
    """
    Kimi Delta Attention with channel-wise gating.
    Optimized implementation for H100.
    """

    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_dplr: bool = False,
        dplr_rank: int = 4,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        # Channel-wise gating: d_v gates per head
        self.a_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=True)
        self.b_proj = nn.Linear(hidden_size, num_heads, bias=True)

        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)

        if use_short_conv:
            self.q_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.k_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.v_conv = nn.Conv1d(
                num_heads * head_dim_v, num_heads * head_dim_v,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_v,
                padding=conv_kernel_size - 1
            )

        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape

        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)

        if self.use_short_conv:
            q = self.q_conv(q.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            k = self.k_conv(k.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            v = self.v_conv(v.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            q = F.silu(q)
            k = F.silu(k)
            v = F.silu(v)

        # Reshape to (B, H, T, D)
        q = q.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        k = k.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        v = v.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        # Channel-wise gates (B, H, T, d_v)
        a = torch.sigmoid(self.a_proj(x))
        a = a.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        beta = torch.sigmoid(self.b_proj(x)).transpose(1, 2)  # (B, H, T)

        # Chunk-wise parallel attention (fla)
        # fla expects (B, T, H, D)
        q = q.transpose(1, 2)
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)
        a = a.transpose(1, 2)
        beta = beta.transpose(1, 2)
        
        g = a.clamp(min=1e-6).log()
        o, _ = chunk_kda(q, k, v, g, beta, scale=self.scale)
        
        # o is (B, T, H, D_v)

        o = self.o_norm(o)

        g = torch.sigmoid(self.g_proj(x))
        g = g.view(batch_size, seq_len, self.num_heads, self.head_dim_v)
        o = o * g

        o = o.reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
        o = self.o_proj(o)

        return o


batch_size = 4

head_dim_v = 128


def get_inputs():
