import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized CUDA softmax kernel with:
# 1. Vectorized float4 loads for coalesced memory access
# 2. Warp-level reductions using __shfl_xor_sync
# 3. Online/stable softmax (subtract max before exp)
# 4. Each block handles one row, threads cooperate on the row
# 5. For 16384 cols: 256 threads * 64 elements each = 16384

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>

// Warp reduce max
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, mask));
    return val;
}

// Warp reduce sum
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = 16; mask > 0; mask >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, mask);
    return val;
}

// Float softmax kernel: 1 block per row
// threads = 256, each thread processes (n_cols / 256) elements via vectorized float4
__global__ void softmax_f32_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int n_cols
) {
    // Shared memory for warp reductions
    __shared__ float smem[32]; // max 32 warps per block

    int row = blockIdx.x;
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    int num_warps = block_size / 32;
    int warp_id = tid / 32;
    int lane_id = tid % 32;

    const float* row_in = input + row * n_cols;
    float* row_out = output + row * n_cols;

    // Step 1: find row max using vectorized loads
    float thread_max = -FLT_MAX;

    // Use float4 loads when possible
    int vec_cols = n_cols / 4;
    const float4* row_in4 = reinterpret_cast<const float4*>(row_in);

    for (int i = tid; i < vec_cols; i += block_size) {
        float4 v = row_in4[i];
        thread_max = fmaxf(thread_max, fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w)));
    }
    // Handle remainder
    for (int i = vec_cols * 4 + tid; i < n_cols; i += block_size) {
        thread_max = fmaxf(thread_max, row_in[i]);
    }

    // Warp reduce max
    thread_max = warp_reduce_max(thread_max);
    if (lane_id == 0) smem[warp_id] = thread_max;
    __syncthreads();

    // Final reduce across warps (by first warp)
    float row_max;
    if (warp_id == 0) {
        float val = (lane_id < num_warps) ? smem[lane_id] : -FLT_MAX;
        val = warp_reduce_max(val);
        if (lane_id == 0) smem[0] = val;
    }
    __syncthreads();
    row_max = smem[0];

    // Step 2: compute exp(x - max) and sum
    float thread_sum = 0.0f;
    float4* row_out4 = reinterpret_cast<float4*>(row_out);

    for (int i = tid; i < vec_cols; i += block_size) {
        float4 v = row_in4[i];
        float4 ev;
        ev.x = __expf(v.x - row_max);
        ev.y = __expf(v.y - row_max);
        ev.z = __expf(v.z - row_max);
        ev.w = __expf(v.w - row_max);
        row_out4[i] = ev;
        thread_sum += ev.x + ev.y + ev.z + ev.w;
    }
    for (int i = vec_cols * 4 + tid; i < n_cols; i += block_size) {
        float ev = __expf(row_in[i] - row_max);
        row_out[i] = ev;
        thread_sum += ev;
    }

    // Warp reduce sum
    thread_sum = warp_reduce_sum(thread_sum);
    if (lane_id == 0) smem[warp_id] = thread_sum;
    __syncthreads();

    float row_sum;
    if (warp_id == 0) {
        float val = (lane_id < num_warps) ? smem[lane_id] : 0.0f;
        val = warp_reduce_sum(val);
        if (lane_id == 0) smem[0] = val;
    }
    __syncthreads();
    row_sum = smem[0];

    // Step 3: normalize
    float inv_sum = __fdividef(1.0f, row_sum);
    for (int i = tid; i < vec_cols; i += block_size) {
        float4 v = row_out4[i];
        v.x *= inv_sum;
        v.y *= inv_sum;
        v.z *= inv_sum;
        v.w *= inv_sum;
        row_out4[i] = v;
    }
    for (int i = vec_cols * 4 + tid; i < n_cols; i += block_size) {
        row_out[i] *= inv_sum;
    }
}

// Half-precision softmax: load half2, compute in float, store half2
__global__ void softmax_f16_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    int n_cols
) {
    __shared__ float smem[32];

    int row = blockIdx.x;
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    int num_warps = block_size / 32;
    int warp_id = tid / 32;
    int lane_id = tid % 32;

    const __half* row_in = input + row * n_cols;
    __half* row_out = output + row * n_cols;

    float thread_max = -FLT_MAX;
    int vec_cols = n_cols / 2;
    const __half2* row_in2 = reinterpret_cast<const __half2*>(row_in);

    for (int i = tid; i < vec_cols; i += block_size) {
        __half2 v = row_in2[i];
        float2 vf = __half22float2(v);
        thread_max = fmaxf(thread_max, fmaxf(vf.x, vf.y));
    }
    for (int i = vec_cols * 2 + tid; i < n_cols; i += block_size) {
        thread_max = fmaxf(thread_max, __half2float(row_in[i]));
    }

    thread_max = warp_reduce_max(thread_max);
    if (lane_id == 0) smem[warp_id] = thread_max;
    __syncthreads();

    float row_max;
    if (warp_id == 0) {
        float val = (lane_id < num_warps) ? smem[lane_id] : -FLT_MAX;
        val = warp_reduce_max(val);
        if (lane_id == 0) smem[0] = val;
    }
    __syncthreads();
    row_max = smem[0];

    float thread_sum = 0.0f;
    __half2* row_out2 = reinterpret_cast<__half2*>(row_out);

    for (int i = tid; i < vec_cols; i += block_size) {
        __half2 v = row_in2[i];
        float2 vf = __half22float2(v);
        float ex = __expf(vf.x - row_max);
        float ey = __expf(vf.y - row_max);
        row_out2[i] = __float22half2_rn(make_float2(ex, ey));
        thread_sum += ex + ey;
    }
    for (int i = vec_cols * 2 + tid; i < n_cols; i += block_size) {
        float ev = __expf(__half2float(row_in[i]) - row_max);
        row_out[i] = __float2half(ev);
        thread_sum += ev;
    }

    thread_sum = warp_reduce_sum(thread_sum);
    if (lane_id == 0) smem[warp_id] = thread_sum;
    __syncthreads();

    float row_sum;
    if (warp_id == 0) {
        float val = (lane_id < num_warps) ? smem[lane_id] : 0.0f;
        val = warp_reduce_sum(val);
        if (lane_id == 0) smem[0] = val;
    }
    __syncthreads();
    row_sum = smem[0];

    float inv_sum = __fdividef(1.0f, row_sum);
    for (int i = tid; i < vec_cols; i += block_size) {
        __half2 v = row_out2[i];
        float2 vf = __half22float2(v);
        vf.x *= inv_sum;
        vf.y *= inv_sum;
        row_out2[i] = __float22half2_rn(vf);
    }
    for (int i = vec_cols * 2 + tid; i < n_cols; i += block_size) {
        row_out[i] = __float2half(__half2float(row_out[i]) * inv_sum);
    }
}

torch::Tensor softmax_forward(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "Input must be on CUDA");
    TORCH_CHECK(x.dim() == 2, "Input must be 2D");
    x = x.contiguous();

    int n_rows = x.size(0);
    int n_cols = x.size(1);
    auto y = torch::empty_like(x);

    // Choose block size: want enough threads to fill the SM
    // For 16384 cols with float4: 256 threads * 16 iterations = 4096 float4 = 16384 floats
    // Using 512 threads for better occupancy tradeoff
    const int BLOCK_SIZE = 512;

    dim3 grid(n_rows);
    dim3 block(BLOCK_SIZE);

    auto dtype = x.scalar_type();
    if (dtype == torch::kFloat32) {
        softmax_f32_kernel<<<grid, block>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            n_cols
        );
    } else if (dtype == torch::kFloat16) {
        softmax_f16_kernel<<<grid, block>>>(
            reinterpret_cast<const __half*>(x.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(y.data_ptr<at::Half>()),
            n_cols
        );
    } else if (dtype == torch::kBFloat16) {
        softmax_bf16_kernel<<<grid, block>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(y.data_ptr<at::BFloat16>()),
            n_cols
        );
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }

    return y;
}
'''

cpp_source = 'torch::Tensor softmax_forward(torch::Tensor x);'

ext = load_inline(
    name='softmax_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_100', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Simple model that performs a Softmax activation using an optimized CUDA kernel.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features).

        Returns:
            torch.Tensor: Output tensor with Softmax applied, same shape as input.
        """
        return ext.softmax_forward(x)


batch_size = 256

def get_inputs():
