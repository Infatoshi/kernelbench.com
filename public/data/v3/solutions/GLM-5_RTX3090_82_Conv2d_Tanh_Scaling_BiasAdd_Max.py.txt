import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def fast_tanh(x):
    """Fast tanh implementation using exp"""
    # tanh(x) = (exp(2x) - 1) / (exp(2x) + 1)
    exp_2x = tl.exp(2.0 * x)
    return (exp_2x - 1.0) / (exp_2x + 1.0)


@triton.jit
def fused_tanh_scale_bias_kernel(
    input_ptr, output_ptr, bias_ptr,
    scaling_factor,
    n_elements,
    channels, hw,
    BLOCK_SIZE: tl.constexpr,
):
    """Fused tanh + scale + bias kernel"""
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    # Load input
    x = tl.load(input_ptr + offsets, mask=mask, other=0.0)
    
    # Compute channel indices for bias
    c_indices = offsets // hw
    c_indices = c_indices % channels
    
    # Load bias
    bias = tl.load(bias_ptr + c_indices, mask=mask, other=0.0)
    
    # Fused operations
    x = fast_tanh(x)           # tanh
    x = x * scaling_factor     # scale
    x = x + bias               # bias add
    
    # Store output
    tl.store(output_ptr + offsets, x, mask=mask)


def fused_tanh_scale_bias(input_tensor, bias, scaling_factor):
    """Apply fused tanh + scale + bias"""
    batch, channels, height, width = input_tensor.shape
    hw = height * width
    n_elements = batch * channels * hw
    
    output = torch.empty_like(input_tensor)
    
    grid = lambda meta: (triton.cdiv(n_elements, meta['BLOCK_SIZE']),)
    
    fused_tanh_scale_bias_kernel[grid](
        input_tensor, output, bias,
        scaling_factor,
        n_elements,
        channels, hw,
        BLOCK_SIZE=1024,
    )
    
    return output


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Convolution (using PyTorch's optimized implementation)
        x = self.conv(x)
        
        # Fused tanh + scale + bias
        x = fused_tanh_scale_bias(x, self.bias.view(-1), self.scaling_factor)
        
        # Max pooling (using PyTorch's optimized implementation)
        x = F.max_pool2d(x, self.pool_kernel_size)
        
        return x
