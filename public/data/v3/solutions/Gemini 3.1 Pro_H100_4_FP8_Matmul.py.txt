import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def _find_amax_kernel(
    x_ptr,
    out_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    x_abs = tl.abs(x)
    max_val = tl.max(x_abs)
    tl.store(out_ptr + pid, max_val)

@triton.jit
def _quantize_e4m3_kernel(
    x_ptr,
    out_ptr,
    scale_ptr,
    n_elements,
    val_max: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    scale = tl.load(scale_ptr)
    
    x_scaled = x * scale
    x_clamped = tl.where(x_scaled > val_max, val_max, x_scaled)
    x_clamped = tl.where(x_clamped < -val_max, -val_max, x_clamped)
    
    x_fp8 = x_clamped.to(tl.float8e4nv)
    tl.store(out_ptr + offsets, x_fp8, mask=mask)

@triton.jit
def _quantize_e5m2_kernel(
    x_ptr,
    out_ptr,
    scale_ptr,
    n_elements,
    val_max: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask, other=0.0)
    scale = tl.load(scale_ptr)
    
    x_scaled = x * scale
    x_clamped = tl.where(x_scaled > val_max, val_max, x_scaled)
    x_clamped = tl.where(x_clamped < -val_max, -val_max, x_clamped)
    
    x_fp8 = x_clamped.to(tl.float8e5)
    tl.store(out_ptr + offsets, x_fp8, mask=mask)


class Model(nn.Module):
    def __init__(self, M: int, K: int, N: int, use_e4m3: bool = True):
        super().__init__()
        self.M = M
        self.K = K
        self.N = N
        self.use_e4m3 = use_e4m3
        if use_e4m3:
            self.fp8_dtype = torch.float8_e4m3fn
            self.fp8_max = 448.0
        else:
            self.fp8_dtype = torch.float8_e5m2
            self.fp8_max = 57344.0
            
        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.weight = nn.Parameter(torch.randn(K, N) * 0.02)
        torch.random.set_rng_state(rng_state)
        
        self.w_fp8 = None
        self.w_scale_inv = None
        
        # Buffer for intermediate block maximums. Max possible n_elements = M*K, which could vary 
        # but let's size it large enough. Our max expected is seq_len=2048, batch_size=8, K=4096 => 67108864 elements
        # Blocks are 65536, so 1024 blocks. We will allocate dynamically or lazily.
        self.block_maxes = None

    def compute_scale(self, x: torch.Tensor) -> torch.Tensor:
        amax = x.abs().max()
        scale = self.fp8_max / amax.clamp(min=1e-12)
        return scale

    def quantize_to_fp8(self, x: torch.Tensor, scale: torch.Tensor) -> torch.Tensor:
        x_scaled = x * scale
        x_clamped = x_scaled.clamp(-self.fp8_max, self.fp8_max)
        return x_clamped.to(self.fp8_dtype)

    def _fast_quantize(self, x: torch.Tensor):
        n_elements = x.numel()
        BLOCK_SIZE = 65536
        grid_dim = triton.cdiv(n_elements, BLOCK_SIZE)
        
        if self.block_maxes is None or self.block_maxes.numel() < grid_dim:
            self.block_maxes = torch.empty(grid_dim, dtype=torch.float32, device=x.device)
            
        out_blocks = self.block_maxes[:grid_dim]
        _find_amax_kernel[(grid_dim,)](x, out_blocks, n_elements, BLOCK_SIZE=BLOCK_SIZE)
        
        amax = torch.max(out_blocks)
        amax = torch.clamp(amax, min=1e-12)
        s = self.fp8_max / amax
        
        # It's faster to pass s as a tensor to the second kernel. But Triton scalar load is fine too.
        # Actually passing a pointer to torch tensor on GPU avoids device-to-host transfer.
        s_tensor = torch.tensor([s], dtype=torch.float32, device=x.device)
        
        out = torch.empty_like(x, dtype=self.fp8_dtype)
        if self.use_e4m3:
            _quantize_e4m3_kernel[(grid_dim,)](x, out, s_tensor, n_elements, val_max=self.fp8_max, BLOCK_SIZE=BLOCK_SIZE)
        else:
            _quantize_e5m2_kernel[(grid_dim,)](x, out, s_tensor, n_elements, val_max=self.fp8_max, BLOCK_SIZE=BLOCK_SIZE)
            
        return out, s

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        input_dtype = x.dtype
        batch_size = x.shape[0]
        seq_len = x.shape[1]
        x_2d = x.view(-1, self.K)

        if self.w_fp8 is None:
            w_scale = self.compute_scale(self.weight)
            w_t = self.weight.t().contiguous()
            self.w_fp8 = self.quantize_to_fp8(w_t, w_scale)
            self.w_scale_inv = (1.0 / w_scale).to(torch.float32)

        x_fp8, s = self._fast_quantize(x_2d)
        x_scale_inv = (1.0 / s).to(torch.float32)

        out = torch._scaled_mm(
            x_fp8,
            self.w_fp8.t(),
            scale_a=x_scale_inv,
            scale_b=self.w_scale_inv,
            out_dtype=input_dtype,
        )

        return out.view(batch_size, seq_len, self.N)
