import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

constexpr float SQRT_1_2 = 0.70710678118654752440f;

__device__ __forceinline__ float compute_gelu(float x) {
    return x * 0.5f * (1.0f + erff(x * SQRT_1_2));
}

__device__ __forceinline__ float4 compute_gelu_float4(float4 x) {
    return make_float4(compute_gelu(x.x), compute_gelu(x.y), compute_gelu(x.z), compute_gelu(x.w));
}

__device__ __forceinline__ half2 compute_gelu_half2(half2 x) {
    float2 f = __half22float2(x);
    f.x = compute_gelu(f.x);
    f.y = compute_gelu(f.y);
    return __float22half2_rn(f);
}

__device__ __forceinline__ nv_bfloat162 compute_gelu_bf162(nv_bfloat162 x) {
#if __CUDA_ARCH__ >= 800
    float2 f = __bfloat1622float2(x);
    f.x = compute_gelu(f.x);
    f.y = compute_gelu(f.y);
    return __float22bfloat162_rn(f);
#else
    return x;
#endif
}

__global__ void gelu_float_kernel(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        y[i] = compute_gelu(x[i]);
    }
}

__global__ void gelu_float4_kernel(const float4* __restrict__ x, float4* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        y[i] = compute_gelu_float4(x[i]);
    }
}

__global__ void gelu_half8_kernel(const float4* __restrict__ x, float4* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        float4 val = x[i];
        half2* h2 = reinterpret_cast<half2*>(&val);
        h2[0] = compute_gelu_half2(h2[0]);
        h2[1] = compute_gelu_half2(h2[1]);
        h2[2] = compute_gelu_half2(h2[2]);
        h2[3] = compute_gelu_half2(h2[3]);
        y[i] = val;
    }
}

__global__ void gelu_half_kernel(const half* __restrict__ x, half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        float f = __half2float(x[i]);
        y[i] = __float2half(compute_gelu(f));
    }
}

__global__ void gelu_bf168_kernel(const float4* __restrict__ x, float4* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        float4 val = x[i];
        nv_bfloat162* h2 = reinterpret_cast<nv_bfloat162*>(&val);
        h2[0] = compute_gelu_bf162(h2[0]);
        h2[1] = compute_gelu_bf162(h2[1]);
        h2[2] = compute_gelu_bf162(h2[2]);
        h2[3] = compute_gelu_bf162(h2[3]);
        y[i] = val;
    }
}

__global__ void gelu_bf16_kernel(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    for (int i = idx; i < n; i += stride) {
        float f = __bfloat162float(x[i]);
        y[i] = __float2bfloat16(compute_gelu(f));
    }
}

torch::Tensor my_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    if (n == 0) return y;
    
    int threads = 256;
    int blocks = std::min((n + threads - 1) / threads, 65535);

    if (x.scalar_type() == torch::kFloat32) {
        if (n % 4 == 0 && reinterpret_cast<std::uintptr_t>(x.data_ptr()) % 16 == 0 && reinterpret_cast<std::uintptr_t>(y.data_ptr()) % 16 == 0) {
            blocks = std::min((n / 4 + threads - 1) / threads, 65535);
            gelu_float4_kernel<<<blocks, threads>>>(
                reinterpret_cast<const float4*>(x.data_ptr<float>()),
                reinterpret_cast<float4*>(y.data_ptr<float>()),
                n / 4
            );
        } else {
            gelu_float_kernel<<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
        }
    } else if (x.scalar_type() == torch::kFloat16) {
         if (n % 8 == 0 && reinterpret_cast<std::uintptr_t>(x.data_ptr()) % 16 == 0 && reinterpret_cast<std::uintptr_t>(y.data_ptr()) % 16 == 0) {
            blocks = std::min((n / 8 + threads - 1) / threads, 65535);
            gelu_half8_kernel<<<blocks, threads>>>(
                reinterpret_cast<const float4*>(x.data_ptr<at::Half>()),
                reinterpret_cast<float4*>(y.data_ptr<at::Half>()),
                n / 8
            );
        } else {
             gelu_half_kernel<<<blocks, threads>>>(
                 reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
                 reinterpret_cast<half*>(y.data_ptr<at::Half>()),
                 n
             );
        }
    } else if (x.scalar_type() == torch::kBFloat16) {
        if (n % 8 == 0 && reinterpret_cast<std::uintptr_t>(x.data_ptr()) % 16 == 0 && reinterpret_cast<std::uintptr_t>(y.data_ptr()) % 16 == 0) {
            blocks = std::min((n / 8 + threads - 1) / threads, 65535);
            gelu_bf168_kernel<<<blocks, threads>>>(
                reinterpret_cast<const float4*>(x.data_ptr<at::BFloat16>()),
                reinterpret_cast<float4*>(y.data_ptr<at::BFloat16>()),
                n / 8
            );
        } else {
             gelu_bf16_kernel<<<blocks, threads>>>(
                 reinterpret_cast<const nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
                 reinterpret_cast<nv_bfloat16*>(y.data_ptr<at::BFloat16>()),
                 n
             );
        }
    }
    
    return y;
}
'''

ext = load_inline(
    name='my_ext',
    cpp_sources='torch::Tensor my_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-U__CUDA_NO_HALF_OPERATORS__', '-U__CUDA_NO_HALF_CONVERSIONS__', '-U__CUDA_NO_BFLOAT16_CONVERSIONS__', '-U__CUDA_NO_HALF2_OPERATORS__']
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.my_op(x)

batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []  # No special initialization inputs needed
