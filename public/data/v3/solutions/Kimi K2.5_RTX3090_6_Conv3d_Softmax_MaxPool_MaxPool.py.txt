import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size, bias=False)
        self.pool1 = nn.MaxPool3d(pool_kernel_size)
        self.pool2 = nn.MaxPool3d(pool_kernel_size)
        
        # Initialize like the reference would
        nn.init.kaiming_uniform_(self.conv.weight, a=math.sqrt(5))
        
    def forward(self, x):
        # Channels-last format for potential speedup
        x = self.conv(x)
        # Softmax over channels
        x_exp = torch.exp(x - x.max(dim=1, keepdim=True)[0])
        x = x_exp / x_exp.sum(dim=1, keepdim=True)
        x = self.pool1(x)
        x = self.pool2(x)
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
