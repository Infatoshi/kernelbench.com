import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void conv_stats_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    float* __restrict__ sum,
    float* __restrict__ sumsq,
    int N,int Cin,int H,int W,
    int Cout,int KH,int KW,
    int Ho,int Wo)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*Cout*Ho*Wo;
    if(idx>=total) return;

    int wo = idx % Wo;
    int ho = (idx / Wo) % Ho;
    int co = (idx / (Wo*Ho)) % Cout;
    int n  = idx / (Wo*Ho*Cout);

    float v = b ? b[co] : 0.0f;

    for(int ci=0; ci<Cin; ++ci){
        for(int ky=0; ky<KH; ++ky){
            for(int kx=0; kx<KW; ++kx){
                int iy = ho + ky;
                int ix = wo + kx;
                int x_idx = ((n*Cin+ci)*H + iy)*W + ix;
                int w_idx = ((co*Cin+ci)*KH + ky)*KW + kx;
                v += x[x_idx]*w[w_idx];
            }
        }
    }

    y[idx] = v;

    int sc = n*Cout + co;
    atomicAdd(&sum[sc], v);
    atomicAdd(&sumsq[sc], v*v);
}

__global__ void norm_div_kernel(
    float* __restrict__ y,
    const float* __restrict__ sum,
    const float* __restrict__ sumsq,
    int N,int Cout,int Ho,int Wo,
    float eps,float divv)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*Cout*Ho*Wo;
    if(idx>=total) return;

    int wo = idx % Wo;
    int ho = (idx / Wo) % Ho;
    int co = (idx / (Wo*Ho)) % Cout;
    int n  = idx / (Wo*Ho*Cout);

    int sc = n*Cout + co;
    float HW = (float)(Ho*Wo);
    float mean = sum[sc]/HW;
    float var = sumsq[sc]/HW - mean*mean;
    float v = y[idx];
    v = (v - mean) / sqrtf(var + eps);
    v = v / divv;
    y[idx] = v;
}

torch::Tensor fused(torch::Tensor x, torch::Tensor w, torch::Tensor b, float divv){
    int N=x.size(0), Cin=x.size(1), H=x.size(2), W=x.size(3);
    int Cout=w.size(0), KH=w.size(2), KW=w.size(3);
    int Ho = H-KH+1;
    int Wo = W-KW+1;

    auto y = torch::empty({N,Cout,Ho,Wo}, x.options());
    auto sum = torch::zeros({N,Cout}, x.options());
    auto sumsq = torch::zeros({N,Cout}, x.options());

    int total = N*Cout*Ho*Wo;
    int threads = 256;
    int blocks = (total + threads - 1)/threads;

    conv_stats_kernel<<<blocks,threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.numel()?b.data_ptr<float>():nullptr,
        y.data_ptr<float>(),
        sum.data_ptr<float>(),
        sumsq.data_ptr<float>(),
        N,Cin,H,W,Cout,KH,KW,Ho,Wo);

    norm_div_kernel<<<blocks,threads>>>(
        y.data_ptr<float>(),
        sum.data_ptr<float>(),
        sumsq.data_ptr<float>(),
        N,Cout,Ho,Wo,1e-5f,divv);

    return y;
}
'''

ext = load_inline(
    name='fused_conv_instnorm_div',
    cpp_sources='torch::Tensor fused(torch::Tensor x, torch::Tensor w, torch::Tensor b, float divv);',
    cuda_sources=cuda_source,
    functions=['fused'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = divide_by

    def forward(self, x):
        x = x.contiguous()
        w = self.conv.weight.contiguous()
        b = self.conv.bias.contiguous() if self.conv.bias is not None else torch.empty(0, device=x.device)
        return ext.fused(x, w, b, float(self.divide_by))


def get_inputs():
    return [torch.randn(16,3,256,256, device='cuda')]


def get_init_inputs():
    return [3,16,3,2.0]
