import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_matmul_gelu_softmax_kernel(
    X, W, B, Y,
    stride_xm, stride_xk,
    stride_wk, stride_wn,
    stride_ym, stride_yn,
    M, N, K,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """
    Fused kernel: Matmul + GELU + Softmax
    X: [M, K] input
    W: [N, K] weight (transposed for row-major access)
    B: [N] bias
    Y: [M, N] output
    """
    pid_m = tl.program_id(axis=0)
    pid_n = tl.program_id(axis=1)
    
    # Compute row and column indices
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    
    # Load input and weights, compute matmul
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    for k in range(0, K, BLOCK_K):
        offs_k = k + tl.arange(0, BLOCK_K)
        
        # Load X block
        x_ptrs = X + offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk
        x_mask = (offs_m[:, None] < M) & (offs_k[None, :] < K)
        x = tl.load(x_ptrs, mask=x_mask, other=0.0)
        
        # Load W block (weight is [N, K], so we access W[n, k])
        w_ptrs = W + offs_n[None, :] * stride_wk + offs_k[:, None] * stride_wn
        w_mask = (offs_k[:, None] < K) & (offs_n[None, :] < N)
        w = tl.load(w_ptrs, mask=w_mask, other=0.0)
        
        # Matrix multiply
        acc = tl.dot(x, w, acc)
    
    # Add bias
    bias = tl.load(B + offs_n, mask=offs_n < N, other=0.0)
    acc = acc + bias[None, :]
    
    # Apply GELU: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    x_gelu = acc
    x3 = x_gelu * x_gelu * x_gelu
    coeff = 0.7978845608028654  # sqrt(2/pi)
    tanh_arg = coeff * (x_gelu + 0.044715 * x3)
    gelu_out = 0.5 * x_gelu * (1.0 + tl.libdevice.tanh(tanh_arg))
    
    # Apply Softmax along dim=1 (columns)
    # First, compute max for numerical stability
    row_max = tl.max(gelu_out, axis=1, keep_dims=True)
    gelu_out = gelu_out - row_max
    exp_out = tl.exp(gelu_out)
    row_sum = tl.sum(exp_out, axis=1, keep_dims=True)
    softmax_out = exp_out / (row_sum + 1e-9)
    
    # Store result
    out_ptrs = Y + offs_m[:, None] * stride_ym + offs_n[None, :] * stride_yn
    out_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(out_ptrs, softmax_out, mask=out_mask)


class FusedMatmulGeluSoftmax(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias):
        M, K = x.shape
        N = weight.shape[0]
        
        # Ensure contiguous
        x = x.contiguous()
        weight = weight.contiguous()
        if bias is not None:
            bias = bias.contiguous()
        else:
            bias = torch.zeros(N, dtype=x.dtype, device=x.device)
        
        y = torch.empty((M, N), dtype=torch.float32, device=x.device)
        
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_K = 32
        
        grid = (triton.cdiv(M, BLOCK_M), triton.cdiv(N, BLOCK_N))
        
        fused_matmul_gelu_softmax_kernel[grid](
            x, weight, bias, y,
            x.stride(0), x.stride(1),
            weight.stride(0), weight.stride(1),
            y.stride(0), y.stride(1),
            M, N, K,
            BLOCK_M, BLOCK_N, BLOCK_K,
        )
        
        ctx.save_for_backward(x, weight, bias)
        return y


class Model(nn.Module):
    """
    Optimized model that performs a matrix multiplication, applies GELU, and then applies Softmax.
    Uses fused Triton kernel for better performance.
    """
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        # Use the fused kernel
        weight = self.linear.weight
        bias = self.linear.bias
        return FusedMatmulGeluSoftmax.apply(x, weight, bias)


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [4096, 4096]
