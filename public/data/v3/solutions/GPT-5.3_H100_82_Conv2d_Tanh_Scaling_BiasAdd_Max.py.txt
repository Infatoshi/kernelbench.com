import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void fused_kernel(
    const scalar_t* __restrict__ x,
    const scalar_t* __restrict__ w,
    const scalar_t* __restrict__ conv_bias,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ out,
    int N, int Cin, int H, int W,
    int Cout, int K,
    int Hout, int Wout,
    int Hp, int Wp,
    float scale)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * Hp * Wp;
    if (idx >= total) return;

    int wp = idx % Wp;
    int hp = (idx / Wp) % Hp;
    int co = (idx / (Wp*Hp)) % Cout;
    int n = idx / (Wp*Hp*Cout);

    scalar_t max_val = -1e20;

    for (int ph = 0; ph < 2; ++ph) {
        for (int pw = 0; pw < 2; ++pw) {
            int h_out = hp*2 + ph;
            int w_out = wp*2 + pw;

            if (h_out >= Hout || w_out >= Wout) continue;

            float acc = (float)conv_bias[co];

            for (int ci = 0; ci < Cin; ++ci) {
                for (int kh = 0; kh < K; ++kh) {
                    for (int kw = 0; kw < K; ++kw) {
                        int h_in = h_out + kh;
                        int w_in = w_out + kw;

                        int x_idx = ((n*Cin + ci)*H + h_in)*W + w_in;
                        int w_idx = ((co*Cin + ci)*K + kh)*K + kw;

                        acc += (float)x[x_idx] * (float)w[w_idx];
                    }
                }
            }

            float v = tanhf(acc);
            v = v * scale + (float)bias[co];

            max_val = max_val > (scalar_t)v ? max_val : (scalar_t)v;
        }
    }

    out[idx] = max_val;
}

torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor conv_bias, torch::Tensor bias, float scale) {
    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int Cout = w.size(0);
    int K = w.size(2);

    int Hout = H - K + 1;
    int Wout = W - K + 1;

    int Hp = Hout / 2;
    int Wp = Wout / 2;

    auto out = torch::empty({N, Cout, Hp, Wp}, x.options());

    int total = N * Cout * Hp * Wp;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "fused_kernel", ([&] {
        fused_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            w.data_ptr<scalar_t>(),
            conv_bias.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            out.data_ptr<scalar_t>(),
            N, Cin, H, W, Cout, K,
            Hout, Wout, Hp, Wp, scale);
    }));

    return out;
}
'''

ext = load_inline(
    name='fused_conv_tanh_pool',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor conv_bias, torch::Tensor bias, float scale);',
    cuda_sources=cuda_src,
    functions=['fused_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        self.conv_bias = nn.Parameter(torch.randn(out_channels))
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        return ext.fused_op(x, self.weight, self.conv_bias, self.bias.view(-1), float(self.scaling_factor))


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 2.0, (16,1,1), 2]
