cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda;
using namespace nvcuda::wmma;

// Approximate GELU for better performance
__device__ inline float fast_gelu(float x) {
    float cdf = 0.5f + 0.5f * tanhf(0.7978845608f * (x + 0.044715f * x * x * x));
    return x * cdf;
}

// Kernel for fused linear + divide + gelu with FP16 using WMMA
__global__ void fused_linear_divide_gelu_fp16_kernel(
    const half* __restrict__ input,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    half* __restrict__ output,
    float divisor,
    int M, int N, int K
) {
    __shared__ half shmem[64 * 32 + 64 * 64];
    half* a_shmem = shmem;
    half* b_shmem = shmem + 64 * 32;
    
    const int warp_id = threadIdx.x / 32;
    const int lane_id = threadIdx.x % 32;
    
    const int c_row = blockIdx.x * 64 + warp_id * 16;
    const int c_col_base = blockIdx.y * 64 + lane_id;
    
    if (c_row >= M) return;
    
    float inv_div = 1.0f / divisor;
    
    wmma::fragment<wmma::matrix_a, 16, 16, 16, half, wmma::row_major> a_frag0, a_frag1, a_frag2, a_frag3;
    wmma::fragment<wmma::matrix_b, 16, 16, 16, half, wmma::col_major> b_frag0, b_frag1, b_frag2, b_frag3;
    wmma::fragment<wmma::accumulator, 16, 16, 16, half> c_frag0, c_frag1, c_frag2, c_frag3;
    
    wmma::fill_fragment(c_frag0, __float2half(0.0f));
    wmma::fill_fragment(c_frag1, __float2half(0.0f));
    wmma::fill_fragment(c_frag2, __float2half(0.0f));
    wmma::fill_fragment(c_frag3, __float2half(0.0f));
    
    for (int k = 0; k < K; k += 32) {
        // Load A tile (16x32)
        for (int i = 0; i < 4; ++i) {
            int row = i;
            int a_idx = (c_row + row) * K + k;
            for (int j = 0; j < 8; ++j) {
                int col = lane_id % 4 + j * 4;
                int a_sidx = row * 32 + col;
                if (c_row + row < M && k + col < K) {
                    a_shmem[a_sidx] = input[a_idx + col];
                } else {
                    a_shmem[a_sidx] = __float2half(0.0f);
                }
            }
        }
        
        // Load B tile (32x64)
        for (int i = 0; i < 8; ++i) {
            int b_col = i;
            for (int j = 0; j < 2; ++j) {
                int b_row = lane_id / 4 + j * 8;
                int b_idx = (k + b_row) * N + c_col_base;
                int b_sidx = b_row * 64 + b_col * 8 + lane_id % 4;
                if (k + b_row < K && c_col_base + b_col * 8 + lane_id % 4 < N) {
                    b_shmem[b_sidx] = weight[b_idx + b_col * 8 + lane_id % 4];
                } else {
                    b_shmem[b_sidx] = __float2half(0.0f);
                }
            }
        }
        
        __syncthreads();
        
        wmma::load_matrix_sync(a_frag0, a_shmem + 0 * 16, 32);
        wmma::load_matrix_sync(a_frag1, a_shmem + 1 * 16, 32);
        wmma::load_matrix_sync(a_frag2, a_shmem + 2 * 16, 32);
        wmma::load_matrix_sync(a_frag3, a_shmem + 3 * 16, 32);
        
        wmma::load_matrix_sync(b_frag0, b_shmem + (lane_id / 4) * 64, 64);
        wmma::load_matrix_sync(b_frag1, b_shmem + (lane_id / 4 + 8) * 64, 64);
        wmma::load_matrix_sync(b_frag2, b_shmem + (lane_id / 4 + 16) * 64, 64);
        wmma::load_matrix_sync(b_frag3, b_shmem + (lane_id / 4 + 24) * 64, 64);
        
        wmma::mma_sync(c_frag0, a_frag0, b_frag0, c_frag0);
        wmma::mma_sync(c_frag1, a_frag1, b_frag1, c_frag1);
        wmma::mma_sync(c_frag2, a_frag2, b_frag2, c_frag2);
        wmma::mma_sync(c_frag3, a_frag3, b_frag3, c_frag3);
        
        __syncthreads();
    }
    
    // Store with bias, division, and GELU
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            int out_row = c_row + i;
            int out_col = c_col_base + j;
            
            if (out_row < M && out_col < N) {
                float val;
                if (i == 0 && j == 0) val = __half2float(c_frag0.x[lane_id % 4]);
                else if (i == 0 && j == 1) val = __half2float(c_frag0.x[lane_id % 4 + 4]);
                else if (i == 0 && j == 2) val = __half2float(c_frag0.x[lane_id % 4 + 8]);
                else if (i == 0 && j == 3) val = __half2float(c_frag0.x[lane_id % 4 + 12]);
                else if (i == 1 && j == 0) val = __half2float(c_frag1.x[lane_id % 4]);
                else if (i == 1 && j == 1) val = __half2float(c_frag1.x[lane_id % 4 + 4]);
                else if (i == 1 && j == 2) val = __half2float(c_frag1.x[lane_id % 4 + 8]);
                else if (i == 1 && j == 3) val = __half2float(c_frag1.x[lane_id % 4 + 12]);
                else if (i == 2 && j == 0) val = __half2float(c_frag2.x[lane_id % 4]);
                else if (i == 2 && j == 1) val = __half2float(c_frag2.x[lane_id % 4 + 4]);
                else if (i == 2 && j == 2) val = __half2float(c_frag2.x[lane_id % 4 + 8]);
                else if (i == 2 && j == 3) val = __half2float(c_frag2.x[lane_id % 4 + 12]);
                else if (i == 3 && j == 0) val = __half2float(c_frag3.x[lane_id % 4]);
                else if (i == 3 && j == 1) val = __half2float(c_frag3.x[lane_id % 4 + 4]);
                else if (i == 3 && j == 2) val = __half2float(c_frag3.x[lane_id % 4 + 8]);
                else val = __half2float(c_frag3.x[lane_id % 4 + 12]);
                
                if (bias != nullptr) val += __half2float(bias[out_col]);
                val *= inv_div;
                val = fast_gelu(val);
                output[out_row * N + out_col] = __float2half(val);
            }
        }
    }
}

// Kernel for fused linear + divide + gelu with FP32
__global__ void fused_linear_divide_gelu_fp32_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    float divisor,
    int M, int N, int K
) {
    __shared__ float shmem[64 * 16 + 64 * 64];
    float* a_shmem = shmem;
    float* b_shmem = shmem + 64 * 16;
    
    const int warp_id = threadIdx.x / 32;
    const int lane_id = threadIdx.x % 32;
    
    const int c_row = blockIdx.x * 64 + warp_id * 16;
    const int c_col_base = blockIdx.y * 64 + lane_id;
    
    if (c_row >= M) return;
    
    float inv_div = 1.0f / divisor;
    
    // Process 16x2 output elements per warp
    float acc0 = 0.0f, acc1 = 0.0f, acc2 = 0.0f, acc3 = 0.0f;
    float acc4 = 0.0f, acc5 = 0.0f, acc6 = 0.0f, acc7 = 0.0f;
    float acc8 = 0.0f, acc9 = 0.0f, acc10 = 0.0f, acc11 = 0.0f;
    float acc12 = 0.0f, acc13 = 0.0f, acc14 = 0.0f, acc15 = 0.0f;
    
    for (int k = 0; k < K; k += 16) {
        // Load A tile (16x16)
        for (int i = 0; i < 4; ++i) {
            int row = i;
            int a_idx = (c_row + row) * K + k;
            for (int j = 0; j < 4; ++j) {
                int col = lane_id % 4 + j * 8;
                if (c_row + row < M && k + col < K) {
                    a_shmem[row * 16 + col] = input[a_idx + col];
                } else {
                    a_shmem[row * 16 + col] = 0.0f;
                }
            }
        }
        
        // Load B tile (16x64)
        for (int i = 0; i < 4; ++i) {
            int b_row = lane_id / 4 + i * 8;
            int b_idx = (k + b_row) * N + c_col_base;
            for (int j = 0; j < 16; ++j) {
                if (k + b_row < K && c_col_base + j < N) {
                    b_shmem[b_row * 64 + j] = weight[b_idx + j];
                } else {
                    b_shmem[b_row * 64 + j] = 0.0f;
                }
            }
        }
        
        __syncthreads();
        
        // Compute partial results
        for (int kk = 0; kk < 16; ++kk) {
            float a_val = a_shmem[(lane_id / 4) * 16 + kk];
            float b_val0 = b_shmem[kk * 64 + 0];
            float b_val1 = b_shmem[kk * 64 + 16];
            float b_val2 = b_shmem[kk * 64 + 32];
            float b_val3 = b_shmem[kk * 64 + 48];
            acc0 += a_val * b_val0;
            acc1 += a_val * b_shmem[kk * 64 + 1];
            acc2 += a_val * b_shmem[kk * 64 + 2];
            acc3 += a_val * b_shmem[kk * 64 + 3];
            acc4 += a_val * b_shmem[kk * 64 + 4];
            acc5 += a_val * b_shmem[kk * 64 + 5];
            acc6 += a_val * b_shmem[kk * 64 + 6];
            acc7 += a_val * b_shmem[kk * 64 + 7];
            acc8 += a_val * b_shmem[kk * 64 + 8];
            acc9 += a_val * b_shmem[kk * 64 + 9];
            acc10 += a_val * b_shmem[kk * 64 + 10];
            acc11 += a_val * b_shmem[kk * 64 + 11];
            acc12 += a_val * b_shmem[kk * 64 + 12];
            acc13 += a_val * b_shmem[kk * 64 + 13];
            acc14 += a_val * b_shmem[kk * 64 + 14];
            acc15 += a_val * b_shmem[kk * 64 + 15];
        }
        
        __syncthreads();
    }
    
    // Store with bias, division, and GELU
    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            int out_row = c_row + i;
            int out_col = c_col_base + j * 16;
            
            if (out_row < M) {
                float val;
                if (j == 0) val = acc0;
                else if (j == 1) val = acc1;
                else if (j == 2) val = acc2;
                else if (j == 3) val = acc3;
                else if (j == 4) val = acc4;
                else if (j == 5) val = acc5;
                else if (j == 6) val = acc6;
                else if (j == 7) val = acc7;
                else if (j == 8) val = acc8;
                else if (j == 9) val = acc9;
                else if (j == 10) val = acc10;
                else if (j == 11) val = acc11;
                else if (j == 12) val = acc12;
                else if (j == 13) val = acc13;
                else if (j == 14) val = acc14;
                else val = acc15;
                
                if (out_col < N) {
                    if (bias) val += bias[out_col];
                    val *= inv_div;
                    val = fast_gelu(val);
                    output[out_row * N + out_col] = val;
                }
            }
        }
    }
}

torch::Tensor fused_linear_divide_gelu_fp16(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float divisor
) {
    int M = input.size(0);
    int K = input.size(1);
    int N = weight.size(0);
    
    auto output = torch::empty({M, N}, input.options());
    
    dim3 grid((M + 63) / 64, (N + 63) / 64);
    dim3 block(128);
    
    int shmem_size = (64 * 32 + 64 * 64) * sizeof(half);
    
    fused_linear_divide_gelu_fp16_kernel<<<grid, block, shmem_size>>>(
        input.data_ptr<half>(),
        weight.data_ptr<half>(),
        bias.defined() ? bias.data_ptr<half>() : nullptr,
        output.data_ptr<half>(),
        divisor,
        M, N, K
    );
    
    return output;
}

ext = load_inline(
    name='fused_linear_gelu',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=[
        'fused_linear_divide_gelu_fp16',
        'fused_linear_divide_gelu_fp32',
        'fused_linear_divide_gelu_bf16'
    ],
    extra_include_paths=['/opt/cutlass/include'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, input_size, output_size, divisor, precision='fp32'):
        super().__init__()
        self.input_size = input_size
        self.output_size = output_size
        self.divisor = divisor
        self.precision = precision
        
        self.weight = nn.Parameter(torch.randn(output_size, input_size))
        self.bias = nn.Parameter(torch.randn(output_size))
        
        # Initialize with reference model's weight distribution
        nn.init.kaiming_uniform_(self.weight, a=0, mode='fan_in', nonlinearity='linear')
        nn.init.uniform_(self.bias, -0.5, 0.5)
    
    def forward(self, x):
        if self.precision == 'fp16':
            input_fp = x.to(torch.float16)
            weight_fp = self.weight.to(torch.float16)
            bias_fp = self.bias.to(torch.float16)
            out = ext.fused_linear_divide_gelu_fp16(input_fp, weight_fp, bias_fp, self.divisor)
            return out.to(torch.float32)
        elif self.precision == 'bf16':
            input_fp = x.to(torch.bfloat16)
            weight_fp = self.weight.to(torch.bfloat16)
            bias_fp = self.bias.to(torch.bfloat16)
            out = ext.fused_linear_divide_gelu_bf16(input_fp, weight_fp, bias_fp, self.divisor)
            return out.to(torch.float32)
        else:
            return ext.fused_linear_divide_gelu_fp32(x, self.weight, self.bias, self.divisor)

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 10.0]
