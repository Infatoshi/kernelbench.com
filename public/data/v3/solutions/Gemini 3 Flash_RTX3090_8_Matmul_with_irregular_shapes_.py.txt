import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

struct CublasContext {
    cublasHandle_t handle;
    CublasContext() { cublasCreate(&handle); }
    ~CublasContext() { cublasDestroy(handle); }
};

torch::Tensor matmul_irregular(torch::Tensor A, torch::Tensor B) {
    static CublasContext ctx;
    
    auto M = A.size(0);
    auto K = A.size(1);
    auto N = B.size(1);
    auto C = torch::empty({M, N}, A.options());

    float alpha = 1.0f;
    float beta = 0.0f;

    cublasSgemm(ctx.handle, CUBLAS_OP_N, CUBLAS_OP_N, 
                N, M, K, 
                &alpha, 
                B.data_ptr<float>(), B.stride(0), // leading dim is N
                A.data_ptr<float>(), A.stride(0), // leading dim is K
                &beta, 
                C.data_ptr<float>(), C.size(1)); // leading dim is N

    return C;
}
'''

ext = load_inline(
    name='matmul_ext',
    cpp_sources='torch::Tensor matmul_irregular(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matmul_irregular'],
    with_cuda=True,
    extra_ldflags=['-lcublas'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        A = A.contiguous()
        B = B.contiguous()
        return ext.matmul_irregular(A, B)

def get_inputs():
    M, K, N = 8205, 2949, 5921
    A = torch.randn(M, K, device='cuda')
    B = torch.randn(K, N, device='cuda')
    return [A, B]
