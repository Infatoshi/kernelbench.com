
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void gemv_fp4_kernel(
    const int8_t* __restrict__ x_q,
    const float x_scale,
    const int8_t* __restrict__ w_q,
    const float w_scale,
    at::Half* __restrict__ out,
    int M, int K, int N) // M=batch, K=in_features, N=out_features
{
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    int batch_idx = blockIdx.y;

    if (row < N && batch_idx < M) {
        float sum = 0.0f;
        const int8_t* x_row = x_q + batch_idx * K;
        const int8_t* w_col = w_q + row * K;
        
        int k = 0;
        // Use dp4a style if we want, but let's just stick to vector loads for now.
        // For Blackwell, tcgen05.mma would be the way but we are doing a simpler GEMV-like kernel.
        // The scale is constant per weight/activation, so we can do it at the end.
        
        #pragma unroll 4
        for (; k <= K - 16; k += 16) {
            int4 x4 = *(int4*)(x_row + k);
            int4 w4 = *(int4*)(w_col + k);
            
            #pragma unroll
            for(int i=0; i<4; ++i) {
               // dp4a handles 4 bytes at a time
               sum = __dp4a(((int*)&x4)[i], ((int*)&w4)[i], sum);
            }
        }
        for (; k < K; ++k) {
            sum += (float)x_row[k] * (float)w_col[k];
        }
        out[batch_idx * N + row] = (at::Half)__float2half(sum * (x_scale * w_scale));
    }
}

torch::Tensor gemv_fp4_op(torch::Tensor x_q, float x_scale, torch::Tensor w_q, float w_scale) {
    auto M = x_q.size(0);
    auto K = x_q.size(1);
    auto N = w_q.size(0);
    auto options = torch::TensorOptions().dtype(torch::kHalf).device(x_q.device());
    auto out = torch::empty({M, N}, options);

    dim3 block(256);
    dim3 grid((N + block.x - 1) / block.x, M);

    gemv_fp4_kernel<<<grid, block>>>(
        (const int8_t*)x_q.data_ptr(),
        x_scale,
        (const int8_t*)w_q.data_ptr(),
        w_scale,
        (at::Half*)out.data_ptr(),
        M, K, N
    );
    return out;
}
'''

cpp_source = 'torch::Tensor gemv_fp4_op(torch::Tensor x_q, float x_scale, torch::Tensor w_q, float w_scale);'

ext = load_inline(
    name='gemv_fp4_ext4',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp4_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8).contiguous())
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp4_op(x_q, x_scale.item(), self.weight_q, self.weight_scale.item())

def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.08, dtype=torch.float32)
    return [x_q, x_scale]

def get_init_inputs():
    return [4096, 14336]
