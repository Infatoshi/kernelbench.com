import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rms_norm_kernel(
    x_ptr,          # Pointer to input tensor
    output_ptr,     # Pointer to output tensor
    num_features,   # Number of features (dim 1)
    eps,            # Epsilon for numerical stability
    stride_x_batch, # Stride for batch dimension in x
    stride_x_feat,  # Stride for feature dimension in x
    stride_out_batch, # Stride for batch dimension in output
    stride_out_feat,  # Stride for feature dimension in output
    total_elements_per_sample,  # Total elements per sample (excluding batch)
    BLOCK_SIZE: tl.constexpr,   # Block size for feature dimension
):
    """
    RMSNorm kernel using Triton.
    Processes one row (one sample's feature vector) at a time.
    """
    # Get the row index (flattened batch * remaining_dims)
    row_idx = tl.program_id(0)
    
    # Compute base pointers for this row
    x_row_ptr = x_ptr + row_idx * total_elements_per_sample
    out_row_ptr = output_ptr + row_idx * total_elements_per_sample
    
    # Load all elements for this row and compute sum of squares
    # We process in blocks for large feature dimensions
    sum_sq = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for block_start in range(0, num_features, BLOCK_SIZE):
        offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < num_features
        
        # Load input values
        x_vals = tl.load(x_row_ptr + offsets * stride_x_feat, mask=mask, other=0.0)
        
        # Convert to float32 for computation
        x_vals_f32 = x_vals.to(tl.float32)
        
        # Accumulate sum of squares
        sum_sq += x_vals_f32 * x_vals_f32
    
    # Reduce sum of squares across the block
    total_sum_sq = tl.sum(sum_sq)
    
    # Compute RMS
    rms = tl.sqrt(total_sum_sq / num_features + eps)
    
    # Inverse RMS for normalization (avoid division)
    inv_rms = 1.0 / rms
    
    # Normalize and store
    for block_start in range(0, num_features, BLOCK_SIZE):
        offsets = block_start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < num_features
        
        # Load input values again
        x_vals = tl.load(x_row_ptr + offsets * stride_x_feat, mask=mask, other=0.0)
        
        # Normalize
        x_vals_f32 = x_vals.to(tl.float32)
        normalized = x_vals_f32 * inv_rms
        
        # Store result (preserve original dtype)
        tl.store(out_row_ptr + offsets * stride_x_feat, normalized.to(x_vals.dtype), mask=mask)


@triton.jit
def rms_norm_kernel_fused(
    x_ptr,          # Pointer to input tensor (row-major)
    output_ptr,     # Pointer to output tensor
    num_features,   # Number of features (dim 1)
    eps,            # Epsilon for numerical stability
    stride_x_feat,  # Stride for feature dimension in x
    stride_out_feat,  # Stride for feature dimension in output
    BLOCK_FEATURES: tl.constexpr,   # Block size for feature dimension
):
    """
    Optimized RMSNorm kernel for row-major tensors.
    Each program handles one row (one sample).
    """
    row_idx = tl.program_id(0)
    
    # Compute base pointers
    x_row_ptr = x_ptr + row_idx * num_features * stride_x_feat
    out_row_ptr = output_ptr + row_idx * num_features * stride_out_feat
    
    # Compute sum of squares using a reduction
    sum_sq = 0.0
    
    # Process in blocks
    for start in range(0, num_features, BLOCK_FEATURES):
        offsets = start + tl.arange(0, BLOCK_FEATURES)
        mask = offsets < num_features
        
        x_vals = tl.load(x_row_ptr + offsets, mask=mask, other=0.0)
        x_f32 = x_vals.to(tl.float32)
        sum_sq += tl.sum(x_f32 * x_f32 * mask.to(tl.float32))
    
    # Compute RMS
    mean_sq = sum_sq / num_features
    rms = tl.sqrt(mean_sq + eps)
    inv_rms = 1.0 / rms
    
    # Normalize and store
    for start in range(0, num_features, BLOCK_FEATURES):
        offsets = start + tl.arange(0, BLOCK_FEATURES)
        mask = offsets < num_features
        
        x_vals = tl.load(x_row_ptr + offsets, mask=mask, other=0.0)
        x_f32 = x_vals.to(tl.float32)
        normalized = x_f32 * inv_rms
        
        tl.store(out_row_ptr + offsets, normalized.to(x_vals.dtype), mask=mask)


def rms_norm_forward(x: torch.Tensor, eps: float = 1e-5) -> torch.Tensor:
    """
    Apply RMS normalization along dimension 1.
    Input shape: (batch_size, num_features, *rest)
    Output shape: same as input
    """
    assert x.is_cuda, "Input must be on CUDA"
    
    # Get input shape
    original_shape = x.shape
    batch_size = x.shape[0]
    num_features = x.shape[1]
    
    # Calculate the size of remaining dimensions
    rest_dims = list(original_shape[2:])
    rest_size = 1
    for d in rest_dims:
        rest_size *= d
    
    # Reshape to (batch * rest, num_features) for processing
    # First, move feature dimension to the end, then reshape
    if len(original_shape) > 2:
        # Permute to (batch, *rest, num_features)
        perm = [0] + list(range(2, len(original_shape))) + [1]
        x_permuted = x.permute(*perm)
        # Reshape to (batch * rest, num_features)
        x_flat = x_permuted.reshape(-1, num_features)
    else:
        x_flat = x.reshape(-1, num_features)
    
    # Create output tensor
    output_flat = torch.empty_like(x_flat)
    
    # Number of rows to process
    num_rows = x_flat.shape[0]
    
    # Choose block size based on num_features
    # Use power of 2, at most 1024
    BLOCK_SIZE = min(1024, triton.next_power_of_2(num_features))
    if BLOCK_SIZE < 32:
        BLOCK_SIZE = 32
    
    # Launch kernel
    grid = (num_rows,)
    
    rms_norm_kernel_fused[grid](
        x_flat,
        output_flat,
        num_features,
        eps,
        1,  # stride_x_feat (row-major after reshape)
        1,  # stride_out_feat
        BLOCK_FEATURES=BLOCK_SIZE,
    )
    
    # Reshape back to original shape
    if len(original_shape) > 2:
        # output_flat is (batch * rest, num_features)
        # Reshape to (batch, *rest, num_features)
        output = output_flat.reshape(batch_size, *rest_dims, num_features)
        # Permute back from (batch, *rest, feat) to (batch, feat, *rest)
        inv_perm = [0, len(original_shape) - 1] + list(range(1, len(original_shape) - 1))
        output = output.permute(*inv_perm)
        output = output.reshape(original_shape)
    else:
        output = output_flat.reshape(original_shape)
    
    return output


class Model(nn.Module):
    """
    RMSNorm model using optimized Triton kernel.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return rms_norm_forward(x, self.eps)

    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]
