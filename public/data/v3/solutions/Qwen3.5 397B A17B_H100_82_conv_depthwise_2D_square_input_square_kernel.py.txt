import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

using half_t = at::Half;
using bfloat16_t = at::BFloat16;

// Depthwise 2D convolution kernel optimized for H100
// Each channel is processed independently

template<typename scalar_t>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ input,   // [N, C, H, W]
    const scalar_t* __restrict__ weight,  // [C, 1, K, K]
    scalar_t* __restrict__ output,        // [N, C, H_out, W_out]
    int N, int C, int H, int W,
    int K, int stride, int padding,
    int H_out, int W_out) {
    
    // Each block handles one (n, c) pair
    int n = blockIdx.z;
    int c = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (n >= N || c >= C) return;
    
    // Output position within the channel
    int base_idx = n * C * H_out * W_out + c * H_out * W_out;
    int input_base = n * C * H * W + c * H * W;
    int weight_base = c * K * K;
    
    // Process multiple output pixels per thread for better occupancy
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_pixels = H_out * W_out;
    
    for (int out_idx = tid; out_idx < total_pixels; out_idx += blockDim.x * gridDim.x) {
        int out_h = out_idx / W_out;
        int out_w = out_idx % W_out;
        
        int in_h_start = out_h * stride - padding;
        int in_w_start = out_w * stride - padding;
        
        scalar_t sum = 0;
        
        // Unroll kernel loop for small kernels
        #pragma unroll
        for (int kh = 0; kh < K; kh++) {
            #pragma unroll
            for (int kw = 0; kw < K; kw++) {
                int in_h = in_h_start + kh;
                int in_w = in_w_start + kw;
                
                if (in_h >= 0 && in_h < H && in_w >= 0 && in_w < W) {
                    scalar_t inp_val = input[input_base + in_h * W + in_w];
                    scalar_t wgt_val = weight[weight_base + kh * K + kw];
                    sum += inp_val * wgt_val;
                }
            }
        }
        
        output[base_idx + out_idx] = sum;
    }
}

// FP16 specialized kernel
__global__ void depthwise_conv2d_fp16_kernel(
    const half_t* __restrict__ input,
    const half_t* __restrict__ weight,
    half_t* __restrict__ output,
    int N, int C, int H, int W,
    int K, int stride, int padding,
    int H_out, int W_out) {
    
    int n = blockIdx.z;
    int c = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (n >= N || c >= C) return;
    
    int base_idx = n * C * H_out * W_out + c * H_out * W_out;
    int input_base = n * C * H * W + c * H * W;
    int weight_base = c * K * K;
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_pixels = H_out * W_out;
    
    // Load weights to registers
    half_t w[9];
    #pragma unroll
    for (int i = 0; i < K * K; i++) {
        w[i] = weight[weight_base + i];
    }
    
    for (int out_idx = tid; out_idx < total_pixels; out_idx += blockDim.x * gridDim.x) {
        int out_h = out_idx / W_out;
        int out_w = out_idx % W_out;
        
        int in_h_start = out_h * stride - padding;
        int in_w_start = out_w * stride - padding;
        
        float sum = 0;
        
        #pragma unroll
        for (int kh = 0; kh < K; kh++) {
            #pragma unroll
            for (int kw = 0; kw < K; kw++) {
                int in_h = in_h_start + kh;
                int in_w = in_w_start + kw;
                
                if (in_h >= 0 && in_h < H && in_w >= 0 && in_w < W) {
                    float inp_val = __half2float((half)input[input_base + in_h * W + in_w]);
                    float wgt_val = __half2float((half)w[kh * K + kw]);
                    sum += inp_val * wgt_val;
                }
            }
        }
        
        output[base_idx + out_idx] = (half_t)__float2half(sum);
    }
}

// BF16 specialized kernel
__global__ void depthwise_conv2d_bf16_kernel(
    const bfloat16_t* __restrict__ input,
    const bfloat16_t* __restrict__ weight,
    bfloat16_t* __restrict__ output,
    int N, int C, int H, int W,
    int K, int stride, int padding,
    int H_out, int W_out) {
    
    int n = blockIdx.z;
    int c = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (n >= N || c >= C) return;
    
    int base_idx = n * C * H_out * W_out + c * H_out * W_out;
    int input_base = n * C * H * W + c * H * W;
    int weight_base = c * K * K;
    
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int total_pixels = H_out * W_out;
    
    // Load weights to registers
    bfloat16_t w[9];
    #pragma unroll
    for (int i = 0; i < K * K; i++) {
        w[i] = weight[weight_base + i];
    }
    
    for (int out_idx = tid; out_idx < total_pixels; out_idx += blockDim.x * gridDim.x) {
        int out_h = out_idx / W_out;
        int out_w = out_idx % W_out;
        
        int in_h_start = out_h * stride - padding;
        int in_w_start = out_w * stride - padding;
        
        float sum = 0;
        
        #pragma unroll
        for (int kh = 0; kh < K; kh++) {
            #pragma unroll
            for (int kw = 0; kw < K; kw++) {
                int in_h = in_h_start + kh;
                int in_w = in_w_start + kw;
                
                if (in_h >= 0 && in_h < H && in_w >= 0 && in_w < W) {
                    float inp_val = __bfloat162float((nv_bfloat16)input[input_base + in_h * W + in_w]);
                    float wgt_val = __bfloat162float((nv_bfloat16)w[kh * K + kw]);
                    sum += inp_val * wgt_val;
                }
            }
        }
        
        output[base_idx + out_idx] = (bfloat16_t)__float2bfloat16(sum);
    }
}

torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, int stride, int padding) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int K = weight.size(2);
    
    int H_out = (H + 2 * padding - K) / stride + 1;
    int W_out = (W + 2 * padding - K) / stride + 1;
    
    auto output = torch::empty({N, C, H_out, W_out}, input.options());
    
    int threads_per_block = 256;
    int blocks_x = 32;
    dim3 block_dim(threads_per_block);
    dim3 grid_dim(blocks_x, (C + 7) / 8, N);
    
    if (input.dtype() == torch::kHalf) {
        depthwise_conv2d_fp16_kernel<<<grid_dim, block_dim>>>(
            input.data_ptr<half_t>(),
            weight.data_ptr<half_t>(),
            output.data_ptr<half_t>(),
            N, C, H, W, K, stride, padding, H_out, W_out);
    } else if (input.dtype() == torch::kBFloat16) {
        depthwise_conv2d_bf16_kernel<<<grid_dim, block_dim>>>(
            input.data_ptr<bfloat16_t>(),
            weight.data_ptr<bfloat16_t>(),
            output.data_ptr<bfloat16_t>(),
            N, C, H, W, K, stride, padding, H_out, W_out);
    } else {
        // FP32
        depthwise_conv2d_kernel<float><<<grid_dim, block_dim>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            output.data_ptr<float>(),
            N, C, H, W, K, stride, padding, H_out, W_out);
    }
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor depthwise_conv2d(torch::Tensor input, torch::Tensor weight, int stride, int padding);
"""

# Compile the extension
ext = load_inline(
    name='depthwise_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d'],
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
    verbose=False,
)


class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.bias = bias
        
        # Weight shape: [C, 1, K, K] for depthwise
        self.weight = nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        if bias:
            self.bias_param = nn.Parameter(torch.randn(in_channels))
        else:
            self.register_buffer('bias_param', torch.zeros(in_channels))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.depthwise_conv2d(x, self.weight, self.stride, self.padding)


def get_inputs():
    batch_size = 16
    in_channels = 3
    kernel_size = 3
    width = 256
    height = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
