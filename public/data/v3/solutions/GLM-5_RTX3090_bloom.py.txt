import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized bloom kernel with zero padding (matching avg_pool2d behavior)
template<typename T>
__global__ void bloom_kernel_opt(
    const T* __restrict__ image,
    T* __restrict__ output,
    const int N, const int C, const int H, const int W,
    const float threshold, const float intensity
) {
    const int halo = 2;  // for 5x5 kernel
    const int BLOCK_SIZE = 16;
    const int TILE_SIZE = BLOCK_SIZE + 2 * halo;  // 20
    
    __shared__ float s_tile[TILE_SIZE][TILE_SIZE];
    
    const int batch_idx = blockIdx.z / C;
    const int channel_idx = blockIdx.z % C;
    
    const int tile_x = blockIdx.x * BLOCK_SIZE;
    const int tile_y = blockIdx.y * BLOCK_SIZE;
    
    const int local_x = threadIdx.x;
    const int local_y = threadIdx.y;
    
    // Cooperative loading: each thread loads up to 4 elements
    // Store bright values (relu(image - threshold)) with zero padding at boundaries
    for (int dy = 0; dy < 2; dy++) {
        for (int dx = 0; dx < 2; dx++) {
            int load_y = local_y * 2 + dy;
            int load_x = local_x * 2 + dx;
            
            if (load_y < TILE_SIZE && load_x < TILE_SIZE) {
                int global_y = tile_y - halo + load_y;
                int global_x = tile_x - halo + load_x;
                
                // Zero padding for out-of-bounds (matching avg_pool2d with count_include_pad=True)
                if (global_y >= 0 && global_y < H && global_x >= 0 && global_x < W) {
                    int idx = (batch_idx * C + channel_idx) * H * W + global_y * W + global_x;
                    float val = static_cast<float>(image[idx]);
                    // Apply threshold and relu
                    s_tile[load_y][load_x] = val > threshold ? (val - threshold) : 0.0f;
                } else {
                    // Zero padding for out-of-bounds
                    s_tile[load_y][load_x] = 0.0f;
                }
            }
        }
    }
    
    __syncthreads();
    
    const int out_x = tile_x + local_x;
    const int out_y = tile_y + local_y;
    
    if (out_x < W && out_y < H) {
        const int cx = local_x + halo;
        const int cy = local_y + halo;
        
        // 5x5 blur (matching avg_pool2d with count_include_pad=True)
        const float kernel_weight = 0.04f;  // 1/25
        float blur_sum = 0.0f;
        
        #pragma unroll
        for (int ky = -2; ky <= 2; ky++) {
            #pragma unroll
            for (int kx = -2; kx <= 2; kx++) {
                blur_sum += s_tile[cy + ky][cx + kx];
            }
        }
        blur_sum *= kernel_weight;
        
        // Original pixel (no threshold applied)
        float original = static_cast<float>(image[(batch_idx * C + channel_idx) * H * W + out_y * W + out_x]);
        
        // Blend and clamp
        float result = original + intensity * blur_sum;
        result = fmaxf(0.0f, fminf(10.0f, result));
        
        int out_idx = (batch_idx * C + channel_idx) * H * W + out_y * W + out_x;
        output[out_idx] = static_cast<T>(result);
    }
}

torch::Tensor bloom_forward(
    torch::Tensor image,
    double threshold,
    double intensity,
    int64_t blur_kernel
) {
    const int N = image.size(0);
    const int C = image.size(1);
    const int H = image.size(2);
    const int W = image.size(3);
    
    auto output = torch::empty_like(image);
    
    const int BLOCK_SIZE = 16;
    dim3 block(BLOCK_SIZE, BLOCK_SIZE);
    dim3 grid((W + BLOCK_SIZE - 1) / BLOCK_SIZE,
              (H + BLOCK_SIZE - 1) / BLOCK_SIZE,
              N * C);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(image.scalar_type(), "bloom_forward", ([&] {
        bloom_kernel_opt<scalar_t><<<grid, block>>>(
            image.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            N, C, H, W,
            static_cast<float>(threshold),
            static_cast<float>(intensity)
        );
    }));
    
    return output;
}
'''

cpp_source = '''
torch::Tensor bloom_forward(torch::Tensor image, double threshold, double intensity, int64_t blur_kernel);
'''

ext = load_inline(
    name='bloom_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bloom_forward'],
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
    verbose=False,
)

class Model(nn.Module):
    """Bloom effect approximation: threshold -> blur -> additive blend."""
    
    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel
    
    def forward(self, image: torch.Tensor) -> torch.Tensor:
        return ext.bloom_forward(image, self.threshold, self.intensity, self.blur_kernel)

def get_inputs():
    return [torch.randn(2, 3, 1024, 1024) * 1.5 + 0.2]

def get_init_inputs():
    return [1.0, 0.6, 5]
