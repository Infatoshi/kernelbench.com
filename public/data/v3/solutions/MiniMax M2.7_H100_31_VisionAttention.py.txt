import torch
import torch.nn as nn
import torch.nn.functional as F

class OptimizedAttention(nn.Module):
    """Optimized multi-head self-attention using scaled_dot_product_attention"""
    
    def __init__(self, embed_dim, num_heads, dropout=0.0):
        super().__init__()
        assert embed_dim % num_heads == 0, "embed_dim must be divisible by num_heads"
        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        
        # Combined QKV projection
        self.qkv = nn.Linear(embed_dim, 3 * embed_dim, bias=False)
        # Output projection
        self.proj = nn.Linear(embed_dim, embed_dim)
        self.dropout = dropout
        
    def forward(self, x):
        """
        Forward pass with scaled_dot_product_attention
        x: (B, C, H, W) -> (B, embed_dim, H*W)
        """
        B, C, H, W = x.shape
        seq_len = H * W
        
        # Reshape to (B, seq_len, C)
        x = x.view(B, C, seq_len).permute(0, 2, 1)  # (B, seq_len, C)
        
        # QKV projection
        qkv = self.qkv(x)  # (B, seq_len, 3*C)
        qkv = qkv.reshape(B, seq_len, 3, self.num_heads, self.head_dim)
        qkv = qkv.permute(2, 0, 3, 1, 4)  # (3, B, num_heads, seq_len, head_dim)
        q, k, v = qkv[0], qkv[1], qkv[2]
        
        # Scaled dot-product attention (Flash Attention via cuDNN)
        attn = F.scaled_dot_product_attention(
            q, k, v,
            attn_mask=None,
            dropout_p=self.dropout if self.training else 0.0,
            is_causal=False
        )  # (B, num_heads, seq_len, head_dim)
        
        # Reshape and project
        attn = attn.transpose(1, 2).reshape(B, seq_len, C)  # (B, seq_len, C)
        out = self.proj(attn)  # (B, seq_len, C)
        
        # Reshape back to (B, C, H, W)
        out = out.permute(0, 2, 1).reshape(B, C, H, W)
        return out


class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        """
        Optimized Vision Attention Block.
        """
        super().__init__()
        self.attn = OptimizedAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        """
        Forward pass of the Optimized AttentionBlock.
        :param x: Input tensor of shape (B, C, H, W)
        :return: Output tensor of the same shape (B, C, H, W)
        """
        B, C, H, W = x.shape
        
        # Attention with residual
        attn_out = self.attn(x)
        x = x + attn_out
        
        # LayerNorm in-place friendly way
        x_flat = x.flatten(2).permute(0, 2, 1)  # (B, H*W, C)
        x_flat = self.norm(x_flat)
        x = x_flat.permute(0, 2, 1).reshape(B, C, H, W)
        
        return x
