import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(x_ptr, mean_ptr, var_ptr, scale_ptr, out_ptr,
                 N, C, H, W, G, Cg, out_H, out_W,
                 clamp_min, clamp_max,
                 eps,
                 BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    n = pid // (out_H * out_W)
    hw = pid % (out_H * out_W)
    oh = hw // out_W
    ow = hw % out_W

    for c in range(0, C, BLOCK):
        offs = c + tl.arange(0, BLOCK)
        mask = offs < C

        g = offs // Cg

        mean = tl.load(mean_ptr + n*G + g, mask=mask)
        var = tl.load(var_ptr + n*G + g, mask=mask)
        scale = tl.load(scale_ptr + offs, mask=mask)

        ih0 = oh*2
        iw0 = ow*2

        idx0 = ((n*C + offs)*H + ih0)*W + iw0
        idx1 = ((n*C + offs)*H + ih0)*W + (iw0+1)
        idx2 = ((n*C + offs)*H + (ih0+1))*W + iw0
        idx3 = ((n*C + offs)*H + (ih0+1))*W + (iw0+1)

        v0 = tl.load(x_ptr + idx0, mask=mask)
        v1 = tl.load(x_ptr + idx1, mask=mask)
        v2 = tl.load(x_ptr + idx2, mask=mask)
        v3 = tl.load(x_ptr + idx3, mask=mask)

        v0 = (v0 - mean) / tl.sqrt(var + eps)
        v1 = (v1 - mean) / tl.sqrt(var + eps)
        v2 = (v2 - mean) / tl.sqrt(var + eps)
        v3 = (v3 - mean) / tl.sqrt(var + eps)

        v0 = v0 * scale
        v1 = v1 * scale
        v2 = v2 * scale
        v3 = v3 * scale

        v = tl.maximum(tl.maximum(v0, v1), tl.maximum(v2, v3))
        v = tl.maximum(v, clamp_min)
        v = tl.minimum(v, clamp_max)

        out_idx = ((n*C + offs)*out_H + oh)*out_W + ow
        tl.store(out_ptr + out_idx, v, mask=mask)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.num_groups = num_groups
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)
        N, C, H, W = x.shape
        G = self.num_groups
        Cg = C // G

        x_ = x.view(N, G, Cg, H, W)
        mean = x_.mean(dim=(2,3,4))
        var = x_.var(dim=(2,3,4), unbiased=False)

        out_H = H // 2
        out_W = W // 2
        out = torch.empty((N, C, out_H, out_W), device=x.device, dtype=x.dtype)

        total = N * out_H * out_W
        fused_kernel[(total,)](
            x, mean, var, self.scale.view(-1), out,
            N, C, H, W, G, Cg, out_H, out_W,
            self.clamp_min, self.clamp_max,
            1e-5,
            BLOCK=32
        )
        return out
