import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Vectorized float32 kernel - processes 2 elements per thread
__global__ void layernorm_fwd_float2_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float eps,
    int B, int HW, int C) {
    
    using LoadType = float2;
    constexpr int VEC_ELEMS = 2;
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x;
    int vec_idx = tid;
    
    const float* x_batch = x + batch_idx * HW * C;
    float* y_batch = y + batch_idx * HW * C;
    
    int N = HW * C;
    int vec_count = N / VEC_ELEMS;
    
    // Phase 1: Compute sum using vectorized loads
    float thread_sum = 0.0f;
    float thread_sq_sum = 0.0f;
    
    for (int i = vec_idx; i < vec_count; i += blockDim.x) {
        float2 val = reinterpret_cast<const LoadType*>(x_batch)[i];
        thread_sum += val.x + val.y;
        thread_sq_sum += val.x * val.x + val.y * val.y;
    }
    
    // Handle odd element
    if (N % 2 && vec_idx == vec_count) {
        float val = x_batch[N-1];
        thread_sum += val;
        thread_sq_sum += val * val;
    }
    
    // Warp reduction
    __shared__ float shared_sum[32];
    __shared__ float shared_sq_sum[32];
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        thread_sum += __shfl_xor_sync(0xffffffff, thread_sum, offset);
        thread_sq_sum += __shfl_xor_sync(0xffffffff, thread_sq_sum, offset);
    }
    
    if (lane_id == 0) {
        shared_sum[warp_id] = thread_sum;
        shared_sq_sum[warp_id] = thread_sq_sum;
    }
    __syncthreads();
    
    // Block reduction
    if (warp_id == 0) {
        thread_sum = (lane_id < (blockDim.x / 32)) ? shared_sum[lane_id] : 0.0f;
        thread_sq_sum = (lane_id < (blockDim.x / 32)) ? shared_sq_sum[lane_id] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            thread_sum += __shfl_xor_sync(0xffffffff, thread_sum, offset);
            thread_sq_sum += __shfl_xor_sync(0xffffffff, thread_sq_sum, offset);
        }
        
        if (lane_id == 0) {
            shared_sum[0] = thread_sum;
            shared_sq_sum[0] = thread_sq_sum;
        }
    }
    __syncthreads();
    
    float mean = shared_sum[0] / N;
    float var = shared_sq_sum[0] / N - mean * mean;
    float inv_std = rsqrtf(var + eps);
    
    // Phase 2: Compute output with vectorized stores
    for (int i = vec_idx; i < vec_count; i += blockDim.x) {
        float2 x_val = reinterpret_cast<const LoadType*>(x_batch)[i];
        
        float2 g_val = {1.0f, 1.0f};
        float2 b_val = {0.0f, 0.0f};
        
        if (gamma != nullptr) {
            g_val = reinterpret_cast<const LoadType*>(gamma)[i];
        }
        if (beta != nullptr) {
            b_val = reinterpret_cast<const LoadType*>(beta)[i];
        }
        
        float2 out_val;
        out_val.x = g_val.x * ((x_val.x - mean) * inv_std) + b_val.x;
        out_val.y = g_val.y * ((x_val.y - mean) * inv_std) + b_val.y;
        
        reinterpret_cast<LoadType*>(y_batch)[i] = out_val;
    }
    
    // Handle odd element
    if (N % 2 && vec_idx == vec_count) {
        int idx = N - 1;
        float val = (x_batch[idx] - mean) * inv_std;
        float g = (gamma != nullptr) ? gamma[idx] : 1.0f;
        float b = (beta != nullptr) ? beta[idx] : 0.0f;
        y_batch[idx] = g * val + b;
    }
}

// Generic scalar kernel
template <typename scalar_t>
__global__ void layernorm_fwd_generic_kernel(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    const scalar_t* __restrict__ gamma,
    const scalar_t* __restrict__ beta,
    float eps,
    int B, int HW, int C) {
    
    int tid = threadIdx.x;
    int batch_idx = blockIdx.x;
    
    const scalar_t* x_batch = x + batch_idx * HW * C;
    scalar_t* y_batch = y + batch_idx * HW * C;
    
    int N = HW * C;
    
    // Phase 1: Compute sum using float accumulation
    float thread_sum = 0.0f;
    float thread_sq_sum = 0.0f;
    
    for (int i = tid; i < N; i += blockDim.x) {
        float val = static_cast<float>(x_batch[i]);
        thread_sum += val;
        thread_sq_sum += val * val;
    }
    
    // Warp reduction
    __shared__ float shared_sum[32];
    __shared__ float shared_sq_sum[32];
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        thread_sum += __shfl_xor_sync(0xffffffff, thread_sum, offset);
        thread_sq_sum += __shfl_xor_sync(0xffffffff, thread_sq_sum, offset);
    }
    
    if (lane_id == 0) {
        shared_sum[warp_id] = thread_sum;
        shared_sq_sum[warp_id] = thread_sq_sum;
    }
    __syncthreads();
    
    // Block reduction
    if (warp_id == 0) {
        thread_sum = (lane_id < (blockDim.x / 32)) ? shared_sum[lane_id] : 0.0f;
        thread_sq_sum = (lane_id < (blockDim.x / 32)) ? shared_sq_sum[lane_id] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            thread_sum += __shfl_xor_sync(0xffffffff, thread_sum, offset);
            thread_sq_sum += __shfl_xor_sync(0xffffffff, thread_sq_sum, offset);
        }
        
        if (lane_id == 0) {
            shared_sum[0] = thread_sum;
            shared_sq_sum[0] = thread_sq_sum;
        }
    }
    __syncthreads();
    
    float mean = shared_sum[0] / N;
    float var = shared_sq_sum[0] / N - mean * mean;
    float inv_std = rsqrtf(var + eps);
    
    // Phase 2: Compute output
    for (int i = tid; i < N; i += blockDim.x) {
        float val = (static_cast<float>(x_batch[i]) - mean) * inv_std;
        float g = (gamma != nullptr) ? static_cast<float>(gamma[i]) : 1.0f;
        float b = (beta != nullptr) ? static_cast<float>(beta[i]) : 0.0f;
        y_batch[i] = static_cast<scalar_t>(g * val + b);
    }
}

// Dispatcher
torch::Tensor layernorm_cuda(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps) {
    TORCH_CHECK(x.is_cuda(), "x must be a CUDA tensor");
    
    int B = x.size(0);
    int C = x.size(1);
    int D1 = x.size(2);
    int D2 = x.size(3);
    int HW = D1 * D2;
    int N = C * HW;
    
    auto y = torch::empty_like(x);
    
    const int BLOCK_SIZE = 256;
    const int grid_size = B;
    
    // Choose kernel based on dtype
    if (x.dtype() == torch::kFloat32) {
        layernorm_fwd_float2_kernel<<<grid_size, BLOCK_SIZE>>>(
            x.data_ptr<float>(),
            y.data_ptr<float>(),
            gamma.defined() ? gamma.data_ptr<float>() : nullptr,
            beta.defined() ? beta.data_ptr<float>() : nullptr,
            eps,
            B, HW, C
        );
    } else if (x.dtype() == torch::kHalf) {
        layernorm_fwd_generic_kernel<torch::Half><<<grid_size, BLOCK_SIZE>>>(
            x.data_ptr<torch::Half>(),
            y.data_ptr<torch::Half>(),
            gamma.defined() ? gamma.data_ptr<torch::Half>() : nullptr,
            beta.defined() ? beta.data_ptr<torch::Half>() : nullptr,
            eps,
            B, HW, C
        );
    } else if (x.dtype() == torch::kBFloat16) {
        layernorm_fwd_generic_kernel<torch::BFloat16><<<grid_size, BLOCK_SIZE>>>(
            x.data_ptr<torch::BFloat16>(),
            y.data_ptr<torch::BFloat16>(),
            gamma.defined() ? gamma.data_ptr<torch::BFloat16>() : nullptr,
            beta.defined() ? beta.data_ptr<torch::BFloat16>() : nullptr,
            eps,
            B, HW, C
        );
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return y;
}

// Python binding
torch::Tensor layernorm_op(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, double eps) {
    return layernorm_cuda(x, gamma, beta, eps);
}
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources='''
torch::Tensor layernorm_op(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, double eps);
''',
    cuda_sources=cuda_source,
    functions=['layernorm_op'],
    extra_include_paths=['/usr/local/cuda/include'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized LayerNorm using custom CUDA kernel with warp-level reductions
    and vectorized memory access for float32.
    """
    def __init__(self, normalized_shape: tuple, eps: float = 1e-5):
        super().__init__()
        self.normalized_shape = normalized_shape
        self.eps = eps
        # Use registered buffer for learnable parameters
        self.register_buffer('gamma', torch.ones(normalized_shape))
        self.register_buffer('beta', torch.zeros(normalized_shape))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_op(x, self.gamma, self.beta, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(64, 256, 256)]
