
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>

template <typename T>
__global__ void layernorm_kernel(
    const T* __restrict__ x,
    const T* __restrict__ gamma,
    const T* __restrict__ beta,
    T* __restrict__ y,
    int N,
    float eps) {
    
    int row_idx = blockIdx.x;
    const T* row_x = x + row_idx * (long long)N;
    T* row_y = y + row_idx * (long long)N;

    // First pass: sum and sum of squares
    float sum = 0.0f;
    float sum_sq = 0.0f;

    #pragma unroll 4
    for (int i = threadIdx.x; i < N; i += blockDim.x) {
        float val = (float)row_x[i];
        sum += val;
        sum_sq += val * val;
    }

    // Block reduction using shuffle
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xFFFFFFFF, sum, offset);
        sum_sq += __shfl_down_sync(0xFFFFFFFF, sum_sq, offset);
    }

    extern __shared__ float shared_mem[];
    float* s_sum = shared_mem;
    float* s_sum_sq = shared_mem + (blockDim.x / 32);

    if (threadIdx.x % 32 == 0) {
        s_sum[threadIdx.x / 32] = sum;
        s_sum_sq[threadIdx.x / 32] = sum_sq;
    }
    __syncthreads();

    if (threadIdx.x < 32) {
        sum = (threadIdx.x < blockDim.x / 32) ? s_sum[threadIdx.x] : 0.0f;
        sum_sq = (threadIdx.x < blockDim.x / 32) ? s_sum_sq[threadIdx.x] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xFFFFFFFF, sum, offset);
            sum_sq += __shfl_down_sync(0xFFFFFFFF, sum_sq, offset);
        }
        if (threadIdx.x == 0) {
            s_sum[0] = sum;
            s_sum_sq[0] = sum_sq;
        }
    }
    __syncthreads();

    // Single pass: normalize
    const float mean = s_sum[0] / N;
    float var = (s_sum_sq[0] / N) - (mean * mean);
    if (var < 0.0f) var = 0.0f;
    const float inv_std = rsqrtf(var + eps);

    // Second pass: normalize
    for (int i = threadIdx.x; i < N; i += blockDim.x) {
        float val = (float)row_x[i];
        float g = (float)gamma[i];
        float b = (float)beta[i];
        row_y[i] = (T)((val - mean) * (inv_std * g) + b);
    }
}

torch::Tensor layernorm_cuda(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps) {
    auto y = torch::empty_like(x);
    int N = gamma.numel();
    long long total_elements = x.numel();
    int num_rows = total_elements / N;

    int threads = 256;
    if (N > 512) threads = 512;
    if (N > 1024) threads = 1024;
    
    size_t shared_mem_size = (threads / 32) * 2 * sizeof(float);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "layernorm_cuda", ([&] {
        layernorm_kernel<scalar_t><<<num_rows, threads, shared_mem_size>>>(
            x.data_ptr<scalar_t>(),
            gamma.data_ptr<scalar_t>(),
            beta.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            N,
            eps);
    }));

    return y;
}
'''

cpp_source = r'''
torch::Tensor layernorm_cuda(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        if isinstance(normalized_shape, int):
            normalized_shape = (normalized_shape,)
        self.normalized_shape = normalized_shape
        self.gamma = nn.Parameter(torch.ones(normalized_shape))
        self.beta = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Match input precision for gamma and beta
        gamma = self.gamma.to(x.dtype)
        beta = self.beta.to(x.dtype)
        return ext.layernorm_cuda(x, gamma, beta, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]

def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
