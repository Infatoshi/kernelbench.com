import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void stats_kernel(const float* __restrict__ x, float* mean, float* var,
                             int N,int C,int H,int W,int G){
    int ng = blockIdx.x; // n*G + g
    int n = ng / G;
    int g = ng % G;
    int Cg = C / G;
    int HW = H*W;
    int start_c = g * Cg;
    int elems = Cg * HW;

    float sum=0.f, sq=0.f;
    for(int i=threadIdx.x;i<elems;i+=blockDim.x){
        int c = start_c + i / HW;
        int hw = i % HW;
        int h = hw / W;
        int w = hw % W;
        int idx = ((n*C + c)*H + h)*W + w;
        float v = x[idx];
        sum += v;
        sq += v*v;
    }

    __shared__ float ssum[256];
    __shared__ float ssq[256];
    ssum[threadIdx.x]=sum;
    ssq[threadIdx.x]=sq;
    __syncthreads();

    for(int s=blockDim.x/2;s>0;s>>=1){
        if(threadIdx.x < s){
            ssum[threadIdx.x]+=ssum[threadIdx.x+s];
            ssq[threadIdx.x]+=ssq[threadIdx.x+s];
        }
        __syncthreads();
    }

    if(threadIdx.x==0){
        float m = ssum[0] / elems;
        float v = ssq[0] / elems - m*m;
        mean[ng]=m;
        var[ng]=v;
    }
}

__global__ void fused_kernel(const float* __restrict__ x, const float* __restrict__ mean, const float* __restrict__ var,
                             const float* __restrict__ gamma, const float* __restrict__ beta,
                             const float* __restrict__ scale,
                             float* __restrict__ y,
                             int N,int C,int H,int W,int G,float eps,
                             float clamp_min,float clamp_max){
    int Ho = H/2;
    int Wo = W/2;
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*C*Ho*Wo;
    if(idx>=total) return;

    int wo = idx % Wo;
    int ho = (idx / Wo) % Ho;
    int c = (idx / (Wo*Ho)) % C;
    int n = idx / (Wo*Ho*C);

    int g = c / (C/G);
    float m = mean[n*G + g];
    float v = var[n*G + g];
    float inv = rsqrtf(v + eps);

    int h0 = ho*2;
    int w0 = wo*2;

    float maxv = -1e30f;
    for(int dh=0; dh<2; ++dh){
        for(int dw=0; dw<2; ++dw){
            int h = h0 + dh;
            int w = w0 + dw;
            int in_idx = ((n*C + c)*H + h)*W + w;
            float v0 = x[in_idx];
            float nrm = (v0 - m) * inv;
            float a = nrm * gamma[c] + beta[c];
            a = a * scale[c];
            maxv = fmaxf(maxv, a);
        }
    }

    maxv = fminf(fmaxf(maxv, clamp_min), clamp_max);
    y[idx] = maxv;
}

std::vector<torch::Tensor> fused_forward(torch::Tensor x,
        torch::Tensor gamma, torch::Tensor beta, torch::Tensor scale,
        int G, float eps, float clamp_min, float clamp_max){

    int N=x.size(0), C=x.size(1), H=x.size(2), W=x.size(3);
    int Ho=H/2, Wo=W/2;

    auto opts = x.options();
    auto mean = torch::empty({N*G}, opts);
    auto var = torch::empty({N*G}, opts);
    auto y = torch::empty({N,C,Ho,Wo}, opts);

    stats_kernel<<<N*G,256>>>(x.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(), N,C,H,W,G);

    int total = N*C*Ho*Wo;
    int threads = 256;
    int blocks = (total + threads -1)/threads;

    fused_kernel<<<blocks,threads>>>(x.data_ptr<float>(), mean.data_ptr<float>(), var.data_ptr<float>(),
        gamma.data_ptr<float>(), beta.data_ptr<float>(), scale.data_ptr<float>(),
        y.data_ptr<float>(), N,C,H,W,G,eps,clamp_min,clamp_max);

    return {y};
}
'''

ext = load_inline(name='fused_gn_pool',
                  cpp_sources='std::vector<torch::Tensor> fused_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, torch::Tensor scale, int G, float eps, float clamp_min, float clamp_max);',
                  cuda_sources=cuda_source,
                  functions=['fused_forward'],
                  verbose=False)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.num_groups = num_groups
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)
        y = ext.fused_forward(x.contiguous(), self.group_norm.weight.contiguous(), self.group_norm.bias.contiguous(), self.scale.view(-1).contiguous(), self.num_groups, self.group_norm.eps, self.clamp_min, self.clamp_max)[0]
        return y


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width, device='cuda')]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
