import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def cross_entropy_kernel(
    input_ptr,
    target_ptr,
    output_ptr,
    n_cols,
    batch_stride,
    BLOCK_SIZE: tl.constexpr,
):
    # Get the row (sample) index
    row_idx = tl.program_id(0)
    
    # Compute offsets for this row
    row_start_ptr = input_ptr + row_idx * batch_stride
    
    # Load target
    target = tl.load(target_ptr + row_idx)
    
    # First pass: find max for numerical stability
    max_val = float('-inf')
    for block_start in range(0, n_cols, BLOCK_SIZE):
        cols = block_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < n_cols
        vals = tl.load(row_start_ptr + cols, mask=mask, other=float('-inf'))
        block_max = tl.max(vals, axis=0)
        max_val = tl.maximum(max_val, block_max)
    
    # Second pass: compute sum of exp(x - max)
    sum_exp = 0.0
    for block_start in range(0, n_cols, BLOCK_SIZE):
        cols = block_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < n_cols
        vals = tl.load(row_start_ptr + cols, mask=mask, other=0.0)
        sum_exp += tl.sum(tl.exp(vals - max_val), axis=0)
    
    log_sum_exp = tl.log(sum_exp) + max_val
    
    # Load target value
    target_idx = target.to(tl.int32)
    target_val = tl.load(row_start_ptr + target_idx)
    
    # Compute loss for this sample
    loss = log_sum_exp - target_val
    
    # Atomic add to global loss (for reduction across batch)
    tl.atomic_add(output_ptr, loss)

def cross_entropy_triton(input, target):
    batch_size, num_classes = input.shape
    
    # Allocate output (initialized to 0)
    output = torch.zeros(1, device=input.device, dtype=torch.float32)
    
    # Determine block size
    BLOCK_SIZE = triton.next_power_of_2(num_classes)
    BLOCK_SIZE = min(BLOCK_SIZE, 2048)  # Cap for performance
    
    # Convert to contiguous if needed
    input = input.contiguous()
    target = target.contiguous()
    
    grid = (batch_size,)
    cross_entropy_kernel[grid](
        input, target, output,
        num_classes, input.stride(0),
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return output / batch_size

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return cross_entropy_triton(predictions, targets)
