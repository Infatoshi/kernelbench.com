import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE 32

__global__ void gemm_int8_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K,
    float scale
) {
    __shared__ int8_t As[TILE][TILE];
    __shared__ int8_t Bs[TILE][TILE];

    int row = blockIdx.y * TILE + threadIdx.y;
    int col = blockIdx.x * TILE + threadIdx.x;

    int32_t acc = 0;

    for (int t = 0; t < K; t += TILE) {
        if (row < M && (t + threadIdx.x) < K)
            As[threadIdx.y][threadIdx.x] = A[row * K + t + threadIdx.x];
        else
            As[threadIdx.y][threadIdx.x] = 0;

        if (col < N && (t + threadIdx.y) < K)
            Bs[threadIdx.y][threadIdx.x] = B[(t + threadIdx.y) * N + col];
        else
            Bs[threadIdx.y][threadIdx.x] = 0;

        __syncthreads();

        #pragma unroll
        for (int i = 0; i < TILE; ++i) {
            acc += int32_t(As[threadIdx.y][i]) * int32_t(Bs[i][threadIdx.x]);
        }

        __syncthreads();
    }

    if (row < M && col < N) {
        C[row * N + col] = acc * scale;
    }
}

torch::Tensor gemm_int8(torch::Tensor A, torch::Tensor B, float scale) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);

    auto C = torch::empty({M, N}, torch::dtype(torch::kFloat32).device(A.device()));

    dim3 block(TILE, TILE);
    dim3 grid((N + TILE - 1) / TILE, (M + TILE - 1) / TILE);

    gemm_int8_kernel<<<grid, block>>>(
        A.data_ptr<int8_t>(),
        B.data_ptr<int8_t>(),
        C.data_ptr<float>(),
        M, N, K,
        scale
    );

    return C;
}
'''

ext = load_inline(
    name='fp4_gemm_ext',
    cpp_sources='torch::Tensor gemm_int8(torch::Tensor A, torch::Tensor B, float scale);',
    cuda_sources=cuda_source,
    functions=['gemm_int8'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(self, a_q, b_q, scale_a, scale_b):
        scale = float(scale_a.item() * scale_b.item())
        out = ext.gemm_int8(a_q.contiguous(), b_q.contiguous(), scale)
        return out.to(torch.float16)


def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-8, 8, (m, k), dtype=torch.int8, device='cuda')
    b_q = torch.randint(-8, 8, (k, n), dtype=torch.int8, device='cuda')
    scale_a = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    scale_b = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    return [a_q, b_q, scale_a, scale_b]
