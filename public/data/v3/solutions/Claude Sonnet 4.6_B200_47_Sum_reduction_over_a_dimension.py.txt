import torch
import torch.nn as nn
import triton
import triton.language as tl


# -----------------------------------------------------------------------
# Triton kernel: sum reduction over a middle dimension
#
# Tensor layout: [outer_size, reduce_size, inner_size]
# We handle one (outer, inner) pair per program instance.
# inner_size is tiled with BLOCK_INNER so we can vectorise.
# -----------------------------------------------------------------------

@triton.jit
def sum_reduce_kernel(
    input_ptr,
    output_ptr,
    outer_size,
    reduce_size,
    inner_size,
    BLOCK_REDUCE: tl.constexpr,
    BLOCK_INNER:  tl.constexpr,
):
    # Programme ID encodes (outer_idx * num_inner_blocks + inner_block_idx)
    pid = tl.program_id(0)
    num_inner_blocks = tl.cdiv(inner_size, BLOCK_INNER)
    outer_idx  = pid // num_inner_blocks
    inner_bid  = pid %  num_inner_blocks

    inner_off = inner_bid * BLOCK_INNER + tl.arange(0, BLOCK_INNER)
    inner_mask = inner_off < inner_size

    acc = tl.zeros([BLOCK_INNER], dtype=tl.float32)

    # Base pointer for this outer slice
    base = outer_idx * reduce_size * inner_size

    # Loop over reduce_size in tiles of BLOCK_REDUCE
    for r_start in range(0, reduce_size, BLOCK_REDUCE):
        r_off = r_start + tl.arange(0, BLOCK_REDUCE)
        r_mask = r_off < reduce_size
        # 2-D offset: r_off[:,None]*inner_size + inner_off[None,:]
        off2d  = r_off[:, None] * inner_size + inner_off[None, :]
        mask2d = r_mask[:, None] & inner_mask[None, :]
        vals = tl.load(input_ptr + base + off2d, mask=mask2d, other=0.0)
        acc  = acc + tl.sum(vals, axis=0)

    # Write output
    out_base = outer_idx * inner_size
    tl.store(output_ptr + out_base + inner_off, acc.to(input_ptr.dtype.element_ty), mask=inner_mask)


def _ceil_pow2(n: int) -> int:
    p = 1
    while p < n:
        p <<= 1
    return p


def sum_reduce_triton(x: torch.Tensor, dim: int) -> torch.Tensor:
    ndim = x.dim()
    if dim < 0:
        dim += ndim
    if not x.is_contiguous():
        x = x.contiguous()

    outer_size  = 1
    reduce_size = x.size(dim)
    inner_size  = 1
    for i in range(dim):
        outer_size *= x.size(i)
    for i in range(dim + 1, ndim):
        inner_size *= x.size(i)

    out_sizes = list(x.shape)
    out_sizes[dim] = 1
    output = torch.empty(out_sizes, dtype=x.dtype, device=x.device)

    # Choose tile sizes
    BLOCK_REDUCE = _ceil_pow2(reduce_size)
    if BLOCK_REDUCE > 256:
        BLOCK_REDUCE = 256
    BLOCK_INNER = min(256, _ceil_pow2(inner_size))
    if BLOCK_INNER > inner_size:
        BLOCK_INNER = _ceil_pow2(inner_size)
    # Keep BLOCK_REDUCE * BLOCK_INNER manageable
    while BLOCK_REDUCE * BLOCK_INNER > 1024 and BLOCK_INNER > 1:
        BLOCK_INNER >>= 1

    num_inner_blocks = triton.cdiv(inner_size, BLOCK_INNER)
    grid = (outer_size * num_inner_blocks,)

    sum_reduce_kernel[grid](
        x, output,
        outer_size, reduce_size, inner_size,
        BLOCK_REDUCE=BLOCK_REDUCE,
        BLOCK_INNER=BLOCK_INNER,
        num_warps=max(1, BLOCK_REDUCE * BLOCK_INNER // 32),
    )
    return output


class Model(nn.Module):
    """Optimised sum reduction with keepdim=True (Triton backend)."""
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return sum_reduce_triton(x, self.dim)


# Problem constants
batch_size = 64
dim1       = 256
dim2       = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
