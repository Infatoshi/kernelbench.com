import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void int4_gemm_kernel(
    const half* __restrict__ X,
    const uint8_t* __restrict__ W_packed,
    const half* __restrict__ Scales,
    half* __restrict__ Y,
    int M, int N, int K, int group_size) {

    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;

    if (row < M && col < N) {
        float acc = 0.0f;
        int num_groups = K / group_size;
        int K_half = K / 2;

        for (int g = 0; g < num_groups; ++g) {
            half scale_h = Scales[col * num_groups + g];
            int k_start = g * group_size;
            
            for (int k_off = 0; k_off < group_size; k_off += 2) {
                int k = k_start + k_off;
                uint8_t packed = W_packed[col * K_half + (k >> 1)];
                
                half w0 = __float2half((float)(packed & 0x0F) - 8.0f);
                half w1 = __float2half((float)(packed >> 4) - 8.0f);
                
                half dq0 = __hmul(scale_h, w0);
                half dq1 = __hmul(scale_h, w1);

                acc += __half2float(X[row * K + k]) * __half2float(dq0);
                acc += __half2float(X[row * K + k + 1]) * __half2float(dq1);
            }
        }
        Y[row * N + col] = __float2half(acc);
    }
}

torch::Tensor int4_gemm_cuda(torch::Tensor x, torch::Tensor weight_packed, torch::Tensor scales, int group_size) {
    int M_batch = x.size(0);
    int M_seq = x.size(1);
    int M = M_batch * M_seq;
    int K = x.size(2);
    int N = weight_packed.size(0);

    auto x_reshape = x.reshape({M, K});
    auto y = torch::empty({M, N}, x.options());

    dim3 block(32, 8);
    dim3 grid((N + block.x - 1) / block.x, (M + block.y - 1) / block.y);

    int4_gemm_kernel<<<grid, block>>>(
        (const half*)x_reshape.data_ptr<at::Half>(),
        (const uint8_t*)weight_packed.data_ptr<uint8_t>(),
        (const half*)scales.data_ptr<at::Half>(),
        (half*)y.data_ptr<at::Half>(),
        M, N, K, group_size
    );

    return y.reshape({M_batch, M_seq, N});
}
'''

cpp_source = r'''
torch::Tensor int4_gemm_cuda(torch::Tensor x, torch::Tensor weight_packed, torch::Tensor scales, int group_size);
'''

ext = load_inline(
    name='int4_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['int4_gemm_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        self.register_buffer("weight_packed", torch.empty((N, K // 2), dtype=torch.uint8))
        self.register_buffer("scales", torch.empty((N, self.num_groups), dtype=torch.float16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.int4_gemm_cuda(x, self.weight_packed, self.scales, self.group_size)

def get_inputs():
    batch_size, seq_len, K = 4, 2048, 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16, device='cuda')]

def get_init_inputs():
    return [4096, 11008, 128]
