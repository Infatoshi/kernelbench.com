import torch
import torch.nn as nn
import triton
import triton.language as tl

# Optimized 2D convolution using Triton
# For this problem: batch=16, in_ch=3, out_ch=64, kernel=3x3, stride=1, padding=0

@triton.jit
def conv2d_kernel(
    x_ptr, w_ptr, y_ptr,
    batch_size, in_channels, out_channels,
    height, width, kernel_size,
    out_height, out_width,
    stride, padding, dilation,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """
    Optimized 2D convolution kernel using Triton.
    Each program instance computes a tile of the output.
    
    For convolution: y[b, oc, oh, ow] = sum over (ic, kh, kw) of x[b, ic, oh*s + kh - pad, ow*s + kw - pad] * w[oc, ic, kh, kw]
    
    We reshape this as a matrix multiplication:
    - Input patch: (1, in_channels * kernel_size * kernel_size)
    - Weight: (out_channels, in_channels * kernel_size * kernel_size)
    - Output: (1, out_channels)
    
    BLOCK_M: tile size for output positions (flattened batch * out_h * out_w)
    BLOCK_N: tile size for output channels
    BLOCK_K: tile size for input channels * kernel_h * kernel_w (unrolled)
    """
    # Program IDs
    pid_m = tl.program_id(0)  # Tile in M dimension
    pid_n = tl.program_id(1)  # Tile in N dimension
    
    # Offsets for the output tile
    m_start = pid_m * BLOCK_M
    n_start = pid_n * BLOCK_N
    
    # Accumulator for the output tile
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # K dimension: in_channels * kernel_size * kernel_size
    K_total = in_channels * kernel_size * kernel_size
    
    # Iterate over K in tiles
    for k_start in range(0, K_total, BLOCK_K):
        k_end = min(k_start + BLOCK_K, K_total)
        k_range = k_end - k_start
        
        # Load weight tile: shape (BLOCK_K, BLOCK_N)
        # Weight is (out_channels, in_channels, kernel_size, kernel_size)
        # Reshaped as (out_channels, K_total)
        w_offsets = n_start + tl.arange(0, BLOCK_N)  # Output channel indices
        k_offsets = k_start + tl.arange(0, BLOCK_K)  # K indices
        
        w_mask = (w_offsets[None, :] < out_channels) & (k_offsets[:, None] < K_total)
        w_ptr_offsets = w_offsets[None, :] * K_total + k_offsets[:, None]
        w_tile = tl.load(w_ptr + w_ptr_offsets, mask=w_mask, other=0.0)
        
        # Load input tile: shape (BLOCK_M, BLOCK_K)
        # For each output position (m_idx), we need to gather input values
        # m_idx = b * out_height * out_width + oh * out_width + ow
        
        m_offsets = m_start + tl.arange(0, BLOCK_M)
        m_mask = m_offsets < (batch_size * out_height * out_width)
        
        # Decode m_idx into batch, oh, ow
        m_idx = m_offsets
        b_idx = m_idx // (out_height * out_width)
        remaining = m_idx % (out_height * out_width)
        oh_idx = remaining // out_width
        ow_idx = remaining % out_width
        
        # For each k offset, compute the input position
        # k_idx = ic * kernel_size * kernel_size + kh * kernel_size + kw
        k_idx = k_start + tl.arange(0, BLOCK_K)
        ic_idx = k_idx // (kernel_size * kernel_size)
        k_remaining = k_idx % (kernel_size * kernel_size)
        kh_idx = k_remaining // kernel_size
        kw_idx = k_remaining % kernel_size
        
        # Input position: ih = oh * stride + kh * dilation - padding
        ih_idx = oh_idx[:, None] * stride + kh_idx[None, :] * dilation - padding
        iw_idx = ow_idx[:, None] * stride + kw_idx[None, :] * dilation - padding
        
        # Compute flat input indices
        # x is (batch, in_channels, height, width)
        # idx = b * in_channels * height * width + ic * height * width + ih * width + iw
        x_offsets = (
            b_idx[:, None] * in_channels * height * width +
            ic_idx[None, :] * height * width +
            ih_idx * width +
            iw_idx
        )
        
        # Boundary check for input positions
        input_mask = (
            m_mask[:, None] & 
            (k_idx[None, :] < K_total) &
            (ih_idx >= 0) & (ih_idx < height) &
            (iw_idx >= 0) & (iw_idx < width)
        )
        
        x_tile = tl.load(x_ptr + x_offsets, mask=input_mask, other=0.0)
        
        # Matrix multiply: (BLOCK_M, BLOCK_K) @ (BLOCK_K, BLOCK_N) -> (BLOCK_M, BLOCK_N)
        acc += tl.dot(x_tile, w_tile)
    
    # Store output tile
    out_offsets_m = m_start + tl.arange(0, BLOCK_M)
    out_offsets_n = n_start + tl.arange(0, BLOCK_N)
    out_mask = (out_offsets_m[:, None] < batch_size * out_height * out_width) & \
               (out_offsets_n[None, :] < out_channels)
    
    # Output is (batch, out_channels, out_height, out_width)
    # Flatten: (batch * out_height * out_width, out_channels)
    # idx = b * out_channels * out_height * out_width + oc * out_height * out_width + oh * out_width + ow
    # But we want to store as (batch * out_h * out_w, out_channels) for simplicity
    # Actually PyTorch stores as (batch, out_channels, out_height, out_width)
    
    # Convert m_idx back to output coordinates for proper layout
    m_idx_store = out_offsets_m
    b_store = m_idx_store // (out_height * out_width)
    remaining_store = m_idx_store % (out_height * out_width)
    oh_store = remaining_store // out_width
    ow_store = remaining_store % out_width
    
    # Output flat index for (batch, out_channels, out_height, out_width)
    y_offsets = (
        b_store[:, None] * out_channels * out_height * out_width +
        out_offsets_n[None, :] * out_height * out_width +
        oh_store[:, None] * out_width +
        ow_store[:, None]
    )
    
    tl.store(y_ptr + y_offsets, acc, mask=out_mask)


class Conv2dFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, stride, padding, dilation):
        batch_size, in_channels, height, width = x.shape
        out_channels, in_channels_w, kernel_size, _ = weight.shape
        
        assert in_channels == in_channels_w
        
        # Output dimensions
        out_height = (height + 2 * padding - dilation * (kernel_size - 1) - 1) // stride + 1
        out_width = (width + 2 * padding - dilation * (kernel_size - 1) - 1) // stride + 1
        
        # Output tensor
        y = torch.empty(batch_size, out_channels, out_height, out_width, 
                       device=x.device, dtype=x.dtype)
        
        # Grid and block sizes
        M = batch_size * out_height * out_width
        N = out_channels
        K = in_channels * kernel_size * kernel_size
        
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_K = 32
        
        grid = (triton.cdiv(M, BLOCK_M), triton.cdiv(N, BLOCK_N))
        
        conv2d_kernel[grid](
            x, weight, y,
            batch_size, in_channels, out_channels,
            height, width, kernel_size,
            out_height, out_width,
            stride, padding, dilation,
            BLOCK_M, BLOCK_N, BLOCK_K,
        )
        
        ctx.save_for_backward(x, weight)
        ctx.stride = stride
        ctx.padding = padding
        ctx.dilation = dilation
        
        return y


def conv2d_triton(x, weight, stride=1, padding=0, dilation=1):
    return Conv2dFunction.apply(x, weight, stride, padding, dilation)


class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, 
                 stride: int = 1, padding: int = 0, dilation: int = 1, 
                 groups: int = 1, bias: bool = False):
        super().__init__()
        assert groups == 1, "Only groups=1 supported"
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        
        # Create weight parameter (out_channels, in_channels, kernel_size, kernel_size)
        self.weight = nn.Parameter(torch.empty(out_channels, in_channels, kernel_size, kernel_size))
        nn.init.kaiming_uniform_(self.weight)
        
        if bias:
            self.bias = nn.Parameter(torch.empty(out_channels))
        else:
            self.register_parameter('bias', None)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return conv2d_triton(x, self.weight, self.stride, self.padding, self.dilation)


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width, device='cuda', dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [3, 64, 3]  # in_channels, out_channels, kernel_size
