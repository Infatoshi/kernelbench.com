import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Warp-level reduction for sum
template<typename T>
__device__ __forceinline__ T warp_reduce_sum(T val) {
    for (int offset = warpSize / 2; offset > 0; offset >>= 1) {
        val += __shfl_xor_sync(0xffffffff, val, offset);
    }
    return val;
}

// FP32 RMS Norm kernel
__global__ void rms_norm_fp32_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    const int batch, const int features, const int dim1, const int dim2,
    const float eps
) {
    extern __shared__ float rms_fp32_cache[];
    
    const int b = blockIdx.z;
    const int d1 = blockIdx.y;
    const int d2_base = blockIdx.x * blockDim.x;
    const int tid = threadIdx.x;
    
    const int d2 = d2_base + tid;
    
    if (d2 < dim2) {
        float sum_sq = 0.0f;
        
        for (int f = 0; f < features; f++) {
            const float val = x[(b * features + f) * dim1 * dim2 + d1 * dim2 + d2];
            sum_sq += val * val;
        }
        
        sum_sq = warp_reduce_sum(sum_sq);
        
        if (tid == 0) {
            sum_sq = sum_sq / features + eps;
            rms_fp32_cache[tid] = rsqrtf(sum_sq);
        }
        
        __syncthreads();
        
        const float rms = rms_fp32_cache[0];
        
        for (int f = 0; f < features; f++) {
            const int idx = (b * features + f) * dim1 * dim2 + d1 * dim2 + d2;
            y[idx] = x[idx] * rms;
        }
    }
}

// FP16 RMS Norm kernel - vectorized version
__global__ void rms_norm_fp16_kernel(
    const half2* __restrict__ x,
    half2* __restrict__ y,
    const int batch, const int features_half2, const int dim1, const int dim2,
    const float eps
) {
    extern __shared__ float2 rms_fp16_cache[];
    
    const int b = blockIdx.z;
    const int d1 = blockIdx.y;
    const int d2_base = blockIdx.x * blockDim.x;
    const int tid = threadIdx.x;
    
    const int d2 = d2_base + tid;
    
    if (d2 < dim2) {
        float2 sum_sq = make_float2(0.0f, 0.0f);
        
        for (int f = 0; f < features_half2; f++) {
            const half2 val = x[(b * features_half2 + f) * dim1 * dim2 + d1 * dim2 + d2];
            const float2 fval = __half22float2(val);
            sum_sq.x += fval.x * fval.x;
            sum_sq.y += fval.y * fval.y;
        }
        
        sum_sq.x = warp_reduce_sum(sum_sq.x);
        sum_sq.y = warp_reduce_sum(sum_sq.y);
        
        if (tid == 0) {
            sum_sq.x = sum_sq.x / (features_half2 * 2) + eps;
            sum_sq.y = sum_sq.y / (features_half2 * 2) + eps;
            rms_fp16_cache[tid] = make_float2(rsqrtf(sum_sq.x), rsqrtf(sum_sq.y));
        }
        
        __syncthreads();
        
        const float2 rms = rms_fp16_cache[0];
        
        for (int f = 0; f < features_half2; f++) {
            const int idx = (b * features_half2 + f) * dim1 * dim2 + d1 * dim2 + d2;
            const half2 val = x[idx];
            const float2 fval = __half22float2(val);
            y[idx] = __float22half2_rn(make_float2(fval.x * rms.x, fval.y * rms.y));
        }
    }
}

// BF16 RMS Norm kernel - vectorized version  
__global__ void rms_norm_bf16_kernel(
    const nv_bfloat162* __restrict__ x,
    nv_bfloat162* __restrict__ y,
    const int batch, const int features_bf162, const int dim1, const int dim2,
    const float eps
) {
    extern __shared__ float2 rms_bf16_cache[];
    
    const int b = blockIdx.z;
    const int d1 = blockIdx.y;
    const int d2_base = blockIdx.x * blockDim.x;
    const int tid = threadIdx.x;
    
    const int d2 = d2_base + tid;
    
    if (d2 < dim2) {
        float2 sum_sq = make_float2(0.0f, 0.0f);
        
        for (int f = 0; f < features_bf162; f++) {
            const nv_bfloat162 val = x[(b * features_bf162 + f) * dim1 * dim2 + d1 * dim2 + d2];
            const float2 fval = __bfloat1622float2(val);
            sum_sq.x += fval.x * fval.x;
            sum_sq.y += fval.y * fval.y;
        }
        
        sum_sq.x = warp_reduce_sum(sum_sq.x);
        sum_sq.y = warp_reduce_sum(sum_sq.y);
        
        if (tid == 0) {
            sum_sq.x = sum_sq.x / (features_bf162 * 2) + eps;
            sum_sq.y = sum_sq.y / (features_bf162 * 2) + eps;
            rms_bf16_cache[tid] = make_float2(rsqrtf(sum_sq.x), rsqrtf(sum_sq.y));
        }
        
        __syncthreads();
        
        const float2 rms = rms_bf16_cache[0];
        
        for (int f = 0; f < features_bf162; f++) {
            const int idx = (b * features_bf162 + f) * dim1 * dim2 + d1 * dim2 + d2;
            const nv_bfloat162 val = x[idx];
            const float2 fval = __bfloat1622float2(val);
            y[idx] = __float22bfloat162_rn(make_float2(fval.x * rms.x, fval.y * rms.y));
        }
    }
}

torch::Tensor rms_norm_fp32_op(torch::Tensor x, float eps) {
    auto y = torch::empty_like(x);
    
    const int batch = x.size(0);
    const int features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    const int total_blocks_x = (dim2 + 255) / 256;
    dim3 grid(total_blocks_x, dim1, batch);
    dim3 block(256);
    
    const int shared_mem = 256 * sizeof(float);
    
    rms_norm_fp32_kernel<<<grid, block, shared_mem>>>(
        x.data_ptr<float>(), y.data_ptr<float>(),
        batch, features, dim1, dim2, eps
    );
    
    return y;
}

torch::Tensor rms_norm_fp16_op(torch::Tensor x, float eps) {
    auto y = torch::empty_like(x);
    
    const int batch = x.size(0);
    const int features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    const int total_blocks_x = (dim2 + 255) / 256;
    dim3 grid(total_blocks_x, dim1, batch);
    dim3 block(256);
    
    const int features_half2 = features / 2;
    const int shared_mem = 256 * sizeof(float2);
    
    rms_norm_fp16_kernel<<<grid, block, shared_mem>>>(
        (const half2*)x.data_ptr<at::Half>(), (half2*)y.data_ptr<at::Half>(),
        batch, features_half2, dim1, dim2, eps
    );
    
    return y;
}

torch::Tensor rms_norm_bf16_op(torch::Tensor x, float eps) {
    auto y = torch::empty_like(x);
    
    const int batch = x.size(0);
    const int features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    const int total_blocks_x = (dim2 + 255) / 256;
    dim3 grid(total_blocks_x, dim1, batch);
    dim3 block(256);
    
    const int features_bf162 = features / 2;
    const int shared_mem = 256 * sizeof(float2);
    
    rms_norm_bf16_kernel<<<grid, block, shared_mem>>>(
        (const nv_bfloat162*)x.data_ptr<at::BFloat16>(), (nv_bfloat162*)y.data_ptr<at::BFloat16>(),
        batch, features_bf162, dim1, dim2, eps
    );
    
    return y;
}
'''

ext = load_inline(
    name='rms_norm_ext',
    cpp_sources='''
torch::Tensor rms_norm_fp32_op(torch::Tensor x, float eps);
torch::Tensor rms_norm_fp16_op(torch::Tensor x, float eps);
torch::Tensor rms_norm_bf16_op(torch::Tensor x, float eps);
''',
    cuda_sources=cuda_source,
    functions=['rms_norm_fp32_op', 'rms_norm_fp16_op', 'rms_norm_bf16_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if x.dtype == torch.float32:
            return ext.rms_norm_fp32_op(x, self.eps)
        elif x.dtype == torch.float16:
            return ext.rms_norm_fp16_op(x, self.eps)
        elif x.dtype == torch.bfloat16:
            return ext.rms_norm_bf16_op(x, self.eps)
        else:
            # Fallback to PyTorch for unsupported types
            rms = torch.sqrt(torch.mean(x ** 2, dim=1, keepdim=True) + self.eps)
            return x / rms

def get_inputs():
    return [torch.randn(16, 64, 256, 256, device='cuda')]

def get_init_inputs():
    return [64]
