import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import time

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <ATen/cuda/CUDAContext.h>

template <typename scalar_t>
__global__ void particle_step_kernel(
    const scalar_t* __restrict__ pos,
    const scalar_t* __restrict__ vel,
    scalar_t* __restrict__ out,
    const float dt,
    const float gravity_x,
    const float gravity_y,
    const float damping,
    const int n) 
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float px = static_cast<float>(pos[idx * 2]);
        float py = static_cast<float>(pos[idx * 2 + 1]);
        float vx = static_cast<float>(vel[idx * 2]);
        float vy = static_cast<float>(vel[idx * 2 + 1]);
        
        float new_vx = vx + gravity_x * dt;
        float new_vy = vy + gravity_y * dt;
        
        float new_px = px + new_vx * dt;
        float new_py = py + new_vy * dt;
        
        if (fabsf(new_px) > 1.0f) {
            new_vx = -new_vx * damping;
        }
        if (fabsf(new_py) > 1.0f) {
            new_vy = -new_vy * damping;
        }
        
        new_px = fmaxf(-1.0f, fminf(1.0f, new_px));
        new_py = fmaxf(-1.0f, fminf(1.0f, new_py));
        
        out[idx * 4]     = static_cast<scalar_t>(new_px);
        out[idx * 4 + 1] = static_cast<scalar_t>(new_py);
        out[idx * 4 + 2] = static_cast<scalar_t>(new_vx);
        out[idx * 4 + 3] = static_cast<scalar_t>(new_vy);
    }
}

torch::Tensor particle_step_cuda(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping)
{
    int n = positions.size(0);
    auto out = torch::empty({n, 4}, positions.options());
    
    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, positions.scalar_type(), "particle_step_cuda", ([&] {
        particle_step_kernel<scalar_t><<<blocks, threads>>>(
            positions.data_ptr<scalar_t>(),
            velocities.data_ptr<scalar_t>(),
            out.data_ptr<scalar_t>(),
            dt, gravity_x, gravity_y, damping, n
        );
    }));
    
    return out;
}
'''

cpp_source = r'''
torch::Tensor particle_step_cuda(
    torch::Tensor positions,
    torch::Tensor velocities,
    float dt,
    float gravity_x,
    float gravity_y,
    float damping);
'''

ext = load_inline(
    name='particle_step_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_step_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math']
)

class Model(nn.Module):
    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        dt_value = dt.item() if isinstance(dt, torch.Tensor) else float(dt)
        return ext.particle_step_cuda(
            positions,
            velocities,
            dt_value,
            gravity[0].item(),
            gravity[1].item(),
            self.damping
        )

def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]

def get_init_inputs():
    return [0.9]
