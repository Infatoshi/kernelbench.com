import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_gemm_gelu_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_per_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // pid_per_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % pid_per_group) // group_size_m

    offs_am = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_bn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    a_ptr = X_ptr + (offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    b_ptr = W_ptr + (offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_remaining = K - k * BLOCK_K
        # Vectorized loads
        a = tl.load(a_ptr, mask=offs_k[None, :] < k_remaining, other=0.0)
        b = tl.load(b_ptr, mask=offs_k[:, None] < k_remaining, other=0.0)
        accumulator += tl.dot(a, b)
        a_ptr += BLOCK_K * stride_xk
        b_ptr += BLOCK_K * stride_wk

    if B_ptr is not None:
        bias = tl.load(B_ptr + offs_bn)[None, :]
        accumulator += bias

    # GELU
    x_val = accumulator
    f_val = x_val * 0.70710678118
    erf_val = tl.extra.cuda.libdevice.erf(f_val)
    out_val = x_val * 0.5 * (1.0 + erf_val)

    # Store
    out_ptr = Out_ptr + (offs_am[:, None] * stride_om + offs_bn[None, :] * stride_on)
    tl.store(out_ptr, out_val.to(Out_ptr.dtype.element_ty))

@triton.jit
def softmax_kernel(
    Out_ptr,
    M, N,
    stride_om, stride_on,
    N_TILES: tl.constexpr,
    BLOCK_SIZE_N: tl.constexpr,
):
    row_idx = tl.program_id(0)
    if row_idx >= M:
        return
    
    # Large N Softmax using online algorithm (w/o full row in SRAM if possible, but 4096 is fine)
    # Actually, for N=4096, 4096 * 4 bytes = 16KB. 
    # A single block in 3090 can easily hold 16KB.
    
    offs_n = tl.arange(0, BLOCK_SIZE_N)
    row_ptr = Out_ptr + row_idx * stride_om + offs_n * stride_on
    
    mask = offs_n < N
    row = tl.load(row_ptr, mask=mask, other=-float('inf'))
    
    row_max = tl.max(row, axis=0)
    numerator = tl.exp(row.to(tl.float32) - row_max)
    denominator = tl.sum(numerator, axis=0)
    softmax_out = numerator / denominator
    
    tl.store(row_ptr, softmax_out.to(Out_ptr.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))

    def forward(self, x):
        if x.dim() == 1:
            x = x.unsqueeze(0)
        M, K = x.shape
        N = self.out_features
        out = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        # Grid for GEMM - Tiled for Ampere
        # 3090: 82 SMs. For M=128, N=4096, we have 128*4096/(128*128) = 32 tiles. 
        # Using smaller tiles to fill more SMs.
        BLOCK_M = 64
        BLOCK_N = 64
        BLOCK_K = 32
        GROUP_SIZE_M = 8
        grid_gemm = (triton.cdiv(M, BLOCK_M) * triton.cdiv(N, BLOCK_N),)
        fused_gemm_gelu_kernel[grid_gemm](
            x, self.weight, self.bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
            num_warps=8,
        )
        
        # Softmax pass
        grid_softmax = (M,)
        BLOCK_SIZE_N = triton.next_power_of_2(N)
        softmax_kernel[grid_softmax](
            out, M, N,
            out.stride(0), out.stride(1),
            N_TILES=1,
            BLOCK_SIZE_N=BLOCK_SIZE_N,
            num_warps=16, # More warps for better reduction
        )
        return out

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096]
