import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Use a single, high-performance CUDA kernel for 2D convolution.
# We'll stick with a straightforward global memory implementation first,
# but ensure tensors are processed correctly based on groups and stride.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv2d_nchw_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_h, int in_w,
    int out_channels, int out_h, int out_w,
    int kernel_size, int stride, int padding, int dilation, int groups,
    int in_channels_per_group, int out_channels_per_group) {

    int index = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * out_channels * out_h * out_w;

    if (index >= total_elements) return;

    // Decode index: [n, oc, oh, ow]
    int ow = index % out_w;
    int oh = (index / out_w) % out_h;
    int oc = (index / (out_w * out_h)) % out_channels;
    int n = index / (out_w * out_h * out_channels);

    int g = oc / out_channels_per_group;
    int oc_in_g = oc % out_channels_per_group;
    int ic_start = g * in_channels_per_group;

    float val = 0.0f;

    for (int ic_off = 0; ic_off < in_channels_per_group; ++ic_off) {
        int ic = ic_start + ic_off;
        for (int kh = 0; kh < kernel_size; ++kh) {
            for (int kw = 0; kw < kernel_size; ++kw) {
                int ih = oh * stride - padding + kh * dilation;
                int iw = ow * stride - padding + kw * dilation;

                if (ih >= 0 && ih < in_h && iw >= 0 && iw < in_w) {
                    float x_val = input[((n * in_channels + ic) * in_h + ih) * in_w + iw];
                    float w_val = weight[((oc * in_channels_per_group + ic_off) * kernel_size + kh) * kernel_size + kw];
                    val += x_val * w_val;
                }
            }
        }
    }

    if (bias != nullptr) {
        val += bias[oc];
    }

    output[index] = val;
}

torch::Tensor conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    std::optional<torch::Tensor> bias,
    int stride,
    int padding,
    int dilation,
    int groups) {
    
    auto batch_size = input.size(0);
    auto in_channels = input.size(1);
    auto in_h = input.size(2);
    auto in_w = input.size(3);

    auto out_channels = weight.size(0);
    auto kernel_size = weight.size(2);
    
    auto out_h = (in_h + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    auto out_w = (in_w + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;

    auto output = torch::empty({batch_size, out_channels, out_h, out_w}, input.options());

    int total_elements = batch_size * out_channels * out_h * out_w;
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;

    int in_channels_per_group = in_channels / groups;
    int out_channels_per_group = out_channels / groups;

    conv2d_nchw_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias.value().data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size, in_channels, in_h, in_w,
        out_channels, (int)out_h, (int)out_w,
        kernel_size, stride, padding, dilation, groups,
        in_channels_per_group, out_channels_per_group
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    std::optional<torch::Tensor> bias,
    int stride,
    int padding,
    int dilation,
    int groups);
'''

ext = load_inline(
    name='conv2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['conv2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, out_channels, (kernel_size, kernel_size), stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.conv2d_forward(x.contiguous(), self.conv2d.weight, self.conv2d.bias, self.stride, self.padding, self.dilation, self.groups)

def get_inputs():
    batch_size = 16
    in_channels = 3
    out_channels = 64
    kernel_size = 3
    width = 256
    height = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]

def get_init_inputs():
    return [3, 64, 3]
