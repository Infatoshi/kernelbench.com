import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__device__ inline float act(float x) {
    float sp = log1pf(expf(x));
    float t = tanhf(sp);
    return t * x;
}

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    const float* __restrict__ bn_mean,
    const float* __restrict__ bn_var,
    const float* __restrict__ bn_weight,
    const float* __restrict__ bn_bias,
    float* __restrict__ out,
    int N, int C, int H, int W,
    int OC, int KH, int KW,
    int OH, int OW,
    float eps
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int oc = (idx / (OW*OH)) % OC;
    int n = idx / (OW*OH*OC);

    float sum = 0.0f;
    for (int ic = 0; ic < C; ++ic) {
        for (int kh = 0; kh < KH; ++kh) {
            for (int kw = 0; kw < KW; ++kw) {
                int ih = oh + kh;
                int iw = ow + kw;
                int x_idx = ((n*C + ic)*H + ih)*W + iw;
                int w_idx = ((oc*C + ic)*KH + kh)*KW + kw;
                sum += x[x_idx] * w[w_idx];
            }
        }
    }
    sum += bias[oc];

    float a = act(sum);

    float mean = bn_mean[oc];
    float var = bn_var[oc];
    float gamma = bn_weight[oc];
    float beta = bn_bias[oc];

    float norm = (a - mean) * rsqrtf(var + eps);
    float y = norm * gamma + beta;

    out[idx] = y;
}

torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    float eps
) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int KH = w.size(2);
    int KW = w.size(3);

    int OH = H - KH + 1;
    int OW = W - KW + 1;

    auto out = torch::empty({N, OC, OH, OW}, x.options());

    int total = N * OC * OH * OW;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.data_ptr<float>(),
        bn_mean.data_ptr<float>(),
        bn_var.data_ptr<float>(),
        bn_weight.data_ptr<float>(),
        bn_bias.data_ptr<float>(),
        out.data_ptr<float>(),
        N, C, H, W, OC, KH, KW, OH, OW, eps
    );

    return out;
}
'''

ext = load_inline(
    name='fused_conv_bn_act',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, float);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        return ext.fused_forward(
            x.contiguous(),
            self.conv.weight.contiguous(),
            self.conv.bias.contiguous(),
            self.bn.running_mean.contiguous(),
            self.bn.running_var.contiguous(),
            self.bn.weight.contiguous(),
            self.bn.bias.contiguous(),
            self.bn.eps
        )


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3]
