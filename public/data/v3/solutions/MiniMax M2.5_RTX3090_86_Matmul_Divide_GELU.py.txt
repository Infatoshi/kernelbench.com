import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Get the reference parameters
batch_size = 128
input_size = 4096
output_size = 4096
divisor = 10.0

# Optimized CUDA source with a well-tuned fused divide+GELU kernel
# Using PyTorch's optimized mm for matrix multiplication
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Fast GELU using tanh approximation
__device__ __forceinline__ float gelu(float x) {
    float c = static_cast<float>(0.7978845608028654);
    float x3 = x * x * x;
    return x * 0.5f * (1.0f + tanhf(c * (x + 0.044715f * x3)));
}

__device__ __forceinline__ half gelu_half(half x) {
    float v = __half2float(x);
    float c = static_cast<float>(0.7978845608028654);
    float x3 = v * v * v;
    float result = v * 0.5f * (1.0f + tanhf(c * (v + 0.044715f * x3)));
    return __float2half(result);
}

// Fused division + GELU kernel with vectorized operations
// Uses 128 threads per block, each processing 8 elements
template<typename T>
__global__ void fused_div_gelu_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int total_elements,
    float divisor) {
    
    // Each thread processes a contiguous chunk with strided access
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int stride = blockDim.x * gridDim.x;
    
    // Process 8 elements per thread for better throughput
    #pragma unroll
    for (int i = 0; i < 8; i++) {
        int idx = tid + i * stride;
        if (idx < total_elements) {
            float val = static_cast<float>(input[idx]);
            val = val / divisor;
            val = gelu(val);
            output[idx] = static_cast<T>(val);
        }
    }
}

// Usingwmma for tensor core operations (optional for future optimization)
__global__ void wmma_gemm_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    half* __restrict__ C,
    int M, int N, int K) {
    // Placeholder for WMMA-based GEMM if needed
}

torch::Tensor fused_div_gelu(
    torch::Tensor input,
    float divisor) {
    
    int num_elements = input.numel();
    auto output = torch::empty_like(input);
    
    const int threads = 256;
    const int blocks = (num_elements + threads * 8 - 1) / (threads * 8);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "fused_div_gelu", ([&] {
        fused_div_gelu_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            num_elements,
            divisor
        );
    }));
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor fused_div_gelu(torch::Tensor input, float divisor);
'''

# Compile the extension
ext = load_inline(
    name='fused_div_gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_div_gelu'],
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '-std=c++17', '-use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that fuses matrix multiplication, division, and GELU activation.
    Uses PyTorch's optimized mm/bmm for GEMM and custom kernel for fused divide+GELU.
    """
    def __init__(self, input_size, output_size, divisor):
        super(Model, self).__init__()
        self.linear = nn.Linear(input_size, output_size)
        self.divisor = divisor

    def forward(self, x):
        """
        Fused operation: linear(x) / divisor + gelu
        Uses PyTorch's optimized matrix multiply and our fused divide+GELU kernel.
        """
        # Use PyTorch's optimized mm for matrix multiplication
        # Linear: y = x @ W^T + b
        weight = self.linear.weight  # (output_size, input_size)
        bias = self.linear.bias      # (output_size,)
        
        # Efficient matrix multiplication
        x = torch.mm(x, weight.t())  # (batch, input) @ (output, input)^T -> (batch, output)
        
        # Add bias (broadcasting)
        if bias is not None:
            x = x + bias
        
        # Apply fused division and GELU using custom kernel
        x = ext.fused_div_gelu(x, self.divisor)
        
        return x

def get_inputs():
    return [torch.randn(batch_size, input_size, device='cuda')]

def get_init_inputs():
    return [input_size, output_size, divisor]
