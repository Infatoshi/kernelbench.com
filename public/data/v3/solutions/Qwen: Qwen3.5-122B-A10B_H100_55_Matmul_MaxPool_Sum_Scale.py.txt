import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

// Fused kernel for max pool, sum, and scale
template<typename T>
__global__ void fused_post_process_kernel(
    const T* input,      // [batch, out_features]
    T* output,           // [batch]
    int batch_size,
    int out_features,
    int kernel_size,
    float scale_factor
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    const T* row = input + batch_idx * out_features;
    T sum_val = T(0);
    
    // MaxPool1d(kernel_size=2): for each pair, take max, then sum
    int num_pairs = out_features / kernel_size;
    for (int i = 0; i < num_pairs; i++) {
        T a = row[i * kernel_size];
        T b = row[i * kernel_size + 1];
        T max_val = (a > b) ? a : b;
        sum_val += max_val;
    }
    
    output[batch_idx] = sum_val * T(scale_factor);
}

torch::Tensor fused_matmul_pool_sum_scale(
    torch::Tensor x,           // [batch, in_features]
    torch::Tensor weight,      // [out_features, in_features]
    torch::Tensor bias,        // [out_features]
    int kernel_size,
    float scale_factor
) {
    int batch_size = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);
    
    // Use float32 for computation
    auto options = x.options().dtype(torch::kFloat32);
    
    // Cast inputs to float32 if needed
    torch::Tensor x_fp32 = x.to(torch::kFloat32);
    torch::Tensor weight_fp32 = weight.to(torch::kFloat32);
    torch::Tensor bias_fp32 = bias.to(torch::kFloat32);
    
    // Allocate output for matmul: [batch, out_features]
    torch::Tensor matmul_out = torch::empty({batch_size, out_features}, options);
    
    const float alpha = 1.0f;
    const float beta = 0.0f;
    
    // Create cuBLAS handle
    cublasHandle_t cublas_handle;
    cublasCreate(&cublas_handle);
    
    // For each batch element, compute GEMM
    // C = alpha * A @ B^T + beta * C
    // A: [1, in_features], B: [out_features, in_features] -> C: [1, out_features]
    for (int b = 0; b < batch_size; b++) {
        const float* A = x_fp32.data_ptr<float>() + b * in_features;
        const float* B = weight_fp32.data_ptr<float>();
        float* C = matmul_out.data_ptr<float>() + b * out_features;
        
        cublasSgemm(
            cublas_handle,
            CUBLAS_OP_N, CUBLAS_OP_T,
            out_features, 1, in_features,
            &alpha,
            B, out_features,
            A, in_features,
            &beta,
            C, out_features
        );
    }
    
    cublasDestroy(cublas_handle);
    
    // Add bias
    torch::Tensor with_bias = matmul_out + bias_fp32;
    
    // Fused post-processing: max pool, sum, scale
    torch::Tensor final_out = torch::empty({batch_size}, options);
    
    int threads = 256;
    int blocks = batch_size;
    
    fused_post_process_kernel<float><<<blocks, threads>>>(
        with_bias.data_ptr<float>(),
        final_out.data_ptr<float>(),
        batch_size,
        out_features,
        kernel_size,
        scale_factor
    );
    
    // Cast back to original dtype
    return final_out.to(x.dtype());
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='',
    cuda_sources=cuda_source,
    functions=['fused_matmul_pool_sum_scale'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        return ext.fused_matmul_pool_sum_scale(
            x,
            self.matmul.weight,
            self.matmul.bias,
            self.kernel_size,
            self.scale_factor
        )

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 2, 0.5]
