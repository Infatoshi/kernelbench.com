import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void fused_conv_instancenorm_divide_kernel(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int N, int C_in, int C_out, int H, int W, int K, int H_out, int W_out,
    float divide_const
) {
    int n = blockIdx.x;
    int c_out = blockIdx.y;
    int tid = threadIdx.x;
    
    int spatial_size = H_out * W_out;
    int num_threads = blockDim.x;
    
    extern __shared__ float s_mem[];
    float* s_sum = s_mem;
    float* s_sum_sq = s_mem + 1;
    
    float thread_sum = 0.0f;
    float thread_sum_sq = 0.0f;
    
    // First pass: compute conv and accumulate stats
    for (int hw = tid; hw < spatial_size; hw += num_threads) {
        int h_out = hw / W_out;
        int w_out = hw % W_out;
        
        float val = bias[c_out];
        for (int c_in = 0; c_in < C_in; c_in++) {
            for (int kh = 0; kh < K; kh++) {
                for (int kw = 0; kw < K; kw++) {
                    int h_in = h_out + kh;
                    int w_in = w_out + kw;
                    int in_idx = ((n * C_in + c_in) * H + h_in) * W + w_in;
                    int w_idx = ((c_out * C_in + c_in) * K + kh) * K + kw;
                    val += input[in_idx] * weight[w_idx];
                }
            }
        }
        
        thread_sum += val;
        thread_sum_sq += val * val;
        
        // Store for second pass
        int out_idx = ((n * C_out + c_out) * H_out + h_out) * W_out + w_out;
        output[out_idx] = val;
    }
    
    // Reduce within thread block
    for (int offset = num_threads/2; offset > 0; offset /= 2) {
        thread_sum += __shfl_down_sync(0xFFFFFFFF, thread_sum, offset);
        thread_sum_sq += __shfl_down_sync(0xFFFFFFFF, thread_sum_sq, offset);
    }
    
    if (tid == 0) {
        s_sum[0] = thread_sum;
        s_sum_sq[0] = thread_sum_sq;
    }
    __syncthreads();
    
    float total_sum = s_sum[0];
    float total_sum_sq = s_sum_sq[0];
    float mean = total_sum / spatial_size;
    float variance = total_sum_sq / spatial_size - mean * mean;
    float inv_std = rsqrtf(variance + 1e-5f);
    
    // Second pass: apply norm and divide
    for (int hw = tid; hw < spatial_size; hw += num_threads) {
        int h_out = hw / W_out;
        int w_out = hw % W_out;
        int out_idx = ((n * C_out + c_out) * H_out + h_out) * W_out + w_out;
        float val = output[out_idx];
        output[out_idx] = (val - mean) * inv_std / divide_const;
    }
}

torch::Tensor fused_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float divide_const) {
    int N = input.size(0);
    int C_in = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int C_out = weight.size(0);
    int K = weight.size(2);
    int H_out = H - K + 1;
    int W_out = W - K + 1;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    dim3 grid(N, C_out);
    dim3 block(256);
    size_t smem_size = 2 * sizeof(float);
    
    fused_conv_instancenorm_divide_kernel<<<grid, block, smem_size>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C_in, C_out, H, W, K, H_out, W_out,
        divide_const
    );
    
    return output;
}
'''

cpp_source = 'torch::Tensor fused_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float divide_const);'

fused_module = load_inline(
    name='fused_conv_instancenorm',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels)
        self.divide_by = divide_by
        
    def forward(self, x):
        return fused_module.fused_forward(x, self.conv.weight, self.conv.bias, self.divide_by)

def get_inputs():
    return [torch.randn(16, 3, 256, 256)]

def get_init_inputs():
    return [3, 16, 3, 2.0]
