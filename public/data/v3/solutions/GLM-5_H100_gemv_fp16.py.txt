"""
FP16 GEMV for autoregressive decoding - Optimized with CUDA C++.

GEMV is memory-bound, so we need to maximize memory bandwidth utilization.
Using CUDA C++ for efficient GPU kernel implementation on H100.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Simple GEMV kernel: each warp handles one output element
// Optimized for memory bandwidth on H100

__global__ void gemv_kernel_simple(
    const half* __restrict__ x,
    const half* __restrict__ w,
    half* __restrict__ y,
    const int batch_size,
    const int in_features,
    const int out_features,
    const int stride_xb,
    const int stride_xi,
    const int stride_wo,
    const int stride_wi,
    const int stride_yb,
    const int stride_yo
) {
    // Each warp computes one output element
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x * blockDim.y + threadIdx.y;
    
    if (out_idx >= out_features) return;
    
    const int lane_id = threadIdx.x;
    const int warp_size = 32;
    
    // Pointers
    const half* x_row = x + batch_idx * stride_xb;
    const half* w_row = w + out_idx * stride_wo;
    
    // Accumulator (float for precision)
    float sum = 0.0f;
    
    // Each lane processes a subset of input features
    for (int i = lane_id; i < in_features; i += warp_size) {
        half x_val = x_row[i];
        half w_val = w_row[i];
        sum += __half2float(x_val) * __half2float(w_val);
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = warp_size / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Lane 0 writes result
    if (lane_id == 0) {
        y[batch_idx * stride_yb + out_idx * stride_yo] = __float2half(sum);
    }
}

// Optimized kernel with vectorized loads
__global__ void gemv_kernel_vectorized(
    const half* __restrict__ x,
    const half* __restrict__ w,
    half* __restrict__ y,
    const int batch_size,
    const int in_features,
    const int out_features,
    const int stride_xb,
    const int stride_xi,
    const int stride_wo,
    const int stride_wi,
    const int stride_yb,
    const int stride_yo
) {
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x * blockDim.y + threadIdx.y;
    
    if (out_idx >= out_features) return;
    
    const int lane_id = threadIdx.x;
    const int warp_size = 32;
    
    const half* x_row = x + batch_idx * stride_xb;
    const half* w_row = w + out_idx * stride_wo;
    
    // Vectorized load (half2)
    float sum = 0.0f;
    const int vec_in_features = in_features / 2;  // half2 elements
    
    // Process half2 elements
    const half2* x_vec = reinterpret_cast<const half2*>(x_row);
    const half2* w_vec = reinterpret_cast<const half2*>(w_row);
    
    for (int i = lane_id; i < vec_in_features; i += warp_size) {
        half2 x_val = x_vec[i];
        half2 w_val = w_vec[i];
        // Multiply and add both elements
        sum += __half2float(x_val.x) * __half2float(w_val.x);
        sum += __half2float(x_val.y) * __half2float(w_val.y);
    }
    
    // Handle remaining elements if in_features is odd
    if (in_features % 2 == 1 && lane_id == 0) {
        int last_idx = in_features - 1;
        sum += __half2float(x_row[last_idx]) * __half2float(w_row[last_idx]);
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = warp_size / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        y[batch_idx * stride_yb + out_idx * stride_yo] = __float2half(sum);
    }
}

// Kernel with multiple warps per output for better memory utilization
__global__ void gemv_kernel_multi_warp(
    const half* __restrict__ x,
    const half* __restrict__ w,
    half* __restrict__ y,
    const int batch_size,
    const int in_features,
    const int out_features,
    const int stride_xb,
    const int stride_xi,
    const int stride_wo,
    const int stride_wi,
    const int stride_yb,
    const int stride_yo
) {
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.x;
    
    if (out_idx >= out_features) return;
    
    const int warp_id = threadIdx.y;
    const int lane_id = threadIdx.x;
    const int num_warps = blockDim.y;
    const int warp_size = 32;
    
    const half* x_row = x + batch_idx * stride_xb;
    const half* w_row = w + out_idx * stride_wo;
    
    // Each warp processes a chunk of the input
    float sum = 0.0f;
    const int chunk_size = (in_features + num_warps - 1) / num_warps;
    const int start = warp_id * chunk_size;
    const int end = min(start + chunk_size, in_features);
    
    for (int i = start + lane_id; i < end; i += warp_size) {
        sum += __half2float(x_row[i]) * __half2float(w_row[i]);
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = warp_size / 2; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Block reduction using shared memory
    __shared__ float partial_sums[32];  // Max 32 warps
    if (lane_id == 0) {
        partial_sums[warp_id] = sum;
    }
    __syncthreads();
    
    // First warp does final reduction
    if (warp_id == 0) {
        float total = (lane_id < num_warps) ? partial_sums[lane_id] : 0.0f;
        #pragma unroll
        for (int offset = warp_size / 2; offset > 0; offset /= 2) {
            total += __shfl_down_sync(0xffffffff, total, offset);
        }
        if (lane_id == 0) {
            y[batch_idx * stride_yb + out_idx * stride_yo] = __float2half(total);
        }
    }
}

torch::Tensor gemv_cuda(torch::Tensor x, torch::Tensor w) {
    const int batch_size = x.size(0);
    const int in_features = x.size(1);
    const int out_features = w.size(0);
    
    auto y = torch::empty({batch_size, out_features}, x.options());
    
    // Use multi-warp kernel for better memory bandwidth
    const int num_warps = 4;  // 4 warps per output
    const int threads_per_warp = 32;
    
    dim3 block(threads_per_warp, num_warps);
    dim3 grid(out_features, batch_size);
    
    gemv_kernel_multi_warp<<<grid, block>>>(
        x.data_ptr<at::Half>(),
        w.data_ptr<at::Half>(),
        y.data_ptr<at::Half>(),
        batch_size,
        in_features,
        out_features,
        x.stride(0),
        x.stride(1),
        w.stride(0),
        w.stride(1),
        y.stride(0),
        y.stride(1)
    );
    
    return y;
}
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources='torch::Tensor gemv_cuda(torch::Tensor x, torch::Tensor w);',
    cuda_sources=cuda_source,
    functions=['gemv_cuda'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_90', '-O3'],
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        return ext.gemv_cuda(x_fp16, self.weight)


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp16"]
HARDWARE_REQUIRED = ["RTX3090", "A100", "H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
