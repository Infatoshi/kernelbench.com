import torch
import torch.nn as nn
import triton
import triton.language as tl


# Triton GELU kernel - highly optimized for B200
@triton.jit
def gelu_kernel(
    x_ptr,
    y_ptr,
    n_elements,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    block_start = pid * BLOCK_SIZE
    offsets = block_start + tl.arange(0, BLOCK_SIZE)
    mask = offsets < n_elements
    
    x = tl.load(x_ptr + offsets, mask=mask)
    
    # Exact GELU: x * 0.5 * (1 + erf(x / sqrt(2)))
    # Use tl.math.erf for exact computation
    x_f32 = x.to(tl.float32)
    sqrt2_inv = 0.7071067811865475
    result = x_f32 * 0.5 * (1.0 + tl.math.erf(x_f32 * sqrt2_inv))
    result = result.to(x.dtype)
    
    tl.store(y_ptr + offsets, result, mask=mask)


def gelu_forward(x: torch.Tensor) -> torch.Tensor:
    y = torch.empty_like(x)
    n = x.numel()
    
    # Choose block size based on dtype for vectorization
    # Triton handles vectorization internally
    BLOCK_SIZE = 2048
    
    grid = lambda meta: (triton.cdiv(n, meta['BLOCK_SIZE']),)
    
    gelu_kernel[grid](x, y, n, BLOCK_SIZE=BLOCK_SIZE)
    return y


class Model(nn.Module):
    """
    Optimized GELU activation using Triton kernel.
    Supports fp16, bf16, fp32 with vectorized memory access.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_cuda:
            x = x.cuda()
        return gelu_forward(x)
