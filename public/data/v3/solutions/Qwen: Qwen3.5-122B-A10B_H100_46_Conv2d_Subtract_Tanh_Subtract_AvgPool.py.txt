import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused kernel for: (x - sub1) -> tanh -> (x - sub2) -> avgpool
# This fuses all element-wise operations into a single kernel

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Fused kernel: (x - sub1) -> tanh -> (x - sub2) -> avgpool
// Each thread computes one output element by aggregating over the pool window
template <typename T>
__global__ void fused_post_process_kernel(
    const T* __restrict__ conv_output,
    T* __restrict__ output,
    float sub1,
    float sub2,
    int64_t channels,
    int64_t out_height,
    int64_t out_width,
    int64_t batch_size,
    int64_t pool_kernel)
{
    int64_t pool_out_height = (out_height + pool_kernel - 1) / pool_kernel;
    int64_t pool_out_width = (out_width + pool_kernel - 1) / pool_kernel;
    
    int64_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    int64_t total_pool_elements = batch_size * channels * pool_out_height * pool_out_width;
    
    if (idx >= total_pool_elements) return;
    
    // Calculate pooled position
    int64_t tmp = idx;
    int64_t pool_pw = tmp % pool_out_width;
    tmp /= pool_out_width;
    int64_t pool_ph = tmp % pool_out_height;
    tmp /= pool_out_height;
    int64_t pc = tmp % channels;
    int64_t pb = tmp / channels;
    
    // Accumulate values from pool window
    float sum = 0.0f;
    int count = 0;
    
    int64_t start_h = pool_ph * pool_kernel;
    int64_t start_w = pool_pw * pool_kernel;
    
    for (int64_t ph = start_h; ph < start_h + pool_kernel && ph < out_height; ph++) {
        for (int64_t pw = start_w; pw < start_w + pool_kernel && pw < out_width; pw++) {
            int64_t input_idx = ((pb * channels + pc) * out_height + ph) * out_width + pw;
            float val = (float)conv_output[input_idx];
            val = val - sub1;
            val = tanhf(val);
            val = val - sub2;
            sum += val;
            count++;
        }
    }
    
    output[idx] = (T)(sum / count);
}

torch::Tensor fused_post_process(
    torch::Tensor conv_output,
    float sub1,
    float sub2,
    int64_t pool_kernel)
{
    int64_t batch_size = conv_output.size(0);
    int64_t channels = conv_output.size(1);
    int64_t out_height = conv_output.size(2);
    int64_t out_width = conv_output.size(3);
    
    int64_t pool_out_height = (out_height + pool_kernel - 1) / pool_kernel;
    int64_t pool_out_width = (out_width + pool_kernel - 1) / pool_kernel;
    
    auto output = torch::empty({batch_size, channels, pool_out_height, pool_out_width}, 
                               conv_output.options());
    
    int64_t total_pool_elements = batch_size * channels * pool_out_height * pool_out_width;
    int threads = 256;
    int blocks = (total_pool_elements + threads - 1) / threads;
    
    if (conv_output.dtype() == torch::kFloat32) {
        fused_post_process_kernel<float><<<blocks, threads>>>(
            conv_output.data_ptr<float>(),
            output.data_ptr<float>(),
            sub1, sub2,
            channels, out_height, out_width,
            batch_size, pool_kernel);
    } else if (conv_output.dtype() == torch::kBFloat16) {
        fused_post_process_kernel<at::BFloat16><<<blocks, threads>>>(
            conv_output.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            sub1, sub2,
            channels, out_height, out_width,
            batch_size, pool_kernel);
    } else if (conv_output.dtype() == torch::kFloat16) {
        fused_post_process_kernel<at::Half><<<blocks, threads>>>(
            conv_output.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            sub1, sub2,
            channels, out_height, out_width,
            batch_size, pool_kernel);
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_post_ext',
    cpp_sources='''
torch::Tensor fused_post_process(
    torch::Tensor conv_output,
    float sub1,
    float sub2,
    int64_t pool_kernel);
    ''',
    cuda_sources=cuda_source,
    functions=['fused_post_process'],
    verbose=False,
)


class Model(nn.Module):
    """
    Optimized model that fuses: Conv2d -> Subtract -> Tanh -> Subtract -> AvgPool2d
    Uses fused kernel for element-wise operations after convolution
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        # Conv2d - PyTorch's conv2d is already highly optimized on H100
        conv_out = self.conv(x)
        
        # Fused post-processing: (x - sub1) -> tanh -> (x - sub2) -> avgpool
        result = ext.fused_post_process(
            conv_out,
            self.subtract1_value,
            self.subtract2_value,
            self.kernel_size_pool
        )
        
        return result


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]


# Constants from reference
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2
