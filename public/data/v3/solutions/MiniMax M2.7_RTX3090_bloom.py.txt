import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = '''
#include <torch/extension.h>

torch::Tensor bloom_fused_forward(torch::Tensor image, float threshold, float intensity, float max_val);
'''

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

constexpr int BLUR_KERNEL = 5;
constexpr int BLUR_RADIUS = BLUR_KERNEL / 2;
constexpr int BLOCK_X = 32;
constexpr int BLOCK_Y = 8;
constexpr int SHMEM_Y = BLOCK_Y + BLUR_KERNEL - 1;  // 12
constexpr int SHMEM_X = BLOCK_X + BLUR_KERNEL - 1;  // 36

template <typename T>
__global__ void bloom_single_pass_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int N, int C, int H, int W,
    T threshold,
    T intensity,
    T max_val)
{
    extern __shared__ char shmem_char[];
    float* shmem = reinterpret_cast<float*>(shmem_char);
    
    int tid = threadIdx.y * blockDim.x + threadIdx.x;
    int block_total = blockDim.x * blockDim.y;
    
    int out_y = blockIdx.y * BLOCK_Y + threadIdx.y;
    int out_x = blockIdx.x * BLOCK_X + threadIdx.x;
    int in_base_y = blockIdx.y * BLOCK_Y;
    int in_base_x = blockIdx.x * BLOCK_X;
    
    // Process each channel
    for (int c = 0; c < C; ++c) {
        // Load input with halo into shared memory
        for (int i = tid; i < SHMEM_Y * SHMEM_X; i += block_total) {
            int sy = i / SHMEM_X;
            int sx = i % SHMEM_X;
            
            int gx = in_base_x + sx - BLUR_RADIUS;
            int gy = in_base_y + sy - BLUR_RADIUS;
            
            // Clamp to image bounds
            gx = gx < 0 ? 0 : (gx >= W ? W - 1 : gx);
            gy = gy < 0 ? 0 : (gy >= H ? H - 1 : gy);
            
            T val = input[((c * H + gy) * W + gx)];
            shmem[sy * SHMEM_X + sx] = static_cast<float>(val);
        }
        
        __syncthreads();
        
        // Compute output for threads that are in bounds
        if (out_y < H && out_x < W) {
            // Compute bright value and 5x5 blur
            int local_y = threadIdx.y + BLUR_RADIUS;
            int local_x = threadIdx.x + BLUR_RADIUS;
            
            T input_val = static_cast<T>(shmem[local_y * SHMEM_X + local_x]);
            T bright_center = input_val - threshold;
            bright_center = bright_center > static_cast<T>(0.0f) ? bright_center : static_cast<T>(0.0f);
            
            // 5x5 box blur
            float blur_sum = 0.0f;
            for (int ky = -BLUR_RADIUS; ky <= BLUR_RADIUS; ++ky) {
                for (int kx = -BLUR_RADIUS; kx <= BLUR_RADIUS; ++kx) {
                    float val = shmem[(local_y + ky) * SHMEM_X + (local_x + kx)];
                    float bright = val - static_cast<float>(threshold);
                    bright = bright > 0.0f ? bright : 0.0f;
                    blur_sum += bright;
                }
            }
            
            T blurred = static_cast<T>(blur_sum / static_cast<float>(BLUR_KERNEL * BLUR_KERNEL));
            
            // Compute final result with clamp
            T result = input_val + intensity * blurred;
            result = result > static_cast<T>(0.0f) ? result : static_cast<T>(0.0f);
            result = result < max_val ? result : max_val;
            
            output[((c * H + out_y) * W + out_x)] = result;
        }
        
        __syncthreads();
    }
}

// Explicit template instantiations
template __global__ void bloom_single_pass_kernel<float>(
    const float* __restrict__,
    float* __restrict__,
    int, int, int, int,
    float, float, float);

template __global__ void bloom_single_pass_kernel<double>(
    const double* __restrict__,
    double* __restrict__,
    int, int, int, int,
    double, double, double);

torch::Tensor bloom_fused_forward(torch::Tensor image, float threshold, float intensity, float max_val) {
    int N = image.size(0);
    int C = image.size(1);
    int H = image.size(2);
    int W = image.size(3);
    
    auto output = torch::empty_like(image);
    
    const dim3 block(BLOCK_X, BLOCK_Y);
    const dim3 grid((W + BLOCK_X - 1) / BLOCK_X, (H + BLOCK_Y - 1) / BLOCK_Y);
    
    size_t shmem_size = SHMEM_Y * SHMEM_X * sizeof(float);
    
    if (image.scalar_type() == at::kFloat) {
        bloom_single_pass_kernel<float><<<grid, block, shmem_size>>>(
            image.data_ptr<float>(),
            output.data_ptr<float>(),
            N, C, H, W,
            threshold, intensity, max_val
        );
    } else if (image.scalar_type() == at::kDouble) {
        bloom_single_pass_kernel<double><<<grid, block, shmem_size>>>(
            image.data_ptr<double>(),
            output.data_ptr<double>(),
            N, C, H, W,
            static_cast<double>(threshold),
            static_cast<double>(intensity),
            static_cast<double>(max_val)
        );
    }
    
    return output;
}
'''

ext = load_inline(
    name='bloom_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bloom_fused_forward'],
    verbose=False,
)

class Model(nn.Module):
    """Optimized Bloom effect with fused CUDA kernel."""

    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel
        self.max_val = 10.0

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        return ext.bloom_fused_forward(
            image,
            self.threshold,
            self.intensity,
            self.max_val
        )


def get_inputs():
    return [torch.randn(2, 3, 1024, 1024) * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
