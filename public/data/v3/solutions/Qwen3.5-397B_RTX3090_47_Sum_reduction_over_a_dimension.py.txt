import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

template<typename T>
__global__ void sum_reduce_kernel(const T* x, T* y, int batch_size, int reduce_dim, int out_dim) {
    // Each block handles one (batch_idx, out_col) pair
    int batch_idx = blockIdx.y;
    int out_col = blockIdx.x;
    
    // Shared memory for partial sums
    extern __shared__ float shared[];
    int tid = threadIdx.x;
    
    // Initialize shared memory
    shared[tid] = 0.0f;
    
    // Each thread processes multiple elements along reduce_dim
    int stride = blockDim.x;
    float local_sum = 0.0f;
    
    for (int reduce_idx = tid; reduce_idx < reduce_dim; reduce_idx += stride) {
        int input_idx = batch_idx * reduce_dim * out_dim + reduce_idx * out_dim + out_col;
        local_sum += static_cast<float>(x[input_idx]);
    }
    
    shared[tid] = local_sum;
    __syncthreads();
    
    // Reduction in shared memory
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared[tid] += shared[tid + s];
        }
        __syncthreads();
    }
    
    // Write result
    if (tid == 0) {
        y[batch_idx * out_dim + out_col] = static_cast<T>(shared[0]);
    }
}

torch::Tensor sum_reduce(torch::Tensor x, int dim) {
    // Assume dim is the middle dimension (dim=1 for 3D tensor)
    auto sizes = x.sizes();
    int ndim = sizes.size();
    
    // Permute so that reduce dim is in the middle
    std::vector<int64_t> permute_dims;
    for (int i = 0; i < ndim; i++) {
        if (i != dim) permute_dims.push_back(i);
    }
    permute_dims.insert(permute_dims.begin() + 1, dim);
    
    auto x_perm = x.permute(permute_dims).contiguous();
    auto sizes_perm = x_perm.sizes();
    
    int batch_size = 1;
    for (int i = 0; i < ndim - 2; i++) {
        batch_size *= sizes_perm[i];
    }
    int reduce_dim = sizes_perm[ndim - 2];
    int out_dim = sizes_perm[ndim - 1];
    
    auto y_perm = torch::empty({batch_size, 1, out_dim}, x_perm.options());
    auto y_perm_flat = y_perm.view({batch_size, out_dim});
    
    dim3 block(256);
    dim3 grid(out_dim, batch_size);
    int shared_mem = 256 * sizeof(float);
    
    if (x.dtype() == torch::kFloat32) {
        sum_reduce_kernel<float><<<grid, block, shared_mem>>>(
            x_perm.data_ptr<float>(),
            y_perm_flat.data_ptr<float>(),
            batch_size, reduce_dim, out_dim
        );
    } else if (x.dtype() == torch::kFloat16) {
        sum_reduce_kernel<at::Half><<<grid, block, shared_mem>>>(
            x_perm.data_ptr<at::Half>(),
            y_perm_flat.data_ptr<at::Half>(),
            batch_size, reduce_dim, out_dim
        );
    } else if (x.dtype() == torch::kBFloat16) {
        sum_reduce_kernel<at::BFloat16><<<grid, block, shared_mem>>>(
            x_perm.data_ptr<at::BFloat16>(),
            y_perm_flat.data_ptr<at::BFloat16>(),
            batch_size, reduce_dim, out_dim
        );
    }
    
    // Reverse permute
    std::vector<int64_t> reverse_permute(ndim);
    for (int i = 0; i < ndim - 1; i++) {
        reverse_permute[i] = i;
    }
    reverse_permute[ndim - 1] = permute_dims[0];
    
    // Actually we need to reverse the permutation properly
    std::vector<int64_t> final_permute(ndim);
    for (int i = 0; i < ndim; i++) {
        final_permute[permute_dims[i]] = i;
    }
    
    // Reshape to final output shape
    std::vector<int64_t> out_sizes;
    for (int i = 0; i < ndim; i++) {
        if (i == dim) {
            out_sizes.push_back(1);
        } else {
            out_sizes.push_back(sizes[i]);
        }
    }
    
    auto y = y_perm.view(out_sizes).permute(final_permute).contiguous();
    return y;
}
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources='torch::Tensor sum_reduce(torch::Tensor x, int dim);',
    cuda_sources=cuda_source,
    functions=['sum_reduce'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce(x, self.dim)

def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [1]
