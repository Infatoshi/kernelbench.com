import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

__device__ __forceinline__ float mish_activation(float x) {
    if (x > 20.0f) return x;
    return x * tanhf(log1pf(expf(x)));
}

template <typename T>
__global__ void conv_mish_bn_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    const T* __restrict__ bn_weight,
    const T* __restrict__ bn_bias,
    const T* __restrict__ running_mean,
    const T* __restrict__ running_var,
    T* __restrict__ output,
    int N, int C_in, int H_in, int W_in,
    int C_out, int K, int H_out, int W_out,
    float eps) {

    int index = blockIdx.x * blockDim.x + threadIdx.x;
    int spatial_out = H_out * W_out;
    int total_elements = N * C_out * spatial_out;
    if (index >= total_elements) return;

    int w_out = index % W_out;
    int h_out = (index / W_out) % H_out;
    int c_out = (index / spatial_out) % C_out;
    int n     = index / (spatial_out * C_out);

    float sum = (float)bias[c_out];
    const T* weight_ptr = weight + c_out * C_in * K * K;
    
    #pragma unroll
    for (int ci = 0; ci < C_in; ++ci) {
        const T* in_ptr = input + (n * C_in + ci) * H_in * W_in + h_out * W_in + w_out;
        const T* w_ci_ptr = weight_ptr + ci * K * K;
        #pragma unroll
        for (int ki = 0; ki < K; ++ki) {
            #pragma unroll
            for (int kj = 0; kj < K; ++kj) {
                sum += (float)in_ptr[ki * W_in + kj] * (float)w_ci_ptr[ki * K + kj];
            }
        }
    }

    float activated = mish_activation(sum);
    float inv_std = 1.0f / sqrtf((float)running_var[c_out] + eps);
    float out_val = (activated - (float)running_mean[c_out]) * inv_std * (float)bn_weight[c_out] + (float)bn_bias[c_out];
    output[index] = (T)out_val;
}

torch::Tensor forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    float eps) {
    
    int N = input.size(0);
    int C_in = input.size(1);
    int H_in = input.size(2);
    int W_in = input.size(3);
    int C_out = weight.size(0);
    int K = weight.size(2);
    int H_out = H_in - K + 1;
    int W_out = W_in - K + 1;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    int total_elements = N * C_out * H_out * W_out;
    int threads = 256;
    int blocks = (total_elements + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "conv_mish_bn_kernel", ([&] {
        conv_mish_bn_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            bn_weight.data_ptr<scalar_t>(),
            bn_bias.data_ptr<scalar_t>(),
            running_mean.data_ptr<scalar_t>(),
            running_var.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            N, C_in, H_in, W_in,
            C_out, K, H_out, W_out,
            eps
        );
    }));
    
    return output;
}
'''

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps
        
        self.ext = load_inline(
            name='fused_conv_mish_bn_v5',
            cpp_sources='torch::Tensor forward(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, float);',
            cuda_sources=cuda_source,
            functions=['forward'],
            verbose=False,
        )

    def forward(self, x):
        return self.ext.forward(
            x.contiguous(),
            self.conv.weight.contiguous(),
            self.conv.bias.contiguous(),
            self.bn.weight.contiguous(),
            self.bn.bias.contiguous(),
            self.bn.running_mean.contiguous(),
            self.bn.running_var.contiguous(),
            self.eps
        )

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3]
