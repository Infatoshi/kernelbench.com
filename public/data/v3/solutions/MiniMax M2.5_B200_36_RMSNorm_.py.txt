import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized CUDA kernel for RMSNorm with fusion and warp-level reduction
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Warp-level reduction using shuffle
__device__ __forceinline__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// RMSNorm kernel for float32
__global__ void rmsnorm_kernel_float(
    const float* __restrict__ x,
    float* __restrict__ output,
    const int batch_size,
    const int features,
    const int dim1,
    const int dim2,
    const float eps
) {
    const int tid = threadIdx.x;
    const int nthreads = blockDim.x;
    
    const int row_idx = blockIdx.x;
    const int num_rows = batch_size * dim1 * dim2;
    
    if (row_idx >= num_rows) return;
    
    // Compute row coordinates
    const int batch_idx = row_idx / (dim1 * dim2);
    const int rem = row_idx % (dim1 * dim2);
    const int d1_idx = rem / dim2;
    const int d2_idx = rem % dim2;
    
    const int row_offset = ((batch_idx * dim1 + d1_idx) * dim2 + d2_idx) * features;
    
    // Phase 1: Compute sum of squares
    float sum = 0.0f;
    for (int i = tid; i < features; i += nthreads) {
        float val = x[row_offset + i];
        sum += val * val;
    }
    
    // Warp reduction
    float warp_sum = warp_reduce_sum(sum);
    
    // Store warp sums in shared memory
    extern __shared__ float sdata[];
    if (tid % 32 == 0) {
        sdata[tid / 32] = warp_sum;
    }
    __syncthreads();
    
    // Final reduction - first warp does the final sum
    if (tid < 32) {
        float block_sum = 0.0f;
        for (int i = 0; i < (nthreads + 31) / 32; i++) {
            block_sum += sdata[i];
        }
        block_sum = warp_reduce_sum(block_sum);
        if (tid == 0) {
            sdata[0] = block_sum;
        }
    }
    __syncthreads();
    
    float rms = sqrtf(sdata[0] / (float)features + eps);
    
    // Phase 2: Normalize
    for (int i = tid; i < features; i += nthreads) {
        output[row_offset + i] = x[row_offset + i] / rms;
    }
}

// RMSNorm kernel for half precision
__global__ void rmsnorm_kernel_half(
    const __half* __restrict__ x,
    __half* __restrict__ output,
    const int batch_size,
    const int features,
    const int dim1,
    const int dim2,
    const float eps
) {
    const int tid = threadIdx.x;
    const int nthreads = blockDim.x;
    
    const int row_idx = blockIdx.x;
    const int num_rows = batch_size * dim1 * dim2;
    
    if (row_idx >= num_rows) return;
    
    // Compute row coordinates
    const int batch_idx = row_idx / (dim1 * dim2);
    const int rem = row_idx % (dim1 * dim2);
    const int d1_idx = rem / dim2;
    const int d2_idx = rem % dim2;
    
    const int row_offset = ((batch_idx * dim1 + d1_idx) * dim2 + d2_idx) * features;
    
    // Phase 1: Compute sum of squares
    float sum = 0.0f;
    for (int i = tid; i < features; i += nthreads) {
        float val = __half2float(x[row_offset + i]);
        sum += val * val;
    }
    
    // Warp reduction
    float warp_sum = warp_reduce_sum(sum);
    
    // Store warp sums in shared memory
    extern __shared__ float sdata[];
    if (tid % 32 == 0) {
        sdata[tid / 32] = warp_sum;
    }
    __syncthreads();
    
    // Final reduction - first warp does the final sum
    if (tid < 32) {
        float block_sum = 0.0f;
        for (int i = 0; i < (nthreads + 31) / 32; i++) {
            block_sum += sdata[i];
        }
        block_sum = warp_reduce_sum(block_sum);
        if (tid == 0) {
            sdata[0] = block_sum;
        }
    }
    __syncthreads();
    
    float rms = sqrtf(sdata[0] / (float)features + eps);
    
    // Phase 2: Normalize
    for (int i = tid; i < features; i += nthreads) {
        float val = __half2float(x[row_offset + i]);
        output[row_offset + i] = __float2half(val / rms);
    }
}

torch::Tensor rmsnorm_op_float(torch::Tensor x, float eps) {
    const int batch_size = x.size(0);
    const int features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    auto output = torch::empty_like(x);
    
    const int num_rows = batch_size * dim1 * dim2;
    const int block_size = 256;
    const int smem_size = (block_size / 32) * sizeof(float);
    
    rmsnorm_kernel_float<<<num_rows, block_size, smem_size, 0>>>(
        x.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, features, dim1, dim2, eps
    );
    
    cudaDeviceSynchronize();
    return output;
}

torch::Tensor rmsnorm_op_half(torch::Tensor x, float eps) {
    const int batch_size = x.size(0);
    const int features = x.size(1);
    const int dim1 = x.size(2);
    const int dim2 = x.size(3);
    
    auto output = torch::empty_like(x);
    
    const int num_rows = batch_size * dim1 * dim2;
    const int block_size = 256;
    const int smem_size = (block_size / 32) * sizeof(float);
    
    rmsnorm_kernel_half<<<num_rows, block_size, smem_size, 0>>>(
        reinterpret_cast<__half*>(x.data_ptr<at::Half>()),
        reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
        batch_size, features, dim1, dim2, eps
    );
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = r'''
torch::Tensor rmsnorm_op_float(torch::Tensor x, float eps);
torch::Tensor rmsnorm_op_half(torch::Tensor x, float eps);
'''

ext = load_inline(
    name='rmsnorm_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['rmsnorm_op_float', 'rmsnorm_op_half'],
    verbose=False,
    extra_cuda_cflags=['-res-usage', '--maxrregcount=128', '-use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Dispatch based on dtype
        if x.dtype == torch.float32:
            return ext.rmsnorm_op_float(x, self.eps)
        elif x.dtype == torch.float16:
            return ext.rmsnorm_op_half(x, self.eps)
        elif x.dtype == torch.bfloat16:
            # Convert to float32 for computation, then back
            x_fp32 = x.float()
            result = ext.rmsnorm_op_float(x_fp32, self.eps)
            return result.bfloat16()
        else:
            # Fallback for unsupported dtypes
            rms = torch.sqrt(torch.mean(x ** 2, dim=1, keepdim=True) + self.eps)
            return x / rms


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [64]
