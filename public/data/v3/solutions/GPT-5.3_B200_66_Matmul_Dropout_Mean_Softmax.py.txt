import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void matmul_mean_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int B, int K, int N)
{
    int j = blockIdx.x * blockDim.x + threadIdx.x;
    if (j >= N) return;

    float sum = 0.0f;

    for (int i = 0; i < B; ++i) {
        float dot = 0.0f;
        const float* x_row = x + i * K;
        const float* w_row = w + j * K;
        for (int k = 0; k < K; ++k) {
            dot += x_row[k] * w_row[k];
        }
        dot += b[j];
        sum += dot;
    }

    out[j] = sum / B;
}

__global__ void softmax_kernel(float* x, int N) {
    __shared__ float max_val;
    __shared__ float sum;

    int tid = threadIdx.x;

    float val = -1e20f;
    for (int i = tid; i < N; i += blockDim.x) {
        val = fmaxf(val, x[i]);
    }

    atomicMax((int*)&max_val, __float_as_int(val));
    __syncthreads();

    float exp_sum = 0.0f;
    for (int i = tid; i < N; i += blockDim.x) {
        float e = expf(x[i] - max_val);
        x[i] = e;
        exp_sum += e;
    }

    atomicAdd(&sum, exp_sum);
    __syncthreads();

    for (int i = tid; i < N; i += blockDim.x) {
        x[i] /= sum;
    }
}


torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b) {
    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);

    auto out = torch::empty({N}, x.options());

    int threads = 256;
    int blocks = (N + threads - 1) / threads;

    matmul_mean_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        B, K, N);

    softmax_kernel<<<1, 256>>>(out.data_ptr<float>(), N);

    return out.unsqueeze(0);
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        return ext.fused_op(x.contiguous(), self.weight.contiguous(), self.bias.contiguous())


def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]


def get_init_inputs():
    return [4096, 4096, 0.2]
