import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMV: C[M,1] = A[M,K] * B[K,1]
// M=256, K=131072
// This is memory-bandwidth bound. Key: maximize bandwidth utilization.
// Strategy: Use multiple thread blocks per row to increase parallelism.
// Each thread block handles a chunk of K, does partial reduction, then atomic add.

// Each thread loads multiple float4s (vectorized loads)
// Block size: 256 threads
// Each thread handles ITEMS_PER_THREAD elements
// Per block: 256 * ITEMS_PER_THREAD elements of K
// We split each row across multiple blocks

template<int BLOCK_SIZE, int ITEMS_PER_THREAD>
__global__ void gemv_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int K
) {
    const int row = blockIdx.x;
    const int k_block = blockIdx.y;
    const int tid = threadIdx.x;
    
    const int elements_per_block = BLOCK_SIZE * ITEMS_PER_THREAD;
    const int k_start = k_block * elements_per_block;
    
    // Use float4 for vectorized loads (4 floats = 16 bytes)
    const int VEC_SIZE = 4;
    const int vec_items = ITEMS_PER_THREAD / VEC_SIZE;
    
    float sum = 0.0f;
    
    const float4* A_vec = reinterpret_cast<const float4*>(A + row * K + k_start);
    const float4* B_vec = reinterpret_cast<const float4*>(B + k_start);
    
    #pragma unroll
    for (int i = 0; i < vec_items; i++) {
        int vec_idx = tid * vec_items + i;
        int k_idx = k_start + vec_idx * VEC_SIZE;
        if (k_idx + VEC_SIZE <= K) {
            float4 a4 = A_vec[vec_idx];
            float4 b4 = B_vec[vec_idx];
            sum += a4.x * b4.x + a4.y * b4.y + a4.z * b4.z + a4.w * b4.w;
        }
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Block reduction using shared memory
    __shared__ float warp_sums[BLOCK_SIZE / 32];
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();
    
    if (tid < BLOCK_SIZE / 32) {
        sum = warp_sums[tid];
        #pragma unroll
        for (int offset = BLOCK_SIZE / 64; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (tid == 0) {
            atomicAdd(&C[row], sum);
        }
    }
}

torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::zeros({M, 1}, A.options());
    
    constexpr int BLOCK_SIZE = 256;
    constexpr int ITEMS_PER_THREAD = 16;  // Each thread handles 16 elements
    constexpr int elements_per_block = BLOCK_SIZE * ITEMS_PER_THREAD; // 4096
    
    int num_k_blocks = (K + elements_per_block - 1) / elements_per_block;
    
    dim3 grid(M, num_k_blocks);
    dim3 block(BLOCK_SIZE);
    
    gemv_kernel<BLOCK_SIZE, ITEMS_PER_THREAD><<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K
    );
    
    return C;
}
'''

cpp_source = "torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B);"

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemv_cuda(A, B)


M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
