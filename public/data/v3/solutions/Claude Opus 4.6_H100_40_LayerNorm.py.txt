import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// Two-pass LayerNorm for very large normalized dimensions
// Pass 1: Compute partial sums and partial sum-of-squares per block
// Pass 2: Normalize using global mean/variance

// We use multiple blocks per row. Each block handles a chunk of the normalized dimension.

// Warp-level reduction
__device__ __forceinline__ float warpReduceSum(float val) {
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, offset);
    return val;
}

// Block-level reduction using shared memory
__device__ float blockReduceSum(float val) {
    __shared__ float shared[32]; // one per warp
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    val = warpReduceSum(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    int nwarps = blockDim.x >> 5;
    val = (threadIdx.x < nwarps) ? shared[threadIdx.x] : 0.0f;
    if (wid == 0) val = warpReduceSum(val);
    
    return val;
}

// Pass 1: compute partial sums
__global__ void layernorm_partial_sums(
    const float* __restrict__ input,
    float* __restrict__ partial_sum,      // [batch, blocks_per_row]
    float* __restrict__ partial_sum_sq,   // [batch, blocks_per_row]
    int N,  // normalized dimension size
    int blocks_per_row
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    
    int chunk_size = (N + blocks_per_row - 1) / blocks_per_row;
    int start = block_in_row * chunk_size;
    int end = min(start + chunk_size, N);
    
    const float* row_input = input + (long long)row * N;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = start + threadIdx.x; i < end; i += blockDim.x) {
        float val = row_input[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    if (threadIdx.x == 0) {
        int idx = row * blocks_per_row + block_in_row;
        partial_sum[idx] = local_sum;
        partial_sum_sq[idx] = local_sum_sq;
    }
}

// Pass 2: finalize stats and normalize
__global__ void layernorm_normalize(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ partial_sum,
    const float* __restrict__ partial_sum_sq,
    float* __restrict__ output,
    int N,
    int blocks_per_row,
    float eps
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    
    // First, compute mean and variance from partial sums
    // Use shared memory for this
    __shared__ float s_mean;
    __shared__ float s_inv_std;
    
    if (threadIdx.x == 0) {
        float total_sum = 0.0f;
        float total_sum_sq = 0.0f;
        int base = row * blocks_per_row;
        for (int i = 0; i < blocks_per_row; i++) {
            total_sum += partial_sum[base + i];
            total_sum_sq += partial_sum_sq[base + i];
        }
        float mean = total_sum / N;
        float var = total_sum_sq / N - mean * mean;
        s_mean = mean;
        s_inv_std = rsqrtf(var + eps);
    }
    __syncthreads();
    
    float mean = s_mean;
    float inv_std = s_inv_std;
    
    int chunk_size = (N + blocks_per_row - 1) / blocks_per_row;
    int start = block_in_row * chunk_size;
    int end = min(start + chunk_size, N);
    
    const float* row_input = input + (long long)row * N;
    float* row_output = output + (long long)row * N;
    
    for (int i = start + threadIdx.x; i < end; i += blockDim.x) {
        float val = row_input[i];
        float norm = (val - mean) * inv_std;
        row_output[i] = norm * weight[i] + bias[i];
    }
}

// Vectorized version using float4 for better memory throughput
__global__ void layernorm_partial_sums_vec4(
    const float* __restrict__ input,
    float* __restrict__ partial_sum,
    float* __restrict__ partial_sum_sq,
    int N,
    int blocks_per_row
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    
    int chunk_size = (N + blocks_per_row - 1) / blocks_per_row;
    int start = block_in_row * chunk_size;
    int end = min(start + chunk_size, N);
    
    // Align to float4 boundaries
    int start4 = (start + 3) / 4 * 4;
    int end4 = end / 4 * 4;
    
    const float* row_input = input + (long long)row * N;
    const float4* row_input4 = (const float4*)(row_input);
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    // Handle prefix
    for (int i = start + threadIdx.x; i < start4; i += blockDim.x) {
        float val = row_input[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    // Vectorized main loop
    int n4 = (end4 - start4) / 4;
    int base4 = start4 / 4;
    for (int i = threadIdx.x; i < n4; i += blockDim.x) {
        float4 v = row_input4[base4 + i];
        local_sum += v.x + v.y + v.z + v.w;
        local_sum_sq += v.x*v.x + v.y*v.y + v.z*v.z + v.w*v.w;
    }
    
    // Handle suffix
    for (int i = end4 + threadIdx.x; i < end; i += blockDim.x) {
        float val = row_input[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    if (threadIdx.x == 0) {
        int idx = row * blocks_per_row + block_in_row;
        partial_sum[idx] = local_sum;
        partial_sum_sq[idx] = local_sum_sq;
    }
}

__global__ void layernorm_normalize_vec4(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ partial_sum,
    const float* __restrict__ partial_sum_sq,
    float* __restrict__ output,
    int N,
    int blocks_per_row,
    float eps
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    
    __shared__ float s_mean;
    __shared__ float s_inv_std;
    
    if (threadIdx.x == 0) {
        float total_sum = 0.0f;
        float total_sum_sq = 0.0f;
        int base = row * blocks_per_row;
        for (int i = 0; i < blocks_per_row; i++) {
            total_sum += partial_sum[base + i];
            total_sum_sq += partial_sum_sq[base + i];
        }
        float mean = total_sum / N;
        float var = total_sum_sq / N - mean * mean;
        s_mean = mean;
        s_inv_std = rsqrtf(var + eps);
    }
    __syncthreads();
    
    float mean = s_mean;
    float inv_std = s_inv_std;
    
    int chunk_size = (N + blocks_per_row - 1) / blocks_per_row;
    int start = block_in_row * chunk_size;
    int end = min(start + chunk_size, N);
    
    int start4 = (start + 3) / 4 * 4;
    int end4 = end / 4 * 4;
    
    const float* row_input = input + (long long)row * N;
    float* row_output = output + (long long)row * N;
    const float4* row_input4 = (const float4*)(row_input);
    float4* row_output4 = (float4*)(row_output);
    const float4* weight4 = (const float4*)(weight);
    const float4* bias4 = (const float4*)(bias);
    
    // Handle prefix
    for (int i = start + threadIdx.x; i < start4; i += blockDim.x) {
        float val = row_input[i];
        float norm = (val - mean) * inv_std;
        row_output[i] = norm * weight[i] + bias[i];
    }
    
    // Vectorized main loop
    int n4 = (end4 - start4) / 4;
    int base4 = start4 / 4;
    for (int i = threadIdx.x; i < n4; i += blockDim.x) {
        float4 v = row_input4[base4 + i];
        float4 w = weight4[base4 + i];
        float4 b = bias4[base4 + i];
        float4 out;
        out.x = (v.x - mean) * inv_std * w.x + b.x;
        out.y = (v.y - mean) * inv_std * w.y + b.y;
        out.z = (v.z - mean) * inv_std * w.z + b.z;
        out.w = (v.w - mean) * inv_std * w.w + b.w;
        row_output4[base4 + i] = out;
    }
    
    // Handle suffix
    for (int i = end4 + threadIdx.x; i < end; i += blockDim.x) {
        float val = row_input[i];
        float norm = (val - mean) * inv_std;
        row_output[i] = norm * weight[i] + bias[i];
    }
}


torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float eps) {
    // input: [batch, ...normalized_shape]
    // weight, bias: [normalized_shape_flat]
    
    auto input_contig = input.contiguous();
    auto output = torch::empty_like(input_contig);
    
    int batch = input_contig.size(0);
    int N = input_contig.numel() / batch;
    
    // Choose blocks_per_row to keep enough parallelism
    // With 16 rows, we want enough blocks to fill the GPU
    // H100 has 132 SMs
    int blocks_per_row = 64;  // 16 * 64 = 1024 blocks total
    int threads = 512;
    
    int total_blocks = batch * blocks_per_row;
    
    auto partial_sum = torch::empty({total_blocks}, input_contig.options());
    auto partial_sum_sq = torch::empty({total_blocks}, input_contig.options());
    
    layernorm_partial_sums_vec4<<<total_blocks, threads>>>(
        input_contig.data_ptr<float>(),
        partial_sum.data_ptr<float>(),
        partial_sum_sq.data_ptr<float>(),
        N, blocks_per_row
    );
    
    layernorm_normalize_vec4<<<total_blocks, threads>>>(
        input_contig.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        partial_sum.data_ptr<float>(),
        partial_sum_sq.data_ptr<float>(),
        output.data_ptr<float>(),
        N, blocks_per_row, eps
    );
    
    return output;
}
''';

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, float eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(
            x, 
            self.ln.weight.reshape(-1),
            self.ln.bias.reshape(-1),
            self.ln.eps
        )

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
