cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// FP32 kernel - main optimized kernel
__global__ void fused_conv_bn_scale_fp32_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ bn_mean,
    const float* __restrict__ bn_var,
    const float* __restrict__ bn_gamma,
    const float* __restrict__ bn_beta,
    float* __restrict__ output,
    int batch_size, int in_channels, int height, int width,
    int out_channels, int kernel_size, int stride, int padding,
    float scaling_factor, float eps
) {
    int out_h = (height + 2 * padding - kernel_size) / stride + 1;
    int out_w = (width + 2 * padding - kernel_size) / stride + 1;
    
    int out_ch = blockIdx.x;
    int b = blockIdx.y;
    
    if (out_ch >= out_channels || b >= batch_size) return;
    
    int out_y = threadIdx.y;
    int out_x = threadIdx.x;
    
    if (out_y >= out_h || out_x >= out_w) return;
    
    // Precompute fused scale and bias
    float bn_scale = bn_gamma[out_ch] / sqrtf(bn_var[out_ch] + eps);
    float bn_bias = bn_beta[out_ch] - bn_scale * bn_mean[out_ch];
    float final_scale = bn_scale * scaling_factor;
    float bias_val = bias ? bias[out_ch] : 0.0f;
    float final_bias = bn_bias * scaling_factor + bias_val * final_scale;
    
    // Convolution accumulator
    float acc = 0.0f;
    
    // Convolution loop
    for (int in_ch = 0; in_ch < in_channels; in_ch++) {
        for (int ky = 0; ky < kernel_size; ky++) {
            int in_y = out_y * stride + ky - padding;
            if (in_y >= 0 && in_y < height) {
                for (int kx = 0; kx < kernel_size; kx++) {
                    int in_x = out_x * stride + kx - padding;
                    if (in_x >= 0 && in_x < width) {
                        int in_idx = ((b * in_channels + in_ch) * height + in_y) * width + in_x;
                        int weight_idx = ((out_ch * in_channels + in_ch) * kernel_size + ky) * kernel_size + kx;
                        acc += input[in_idx] * weight[weight_idx];
                    }
                }
            }
        }
    }
    
    // Apply fused BN and scaling
    acc = acc * final_scale + final_bias;
    
    // Store result
    int out_idx = ((b * out_channels + out_ch) * out_h + out_y) * out_w + out_x;
    output[out_idx] = acc;
}

// Host function for FP32
torch::Tensor fused_conv_bn_scale_fp32(
    torch::Tensor input, torch::Tensor weight, torch::Tensor bias,
    torch::Tensor bn_mean, torch::Tensor bn_var, torch::Tensor bn_gamma, torch::Tensor bn_beta,
    int stride, int padding, float scaling_factor, float eps
) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_h = (height + 2 * padding - kernel_size) / stride + 1;
    int out_w = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_h, out_w}, input.options());
    
    const int threads_y = 16;
    const int threads_x = 16;
    dim3 threads(threads_x, threads_y);
    dim3 blocks(out_channels, batch_size);
    
    fused_conv_bn_scale_fp32_kernel<<<blocks, threads>>>(
        input.data_ptr<float>(), weight.data_ptr<float>(), 
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        bn_mean.data_ptr<float>(), bn_var.data_ptr<float>(),
        bn_gamma.data_ptr<float>(), bn_beta.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_channels, height, width,
        out_channels, kernel_size, stride, padding,
        scaling_factor, eps
    );
    
    return output;
}

ext = load_inline(
    name='fused_conv_bn_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=[
        'fused_conv_bn_scale_fp32',
        'fused_conv_bn_scale_fp16', 
        'fused_conv_bn_scale_bf16'
    ],
    extra_include_paths=[],
    verbose=False
)

class Model(nn.Module):
    """
    Fused Conv2d + BatchNorm + Scaling model using optimized CUDA kernel.
    Precomputes fused BN statistics and performs all operations in a single kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, precision='fp32'):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        self.precision = precision
        
        # Compute BN running statistics
        self.register_buffer('bn_mean', self.bn.running_mean.clone())
        self.register_buffer('bn_var', self.bn.running_var.clone())
        self.register_buffer('bn_gamma', self.bn.weight.clone())
        self.register_buffer('bn_beta', self.bn.bias.clone())
        
        self.eps = self.bn.eps
        
    def forward(self, x):
        if self.precision == 'fp16':
            x = x.half()
            weight = self.conv.weight.half()
            bias = self.conv.bias.half() if self.conv.bias is not None else torch.tensor([], dtype=torch.half, device=x.device)
            bn_mean = self.bn_mean.float()
            bn_var = self.bn_var.float()
            bn_gamma = self.bn_gamma.float()
            bn_beta = self.bn_beta.float()
            output = ext.fused_conv_bn_scale_fp16(
                x, weight, bias, bn_mean, bn_var, bn_gamma, bn_beta,
                self.conv.stride[0], self.conv.padding[0], self.scaling_factor, self.eps
            )
            return output.to(torch.float32)
        elif self.precision == 'bf16':
            x = x.bfloat16()
            weight = self.conv.weight.bfloat16()
            bias = self.conv.bias.bfloat16() if self.conv.bias is not None else torch.tensor([], dtype=torch.bfloat16, device=x.device)
            bn_mean = self.bn_mean.float()
            bn_var = self.bn_var.float()
            bn_gamma = self.bn_gamma.float()
            bn_beta = self.bn_beta.float()
            output = ext.fused_conv_bn_scale_bf16(
                x, weight, bias, bn_mean, bn_var, bn_gamma, bn_beta,
                self.conv.stride[0], self.conv.padding[0], self.scaling_factor, self.eps
            )
            return output.to(torch.float32)
        else:
            weight = self.conv.weight.float()
            bias = self.conv.bias.float() if self.conv.bias is not None else torch.tensor([], dtype=torch.float32, device=x.device)
            bn_mean = self.bn_mean.float()
            bn_var = self.bn_var.float()
            bn_gamma = self.bn_gamma.float()
            bn_beta = self.bn_beta.float()
            output = ext.fused_conv_bn_scale_fp32(
                x, weight, bias, bn_mean, bn_var, bn_gamma, bn_beta,
                self.conv.stride[0], self.conv.padding[0], self.scaling_factor, self.eps
            )
            return output


def get_inputs():
