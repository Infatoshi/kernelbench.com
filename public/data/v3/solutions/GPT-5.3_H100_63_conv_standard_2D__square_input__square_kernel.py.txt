import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def conv2d_kernel(x, w, y,
                  N, C, H, W,
                  K, R, S,
                  outH, outW,
                  stride, pad,
                  BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    n = pid // (K * outH * outW)
    rem = pid % (K * outH * outW)
    k = rem // (outH * outW)
    rem = rem % (outH * outW)
    oh = rem // outW
    ow = rem % outW

    acc = 0.0
    for c in range(0, C):
        for r in range(0, R):
            for s in range(0, S):
                ih = oh * stride - pad + r
                iw = ow * stride - pad + s
                if (ih >= 0) and (ih < H) and (iw >= 0) and (iw < W):
                    x_idx = ((n*C + c)*H + ih)*W + iw
                    w_idx = ((k*C + c)*R + r)*S + s
                    acc += tl.load(x + x_idx) * tl.load(w + w_idx)

    y_idx = ((n*K + k)*outH + oh)*outW + ow
    tl.store(y + y_idx, acc)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1, bias=False):
        super().__init__()
        self.conv2d = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=False)

    def forward(self, x):
        return torch.ops.aten.convolution(
            x, self.conv.weight, None,
            self.conv.stride,
            self.conv.padding,
            self.conv.dilation,
            False, [0,0], 1
        )
