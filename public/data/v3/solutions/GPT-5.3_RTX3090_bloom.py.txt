import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void bloom_kernel(const float* __restrict__ img, float* __restrict__ out,
                             int B,int C,int H,int W,
                             float thr,float intensity,int k,int pad) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int N = B*C*H*W;
    if (idx >= N) return;

    int w = idx % W;
    int h = (idx / W) % H;
    int c = (idx / (W*H)) % C;
    int b = idx / (W*H*C);

    float sum = 0.0f;
    int ks = k;

    for (int dy = -pad; dy <= pad; ++dy) {
        int yy = h + dy;
        if (yy < 0 || yy >= H) continue;
        for (int dx = -pad; dx <= pad; ++dx) {
            int xx = w + dx;
            if (xx < 0 || xx >= W) continue;
            int nidx = (((b*C + c)*H + yy)*W + xx);
            float v = img[nidx] - thr;
            if (v > 0.f) sum += v;
        }
    }

    float blurred = sum / (ks*ks);
    float base = img[idx];
    float val = base + intensity * blurred;
    if (val < 0.f) val = 0.f;
    if (val > 10.f) val = 10.f;
    out[idx] = val;
}

torch::Tensor bloom(torch::Tensor img, float thr, float intensity, int k) {
    int B = img.size(0);
    int C = img.size(1);
    int H = img.size(2);
    int W = img.size(3);

    auto out = torch::empty_like(img);

    int N = B*C*H*W;
    int threads = 256;
    int blocks = (N + threads - 1) / threads;

    bloom_kernel<<<blocks, threads>>>(img.data_ptr<float>(), out.data_ptr<float>(),
        B,C,H,W,thr,intensity,k,k/2);

    return out;
}
'''

ext = load_inline(
    name='bloom_ext',
    cpp_sources='torch::Tensor bloom(torch::Tensor img, float thr, float intensity, int k);',
    cuda_sources=cuda_src,
    functions=['bloom'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        orig_dtype = image.dtype
        if image.dtype != torch.float32:
            image = image.float()
        image = image.contiguous()
        out = ext.bloom(image, float(self.threshold), float(self.intensity), int(self.blur_kernel))
        if orig_dtype != torch.float32:
            out = out.to(orig_dtype)
        return out


def get_inputs():
    return [torch.randn(2, 3, 1024, 1024, device='cuda') * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
