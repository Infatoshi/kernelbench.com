import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "fused"

# Fused GEMM + Swish + Bias kernel
@triton.jit
def gemm_swish_bias_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, bias_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Meta-parameters
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """Kernel for computing matmul + swish + bias"""
    # Block IDs
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Pointers to blocks
    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = b_ptr + (offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn)

    # Initialize accumulator
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    # Main loop over K dimension
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Load blocks
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        b = tl.load(b_ptrs, mask=offs_k[:, None] < K - k * BLOCK_SIZE_K, other=0.0)
        # Matrix multiply
        accumulator += tl.dot(a, b)
        # Advance pointers
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_bk

    # Apply swish: x * sigmoid(x)
    c = accumulator * tl.sigmoid(accumulator)
    
    # Load linear_bias and add_bias and add both
    offs_bias = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    linear_bias_val = tl.load(bias_ptr + offs_bias, mask=offs_bias < N, other=0.0)
    c = c + linear_bias_val[None, :]

    # Store result
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, c, mask=c_mask)


# GroupNorm forward kernel
@triton.jit
def group_norm_kernel(
    x_ptr, y_ptr, weight_ptr, bias_ptr,
    N, C, G,
    eps,
    BLOCK_SIZE_C: tl.constexpr,
):
    """GroupNorm forward kernel"""
    # Get group and batch indices
    pid = tl.program_id(0)
    batch_id = pid // G
    group_id = pid % G
    
    # Compute channels per group
    C_PER_G = C // G
    channel_start = group_id * C_PER_G
    
    # Pointers for this group
    x_offset = batch_id * C + channel_start + tl.arange(0, BLOCK_SIZE_C)
    x_ptrs = x_ptr + x_offset
    
    # Compute mean
    mean = tl.zeros([1], dtype=tl.float32)
    for c in range(0, C_PER_G, BLOCK_SIZE_C):
        mask = (tl.arange(0, BLOCK_SIZE_C) + c) < C_PER_G
        x = tl.load(x_ptrs + c, mask=mask, other=0.0).to(tl.float32)
        mean += tl.sum(x, axis=0)
    mean = mean / C_PER_G
    
    # Compute variance
    var = tl.zeros([1], dtype=tl.float32)
    for c in range(0, C_PER_G, BLOCK_SIZE_C):
        mask = (tl.arange(0, BLOCK_SIZE_C) + c) < C_PER_G
        x = tl.load(x_ptrs + c, mask=mask, other=0.0).to(tl.float32)
        x = x - mean
        var += tl.sum(x * x, axis=0)
    var = var / C_PER_G
    rstd = tl.rsqrt(var + eps)
    
    # Normalize and apply weight/bias
    w_ptrs = weight_ptr + channel_start + tl.arange(0, BLOCK_SIZE_C)
    b_ptrs = bias_ptr + channel_start + tl.arange(0, BLOCK_SIZE_C)
    y_ptrs = y_ptr + x_offset
    
    for c in range(0, C_PER_G, BLOCK_SIZE_C):
        mask = (tl.arange(0, BLOCK_SIZE_C) + c) < C_PER_G
        x = tl.load(x_ptrs + c, mask=mask, other=0.0).to(tl.float32)
        w = tl.load(w_ptrs + c, mask=mask, other=0.0).to(tl.float32)
        b = tl.load(b_ptrs + c, mask=mask, other=0.0).to(tl.float32)
        y = (x - mean) * rstd * w + b
        tl.store(y_ptrs + c, y, mask=mask)

    add_bias = add_bias.contiguous()

def triton_group_norm(x, weight, bias, num_groups, eps=1e-5):
    """GroupNorm using Triton"""
    N, C = x.shape
    G = num_groups
    C_PER_G = C // G
    
    # Allocate output
    y = torch.empty_like(x)
    
    BLOCK_SIZE_C = 128
    grid = (N * G,)
    
    group_norm_kernel[grid](
        x, y, weight, bias,
        N, C, G,
        eps,
        BLOCK_SIZE_C=BLOCK_SIZE_C,
    )
    
    return y


class Model(nn.Module):
    """
    A model that performs a matrix multiplication, applies Swish activation, sums with a bias term, and normalizes with GroupNorm.
    Optimized with Triton kernels for H100.
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.num_groups = num_groups

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).
        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_features).
        """
        # Combine biases: linear bias + add bias
        combined_bias = self.matmul.bias.float() + self.bias.float()
        
        # Fused GEMM + Swish + Bias using Triton
        x = fused_gemm_swish_bias(x.float(), self.matmul.weight.float(), combined_bias)
        
        # GroupNorm using Triton
        x = triton_group_norm(x, self.group_norm.weight.float(), self.group_norm.bias.float(), 
                             self.num_groups)
        
        return x
