import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float scale,
    float eps,
    float* __restrict__ out,
    int N,int C,int H,int W,
    int OC,int KH,int KW,
    int OH,int OW
){
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*OC*OH*OW;
    if(idx>=total) return;

    int ow = idx % OW;
    int oh = (idx/OW) % OH;
    int oc = (idx/(OW*OH)) % OC;
    int n  = idx/(OW*OH*OC);

    float v = 0.0f;

    for(int ic=0; ic<C; ++ic){
        for(int kh=0; kh<KH; ++kh){
            for(int kw=0; kw<KW; ++kw){
                int ih = oh + kh;
                int iw = ow + kw;
                int x_idx = ((n*C + ic)*H + ih)*W + iw;
                int w_idx = ((oc*C + ic)*KH + kh)*KW + kw;
                v += x[x_idx] * w[w_idx];
            }
        }
    }

    if(bias) v += bias[oc];

    float mean = running_mean[oc];
    float var = running_var[oc];
    float inv_std = rsqrtf(var + eps);

    float y = (v - mean) * inv_std;
    y = y * gamma[oc] + beta[oc];
    y = y * scale;

    out[idx] = y;
}

torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    torch::Tensor gamma,
    torch::Tensor beta,
    float scale,
    float eps
){
    x = x.contiguous();
    w = w.contiguous();

    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int KH = w.size(2);
    int KW = w.size(3);

    int OH = H - KH + 1;
    int OW = W - KW + 1;

    auto out = torch::empty({N,OC,OH,OW}, x.options());

    int total = N*OC*OH*OW;
    int threads = 256;
    int blocks = (total + threads - 1)/threads;

    fused_kernel<<<blocks,threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        bias.numel()?bias.data_ptr<float>():nullptr,
        running_mean.data_ptr<float>(),
        running_var.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        scale,
        eps,
        out.data_ptr<float>(),
        N,C,H,W,OC,KH,KW,OH,OW
    );

    return out;
}
'''

cpp_source = "torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor bias, torch::Tensor running_mean, torch::Tensor running_var, torch::Tensor gamma, torch::Tensor beta, float scale, float eps);"

ext = load_inline(
    name='fused_conv_bn_scale',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        return ext.fused_forward(
            x,
            self.conv.weight,
            self.conv.bias if self.conv.bias is not None else torch.tensor([] ,device=x.device),
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            float(self.scaling_factor),
            float(self.bn.eps)
        )

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width, device='cuda', dtype=torch.float32)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor]
