import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    float* __restrict__ y,
    int N,int C,int H,int W,
    int K,int R,int S,
    int outH,int outW,
    int stride,int pad,int dilation)
{
    int ow = blockIdx.x * blockDim.x + threadIdx.x;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;

    int bz = blockIdx.z;
    int oc = bz % K;
    int n = bz / K;

    if (ow >= outW || oh >= outH) return;

    float acc = 0.0f;

    for(int c=0;c<C;c++){
        for(int r=0;r<R;r++){
            for(int s=0;s<S;s++){
                int ih = oh*stride - pad + r*dilation;
                int iw = ow*stride - pad + s*dilation;
                if(ih>=0 && ih<H && iw>=0 && iw<W){
                    int x_idx = ((n*C + c)*H + ih)*W + iw;
                    int w_idx = ((oc*C + c)*R + r)*S + s;
                    acc += x[x_idx] * w[w_idx];
                }
            }
        }
    }

    int y_idx = ((n*K + oc)*outH + oh)*outW + ow;
    y[y_idx] = acc;
}


torch::Tensor conv2d_forward(
    torch::Tensor x,
    torch::Tensor w,
    int stride,int pad,int dilation)
{
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int K = w.size(0);
    int R = w.size(2);
    int S = w.size(3);

    int outH = (H + 2*pad - dilation*(R-1) - 1)/stride + 1;
    int outW = (W + 2*pad - dilation*(S-1) - 1)/stride + 1;

    auto y = torch::empty({N,K,outH,outW}, x.options());

    dim3 block(16,16);
    dim3 grid((outW+15)/16,(outH+15)/16,N*K);

    conv2d_kernel<<<grid,block>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        y.data_ptr<float>(),
        N,C,H,W,K,R,S,outH,outW,stride,pad,dilation
    );

    return y;
}
'''

ext = load_inline(
    name='conv2d_ext',
    cpp_sources='torch::Tensor conv2d_forward(torch::Tensor x, torch::Tensor w, int stride, int pad, int dilation);',
    cuda_sources=cuda_source,
    functions=['conv2d_forward'],
    extra_cuda_cflags=['-O3'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        self.conv2d = nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.conv2d.weight
        return ext.conv2d_forward(x.contiguous(), w.contiguous(), self.conv2d.stride[0], self.conv2d.padding[0], self.conv2d.dilation[0])


def get_inputs():
    batch_size = 16
    in_channels = 3
    out_channels = 64
    kernel_size = 3
    width = 256
    height = 256
    x = torch.randn(batch_size, in_channels, height, width, device='cuda')
    return [x]


def get_init_inputs():
    in_channels = 3
    out_channels = 64
    kernel_size = 3
    return [in_channels, out_channels, kernel_size]
