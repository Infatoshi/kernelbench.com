import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

template <typename T>
__device__ __forceinline__ float to_float(T x) {
    return static_cast<float>(x);
}

__device__ __forceinline__ float to_float(at::Half x) {
    return __half2float(x);
}

__device__ __forceinline__ float to_float(at::BFloat16 x) {
    return static_cast<float>(x);
}

template <typename T>
__global__ void cross_entropy_forward_kernel(
    const T* __restrict__ input,
    const int64_t* __restrict__ target,
    float* __restrict__ output,
    int batch_size,
    int num_classes) {
    
    int bid = blockIdx.x;
    if (bid >= batch_size) return;
    
    const T* row_input = input + bid * num_classes;
    int tid = threadIdx.x;
    int lane = tid & 31;
    int warp_id = tid >> 5;
    
    // Find max for numerical stability
    float local_max = -1e20f;
    
    #pragma unroll 4
    for (int i = tid; i < num_classes; i += blockDim.x) {
        local_max = fmaxf(local_max, to_float(row_input[i]));
    }
    
    // Warp reduction for max
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_max = fmaxf(local_max, __shfl_down_sync(0xffffffff, local_max, offset));
    }
    
    __shared__ float shared_max[8]; 
    if (lane == 0) shared_max[warp_id] = local_max;
    __syncthreads();
    
    // Cross-warp reduction
    if (warp_id == 0) {
        float val = (tid < blockDim.x / 32) ? shared_max[tid] : -1e20f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
        }
        if (tid == 0) shared_max[0] = val;
    }
    __syncthreads();
    
    float row_max = shared_max[0];
    
    // Compute sum of exp(x - max)
    float local_sum = 0.0f;
    
    #pragma unroll 4
    for (int i = tid; i < num_classes; i += blockDim.x) {
        float val = to_float(row_input[i]);
        local_sum += expf(val - row_max);
    }
    
    // Warp reduction for sum
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
    }
    
    __shared__ float shared_sum[8];
    if (lane == 0) shared_sum[warp_id] = local_sum;
    __syncthreads();
    
    // Cross-warp reduction
    if (warp_id == 0) {
        float val = (tid < blockDim.x / 32) ? shared_sum[tid] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            val += __shfl_down_sync(0xffffffff, val, offset);
        }
        if (tid == 0) shared_sum[0] = val;
    }
    __syncthreads();
    
    float total_sum = shared_sum[0];
    
    // Compute loss
    if (tid == 0) {
        int target_idx = static_cast<int>(target[bid]);
        float target_val = to_float(row_input[target_idx]);
        float loss = logf(total_sum) + row_max - target_val;
        atomicAdd(output, loss / static_cast<float>(batch_size));
    }
}

torch::Tensor cross_entropy_forward(torch::Tensor input, torch::Tensor target) {
    int batch_size = input.size(0);
    int num_classes = input.size(1);
    
    auto output = torch::zeros({}, torch::dtype(torch::kFloat32).device(input.device()));
    
    const int threads = 256;
    const int blocks = batch_size;
    
    auto scalar_type = input.scalar_type();
    
    if (scalar_type == at::ScalarType::Float) {
        cross_entropy_forward_kernel<float><<<blocks, threads>>>(
            input.data_ptr<float>(),
            target.data_ptr<int64_t>(),
            output.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (scalar_type == at::ScalarType::Half) {
        cross_entropy_forward_kernel<at::Half><<<blocks, threads>>>(
            input.data_ptr<at::Half>(),
            target.data_ptr<int64_t>(),
            output.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else if (scalar_type == at::ScalarType::BFloat16) {
        cross_entropy_forward_kernel<at::BFloat16><<<blocks, threads>>>(
            input.data_ptr<at::BFloat16>(),
            target.data_ptr<int64_t>(),
            output.data_ptr<float>(),
            batch_size,
            num_classes
        );
    } else {
        AT_ERROR("Unsupported dtype for cross_entropy");
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor cross_entropy_forward(torch::Tensor input, torch::Tensor target);
'''

ext = load_inline(
    name='cross_entropy_fused',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
    extra_cflags=['-O3'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets).squeeze()

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    predictions = torch.randn(batch_size, num_classes, device='cuda')
    targets = torch.randint(0, num_classes, (batch_size,), device='cuda')
    return [predictions, targets]

def get_init_inputs():
    return []
