import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_fp16.h>

__global__ void int4_gemm_kernel(
    const half* __restrict__ X, // M,K
    const unsigned char* __restrict__ Wp, // N,K/2
    const half* __restrict__ scales, // N, num_groups
    half* __restrict__ Y, // M,N
    int M, int N, int K, int group_size, int num_groups)
{
    int m = blockIdx.y;
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (m >= M || n >= N) return;

    const half* xrow = X + m * K;
    const unsigned char* wrow = Wp + n * (K/2);

    float acc = 0.0f;

    for (int k = 0; k < K; ++k) {
        half xv = xrow[k];

        unsigned char byte = wrow[k >> 1];
        int wq;
        if ((k & 1) == 0) wq = byte & 0xF;
        else wq = (byte >> 4) & 0xF;

        int g = k / group_size;
        half scale_h = scales[n * num_groups + g];
        half wq_h = __float2half((float)wq);
        half w_h = __hmul(scale_h, __hsub(wq_h, __float2half(8.0f)));
        float w = __half2float(w_h);
        acc += __half2float(xv) * w;
    }

    Y[m * N + n] = __float2half(acc);
}

torch::Tensor int4_gemm(torch::Tensor x, torch::Tensor w_packed, torch::Tensor scales, int group_size) {
    int M = x.size(0);
    int K = x.size(1);
    int N = w_packed.size(0);
    int num_groups = scales.size(1);

    auto y = torch::empty({M, N}, x.options());

    dim3 block(128);
    dim3 grid((N + 127)/128, M);

    int4_gemm_kernel<<<grid, block>>>(
        (half*)x.data_ptr<at::Half>(),
        w_packed.data_ptr<unsigned char>(),
        (half*)scales.data_ptr<at::Half>(),
        (half*)y.data_ptr<at::Half>(),
        M,N,K,group_size,num_groups
    );

    return y;
}
'''

ext = load_inline(
    name='int4_gemm_ext',
    cpp_sources='torch::Tensor int4_gemm(torch::Tensor x, torch::Tensor w_packed, torch::Tensor scales, int group_size);',
    cuda_sources=cuda_src,
    functions=['int4_gemm'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0
        assert K % 2 == 0

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        x2d = x.view(-1, self.K).contiguous()
        out = ext.int4_gemm(x2d, self.weight_packed, self.scales, self.group_size)
        return out.view(batch_size, seq_len, self.N)


def get_inputs():
    batch_size = 4
    seq_len = 2048
    K = 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16, device='cuda')]


def get_init_inputs():
    K = 4096
    N = 11008
    group_size = 128
    return [K, N, group_size]
