import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def gemv_fp16_kernel(
    x_ptr,
    w_ptr,
    y_ptr,
    B,
    K,
    N,
    stride_xb,
    stride_xk,
    stride_wn,
    stride_wk,
    stride_yb,
    stride_yn,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_n = tl.cdiv(N, BLOCK_N)
    b = pid // num_n
    n_block = pid % num_n

    offs_n = n_block * BLOCK_N + tl.arange(0, BLOCK_N)
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)

    k_iter = tl.arange(0, BLOCK_K)
    for k0 in range(0, K, BLOCK_K):
        offs_k = k0 + k_iter
        x = tl.load(x_ptr + b * stride_xb + offs_k * stride_xk, mask=offs_k < K, other=0.0)
        w = tl.load(
            w_ptr + offs_n[:, None] * stride_wn + offs_k[None, :] * stride_wk,
            mask=(offs_n[:, None] < N) & (offs_k[None, :] < K),
            other=0.0,
        )
        acc += tl.sum(w * x[None, :], axis=1)

    out_ptrs = y_ptr + b * stride_yb + offs_n * stride_yn
    tl.store(out_ptrs, acc.to(tl.float16), mask=offs_n < N)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        if not x_fp16.is_cuda:
            x_fp16 = x_fp16.cuda()
        w = self.weight
        B, K = x_fp16.shape
        N = w.shape[0]
        y = torch.empty((B, N), device=x_fp16.device, dtype=torch.float16)
        grid = (B * triton.cdiv(N, 128),)
        gemv_fp16_kernel[grid](
            x_fp16,
            w,
            y,
            B,
            K,
            N,
            x_fp16.stride(0),
            x_fp16.stride(1),
            w.stride(0),
            w.stride(1),
            y.stride(0),
            y.stride(1),
            BLOCK_N=128,
            BLOCK_K=128,
            num_warps=4,
            num_stages=3,
        )
        return y


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp16"]
HARDWARE_REQUIRED = ["RTX3090", "A100", "H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
