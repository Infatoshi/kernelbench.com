import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMV: C[m] = sum_k A[m,k] * B[k]
// M=256, K=131072
// Key insight: This is memory-bandwidth bound.
// Strategy: Assign multiple blocks per row to increase parallelism.
// Use float4 vectorized loads.
// Then do a two-level reduction.

// Kernel 1: Each block handles (row, k_chunk). 
// Grid: (M, num_k_blocks), Block: 256 threads
// Partial sums stored in temp buffer, then reduced.

__global__ void gemv_partial(
    const float* __restrict__ A,   // [M, K]
    const float* __restrict__ B,   // [K]
    float* __restrict__ partial,   // [M, num_k_blocks]
    int M, int K, int num_k_blocks)
{
    int row = blockIdx.x;
    int kb = blockIdx.y;

    // This block handles K elements starting at kb*chunk_size
    int chunk_size = (K + num_k_blocks - 1) / num_k_blocks;
    // Align chunk to float4 (16 bytes = 4 floats)
    int k_start = kb * chunk_size;
    int k_end = min(k_start + chunk_size, K);

    if (k_start >= K) return;

    const float* A_row = A + (long)row * K;
    
    float sum = 0.0f;

    // Use float4 loads for coalesced 128-bit access
    // Align to float4 boundary
    int k_start_aligned = (k_start + 3) & ~3;
    int k_end_aligned = k_end & ~3;

    // Scalar prefix
    for (int i = k_start + threadIdx.x; i < min(k_start_aligned, k_end); i += blockDim.x) {
        sum += A_row[i] * B[i];
    }

    // Vectorized main loop
    if (k_start_aligned < k_end_aligned) {
        const float4* A4 = reinterpret_cast<const float4*>(A_row + k_start_aligned);
        const float4* B4 = reinterpret_cast<const float4*>(B + k_start_aligned);
        int len4 = (k_end_aligned - k_start_aligned) / 4;
        
        for (int i = threadIdx.x; i < len4; i += blockDim.x) {
            float4 a = __ldg(A4 + i);
            float4 b = __ldg(B4 + i);
            sum += a.x*b.x + a.y*b.y + a.z*b.z + a.w*b.w;
        }
    }

    // Scalar suffix
    for (int i = max(k_start, k_end_aligned) + threadIdx.x; i < k_end; i += blockDim.x) {
        sum += A_row[i] * B[i];
    }

    // Warp reduction
    unsigned mask = 0xffffffff;
    #pragma unroll
    for (int off = 16; off >= 1; off >>= 1)
        sum += __shfl_down_sync(mask, sum, off);

    // Shared memory reduction across warps
    __shared__ float shmem[8]; // up to 256/32=8 warps
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    
    if (lane == 0) shmem[warp] = sum;
    __syncthreads();

    int WARPS = blockDim.x >> 5;
    if (warp == 0) {
        sum = (lane < WARPS) ? shmem[lane] : 0.0f;
        #pragma unroll
        for (int off = 16; off >= 1; off >>= 1)
            sum += __shfl_down_sync(mask, sum, off);
        if (lane == 0)
            partial[row * num_k_blocks + kb] = sum;
    }
}

// Kernel 2: Reduce partial sums for each row
__global__ void gemv_reduce(
    const float* __restrict__ partial, // [M, num_k_blocks]
    float* __restrict__ C,             // [M, 1]
    int M, int num_k_blocks)
{
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    if (row >= M) return;
    
    float sum = 0.0f;
    const float* p = partial + row * num_k_blocks;
    for (int i = 0; i < num_k_blocks; i++)
        sum += p[i];
    C[row] = sum;
}

// Single-pass kernel: each row gets one block of 1024 threads
// For K=131072: 1024 threads, each does K/1024=128 float4 loads = 512 floats
// Memory per thread: 512 * 4 bytes = 2048 bytes
// This maximizes ILP
__global__ void gemv_single(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int K)
{
    int row = blockIdx.x;
    if (row >= M) return;

    const float* A_row = A + (long)row * K;
    
    float sum = 0.0f;
    int k4 = K >> 2;
    const float4* A4 = reinterpret_cast<const float4*>(A_row);
    const float4* B4 = reinterpret_cast<const float4*>(B);

    // Each thread: processes k4/blockDim.x float4 elements
    // K=131072, blockDim.x=1024: each thread does 32 float4 = 128 floats
    for (int i = threadIdx.x; i < k4; i += blockDim.x) {
        float4 a = __ldg(A4 + i);
        float4 b = __ldg(B4 + i);
        sum += a.x*b.x + a.y*b.y + a.z*b.z + a.w*b.w;
    }

    // Warp reduction
    unsigned mask = 0xffffffff;
    #pragma unroll
    for (int off = 16; off >= 1; off >>= 1)
        sum += __shfl_down_sync(mask, sum, off);

    __shared__ float shmem[32]; // 32 warps max for 1024 threads
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;

    if (lane == 0) shmem[warp] = sum;
    __syncthreads();

    if (warp == 0) {
        int WARPS = blockDim.x >> 5; // 32 for 1024 threads
        sum = (lane < WARPS) ? shmem[lane] : 0.0f;
        #pragma unroll
        for (int off = 16; off >= 1; off >>= 1)
            sum += __shfl_down_sync(mask, sum, off);
        if (lane == 0)
            C[row] = sum;
    }
}

torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);

    // B may be [K,1] — flatten to [K]
    auto B_flat = B.reshape({-1});
    auto C = torch::empty({M, 1}, A.options());

    // Single-pass: 256 blocks (one per row), 1024 threads each
    // For K=131072, each thread loads 128 floats (32 float4)
    // Very good for memory bandwidth
    gemv_single<<<M, 1024>>>(
        A.data_ptr<float>(),
        B_flat.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K
    );

    return C;
}
'''

cpp_source = '''
torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_ext_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math',
                       '-Xptxas', '-O3,--warn-on-spills',
                       '--ptxas-options=-v'],
)


class Model(nn.Module):
    """
    Optimized matrix-vector multiplication using custom CUDA kernel.
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.gemv_forward(A, B)


M = 256
K = 131072


def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]


def get_init_inputs():
    return []
