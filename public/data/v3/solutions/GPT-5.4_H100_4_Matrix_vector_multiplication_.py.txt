import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ['fp4', 'fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def matvec_kernel(
    A_ptr, B_ptr, C_ptr,
    M, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    BLOCK_K: tl.constexpr,
):
    pid_m = tl.program_id(0)
    offs_m = pid_m
    offs_k = tl.arange(0, BLOCK_K)

    acc = 0.0
    for k0 in range(0, K, BLOCK_K):
        k = k0 + offs_k
        mask = k < K
        a = tl.load(A_ptr + offs_m * stride_am + k * stride_ak, mask=mask, other=0.0)
        b = tl.load(B_ptr + k * stride_bk + 0 * stride_bn, mask=mask, other=0.0)
        acc += tl.sum(a * b, axis=0)

    tl.store(C_ptr + offs_m * stride_cm + 0 * stride_cn, acc)


class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        assert A.is_cuda and B.is_cuda
        assert A.ndim == 2 and B.ndim == 2 and B.shape[1] == 1
        M, K = A.shape
        assert B.shape[0] == K
        # Match reference behavior for arbitrary dtype by using torch's promotion semantics where practical.
        # Fast path for floating dtypes commonly used in benchmark.
        out_dtype = torch.promote_types(A.dtype, B.dtype)
        if out_dtype not in (torch.float16, torch.bfloat16, torch.float32):
            A = A.float()
            B = B.float()
            out_dtype = torch.float32
        else:
            if A.dtype != out_dtype:
                A = A.to(out_dtype)
            if B.dtype != out_dtype:
                B = B.to(out_dtype)

        C = torch.empty((M, 1), device=A.device, dtype=torch.float32)
        grid = (M,)
        matvec_kernel[grid](
            A, B, C,
            M, K,
            A.stride(0), A.stride(1),
            B.stride(0), B.stride(1),
            C.stride(0), C.stride(1),
            BLOCK_K=1024,
        )
        return C.to(out_dtype)


M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]
