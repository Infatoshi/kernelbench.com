import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <algorithm>

__global__ void conv_fused_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ conv_bias,
    const float* __restrict__ add_bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_h, int in_w,
    int out_channels, int out_h, int out_w,
    int kernel_size, float scaling_factor,
    int pool_k, int p_out_h, int p_out_w) 
{
    // Output dimensions: (batch, out_channels, p_out_h, p_out_w)
    // Each thread computes one output element (after pooling)
    int pw = blockIdx.x * blockDim.x + threadIdx.x;
    int ph = blockIdx.y * blockDim.y + threadIdx.y;
    int oc_b = blockIdx.z;
    int oc = oc_b % out_channels;
    int b = oc_b / out_channels;

    if (b < batch_size && oc < out_channels && ph < p_out_h && pw < p_out_w) {
        float max_val = -1e38f;
        float c_bias = conv_bias[oc];
        float a_bias = add_bias[oc];

        for (int kh_p = 0; kh_p < pool_k; ++kh_p) {
            int oh = ph * pool_k + kh_p;
            for (int kw_p = 0; kw_p < pool_k; ++kw_p) {
                int ow = pw * pool_k + kw_p;

                float sum = c_bias;
                for (int ic = 0; ic < in_channels; ++ic) {
                    for (int kh = 0; kh < kernel_size; ++kh) {
                        int ih = oh + kh;
                        for (int kw = 0; kw < kernel_size; ++kw) {
                            int iw = ow + kw;
                            float val = input[((b * in_channels + ic) * in_h + ih) * in_w + iw];
                            float w = weight[((oc * in_channels + ic) * kernel_size + kh) * kernel_size + kw];
                            sum += val * w;
                        }
                    }
                }

                float res = tanhf(sum) * scaling_factor + a_bias;
                if (res > max_val) max_val = res;
            }
        }
        output[((b * out_channels + oc) * p_out_h + ph) * p_out_w + pw] = max_val;
    }
}

torch::Tensor fused_op(
    torch::Tensor input, 
    torch::Tensor weight, 
    torch::Tensor conv_bias, 
    float scaling_factor, 
    torch::Tensor add_bias, 
    int pool_k) 
{
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_channels = weight.size(0);
    int k_size = weight.size(2);
    int out_h = in_h - k_size + 1;
    int out_w = in_w - k_size + 1;
    int p_out_h = out_h / pool_k;
    int p_out_w = out_w / pool_k;

    auto output = torch::empty({batch_size, out_channels, p_out_h, p_out_w}, input.options());

    dim3 block(16, 16);
    dim3 grid((p_out_w + 15) / 16, (p_out_h + 15) / 16, batch_size * out_channels);

    conv_fused_kernel<<<grid, block>>>(
        input.contiguous().data_ptr<float>(),
        weight.contiguous().data_ptr<float>(),
        conv_bias.contiguous().data_ptr<float>(),
        add_bias.contiguous().data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_channels, in_h, in_w,
        out_channels, out_h, out_w,
        k_size, scaling_factor,
        pool_k, p_out_h, p_out_w
    );

    return output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='torch::Tensor fused_op(torch::Tensor, torch::Tensor, torch::Tensor, float, torch::Tensor, int);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = float(scaling_factor)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        return ext.fused_op(x, self.conv.weight, self.conv.bias, self.scaling_factor, self.bias.view(-1), self.pool_kernel_size)

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0, (16, 1, 1), 2]
