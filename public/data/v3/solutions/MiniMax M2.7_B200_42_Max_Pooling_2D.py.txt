import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Highly optimized max pool 2d for small kernels (2x2, 3x3, etc.)
template <typename scalar_t>
__global__ void max_pool2d_v2_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int stride,
    int padding,
    int dilation,
    int ksize_h,
    int ksize_w
) {
    // Grid-stride loop for maximum occupancy
    int total_threads = gridDim.x * blockDim.x;
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    
    #pragma unroll 4
    for (int iter = 0; iter < 4; ++iter) {
        int out_idx = tid + iter * total_threads;
        
        if (out_idx >= batch_size * channels * out_height * out_width) break;
        
        int remainder = out_idx;
        
        // Decode output coordinates
        int w_out = remainder % out_width;
        remainder /= out_width;
        int h_out = remainder % out_height;
        remainder /= out_height;
        int c = remainder % channels;
        int b = remainder / channels;
        
        // Calculate corresponding input start position
        int h_start = h_out * stride - padding;
        int w_start = w_out * stride - padding;
        
        // Load first valid value
        scalar_t max_val;
        bool valid = false;
        
        // Special unrolled path for 2x2 kernel
        if (ksize_h == 2 && ksize_w == 2) {
            int base_idx = b * channels * in_height * in_width 
                         + c * in_height * in_width 
                         + h_start * in_width 
                         + w_start;
            
            // Load all 4 values at once if aligned
            scalar_t v0, v1, v2, v3;
            int h0 = h_start;
            int h1 = h_start + dilation;
            int w0 = w_start;
            int w1 = w_start + dilation;
            
            // Clamp and load with bounds checking
            v0 = (h0 >= 0 && h0 < in_height && w0 >= 0 && w0 < in_width) 
                 ? input[base_idx] : -std::numeric_limits<scalar_t>::max();
            
            v1 = (h0 >= 0 && h0 < in_height && w1 >= 0 && w1 < in_width) 
                 ? input[base_idx + dilation] : -std::numeric_limits<scalar_t>::max();
            
            int base_idx2 = base_idx + dilation * in_width;
            v2 = (h1 >= 0 && h1 < in_height && w0 >= 0 && w0 < in_width) 
                 ? input[base_idx2] : -std::numeric_limits<scalar_t>::max();
            
            v3 = (h1 >= 0 && h1 < in_height && w1 >= 0 && w1 < in_width) 
                 ? input[base_idx2 + dilation] : -std::numeric_limits<scalar_t>::max();
            
            max_val = max(max(v0, v1), max(v2, v3));
            valid = true;
        } else {
            // General kernel path
            for (int kh = 0; kh < ksize_h; ++kh) {
                int h_in = h_start + kh * dilation;
                if (h_in < 0 || h_in >= in_height) continue;
                
                for (int kw = 0; kw < ksize_w; ++kw) {
                    int w_in = w_start + kw * dilation;
                    if (w_in < 0 || w_in >= in_width) continue;
                    
                    int in_idx = b * channels * in_height * in_width 
                               + c * in_height * in_width 
                               + h_in * in_width 
                               + w_in;
                    
                    scalar_t val = input[in_idx];
                    
                    if (!valid || val > max_val) {
                        max_val = val;
                        valid = true;
                    }
                }
            }
        }
        
        if (valid) {
            output[out_idx] = max_val;
        }
    }
}

// Warp-coalesced version - each warp cooperates on reduction
template <typename scalar_t>
__global__ void max_pool2d_warp_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int stride,
    int padding,
    int dilation,
    int ksize_h,
    int ksize_w
) {
    // Each warp (32 threads) processes one output
    int warp_id = blockIdx.x * (blockDim.x / 32) + threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    
    int outputs_per_batch = channels * out_height * out_width;
    int total_warps_per_batch = outputs_per_batch;
    
    for (int out_base = warp_id; out_base < batch_size * total_warps_per_batch; 
         out_base += gridDim.x * (blockDim.x / 32)) {
        
        int b = out_base / outputs_per_batch;
        int out_idx_in_batch = out_base % outputs_per_batch;
        
        int remainder = out_idx_in_batch;
        int w_out = remainder % out_width;
        remainder /= out_width;
        int h_out = remainder % out_height;
        int c = remainder / out_height;
        
        if (c >= channels) continue;
        
        int h_start = h_out * stride - padding;
        int w_start = w_out * stride - padding;
        
        // Each lane processes kernel positions with stride 32
        int max_pos = ksize_h * ksize_w;
        int pos = lane_id;
        
        scalar_t my_max = -std::numeric_limits<scalar_t>::max();
        
        while (pos < max_pos) {
            int kh = pos / ksize_w;
            int kw = pos % ksize_w;
            
            int h_in = h_start + kh * dilation;
            int w_in = w_start + kw * dilation;
            
            if (h_in >= 0 && h_in < in_height && w_in >= 0 && w_in < in_width) {
                int in_idx = b * channels * in_height * in_width 
                           + c * in_height * in_width 
                           + h_in * in_width 
                           + w_in;
                my_max = max(my_max, input[in_idx]);
            }
            
            pos += 32;
        }
        
        // Warp reduction using shuffle
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            scalar_t other = __shfl_down_sync(0xffffffff, my_max, offset);
            my_max = max(my_max, other);
        }
        
        // Lane 0 writes result
        if (lane_id == 0) {
            int final_out = b * channels * out_height * out_width 
                          + c * out_height * out_width 
                          + h_out * out_width 
                          + w_out;
            output[final_out] = my_max;
        }
    }
}

// Specialized 2x2 kernel with complete unrolling
template <typename scalar_t>
__global__ void max_pool2d_2x2_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int batch_size,
    int channels,
    int in_height,
    int in_width,
    int out_height,
    int out_width,
    int stride,
    int padding,
    int dilation
) {
    int total_threads = gridDim.x * blockDim.x;
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    
    for (int out_idx = tid; out_idx < batch_size * channels * out_height * out_width; 
         out_idx += total_threads) {
        
        int remainder = out_idx;
        int w_out = remainder % out_width;
        remainder /= out_width;
        int h_out = remainder % out_height;
        remainder /= out_height;
        int c = remainder % channels;
        int b = remainder / channels;
        
        int h_start = h_out * stride - padding;
        int w_start = w_out * stride - padding;
        
        int base_idx = b * channels * in_height * in_width 
                     + c * in_height * in_width 
                     + h_start * in_width 
                     + w_start;
        
        int h0 = h_start;
        int h1 = h_start + dilation;
        int w0 = w_start;
        int w1 = w_start + dilation;
        
        bool h0_valid = h0 >= 0 && h0 < in_height;
        bool h1_valid = h1 >= 0 && h1 < in_height;
        bool w0_valid = w0 >= 0 && w0 < in_width;
        bool w1_valid = w1 >= 0 && w1 < in_width;
        
        scalar_t v0 = (h0_valid && w0_valid) ? input[base_idx] : -std::numeric_limits<scalar_t>::max();
        scalar_t v1 = (h0_valid && w1_valid) ? input[base_idx + dilation] : -std::numeric_limits<scalar_t>::max();
        scalar_t v2 = (h1_valid && w0_valid) ? input[base_idx + dilation * in_width] : -std::numeric_limits<scalar_t>::max();
        scalar_t v3 = (h1_valid && w1_valid) ? input[base_idx + dilation * (in_width + 1)] : -std::numeric_limits<scalar_t>::max();
        
        output[out_idx] = max(max(v0, v1), max(v2, v3));
    }
}

torch::Tensor max_pool2d_forward_opt(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    int ksize_h = kernel_size;
    int ksize_w = kernel_size;
    
    int batch_size = input.size(0);
    int channels = input.size(1);
    int in_height = input.size(2);
    int in_width = input.size(3);
    
    int out_height = (in_height + 2 * padding - dilation * (ksize_h - 1) - 1) / stride + 1;
    int out_width = (in_width + 2 * padding - dilation * (ksize_w - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                               input.options());
    
    int total_out = batch_size * channels * out_height * out_width;
    
    // Use specialized kernel for 2x2
    if (kernel_size == 2 && stride == 2) {
        int threads = 256;
        int blocks = (total_out + threads - 1) / threads;
        
        AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "max_pool2d_forward_opt", ([&] {
            max_pool2d_2x2_kernel<scalar_t><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, in_height, in_width,
                out_height, out_width,
                stride, padding, dilation
            );
        }));
    } else {
        // General kernel
        int threads = 256;
        int blocks = (total_out + threads - 1) / threads;
        
        AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "max_pool2d_forward_opt", ([&] {
            max_pool2d_v2_kernel<scalar_t><<<blocks, threads>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, in_height, in_width,
                out_height, out_width,
                stride, padding, dilation,
                ksize_h, ksize_w
            );
        }));
    }
    
    return output;
}
'''

ext = load_inline(
    name='maxpool2d_fast_ext',
    cpp_sources=r'''
torch::Tensor max_pool2d_forward_opt(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
);
''',
    cuda_sources=cuda_source,
    functions=['max_pool2d_forward_opt'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Max Pooling 2D with specialized kernels.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.max_pool2d_forward_opt(
            x, 
            self.kernel_size, 
            self.stride, 
            self.padding, 
            self.dilation
        )

def get_inputs():
    return [torch.randn(16, 32, 512, 512, device='cuda')]

def get_init_inputs():
    return [2, 2, 1, 3]  # kernel_size, stride, padding, dilation
