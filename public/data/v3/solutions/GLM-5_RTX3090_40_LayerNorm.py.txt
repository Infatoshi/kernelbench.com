import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>

constexpr int WARP_SIZE = 32;
constexpr int BLOCK_SIZE = 512;

// Warp-level reduction using shuffles
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = WARP_SIZE >> 1; mask > 0; mask >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, mask);
    }
    return val;
}

// LayerNorm kernel - each block handles one sample (batch element)
__global__ void layernorm_kernel_fp32(
    const float* __restrict__ input,
    float* __restrict__ output,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    const int normalized_size,
    const float eps
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    const int lane_id = tid & (WARP_SIZE - 1);
    const int warp_id = tid / WARP_SIZE;
    const int num_warps = blockDim.x / WARP_SIZE;
    
    const float* x = input + batch_idx * normalized_size;
    float* y = output + batch_idx * normalized_size;
    
    __shared__ float s_mean;
    __shared__ float s_rstd;
    __shared__ float s_sum_warp[16];
    __shared__ float s_sum_sq_warp[16];
    
    // Compute mean and variance
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = tid; i < normalized_size; i += blockDim.x) {
        float val = x[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = warp_reduce_sum(local_sum);
    local_sum_sq = warp_reduce_sum(local_sum_sq);
    
    if (lane_id == 0) {
        s_sum_warp[warp_id] = local_sum;
        s_sum_sq_warp[warp_id] = local_sum_sq;
    }
    __syncthreads();
    
    float total_sum = 0.0f;
    float total_sum_sq = 0.0f;
    if (warp_id == 0) {
        if (lane_id < num_warps) {
            total_sum = s_sum_warp[lane_id];
            total_sum_sq = s_sum_sq_warp[lane_id];
        }
        total_sum = warp_reduce_sum(total_sum);
        total_sum_sq = warp_reduce_sum(total_sum_sq);
        
        if (tid == 0) {
            s_mean = total_sum / normalized_size;
            float variance = total_sum_sq / normalized_size - s_mean * s_mean;
            s_rstd = rsqrtf(variance + eps);
        }
    }
    __syncthreads();
    
    const float mean = s_mean;
    const float rstd = s_rstd;
    
    for (int i = tid; i < normalized_size; i += blockDim.x) {
        float val = x[i];
        y[i] = (val - mean) * rstd * gamma[i] + beta[i];
    }
}

// PyTorch interface
torch::Tensor layernorm_forward(
    torch::Tensor input,
    torch::Tensor gamma,
    torch::Tensor beta,
    double eps
) {
    auto output = torch::empty_like(input);
    
    int normalized_size = static_cast<int>(gamma.numel());
    int batch_size = static_cast<int>(input.numel() / normalized_size);
    
    auto stream = at::cuda::getCurrentCUDAStream();
    
    int threads = BLOCK_SIZE;
    int blocks = batch_size;
    
    if (input.scalar_type() == at::ScalarType::Float) {
        layernorm_kernel_fp32<<<blocks, threads, 0, stream>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            gamma.data_ptr<float>(),
            beta.data_ptr<float>(),
            normalized_size,
            static_cast<float>(eps)
        );
    } else if (input.scalar_type() == at::ScalarType::Half) {
        layernorm_kernel_fp16<<<blocks, threads, 0, stream>>>(
            reinterpret_cast<const half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<half*>(output.data_ptr<at::Half>()),
            gamma.data_ptr<float>(),
            beta.data_ptr<float>(),
            normalized_size,
            static_cast<float>(eps)
        );
    } else if (input.scalar_type() == at::ScalarType::BFloat16) {
        layernorm_kernel_bf16<<<blocks, threads, 0, stream>>>(
            reinterpret_cast<const nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            gamma.data_ptr<float>(),
            beta.data_ptr<float>(),
            normalized_size,
            static_cast<float>(eps)
        );
    } else {
        AT_ERROR("Unsupported scalar type for layernorm");
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor gamma, torch::Tensor beta, double eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        weight = self.weight.float()
        bias = self.bias.float()
        return ext.layernorm_forward(x, weight, bias, self.eps)

def get_inputs():
    return [torch.randn(16, 64, 256, 256, device='cuda')]

def get_init_inputs():
    return [(64, 256, 256)]
