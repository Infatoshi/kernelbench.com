import torch

from torch.utils.cpp_extension import load_inline

# Optimized CUDA implementation for LayerNorm
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// Warp-level sum reduction using shuffle
template<typename T>
__device__ __forceinline__ T warp_reduce_sum(T val) {
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Warp-level square sum reduction
template<typename T>
__device__ __forceinline__ T warp_reduce_sq(T val) {
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Block-level reduction using shared memory
template<typename T>
__device__ __forceinline__ T block_reduce_sum(T* sdata, T val, int tid) {
    // Store to shared memory
    sdata[tid] = val;
    __syncthreads();

    // Tree reduction in shared memory
    for (unsigned int s = blockDim.x / 2; s > 32; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    // Warp reduction for last 32 threads
    if (tid < 32) {
        if (blockDim.x >= 64) sdata[tid] += sdata[tid + 32];
        __syncwarp();
        val = warp_reduce_sum(sdata[tid]);
    }
    return val;
}

// Fused LayerNorm kernel - computes mean, variance, and output in one pass
template<typename T>
__global__ void layernorm_fused_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    const T* __restrict__ gamma,
    const T* __restrict__ beta,
    const T epsilon,
    const int batch_size,
    const int norm_size  // product of normalized dimensions
) {
    extern __shared__ float sdata[];
    
    const int tid = threadIdx.x;
    const int bid = blockIdx.x;
    const int batch_idx = bid;  // One block per batch item for simplicity
    
    if (batch_idx >= batch_size) return;
    
    // Calculate base offset for this batch
    const int base_offset = batch_idx * norm_size;
    const T* x_ptr = x + base_offset;
    const T* y_ptr = y + base_offset;
    
    // First pass: compute mean
    float thread_sum = 0.0f;
    for (int i = tid; i < norm_size; i += blockDim.x) {
        thread_sum += static_cast<float>(x_ptr[i]);
    }
    
    // Block reduction for sum
    float block_sum = block_reduce_sum(sdata, thread_sum, tid);
    
    // Broadcast mean to all threads
    const float mean = block_sum / static_cast<float>(norm_size);
    const float mean_shuffled = __shfl_sync(0xffffffff, mean, 0);
    
    // Second pass: compute variance
    float thread_sq_sum = 0.0f;
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float diff = static_cast<float>(x_ptr[i]) - mean_shuffled;
        thread_sq_sum += diff * diff;
    }
    
    // Block reduction for squared sum
    float block_sq_sum = block_reduce_sum(sdata, thread_sq_sum, tid);
    
    // Broadcast variance to all threads
    const float variance = block_sq_sum / static_cast<float>(norm_size);
    const float inv_std = rsqrtf(variance + epsilon);
    const float inv_std_shuffled = __shfl_sync(0xffffffff, inv_std, 0);
    
    // Third pass: normalize and write output
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float x_val = static_cast<float>(x_ptr[i]);
        float normalized = (x_val - mean_shuffled) * inv_std_shuffled;
        
        // Apply gamma and beta
        if (gamma != nullptr && beta != nullptr) {
            // Gamma and beta are for the normalized shape (flattened)
            // For this case, gamma=1, beta=0 (learned params)
            const int norm_idx = i % norm_size;
            normalized = normalized * static_cast<float>(gamma[norm_idx]) + 
                         static_cast<float>(beta[norm_idx]);
        }
        
        y_ptr[i] = static_cast<T>(normalized);
    }
}

// Optimized kernel with multiple blocks per batch for better parallelism
template<typename T>
__global__ void layernorm_fused_kernel_v2(
    const T* __restrict__ x,
    T* __restrict__ y,
    const T* __restrict__ gamma,
    const T* __restrict__ beta,
    const T epsilon,
    const int batch_size,
    const int norm_size,
    const int num_blocks_per_batch
) {
    extern __shared__ float sdata[];
    
    const int tid = threadIdx.x;
    const int block_id = blockIdx.x;
    
    // Calculate batch and element indices
    const int batch_idx = block_id / num_blocks_per_batch;
    const int block_elem_idx = block_id % num_blocks_per_batch;
    
    if (batch_idx >= batch_size) return;
    
    // Each block processes a portion of elements
    const int elements_per_block = (norm_size + num_blocks_per_batch - 1) / num_blocks_per_batch;
    const int start_elem = block_elem_idx * elements_per_block;
    const int end_elem = min(start_elem + elements_per_block, norm_size);
    
    const int base_offset = batch_idx * norm_size;
    const T* x_ptr = x + base_offset;
    const T* y_ptr = y + base_offset;
    
    // First pass: compute local sum
    float thread_sum = 0.0f;
    int count = 0;
    for (int i = start_elem + tid; i < end_elem; i += blockDim.x) {
        thread_sum += static_cast<float>(x_ptr[i]);
        count++;
    }
    
    // Block reduction for sum
    float block_sum = block_reduce_sum(sdata, thread_sum, tid);
    
    // Use atomicAdd to accumulate across blocks for mean
    __shared__ float mean_shared;
    __shared__ float sq_sum_shared;
    
    if (tid == 0) {
        mean_shared = block_sum;
        sq_sum_shared = 0.0f;
    }
    __syncthreads();
    
    // Store local sum to shared for broadcast
    sdata[tid] = thread_sum;
    __syncthreads();
    
    // Compute mean from local sums
    if (tid == 0) {
        float total_sum = 0.0f;
        for (int i = 0; i < blockDim.x && i < (end_elem - start_elem); i++) {
            total_sum += sdata[i];
        }
        mean_shared = total_sum / static_cast<float>(norm_size);
    }
    __syncthreads();
    
    const float mean = mean_shared;
    const float mean_shuffled = __shfl_sync(0xffffffff, mean, 0);
    
    // Second pass: compute local variance
    float thread_sq_sum = 0.0f;
    for (int i = start_elem + tid; i < end_elem; i += blockDim.x) {
        float diff = static_cast<float>(x_ptr[i]) - mean_shuffled;
        thread_sq_sum += diff * diff;
    }
    
    // Block reduction for squared sum
    float block_sq_sum = block_reduce_sum(sdata, thread_sq_sum, tid);
    
    if (tid == 0) {
        sq_sum_shared = block_sq_sum;
    }
    __syncthreads();
    
    // Accumulate squared sums
    sdata[tid] = thread_sq_sum;
    __syncthreads();
    
    if (tid == 0) {
        float total_sq_sum = 0.0f;
        for (int i = 0; i < blockDim.x && i < (end_elem - start_elem); i++) {
            total_sq_sum += sdata[i];
        }
        sq_sum_shared = total_sq_sum;
    }
    __syncthreads();
    
    const float variance = sq_sum_shared / static_cast<float>(norm_size);
    const float inv_std = rsqrtf(variance + epsilon);
    const float inv_std_shuffled = __shfl_sync(0xffffffff, inv_std, 0);
    
    // Third pass: normalize and write output
    for (int i = start_elem + tid; i < end_elem; i += blockDim.x) {
        float x_val = static_cast<float>(x_ptr[i]);
        float normalized = (x_val - mean_shuffled) * inv_std_shuffled;
        
        if (gamma != nullptr && beta != nullptr) {
            normalized = normalized * static_cast<float>(gamma[i]) + 
                         static_cast<float>(beta[i]);
        }
        
        y_ptr[i] = static_cast<T>(normalized);
    }
}

// Simple efficient kernel - one block per batch
template<typename T>
__global__ void layernorm_simple_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    const T* __restrict__ gamma,
    const T* __restrict__ beta,
    const T epsilon,
    const int batch_size,
    const int norm_size
) {
    extern __shared__ float sdata[];
    
    const int tid = threadIdx.x;
    const int batch_idx = blockIdx.x;
    
    if (batch_idx >= batch_size) return;
    
    const int base_offset = batch_idx * norm_size;
    const T* x_batch = x + base_offset;
    T* y_batch = y + base_offset;
    
    // === First pass: Compute mean ===
    float thread_sum = 0.0f;
    for (int i = tid; i < norm_size; i += blockDim.x) {
        thread_sum += static_cast<float>(x_batch[i]);
    }
    
    // Reduction within block
    __syncthreads();
    sdata[tid] = thread_sum;
    __syncthreads();
    
    if (tid < 128) {
        if (tid + 128 < blockDim.x) sdata[tid] += sdata[tid + 128];
    }
    __syncthreads();
    if (tid < 64) {
        if (tid + 64 < blockDim.x) sdata[tid] += sdata[tid + 64];
        sdata[tid] += sdata[tid + 32];
        sdata[tid] += sdata[tid + 16];
        sdata[tid] += sdata[tid + 8];
        sdata[tid] += sdata[tid + 4];
        sdata[tid] += sdata[tid + 2];
        sdata[tid] += sdata[tid + 1];
    }
    __syncthreads();
    
    const float mean = sdata[0] / static_cast<float>(norm_size);
    
    // === Second pass: Compute variance ===
    float thread_sq_sum = 0.0f;
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float diff = static_cast<float>(x_batch[i]) - mean;
        thread_sq_sum += diff * diff;
    }
    
    __syncthreads();
    sdata[tid] = thread_sq_sum;
    __syncthreads();
    
    if (tid < 128) {
        if (tid + 128 < blockDim.x) sdata[tid] += sdata[tid + 128];
    }
    __syncthreads();
    if (tid < 64) {
        if (tid + 64 < blockDim.x) sdata[tid] += sdata[tid + 64];
        sdata[tid] += sdata[tid + 32];
        sdata[tid] += sdata[tid + 16];
        sdata[tid] += sdata[tid + 8];
        sdata[tid] += sdata[tid + 4];
        sdata[tid] += sdata[tid + 2];
        sdata[tid] += sdata[tid + 1];
    }
    __syncthreads();
    
    const float variance = sdata[0] / static_cast<float>(norm_size);
    const float inv_std = rsqrtf(variance + epsilon);
    
    // === Third pass: Normalize and write ===
    for (int i = tid; i < norm_size; i += blockDim.x) {
        float x_val = static_cast<float>(x_batch[i]);
        float normalized = (x_val - mean) * inv_std;
        
        if (gamma != nullptr && beta != nullptr) {
            normalized = normalized * static_cast<float>(gamma[i]) + 
                         static_cast<float>(beta[i]);
        }
        
        y_batch[i] = static_cast<T>(normalized);
    }
}

// Wrapper functions
torch::Tensor layernorm_op(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float epsilon) {
    auto y = torch::empty_like(x);
    
    const int batch_size = x.size(0);
    const int norm_size = x.numel() / batch_size;
    
    const int threads = 256;
    const int blocks = batch_size;  // One block per batch item
    const int smem = threads * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "layernorm_cuda", ([&] {
        layernorm_simple_kernel<scalar_t><<<blocks, threads, smem>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            gamma.defined() ? gamma.data_ptr<scalar_t>() : nullptr,
            beta.defined() ? beta.data_ptr<scalar_t>() : nullptr,
            static_cast<scalar_t>(epsilon),
            batch_size,
            norm_size
        );
    }));
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor layernorm_op(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float epsilon);
'''

# Load the CUDA extension
ext = load_inline(
    name='layernorm_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_op'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized Layer Normalization using custom CUDA kernel.
    """
    def __init__(self, normalized_shape: tuple, eps: float = 1e-5):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        self.eps = eps
        
        # Learnable parameters
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Flatten the normalized dimensions for the kernel
        # Input: (batch, feat, dim1, dim2) -> we normalize over last 3 dims
        # For LayerNorm: (batch, *) where * = feat * dim1 * dim2
        
        # Our kernel expects (batch, norm_size) layout
        # Reshape: (batch, feat, dim1, dim2) -> (batch, feat*dim1*dim2)
        batch_size = x.size(0)
        
        # For the kernel, we need to flatten the normalized dims
        norm_size = 1
        for i in range(1, x.dim()):
            norm_size *= x.size(i)
        
        # Reshape to (batch, norm_size)
        x_flat = x.view(batch_size, norm_size)
        
        # Call our custom kernel
        y_flat = ext.layernorm_op(x_flat, self.weight, self.bias, self.eps)
        
        # Reshape back to original shape
        y = y_flat.view(x.shape)
        
        return y
