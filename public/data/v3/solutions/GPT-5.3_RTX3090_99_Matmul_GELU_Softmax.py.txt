import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <math_constants.h>

__device__ inline float gelu(float x){
    const float k0 = 0.7978845608f; // sqrt(2/pi)
    const float k1 = 0.044715f;
    float y = k0 * (x + k1 * x * x * x);
    return 0.5f * x * (1.0f + tanhf(y));
}

__global__ void fused_kernel(const float* __restrict__ x,
                             const float* __restrict__ w,
                             const float* __restrict__ b,
                             float* __restrict__ y,
                             int B, int IN, int OUT) {
    int row = blockIdx.x;
    int tid = threadIdx.x;

    extern __shared__ float smem[];
    float* vals = smem;
    float* red = smem + OUT;

    const float* xrow = x + row * IN;

    float local_max = -1e30f;

    for(int j = tid; j < OUT; j += blockDim.x){
        const float* wrow = w + j * IN;
        float acc = 0.f;
        for(int k=0;k<IN;k++){
            acc += xrow[k] * wrow[k];
        }
        acc += b[j];
        acc = gelu(acc);
        vals[j] = acc;
        if(acc > local_max) local_max = acc;
    }

    __syncthreads();

    __shared__ float red[256];
    __syncthreads();
    red[tid] = local_max;
    __syncthreads();

    for(int s=blockDim.x/2; s>0; s>>=1){
        if(tid < s) red[tid] = fmaxf(red[tid], red[tid+s]);
        __syncthreads();
    }
    __syncthreads();

    float m = red[0];

    float local_sum = 0.f;
    for(int j = tid; j < OUT; j += blockDim.x){
        float v = __expf(vals[j] - m);
        vals[j] = v;
        local_sum += v;
    }

    red[tid] = local_sum;
    __syncthreads();

    for(int s=blockDim.x/2; s>0; s>>=1){
        if(tid < s) red[tid] += red[tid+s];
        __syncthreads();
    }

    float denom = red[0];

    for(int j = tid; j < OUT; j += blockDim.x){
        y[row*OUT + j] = vals[j] / denom;
    }
}


torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b){
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = w.size(0);

    auto y = torch::empty({B, OUT}, x.options());

    int threads = 256;
    size_t smem = OUT * sizeof(float);

    fused_kernel<<<B, threads, smem>>>(x.data_ptr<float>(), w.data_ptr<float>(), b.data_ptr<float>(), y.data_ptr<float>(), B, IN, OUT);

    return y;
}
'''

ext = load_inline(
    name='fused_matmul_gelu_softmax',
    cpp_sources='torch::Tensor fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b);',
    cuda_sources=cuda_src,
    functions=['fused_forward'],
    extra_cuda_cflags=['-O3','-arch=sm_86'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        x = x.contiguous()
        return ext.fused_forward(x, self.weight.contiguous(), self.bias.contiguous())


def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]


def get_init_inputs():
    return [4096, 4096]
