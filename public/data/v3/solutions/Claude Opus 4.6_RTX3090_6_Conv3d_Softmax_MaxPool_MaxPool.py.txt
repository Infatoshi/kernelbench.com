import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>
#include <math.h>

// Fused softmax (along channel dim=1) + maxpool3d(4, stride=4)
// Input: (B, C, D, H, W) in float
// Output: (B, C, OD, OH, OW) in float

// Strategy: Each block handles multiple output spatial positions.
// We process one (b, od, oh, ow) per warp, with C threads per position.
// For C=16, we have 2 output positions per warp (32/16=2), 
// and multiple warps per block for better occupancy.

// Alternative: each thread handles one output element (b, c, od, oh, ow)
// and loops over the 64 pool positions, computing softmax inline.
// For C=16, we can use warp-level primitives since 16 < 32.

__global__ void fused_softmax_maxpool4_v2(
    const float* __restrict__ input,  // (B, C, D, H, W)
    float* __restrict__ output,       // (B, C, OD, OH, OW)
    int B, int C, int D, int H, int W,
    int OD, int OH, int OW
) {
    // Grid: blocks cover output elements
    // Each warp handles C consecutive channel outputs for one spatial position
    // If C <= 32, one warp handles one (b, od, oh, ow) with threads for channels
    
    int warp_id_global = (blockIdx.x * blockDim.x + threadIdx.x) / C;
    int c = (blockIdx.x * blockDim.x + threadIdx.x) % C;
    
    int total_spatial = B * OD * OH * OW;
    if (warp_id_global >= total_spatial) return;
    
    int idx = warp_id_global;
    int ow_idx = idx % OW; idx /= OW;
    int oh_idx = idx % OH; idx /= OH;
    int od_idx = idx % OD; idx /= OD;
    int b = idx;
    
    int d_start = od_idx * 4;
    int h_start = oh_idx * 4;
    int w_start = ow_idx * 4;
    
    int HW = H * W;
    int DHW = D * H * W;
    int base_b = b * C * DHW;
    
    // Lane within the C-group
    // Use warp shuffle for reductions across C channels
    // We need the lane_id within the C-group and the mask
    int lane_in_warp = threadIdx.x % 32;  // 0..31
    int c_group_start = (lane_in_warp / C) * C;  // start lane of our C-group in the warp
    
    // Create mask for our C-group within the warp
    unsigned int group_mask = ((1u << C) - 1u) << c_group_start;
    
    float max_pool_val = -FLT_MAX;
    
    // Iterate over 4x4x4 pool region
    for (int dd = 0; dd < 4; dd++) {
        int d_idx = d_start + dd;
        if (d_idx >= D) continue;
        for (int dh = 0; dh < 4; dh++) {
            int h_idx = h_start + dh;
            if (h_idx >= H) continue;
            for (int dw = 0; dw < 4; dw++) {
                int w_idx = w_start + dw;
                if (w_idx >= W) continue;
                
                int spatial_idx = d_idx * HW + h_idx * W + w_idx;
                float val = input[base_b + c * DHW + spatial_idx];
                
                // Softmax across C channels using warp shuffle
                // Step 1: find max across C channels in group
                float ch_max = val;
                for (int offset = C/2; offset > 0; offset /= 2) {
                    float other = __shfl_xor_sync(group_mask, ch_max, offset);
                    ch_max = fmaxf(ch_max, other);
                }
                // Broadcast max (all lanes in group now have the same ch_max)
                
                // Step 2: exp(val - max)
                float exp_val = __expf(val - ch_max);
                
                // Step 3: sum across channels
                float ch_sum = exp_val;
                for (int offset = C/2; offset > 0; offset /= 2) {
                    ch_sum += __shfl_xor_sync(group_mask, ch_sum, offset);
                }
                
                float softmax_val = exp_val / ch_sum;
                max_pool_val = fmaxf(max_pool_val, softmax_val);
            }
        }
    }
    
    // Write output
    int out_spatial = OD * OH * OW;
    int out_idx = b * C * out_spatial + c * out_spatial + od_idx * OH * OW + oh_idx * OW + ow_idx;
    output[out_idx] = max_pool_val;
}

torch::Tensor fused_softmax_maxpool4(torch::Tensor input) {
    auto sizes = input.sizes();
    int B = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];
    
    int OD = D / 4;
    int OH = H / 4;
    int OW = W / 4;
    
    auto output = torch::empty({B, C, OD, OH, OW}, input.options());
    
    int total_output_positions = B * OD * OH * OW;
    int total_threads = total_output_positions * C;
    
    // Choose block size as multiple of 32 for warp alignment
    // C must divide 32 for proper grouping (C=16 works: 2 groups per warp)
    int block_size = 256;
    int num_blocks = (total_threads + block_size - 1) / block_size;
    
    fused_softmax_maxpool4_v2<<<num_blocks, block_size>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        B, C, D, H, W, OD, OH, OW
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_softmax_maxpool4(torch::Tensor input);
'''

fused_ext = load_inline(
    name='fused_softmax_maxpool4',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_softmax_maxpool4'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = fused_ext.fused_softmax_maxpool4(x)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
depth, height, width = 16, 128, 128
kernel_size = 3
pool_kernel_size = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
