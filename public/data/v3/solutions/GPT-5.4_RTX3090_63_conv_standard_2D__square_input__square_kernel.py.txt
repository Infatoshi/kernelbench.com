import torch
import torch.nn as nn


OP_TYPE = "conv"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(Model, self).__init__()
        # Use channels_last to unlock cuDNN's fastest Tensor Core kernels on Ampere.
        self.conv2d = nn.Conv2d(
            in_channels,
            out_channels,
            (kernel_size, kernel_size),
            stride=stride,
            padding=padding,
            dilation=dilation,
            groups=groups,
            bias=bias,
        )
        self.conv2d = self.conv2d.to(memory_format=torch.channels_last)

        # Prefer fastest cuDNN algorithm search for fixed benchmark shapes.
        torch.backends.cudnn.benchmark = True
        torch.backends.cudnn.allow_tf32 = True
        torch.backends.cuda.matmul.allow_tf32 = True

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Reference benchmark uses NCHW contiguous input; convert once to NHWC/channels_last.
        if x.dim() == 4:
            x = x.contiguous(memory_format=torch.channels_last)
        return self.conv2d(x)


batch_size = 16
in_channels = 3
out_channels = 64
kernel_size = 3
width = 256
height = 256


def get_inputs():
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
