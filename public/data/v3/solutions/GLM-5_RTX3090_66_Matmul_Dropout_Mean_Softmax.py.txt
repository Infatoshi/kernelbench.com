import torch
import torch.nn as nn
import triton
import triton.language as tl
import random
import math

# Triton kernel for GEMM (matrix multiplication) with tensor cores
@triton.jit
def gemm_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, c_ptr,
    # Matrix dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Meta-parameters
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """Kernel for computing C = A x B^T (B is stored transposed)"""
    # Block indices
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Create pointers for the first blocks of A and B
    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = b_ptr + (offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn)

    # Iterate to compute a block of the C matrix
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Load the next block of A and B
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        b = tl.load(b_ptrs, mask=offs_k[:, None] < K - k * BLOCK_SIZE_K, other=0.0)
        # Compute dot product using tensor cores
        accumulator += tl.dot(a, b)
        # Advance the ptrs to the next K block
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_bk

    # Store the result
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, accumulator, mask=c_mask)


# Triton kernel for fused GEMM + bias + dropout + mean
@triton.jit
def fused_gemm_dropout_mean_kernel(
    # Pointers
    a_ptr, b_ptr, bias_ptr, mean_ptr,
    # Dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Dropout params
    dropout_p, scale, seed,
    # Meta-parameters
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    """Fused kernel for GEMM + bias + dropout + mean"""
    # Each block handles a column strip
    pid_n = tl.program_id(axis=0)
    
    # Column range for this block
    col_start = pid_n * BLOCK_SIZE_N
    col_offsets = col_start + tl.arange(0, BLOCK_SIZE_N)
    col_mask = col_offsets < N
    
    # Load bias for these columns
    bias = tl.load(bias_ptr + col_offsets, mask=col_mask, other=0.0).to(tl.float32)
    
    # Accumulator for mean
    mean_acc = tl.zeros([BLOCK_SIZE_N], dtype=tl.float32)
    
    # Process row strips
    for row_start in range(0, M, BLOCK_SIZE_M):
        row_offsets = row_start + tl.arange(0, BLOCK_SIZE_M)
        row_mask = row_offsets < M
        
        # GEMM for this tile: (BLOCK_SIZE_M, K) @ (K, BLOCK_SIZE_N)
        # We compute dot product with the full K dimension
        acc = tl.zeros([BLOCK_SIZE_M, BLOCK_SIZE_N], dtype=tl.float32)
        
        # Iterate over K
        for k_start in range(0, K, BLOCK_SIZE_K):
            k_offsets = k_start + tl.arange(0, BLOCK_SIZE_K)
            k_mask = k_offsets < K
            
            # Load A tile: (BLOCK_SIZE_M, BLOCK_SIZE_K)
            a_ptrs = a_ptr + row_offsets[:, None] * stride_am + k_offsets[None, :] * stride_ak
            a = tl.load(a_ptrs, mask=row_mask[:, None] & k_mask[None, :], other=0.0)
            
            # Load B tile: (BLOCK_SIZE_K, BLOCK_SIZE_N)
            b_ptrs = b_ptr + k_offsets[:, None] * stride_bk + col_offsets[None, :] * stride_bn
            b = tl.load(b_ptrs, mask=k_mask[:, None] & col_mask[None, :], other=0.0)
            
            # Accumulate
            acc += tl.dot(a, b)
        
        # Add bias
        acc = acc + bias[None, :]
        
        # Apply dropout and accumulate for mean
        # Generate random numbers for each element
        for row_idx in range(BLOCK_SIZE_M):
            if row_start + row_idx < M:
                row_vals = acc[row_idx, :]  # (BLOCK_SIZE_N,)
                
                # Random number for dropout
                rand_offset = (row_start + row_idx) * N + col_offsets
                rand_state = (seed + rand_offset) * 1103515245 + 12345
                rand_val = (rand_state & 0x7fffffff).to(tl.float32) / 2147483647.0
                
                # Apply dropout
                dropped = tl.where(rand_val >= dropout_p, row_vals * scale, 0.0)
                
                # Accumulate for mean
                mean_acc += dropped
    
    # Compute mean and store
    mean_val = mean_acc / M
    tl.store(mean_ptr + col_offsets, mean_val, mask=col_mask)


# Simpler fused kernel that processes entire columns
@triton.jit
def fused_col_gemm_dropout_mean_kernel(
    # Pointers
    a_ptr, b_ptr, bias_ptr, mean_ptr,
    # Dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    # Dropout params
    dropout_p, scale, seed,
    # Meta-parameters
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """Fused kernel where each block handles a set of columns"""
    pid = tl.program_id(axis=0)
    
    # Column range for this block
    col_start = pid * BLOCK_N
    col_offsets = col_start + tl.arange(0, BLOCK_N)
    col_mask = col_offsets < N
    
    # Load bias for these columns
    bias = tl.load(bias_ptr + col_offsets, mask=col_mask, other=0.0).to(tl.float32)
    
    # Initialize mean accumulator
    mean_val = tl.zeros([BLOCK_N], dtype=tl.float32)
    
    # Iterate over rows
    for row in range(M):
        # Compute dot product for this row across all K
        row_acc = tl.zeros([BLOCK_N], dtype=tl.float32)
        
        for k_start in range(0, K, BLOCK_K):
            k_offsets = k_start + tl.arange(0, BLOCK_K)
            k_mask = k_offsets < K
            
            # Load row of A: (BLOCK_K,)
            a_ptrs = a_ptr + row * stride_am + k_offsets * stride_ak
            a = tl.load(a_ptrs, mask=k_mask, other=0.0)
            
            # Load columns of B: (BLOCK_K, BLOCK_N)
            b_ptrs = b_ptr + k_offsets[:, None] * stride_bk + col_offsets[None, :] * stride_bn
            b = tl.load(b_ptrs, mask=k_mask[:, None] & col_mask[None, :], other=0.0)
            
            # Accumulate: (BLOCK_K,) @ (BLOCK_K, BLOCK_N) -> (BLOCK_N,)
            row_acc += tl.sum(a[:, None] * b, axis=0)
        
        # Add bias
        row_acc = row_acc + bias
        
        # Apply dropout
        rand_offset = row * N + col_offsets
        rand_state = (seed + rand_offset) * 1103515245 + 12345
        rand_val = (rand_state & 0x7fffffff).to(tl.float32) / 2147483647.0
        dropped = tl.where(rand_val >= dropout_p, row_acc * scale, 0.0)
        
        # Accumulate for mean
        mean_val += dropped
    
    # Final mean
    mean_val = mean_val / M
    
    # Store
    tl.store(mean_ptr + col_offsets, mean_val, mask=col_mask)


# Triton kernel for softmax
@triton.jit
def softmax_kernel(
    # Pointers
    input_ptr, output_ptr,
    # Dimensions
    N,
    # Meta-parameters
    BLOCK_SIZE: tl.constexpr,
):
    """Kernel for softmax along dimension 1"""
    # Load entire row (since N=4096 fits in one block)
    offsets = tl.arange(0, BLOCK_SIZE)
    mask = offsets < N
    
    # Load input
    x = tl.load(input_ptr + offsets, mask=mask, other=-float('inf'))
    
    # Compute max for numerical stability
    x_max = tl.max(x, axis=0)
    
    # Compute exp(x - max)
    x_exp = tl.exp(x - x_max)
    
    # Compute sum
    x_sum = tl.sum(x_exp, axis=0)
    
    # Compute softmax
    softmax_out = x_exp / x_sum
    
    # Store output
    tl.store(output_ptr + offsets, softmax_out, mask=mask)


class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.dropout_p = dropout_p
        
        # Create weights like nn.Linear (weight shape: out_features x in_features)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize weights like nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
        
        self.seed = 42

    def forward(self, x):
        # x: (M, K) where K = in_features
        # weight: (N, K) where N = out_features
        # We compute x @ weight^T + bias
        
        # Convert to float32 for computation
        x = x.to(torch.float32)
        w = self.weight.to(torch.float32)
        
        # GEMM: (M, K) @ (K, N) = (M, N)
        M, K = x.shape
        N = w.shape[0]
        
        # Allocate output
        gemm_out = torch.empty((M, N), device=x.device, dtype=torch.float32)
        
        # Optimized block sizes for RTX 3090
        BLOCK_SIZE_M = 128
        BLOCK_SIZE_N = 64
        BLOCK_SIZE_K = 32
        GROUP_SIZE_M = 8
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        
        gemm_kernel[grid](
            x, w, gemm_out,
            M, N, K,
            x.stride(0), x.stride(1),
            w.stride(1), w.stride(0),  # B^T: swap strides
            gemm_out.stride(0), gemm_out.stride(1),
            BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
        )
        
        # Add bias
        gemm_out = gemm_out + self.bias.to(torch.float32)
        
        # Dropout + Mean using optimized kernel
        scale = 1.0 / (1.0 - self.dropout_p)
        mean_out = torch.empty((N,), device=x.device, dtype=torch.float32)
        
        # Process columns in blocks
        BLOCK_N = 64
        grid_mean = (triton.cdiv(N, BLOCK_N),)
        
        fused_col_gemm_dropout_mean_kernel[grid_mean](
            x, w, self.bias.to(torch.float32), mean_out,
            M, N, K,
            x.stride(0), x.stride(1),
            w.stride(1), w.stride(0),
            self.dropout_p, scale, self.seed,
            BLOCK_M=16, BLOCK_N=BLOCK_N, BLOCK_K=32,
        )
        
        # Softmax
        softmax_out = torch.empty_like(mean_out)
        BLOCK_SIZE_SOFTMAX = triton.next_power_of_2(N)
        
        softmax_kernel[(1,)](
            mean_out, softmax_out,
            N,
            BLOCK_SIZE=BLOCK_SIZE_SOFTMAX,
        )
        
        return softmax_out.view(1, N)


batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
