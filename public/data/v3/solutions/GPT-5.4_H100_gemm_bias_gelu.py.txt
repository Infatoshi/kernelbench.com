import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _bias_gelu_kernel(
    x_ptr,
    bias_ptr,
    y_ptr,
    n_cols,
    BLOCK: tl.constexpr,
):
    row = tl.program_id(0)
    pid_col = tl.program_id(1)
    offs = pid_col * BLOCK + tl.arange(0, BLOCK)
    mask = offs < n_cols
    x = tl.load(x_ptr + row * n_cols + offs, mask=mask, other=0.0).to(tl.float32)
    b = tl.load(bias_ptr + offs, mask=mask, other=0.0).to(tl.float32)
    z = x + b
    erf_u = tl.erf(z * 0.7071067811865475)
    gelu = 0.5 * z * (1.0 + erf_u)
    tl.store(y_ptr + row * n_cols + offs, gelu.to(tl.float16), mask=mask)


def fused_bias_gelu(x: torch.Tensor, bias: torch.Tensor) -> torch.Tensor:
    assert x.is_cuda and bias.is_cuda
    assert x.dtype == torch.float16 and bias.dtype == torch.float16
    m, n = x.shape
    y = torch.empty_like(x)
    grid = (m, triton.cdiv(n, 256))
    _bias_gelu_kernel[grid](x, bias, y, n, BLOCK=256)
    return y


class Model(nn.Module):
    def __init__(self, n: int = 4096):
        super().__init__()
        self.bias = nn.Parameter(torch.randn(n, dtype=torch.float16) * 0.02)

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        c = torch.ops.aten.mm(a.to(torch.float16), b.to(torch.float16))
        return fused_bias_gelu(c, self.bias)
