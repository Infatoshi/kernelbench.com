import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// FP16 specialized kernel
__global__ void fused_matmul_scale_residual_fp16_kernel(
    const at::Half* input,
    const at::Half* weight,
    const at::Half* bias,
    at::Half* output,
    int batch,
    int in_features,
    int out_features,
    float scale_factor
) {
    int row = blockIdx.y * 16 + threadIdx.y;
    int col = blockIdx.x * 16 + threadIdx.x;
    
    if (row < batch && col < out_features) {
        float acc = 0.0f;
        for (int k = 0; k < in_features; k++) {
            float in_val = __half2float(static_cast<__half>(input[row * in_features + k]));
            float w_val = __half2float(static_cast<__half>(weight[col * in_features + k]));
            acc += in_val * w_val;
        }
        acc += __half2float(static_cast<__half>(bias[col]));
        acc = acc * (1.0f + scale_factor);
        output[row * out_features + col] = static_cast<at::Half>(__float2half(acc));
    }
}

// BF16 specialized kernel
__global__ void fused_matmul_scale_residual_bf16_kernel(
    const __nv_bfloat16* input,
    const __nv_bfloat16* weight,
    const __nv_bfloat16* bias,
    __nv_bfloat16* output,
    int batch,
    int in_features,
    int out_features,
    float scale_factor
) {
    int row = blockIdx.y * 16 + threadIdx.y;
    int col = blockIdx.x * 16 + threadIdx.x;
    
    if (row < batch && col < out_features) {
        float acc = 0.0f;
        for (int k = 0; k < in_features; k++) {
            float in_val = __bfloat162float(input[row * in_features + k]);
            float w_val = __bfloat162float(weight[col * in_features + k]);
            acc += in_val * w_val;
        }
        acc += __bfloat162float(bias[col]);
        acc = acc * (1.0f + scale_factor);
        output[row * out_features + col] = __float2bfloat16(acc);
    }
}

// FP32 kernel with shared memory tiling
template<int BLOCK_SIZE = 16>
__global__ void fused_matmul_scale_residual_fp32_kernel(
    const float* input,
    const float* weight,
    const float* bias,
    float* output,
    int batch,
    int in_features,
    int out_features,
    float scale_factor
) {
    __shared__ float shared_input[BLOCK_SIZE][BLOCK_SIZE];
    __shared__ float shared_weight[BLOCK_SIZE][BLOCK_SIZE];
    
    int row = blockIdx.y * BLOCK_SIZE + threadIdx.y;
    int col = blockIdx.x * BLOCK_SIZE + threadIdx.x;
    
    float acc = 0.0f;
    
    for (int t = 0; t < (in_features + BLOCK_SIZE - 1) / BLOCK_SIZE; t++) {
        // Load input tile
        int in_col = t * BLOCK_SIZE + threadIdx.x;
        if (row < batch && in_col < in_features) {
            shared_input[threadIdx.y][threadIdx.x] = input[row * in_features + in_col];
        } else {
            shared_input[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        // Load weight tile
        int w_col = t * BLOCK_SIZE + threadIdx.y;
        if (col < out_features && w_col < in_features) {
            shared_weight[threadIdx.y][threadIdx.x] = weight[col * in_features + w_col];
        } else {
            shared_weight[threadIdx.y][threadIdx.x] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute dot product for this tile
        for (int k = 0; k < BLOCK_SIZE; k++) {
            acc += shared_input[threadIdx.y][k] * shared_weight[k][threadIdx.x];
        }
        
        __syncthreads();
    }
    
    // Add bias and apply scaling
    if (row < batch && col < out_features) {
        acc += bias[col];
        acc = acc * (1.0f + scale_factor);
        output[row * out_features + col] = acc;
    }
}

torch::Tensor fused_matmul_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor
) {
    int batch = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    auto output = torch::empty({batch, out_features}, input.options());
    
    if (input.dtype() == torch::kFloat16) {
        dim3 block(16, 16);
        dim3 grid((out_features + 15) / 16, (batch + 15) / 16);
        fused_matmul_scale_residual_fp16_kernel<<<grid, block>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            bias.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            batch, in_features, out_features, scale_factor
        );
    } else if (input.dtype() == torch::kBFloat16) {
        dim3 block(16, 16);
        dim3 grid((out_features + 15) / 16, (batch + 15) / 16);
        fused_matmul_scale_residual_bf16_kernel<<<grid, block>>>(
            reinterpret_cast<const __nv_bfloat16*>(input.data_ptr<at::BFloat16>()),
            reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr<at::BFloat16>()),
            reinterpret_cast<const __nv_bfloat16*>(bias.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            batch, in_features, out_features, scale_factor
        );
    } else {
        // FP32
        dim3 block(16, 16);
        dim3 grid((out_features + 15) / 16, (batch + 15) / 16);
        fused_matmul_scale_residual_fp32_kernel<16><<<grid, block>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            batch, in_features, out_features, scale_factor
        );
    }
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_matmul_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor
);
"""

ext = load_inline(
    name='fused_matmul_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_scale_residual'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)


class Model(nn.Module):
    """
    A model that performs a matrix multiplication, scaling, and residual addition.
    Uses fused CUDA kernel for optimization.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        weight = self.matmul.weight
        bias = self.matmul.bias
        return ext.fused_matmul_scale_residual(x, weight, bias, self.scaling_factor)


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    scaling_factor = 0.5
    return [in_features, out_features, scaling_factor]
