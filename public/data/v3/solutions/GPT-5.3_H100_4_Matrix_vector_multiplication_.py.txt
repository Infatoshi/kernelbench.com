import torch
import torch.nn as nn
import triton
import triton.language as tl

# Tunable
@triton.jit
def matvec_kernel(A_ptr, B_ptr, C_ptr, M, K, stride_am, stride_ak, BLOCK_K: tl.constexpr):
    row = tl.program_id(0)
    if row >= M:
        return

    offs_k = tl.arange(0, BLOCK_K)
    a_ptrs = A_ptr + row * stride_am + offs_k * stride_ak
    b_ptrs = B_ptr + offs_k

    acc = tl.zeros((), dtype=tl.float32)

    for k in range(0, K, BLOCK_K):
        k_mask = offs_k + k < K
        a = tl.load(a_ptrs + k * stride_ak, mask=k_mask, other=0.0)
        b = tl.load(b_ptrs + k, mask=k_mask, other=0.0)
        acc += tl.sum(a * b)

    tl.store(C_ptr + row, acc)


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        M, K = A.shape
        C = torch.empty((M, 1), device=A.device, dtype=A.dtype)

        matvec_kernel[(M,)](
            A, B, C,
            M, K,
            A.stride(0), A.stride(1),
            BLOCK_K=1024
        )

        return C
