import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def _fused_gn_scale_pool_clamp_kernel(
    x_ptr,
    scale_ptr,
    y_ptr,
    N,
    C,
    H,
    W,
    OH,
    OW,
    groups,
    clamp_min,
    clamp_max,
    eps,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    n = pid // C
    c = pid % C
    if n >= N:
        return

    channels_per_group = C // groups
    g = c // channels_per_group
    c_start = g * channels_per_group

    HW = H * W
    group_elems = channels_per_group * HW

    offs = tl.arange(0, BLOCK)

    acc = tl.zeros([BLOCK], dtype=tl.float32)
    sum_val = tl.zeros([1], dtype=tl.float32)
    sumsq_val = tl.zeros([1], dtype=tl.float32)

    base_nc = n * C * HW
    for start in range(0, group_elems, BLOCK):
        idx = start + offs
        mask = idx < group_elems
        gc = idx // HW
        hw = idx - gc * HW
        ptrs = x_ptr + base_nc + (c_start + gc) * HW + hw
        vals = tl.load(ptrs, mask=mask, other=0.0).to(tl.float32)
        sum_val += tl.sum(vals, axis=0)
        sumsq_val += tl.sum(vals * vals, axis=0)

    mean = sum_val / group_elems
    var = sumsq_val / group_elems - mean * mean
    inv_std = tl.rsqrt(var + eps)

    scale = tl.load(scale_ptr + c).to(tl.float32)

    h2 = (tl.arange(0, BLOCK) // OW)
    w2 = (tl.arange(0, BLOCK) % OW)
    out_mask = tl.arange(0, BLOCK) < (OH * OW)

    for start in range(0, OH * OW, BLOCK):
        idx = start + offs
        mask = idx < (OH * OW)
        h_out = idx // OW
        w_out = idx - h_out * OW
        h0 = h_out * 2
        w0 = w_out * 2

        idx00 = base_nc + c * HW + h0 * W + w0
        idx01 = idx00 + 1
        idx10 = idx00 + W
        idx11 = idx10 + 1

        v00 = tl.load(x_ptr + idx00, mask=mask, other=-float('inf')).to(tl.float32)
        v01 = tl.load(x_ptr + idx01, mask=mask, other=-float('inf')).to(tl.float32)
        v10 = tl.load(x_ptr + idx10, mask=mask, other=-float('inf')).to(tl.float32)
        v11 = tl.load(x_ptr + idx11, mask=mask, other=-float('inf')).to(tl.float32)

        v00 = (v00 - mean) * inv_std * scale
        v01 = (v01 - mean) * inv_std * scale
        v10 = (v10 - mean) * inv_std * scale
        v11 = (v11 - mean) * inv_std * scale

        vmax = tl.maximum(tl.maximum(v00, v01), tl.maximum(v10, v11))
        vmax = tl.maximum(vmax, clamp_min)
        vmax = tl.minimum(vmax, clamp_max)

        out_ptrs = y_ptr + n * C * OH * OW + c * OH * OW + idx
        tl.store(out_ptrs, vmax, mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)
        assert x.is_cuda
        N, C, H, W = x.shape
        OH, OW = H // 2, W // 2
        y = torch.empty((N, C, OH, OW), device=x.device, dtype=x.dtype)

        scale = self.scale.view(-1).contiguous().to(dtype=x.dtype)
        grid = (N * C,)
        _fused_gn_scale_pool_clamp_kernel[grid](
            x,
            scale,
            y,
            N,
            C,
            H,
            W,
            OH,
            OW,
            self.group_norm.num_groups,
            self.clamp_min,
            self.clamp_max,
            self.group_norm.eps,
            BLOCK=1024,
        )
        return y


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]


def get_init_inputs():
    return [3, 16, 3, 8, (16, 1, 1), 2, 0.0, 1.0]
