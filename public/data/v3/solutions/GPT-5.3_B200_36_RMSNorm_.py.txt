import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

using namespace at;

template <typename scalar_t>
__global__ void rmsnorm_kernel(const scalar_t* __restrict__ x,
                               scalar_t* __restrict__ y,
                               int N, int C, int H, int W,
                               float eps) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int outer = N * H * W;
    if (idx >= outer) return;

    int hw = H * W;
    int n = idx / hw;
    int rem = idx % hw;
    int h = rem / W;
    int w = rem % W;

    int base = n * C * H * W + h * W + w;

    float sum = 0.0f;
    for (int c = 0; c < C; ++c) {
        int off = base + c * H * W;
        float v = (float)x[off];
        sum += v * v;
    }

    float mean = sum / C;
    float inv_rms = rsqrtf(mean + eps);

    for (int c = 0; c < C; ++c) {
        int off = base + c * H * W;
        float v = (float)x[off];
        y[off] = (scalar_t)(v * inv_rms);
    }
}

torch::Tensor rmsnorm(torch::Tensor x, float eps) {
    auto y = torch::empty_like(x);

    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int outer = N * H * W;
    int threads = 256;
    int blocks = (outer + threads - 1) / threads;

    AT_DISPATCH_FLOATING_TYPES_AND2(at::kHalf, at::kBFloat16, x.scalar_type(), "rmsnorm", ([&] {
        rmsnorm_kernel<scalar_t><<<blocks, threads>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            N, C, H, W, eps
        );
    }));

    return y;
}
'''

ext = load_inline(
    name='rmsnorm_ext',
    cpp_sources='torch::Tensor rmsnorm(torch::Tensor x, float eps);',
    cuda_sources=cuda_source,
    functions=['rmsnorm'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rmsnorm(x, self.eps)


def get_inputs():
    x = torch.randn(16, 64, 256, 256, device='cuda')
    return [x]


def get_init_inputs():
    return [64]
