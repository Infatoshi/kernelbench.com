from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Warp-level reduction using shuffle instructions
template <typename T, int WARP_SIZE>
__device__ __forceinline__ T warp_reduce_sum(T val) {
    #pragma unroll
    for (int offset = WARP_SIZE / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Optimized LayerNorm kernel - one block per row (batch element)
// Uses cooperative grid reduction for large normalized shapes
template <typename T, int BLOCK_SIZE, int VEC_SIZE>
__global__ void layernorm_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    const float eps,
    const int normalized_size
) {
    // Each block handles one batch element
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    
    // Pointers for this batch
    const T* x = input + batch_idx * normalized_size;
    T* y = output + batch_idx * normalized_size;
    
    // Use vectorized loads if possible
    using VecT = typename std::conditional<VEC_SIZE == 4, float4, 
                   typename std::conditional<VEC_SIZE == 2, float2, T>::type>::type;
    
    const int vec_size = VEC_SIZE;
    const int num_vecs = normalized_size / vec_size;
    const int remainder = normalized_size % vec_size;
    
    // First pass: compute mean
    float local_sum = 0.0f;
    
    // Vectorized loop
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        VecT vec = reinterpret_cast<const VecT*>(x)[i];
        if constexpr (VEC_SIZE == 4) {
            local_sum += static_cast<float>(vec.x) + static_cast<float>(vec.y) 
                       + static_cast<float>(vec.z) + static_cast<float>(vec.w);
        } else if constexpr (VEC_SIZE == 2) {
            local_sum += static_cast<float>(vec.x) + static_cast<float>(vec.y);
        } else {
            local_sum += static_cast<float>(vec);
        }
    }
    
    // Handle remainder
    const int remainder_start = num_vecs * vec_size;
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        local_sum += static_cast<float>(x[remainder_start + i]);
    }
    
    // Block-level reduction
    __shared__ float shared_sum[32];  // One per warp
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    const int num_warps = BLOCK_SIZE / 32;
    
    // Warp reduction
    float warp_sum = warp_reduce_sum<float, 32>(local_sum);
    
    // Write warp sums to shared memory
    if (lane_id == 0) {
        shared_sum[warp_id] = warp_sum;
    }
    __syncthreads();
    
    // First warp reduces shared sums
    float block_sum = 0.0f;
    if (tid < num_warps) {
        block_sum = shared_sum[tid];
    }
    if (warp_id == 0) {
        block_sum = warp_reduce_sum<float, 32>(block_sum);
    }
    
    // Broadcast mean to all threads
    __shared__ float mean;
    if (tid == 0) {
        mean = block_sum / normalized_size;
    }
    __syncthreads();
    
    // Second pass: compute variance
    float local_var = 0.0f;
    
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        VecT vec = reinterpret_cast<const VecT*>(x)[i];
        if constexpr (VEC_SIZE == 4) {
            float v0 = static_cast<float>(vec.x) - mean;
            float v1 = static_cast<float>(vec.y) - mean;
            float v2 = static_cast<float>(vec.z) - mean;
            float v3 = static_cast<float>(vec.w) - mean;
            local_var += v0*v0 + v1*v1 + v2*v2 + v3*v3;
        } else if constexpr (VEC_SIZE == 2) {
            float v0 = static_cast<float>(vec.x) - mean;
            float v1 = static_cast<float>(vec.y) - mean;
            local_var += v0*v0 + v1*v1;
        } else {
            float v = static_cast<float>(vec) - mean;
            local_var += v * v;
        }
    }
    
    // Handle remainder
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        float v = static_cast<float>(x[remainder_start + i]) - mean;
        local_var += v * v;
    }
    
    // Block-level reduction for variance
    float warp_var = warp_reduce_sum<float, 32>(local_var);
    
    if (lane_id == 0) {
        shared_sum[warp_id] = warp_var;
    }
    __syncthreads();
    
    float block_var = 0.0f;
    if (tid < num_warps) {
        block_var = shared_sum[tid];
    }
    if (warp_id == 0) {
        block_var = warp_reduce_sum<float, 32>(block_var);
    }
    
    __shared__ float rstd;
    if (tid == 0) {
        rstd = rsqrtf(block_var / normalized_size + eps);
    }
    __syncthreads();
    
    // Third pass: apply normalization
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        VecT vec = reinterpret_cast<const VecT*>(x)[i];
        VecT out_vec;
        
        if constexpr (VEC_SIZE == 4) {
            out_vec.x = static_cast<float>((static_cast<float>(vec.x) - mean) * rstd) * static_cast<float>(weight[i*4]) + static_cast<float>(bias[i*4]);
            out_vec.y = static_cast<float>((static_cast<float>(vec.y) - mean) * rstd) * static_cast<float>(weight[i*4+1]) + static_cast<float>(bias[i*4+1]);
            out_vec.z = static_cast<float>((static_cast<float>(vec.z) - mean) * rstd) * static_cast<float>(weight[i*4+2]) + static_cast<float>(bias[i*4+2]);
            out_vec.w = static_cast<float>((static_cast<float>(vec.w) - mean) * rstd) * static_cast<float>(weight[i*4+3]) + static_cast<float>(bias[i*4+3]);
        } else if constexpr (VEC_SIZE == 2) {
            out_vec.x = static_cast<float>((static_cast<float>(vec.x) - mean) * rstd) * static_cast<float>(weight[i*2]) + static_cast<float>(bias[i*2]);
            out_vec.y = static_cast<float>((static_cast<float>(vec.y) - mean) * rstd) * static_cast<float>(weight[i*2+1]) + static_cast<float>(bias[i*2+1]);
        } else {
            out_vec = static_cast<float>((static_cast<float>(vec) - mean) * rstd) * static_cast<float>(weight[i]) + static_cast<float>(bias[i]);
        }
        
        reinterpret_cast<VecT*>(y)[i] = out_vec;
    }
    
    // Handle remainder
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        int idx = remainder_start + i;
        float normalized = (static_cast<float>(x[idx]) - mean) * rstd;
        y[idx] = static_cast<T>(normalized * static_cast<float>(weight[idx]) + static_cast<float>(bias[idx]));
    }
}

// Optimized kernel for half precision (FP16)
template <int BLOCK_SIZE>
__global__ void layernorm_half_kernel(
    const half* __restrict__ input,
    half* __restrict__ output,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    const float eps,
    const int normalized_size
) {
    const int batch_idx = blockIdx.x;
    const int tid = threadIdx.x;
    
    const half* x = input + batch_idx * normalized_size;
    half* y = output + batch_idx * normalized_size;
    
    // Use float2 for 4 half values at once (8 bytes)
    const int vec_size = 4;  // Process 4 halfs = 8 bytes
    const int num_vecs = normalized_size / vec_size;
    const int remainder = normalized_size % vec_size;
    
    // First pass: compute mean
    float local_sum = 0.0f;
    
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        float2 vec = reinterpret_cast<const float2*>(x)[i];
        // Unpack 4 halfs from float2
        half2 h2_0 = reinterpret_cast<half2*>(&vec.x)[0];
        half2 h2_1 = reinterpret_cast<half2*>(&vec.y)[0];
        local_sum += __half2float(h2_0.x) + __half2float(h2_0.y) 
                   + __half2float(h2_1.x) + __half2float(h2_1.y);
    }
    
    const int remainder_start = num_vecs * vec_size;
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        local_sum += __half2float(x[remainder_start + i]);
    }
    
    // Block-level reduction
    __shared__ float shared_sum[32];
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    const int num_warps = BLOCK_SIZE / 32;
    
    float warp_sum = warp_reduce_sum<float, 32>(local_sum);
    
    if (lane_id == 0) {
        shared_sum[warp_id] = warp_sum;
    }
    __syncthreads();
    
    float block_sum = 0.0f;
    if (tid < num_warps) {
        block_sum = shared_sum[tid];
    }
    if (warp_id == 0) {
        block_sum = warp_reduce_sum<float, 32>(block_sum);
    }
    
    __shared__ float mean;
    if (tid == 0) {
        mean = block_sum / normalized_size;
    }
    __syncthreads();
    
    // Second pass: compute variance
    float local_var = 0.0f;
    
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        float2 vec = reinterpret_cast<const float2*>(x)[i];
        half2 h2_0 = reinterpret_cast<half2*>(&vec.x)[0];
        half2 h2_1 = reinterpret_cast<half2*>(&vec.y)[0];
        
        float v0 = __half2float(h2_0.x) - mean;
        float v1 = __half2float(h2_0.y) - mean;
        float v2 = __half2float(h2_1.x) - mean;
        float v3 = __half2float(h2_1.y) - mean;
        local_var += v0*v0 + v1*v1 + v2*v2 + v3*v3;
    }
    
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        float v = __half2float(x[remainder_start + i]) - mean;
        local_var += v * v;
    }
    
    float warp_var = warp_reduce_sum<float, 32>(local_var);
    
    if (lane_id == 0) {
        shared_sum[warp_id] = warp_var;
    }
    __syncthreads();
    
    float block_var = 0.0f;
    if (tid < num_warps) {
        block_var = shared_sum[tid];
    }
    if (warp_id == 0) {
        block_var = warp_reduce_sum<float, 32>(block_var);
    }
    
    __shared__ float rstd;
    if (tid == 0) {
        rstd = rsqrtf(block_var / normalized_size + eps);
    }
    __syncthreads();
    
    // Third pass: normalize
    for (int i = tid; i < num_vecs; i += BLOCK_SIZE) {
        float2 vec = reinterpret_cast<const float2*>(x)[i];
        half2 h2_0 = reinterpret_cast<half2*>(&vec.x)[0];
        half2 h2_1 = reinterpret_cast<half2*>(&vec.y)[0];
        
        half2 w0 = reinterpret_cast<const half2*>(weight)[i*2];
        half2 w1 = reinterpret_cast<const half2*>(weight)[i*2+1];
        half2 b0 = reinterpret_cast<const half2*>(bias)[i*2];
        half2 b1 = reinterpret_cast<const half2*>(bias)[i*2+1];
        
        half2 out0, out1;
        out0.x = __float2half((__half2float(h2_0.x) - mean) * rstd * __half2float(w0.x) + __half2float(b0.x));
        out0.y = __float2half((__half2float(h2_0.y) - mean) * rstd * __half2float(w0.y) + __half2float(b0.y));
        out1.x = __float2half((__half2float(h2_1.x) - mean) * rstd * __half2float(w1.x) + __half2float(b1.x));
        out1.y = __float2half((__half2float(h2_1.y) - mean) * rstd * __half2float(w1.y) + __half2float(b1.y));
        
        float2 out_vec;
        reinterpret_cast<half2*>(&out_vec.x)[0] = out0;
        reinterpret_cast<half2*>(&out_vec.y)[0] = out1;
        reinterpret_cast<float2*>(y)[i] = out_vec;
    }
    
    for (int i = tid; i < remainder; i += BLOCK_SIZE) {
        int idx = remainder_start + i;
        float normalized = (__half2float(x[idx]) - mean) * rstd;
        y[idx] = __float2half(normalized * __half2float(weight[idx]) + __half2float(bias[idx]));
    }
}

// Dispatch function
torch::Tensor layernorm_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double eps
) {
    const int batch_size = input.numel() / weight.numel();
    const int normalized_size = weight.numel();
    
    auto output = torch::empty_like(input);
    
    const int block_size = 512;
    
    if (input.scalar_type() == torch::kFloat32) {
        layernorm_kernel<float, block_size, 4><<<batch_size, block_size>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            static_cast<float>(eps),
            normalized_size
        );
    } else if (input.scalar_type() == torch::kHalf) {
        layernorm_half_kernel<block_size><<<batch_size, block_size>>>(
            reinterpret_cast<const half*>(input.data_ptr<at::Half>()),
            reinterpret_cast<half*>(output.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(weight.data_ptr<at::Half>()),
            reinterpret_cast<const half*>(bias.data_ptr<at::Half>()),
            static_cast<float>(eps),
            normalized_size
        );
    } else {
        TORCH_CHECK(false, "Unsupported dtype for layernorm");
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, double eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-std=c++17'],
)


class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5
        self.normalized_shape = normalized_shape

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(x, self.weight, self.bias, self.eps)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [(64, 256, 256)]
