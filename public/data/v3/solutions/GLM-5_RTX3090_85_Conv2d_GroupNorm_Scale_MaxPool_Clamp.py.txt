import torch
import torch.nn as nn
import triton
import triton.language as tl

# Fully fused kernel: GroupNorm + Scale + MaxPool + Clamp
@triton.jit
def fused_group_norm_scale_maxpool_clamp_kernel(
    x_ptr, output_ptr,
    weight_ptr, bias_ptr, scale_ptr,
    mean_ptr, rstd_ptr,
    N, C, H, W, OH, OW,
    num_groups,
    pool_size,
    clamp_min, clamp_max,
    eps,
    BLOCK_HW: tl.constexpr,
):
    """
    Fused GroupNorm + Scale + MaxPool2d + Clamp kernel.
    Each program instance handles one (batch, group) pair for norm computation,
    then outputs max-pooled result.
    """
    # Get batch and group indices
    batch_idx = tl.program_id(0)
    group_idx = tl.program_id(1)
    
    channels_per_group = C // num_groups
    channel_start = group_idx * channels_per_group
    channel_end = channel_start + channels_per_group
    
    # Compute mean
    count = channels_per_group * H * W
    sum_x = 0.0
    sum_x2 = 0.0
    
    # First pass: compute mean and variance
    for c in range(channel_start, channel_end):
        base_offset = batch_idx * C * H * W + c * H * W
        for hw in range(0, H * W, BLOCK_HW):
            offsets = hw + tl.arange(0, BLOCK_HW)
            mask = offsets < H * W
            x_vals = tl.load(x_ptr + base_offset + offsets, mask=mask, other=0.0).to(tl.float32)
            sum_x += tl.sum(x_vals)
            sum_x2 += tl.sum(x_vals * x_vals)
    
    mean = sum_x / count
    var = (sum_x2 / count) - (mean * mean)
    rstd = 1.0 / tl.sqrt(var + eps)
    
    # Store mean and rstd
    mean_idx = batch_idx * num_groups + group_idx
    tl.store(mean_ptr + mean_idx, mean)
    tl.store(rstd_ptr + mean_idx, rstd)
    
    # Second pass: normalize, scale, maxpool, and clamp
    for c in range(channel_start, channel_end):
        w = tl.load(weight_ptr + c).to(tl.float32)
        b = tl.load(bias_ptr + c).to(tl.float32)
        s = tl.load(scale_ptr + c).to(tl.float32)
        
        in_base = batch_idx * C * H * W + c * H * W
        out_base = batch_idx * C * OH * OW + c * OH * OW
        
        # Process each output pixel
        for ohw in range(0, OH * OW, BLOCK_HW):
            out_offsets = ohw + tl.arange(0, BLOCK_HW)
            out_mask = out_offsets < OH * OW
            
            # Initialize max values to negative infinity
            max_vals = tl.full([BLOCK_HW], -float('inf'), tl.float32)
            
            # For each output position, compute max over pool window
            for pool_idx in range(pool_size * pool_size):
                pool_offsets = out_offsets
                # Reconstruct oh, ow from linear index
                oh_vals = out_offsets // OW
                ow_vals = out_offsets % OW
                
                # Pool window offset
                kh = pool_idx // pool_size
                kw = pool_idx % pool_size
                
                # Input coordinates
                ih_vals = oh_vals * pool_size + kh
                iw_vals = ow_vals * pool_size + kw
                
                # Load input values
                in_offsets = ih_vals * W + iw_vals
                
                # Mask for valid positions
                valid_mask = out_mask & (ih_vals < H) & (iw_vals < W)
                
                in_vals = tl.load(x_ptr + in_base + in_offsets, mask=valid_mask, other=-float('inf')).to(tl.float32)
                
                # Normalize, scale, bias
                in_vals = (in_vals - mean) * rstd
                in_vals = in_vals * w + b
                in_vals = in_vals * s
                
                # Update max
                max_vals = tl.maximum(max_vals, in_vals)
            
            # Clamp and store
            max_vals = tl.maximum(clamp_min, tl.minimum(clamp_max, max_vals))
            tl.store(output_ptr + out_base + out_offsets, max_vals.to(tl.float32), mask=out_mask)


def fused_op(x, weight, bias, scale, num_groups, pool_size, clamp_min, clamp_max, eps=1e-5):
    """
    Fused GroupNorm + Scale + MaxPool2d + Clamp operation.
    """
    N, C, H, W = x.shape
    OH = H // pool_size
    OW = W // pool_size
    output = torch.empty(N, C, OH, OW, device=x.device, dtype=torch.float32)
    mean = torch.empty(N, num_groups, device=x.device, dtype=torch.float32)
    rstd = torch.empty(N, num_groups, device=x.device, dtype=torch.float32)
    
    BLOCK_HW = 256  # Smaller block size for maxpool
    grid = (N, num_groups)
    
    x = x.contiguous()
    
    fused_group_norm_scale_maxpool_clamp_kernel[grid](
        x, output,
        weight.contiguous(), bias.contiguous(), scale.contiguous(),
        mean, rstd,
        N, C, H, W, OH, OW,
        num_groups,
        pool_size,
        clamp_min, clamp_max,
        eps,
        BLOCK_HW=BLOCK_HW,
    )
    
    return output


class Model(nn.Module):
    """
    Optimized model with fully fused kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.num_groups = num_groups

    def forward(self, x):
        """
        Forward pass with fully fused kernel.
        """
        # Conv2d (keep as PyTorch implementation - highly optimized with cuDNN)
        x = self.conv(x)
        
        # Fused GroupNorm + Scale + MaxPool + Clamp
        x = fused_op(
            x,
            self.group_norm.weight,
            self.group_norm.bias,
            self.scale,
            self.num_groups,
            self.maxpool_kernel_size,
            self.clamp_min,
            self.clamp_max,
        )
        
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
