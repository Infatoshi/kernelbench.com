"""
FP4-like GEMV optimized using efficient INT8 dequantization and CUDA kernels.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Efficient GEMV kernel with INT8 storage (simulating FP4 range)
// Each thread computes one output element
template <int BLOCK_K>
__global__ void gemv_fp4_kernel(
    const int8_t* __restrict__ weight_q,  // (out_features, in_features) INT8
    const float* __restrict__ w_scale,    // Weight scale
    const int8_t* __restrict__ x_q,       // (batch, in_features) INT8
    const float* __restrict__ x_scale,    // Input scale
    half* __restrict__ y,                  // Output (batch, out_features)
    int batch_size,
    int in_features,
    int out_features
) {
    const int out_col = blockIdx.x * blockDim.x + threadIdx.x;
    const int batch_idx = blockIdx.y;

    if (out_col >= out_features || batch_idx >= batch_size) return;

    float acc = 0.0f;
    float x_scale_val = *x_scale;
    float w_scale_val = *w_scale;

    // Process in blocks for cache efficiency
    const int k_blocks = (in_features + BLOCK_K - 1) / BLOCK_K;
    
    for (int kb = 0; kb < k_blocks; kb++) {
        int k_start = kb * BLOCK_K;
        int k_end = min(k_start + BLOCK_K, in_features);
        
        // Sequential access for x, random access for weights
        // x[batch_idx, k] - sequential
        // W[out_col, k] - strided with stride = in_features
        
        for (int k = k_start; k < k_end; k++) {
            float x_val = static_cast<float>(x_q[batch_idx * in_features + k]) * x_scale_val;
            float w_val = static_cast<float>(weight_q[out_col * in_features + k]) * w_scale_val;
            acc += x_val * w_val;
        }
    }

    y[batch_idx * out_features + out_col] = __float2half(acc);
}

// Version with loop unrolling hints for BLOCK_K = 64
template <>
__global__ void gemv_fp4_kernel<64>(
    const int8_t* __restrict__ weight_q,
    const float* __restrict__ w_scale,
    const int8_t* __restrict__ x_q,
    const float* __restrict__ x_scale,
    half* __restrict__ y,
    int batch_size,
    int in_features,
    int out_features
) {
    const int out_col = blockIdx.x * blockDim.x + threadIdx.x;
    const int batch_idx = blockIdx.y;

    if (out_col >= out_features || batch_idx >= batch_size) return;

    float acc = 0.0f;
    float x_scale_val = *x_scale;
    float w_scale_val = *w_scale;

    const int BLOCK_K = 64;
    const int k_blocks = (in_features + BLOCK_K - 1) / BLOCK_K;
    
    for (int kb = 0; kb < k_blocks; kb++) {
        int k_base = kb * BLOCK_K;
        
        // Unroll-friendly inner loop
        #pragma unroll 4
        for (int i = 0; i < BLOCK_K; i++) {
            int k = k_base + i;
            if (k < in_features) {
                float x_val = static_cast<float>(x_q[batch_idx * in_features + k]) * x_scale_val;
                float w_val = static_cast<float>(weight_q[out_col * in_features + k]) * w_scale_val;
                acc += x_val * w_val;
            }
        }
    }

    y[batch_idx * out_features + out_col] = __float2half(acc);
}

// Shared memory blocking version for better memory efficiency
template <int BLOCK_M, int BLOCK_K>
__global__ void gemv_fp4_smem_kernel(
    const int8_t* __restrict__ weight_q,
    const float* __restrict__ w_scale,
    const int8_t* __restrict__ x_q,
    const float* __restrict__ x_scale,
    half* __restrict__ y,
    int batch_size,
    int in_features,
    int out_features
) {
    __shared__ float smem_x[32][BLOCK_K];  // x values for one batch
    __shared__ float smem_w[32][BLOCK_K];  // w values for 32 outputs

    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    const int bx = blockIdx.x;
    const int by = blockIdx.y;

    const int out_col_base = bx * BLOCK_M;
    const int batch_idx = by;

    // Each block handles BLOCK_M outputs for one batch element
    // Threads are arranged as BLOCK_M x (BLOCK_K / ELEMS_PER_THREAD)
    // For simplicity, use 1D block with BLOCK_M threads
    
    if (batch_idx >= batch_size) return;

    float x_scale_val = *x_scale;
    float w_scale_val = *w_scale;

    float acc = 0.0f;

    // Process in K blocks
    const int num_k_blocks = (in_features + BLOCK_K - 1) / BLOCK_K;
    
    for (int kb = 0; kb < num_k_blocks; kb++) {
        int k_base = kb * BLOCK_K;
        
        // Load x values into shared memory
        // x[batch_idx, k_base + k] for k in [0, BLOCK_K)
        if (k_base + tx < in_features) {
            smem_x[ty][tx] = static_cast<float>(x_q[batch_idx * in_features + k_base + tx]) * x_scale_val;
        } else {
            smem_x[ty][tx] = 0.0f;
        }
        
        // Load w values for our output columns
        // w[out_col_base + ty, k_base + tx]
        int out_col = out_col_base + ty;
        if (out_col < out_features && k_base + tx < in_features) {
            smem_w[ty][tx] = static_cast<float>(weight_q[out_col * in_features + k_base + tx]) * w_scale_val;
        } else {
            smem_w[ty][tx] = 0.0f;
        }
        
        __syncthreads();
        
        // Compute partial dot product
        #pragma unroll
        for (int k = 0; k < BLOCK_K; k++) {
            acc += smem_x[ty][k] * smem_w[ty][k];
        }
        
        __syncthreads();
    }

    // Write result
    int out_col = out_col_base + ty;
    if (out_col < out_features) {
        y[batch_idx * out_features + out_col] = __float2half(acc);
    }
}

// High-performance version with vectorized loads and warp-level reductions
template <int WARP_THREADS, int ELEMS_PER_THREAD>
__global__ void gemv_fp4_warp_kernel(
    const int8_t* __restrict__ weight_q,
    const float* __restrict__ w_scale,
    const int8_t* __restrict__ x_q,
    const float* __restrict__ x_scale,
    half* __restrict__ y,
    int batch_size,
    int in_features,
    int out_features
) {
    const int warp_id = threadIdx.x / WARP_THREADS;
    const int lane_id = threadIdx.x % WARP_THREADS;
    
    // Each warp handles one output element
    const int out_col = blockIdx.x * (blockDim.x / WARP_THREADS) + warp_id;
    const int batch_idx = blockIdx.y;
    
    if (out_col >= out_features || batch_idx >= batch_size) return;

    float acc = 0.0f;
    float x_scale_val = *x_scale;
    float w_scale_val = *w_scale;

    // Each thread in warp handles multiple K elements
    const int k_step = WARP_THREADS * ELEMS_PER_THREAD;
    
    for (int k_base = lane_id; k_base < in_features; k_base += k_step) {
        // Load x values for this k_base
        float x_vals[ELEMS_PER_THREAD];
        float w_vals[ELEMS_PER_THREAD];
        
        #pragma unroll
        for (int i = 0; i < ELEMS_PER_THREAD; i++) {
            int k = k_base + i * WARP_THREADS;
            if (k < in_features) {
                x_vals[i] = static_cast<float>(x_q[batch_idx * in_features + k]) * x_scale_val;
                w_vals[i] = static_cast<float>(weight_q[out_col * in_features + k]) * w_scale_val;
            } else {
                x_vals[i] = 0.0f;
                w_vals[i] = 0.0f;
            }
        }
        
        // Accumulate
        #pragma unroll
        for (int i = 0; i < ELEMS_PER_THREAD; i++) {
            acc += x_vals[i] * w_vals[i];
        }
    }

    // Warp-level reduction using shuffle
    #pragma unroll
    for (int offset = WARP_THREADS / 2; offset > 0; offset /= 2) {
        acc += __shfl_down_sync(0xffffffff, acc, offset);
    }

    // Lane 0 writes the result
    if (lane_id == 0) {
        y[batch_idx * out_features + out_col] = __float2half(acc);
    }
}

torch::Tensor gemv_fp4_forward(
    torch::Tensor weight_q,      // (out_features, in_features) int8
    torch::Tensor weight_scale,
    torch::Tensor x_q,           // (batch, in_features) int8
    torch::Tensor x_scale
) {
    int batch_size = x_q.size(0);
    int in_features = x_q.size(1);
    int out_features = weight_q.size(0);

    auto y = torch::empty({batch_size, out_features}, 
                          torch::TensorOptions().dtype(at::kHalf).device(at::kCUDA));

    const int BLOCK_K = 64;
    const int THREADS_PER_BLOCK = 256;

    // Use simple kernel: one thread per output element
    dim3 block_dim(THREADS_PER_BLOCK);
    dim3 grid_dim((out_features + THREADS_PER_BLOCK - 1) / THREADS_PER_BLOCK, batch_size);

    gemv_fp4_kernel<BLOCK_K><<<grid_dim, block_dim>>>(
        weight_q.data_ptr<int8_t>(),
        weight_scale.data_ptr<float>(),
        x_q.data_ptr<int8_t>(),
        x_scale.data_ptr<float>(),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        batch_size,
        in_features,
        out_features
    );

    return y;
}
'''

ext = load_inline(
    name='gemv_fp4_ext',
    cpp_sources='''
torch::Tensor gemv_fp4_forward(
    torch::Tensor weight_q,
    torch::Tensor weight_scale,
    torch::Tensor x_q,
    torch::Tensor x_scale
);
''',
    cuda_sources=cuda_source,
    functions=['gemv_fp4_forward'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        # Store weights in INT8 (simulating INT4 range: -8 to 8)
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # Use the custom kernel for efficient computation
        return ext.gemv_fp4_forward(
            self.weight_q,
            self.weight_scale,
            x_q,
            x_scale
        )


def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.08, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
