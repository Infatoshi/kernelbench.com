import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Reduce dim=1 of (64, 256, 256) tensor -> (64, 1, 256)
// B=64, D1=256 (reduce), D2=256
// Each output element = sum of 256 floats with stride 256
// Total output = 64*256 = 16384 elements
// Strategy: each thread block handles multiple output elements
// Use vectorized float4 loads where possible

// Kernel: each block of 256 threads handles 256 consecutive d2 values for one batch
// i.e., one block handles one complete batch row
// Grid: (B,) = (64,)
// Within block: threadIdx.x = d2 index
// Each thread sums D1=256 values

__global__ void sum_reduce_256x256(const float* __restrict__ x, float* __restrict__ out,
                                    int B, int D1, int D2) {
    const int b = blockIdx.x;
    const int d2 = threadIdx.x;  // 0..255
    
    const float* base = x + b * D1 * D2 + d2;
    float sum = 0.0f;
    
    #pragma unroll 16
    for (int k = 0; k < 256; k++) {
        sum += base[k * D2];
    }
    
    out[b * D2 + d2] = sum;
}

// General version for arbitrary sizes
__global__ void sum_reduce_general(const float* __restrict__ x, float* __restrict__ out,
                                    int B, int D1, int D2) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * D2;
    if (idx >= total) return;
    
    int b = idx / D2;
    int d2 = idx % D2;
    
    const float* base = x + b * D1 * D2 + d2;
    float sum = 0.0f;
    for (int k = 0; k < D1; k++) {
        sum += base[k * D2];
    }
    out[idx] = sum;
}

torch::Tensor sum_reduce_dim1(torch::Tensor x, int dim) {
    TORCH_CHECK(x.is_cuda() && x.is_contiguous());
    
    auto sizes = x.sizes();
    int ndim = sizes.size();
    if (dim < 0) dim += ndim;
    
    int64_t B = 1, D1 = sizes[dim], D2 = 1;
    for (int i = 0; i < dim; i++) B *= sizes[i];
    for (int i = dim + 1; i < ndim; i++) D2 *= sizes[i];
    
    // Output shape
    std::vector<int64_t> out_shape(sizes.begin(), sizes.end());
    out_shape[dim] = 1;
    
    auto out_flat = torch::empty({B, D2}, x.options());
    
    if (D1 == 256 && D2 == 256) {
        sum_reduce_256x256<<<B, 256>>>(x.data_ptr<float>(), out_flat.data_ptr<float>(), B, D1, D2);
    } else {
        int total = B * D2;
        int threads = 256;
        int blocks = (total + threads - 1) / threads;
        sum_reduce_general<<<blocks, threads>>>(x.data_ptr<float>(), out_flat.data_ptr<float>(), B, D1, D2);
    }
    
    return out_flat.view(out_shape);
}
'''

cpp_source = r'''
torch::Tensor sum_reduce_dim1(torch::Tensor x, int dim);
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduce_dim1'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce_dim1(x, self.dim)


batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
