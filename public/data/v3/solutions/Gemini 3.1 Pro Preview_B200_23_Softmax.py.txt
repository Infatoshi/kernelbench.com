import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template<typename T>
struct SumOp {
    __device__ __forceinline__ T operator()(const T& a, const T& b) const { return a + b; }
};

template<typename T>
struct MaxOp {
    __device__ __forceinline__ T operator()(const T& a, const T& b) const { return max(a, b); }
};

template <template <typename> class ReductionOp, typename T>
__device__ __forceinline__ T warpReduce(T val) {
    ReductionOp<T> op;
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        val = op(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    return val;
}

template <template <typename> class ReductionOp, typename T>
__device__ __forceinline__ T blockReduce(T val) {
    static __shared__ T shared[32];
    int lane = threadIdx.x % warpSize;
    int wid = threadIdx.x / warpSize;
    
    val = warpReduce<ReductionOp, T>(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    val = (threadIdx.x < blockDim.x / warpSize) ? shared[lane] : (ReductionOp<T>()(T(-INFINITY), T(-INFINITY)) == T(-INFINITY) ? T(-INFINITY) : T(0)); // zero or -inf depending on op
    if (wid == 0) val = warpReduce<ReductionOp, T>(val);
    return val;
}

// Specializations for T(0) and T(-inf)
template <>
__device__ __forceinline__ float blockReduce<MaxOp, float>(float val) {
    static __shared__ float shared[32];
    int lane = threadIdx.x % warpSize;
    int wid = threadIdx.x / warpSize;
    val = warpReduce<MaxOp, float>(val);
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    val = (threadIdx.x < blockDim.x / warpSize) ? shared[lane] : -INFINITY;
    if (wid == 0) val = warpReduce<MaxOp, float>(val);
    return val;
}

template <>
__device__ __forceinline__ float blockReduce<SumOp, float>(float val) {
    static __shared__ float shared[32];
    int lane = threadIdx.x % warpSize;
    int wid = threadIdx.x / warpSize;
    val = warpReduce<SumOp, float>(val);
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    val = (threadIdx.x < blockDim.x / warpSize) ? shared[lane] : 0.0f;
    if (wid == 0) val = warpReduce<SumOp, float>(val);
    return val;
}


template <typename scalar_t>
__global__ void softmax_kernel(const scalar_t* __restrict__ input, scalar_t* __restrict__ output, int n_cols) {
    int row = blockIdx.x;
    const scalar_t* x = input + row * n_cols;
    scalar_t* y = output + row * n_cols;
    
    float thread_max = -INFINITY;
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        thread_max = max(thread_max, static_cast<float>(x[i]));
    }
    
    float row_max = blockReduce<MaxOp, float>(thread_max);
    
    float thread_sum = 0.0f;
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        thread_sum += expf(static_cast<float>(x[i]) - row_max);
    }
    
    float row_sum = blockReduce<SumOp, float>(thread_sum);
    
    for (int i = threadIdx.x; i < n_cols; i += blockDim.x) {
        y[i] = static_cast<scalar_t>(expf(static_cast<float>(x[i]) - row_max) / row_sum);
    }
}

// Float4 vectorization
template <typename scalar_t>
__global__ void softmax_kernel_vec(const scalar_t* __restrict__ input, scalar_t* __restrict__ output, int n_cols) {
    int row = blockIdx.x;
    const scalar_t* x = input + row * n_cols;
    scalar_t* y = output + row * n_cols;
    
    int n_cols_vec = n_cols / 4;
    const float4* x_vec = reinterpret_cast<const float4*>(x);
    float4* y_vec = reinterpret_cast<float4*>(y);
    
    float thread_max = -INFINITY;
    for (int i = threadIdx.x; i < n_cols_vec; i += blockDim.x) {
        float4 vec = x_vec[i];
        thread_max = max(thread_max, vec.x);
        thread_max = max(thread_max, vec.y);
        thread_max = max(thread_max, vec.z);
        thread_max = max(thread_max, vec.w);
    }
    // Handle remainder
    for (int i = n_cols_vec * 4 + threadIdx.x; i < n_cols; i += blockDim.x) {
        thread_max = max(thread_max, static_cast<float>(x[i]));
    }
    
    float row_max = blockReduce<MaxOp, float>(thread_max);
    
    float thread_sum = 0.0f;
    for (int i = threadIdx.x; i < n_cols_vec; i += blockDim.x) {
        float4 vec = x_vec[i];
        thread_sum += expf(vec.x - row_max);
        thread_sum += expf(vec.y - row_max);
        thread_sum += expf(vec.z - row_max);
        thread_sum += expf(vec.w - row_max);
    }
    for (int i = n_cols_vec * 4 + threadIdx.x; i < n_cols; i += blockDim.x) {
        thread_sum += expf(static_cast<float>(x[i]) - row_max);
    }
    
    float row_sum = blockReduce<SumOp, float>(thread_sum);
    
    for (int i = threadIdx.x; i < n_cols_vec; i += blockDim.x) {
        float4 vec = x_vec[i];
        vec.x = expf(vec.x - row_max) / row_sum;
        vec.y = expf(vec.y - row_max) / row_sum;
        vec.z = expf(vec.z - row_max) / row_sum;
        vec.w = expf(vec.w - row_max) / row_sum;
        y_vec[i] = vec;
    }
    for (int i = n_cols_vec * 4 + threadIdx.x; i < n_cols; i += blockDim.x) {
        y[i] = static_cast<scalar_t>(expf(static_cast<float>(x[i]) - row_max) / row_sum);
    }
}

torch::Tensor my_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n_rows = x.size(0);
    int n_cols = x.size(1);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(x.scalar_type(), "softmax_kernel", ([&] {
        if (n_cols % 4 == 0) {
            softmax_kernel_vec<scalar_t><<<n_rows, 1024>>>(x.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(), n_cols);
        } else {
            softmax_kernel<scalar_t><<<n_rows, 1024>>>(x.data_ptr<scalar_t>(), y.data_ptr<scalar_t>(), n_cols);
        }
    }));
    return y;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor my_op(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=False,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100']
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.my_op(x)

def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]

def get_init_inputs():
    return []
