"""
Optimized FP8-like GEMV using Triton on B200.
"""

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def gemv_fp8_kernel(
    # Pointers to matrices
    x_q_ptr, x_scale_ptr,
    w_q_ptr, w_scale_ptr,
    output_ptr,
    # Matrix dimensions
    batch_size, in_features, out_features,
    # Strides
    stride_x_batch, stride_x_in,
    stride_w_out, stride_w_in,
    stride_out_batch, stride_out_out,
    # Meta-parameters
    BLOCK_K: tl.constexpr,
):
    """Optimized GEMV kernel for int8 quantized data."""
    
    # Get batch and output row index
    batch_idx = tl.program_id(0)
    out_idx = tl.program_id(1)
    
    # Bounds check for output
    if batch_idx >= batch_size or out_idx >= out_features:
        return
    
    # Get scale factors (scalars)
    x_scale = tl.load(x_scale_ptr)
    w_scale = tl.load(w_scale_ptr)
    scale = x_scale * w_scale
    
    # Initialize accumulator as a scalar (not block)
    acc = 0.0
    
    # Loop over input features in blocks
    for block_start in range(0, in_features, BLOCK_K):
        # Load x block: shape [BLOCK_K]
        k_offsets = block_start + tl.arange(0, BLOCK_K)
        k_mask = k_offsets < in_features
        
        # Load input vector block
        x_q = tl.load(
            x_q_ptr + batch_idx * stride_x_batch + k_offsets,
            mask=k_mask,
            other=0
        )
        
        # Load weight row block  
        w_q = tl.load(
            w_q_ptr + out_idx * stride_w_out + k_offsets,
            mask=k_mask,
            other=0
        )
        
        # Compute partial dot product
        acc += tl.sum(x_q.to(tl.float32) * w_q.to(tl.float32))
    
    # Apply scale and store result
    result = acc * scale
    
    # Store output
    tl.store(
        output_ptr + batch_idx * stride_out_batch + out_idx * stride_out_out,
        result.to(tl.float16)
    )


def gemv_fp8_triton(x_q, x_scale, w_q, w_scale):
    """Dispatch function for GEMV kernel."""
    batch_size, in_features = x_q.shape
    out_features, _ = w_q.shape
    
    # Allocate output
    output = torch.empty((batch_size, out_features), dtype=torch.float16, device=x_q.device)
    
    # Choose block size based on hardware
    BLOCK_K = 256  # Good balance for B200
    
    # Grid: (batch_size, out_features)
    grid = (batch_size, out_features)
    
    gemv_fp8_kernel[grid](
        x_q, x_scale,
        w_q, w_scale,
        output,
        batch_size, in_features, out_features,
        x_q.stride(0), x_q.stride(1),
        w_q.stride(0), w_q.stride(1),
        output.stride(0), output.stride(1),
        BLOCK_K=BLOCK_K,
    )
    
    return output


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # Ensure inputs are on GPU
        x_q = x_q.cuda() if not x_q.is_cuda else x_q
        x_scale = x_scale.cuda() if not x_scale.is_cuda else x_scale
        w_q = self.weight_q
        w_scale = self.weight_scale
        
        # Use optimized Triton kernel
        return gemv_fp8_triton(x_q, x_scale, w_q, w_scale)


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1

    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]
