
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def mv_kernel(
    A_ptr, B_ptr, C_ptr,
    M, K,
    stride_am, stride_ak,
    stride_bk, stride_b1,
    stride_cm, stride_c1,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    pid = tl.program_id(0)
    
    # Each program handles BLOCK_SIZE_M rows
    rm = pid * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    
    # Accumulator
    acc = tl.zeros((BLOCK_SIZE_M,), dtype=tl.float32)
    
    for k in range(0, K, BLOCK_SIZE_K):
        rk = k + tl.arange(0, BLOCK_SIZE_K)
        
        # Load A: (BLOCK_SIZE_M, BLOCK_SIZE_K)
        a = tl.load(A_ptr + rm[:, None] * stride_am + rk[None, :] * stride_ak)
        
        # Load B: (BLOCK_SIZE_K, 1)
        b = tl.load(B_ptr + rk[:, None] * stride_bk)
        
        # Matrix-vector multiply (using dot for 2D*2D where 2nd is vector-like)
        # However, for vector, we can just do sum(a * b_trans)
        acc += tl.sum(a * tl.trans(b), axis=1)
        
    # Store result
    tl.store(C_ptr + rm * stride_cm, acc)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        M, K = A.shape
        C = torch.empty((M, 1), device=A.device, dtype=A.dtype)
        
        BLOCK_SIZE_M = 1
        BLOCK_SIZE_K = 2048
        
        grid = (M // BLOCK_SIZE_M,)
        
        mv_kernel[grid](
            A, B, C,
            M, K,
            A.stride(0), A.stride(1),
            B.stride(0), B.stride(1),
            C.stride(0), C.stride(1),
            BLOCK_SIZE_M=BLOCK_SIZE_M,
            BLOCK_SIZE_K=BLOCK_SIZE_K,
            num_warps=8,
            num_stages=4,
        )
        return C
