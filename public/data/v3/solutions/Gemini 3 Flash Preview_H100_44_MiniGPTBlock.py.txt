import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Model(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        self.ln_1 = nn.LayerNorm(n_embd)
        self.attn = nn.Module()
        self.attn.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.attn.c_proj = nn.Linear(n_embd, n_embd)
        
        self.ln_2 = nn.LayerNorm(n_embd)
        self.mlp = nn.ModuleDict(dict(
            c_fc    = nn.Linear(n_embd, 4 * n_embd),
            c_proj  = nn.Linear(4 * n_embd, n_embd),
        ))
        
        self.n_head = n_head
        self.n_embd = n_embd
        self.head_dim = n_embd // n_head

    def forward(self, x):
        B, T, C = x.size()
        
        # --- Attention Block ---
        x_ln1 = self.ln_1(x)
        
        # Combined QKV projection
        qkv = self.attn.c_attn(x_ln1)
        q, k, v = qkv.split(self.n_embd, dim=2)
        
        q = q.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        k = k.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        v = v.view(B, T, self.n_head, self.head_dim).transpose(1, 2)
        
        y = F.scaled_dot_product_attention(q, k, v, attn_mask=None, dropout_p=0.0, is_causal=True)
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        
        y = self.attn.c_proj(y)
        x = x + y

        # --- MLP Block ---
        x_ln2 = self.ln_2(x)
        
        x_mlp = self.mlp.c_fc(x_ln2)
        # NewGELU: 0.5 * x * (1.0 + torch.tanh(math.sqrt(2.0/math.pi) * (x + 0.044715 * x^3)))
        x_mlp = 0.5 * x_mlp * (1.0 + torch.tanh(0.7978845608 * (x_mlp + 0.044715 * x_mlp * x_mlp * x_mlp)))
        x_mlp = self.mlp.c_proj(x_mlp)
        
        x = x + x_mlp
        
        return x
