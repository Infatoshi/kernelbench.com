import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ y,
    int N, int C, int H, int W,
    int OC, int KH, int KW,
    int OH, int OW
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int oc = (idx / (OW * OH)) % OC;
    int n = idx / (OW * OH * OC);

    float sum = b[oc];

    for (int c = 0; c < C; ++c) {
        for (int kh = 0; kh < KH; ++kh) {
            for (int kw = 0; kw < KW; ++kw) {
                int ih = oh + kh;
                int iw = ow + kw;
                float xv = x[((n*C + c)*H + ih)*W + iw];
                float wv = w[(((oc*C + c)*KH + kh)*KW + kw)];
                sum += xv * wv;
            }
        }
    }

    y[idx] = sum;
}

torch::Tensor conv2d_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int KH = w.size(2);
    int KW = w.size(3);

    int OH = H - KH + 1;
    int OW = W - KW + 1;

    auto y = torch::empty({N, OC, OH, OW}, x.options());

    int total = N * OC * OH * OW;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    conv2d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        y.data_ptr<float>(),
        N, C, H, W, OC, KH, KW, OH, OW
    );

    return y;
}
'''

ext = load_inline(
    name='fused_conv_bn',
    cpp_sources='torch::Tensor conv2d_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b);',
    cuda_sources=cuda_src,
    functions=['conv2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        # fuse conv + bn + scale (eval mode expected)
        w = self.conv.weight
        b = self.conv.bias if self.conv.bias is not None else torch.zeros(w.size(0), device=w.device, dtype=w.dtype)

        gamma = self.bn.weight
        beta = self.bn.bias
        mean = self.bn.running_mean
        var = self.bn.running_var
        eps = self.bn.eps

        inv_std = gamma / torch.sqrt(var + eps)

        w_fused = w * inv_std.view(-1,1,1,1) * self.scaling_factor
        b_fused = ((b - mean) * inv_std + beta) * self.scaling_factor

        return ext.conv2d_forward(x.contiguous(), w_fused.contiguous(), b_fused.contiguous())


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 2.0]
