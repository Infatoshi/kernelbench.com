import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename T>
__device__ __forceinline__ float to_float(T val) { return (float)val; }

template <>
__device__ __forceinline__ float to_float(at::Half val) { 
    return __half2float(*reinterpret_cast<const __half*>(&val)); 
}

template <>
__device__ __forceinline__ float to_float(at::BFloat16 val) { 
    return __bfloat162float(*reinterpret_cast<const __nv_bfloat16*>(&val)); 
}

template <typename T>
__device__ __forceinline__ T from_float(float val) { return (T)val; }

template <>
__device__ __forceinline__ at::Half from_float(float val) { 
    __half h = __float2half(val);
    return *reinterpret_cast<at::Half*>(&h);
}

template <>
__device__ __forceinline__ at::BFloat16 from_float(float val) { 
    __nv_bfloat16 bf = __float2bfloat16(val);
    return *reinterpret_cast<at::BFloat16*>(&bf);
}

template <typename T>
__global__ void rms_norm_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int batch_size,
    int num_features,
    int inner_dim,
    float eps) {
    
    int b = blockIdx.y;
    int i = blockIdx.x;
    
    if (b >= batch_size || i >= inner_dim) return;

    extern __shared__ float sdata[];

    float sum_sq = 0.0f;
    for (int f = threadIdx.x; f < num_features; f += blockDim.x) {
        float val = to_float(input[b * num_features * inner_dim + f * inner_dim + i]);
        sum_sq += val * val;
    }

    sdata[threadIdx.x] = sum_sq;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (threadIdx.x < s) {
            sdata[threadIdx.x] += sdata[threadIdx.x + s];
        }
        __syncthreads();
    }

    float rms = rsqrtf(sdata[0] / num_features + eps);

    for (int f = threadIdx.x; f < num_features; f += blockDim.x) {
        int idx = b * num_features * inner_dim + f * inner_dim + i;
        float val = to_float(input[idx]);
        output[idx] = from_float<T>(val * rms);
    }
}

torch::Tensor rms_norm_forward(torch::Tensor input, float eps) {
    auto sizes = input.sizes().vec();
    int batch_size = sizes[0];
    int num_features = (sizes.size() > 1) ? sizes[1] : 1;
    int inner_dim = 1;
    for (size_t i = 2; i < sizes.size(); ++i) {
        inner_dim *= sizes[i];
    }

    auto output = torch::empty_like(input);
    
    int threads_per_block = 256;
    if (num_features < threads_per_block) {
        threads_per_block = 32;
        while (threads_per_block < num_features && threads_per_block < 256) threads_per_block <<= 1;
    }

    dim3 threads(threads_per_block);
    dim3 blocks(inner_dim, batch_size);
    int shared_mem = threads.x * sizeof(float);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, input.scalar_type(), "rms_norm_forward", ([&] {
        rms_norm_kernel<scalar_t><<<blocks, threads, shared_mem>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size,
            num_features,
            inner_dim,
            eps);
    }));

    return output;
}
'''

ext = load_inline(
    name='rms_norm_ext',
    cpp_sources='torch::Tensor rms_norm_forward(torch::Tensor input, float eps);',
    cuda_sources=cuda_source,
    functions=['rms_norm_forward'],
    with_cuda=True,
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.rms_norm_forward(x, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]

def get_init_inputs():
    return [64]
