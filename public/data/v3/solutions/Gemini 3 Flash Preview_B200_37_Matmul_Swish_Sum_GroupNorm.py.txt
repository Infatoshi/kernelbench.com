import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_gemm_kernel(
    X_ptr, W_ptr, LinBias_ptr, AddBias_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = (M + BLOCK_M - 1) // BLOCK_M
    num_pid_n = (N + BLOCK_N - 1) // BLOCK_N
    pid_m = (pid % (num_pid_m * GROUP_SIZE_M)) // GROUP_SIZE_M
    pid_n = pid // (num_pid_m * GROUP_SIZE_M) * GROUP_SIZE_M + (pid % GROUP_SIZE_M)
    if pid_n >= num_pid_n:
        pid_m = pid // num_pid_n
        pid_n = pid % num_pid_n

    offs_m = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_n = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    X_ptrs = X_ptr + (offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    W_ptrs = W_ptr + (offs_n[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_mask = (k * BLOCK_K + tl.arange(0, BLOCK_K)) < K
        x = tl.load(X_ptrs, mask=k_mask[None, :], other=0.0)
        w = tl.load(W_ptrs, mask=k_mask[:, None], other=0.0)
        acc += tl.dot(x, w)
        X_ptrs += BLOCK_K * stride_xk
        W_ptrs += BLOCK_K * stride_wk

    lin_bias = tl.load(LinBias_ptr + offs_n[None, :], mask=offs_n[None, :] < N, other=0.0)
    acc += lin_bias
    acc = acc * tl.sigmoid(acc)
    add_bias = tl.load(AddBias_ptr + offs_n[None, :], mask=offs_n[None, :] < N, other=0.0)
    acc += add_bias

    out_ptrs = Out_ptr + (offs_m[:, None] * stride_om + offs_n[None, :] * stride_on)
    tl.store(out_ptrs, acc.to(Out_ptr.dtype.element_ty), mask=(offs_m[:, None] < M) & (offs_n[None, :] < N))

class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        out = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        num_pid_m = (M + 128 - 1) // 128
        num_pid_n = (N + 128 - 1) // 128
        grid = (num_pid_m * num_pid_n,)
        
        fused_gemm_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, self.bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_M=128, BLOCK_N=128, BLOCK_K=32,
            GROUP_SIZE_M=8,
            num_warps=8,
            num_stages=3
        )
        return self.group_norm(out)

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 32, (4096,)]
