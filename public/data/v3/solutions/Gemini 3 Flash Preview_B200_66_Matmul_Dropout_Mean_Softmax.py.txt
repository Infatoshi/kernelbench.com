import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(
    X_ptr, W_ptr, B_ptr, Sum_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    dropout_p, seed,
    IS_TRAIN: tl.constexpr,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    # Swizzling for better L2 cache reuse
    width = GROUP_SIZE_M * num_pid_n
    group_id = pid // width
    group_size = min(num_pid_m - group_id * GROUP_SIZE_M, GROUP_SIZE_M)
    pid_m = group_id * GROUP_SIZE_M + (pid % group_size)
    pid_n = (pid % width) // group_size

    rm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    rn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    rk = tl.arange(0, BLOCK_SIZE_K)

    # Base pointers
    X_ptr_base = X_ptr + rm[:, None] * stride_xm
    W_ptr_base = W_ptr + rn[None, :] * stride_wn

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        k_rem = K - k * BLOCK_SIZE_K
        # Transposed K access for weight B (it's weight.t())
        # The reference nn.Linear uses (out, in), so we need weight.t() in matmul
        # Weight passed is nn.Linear.weight which is (N, K). 
        # So W_ptr[k, n] means nn.Linear.weight[n, k]
        # stride_wn is stride(0), stride_wk is stride(1)
        x = tl.load(X_ptr_base + (k * BLOCK_SIZE_K + rk[None, :]) * stride_xk, 
                   mask=(rm[:, None] < M) & (rk[None, :] < k_rem), other=0.0)
        # Weight load: W[rn, rk] = W_ptr + rn*stride_wn + rk*stride_wk
        w = tl.load(W_ptr_base + (k * BLOCK_SIZE_K + rk[:, None]) * stride_wk, 
                   mask=(rk[:, None] < k_rem) & (rn[None, :] < N), other=0.0)
        accumulator += tl.dot(x, w)

    # Add Bias
    if B_ptr is not None:
        bias = tl.load(B_ptr + rn, mask=rn < N, other=0.0)
        accumulator += bias[None, :]

    # Dropout
    if IS_TRAIN and dropout_p > 0.0:
        keep_p = 1.0 - dropout_p
        offsets = rm[:, None] * N + rn[None, :]
        mask = tl.rand(seed, offsets) > dropout_p
        accumulator = tl.where(mask, accumulator / keep_p, 0.0)

    # Sum reduction along M
    mask_m = rm < M
    accumulator = tl.where(mask_m[:, None], accumulator, 0.0)
    sum_val = tl.sum(accumulator, axis=0)
    
    tl.atomic_add(Sum_ptr + rn, sum_val, mask=rn < N)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout_p = dropout_p
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        x = x.contiguous()
        m_input = x.shape[0]
        n_output = self.out_features
        k_inner = self.in_features
        
        sums = torch.zeros((n_output,), device=x.device, dtype=torch.float32)
        
        # Grid parameters
        BLOCK_SIZE_M = 32
        BLOCK_SIZE_N = 128
        BLOCK_SIZE_K = 32
        GROUP_SIZE_M = 8
        
        grid = (triton.cdiv(m_input, BLOCK_SIZE_M) * triton.cdiv(n_output, BLOCK_SIZE_N),)
        
        seed = torch.randint(0, 1000000, (1,)).item() if self.training else 0

        fused_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, sums,
            m_input, n_output, k_inner,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            self.dropout_p, seed,
            self.training,
            BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
        )
        
        res = (sums / m_input).unsqueeze(0)
        return torch.softmax(res, dim=1)

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 0.2]
