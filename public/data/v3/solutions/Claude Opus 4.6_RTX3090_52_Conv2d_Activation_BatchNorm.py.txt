import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _tanh(x):
    # tanh(x) = 1 - 2/(exp(2x) + 1)
    return 1.0 - 2.0 / (tl.exp(2.0 * x) + 1.0)


@triton.jit
def _mish(x):
    # mish(x) = x * tanh(softplus(x))
    # softplus(x) = log(1 + exp(x)), stabilized for large x
    sp = tl.where(x > 20.0, x, tl.log(1.0 + tl.exp(x)))
    return x * _tanh(sp)


@triton.jit
def fused_mish_bn_eval_kernel(
    X_ptr, Y_ptr,
    gamma_ptr, beta_ptr, mean_ptr, var_ptr,
    eps: tl.constexpr,
    C: tl.constexpr,
    HW: tl.constexpr,
    N_total: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Fused Mish + BN eval: y = gamma * (mish(x) - mean) / sqrt(var + eps) + beta"""
    pid = tl.program_id(0)
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < N_total
    
    x = tl.load(X_ptr + offs, mask=mask)
    
    # Compute channel index: layout is (N, C, H, W), so channel = (linear_idx // HW) % C
    c = (offs // HW) % C
    
    # Load BN params per channel
    gamma = tl.load(gamma_ptr + c, mask=mask)
    beta = tl.load(beta_ptr + c, mask=mask)
    mean = tl.load(mean_ptr + c, mask=mask)
    var = tl.load(var_ptr + c, mask=mask)
    
    # Mish activation
    mish_x = _mish(x)
    
    # BN eval
    inv_std = 1.0 / tl.sqrt(var + eps)
    y = gamma * (mish_x - mean) * inv_std + beta
    
    tl.store(Y_ptr + offs, y, mask=mask)


@triton.jit
def mish_bn_train_pass1_kernel(
    X_ptr,
    sum_ptr, sum_sq_ptr,
    C: tl.constexpr,
    HW: tl.constexpr,
    N_total: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Compute per-channel sum and sum_sq of mish(x) for BN training."""
    pid = tl.program_id(0)
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < N_total
    
    x = tl.load(X_ptr + offs, mask=mask, other=0.0)
    
    mish_x = _mish(x)
    
    c = (offs // HW) % C
    
    # Atomic add to per-channel accumulators
    tl.atomic_add(sum_ptr + c, mish_x, mask=mask)
    tl.atomic_add(sum_sq_ptr + c, mish_x * mish_x, mask=mask)


@triton.jit
def mish_bn_train_pass2_kernel(
    X_ptr, Y_ptr,
    sum_ptr, sum_sq_ptr,
    gamma_ptr, beta_ptr,
    eps: tl.constexpr,
    C: tl.constexpr,
    HW: tl.constexpr,
    NHW: tl.constexpr,
    N_total: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Apply BN normalization using computed statistics."""
    pid = tl.program_id(0)
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < N_total
    
    x = tl.load(X_ptr + offs, mask=mask)
    
    mish_x = _mish(x)
    
    c = (offs // HW) % C
    
    s = tl.load(sum_ptr + c, mask=mask)
    sq = tl.load(sum_sq_ptr + c, mask=mask)
    
    nhw_f = NHW.to(tl.float32)
    mean = s / nhw_f
    var = sq / nhw_f - mean * mean
    
    gamma = tl.load(gamma_ptr + c, mask=mask)
    beta = tl.load(beta_ptr + c, mask=mask)
    
    inv_std = 1.0 / tl.sqrt(var + eps)
    y = gamma * (mish_x - mean) * inv_std + beta
    
    tl.store(Y_ptr + offs, y, mask=mask)


@triton.jit  
def update_running_stats_kernel(
    sum_ptr, sum_sq_ptr,
    running_mean_ptr, running_var_ptr,
    momentum: tl.constexpr,
    NHW: tl.constexpr,
    C: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < C
    
    s = tl.load(sum_ptr + offs, mask=mask)
    sq = tl.load(sum_sq_ptr + offs, mask=mask)
    
    nhw_f = NHW.to(tl.float32)
    mean = s / nhw_f
    var = sq / nhw_f - mean * mean
    # Bessel's correction for running var
    unbiased_var = var * nhw_f / (nhw_f - 1.0)
    
    rm = tl.load(running_mean_ptr + offs, mask=mask)
    rv = tl.load(running_var_ptr + offs, mask=mask)
    
    new_rm = (1.0 - momentum) * rm + momentum * mean
    new_rv = (1.0 - momentum) * rv + momentum * unbiased_var
    
    tl.store(running_mean_ptr + offs, new_rm, mask=mask)
    tl.store(running_var_ptr + offs, new_rv, mask=mask)


class FusedMishBNFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, gamma, beta, running_mean, running_var, training, momentum, eps):
        B, C, H, W = x.shape
        HW = H * W
        N_total = x.numel()
        NHW = B * H * W
        
        y = torch.empty_like(x)
        
        BLOCK_SIZE = 1024
        grid = ((N_total + BLOCK_SIZE - 1) // BLOCK_SIZE,)
        
        if not training:
            fused_mish_bn_eval_kernel[grid](
                x, y,
                gamma, beta, running_mean, running_var,
                eps, C, HW, N_total, BLOCK_SIZE,
            )
        else:
            # Training: two-pass approach
            sum_buf = torch.zeros(C, device=x.device, dtype=torch.float32)
            sum_sq_buf = torch.zeros(C, device=x.device, dtype=torch.float32)
            
            # Pass 1: compute sums (with atomics)
            mish_bn_train_pass1_kernel[grid](
                x, sum_buf, sum_sq_buf,
                C, HW, N_total, BLOCK_SIZE,
            )
            
            # Pass 2: normalize and write output
            mish_bn_train_pass2_kernel[grid](
                x, y,
                sum_buf, sum_sq_buf,
                gamma, beta,
                eps, C, HW, NHW, N_total, BLOCK_SIZE,
            )
            
            # Update running stats
            update_running_stats_kernel[(1,)](
                sum_buf, sum_sq_buf,
                running_mean, running_var,
                momentum, NHW, C, 
                triton.next_power_of_2(C),
            )
        
        return y


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps
        self.momentum = momentum

    def forward(self, x):
        # Conv2d using aten op
        x = torch.ops.aten.convolution(
            x, self.conv.weight, self.conv.bias,
            [1, 1], [0, 0], [1, 1], False, [0, 0], 1
        )
        
        # Fused Mish + BatchNorm
        x = FusedMishBNFunction.apply(
            x, self.bn.weight, self.bn.bias,
            self.bn.running_mean, self.bn.running_var,
            self.bn.training, self.bn.momentum, self.bn.eps
        )
        
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
