import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv_weight = nn.Parameter(torch.Tensor(out_channels, in_channels, kernel_size, kernel_size))
        self.conv_bias = nn.Parameter(torch.Tensor(out_channels))
        self.gn_weight = nn.Parameter(torch.Tensor(out_channels))
        self.gn_bias = nn.Parameter(torch.Tensor(out_channels))
        self.scale = nn.Parameter(torch.Tensor(scale_shape))
        self.num_groups = num_groups
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        
        nn.init.kaiming_uniform_(self.conv_weight, a=math.sqrt(5))
        nn.init.zeros_(self.conv_bias)
        nn.init.ones_(self.gn_weight)
        nn.init.zeros_(self.gn_bias)
        nn.init.ones_(self.scale)
        
    def forward(self, x):
        x = F.conv2d(x, self.conv_weight, self.conv_bias)
        x = F.group_norm(x, self.num_groups, self.gn_weight, self.gn_bias)
        x = x * self.scale
        x = F.max_pool2d(x, self.maxpool_kernel_size)
        x = torch.clamp(x, min=self.clamp_min, max=self.clamp_max)
        return x

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda', dtype=torch.float16)]

def get_init_inputs():
    return [3, 16, 3, 8, (16, 1, 1), 2, 0.0, 1.0]
