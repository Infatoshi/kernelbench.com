import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>

static cublasHandle_t handle = nullptr;

torch::Tensor matmul_cublas(torch::Tensor A, torch::Tensor B) {
    if (handle == nullptr) {
        cublasCreate(&handle);
    }
    
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    // Rows-major format: C = A * B
    // cuBLAS is column-major: C^T = B^T * A^T
    // A (M, K) -> A^T (K, M). B (K, N) -> B^T (N, K). C (M, N) -> C^T (N, M).
    // result = cublasSgemm(handle, opB, opA, N, M, K, &alpha, B, ldb, A, lda, &beta, C, ldc)
    
    cublasSgemm(handle, 
                CUBLAS_OP_N, CUBLAS_OP_N, 
                N, M, K, 
                &alpha, 
                B.data_ptr<float>(), N, 
                A.data_ptr<float>(), K, 
                &beta, 
                C.data_ptr<float>(), N);
    
    return C;
}
'''

# Use extra_ldflags instead of libraries
ext = load_inline(
    name='matmul_ext',
    cpp_sources='torch::Tensor matmul_cublas(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matmul_cublas'],
    extra_ldflags=['-lcublas'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        A = A.cuda().contiguous()
        B = B.cuda().contiguous()
        return ext.matmul_cublas(A, B)

def get_inputs():
    M, K, N = 8205, 2949, 5921
    A = torch.randn(M, K, device='cuda')
    B = torch.randn(K, N, device='cuda')
    return [A, B]

    return []
