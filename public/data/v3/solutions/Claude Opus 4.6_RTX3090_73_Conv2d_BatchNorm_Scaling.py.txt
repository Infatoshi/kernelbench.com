import torch
import torch.nn as nn

# The key optimization: in eval mode, fuse BN + scaling into convolution weights/bias.
# This eliminates two extra kernel launches (BN + multiply) and replaces them with nothing.
# We precompute the fused weights once and use a single nn.Conv2d forward pass.

class Model(nn.Module):
    """
    Fused Conv2d + BatchNorm + Scaling.
    In eval mode, BN and scaling are folded into the conv weights/bias.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        self._fused = False

    def _fuse_bn_conv(self):
        """Fuse BN parameters and scaling factor into conv weight and bias in-place."""
        conv_w = self.conv.weight.data  # [out_ch, in_ch, kH, kW]
        conv_b = self.conv.bias.data    # [out_ch]
        
        bn_gamma = self.bn.weight.data        # [out_ch]
        bn_beta = self.bn.bias.data           # [out_ch]
        bn_mean = self.bn.running_mean        # [out_ch]
        bn_var = self.bn.running_var          # [out_ch]
        bn_eps = self.bn.eps
        
        # Compute the scale factor per output channel
        inv_std = 1.0 / torch.sqrt(bn_var + bn_eps)
        scale = self.scaling_factor * bn_gamma * inv_std  # [out_ch]
        
        # Modify conv weights and bias in-place
        self.conv.weight.data = conv_w * scale.view(-1, 1, 1, 1)
        self.conv.bias.data = scale * conv_b - scale * bn_mean + self.scaling_factor * bn_beta
        
        self._fused = True

    def forward(self, x):
        if not self._fused:
            self._fuse_bn_conv()
        
        # Single conv forward - uses cuDNN internally via nn.Conv2d.__call__
        return self.conv(x)


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor]
