import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda::wmma;

// Tile sizes for optimal performance on RTX 3090
constexpr int WMMA_M = 16;
constexpr int WMMA_N = 16;
constexpr int WMMA_K = 16;

constexpr int BLOCK_TILE_M = 64;   // Reduced for shared memory
constexpr int BLOCK_TILE_N = 64;
constexpr int BLOCK_TILE_K = 32;

// Each block has 64 threads = 2 warps
constexpr int WARPS_M = 2;  // 2 warps in M direction  
constexpr int WARPS_N = 1;  // 1 warp in N direction

__global__ void __launch_bounds__(64) gemm_scale_residual_wmma_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    const float* __restrict__ bias,
    half* __restrict__ D,
    float scaling_factor,
    int M, int N, int K
) {
    // Shared memory for input tiles
    extern __shared__ half smem[];
    half* s_A = smem;                             // [BLOCK_TILE_M, BLOCK_TILE_K]
    half* s_B = smem + BLOCK_TILE_M * BLOCK_TILE_K; // [BLOCK_TILE_N, BLOCK_TILE_K]
    
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    
    // Thread and warp identification
    const int tid = threadIdx.x;
    const int warp_id = tid / 32;
    
    // Warp position within block
    const int warp_m = warp_id;
    
    // Each warp computes 32x64 output (2 WMMA tiles in M, 4 in N)
    const int warp_out_m = 32;
    const int warp_out_n = 64;
    
    // Global output coordinates for this warp
    const int g_row_base = by * BLOCK_TILE_M + warp_m * 32;
    const int g_col_base = bx * BLOCK_TILE_N;
    
    // Accumulator fragments - 2x4 WMMA tiles per warp (32x64 output)
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc_frag[2][4];
    
    // Initialize accumulators
    #pragma unroll
    for (int i = 0; i < 2; i++) {
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            fill_fragment(acc_frag[i][j], 0.0f);
        }
    }
    
    // Number of K tiles
    const int num_k_tiles = (K + BLOCK_TILE_K - 1) / BLOCK_TILE_K;
    
    // Loop over K tiles
    for (int k_tile = 0; k_tile < num_k_tiles; k_tile++) {
        const int k_offset = k_tile * BLOCK_TILE_K;
        
        // Cooperative loading of A tile [BLOCK_TILE_M x BLOCK_TILE_K]
        for (int i = tid; i < BLOCK_TILE_M * BLOCK_TILE_K; i += 64) {
            const int row = i / BLOCK_TILE_K;
            const int col = i % BLOCK_TILE_K;
            const int g_row = by * BLOCK_TILE_M + row;
            const int g_col = k_offset + col;
            if (g_row < M && g_col < K) {
                s_A[i] = A[g_row * K + g_col];
            } else {
                s_A[i] = __float2half(0.0f);
            }
        }
        
        // Cooperative loading of B tile [BLOCK_TILE_N x BLOCK_TILE_K]
        for (int i = tid; i < BLOCK_TILE_N * BLOCK_TILE_K; i += 64) {
            const int row = i / BLOCK_TILE_K;
            const int col = i % BLOCK_TILE_K;
            const int g_row = bx * BLOCK_TILE_N + row;
            const int g_col = k_offset + col;
            if (g_row < N && g_col < K) {
                s_B[i] = B[g_row * K + g_col];
            } else {
                s_B[i] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Process K tiles within shared memory tile
        for (int k_sub = 0; k_sub < BLOCK_TILE_K; k_sub += WMMA_K) {
            #pragma unroll
            for (int i = 0; i < 2; i++) {
                #pragma unroll
                for (int j = 0; j < 4; j++) {
                    fragment<matrix_a, WMMA_M, WMMA_N, WMMA_K, half, row_major> a_frag;
                    fragment<matrix_b, WMMA_M, WMMA_N, WMMA_K, half, row_major> b_frag;
                    
                    // Load A fragment
                    const int a_row = warp_m * 32 + i * WMMA_M;
                    const int a_col = k_sub;
                    load_matrix_sync(a_frag, s_A + a_row * BLOCK_TILE_K + a_col, BLOCK_TILE_K);
                    
                    // Load B fragment
                    const int b_row = j * WMMA_N;
                    const int b_col = k_sub;
                    load_matrix_sync(b_frag, s_B + b_row * BLOCK_TILE_K + b_col, BLOCK_TILE_K);
                    
                    // MMA
                    mma_sync(acc_frag[i][j], a_frag, b_frag, acc_frag[i][j]);
                }
            }
        }
        
        __syncthreads();
    }
    
    // Apply fused operations: scale + residual
    const float scale = 1.0f + scaling_factor;
    
    // Store results with scaling and bias
    // Use a small shared memory buffer for each warp's accumulator
    __shared__ float acc_smem[2][32][64];  // [warps, rows, cols]
    
    #pragma unroll
    for (int i = 0; i < 2; i++) {
        #pragma unroll
        for (int j = 0; j < 4; j++) {
            // Scale accumulator
            for (int e = 0; e < acc_frag[i][j].num_elements; e++) {
                acc_frag[i][j].x[e] *= scale;
            }
            
            // Store to shared memory
            store_matrix_sync(&acc_smem[warp_m][i * WMMA_M][j * WMMA_N], 
                            acc_frag[i][j], 64, mem_row_major);
        }
    }
    
    __syncthreads();
    
    // Write to global memory with bias
    // Each thread handles multiple elements
    const int lane_id = tid % 32;
    
    for (int local_row = lane_id; local_row < 32; local_row += 32) {
        for (int local_col = 0; local_col < 64; local_col++) {
            const int global_row = g_row_base + local_row;
            const int global_col = g_col_base + local_col;
            
            if (global_row < M && global_col < N) {
                float val = acc_smem[warp_m][local_row][local_col];
                if (bias != nullptr) {
                    val += bias[global_col] * scale;
                }
                D[global_row * N + global_col] = __float2half(val);
            }
        }
    }
}

torch::Tensor fused_gemm_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double scaling_factor
) {
    // Convert to FP16 for tensor cores
    auto input_fp16 = input.to(torch::kHalf);
    auto weight_fp16 = weight.to(torch::kHalf);
    
    const int M = input.size(0);       // batch_size
    const int K = input.size(1);       // in_features
    const int N = weight.size(0);      // out_features
    
    auto output = torch::empty({M, N}, input_fp16.options());
    
    // Grid and block dimensions
    dim3 block(64);  // 64 threads per block
    dim3 grid((N + BLOCK_TILE_N - 1) / BLOCK_TILE_N, 
              (M + BLOCK_TILE_M - 1) / BLOCK_TILE_M);
    
    size_t smem_size = (BLOCK_TILE_M + BLOCK_TILE_N) * BLOCK_TILE_K * sizeof(half);
    
    gemm_scale_residual_wmma_kernel<<<grid, block, smem_size>>>(
        input_fp16.data_ptr<half>(),
        weight_fp16.data_ptr<half>(),
        bias.defined() && bias.numel() > 0 ? bias.data_ptr<float>() : nullptr,
        output.data_ptr<half>(),
        static_cast<float>(scaling_factor),
        M, N, K
    );
    
    // Return in same dtype as input
    return output.to(input.scalar_type());
}
'''

cpp_source = r'''
torch::Tensor fused_gemm_scale_residual(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double scaling_factor
);
'''

ext = load_inline(
    name='fused_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_gemm_scale_residual'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Create weight and bias parameters (matching nn.Linear format)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize weights same as nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        return ext.fused_gemm_scale_residual(x, self.weight, self.bias, self.scaling_factor)

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 0.5]
