import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    """
    Optimized Model for B200 (Blackwell).
    Fuses the two MaxPool3d operations into a single MaxPool3d(4, 4) operation.
    In 3D convolutions and spatial operations, memory bandwidth is often the bottleneck.
    Fusing pooling layers reduces memory round-trips.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # 1. 3D Convolution: Built-in is highly optimized for B200 via cuDNN.
        x = self.conv(x)
        
        # 2. Softmax: Standard dim=1 softmax.
        x = torch.softmax(x, dim=1)
        
        # 3. Fused MaxPool3d: Two layers of 2x2x2 pooling with stride 2
        # is equivalent to a single 4x4x4 pooling with stride 4.
        # This reduces kernel launch overhead and memory bandwidth requirements.
        # kernel_size=4, stride=4
        x = F.max_pool3d(x, kernel_size=4, stride=4)
        
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width, device='cuda')]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
