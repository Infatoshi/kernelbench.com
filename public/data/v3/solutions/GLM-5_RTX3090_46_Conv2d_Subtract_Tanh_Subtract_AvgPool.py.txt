import torch
import torch.nn as nn
import triton
import triton.language as tl
from torch.utils.cpp_extension import load_inline

# Fused kernel for post-conv operations: subtract, tanh, subtract, avgpool
@triton.jit
def fused_subtract_tanh_subtract_avgpool_kernel(
    # Input tensor (after conv) [N, C, H, W]
    input_ptr,
    # Output tensor [N, C, H//2, W//2]
    output_ptr,
    # Tensor dimensions
    N, C, H, W,
    # Subtract values
    subtract1,
    subtract2,
    # Strides for input (NCHW layout)
    stride_n_in, stride_c_in, stride_h_in, stride_w_in,
    # Strides for output (NCHW layout)
    stride_n_out, stride_c_out, stride_h_out, stride_w_out,
    # Output dimensions
    out_H, out_W,
):
    # Get output pixel coordinates
    pid = tl.program_id(0)
    
    # Compute n, c, h, w from linear index
    # Output is [N, C, out_H, out_W]
    out_w = pid % out_W
    out_h = (pid // out_W) % out_H
    c = (pid // (out_W * out_H)) % C
    n = pid // (out_W * out_H * C)
    
    # Each output pixel is average of 2x2 input block
    # Load 4 input values and compute avgpool + operations
    sum_val = 0.0
    
    # Unroll the 2x2 pooling
    for dh in range(2):
        for dw in range(2):
            in_h = out_h * 2 + dh
            in_w = out_w * 2 + dw
            
            # Compute input offset
            in_offset = n * stride_n_in + c * stride_c_in + in_h * stride_h_in + in_w * stride_w_in
            
            # Load input value
            val = tl.load(input_ptr + in_offset).to(tl.float32)
            
            # Apply subtract1, tanh, subtract2
            val = val - subtract1
            val = tl.math.tanh(val)
            val = val - subtract2
            
            sum_val += val
    
    # Average over 2x2 pool
    result = sum_val * 0.5 * 0.5  # divide by 4
    
    # Store result
    out_offset = n * stride_n_out + c * stride_c_out + out_h * stride_h_out + out_w * stride_w_out
    tl.store(output_ptr + out_offset, result)


def fused_post_conv_ops(input_tensor, subtract1, subtract2):
    """
    Fused operation: subtract, tanh, subtract, avgpool(2x2)
    Input: [N, C, H, W]
    Output: [N, C, H//2, W//2]
    """
    N, C, H, W = input_tensor.shape
    assert H % 2 == 0 and W % 2 == 0, "H and W must be divisible by 2 for 2x2 avgpool"
    
    out_H = H // 2
    out_W = W // 2
    
    output = torch.empty((N, C, out_H, out_W), dtype=input_tensor.dtype, device=input_tensor.device)
    
    total_outputs = N * C * out_H * out_W
    
    grid = lambda meta: (total_outputs,)
    
    fused_subtract_tanh_subtract_avgpool_kernel[grid](
        input_tensor,
        output,
        N, C, H, W,
        subtract1,
        subtract2,
        input_tensor.stride(0), input_tensor.stride(1), input_tensor.stride(2), input_tensor.stride(3),
        output.stride(0), output.stride(1), output.stride(2), output.stride(3),
        out_H, out_W,
    )
    
    return output


# CUDA source for custom conv2d with fused operations
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fast tanh approximation for float
__device__ __forceinline__ float fast_tanh(float x) {
    // Use built-in which is fast on modern GPUs
    return tanhf(x);
}

// Optimized 3x3 convolution kernel with fused post-processing
// Each thread block processes a tile of output pixels
template<typename scalar_t>
__global__ void conv2d_fused_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    const scalar_t* __restrict__ bias,
    scalar_t* __restrict__ output,
    int N, int C_in, int H_in, int W_in,
    int C_out, int H_out, int W_out,
    int kernel_size, int padding,
    float subtract1, float subtract2,
    int H_pool_out, int W_pool_out
) {
    // Shared memory for input tile (with halo for 3x3 conv)
    // We load a tile of input and compute multiple output pixels
    // Then apply fused ops and write to output
    
    // For simplicity, each thread computes one final output pixel (after pooling)
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    int total_output = N * C_out * H_pool_out * W_pool_out;
    if (idx >= total_output) return;
    
    // Decode output indices
    int w_pool_out = idx % W_pool_out;
    int h_pool_out = (idx / W_pool_out) % H_pool_out;
    int c_out = (idx / (W_pool_out * H_pool_out)) % C_out;
    int n = idx / (W_pool_out * H_pool_out * C_out);
    
    // Each pooled output is average of 2x2 conv outputs
    float pool_sum = 0.0f;
    
    for (int dh_pool = 0; dh_pool < 2; dh_pool++) {
        for (int dw_pool = 0; dw_pool < 2; dw_pool++) {
            // Conv output position (before pooling)
            int h_out = h_pool_out * 2 + dh_pool;
            int w_out = w_pool_out * 2 + dw_pool;
            
            // Compute convolution for this output pixel
            float conv_val = (bias != nullptr) ? static_cast<float>(bias[c_out]) : 0.0f;
            
            // 3x3 convolution
            for (int c_in = 0; c_in < C_in; c_in++) {
                for (int kh = 0; kh < kernel_size; kh++) {
                    for (int kw = 0; kw < kernel_size; kw++) {
                        // Input position (assuming padding=0 for 3x3 conv)
                        int h_in = h_out + kh;
                        int w_in = w_out + kw;
                        
                        if (h_in >= 0 && h_in < H_in && w_in >= 0 && w_in < W_in) {
                            int input_idx = n * C_in * H_in * W_in + c_in * H_in * W_in + h_in * W_in + w_in;
                            int weight_idx = c_out * C_in * kernel_size * kernel_size + 
                                            c_in * kernel_size * kernel_size + 
                                            kh * kernel_size + kw;
                            
                            float in_val = static_cast<float>(input[input_idx]);
                            float w_val = static_cast<float>(weight[weight_idx]);
                            conv_val += in_val * w_val;
                        }
                    }
                }
            }
            
            // Apply fused operations: subtract1, tanh, subtract2
            conv_val = conv_val - subtract1;
            conv_val = fast_tanh(conv_val);
            conv_val = conv_val - subtract2;
            
            pool_sum += conv_val;
        }
    }
    
    // Average pooling (divide by 4)
    float result = pool_sum * 0.25f;
    
    // Write output
    int out_idx = n * C_out * H_pool_out * W_pool_out + c_out * H_pool_out * W_pool_out + 
                  h_pool_out * W_pool_out + w_pool_out;
    output[out_idx] = static_cast<scalar_t>(result);
}

torch::Tensor fused_conv_subtract_tanh_subtract_avgpool(
    torch::Tensor input,
    torch::Tensor weight,
    c10::optional<torch::Tensor> bias_opt,
    int64_t kernel_size,
    double subtract1,
    double subtract2,
    int64_t kernel_size_pool
) {
    auto bias = bias_opt.value_or(torch::Tensor());
    
    int N = input.size(0);
    int C_in = input.size(1);
    int H_in = input.size(2);
    int W_in = input.size(3);
    
    int C_out = weight.size(0);
    int H_out = H_in - kernel_size + 1;  // No padding
    int W_out = W_in - kernel_size + 1;
    
    int H_pool_out = H_out / kernel_size_pool;
    int W_pool_out = W_out / kernel_size_pool;
    
    auto output = torch::empty({N, C_out, H_pool_out, W_pool_out}, input.options());
    
    int total_output = N * C_out * H_pool_out * W_pool_out;
    int threads = 256;
    int blocks = (total_output + threads - 1) / threads;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "conv2d_fused", ([&] {
        conv2d_fused_kernel<scalar_t><<<blocks, threads>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.defined() ? bias.data_ptr<scalar_t>() : nullptr,
            output.data_ptr<scalar_t>(),
            N, C_in, H_in, W_in,
            C_out, H_out, W_out,
            kernel_size, 0,  // padding = 0
            static_cast<float>(subtract1),
            static_cast<float>(subtract2),
            H_pool_out, W_pool_out
        );
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_conv_subtract_tanh_subtract_avgpool(
    torch::Tensor input,
    torch::Tensor weight,
    c10::optional<torch::Tensor> bias,
    int64_t kernel_size,
    double subtract1,
    double subtract2,
    int64_t kernel_size_pool
);
'''

# Load the extension
ext = load_inline(
    name='conv_fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_subtract_tanh_subtract_avgpool'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)


class Model(nn.Module):
    """
    Optimized model with fused convolution and post-processing operations.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool
        
        # Use a conv module to match reference state dict keys
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
    
    def forward(self, x):
        return ext.fused_conv_subtract_tanh_subtract_avgpool(
            x, 
            self.conv.weight,
            self.conv.bias,
            self.kernel_size,
            self.subtract1_value,
            self.subtract2_value,
            self.kernel_size_pool
        )
