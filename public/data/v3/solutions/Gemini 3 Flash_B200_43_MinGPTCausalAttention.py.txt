
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class Model(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.n_head = n_head
        self.n_embd = n_embd
        self.head_dim = n_embd // n_head

    def forward(self, x):
        # We can optimize slightly by doing a single matrix multiplication
        # and then using Flash Attention via scaled_dot_product_attention.
        B, T, C = x.size()
        
        # Projection for Q, K, V
        qkv = self.c_attn(x) # (B, T, 3*C)
        
        # Reshape and separate Q, K, V
        # Torch's scaled_dot_product_attention expects (B, n_head, T, head_dim)
        # We can optimize the view/transpose sequence
        qkv = qkv.view(B, T, 3, self.n_head, self.head_dim)
        q = qkv[:, :, 0, :, :].transpose(1, 2) # (B, nh, T, hs)
        k = qkv[:, :, 1, :, :].transpose(1, 2) # (B, nh, T, hs)
        v = qkv[:, :, 2, :, :].transpose(1, 2) # (B, nh, T, hs)
        
        # Efficient Flash Attention 2 / Memory Efficient Attention
        # is_causal=True automatically handles the triangular mask
        y = F.scaled_dot_product_attention(
            q, k, v, 
            attn_mask=None, 
            dropout_p=self.attn_dropout.p if self.training else 0.0, 
            is_causal=True
        )
        
        # Re-assemble heads
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        
        # Final projection
        y = self.resid_dropout(self.c_proj(y))
        return y

def get_inputs():
    batch_size = 64
    seq_len = 512
    n_embd = 768
    return [torch.randn(batch_size, seq_len, n_embd, device='cuda')]

def get_init_inputs():
    n_embd = 768
    n_head = 8
    attn_pdrop = 0.0
    resid_pdrop = 0.0
    max_seqlen = 1024
    return [n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen]
