import torch
import torch.nn as nn
import torch.nn.functional as F

# Gated DeltaNet: Linear Attention with Gated Delta Rule
# Optimized implementation

OP_TYPE = "model"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


def gated_delta_attention_optimized(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    alpha: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
) -> torch.Tensor:
    """
    Optimized gated delta rule attention using efficient matrix operations.
    
    The recurrence is:
    S_t = alpha_t * S_{t-1} + beta_t * (v_t - S_{t-1} @ k_t^T) @ k_t
    o_t = S_t @ q_t
    
    q, k: (batch, heads, seq, head_dim_qk)
    v: (batch, heads, seq, head_dim_v)
    alpha: (batch, heads, seq) - decay gate (0-1)
    beta: (batch, heads, seq) - update gate (0-1)
    """
    batch, heads, seq_len, head_dim_qk = q.shape
    _, _, _, head_dim_v = v.shape
    
    # Convert alpha to log-space for numerical stability
    g = alpha.clamp(min=1e-6).log()
    
    # Use efficient matrix operations
    # The key is to use cumsum and cumprod for the decay factors
    
    # Compute cumulative sum of g for decay computation
    g_cumsum = F.pad(torch.cumsum(g, dim=-1), (1, 0), value=0.0)  # (B, H, T+1)
    
    # Create position indices
    pos = torch.arange(seq_len, device=q.device)
    
    # Initialize output
    output = torch.zeros(batch, heads, seq_len, head_dim_v, device=q.device, dtype=torch.float32)
    
    # Process in chunks for memory efficiency
    chunk_size = 64
    
    for chunk_start in range(0, seq_len, chunk_size):
        chunk_end = min(chunk_start + chunk_size, seq_len)
        chunk_len = chunk_end - chunk_start
        
        # Get q for this chunk
        q_chunk = q[:, :, chunk_start:chunk_end, :]  # (B, H, chunk_len, D_qk)
        
        # Compute attention scores: q_chunk @ k^T
        # scores[b, h, m, n] = q[b, h, chunk_start+m] @ k[b, h, n]
        scores = torch.einsum('bhmd,bhnd->bhmn', q_chunk, k) * scale  # (B, H, chunk_len, seq)
        
        # Compute decay factors
        # For position m in chunk and position n in sequence (n <= chunk_start + m)
        # decay = exp(g_cumsum[chunk_start+m] - g_cumsum[n+1])
        
        # Create mask for causal attention
        m_pos = chunk_start + torch.arange(chunk_len, device=q.device)  # (chunk_len,)
        n_pos = torch.arange(seq_len, device=q.device)  # (seq_len,)
        
        # decay_matrix[m, n] = exp(g_cumsum[m_pos[m]] - g_cumsum[n_pos[n] + 1]) if n <= m_pos[m] else 0
        # But we need to be careful: the decay is from n to m, so it's exp(sum of g from n+1 to m)
        
        # g_cumsum[i] = sum of g[0:i]
        # sum of g from n+1 to m = g_cumsum[m+1] - g_cumsum[n+1]
        
        # For each m in chunk, we want decay from n to m where n <= m
        # decay[n, m] = exp(g_cumsum[m+1] - g_cumsum[n+1]) for n <= m
        
        # Reshape for broadcasting
        g_cumsum_m = g_cumsum[:, :, m_pos + 1]  # (B, H, chunk_len)
        g_cumsum_n = g_cumsum[:, :, 1:]  # (B, H, seq_len)
        
        # decay_matrix[b, h, m, n] = exp(g_cumsum_m[b, h, m] - g_cumsum_n[b, h, n])
        # But only for n <= m_pos[m]
        
        # Create causal mask
        causal_mask = (n_pos <= m_pos[:, None])  # (chunk_len, seq_len)
        
        # Compute decay
        decay = torch.exp(g_cumsum_m[:, :, :, None] - g_cumsum_n[:, :, None, :])  # (B, H, chunk_len, seq)
        decay = decay * causal_mask.to(decay.dtype)  # Apply causal mask
        
        # Also need to incorporate beta
        # The full contribution from position n to position m is:
        # beta[n] * decay[n+1:m] * (q[m] @ k[n]) * v[n]
        
        # beta_decay[b, h, n, m] = beta[b, h, n] * decay[b, h, n, m]
        beta_expanded = beta[:, :, :, None]  # (B, H, seq, 1)
        beta_decay = beta_expanded * decay  # (B, H, seq, chunk_len) - wait, shapes don't match
        
        # Let me reconsider the formula
        # The output at position m is:
        # o[m] = sum_{n=0}^{m} beta[n] * decay(n+1, m) * (q[m] @ k[n]) * v[n]
        # where decay(n+1, m) = exp(sum of g from n+1 to m)
        
        # So we need:
        # 1. scores[m, n] = q[m] @ k[n] (already computed)
        # 2. beta_decay[n, m] = beta[n] * decay(n+1, m) for n <= m
        # 3. output[m] = sum_n beta_decay[n, m] * scores[m, n] * v[n]
        
        # Reshape decay to match
        # decay is (B, H, chunk_len, seq) where decay[m, n] is decay from n to m
        # We want beta_decay[n, m] = beta[n] * decay[m, n]
        
        beta_for_decay = beta[:, :, :, None]  # (B, H, seq, 1)
        beta_decay = decay * beta_for_decay  # (B, H, chunk_len, seq)
        
        # Now compute: output[m] = sum_n beta_decay[m, n] * scores[m, n] * v[n]
        # This is: output = sum_n (beta_decay * scores)[m, n] * v[n]
        # Which is: output = (beta_decay * scores) @ v
        
        # weighted_scores = beta_decay * scores  # (B, H, chunk_len, seq)
        # output_chunk = weighted_scores @ v  # (B, H, chunk_len, D_v)
        
        weighted_scores = beta_decay * scores  # (B, H, chunk_len, seq)
        output_chunk = torch.einsum('bhmn,bhnd->bhmd', weighted_scores, v)  # (B, H, chunk_len, D_v)
        
        output[:, :, chunk_start:chunk_end, :] = output_chunk
    
    return output.to(q.dtype)


class Model(nn.Module):
    """
    Gated DeltaNet: Linear Attention with Gated Delta Rule
    Optimized implementation.
    """

    def __init__(
        self,
        hidden_size: int,
        num_heads: int,
        head_dim_qk: int,
        head_dim_v: int,
        use_short_conv: bool = True,
        conv_kernel_size: int = 4,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_heads = num_heads
        self.head_dim_qk = head_dim_qk
        self.head_dim_v = head_dim_v
        self.use_short_conv = use_short_conv

        self.q_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.k_proj = nn.Linear(hidden_size, num_heads * head_dim_qk, bias=False)
        self.v_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)

        self.a_proj = nn.Linear(hidden_size, num_heads, bias=True)
        self.b_proj = nn.Linear(hidden_size, num_heads, bias=True)

        self.o_proj = nn.Linear(num_heads * head_dim_v, hidden_size, bias=False)

        if use_short_conv:
            self.q_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.k_conv = nn.Conv1d(
                num_heads * head_dim_qk, num_heads * head_dim_qk,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_qk,
                padding=conv_kernel_size - 1
            )
            self.v_conv = nn.Conv1d(
                num_heads * head_dim_v, num_heads * head_dim_v,
                kernel_size=conv_kernel_size, groups=num_heads * head_dim_v,
                padding=conv_kernel_size - 1
            )

        self.g_proj = nn.Linear(hidden_size, num_heads * head_dim_v, bias=False)
        self.o_norm = nn.LayerNorm(head_dim_v)
        self.scale = head_dim_qk ** -0.5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape

        q = self.q_proj(x)
        k = self.k_proj(x)
        v = self.v_proj(x)

        if self.use_short_conv:
            q = self.q_conv(q.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            k = self.k_conv(k.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            v = self.v_conv(v.transpose(1, 2))[:, :, :seq_len].transpose(1, 2)
            q = F.silu(q)
            k = F.silu(k)
            v = F.silu(v)

        # Reshape to (B, H, T, D) for recurrence
        q = q.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        k = k.view(batch_size, seq_len, self.num_heads, self.head_dim_qk).transpose(1, 2)
        v = v.view(batch_size, seq_len, self.num_heads, self.head_dim_v).transpose(1, 2)

        alpha = torch.sigmoid(self.a_proj(x)).transpose(1, 2)  # (B, H, T)
        beta = torch.sigmoid(self.b_proj(x)).transpose(1, 2)

        # Optimized gated delta attention
        o = gated_delta_attention_optimized(q, k, v, alpha, beta, scale=self.scale)

        # (B, H, T, d_v) -> (B, T, H, d_v)
        o = o.transpose(1, 2)

        o = self.o_norm(o)

        g = torch.sigmoid(self.g_proj(x))
        g = g.view(batch_size, seq_len, self.num_heads, self.head_dim_v)
        o = o * g

        o = o.reshape(batch_size, seq_len, self.num_heads * self.head_dim_v)
        o = self.o_proj(o)

        return o


batch_size = 4
seq_len = 2048
hidden_size = 2048
num_heads = 16
head_dim_qk = 128
head_dim_v = 128


def get_inputs():
    return [torch.randn(batch_size, seq_len, hidden_size)]


def get_init_inputs():
    return [hidden_size, num_heads, head_dim_qk, head_dim_v]
