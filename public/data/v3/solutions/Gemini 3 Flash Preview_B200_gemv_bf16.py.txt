
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def gemv_kernel(
    y_ptr,
    x_ptr,
    w_ptr,
    M,
    N,
    K,
    stride_ym, stride_yn,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    # This kernel handles Batch GEMV (M, N, K) 
    # M is the batch dimension (often small, e.g., 32)
    # N is out_features
    # K is in_features
    
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n
    
    offs_m = pid_m
    offs_n = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    # Pointers
    x_ptrs = x_ptr + offs_m * stride_xm + offs_k * stride_xk
    w_ptrs = w_ptr + offs_n[:, None] * stride_wn + offs_k[None, :] * stride_wk
    
    acc = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
    
    for k in range(0, K, BLOCK_SIZE_K):
        # Load x: [BLOCK_SIZE_K]
        x = tl.load(x_ptr + offs_m * stride_xm + (k + tl.arange(0, BLOCK_SIZE_K)) * stride_xk, mask=(k + tl.arange(0, BLOCK_SIZE_K)) < K, other=0.0)
        # Load w: [BLOCK_SIZE_N, BLOCK_SIZE_K]
        w = tl.load(w_ptr + offs_n[:, None] * stride_wn + (k + tl.arange(0, BLOCK_SIZE_K))[None, :] * stride_wk, mask=(offs_n[:, None] < N) & (k + tl.arange(0, BLOCK_SIZE_K)[None, :] < K), other=0.0)
        
        # Multiply and accumulate
        acc += tl.sum(x[None, :].to(tl.float32) * w.to(tl.float32), axis=1)
        
    y_ptrs = y_ptr + offs_m * stride_ym + offs_n * stride_yn
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=offs_n < N)

def gemv_bf16(x, weight):
    # x shape: [M, K]
    # weight shape: [N, K]
    # output shape: [M, N]
    M, K = x.shape
    N, K_w = weight.shape
    assert K == K_w, "Incompatible dimensions"
    
    y = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)
    
    # Tuning parameters for B200 gemv-like workload
    BLOCK_SIZE_N = 32
    BLOCK_SIZE_K = 512
    num_warps = 8
    
    grid = (M * triton.cdiv(N, BLOCK_SIZE_N),)
    
    gemv_kernel[grid](
        y, x, weight,
        M, N, K,
        y.stride(0), y.stride(1),
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        BLOCK_SIZE_N=BLOCK_SIZE_N,
        BLOCK_SIZE_K=BLOCK_SIZE_K,
        GROUP_SIZE_M=1,
        num_warps=num_warps
    )
    return y

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Reference is x @ weight.t()
        # x: [32, 4096], weight: [14336, 4096]
        return gemv_bf16(x, self.weight)

def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16)]

def get_init_inputs():
    return []
