
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def layernorm_kernel(
    X, Y, W, B, N, eps,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0).to(tl.int64)
    X_ptr = X + row_idx * N
    Y_ptr = Y + row_idx * N

    # Loading into registers if N <= BLOCK_SIZE
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < N
    x = tl.load(X_ptr + cols, mask=mask, other=0.0).to(tl.float32)
    
    # Simple one-pass reduction within the block
    mean = tl.sum(x, axis=0) / N
    x_centered = tl.where(mask, x - mean, 0.0)
    var = tl.sum(x_centered * x_centered, axis=0) / N
    rstd = 1.0 / tl.sqrt(var + eps)
    
    w = tl.load(W + cols, mask=mask).to(tl.float32)
    b = tl.load(B + cols, mask=mask).to(tl.float32)
    y = x_centered * rstd * w + b
    tl.store(Y_ptr + cols, y.to(X.dtype.element_ty), mask=mask)

@triton.jit
def layernorm_kernel_tiled(
    X, Y, W, B, N, eps,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0).to(tl.int64)
    X_ptr = X + row_idx * N
    Y_ptr = Y + row_idx * N

    # Compute mean
    sum_val = 0.0
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        x = tl.load(X_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        sum_val += tl.sum(x, axis=0)
    mean = sum_val / N

    # Compute variance sum
    var_sum = 0.0
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        x = tl.load(X_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        diff = tl.where(mask, x - mean, 0.0)
        var_sum += tl.sum(diff * diff, axis=0)
    var = var_sum / N
    rstd = 1.0 / tl.sqrt(var + eps)

    # Normalize and store
    for off in range(0, N, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        x = tl.load(X_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        w = tl.load(W + cols, mask=mask).to(tl.float32)
        b = tl.load(B + cols, mask=mask).to(tl.float32)
        y = (x - mean) * rstd * w + b
        tl.store(Y_ptr + cols, y.to(X.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        if isinstance(normalized_shape, int):
            normalized_shape = (normalized_shape,)
        self.normalized_shape = tuple(normalized_shape)
        self.weight = nn.Parameter(torch.ones(self.normalized_shape))
        self.bias = nn.Parameter(torch.zeros(self.normalized_shape))
        
        self.n_elements = 1
        for s in self.normalized_shape:
            self.n_elements *= s

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        input_shape = x.shape
        N = self.n_elements
        M = x.numel() // N
        
        x_reshaped = x.view(M, N)
        y = torch.empty_like(x_reshaped)
        
        weight_flat = self.weight.view(-1)
        bias_flat = self.bias.view(-1)
        
        # Max tensor size for Triton is 1048576. 
        # If N exceeds this, we Must use tiled kernel.
        # If N <= 1024*8, we can use the single-pass kernel efficiently.
        
        grid = (M,)
        if N <= 1024*8:
            BLOCK_SIZE = triton.next_power_of_2(N)
            layernorm_kernel[grid](
                x_reshaped, y, weight_flat, bias_flat, N, 1e-5,
                BLOCK_SIZE=BLOCK_SIZE, num_warps=min(32, BLOCK_SIZE // 32 if BLOCK_SIZE > 32 else 1)
            )
        else:
            # For large N, use tiled. 
            # Note: Triton's arange limit is 1048576.
            # We use a manageable BLOCK_SIZE.
            BLOCK_SIZE = 4096 
            layernorm_kernel_tiled[grid](
                x_reshaped, y, weight_flat, bias_flat, N, 1e-5,
                BLOCK_SIZE=BLOCK_SIZE, num_warps=8
            )
        
        return y.view(input_shape)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]

def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
