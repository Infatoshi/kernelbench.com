import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <cuda_fp16.h>

// Shared memory tiled max pooling for better memory access
template<typename T, int KERNEL_SIZE, int DILATION, int TILE_OUT_H, int TILE_OUT_W>
__global__ void __launch_bounds__(256) maxpool2d_kernel_tiled(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int input_height,
    const int input_width,
    const int output_height,
    const int output_width,
    const int stride,
    const int padding
) {
    // Shared memory for input tile
    // Need to load enough input to cover all output tiles with the kernel
    const int tile_input_h = TILE_OUT_H * stride + (KERNEL_SIZE - 1) * DILATION;
    const int tile_input_w = TILE_OUT_W * stride + (KERNEL_SIZE - 1) * DILATION;
    
    __shared__ float shared_input[TILE_OUT_H * stride + (KERNEL_SIZE - 1) * DILATION]
                                  [TILE_OUT_W * stride + (KERNEL_SIZE - 1) * DILATION];
    
    const int oh_tile = blockIdx.y * TILE_OUT_H;
    const int ow_tile = blockIdx.z * TILE_OUT_W;
    const int bc = blockIdx.x;
    
    const int b = bc / channels;
    const int c = bc % channels;
    
    // Calculate the input region we need
    const int ih_base = oh_tile * stride - padding;
    const int iw_base = ow_tile * stride - padding;
    
    // Cooperatively load input tile into shared memory
    const int threads_per_block = blockDim.y * blockDim.z;
    const int thread_id = threadIdx.y * blockDim.z + threadIdx.z;
    
    for (int i = thread_id; i < tile_input_h * tile_input_w; i += threads_per_block) {
        const int local_ih = i / tile_input_w;
        const int local_iw = i % tile_input_w;
        const int global_ih = ih_base + local_ih;
        const int global_iw = iw_base + local_iw;
        
        float val = -FLT_MAX;
        if (global_ih >= 0 && global_ih < input_height && global_iw >= 0 && global_iw < input_width) {
            const int input_idx = ((b * channels + c) * input_height + global_ih) * input_width + global_iw;
            val = static_cast<float>(input[input_idx]);
        }
        shared_input[local_ih][local_iw] = val;
    }
    
    __syncthreads();
    
    // Each thread computes one output element
    const int oh_local = threadIdx.y;
    const int ow_local = threadIdx.z;
    const int oh = oh_tile + oh_local;
    const int ow = ow_tile + ow_local;
    
    if (oh >= output_height || ow >= output_width) return;
    
    // Compute max from shared memory
    const int shared_ih_base = oh_local * stride;
    const int shared_iw_base = ow_local * stride;
    
    float max_val = -FLT_MAX;
    
    #pragma unroll
    for (int kh = 0; kh < KERNEL_SIZE; kh++) {
        const int shared_ih = shared_ih_base + kh * DILATION;
        #pragma unroll
        for (int kw = 0; kw < KERNEL_SIZE; kw++) {
            const int shared_iw = shared_iw_base + kw * DILATION;
            const float val = shared_input[shared_ih][shared_iw];
            max_val = fmaxf(max_val, val);
        }
    }
    
    const int output_idx = ((b * channels + c) * output_height + oh) * output_width + ow;
    output[output_idx] = static_cast<T>(max_val);
}

// Optimized simple kernel with loop unrolling
template<typename T, int KERNEL_SIZE, int DILATION, int N_OUTPUTS>
__global__ void __launch_bounds__(256) maxpool2d_kernel_multi(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int input_height,
    const int input_width,
    const int output_height,
    const int output_width,
    const int stride,
    const int padding
) {
    const int ow_base = (blockIdx.y * blockDim.y + threadIdx.y) * N_OUTPUTS;
    const int oh = blockIdx.z * blockDim.z + threadIdx.z;
    const int bc = blockIdx.x;
    
    if (oh >= output_height || bc >= batch_size * channels) return;
    
    const int b = bc / channels;
    const int c = bc % channels;
    
    float max_vals[N_OUTPUTS];
    #pragma unroll
    for (int i = 0; i < N_OUTPUTS; i++) {
        max_vals[i] = -FLT_MAX;
    }
    
    const int ih_base = oh * stride - padding;
    
    #pragma unroll
    for (int kh = 0; kh < KERNEL_SIZE; kh++) {
        const int ih = ih_base + kh * DILATION;
        if (ih >= 0 && ih < input_height) {
            #pragma unroll
            for (int kw = 0; kw < KERNEL_SIZE; kw++) {
                #pragma unroll
                for (int i = 0; i < N_OUTPUTS; i++) {
                    const int ow = ow_base + i;
                    if (ow >= output_width) continue;
                    
                    const int iw = (ow * stride - padding) + kw * DILATION;
                    if (iw >= 0 && iw < input_width) {
                        const int input_idx = ((b * channels + c) * input_height + ih) * input_width + iw;
                        const float val = static_cast<float>(input[input_idx]);
                        max_vals[i] = fmaxf(max_vals[i], val);
                    }
                }
            }
        }
    }
    
    #pragma unroll
    for (int i = 0; i < N_OUTPUTS; i++) {
        const int ow = ow_base + i;
        if (ow < output_width) {
            const int output_idx = ((b * channels + c) * output_height + oh) * output_width + ow;
            output[output_idx] = static_cast<T>(max_vals[i]);
        }
    }
}

// Generic kernel for any parameters
template<typename T>
__global__ void __launch_bounds__(256) maxpool2d_kernel_generic(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int input_height,
    const int input_width,
    const int output_height,
    const int output_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation
) {
    const int ow = blockIdx.y * blockDim.y + threadIdx.y;
    const int oh = blockIdx.z * blockDim.z + threadIdx.z;
    const int bc = blockIdx.x;
    
    if (ow >= output_width || oh >= output_height || bc >= batch_size * channels) return;
    
    const int b = bc / channels;
    const int c = bc % channels;
    
    const int ih_base = oh * stride - padding;
    const int iw_base = ow * stride - padding;
    
    float max_val = -FLT_MAX;
    
    for (int kh = 0; kh < kernel_size; kh++) {
        const int ih = ih_base + kh * dilation;
        if (ih >= 0 && ih < input_height) {
            for (int kw = 0; kw < kernel_size; kw++) {
                const int iw = iw_base + kw * dilation;
                if (iw >= 0 && iw < input_width) {
                    const int input_idx = ((b * channels + c) * input_height + ih) * input_width + iw;
                    const float val = static_cast<float>(input[input_idx]);
                    max_val = fmaxf(max_val, val);
                }
            }
        }
    }
    
    const int output_idx = ((b * channels + c) * output_height + oh) * output_width + ow;
    output[output_idx] = static_cast<T>(max_val);
}

torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int input_height = input.size(2);
    const int input_width = input.size(3);
    
    // Calculate output dimensions
    const int output_height = (input_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    const int output_width = (input_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, output_height, output_width}, input.options());
    
    const int bc_size = batch_size * channels;
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "maxpool2d_cuda", ([&] {
        if (kernel_size == 2 && dilation == 3) {
            // Use multi-output kernel for best performance
            const int N_OUTPUTS = 4;
            dim3 blockDim(1, 16, 16);  // 256 threads
            dim3 gridDim(
                bc_size,
                (output_width + blockDim.y * N_OUTPUTS - 1) / (blockDim.y * N_OUTPUTS),
                (output_height + blockDim.z - 1) / blockDim.z
            );
            maxpool2d_kernel_multi<scalar_t, 2, 3, N_OUTPUTS><<<gridDim, blockDim>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                input_height, input_width,
                output_height, output_width,
                stride, padding
            );
        } else {
            // Generic kernel for other configurations
            dim3 blockDim(1, 16, 16);
            dim3 gridDim(
                bc_size,
                (output_width + blockDim.y - 1) / blockDim.y,
                (output_height + blockDim.z - 1) / blockDim.z
            );
            maxpool2d_kernel_generic<scalar_t><<<gridDim, blockDim>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels,
                input_height, input_width,
                output_height, output_width,
                kernel_size, stride, padding, dilation
            );
        }
    }));
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor maxpool2d_cuda(torch::Tensor input, int kernel_size, int stride, int padding, int dilation);
'''

ext = load_inline(
    name='maxpool2d_ext3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['maxpool2d_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    """
    Simple model that performs Max Pooling 2D.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.

        Args:
            kernel_size (int): Size of the pooling window.
            stride (int): Stride of the pooling window.
            padding (int): Padding to be applied before pooling.
            dilation (int): Spacing between kernel elements.
        """
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Max Pooling 2D to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, channels, height, width).

        Returns:
            torch.Tensor: Output tensor after Max Pooling 2D, shape (batch_size, channels, pooled_height, pooled_width).
        """
        return ext.maxpool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)

batch_size = 16
channels = 32
height = 512
width = 512
kernel_size = 2
stride = 2
padding = 1
dilation = 3

def get_inputs():
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    return [kernel_size, stride, padding, dilation]
