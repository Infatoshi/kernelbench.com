import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        # Use a more efficient convolution if possible. 
        # But nn.Conv3d is already using cuDNN.
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool = nn.MaxPool3d(kernel_size=4, stride=4)

    def forward(self, x):
        # We can try to use torch.empty and then use fused softmax/pool if we have a better kernel.
        # But for now, let's look at the sequence.
        x = self.conv(x)
        # Note: torch.softmax(x, dim=1).max_pool3d(kernel_size=4, stride=4)
        x = torch.softmax(x, dim=1)
        x = self.pool(x)
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width, device='cuda')]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
