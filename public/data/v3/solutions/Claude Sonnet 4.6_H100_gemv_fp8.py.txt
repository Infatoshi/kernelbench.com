"""
Optimized FP8-like GEMV using INT8 dot product (dp4a) with shared memory tiling.
Matrix: weight_q [out_features=14336, in_features=4096] INT8
Input:  x_q [batch=32, in_features=4096] INT8
Output: [batch=32, out_features=14336] FP16

Strategy:
- Use CUTLASS INT8 GEMM via load_inline for peak INT8 throughput
- Apply combined scale (x_scale * weight_scale) as a post-processing step
- The combined_scale = x_scale * weight_scale converts INT32 accumulator -> FP16
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <stdint.h>

// Optimized INT8 GEMV kernel using dp4a (int8 dot product accumulate 4 elements)
// Computes: C[M, N] = A[M, K] (int8) * B[N, K]^T (int8), then scale -> fp16
// A: x_q    [M x K]  int8, row-major
// B: weight_q [N x K] int8, row-major (weight stored as [out, in])
// C: output [M x N]  fp16

// Tile: each warp handles multiple output rows and one output column block
// We use dp4a: __dp4a(a, b, c) -> c + dot(a[4bytes], b[4bytes])

#define TILE_K 64   // K-dimension tile size (must be multiple of 4 for dp4a)
#define BLOCK_N 4   // output rows per thread block (in N dimension)

// Each thread block handles: blockDim.y output rows of N, blockDim.x threads reduce K
// Grid: (N / BLOCK_N_TILE, M) -- but we'll do one N per block, many K threads

// Kernel: each block computes one output element C[m, n] accumulated over K
// But that's inefficient. Better: each block handles [BM x BN] output tile.

// Let's use a simple but effective approach:
// - Grid: (ceil(N/32), M)  -- each block handles 32 consecutive N values for one M row
// - Each block: 32 threads, each thread computes one (m, n) output
//   by iterating over K with dp4a accumulation
// This gives good parallelism and uses dp4a efficiently

// Better: tile over K with shared memory
// Block: (128, 1) threads  
// Grid: (N, M/1) - each block handles one m row, one n column, all K
// Too many blocks. Let's do BN=16 per block.

// Final approach: 
// Grid: (N / BN, M)  where BN=32
// Block: (BN=32, BT=4) -- BT threads cooperate on K reduction for each n
// Shared memory: intermediate int32 partial sums
// Each thread: iterates K/BT elements with dp4a, then tree reduction

// Actually simplest effective approach for M=32, N=14336, K=4096:
// Grid: (N, 1) = 14336 blocks
// Block: 128 threads, reduce over K=4096 for all 32 M rows in a loop
// Each thread handles K/128 = 32 elements for one (m, n) pair
// But M=32, so each block actually loops over m.

// Better: 
// Grid: (N / BN, M / BM) where BN * BM is computed per block
// Block does BM x BN output tile, reducing over K with shared memory

// For M=32, N=14336, K=4096:
// Let BN=32, BM=1 (or BM=2)
// Grid: (14336/32, 32) = (448, 32)
// Block: 128 threads
// Each block computes [BM x BN] = [1 x 32] outputs
// Each thread handles K/4 dp4a operations per output element... 
// That's 4096/4 = 1024 per element, but 32 outputs means 32768 dp4a per thread
// With 128 threads, total work = 128 * ops... 

// Simplest working approach: one thread per output element
// Grid: (ceil(N/BN), M), Block: (BN, BM)
// BN=32, BM=4, so 128 threads per block
// Each thread computes ONE output element, iterates K=4096 with dp4a

// Load A row and B row into registers/shared memory, use __dp4a

__global__ void int8_gemv_kernel(
    const int8_t* __restrict__ A,   // [M, K]
    const int8_t* __restrict__ B,   // [N, K] 
    __half* __restrict__ C,         // [M, N]
    const float combined_scale,
    int M, int N, int K
) {
    // Each thread computes one C[m, n]
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    int m = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (m >= M || n >= N) return;
    
    const int8_t* a_row = A + m * K;
    const int8_t* b_row = B + n * K;
    
    int32_t acc = 0;
    
    // Process K in chunks of 4 using dp4a
    int k4 = K / 4;
    const int32_t* a4 = reinterpret_cast<const int32_t*>(a_row);
    const int32_t* b4 = reinterpret_cast<const int32_t*>(b_row);
    
    #pragma unroll 8
    for (int k = 0; k < k4; k++) {
        acc = __dp4a(a4[k], b4[k], acc);
    }
    
    // Scale and convert to fp16
    C[m * N + n] = __float2half(acc * combined_scale);
}

// Optimized version with shared memory tiling for better cache utilization
// Tile K dimension in shared memory
#define TILE_SIZE 64

__global__ void int8_gemv_tiled_kernel(
    const int8_t* __restrict__ A,   // [M, K]
    const int8_t* __restrict__ B,   // [N, K] 
    __half* __restrict__ C,         // [M, N]
    const float combined_scale,
    int M, int N, int K
) {
    // Block: (BN=32, BM=4) = 128 threads
    // Each thread handles one (m, n) pair
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    int m = blockIdx.y * blockDim.y + threadIdx.y;
    
    // Shared memory for A tile: [BM x TILE_SIZE] = [4 x 64]
    // Shared memory for B tile: [BN x TILE_SIZE] = [32 x 64]
    __shared__ int8_t sA[4][TILE_SIZE];   // BM x TILE_SIZE
    __shared__ int8_t sB[32][TILE_SIZE];  // BN x TILE_SIZE
    
    int32_t acc = 0;
    
    int local_m = threadIdx.y;  // 0..3
    int local_n = threadIdx.x;  // 0..31
    
    for (int k_tile = 0; k_tile < K; k_tile += TILE_SIZE) {
        // Load A tile: thread (local_n, local_m) loads sA[local_m][...]
        // Each row of A needs TILE_SIZE bytes, load cooperatively
        // 128 threads load 4*64=256 bytes of A
        int a_m_idx = blockIdx.y * blockDim.y + local_m;
        if (local_n < TILE_SIZE/4) {  // 16 threads per row of A
            int k_idx = k_tile + local_n * 4;
            if (a_m_idx < M && k_idx + 3 < K) {
                *reinterpret_cast<int32_t*>(&sA[local_m][local_n*4]) = 
                    *reinterpret_cast<const int32_t*>(&A[a_m_idx * K + k_idx]);
            } else if (a_m_idx < M) {
                for (int i = 0; i < 4 && k_idx + i < K; i++)
                    sA[local_m][local_n*4 + i] = A[a_m_idx * K + k_idx + i];
            }
        }
        
        // Load B tile: thread (local_n, local_m) loads sB[local_n][...]
        // 32 threads * 4 rows = 128 threads load 32*64=2048 bytes of B
        int b_n_idx = blockIdx.x * blockDim.x + local_n;
        // Use local_m to load TILE_SIZE/4 = 16 elements per thread for each B row
        for (int i = local_m; i < TILE_SIZE/4; i += blockDim.y) {
            int k_idx = k_tile + i * 4;
            if (b_n_idx < N && k_idx + 3 < K) {
                *reinterpret_cast<int32_t*>(&sB[local_n][i*4]) = 
                    *reinterpret_cast<const int32_t*>(&B[b_n_idx * K + k_idx]);
            }
        }
        
        __syncthreads();
        
        // Compute dot product for this tile
        if (m < M && n < N) {
            const int32_t* sa4 = reinterpret_cast<const int32_t*>(sA[local_m]);
            const int32_t* sb4 = reinterpret_cast<const int32_t*>(sB[local_n]);
            int tile_k4 = TILE_SIZE / 4;
            #pragma unroll
            for (int k = 0; k < tile_k4; k++) {
                acc = __dp4a(sa4[k], sb4[k], acc);
            }
        }
        
        __syncthreads();
    }
    
    if (m < M && n < N) {
        C[m * N + n] = __float2half(acc * combined_scale);
    }
}


// High-performance version: 
// - Multiple M rows per thread block for better reuse of B rows
// - Vectorized loading with int4 (16 bytes at once)
// - Warp-level parallelism with __dp4a

// Block: (WARP_SIZE=32, WARPS=4) = 128 threads
// Grid: (N/BN, M/BM) where BN=32, BM=4
// Each warp (32 threads) handles BN=32 outputs for one M row
// So 4 warps handle 4 M rows simultaneously

__global__ void int8_gemv_fast_kernel(
    const int8_t* __restrict__ A,   // [M, K] row-major
    const int8_t* __restrict__ B,   // [N, K] row-major  
    __half* __restrict__ C,         // [M, N]
    const float combined_scale,
    int M, int N, int K
) {
    const int BN = 32;  // blockDim.x
    const int BM = 4;   // blockDim.y
    
    int n = blockIdx.x * BN + threadIdx.x;
    int m_base = blockIdx.y * BM;
    int m = m_base + threadIdx.y;
    
    if (n >= N) return;
    
    // Each thread accumulates K dot-product elements
    int32_t acc = 0;
    
    const int8_t* b_row = B + n * K;
    const int8_t* a_row = (m < M) ? A + m * K : nullptr;
    
    if (m < M) {
        const int32_t* a4 = reinterpret_cast<const int32_t*>(a_row);
        const int32_t* b4 = reinterpret_cast<const int32_t*>(b_row);
        int k4 = K / 4;
        
        #pragma unroll 16
        for (int k = 0; k < k4; k++) {
            acc = __dp4a(a4[k], b4[k], acc);
        }
        
        C[m * N + n] = __float2half(acc * combined_scale);
    }
}

// Most optimized: use vectorized loads (int4 = 16 bytes) and maximize ILP
// Each thread computes ONE output, but loads 16 bytes at a time
__global__ void int8_gemv_vec16_kernel(
    const int8_t* __restrict__ A,   // [M, K] row-major
    const int8_t* __restrict__ B,   // [N, K] row-major  
    __half* __restrict__ C,         // [M, N]
    const float combined_scale,
    int M, int N, int K
) {
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    int m = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (m >= M || n >= N) return;
    
    const int32_t* a4 = reinterpret_cast<const int32_t*>(A + m * K);
    const int32_t* b4 = reinterpret_cast<const int32_t*>(B + n * K);
    
    int32_t acc0 = 0, acc1 = 0, acc2 = 0, acc3 = 0;
    int k4 = K / 4;
    
    // 4-way unroll for ILP
    #pragma unroll 4
    for (int k = 0; k < k4; k += 4) {
        if (k + 3 < k4) {
            acc0 = __dp4a(a4[k+0], b4[k+0], acc0);
            acc1 = __dp4a(a4[k+1], b4[k+1], acc1);
            acc2 = __dp4a(a4[k+2], b4[k+2], acc2);
            acc3 = __dp4a(a4[k+3], b4[k+3], acc3);
        } else {
            for (int kk = k; kk < k4; kk++) {
                acc0 = __dp4a(a4[kk], b4[kk], acc0);
            }
        }
    }
    
    int32_t acc = (acc0 + acc1) + (acc2 + acc3);
    C[m * N + n] = __float2half(acc * combined_scale);
}


torch::Tensor int8_scaled_gemv(
    torch::Tensor A,        // [M, K] int8
    torch::Tensor B,        // [N, K] int8
    float x_scale,
    float w_scale
) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(0);
    
    auto C = torch::empty({M, N}, torch::dtype(torch::kFloat16).device(A.device()));
    
    float combined_scale = x_scale * w_scale;
    
    // Use 32x4=128 threads per block
    dim3 block(32, 4);
    dim3 grid((N + 31) / 32, (M + 3) / 4);
    
    int8_gemv_vec16_kernel<<<grid, block>>>(
        A.data_ptr<int8_t>(),
        B.data_ptr<int8_t>(),
        reinterpret_cast<__half*>(C.data_ptr()),
        combined_scale,
        M, N, K
    );
    
    return C;
}
'''

ext = load_inline(
    name='int8_gemv_ext',
    cpp_sources='torch::Tensor int8_scaled_gemv(torch::Tensor A, torch::Tensor B, float x_scale, float w_scale);',
    cuda_sources=cuda_source,
    functions=['int8_scaled_gemv'],
    verbose=True,
    extra_cuda_cflags=['-O3', '-arch=sm_90', '--use_fast_math', '-lineinfo'],
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.int8_scaled_gemv(
            x_q,
            self.weight_q,
            x_scale.item(),
            self.weight_scale.item()
        )


OP_TYPE = "gemv"
SUPPORTED_PRECISIONS = ["fp8"]
HARDWARE_REQUIRED = ["H100", "B200"]
SPECIALIZED_LEVEL = 1


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
