"""
GEMM + Bias + ReLU fusion optimized for H100.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>

using namespace nvcuda::wmma;

// Optimized kernel using WMMA for tensor cores
__global__ void __launch_bounds__(256) gemm_bias_relu_wmma(
    const __half* __restrict__ A,
    const __half* __restrict__ B,
    const __half* __restrict__ bias,
    __half* __restrict__ C,
    int M, int N, int K) {
    
    // Tile dimensions
    const int WMMA_M = 16;
    const int WMMA_N = 16;
    const int WMMA_K = 16;
    
    const int BLOCK_M = 128;
    const int BLOCK_N = 128;
    const int BLOCK_K = 16;
    
    // Shared memory with padding to avoid bank conflicts
    __shared__ __half A_shared[BLOCK_M][BLOCK_K + 8];
    __shared__ __half B_shared[BLOCK_K][BLOCK_N + 8];
    
    // Block coordinates
    int block_m = blockIdx.y;
    int block_n = blockIdx.x;
    
    int base_m = block_m * BLOCK_M;
    int base_n = block_n * BLOCK_N;
    
    if (base_m >= M || base_n >= N) return;
    
    // Warp and lane IDs
    int warp_id = (threadIdx.y * blockDim.x + threadIdx.x) / 32;
    
    // Each warp handles 32x32 output (2x2 WMMA tiles)
    // 4 warps per block in each dimension (16 warps total)
    int warp_row = warp_id / 4;  // 0-3
    int warp_col = warp_id % 4;  // 0-3
    
    // Warp's starting position in the output tile
    int warp_m = warp_row * 32;  // 0, 32, 64, 96
    int warp_n = warp_col * 32;  // 0, 32, 64, 96
    
    // Accumulator fragments (4 WMMA tiles per warp: 2x2)
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc[2][2];
    
    // Initialize accumulators to zero
    for (int i = 0; i < 2; ++i) {
        for (int j = 0; j < 2; ++j) {
            fill_fragment(acc[i][j], 0.0f);
        }
    }
    
    // Loop over K in tiles
    for (int k_tile = 0; k_tile < (K + BLOCK_K - 1) / BLOCK_K; ++k_tile) {
        int base_k = k_tile * BLOCK_K;
        
        // Cooperative load of tiles to shared memory
        int tid = threadIdx.y * blockDim.x + threadIdx.x;
        
        // Load A tile: BLOCK_M x BLOCK_K
        for (int i = tid; i < BLOCK_M * BLOCK_K; i += 256) {
            int row = i / BLOCK_K;
            int col = i % BLOCK_K;
            int global_row = base_m + row;
            int global_col = base_k + col;
            if (global_row < M && global_col < K) {
                A_shared[row][col] = A[global_row * K + global_col];
            } else {
                A_shared[row][col] = __float2half(0.0f);
            }
        }
        
        // Load B tile: BLOCK_K x BLOCK_N
        for (int i = tid; i < BLOCK_K * BLOCK_N; i += 256) {
            int row = i / BLOCK_N;
            int col = i % BLOCK_N;
            int global_row = base_k + row;
            int global_col = base_n + col;
            if (global_row < K && global_col < N) {
                B_shared[row][col] = B[global_row * N + global_col];
            } else {
                B_shared[row][col] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute WMMA operations
        // Each warp computes 2x2 WMMA tiles
        // Load A and B fragments
        fragment<matrix_a, WMMA_M, WMMA_N, WMMA_K, half, row_major> a_frag[2];
        fragment<matrix_b, WMMA_M, WMMA_N, WMMA_K, half, col_major> b_frag[2];
        
        // Load A fragments (two rows of WMMA tiles)
        for (int i = 0; i < 2; ++i) {
            load_matrix_sync(a_frag[i], &A_shared[warp_m + i * WMMA_M][0], BLOCK_K + 8);
        }
        
        // Load B fragments (two columns of WMMA tiles)
        for (int j = 0; j < 2; ++j) {
            load_matrix_sync(b_frag[j], &B_shared[0][warp_n + j * WMMA_N], BLOCK_N + 8);
        }
        
        // Perform WMMA operations
        for (int i = 0; i < 2; ++i) {
            for (int j = 0; j < 2; ++j) {
                mma_sync(acc[i][j], a_frag[i], b_frag[j], acc[i][j]);
            }
        }
        
        __syncthreads();
    }
    
    // Store results with bias + ReLU
    // Use store_matrix_sync to write out each accumulator tile
    __shared__ float result_shared[BLOCK_M][BLOCK_N];
    
    // Each warp stores its accumulator to shared memory
    for (int i = 0; i < 2; ++i) {
        for (int j = 0; j < 2; ++j) {
            int tile_m = warp_m + i * WMMA_M;
            int tile_n = warp_n + j * WMMA_N;
            
            // Store accumulator to shared memory
            store_matrix_sync(&result_shared[tile_m][tile_n], acc[i][j], BLOCK_N, mem_row_major);
        }
    }
    
    __syncthreads();
    
    // Apply bias + ReLU and write to global memory
    // Each thread handles multiple output elements
    int tid = threadIdx.y * blockDim.x + threadIdx.x;
    for (int idx = tid; idx < BLOCK_M * BLOCK_N; idx += 256) {
        int local_row = idx / BLOCK_N;
        int local_col = idx % BLOCK_N;
        
        int global_row = base_m + local_row;
        int global_col = base_n + local_col;
        
        if (global_row < M && global_col < N) {
            float val = result_shared[local_row][local_col];
            val += __half2float(bias[global_col]);
            val = fmaxf(0.0f, val);
            C[global_row * N + global_col] = __float2half(val);
        }
    }
}

torch::Tensor gemm_bias_relu_cuda(torch::Tensor A, torch::Tensor B, torch::Tensor bias) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    const int BLOCK_M = 128;
    const int BLOCK_N = 128;
    
    dim3 block(64, 4);  // 256 threads total
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);
    
    gemm_bias_relu_wmma<<<grid, block>>>(
        reinterpret_cast<__half*>(A.data_ptr<at::Half>()),
        reinterpret_cast<__half*>(B.data_ptr<at::Half>()),
        reinterpret_cast<__half*>(bias.data_ptr<at::Half>()),
        reinterpret_cast<__half*>(C.data_ptr<at::Half>()),
        M, N, K
    );
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemm_bias_relu_cuda(torch::Tensor A, torch::Tensor B, torch::Tensor bias);
'''

ext = load_inline(
    name='gemm_bias_relu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_bias_relu_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_90', '-use_fast_math'],
    verbose=True,
)

class Model(nn.Module):
    def __init__(self, n: int = 4096):
        super().__init__()
        self.bias = nn.Parameter(torch.randn(n, dtype=torch.float16) * 0.02)

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        a_fp16 = a.to(torch.float16)
        b_fp16 = b.to(torch.float16)
        return ext.gemm_bias_relu_cuda(a_fp16, b_fp16, self.bias)

def get_inputs():
    m = 2048
    n = 4096
    k = 2048
    return [torch.randn(m, k, dtype=torch.float16), torch.randn(k, n, dtype=torch.float16)]
