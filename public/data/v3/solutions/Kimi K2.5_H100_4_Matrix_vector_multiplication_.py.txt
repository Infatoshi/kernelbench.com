import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = '''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

template <typename scalar_t>
__global__ void gemv_kernel(const scalar_t* A, const scalar_t* B, scalar_t* C, int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    scalar_t sum = 0;
    for (int k = threadIdx.x; k < K; k += blockDim.x) {
        sum += A[row * K + k] * B[k];
    }
    
    // Shared memory for reduction
    extern __shared__ char shared_memory[];
    scalar_t* shared = reinterpret_cast<scalar_t*>(shared_memory);
    
    shared[threadIdx.x] = sum;
    __syncthreads();
    
    // Reduction
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            shared[threadIdx.x] += shared[threadIdx.x + stride];
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        C[row] = shared[0];
    }
}

torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::zeros({M, 1}, A.options());
    
    const int threads = 512;
    
    AT_DISPATCH_FLOATING_TYPES(A.scalar_type(), "gemv_cuda", ([&] {
        gemv_kernel<scalar_t><<<M, threads, threads * sizeof(scalar_t)>>>(
            A.data_ptr<scalar_t>(),
            B.data_ptr<scalar_t>(),
            C.data_ptr<scalar_t>(),
            M,
            K
        );
    }));
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemv_cuda(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_simple',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_cuda'],
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs optimized matrix-vector multiplication C = A @ B.
        
        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).
            
        Returns:
            Output vector of shape (M, 1).
        """
        return ext.gemv_cuda(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]
