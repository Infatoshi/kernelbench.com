import torch
import torch.nn as nn


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Ensure contiguous memory layout for better performance
        if not x.is_contiguous():
            x = x.contiguous()
        
        # Compute mean of squares along the feature dimension (dim=1)
        mean_sq = torch.mean(x * x, dim=1, keepdim=True)
        
        # Use rsqrt for efficiency: 1/sqrt(mean_sq + eps)
        inv_rms = torch.rsqrt(mean_sq + self.eps)
        
        # Element-wise multiplication (equivalent to x / rms)
        return x * inv_rms
