import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def rms_norm_kernel(
    X, Y,
    stride_batch, stride_feat, stride_d1, stride_d2,
    B, F, D1, D2,
    eps,
    BLOCK_SIZE: tl.constexpr
):
    # Each program handles one (b, d1, d2) location and reduces across F
    # Map program ID to (batch, d1, d2)
    pid = tl.program_id(0)
    
    # pid = b * (D1 * D2) + d1 * D2 + d2
    b = pid // (D1 * D2)
    rem_loc = pid % (D1 * D2)
    d1 = rem_loc // D2
    d2 = rem_loc % D2

    # Pointer to the start of the feature column for this (b, d1, d2)
    # X has shape (B, F, D1, D2)
    # offset = b*stride_batch + d1*stride_d1 + d2*stride_d2
    X_ptr = X + b * stride_batch + d1 * stride_d1 + d2 * stride_d2
    Y_ptr = Y + b * stride_batch + d1 * stride_d1 + d2 * stride_d2

    # Standard RMS computation
    # mean(x^2) along dimension F
    # Use float32 for accumulation
    var_sum = 0.0
    
    for f_off in range(0, F, BLOCK_SIZE):
        cols = f_off + tl.arange(0, BLOCK_SIZE)
        mask = cols < F
        x = tl.load(X_ptr + cols * stride_feat, mask=mask, other=0.0).to(tl.float32)
        var_sum += tl.sum(x * x, axis=0)
    
    rms = tl.math.sqrt(var_sum / F + eps)
    
    # Normalize and write back
    for f_off in range(0, F, BLOCK_SIZE):
        cols = f_off + tl.arange(0, BLOCK_SIZE)
        mask = cols < F
        x = tl.load(X_ptr + cols * stride_feat, mask=mask, other=0.0).to(tl.float32)
        out = x / rms
        tl.store(Y_ptr + cols * stride_feat, out.to(Y.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x shape: (batch_size, num_features, dim1, dim2)
        # We normalize along dim=1 (num_features)
        B, F, D1, D2 = x.shape
        y = torch.empty_like(x)
        
        # Grid covers all elements except the feature dimension
        grid = (B * D1 * D2,)
        
        # Heuristic for BLOCK_SIZE
        BLOCK_SIZE = triton.next_power_of_2(min(F, 1024))
        
        rms_norm_kernel[grid](
            x, y,
            x.stride(0), x.stride(1), x.stride(2), x.stride(3),
            B, F, D1, D2,
            self.eps,
            BLOCK_SIZE=BLOCK_SIZE
        )
        return y
