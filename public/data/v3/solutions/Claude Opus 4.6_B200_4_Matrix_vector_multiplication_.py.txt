import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

M = 256
K = 131072

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMV kernel: C[M,1] = A[M,K] * B[K,1]
// Each row handled by multiple warps. Grid: (M, NUM_BLOCKS_PER_ROW)
// Each block does partial reduction, then atomic add to output.

// Using float4 vectorized loads (128-bit)

template<int BLOCK_SIZE, int ELEMENTS_PER_THREAD>
__global__ void gemv_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M_val, int K_val
) {
    const int row = blockIdx.x;
    const int block_k = blockIdx.y;
    
    // Each block handles BLOCK_SIZE * ELEMENTS_PER_THREAD elements
    const int chunk_size = BLOCK_SIZE * ELEMENTS_PER_THREAD;
    const int k_start = block_k * chunk_size;
    
    const float* A_row = A + row * K_val + k_start;
    const float* B_start = B + k_start;
    
    float sum = 0.0f;
    
    // Vectorized loads using float4
    const int tid = threadIdx.x;
    const int vec_elements = ELEMENTS_PER_THREAD / 4;  // number of float4 loads per thread
    
    const float4* A_vec = reinterpret_cast<const float4*>(A_row);
    const float4* B_vec = reinterpret_cast<const float4*>(B_start);
    
    #pragma unroll
    for (int i = 0; i < vec_elements; i++) {
        int idx = tid * vec_elements + i;
        float4 a = A_vec[idx];
        float4 b = B_vec[idx];
        sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
    }
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_xor_sync(0xffffffff, sum, offset);
    }
    
    // Block reduction using shared memory
    __shared__ float warp_sums[32];  // max 32 warps per block
    
    const int warp_id = tid / 32;
    const int lane_id = tid % 32;
    
    if (lane_id == 0) {
        warp_sums[warp_id] = sum;
    }
    __syncthreads();
    
    // First warp reduces all warp sums
    if (warp_id == 0) {
        const int num_warps = BLOCK_SIZE / 32;
        sum = (lane_id < num_warps) ? warp_sums[lane_id] : 0.0f;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum += __shfl_xor_sync(0xffffffff, sum, offset);
        }
        if (lane_id == 0) {
            atomicAdd(C + row, sum);
        }
    }
}

torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B) {
    int M_val = A.size(0);
    int K_val = A.size(1);
    
    auto C = torch::zeros({M_val, 1}, A.options());
    
    constexpr int BLOCK_SIZE = 256;
    constexpr int ELEMENTS_PER_THREAD = 16;  // each thread handles 16 elements (4 float4)
    constexpr int CHUNK = BLOCK_SIZE * ELEMENTS_PER_THREAD;  // 256 * 16 = 4096
    
    int num_blocks_k = K_val / CHUNK;  // 131072 / 4096 = 32
    
    dim3 grid(M_val, num_blocks_k);
    dim3 block(BLOCK_SIZE);
    
    gemv_kernel<BLOCK_SIZE, ELEMENTS_PER_THREAD><<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M_val, K_val
    );
    
    return C;
}
'''

cpp_source = '''
torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        B_flat = B.view(-1)
        return ext.gemv_forward(A, B_flat)


def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
