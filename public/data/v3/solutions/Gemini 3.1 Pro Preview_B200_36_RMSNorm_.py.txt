import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename scalar_t>
__global__ void rms_norm_kernel_shmem(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int B, int F, int M, float eps)
{
    // Shared memory size is dynamically allocated: F * blockDim.x * sizeof(scalar_t)
    extern __shared__ char smem[];
    scalar_t* shared_x = reinterpret_cast<scalar_t*>(smem);

    int m_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int b_idx = blockIdx.y;
    int tid = threadIdx.x;
    
    if (m_idx < M && b_idx < B) {
        float sum_sq = 0.0f;
        int base_idx = b_idx * F * M + m_idx;
        
        // Load into shared memory and compute sum of squares
        for (int f = 0; f < F; ++f) {
            scalar_t val = x[base_idx + f * M];
            shared_x[f * blockDim.x + tid] = val;
            float fval = static_cast<float>(val);
            sum_sq += fval * fval;
        }
        
        float rms = sqrtf(sum_sq / static_cast<float>(F) + eps);
        float inv_rms = 1.0f / rms;
        
        // Read from shared memory and write to outputs
        for (int f = 0; f < F; ++f) {
            float fval = static_cast<float>(shared_x[f * blockDim.x + tid]);
            y[base_idx + f * M] = static_cast<scalar_t>(fval * inv_rms);
        }
    }
}

template <typename scalar_t>
__global__ void rms_norm_kernel_global(
    const scalar_t* __restrict__ x,
    scalar_t* __restrict__ y,
    int B, int F, int M, float eps)
{
    int m_idx = blockIdx.x * blockDim.x + threadIdx.x;
    int b_idx = blockIdx.y;
    
    if (m_idx < M && b_idx < B) {
        float sum_sq = 0.0f;
        int base_idx = b_idx * F * M + m_idx;
        
        // Compute sum of squares
        for (int f = 0; f < F; ++f) {
            float val = static_cast<float>(x[base_idx + f * M]);
            sum_sq += val * val;
        }
        
        float rms = sqrtf(sum_sq / static_cast<float>(F) + eps);
        float inv_rms = 1.0f / rms;
        
        // Write outputs
        for (int f = 0; f < F; ++f) {
            float val = static_cast<float>(x[base_idx + f * M]);
            y[base_idx + f * M] = static_cast<scalar_t>(val * inv_rms);
        }
    }
}

// BFloat16 requires AT_DISPATCH_FLOATING_TYPES_AND_HALF_AND_BF16, but PyTorch extension
// typically defines AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, ...)
torch::Tensor my_op(torch::Tensor x, float eps) {
    auto y = torch::empty_like(x);
    
    int B = x.size(0);
    int F = x.size(1);
    int M = 1;
    for (int i = 2; i < x.dim(); ++i) {
        M *= x.size(i);
    }
    
    int threads_per_block = 256;
    dim3 threads(threads_per_block);
    dim3 blocks((M + threads_per_block - 1) / threads_per_block, B);
    
    int element_size = x.element_size();
    size_t shmem_size = F * threads_per_block * element_size;
    size_t max_shmem = 49152; 
    
    if (shmem_size <= max_shmem) {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half, 
            at::ScalarType::BFloat16, 
            x.scalar_type(), "rms_norm_kernel_shmem", [&] {
            rms_norm_kernel_shmem<scalar_t><<<blocks, threads, shmem_size>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                B, F, M, eps
            );
        });
    } else {
        AT_DISPATCH_FLOATING_TYPES_AND2(
            at::ScalarType::Half, 
            at::ScalarType::BFloat16, 
            x.scalar_type(), "rms_norm_kernel_global", [&] {
            rms_norm_kernel_global<scalar_t><<<blocks, threads>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                B, F, M, eps
            );
        });
    }
    
    return y;
}
'''

cpp_source = r'''
torch::Tensor my_op(torch::Tensor x, float eps);
'''

ext = load_inline(
    name='my_ext3',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['my_op'],
    verbose=True,
    extra_cflags=['-O3'],
    extra_cuda_cflags=['-O3', '--use_fast_math']
)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.my_op(x, self.eps)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    return [torch.randn(batch_size, features, dim1, dim2, device='cuda')]

def get_init_inputs():
    features = 64
    return [features]
