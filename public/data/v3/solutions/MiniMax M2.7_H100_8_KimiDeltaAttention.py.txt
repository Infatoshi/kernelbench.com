cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cublas_v2.h>
#include <stdio.h>

// ============================================================================
// Fused CUDA Kernels for Kimi Delta Attention
// ============================================================================
// Key optimizations:
// 1. Fused SiLU + Linear projection (eliminates memory round-trip)
// 2. Fused sigmoid + element-wise multiply (reduces kernel launches)
// 3. Efficient memory access patterns
// ============================================================================

#define CUDA_CHECK(call) \
    do { \
        cudaError_t err = call; \
        if (err != cudaSuccess) { \
            fprintf(stderr, "CUDA error at %s:%d: %s\\n", __FILE__, __LINE__, \
                    cudaGetErrorString(err)); \
            exit(1); \
        } \
    } while(0)

// --------------------------------------------------------------------------
// Fused SiLU (Swish) + Linear: y = silu(x) @ W^T
// This fuses the activation with the projection in a single kernel launch
// --------------------------------------------------------------------------

template <int BLOCK_M, int BLOCK_N, int BLOCK_K>
__global__ void fused_silu_linear_kernel(
    const half* __restrict__ x,      // (M, K) - input
    const half* __restrict__ w,      // (N, K) - transposed weight [to avoid transpose]
    half* __restrict__ y,            // (M, N) - output
    int M, int N, int K,
    float scale) 
{
    // Block coordinates
    int by = blockIdx.y;
    int bx = blockIdx.x;
    
    // Thread coordinates within block
    int ty = threadIdx.y;
    int tx = threadIdx.x;
    
    // Stride for block moving through K dimension
    int k_stride = BLOCK_K * gridStrides3D().y;
    
    // Gemm state
    float acc[BLOCK_M][BLOCK_N] = {0.0f};
    
    // Pointer to current K tile of x and w
    const half* x_ptr = x + by * BLOCK_M * K + bx * BLOCK_K;
    const half* w_ptr = w + bx * BLOCK_N * K;  // w is (N, K)
    
    // Iterate over K tiles
    for (int k = 0; k < K; k += BLOCK_K) {
        // Declare shared memory for this block
        __shared__ half x_smem[BLOCK_M][BLOCK_K];
        __shared__ half w_smem[BLOCK_K][BLOCK_N];
        
        // Bounds check
        int k_remaining = min(BLOCK_K, K - k);
        
        // Load x tile: x_smem[ty][tx]
        for (int i = 0; i < BLOCK_M; i += 16) {
            for (int j = 0; j < BLOCK_K; j += 16) {
                int row = by * BLOCK_M + i + ty;
                int col = k + j + tx;
                if (row < M && col < K) {
                    x_smem[i + ty][j + tx] = x[row * K + col];
                } else {
                    x_smem[i + ty][j + tx] = __float2half(0.0f);
                }
            }
        }
        
        // Load w tile: w_smem[ty][tx]
        for (int i = 0; i < BLOCK_K; i += 16) {
            for (int j = 0; j < BLOCK_N; j += 16) {
                int row = k + i + ty;
                int col = bx * BLOCK_N + j + tx;
                if (row < K && col < N) {
                    w_smem[i + ty][j + tx] = w[row * N + col];
                } else {
                    w_smem[i + ty][j + tx] = __float2half(0.0f);
                }
            }
        }
        
        __syncthreads();
        
        // Compute partial gemm
        for (int kk = 0; kk < k_remaining; kk++) {
            // x_smem[ty][kk] and w_smem[kk][tx]
            half x_val = x_smem[ty][kk];
            
            // Apply SiLU: silu(x) = x / (1 + exp(-x))
            // Fast sigmoid approximation: 1 / (1 + exp(-x)) ≈ 0.5 + 0.5 * tanh(x)
            float f = __half2float(x_val);
            float sigmoid = 1.0f / (1.0f + expf(-f));
            float silu = f * sigmoid;
            half x_silu = __float2half(silu);
            
            // Multiply accumulate
            #pragma unroll
            for (int n = 0; n < BLOCK_N; n++) {
                acc[ty][n] += __half2float(x_silu) * __half2float(w_smem[kk][n]);
            }
        }
        
        __syncthreads();
    }
    
    // Store result
    #pragma unroll
    for (int i = 0; i < BLOCK_M; i++) {
        #pragma unroll
        for (int j = 0; j < BLOCK_N; j++) {
            int row = by * BLOCK_M + i * 16 + ty;
            int col = bx * BLOCK_N + j * 16 + tx;
            if (row < M && col < N) {
                y[row * N + col] = __float2half(acc[i][j] * scale);
            }
        }
    }
}

// --------------------------------------------------------------------------
// Fused Linear + SiLU kernel (when SiLU follows Linear)
// y = silu(x @ W^T)
// --------------------------------------------------------------------------

template <int BLOCK_M, int BLOCK_N, int BLOCK_K>
__global__ void fused_linear_silu_kernel(
    const half* __restrict__ x,      // (M, K)
    const half* __restrict__ w,      // (N, K) transposed
    half* __restrict__ y,            // (M, N)
    int M, int N, int K,
    float scale)
{
    int by = blockIdx.y;
    int bx = blockIdx.x;
    int ty = threadIdx.y;
    int tx = threadIdx.x;
    
    // Gemm state for thread tile
    float acc = 0.0f;
    
    const half* x_ptr = x + by * BLOCK_M * K + bx * BLOCK_K;
    const half* w_ptr = w + bx * BLOCK_N * K;
    
    // Iterate over K
    for (int k = 0; k < K; k += BLOCK_K) {
        __shared__ half x_smem[BLOCK_M][BLOCK_K];
        __shared__ half w_smem[BLOCK_K][BLOCK_N];
        
        int k_remaining = min(BLOCK_K, K - k);
        
        // Cooperative load x tile
        for (int i = 0; i < BLOCK_M * BLOCK_K; i += blockDim.x * blockDim.y) {
            int row = i / BLOCK_K;
            int col = i % BLOCK_K;
            int global_row = by * BLOCK_M + row;
            int global_col = k + col;
            if (global_row < M && global_col < K) {
                x_smem[row][col] = x[global_row * K + global_col];
            } else {
                x_smem[row][col] = __float2half(0.0f);
            }
        }
        
        // Cooperative load w tile
        for (int i = 0; i < BLOCK_K * BLOCK_N; i += blockDim.x * blockDim.y) {
            int row = i / BLOCK_N;
            int col = i % BLOCK_N;
            int global_row = k + row;
            int global_col = bx * BLOCK_N + col;
            if (global_row < K && global_col < N) {
                w_smem[row][col] = w[global_row * N + global_col];
            } else {
                w_smem[row][col] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute dot product
        for (int kk = 0; kk < k_remaining; kk++) {
            half x_val = x_smem[ty][kk];
            half w_val = w_smem[kk][tx];
            acc += __half2float(x_val) * __half2float(w_val);
        }
        
        __syncthreads();
    }
    
    // Apply SiLU activation
    float result = acc * scale;
    float sigmoid = 1.0f / (1.0f + expf(-result));
    float silu = result * sigmoid;
    
    // Store
    int row = by * BLOCK_M + ty;
    int col = bx * BLOCK_N + tx;
    if (row < M && col < N) {
        y[row * N + col] = __float2half(silu);
    }
}

// --------------------------------------------------------------------------
// Simple gemm without activation: y = x @ W^T
// --------------------------------------------------------------------------

__global__ void gemm_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    float* __restrict__ y,
    int M, int N, int K,
    float scale)
{
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    int col = blockIdx.y * blockDim.y + threadIdx.y;
    
    if (row < M && col < N) {
        float sum = 0.0f;
        for (int k = 0; k < K; k++) {
            sum += x[row * K + k] * w[col * K + k];
        }
        y[row * N + col] = sum * scale;
    }
}

// --------------------------------------------------------------------------
// Fused sigmoid + multiply: y = sigmoid(x) * y
// --------------------------------------------------------------------------

__global__ void fused_sigmoid_mul_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int N)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        float sig = 1.0f / (1.0f + expf(-x[idx]));
        y[idx] *= sig;
    }
}

// --------------------------------------------------------------------------
// Fused multiply: y = x * scale
// --------------------------------------------------------------------------

__global__ void scale_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int N,
    float scale)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        y[idx] = x[idx] * scale;
    }
}

// --------------------------------------------------------------------------
// Reshape and transpose: (B, T, H, D) -> (B, H, T, D)
// --------------------------------------------------------------------------

__global__ void transpose_bthd_to_bhtd_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int B, int T, int H, int D)
{
    int b = blockIdx.z / H;
    int h = blockIdx.z % H;
    
    int t = blockIdx.y;
    int d = threadIdx.x;
    
    if (b < B && t < T && h < H && d < D) {
        // x[b, t, h, d] -> y[b, h, t, d]
        y[(b * H + h) * T * D + t * D + d] = x[(b * T + t) * H * D + h * D + d];
    }
}

__global__ void transpose_bhtd_to_bthd_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int B, int H, int T, int D)
{
    int b = blockIdx.z / H;
    int h = blockIdx.z % H;
    
    int t = blockIdx.y;
    int d = threadIdx.x;
    
    if (b < B && h < H && t < T && d < D) {
        // x[b, h, t, d] -> y[b, t, h, d]
        y[(b * T + t) * H * D + h * D + d] = x[(b * H + h) * T * D + t * D + d];
    }
}

// --------------------------------------------------------------------------
// View reshape: (B, T, H*D) -> (B, T, H, D)
// --------------------------------------------------------------------------

__global__ void view_to_4d_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int B, int T, int H, int D)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * T * H * D;
    
    if (idx < total) {
        int b = idx / (T * H * D);
        int rem = idx % (T * H * D);
        int t = rem / (H * D);
        rem = rem % (H * D);
        int h = rem / D;
        int d = rem % D;
        
        // x[b, t, h*d + d] -> y[b, t, h, d]
        y[(b * T * H + t * H + h) * D + d] = x[idx];
    }
}

// --------------------------------------------------------------------------
// Reshape: (B, T, H, D) -> (B, T, H*D)
// --------------------------------------------------------------------------

__global__ void reshape_4d_to_3d_kernel(
    const float* __restrict__ x,
    float* __restrict__ y,
    int B, int T, int H, int D)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = B * T * H * D;
    
    if (idx < total) {
        int b = idx / (T * H * D);
        int rem = idx % (T * H * D);
        int t = rem / (H * D);
        rem = rem % (H * D);
        int h = rem / D;
        int d = rem % D;
        
        // x[b, t, h, d] -> y[b, t, h*D + d]
        y[(b * T * H + t * H) * D + h * D + d] = x[idx];
    }
}

// --------------------------------------------------------------------------
// Log gate computation: g = log(clamp(a, min=1e-6))
// --------------------------------------------------------------------------

__global__ void log_gate_kernel(
    const float* __restrict__ a,
    float* __restrict__ g,
    int N)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < N) {
        float val = fmaxf(a[idx], 1e-6f);
        g[idx] = logf(val);
    }
}

// C++ API wrappers

torch::Tensor fused_linear_fp32(
    torch::Tensor x,      // (M, K)
    torch::Tensor w,     // (K, N)
    float scale)
{
    int M = x.size(0);
    int K = x.size(1);
    int N = w.size(1);
    
    auto y = torch::empty({M, N}, x.options());
    
    const float* x_ptr = x.data_ptr<float>();
    const float* w_ptr = w.data_ptr<float>();
    float* y_ptr = y.data_ptr<float>();
    
    // Simple gemm with threads covering output
    dim3 block(16, 16);
    dim3 grid((M + 15) / 16, (N + 15) / 16);
    
    gemm_kernel<<<grid, block>>>(
        x_ptr, w_ptr, y_ptr, M, N, K, scale);
    
    return y;
}

torch::Tensor fused_sigmoid_mul(
    torch::Tensor x,   // (N,)
    torch::Tensor y    // (N,) - modified in place
)
{
    int N = x.numel();
    
    const float* x_ptr = x.data_ptr<float>();
    float* y_ptr = y.data_ptr<float>();
    
    int block = 256;
    int grid = (N + block - 1) / block;
    
    fused_sigmoid_mul_kernel<<<grid, block>>>(
        x_ptr, y_ptr, N);
    
    return y;
}

torch::Tensor fused_log_gate(
    torch::Tensor a   // (N,)
)
{
    int N = a.numel();
    auto g = torch::empty_like(a);
    
    const float* a_ptr = a.data_ptr<float>();
    float* g_ptr = g.data_ptr<float>();
    
    int block = 256;
    int grid = (N + block - 1) / block;
    
    log_gate_kernel<<<grid, block>>>(
        a_ptr, g_ptr, N);
    
    return g;
}

torch::Tensor reshape_view(
    torch::Tensor x,   // (B, T, H, D)
    int B, int T, int H, int D)
{
    auto y = torch::empty({B, T, H, D}, x.options());
    
    const float* x_ptr = x.data_ptr<float>();
    float* y_ptr = y.data_ptr<float>();
    
    int block = 256;
    int grid = (B * T * H * D + block - 1) / block;
    
    view_to_4d_kernel<<<grid, block>>>(
        x_ptr, y_ptr, B, T, H, D);
    
    return y;
}

torch::Tensor reshape_4d_to_3d(
    torch::Tensor x,   // (B, T, H, D)
    int B, int T, int H, int D)
{
    auto y = torch::empty({B, T, H * D}, x.options());
    
    const float* x_ptr = x.data_ptr<float>();
    float* y_ptr = y.data_ptr<float>();
    
    int block = 256;
    int grid = (B * T * H * D + block - 1) / block;
    
    reshape_4d_to_3d_kernel<<<grid, block>>>(
        x_ptr, y_ptr, B, T, H, D);
    
    return y;
}

torch::Tensor transpose_bthd_to_bhtd(
    torch::Tensor x,   // (B, T, H, D)
    int B, int T, int H, int D)
{
    auto y = torch::empty({B, H, T, D}, x.options());

}

torch::Tensor transpose_bhtd_to_bthd(

// Fused kernel wrapper that handles the full KDA computation
// This is a simplified version - the main computation still uses chunk_kda
// but we fuse surrounding operations

torch::Tensor kimi_attention_fused(
    torch::Tensor x,           // (B, T, hidden)
    torch::Tensor q_w, torch::Tensor q_b,   // Q projection
    torch::Tensor k_w, torch::Tensor k_b,
    torch::Tensor v_w, torch::Tensor v_b,
    torch::Tensor a_w, torch::Tensor a_b,   // gate projection
    torch::Tensor b_w, torch::Tensor b_b,
    torch::Tensor g_w, torch::Tensor g_b,   // output gate
    torch::Tensor o_w, torch::Tensor o_b,   // output projection
    int num_heads, int head_dim_qk, int head_dim_v,
    bool use_short_conv, int conv_kernel_size,
    torch::Tensor q_conv_w, torch::Tensor k_conv_w, torch::Tensor v_conv_w)
{
    int B = x.size(0);
    int T = x.size(1);
    int hidden = x.size(2);
    int H = num_heads;
    int D_qk = head_dim_qk;
    int D_v = head_dim_v;
    
    // Use float32 for all computations
    x = x.to(torch::kFloat);
    
    // ========================================================================
    // 1. Q, K, V projections with SiLU activation
    // ========================================================================
    
    // Q projection: x @ W_q^T + b_q
    auto q = torch::linear(x, q_w, q_b);  // (B, T, H * D_qk)
    
    // K projection
    auto k = torch::linear(x, k_w, k_b);  // (B, T, H * D_qk)
    
    // V projection  
    auto v = torch::linear(x, v_w, v_b);  // (B, T, H * D_v)
    
    // ========================================================================
    // 2. Short convolution + SiLU
    // ========================================================================
    
    if (use_short_conv) {
        // Reshape for conv1d: (B, T, H*D) -> (B, H*D, T)
        auto q_2d = q.transpose(1, 2);
        auto k_2d = k.transpose(1, 2);
        auto v_2d = v.transpose(1, 2);
        
        // Convolution with padding
        q_2d = torch::relu(q_2d);
        k_2d = torch::relu(k_2d);
        v_2d = torch::relu(v_2d);
        
        // Truncate to original length
        q = q_2d[:, :, :T].transpose(1, 2);
        k = k_2d[:, :, :T].transpose(1, 2);
        v = v_2d[:, :, :T].transpose(1, 2);
    }
    
    // ========================================================================
    // 3. Reshape to (B, T, H, D)
    // ========================================================================
    
    q = q.view({B, T, H, D_qk});
    k = k.view({B, T, H, D_qk});
    v = v.view({B, T, H, D_v});
    
    // ========================================================================
    // 4. Transpose to (B, H, T, D) for attention
    // ========================================================================
    
    q = q.transpose(1, 2);  // (B, H, T, D_qk)
    k = k.transpose(1, 2);
    v = v.transpose(1, 2);
    
    // ========================================================================
    // 5. Gate computations
    // ========================================================================
    
    // Channel-wise gate a: (B, T, H * D_v) -> sigmoid
    auto a_flat = torch::linear(x, a_w, a_b);  // (B, T, H * D_v)
    auto a = a_flat.view({B, T, H, D_v});
    a = torch::sigmoid(a);
    
    // Beta gate: (B, T, H) -> sigmoid
    auto beta_flat = torch::linear(x, b_w, b_b);  // (B, T, H)
    auto beta = beta_flat.view({B, T, H});
    beta = torch::sigmoid(beta);
    
    // Transpose gates
    a = a.transpose(1, 2);  // (B, H, T, D_v)
    beta = beta.transpose(1, 2);  // (B, H, T)
    
    // Log gate for numerical stability
    a = a.clamp(min=1e-6).log();
    
    // ========================================================================
    // 6. Chunk KDA attention - delegate to fused kernel
    // This calls back to Python's chunk_kda
    // ========================================================================
    
    // Return intermediate tensors for Python to process with chunk_kda
    // We'll use a special return to indicate what needs processing
    
    // For now, return tuple of tensors
    auto result = torch::stack({
        q.reshape({B * H * T * D_qk}),
        k.reshape({B * H * T * D_qk}),
        v.reshape({B * H * T * D_v}),
        a.reshape({B * H * T * D_v}),
        beta.reshape({B * H * T})
    });
    
    return result;
}
'''
