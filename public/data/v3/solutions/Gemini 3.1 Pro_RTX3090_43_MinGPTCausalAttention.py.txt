import torch
import torch.nn as nn
import torch.nn.functional as F
import math
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>

torch::Tensor my_linear(torch::Tensor input, torch::Tensor weight, torch::optional<torch::Tensor> bias) {
    return at::linear(input, weight, bias);
}
'''

ext = load_inline(
    name='my_ext',
    cpp_sources='torch::Tensor my_linear(torch::Tensor, torch::Tensor, torch::optional<torch::Tensor>);',
    cuda_sources=cuda_source,
    functions=['my_linear'],
    verbose=True
)

class Model(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.n_head = n_head
        self.n_embd = n_embd

    def forward(self, x):
        B, T, C = x.size()
        
        qkv = ext.my_linear(x, self.c_attn.weight, self.c_attn.bias)
        q, k, v = qkv.split(self.n_embd, dim=2)
        k = k.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)
        q = q.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)
        v = v.view(B, T, self.n_head, C // self.n_head).transpose(1, 2)
        
        y = F.scaled_dot_product_attention(q, k, v, attn_mask=None, dropout_p=0.0, is_causal=True)
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        
        y = self.resid_dropout(ext.my_linear(y, self.c_proj.weight, self.c_proj.bias))
        return y
