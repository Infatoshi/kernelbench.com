import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define TILE_X 16
#define TILE_Y 16
#define KERNEL_RADIUS 2
#define KERNEL_SIZE 5
#define SMEM_W (TILE_X + 2 * KERNEL_RADIUS)
#define SMEM_H (TILE_Y + 2 * KERNEL_RADIUS)

__global__ void bloom_kernel(
    const float* __restrict__ input,
    float* __restrict__ output,
    int B, int C, int H, int W,
    float threshold, float intensity
) {
    const int bc = blockIdx.z;
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    const int gx = bx * TILE_X + tx;
    const int gy = by * TILE_Y + ty;
    
    __shared__ float smem[SMEM_H][SMEM_W];
    
    const int tid = ty * TILE_X + tx;
    const int smem_size = SMEM_H * SMEM_W;
    
    #pragma unroll
    for (int i = tid; i < smem_size; i += TILE_X * TILE_Y) {
        const int sm_y = i / SMEM_W;
        const int sm_x = i % SMEM_W;
        const int g_y = by * TILE_Y + sm_y - KERNEL_RADIUS;
        const int g_x = bx * TILE_X + sm_x - KERNEL_RADIUS;
        
        float val = 0.0f;
        if (g_y >= 0 && g_y < H && g_x >= 0 && g_x < W) {
            const int idx = bc * H * W + g_y * W + g_x;
            val = input[idx];
        }
        smem[sm_y][sm_x] = val;
    }
    
    __syncthreads();
    
    if (gx < W && gy < H) {
        float sum = 0.0f;
        
        #pragma unroll
        for (int dy = 0; dy < KERNEL_SIZE; ++dy) {
            #pragma unroll
            for (int dx = 0; dx < KERNEL_SIZE; ++dx) {
                float in_val = smem[ty + dy][tx + dx];
                float bright = in_val > threshold ? (in_val - threshold) : 0.0f;
                sum += bright;
            }
        }
        
        const float blurred = sum * (1.0f / 25.0f);
        const float orig = smem[ty + KERNEL_RADIUS][tx + KERNEL_RADIUS];
        float result = orig + intensity * blurred;
        
        result = fminf(fmaxf(result, 0.0f), 10.0f);
        
        const int out_idx = bc * H * W + gy * W + gx;
        output[out_idx] = result;
    }
}

torch::Tensor bloom_op(torch::Tensor input, float threshold, float intensity) {
    const int B = input.size(0);
    const int C = input.size(1);
    const int H = input.size(2);
    const int W = input.size(3);
    
    auto output = torch::empty_like(input);
    
    const dim3 block(TILE_X, TILE_Y);
    const dim3 grid((W + TILE_X - 1) / TILE_X, (H + TILE_Y - 1) / TILE_Y, B * C);
    
    bloom_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        B, C, H, W,
        threshold, intensity
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor bloom_op(torch::Tensor input, float threshold, float intensity);
'''

ext = load_inline(
    name='bloom_opt',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['bloom_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
)

class Model(nn.Module):
    def __init__(self, threshold: float = 1.0, intensity: float = 0.6, blur_kernel: int = 5):
        super().__init__()
        self.threshold = threshold
        self.intensity = intensity
        self.blur_kernel = blur_kernel
        if blur_kernel != 5:
            raise ValueError(f"Optimized kernel only supports blur_kernel=5, got {blur_kernel}")

    def forward(self, image: torch.Tensor) -> torch.Tensor:
        assert image.is_cuda, "Input must be CUDA tensor"
        image = image.contiguous()
        return ext.bloom_op(image, self.threshold, self.intensity)


def get_inputs():
    return [torch.randn(2, 3, 1024, 1024, device='cuda') * 1.5 + 0.2]


def get_init_inputs():
    return [1.0, 0.6, 5]
