import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def layer_norm_kernel(
    X_ptr,
    Y_ptr,
    Weight_ptr,
    Bias_ptr,
    Mean_ptr,
    Rstd_ptr,
    stride_x_row,
    stride_y_row,
    N: tl.constexpr,
    eps,
    BLOCK_SIZE: tl.constexpr,
):
    """
    LayerNorm kernel using Triton.
    Each program handles one row of the input.
    Uses multiple passes for large N.
    """
    row_idx = tl.program_id(0)
    
    # Pointers to the start of the row
    x_row_ptr = X_ptr + row_idx * stride_x_row
    y_row_ptr = Y_ptr + row_idx * stride_y_row
    
    # Process in chunks
    for block_start in range(0, N, BLOCK_SIZE):
        cols = block_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        
        x = tl.load(x_row_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        weight = tl.load(Weight_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        bias = tl.load(Bias_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        
        # Load pre-computed mean and rstd
        mean = tl.load(Mean_ptr + row_idx).to(tl.float32)
        rstd = tl.load(Rstd_ptr + row_idx).to(tl.float32)
        
        # Normalize and apply weight/bias
        x_norm = (x - mean) * rstd
        y = x_norm * weight + bias
        
        # Store result
        tl.store(y_row_ptr + cols, y, mask=mask)


@triton.jit
def compute_stats_kernel(
    X_ptr,
    Mean_ptr,
    Rstd_ptr,
    stride_x_row,
    N: tl.constexpr,
    eps,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Kernel to compute mean and variance for each row.
    """
    row_idx = tl.program_id(0)
    
    x_row_ptr = X_ptr + row_idx * stride_x_row
    
    # Accumulate sum and sum of squares
    total_sum = 0.0
    total_sq_sum = 0.0
    count = 0
    
    for block_start in range(0, N, BLOCK_SIZE):
        cols = block_start + tl.arange(0, BLOCK_SIZE)
        mask = cols < N
        
        x = tl.load(x_row_ptr + cols, mask=mask, other=0.0).to(tl.float32)
        
        total_sum += tl.sum(x, axis=0)
        total_sq_sum += tl.sum(x * x, axis=0)
        count += tl.sum(mask.to(tl.int32), axis=0)
    
    # Compute mean and variance
    mean = total_sum / count
    variance = total_sq_sum / count - mean * mean
    rstd = 1.0 / tl.sqrt(variance + eps)
    
    # Store results
    tl.store(Mean_ptr + row_idx, mean)
    tl.store(Rstd_ptr + row_idx, rstd)


class TritonLayerNorm(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, eps):
        # Get the normalized shape - last dimension(s) to normalize over
        normalized_shape = weight.shape
        
        # Calculate N - the number of elements to normalize over per row
        N = 1
        for dim in normalized_shape:
            N *= dim
        
        # Reshape x to (num_rows, N) where num_rows is the product of all other dimensions
        orig_shape = x.shape
        num_rows = 1
        for dim in orig_shape[:-len(normalized_shape)]:
            num_rows *= dim
        
        x_flat = x.reshape(num_rows, N).contiguous()
        
        # Choose block size - power of 2, max 1024 to stay within Triton limits
        BLOCK_SIZE = min(triton.next_power_of_2(N), 1024)
        
        # Allocate output and intermediate buffers
        y_flat = torch.empty_like(x_flat)
        mean_buf = torch.empty(num_rows, dtype=torch.float32, device=x.device)
        rstd_buf = torch.empty(num_rows, dtype=torch.float32, device=x.device)
        
        # First pass: compute mean and variance
        grid = (num_rows,)
        compute_stats_kernel[grid](
            x_flat, mean_buf, rstd_buf,
            x_flat.stride(0),
            N, eps,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        # Second pass: normalize and apply weight/bias
        layer_norm_kernel[grid](
            x_flat, y_flat, weight, bias,
            mean_buf, rstd_buf,
            x_flat.stride(0), y_flat.stride(0),
            N, eps,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        ctx.save_for_backward(x_flat, weight, bias, mean_buf, rstd_buf)
        ctx.eps = eps
        ctx.orig_shape = orig_shape
        
        return y_flat.reshape(orig_shape)


def triton_layernorm(x, weight, bias, eps=1e-5):
    return TritonLayerNorm.apply(x, weight, bias, eps)


class Model(nn.Module):
    """
    Optimized LayerNorm using Triton.
    """
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        
        # Create weight and bias with the correct shape
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias = nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return triton_layernorm(x, self.weight, self.bias, self.eps)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
