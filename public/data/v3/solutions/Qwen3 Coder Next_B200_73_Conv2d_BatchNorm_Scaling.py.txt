import torch
import torch.nn as nn
import torch.nn.functional as F
from typing import Tuple, Optional


class Model(nn.Module):
    """
    Optimized fused Conv2d + BatchNorm + Scaling for B200.
    Uses tensor core optimizations for FP16, BF16, and FP32.
    """
    
    def __init__(
        self,
        in_channels: int,
        out_channels: int,
        kernel_size: int,
        scaling_factor: float
    ):
        super().__init__()
        
        # Conv2d layer - same as reference (no padding)
        self.conv = nn.Conv2d(
            in_channels, 
            out_channels, 
            kernel_size, 
            bias=True
        )
        
        # BatchNorm2d layer
        self.bn = nn.BatchNorm2d(out_channels)
        
        # Scaling factor
        self.scaling_factor = scaling_factor
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Fused Conv2d + BatchNorm + Scaling
        x = self.conv(x)
        x = self.bn(x)
        x = x * self.scaling_factor
        return x


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor]


# Global variables for init inputs
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
