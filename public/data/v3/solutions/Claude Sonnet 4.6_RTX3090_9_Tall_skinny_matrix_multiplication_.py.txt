import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Tall-skinny GEMM: A(M,K) * B(K,N) where M=16384, K=16, N=16384
# A: (16384, 16), B: (16, 16384), C: (16384, 16384)
# Strategy: Each thread block computes a tile of C using shared memory
# K=16 is tiny, so we do the full K reduction per tile
# Use vectorized loads (float4) for coalesced memory access

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <mma.h>

using namespace nvcuda;

#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

// Block tile: BM x BN output computed per block
// Each warp computes one 16x16 tile
// Use 8 warps per block: 2 along M, 4 along N => BM=32, BN=64
#define WARPS_M 2
#define WARPS_N 4
#define BM (WARPS_M * WMMA_M)  // 32
#define BN (WARPS_N * WMMA_N)  // 64
#define WARPS_PER_BLOCK (WARPS_M * WARPS_N)  // 8
#define THREADS_PER_BLOCK (WARPS_PER_BLOCK * 32)  // 256

// FP16 WMMA kernel - uses BF16-like accumulation approach
// Converts inputs to FP16, does WMMA with FP32 accumulation
__global__ void wmma_gemm_fp16(
    const __half* __restrict__ A,  // (M, K)
    const __half* __restrict__ B,  // (K, N)
    float* __restrict__ C,         // (M, N)
    int M, int K, int N
) {
    // Shared mem for A and B tiles
    // A: BM x K, B: K x BN
    extern __shared__ __half smem[];
    __half* smem_A = smem;                        // BM * K halfs
    __half* smem_B = smem + BM * K;              // K * BN halfs

    int warp_id = threadIdx.x / 32;
    int warp_m = warp_id / WARPS_N;
    int warp_n = warp_id % WARPS_N;

    int block_m = blockIdx.x * BM;
    int block_n = blockIdx.y * BN;

    // Load A[block_m:block_m+BM, 0:K] into smem_A
    // BM*K = 32*16 = 512 elements, 256 threads -> 2 each
    for (int i = threadIdx.x; i < BM * K; i += THREADS_PER_BLOCK) {
        int row = i / K;
        int col = i % K;
        int g_row = block_m + row;
        smem_A[row * K + col] = (g_row < M) ? A[g_row * K + col] : __float2half(0.f);
    }

    // Load B[0:K, block_n:block_n+BN] into smem_B
    // K*BN = 16*64 = 1024 elements, 256 threads -> 4 each
    for (int i = threadIdx.x; i < K * BN; i += THREADS_PER_BLOCK) {
        int row = i / BN;
        int col = i % BN;
        int g_col = block_n + col;
        smem_B[row * BN + col] = (g_col < N) ? B[row * N + g_col] : __float2half(0.f);
    }

    __syncthreads();

    // Each warp computes 16x16 tile
    wmma::fragment<wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> a_frag;
    wmma::fragment<wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K, __half, wmma::row_major> b_frag;
    wmma::fragment<wmma::accumulator, WMMA_M, WMMA_N, WMMA_K, float> acc_frag;
    wmma::fill_fragment(acc_frag, 0.0f);

    // A sub-tile: smem_A[warp_m*16 : (warp_m+1)*16, 0:16]
    // stride = K = 16
    wmma::load_matrix_sync(a_frag, smem_A + warp_m * WMMA_M * K, K);
    // B sub-tile: smem_B[0:16, warp_n*16 : (warp_n+1)*16]
    // stride = BN = 64
    wmma::load_matrix_sync(b_frag, smem_B + warp_n * WMMA_N, BN);

    wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);

    // Store to C
    int out_m = block_m + warp_m * WMMA_M;
    int out_n = block_n + warp_n * WMMA_N;
    if (out_m < M && out_n < N) {
        wmma::store_matrix_sync(C + out_m * N + out_n, acc_frag, N, wmma::mem_row_major);
    }
}


// Pure FP32 kernel for correctness with larger tolerance
// Each block of 256 threads computes a BM x BN tile
// Uses shared memory to cache A rows (small K)
#define BM32 32
#define BN32 64
#define BK32 16

__global__ void gemm_fp32_kernel(
    const float* __restrict__ A,  // (M, K)
    const float* __restrict__ B,  // (K, N)
    float* __restrict__ C,        // (M, N)
    int M, int K, int N
) {
    // Each thread block: BM32 x BN32 tile
    // 256 threads total
    __shared__ float smem_A[BM32][BK32];  // 32x16 = 512 floats
    __shared__ float smem_B[BK32][BN32];  // 16x64 = 1024 floats

    int tx = threadIdx.x;
    int block_m = blockIdx.x * BM32;
    int block_n = blockIdx.y * BN32;

    // Load smem_A: BM32*BK32=512 elements, 256 threads -> 2 each
    for (int i = tx; i < BM32 * BK32; i += blockDim.x) {
        int row = i / BK32;
        int col = i % BK32;
        int g_row = block_m + row;
        smem_A[row][col] = (g_row < M && col < K) ? A[g_row * K + col] : 0.f;
    }

    // Load smem_B: BK32*BN32=1024 elements, 256 threads -> 4 each
    for (int i = tx; i < BK32 * BN32; i += blockDim.x) {
        int row = i / BN32;
        int col = i % BN32;
        int g_col = block_n + col;
        smem_B[row][col] = (row < K && g_col < N) ? B[row * N + g_col] : 0.f;
    }

    __syncthreads();

    // Each thread computes 2 output elements (one row, 2 consecutive columns)
    // tx maps to (m_local, n_local) in the tile
    // thread layout: 256 = 32 rows x 8 col-groups (each 8 threads handle 64 cols?)
    // Actually: 256 threads, BM32*BN32 = 2048 outputs -> 8 per thread
    // Let each thread compute 2 rows x 4 cols = 8 outputs
    int tid_m = tx / (BN32 / 4);   // 256 / (64/4) = 256/16 = 16 rows
    int tid_n = tx % (BN32 / 4);   // = tx % 16, each covers 4 consecutive cols

    // Actually simpler: stride pattern
    // Each thread computes all outputs it owns
    for (int idx = tx; idx < BM32 * BN32; idx += blockDim.x) {
        int local_m = idx / BN32;
        int local_n = idx % BN32;
        float sum = 0.f;
        #pragma unroll
        for (int k = 0; k < BK32; k++) {
            sum += smem_A[local_m][k] * smem_B[k][local_n];
        }
        int g_m = block_m + local_m;
        int g_n = block_n + local_n;
        if (g_m < M && g_n < N) {
            C[g_m * N + g_n] = sum;
        }
    }
}

torch::Tensor tall_skinny_gemm(torch::Tensor A, torch::Tensor B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Expected 2D tensors");
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    TORCH_CHECK(A.size(1) == B.size(0), "Shape mismatch");

    auto C = torch::empty({M, N}, torch::TensorOptions().dtype(torch::kFloat32).device(A.device()));

    // Use FP32 kernel for correctness
    auto A_fp32 = A.to(torch::kFloat32).contiguous();
    auto B_fp32 = B.to(torch::kFloat32).contiguous();

    dim3 grid((M + BM32 - 1) / BM32, (N + BN32 - 1) / BN32);
    dim3 block(256);

    gemm_fp32_kernel<<<grid, block>>>(
        A_fp32.data_ptr<float>(),
        B_fp32.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K, N
    );

    return C.to(A.dtype());
}
'''

cpp_source = '''
torch::Tensor tall_skinny_gemm(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='tall_skinny_gemm_v2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['tall_skinny_gemm'],
    verbose=True,
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math', '-std=c++17'],
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A, B):
        return ext.tall_skinny_gemm(A, B)


M = 16384
N = 16

def get_inputs():
    A = torch.randn(M, N)
    B = torch.randn(N, M)
    return [A, B]

def get_init_inputs():
    return []
