import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
// Constants
constexpr float SQRT_2_OVER_PI = 0.7978845608028654f;
constexpr float COEF = 0.044715f;

__global__ void gelu_kernel_float(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float xi = x[idx];
        float x3 = xi * xi * xi;
        float inner = SQRT_2_OVER_PI * (xi + COEF * x3);
        float tanh_inner = tanhf(inner);
        y[idx] = 0.5f * xi * (1.0f + tanh_inner);
    }
}

__global__ void gelu_kernel_half(const __half* __restrict__ x, __half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float xi = __half2float(x[idx]);
        float x3 = xi * xi * xi;
        float inner = SQRT_2_OVER_PI * (xi + COEF * x3);
        float tanh_inner = tanhf(inner);
        float result = 0.5f * xi * (1.0f + tanh_inner);
        y[idx] = __float2half(result);
    }
}

__global__ void gelu_kernel_bfloat16(const __nv_bfloat16* __restrict__ x, __nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float xi = __bfloat162float(x[idx]);
        float x3 = xi * xi * xi;
        float inner = SQRT_2_OVER_PI * (xi + COEF * x3);
        float tanh_inner = tanhf(inner);
        float result = 0.5f * xi * (1.0f + tanh_inner);
        y[idx] = __float2bfloat16(result);
    }
}

torch::Tensor gelu_op(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    if (n == 0) return y;
    
    int threads = 512;
    int blocks = (n + threads - 1) / threads;
    
    auto dtype = x.scalar_type();
    
    if (dtype == torch::kFloat32) {
        gelu_kernel_float<<<blocks, threads>>>(
            x.data_ptr<float>(), 
            y.data_ptr<float>(), 
            n
        );
    } else if (dtype == torch::kFloat16) {
        gelu_kernel_half<<<blocks, threads>>>(
            reinterpret_cast<const __half*>(x.data_ptr<c10::Half>()), 
            reinterpret_cast<__half*>(y.data_ptr<c10::Half>()), 
            n
        );
    } else if (dtype == torch::kBFloat16) {
        gelu_kernel_bfloat16<<<blocks, threads>>>(
            reinterpret_cast<const __nv_bfloat16*>(x.data_ptr<c10::BFloat16>()), 
            reinterpret_cast<__nv_bfloat16*>(y.data_ptr<c10::BFloat16>()), 
            n
        );
    } else {
        // Fallback for other types - convert to float
        gelu_kernel_float<<<blocks, threads>>>(
            x.data_ptr<float>(), 
            y.data_ptr<float>(), 
            n
        );
    }
    
    cudaDeviceSynchronize();
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_op(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['gelu_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_op(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    return [torch.randn(batch_size, dim, device='cuda')]

def get_init_inputs():
    return []
