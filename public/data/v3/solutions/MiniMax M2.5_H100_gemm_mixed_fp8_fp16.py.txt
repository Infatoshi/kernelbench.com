"""
Mixed-precision GEMM: FP8-like activations with FP16 weights/accumulation.
Optimized CUDA implementation for H100.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GEMM kernel: int8 * fp16 -> fp16 with scaling
// Uses simple blocked tiling for clarity

__global__ void gemm_int8_fp16_kernel(
    const int8_t* __restrict__ a_q,
    const half* __restrict__ b_fp16,
    half* __restrict__ c,
    const float scale,
    const int M,
    const int N,
    const int K
) {
    // Block tile sizes
    const int BM = 64;
    const int BN = 64;
    const int BK = 16;
    
    // Shared memory
    __shared__ half a_shared[BM * BK];
    __shared__ half b_shared[BK * BN];
    
    // Block indices
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    
    // Thread indices within block
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    // Row and column of output that this thread computes
    const int row = by * BM + ty;
    const int col = bx * BN + tx;
    
    // Accumulator
    half sum = __float2half_rn(0.0f);
    
    // Number of K tiles
    const int num_k_tiles = (K + BK - 1) / BK;
    
    // Loop over K tiles
    for (int kt = 0; kt < num_k_tiles; ++kt) {
        // Load A tile into shared memory
        // Each thread loads one element from A
        const int a_col = kt * BK + tx;
        if (row < M && a_col < K) {
            int8_t val = a_q[row * K + a_col];
            a_shared[ty * BK + tx] = __float2half_rn(static_cast<float>(val) * scale);
        } else {
            a_shared[ty * BK + tx] = __float2half_rn(0.0f);
        }
        
        // Load B tile into shared memory
        const int b_row = kt * BK + ty;
        if (b_row < K && col < N) {
            b_shared[ty * BN + tx] = b_fp16[b_row * N + col];
        } else {
            b_shared[ty * BN + tx] = __float2half_rn(0.0f);
        }
        
        __syncthreads();
        
        // Compute partial dot product
        if (row < M && col < N) {
            for (int kk = 0; kk < BK; ++kk) {
                half a_val = a_shared[ty * BK + kk];
                half b_val = b_shared[kk * BN + tx];
                sum = __hfma(a_val, b_val, sum);
            }
        }
        
        __syncthreads();
    }
    
    // Store result
    if (row < M && col < N) {
        c[row * N + col] = sum;
    }
}

// Optimized version using vectorized loads and larger tiles
__global__ void gemm_int8_fp16_kernel_v2(
    const int8_t* __restrict__ a_q,
    const half* __restrict__ b_fp16,
    half* __restrict__ c,
    const float scale,
    const int M,
    const int N,
    const int K
) {
    // Larger tile sizes for better occupancy
    const int BM = 128;
    const int BN = 128;
    const int BK = 32;
    
    // Shared memory
    __shared__ half a_shared[BM * BK];
    __shared__ half b_shared[BK * BN];
    
    // Block indices
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    
    // Thread indices - each thread handles multiple elements
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    // Number of threads per block
    const int num_threads_x = blockDim.x;
    const int num_threads_y = blockDim.y;
    
    // Each thread computes a tile of size (thread_tile_m x thread_tile_n)
    const int thread_tile_m = 1;
    const int thread_tile_n = 1;
    
    // Loop over output blocks
    for (int m_block = by * BM; m_block < M; m_block += BM * gridDim.y) {
        for (int n_block = bx * BN; n_block < N; n_block += BN * gridDim.x) {
            
            // Accumulator for this output block
            half sum[thread_tile_m][thread_tile_n];
            #pragma unroll
            for (int i = 0; i < thread_tile_m; ++i) {
                #pragma unroll
                for (int j = 0; j < thread_tile_n; ++j) {
                    sum[i][j] = __float2half_rn(0.0f);
                }
            }
            
            // Loop over K
            const int k_start = 0;
            const int k_end = K;
            
            for (int kk = k_start; kk < k_end; kk += BK) {
                // Cooperative load A tile
                const int a_load_m = (M - m_block + num_threads_y - 1) / num_threads_y;
                const int a_load_k = min(BK, K - kk);
                
                // Each thread loads a[ty + i*num_threads_y][tx + j*num_threads_x] for A
                for (int i = 0; i < (BM + num_threads_y - 1) / num_threads_y; ++i) {
                    int row = m_block + ty + i * num_threads_y;
                    int col = kk + tx;
                    if (row < M && col < K) {
                        int8_t val = a_q[row * K + col];
                        a_shared[(ty + i * num_threads_y) * BK + tx] = __float2half_rn(static_cast<float>(val) * scale);
                    }
                }
                
                // Cooperative load B tile
                for (int i = 0; i < (BK + num_threads_y - 1) / num_threads_y; ++i) {
                    int row = kk + ty + i * num_threads_y;
                    int col = n_block + tx;
                    if (row < K && col < N) {
                        b_shared[(ty + i * num_threads_y) * BN + tx] = b_fp16[row * N + col];
                    }
                }
                
                __syncthreads();
                
                // Compute partial GEMM
                for (int k = 0; k < BK && kk + k < K; ++k) {
                    #pragma unroll
                    for (int i = 0; i < thread_tile_m; ++i) {
                        #pragma unroll
                        for (int j = 0; j < thread_tile_n; ++j) {
                            int row = ty + i * num_threads_y;
                            int col = tx + j * num_threads_x;
                            half a_val = a_shared[row * BK + k];
                            half b_val = b_shared[k * BN + col];
                            sum[i][j] = __hfma(a_val, b_val, sum[i][j]);
                        }
                    }
                }
                
                __syncthreads();
            }
            
            // Store result
            #pragma unroll
            for (int i = 0; i < thread_tile_m; ++i) {
                #pragma unroll
                for (int j = 0; j < thread_tile_n; ++j) {
                    int row = m_block + ty + i * num_threads_y;
                    int col = n_block + tx + j * num_threads_x;
                    if (row < M && col < N) {
                        c[row * N + col] = sum[i][j];
                    }
                }
            }
        }
    }
}

// Wrapper function - uses v2 kernel
torch::Tensor gemm_forward(
    torch::Tensor a_q,
    torch::Tensor b_fp16,
    torch::Tensor scale_a
) {
    const int M = a_q.size(0);
    const int K = a_q.size(1);
    const int N = b_fp16.size(1);
    const float scale = scale_a.item<float>();
    
    auto output = torch::empty({M, N}, a_q.options().dtype(at::kHalf));
    
    // Kernel config - 256 threads per block (16x16)
    dim3 block(16, 16);  // 256 threads
    dim3 grid((N + 127) / 128, (M + 127) / 128);
    
    // Make sure grid covers all output
    grid.x = min(grid.x, (N + 127) / 128);
    grid.y = min(grid.y, (M + 127) / 128);
    
    gemm_int8_fp16_kernel_v2<<<grid, block>>>(
        a_q.data_ptr<int8_t>(),
        reinterpret_cast<half*>(b_fp16.data_ptr<at::Half>()),
        reinterpret_cast<half*>(output.data_ptr<at::Half>()),
        scale,
        M, N, K
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='gemm_mixed_fp8_fp16',
    cpp_sources=r'''
torch::Tensor gemm_forward(
    torch::Tensor a_q,
    torch::Tensor b_fp16,
    torch::Tensor scale_a
);
''',
    cuda_sources=cuda_source,
    functions=['gemm_forward'],
    verbose=True,
    extra_cflags=['-O3', '-std=c++17'],
    extra_include_paths=[
        '/opt/cutlass/include',
        '/usr/local/cuda/include'
    ]
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, a_q: torch.Tensor, b_fp16: torch.Tensor, scale_a: torch.Tensor) -> torch.Tensor:
        return ext.gemm_forward(a_q, b_fp16, scale_a)


def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-127, 127, (m, k), dtype=torch.int8)
    b_fp16 = torch.randn(k, n, dtype=torch.float16)
    scale_a = torch.tensor(0.01, dtype=torch.float32)
    return [a_q, b_fp16, scale_a]


def get_init_inputs():
    return []
