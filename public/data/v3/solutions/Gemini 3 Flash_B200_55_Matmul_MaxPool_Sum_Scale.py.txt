import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_maxpool_sum_scale_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_out,
    KERNEL_SIZE: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    """
    X: (M, K)
    W: (N, K) -> Linear layer uses (out_features, in_features)
    B: (N,)
    Out: (M,)
    Operation:
    1. Y = X @ W.T + B  -> (M, N)
    2. Y_pooled = MaxPool1d(Y.unsqueeze(1), KERNEL_SIZE).squeeze(1) -> (M, N // KERNEL_SIZE) if we assume N is divisible
       Wait, MaxPool1d kernel_size on the out_features dimension.
    3. Y_sum = sum(Y_pooled, dim=1) -> (M,)
    4. Y_final = Y_sum * SCALE
    """
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid % num_pid_m
    
    # We want to process each row (M) independently for the final sum.
    # To optimize matmul, we typically use 2D blocks.
    # But here we need to sum across N after pooling. 
    # Let's simplify: Mỗi program_id xử lý 1 khối BLOCK_M hàng.
    # Trong mỗi hàng, trượt qua N để tính MaxPool và Sum.
    
    rm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    rm_mask = rm < M
    
    # Accumulator for the final sum per row in the block
    final_sum = tl.zeros((BLOCK_M,), dtype=tl.float32)
    
    # We need to compute MaxPool1d. MaxPool1d(x, K) on (M, N) gives (M, N/K)
    # Each element of the pooled output is max(Y[m, n*K : (n+1)*K])
    # The sum is sum_{n=0}^{N/K - 1} max(Y[m, n*K : (n+1)*K])
    
    # Iterate over pooled elements
    # N_pooled = N // KERNEL_SIZE (assuming divisible for simplicity, or handle remainder)
    N_pooled = N // KERNEL_SIZE
    
    # To make matmul efficient, we should still iterate over K in an outer loop if possible.
    # But we need the full row (or chunks of K) to compute MaxPool? 
    # No, MaxPool is over the N dimension.
    
    # Better approach:
    # For each row m:
    #   For each pooled_idx j from 0 to N_pooled - 1:
    #     For each k_idx in 0..KERNEL_SIZE-1:
    #       n = j * KERNEL_SIZE + k_idx
    #       Y[m, n] = sum_{k=0}^{K-1} X[m, k] * W[n, k] + B[n]
    #     val = max(Y[m, j*K : (j+1)*K])
    #     sum += val
    
    # Optimization: BLOCK_N should be a multiple of KERNEL_SIZE.
    # BLOCK_N = 64, KERNEL_SIZE = 2, then we have 32 pooled results per BLOCK_N.
    
    # Since we need to sum over all N, we iterate through N in blocks.
    for n_start in range(0, N, BLOCK_N):
        # Compute Y[BLOCK_M, BLOCK_N]
        rn = n_start + tl.arange(0, BLOCK_N)
        rn_mask = rn < N
        
        acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
        
        for k_start in range(0, K, BLOCK_K):
            rk = k_start + tl.arange(0, BLOCK_K)
            rk_mask = rk < K
            
            # Load X: (BLOCK_M, BLOCK_K)
            x = tl.load(X_ptr + (rm[:, None] * stride_xm + rk[None, :] * stride_xk), 
                        mask=rm_mask[:, None] & rk_mask[None, :], other=0.0)
            # Load W: (BLOCK_N, BLOCK_K) 
            w = tl.load(W_ptr + (rn[:, None] * stride_wn + rk[None, :] * stride_wk), 
                        mask=rn_mask[:, None] & rk_mask[None, :], other=0.0)
            
            acc += tl.dot(x, tl.trans(w))
            
        # Add Bias
        bias = tl.load(B_ptr + rn, mask=rn_mask, other=0.0)
        acc += bias[None, :]
        
        # Now we have acc: (BLOCK_M, BLOCK_N)
        # We need to perform MaxPool1d on the BLOCK_N dimension.
        # Reshape acc to (BLOCK_M, BLOCK_N / KERNEL_SIZE, KERNEL_SIZE)
        # Then max over the last dimension.
        
        # Note: This requires BLOCK_N to be divisible by KERNEL_SIZE.
        # If not, it's more complex.
        acc_reshaped = tl.reshape(acc, (BLOCK_M, BLOCK_N // KERNEL_SIZE, KERNEL_SIZE))
        pooled = tl.max(acc_reshaped, axis=2) # (BLOCK_M, BLOCK_N // KERNEL_SIZE)
        
        final_sum += tl.sum(pooled, axis=1)
        
    final_sum *= SCALE
    tl.store(Out_ptr + rm, final_sum, mask=rm_mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        out = torch.empty((M,), device=x.device, dtype=x.dtype)
        
        # grid = lambda META: (triton.cdiv(M, META['BLOCK_M']),)
        # We'll use fixed block sizes for simplicity, or tune them.
        # BLOCK_N must be a multiple of kernel_size.
        BLOCK_M = 32
        BLOCK_N = 128
        while BLOCK_N % self.kernel_size != 0:
            BLOCK_N += 1
            
        grid = (triton.cdiv(M, BLOCK_M),)
        
        matmul_maxpool_sum_scale_kernel[grid](
            x, self.weight, self.bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.weight.stride(0), self.weight.stride(1),
            out.stride(0),
            KERNEL_SIZE=self.kernel_size,
            SCALE=self.scale_factor,
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=32,
            GROUP_SIZE_M=8
        )
        return out

import math
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_maxpool_sum_scale_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_out,
    KERNEL_SIZE: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    rm = pid * BLOCK_M + tl.arange(0, BLOCK_M)
    rm_mask = rm < M
    
    final_sum = tl.zeros((BLOCK_M,), dtype=tl.float32)
    
    # Loop over N in chunks of BLOCK_N
    for n_start in range(0, N, BLOCK_N):
        rn = n_start + tl.arange(0, BLOCK_N)
        rn_mask = rn < N
        
        acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
        
        for k_start in range(0, K, BLOCK_K):
            rk = k_start + tl.arange(0, BLOCK_K)
            rk_mask = rk < K
            
            x = tl.load(X_ptr + (rm[:, None] * stride_xm + rk[None, :] * stride_xk), 
                        mask=rm_mask[:, None] & rk_mask[None, :], other=0.0)
            w = tl.load(W_ptr + (rn[:, None] * stride_wn + rk[None, :] * stride_wk), 
                        mask=rn_mask[:, None] & rk_mask[None, :], other=0.0)
            
            acc = tl.dot(x, tl.trans(w), acc)
            
        bias = tl.load(B_ptr + rn, mask=rn_mask, other=0.0)
        acc += bias[None, :]
        
        # MaxPool1d along N dimension
        # Reshape to (BLOCK_M, BLOCK_N // KERNEL_SIZE, KERNEL_SIZE)
        acc_reshaped = tl.reshape(acc, (BLOCK_M, BLOCK_N // KERNEL_SIZE, KERNEL_SIZE))
        pooled = tl.max(acc_reshaped, axis=2)
        
        final_sum += tl.sum(pooled, axis=1)
        
    final_sum *= SCALE
    tl.store(Out_ptr + rm, final_sum, mask=rm_mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        self.matmul = nn.Linear(in_features, out_features)

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        # Ensure x is float32 for dot if necessary, but Triton handles half
        # However, it's safer to keep accumulation in float32.
        out = torch.empty((M,), device=x.device, dtype=x.dtype)
        
        # Use more efficient block sizes for B200 matmul
        BLOCK_M = 16 
        BLOCK_N = 128
        BLOCK_K = 128

        grid = (triton.cdiv(M, BLOCK_M),)
        
        # Kernel needs inputs in fp16/bf16/fp32.
        # Ensure weights and input types match for tl.dot
        weight = self.matmul.weight.to(x.dtype)
        bias = self.matmul.bias.to(x.dtype)

        matmul_maxpool_sum_scale_kernel[grid](
            x, weight, bias, out,
            M, N, K,
            x.stride(0), x.stride(1),
            weight.stride(0), weight.stride(1),
            out.stride(0),
            KERNEL_SIZE=self.kernel_size,
            SCALE=float(self.scale_factor),
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
        )
        return out
