import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

# Constants matching reference
in_channels = 3
out_channels = 16
kernel_size = 3
batch_size = 16
height = 256
width = 256

# CUDA kernel for fused Conv(3x3) + Activation + BN
# Optimized for RTX 3090 Ampere
cuda_source = """
#include <torch/extension.h>

__global__ void fused_fwd_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ bn_scale,
    const float* __restrict__ bn_shift,
    float* __restrict__ output,
    int N, int H, int W, int C_out, int H_out, int W_out
) {
    int oc = blockIdx.x;  // output channel
    int n = blockIdx.y;   // batch index
    
    __shared__ float s_w[27];
    int tid = threadIdx.x;
    if (tid < 27) {
        s_w[tid] = weight[oc * 27 + tid];
    }
    __syncthreads();
    
    float b = bias[oc];
    float sc = bn_scale[oc];
    float sh = bn_shift[oc];
    
    int total_out = H_out * W_out;
    for (int hw = tid; hw < total_out; hw += blockDim.x) {
        int h = hw / W_out;
        int w = hw % W_out;
        
        float acc = 0.0f;
        
        // Input indices base calculation
        // For each input channel ic (0, 1, 2), access 3x3 neighborhood
        
        // Channel 0
        int b0 = ((n * 3 + 0) * H + h) * W + w;
        acc += input[b0] * s_w[0];
        acc += input[b0 + 1] * s_w[1];
        acc += input[b0 + 2] * s_w[2];
        acc += input[b0 + W] * s_w[3];
        acc += input[b0 + W + 1] * s_w[4];
        acc += input[b0 + W + 2] * s_w[5];
        acc += input[b0 + 2*W] * s_w[6];
        acc += input[b0 + 2*W + 1] * s_w[7];
        acc += input[b0 + 2*W + 2] * s_w[8];
        
        // Channel 1
        int b1 = ((n * 3 + 1) * H + h) * W + w;
        acc += input[b1] * s_w[9];
        acc += input[b1 + 1] * s_w[10];
        acc += input[b1 + 2] * s_w[11];
        acc += input[b1 + W] * s_w[12];
        acc += input[b1 + W + 1] * s_w[13];
        acc += input[b1 + W + 2] * s_w[14];
        acc += input[b1 + 2*W] * s_w[15];
        acc += input[b1 + 2*W + 1] * s_w[16];
        acc += input[b1 + 2*W + 2] * s_w[17];
        
        // Channel 2
        int b2 = ((n * 3 + 2) * H + h) * W + w;
        acc += input[b2] * s_w[18];
        acc += input[b2 + 1] * s_w[19];
        acc += input[b2 + 2] * s_w[20];
        acc += input[b2 + W] * s_w[21];
        acc += input[b2 + W + 1] * s_w[22];
        acc += input[b2 + W + 2] * s_w[23];
        acc += input[b2 + 2*W] * s_w[24];
        acc += input[b2 + 2*W + 1] * s_w[25];
        acc += input[b2 + 2*W + 2] * s_w[26];
        
        acc += b;
        
        // Activation: x * tanh(softplus(x))
        float splus = acc > 20.0f ? acc : (acc < -20.0f ? expf(acc) : log1pf(expf(acc)));
        float activated = acc * tanhf(splus);
        
        // BN fusion: y = x * scale + shift
        output[((n * C_out + oc) * H_out + h) * W_out + w] = activated * sc + sh;
    }
}

torch::Tensor forward_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_scale,
    torch::Tensor bn_shift
) {
    int N = input.size(0);
    int H = input.size(2);
    int W = input.size(3);
    int C_out = weight.size(0);
    int H_out = H - 2;
    int W_out = W - 2;
    
    auto output = torch::empty({N, C_out, H_out, W_out}, input.options());
    
    dim3 grid(C_out, N);
    int threads = 128;
    
    fused_fwd_kernel<<<grid, threads>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        bn_scale.data_ptr<float>(),
        bn_shift.data_ptr<float>(),
        output.data_ptr<float>(),
        N, H, W, C_out, H_out, W_out
    );
    
    return output;
}
"""

# Load inline CUDA extension
fused_ops = load_inline(
    name="fused_conv_act_bn_52",
    cpp_sources="torch::Tensor forward_cuda(torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor, torch::Tensor);",
    cuda_sources=cuda_source,
    functions=["forward_cuda"],
    extra_cuda_cflags=["-O3", "--use_fast_math", "-arch=sm_86", "-std=c++17"],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        
        # Buffers for fused BN params
        self.register_buffer('bn_scale', torch.empty(out_channels))
        self.register_buffer('bn_shift', torch.empty(out_channels))
    
    def _compute_bn_params(self):
        with torch.no_grad():
            std = torch.sqrt(self.bn.running_var + self.bn.eps)
            self.bn_scale.copy_(self.bn.weight / std)
            self.bn_shift.copy_(self.bn.bias - self.bn.running_mean * self.bn.scale if hasattr(self.bn, 'scale') else self.bn.weight / std)
    
    def forward(self, x):
        # Fast path: eval mode, 3x3 conv, 3 input channels, stride=1, padding=0
        if (not self.training and 
            self.conv.in_channels == 3 and 
            self.conv.kernel_size[0] == 3 and self.conv.kernel_size[1] == 3 and
            self.conv.stride[0] == 1 and self.conv.stride[1] == 1 and
            self.conv.padding[0] == 0 and self.conv.padding[1] == 0):
            
            # Compute BN params
            with torch.no_grad():
                std = torch.sqrt(self.bn.running_var + self.bn.eps)
                self.bn_scale.copy_(self.bn.weight / std)
                self.bn_shift.copy_(self.bn.bias - self.bn.running_mean * self.bn.weight / std)
            
            if not x.is_contiguous():
                x = x.contiguous()
            
            return fused_ops.forward_cuda(
                x, self.conv.weight, self.conv.bias, 
                self.bn_scale, self.bn_shift
            )
        
        # Standard path
        x = self.conv(x)
        x = x * torch.tanh(F.softplus(x))
        x = self.bn(x)
        return x

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
