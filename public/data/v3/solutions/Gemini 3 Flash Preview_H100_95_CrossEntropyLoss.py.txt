import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def cross_entropy_kernel(
    logits_ptr, 
    targets_ptr, 
    output_ptr, 
    n_rows, 
    n_cols, 
    stride_row, 
    BLOCK_SIZE: tl.constexpr
):
    row_idx = tl.program_id(0).to(tl.int64)
    if row_idx >= n_rows:
        return

    # Load target
    target = tl.load(targets_ptr + row_idx)
    
    # Load logits for this row
    row_start_ptr = logits_ptr + row_idx * stride_row
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols
    
    # Load logits
    logits = tl.load(row_start_ptr + col_offsets, mask=mask, other=-float('inf'))
    
    # Use max for numerical stability
    m = tl.max(logits, axis=0)
    
    # exp(logits - m) and sum
    exp_logits = tl.exp(logits - m)
    sum_exp = tl.sum(exp_logits, axis=0)
    
    # Get the specific logit for the target class
    # Safe load using target index
    target_logit = tl.load(row_start_ptr + target)
    
    # loss = log(sum(exp(x_i - m))) + m - x_target
    loss = tl.log(sum_exp) + m - target_logit
    
    tl.store(output_ptr + row_idx, loss)

class CrossEntropyFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, logits, targets):
        n_rows, n_cols = logits.shape
        output = torch.empty((n_rows,), device=logits.device, dtype=torch.float32)
        
        # Power of 2 block size
        BLOCK_SIZE = triton.next_power_of_2(n_cols)
        
        # Higher occupancy and performance tuning
        num_warps = 8 if BLOCK_SIZE >= 4096 else 4
        
        cross_entropy_kernel[(n_rows,)](
            logits, 
            targets, 
            output, 
            n_rows, 
            n_cols, 
            logits.stride(0),
            BLOCK_SIZE=BLOCK_SIZE,
            num_warps=num_warps
        )
        return output.mean()

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return CrossEntropyFunction.apply(predictions, targets)

def get_inputs():
    batch_size = 4096
    num_classes = 1024
    return [torch.randn(batch_size, num_classes, device='cuda'), torch.randint(0, num_classes, (batch_size,), device='cuda')]

    return []
