import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void softmax_channel_kernel(const float* __restrict__ x, float* __restrict__ y,
                                        int N,int C,int D,int H,int W) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int spatial = N*D*H*W;
    if(idx>=spatial) return;
    int w = idx % W;
    int h = (idx / W) % H;
    int d = (idx / (W*H)) % D;
    int n = idx / (W*H*D);

    int base = (((n*C)*D + d)*H + h)*W + w;

    float maxv = -1e30f;
    for(int c=0;c<C;++c){
        float v = x[base + c*D*H*W];
        if(v>maxv) maxv=v;
    }
    float sum=0.f;
    for(int c=0;c<C;++c){
        float e = expf(x[base + c*D*H*W]-maxv);
        sum += e;
        y[base + c*D*H*W] = e;
    }
    for(int c=0;c<C;++c){
        y[base + c*D*H*W] /= sum;
    }
}

__global__ void maxpool3d_kernel(const float* __restrict__ x, float* __restrict__ y,
                                 int N,int C,int D,int H,int W,
                                 int k,
                                 int Do,int Ho,int Wo) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N*C*Do*Ho*Wo;
    if(idx>=total) return;

    int wo = idx % Wo;
    int ho = (idx / Wo) % Ho;
    int do_ = (idx / (Wo*Ho)) % Do;
    int c = (idx / (Wo*Ho*Do)) % C;
    int n = idx / (Wo*Ho*Do*C);

    int d0 = do_ * k;
    int h0 = ho * k;
    int w0 = wo * k;

    float m = -1e30f;
    for(int dz=0; dz<k; ++dz)
    for(int dy=0; dy<k; ++dy)
    for(int dx=0; dx<k; ++dx){
        int d = d0+dz, h=h0+dy, w=w0+dx;
        int in_idx = (((n*C + c)*D + d)*H + h)*W + w;
        float v = x[in_idx];
        if(v>m) m=v;
    }

    int out_idx = (((n*C + c)*Do + do_)*Ho + ho)*Wo + wo;
    y[out_idx] = m;
}

std::vector<torch::Tensor> fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, int pool_k){
    auto y = at::conv3d(x, w, b);
    y = at::softmax(y, 1);
    y = at::max_pool3d(y, {pool_k,pool_k,pool_k});
    y = at::max_pool3d(y, {pool_k,pool_k,pool_k});
    return {y};
}
'''

ext = load_inline(name='fused_conv_soft_pool',
                  cpp_sources='std::vector<torch::Tensor> fused_forward(torch::Tensor x, torch::Tensor w, torch::Tensor b, int pool_k);',
                  cuda_sources=cuda_src,
                  functions=['fused_forward'],
                  verbose=False)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super().__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_k = pool_kernel_size

    def forward(self, x):
        w = self.conv.weight
        b = self.conv.bias
        return ext.fused_forward(x, w, b, self.pool_k)[0]

batch_size = 16
in_channels = 3
out_channels = 16
depth, height, width = 16, 128, 128
kernel_size = 3
pool_kernel_size = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, depth, height, width, device='cuda')]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
