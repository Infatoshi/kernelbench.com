import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def batched_gemm_kernel(
    # Pointers to matrices
    a_ptr, b_ptr, c_ptr,
    # Matrix dimensions
    M, K, N,
    # Strides for A (batch, m, k)
    stride_a_batch, stride_a_m, stride_a_k,
    # Strides for B (batch, k, n)
    stride_b_batch, stride_b_k, stride_b_n,
    # Strides for C (batch, m, n)
    stride_c_batch, stride_c_m, stride_c_n,
    # Block sizes
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """Batched GEMM kernel using Triton."""
    # Get batch index and position within batch
    pid_batch = tl.program_id(0)
    pid_m = tl.program_id(1)
    pid_n = tl.program_id(2)
    
    # Compute the starting row and column for this block
    rm = pid_m * BLOCK_M
    rn = pid_n * BLOCK_N
    
    # Initialize accumulator for C block
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Compute base pointers for this batch
    a_batch_ptr = a_ptr + pid_batch * stride_a_batch
    b_batch_ptr = b_ptr + pid_batch * stride_b_batch
    c_batch_ptr = c_ptr + pid_batch * stride_c_batch
    
    # K-dimension loop with tiling
    for k in range(0, K, BLOCK_K):
        # Load A block [BLOCK_M, BLOCK_K]
        offs_am = (rm + tl.arange(0, BLOCK_M)) % M
        offs_ak = (k + tl.arange(0, BLOCK_K)) % K
        a_ptrs = a_batch_ptr + offs_am[:, None] * stride_a_m + offs_ak[None, :] * stride_a_k
        a = tl.load(a_ptrs)
        
        # Load B block [BLOCK_K, BLOCK_N]
        offs_bk = (k + tl.arange(0, BLOCK_K)) % K
        offs_bn = (rn + tl.arange(0, BLOCK_N)) % N
        b_ptrs = b_batch_ptr + offs_bk[:, None] * stride_b_k + offs_bn[None, :] * stride_b_n
        b = tl.load(b_ptrs)
        
        # Matrix multiply accumulate
        acc += tl.dot(a, b)
    
    # Store C block
    offs_cm = rm + tl.arange(0, BLOCK_M)
    offs_cn = rn + tl.arange(0, BLOCK_N)
    c_ptrs = c_batch_ptr + offs_cm[:, None] * stride_c_m + offs_cn[None, :] * stride_c_n
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, acc, mask=c_mask)


def batched_gemm_triton(A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
    """Batched GEMM using Triton kernel."""
    assert A.dim() == 3 and B.dim() == 3, "Inputs must be 3D tensors"
    batch_size, M, K = A.shape
    batch_size_b, K_b, N = B.shape
    assert batch_size == batch_size_b, "Batch dimensions must match"
    assert K == K_b, "K dimensions must match"
    
    # Allocate output
    C = torch.empty((batch_size, M, N), device=A.device, dtype=torch.float32)
    
    # Choose block sizes - tuned for B200 tensor cores
    BLOCK_M = 64
    BLOCK_N = 64
    BLOCK_K = 32
    
    # Grid dimensions
    grid = (batch_size, triton.cdiv(M, BLOCK_M), triton.cdiv(N, BLOCK_N))
    
    # Launch kernel
    batched_gemm_kernel[grid](
        A, B, C,
        M, K, N,
        A.stride(0), A.stride(1), A.stride(2),
        B.stride(0), B.stride(1), B.stride(2),
        C.stride(0), C.stride(1), C.stride(2),
        BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
    )
    
    return C


class Model(nn.Module):
    """
    Performs batched matrix multiplication (C = A * B) where A, B, and C have the same batch dimension.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs batched matrix multiplication.

        Args:
            A: Input tensor of shape (batch_size, m, k).
            B: Input tensor of shape (batch_size, k, n).

        Returns:
            C: Output tensor of shape (batch_size, m, n).
        """
        return batched_gemm_triton(A, B)


batch_size = 128
m = 128
k = 256
n = 512

def get_inputs():
    A = torch.randn(batch_size, m, k)
    B = torch.randn(batch_size, k, n)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
