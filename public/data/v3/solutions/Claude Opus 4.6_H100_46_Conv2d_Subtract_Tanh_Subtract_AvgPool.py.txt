import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_sub_tanh_sub_avgpool_kernel(
    input_ptr,    # conv output: (N, C, H, W)
    output_ptr,   # pooled output: (N, C, H_out, W_out)
    subtract1_value,
    subtract2_value,
    N, C, H, W,
    H_out, W_out,
    pool_size: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one element of the output
    pid = tl.program_id(0)
    idx = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    
    total_out = N * C * H_out * W_out
    mask = idx < total_out
    
    # Decode output index
    w_out = idx % W_out
    tmp = idx // W_out
    h_out = tmp % H_out
    tmp2 = tmp // H_out
    c = tmp2 % C
    n = tmp2 // C
    
    # Compute avg pool over pool_size x pool_size window
    h_start = h_out * pool_size
    w_start = w_out * pool_size
    
    inv_pool = 1.0 / (pool_size * pool_size)
    
    # For pool_size=2, unroll manually
    # input index: n * C * H * W + c * H * W + h * W + w
    base = n * C * H * W + c * H * W
    
    acc = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    # Unrolled for pool_size=2
    if pool_size == 2:
        # (0, 0)
        h0 = h_start
        w0 = w_start
        in_idx = base + h0 * W + w0
        val = tl.load(input_ptr + in_idx, mask=mask, other=0.0)
        val = val - subtract1_value
        val = tl.extra.cuda.libdevice.tanh(val.to(tl.float32))
        val = val - subtract2_value
        acc += val
        
        # (0, 1)
        w1 = w_start + 1
        in_idx = base + h0 * W + w1
        val = tl.load(input_ptr + in_idx, mask=mask, other=0.0)
        val = val - subtract1_value
        val = tl.extra.cuda.libdevice.tanh(val.to(tl.float32))
        val = val - subtract2_value
        acc += val
        
        # (1, 0)
        h1 = h_start + 1
        in_idx = base + h1 * W + w0
        val = tl.load(input_ptr + in_idx, mask=mask, other=0.0)
        val = val - subtract1_value
        val = tl.extra.cuda.libdevice.tanh(val.to(tl.float32))
        val = val - subtract2_value
        acc += val
        
        # (1, 1)
        in_idx = base + h1 * W + w1
        val = tl.load(input_ptr + in_idx, mask=mask, other=0.0)
        val = val - subtract1_value
        val = tl.extra.cuda.libdevice.tanh(val.to(tl.float32))
        val = val - subtract2_value
        acc += val
    else:
        for i in range(pool_size):
            for j in range(pool_size):
                h_in = h_start + i
                w_in = w_start + j
                in_idx = base + h_in * W + w_in
                val = tl.load(input_ptr + in_idx, mask=mask, other=0.0)
                val = val - subtract1_value
                val = tl.math.tanh(val.to(tl.float32))
                val = val - subtract2_value
                acc += val
    
    result = acc * inv_pool
    tl.store(output_ptr + idx, result.to(tl.float32), mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.kernel_size_pool = kernel_size_pool

    def forward(self, x):
        # Use aten::convolution directly (not F.conv2d)
        conv_out = torch.ops.aten.convolution(
            x, self.conv.weight, self.conv.bias,
            [1, 1],  # stride
            [0, 0],  # padding
            [1, 1],  # dilation
            False,    # transposed
            [0, 0],  # output_padding
            1         # groups
        )
        
        N, C, H, W = conv_out.shape
        pool_size = self.kernel_size_pool
        H_out = H // pool_size
        W_out = W // pool_size
        
        output = torch.empty(N, C, H_out, W_out, device=conv_out.device, dtype=conv_out.dtype)
        
        total_elements = N * C * H_out * W_out
        BLOCK_SIZE = 1024
        grid = ((total_elements + BLOCK_SIZE - 1) // BLOCK_SIZE,)
        
        fused_sub_tanh_sub_avgpool_kernel[grid](
            conv_out,
            output,
            self.subtract1_value,
            self.subtract2_value,
            N, C, H, W,
            H_out, W_out,
            pool_size=pool_size,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        return output


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
subtract1_value = 0.5
subtract2_value = 0.2
kernel_size_pool = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool]
