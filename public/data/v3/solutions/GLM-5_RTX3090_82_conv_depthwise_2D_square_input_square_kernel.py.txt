import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized depthwise 2D convolution kernel
// Each thread block processes one (batch, channel) pair
// Threads cooperate to load input tiles into shared memory

__global__ void depthwise_conv2d_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int channels, int height, int width,
    int kernel_size, int stride, int padding, int output_height, int output_width
) {
    // Block indices
    int bc_idx = blockIdx.x;  // batch * channels combined
    int batch = bc_idx / channels;
    int ch = bc_idx % channels;
    
    // Thread indices within block
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Output coordinates
    int out_x = blockIdx.y * blockDim.x + tx;
    int out_y = blockIdx.z * blockDim.y + ty;
    
    if (out_x >= output_width || out_y >= output_height) return;
    
    // Input base pointer for this batch and channel
    const float* in_ptr = input + (batch * channels + ch) * height * width;
    
    // Weight pointer for this channel (depthwise: one filter per channel)
    // Weight layout: (channels, 1, kernel_size, kernel_size)
    const float* w_ptr = weight + ch * kernel_size * kernel_size;
    
    // Compute input coordinates with padding offset
    int in_x_base = out_x * stride - padding;
    int in_y_base = out_y * stride - padding;
    
    // Perform convolution
    float sum = 0.0f;
    
    #pragma unroll
    for (int ky = 0; ky < kernel_size; ky++) {
        int in_y = in_y_base + ky;
        if (in_y >= 0 && in_y < height) {
            #pragma unroll
            for (int kx = 0; kx < kernel_size; kx++) {
                int in_x = in_x_base + kx;
                if (in_x >= 0 && in_x < width) {
                    sum += in_ptr[in_y * width + in_x] * w_ptr[ky * kernel_size + kx];
                }
            }
        }
    }
    
    // Add bias if present
    if (bias != nullptr) {
        sum += bias[ch];
    }
    
    // Store output
    output[(batch * channels + ch) * output_height * output_width + out_y * output_width + out_x] = sum;
}

// Shared memory optimized version for larger kernels
template<int KERNEL_SIZE, int TILE_OUT_W, int TILE_OUT_H>
__global__ void depthwise_conv2d_shared_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int channels, int height, int width,
    int stride, int padding, int output_height, int output_width
) {
    // Shared memory for input tile (including halo for kernel)
    // Need extra pixels on each side for the kernel
    constexpr int TILE_IN_W = TILE_OUT_W * 1 + KERNEL_SIZE - 1;  // assuming stride=1
    constexpr int TILE_IN_H = TILE_OUT_H * 1 + KERNEL_SIZE - 1;
    
    __shared__ float tile[TILE_IN_H][TILE_IN_W];
    __shared__ float w_shared[KERNEL_SIZE][KERNEL_SIZE];
    
    int bc_idx = blockIdx.x;
    int batch = bc_idx / channels;
    int ch = bc_idx % channels;
    
    // Output tile origin
    int out_x0 = blockIdx.y * TILE_OUT_W;
    int out_y0 = blockIdx.z * TILE_OUT_H;
    
    // Thread coordinates in output tile
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Cooperative loading of weights into shared memory
    if (tx < KERNEL_SIZE && ty < KERNEL_SIZE) {
        w_shared[ty][tx] = weight[ch * KERNEL_SIZE * KERNEL_SIZE + ty * KERNEL_SIZE + tx];
    }
    
    // Cooperative loading of input tile into shared memory
    int in_x0 = out_x0 * stride - padding;
    int in_y0 = out_y0 * stride - padding;
    
    const float* in_ptr = input + (batch * channels + ch) * height * width;
    
    // Load tile cooperatively
    for (int i = ty; i < TILE_IN_H; i += blockDim.y) {
        for (int j = tx; j < TILE_IN_W; j += blockDim.x) {
            int in_x = in_x0 + j;
            int in_y = in_y0 + i;
            float val = 0.0f;
            if (in_x >= 0 && in_x < width && in_y >= 0 && in_y < height) {
                val = in_ptr[in_y * width + in_x];
            }
            tile[i][j] = val;
        }
    }
    
    __syncthreads();
    
    // Compute output
    int out_x = out_x0 + tx;
    int out_y = out_y0 + ty;
    
    if (out_x >= output_width || out_y >= output_height) return;
    
    // Input position in shared tile
    int tile_x = tx;
    int tile_y = ty;
    
    float sum = 0.0f;
    
    #pragma unroll
    for (int ky = 0; ky < KERNEL_SIZE; ky++) {
        #pragma unroll
        for (int kx = 0; kx < KERNEL_SIZE; kx++) {
            sum += tile[tile_y + ky][tile_x + kx] * w_shared[ky][kx];
        }
    }
    
    if (bias != nullptr) {
        sum += bias[ch];
    }
    
    output[(batch * channels + ch) * output_height * output_width + out_y * output_width + out_x] = sum;
}

torch::Tensor depthwise_conv2d(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride, int padding
) {
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    int kernel_size = weight.size(2);  // square kernel
    
    int output_height = (height + 2 * padding - kernel_size) / stride + 1;
    int output_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::zeros({batch_size, channels, output_height, output_width}, 
                                input.options());
    
    // Choose kernel based on problem size
    dim3 blockDim(16, 16);  // 256 threads per block
    
    // Grid: x = batch * channels, y = output_width / 16, z = output_height / 16
    dim3 gridDim(batch_size * channels, 
                 (output_width + blockDim.x - 1) / blockDim.x,
                 (output_height + blockDim.y - 1) / blockDim.y);
    
    const float* bias_ptr = bias.defined() ? bias.data_ptr<float>() : nullptr;
    
    // Launch optimized kernel
    depthwise_conv2d_kernel<<<gridDim, blockDim>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias_ptr,
        output.data_ptr<float>(),
        batch_size, channels, height, width,
        kernel_size, stride, padding, output_height, output_width
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor depthwise_conv2d(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride, int padding
);
'''

ext = load_inline(
    name='depthwise_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
        # Create a Conv2d to match state_dict keys, but we won't use it for forward
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Ensure contiguous and correct format
        x = x.contiguous()
        weight = self.conv2d.weight.squeeze(1).contiguous()  # (in_channels, kernel_size, kernel_size)
        
        # For now, use the simple CUDA kernel
        # Weight format for our kernel: (channels, kernel_size, kernel_size)
        bias_tensor = self.conv2d.bias if self.conv2d.bias is not None else torch.Tensor()
        
        return ext.depthwise_conv2d(x, weight, bias_tensor, self.stride, self.padding)

def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [3, 3, 1, 0]  # in_channels, kernel_size, stride, padding
