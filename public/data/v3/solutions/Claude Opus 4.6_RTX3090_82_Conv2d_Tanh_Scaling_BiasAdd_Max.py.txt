import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>
#include <float.h>

// Fused kernel: tanh + scaling + bias + 2x2 max-pool
// Input: conv_out [N, C, H, W]
// Output: [N, C, H/2, W/2]
// Grid: (N*C*outH, ceil(outW/blockDim.x))
// Each thread computes one output pixel
__global__ void fused_tanh_scale_bias_maxpool_kernel(
    const float* __restrict__ conv_out,
    const float* __restrict__ bias,
    float* __restrict__ output,
    const float scaling_factor,
    const int N, const int C, const int H, const int W,
    const int outH, const int outW)
{
    int nc_oh = blockIdx.x;
    int ow = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (ow >= outW) return;
    
    int oh = nc_oh % outH;
    int nc = nc_oh / outH;
    int c = nc % C;
    
    float b = bias[c];
    
    int h0 = oh * 2;
    int w0 = ow * 2;
    
    int base = nc * H * W;
    
    // Read 2x2 window, apply tanh+scale+bias, find max
    // For H=254 and pool=2, outH=127, last window starts at h=252, reaches h=253 < 254, always valid
    int row0 = base + h0 * W;
    int row1 = base + (h0 + 1) * W;
    
    float v00 = conv_out[row0 + w0];
    float v01 = conv_out[row0 + w0 + 1];
    float v10 = conv_out[row1 + w0];
    float v11 = conv_out[row1 + w0 + 1];
    
    v00 = __fmul_rn(tanhf(v00), scaling_factor) + b;
    v01 = __fmul_rn(tanhf(v01), scaling_factor) + b;
    v10 = __fmul_rn(tanhf(v10), scaling_factor) + b;
    v11 = __fmul_rn(tanhf(v11), scaling_factor) + b;
    
    float mx = fmaxf(fmaxf(v00, v01), fmaxf(v10, v11));
    
    int out_idx = (nc * outH + oh) * outW + ow;
    output[out_idx] = mx;
}

torch::Tensor fused_tanh_scale_bias_maxpool(
    torch::Tensor conv_out,
    torch::Tensor bias,
    float scaling_factor,
    int pool_size)
{
    auto N = conv_out.size(0);
    auto C = conv_out.size(1);
    auto H = conv_out.size(2);
    auto W = conv_out.size(3);
    
    int outH = H / pool_size;
    int outW = W / pool_size;
    
    auto output = torch::empty({N, C, outH, outW}, conv_out.options());
    
    int nc_oh = N * C * outH;
    int block_x = 128;
    int grid_y = (outW + block_x - 1) / block_x;
    
    dim3 grid(nc_oh, grid_y);
    dim3 block(block_x);
    
    fused_tanh_scale_bias_maxpool_kernel<<<grid, block>>>(
        conv_out.data_ptr<float>(),
        bias.data_ptr<float>(),
        output.data_ptr<float>(),
        scaling_factor,
        N, C, H, W,
        outH, outW
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_tanh_scale_bias_maxpool(
    torch::Tensor conv_out,
    torch::Tensor bias,
    float scaling_factor,
    int pool_size);
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_tanh_scale_bias_maxpool'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86']
)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Use cuDNN for conv
        x = self.conv(x)
        # Fused tanh + scale + bias + maxpool
        x = ext.fused_tanh_scale_bias_maxpool(
            x, self.bias.view(-1), self.scaling_factor, self.pool_kernel_size)
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
