import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void gemv_int8_kernel(
    const int8_t* __restrict__ x,
    const int8_t* __restrict__ w,
    float* __restrict__ y,
    int B, int IN, int OUT,
    float scale)
{
    int b = blockIdx.x;
    int o = blockIdx.y;

    if (b >= B || o >= OUT) return;

    const int8_t* x_row = x + b * IN;
    const int8_t* w_row = w + o * IN;

    int tid = threadIdx.x;
    int stride = blockDim.x;

    int sum = 0;

    for (int i = tid; i < IN; i += stride) {
        sum += (int)x_row[i] * (int)w_row[i];
    }

    __shared__ int sdata[256];
    sdata[tid] = sum;
    __syncthreads();

    // reduction
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    if (tid == 0) {
        y[b * OUT + o] = (float)sdata[0] * scale;
    }
}

torch::Tensor gemv_int8(torch::Tensor x, torch::Tensor w, float scale) {
    int B = x.size(0);
    int IN = x.size(1);
    int OUT = w.size(0);

    auto y = torch::empty({B, OUT}, torch::dtype(torch::kFloat32).device(x.device()));

    dim3 grid(B, OUT);
    dim3 block(256);

    gemv_int8_kernel<<<grid, block>>>(
        x.data_ptr<int8_t>(),
        w.data_ptr<int8_t>(),
        y.data_ptr<float>(),
        B, IN, OUT,
        scale
    );

    return y;
}
'''

ext = load_inline(
    name='gemv_int8_ext',
    cpp_sources='torch::Tensor gemv_int8(torch::Tensor x, torch::Tensor w, float scale);',
    cuda_sources=cuda_src,
    functions=['gemv_int8'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        scale = float(x_scale.item() * self.weight_scale.item())
        y = ext.gemv_int8(x_q.contiguous(), self.weight_q.contiguous(), scale)
        return y.to(torch.float16)


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8, device='cuda')
    x_scale = torch.tensor(0.02, dtype=torch.float32, device='cuda')
    return [x_q, x_scale]


def get_init_inputs():
    return []
