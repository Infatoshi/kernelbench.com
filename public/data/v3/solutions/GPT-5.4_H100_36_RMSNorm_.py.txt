import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "layernorm"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _rmsnorm_kernel(
    x_ptr,
    y_ptr,
    outer_size,
    C,
    inner_size,
    eps,
    BLOCK_INNER: tl.constexpr,
):
    pid = tl.program_id(0)
    total = outer_size * inner_size
    if pid >= total:
        return

    outer_idx = pid // inner_size
    inner_idx = pid % inner_size

    base = outer_idx * C * inner_size + inner_idx

    acc = tl.zeros([BLOCK_INNER], dtype=tl.float32)
    offs = tl.arange(0, BLOCK_INNER)
    mask = offs < C
    ptrs = x_ptr + base + offs * inner_size
    x = tl.load(ptrs, mask=mask, other=0.0).to(tl.float32)
    acc += x * x
    mean_sq = tl.sum(acc, axis=0) / C
    inv_rms = tl.rsqrt(mean_sq + eps)
    y = x * inv_rms
    tl.store(y_ptr + base + offs * inner_size, y, mask=mask)


class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_cuda
        assert x.shape[1] == self.num_features
        y = torch.empty_like(x)

        outer_size = x.shape[0]
        inner_size = x.numel() // (outer_size * self.num_features)
        total = outer_size * inner_size

        if self.num_features <= 64:
            block_inner = 64
        elif self.num_features <= 128:
            block_inner = 128
        elif self.num_features <= 256:
            block_inner = 256
        else:
            raise RuntimeError("Unsupported num_features for this kernel")

        grid = (total,)
        _rmsnorm_kernel[grid](
            x,
            y,
            outer_size,
            self.num_features,
            inner_size,
            self.eps,
            BLOCK_INNER=block_inner,
        )
        return y


batch_size = 16
features = 64
dim1 = 256
dim2 = 256


def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [features]
