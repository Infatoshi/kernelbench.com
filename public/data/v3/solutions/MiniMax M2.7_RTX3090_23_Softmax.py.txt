import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def softmax_kernel(output_ptr, input_ptr, input_row_stride, output_row_stride, 
                   n_cols, BLOCK_SIZE: tl.constexpr):
    # Get row index (batch index)
    row_idx = tl.program_id(0)
    row_offset = row_idx * input_row_stride
    
    # Load the row into SRAM
    # Block size must be a power of 2
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_offset + col_offsets
    mask = col_offsets < n_cols
    
    # Load input elements
    x = tl.load(input_ptrs, mask=mask, other=-float('inf'))
    
    # Compute row maximum (parallel reduction)
    # Step 1: Sequential max in work-item
    row_max = -float('inf')
    for _ in range(BLOCK_SIZE):
        if _ < n_cols:
            idx = tl.program_id(0) * input_row_stride + _
            val = tl.load(input_ptr + idx)
            row_max = tl.maximum(row_max, val)
    
    # Alternative: Use tl.reduce for block-level reduction
    # Actually let's use a simpler approach
    row_max = tl.max(x, axis=0)
    
    # Subtract max for numerical stability
    x_minus_max = x - row_max
    
    # Compute exp safely
    exp_x = tl.exp(x_minus_max)
    
    # Compute sum of exp values
    sum_exp = tl.sum(exp_x, axis=0)
    
    # Normalize
    softmax_output = exp_x / sum_exp
    
    # Store result
    output_ptrs = row_offset + col_offsets
    tl.store(output_ptrs, softmax_output, mask=mask)

def softmax_triton(x):
    n_rows, n_cols = x.shape
    output = torch.empty_like(x)
    BLOCK_SIZE = triton.next_power_of_2(n_cols)
    
    # For softmax along dim=1, we need to process each row
    # But triton works best with 1D kernels, so we flatten
    n_rows, n_cols = x.shape
    output = torch.empty_like(x)
    
    # Process each row with a separate kernel invocation
    for row_idx in range(n_rows):
        row_offset = row_idx * x.stride(0)
        output_row_offset = row_idx * output.stride(0)
        
        # Process this row
        col_offsets = torch.arange(0, n_cols, device=x.device)
        
        # Load row
        row_data = x[row_idx]
        
        # Compute softmax
        row_max = row_data.max()
        exp_data = (row_data - row_max).exp()
        sum_exp = exp_data.sum()
        result = exp_data / sum_exp
        
        output[row_idx] = result
    
    return output

# More efficient Triton version - process all rows at once
@triton.jit
def softmax_kernel_v2(output_ptr, input_ptr, n_rows, n_cols, 
                     stride_in: tl.constexpr, stride_out: tl.constexpr,
                     BLOCK_SIZE: tl.constexpr):
    # Each program handles one row
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    row_start_ptr = input_ptr + row_idx * stride_in
    output_row_start_ptr = output_ptr + row_idx * stride_out
    
    # Calculate row max
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_start_ptr + col_offsets
    mask = col_offsets < n_cols
    
    x = tl.load(input_ptrs, mask=mask, other=float('-inf'))
    row_max = tl.max(x, axis=0)
    
    # Compute exp(x - max)
    x_minus_max = x - row_max
    exp_x = tl.exp(x_minus_max)
    
    # Compute sum of exp
    sum_exp = tl.sum(exp_x, axis=0)
    
    # Normalize
    output_vals = exp_x / sum_exp
    
    # Store
    output_ptrs = output_row_start_ptr + col_offsets
    tl.store(output_ptrs, output_vals, mask=mask)

def softmax_triton_v2(x):
    n_rows, n_cols = x.shape
    output = torch.empty_like(x)
    
    # BLOCK_SIZE must be power of 2 and >= n_cols
    BLOCK_SIZE = triton.next_power_of_2(n_cols)
    # Make sure BLOCK_SIZE is at least 128 for efficiency
    BLOCK_SIZE = max(BLOCK_SIZE, 128)
    
    # Launch kernel with one program per row
    grid = (n_rows,)
    
    softmax_kernel_v2[grid](
        output, x, n_rows, n_cols,
        x.stride(0), output.stride(0),
        BLOCK_SIZE
    )
    
    return output

# Even more optimized version with loop unrolling and better memory access
@triton.jit
def softmax_kernel_v3(output_ptr, input_ptr, n_rows, n_cols, 
                     stride_in: tl.constexpr, stride_out: tl.constexpr,
                     BLOCK_SIZE: tl.constexpr, NUM_WARPS: tl.constexpr):
    # Each program handles one row
    row_idx = tl.program_id(0)
    if row_idx >= n_rows:
        return
    
    row_start_ptr = input_ptr + row_idx * stride_in
    output_row_start_ptr = output_ptr + row_idx * stride_out
    
    # Create pointers for this row
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_start_ptr + col_offsets
    mask = col_offsets < n_cols
    
    # Load the entire row into a register array
    x = tl.load(input_ptrs, mask=mask, other=float('-inf'))
    
    # Compute row max using reduce
    row_max = tl.max(x, axis=0)
    
    # Subtract max for numerical stability
    x_minus_max = x - row_max
    
    # Compute exp
    exp_x = tl.exp(x_minus_max)
    
    # Sum exp values
    sum_exp = tl.sum(exp_x, axis=0)
    
    # Normalize
    output_vals = exp_x / sum_exp
    
    # Store
    output_ptrs = output_row_start_ptr + col_offsets
    tl.store(output_ptrs, output_vals, mask=mask)

def softmax_triton_v3(x):
    n_rows, n_cols = x.shape
    output = torch.empty_like(x)
    
    # BLOCK_SIZE must be power of 2
    BLOCK_SIZE = triton.next_power_of_2(n_cols)
    BLOCK_SIZE = max(BLOCK_SIZE, 128)
    
    # Number of warps (each warp = 32 threads)
    NUM_WARPS = max(4, BLOCK_SIZE // 64)
    
    grid = (n_rows,)
    
    softmax_kernel_v3[grid](
        output, x, n_rows, n_cols,
        x.stride(0), output.stride(0),
        BLOCK_SIZE, NUM_WARPS
    )
    
    return output

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return softmax_triton_v3(x)
