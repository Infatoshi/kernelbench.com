import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cublas_v2.h>
#include <c10/cuda/CUDAStream.h>
#include <math.h>

static cublasHandle_t cublas_handle = nullptr;

cublasHandle_t get_cublas_handle() {
    if (cublas_handle == nullptr) {
        cublasCreate(&cublas_handle);
        cublasSetMathMode(cublas_handle, CUBLAS_TF32_TENSOR_OP_MATH);
    }
    return cublas_handle;
}

// Fused: (matmul_out + linear_bias) -> swish -> + bias_add -> GroupNorm
// channels_per_group = 128, threads = 128 -> 1 element per thread
__global__ void fused_swish_bias_groupnorm_kernel(
    const float* __restrict__ matmul_out,
    const float* __restrict__ linear_bias,
    const float* __restrict__ bias_add,
    const float* __restrict__ gn_weight,
    const float* __restrict__ gn_bias,
    float* __restrict__ output,
    int out_features,
    int channels_per_group
) {
    int batch_idx = blockIdx.x;
    int group_idx = blockIdx.y;
    int group_start = group_idx * channels_per_group;
    int tid = threadIdx.x;
    
    int c = group_start + tid;
    int idx = batch_idx * out_features + c;
    
    float val = matmul_out[idx] + linear_bias[c];
    float sig = __fdividef(1.0f, 1.0f + __expf(-val));
    float result = val * sig + bias_add[c];
    
    float local_sum = result;
    float local_sum_sq = result * result;
    
    // Warp reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
        local_sum_sq += __shfl_down_sync(0xffffffff, local_sum_sq, offset);
    }
    
    __shared__ float shared_sum[4];
    __shared__ float shared_sum_sq[4];
    
    int lane = tid & 31;
    int warp_id = tid >> 5;
    
    if (lane == 0) {
        shared_sum[warp_id] = local_sum;
        shared_sum_sq[warp_id] = local_sum_sq;
    }
    __syncthreads();
    
    __shared__ float group_mean;
    __shared__ float group_inv_std;
    
    if (tid < 4) {
        local_sum = shared_sum[tid];
        local_sum_sq = shared_sum_sq[tid];
        local_sum += __shfl_down_sync(0xf, local_sum, 2);
        local_sum_sq += __shfl_down_sync(0xf, local_sum_sq, 2);
        local_sum += __shfl_down_sync(0xf, local_sum, 1);
        local_sum_sq += __shfl_down_sync(0xf, local_sum_sq, 1);
        
        if (tid == 0) {
            float mean = local_sum * (1.0f / channels_per_group);
            float var = local_sum_sq * (1.0f / channels_per_group) - mean * mean;
            group_mean = mean;
            group_inv_std = rsqrtf(var + 1e-5f);
        }
    }
    __syncthreads();
    
    float mean = group_mean;
    float inv_std = group_inv_std;
    
    float normalized = (result - mean) * inv_std;
    output[idx] = __fmaf_rn(normalized, gn_weight[c], gn_bias[c]);
}

// Version that reads matmul_out from half precision
__global__ void fused_swish_bias_groupnorm_kernel_fp16in(
    const half* __restrict__ matmul_out,
    const float* __restrict__ linear_bias,
    const float* __restrict__ bias_add,
    const float* __restrict__ gn_weight,
    const float* __restrict__ gn_bias,
    float* __restrict__ output,
    int out_features,
    int channels_per_group
) {
    int batch_idx = blockIdx.x;
    int group_idx = blockIdx.y;
    int group_start = group_idx * channels_per_group;
    int tid = threadIdx.x;
    
    int c = group_start + tid;
    int idx = batch_idx * out_features + c;
    
    float val = __half2float(matmul_out[idx]) + linear_bias[c];
    float sig = __fdividef(1.0f, 1.0f + __expf(-val));
    float result = val * sig + bias_add[c];
    
    float local_sum = result;
    float local_sum_sq = result * result;
    
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        local_sum += __shfl_down_sync(0xffffffff, local_sum, offset);
        local_sum_sq += __shfl_down_sync(0xffffffff, local_sum_sq, offset);
    }
    
    __shared__ float shared_sum[4];
    __shared__ float shared_sum_sq[4];
    
    int lane = tid & 31;
    int warp_id = tid >> 5;
    
    if (lane == 0) {
        shared_sum[warp_id] = local_sum;
        shared_sum_sq[warp_id] = local_sum_sq;
    }
    __syncthreads();
    
    __shared__ float group_mean;
    __shared__ float group_inv_std;
    
    if (tid < 4) {
        local_sum = shared_sum[tid];
        local_sum_sq = shared_sum_sq[tid];
        local_sum += __shfl_down_sync(0xf, local_sum, 2);
        local_sum_sq += __shfl_down_sync(0xf, local_sum_sq, 2);
        local_sum += __shfl_down_sync(0xf, local_sum, 1);
        local_sum_sq += __shfl_down_sync(0xf, local_sum_sq, 1);
        
        if (tid == 0) {
            float mean = local_sum * (1.0f / channels_per_group);
            float var = local_sum_sq * (1.0f / channels_per_group) - mean * mean;
            group_mean = mean;
            group_inv_std = rsqrtf(var + 1e-5f);
        }
    }
    __syncthreads();
    
    float mean = group_mean;
    float inv_std = group_inv_std;
    
    float normalized = (result - mean) * inv_std;
    output[idx] = __fmaf_rn(normalized, gn_weight[c], gn_bias[c]);
}

torch::Tensor fused_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor linear_bias,
    torch::Tensor bias_add,
    torch::Tensor gn_weight,
    torch::Tensor gn_bias,
    int num_groups
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    auto handle = get_cublas_handle();
    cublasSetStream(handle, at::cuda::getCurrentCUDAStream());
    
    // Try FP16 GEMM with FP32 accumulation for speed
    // Convert input and weight to FP16
    auto input_h = input.to(torch::kFloat16);
    auto weight_h = weight.to(torch::kFloat16);
    auto matmul_out_h = torch::empty({batch_size, out_features}, input.options().dtype(torch::kFloat16));
    
    __half alpha_h = __float2half(1.0f);
    __half beta_h = __float2half(0.0f);
    
    // Use cublasHgemm for FP16
    cublasHgemm(handle,
        CUBLAS_OP_T,
        CUBLAS_OP_N,
        out_features,
        batch_size,
        in_features,
        &alpha_h,
        (const __half*)weight_h.data_ptr<at::Half>(), in_features,
        (const __half*)input_h.data_ptr<at::Half>(), in_features,
        &beta_h,
        (__half*)matmul_out_h.data_ptr<at::Half>(), out_features
    );
    
    auto output = torch::empty({batch_size, out_features}, input.options());
    
    int channels_per_group = out_features / num_groups;
    int threads = channels_per_group;
    
    dim3 grid(batch_size, num_groups);
    
    fused_swish_bias_groupnorm_kernel_fp16in<<<grid, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        (const half*)matmul_out_h.data_ptr<at::Half>(),
        linear_bias.data_ptr<float>(),
        bias_add.data_ptr<float>(),
        gn_weight.data_ptr<float>(),
        gn_bias.data_ptr<float>(),
        output.data_ptr<float>(),
        out_features,
        channels_per_group
    );
    
    return output;
}

torch::Tensor fused_forward_fp32(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor linear_bias,
    torch::Tensor bias_add,
    torch::Tensor gn_weight,
    torch::Tensor gn_bias,
    int num_groups
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    auto matmul_out = torch::empty({batch_size, out_features}, input.options());
    
    auto handle = get_cublas_handle();
    cublasSetStream(handle, at::cuda::getCurrentCUDAStream());
    
    float alpha = 1.0f;
    float beta = 0.0f;
    
    cublasSgemm(handle,
        CUBLAS_OP_T,
        CUBLAS_OP_N,
        out_features,
        batch_size,
        in_features,
        &alpha,
        weight.data_ptr<float>(), in_features,
        input.data_ptr<float>(), in_features,
        &beta,
        matmul_out.data_ptr<float>(), out_features
    );
    
    auto output = torch::empty_like(matmul_out);
    
    int channels_per_group = out_features / num_groups;
    int threads = channels_per_group;
    
    dim3 grid(batch_size, num_groups);
    
    fused_swish_bias_groupnorm_kernel<<<grid, threads, 0, at::cuda::getCurrentCUDAStream()>>>(
        matmul_out.data_ptr<float>(),
        linear_bias.data_ptr<float>(),
        bias_add.data_ptr<float>(),
        gn_weight.data_ptr<float>(),
        gn_bias.data_ptr<float>(),
        output.data_ptr<float>(),
        out_features,
        channels_per_group
    );
    
    return output;
}

cpp_source = r'''

ext = load_inline(
    name='fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward', 'fused_forward_fp32'],
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
    extra_ldflags=['-lcublas'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.num_groups = num_groups
        # Pre-converted weights (will be set after load_state_dict)
        self._weight_h = None
        
    def _ensure_fp16_weights(self):
        if self._weight_h is None or self._weight_h.data_ptr() == 0:
            self._weight_h = self.matmul.weight.half()

    def forward(self, x):
        return ext.fused_forward_fp32(
            x,
            self.matmul.weight,
            self.matmul.bias,
            self.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.num_groups
        )


batch_size = 128
in_features = 4096
out_features = 4096
num_groups = 32
bias_shape = (out_features,)

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, num_groups, bias_shape]
