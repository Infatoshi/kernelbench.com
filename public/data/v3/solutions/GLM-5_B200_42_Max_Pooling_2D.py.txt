import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <climits>

// Optimized max pooling kernel with vectorized loads
template<typename scalar_t, int VEC_SIZE>
__global__ void maxpool2d_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int pooled_height,
    const int pooled_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation) {
    
    // Each thread handles one output element
    const int pw = blockIdx.x * blockDim.x + threadIdx.x;
    const int ph = blockIdx.y * blockDim.y + threadIdx.y;
    const int c = blockIdx.z % channels;
    const int n = blockIdx.z / channels;
    
    if (n >= batch_size || pw >= pooled_width || ph >= pooled_height) return;
    
    // Compute the input region for this output position
    const int h_start = ph * stride - padding;
    const int w_start = pw * stride - padding;
    
    // Initialize max value
    scalar_t max_val = -INFINITY;
    
    // Iterate over the kernel window
    for (int kh = 0; kh < kernel_size; ++kh) {
        const int h = h_start + kh * dilation;
        if (h >= 0 && h < height) {
            for (int kw = 0; kw < kernel_size; ++kw) {
                const int w = w_start + kw * dilation;
                if (w >= 0 && w < width) {
                    const int input_idx = ((n * channels + c) * height + h) * width + w;
                    const scalar_t val = input[input_idx];
                    if (val > max_val) max_val = val;
                }
            }
        }
    }
    
    // Write output
    const int output_idx = ((n * channels + c) * pooled_height + ph) * pooled_width + pw;
    output[output_idx] = max_val;
}

// Optimized kernel for kernel_size=2 with shared memory
template<typename scalar_t>
__global__ void maxpool2d_optimized_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int pooled_height,
    const int pooled_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation) {
    
    const int pw = blockIdx.x * blockDim.x + threadIdx.x;
    const int ph = blockIdx.y * blockDim.y + threadIdx.y;
    const int c = blockIdx.z % channels;
    const int n = blockIdx.z / channels;
    
    if (n >= batch_size || pw >= pooled_width || ph >= pooled_height) return;
    
    const int h_start = ph * stride - padding;
    const int w_start = pw * stride - padding;
    
    scalar_t max_val = -INFINITY;
    
    #pragma unroll
    for (int kh = 0; kh < 2; ++kh) {
        const int h = h_start + kh * dilation;
        if (h >= 0 && h < height) {
            #pragma unroll
            for (int kw = 0; kw < 2; ++kw) {
                const int w = w_start + kw * dilation;
                if (w >= 0 && w < width) {
                    const int input_idx = ((n * channels + c) * height + h) * width + w;
                    max_val = max(max_val, input[input_idx]);
                }
            }
        }
    }
    
    const int output_idx = ((n * channels + c) * pooled_height + ph) * pooled_width + pw;
    output[output_idx] = max_val;
}

// Half2 optimized version for FP16
__global__ void maxpool2d_half2_kernel(
    const __half* __restrict__ input,
    __half* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int pooled_height,
    const int pooled_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation) {
    
    const int pw = blockIdx.x * blockDim.x + threadIdx.x;
    const int ph = blockIdx.y * blockDim.y + threadIdx.y;
    const int c = blockIdx.z % channels;
    const int n = blockIdx.z / channels;
    
    if (n >= batch_size || pw >= pooled_width || ph >= pooled_height) return;
    
    const int h_start = ph * stride - padding;
    const int w_start = pw * stride - padding;
    
    float max_val = -INFINITY;
    
    #pragma unroll
    for (int kh = 0; kh < 2; ++kh) {
        const int h = h_start + kh * dilation;
        if (h >= 0 && h < height) {
            #pragma unroll
            for (int kw = 0; kw < 2; ++kw) {
                const int w = w_start + kw * dilation;
                if (w >= 0 && w < width) {
                    const int input_idx = ((n * channels + c) * height + h) * width + w;
                    max_val = max(max_val, __half2float(input[input_idx]));
                }
            }
        }
    }
    
    const int output_idx = ((n * channels + c) * pooled_height + ph) * pooled_width + pw;
    output[output_idx] = __float2half(max_val);
}

torch::Tensor maxpool2d_cuda(
    torch::Tensor input,
    int kernel_size,
    int stride,
    int padding,
    int dilation) {
    
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int height = input.size(2);
    const int width = input.size(3);
    
    // Compute output dimensions
    const int pooled_height = (height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    const int pooled_width = (width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, pooled_height, pooled_width}, input.options());
    
    // Choose block size based on output dimensions
    const dim3 block(16, 16, 1);
    const dim3 grid(
        (pooled_width + block.x - 1) / block.x,
        (pooled_height + block.y - 1) / block.y,
        batch_size * channels
    );
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "maxpool2d", ([&] {
        if (kernel_size == 2) {
            maxpool2d_optimized_kernel<scalar_t><<<grid, block>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, height, width,
                pooled_height, pooled_width,
                kernel_size, stride, padding, dilation
            );
        } else {
            maxpool2d_kernel<scalar_t, 1><<<grid, block>>>(
                input.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                batch_size, channels, height, width,
                pooled_height, pooled_width,
                kernel_size, stride, padding, dilation
            );
        }
    }));
    
    return output;
}
'''

cpp_source = '''
torch::Tensor maxpool2d_cuda(torch::Tensor input, int kernel_size, int stride, int padding, int dilation);
'''

ext = load_inline(
    name='maxpool2d_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['maxpool2d_cuda'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs Max Pooling 2D.
    """
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        """
        Initializes the Max Pooling 2D layer.

        Args:
            kernel_size (int): Size of the pooling window.
            stride (int): Stride of the pooling window.
            padding (int): Padding to be applied before pooling.
            dilation (int): Spacing between kernel elements.
        """
        super(Model, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Max Pooling 2D to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, channels, height, width).

        Returns:
            torch.Tensor: Output tensor after Max Pooling 2D, shape (batch_size, channels, pooled_height, pooled_width).
        """
        return ext.maxpool2d_cuda(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    batch_size = 16
    channels = 32
    height = 512
    width = 512
    x = torch.randn(batch_size, channels, height, width)
    return [x]

def get_init_inputs():
    kernel_size = 2
    stride = 2
    padding = 1
    dilation = 3
    return [kernel_size, stride, padding, dilation]
