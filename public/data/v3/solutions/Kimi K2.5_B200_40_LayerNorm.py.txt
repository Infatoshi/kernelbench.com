import torch
import torch.nn as nn

# Constants matching reference.py
batch_size = 16
features = 64
dim1 = 256
dim2 = 256


class Model(nn.Module):
    """
    Simple model that performs Layer Normalization.
    """
    def __init__(self, normalized_shape: tuple):
        """
        Initializes the LayerNorm layer.

        Args:
            normalized_shape (tuple): Shape of the input tensor to be normalized.
        """
        super(Model, self).__init__()
        
        # Compute total size from normalized_shape tuple
        N = 1
        for dim in normalized_shape:
            N *= dim
        
        self.N = N
        self.normalized_shape = normalized_shape
        
        # LayerNorm affine parameters
        self.eps = 1e-5
        self.weight = nn.Parameter(torch.ones(N))
        self.bias = nn.Parameter(torch.zeros(N))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Layer Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (*, normalized_shape).

        Returns:
            torch.Tensor: Output tensor with Layer Normalization applied, same shape as input.
        """
        original_shape = x.shape
        batch = original_shape[0]
        
        # Reshape to [batch, N]
        x_2d = x.reshape(batch, self.N)
        
        # Compute mean and variance
        mean = x_2d.mean(dim=1, keepdim=True)
        var = x_2d.var(dim=1, keepdim=True, unbiased=False)
        inv_std = torch.rsqrt(var + self.eps)
        
        # Normalize
        x_norm = (x_2d - mean) * inv_std
        
        # Apply affine transform
        output = x_norm * self.weight + self.bias
        
        # Reshape back
        return output.reshape(original_shape)


def get_inputs():
    """
    Returns a list of input tensors for testing.
    """
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    """
    Returns a list of arguments to initialize the Model.
    """
    return [(features, dim1, dim2)]
