import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>

// Fused activation: x * tanh(softplus(x))
// softplus(x) = log(1 + exp(x))
// tanh(softplus(x)) = tanh(log(1 + exp(x)))
__device__ __forceinline__ float fused_activation(float x) {
    float softplus_x = logf(1.0f + expf(x));
    float tanh_sp = tanhf(softplus_x);
    return x * tanh_sp;
}

__device__ __forceinline__ float batchnorm_apply(float x, float mean, float var, float weight, float bias, float eps) {
    return weight * (x - mean) / sqrtf(var + eps) + bias;
}

// Direct 3x3 convolution with fused activation and batchnorm
// Input: [N, C, H, W], Weight: [out_ch, in_ch, 3, 3]
// Output: [N, out_ch, H-2, W-2] (no padding)
__global__ void fused_conv_act_bn_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ bn_weight,
    const float* __restrict__ bn_bias,
    const float* __restrict__ bn_running_mean,
    const float* __restrict__ bn_running_var,
    float* __restrict__ output,
    int N, int C, int H, int W,
    int out_ch, int kernel_size,
    float eps
) {
    int n = blockIdx.z;
    int oc = blockIdx.y;
    int oh = blockIdx.x * blockDim.y + threadIdx.y;
    int ow = threadIdx.x;
    
    // Output dimensions are (H - kernel_size + 1) x (W - kernel_size + 1)
    int out_H = H - kernel_size + 1;
    int out_W = W - kernel_size + 1;
    
    if (n >= N || oc >= out_ch || oh >= out_H || ow >= out_W) return;
    
    float sum = bias[oc];
    
    // 3x3 convolution (no padding)
    for (int ic = 0; ic < C; ic++) {
        for (int kh = 0; kh < kernel_size; kh++) {
            for (int kw = 0; kw < kernel_size; kw++) {
                int ih = oh + kh;
                int iw = ow + kw;
                
                int in_idx = ((n * C + ic) * H + ih) * W + iw;
                int w_idx = ((oc * C + ic) * kernel_size + kh) * kernel_size + kw;
                sum += input[in_idx] * weight[w_idx];
            }
        }
    }
    
    // Apply activation: x * tanh(softplus(x))
    float activated = fused_activation(sum);
    
    // Apply batchnorm
    float mean = bn_running_mean[oc];
    float var = bn_running_var[oc];
    float w = bn_weight[oc];
    float b = bn_bias[oc];
    
    output[((n * out_ch + oc) * out_H + oh) * out_W + ow] = batchnorm_apply(activated, mean, var, w, b, eps);
}

torch::Tensor fused_conv_act_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    float eps
) {
    int N = input.size(0);
    int C = input.size(1);
    int H = input.size(2);
    int W = input.size(3);
    int out_ch = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_H = H - kernel_size + 1;
    int out_W = W - kernel_size + 1;
    
    auto output = torch::empty({N, out_ch, out_H, out_W}, input.options());
    
    dim3 blockDim(16, 16);
    dim3 gridDim((out_H + blockDim.y - 1) / blockDim.y, out_ch, N);
    
    fused_conv_act_bn_kernel<<<gridDim, blockDim>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        bn_weight.data_ptr<float>(),
        bn_bias.data_ptr<float>(),
        bn_running_mean.data_ptr<float>(),
        bn_running_var.data_ptr<float>(),
        output.data_ptr<float>(),
        N, C, H, W, out_ch, kernel_size, eps
    );
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_conv_act_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    torch::Tensor bn_running_mean,
    torch::Tensor bn_running_var,
    float eps
);
"""

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_act_bn'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
)

class Model(nn.Module):
    """
    Simple model that performs a convolution, applies activation, and then applies Batch Normalization.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        # Get conv parameters
        weight = self.conv.weight
        bias = self.conv.bias
        
        # Get batchnorm parameters (in eval mode, use running stats)
        bn_weight = self.bn.weight
        bn_bias = self.bn.bias
        bn_running_mean = self.bn.running_mean
        bn_running_var = self.bn.running_var
        
        return ext.fused_conv_act_bn(
            x, weight, bias,
            bn_weight, bn_bias,
            bn_running_mean, bn_running_var,
            self.eps
        )

batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
