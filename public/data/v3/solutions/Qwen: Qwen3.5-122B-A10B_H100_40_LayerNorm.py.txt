import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def layer_norm_fwd_kernel(
    X,  # pointer to the input
    Y,  # pointer to the output
    W,  # pointer to the weights
    B,  # pointer to the biases
    Mean,  # pointer to the mean
    Rstd,  # pointer to the 1/std
    stride,  # row stride
    N,  # number of columns
    eps,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Triton kernel for LayerNorm forward pass.
    Each program instance handles one row.
    """
    row_idx = tl.program_id(0)
    
    # Compute the starting offset for this row
    row_start = row_idx * stride
    
    # First pass: compute mean
    acc_mean = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    for start in range(0, N, BLOCK_SIZE):
        cols = start + tl.arange(0, BLOCK_SIZE)
        x = tl.load(X + row_start + cols, mask=cols < N, other=0.0)
        acc_mean += x
    mean = tl.sum(acc_mean) / N
    
    # Second pass: compute variance
    acc_var = tl.zeros((BLOCK_SIZE,), dtype=tl.float32)
    for start in range(0, N, BLOCK_SIZE):
        cols = start + tl.arange(0, BLOCK_SIZE)
        x = tl.load(X + row_start + cols, mask=cols < N, other=0.0)
        diff = x - mean
        acc_var += diff * diff
    var = tl.sum(acc_var) / N
    rstd = 1.0 / tl.sqrt(var + eps)
    
    # Store mean and rstd
    tl.store(Mean + row_idx, mean)
    tl.store(Rstd + row_idx, rstd)
    
    # Third pass: normalize and write output
    for start in range(0, N, BLOCK_SIZE):
        cols = start + tl.arange(0, BLOCK_SIZE)
        x = tl.load(X + row_start + cols, mask=cols < N, other=0.0)
        
        if W is not None:
            w = tl.load(W + cols, mask=cols < N, other=1.0)
        else:
            w = 1.0
        
        if B is not None:
            b = tl.load(B + cols, mask=cols < N, other=0.0)
        else:
            b = 0.0
        
        x_hat = (x - mean) * rstd
        y = x_hat * w + b
        
        tl.store(Y + row_start + cols, y, mask=cols < N)


class TritonLayerNorm(nn.Module):
    """
    LayerNorm implementation using Triton kernels.
    """
    def __init__(self, normalized_shape, eps=1e-5, elementwise_affine=True):
        super().__init__()
        if isinstance(normalized_shape, int):
            normalized_shape = (normalized_shape,)
        self.normalized_shape = tuple(normalized_shape)
        self.eps = eps
        self.elementwise_affine = elementwise_affine
        
        # Compute total number of elements in normalized shape
        self.total_normalized_size = 1
        for s in self.normalized_shape:
            self.total_normalized_size *= s
        
        if self.elementwise_affine:
            self.weight = nn.Parameter(torch.ones(self.total_normalized_size))
            self.bias = nn.Parameter(torch.zeros(self.total_normalized_size))
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)
    
    def load_state_dict(self, state_dict, strict=True, assign=False):
        """
        Override load_state_dict to handle shape conversion from multi-dimensional to 1D.
        """
        # Check if we need to flatten the weight and bias
        if 'ln.weight' in state_dict and self.weight is not None:
            ref_weight = state_dict['ln.weight']
            if ref_weight.dim() > 1:
                state_dict['ln.weight'] = ref_weight.flatten()
        
        if 'ln.bias' in state_dict and self.bias is not None:
            ref_bias = state_dict['ln.bias']
            if ref_bias.dim() > 1:
                state_dict['ln.bias'] = ref_bias.flatten()
        
        return super().load_state_dict(state_dict, strict=strict, assign=assign)
    
    def forward(self, x):
        # Get the shape
        original_shape = x.shape
        *batch_dims, N = x.shape
        
        # Reshape to 2D for processing - flatten the normalized dimensions
        x = x.reshape(-1, N)
        M, N = x.shape
        
        # Allocate output and statistics
        y = torch.empty_like(x)
        mean = torch.empty(M, dtype=torch.float32, device=x.device)
        rstd = torch.empty(M, dtype=torch.float32, device=x.device)
        
        # Choose block size based on N
        BLOCK_SIZE = triton.next_power_of_2(N)
        if BLOCK_SIZE > 4096:
            BLOCK_SIZE = 4096
        
        # Launch kernel
        grid = (M,)
        layer_norm_fwd_kernel[grid](
            x, y, 
            self.weight if self.elementwise_affine else None,
            self.bias if self.elementwise_affine else None,
            mean, rstd,
            N, N,  # stride is N for contiguous rows
            self.eps,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        # Reshape back to original shape
        return y.reshape(original_shape)


class Model(nn.Module):
    """
    Model that performs Layer Normalization using optimized Triton kernel.
    """
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = TritonLayerNorm(normalized_shape=normalized_shape)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.ln(x)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
