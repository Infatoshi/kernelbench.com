import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

__global__ void fused_conv_pool_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size, int in_channels, int in_h, int in_w,
    int out_channels, int conv_h, int conv_w,
    int kernel_size, float sub1, float sub2,
    int pool_size, int pool_out_h, int pool_out_w) 
{
    int pw = blockIdx.x * blockDim.x + threadIdx.x;
    int ph = blockIdx.y * blockDim.y + threadIdx.y;
    int bc = blockIdx.z;
    int b = bc / out_channels;
    int c = bc % out_channels;

    if (b >= batch_size || ph >= pool_out_h || pw >= pool_out_w) return;

    float pool_sum = 0.0f;
    float b_val = bias ? bias[c] : 0.0f;
    
    float weight_cache[27]; 
    if (in_channels == 3 && kernel_size == 3) {
        for (int i = 0; i < 27; ++i) {
            weight_cache[i] = weight[c * 27 + i];
        }
    }

    for (int i = 0; i < pool_size; ++i) {
        for (int j = 0; j < pool_size; ++j) {
            int ch = ph * pool_size + i;
            int cw = pw * pool_size + j;
            float val = b_val;
            
            if (in_channels == 3 && kernel_size == 3) {
                #pragma unroll
                for (int ic = 0; ic < 3; ++ic) {
                    const float* input_ptr = &input[((b * 3 + ic) * in_h + ch) * in_w + cw];
                    val += input_ptr[0] * weight_cache[ic * 9 + 0];
                    val += input_ptr[1] * weight_cache[ic * 9 + 1];
                    val += input_ptr[2] * weight_cache[ic * 9 + 2];
                    input_ptr += in_w;
                    val += input_ptr[0] * weight_cache[ic * 9 + 3];
                    val += input_ptr[1] * weight_cache[ic * 9 + 4];
                    val += input_ptr[2] * weight_cache[ic * 9 + 5];
                    input_ptr += in_w;
                    val += input_ptr[0] * weight_cache[ic * 9 + 6];
                    val += input_ptr[1] * weight_cache[ic * 9 + 7];
                    val += input_ptr[2] * weight_cache[ic * 9 + 8];
                }
            } else {
                for (int ic = 0; ic < in_channels; ++ic) {
                    for (int kh = 0; kh < kernel_size; ++kh) {
                        for (int kw = 0; kw < kernel_size; ++kw) {
                            int ih = ch + kh;
                            int iw = cw + kw;
                            val += input[((b * in_channels + ic) * in_h + ih) * in_w + iw] * 
                                   weight[((c * in_channels + ic) * kernel_size + kh) * kernel_size + kw];
                        }
                    }
                }
            }
            pool_sum += tanhf(val - sub1) - sub2;
        }
    }
    output[((b * out_channels + c) * pool_out_h + ph) * pool_out_w + pw] = pool_sum / (pool_size * pool_size);
}

torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, at::optional<torch::Tensor> bias, 
                      float sub1, float sub2, int pool_size) {
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    int conv_h = in_h - kernel_size + 1;
    int conv_w = in_w - kernel_size + 1;
    int pool_out_h = conv_h / pool_size;
    int pool_out_w = conv_w / pool_size;
    
    auto output = torch::empty({batch_size, out_channels, pool_out_h, pool_out_w}, input.options());
    
    dim3 block(16, 16);
    dim3 grid((pool_out_w + block.x - 1) / block.x, (pool_out_h + block.y - 1) / block.y, batch_size * out_channels);
    
    fused_conv_pool_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias.value().data_ptr<float>() : nullptr,
        output.data_ptr<float>(),
        batch_size, in_channels, in_h, in_w,
        out_channels, conv_h, conv_w,
        kernel_size, sub1, sub2,
        pool_size, pool_out_h, pool_out_w
    );
    
    return output;
}
'''

fused_lib = load_inline(
    name='fused_lib',
    cpp_sources='torch::Tensor fused_op(torch::Tensor input, torch::Tensor weight, at::optional<torch::Tensor> bias, float sub1, float sub2, int pool_size);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = float(subtract1_value)
        self.subtract2_value = float(subtract2_value)
        self.kernel_size_pool = int(kernel_size_pool)

    def forward(self, x):
        return fused_lib.fused_op(x.contiguous(), self.conv.weight.contiguous(), self.conv.bias.contiguous() if self.conv.bias is not None else None, self.subtract1_value, self.subtract2_value, self.kernel_size_pool)

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
