import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# CUDA kernel for matrix-vector multiplication C = A * B
# A: (M, K), B: (K,), C: (M,)
# Each block processes one row of A. Threads within a block cooperate to compute
# the dot product of the row with B using vectorized float4 loads and a tree reduction.
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

#define BLOCK_SIZE 256

__global__ void gemv_kernel(const float* __restrict__ A,
                            const float* __restrict__ B,
                            float* __restrict__ C,
                            int M,
                            int K) {
    int row = blockIdx.x;
    if (row >= M) return;

    int tid = threadIdx.x;
    // Number of float4 elements (K must be multiple of 4 for this kernel)
    int n4 = K >> 2;

    const float4* A4 = reinterpret_cast<const float4*>(A + row * K);
    const float4* B4 = reinterpret_cast<const float4*>(B);

    float sum = 0.0f;

    // Grid-stride loop over the K dimension with float4 vectorized loads
    for (int i = tid; i < n4; i += blockDim.x) {
        float4 a = __ldg(A4 + i);
        float4 b = __ldg(B4 + i);
        sum += a.x * b.x;
        sum += a.y * b.y;
        sum += a.z * b.z;
        sum += a.w * b.w;
    }

    // Store partial sum in shared memory and perform tree reduction
    __shared__ float sdata[BLOCK_SIZE];
    sdata[tid] = sum;
    __syncthreads();

    // Reduce within the block
    #pragma unroll
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }

    // Write the result
    if (tid == 0) {
        C[row] = sdata[0];
    }
}

torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);

    // Ensure contiguous memory layout
    auto A_contig = A.contiguous();
    auto B_contig = B.contiguous();

    auto C = torch::empty({M}, A.options());

    dim3 grid(M);
    dim3 block(BLOCK_SIZE);

    gemv_kernel<<<grid, block>>>(
        A_contig.data_ptr<float>(),
        B_contig.data_ptr<float>(),
        C.data_ptr<float>(),
        M,
        K);

    return C;
}
'''

cpp_source = r'''
torch::Tensor gemv_forward(torch::Tensor A, torch::Tensor B);
'''

# Load the inline CUDA extension
gemv_cuda = load_inline(
    name='gemv_cuda',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_forward'],
    extra_cuda_cflags=['-O3', '-arch=sm_86'],
    verbose=False,
)


class Model(nn.Module):
    """
    Optimized model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1) or (K,).

        Returns:
            Output vector of shape (M, 1).
        """
        # Flatten B to 1-D if needed
        if B.dim() == 2 and B.size(1) == 1:
            B = B.squeeze(1)
        # Compute GEMV
        C = gemv_cuda.gemv_forward(A, B)
        # Reshape to (M, 1) to match reference
        return C.unsqueeze(1)
