import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused kernel: after GEMM produces (batch, out_features),
# compute: result[b] = scale * sum_{p=0}^{out_features/2-1} max(mat[b,2p], mat[b,2p+1])
# This avoids materializing the maxpool output tensor.
#
# We also handle general kernel_size (not just 2).
# The GEMM is done via cublas for maximum efficiency.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <float.h>

// -----------------------------------------------------------------------
// Fused MaxPool1d(kernel_size) + Sum(dim=1) + Scale kernel
// Input:  mat[batch, out_features]  (row-major)
// Output: out[batch]
// Each block handles one batch row.
// Within a row, we compute sum of max(mat[b, ks*p : ks*(p+1)]) over p.
// -----------------------------------------------------------------------
template<int BLOCK_SIZE>
__global__ void fused_maxpool_sum_scale_kernel(
    const float* __restrict__ mat,
    float* __restrict__ out,
    int batch,
    int out_features,
    int kernel_size,
    float scale_factor
) {
    int b = blockIdx.x;
    if (b >= batch) return;

    const float* row = mat + b * out_features;
    int num_pools = out_features / kernel_size;

    float thread_sum = 0.0f;

    // Each thread processes multiple pool windows
    for (int p = threadIdx.x; p < num_pools; p += BLOCK_SIZE) {
        int base = p * kernel_size;
        float val = row[base];
        for (int k = 1; k < kernel_size; k++) {
            float v = row[base + k];
            if (v > val) val = v;
        }
        thread_sum += val;
    }

    // Warp reduction
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 16);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 8);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 4);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 2);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 1);

    // Block reduction using shared memory
    __shared__ float smem[BLOCK_SIZE / 32];
    int warp_id = threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    if (lane_id == 0) {
        smem[warp_id] = thread_sum;
    }
    __syncthreads();

    if (warp_id == 0) {
        int num_warps = BLOCK_SIZE / 32;
        float val = (lane_id < num_warps) ? smem[lane_id] : 0.0f;
        val += __shfl_down_sync(0xffffffff, val, 16);
        val += __shfl_down_sync(0xffffffff, val, 8);
        val += __shfl_down_sync(0xffffffff, val, 4);
        val += __shfl_down_sync(0xffffffff, val, 2);
        val += __shfl_down_sync(0xffffffff, val, 1);
        if (lane_id == 0) {
            out[b] = val * scale_factor;
        }
    }
}

// Specialization for kernel_size=2 (most common case) with vectorized loads
template<int BLOCK_SIZE>
__global__ void fused_maxpool2_sum_scale_kernel(
    const float* __restrict__ mat,
    float* __restrict__ out,
    int batch,
    int out_features,
    float scale_factor
) {
    int b = blockIdx.x;
    if (b >= batch) return;

    const float2* row = reinterpret_cast<const float2*>(mat + b * out_features);
    int num_pools = out_features / 2;

    float thread_sum = 0.0f;

    for (int p = threadIdx.x; p < num_pools; p += BLOCK_SIZE) {
        float2 v = row[p];
        thread_sum += (v.x > v.y) ? v.x : v.y;
    }

    // Warp reduction
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 16);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 8);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 4);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 2);
    thread_sum += __shfl_down_sync(0xffffffff, thread_sum, 1);

    __shared__ float smem[BLOCK_SIZE / 32];
    int warp_id = threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    if (lane_id == 0) {
        smem[warp_id] = thread_sum;
    }
    __syncthreads();

    if (warp_id == 0) {
        int num_warps = BLOCK_SIZE / 32;
        float val = (lane_id < num_warps) ? smem[lane_id] : 0.0f;
        val += __shfl_down_sync(0xffffffff, val, 16);
        val += __shfl_down_sync(0xffffffff, val, 8);
        val += __shfl_down_sync(0xffffffff, val, 4);
        val += __shfl_down_sync(0xffffffff, val, 2);
        val += __shfl_down_sync(0xffffffff, val, 1);
        if (lane_id == 0) {
            out[b] = val * scale_factor;
        }
    }
}

// -----------------------------------------------------------------------
// Full fused forward: linear + maxpool + sum + scale
// Uses cuBLAS for the GEMM part
// -----------------------------------------------------------------------
static cublasHandle_t cublas_handle = nullptr;

void ensure_cublas() {
    if (cublas_handle == nullptr) {
        cublasCreate(&cublas_handle);
        cublasSetMathMode(cublas_handle, CUBLAS_DEFAULT_MATH);
    }
}

torch::Tensor fused_forward(
    torch::Tensor x,        // (batch, in_features)
    torch::Tensor weight,   // (out_features, in_features)
    torch::Tensor bias,     // (out_features,)
    int kernel_size,
    float scale_factor
) {
    ensure_cublas();

    int batch = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);

    // GEMM: out_mat = x @ weight.T + bias
    // out_mat shape: (batch, out_features)
    auto out_mat = torch::empty({batch, out_features}, x.options());

    // cuBLAS is column-major. We compute:
    // out_mat^T (out_features x batch) = weight (out_features x in_features) * x^T (in_features x batch)
    // In cuBLAS terms: C = alpha * A * B + beta * C
    // A = weight (out_features x in_features), stored row-major => col-major transpose: (in_features x out_features)
    // B = x^T stored row-major => col-major: (batch x in_features) 
    // We want: out_mat (batch x out_features) row-major = x (batch x in_features) * weight^T (in_features x out_features)
    // In col-major: out_mat^T (out_features x batch) = weight (out_features x in_features) * x^T (in_features x batch)
    // cublasSgemm(handle, CUBLAS_OP_T, CUBLAS_OP_N, out_features, batch, in_features, &alpha, weight, in_features, x, in_features, &beta, out_mat, out_features)

    float alpha = 1.0f, beta = 0.0f;

    // First fill out_mat with bias broadcast
    // out_mat[b, j] = bias[j] for all b
    // We do this by setting beta=1.0 and initializing out_mat with broadcast bias
    // Use cublasSger or just expand bias
    // Simpler: use cudaMemcpy2D or a small kernel to broadcast bias

    // Broadcast bias: out_mat[b,:] = bias for all b
    // Then use beta=1.0 in GEMM
    {
        // Use cublas to set out_mat = ones_col * bias_row
        // Actually just do it with a small kernel or torch ops
        // Since bias is small (4096), we can use torch::addmm equivalent
        // Let's just expand bias manually with a kernel call
        
        // Actually: copy bias to each row of out_mat
        // out_mat shape (batch, out_features), bias shape (out_features,)
        // Use cudaMemcpy2D-like approach: each row = bias
        // Simple: launch small kernel
        // 
        // Or: use cublas with a column of ones
        // Let's use the simplest approach: expand bias
        
        auto bias_expanded = bias.unsqueeze(0).expand({batch, out_features}).contiguous();
        out_mat.copy_(bias_expanded);
        beta = 1.0f;
    }

    cublasStatus_t status = cublasSgemm(
        cublas_handle,
        CUBLAS_OP_T,    // weight is transposed (we want weight^T for row-major matmul)
        CUBLAS_OP_N,    // x is not transposed
        out_features,   // m
        batch,          // n
        in_features,    // k
        &alpha,
        weight.data_ptr<float>(),   // A: weight (out_features x in_features) in row-major = (in_features x out_features) col-major
        in_features,                // lda
        x.data_ptr<float>(),        // B: x (batch x in_features) in row-major = (in_features x batch) col-major... wait
        in_features,                // ldb
        &beta,
        out_mat.data_ptr<float>(),  // C: (batch x out_features) row-major = (out_features x batch) col-major
        out_features                // ldc
    );

    if (status != CUBLAS_STATUS_SUCCESS) {
        throw std::runtime_error("cublasSgemm failed");
    }

    // Now fuse maxpool + sum + scale
    auto result = torch::empty({batch}, x.options());

    constexpr int BLOCK_SIZE = 256;

    if (kernel_size == 2 && out_features % 2 == 0) {
        fused_maxpool2_sum_scale_kernel<BLOCK_SIZE><<<batch, BLOCK_SIZE>>>(
            out_mat.data_ptr<float>(),
            result.data_ptr<float>(),
            batch, out_features, scale_factor
        );
    } else {
        fused_maxpool_sum_scale_kernel<BLOCK_SIZE><<<batch, BLOCK_SIZE>>>(
            out_mat.data_ptr<float>(),
            result.data_ptr<float>(),
            batch, out_features, kernel_size, scale_factor
        );
    }

    return result;
}
'''

cpp_source = r'''
torch::Tensor fused_forward(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    int kernel_size,
    float scale_factor
);
'''

ext = load_inline(
    name='fused_matmul_maxpool_sum_scale',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_forward'],
    extra_ldflags=['-lcublas'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
)


class Model(nn.Module):
    """
    Model that performs matrix multiplication, max pooling, sum, and scaling.
    Fused CUDA implementation for maximum performance.
    """
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        return ext.fused_forward(
            x,
            self.matmul.weight,
            self.matmul.bias,
            self.kernel_size,
            self.scale_factor
        )


batch_size = 128
in_features = 4096
out_features = 4096
kernel_size = 2
scale_factor = 0.5

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, kernel_size, scale_factor]
