"""
BF16 GEMV optimized for B200 (Blackwell SM100) using CUDA.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>

// Simple but efficient GEMV kernel with vectorized loads
__global__ void gemv_simple(
    const __nv_bfloat16* __restrict__ x,
    const __nv_bfloat16* __restrict__ weight,
    __nv_bfloat16* __restrict__ output,
    int batch,
    int in_features,
    int out_features
) {
    int out_idx = blockIdx.x;
    int batch_idx = blockIdx.y;
    
    if (out_idx >= out_features || batch_idx >= batch) return;
    
    const __nv_bfloat16* x_row = x + batch_idx * in_features;
    const __nv_bfloat16* w_row = weight + out_idx * in_features;
    
    float accum = 0.0f;
    
    // Vectorized load using float2 (2 bfloat16 per load)
    int vec_len = in_features / 2;
    const __nv_bfloat162* x_vec = reinterpret_cast<const __nv_bfloat162*>(x_row);
    const __nv_bfloat162* w_vec = reinterpret_cast<const __nv_bfloat162*>(w_row);
    
    for (int i = threadIdx.x; i < vec_len; i += blockDim.x) {
        __nv_bfloat162 x_pair = x_vec[i];
        __nv_bfloat162 w_pair = w_vec[i];
        
        float2 x_f = __bfloat1622float2(x_pair);
        float2 w_f = __bfloat1622float2(w_pair);
        
        accum += x_f.x * w_f.x + x_f.y * w_f.y;
    }
    
    // Handle odd element
    if (in_features % 2 == 1 && threadIdx.x == 0) {
        int last = in_features - 1;
        accum += __bfloat162float(x_row[last]) * __bfloat162float(w_row[last]);
    }
    
    // Block reduction
    extern __shared__ float sdata[];
    sdata[threadIdx.x] = accum;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            sdata[threadIdx.x] += sdata[threadIdx.x + stride];
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        output[batch_idx * out_features + out_idx] = __float2bfloat16(sdata[0]);
    }
}

torch::Tensor gemv_bf16(torch::Tensor x, torch::Tensor weight) {
    // x: (batch, in_features)
    // weight: (out_features, in_features)
    // output: (batch, out_features)
    
    int batch = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);
    
    auto options = torch::TensorOptions()
        .dtype(torch::kBFloat16)
        .device(x.device());
    auto output = torch::empty({batch, out_features}, options);
    
    // Configure kernel launch
    int threads_per_block = 256;
    dim3 block_dim(threads_per_block);
    dim3 grid_dim(out_features, batch);
    
    // Calculate shared memory size
    size_t shared_mem = threads_per_block * sizeof(float);
    
    // Launch kernel
    gemv_simple<<<grid_dim, block_dim, shared_mem>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(output.data_ptr()),
        batch, in_features, out_features
    );
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor gemv_bf16(torch::Tensor x, torch::Tensor weight);
'''

# Compile the extension
ext = load_inline(
    name='gemv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_bf16'],
    verbose=False,
    extra_cuda_cflags=[
        '-arch=sm_100',
        '-O3',
        '--use_fast_math',
    ],
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf16 = x.to(dtype=torch.bfloat16)
        return ext.gemv_bf16(x_bf16, self.weight)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16)]


def get_init_inputs():
    return []
