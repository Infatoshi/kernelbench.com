import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


# Fused kernel for eval mode: apply mish then BN with precomputed running stats
@triton.jit
def mish_bn_eval_kernel(
    x_ptr,           # input [N, C, H, W] contiguous
    out_ptr,         # output [N, C, H, W]
    running_mean_ptr,  # [C]
    running_var_ptr,   # [C]
    weight_ptr,      # BN gamma [C]
    bias_ptr,        # BN beta [C]
    eps,
    total_elements,
    C,
    HW,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < total_elements
    
    # NCHW layout: element at flat index i
    # c = (i // HW) % C
    c_idx = (offs // HW) % C
    
    val = tl.load(x_ptr + offs, mask=mask, other=0.0).to(tl.float32)
    
    # Mish: x * tanh(softplus(x))
    sp = tl.where(val > 20.0, val, tl.log(1.0 + tl.exp(val)))
    tanh_sp = 2.0 * tl.sigmoid(2.0 * sp) - 1.0
    mish_val = val * tanh_sp
    
    # BN with running stats
    mean = tl.load(running_mean_ptr + c_idx, mask=mask)
    var = tl.load(running_var_ptr + c_idx, mask=mask)
    gamma = tl.load(weight_ptr + c_idx, mask=mask)
    beta = tl.load(bias_ptr + c_idx, mask=mask)
    
    inv_std = 1.0 / tl.sqrt(var + eps)
    result = (mish_val - mean) * inv_std * gamma + beta
    
    tl.store(out_ptr + offs, result.to(val.dtype), mask=mask)


# Fused kernel 1: compute mish(x) per-channel sum and sum-of-squares for BN stats
@triton.jit
def mish_bn_stats_kernel(
    x_ptr,          # input [N, C, HW] contiguous
    sum_ptr,        # output: per-channel sum [C]
    sum_sq_ptr,     # output: per-channel sum of squares [C]
    N_HW,           # N * H * W
    C,
    HW,             # H * W
    BLOCK_SIZE: tl.constexpr,
):
    c = tl.program_id(0)
    
    acc_sum = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    acc_sum_sq = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    
    for start in range(0, N_HW, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < N_HW
        
        n_idx = offs // HW
        hw_idx = offs % HW
        global_idx = n_idx * (C * HW) + c * HW + hw_idx
        
        val = tl.load(x_ptr + global_idx, mask=mask, other=0.0).to(tl.float32)
        
        sp = tl.where(val > 20.0, val, tl.log(1.0 + tl.exp(val)))
        tanh_sp = 2.0 * tl.sigmoid(2.0 * sp) - 1.0
        mish_val = val * tanh_sp
        
        acc_sum += tl.where(mask, mish_val, 0.0)
        acc_sum_sq += tl.where(mask, mish_val * mish_val, 0.0)
    
    total_sum = tl.sum(acc_sum)
    total_sum_sq = tl.sum(acc_sum_sq)
    
    tl.store(sum_ptr + c, total_sum)
    tl.store(sum_sq_ptr + c, total_sum_sq)


# Fused kernel 2: apply mish then BN using precomputed batch stats
@triton.jit
def mish_bn_train_apply_kernel(
    x_ptr,
    out_ptr,
    sum_ptr,
    sum_sq_ptr,
    weight_ptr,
    bias_ptr,
    eps,
    inv_count,
    total_elements,
    C,
    HW,
    BLOCK_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    
    offs = pid * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < total_elements
    
    c_idx = (offs // HW) % C
    
    val = tl.load(x_ptr + offs, mask=mask, other=0.0).to(tl.float32)
    
    # Mish
    sp = tl.where(val > 20.0, val, tl.log(1.0 + tl.exp(val)))
    tanh_sp = 2.0 * tl.sigmoid(2.0 * sp) - 1.0
    mish_val = val * tanh_sp
    
    # BN with batch stats
    chan_sum = tl.load(sum_ptr + c_idx, mask=mask)
    chan_sum_sq = tl.load(sum_sq_ptr + c_idx, mask=mask)
    mean = chan_sum * inv_count
    var = chan_sum_sq * inv_count - mean * mean
    inv_std = 1.0 / tl.sqrt(var + eps)
    gamma = tl.load(weight_ptr + c_idx, mask=mask)
    beta = tl.load(bias_ptr + c_idx, mask=mask)
    
    result = (mish_val - mean) * inv_std * gamma + beta
    
    tl.store(out_ptr + offs, result.to(val.dtype), mask=mask)


def fused_mish_bn(x, bn_weight, bn_bias, bn_running_mean, bn_running_var, eps, momentum, training):
    N, C, H, W = x.shape
    HW = H * W
    N_HW = N * HW
    total_elements = x.numel()
    
    out = torch.empty_like(x)
    
    BLOCK_SIZE = 1024
    
    if not training:
        # Eval mode: use running stats
        num_blocks = triton.cdiv(total_elements, BLOCK_SIZE)
        mish_bn_eval_kernel[(num_blocks,)](
            x, out,
            bn_running_mean, bn_running_var,
            bn_weight, bn_bias,
            eps,
            total_elements, C, HW,
            BLOCK_SIZE=BLOCK_SIZE,
        )
    else:
        # Training mode: compute batch stats
        chan_sum = torch.empty(C, dtype=torch.float32, device=x.device)
        chan_sum_sq = torch.empty(C, dtype=torch.float32, device=x.device)
        
        mish_bn_stats_kernel[(C,)](
            x, chan_sum, chan_sum_sq,
            N_HW, C, HW,
            BLOCK_SIZE=4096,
        )
        
        num_blocks = triton.cdiv(total_elements, BLOCK_SIZE)
        mish_bn_train_apply_kernel[(num_blocks,)](
            x, out, chan_sum, chan_sum_sq,
            bn_weight, bn_bias,
            eps,
            1.0 / N_HW,
            total_elements, C, HW,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        # Update running stats
        with torch.no_grad():
            mean = chan_sum / N_HW
            var = chan_sum_sq / N_HW - mean * mean
            unbiased_var = var * N_HW / (N_HW - 1)
            bn_running_mean.mul_(1 - momentum).add_(momentum * mean)
            bn_running_var.mul_(1 - momentum).add_(momentum * unbiased_var)
    
    return out


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
    
    def forward(self, x):
        x = self.conv(x)
        x = x.contiguous()
        x = fused_mish_bn(
            x, self.bn.weight, self.bn.bias,
            self.bn.running_mean, self.bn.running_var,
            self.bn.eps, self.bn.momentum, self.training
        )
        return x


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size]
