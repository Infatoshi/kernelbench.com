import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ['fp4', 'fp8', 'fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']

M = 256
K = 131072


@triton.jit
def matvec_kernel(
    A_ptr, B_ptr, C_ptr,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    M, K,
    BLOCK_K: tl.constexpr,
):
    row = tl.program_id(0)
    offs_k = tl.arange(0, BLOCK_K)
    acc = 0.0

    for k_start in range(0, K, BLOCK_K):
        k = k_start + offs_k
        mask = k < K
        a = tl.load(A_ptr + row * stride_am + k * stride_ak, mask=mask, other=0.0)
        b = tl.load(B_ptr + k * stride_bk, mask=mask, other=0.0)
        acc += tl.sum(a * b, axis=0)

    tl.store(C_ptr + row * stride_cm, acc)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        assert A.is_cuda and B.is_cuda
        assert A.ndim == 2 and B.ndim == 2
        M_, K_ = A.shape
        assert B.shape[0] == K_ and B.shape[1] == 1

        A_c = A.contiguous()
        B_c = B.contiguous()
        C = torch.empty((M_, 1), device=A.device, dtype=torch.float32)

        matvec_kernel[(M_,)](
            A_c, B_c, C,
            A_c.stride(0), A_c.stride(1),
            B_c.stride(0), B_c.stride(1),
            C.stride(0), C.stride(1),
            M_, K_,
            BLOCK_K=1024,
            num_warps=4,
            num_stages=4,
        )
        return C.to(dtype=A.dtype)


def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]


def get_init_inputs():
    return []
