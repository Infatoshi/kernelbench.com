import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void fused_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N, int Cin, int H, int W,
    int Cout, int K,
    float sub1, float sub2,
    int Hc, int Wc,
    int Hp, int Wp
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * Cout * Hp * Wp;
    if (idx >= total) return;

    int pw = idx % Wp;
    int ph = (idx / Wp) % Hp;
    int co = (idx / (Wp * Hp)) % Cout;
    int n = idx / (Wp * Hp * Cout);

    float acc_pool = 0.0f;

    for (int pi = 0; pi < 2; ++pi) {
        for (int pj = 0; pj < 2; ++pj) {
            int h = ph * 2 + pi;
            int w_out = pw * 2 + pj;

            float val = 0.0f;

            for (int ci = 0; ci < Cin; ++ci) {
                for (int ky = 0; ky < K; ++ky) {
                    for (int kx = 0; kx < K; ++kx) {
                        int ih = h + ky;
                        int iw = w_out + kx;
                        float xv = x[((n * Cin + ci) * H + ih) * W + iw];
                        float wv = w[(((co * Cin + ci) * K + ky) * K) + kx];
                        val += xv * wv;
                    }
                }
            }
            val += b[co];
            val -= sub1;
            val = tanhf(val);
            val -= sub2;

            acc_pool += val;
        }
    }

    acc_pool *= 0.25f;

    out[((n * Cout + co) * Hp + ph) * Wp + pw] = acc_pool;
}

torch::Tensor fused_op(
    torch::Tensor x,
    torch::Tensor w,
    torch::Tensor b,
    float sub1,
    float sub2
) {
    int N = x.size(0);
    int Cin = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int Cout = w.size(0);
    int K = w.size(2);

    int Hc = H - K + 1;
    int Wc = W - K + 1;

    int Hp = Hc / 2;
    int Wp = Wc / 2;

    auto out = torch::empty({N, Cout, Hp, Wp}, x.options());

    int total = N * Cout * Hp * Wp;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    fused_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.data_ptr<float>(),
        out.data_ptr<float>(),
        N, Cin, H, W,
        Cout, K,
        sub1, sub2,
        Hc, Wc,
        Hp, Wp
    );

    return out;
}
'''

ext = load_inline(
    name='fused_conv_act_pool',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b, float sub1, float sub2);',
    cuda_sources=cuda_source,
    functions=['fused_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_channels, in_channels, kernel_size, kernel_size))
        self.bias = nn.Parameter(torch.randn(out_channels))
        self.sub1 = subtract1_value
        self.sub2 = subtract2_value

    def forward(self, x):
        return ext.fused_op(x.contiguous(), self.weight.contiguous(), self.bias.contiguous(), self.sub1, self.sub2)


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 0.5, 0.2, 2]
