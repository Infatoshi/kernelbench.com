import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def tall_skinny_gemm_kernel(
    A_ptr, B_ptr, C_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    pid = tl.program_id(0)
    grid_m = tl.cdiv(M, BLOCK_SIZE_M)
    grid_n = tl.cdiv(N, BLOCK_SIZE_N)
    pid_m = pid // grid_n
    pid_n = pid % grid_n

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    A_ptr = A_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    B_ptr = B_ptr + offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Correctly mask loading for inner loop K dimension
        k_mask = (offs_k[None, :] < K - k * BLOCK_SIZE_K)
        a = tl.load(A_ptr, mask=(offs_am[:, None] < M) & k_mask, other=0.0)
        b = tl.load(B_ptr, mask=(offs_k[:, None] < K - k * BLOCK_SIZE_K) & (offs_bn[None, :] < N), other=0.0)
        accumulator = tl.dot(a, b, accumulator)
        A_ptr += BLOCK_SIZE_K * stride_ak
        B_ptr += BLOCK_SIZE_K * stride_bk

    # Precision update
    if C_ptr.dtype.is_fp16() or C_ptr.dtype.is_bf16():
        accumulator = accumulator.to(C_ptr.dtype.element_ty)
    
    offs_cm = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M))
    offs_cn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N))
    C_ptr = C_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    tl.store(C_ptr, accumulator, mask=(offs_cm[:, None] < M) & (offs_cn[None, :] < N))

def tall_skinny_gemm(A, B):
    M, K = A.shape
    K2, N = B.shape
    assert K == K2
    C = torch.empty((M, N), device=A.device, dtype=A.dtype)
    
    # Simple tuning for H100
    BLOCK_SIZE_M = 128
    BLOCK_SIZE_N = 128
    BLOCK_SIZE_K = 32
    num_warps = 8
    num_stages = 3
    
    if N <= 128:
        BLOCK_SIZE_N = 128 if N > 64 else 64 if N > 32 else 32
    if M <= 128:
        BLOCK_SIZE_M = 128 if M > 64 else 64 if M > 32 else 32

    grid = (triton.cdiv(M, BLOCK_SIZE_M) * triton.cdiv(N, BLOCK_SIZE_N),)
    tall_skinny_gemm_kernel[grid](
        A, B, C,
        M, N, K,
        A.stride(0), A.stride(1),
        B.stride(0), B.stride(1),
        C.stride(0), C.stride(1),
        BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
        num_warps=num_warps,
        num_stages=num_stages,
    )
    return C

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, A, B):
        # The reference performs A*B. A is (M, N), B is (N, M) as per get_inputs().
        # M=16384, N=16. 
        # A is (16384, 16), B is (16, 16384) -> Out is (16384, 16384)
        return tall_skinny_gemm(A, B)

def get_inputs():
    M = 16384
    N = 16
    A = torch.randn(M, N, device='cuda')
    B = torch.randn(N, M, device='cuda')
    return [A, B]

def get_init_inputs():
    return []
