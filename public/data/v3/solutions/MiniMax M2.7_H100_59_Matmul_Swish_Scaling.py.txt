import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// FP16 kernel: simple version with 16x16 thread block
__global__ void matmul_swish_scale_fp16_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    half* __restrict__ C,
    half scale,
    int M, int N, int K) {
    
    const int row = blockIdx.y * blockDim.y + threadIdx.y;
    const int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    half sum = __float2half(0.0f);
    
    for (int k = 0; k < K; k++) {
        sum = __hfma(A[row * K + k], B[k * N + col], sum);
    }
    
    // Swish: x * sigmoid(x) then scale
    float fsum = __half2float(sum);
    float fsig = 1.0f / (1.0f + expf(-fsum));
    float fswish = fsum * fsig;
    float fout = fswish * __half2float(scale) * 2.0f;
    
    C[row * N + col] = __float2half(fout);
}

// FP32 kernel
__global__ void matmul_swish_scale_fp32_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    float scale,
    int M, int N, int K) {
    
    const int row = blockIdx.y * blockDim.y + threadIdx.y;
    const int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    for (int k = 0; k < K; k++) {
        sum += A[row * K + k] * B[k * N + col];
    }
    
    // Swish: x * sigmoid(x) then scale
    float sig = 1.0f / (1.0f + expf(-sum));
    float swish = sum * sig;
    float final = swish * scale * 2.0f;
    
    C[row * N + col] = final;
}

// BF16 helpers
__device__ __forceinline__ float bf16_to_float(unsigned short bf16) {
    float f;
    unsigned int bits = static_cast<unsigned int>(bf16) << 16;
    asm volatile("mov.f32 %0, %1;" : "=f"(f) : "r"(bits));
    return f;
}

__device__ __forceinline__ unsigned short float_to_bf16(float f) {
    unsigned int bits;
    asm volatile("mov.f32 %0, %1;" : "=r"(bits) : "f"(f));
    bits += 0x00008000;
    return static_cast<unsigned short>(bits >> 16);
}

__global__ void matmul_swish_scale_bf16_kernel(
    const unsigned short* __restrict__ A,
    const unsigned short* __restrict__ B,
    unsigned short* __restrict__ C,
    float scale,
    int M, int N, int K) {
    
    const int row = blockIdx.y * blockDim.y + threadIdx.y;
    const int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    
    for (int k = 0; k < K; k++) {
        float a = bf16_to_float(A[row * K + k]);
        float b = bf16_to_float(B[k * N + col]);
        sum += a * b;
    }
    
    // Swish
    float sig = 1.0f / (1.0f + expf(-sum));
    float swish = sum * sig;
    float final = swish * scale * 2.0f;
    
    C[row * N + col] = float_to_bf16(final);
}

// Host function for FP16
torch::Tensor matmul_swish_scale_fp16(
    torch::Tensor A,
    torch::Tensor B,
    torch::Tensor C,
    float scale) {
    
    const int M = A.size(0);
    const int K = A.size(1);
    const int N = B.size(1);
    
    const int BLOCK = 16;
    dim3 block(BLOCK, BLOCK);
    dim3 grid((N + BLOCK - 1) / BLOCK, (M + BLOCK - 1) / BLOCK);
    
    half scale_h = __float2half(scale);
    
    matmul_swish_scale_fp16_kernel<<<grid, block>>>(
        static_cast<half*>(A.data_ptr()),
        static_cast<half*>(B.data_ptr()),
        static_cast<half*>(C.data_ptr()),
        scale_h,
        M, N, K
    );
    
    return C;
}

// Host function for FP32
torch::Tensor matmul_swish_scale_fp32(
    torch::Tensor A,
    torch::Tensor B,
    torch::Tensor C,
    float scale) {
    
    const int M = A.size(0);
    const int K = A.size(1);
    const int N = B.size(1);
    
    const int BLOCK = 16;
    dim3 block(BLOCK, BLOCK);
    dim3 grid((N + BLOCK - 1) / BLOCK, (M + BLOCK - 1) / BLOCK);
    
    matmul_swish_scale_fp32_kernel<<<grid, block>>>(
        static_cast<float*>(A.data_ptr()),
        static_cast<float*>(B.data_ptr()),
        static_cast<float*>(C.data_ptr()),
        scale,
        M, N, K
    );
    
    return C;
}

// Host function for BF16
torch::Tensor matmul_swish_scale_bf16(
    torch::Tensor A,
    torch::Tensor B,
    torch::Tensor C,
    float scale) {
    
    const int M = A.size(0);
    const int K = A.size(1);
    const int N = B.size(1);
    
    const int BLOCK = 16;
    dim3 block(BLOCK, BLOCK);
    dim3 grid((N + BLOCK - 1) / BLOCK, (M + BLOCK - 1) / BLOCK);
    
    matmul_swish_scale_bf16_kernel<<<grid, block>>>(
        static_cast<unsigned short*>(A.data_ptr()),
        static_cast<unsigned short*>(B.data_ptr()),
        static_cast<unsigned short*>(C.data_ptr()),
        scale,
        M, N, K
    );
    
    return C;
}
'''

ext = load_inline(
    name='matmul_swish_ext',
    cpp_sources='''
torch::Tensor matmul_swish_scale_fp16(torch::Tensor A, torch::Tensor B, torch::Tensor C, float scale);
torch::Tensor matmul_swish_scale_fp32(torch::Tensor A, torch::Tensor B, torch::Tensor C, float scale);
torch::Tensor matmul_swish_scale_bf16(torch::Tensor A, torch::Tensor B, torch::Tensor C, float scale);
''',
    cuda_sources=cuda_source,
    functions=['matmul_swish_scale_fp16', 'matmul_swish_scale_fp32', 'matmul_swish_scale_bf16'],
    extra_include_paths=['/usr/local/cuda/include'],
    verbose=False
)

class Model(nn.Module):
    """
    Optimized fused model: Matmul + Swish + Scale
    Uses custom CUDA kernels for better performance on H100.
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.randn(out_features))
        self.scaling_factor = scaling_factor
        self.in_features = in_features
        self.out_features = out_features
        
        nn.init.kaiming_normal_(self.weight, mode='fan_out')
        nn.init.zeros_(self.bias)
    
    def forward(self, x):
        dtype = x.dtype
        
        if dtype == torch.float16:
            weight_fp16 = self.weight.half()
            bias_fp16 = self.bias.half()
            
            output = torch.empty(x.shape[0], self.out_features, dtype=torch.float16, device=x.device)
            weight_t = weight_fp16.t()
            
            ext.matmul_swish_scale_fp16(
                x.half(), weight_t, output, self.scaling_factor
            )
            
            output = output + bias_fp16.unsqueeze(0)
            return output
            
        elif dtype == torch.bfloat16:
            weight_bf16 = self.weight.bfloat16()
            bias_bf16 = self.bias.bfloat16()
            
            output = torch.empty(x.shape[0], self.out_features, dtype=torch.bfloat16, device=x.device)
            weight_t = weight_bf16.t()
            
            ext.matmul_swish_scale_bf16(
                x.bfloat16(), weight_t, output, self.scaling_factor
            )
            
            output = output + bias_bf16.unsqueeze(0)
            return output
            
        else:  # float32
            weight_fp32 = self.weight.float()
            bias_fp32 = self.bias.float()
            
            output = torch.empty(x.shape[0], self.out_features, dtype=torch.float32, device=x.device)
            weight_t = weight_fp32.t()
            
            ext.matmul_swish_scale_fp32(
                x.float(), weight_t, output, self.scaling_factor
            )
            
            output = output + bias_fp32.unsqueeze(0)
            return output

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 2.0]
