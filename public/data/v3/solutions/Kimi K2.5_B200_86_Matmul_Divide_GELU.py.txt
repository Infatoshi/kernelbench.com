import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Module constants - MUST be defined before use
batch_size = 128
input_size = 4096
output_size = 4096
divisor = 10.0

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

constexpr float SQRT_2_OVER_PI = 0.7978845608028654f;
constexpr float COEFF = 0.044715f;

__device__ __forceinline__ float gelu_tanh(float x) {
    float x3 = x * x * x;
    return 0.5f * x * (1.0f + tanhf(SQRT_2_OVER_PI * (x + COEFF * x3)));
}

template <typename T>
__global__ void matmul_div_gelu_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    int M, int N, int K,
    float divisor
) {
    const int row = blockIdx.y * blockDim.y + threadIdx.y;
    const int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    float acc = 0.0f;
    for (int k = 0; k < K; ++k) {
        acc += float(input[row * K + k]) * float(weight[col * K + k]);
    }
    
    acc += float(bias[col]);
    acc /= divisor;
    acc = gelu_tanh(acc);
    
    output[row * N + col] = T(acc);
}

torch::Tensor matmul_div_gelu_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float divisor
) {
    auto M = input.size(0);
    auto K = input.size(1);
    auto N = weight.size(0);
    
    auto output = torch::empty({M, N}, input.options());
    
    const dim3 block_size(16, 16);
    const dim3 grid_size(
        (N + block_size.x - 1) / block_size.x,
        (M + block_size.y - 1) / block_size.y
    );
    
    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half,
        at::ScalarType::BFloat16,
        input.scalar_type(),
        "matmul_div_gelu_cuda",
        [&]() {
            matmul_div_gelu_kernel<scalar_t><<<grid_size, block_size>>>(
                input.data_ptr<scalar_t>(),
                weight.data_ptr<scalar_t>(),
                bias.data_ptr<scalar_t>(),
                output.data_ptr<scalar_t>(),
                M, N, K,
                divisor
            );
        }
    );
    
    return output;
}
'''

cpp_source = '''
torch::Tensor matmul_div_gelu_cuda(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float divisor);
'''

# Load the extension
_ext = load_inline(
    name='matmul_div_gelu',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['matmul_div_gelu_cuda'],
    extra_cuda_cflags=['-O3'],
    verbose=False,
)

def get_inputs():
    return [torch.randn(batch_size, input_size)]

def get_init_inputs():
    return [input_size, output_size, divisor]

class Model(nn.Module):
    def __init__(self, input_size, output_size, divisor):
        super(Model, self).__init__()
        self.linear = nn.Linear(input_size, output_size)
        self.divisor = divisor

    def forward(self, x):
        return _ext.matmul_div_gelu_cuda(
            x, 
            self.linear.weight, 
            self.linear.bias, 
            self.divisor
        )
