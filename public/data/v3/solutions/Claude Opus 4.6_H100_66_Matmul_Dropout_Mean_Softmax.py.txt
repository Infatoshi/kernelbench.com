import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_softmax_kernel(
    X_ptr, Out_ptr,
    N,
    BLOCK_SIZE: tl.constexpr,
):
    # Single program for single row - online softmax
    # First pass: find max
    m = float('-inf')
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(X_ptr + offsets, mask=mask, other=float('-inf'))
        m = tl.maximum(m, tl.max(x, axis=0))
    
    # Second pass: compute sum of exp(x - max)
    s = tl.zeros([1], dtype=tl.float32)
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(X_ptr + offsets, mask=mask, other=float('-inf'))
        s += tl.sum(tl.exp(x - m))
    
    # Third pass: write softmax output
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(X_ptr + offsets, mask=mask, other=float('-inf'))
        result = tl.exp(x - m) / s
        tl.store(Out_ptr + offsets, result, mask=mask)


@triton.jit
def matvec_kernel(
    X_ptr, W_ptr, Bias_ptr, Out_ptr,
    K, N,
    stride_w_n, stride_w_k,
    BLOCK_K: tl.constexpr, BLOCK_N: tl.constexpr,
):
    """Compute out[n] = sum_k(x[k] * W[n, k]) + bias[n] for a block of n values."""
    pid = tl.program_id(0)
    n_offsets = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = n_offsets < N
    
    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    
    for k_start in range(0, K, BLOCK_K):
        k_offsets = k_start + tl.arange(0, BLOCK_K)
        k_mask = k_offsets < K
        
        x_vals = tl.load(X_ptr + k_offsets, mask=k_mask, other=0.0).to(tl.float32)
        
        # W[n, k]
        w_ptrs = W_ptr + n_offsets[:, None] * stride_w_n + k_offsets[None, :]
        w_mask = n_mask[:, None] & k_mask[None, :]
        w_vals = tl.load(w_ptrs, mask=w_mask, other=0.0).to(tl.float32)
        
        acc += tl.sum(w_vals * x_vals[None, :], axis=1)
    
    bias = tl.load(Bias_ptr + n_offsets, mask=n_mask, other=0.0).to(tl.float32)
    acc += bias
    
    tl.store(Out_ptr + n_offsets, acc, mask=n_mask)


@triton.jit
def mean_reduce_kernel(
    X_ptr, Out_ptr,
    batch_size, features,
    stride_batch, stride_feat,
    BLOCK_F: tl.constexpr,
):
    """Compute mean across batch dimension for a block of features."""
    pid = tl.program_id(0)
    f_offsets = pid * BLOCK_F + tl.arange(0, BLOCK_F)
    f_mask = f_offsets < features
    
    acc = tl.zeros([BLOCK_F], dtype=tl.float32)
    for b in range(batch_size):
        vals = tl.load(X_ptr + b * stride_batch + f_offsets * stride_feat, mask=f_mask, other=0.0)
        acc += vals.to(tl.float32)
    
    acc = acc / batch_size
    tl.store(Out_ptr + f_offsets, acc, mask=f_mask)


class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout = nn.Dropout(dropout_p)
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        # In eval mode, dropout is no-op
        # mean(X @ W^T + b, dim=0) = mean(X, dim=0) @ W^T + b
        
        # Step 1: Mean across batch dimension -> (in_features,)
        x_mean = x.mean(dim=0)  # Fast PyTorch reduction
        
        # Step 2: Matrix-vector product + bias using Triton
        weight = self.matmul.weight  # (out_features, in_features)
        bias = self.matmul.bias  # (out_features,)
        
        out = torch.empty(self.out_features, device=x.device, dtype=torch.float32)
        
        BLOCK_K = 256
        BLOCK_N = 32
        grid = ((self.out_features + BLOCK_N - 1) // BLOCK_N,)
        
        matvec_kernel[grid](
            x_mean, weight, bias, out,
            self.in_features, self.out_features,
            weight.stride(0), weight.stride(1),
            BLOCK_K=BLOCK_K, BLOCK_N=BLOCK_N,
        )
        
        # Step 3: Softmax on single row
        result = torch.empty(1, self.out_features, device=x.device, dtype=torch.float32)
        fused_softmax_kernel[(1,)](
            out, result,
            self.out_features,
            BLOCK_SIZE=4096,
        )
        
        return result


batch_size = 128
in_features = 4096
out_features = 4096
dropout_p = 0.2

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, dropout_p]
