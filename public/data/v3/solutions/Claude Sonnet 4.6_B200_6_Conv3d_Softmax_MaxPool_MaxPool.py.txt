import torch
import torch.nn as nn
import triton
import triton.language as tl


# Triton kernel: fused softmax (over channel dim) + 2x MaxPool3d(2)
# Input:  [B, C, D, H, W]  (after conv3d)
# Output: [B, C, OD, OH, OW]  where OD=D//4, OH=H//4, OW=W//4
# For each output (b, od, oh, ow):
#   Pool window spans input [4*od:4*od+4, 4*oh:4*oh+4, 4*ow:4*ow+4]
#   For each spatial in window, compute softmax over C channels
#   Output[c] = max over all 64 spatial positions of softmax[c]
@triton.jit
def fused_softmax_pool_kernel(
    inp_ptr,        # [B, C, D, H, W]
    out_ptr,        # [B, C, OD, OH, OW]
    B, C, D, H, W,
    OD, OH, OW,
    stride_b, stride_c, stride_d, stride_h, stride_w,
    out_stride_b, out_stride_c, out_stride_od, out_stride_oh, out_stride_ow,
    POOL: tl.constexpr,   # pool factor (4 for 2x pool of size 2)
    C_BLOCK: tl.constexpr,  # must be >= C, power of 2
):
    # Each program handles one output spatial position (b, od, oh, ow)
    pid = tl.program_id(0)
    total_spatial = OD * OH * OW
    b = pid // total_spatial
    spatial = pid % total_spatial
    od = spatial // (OH * OW)
    oh = (spatial % (OH * OW)) // OW
    ow = spatial % OW

    # Channel indices
    c_idx = tl.arange(0, C_BLOCK)
    c_mask = c_idx < C

    # Running max of softmax values per channel
    running_max = tl.full([C_BLOCK], float('-inf'), dtype=tl.float32)

    # Iterate over the 4x4x4 (POOL^3) pool window
    for pd in range(POOL):
        d_in = od * POOL + pd
        if d_in < D:
            for ph in range(POOL):
                h_in = oh * POOL + ph
                if h_in < H:
                    for pw in range(POOL):
                        w_in = ow * POOL + pw
                        if w_in < W:
                            # Load C values for this spatial position
                            base = b * stride_b + d_in * stride_d + h_in * stride_h + w_in * stride_w
                            x = tl.load(inp_ptr + base + c_idx * stride_c, mask=c_mask, other=float('-inf'))
                            x = x.to(tl.float32)

                            # Softmax over C channels for this spatial position
                            x_max = tl.max(x, axis=0)
                            x = x - x_max
                            x_exp = tl.exp(x)
                            x_sum = tl.sum(x_exp, axis=0)
                            softmax_vals = x_exp / x_sum

                            # Update running max per channel
                            running_max = tl.maximum(running_max, softmax_vals)

    # Write output
    out_base = b * out_stride_b + od * out_stride_od + oh * out_stride_oh + ow * out_stride_ow
    tl.store(out_ptr + out_base + c_idx * out_stride_c, running_max.to(tl.float16), mask=c_mask)


@triton.jit
def fused_softmax_pool_fp32_kernel(
    inp_ptr,        # [B, C, D, H, W]
    out_ptr,        # [B, C, OD, OH, OW]
    B, C, D, H, W,
    OD, OH, OW,
    stride_b, stride_c, stride_d, stride_h, stride_w,
    out_stride_b, out_stride_c, out_stride_od, out_stride_oh, out_stride_ow,
    POOL: tl.constexpr,
    C_BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    total_spatial = OD * OH * OW
    b = pid // total_spatial
    spatial = pid % total_spatial
    od = spatial // (OH * OW)
    oh = (spatial % (OH * OW)) // OW
    ow = spatial % OW

    c_idx = tl.arange(0, C_BLOCK)
    c_mask = c_idx < C

    running_max = tl.full([C_BLOCK], float('-inf'), dtype=tl.float32)

    for pd in range(POOL):
        d_in = od * POOL + pd
        if d_in < D:
            for ph in range(POOL):
                h_in = oh * POOL + ph
                if h_in < H:
                    for pw in range(POOL):
                        w_in = ow * POOL + pw
                        if w_in < W:
                            base = b * stride_b + d_in * stride_d + h_in * stride_h + w_in * stride_w
                            x = tl.load(inp_ptr + base + c_idx * stride_c, mask=c_mask, other=float('-inf'))
                            x = x.to(tl.float32)
                            x_max = tl.max(x, axis=0)
                            x = x - x_max
                            x_exp = tl.exp(x)
                            x_sum = tl.sum(x_exp, axis=0)
                            softmax_vals = x_exp / x_sum
                            running_max = tl.maximum(running_max, softmax_vals)

    out_base = b * out_stride_b + od * out_stride_od + oh * out_stride_oh + ow * out_stride_ow
    tl.store(out_ptr + out_base + c_idx * out_stride_c, running_max, mask=c_mask)


def fused_softmax_pool(x, pool=4):
    """
    x: [B, C, D, H, W] float32 or float16
    Computes softmax(x, dim=1) then 2x MaxPool3d(2) (= effective pool of 4 with stride 4)
    Returns [B, C, OD, OH, OW]
    """
    B, C, D, H, W = x.shape
    OD = D // pool
    OH = H // pool
    OW = W // pool

    # Use float16 output for fp16/bf16 input, float32 for fp32 input
    if x.dtype == torch.float32:
        out = torch.empty(B, C, OD, OH, OW, dtype=torch.float32, device=x.device)
    else:
        out = torch.empty(B, C, OD, OH, OW, dtype=torch.float16, device=x.device)

    # Ensure contiguous
    if not x.is_contiguous():
        x = x.contiguous()

    # Convert to float16 for faster processing if bf16
    if x.dtype == torch.bfloat16:
        x = x.to(torch.float16)

    stride_b, stride_c, stride_d, stride_h, stride_w = x.stride()
    out_stride_b, out_stride_c, out_stride_od, out_stride_oh, out_stride_ow = out.stride()

    C_BLOCK = triton.next_power_of_2(C)
    total = B * OD * OH * OW

    if x.dtype == torch.float32:
        fused_softmax_pool_fp32_kernel[(total,)](
            x, out,
            B, C, D, H, W,
            OD, OH, OW,
            stride_b, stride_c, stride_d, stride_h, stride_w,
            out_stride_b, out_stride_c, out_stride_od, out_stride_oh, out_stride_ow,
            POOL=pool,
            C_BLOCK=C_BLOCK,
            num_warps=4,
        )
    else:
        fused_softmax_pool_kernel[(total,)](
            x, out,
            B, C, D, H, W,
            OD, OH, OW,
            stride_b, stride_c, stride_d, stride_h, stride_w,
            out_stride_b, out_stride_c, out_stride_od, out_stride_oh, out_stride_ow,
            POOL=pool,
            C_BLOCK=C_BLOCK,
            num_warps=4,
        )

    return out


class Model(nn.Module):
    """
    Model that performs a 3D convolution, applies Softmax, and performs two max pooling operations.
    Fused: softmax + 2x MaxPool3d(2) in a single Triton kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size
        # pool_kernel_size=2 applied twice => effective pool of 4
        self.effective_pool = pool_kernel_size * pool_kernel_size  # = 4

    def forward(self, x):
        x = self.conv(x)
        # Fused: softmax over channels + 2x maxpool
        x = fused_softmax_pool(x, pool=self.effective_pool)
        return x

batch_size = 16

depth, height, width = 16, 128, 128

def get_inputs():
    return [torch.randn(batch_size, in_channels, depth, height, width)]

    return [in_channels, out_channels, kernel_size, pool_kernel_size]
