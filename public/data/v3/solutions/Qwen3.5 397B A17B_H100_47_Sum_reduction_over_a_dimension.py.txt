import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp-level reduction using shuffle instructions
template<typename T>
__device__ inline T warp_reduce_sum(T val) {
    for (int offset = 16; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Optimized sum reduction kernel for H100
// Reduces over dimension 1 of a (B, D1, D2) tensor
// Output shape: (B, 1, D2)
template<typename T>
__global__ void sum_reduce_dim1_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    int B,  // batch size (dim 0)
    int D1, // reduction dimension (dim 1)
    int D2  // output dimension (dim 2)
) {
    // Each block handles one (b, d2) pair
    // Threads within block reduce over d1 dimension
    
    int b = blockIdx.x;
    int d2 = blockIdx.y;
    
    if (b >= B || d2 >= D2) return;
    
    // Shared memory for block-level reduction
    extern __shared__ float shared_mem[];
    
    float local_sum = 0.0f;
    
    // Each thread processes multiple elements along D1
    int tid = threadIdx.x;
    int threads_per_block = blockDim.x;
    
    // Strided access over D1 dimension
    for (int d1 = tid; d1 < D1; d1 += threads_per_block) {
        int input_idx = b * D1 * D2 + d1 * D2 + d2;
        local_sum += static_cast<float>(input[input_idx]);
    }
    
    // Store partial sum in shared memory
    shared_mem[tid] = local_sum;
    __syncthreads();
    
    // Reduce within shared memory (tree reduction)
    for (int stride = threads_per_block / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            shared_mem[tid] += shared_mem[tid + stride];
        }
        __syncthreads();
    }
    
    // Write result
    if (tid == 0) {
        int output_idx = b * D2 + d2;  // Output shape: (B, 1, D2) -> flattened as (B, D2)
        output[output_idx] = static_cast<T>(shared_mem[0]);
    }
}

// Generic launcher
torch::Tensor sum_reduce_dim1(torch::Tensor x, int dim) {
    auto input_sizes = x.sizes();
    int ndim = x.dim();
    
    // Normalize negative dim
    if (dim < 0) dim += ndim;
    
    // Get dimensions
    int B = 1, D1 = 1, D2 = 1;
    
    // Product of dimensions before reduction dim
    for (int i = 0; i < dim; i++) B *= input_sizes[i];
    D1 = input_sizes[dim];
    // Product of dimensions after reduction dim
    for (int i = dim + 1; i < ndim; i++) D2 *= input_sizes[i];
    
    // Output shape: same as input but with reduction dim = 1
    std::vector<int64_t> output_sizes;
    for (int i = 0; i < ndim; i++) {
        if (i == dim) output_sizes.push_back(1);
        else output_sizes.push_back(input_sizes[i]);
    }
    
    auto output = torch::empty(output_sizes, x.options());
    
    int threads_per_block = 256;
    dim3 block_dim(threads_per_block);
    dim3 grid_dim(B, D2);
    
    size_t shared_mem_size = threads_per_block * sizeof(float);
    
    if (x.dtype() == torch::kFloat32) {
        sum_reduce_dim1_kernel<float><<<grid_dim, block_dim, shared_mem_size>>>(
            x.data_ptr<float>(),
            output.data_ptr<float>(),
            B, D1, D2
        );
    } else if (x.dtype() == torch::kFloat16) {
        sum_reduce_dim1_kernel<at::Half><<<grid_dim, block_dim, shared_mem_size>>>(
            x.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            B, D1, D2
        );
    } else if (x.dtype() == torch::kBFloat16) {
        sum_reduce_dim1_kernel<at::BFloat16><<<grid_dim, block_dim, shared_mem_size>>>(
            x.data_ptr<at::BFloat16>(),
            output.data_ptr<at::BFloat16>(),
            B, D1, D2
        );
    }
    
    cudaDeviceSynchronize();
    
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor sum_reduce_dim1(torch::Tensor x, int dim);
'''

ext = load_inline(
    name='sum_reduce_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['sum_reduce_dim1'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduce_dim1(x, self.dim)

def get_inputs():
    batch_size = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [1]  # reduce_dim = 1
