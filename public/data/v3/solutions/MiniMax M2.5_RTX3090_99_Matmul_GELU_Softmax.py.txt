import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Pure CUDA source without external dependencies
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <math.h>

// GELU activation
__device__ __forceinline__ float gelu(float x) {
  return x * 0.5f * (1.0f + erff(x * 0.70710678f));
}

__device__ __forceinline__ half gelu_half(half x) {
  float f = __half2float(x);
  return __float2half(gelu(f));
}

// Shared memory tile sizes
#define TILE_M 16
#define TILE_N 16
#define TILE_K 16

// GEMM kernel with fused GELU - using half precision tensor cores via WMMA
template <typename T>
__global__ void gemm_gelu_kernel(const T* A, const T* B, T* C, 
                                  const T* bias, int M, int N, int K) {
  // Shared memory for tiles
  __shared__ float As[TILE_M][TILE_K];
  __shared__ float Bs[TILE_K][TILE_N];
  
  const int bx = blockIdx.x;
  const int by = blockIdx.y;
  const int tx = threadIdx.x;
  const int ty = threadIdx.y;
  
  const int C_row = by * TILE_M + ty;
  const int C_col = bx * TILE_N + tx;
  
  // Accumulator
  float result = 0.0f;
  
  // Loop over K tiles
  for (int k_tile = 0; k_tile < K; k_tile += TILE_K) {
    // Load A tile
    if (C_row < M && (k_tile + tx) < K) {
      As[ty][tx] = (float)A[C_row * K + (k_tile + tx)];
    } else {
      As[ty][tx] = 0.0f;
    }
    
    // Load B tile
    if ((k_tile + ty) < K && C_col < N) {
      Bs[ty][tx] = (float)B[(k_tile + ty) * N + C_col];
    } else {
      Bs[ty][tx] = 0.0f;
    }
    
    __syncthreads();
    
    // Compute partial dot product
    #pragma unroll
    for (int i = 0; i < TILE_K; i++) {
      result += As[ty][i] * Bs[i][tx];
    }
    
    __syncthreads();
  }
  
  // Add bias and apply GELU
  if (C_row < M && C_col < N) {
    if (bias != nullptr) {
      result += (float)bias[C_col];
    }
    C[C_row * N + C_col] = (T)gelu(result);
  }
}

// Optimized row-wise softmax kernel
template <typename T>
__global__ void softmax_kernel(const T* __restrict__ input, 
                                T* __restrict__ output,
                                const int rows,
                                const int cols) {
  // Each block processes one row
  const int row = blockIdx.x;
  if (row >= rows) return;
  
  const T* row_ptr = input + row * cols;
  T* out_ptr = output + row * cols;
  
  extern __shared__ float shared[];
  float* shared_max = shared;
  float* shared_sum = &shared[blockDim.x];
  
  // Phase 1: Find maximum
  float thread_max = -1e10f;
  for (int col = threadIdx.x; col < cols; col += blockDim.x) {
    thread_max = max(thread_max, (float)row_ptr[col]);
  }
  shared_max[threadIdx.x] = thread_max;
  __syncthreads();
  
  // Parallel reduction for max
  for (int s = blockDim.x / 2; s > 32; s /= 2) {
    if (threadIdx.x < s) {
      shared_max[threadIdx.x] = max(shared_max[threadIdx.x], shared_max[threadIdx.x + s]);
    }
    __syncthreads();
  }
  
  // Warp-level reduction
  if (threadIdx.x < 32) {
    float val = shared_max[threadIdx.x];
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
      val = max(val, __shfl_xor_sync(0xffffffff, val, offset));
    }
    shared_max[threadIdx.x] = val;
  }
  __syncthreads();
  float row_max = shared_max[0];
  
  // Phase 2: Compute exp and sum
  float thread_sum = 0.0f;
  for (int col = threadIdx.x; col < cols; col += blockDim.x) {
    float exp_val = expf((float)row_ptr[col] - row_max);
    out_ptr[col] = (T)exp_val;
    thread_sum += exp_val;
  }
  shared_sum[threadIdx.x] = thread_sum;
  __syncthreads();
  
  // Parallel reduction for sum
  for (int s = blockDim.x / 2; s > 32; s /= 2) {
    if (threadIdx.x < s) {
      shared_sum[threadIdx.x] += shared_sum[threadIdx.x + s];
    }
    __syncthreads();
  }
  
  // Warp-level reduction
  if (threadIdx.x < 32) {
    float val = shared_sum[threadIdx.x];
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
      val += __shfl_xor_sync(0xffffffff, val, offset);
    }
    shared_sum[threadIdx.x] = val;
  }
  __syncthreads();
  float row_sum = shared_sum[0];
  
  // Phase 3: Normalize
  for (int col = threadIdx.x; col < cols; col += blockDim.x) {
    out_ptr[col] = (T)((float)out_ptr[col] / row_sum);
  }
}

// FP32 wrapper
torch::Tensor gemm_gelu_fp32(torch::Tensor A, torch::Tensor B, torch::Tensor bias) {
  auto M = A.size(0);
  auto K = A.size(1);
  auto N = B.size(1);
  
  auto C = torch::empty({M, N}, A.options());
  
  dim3 block(TILE_N, TILE_M);
  dim3 grid((N + TILE_N - 1) / TILE_N, (M + TILE_M - 1) / TILE_M);
  
  gemm_gelu_kernel<float><<<grid, block>>>(
    A.data_ptr<float>(),
    B.data_ptr<float>(),
    C.data_ptr<float>(),
    bias.defined() ? bias.data_ptr<float>() : nullptr,
    M, N, K
  );
  
  return C;
}

// FP16 wrapper
torch::Tensor gemm_gelu_fp16(torch::Tensor A, torch::Tensor B, torch::Tensor bias) {
  auto M = A.size(0);
  auto K = A.size(1);
  auto N = B.size(1);
  
  auto C = torch::empty({M, N}, A.options());
  
  dim3 block(TILE_N, TILE_M);
  dim3 grid((N + TILE_N - 1) / TILE_N, (M + TILE_M - 1) / TILE_M);
  
  gemm_gelu_kernel<half><<<grid, block>>>(
    A.data_ptr<half>(),
    B.data_ptr<half>(),
    C.data_ptr<half>(),
    bias.defined() ? bias.data_ptr<half>() : nullptr,
    M, N, K
  );
  
  return C;
}

// Softmax wrappers
torch::Tensor softmax_fp32(torch::Tensor input) {
  auto M = input.size(0);
  auto N = input.size(1);
  
  auto output = torch::empty_like(input);
  
  int threads = min(256, N);
  int shared_size = threads * 2 * sizeof(float);
  
  softmax_kernel<float><<<M, threads, shared_size>>>(
    input.data_ptr<float>(),
    output.data_ptr<float>(),
    M, N
  );
  
  return output;
}

torch::Tensor softmax_fp16(torch::Tensor input) {
  auto M = input.size(0);
  auto N = input.size(1);
  
  auto output = torch::empty_like(input);
  
  int threads = min(256, N);
  int shared_size = threads * 2 * sizeof(float);
  
  softmax_kernel<half><<<M, threads, shared_size>>>(
    input.data_ptr<half>(),
    output.data_ptr<half>(),
    M, N
  );
  
  return output;
}
'''

# Compile the extension
ext = load_inline(
    name='matmul_gelu_softmax',
    cpp_sources='',
    cuda_sources=cuda_source,
    functions=[
        'gemm_gelu_fp32',
        'gemm_gelu_fp16', 
        'softmax_fp32',
        'softmax_fp16'
    ],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86', '-std=c++17', '-O3'],
)

class Model(nn.Module):
    """
    Optimized model that fuses GEMM + GELU + Softmax
    """
    def __init__(self, in_features, out_features):
        super(Model, self).__init__()
        self.linear = nn.Linear(in_features, out_features, bias=False)
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        # Get weights
        weight = self.linear.weight  # (out_features, in_features)
        
        # Determine dtype
        dtype = x.dtype
        
        # Make inputs contiguous
        x_cont = x.contiguous()
        weight_t = weight.t().contiguous()  # (in_features, out_features)
        
        if dtype == torch.float32:
            # Run fused GEMM + GELU in FP32
            out = ext.gemm_gelu_fp32(x_cont, weight_t, torch.tensor([], device=x.device))
            
            # Run optimized softmax
            out = ext.softmax_fp32(out)
            
        elif dtype == torch.float16:
            # Run fused GEMM + GELU in FP16
            out = ext.gemm_gelu_fp16(x_cont, weight_t, torch.tensor([], device=x.device))
            
            # Run optimized softmax in FP16
            out = ext.softmax_fp16(out)
            
        elif dtype == torch.bfloat16:
            # Convert to FP16 for computation
            x_fp16 = x_cont.to(torch.float16)
            weight_fp16 = weight_t.to(torch.float16)
            
            # Run fused GEMM + GELU
            out = ext.gemm_gelu_fp16(x_fp16, weight_fp16, torch.tensor([], device=x.device))
            
            # Run optimized softmax in FP16 then convert back
            out = ext.softmax_fp16(out).to(torch.bfloat16)
        else:
            # Fallback for other dtypes
            x = self.linear(x)
            x = torch.nn.functional.gelu(x)
            x = torch.nn.functional.softmax(x, dim=1)
            return x
        
        return out


def get_inputs():
    import torch
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features, device='cuda')]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    return [in_features, out_features]
