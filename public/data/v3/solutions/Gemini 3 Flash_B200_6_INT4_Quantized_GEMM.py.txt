
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def int4_gemm_kernel(
    a_ptr, b_ptr, scales_ptr, c_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    stride_sn, stride_sk,
    stride_cm, stride_cn,
    group_size,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptrs = a_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    # b_ptr points to (N, K//2) uint8. 
    # Logic: we load BLOCK_SIZE_K elements of K. Since it's packed, we load K//2.
    # We'll load BLOCK_SIZE_K // 2 bytes.
    b_ptrs = b_ptr + (offs_bn[:, None] * stride_bn + (offs_k[None, :] // 2) * stride_bk)
    
    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Load A
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0.0)
        
        # Load B (packed uint8)
        # Note: B is (N, K//2). stride_bk is 1. stride_bn is K//2.
        b_packed = tl.load(b_ptrs, mask=offs_k[None, :] < K - k * BLOCK_SIZE_K, other=0)
        
        # Unpack INT4
        # shifter = 0 if offs_k is even, 4 if odd
        shifter = (offs_k % 2) * 4
        b_int = (b_packed >> shifter) & 0xF
        b_fp16 = (b_int.to(tl.float16) - 8.0)
        
        # Load Scales: scales are (N, K // group_size)
        # current_k = k * BLOCK_SIZE_K + offs_k
        # group_idx = current_k // group_size
        k_base = k * BLOCK_SIZE_K
        offs_sn_group = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
        # Scale is same for all elements in a group.
        # Simple approach: load scale for each K in BLOCK_SIZE_K
        # Since BLOCK_SIZE_K (32) < group_size (128), usually only 1 or 2 scales per row.
        group_idx = (k_base + offs_k) // group_size
        s_ptrs = scales_ptr + (offs_bn[:, None] * stride_sn + group_idx[None, :] * stride_sk)
        scales = tl.load(s_ptrs)
        
        b_dequant = b_fp16 * scales
        
        accumulator += tl.dot(a, b_dequant.trans(), out_dtype=tl.float32)
        
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += (BLOCK_SIZE_K // 2) * stride_bk

    c = accumulator.to(tl.float16)
    
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, c, mask=c_mask)

class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, K = x.shape
        M = batch_size * seq_len
        N = self.N
        
        x_2d = x.reshape(M, K)
        out = torch.empty((M, N), device=x.device, dtype=torch.float16)
        
        grid = lambda META: (
            triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),
        )
        
        int4_gemm_kernel[grid](
            x_2d, self.weight_packed, self.scales, out,
            M, N, K,
            x_2d.stride(0), x_2d.stride(1),
            self.weight_packed.stride(0), self.weight_packed.stride(1),
            self.scales.stride(0), self.scales.stride(1),
            out.stride(0), out.stride(1),
            self.group_size
        )
        
        return out.reshape(batch_size, seq_len, N)

def get_inputs():
    batch_size = 4
    seq_len = 2048
    K = 4096
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16, device='cuda')]

def get_init_inputs():
    K = 4096
    N = 11008
    group_size = 128
    return [K, N, group_size]
