import torch
import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(Model, self).__init__()
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)
        self.num_heads = num_heads

    def forward(self, x):
        B, C, H, W = x.shape
        L = H * W
        # Flattened spatial to (L, B, C)
        x_seq = x.view(B, C, L).permute(2, 0, 1)

        # Use fast fused path with need_weights=False
        attn_output, _ = self.attn(x_seq, x_seq, x_seq, need_weights=False)
        
        # Residual and LayerNorm
        x_res = self.norm(attn_output + x_seq)
        
        # Back to (B, C, H, W)
        return x_res.permute(1, 2, 0).view(B, C, H, W)

def get_inputs():
    batch_size = 2
    num_channels = 128
    image_height = 128
    image_width = 128
    return [torch.randn(batch_size, num_channels, image_height, image_width)]

def get_init_inputs():
    embed_dim = 128
    num_heads = 4
    return [embed_dim, num_heads]
