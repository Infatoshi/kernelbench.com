"""
Optimized FP4-like GEMV using CUDA C++ for H100.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized GEMV kernel with shared memory and warp reduction
__global__ void gemv_fp4_kernel_shared(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    float x_scale,
    float weight_scale,
    half* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features
) {
    constexpr int TILE_K = 128;
    constexpr int THREADS = 256;
    
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    int out_idx = blockIdx.y * TILE_K + threadIdx.x;
    if (out_idx >= out_features) return;
    
    __shared__ float weight_shared[TILE_K];
    
    float acc = 0.0f;
    int x_row = batch_idx * in_features;
    
    // Process weight tiles
    for (int tile = 0; tile < (in_features + TILE_K - 1) / TILE_K; tile++) {
        int k_start = tile * TILE_K;
        int k_end = min(k_start + TILE_K, in_features);
        int local_k = threadIdx.x;
        
        // Load weight tile into shared memory
        if (k_start + local_k < k_end) {
            weight_shared[local_k] = static_cast<float>(weight_q[out_idx * in_features + k_start + local_k]) * weight_scale;
        }
        __syncthreads();
        
        // Load x values and compute
        float local_acc = 0.0f;
        for (int k = local_k; k < (k_end - k_start); k += THREADS) {
            float x_val = static_cast<float>(x_q[x_row + k_start + k]) * x_scale;
            local_acc += x_val * weight_shared[k];
        }
        
        // Warp reduce
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            local_acc += __shfl_down_sync(0xffffffff, local_acc, offset);
        }
        
        if (local_k % 32 == 0) {
            acc += local_acc;
        }
        __syncthreads();
    }
    
    output[batch_idx * out_features + out_idx] = __float2half(acc);
}

torch::Tensor gemv_fp4_op(torch::Tensor x_q, torch::Tensor weight_q, float x_scale, float weight_scale) {
    int batch_size = x_q.size(0);
    int in_features = x_q.size(1);
    int out_features = weight_q.size(0);
    
    auto output = torch::empty({batch_size, out_features}, x_q.options().dtype(torch::kFloat16));
    
    int threads = 256;
    dim3 grid(batch_size, (out_features + threads - 1) / threads);
    dim3 block(threads);
    
    gemv_fp4_kernel_shared<<<grid, block>>>(
        x_q.data_ptr<int8_t>(),
        weight_q.data_ptr<int8_t>(),
        x_scale,
        weight_scale,
        reinterpret_cast<half*>(output.data_ptr<at::Half>()),
        batch_size,
        in_features,
        out_features
    );
    
    cudaDeviceSynchronize();
    return output;
}
'''

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor gemv_fp4_op(torch::Tensor x_q, torch::Tensor weight_q, float x_scale, float weight_scale);
'''

ext = load_inline(
    name='gemv_fp4_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp4_op'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math', '-arch=sm_90'],
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        # Simulate fp4 range in int8 storage.
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp4_op(x_q, self.weight_q, x_scale.item(), self.weight_scale.item())


def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.08, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
