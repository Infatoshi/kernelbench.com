import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_gemm_scale_residual(
    X, W, B, Out,
    M, N, K,
    stride_xm, stride_xk,
    stride_wk, stride_wn,
    stride_bn,
    stride_outm, stride_outn,
    scale_factor,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """
    Fused GEMM with scaling and residual: Out = (X @ W + B) * scale_factor
    
    X: (M, K) - input
    W: (K, N) - weight (transposed from nn.Linear weight)
    B: (N,) - bias
    Out: (M, N) - output
    """
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n
    
    m_start = pid_m * BLOCK_M
    n_start = pid_n * BLOCK_N
    
    m_offs = m_start + tl.arange(0, BLOCK_M)
    n_offs = n_start + tl.arange(0, BLOCK_N)
    
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    for k in range(0, K, BLOCK_K):
        k_offs = k + tl.arange(0, BLOCK_K)
        
        x_ptrs = X + m_offs[:, None] * stride_xm + k_offs[None, :] * stride_xk
        x = tl.load(x_ptrs, mask=(m_offs[:, None] < M) & (k_offs[None, :] < K), other=0.0)
        
        w_ptrs = W + k_offs[:, None] * stride_wk + n_offs[None, :] * stride_wn
        w = tl.load(w_ptrs, mask=(k_offs[:, None,] < K) & (n_offs[None, :] < N), other=0.0)
        
        acc += tl.dot(x, w)
    
    # Add bias
    b_ptrs = B + n_offs * stride_bn
    b = tl.load(b_ptrs, mask=n_offs < N, other=0.0)
    acc += b[None, :]
    
    # Apply scale factor (which is 1 + scaling_factor from reference)
    acc = acc * scale_factor
    
    # Store result
    out_ptrs = Out + m_offs[:, None] * stride_outm + n_offs[None, :] * stride_outn
    tl.store(out_ptrs, acc, mask=(m_offs[:, None] < M) & (n_offs[None, :] < N))


class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
    
    def forward(self, x):
        batch_size, in_features = x.shape
        out_features = self.matmul.out_features
        
        # Transpose weight to (in_features, out_features) for our kernel
        # nn.Linear weight is (out_features, in_features)
        weight_t = self.matmul.weight.t().contiguous()
        bias = self.matmul.bias.contiguous()
        
        out = torch.empty(batch_size, out_features, dtype=x.dtype, device=x.device)
        
        # Block sizes tuned for B200
        BLOCK_M, BLOCK_N, BLOCK_K = 64, 64, 32
        grid = (triton.cdiv(batch_size, BLOCK_M) * triton.cdiv(out_features, BLOCK_N),)
        
        fused_gemm_scale_residual[grid](
            x, weight_t, bias, out,
            batch_size, out_features, in_features,
            x.stride(0), x.stride(1),
            weight_t.stride(0), weight_t.stride(1),
            bias.stride(0),
            out.stride(0), out.stride(1),
            1.0 + self.scaling_factor,
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
        )
        
        return out


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    scaling_factor = 0.5
    return [in_features, out_features, scaling_factor]
