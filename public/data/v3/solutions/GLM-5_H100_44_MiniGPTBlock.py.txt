import torch.nn.functional as F
import math

# CUDA kernels for optimized operations
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cmath>
#include <cfloat>

// Flash Attention kernel - optimized for causal attention
template<int BLOCK_SIZE, int HEAD_DIM>
__global__ void flash_attention_kernel(
    const float* __restrict__ q,
    const float* __restrict__ k,
    const float* __restrict__ v,
    float* __restrict__ out,
    int B, int H, int T, int head_dim,
    float sm_scale
) {
    // Each block handles one (batch, head) pair
    int b = blockIdx.x / H;
    int h = blockIdx.x % H;
    
    int tid = threadIdx.x;
    int t = tid;  // Query position handled by this thread
    
    // Shared memory for k, v cache for this head
    extern __shared__ float shared_mem[];
    float* k_cache = shared_mem;           // [T, HEAD_DIM]
    float* v_cache = shared_mem + T * HEAD_DIM;  // [T, HEAD_DIM]
    
    // Base pointers for this batch and head
    const float* q_base = q + (b * H + h) * T * head_dim;
    const float* k_base = k + (b * H + h) * T * head_dim;
    const float* v_base = v + (b * H + h) * T * head_dim;
    float* o_base = out + (b * H + h) * T * head_dim;
    
    // Load k and v into shared memory (all threads cooperate)
    for (int i = tid; i < T * head_dim; i += blockDim.x) {
        k_cache[i] = k_base[i];
        v_cache[i] = v_base[i];
    }
    __syncthreads();
    
    // Each thread handles one query position
    if (t >= T) return;
    
    // Load query vector for this position
    float q_vec[HEAD_DIM];
    for (int d = 0; d < head_dim; d++) {
        q_vec[d] = q_base[t * head_dim + d];
    }
    
    // Compute attention scores and find max (for numerical stability)
    float max_score = -FLT_MAX;
    float scores[BLOCK_SIZE];
    
    for (int k_pos = 0; k_pos <= t; k_pos++) {  // Causal: only attend to positions <= t
        float score = 0.0f;
        for (int d = 0; d < head_dim; d++) {
            score += q_vec[d] * k_cache[k_pos * head_dim + d];
        }
        scores[k_pos] = score * sm_scale;
        max_score = fmaxf(max_score, scores[k_pos]);
    }
    
    // Compute softmax
    float sum_exp = 0.0f;
    for (int k_pos = 0; k_pos <= t; k_pos++) {
        scores[k_pos] = expf(scores[k_pos] - max_score);
        sum_exp += scores[k_pos];
    }
    
    // Compute weighted sum of values
    for (int d = 0; d < head_dim; d++) {
        float out_val = 0.0f;
        for (int k_pos = 0; k_pos <= t; k_pos++) {
            out_val += scores[k_pos] * v_cache[k_pos * head_dim + d];
        }
        o_base[t * head_dim + d] = out_val / sum_exp;
    }
}

torch::Tensor flash_attention_cuda(torch::Tensor q, torch::Tensor k, torch::Tensor v, double sm_scale) {
    auto shape = q.sizes();
    int B = shape[0];
    int H = shape[1];
    int T = shape[2];
    int head_dim = shape[3];
    
    auto out = torch::empty_like(q);
    
    int block_size = T;
    size_t shared_mem = 2 * T * head_dim * sizeof(float);
    
    // Choose kernel based on head_dim
    if (head_dim <= 64) {
        flash_attention_kernel<1024, 64><<<B * H, block_size, shared_mem>>>(
            q.data_ptr<float>(),
            k.data_ptr<float>(),
            v.data_ptr<float>(),
            out.data_ptr<float>(),
            B, H, T, head_dim,
            static_cast<float>(sm_scale)
        );
    } else if (head_dim <= 96) {
        flash_attention_kernel<1024, 96><<<B * H, block_size, shared_mem>>>(
            q.data_ptr<float>(),
            k.data_ptr<float>(),
            v.data_ptr<float>(),
            out.data_ptr<float>(),
            B, H, T, head_dim,
            static_cast<float>(sm_scale)
        );
    } else {
        flash_attention_kernel<1024, 128><<<B * H, block_size, shared_mem>>>(
            q.data_ptr<float>(),
            k.data_ptr<float>(),
            v.data_ptr<float>(),
            out.data_ptr<float>(),
            B, H, T, head_dim,
            static_cast<float>(sm_scale)
        );
    }
    
    return out;
}

// Fused GELU kernel
__global__ void gelu_kernel(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n) return;
    
    float val = x[idx];
    float sqrt_2_over_pi = 0.7978845608028654f;
    float coeff = 0.044715f;
    
    float inner = sqrt_2_over_pi * (val + coeff * val * val * val);
    float tanh_inner = tanhf(inner);
    y[idx] = 0.5f * val * (1.0f + tanh_inner);
}

torch::Tensor gelu_cuda(torch::Tensor x) {
    int n = x.numel();
    auto y = torch::empty_like(x);
    
    int block = 256;
    int grid = (n + block - 1) / block;
    
    gelu_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        y.data_ptr<float>(),
        n
    );
    
    return y;
}

// Fused LayerNorm kernel
__global__ void layer_norm_kernel(
    const float* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ y,
    int M, int N, float eps
) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int lane = tid & 31;
    
    // Compute mean
    float sum = 0.0f;
    for (int i = tid; i < N; i += blockDim.x) {
        sum += x[row * N + i];
    }
    
    // Warp reduce
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Block reduce (first warp broadcasts)
    __shared__ float mean_shared;
    if (tid == 0) {
        mean_shared = sum / N;
    }
    __syncthreads();
    
    float mean = mean_shared;
    
    // Compute variance
    float sq_sum = 0.0f;
    for (int i = tid; i < N; i += blockDim.x) {
        float diff = x[row * N + i] - mean;
        sq_sum += diff * diff;
    }
    
    // Warp reduce
    for (int offset = 16; offset > 0; offset >>= 1) {
        sq_sum += __shfl_down_sync(0xffffffff, sq_sum, offset);
    }
    
    __shared__ float rstd_shared;
    if (tid == 0) {
        rstd_shared = rsqrtf(sq_sum / N + eps);
    }
    __syncthreads();
    
    float rstd = rstd_shared;
    
    // Apply normalization
    for (int i = tid; i < N; i += blockDim.x) {
        float val = (x[row * N + i] - mean) * rstd;
        y[row * N + i] = val * weight[i] + bias[i];
    }
}

torch::Tensor layer_norm_cuda(torch::Tensor x, torch::Tensor weight, torch::Tensor bias, double eps) {
    auto shape = x.sizes();
    int M = shape[0] * shape[1];
    int N = shape[2];
    
    auto y = torch::empty_like(x);
    
    int block = 256;
    dim3 grid(M);
    
    layer_norm_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        y.data_ptr<float>(),
        M, N, static_cast<float>(eps)
    );
    
    return y;
}
'''

cpp_source = r'''

torch::Tensor gelu_cuda(torch::Tensor x);

# Load extension
ext = load_inline(
    name='gpt_optimized',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['flash_attention_cuda', 'gelu_cuda', 'layer_norm_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_90', '-maxrregcount=64'],
    verbose=False,
)


class NewGELU(nn.Module):
    def __init__(self):
        super(NewGELU, self).__init__()
    
    def forward(self, x):
        if x.is_cuda and x.is_contiguous():
            return ext.gelu_cuda(x.float()).to(x.dtype)
        return 0.5 * x * (1.0 + torch.tanh(math.sqrt(2.0 / math.pi) * (x + 0.044715 * torch.pow(x, 3.0))))


class CausalSelfAttention(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.register_buffer("bias", torch.tril(torch.ones(max_seqlen, max_seqlen))
                                     .view(1, 1, max_seqlen, max_seqlen))
        self.n_head = n_head
        self.n_embd = n_embd

    def forward(self, x):
        B, T, C = x.size()
        
        # QKV projection
        qkv = self.c_attn(x)
        q, k, v = qkv.split(self.n_embd, dim=2)
        
        head_dim = C // self.n_head
        k = k.view(B, T, self.n_head, head_dim).transpose(1, 2)
        q = q.view(B, T, self.n_head, head_dim).transpose(1, 2)
        v = v.view(B, T, self.n_head, head_dim).transpose(1, 2)
        
        # Flash Attention
        sm_scale = 1.0 / math.sqrt(head_dim)
        y = ext.flash_attention_cuda(
            q.float(), k.float(), v.float(), sm_scale
        ).to(q.dtype)
        
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        
        # Output projection
        y = self.resid_dropout(self.c_proj(y))
        return y


class Model(nn.Module):
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        self.ln_1 = nn.LayerNorm(n_embd)
        self.attn = CausalSelfAttention(n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen)
        self.ln_2 = nn.LayerNorm(n_embd)
        self.mlp = nn.ModuleDict(dict(
            c_fc    = nn.Linear(n_embd, 4 * n_embd),
            c_proj  = nn.Linear(4 * n_embd, n_embd),
            act     = NewGELU(),
            dropout = nn.Dropout(resid_pdrop),
        ))
        m = self.mlp
        self.mlpf = lambda x: m.dropout(m.c_proj(m.act(m.c_fc(x))))

    def forward(self, x):
        x = x + self.attn(self.ln_1(x))
        x = x + self.mlpf(self.ln_2(x))
        return x


batch_size = 128

n_head = 8
attn_pdrop = 0.0
