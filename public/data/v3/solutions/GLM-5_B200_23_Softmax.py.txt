import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "softmax"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def softmax_kernel(
    input_ptr,
    output_ptr,
    stride_batch,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Optimized softmax kernel for B200.
    Each program instance processes one row.
    """
    row_idx = tl.program_id(0)
    
    # Compute pointers to the row
    row_start = input_ptr + row_idx * stride_batch
    col_offsets = tl.arange(0, BLOCK_SIZE)
    input_ptrs = row_start + col_offsets
    
    # Load the row with masking
    mask = col_offsets < n_cols
    row = tl.load(input_ptrs, mask=mask, other=-float('inf')).to(tl.float32)
    
    # Numerical stability: subtract max
    row_max = tl.max(row, axis=0)
    numerator = tl.exp(row - row_max)
    
    # Compute denominator (sum of exponentials)
    denominator = tl.sum(numerator, axis=0)
    
    # Compute softmax output
    softmax_output = numerator / denominator
    
    # Store the result
    output_ptrs = output_ptr + row_idx * stride_batch + col_offsets
    tl.store(output_ptrs, softmax_output, mask=mask)


def softmax_triton(x):
    """Optimized softmax using Triton kernels."""
    n_rows, n_cols = x.shape
    
    # Ensure contiguous memory
    if not x.is_contiguous():
        x = x.contiguous()
    
    output = torch.empty_like(x)
    
    # Choose block size (power of 2, at least n_cols)
    BLOCK_SIZE = triton.next_power_of_2(n_cols)
    
    # Launch kernel - one program per row
    grid = (n_rows,)
    
    softmax_kernel[grid](
        x,
        output,
        x.stride(0),
        n_cols,
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return output


class Model(nn.Module):
    """
    Optimized model that performs a Softmax activation using Triton kernels.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features).

        Returns:
            torch.Tensor: Output tensor with Softmax applied, same shape as input.
        """
        return softmax_triton(x)
