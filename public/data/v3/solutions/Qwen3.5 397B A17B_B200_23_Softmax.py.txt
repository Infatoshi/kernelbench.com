import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def softmax_kernel(
    X_ptr,
    Y_ptr,
    stride_x_row,
    stride_y_row,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Triton softmax kernel optimized for B200.
    Each program handles one row.
    """
    row_idx = tl.program_id(0)
    
    # Pointers to the start of the row
    x_row_ptr = X_ptr + row_idx * stride_x_row
    y_row_ptr = Y_ptr + row_idx * stride_y_row
    
    # First pass: find the maximum value in the row (for numerical stability)
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols
    
    x = tl.load(x_row_ptr + col_offsets, mask=mask, other=-float('inf'))
    row_max = tl.max(x, axis=0)
    
    # Second pass: compute exp(x - max) and sum
    x = tl.load(x_row_ptr + col_offsets, mask=mask, other=0.0)
    x_shifted = x - row_max
    exp_x = tl.exp(x_shifted)
    row_sum = tl.sum(exp_x, axis=0)
    
    # Third pass: normalize and store
    y = exp_x / row_sum
    tl.store(y_row_ptr + col_offsets, y, mask=mask)


@triton.jit
def softmax_kernel_fused(
    X_ptr,
    Y_ptr,
    stride_x_row,
    stride_y_row,
    n_cols,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Fused softmax kernel - single pass with atomic operations for reduction.
    More efficient for smaller row sizes.
    """
    row_idx = tl.program_id(0)
    
    # Pointers to the start of the row
    x_row_ptr = X_ptr + row_idx * stride_x_row
    y_row_ptr = Y_ptr + row_idx * stride_y_row
    
    col_offsets = tl.arange(0, BLOCK_SIZE)
    mask = col_offsets < n_cols
    
    # Load the row
    x = tl.load(x_row_ptr + col_offsets, mask=mask, other=-float('inf'))
    
    # Compute max
    row_max = tl.max(x, axis=0)
    
    # Compute exp(x - max)
    x_shifted = x - row_max
    exp_x = tl.exp(x_shifted)
    
    # Compute sum
    row_sum = tl.sum(exp_x, axis=0)
    
    # Normalize
    y = exp_x / row_sum
    
    # Store result
    tl.store(y_row_ptr + col_offsets, y, mask=mask)


class SoftmaxFunction(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, dim):
        ctx.dim = dim
        ctx.save_for_backward(x)
        
        # Ensure contiguous
        x = x.contiguous()
        
        # Move dim to last position if needed, but for dim=1 we handle it directly
        if dim != 1 and dim != -1:
            # Transpose to make dim=1 the reduction dimension
            x = x.transpose(dim, 1)
            needs_transpose_back = True
        else:
            needs_transpose_back = False
        
        original_shape = x.shape
        x_2d = x.reshape(-1, x.shape[-1])
        
        batch_size, n_cols = x_2d.shape
        
        # Choose block size based on n_cols
        if n_cols <= 512:
            BLOCK_SIZE = triton.next_power_of_2(n_cols)
        elif n_cols <= 1024:
            BLOCK_SIZE = 1024
        elif n_cols <= 2048:
            BLOCK_SIZE = 2048
        elif n_cols <= 4096:
            BLOCK_SIZE = 4096
        else:
            BLOCK_SIZE = 8192
        
        y_2d = torch.empty_like(x_2d)
        
        grid = (batch_size,)
        softmax_kernel_fused[grid](
            x_2d,
            y_2d,
            x_2d.stride(0),
            y_2d.stride(0),
            n_cols,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        y = y_2d.reshape(original_shape)
        
        if needs_transpose_back:
            y = y.transpose(dim, 1)
        
        return y
    
    @staticmethod
    def backward(ctx, grad_output):
        x = ctx.saved_tensors[0]
        dim = ctx.dim
        
        # For softmax backward: grad_input = grad_output * output - (grad_output * output).sum(dim) * output
        # We need to recompute output
        output = torch.softmax(x, dim=dim)
        
        grad_output = grad_output.contiguous()
        output = output.contiguous()
        
        # Compute grad_output * output
        grad_times_output = grad_output * output
        
        # Sum along the dim
        sum_grad_times_output = grad_times_output.sum(dim=dim, keepdim=True)
        
        # grad_input = output * (grad_output - sum_grad_times_output)
        grad_input = output * (grad_output - sum_grad_times_output)
        
        return grad_input, None


def triton_softmax(x, dim=1):
    return SoftmaxFunction.apply(x, dim)


class Model(nn.Module):
    """
    Optimized softmax model using Triton kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.
        """
        return triton_softmax(x, dim=1)
