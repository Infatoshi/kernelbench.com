import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Warp-level reduction utilities
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int mask = 16; mask >= 1; mask >>= 1) {
        float other = __shfl_xor_sync(0xffffffff, val, mask);
        val = fmaxf(val, other);
    }
    return val;
}

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = 16; mask >= 1; mask >>= 1) {
        val += __shfl_xor_sync(0xffffffff, val, mask);
    }
    return val;
}

// Block-level reduce max using shared memory
// num_warps must match actual number of warps in block
template<int NUM_WARPS>
__device__ __forceinline__ float block_reduce_max(float val, float* smem) {
    int lane = threadIdx.x & 31;
    int warp_id = threadIdx.x >> 5;
    
    val = warp_reduce_max(val);
    
    if (lane == 0) smem[warp_id] = val;
    __syncthreads();
    
    float result = (threadIdx.x < NUM_WARPS) ? smem[threadIdx.x] : -1e38f;
    // Only first warp does final reduction
    if (warp_id == 0) {
        result = warp_reduce_max(result);
    }
    result = __shfl_sync(0xffffffff, result, 0);
    return result;
}

template<int NUM_WARPS>
__device__ __forceinline__ float block_reduce_sum(float val, float* smem) {
    int lane = threadIdx.x & 31;
    int warp_id = threadIdx.x >> 5;
    
    val = warp_reduce_sum(val);
    
    if (lane == 0) smem[warp_id] = val;
    __syncthreads();
    
    float result = (threadIdx.x < NUM_WARPS) ? smem[threadIdx.x] : 0.0f;
    if (warp_id == 0) {
        result = warp_reduce_sum(result);
    }
    result = __shfl_sync(0xffffffff, result, 0);
    return result;
}

// Cross entropy with float4 vectorized loads
// BLOCK_SIZE=256, NUM_WARPS=8
template <int BLOCK_SIZE, int NUM_WARPS>
__global__ void cross_entropy_kernel_f32(
    const float* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ losses,
    int num_classes
) {
    __shared__ float smem[NUM_WARPS];
    int row = blockIdx.x;
    const float* row_ptr = logits + row * num_classes;
    int target = (int)targets[row];

    // Step 1: find max for numerical stability
    float local_max = -1e38f;
    
    // Vectorized load: float4 = 4 floats
    int vec_count = num_classes / 4;
    const float4* row_ptr4 = reinterpret_cast<const float4*>(row_ptr);
    
    for (int i = threadIdx.x; i < vec_count; i += BLOCK_SIZE) {
        float4 v = __ldg(row_ptr4 + i);
        float m = fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w));
        local_max = fmaxf(local_max, m);
    }
    // Handle remainder
    for (int i = vec_count * 4 + threadIdx.x; i < num_classes; i += BLOCK_SIZE) {
        local_max = fmaxf(local_max, __ldg(row_ptr + i));
    }

    float row_max = block_reduce_max<NUM_WARPS>(local_max, smem);
    __syncthreads();

    // Step 2: sum of exp(x - max)
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < vec_count; i += BLOCK_SIZE) {
        float4 v = __ldg(row_ptr4 + i);
        local_sum += expf(v.x - row_max) + expf(v.y - row_max) 
                   + expf(v.z - row_max) + expf(v.w - row_max);
    }
    for (int i = vec_count * 4 + threadIdx.x; i < num_classes; i += BLOCK_SIZE) {
        local_sum += expf(__ldg(row_ptr + i) - row_max);
    }

    float total_sum = block_reduce_sum<NUM_WARPS>(local_sum, smem);

    // Step 3: loss = log(sum_exp) + max - logit[target]
    if (threadIdx.x == 0) {
        losses[row] = logf(total_sum) + row_max - __ldg(row_ptr + target);
    }
}

// Cross entropy for half/bfloat16 types
template <typename scalar_t, int BLOCK_SIZE, int NUM_WARPS>
__global__ void cross_entropy_kernel_half(
    const scalar_t* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ losses,
    int num_classes
) {
    __shared__ float smem[NUM_WARPS];
    int row = blockIdx.x;
    const scalar_t* row_ptr = logits + row * num_classes;
    int target = (int)targets[row];

    // Step 1: find max
    float local_max = -1e38f;
    for (int i = threadIdx.x; i < num_classes; i += BLOCK_SIZE) {
        float val = (float)__ldg(row_ptr + i);
        local_max = fmaxf(local_max, val);
    }

    float row_max = block_reduce_max<NUM_WARPS>(local_max, smem);
    __syncthreads();

    // Step 2: sum of exp(x - max)
    float local_sum = 0.0f;
    for (int i = threadIdx.x; i < num_classes; i += BLOCK_SIZE) {
        local_sum += expf((float)__ldg(row_ptr + i) - row_max);
    }

    float total_sum = block_reduce_sum<NUM_WARPS>(local_sum, smem);

    if (threadIdx.x == 0) {
        losses[row] = logf(total_sum) + row_max - (float)__ldg(row_ptr + target);
    }
}

// Parallel reduction to compute mean of losses
__global__ void mean_kernel(const float* __restrict__ losses, float* output, int n) {
    __shared__ float smem[256];
    float val = 0.0f;
    for (int i = threadIdx.x; i < n; i += blockDim.x) {
        val += losses[i];
    }
    smem[threadIdx.x] = val;
    __syncthreads();
    
    for (int s = 128; s >= 1; s >>= 1) {
        if (threadIdx.x < s) smem[threadIdx.x] += smem[threadIdx.x + s];
        __syncthreads();
    }
    if (threadIdx.x == 0) output[0] = smem[0] * (1.0f / n);
}

torch::Tensor cross_entropy_forward(torch::Tensor logits, torch::Tensor targets) {
    int batch = logits.size(0);
    int num_classes = logits.size(1);

    auto losses = torch::empty({batch}, torch::TensorOptions().dtype(torch::kFloat32).device(logits.device()));
    auto output = torch::empty({1}, torch::TensorOptions().dtype(torch::kFloat32).device(logits.device()));

    constexpr int BLOCK_SIZE = 256;
    constexpr int NUM_WARPS = BLOCK_SIZE / 32;

    if (logits.dtype() == torch::kFloat32) {
        cross_entropy_kernel_f32<BLOCK_SIZE, NUM_WARPS><<<batch, BLOCK_SIZE>>>(
            logits.data_ptr<float>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            num_classes
        );
    } else if (logits.dtype() == torch::kFloat16) {
        cross_entropy_kernel_half<at::Half, BLOCK_SIZE, NUM_WARPS><<<batch, BLOCK_SIZE>>>(
            logits.data_ptr<at::Half>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            num_classes
        );
    } else if (logits.dtype() == torch::kBFloat16) {
        cross_entropy_kernel_half<at::BFloat16, BLOCK_SIZE, NUM_WARPS><<<batch, BLOCK_SIZE>>>(
            logits.data_ptr<at::BFloat16>(),
            targets.data_ptr<int64_t>(),
            losses.data_ptr<float>(),
            num_classes
        );
    }

    mean_kernel<<<1, 256>>>(losses.data_ptr<float>(), output.data_ptr<float>(), batch);

    return output.squeeze(0);
}
'''

cpp_source = r'''
torch::Tensor cross_entropy_forward(torch::Tensor logits, torch::Tensor targets);
'''

ext = load_inline(
    name='cross_entropy_ext2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['cross_entropy_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, predictions, targets):
        return ext.cross_entropy_forward(predictions, targets)

batch_size = 4096
num_classes = 1024
input_shape = (num_classes, )
dim = 1

def get_inputs():
    return [torch.randn(batch_size, *input_shape), torch.randint(0, num_classes, (batch_size,))]

    return []
