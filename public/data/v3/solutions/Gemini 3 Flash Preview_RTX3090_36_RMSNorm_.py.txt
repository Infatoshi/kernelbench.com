import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def rms_norm_kernel(
    x_ptr,
    y_ptr,
    n_features,
    n_elements_after_features,
    eps,
    num_rows,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0)
    if row_idx >= num_rows:
        return

    # row_idx is (b, d1, d2, ...)
    # Row layout has strides: B: C * D1 * D2, C: D1 * D2, D1: D2, D2: 1
    # n_elements_after_features = D1 * D2
    
    b_idx = row_idx // n_elements_after_features
    after_idx = row_idx % n_elements_after_features
    
    # offset for x[b_idx, 0, d1, d2]
    offset_base = b_idx * (n_features * n_elements_after_features) + after_idx
    
    # Thread offsets for dimension 1 (C)
    # We load x[b_idx, 0:C, d1, d2]
    # Memory distance between x[b_idx, c, d1, d2] and x[b_idx, c+1, d1, d2] is n_elements_after_features
    c_offsets = tl.arange(0, BLOCK_SIZE)
    mask = c_offsets < n_features
    
    curr_input_ptrs = x_ptr + offset_base + c_offsets * n_elements_after_features
    x = tl.load(curr_input_ptrs, mask=mask, other=0.0).to(tl.float32)
    
    # Standard RMS calculation
    # Mean is across the C dimension (the dimension being normalized)
    x2 = x * x
    sum_x2 = tl.sum(x2, axis=0)
    rms = tl.sqrt(sum_x2 / n_features + eps)
    
    out = x / rms
    
    curr_output_ptrs = y_ptr + offset_base + c_offsets * n_elements_after_features
    tl.store(curr_output_ptrs, out.to(y_ptr.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Check input rank
        rank = x.dim()
        if rank < 2:
            raise ValueError("Input must have at least 2 dimensions")
            
        shape = x.shape
        B = shape[0]
        C = shape[1] # n_features
        
        # Elements after dimension 1 (dim 1 is normalized)
        n_elements_after_features = 1
        for i in range(2, rank):
            n_elements_after_features *= shape[i]
            
        num_rows = B * n_elements_after_features
        
        y = torch.empty_like(x)
        
        # Pick BLOCK_SIZE to cover all features. If features > 1024,
        # Triton should handle up to 65536 or more, depending on shared memory.
        # Ampere handles up to 1024 threads per block, and we can use larger BLOCK_SIZE.
        BLOCK_SIZE = triton.next_power_of_2(C)
        
        # We launch one program per (batch, d1, d2, ...) "row"
        grid = (num_rows,)
        rms_norm_kernel[grid](
            x, y,
            C, n_elements_after_features,
            self.eps,
            num_rows,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        return y

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]

def get_init_inputs():
    features = 64
    return [features]
