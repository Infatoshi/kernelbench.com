import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_tanh_scale_bias_maxpool_kernel(
    conv_out_ptr,
    bias_ptr,
    out_ptr,
    batch_size: tl.constexpr,
    out_channels: tl.constexpr,
    out_height: tl.constexpr,
    out_width: tl.constexpr,
    scaling_factor: tl.constexpr,
    pool_kernel_size: tl.constexpr,
    # Strides
    conv_stride_b: tl.constexpr,
    conv_stride_c: tl.constexpr,
    conv_stride_h: tl.constexpr,
    conv_stride_w: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Fused kernel: Tanh -> Scale -> BiasAdd -> MaxPool
    Each program handles one output element after maxpool.
    """
    # Linear index
    pid = tl.program_id(0)
    
    # Decode into batch, channel, height, width (after maxpool)
    pre_pool_out_height = out_height * pool_kernel_size
    pre_pool_out_width = out_width * pool_kernel_size
    
    # Calculate output position
    out_idx = pid
    
    # Decode output position: batch, oc, out_h, out_w
    tmp = out_idx
    out_w = tmp % out_width
    tmp = tmp // out_width
    out_h = tmp % out_height
    tmp = tmp // out_height
    oc = tmp % out_channels
    batch_idx = tmp // out_channels
    
    # Compute max over 2x2 region using unrolled loop
    # Initialize with a very small value
    max_val = -1e38
    
    # Unroll the 2x2 maxpool
    # Position (0, 0)
    pre_h_00 = out_h * pool_kernel_size + 0
    pre_w_00 = out_w * pool_kernel_size + 0
    valid_00 = (pre_h_00 < pre_pool_out_height) & (pre_w_00 < pre_pool_out_width)
    conv_idx_00 = batch_idx * conv_stride_b + oc * conv_stride_c + pre_h_00 * conv_stride_h + pre_w_00
    conv_val_00 = tl.load(conv_out_ptr + conv_idx_00).to(tl.float32)
    tanh_00 = tl.math.tanh(conv_val_00)
    scaled_00 = tanh_00 * scaling_factor
    bias_val = tl.load(bias_ptr + oc).to(tl.float32)
    final_00 = scaled_00 + bias_val
    max_val = tl.where(valid_00, tl.maximum(max_val, final_00), max_val)
    
    # Position (0, 1)
    pre_h_01 = out_h * pool_kernel_size + 0
    pre_w_01 = out_w * pool_kernel_size + 1
    valid_01 = (pre_h_01 < pre_pool_out_height) & (pre_w_01 < pre_pool_out_width)
    conv_idx_01 = batch_idx * conv_stride_b + oc * conv_stride_c + pre_h_01 * conv_stride_h + pre_w_01
    conv_val_01 = tl.load(conv_out_ptr + conv_idx_01).to(tl.float32)
    tanh_01 = tl.math.tanh(conv_val_01)
    scaled_01 = tanh_01 * scaling_factor
    final_01 = scaled_01 + bias_val
    max_val = tl.where(valid_01, tl.maximum(max_val, final_01), max_val)
    
    # Position (1, 0)
    pre_h_10 = out_h * pool_kernel_size + 1
    pre_w_10 = out_w * pool_kernel_size + 0
    valid_10 = (pre_h_10 < pre_pool_out_height) & (pre_w_10 < pre_pool_out_width)
    conv_idx_10 = batch_idx * conv_stride_b + oc * conv_stride_c + pre_h_10 * conv_stride_h + pre_w_10
    conv_val_10 = tl.load(conv_out_ptr + conv_idx_10).to(tl.float32)
    tanh_10 = tl.math.tanh(conv_val_10)
    scaled_10 = tanh_10 * scaling_factor
    final_10 = scaled_10 + bias_val
    max_val = tl.where(valid_10, tl.maximum(max_val, final_10), max_val)
    
    # Position (1, 1)
    pre_h_11 = out_h * pool_kernel_size + 1
    pre_w_11 = out_w * pool_kernel_size + 1
    valid_11 = (pre_h_11 < pre_pool_out_height) & (pre_w_11 < pre_pool_out_width)
    conv_idx_11 = batch_idx * conv_stride_b + oc * conv_stride_c + pre_h_11 * conv_stride_h + pre_w_11
    conv_val_11 = tl.load(conv_out_ptr + conv_idx_11).to(tl.float32)
    tanh_11 = tl.math.tanh(conv_val_11)
    scaled_11 = tanh_11 * scaling_factor
    final_11 = scaled_11 + bias_val
    max_val = tl.where(valid_11, tl.maximum(max_val, final_11), max_val)
    
    # Store result
    tl.store(out_ptr + out_idx, max_val)


def fused_tanh_scale_bias_maxpool(conv_out, bias, scaling_factor, pool_kernel_size):
    """
    Fused kernel for: Tanh -> Scale -> BiasAdd -> MaxPool2d
    """
    batch_size, out_channels, out_h, out_w = conv_out.shape
    
    # Output after maxpool
    pool_out_h = out_h // pool_kernel_size
    pool_out_w = out_w // pool_kernel_size
    
    out = torch.empty(batch_size, out_channels, pool_out_h, pool_out_w, 
                      device=conv_out.device, dtype=conv_out.dtype)
    
    total_elements = batch_size * out_channels * pool_out_h * pool_out_w
    
    if total_elements == 0:
        return out
    
    # Launch kernel - one program per output element
    grid = (total_elements,)
    BLOCK_SIZE = 1024
    
    fused_tanh_scale_bias_maxpool_kernel[grid](
        conv_out,
        bias,
        out,
        batch_size,
        out_channels,
        pool_out_h,
        pool_out_w,
        scaling_factor,
        pool_kernel_size,
        conv_out.stride(0),
        conv_out.stride(1),
        conv_out.stride(2),
        conv_out.stride(3),
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return out


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=False)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size
        
        # Initialize conv weights
        nn.init.kaiming_uniform_(self.conv.weight, a=0.1)
        
    def forward(self, x):
        # Conv2d
        x = self.conv(x)
        
        # Fused: Tanh -> Scale -> BiasAdd -> MaxPool
        x = fused_tanh_scale_bias_maxpool(x, self.bias, self.scaling_factor, self.pool_kernel_size)
        
        return x


def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]


def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]


# Constants
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2
