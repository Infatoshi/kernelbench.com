class Model(nn.Module):
    def __init__(self, M: int, K: int, N: int, use_e4m3: bool = True):
        super().__init__()
        self.M = M
        self.K = K
        self.N = N
        self.use_e4m3 = use_e4m3

        if use_e4m3:
            self.fp8_dtype = torch.float8_e4m3fn
            self.fp8_max = 448.0
        else:
            self.fp8_dtype = torch.float8_e5m2
            self.fp8_max = 57344.0

        self.weight = nn.Parameter(torch.randn(K, N, dtype=torch.float32) * 0.02)
        
        self.register_buffer("w_fp8_t", torch.empty(self.N, self.K, dtype=self.fp8_dtype))
        self.register_buffer("w_scale_inv", torch.empty(1, dtype=torch.float32))
        
        # Track when initialized
        self.is_quantized = False

    def _quantize_weights(self):
        w_scale = self.fp8_max / self.weight.abs().max().clamp(min=1e-12)
        w_t = self.weight.t().contiguous()
        w_fp8 = (w_t * w_scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)
        self.w_fp8_t = w_fp8.t()
        self.w_scale_inv = (1.0 / w_scale).to(torch.float32)
        self.is_quantized = True

    def _load_from_state_dict(self, state_dict, prefix, local_metadata, strict,
                              missing_keys, unexpected_keys, error_msgs):
        super()._load_from_state_dict(state_dict, prefix, local_metadata, strict,
                                      missing_keys, unexpected_keys, error_msgs)
        self._quantize_weights()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not self.is_quantized:
            self._quantize_weights()
            
        input_dtype = x.dtype
        batch_size = x.shape[0]
        seq_len = x.shape[1]

        x_2d = x.view(-1, self.K)
        # Using built-in torch ops for now, but compiled
        # We can write a custom Triton core for this or use torch.compile if not disabled
        x_amax = x_2d.abs().max().clamp(min=1e-12)
        x_scale = self.fp8_max / x_amax
        x_scale_inv = (1.0 / x_scale).to(torch.float32)
        x_fp8 = (x_2d * x_scale).clamp(-self.fp8_max, self.fp8_max).to(self.fp8_dtype)

        out = torch._scaled_mm(
            x_fp8,
            self.w_fp8_t,
            scale_a=x_scale_inv,
            scale_b=self.w_scale_inv,
            out_dtype=input_dtype,
        )
        return out.view(batch_size, seq_len, self.N)
