import torch
import torch.nn as nn
import triton
import triton.language as tl


# Fused kernel: GroupNorm + Scale + MaxPool2d(2) + Clamp
# Input shape: (N, C, H, W) where C is divided into num_groups groups
# Each block handles one (batch, group, spatial_tile) chunk

# Strategy: We fuse GroupNorm + Scale + MaxPool + Clamp
# GroupNorm needs two passes: first compute mean/var, then normalize.
# We'll do it in two kernels:
# 1) Compute per-group mean and variance
# 2) Apply normalization, scale, maxpool, clamp

@triton.jit
def groupnorm_stats_kernel(
    X_ptr,  # input (N, C, H, W)
    Mean_ptr,  # output (N, num_groups)
    Var_ptr,   # output (N, num_groups)
    N: tl.constexpr, C: tl.constexpr, H: tl.constexpr, W: tl.constexpr,
    num_groups: tl.constexpr,
    channels_per_group: tl.constexpr,
    HW: tl.constexpr,
    group_size: tl.constexpr,  # channels_per_group * H * W
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles one (batch, group) pair
    pid = tl.program_id(0)
    batch_idx = pid // num_groups
    group_idx = pid % num_groups
    
    # Base offset for this batch and group
    base_offset = batch_idx * C * HW + group_idx * channels_per_group * HW
    
    # Compute mean and variance using online algorithm
    mean = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    m2 = tl.zeros([BLOCK_SIZE], dtype=tl.float32)
    count = 0
    
    # Iterate over all elements in this group
    for start in range(0, group_size, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < group_size
        x = tl.load(X_ptr + base_offset + offsets, mask=mask, other=0.0).to(tl.float32)
        mean += tl.where(mask, x, 0.0)
        m2 += tl.where(mask, x * x, 0.0)
    
    total_mean = tl.sum(mean, axis=0) / group_size
    total_m2 = tl.sum(m2, axis=0) / group_size
    total_var = total_m2 - total_mean * total_mean
    
    # Store
    out_idx = batch_idx * num_groups + group_idx
    tl.store(Mean_ptr + out_idx, total_mean)
    tl.store(Var_ptr + out_idx, total_var)


@triton.jit
def fused_gn_scale_maxpool_clamp_kernel(
    X_ptr,      # input (N, C, H, W) - after conv
    Mean_ptr,   # (N, num_groups)
    Var_ptr,    # (N, num_groups)
    Weight_ptr, # GroupNorm weight (C,)
    Bias_ptr,   # GroupNorm bias (C,)
    Scale_ptr,  # Scale parameter (C,)
    Out_ptr,    # output (N, C, H//2, W//2)
    N: tl.constexpr, C: tl.constexpr, H: tl.constexpr, W: tl.constexpr,
    outH: tl.constexpr, outW: tl.constexpr,
    num_groups: tl.constexpr,
    channels_per_group: tl.constexpr,
    clamp_min: tl.constexpr, clamp_max: tl.constexpr,
    eps: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    # Each program handles a chunk of output spatial elements for one (batch, channel)
    pid = tl.program_id(0)
    total_out_hw = outH * outW
    
    # Decode pid into (batch, channel, spatial_block)
    num_spatial_blocks = tl.cdiv(total_out_hw, BLOCK_SIZE)
    bc_idx = pid // num_spatial_blocks
    spatial_block = pid % num_spatial_blocks
    
    batch_idx = bc_idx // C
    c_idx = bc_idx % C
    group_idx = c_idx // channels_per_group
    
    # Load stats
    stats_idx = batch_idx * num_groups + group_idx
    mean = tl.load(Mean_ptr + stats_idx)
    var = tl.load(Var_ptr + stats_idx)
    inv_std = 1.0 / tl.sqrt(var + eps)
    
    # Load GN weight, bias, and scale for this channel
    gn_w = tl.load(Weight_ptr + c_idx).to(tl.float32)
    gn_b = tl.load(Bias_ptr + c_idx).to(tl.float32)
    scale = tl.load(Scale_ptr + c_idx).to(tl.float32)
    
    # (x - mean) * inv_std * gn_w * scale + gn_b * scale
    # = x * (inv_std * gn_w * scale) + (-mean * inv_std * gn_w + gn_b) * scale
    combined_scale = inv_std * gn_w * scale
    combined_bias = (-mean * inv_std * gn_w + gn_b) * scale
    
    # Output spatial offsets
    out_offsets = spatial_block * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = out_offsets < total_out_hw
    
    # Convert to (oh, ow) coordinates
    oh = out_offsets // outW
    ow = out_offsets % outW
    
    # Input coordinates for 2x2 maxpool
    ih = oh * 2
    iw = ow * 2
    
    in_base = batch_idx * C * H * W + c_idx * H * W
    
    # Load 4 input values for maxpool
    idx00 = in_base + ih * W + iw
    idx01 = in_base + ih * W + (iw + 1)
    idx10 = in_base + (ih + 1) * W + iw
    idx11 = in_base + (ih + 1) * W + (iw + 1)
    
    # Check bounds for maxpool (handle edge cases)
    mask01 = mask & ((iw + 1) < W)
    mask10 = mask & ((ih + 1) < H)
    mask11 = mask & ((iw + 1) < W) & ((ih + 1) < H)
    
    x00 = tl.load(X_ptr + idx00, mask=mask, other=0.0).to(tl.float32)
    x01 = tl.load(X_ptr + idx01, mask=mask01, other=-1e30).to(tl.float32)
    x10 = tl.load(X_ptr + idx10, mask=mask10, other=-1e30).to(tl.float32)
    x11 = tl.load(X_ptr + idx11, mask=mask11, other=-1e30).to(tl.float32)
    
    # Apply GroupNorm + Scale to each
    x00 = x00 * combined_scale + combined_bias
    x01 = x01 * combined_scale + combined_bias
    x10 = x10 * combined_scale + combined_bias
    x11 = x11 * combined_scale + combined_bias
    
    # Fix values that were -1e30 (out of bounds) - set to -inf after norm
    x01 = tl.where(mask01, x01, -1e30)
    x10 = tl.where(mask10, x10, -1e30)
    x11 = tl.where(mask11, x11, -1e30)
    
    # MaxPool
    maxval = tl.maximum(tl.maximum(x00, x01), tl.maximum(x10, x11))
    
    # Clamp
    maxval = tl.minimum(tl.maximum(maxval, clamp_min), clamp_max)
    
    # Store output
    out_base = batch_idx * C * outH * outW + c_idx * outH * outW
    tl.store(Out_ptr + out_base + out_offsets, maxval.to(Out_ptr.dtype.element_ty), mask=mask)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.num_groups = num_groups

    def forward(self, x):
        # Use cuDNN for conv2d (it's highly optimized)
        x = self.conv(x)
        
        N, C, H, W = x.shape
        num_groups = self.num_groups
        channels_per_group = C // num_groups
        HW = H * W
        group_size = channels_per_group * HW
        
        outH = H // self.maxpool_kernel_size
        outW = W // self.maxpool_kernel_size
        
        # Allocate stats tensors
        mean = torch.empty(N * num_groups, device=x.device, dtype=torch.float32)
        var = torch.empty(N * num_groups, device=x.device, dtype=torch.float32)
        
        # Kernel 1: compute group stats
        STATS_BLOCK = 1024
        grid_stats = (N * num_groups,)
        groupnorm_stats_kernel[grid_stats](
            x, mean, var,
            N, C, H, W,
            num_groups, channels_per_group, HW, group_size,
            BLOCK_SIZE=STATS_BLOCK,
        )
        
        # Kernel 2: fused normalize + scale + maxpool + clamp
        total_out_hw = outH * outW
        BLOCK = 1024
        num_spatial_blocks = (total_out_hw + BLOCK - 1) // BLOCK
        grid_fused = (N * C * num_spatial_blocks,)
        
        out = torch.empty(N, C, outH, outW, device=x.device, dtype=x.dtype)
        
        fused_gn_scale_maxpool_clamp_kernel[grid_fused](
            x, mean, var,
            self.group_norm.weight, self.group_norm.bias,
            self.scale.view(-1),
            out,
            N, C, H, W,
            outH, outW,
            num_groups, channels_per_group,
            self.clamp_min, self.clamp_max,
            1e-5,  # eps
            BLOCK_SIZE=BLOCK,
        )
        
        return out


batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
num_groups = 8
scale_shape = (out_channels, 1, 1)
maxpool_kernel_size = 2
clamp_min = 0.0
clamp_max = 1.0

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max]
