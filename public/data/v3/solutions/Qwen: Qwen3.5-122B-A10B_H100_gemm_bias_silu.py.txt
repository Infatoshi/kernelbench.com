"""
Optimized GEMM + Bias + SiLU fusion for H100 using CUDA C++.
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Simple GEMM + Bias + SiLU kernel with shared memory tiling
// 2D thread block

template <int BM, int BN, int BK, int TM, int TN>
__global__ void gemm_bias_silu_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    const half* __restrict__ bias,
    half* __restrict__ C,
    int M, int N, int K) {
    
    __shared__ half As[BM * BK];
    __shared__ half Bs[BK * BN];
    
    int tid = threadIdx.x;
    int tx = tid % BN;
    int ty = tid / BN;
    
    int block_row = blockIdx.y;
    int block_col = blockIdx.x;
    
    int row = block_row * BM + ty;
    int col = block_col * BN + tx;
    
    float acc = 0.0f;
    
    int num_k_blocks = (K + BK - 1) / BK;
    
    for (int k_block = 0; k_block < num_k_blocks; k_block++) {
        int k = k_block * BK;
        
        // Load A tile: thread (ty, tx) loads As[ty, k+tx]
        if (row < M && k + tx < K) {
            As[ty * BK + tx] = A[row * K + k + tx];
        } else {
            As[ty * BK + tx] = __float2half(0.0f);
        }
        
        // Load B tile: thread (ty, tx) loads Bs[k+ty, tx]
        if (k + ty < K && col < N) {
            Bs[(k + ty) % BK * BN + tx] = B[(k + ty) * N + col];
        } else {
            Bs[(k + ty) % BK * BN + tx] = __float2half(0.0f);
        }
        
        __syncthreads();
        
        // Compute dot product
        for (int kk = 0; kk < BK; kk++) {
            acc += __half2float(As[ty * BK + kk]) * __half2float(Bs[kk * BN + tx]);
        }
        
        __syncthreads();
    }
    
    // Apply bias and silu
    if (row < M && col < N) {
        float val = acc + __half2float(bias[col]);
        val = val / (1.0f + expf(-val));  // SiLU
        C[row * N + col] = __float2half(val);
    }
}

torch::Tensor gemm_bias_silu(torch::Tensor a, torch::Tensor b, torch::Tensor bias) {
    int M = a.size(0);
    int K = a.size(1);
    int N = b.size(1);
    
    auto c = torch::empty({M, N}, a.options());
    
    // Tuned parameters: 32x32 output tile, 16x16 input tile
    const int BM = 32;
    const int BN = 32;
    const int BK = 16;
    dim3 block(BN, BM);  // 32 x 32 = 1024 threads
    
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);
    
    gemm_bias_silu_kernel<BM, BN, BK, 1, 1><<<grid, block>>>(
        reinterpret_cast<half*>(a.data_ptr<at::Half>()),
        reinterpret_cast<half*>(b.data_ptr<at::Half>()),
        reinterpret_cast<half*>(bias.data_ptr<at::Half>()),
        reinterpret_cast<half*>(c.data_ptr<at::Half>()),
        M, N, K
    );
    
    return c;
}
'''

ext = load_inline(
    name='gemm_bias_silu_ext',
    cpp_sources='torch::Tensor gemm_bias_silu(torch::Tensor, torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gemm_bias_silu'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, n: int = 4096):
        super().__init__()
        self.bias = nn.Parameter(torch.randn(n, dtype=torch.float16) * 0.02)

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        return ext.gemm_bias_silu(a, b, self.bias)

def get_inputs():
    m = 2048
    n = 4096
    k = 2048
    return [torch.randn(m, k, dtype=torch.float16), torch.randn(k, n, dtype=torch.float16)]

def get_init_inputs():
    return []
