import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// Optimized for H100: better memory coalescing and warp-level reductions
template <typename T>
__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = warpSize / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

// Fused InstanceNorm2d + Divide kernel for H100
// Optimized for large spatial dimensions (256x256)
template <typename T>
__global__ void fused_instance_norm_divide_kernel(
    const T* __restrict__ x,
    T* __restrict__ output,
    const T divide_by,
    const int N,      // batch_size
    const int C,      // num_channels
    const int H,      // height
    const int W       // width
) {
    const int spatial = H * W;
    const int instance_idx = blockIdx.x;  // batch * channel
    const int batch_idx = instance_idx / C;
    const int channel_idx = instance_idx % C;
    
    if (instance_idx >= N * C) return;

    const int tid = threadIdx.x;
    const int num_threads = blockDim.x;
    
    // Optimized: process multiple elements per thread for better memory throughput
    // Each thread processes a strided set of elements
    float sum = 0.0f;
    float sq_sum = 0.0f;
    
    const int elements_per_thread = (spatial + num_threads - 1) / num_threads;
    
    #pragma unroll 4
    for (int i = 0; i < elements_per_thread; i++) {
        int idx = tid + i * num_threads;
        if (idx < spatial) {
            const int h = idx / W;
            const int w = idx % W;
            const int offset = ((batch_idx * C + channel_idx) * H + h) * W + w;
            const float val = (float)x[offset];
            sum += val;
            sq_sum += val * val;
        }
    }
    
    // First-level warp reduction
    sum = warp_reduce_sum(sum);
    sq_sum = warp_reduce_sum(sq_sum);
    
    // Shared memory for block-level reduction
    extern __shared__ float smem[];
    if (tid % warpSize == 0) {
        smem[tid / warpSize] = sum;
        smem[32 + tid / warpSize] = sq_sum;
    }
    __syncthreads();
    
    // Final reduction
    if (tid < 32) {
        float my_sum = (tid < (num_threads + warpSize - 1) / warpSize) ? smem[tid] : 0.0f;
        float my_sq_sum = (tid < (num_threads + warpSize - 1) / warpSize) ? smem[32 + tid] : 0.0f;
        my_sum = warp_reduce_sum(my_sum);
        my_sq_sum = warp_reduce_sum(my_sq_sum);
        if (tid == 0) {
            smem[0] = my_sum;
            smem[1] = my_sq_sum;
        }
    }
    __syncthreads();
    
    // Compute statistics
    float mean = smem[0] / spatial;
    float var = smem[1] / spatial - mean * mean;
    float inv_std = rsqrtf(var + 1e-5f);
    float scale = inv_std / (float)divide_by;
    
    // Second pass: normalize
    #pragma unroll 4
    for (int i = 0; i < elements_per_thread; i++) {
        int idx = tid + i * num_threads;
        if (idx < spatial) {
            const int h = idx / W;
            const int w = idx % W;
            const int offset = ((batch_idx * C + channel_idx) * H + h) * W + w;
            const float val = (float)x[offset];
            output[offset] = (T)((val - mean) * scale);
        }
    }
}

torch::Tensor fused_instance_norm_divide(
    torch::Tensor x,  // Output of Conv2d: (N, C, H, W)
    double divide_by
) {
    auto output = torch::empty_like(x);

    const int N = x.size(0);
    const int C = x.size(1);
    const int H = x.size(2);
    const int W = x.size(3);
    const int instances = N * C;

    // Use 512 threads for better occupancy on H100
    const int threads = 512;
    const int blocks = instances;
    const int smem_size = 64 * sizeof(float);

    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "fused_instance_norm_divide", ([&] {
        fused_instance_norm_divide_kernel<scalar_t><<<blocks, threads, smem_size>>>(
            x.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            (scalar_t)divide_by,
            N, C, H, W
        );
    }));

    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }

    return output;
}
'''

ext = load_inline(
    name='fused_instancenorm_divide',
    cpp_sources='torch::Tensor fused_instance_norm_divide(torch::Tensor x, double divide_by);',
    cuda_sources=cuda_source,
    functions=['fused_instance_norm_divide'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.instance_norm = nn.InstanceNorm2d(out_channels, affine=False, track_running_stats=False)
        self.divide_by = divide_by

    def forward(self, x):
        x = self.conv(x)
        x = ext.fused_instance_norm_divide(x, self.divide_by)
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    height, width = 256, 256
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    divide_by = 2.0
    return [in_channels, out_channels, kernel_size, divide_by]
