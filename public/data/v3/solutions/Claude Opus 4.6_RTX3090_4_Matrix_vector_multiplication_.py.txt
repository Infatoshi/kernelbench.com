import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Kernel: each block handles one row, using all threads to collaboratively reduce
// We use float4 vectorized loads for coalesced access
// Multiple blocks per row with atomic adds for large K

// Simple approach: one block per row, large number of threads, vectorized loads
__global__ void matvec_kernel(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int K
) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    const float4* A4 = reinterpret_cast<const float4*>(A + row * K);
    const float4* B4 = reinterpret_cast<const float4*>(B);
    
    int K4 = K / 4;
    
    float sum = 0.0f;
    
    for (int i = threadIdx.x; i < K4; i += blockDim.x) {
        float4 a = A4[i];
        float4 b = B4[i];
        sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
    }
    
    // Warp-level reduction
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Block-level reduction using shared memory
    __shared__ float shared[32]; // One per warp
    int lane = threadIdx.x % 32;
    int warp_id = threadIdx.x / 32;
    
    if (lane == 0) shared[warp_id] = sum;
    __syncthreads();
    
    // First warp reduces
    if (warp_id == 0) {
        int num_warps = blockDim.x / 32;
        sum = (lane < num_warps) ? shared[lane] : 0.0f;
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (lane == 0) {
            C[row] = sum;
        }
    }
}

// Multi-block per row kernel for better occupancy
__global__ void matvec_kernel_multiblock(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ partial,  // [M, num_blocks_per_row]
    int M, int K, int blocks_per_row
) {
    int row = blockIdx.x / blocks_per_row;
    int block_in_row = blockIdx.x % blocks_per_row;
    if (row >= M) return;
    
    int chunk_size = (K + blocks_per_row - 1) / blocks_per_row;
    // Align chunk_size to 4
    chunk_size = ((chunk_size + 3) / 4) * 4;
    int k_start = block_in_row * chunk_size;
    int k_end = min(k_start + chunk_size, K);
    
    if (k_start >= K) {
        if (threadIdx.x == 0) {
            partial[row * blocks_per_row + block_in_row] = 0.0f;
        }
        return;
    }
    
    const float4* A4 = reinterpret_cast<const float4*>(A + row * K + k_start);
    const float4* B4 = reinterpret_cast<const float4*>(B + k_start);
    
    int len4 = (k_end - k_start) / 4;
    
    float sum = 0.0f;
    for (int i = threadIdx.x; i < len4; i += blockDim.x) {
        float4 a = A4[i];
        float4 b = B4[i];
        sum += a.x * b.x + a.y * b.y + a.z * b.z + a.w * b.w;
    }
    
    // Warp-level reduction
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    __shared__ float shared[32];
    int lane = threadIdx.x % 32;
    int warp_id = threadIdx.x / 32;
    
    if (lane == 0) shared[warp_id] = sum;
    __syncthreads();
    
    if (warp_id == 0) {
        int num_warps = blockDim.x / 32;
        sum = (lane < num_warps) ? shared[lane] : 0.0f;
        for (int offset = 16; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        if (lane == 0) {
            partial[row * blocks_per_row + block_in_row] = sum;
        }
    }
}

__global__ void reduce_kernel(
    const float* __restrict__ partial,
    float* __restrict__ C,
    int M, int blocks_per_row
) {
    int row = blockIdx.x * blockDim.x + threadIdx.x;
    if (row >= M) return;
    
    float sum = 0.0f;
    for (int i = 0; i < blocks_per_row; i++) {
        sum += partial[row * blocks_per_row + i];
    }
    C[row] = sum;
}

torch::Tensor matvec(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    // Strategy: use multi-block approach for large K
    int threads = 256;
    int blocks_per_row = 8;  // Split K into 8 chunks
    
    auto partial = torch::empty({M, blocks_per_row}, A.options());
    
    dim3 grid(M * blocks_per_row);
    dim3 block(threads);
    
    matvec_kernel_multiblock<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        partial.data_ptr<float>(),
        M, K, blocks_per_row
    );
    
    // Final reduction
    reduce_kernel<<<(M + 255) / 256, 256>>>(
        partial.data_ptr<float>(),
        C.data_ptr<float>(),
        M, blocks_per_row
    );
    
    return C;
}
'''

cpp_source = r'''
torch::Tensor matvec(torch::Tensor A, torch::Tensor B);
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['matvec'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.matvec(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
