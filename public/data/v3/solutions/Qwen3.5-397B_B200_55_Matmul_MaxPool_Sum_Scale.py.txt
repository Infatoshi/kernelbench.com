cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

// Fused kernel: GEMM + MaxPool1d(kernel=2) + Sum + Scale
// Input: x (batch, in_features)
// Weight: (out_features, in_features)
// Bias: (out_features,)
// Output: (batch,) - after pooling (out_features/2) then sum

// FP32 kernel
__global__ void fused_kernel_fp32(
    const float* x,
    const float* weight,
    const float* bias,
    float* output,
    int batch,
    int in_features,
    int out_features,
    float scale_factor
) {
    int b = blockIdx.x;
    if (b >= batch) return;
    
    int tid = threadIdx.x;
    int num_threads = blockDim.x;
    
    int pooled_size = out_features / 2;
    
    const int TILE_SIZE = 8;
    float local_sums[TILE_SIZE] = {0.0f};
    
    for (int p_base = tid * TILE_SIZE; p_base < pooled_size; p_base += num_threads * TILE_SIZE) {
        for (int t = 0; t < TILE_SIZE && (p_base + t) < pooled_size; t++) {
            int p = p_base + t;
            int pos1 = 2 * p;
            int pos2 = 2 * p + 1;
            
            float val1 = 0.0f, val2 = 0.0f;
            
            for (int k = 0; k < in_features; k++) {
                float x_val = x[b * in_features + k];
                float w1_val = weight[pos1 * in_features + k];
                float w2_val = weight[pos2 * in_features + k];
                val1 += x_val * w1_val;
                val2 += x_val * w2_val;
            }
            val1 += bias[pos1];
            val2 += bias[pos2];
            
            float max_val = (val1 > val2) ? val1 : val2;
            local_sums[t] += max_val;
        }
    }
    
    float thread_sum = 0.0f;
    for (int t = 0; t < TILE_SIZE; t++) {
        thread_sum += local_sums[t];
    }
    
    extern __shared__ float sbuf[];
    sbuf[tid] = thread_sum;
    __syncthreads();
    
    for (int stride = num_threads / 2; stride > 0; stride >>= 1) {
        if (tid < stride) {
            sbuf[tid] += sbuf[tid + stride];
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[b] = sbuf[0] * scale_factor;
    }
}

torch::Tensor fused_matmul_maxpool_sum_scale(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor
) {
    int batch = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);
    
    auto output = torch::empty({batch}, x.options());
    
    int threads = 256;
    int blocks = batch;
    int shared_mem = threads * sizeof(float);
    
    if (x.dtype() == at::kHalf) {
        fused_kernel_fp16<<<blocks, threads, shared_mem>>>(
            (const half*)x.data_ptr(),
            (const half*)weight.data_ptr(),
            (const half*)bias.data_ptr(),
            (half*)output.data_ptr(),
            batch, in_features, out_features,
            scale_factor
        );
    } else if (x.dtype() == at::kBFloat16) {
        fused_kernel_bf16<<<blocks, threads, shared_mem>>>(
            (const nv_bfloat16*)x.data_ptr(),
            (const nv_bfloat16*)weight.data_ptr(),
            (const nv_bfloat16*)bias.data_ptr(),
            (nv_bfloat16*)output.data_ptr(),
            batch, in_features, out_features,
            scale_factor
        );
    } else {
        fused_kernel_fp32<<<blocks, threads, shared_mem>>>(
            x.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.data_ptr<float>(),
            output.data_ptr<float>(),
            batch, in_features, out_features,
            scale_factor
        );
    }
    
    return output;
}
'''

cpp_source = """
#include <torch/extension.h>
torch::Tensor fused_matmul_maxpool_sum_scale(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale_factor
);
"""

ext = load_inline(
    name='fused_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_maxpool_sum_scale'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)


class Model(nn.Module):
    """
    Model that performs matrix multiplication, max pooling, sum, and scaling.
    """
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.max_pool = nn.MaxPool1d(kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size,).
        """
        weight = self.matmul.weight
        bias = self.matmul.bias
        return ext.fused_matmul_maxpool_sum_scale(x, weight, bias, self.scale_factor)


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    kernel_size = 2
    scale_factor = 0.5
    return [in_features, out_features, kernel_size, scale_factor]
