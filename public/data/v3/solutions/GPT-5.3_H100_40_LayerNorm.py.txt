import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void layernorm_kernel(
    const float* __restrict__ x,
    const float* __restrict__ gamma,
    const float* __restrict__ beta,
    float* __restrict__ y,
    int N, int D, float eps)
{
    int n = blockIdx.x;
    if (n >= N) return;

    const float* x_ptr = x + n * D;
    float* y_ptr = y + n * D;

    __shared__ float s_mean;
    __shared__ float s_var;

    float sum = 0.0f;
    for (int i = threadIdx.x; i < D; i += blockDim.x) {
        sum += x_ptr[i];
    }

    __shared__ float buf[512];
    int tid = threadIdx.x;
    buf[tid] = sum;
    __syncthreads();

    // reduction
    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) buf[tid] += buf[tid + s];
        __syncthreads();
    }

    float mean = buf[0] / D;
    if (tid == 0) s_mean = mean;
    __syncthreads();

    float var_sum = 0.0f;
    for (int i = threadIdx.x; i < D; i += blockDim.x) {
        float diff = x_ptr[i] - s_mean;
        var_sum += diff * diff;
    }

    buf[tid] = var_sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) buf[tid] += buf[tid + s];
        __syncthreads();
    }

    float var = buf[0] / D;
    if (tid == 0) s_var = var;
    __syncthreads();

    float inv_std = rsqrtf(s_var + eps);

    for (int i = threadIdx.x; i < D; i += blockDim.x) {
        float norm = (x_ptr[i] - s_mean) * inv_std;
        y_ptr[i] = norm * gamma[i] + beta[i];
    }
}

torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps) {
    auto y = torch::empty_like(x);

    int64_t N = x.size(0);
    int64_t D = 1;
    for (int i = 1; i < x.dim(); ++i) D *= x.size(i);

    int threads = 256;
    dim3 blocks(N);

    layernorm_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        gamma.data_ptr<float>(),
        beta.data_ptr<float>(),
        y.data_ptr<float>(),
        N, D, eps);

    return y;
}
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources='torch::Tensor layernorm_forward(torch::Tensor x, torch::Tensor gamma, torch::Tensor beta, float eps);',
    cuda_sources=cuda_src,
    functions=['layernorm_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.normalized_shape = normalized_shape
        self.weight = nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.bias = nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        orig_dtype = x.dtype
        x_f = x.float().contiguous()
        gamma = self.weight.contiguous().view(-1)
        beta = self.bias.contiguous().view(-1)
        y = ext.layernorm_forward(x_f, gamma, beta, self.eps)
        return y.to(orig_dtype)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2, device='cuda')
    return [x]


def get_init_inputs():
    return [(64, 256, 256)]
