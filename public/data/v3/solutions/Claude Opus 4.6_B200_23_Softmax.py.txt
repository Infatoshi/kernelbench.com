import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <float.h>

// Warp-level reduction helpers
__device__ __forceinline__ float warp_reduce_max(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val = fmaxf(val, __shfl_xor_sync(0xffffffff, val, offset));
    return val;
}

__device__ __forceinline__ float warp_reduce_sum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, offset);
    return val;
}

// Block-level reduction using shared memory
// blockDim.x = 32 (warp), blockDim.y = NUM_WARPS
__device__ __forceinline__ float block_reduce_max(float val, float* shared) {
    int lane = threadIdx.x;
    int warp_id = threadIdx.y;
    
    val = warp_reduce_max(val);
    
    if (lane == 0)
        shared[warp_id] = val;
    __syncthreads();
    
    // First warp reduces across warps
    if (warp_id == 0) {
        val = (lane < blockDim.y) ? shared[lane] : -FLT_MAX;
        val = warp_reduce_max(val);
    }
    __syncthreads();
    
    // Broadcast
    if (warp_id == 0 && lane == 0)
        shared[0] = val;
    __syncthreads();
    
    return shared[0];
}

__device__ __forceinline__ float block_reduce_sum(float val, float* shared) {
    int lane = threadIdx.x;
    int warp_id = threadIdx.y;
    
    val = warp_reduce_sum(val);
    
    if (lane == 0)
        shared[warp_id] = val;
    __syncthreads();
    
    if (warp_id == 0) {
        val = (lane < blockDim.y) ? shared[lane] : 0.0f;
        val = warp_reduce_sum(val);
    }
    __syncthreads();
    
    if (warp_id == 0 && lane == 0)
        shared[0] = val;
    __syncthreads();
    
    return shared[0];
}

// Each block processes one row. blockDim = (32, NUM_WARPS)
// n_cols = 16384, each thread processes n_cols / (32 * NUM_WARPS) elements
template<int NUM_WARPS>
__global__ void softmax_kernel(const float* __restrict__ input,
                                float* __restrict__ output,
                                int n_rows, int n_cols) {
    int row = blockIdx.x;
    if (row >= n_rows) return;
    
    const float* row_in = input + row * n_cols;
    float* row_out = output + row * n_cols;
    
    int tid = threadIdx.y * 32 + threadIdx.x;
    int total_threads = 32 * NUM_WARPS;
    
    __shared__ float shared[32];
    
    // Pass 1: find max
    float local_max = -FLT_MAX;
    for (int i = tid; i < n_cols; i += total_threads) {
        local_max = fmaxf(local_max, row_in[i]);
    }
    float row_max = block_reduce_max(local_max, shared);
    
    // Pass 2: compute sum of exp
    float local_sum = 0.0f;
    for (int i = tid; i < n_cols; i += total_threads) {
        local_sum += __expf(row_in[i] - row_max);
    }
    float row_sum = block_reduce_sum(local_sum, shared);
    float inv_sum = 1.0f / row_sum;
    
    // Pass 3: write output
    for (int i = tid; i < n_cols; i += total_threads) {
        row_out[i] = __expf(row_in[i] - row_max) * inv_sum;
    }
}

// Vectorized version using float4
template<int NUM_WARPS>
__global__ void softmax_kernel_vec4(const float* __restrict__ input,
                                     float* __restrict__ output,
                                     int n_rows, int n_cols) {
    int row = blockIdx.x;
    if (row >= n_rows) return;
    
    const float4* row_in = reinterpret_cast<const float4*>(input + row * n_cols);
    float4* row_out = reinterpret_cast<float4*>(output + row * n_cols);
    int n_cols4 = n_cols / 4;
    
    int tid = threadIdx.y * 32 + threadIdx.x;
    int total_threads = 32 * NUM_WARPS;
    
    __shared__ float shared[32];
    
    // Pass 1: find max
    float local_max = -FLT_MAX;
    for (int i = tid; i < n_cols4; i += total_threads) {
        float4 v = row_in[i];
        local_max = fmaxf(local_max, fmaxf(fmaxf(v.x, v.y), fmaxf(v.z, v.w)));
    }
    float row_max = block_reduce_max(local_max, shared);
    
    // Pass 2: compute sum of exp
    float local_sum = 0.0f;
    for (int i = tid; i < n_cols4; i += total_threads) {
        float4 v = row_in[i];
        local_sum += __expf(v.x - row_max) + __expf(v.y - row_max) +
                     __expf(v.z - row_max) + __expf(v.w - row_max);
    }
    float row_sum = block_reduce_sum(local_sum, shared);
    float inv_sum = 1.0f / row_sum;
    
    // Pass 3: write output
    for (int i = tid; i < n_cols4; i += total_threads) {
        float4 v = row_in[i];
        float4 out;
        out.x = __expf(v.x - row_max) * inv_sum;
        out.y = __expf(v.y - row_max) * inv_sum;
        out.z = __expf(v.z - row_max) * inv_sum;
        out.w = __expf(v.w - row_max) * inv_sum;
        row_out[i] = out;
    }
}

torch::Tensor softmax_cuda(torch::Tensor input) {
    auto output = torch::empty_like(input);
    int n_rows = input.size(0);
    int n_cols = input.size(1);
    
    constexpr int NUM_WARPS = 8;
    dim3 block(32, NUM_WARPS);
    dim3 grid(n_rows);
    
    if (n_cols % 4 == 0) {
        softmax_kernel_vec4<NUM_WARPS><<<grid, block>>>(
            input.data_ptr<float>(), output.data_ptr<float>(), n_rows, n_cols);
    } else {
        softmax_kernel<NUM_WARPS><<<grid, block>>>(
            input.data_ptr<float>(), output.data_ptr<float>(), n_rows, n_cols);
    }
    
    return output;
}
'''

cpp_source = "torch::Tensor softmax_cuda(torch::Tensor input);"

ext = load_inline(
    name='softmax_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_100'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.softmax_cuda(x)


batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []
