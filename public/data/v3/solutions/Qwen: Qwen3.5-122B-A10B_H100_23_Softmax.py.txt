import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def softmax_kernel(
    x_ptr,
    output_ptr,
    N: tl.constexpr,
    M: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Optimized softmax kernel using Triton.
    Each program handles one row (one sample in batch).
    """
    row_idx = tl.program_id(0)
    
    # Compute pointers for this row
    x_row = x_ptr + row_idx * M
    out_row = output_ptr + row_idx * M
    
    # First pass: find max for numerical stability
    max_val = tl.full([1], value=-float("inf"), dtype=tl.float32)
    for start in range(0, M, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < M
        x_vals = tl.load(x_row + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        max_val = tl.maximum(max_val, tl.max(x_vals, axis=0))
    
    # Second pass: compute exp(x - max) and sum
    sum_val = tl.zeros([1], dtype=tl.float32)
    for start in range(0, M, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < M
        x_vals = tl.load(x_row + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        sum_val += tl.sum(exp_vals, axis=0)
    
    # Third pass: compute final softmax output
    for start in range(0, M, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < M
        x_vals = tl.load(x_row + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        out_vals = exp_vals / sum_val
        tl.store(out_row + offs, out_vals, mask=mask)


@triton.jit
def softmax_kernel_fused(
    x_ptr,
    output_ptr,
    N: tl.constexpr,
    M: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Fused softmax kernel with better memory coalescing.
    Each program handles one row.
    """
    row_idx = tl.program_id(0)
    
    # Compute pointers for this row
    x_row = x_ptr + row_idx * M
    out_row = output_ptr + row_idx * M
    
    # Load all data for this row (if M <= BLOCK_SIZE, single load)
    # For larger M, we need multiple loads
    offs = tl.arange(0, BLOCK_SIZE)
    
    # First pass: find max
    max_val = tl.full([1], value=-float("inf"), dtype=tl.float32)
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        max_val = tl.maximum(max_val, tl.max(x_vals, axis=0))
    
    # Second pass: compute exp and sum
    sum_val = tl.zeros([1], dtype=tl.float32)
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        sum_val += tl.sum(exp_vals, axis=0)
    
    # Third pass: store result
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        out_vals = exp_vals / sum_val
        tl.store(out_row + block_start + offs, out_vals, mask=mask)


@triton.jit
def softmax_kernel_optimized(
    x_ptr,
    output_ptr,
    stride_x_row,
    stride_out_row,
    M: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """
    Optimized softmax with stride support for flexibility.
    Each program handles one row.
    """
    row_idx = tl.program_id(0)
    
    x_row = x_ptr + row_idx * stride_x_row
    out_row = output_ptr + row_idx * stride_out_row
    
    offs = tl.arange(0, BLOCK_SIZE)
    
    # First pass: find max for numerical stability
    max_val = tl.full([1], value=-float("inf"), dtype=tl.float32)
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        max_val = tl.maximum(max_val, tl.max(x_vals, axis=0))
    
    # Second pass: compute exp and sum
    sum_val = tl.zeros([1], dtype=tl.float32)
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        sum_val += tl.sum(exp_vals, axis=0)
    
    # Third pass: store result
    for block_start in range(0, M, BLOCK_SIZE):
        mask = block_start + offs < M
        x_vals = tl.load(x_row + block_start + offs, mask=mask, other=-float("inf"))
        x_vals = x_vals.to(tl.float32)
        exp_vals = tl.exp(x_vals - max_val)
        out_vals = exp_vals / sum_val
        tl.store(out_row + block_start + offs, out_vals, mask=mask)


def softmax_forward(x: torch.Tensor) -> torch.Tensor:
    """
    Optimized softmax forward pass.
    """
    assert x.dim() == 2, "Input must be 2D tensor"
    
    N, M = x.shape
    output = torch.empty_like(x)
    
    # Choose block size based on M
    # For H100, we can use larger block sizes
    if M <= 1024:
        BLOCK_SIZE = 1024
    elif M <= 2048:
        BLOCK_SIZE = 2048
    elif M <= 4096:
        BLOCK_SIZE = 4096
    elif M <= 8192:
        BLOCK_SIZE = 8192
    else:
        BLOCK_SIZE = 16384
    
    # Ensure BLOCK_SIZE is at least M for efficiency
    BLOCK_SIZE = max(BLOCK_SIZE, triton.next_power_of_2(M))
    
    # Number of program instances = number of rows
    grid = (N,)
    
    softmax_kernel_optimized[grid](
        x,
        output,
        x.stride(0),
        output.stride(0),
        M,
        BLOCK_SIZE,
    )
    
    return output


class Model(nn.Module):
    """
    Optimized model using custom softmax kernel.
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies Softmax activation to the input tensor.
        """
        return softmax_forward(x)
