import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "elementwise"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _gelu_kernel(X, Y, N, BLOCK: tl.constexpr):
    pid = tl.program_id(0)
    offs = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offs < N
    x = tl.load(X + offs, mask=mask)
    y = x * 0.5 * (1.0 + tl.erf(x * 0.7071067811865475))
    tl.store(Y + offs, y, mask=mask)


def gelu_triton(x: torch.Tensor) -> torch.Tensor:
    assert x.is_cuda
    y = torch.empty_like(x)
    n = x.numel()
    grid = lambda meta: (triton.cdiv(n, meta['BLOCK']),)
    _gelu_kernel[grid](x, y, n, BLOCK=1024)
    return y


class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return gelu_triton(x)


batch_size = 256
dim = 16384

    x = torch.randn(batch_size, dim)
    return [x]
