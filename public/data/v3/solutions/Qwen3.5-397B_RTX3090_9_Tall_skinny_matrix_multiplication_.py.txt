import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Tall-skinny GEMM: A (M x K) * B (K x N) = C (M x N)
// For this problem: M=16384, K=16, N=16384
// K is very small (16), so we optimize for that

constexpr int TILE_M = 128;
constexpr int TILE_N = 128;

__global__ void tall_skinny_gemm(
    const float* __restrict__ A,
    const float* __restrict__ B,
    float* __restrict__ C,
    int M, int N, int K_dim
) {
    // Each block computes a TILE_M x TILE_N tile of C
    int block_row = blockIdx.y;
    int block_col = blockIdx.x;
    
    int row_start = block_row * TILE_M;
    int col_start = block_col * TILE_N;
    
    // Shared memory for B tile (K x TILE_N)
    __shared__ float Bs[16 * TILE_N];
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Load B tile into shared memory
    // B is (K x N), we need columns [col_start:col_start+TILE_N]
    for (int k = ty; k < K_dim; k += TILE_M) {
        for (int c = tx; c < TILE_N; c += TILE_N) {
            int col = col_start + c;
            if (col < N) {
                Bs[k * TILE_N + c] = B[k * N + col];
            } else {
                Bs[k * TILE_N + c] = 0.0f;
            }
        }
    }
    
    __syncthreads();
    
    // Compute output tile
    // Each thread computes one element
    int row = row_start + ty;
    int col = col_start + tx;
    
    if (row < M && col < N) {
        float sum = 0.0f;
        #pragma unroll
        for (int k = 0; k < K_dim; k++) {
            sum += A[row * K_dim + k] * Bs[k * TILE_N + tx];
        }
        C[row * N + col] = sum;
    }
}

torch::Tensor tall_skinny_matmul(torch::Tensor A, torch::Tensor B) {
    // A: (M, K), B: (K, N) -> C: (M, N)
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(1);
    
    auto C = torch::empty({M, N}, A.options());
    
    // Grid and block dimensions
    dim3 block(TILE_N, TILE_M);
    dim3 grid((N + TILE_N - 1) / TILE_N, (M + TILE_M - 1) / TILE_M);
    
    tall_skinny_gemm<<<grid, block>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, N, K
    );
    
    return C;
}
'''

ext = load_inline(
    name='tall_skinny_ext',
    cpp_sources='torch::Tensor tall_skinny_matmul(torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['tall_skinny_matmul'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86', '-O3'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A, B):
        return ext.tall_skinny_matmul(A, B)

M = 16384
N = 16

def get_inputs():
    A = torch.randn(M, N)
    B = torch.randn(N, M)
    return [A, B]

def get_init_inputs():
    return []
