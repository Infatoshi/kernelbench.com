import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl


@triton.jit
def moe_gate_up_kernel(
    x_ptr,           # (num_tokens, hidden_size)
    gate_ptr,        # (intermediate_size, hidden_size) - single expert
    up_ptr,          # (intermediate_size, hidden_size) - single expert
    output_ptr,      # (num_tokens, intermediate_size)
    num_tokens: tl.constexpr,
    hidden_size: tl.constexpr,
    intermediate_size: tl.constexpr,
    BLOCK_HIDDEN: tl.constexpr,
    BLOCK_INTER: tl.constexpr,
):
    """
    Kernel for computing gate = SiLU(gate_proj(x)) and up = up_proj(x), then output = gate * up
    for a single expert.
    """
    pid = tl.program_id(0)
    inter_block = tl.program_id(1)
    
    if pid >= num_tokens:
        return
    
    # Compute intermediate index
    inter_start = inter_block * BLOCK_INTER
    inter_offs = inter_start + tl.arange(0, BLOCK_INTER)
    inter_mask = inter_offs < intermediate_size
    
    # Accumulator
    acc = tl.zeros([BLOCK_INTER], dtype=tl.float32)
    
    # Process hidden dimension in blocks
    for hidden_start in range(0, hidden_size, BLOCK_HIDDEN):
        hidden_offs = hidden_start + tl.arange(0, BLOCK_HIDDEN)
        hidden_mask = hidden_offs < hidden_size
        
        # Load x
        x_offs = pid * hidden_size + hidden_offs
        x = tl.load(x_ptr + x_offs, mask=hidden_mask, other=0.0)
        
        # Load gate weights: gate_ptr is (intermediate_size, hidden_size)
        gate_w_offs = inter_offs[:, None] * hidden_size + hidden_offs[None, :]
        gate_w = tl.load(gate_ptr + gate_w_offs, mask=inter_mask[:, None] & hidden_mask[None, :], other=0.0)
        
        # Load up weights
        up_w_offs = inter_offs[:, None] * hidden_size + hidden_offs[None, :]
        up_w = tl.load(up_ptr + up_w_offs, mask=inter_mask[:, None] & hidden_mask[None, :], other=0.0)
        
        # Compute dot products
        gate_val = tl.sum(gate_w * x[None, :], axis=1)
        up_val = tl.sum(up_w * x[None, :], axis=1)
        
        # SiLU activation
        gate_silu = gate_val / (1.0 + tl.exp(-gate_val))
        
        acc += gate_silu * up_val * inter_mask
    
    # Store result
    out_offs = pid * intermediate_size + inter_offs
    tl.store(output_ptr + out_offs, acc, mask=inter_mask)


@triton.jit
def moe_down_kernel(
    inter_ptr,       # (num_tokens, intermediate_size)
    down_ptr,        # (hidden_size, intermediate_size) - single expert
    weights_ptr,     # (num_tokens,)
    output_ptr,      # (num_tokens, hidden_size)
    num_tokens: tl.constexpr,
    hidden_size: tl.constexpr,
    intermediate_size: tl.constexpr,
    BLOCK_INTER: tl.constexpr,
    BLOCK_HIDDEN: tl.constexpr,
):
    """
    Kernel for computing output = down_proj(intermediate) * weight
    for a single expert.
    """
    pid = tl.program_id(0)
    hidden_block = tl.program_id(1)
    
    if pid >= num_tokens:
        return
    
    # Compute hidden index
    hidden_start = hidden_block * BLOCK_HIDDEN
    hidden_offs = hidden_start + tl.arange(0, BLOCK_HIDDEN)
    hidden_mask = hidden_offs < hidden_size
    
    # Accumulator
    acc = tl.zeros([BLOCK_HIDDEN], dtype=tl.float32)
    
    # Load weight
    weight = tl.load(weights_ptr + pid)
    
    # Process intermediate dimension in blocks
    for inter_start in range(0, intermediate_size, BLOCK_INTER):
        inter_offs = inter_start + tl.arange(0, BLOCK_INTER)
        inter_mask = inter_offs < intermediate_size
        
        # Load intermediate
        inter_base = pid * intermediate_size + inter_offs
        inter = tl.load(inter_ptr + inter_base, mask=inter_mask, other=0.0)
        
        # Load down weights: down_ptr is (hidden_size, intermediate_size)
        down_w_offs = hidden_offs[:, None] * intermediate_size + inter_offs[None, :]
        down_w = tl.load(down_ptr + down_w_offs, mask=hidden_mask[:, None] & inter_mask[None, :], other=0.0)
        
        # Compute dot product
        out_val = tl.sum(down_w * inter[None, :], axis=1)
        acc += out_val * inter_mask
    
    # Apply weight
    acc = acc * weight
    
    # Store output
    out_offs = pid * hidden_size + hidden_offs
    tl.store(output_ptr + out_offs, acc, mask=hidden_mask)


class Model(nn.Module):
    """
    MoE Expert with Gated GEMM (SiLU-gated FFN).
    Optimized with Triton kernels.
    """

    def __init__(
        self,
        hidden_size: int,
        intermediate_size: int,
        num_experts: int,
    ):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.gate_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.up_proj = nn.Parameter(
            torch.randn(num_experts, intermediate_size, hidden_size) * 0.02
        )
        self.down_proj = nn.Parameter(
            torch.randn(num_experts, hidden_size, intermediate_size) * 0.02
        )
        torch.random.set_rng_state(rng_state)

    def forward(
        self,
        x: torch.Tensor,
        expert_indices: torch.Tensor,
        expert_weights: torch.Tensor,
    ) -> torch.Tensor:
        batch, seq_len, hidden_size = x.shape
        top_k = expert_indices.shape[-1]
        num_tokens = batch * seq_len
        
        x_flat = x.view(num_tokens, hidden_size)
        indices_flat = expert_indices.view(num_tokens * top_k)
        weights_flat = expert_weights.view(num_tokens * top_k)
        
        # Create token indices
        token_ids = torch.arange(num_tokens, device=x.device)
        token_ids = token_ids.unsqueeze(1).expand(-1, top_k).reshape(-1)
        
        # Sort by expert
        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]
        
        # Get expert boundaries
        expert_counts = torch.bincount(sorted_expert_idx, minlength=self.num_experts)
        expert_offsets = torch.cat([
            torch.zeros(1, dtype=torch.long, device=x.device),
            expert_counts.cumsum(0)
        ])
        
        # Gather sorted inputs
        sorted_x = x_flat[sorted_token_ids]
        
        num_sorted = num_tokens * top_k
        
        # Process each expert and accumulate results
        # Output is accumulated per token (before the sort)
        final_output = torch.zeros(num_tokens, hidden_size, device=x.device, dtype=x.dtype)
        
        BLOCK_HIDDEN = 256
        BLOCK_INTER = 256
        
        for e in range(self.num_experts):
            start = int(expert_offsets[e].item())
            end = int(expert_offsets[e + 1].item())
            if start == end:
                continue
            
            expert_x = sorted_x[start:end].contiguous()  # (n_tokens, H)
            n_tokens = end - start
            
            # Allocate intermediate
            intermediate = torch.empty(n_tokens, self.intermediate_size, device=x.device, dtype=x.dtype)
            
            # Gate-up kernel
            grid_gate_up = (n_tokens, triton.cdiv(self.intermediate_size, BLOCK_INTER))
            moe_gate_up_kernel[grid_gate_up](
                expert_x,
                self.gate_proj[e].contiguous(),
                self.up_proj[e].contiguous(),
                intermediate,
                n_tokens, self.hidden_size, self.intermediate_size,
                BLOCK_HIDDEN, BLOCK_INTER,
            )
            
            # Down kernel for this expert
            expert_output = torch.empty(n_tokens, self.hidden_size, device=x.device, dtype=x.dtype)
            grid_down = (n_tokens, triton.cdiv(self.hidden_size, BLOCK_HIDDEN))
            moe_down_kernel[grid_down](
                intermediate,
                self.down_proj[e].contiguous(),
                sorted_weights[start:end].contiguous(),
                expert_output,
                n_tokens, self.hidden_size, self.intermediate_size,
                BLOCK_INTER, BLOCK_HIDDEN,
            )
            
            # Get original token indices for this expert's tokens
            orig_token_ids = sorted_token_ids[start:end]
            
            # Scatter-add to final output
            final_output.index_add_(0, orig_token_ids, expert_output)
        
        return final_output.view(batch, seq_len, self.hidden_size)


# Mixtral-style configuration
batch_size = 4
seq_len = 2048
hidden_size = 4096
intermediate_size = 14336
num_experts = 8
top_k = 2


def get_inputs():
    x = torch.randn(batch_size, seq_len, hidden_size)
    
    expert_indices = torch.stack([
        torch.randperm(num_experts)[:top_k]
        for _ in range(batch_size * seq_len)
    ]).view(batch_size, seq_len, top_k)
    
    expert_weights = F.softmax(torch.randn(batch_size, seq_len, top_k), dim=-1)
    
    return [x, expert_indices, expert_weights]


def get_init_inputs():
    return [hidden_size, intermediate_size, num_experts]
