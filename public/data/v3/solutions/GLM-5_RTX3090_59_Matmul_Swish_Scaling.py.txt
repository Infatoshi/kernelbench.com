import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <mma.h>
#include <cmath>

using namespace nvcuda::wmma;

// Tile dimensions for tensor cores
constexpr int WMMA_M = 16;
constexpr int WMMA_N = 16;
constexpr int WMMA_K = 16;

// Block tile dimensions
constexpr int BLOCK_M = 64;
constexpr int BLOCK_N = 64;
constexpr int BLOCK_K = 32;

// Each block has 4x4 = 16 warps, each computing 16x16 output tile
constexpr int NUM_WARPS = 16;
constexpr int THREADS_PER_BLOCK = NUM_WARPS * 32;  // 512 threads

__device__ __forceinline__ float fast_sigmoid_float(float x) {
    return 1.0f / (1.0f + expf(-x));
}

// Fused GEMM + Swish + Scale kernel using WMMA
__global__ void __launch_bounds__(512) fused_matmul_swish_scale_kernel(
    const half* __restrict__ input,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    half* __restrict__ output,
    int M, int N, int K,
    float scaling_factor
) {
    // Shared memory for input and weight tiles
    __shared__ half smem_input[BLOCK_K][BLOCK_M + 8];
    __shared__ half smem_weight[BLOCK_K][BLOCK_N + 8];
    __shared__ float smem_acc[BLOCK_M][BLOCK_N + 8];  // Float for accumulator storage
    
    // Warp and lane identification
    int warp_id = threadIdx.x / 32;
    int lane_id = threadIdx.x % 32;
    
    // Determine which 16x16 output tile this warp computes
    int warp_m = warp_id / 4;  // 0-3
    int warp_n = warp_id % 4;  // 0-3
    
    int global_m = blockIdx.y * BLOCK_M + warp_m * WMMA_M;
    int global_n = blockIdx.x * BLOCK_N + warp_n * WMMA_N;
    
    // Fragment registers for WMMA
    fragment<matrix_a, WMMA_M, WMMA_N, WMMA_K, half, row_major> frag_a;
    fragment<matrix_b, WMMA_M, WMMA_N, WMMA_K, half, col_major> frag_b;
    fragment<accumulator, WMMA_M, WMMA_N, WMMA_K, float> frag_c;
    
    // Initialize accumulator to zero
    fill_fragment(frag_c, 0.0f);
    
    // Loop over K dimension
    int num_k_tiles = (K + BLOCK_K - 1) / BLOCK_K;
    int tid = threadIdx.x;
    
    for (int k_tile = 0; k_tile < num_k_tiles; k_tile++) {
        int global_k = k_tile * BLOCK_K;
        
        // Cooperative loading of input and weight tiles into shared memory
        for (int i = tid; i < BLOCK_K * BLOCK_M; i += THREADS_PER_BLOCK) {
            int k_local = i / BLOCK_M;
            int m_local = i % BLOCK_M;
            int k_global = global_k + k_local;
            int m_global = blockIdx.y * BLOCK_M + m_local;
            
            half val = __float2half(0.0f);
            if (k_global < K && m_global < M) {
                val = input[m_global * K + k_global];
            }
            smem_input[k_local][m_local] = val;
        }
        
        for (int i = tid; i < BLOCK_K * BLOCK_N; i += THREADS_PER_BLOCK) {
            int k_local = i / BLOCK_N;
            int n_local = i % BLOCK_N;
            int k_global = global_k + k_local;
            int n_global = blockIdx.x * BLOCK_N + n_local;
            
            half val = __float2half(0.0f);
            if (k_global < K && n_global < N) {
                val = weight[n_global * K + k_global];
            }
            smem_weight[k_local][n_local] = val;
        }
        
        __syncthreads();
        
        // Compute WMMA operations for this tile
        for (int k = 0; k < BLOCK_K; k += WMMA_K) {
            // Load fragments from shared memory
            load_matrix_sync(frag_a, &smem_input[k][warp_m * WMMA_M], BLOCK_M + 8);
            load_matrix_sync(frag_b, &smem_weight[k][warp_n * WMMA_N], BLOCK_N + 8);
            
            // Matrix multiply-accumulate
            mma_sync(frag_c, frag_a, frag_b, frag_c);
        }
        
        __syncthreads();
    }
    
    // Store accumulator to shared memory using WMMA store (float pointer)
    store_matrix_sync(&smem_acc[warp_m * WMMA_M][warp_n * WMMA_N], frag_c, BLOCK_N + 8, mem_row_major);
    
    __syncthreads();
    
    // Now each thread processes elements from shared memory
    // and writes final output with swish and scaling
    for (int i = tid; i < BLOCK_M * BLOCK_N; i += THREADS_PER_BLOCK) {
        int m_local = i / BLOCK_N;
        int n_local = i % BLOCK_N;
        
        int m_global = blockIdx.y * BLOCK_M + m_local;
        int n_global = blockIdx.x * BLOCK_N + n_local;
        
        if (m_global < M && n_global < N) {
            float val = smem_acc[m_local][n_local];
            
            // Add bias
            if (bias != nullptr) {
                val += __half2float(bias[n_global]);
            }
            
            // Swish: x * sigmoid(x)
            float sigmoid_val = fast_sigmoid_float(val);
            float swish_val = val * sigmoid_val;
            
            // Scale
            swish_val *= scaling_factor;
            
            output[m_global * N + n_global] = __float2half(swish_val);
        }
    }
}

torch::Tensor fused_matmul_swish_scale(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double scaling_factor
) {
    int M = input.size(0);  // batch_size
    int K = input.size(1);  // in_features
    int N = weight.size(0); // out_features
    
    auto output = torch::empty({M, N}, input.options().dtype(torch::kHalf));
    
    half* input_ptr = reinterpret_cast<half*>(input.data_ptr<at::Half>());
    half* weight_ptr = reinterpret_cast<half*>(weight.data_ptr<at::Half>());
    half* bias_ptr = bias.defined() ? reinterpret_cast<half*>(bias.data_ptr<at::Half>()) : nullptr;
    half* output_ptr = reinterpret_cast<half*>(output.data_ptr<at::Half>());
    
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);
    dim3 block(THREADS_PER_BLOCK);
    
    fused_matmul_swish_scale_kernel<<<grid, block>>>(
        input_ptr, weight_ptr, bias_ptr, output_ptr,
        M, N, K, static_cast<float>(scaling_factor)
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_matmul_swish_scale(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, double scaling_factor);
'''

ext = load_inline(
    name='fused_matmul_swish_scale_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_swish_scale'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
    verbose=False,
)

import math

class Model(nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.scaling_factor = scaling_factor
        
        # Create weight and bias tensors (matching nn.Linear initialization)
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize weights (kaiming uniform like nn.Linear)
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / math.sqrt(fan_in) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)
    
    def forward(self, x):
        # Convert to FP16 for tensor core utilization
        x_fp16 = x.half()
        w_fp16 = self.weight.half()
        b_fp16 = self.bias.half()
        
        # Fused kernel
        out = ext.fused_matmul_swish_scale(x_fp16, w_fp16, b_fp16, self.scaling_factor)
        
        return out.float()

batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, scaling_factor]
