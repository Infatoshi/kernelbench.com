import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_matmul_maxpool_sum_scale_kernel(
    x_ptr, w_ptr, bias_ptr,
    out_ptr,
    batch_size,
    in_features,
    out_features,
    scale_factor,
    BLOCK_IN: tl.constexpr,
):
    """
    Fused kernel: GEMM + MaxPool(kernel_size=2) + Sum + Scale
    Each program handles one batch element.
    """
    pid = tl.program_id(0)
    
    # Accumulator for the final sum (after maxpool)
    acc = 0.0
    
    # Number of pairs for max pooling (kernel_size=2, stride=2)
    num_pairs = out_features // 2
    
    # Process each pair of output columns
    for pair_idx in range(num_pairs):
        col_a = pair_idx * 2
        col_b = pair_idx * 2 + 1
        
        # Compute dot products for both columns in this pair
        dot_a = 0.0
        dot_b = 0.0
        
        # Stream through input features in blocks
        for in_start in range(0, in_features, BLOCK_IN):
            in_offs = in_start + tl.arange(0, BLOCK_IN)
            mask = in_offs < in_features
            
            # Load x[pid, :] - reuse for both columns
            x_vals = tl.load(x_ptr + pid * in_features + in_offs, 
                            mask=mask, other=0.0).to(tl.float32)
            
            # Load w[col_a, :] and w[col_b, :]
            w_a_vals = tl.load(w_ptr + col_a * in_features + in_offs, 
                              mask=mask, other=0.0).to(tl.float32)
            w_b_vals = tl.load(w_ptr + col_b * in_features + in_offs, 
                              mask=mask, other=0.0).to(tl.float32)
            
            # Accumulate dot products
            dot_a += tl.sum(x_vals * w_a_vals)
            dot_b += tl.sum(x_vals * w_b_vals)
        
        # Add bias to each column
        bias_a = tl.load(bias_ptr + col_a).to(tl.float32)
        bias_b = tl.load(bias_ptr + col_b).to(tl.float32)
        dot_a += bias_a
        dot_b += bias_b
        
        # Max pooling: max of the pair
        pooled = tl.maximum(dot_a, dot_b)
        
        # Accumulate for sum
        acc += pooled
    
    # Apply scale factor and store
    result = acc * scale_factor
    tl.store(out_ptr + pid, result)


class Model(nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.kernel_size = kernel_size
        self.scale_factor = scale_factor
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        batch_size = x.shape[0]
        in_features = self.in_features
        out_features = self.out_features
        
        # Output: one scalar per batch element
        output = torch.empty(batch_size, dtype=x.dtype, device=x.device)
        
        # Choose block size based on input features
        BLOCK_IN = 256 if in_features >= 256 else 128
        
        # Launch kernel: one program per batch element
        fused_matmul_maxpool_sum_scale_kernel[(batch_size,)](
            x,
            self.matmul.weight,
            self.matmul.bias,
            output,
            batch_size,
            in_features,
            out_features,
            float(self.scale_factor),
            BLOCK_IN=BLOCK_IN,
        )
        
        return output


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    in_features = 4096
    out_features = 4096
    kernel_size = 2
    scale_factor = 0.5
    return [in_features, out_features, kernel_size, scale_factor]
