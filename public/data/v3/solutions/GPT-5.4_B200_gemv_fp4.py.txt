import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor fp4_gemv(torch::Tensor x_q, torch::Tensor x_scale, torch::Tensor w_q, torch::Tensor w_scale);
'''

cuda_source = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void fp4_gemv_kernel(
    const int8_t* __restrict__ x,
    const int8_t* __restrict__ w,
    half* __restrict__ y,
    int B,
    int K,
    int N,
    float total_scale)
{
    int row = blockIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    if (row >= B || col >= N) return;

    const int8_t* x_row = x + row * K;
    const int8_t* w_row = w + col * K;

    int acc = 0;
    int k = 0;

    const int4* x4 = reinterpret_cast<const int4*>(x_row);
    const int4* w4 = reinterpret_cast<const int4*>(w_row);
    int vec4 = K / 16;

    #pragma unroll 4
    for (int i = 0; i < vec4; ++i) {
        int4 xv = x4[i];
        int4 wv = w4[i];
        const int8_t* xa = reinterpret_cast<const int8_t*>(&xv);
        const int8_t* wa = reinterpret_cast<const int8_t*>(&wv);
        #pragma unroll
        for (int j = 0; j < 16; ++j) {
            acc += int(xa[j]) * int(wa[j]);
        }
    }
    k = vec4 * 16;
    for (; k < K; ++k) {
        acc += int(x_row[k]) * int(w_row[k]);
    }

    y[row * N + col] = __float2half_rn(float(acc) * total_scale);
}

torch::Tensor fp4_gemv(torch::Tensor x_q, torch::Tensor x_scale, torch::Tensor w_q, torch::Tensor w_scale) {
    TORCH_CHECK(x_q.is_cuda(), "x_q must be CUDA");
    TORCH_CHECK(w_q.is_cuda(), "w_q must be CUDA");
    TORCH_CHECK(x_q.dtype() == torch::kInt8, "x_q must be int8");
    TORCH_CHECK(w_q.dtype() == torch::kInt8, "w_q must be int8");
    TORCH_CHECK(x_q.dim() == 2, "x_q must be 2D");
    TORCH_CHECK(w_q.dim() == 2, "w_q must be 2D");
    TORCH_CHECK(x_q.size(1) == w_q.size(1), "K dimension mismatch");

    auto x = x_q.contiguous();
    auto w = w_q.contiguous();
    int B = x.size(0);
    int K = x.size(1);
    int N = w.size(0);

    float xs = x_scale.item<float>();
    float ws = w_scale.item<float>();
    float total_scale = xs * ws;

    auto y = torch::empty({B, N}, x.options().dtype(torch::kFloat16));

    dim3 block(256);
    dim3 grid((N + block.x - 1) / block.x, B);
    auto stream = at::cuda::getDefaultCUDAStream();
    fp4_gemv_kernel<<<grid, block, 0, stream>>>(
        reinterpret_cast<const int8_t*>(x.data_ptr<int8_t>()),
        reinterpret_cast<const int8_t*>(w.data_ptr<int8_t>()),
        reinterpret_cast<half*>(y.data_ptr<at::Half>()),
        B, K, N, total_scale);
    return y;
}
'''

ext = load_inline(
    name='fp4_gemv_ext_b200_v1',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fp4_gemv'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-lineinfo'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer('weight_q', torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer('weight_scale', torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.fp4_gemv(x_q, x_scale, self.weight_q, self.weight_scale)


OP_TYPE = 'gemv'
SUPPORTED_PRECISIONS = ['fp4']
HARDWARE_REQUIRED = ['B200']
SPECIALIZED_LEVEL = 1


def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.08, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
