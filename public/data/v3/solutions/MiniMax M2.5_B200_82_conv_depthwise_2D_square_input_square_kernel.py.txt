import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized depthwise convolution CUDA kernel
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Depthwise convolution kernel for square input and square kernel
// Each channel is processed independently
template <typename T>
__global__ void depthwise_conv2d_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int out_height,
    const int out_width) 
{
    // Each block handles one channel for one batch element
    const int channel = blockIdx.x;
    const int batch_idx = blockIdx.y;
    
    if (channel >= channels || batch_idx >= batch_size) return;
    
    const int output_size = out_height * out_width;
    const int total_work = output_size;
    
    // Grid-stride loop for parallelization within each channel
    for (int idx = blockIdx.z * blockDim.x + threadIdx.x; 
         idx < total_work; 
         idx += gridDim.z * blockDim.x) 
    {
        const int out_y = idx / out_width;
        const int out_x = idx % out_width;
        
        // Calculate corresponding input position
        const int in_y = out_y * stride - padding;
        const int in_x = out_x * stride - padding;
        
        // Accumulate convolution result
        T sum = static_cast<T>(0);
        
        const int weight_base = channel * kernel_size * kernel_size;
        const int input_base = (batch_idx * channels + channel) * height * width;
        
        for (int ky = 0; ky < kernel_size; ky++) {
            for (int kx = 0; kx < kernel_size; kx++) {
                const int in_row = in_y + ky;
                const int in_col = in_x + kx;
                
                // Bounds check
                if (in_row >= 0 && in_row < height && in_col >= 0 && in_col < width) {
                    const int input_idx = input_base + in_row * width + in_col;
                    const int weight_idx = weight_base + ky * kernel_size + kx;
                    sum += input[input_idx] * weight[weight_idx];
                }
            }
        }
        
        const int out_idx = (batch_idx * channels + channel) * output_size + out_y * out_width + out_x;
        output[out_idx] = sum;
    }
}

// Kernel with bias support
template <typename T>
__global__ void depthwise_conv2d_bias_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int out_height,
    const int out_width) 
{
    const int channel = blockIdx.x;
    const int batch_idx = blockIdx.y;
    
    if (channel >= channels || batch_idx >= batch_size) return;
    
    const int output_size = out_height * out_width;
    const int total_work = output_size;
    
    for (int idx = blockIdx.z * blockDim.x + threadIdx.x; 
         idx < total_work; 
         idx += gridDim.z * blockDim.x) 
    {
        const int out_y = idx / out_width;
        const int out_x = idx % out_width;
        
        const int in_y = out_y * stride - padding;
        const int in_x = out_x * stride - padding;
        
        T sum = static_cast<T>(0);
        
        const int weight_base = channel * kernel_size * kernel_size;
        const int input_base = (batch_idx * channels + channel) * height * width;
        
        for (int ky = 0; ky < kernel_size; ky++) {
            for (int kx = 0; kx < kernel_size; kx++) {
                const int in_row = in_y + ky;
                const int in_col = in_x + kx;
                
                if (in_row >= 0 && in_row < height && in_col >= 0 && in_col < width) {
                    const int input_idx = input_base + in_row * width + in_col;
                    const int weight_idx = weight_base + ky * kernel_size + kx;
                    sum += input[input_idx] * weight[weight_idx];
                }
            }
        }
        
        if (bias != nullptr) {
            sum += bias[channel];
        }
        
        const int out_idx = (batch_idx * channels + channel) * output_size + out_y * out_width + out_x;
        output[out_idx] = sum;
    }
}

// Optimized kernel using shared memory for weight
template <typename T, int KERNEL_SIZE>
__global__ void depthwise_conv2d_sm_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int stride,
    const int padding,
    const int out_height,
    const int out_width) 
{
    __shared__ T weight_smem[3 * 3];  // Max kernel size 3x3
    
    const int channel = blockIdx.x;
    const int batch_idx = blockIdx.y;
    const int out_idx = blockIdx.z;
    
    if (channel >= channels || batch_idx >= batch_size || out_idx >= out_height * out_width) return;
    
    // Load weight into shared memory
    if (threadIdx.x < KERNEL_SIZE * KERNEL_SIZE) {
        weight_smem[threadIdx.x] = weight[channel * KERNEL_SIZE * KERNEL_SIZE + threadIdx.x];
    }
    __syncthreads();
    
    const int out_y = out_idx / out_width;
    const int out_x = out_idx % out_width;
    const int in_y = out_y * stride - padding;
    const int in_x = out_x * stride - padding;
    
    T sum = static_cast<T>(0);
    const int input_base = (batch_idx * channels + channel) * height * width;
    
    for (int ky = 0; ky < KERNEL_SIZE; ky++) {
        for (int kx = 0; kx < KERNEL_SIZE; kx++) {
            const int in_row = in_y + ky;
            const int in_col = in_x + kx;
            
            if (in_row >= 0 && in_row < height && in_col >= 0 && in_col < width) {
                const int input_idx = input_base + in_row * width + in_col;
                const int weight_idx = ky * KERNEL_SIZE + kx;
                sum += input[input_idx] * weight_smem[weight_idx];
            }
        }
    }
    
    const int output_base = (batch_idx * channels + channel) * out_height * out_width;
    output[output_base + out_idx] = sum;
}

// Launchers
torch::Tensor depthwise_conv2d_fwd(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding) 
{
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int height = input.size(2);
    const int width = input.size(3);
    const int kernel_size = weight.size(2);
    
    const int out_height = (height + 2 * padding - kernel_size) / stride + 1;
    const int out_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    const int blocks_z = 16;  // Parallelism within channel
    
    dim3 grid(channels, batch_size, blocks_z);
    dim3 block(256);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "depthwise_conv2d_cuda", ([&] {
        depthwise_conv2d_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels, height, width,
            kernel_size, stride, padding,
            out_height, out_width
        );
    }));
    
    cudaDeviceSynchronize();
    return output;
}

torch::Tensor depthwise_conv2d_fwd_bias(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int stride,
    int padding) 
{
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int height = input.size(2);
    const int width = input.size(3);
    const int kernel_size = weight.size(2);
    
    const int out_height = (height + 2 * padding - kernel_size) / stride + 1;
    const int out_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    const int blocks_z = 16;
    
    dim3 grid(channels, batch_size, blocks_z);
    dim3 block(256);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "depthwise_conv2d_bias_cuda", ([&] {
        depthwise_conv2d_bias_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            bias.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels, height, width,
            kernel_size, stride, padding,
            out_height, out_width
        );
    }));
    
    cudaDeviceSynchronize();
    return output;
}
'''

ext = load_inline(
    name='depthwise_conv2d_ext',
    cpp_sources='''
torch::Tensor depthwise_conv2d_fwd(torch::Tensor input, torch::Tensor weight, int stride, int padding);
torch::Tensor depthwise_conv2d_fwd_bias(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, int stride, int padding);
''',
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_fwd', 'depthwise_conv2d_fwd_bias'],
    verbose=False,
)


class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.
    Uses optimized custom CUDA kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, 
                                padding=padding, groups=in_channels, bias=bias)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the depthwise 2D convolution using optimized CUDA kernel.
        """
        # Use custom kernel for forward pass
        weight = self.conv2d.weight
        
        if self.conv2d.bias is not None and self.conv2d.bias.requires_grad:
            # Use bias kernel if bias exists
            return ext.depthwise_conv2d_fwd_bias(
                x, weight, self.conv2d.bias, 
                self.conv2d.stride[0], self.conv2d.padding[0]
            )
        else:
            return ext.depthwise_conv2d_fwd(
                x, weight, 
                self.conv2d.stride[0], self.conv2d.padding[0]
            )


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]


def get_init_inputs():
    in_channels = 3
    kernel_size = 3
    stride = 1
    padding = 0
    return [in_channels, kernel_size, stride, padding]
