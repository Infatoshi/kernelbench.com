import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>

// Fast tanh approximation using rational function
__device__ __forceinline__ float fast_tanh(float x) {
    // Use cubic approximation for better accuracy
    float x2 = x * x;
    float a = x * (0.98584031065334f + x2 * (0.08196994917293f + x2 * -0.00146750520431f));
    float b = 1.0f + x2 * (0.10283297113876f + x2 * 0.00085154778885f);
    return a / b;
}

// Fused Conv2d + Tanh + Scale + Bias + MaxPool kernel
// Each thread block handles one output tile in the output feature map
// Each thread handles multiple output pixels (for different output channels)

template<typename T>
__global__ void fused_conv_tanh_scale_bias_maxpool_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int in_h, int in_w, int out_h, int out_w,
    int kernel_size, int pool_size, float scale,
    int pooled_h, int pooled_w) {
    
    // Each block handles a tile of output (after pooling)
    // Grid: (batch_size * out_channels, pooled_h_tiles, pooled_w_tiles)
    
    int bc = blockIdx.x;  // batch * channel combined
    int b = bc / out_channels;
    int oc = bc % out_channels;
    
    int pool_tile_y = blockIdx.y;
    int pool_tile_x = blockIdx.z;
    
    // Each thread handles one output pixel (after pooling)
    // pool_size = 2, so we need to compute 2x2 conv outputs and take max
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    // Output position in pooled space
    int pool_out_y = pool_tile_y * blockDim.y + ty;
    int pool_out_x = pool_tile_x * blockDim.x + tx;
    
    if (pool_out_y >= pooled_h || pool_out_x >= pooled_w) return;
    
    // Corresponding position in conv output space (before pooling)
    int conv_out_y_start = pool_out_y * pool_size;
    int conv_out_x_start = pool_out_x * pool_size;
    
    // Max over the pool window
    float max_val = -FLT_MAX;
    
    // Compute 2x2 pool window (unrolled for efficiency)
    #pragma unroll
    for (int py = 0; py < 2; py++) {
        int conv_y = conv_out_y_start + py;
        if (conv_y >= out_h) continue;
        
        #pragma unroll
        for (int px = 0; px < 2; px++) {
            int conv_x = conv_out_x_start + px;
            if (conv_x >= out_w) continue;
            
            // Compute convolution for this position
            float conv_val = 0.0f;
            
            // Input position for 3x3 kernel (padding=0, stride=1)
            int in_y_start = conv_y;  // No padding
            int in_x_start = conv_x;
            
            // 3x3 kernel - unrolled for efficiency
            #pragma unroll
            for (int ic = 0; ic < 3; ic++) {  // in_channels = 3
                #pragma unroll
                for (int ky = 0; ky < 3; ky++) {
                    int in_y = in_y_start + ky;
                    if (in_y >= in_h) continue;
                    
                    #pragma unroll
                    for (int kx = 0; kx < 3; kx++) {
                        int in_x = in_x_start + kx;
                        if (in_x >= in_w) continue;
                        
                        // Weight index: (oc, ic, ky, kx)
                        int weight_idx = oc * (3 * 3 * 3) + ic * (3 * 3) + ky * 3 + kx;
                        int input_idx = b * (in_channels * in_h * in_w) + ic * (in_h * in_w) + in_y * in_w + in_x;
                        
                        conv_val += (float)input[input_idx] * (float)weight[weight_idx];
                    }
                }
            }
            
            // Apply tanh, scale, bias
            conv_val = fast_tanh(conv_val);
            conv_val = conv_val * scale + (float)bias[oc];
            
            max_val = fmaxf(max_val, conv_val);
        }
    }
    
    // Write output
    int out_idx = b * (out_channels * pooled_h * pooled_w) + oc * (pooled_h * pooled_w) + pool_out_y * pooled_w + pool_out_x;
    output[out_idx] = (T)max_val;
}

// Optimized kernel using shared memory for weights
template<typename T>
__global__ void fused_conv_tanh_scale_bias_maxpool_shared_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ conv_bias,
    const T* __restrict__ extra_bias,
    T* __restrict__ output,
    int batch_size, int in_channels, int out_channels,
    int in_h, int in_w, int out_h, int out_w,
    int kernel_size, int pool_size, float scale,
    int pooled_h, int pooled_w) {
    
    // Shared memory for 3x3 kernel weights (27 floats per output channel)
    __shared__ float shared_weight[16 * 27];  // out_channels * (in_channels * 3 * 3)
    __shared__ float shared_conv_bias[16];
    __shared__ float shared_extra_bias[16];
    
    // Load weights and bias into shared memory
    int tid = threadIdx.y * blockDim.x + threadIdx.x;
    for (int i = tid; i < 16 * 27; i += blockDim.x * blockDim.y) {
        shared_weight[i] = (float)weight[i];
    }
    for (int i = tid; i < 16; i += blockDim.x * blockDim.y) {
        shared_conv_bias[i] = (float)conv_bias[i];
        shared_extra_bias[i] = (float)extra_bias[i];
    }
    __syncthreads();
    
    int bc = blockIdx.x;
    int b = bc / out_channels;
    int oc = bc % out_channels;
    
    int pool_tile_y = blockIdx.y;
    int pool_tile_x = blockIdx.z;
    
    int ty = threadIdx.y;
    int tx = threadIdx.x;
    
    int pool_out_y = pool_tile_y * blockDim.y + ty;
    int pool_out_x = pool_tile_x * blockDim.x + tx;
    
    if (pool_out_y >= pooled_h || pool_out_x >= pooled_w) return;
    
    int conv_out_y_start = pool_out_y * pool_size;
    int conv_out_x_start = pool_out_x * pool_size;
    
    float max_val = -FLT_MAX;
    
    // Weight offset for this output channel
    const float* w_ptr = &shared_weight[oc * 27];
    float bias_val = shared_bias[oc];
    
    #pragma unroll
    for (int py = 0; py < 2; py++) {
        int conv_y = conv_out_y_start + py;
        if (conv_y >= out_h) continue;
        
        #pragma unroll
        for (int px = 0; px < 2; px++) {
            int conv_x = conv_out_x_start + px;
            if (conv_x >= out_w) continue;
            
            float conv_val = 0.0f;
            
            int in_y_start = conv_y;
            int in_x_start = conv_x;
            
            // Unrolled 3x3 convolution
            #pragma unroll
            for (int ic = 0; ic < 3; ic++) {
                #pragma unroll
                for (int ky = 0; ky < 3; ky++) {
                    int in_y = in_y_start + ky;
                    if (in_y >= in_h) continue;
                    
                    #pragma unroll
                    for (int kx = 0; kx < 3; kx++) {
                        int in_x = in_x_start + kx;
                        if (in_x >= in_w) continue;
                        
                        int weight_idx = ic * 9 + ky * 3 + kx;
                        int input_idx = b * (in_channels * in_h * in_w) + ic * (in_h * in_w) + in_y * in_w + in_x;
                        
                        conv_val += (float)input[input_idx] * w_ptr[weight_idx];
                    }
                }
            }
            
            conv_val = fast_tanh(conv_val);
            conv_val = conv_val * scale + bias_val;
            
            max_val = fmaxf(max_val, conv_val);
        }
    }
    
    int out_idx = b * (out_channels * pooled_h * pooled_w) + oc * (pooled_h * pooled_w) + pool_out_y * pooled_w + pool_out_x;
    output[out_idx] = (T)max_val;
}

torch::Tensor fused_conv_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor conv_bias,
    torch::Tensor extra_bias,
    float scale,
    int pool_size) {
    
    int batch_size = input.size(0);
    int in_channels = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    
    int out_channels = weight.size(0);
    int kernel_size = weight.size(2);
    
    int out_h = in_h - kernel_size + 1;
    int out_w = in_w - kernel_size + 1;
    
    int pooled_h = (out_h + pool_size - 1) / pool_size;
    int pooled_w = (out_w + pool_size - 1) / pool_size;
    
    auto output = torch::empty({batch_size, out_channels, pooled_h, pooled_w}, input.options());
    
    dim3 block(16, 16);
    dim3 grid(batch_size * out_channels,
              (pooled_h + 15) / 16,
              (pooled_w + 15) / 16);
    
    if (input.dtype() == torch::kFloat16) {
        fused_conv_tanh_scale_bias_maxpool_shared_kernel<at::Half><<<grid, block>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            conv_bias.data_ptr<at::Half>(),
            extra_bias.data_ptr<at::Half>(),
            output.data_ptr<at::Half>(),
            batch_size, in_channels, out_channels,
            in_h, in_w, out_h, out_w,
            kernel_size, pool_size, scale,
            pooled_h, pooled_w);
    } else {
        fused_conv_tanh_scale_bias_maxpool_shared_kernel<float><<<grid, block>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            conv_bias.data_ptr<float>(),
            extra_bias.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size, in_channels, out_channels,
            in_h, in_w, out_h, out_w,
            kernel_size, pool_size, scale,
            pooled_h, pooled_w);
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_conv_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    float scale,
    int pool_size);
'''

ext = load_inline(
    name='fused_conv_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math'],
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.max_pool = nn.MaxPool2d(pool_kernel_size)
        
        # Store for kernel
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Use our fused kernel
        return ext.fused_conv_forward(
            x,
            self.conv.weight,
            self.bias.squeeze(),  # Remove extra dims
            self.scaling_factor,
            self.pool_kernel_size
        )

# Constants
batch_size = 16
in_channels = 3
out_channels = 16
height, width = 256, 256
kernel_size = 3
scaling_factor = 2.0
bias_shape = (out_channels, 1, 1)
pool_kernel_size = 2

def get_inputs():
    return [torch.randn(batch_size, in_channels, height, width)]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size]
