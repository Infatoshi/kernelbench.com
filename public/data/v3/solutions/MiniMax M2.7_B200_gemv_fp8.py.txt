"""
Optimized GEMV for B200 using FP8-style int8 tensor operations.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = r"""
torch::Tensor gemv_fp8_kernel(
    torch::Tensor x_q,
    torch::Tensor w_q,
    torch::Tensor x_scale,
    torch::Tensor w_scale);
"""

cuda_kernel = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// B200 optimized GEMV kernel
// Uses block tiling to improve L2 cache hit rate
// Each block: handles 1 row of M, 128 columns of N
// Thread arrangement: 128 threads per block, each computes 4 outputs

__global__ void gemv_fp8_kernel_impl(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ w_q,
    const float* __restrict__ x_scale,
    const float* __restrict__ w_scale,
    __half* __restrict__ output,
    int M, int N, int K) {
    
    // Scale factor (broadcast from device memory)
    float scale = *x_scale * *w_scale;
    
    // Grid: M rows x (N+127)/128 column blocks
    int row = blockIdx.x;
    int col_block = blockIdx.y * 128;
    int tid = threadIdx.x;
    
    if (row >= M) return;
    if (col_block >= N) return;
    
    // Each thread computes 4 output values
    // This gives us 128*4 = 512 outputs per block
    
    int n0 = col_block + tid;
    int n1 = col_block + tid + 32;
    int n2 = col_block + tid + 64;
    int n3 = col_block + tid + 96;
    
    // Accumulators
    int acc0 = 0, acc1 = 0, acc2 = 0, acc3 = 0;
    
    // Pointer to x row
    const int8_t* x_row = x_q + row * K;
    
    // Process all K elements
    // Inner loop unrolled by compiler
    for (int k = 0; k < K; k++) {
        int8_t xk = x_row[k];
        
        // Bounds-checked loads
        int8_t w0 = (n0 < N) ? w_q[n0 * K + k] : 0;
        int8_t w1 = (n1 < N) ? w_q[n1 * K + k] : 0;
        int8_t w2 = (n2 < N) ? w_q[n2 * K + k] : 0;
        int8_t w3 = (n3 < N) ? w_q[n3 * K + k] : 0;
        
        acc0 += (int)xk * (int)w0;
        acc1 += (int)xk * (int)w1;
        acc2 += (int)xk * (int)w2;
        acc3 += (int)xk * (int)w3;
    }
    
    // Store results with scaling
    if (n0 < N) output[row * N + n0] = __float2half_rn((float)acc0 * scale);
    if (n1 < N) output[row * N + n1] = __float2half_rn((float)acc1 * scale);
    if (n2 < N) output[row * N + n2] = __float2half_rn((float)acc2 * scale);
    if (n3 < N) output[row * N + n3] = __float2half_rn((float)acc3 * scale);
}

torch::Tensor gemv_fp8_kernel(
    torch::Tensor x_q,
    torch::Tensor w_q,
    torch::Tensor x_scale,
    torch::Tensor w_scale) {
    
    int M = x_q.size(0);
    int K = x_q.size(1);
    int N = w_q.size(0);
    
    auto output = torch::empty({M, N}, x_q.options().dtype(torch::kFloat16));
    
    dim3 grid(M, (N + 127) / 128);
    gemv_fp8_kernel_impl<<<grid, 128>>>(
        x_q.data_ptr<int8_t>(),
        w_q.data_ptr<int8_t>(),
        x_scale.data_ptr<float>(),
        w_scale.data_ptr<float>(),
        reinterpret_cast<__half*>(output.data_ptr<torch::Half>()),
        M, N, K
    );
    
    return output;
}
"""

ext = load_inline(
    name='gemv_fp8_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_kernel,
    functions=['gemv_fp8_kernel'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-127, 127, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.02, dtype=torch.float32))
        self.in_features = in_features
        self.out_features = out_features
        
    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        return ext.gemv_fp8_kernel(
            x_q, self.weight_q, x_scale, self.weight_scale
        )


def get_inputs():
    x_q = torch.randint(-127, 127, (32, 4096), dtype=torch.int8)
    x_scale = torch.tensor(0.02, dtype=torch.float32)
    return [x_q, x_scale]


def get_init_inputs():
    return []
