import torch
import torch.nn as nn

class Model(nn.Module):
    """
    A model that performs a matrix multiplication, scaling, and residual addition.
    Optimized to compute: y = matmul(x); return y * (scaling_factor + 1)
    which is equivalent to: y * scaling_factor + y
    """
    def __init__(self, in_features=None, out_features=None, scaling_factor=None):
        super(Model, self).__init__()
        # Default values for compatibility
        if in_features is None:
            in_features = 4096
        if out_features is None:
            out_features = 4096
        if scaling_factor is None:
            scaling_factor = 0.5
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        """
        Forward pass of the model.
        """
        y = self.matmul(x)
        # Fused operation: scaling and residual add
        # y * scaling_factor + y = y * (scaling_factor + 1)
        return y * (self.scaling_factor + 1.0)


def get_inputs():
    """
    Returns sample inputs for benchmarking.
    """
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features, device='cuda')]


def get_init_inputs():
    """
    Returns initialization arguments for Model.
    """
    in_features = 4096
    out_features = 4096
    scaling_factor = 0.5
    return [in_features, out_features, scaling_factor]
