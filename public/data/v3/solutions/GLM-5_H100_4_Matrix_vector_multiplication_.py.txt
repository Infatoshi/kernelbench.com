import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized matrix-vector multiplication kernel
// A: (M, K), B: (K, 1), C: (M, 1)
// Each thread block handles one row of A
// Threads cooperate using shared memory reduction

__global__ void matvec_kernel_fp32(const float* __restrict__ A, 
                                    const float* __restrict__ B, 
                                    float* __restrict__ C, 
                                    int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    const float* row_a = A + row * K;
    
    float sum = 0.0f;
    
    // Vectorized loads for better memory throughput
    int vec_size = 4;
    int k_vec = tid * vec_size;
    int K_vec = (K / vec_size) * vec_size;
    
    // Process with float4 vectorized loads
    for (int k = k_vec; k < K_vec; k += stride * vec_size) {
        float4 a_val = *reinterpret_cast<const float4*>(row_a + k);
        float4 b_val = *reinterpret_cast<const float4*>(B + k);
        sum += a_val.x * b_val.x;
        sum += a_val.y * b_val.y;
        sum += a_val.z * b_val.z;
        sum += a_val.w * b_val.w;
    }
    
    // Handle remaining elements
    for (int k = K_vec + tid; k < K; k += stride) {
        sum += row_a[k] * B[k];
    }
    
    // Shared memory for inter-warp reduction
    __shared__ float shared_sum[32];  // One per warp
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    // Warp-level reduction using shuffle instructions
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // First lane in each warp writes to shared memory
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    
    __syncthreads();
    
    // First warp combines results from all warps
    if (warp_id == 0) {
        sum = (lane_id < blockDim.x / 32) ? shared_sum[lane_id] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        
        if (lane_id == 0) {
            C[row] = sum;
        }
    }
}

__global__ void matvec_kernel_fp16(const at::Half* __restrict__ A, 
                                    const at::Half* __restrict__ B, 
                                    at::Half* __restrict__ C, 
                                    int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    const half* row_a = reinterpret_cast<const half*>(A) + row * K;
    const half* b_ptr = reinterpret_cast<const half*>(B);
    half* c_ptr = reinterpret_cast<half*>(C);
    
    float sum = 0.0f;
    
    // Process 8 half values at a time (128-bit loads)
    int vec_size = 8;
    int k_vec = tid * vec_size;
    int K_vec = (K / vec_size) * vec_size;
    
    for (int k = k_vec; k < K_vec; k += stride * vec_size) {
        const half* a_ptr = row_a + k;
        const half* b_loc = b_ptr + k;
        
        // Load 8 half values
        float a0 = __half2float(a_ptr[0]);
        float a1 = __half2float(a_ptr[1]);
        float a2 = __half2float(a_ptr[2]);
        float a3 = __half2float(a_ptr[3]);
        float a4 = __half2float(a_ptr[4]);
        float a5 = __half2float(a_ptr[5]);
        float a6 = __half2float(a_ptr[6]);
        float a7 = __half2float(a_ptr[7]);
        
        float b0 = __half2float(b_loc[0]);
        float b1 = __half2float(b_loc[1]);
        float b2 = __half2float(b_loc[2]);
        float b3 = __half2float(b_loc[3]);
        float b4 = __half2float(b_loc[4]);
        float b5 = __half2float(b_loc[5]);
        float b6 = __half2float(b_loc[6]);
        float b7 = __half2float(b_loc[7]);
        
        sum += a0 * b0 + a1 * b1 + a2 * b2 + a3 * b3;
        sum += a4 * b4 + a5 * b5 + a6 * b6 + a7 * b7;
    }
    
    // Handle remaining elements
    for (int k = K_vec + tid; k < K; k += stride) {
        sum += __half2float(row_a[k]) * __half2float(b_ptr[k]);
    }
    
    // Shared memory for inter-warp reduction
    __shared__ float shared_sum[32];
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    // Warp-level reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    
    __syncthreads();
    
    if (warp_id == 0) {
        sum = (lane_id < blockDim.x / 32) ? shared_sum[lane_id] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        
        if (lane_id == 0) {
            c_ptr[row] = __float2half(sum);
        }
    }
}

__global__ void matvec_kernel_bf16(const at::BFloat16* __restrict__ A, 
                                    const at::BFloat16* __restrict__ B, 
                                    at::BFloat16* __restrict__ C, 
                                    int M, int K) {
    int row = blockIdx.x;
    if (row >= M) return;
    
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    const __nv_bfloat16* row_a = reinterpret_cast<const __nv_bfloat16*>(A) + row * K;
    const __nv_bfloat16* b_ptr = reinterpret_cast<const __nv_bfloat16*>(B);
    __nv_bfloat16* c_ptr = reinterpret_cast<__nv_bfloat16*>(C);
    
    float sum = 0.0f;
    
    // Process 8 bf16 values at a time
    int vec_size = 8;
    int k_vec = tid * vec_size;
    int K_vec = (K / vec_size) * vec_size;
    
    for (int k = k_vec; k < K_vec; k += stride * vec_size) {
        const __nv_bfloat16* a_ptr = row_a + k;
        const __nv_bfloat16* b_loc = b_ptr + k;
        
        float a0 = __bfloat162float(a_ptr[0]);
        float a1 = __bfloat162float(a_ptr[1]);
        float a2 = __bfloat162float(a_ptr[2]);
        float a3 = __bfloat162float(a_ptr[3]);
        float a4 = __bfloat162float(a_ptr[4]);
        float a5 = __bfloat162float(a_ptr[5]);
        float a6 = __bfloat162float(a_ptr[6]);
        float a7 = __bfloat162float(a_ptr[7]);
        
        float b0 = __bfloat162float(b_loc[0]);
        float b1 = __bfloat162float(b_loc[1]);
        float b2 = __bfloat162float(b_loc[2]);
        float b3 = __bfloat162float(b_loc[3]);
        float b4 = __bfloat162float(b_loc[4]);
        float b5 = __bfloat162float(b_loc[5]);
        float b6 = __bfloat162float(b_loc[6]);
        float b7 = __bfloat162float(b_loc[7]);
        
        sum += a0 * b0 + a1 * b1 + a2 * b2 + a3 * b3;
        sum += a4 * b4 + a5 * b5 + a6 * b6 + a7 * b7;
    }
    
    // Handle remaining elements
    for (int k = K_vec + tid; k < K; k += stride) {
        sum += __bfloat162float(row_a[k]) * __bfloat162float(b_ptr[k]);
    }
    
    // Shared memory for inter-warp reduction
    __shared__ float shared_sum[32];
    
    int warp_id = tid / 32;
    int lane_id = tid % 32;
    
    // Warp-level reduction
    #pragma unroll
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    if (lane_id == 0) {
        shared_sum[warp_id] = sum;
    }
    
    __syncthreads();
    
    if (warp_id == 0) {
        sum = (lane_id < blockDim.x / 32) ? shared_sum[lane_id] : 0.0f;
        
        #pragma unroll
        for (int offset = 16; offset > 0; offset /= 2) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
        
        if (lane_id == 0) {
            c_ptr[row] = __float2bfloat16(sum);
        }
    }
}

torch::Tensor matvec_cuda(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    int threads = 256;  // 8 warps per block
    int blocks = M;
    
    if (A.scalar_type() == torch::kFloat32) {
        matvec_kernel_fp32<<<blocks, threads>>>(
            A.data_ptr<float>(),
            B.data_ptr<float>(),
            C.data_ptr<float>(),
            M, K
        );
    } else if (A.scalar_type() == torch::kFloat16) {
        matvec_kernel_fp16<<<blocks, threads>>>(
            A.data_ptr<at::Half>(),
            B.data_ptr<at::Half>(),
            C.data_ptr<at::Half>(),
            M, K
        );
    } else if (A.scalar_type() == torch::kBFloat16) {
        matvec_kernel_bf16<<<blocks, threads>>>(
            A.data_ptr<at::BFloat16>(),
            B.data_ptr<at::BFloat16>(),
            C.data_ptr<at::BFloat16>(),
            M, K
        );
    } else {
        throw std::runtime_error("Unsupported dtype");
    }
    
    return C;
}
'''

cpp_source = 'torch::Tensor matvec_cuda(torch::Tensor A, torch::Tensor B);'

ext = load_inline(
    name='matvec_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['matvec_cuda'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    """
    Simple model that performs matrix-vector multiplication (C = A * B).
    """
    def __init__(self):
        super(Model, self).__init__()
    
    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        """
        Performs matrix-vector multiplication.

        Args:
            A: Input matrix of shape (M, K).
            B: Input vector of shape (K, 1).

        Returns:
            Output vector of shape (M, 1).
        """
        return ext.matvec_cuda(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []  # No special initialization inputs needed
