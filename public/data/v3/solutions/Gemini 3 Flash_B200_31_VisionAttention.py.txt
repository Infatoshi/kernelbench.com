import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super(Model, self).__init__()
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads
        self.embed_dim = embed_dim
        
        # Match MultiheadAttention structure for state_dict compatibility
        self.attn = nn.MultiheadAttention(embed_dim, num_heads)
        self.norm = nn.LayerNorm(embed_dim)

    def forward(self, x):
        B, C, H, W = x.shape
        L = H * W
        # Input view: (B, C, H, W) -> (L, B, C)
        x_in = x.view(B, C, L).permute(2, 0, 1)
        
        # MultiheadAttention internally uses efficient kernels
        # But we want to bypass the full nn.MultiheadAttention.forward for speed if possible
        # Alternatively, we can use the bits and pieces:
        qkv = torch.nn.functional.linear(x_in, self.attn.in_proj_weight, self.attn.in_proj_bias)
        # qkv is (L, B, 3*C)
        qkv = qkv.view(L, B, 3, self.num_heads, self.head_dim)
        
        q = qkv[:, :, 0].transpose(0, 1) # (B, L, H, D)
        k = qkv[:, :, 1].transpose(0, 1)
        v = qkv[:, :, 2].transpose(0, 1)
        
        # Use fast fused attention
        # (B, H, L, D) is preferred for sdpa
        q = q.transpose(1, 2)
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)
        
        attn_output = torch.nn.functional.scaled_dot_product_attention(q, k, v)
        # (B, H, L, D) -> (B, L, H, D) -> (B, L, C) -> (L, B, C)
        attn_output = attn_output.transpose(1, 2).reshape(B, L, self.embed_dim).transpose(0, 1)
        
        # Final linear projection
        attn_output = torch.nn.functional.linear(attn_output, self.attn.out_proj.weight, self.attn.out_proj.bias)
        
        # Residual and LayerNorm
        out = self.norm(attn_output + x_in)
        
        # Back to (B, C, H, W)
        out = out.permute(1, 2, 0).view(B, C, H, W)
        return out

def get_inputs():
    return [torch.randn(2, 128, 128, 128, device='cuda')]

def get_init_inputs():
    return [128, 4]
