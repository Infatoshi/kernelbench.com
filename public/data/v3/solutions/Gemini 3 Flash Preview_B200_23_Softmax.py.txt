import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

template <typename T>
__device__ __forceinline__ float to_float(T val) {
    return (float)val;
}

template <typename T>
__global__ void softmax_kernel(const T* x, T* y, int batch_size, int dim) {
    int row = blockIdx.x;
    if (row >= batch_size) return;

    const T* row_x = x + row * dim;
    T* row_y = y + row * dim;

    float max_val = -1e38f;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        max_val = fmaxf(max_val, to_float(row_x[i]));
    }

    // Warp reduce max
    for (int offset = 16; offset > 0; offset /= 2) {
        max_val = fmaxf(max_val, __shfl_down_sync(0xffffffff, max_val, offset));
    }

    __shared__ float shared_max;
    __shared__ float warp_maxs[32]; 

    int lane = threadIdx.x % 32;
    int wid = threadIdx.x / 32;

    if (lane == 0) warp_maxs[wid] = max_val;
    __syncthreads();

    if (wid == 0) {
        float val = (lane < (blockDim.x / 32)) ? warp_maxs[lane] : -1e38f;
        for (int offset = 16; offset > 0; offset /= 2) {
            val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
        }
        if (lane == 0) shared_max = val;
    }
    __syncthreads();
    max_val = shared_max;

    float sum = 0.0f;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        float val = expf(to_float(row_x[i]) - max_val);
        sum += val;
    }

    // Warp reduce sum
    for (int offset = 16; offset > 0; offset /= 2) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }

    __shared__ float shared_sum;
    __shared__ float warp_sums[32];
    if (lane == 0) warp_sums[wid] = sum;
    __syncthreads();

    if (wid == 0) {
        float val = (lane < (blockDim.x / 32)) ? warp_sums[lane] : 0.0f;
        for (int offset = 16; offset > 0; offset /= 2) {
            val += __shfl_down_sync(0xffffffff, val, offset);
        }
        if (lane == 0) shared_sum = val;
    }
    __syncthreads();
    sum = shared_sum;

    float inv_sum = 1.0f / sum;
    for (int i = threadIdx.x; i < dim; i += blockDim.x) {
        row_y[i] = (T)(expf(to_float(row_x[i]) - max_val) * inv_sum);
    }
}

torch::Tensor softmax_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int batch_size = x.size(0);
    int dim = x.size(1);
    
    int threads = 256;
    if (dim <= 128) threads = 32;
    else if (dim <= 256) threads = 64;
    else if (dim <= 512) threads = 128;

    dim3 block_dim(threads);
    dim3 grid_dim(batch_size);

    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, x.scalar_type(), "softmax_forward", ([&] {
        softmax_kernel<scalar_t><<<grid_dim, block_dim>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            batch_size,
            dim
        );
    }));

    return y;
}
'''

cpp_source = r'''
torch::Tensor softmax_forward(torch::Tensor x);
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['softmax_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super(Model, self).__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.softmax_forward(x)

def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]

def get_init_inputs():
    return []
