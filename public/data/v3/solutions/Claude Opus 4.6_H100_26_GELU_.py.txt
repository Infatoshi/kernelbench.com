import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <math_constants.h>
#include <cmath>

// Vectorized GELU kernel for float using float4 (128-bit loads)
__global__ void gelu_f32_vec4_kernel(const float4* __restrict__ x, float4* __restrict__ y, int n4) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n4) {
        float4 v = x[idx];
        const float k = 0.7071067811865476f;
        v.x = v.x * 0.5f * (1.0f + erff(v.x * k));
        v.y = v.y * 0.5f * (1.0f + erff(v.y * k));
        v.z = v.z * 0.5f * (1.0f + erff(v.z * k));
        v.w = v.w * 0.5f * (1.0f + erff(v.w * k));
        y[idx] = v;
    }
}

// Scalar fallback for remaining elements
__global__ void gelu_f32_scalar_kernel(const float* __restrict__ x, float* __restrict__ y, int start, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x + start;
    if (idx < n) {
        float v = x[idx];
        y[idx] = v * 0.5f * (1.0f + erff(v * 0.7071067811865476f));
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    TORCH_CHECK(x.is_cuda(), "Input must be CUDA tensor");
    TORCH_CHECK(x.is_contiguous(), "Input must be contiguous");
    
    auto y = torch::empty_like(x);
    int n = x.numel();
    
    if (x.scalar_type() == torch::kFloat32) {
        int n4 = n / 4;
        int remainder = n % 4;
        
        if (n4 > 0) {
            int threads = 256;
            int blocks = (n4 + threads - 1) / threads;
            gelu_f32_vec4_kernel<<<blocks, threads>>>(
                reinterpret_cast<const float4*>(x.data_ptr<float>()),
                reinterpret_cast<float4*>(y.data_ptr<float>()),
                n4
            );
        }
        if (remainder > 0) {
            int start = n4 * 4;
            gelu_f32_scalar_kernel<<<1, remainder>>>(
                x.data_ptr<float>(), y.data_ptr<float>(), start, n
            );
        }
    } else {
        TORCH_CHECK(false, "Unsupported dtype");
    }
    
    return y;
}
'''

cpp_source = "torch::Tensor gelu_forward(torch::Tensor x);"

ext = load_inline(
    name='gelu_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_90'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_forward(x)


batch_size = 256
dim = 16384

def get_inputs():
    x = torch.randn(batch_size, dim)
    return [x]

def get_init_inputs():
    return []
