import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

__device__ inline float gelu_approx(float x) {
    const float k0 = 0.7978845608f; // sqrt(2/pi)
    const float k1 = 0.044715f;
    float x3 = x * x * x;
    float t = k0 * (x + k1 * x3);
    return 0.5f * x * (1.0f + tanhf(t));
}

__global__ void gelu_f32_kernel(const float* __restrict__ x, float* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        y[idx] = gelu_approx(x[idx]);
    }
}

__global__ void gelu_f16_kernel(const half* __restrict__ x, half* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = __half2float(x[idx]);
        float r = gelu_approx(v);
        y[idx] = __float2half(r);
    }
}

__global__ void gelu_bf16_kernel(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = __bfloat162float(x[idx]);
        float r = gelu_approx(v);
        y[idx] = __float2bfloat16(r);
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();
    int block = 256;
    int grid = (n + block - 1) / block;

    if (x.scalar_type() == at::kFloat) {
        gelu_f32_kernel<<<grid, block>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    } else if (x.scalar_type() == at::kHalf) {
        gelu_f16_kernel<<<grid, block>>>((half*)x.data_ptr<at::Half>(), (half*)y.data_ptr<at::Half>(), n);
    } else if (x.scalar_type() == at::kBFloat16) {
        gelu_bf16_kernel<<<grid, block>>>((nv_bfloat16*)x.data_ptr<at::BFloat16>(), (nv_bfloat16*)y.data_ptr<at::BFloat16>(), n);
    }
    return y;
}
'''

ext = load_inline(
    name='gelu_ext',
    cpp_sources='torch::Tensor gelu_forward(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.gelu_forward(x)


def get_inputs():
    return [torch.randn(256, 16384, device='cuda')]


def get_init_inputs():
    return []
