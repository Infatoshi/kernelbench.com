import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>

// Optimized Max Pooling 2D kernel
// Each 16x16 block handles a tile of outputs
template <typename T>
__global__ void maxpool2d_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const int batch_size,
    const int channels,
    const int in_height,
    const int in_width,
    const int out_height,
    const int out_width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation
) {
    // 2D block: 16x16 threads
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    // Output position this thread should compute
    const int out_x = blockIdx.x * 16 + tx;
    const int out_y = blockIdx.y * 16 + ty;
    
    if (out_x >= out_width || out_y >= out_height) return;
    
    // Grid dimensions: z = batch * channels
    const int batch_channel = blockIdx.z;
    const int b = batch_channel / channels;
    const int c = batch_channel % channels;
    
    // Calculate input window start
    const int ih_start = out_y * stride - padding;
    const int iw_start = out_x * stride - padding;
    
    // Find max with unrolling for performance
    float max_val = -FLT_MAX;
    
    for (int kh = 0; kh < kernel_size; kh++) {
        const int ih = ih_start + kh * dilation;
        if (ih < 0 || ih >= in_height) continue;
        
        for (int kw = 0; kw < kernel_size; kw++) {
            const int iw = iw_start + kw * dilation;
            if (iw < 0 || iw >= in_width) continue;
            
            const int in_idx = ((b * channels + c) * in_height + ih) * in_width + iw;
            max_val = fmaxf(max_val, static_cast<float>(input[in_idx]));
        }
    }
    
    const int out_idx = ((b * channels + c) * out_height + out_y) * out_width + out_x;
    output[out_idx] = static_cast<T>(max_val);
}

torch::Tensor maxpool2d_op(
    torch::Tensor input,
    const int kernel_size,
    const int stride,
    const int padding,
    const int dilation
) {
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int in_height = input.size(2);
    const int in_width = input.size(3);
    
    // Calculate output dimensions - correct formula for dilation
    // out = floor((in + 2*padding - dilation*(kernel_size-1) - 1) / stride) + 1
    const int out_height = (in_height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    const int out_width = (in_width + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, input.options());
    
    // Use grid of 16x16 thread blocks
    const int BLOCK_SIZE = 16;
    dim3 blocks((out_width + BLOCK_SIZE - 1) / BLOCK_SIZE, 
                (out_height + BLOCK_SIZE - 1) / BLOCK_SIZE,
                batch_size * channels);
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "maxpool2d_cuda", ([&] {
        maxpool2d_kernel<scalar_t><<<blocks, BLOCK_SIZE*BLOCK_SIZE>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels, in_height, in_width,
            out_height, out_width,
            kernel_size, stride, padding, dilation
        );
    }));
    
    cudaGetLastError();
    return output;
}
'''

ext = load_inline(
    name='maxpool2d_ext',
    cpp_sources='torch::Tensor maxpool2d_op(torch::Tensor input, const int kernel_size, const int stride, const int padding, const int dilation);',
    cuda_sources=cuda_source,
    functions=['maxpool2d_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, kernel_size: int, stride: int, padding: int, dilation: int):
        super().__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.maxpool2d_op(x, self.kernel_size, self.stride, self.padding, self.dilation)

def get_inputs():
    return [torch.randn(16, 32, 512, 512, device='cuda')]

def get_init_inputs():
    return [2, 2, 1, 3]
