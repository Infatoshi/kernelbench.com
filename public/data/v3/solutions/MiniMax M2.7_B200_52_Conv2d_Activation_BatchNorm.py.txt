import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized fused activation + BatchNorm for Blackwell
// Uses vectorized memory access and fast math intrinsics

// Specialized for float with loop unrolling hints
__global__ void fused_activation_bn_kernel_float4(
    const float* __restrict__ conv_out,
    float* __restrict__ output,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const float eps,
    int N, int C, int H, int W) {
    
    int total_elements = N * C * H * W;
    int num_threads = blockDim.x * gridDim.x;
    int base_idx = blockIdx.x * blockDim.x + threadIdx.x;
    
    // Process 4 elements per iteration for better ILP
    #pragma unroll
    for (int i = 0; i < 4 && (base_idx + i * num_threads) < total_elements; ++i) {
        int idx = base_idx + i * num_threads;
        int c = (idx / (H * W)) % C;
        
        float x = conv_out[idx];
        
        // Fast softplus with bounds checking
        float softplus_x;
        if (x > 20.0f) softplus_x = x;
        else if (x < -20.0f) softplus_x = expf(x);
        else softplus_x = logf(expf(x) + 1.0f);
        
        // Fast tanh
        float tanh_softplus = tanhf(softplus_x);
        
        // Activation: tanh(softplus(x)) * x
        float act = tanh_softplus * x;
        
        // BN params - using const pointers for L1 caching
        const float w = (weight != nullptr) ? weight[c] : 1.0f;
        const float b = (bias != nullptr) ? bias[c] : 0.0f;
        const float mean = running_mean[c];
        const float var = running_var[c];
        
        float inv_std = rsqrtf(var + eps);
        float norm = (act - mean) * inv_std;
        float out = norm * w + b;
        
        output[idx] = out;
    }
}

__global__ void fused_activation_bn_kernel_half4(

torch::Tensor fused_activation_bn(
    torch::Tensor conv_out,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps) {
    
    int N = conv_out.size(0);
    int C = conv_out.size(1);
    int H = conv_out.size(2);
    int W = conv_out.size(3);
    
    auto output = torch::empty_like(conv_out);
    
    int total_elements = N * C * H * W;
    int block_size = 256;
    // Use more blocks for better occupancy
    int num_blocks = min((total_elements + 3) / 4, 8192);
    
    // Ensure running stats are float
    auto running_mean_f = running_mean.to(torch::kFloat);
    auto running_var_f = running_var.to(torch::kFloat);
    
    // Use specialized kernels based on dtype
    if (conv_out.scalar_type() == torch::kFloat) {
        fused_activation_bn_kernel_float4<<<num_blocks, block_size>>>(
            conv_out.data_ptr<float>(),
            output.data_ptr<float>(),
            weight.defined() ? weight.data_ptr<float>() : nullptr,
            bias.defined() ? bias.data_ptr<float>() : nullptr,
            running_mean_f.data_ptr<float>(),
            running_var_f.data_ptr<float>(),
            static_cast<float>(eps),
            N, C, H, W);
    } else if (conv_out.scalar_type() == torch::kHalf) {
        fused_activation_bn_kernel_half4<<<num_blocks, block_size>>>(
            reinterpret_cast<const __half*>(conv_out.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(output.data_ptr<at::Half>()),
            weight.defined() ? reinterpret_cast<const __half*>(weight.data_ptr<at::Half>()) : nullptr,
            bias.defined() ? reinterpret_cast<const __half*>(bias.data_ptr<at::Half>()) : nullptr,
            running_mean_f.data_ptr<float>(),
            running_var_f.data_ptr<float>(),
            static_cast<float>(eps),
            N, C, H, W);
    } else if (conv_out.scalar_type() == torch::kBFloat16) {
        fused_activation_bn_kernel_bf164<<<num_blocks, block_size>>>(
            reinterpret_cast<const __nv_bfloat16*>(conv_out.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(output.data_ptr<at::BFloat16>()),
            weight.defined() ? reinterpret_cast<const __nv_bfloat16*>(weight.data_ptr<at::BFloat16>()) : nullptr,
            bias.defined() ? reinterpret_cast<const __nv_bfloat16*>(bias.data_ptr<at::BFloat16>()) : nullptr,
            running_mean_f.data_ptr<float>(),
            running_var_f.data_ptr<float>(),
            static_cast<float>(eps),
            N, C, H, W);
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_conv_act_bn',
    cpp_sources='''
torch::Tensor fused_activation_bn(
    torch::Tensor conv_out,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    double eps);
''',
    cuda_sources=cuda_source,
    functions=['fused_activation_bn'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized model that fuses activation and batch normalization
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        # Convolution - relies on cuDNN optimization
        conv_out = self.conv(x)
        
        # Fused activation + batch norm
        output = ext.fused_activation_bn(
            conv_out,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            self.eps
        )
        
        return output


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3]
