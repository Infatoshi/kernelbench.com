import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def layernorm_kernel(
    x_ptr,
    y_ptr,
    w_ptr,
    b_ptr,
    stride_x,
    stride_y,
    N,
    eps,
    BLOCK_SIZE: tl.constexpr,
):
    row_idx = tl.program_id(0)
    
    x_row_ptr = x_ptr + row_idx * stride_x
    y_row_ptr = y_ptr + row_idx * stride_y
    
    # Compute mean
    mean = 0.0
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(x_row_ptr + offsets, mask=mask, other=0.0)
        mean += tl.sum(x, axis=0)
    mean = mean / N
    
    # Compute variance
    var = 0.0
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(x_row_ptr + offsets, mask=mask, other=0.0)
        diff = x - mean
        var += tl.sum(diff * diff, axis=0)
    var = var / N
    inv_std = tl.rsqrt(var + eps)
    
    # Normalize and apply affine
    for start in range(0, N, BLOCK_SIZE):
        offsets = start + tl.arange(0, BLOCK_SIZE)
        mask = offsets < N
        x = tl.load(x_row_ptr + offsets, mask=mask, other=0.0)
        w = tl.load(w_ptr + offsets, mask=mask, other=0.0)
        b = tl.load(b_ptr + offsets, mask=mask, other=0.0)
        
        normalized = (x - mean) * inv_std
        y = normalized * w + b
        
        tl.store(y_row_ptr + offsets, y, mask=mask)

def layernorm_triton(x, weight, bias, eps=1e-5):
    rows, N = x.shape
    y = torch.empty_like(x)
    
    BLOCK_SIZE = triton.next_power_of_2(N)
    if BLOCK_SIZE > 8192:
        BLOCK_SIZE = 8192
    
    layernorm_kernel[(rows,)](
        x, y, weight, bias,
        x.stride(0), y.stride(0),
        N, eps,
        BLOCK_SIZE=BLOCK_SIZE,
    )
    
    return y

class Model(nn.Module):
    def __init__(self, *args):
        super(Model, self).__init__()
        # Handle both Model((64, 256, 256)) and Model(64, 256, 256)
        if len(args) == 1 and isinstance(args[0], (tuple, list)):
            self.normalized_shape = tuple(args[0])
        else:
            self.normalized_shape = tuple(args)
        
        self.weight = nn.Parameter(torch.ones(self.normalized_shape))
        self.bias = nn.Parameter(torch.zeros(self.normalized_shape))
        self.eps = 1e-5

    def forward(self, x):
        # Save original shape
        orig_shape = x.shape
        
        # Calculate the number of normalized dimensions
        num_norm_dims = len(self.normalized_shape)
        
        # Flatten batch dimensions
        batch_size = 1
        for i in range(len(orig_shape) - num_norm_dims):
            batch_size *= orig_shape[i]
        
        # Calculate normalized size
        norm_size = 1
        for d in self.normalized_shape:
            norm_size *= d
        
        # Reshape to [batch_size, norm_size]
        x_flat = x.reshape(batch_size, norm_size)
        
        # Flatten weight and bias
        w_flat = self.weight.reshape(norm_size)
        b_flat = self.bias.reshape(norm_size)
        
        # Apply layernorm
        out_flat = layernorm_triton(x_flat, w_flat, b_flat, self.eps)
        
        # Reshape back to original shape
        return out_flat.reshape(orig_shape)

def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    features = 64
    dim1 = 256
    dim2 = 256
    return [(features, dim1, dim2)]
