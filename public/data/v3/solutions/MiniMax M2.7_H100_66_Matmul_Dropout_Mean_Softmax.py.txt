cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <curand_kernel.h>

#define CUDA_CHECK(x) do { \
    cudaError_t err = x; \
    if (err != cudaSuccess) { \
        fprintf(stderr, "CUDA error at %s:%d: %s\\n", __FILE__, __LINE__, \
                cudaGetErrorString(err)); \
    } \
} while (0)

#define BLOCK_SIZE 256

// FP32 MatMul + Dropout kernel
__global__ void matmul_dropout_fp32_kernel(
    const float* __restrict__ x,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features,
    float dropout_scale,
    float dropout_p,
    unsigned long long seed,
    unsigned long long offset
) {
    int row = blockIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= batch_size || col >= out_features) return;
    
    float sum = bias ? bias[col] : 0.0f;
    const float* x_row = x + row * in_features;
    const float* w_col = weight + col;
    
    // Vectorized matmul - process 4 elements at a time
    int k = 0;
    for (; k + 3 < in_features; k += 4) {
        sum += x_row[k] * w_col[k * out_features]
             + x_row[k+1] * w_col[(k+1) * out_features]
             + x_row[k+2] * w_col[(k+2) * out_features]
             + x_row[k+3] * w_col[(k+3) * out_features];
    }
    for (; k < in_features; k++) {
        sum += x_row[k] * w_col[k * out_features];
    }
    
    // Dropout using curand
    curandStatePhilox4_32_10_t state;
    curand_init(seed, offset + row * out_features + col, 0, &state);
    float rng = curand_uniform(&state);
    float mask = (rng < dropout_p) ? dropout_scale : 0.0f;
    
    output[row * out_features + col] = sum * mask;
}

// FP32 Softmax + Mean kernel (one thread handles entire column)
__global__ void softmax_mean_fp32_kernel(
    const float* __restrict__ matmul_out,
    float* __restrict__ output,
    int batch_size,
    int out_features
) {
    int col = blockIdx.x;
    if (col >= out_features) return;
    
    if (threadIdx.x == 0) {
        // Find max for numerical stability
        float max_val = -INFINITY;
        for (int row = 0; row < batch_size; row++) {
            max_val = fmaxf(max_val, matmul_out[row * out_features + col]);
        }
        
        // Compute exp and sum
        float exp_sum = 0.0f;
        for (int row = 0; row < batch_size; row++) {
            float val = matmul_out[row * out_features + col] - max_val;
            exp_sum += expf(val);
        }
        
        // Compute softmax mean
        float result = 0.0f;
        for (int row = 0; row < batch_size; row++) {
            float val = matmul_out[row * out_features + col] - max_val;
            result += expf(val) / exp_sum;
        }
        result /= batch_size;
        
        output[col] = result;
    }
}

// FP16 MatMul + Dropout kernel
__global__ void matmul_dropout_fp16_kernel(
    const half* __restrict__ x,
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    half* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features,
    float dropout_scale,
    float dropout_p,
    unsigned long long seed,
    unsigned long long offset
) {
    int row = blockIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= batch_size || col >= out_features) return;
    
    float sum = 0.0f;
    const half* x_row = x + row * in_features;
    const half* w_col = weight + col;
    
    // FP16 matmul - process 2 elements at a time
    int k = 0;
    for (; k + 1 < in_features; k += 2) {
        half2 xh = __halves2half2(x_row[k], x_row[k+1]);
        half2 wh = __halves2half2(w_col[k * out_features], w_col[(k+1) * out_features]);
        float2 xv = __half22float2(xh);
        float2 wv = __half22float2(wh);
        sum += xv.x * wv.x + xv.y * wv.y;
    }
    for (; k < in_features; k++) {
        sum += __half2float(x_row[k]) * __half2float(w_col[k * out_features]);
    }
    
    if (bias) {
        sum += __half2float(bias[col]);
    }
    
    // Dropout using curand
    curandStatePhilox4_32_10_t state;
    curand_init(seed, offset + row * out_features + col, 0, &state);
    float rng = curand_uniform(&state);
    float mask = (rng < dropout_p) ? dropout_scale : 0.0f;
    
    output[row * out_features + col] = __float2half(sum * mask);
}

    half* __restrict__ output,
    int batch_size,
    int out_features
) {
    int col = blockIdx.x;

        for (int row = 0; row < batch_size; row++) {
            max_val = fmaxf(max_val, __half2float(matmul_out[row * out_features + col]));
        }
        
        float exp_sum = 0.0f;
        for (int row = 0; row < batch_size; row++) {
            float val = __half2float(matmul_out[row * out_features + col]) - max_val;
            exp_sum += expf(val);
        }
        
        float result = 0.0f;
        for (int row = 0; row < batch_size; row++) {
            float val = __half2float(matmul_out[row * out_features + col]) - max_val;
            result += expf(val) / exp_sum;
        }
        result /= batch_size;
        
        output[col] = __float2half(result);
    }
}

// Implementation
__host__ void launch_matmul_dropout_fp32(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor output,
    float dropout_p,
    unsigned long long seed,
    unsigned long long offset
) {
    int batch_size = x.size(0);
    int in_features = x.size(1);
    int out_features = weight.size(0);
    
    float dropout_scale = 1.0f / (1.0f - dropout_p);
    
    auto matmul_out = torch::empty({batch_size, out_features}, x.options());
    
    dim3 block(BLOCK_SIZE);
    dim3 grid((out_features + block.x - 1) / block.x, batch_size);
    
    matmul_dropout_fp32_kernel<<<grid, block>>>(
        x.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        matmul_out.data_ptr<float>(),
        batch_size, in_features, out_features,
        dropout_scale, dropout_p, seed, offset
    );
    
    softmax_mean_fp32_kernel<<<dim3(out_features), 1>>>(
        matmul_out.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, out_features
    );
    
    CUDA_CHECK(cudaGetLastError());
}

cpp_source = r'''
#include <torch/extension.h>
#include <random>

torch::Tensor forward_fused(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float dropout_p,
    unsigned long long seed,
    unsigned long long offset
);

torch::Tensor forward_fused(
    torch::Tensor x,
    torch::Tensor weight,
    torch::Tensor bias,
    float dropout_p,
    unsigned long long seed,
    unsigned long long offset
) {
    auto output = torch::empty({1, weight.size(0)}, x.options());
    
    if (x.scalar_type() == at::ScalarType::Half) {
        launch_matmul_dropout_fp16(x, weight, bias, output, dropout_p, seed, offset);
    } else if (x.scalar_type() == at::ScalarType::BFloat16) {
        launch_matmul_dropout_bf16(x, weight, bias, output, dropout_p, seed, offset);
    } else {
        launch_matmul_dropout_fp32(x, weight, bias, output, dropout_p, seed, offset);
    }
    
    return output;
}
'''

ext = load_inline(
    name='fused_ops_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['forward_fused'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized fused MatMul + Dropout + Mean + Softmax using custom CUDA kernels.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        self.dropout_p = dropout_p
        self.seed = 42
        self.offset = 0
        
        # Initialize like nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))
        nn.init.zeros_(self.bias)
    
    def forward(self, x):
        return ext.forward_fused(x, self.weight, self.bias, self.dropout_p, self.seed, self.offset)

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 0.2]
