import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Helper function to get lane ID
__device__ __forceinline__ int get_lane_id() {
    int laneid;
    asm("mov.s32 %0, %laneid;" : "=r"(laneid));
    return laneid;
}

// Optimized reduction kernel with warp-level shuffle reduction
template<typename T>
__global__ void sum_reduction_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    const int batch_size,
    const int dim1,
    const int dim2
) {
    __shared__ float sdata[256];
    
    const int tid = threadIdx.x;
    const int warpId = tid >> 5;
    const int laneId = tid & 31;
    
    // Each block handles one (batch, dim2) position
    const int b = blockIdx.x / dim2;
    const int k = blockIdx.x % dim2;
    
    // Grid-stride reduction with sequential loading
    float sum = 0.0f;
    const int stride = blockDim.x;
    
    for (int i = tid; i < dim1; i += stride) {
        const int idx = b * dim1 * dim2 + i * dim2 + k;
        sum += static_cast<float>(x[idx]);
    }
    
    // Warp-level reduction using shuffle instructions
    for (int offset = 16; offset > 0; offset >>= 1) {
        sum += __shfl_down_sync(0xffffffff, sum, offset);
    }
    
    // Write warp result to shared memory
    if (laneId == 0) {
        sdata[warpId] = sum;
    }
    __syncthreads();
    
    // Final warp reduction from shared memory
    if (warpId == 0) {
        sum = (laneId < 8) ? sdata[laneId] : 0.0f;
        for (int offset = 4; offset > 0; offset >>= 1) {
            sum += __shfl_down_sync(0xffffffff, sum, offset);
        }
    }
    
    if (tid == 0) {
        y[b * dim2 + k] = static_cast<T>(sum);
    }
}

torch::Tensor sum_reduction_cuda(torch::Tensor x, int reduce_dim) {
    int batch_size = x.size(0);
    int dim1 = x.size(1);
    int dim2 = x.size(2);
    
    // Output shape with keepdim=True
    std::vector<long int> output_size = {batch_size, 1, dim2};
    auto y = torch::empty(output_size, x.options());
    
    const int block_size = 256;
    const int grid_size = batch_size * dim2;
    
    AT_DISPATCH_FLOATING_TYPES(x.scalar_type(), "sum_reduction_cuda", ([&] {
        sum_reduction_kernel<scalar_t><<<grid_size, block_size>>>(
            x.data_ptr<scalar_t>(),
            y.data_ptr<scalar_t>(),
            batch_size, dim1, dim2
        );
    }));
    
    return y;
}
'''

ext = load_inline(
    name='sum_reduction_ext',
    cpp_sources='torch::Tensor sum_reduction_cuda(torch::Tensor x, int reduce_dim);',
    cuda_sources=cuda_source,
    functions=['sum_reduction_cuda'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, dim: int):
        super(Model, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.sum_reduction_cuda(x, self.dim)

batch_size = 64
dim1 = 256
dim2 = 256
reduce_dim = 1

def get_inputs():
    x = torch.randn(batch_size, dim1, dim2)
    return [x]

def get_init_inputs():
    return [reduce_dim]
