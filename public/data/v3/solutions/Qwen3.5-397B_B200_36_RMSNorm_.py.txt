import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def rms_norm_kernel(
    X_ptr,
    Y_ptr,
    num_features: tl.constexpr,
    num_rows: tl.constexpr,
    eps: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """
    RMSNorm kernel: y = x / sqrt(mean(x^2) + eps)
    Normalizes along the feature dimension.
    
    Grid: (num_rows,)
    Each program handles one row of shape (num_features,)
    """
    # Get the row index
    row_idx = tl.program_id(0)
    
    # Compute offsets for this row
    offsets = tl.arange(0, BLOCK_SIZE)
    mask = offsets < num_features
    
    # Compute the base pointer for this row
    row_start_x = row_idx * num_features
    row_start_y = row_idx * num_features
    
    # Load the row
    x = tl.load(X_ptr + row_start_x + offsets, mask=mask, other=0.0)
    
    # Compute x^2
    x_sq = x * x
    
    # Compute mean of x^2
    mean_sq = tl.sum(x_sq, axis=0) / num_features
    
    # Compute RMS = sqrt(mean_sq + eps)
    rms = tl.sqrt(mean_sq + eps)
    
    # Normalize: y = x / rms
    y = x / rms
    
    # Store the result
    tl.store(Y_ptr + row_start_y + offsets, y, mask=mask)


class Model(nn.Module):
    """
    Simple model that performs RMS Normalization.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        """
        Initializes the RMSNorm layer.

        Args:
            num_features (int): Number of features in the input tensor.
            eps (float, optional): A small value added to the denominator to avoid division by zero. Defaults to 1e-5.
        """
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Applies RMS Normalization to the input tensor.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, num_features, *).

        Returns:
            torch.Tensor: Output tensor with RMS Normalization applied, same shape as input.
        """
        # Input shape: (batch, features, dim1, dim2)
        # We normalize along dim=1 (features)
        orig_shape = x.shape
        batch_size = orig_shape[0]
        num_features = orig_shape[1]
        
        # Reshape to 2D: (batch * dim1 * dim2, features)
        # Then each row is normalized independently
        if x.dim() > 2:
            # Flatten all dims except the feature dim
            x_flat = x.transpose(1, -1).reshape(-1, num_features).contiguous()
        else:
            x_flat = x
        
        num_rows = x_flat.shape[0]
        
        # Allocate output
        y_flat = torch.empty_like(x_flat)
        
        # Choose block size - round up to power of 2
        BLOCK_SIZE = triton.next_power_of_2(num_features)
        BLOCK_SIZE = min(BLOCK_SIZE, 1024)  # Cap at 1024
        
        # Launch kernel
        grid = (num_rows,)
        rms_norm_kernel[grid](
            x_flat,
            y_flat,
            num_features,
            num_rows,
            self.eps,
            BLOCK_SIZE=BLOCK_SIZE,
        )
        
        # Reshape back to original shape
        if x.dim() > 2:
            y_flat = y_flat.reshape(orig_shape[0], orig_shape[2], orig_shape[3], orig_shape[1])
            y = y_flat.transpose(1, -1).contiguous()
        else:
            y = y_flat
        
        return y


batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [features]
