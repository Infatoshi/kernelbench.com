import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Simple Triton-like or basic optimized CUDA kernel for GEMM
// Given the constraints and the reference, we need to handle int8 inputs (quantized)
// with scalar scales.

__global__ void gemm_fp4_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    half* __restrict__ C,
    float scale_a,
    float scale_b,
    int M, int N, int K) {
    
    extern __shared__ int8_t shared_mem[];
    int8_t* sA = &shared_mem[0];
    int8_t* sB = &shared_mem[32 * 32];

    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int row = blockIdx.y * 32 + ty;
    int col = blockIdx.x * 32 + tx;

    float acc = 0.0f;

    for (int k = 0; k < K; k += 32) {
        if (row < M && (k + tx) < K)
            sA[ty * 32 + tx] = A[row * K + (k + tx)];
        else
            sA[ty * 32 + tx] = 0;

        if (col < N && (k + ty) < K)
            sB[ty * 32 + tx] = B[(k + ty) * N + col];
        else
            sB[ty * 32 + tx] = 0;

        __syncthreads();

        #pragma unroll
        for (int i = 0; i < 32; i+=4) {
            acc += (float)sA[ty * 32 + i] * (float)sB[i * 32 + tx];
            acc += (float)sA[ty * 32 + i + 1] * (float)sB[(i + 1) * 32 + tx];
            acc += (float)sA[ty * 32 + i + 2] * (float)sB[(i + 2) * 32 + tx];
            acc += (float)sA[ty * 32 + i + 3] * (float)sB[(i + 3) * 32 + tx];
        }
        __syncthreads();
    }

    if (row < M && col < N) {
        C[row * N + col] = __float2half(acc * scale_a * scale_b);
    }
}

torch::Tensor gemm_fp4(torch::Tensor a_q, torch::Tensor b_q, torch::Tensor scale_a, torch::Tensor scale_b) {
    int M = a_q.size(0);
    int K = a_q.size(1);
    int N = b_q.size(1);
    
    auto options = torch::TensorOptions().dtype(torch::kFloat16).device(a_q.device());
    auto C = torch::empty({M, N}, options);

    float s_a = scale_a.item<float>();
    float s_b = scale_b.item<float>();

    dim3 threadsPerBlock(32, 32);
    dim3 numBlocks((N + 31) / 32, (M + 31) / 32);
    int shared_mem_size = 2 * 32 * 32 * sizeof(int8_t);

    gemm_fp4_kernel<<<numBlocks, threadsPerBlock, shared_mem_size>>>(
        (const int8_t*)a_q.data_ptr<int8_t>(),
        (const int8_t*)b_q.data_ptr<int8_t>(),
        (half*)C.data_ptr<at::Half>(),
        s_a, s_b, M, N, K
    );

    return C;
}
'''

cpp_source = r'''
torch::Tensor gemm_fp4(torch::Tensor a_q, torch::Tensor b_q, torch::Tensor scale_a, torch::Tensor scale_b);
'''

ext = load_inline(
    name='gemm_fp4_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemm_fp4'],
    verbose=False,
    extra_cuda_cflags=['-O3']
)

class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(
        self,
        a_q: torch.Tensor,
        b_q: torch.Tensor,
        scale_a: torch.Tensor,
        scale_b: torch.Tensor,
    ) -> torch.Tensor:
        return ext.gemm_fp4(a_q, b_q, scale_a, scale_b)

def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-8, 8, (m, k), dtype=torch.int8, device='cuda')
    b_q = torch.randint(-8, 8, (k, n), dtype=torch.int8, device='cuda')
    scale_a = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    scale_b = torch.tensor(0.08, dtype=torch.float32, device='cuda')
    return [a_q, b_q, scale_a, scale_b]

def get_init_inputs():
    return []
