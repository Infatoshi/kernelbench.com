import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def gemv_kernel(
    y_ptr,
    x_ptr,
    w_ptr,
    M,
    N,
    stride_y_b,
    stride_y_m,
    stride_x_b,
    stride_x_n,
    stride_w_m,
    stride_w_n,
    BLOCK_N: tl.constexpr,
):
    m_idx = tl.program_id(0)
    b_idx = tl.program_id(1)

    y_ptr = y_ptr + b_idx * stride_y_b + m_idx * stride_y_m
    x_ptr = x_ptr + b_idx * stride_x_b
    w_ptr = w_ptr + m_idx * stride_w_m

    acc = tl.zeros([BLOCK_N], dtype=tl.float32)
    cols = tl.arange(0, BLOCK_N)
    
    for n_offset in range(0, N, BLOCK_N):
        mask = (n_offset + cols) < N
        x_val = tl.load(x_ptr + (n_offset + cols) * stride_x_n, mask=mask, other=0.0).to(tl.float32)
        w_val = tl.load(w_ptr + (n_offset + cols) * stride_w_n, mask=mask, other=0.0).to(tl.float32)
        acc += x_val * w_val

    res = tl.sum(acc)
    tl.store(y_ptr, res.to(tl.bfloat16))


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.weight = nn.Parameter(
            torch.randn(out_features, in_features, dtype=torch.bfloat16) * 0.02
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x.contiguous()
        original_shape = x.shape
        # Flatten B dimensions
        x_2d = x.view(-1, self.in_features)
        B, N = x_2d.shape
        M = self.out_features

        y = torch.empty((B, M), device=x.device, dtype=torch.bfloat16)

        grid = (M, B)
        BLOCK_N = 1024

        gemv_kernel[grid](
            y,
            x_2d,
            self.weight,
            M,
            N,
            y.stride(0),
            y.stride(1),
            x_2d.stride(0),
            x_2d.stride(1),
            self.weight.stride(0),
            self.weight.stride(1),
            BLOCK_N=BLOCK_N,
            num_warps=8,
        )
        # Reshape back to original batch dimensions
        return y.view(*original_shape[:-1], M)

def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.bfloat16, device='cuda')]

def get_init_inputs():
    return [4096, 14336]
