"""
GEMM + residual add fusion using optimized CUDA kernel.
Handles FP16 on H100 with proper memory access patterns.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// Optimized GEMM + residual add for FP16
// Using register blocking (8x8 tiles) with shared memory for tiles

__global__ void gemm_residual_kernel(
    const half* __restrict__ A,
    const half* __restrict__ B,
    const half* __restrict__ Residual,
    half* __restrict__ C,
    int M, int N, int K
) {
    // Shared memory: A tile (BM x BK) + B tile (BK x BN)
    extern __shared__ half shmem[];
    half* sA = shmem;
    half* sB = shmem + 128 * 64;
    
    const int bx = blockIdx.x;
    const int by = blockIdx.y;
    const int tx = threadIdx.x;
    const int ty = threadIdx.y;
    
    // Thread block tile sizes
    const int BM = 128;
    const int BN = 128;
    const int BK = 64;
    
    // Each thread computes 8x8 output elements
    const int TM = 8;
    const int TN = 8;
    
    // Number of threads per block
    const int nthreads = blockDim.x * blockDim.y;
    const int tid = ty * blockDim.x + tx;
    
    // Output position for this thread block
    const int c_row_start = by * BM;
    const int c_col_start = bx * BN;
    
    // Register accumulator for output tile
    half C_local[TM * TN];
    #pragma unroll
    for (int i = 0; i < TM * TN; ++i) {
        C_local[i] = __int2half_rn(0);
    }
    
    const int num_k_tiles = (K + BK - 1) / BK;
    
    // Iterate over K tiles
    for (int k_tile = 0; k_tile < num_k_tiles; ++k_tile) {
        // ====== Load A tile into shared memory ======
        // A is row-major: M x K
        // Each thread loads (BM*BK)/nthreads elements
        const int a_col_base = k_tile * BK;
        
        for (int i = tid; i < BM * BK; i += nthreads) {
            int local_row = i / BK;
            int local_col = i % BK;
            int global_row = c_row_start + local_row;
            int global_col = a_col_base + local_col;
            
            if (global_row < M && global_col < K) {
                sA[local_row * BK + local_col] = A[global_row * K + global_col];
            }
        }
        
        // ====== Load B tile into shared memory ======
        // B is column-major (stored as row-major in memory, accessed as column-major)
        // We store it in shared memory in row-major for coalesced access
        const int b_row_base = k_tile * BK;
        
        for (int i = tid; i < BK * BN; i += nthreads) {
            int local_row = i / BN;
            int local_col = i % BN;
            int global_row = b_row_base + local_row;
            int global_col = c_col_start + local_col;
            
            if (global_row < K && global_col < N) {
                // Store B in row-major format in shared memory
                sB[local_row * BN + local_col] = B[global_row * N + global_col];
            }
        }
        
        __syncthreads();
        
        // ====== Compute this tile ======
        const int thread_row_start = ty * TM;
        const int thread_col_start = tx * TN;
        
        // For each K element in the tile
        #pragma unroll
        for (int k = 0; k < BK; ++k) {
            // Load A values for this k
            half a_val[TM];
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                a_val[i] = sA[(thread_row_start + i) * BK + k];
            }
            
            // Load B values for this k  
            half b_val[TN];
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                b_val[j] = sB[k * BN + thread_col_start + j];
            }
            
            // Fused multiply-add for each output element
            #pragma unroll
            for (int i = 0; i < TM; ++i) {
                #pragma unroll
                for (int j = 0; j < TN; ++j) {
                    C_local[i * TN + j] = __hfma(a_val[i], b_val[j], C_local[i * TN + j]);
                }
            }
        }
        
        __syncthreads();
    }
    
    // ====== Store result with fused residual add ======
    const int thread_row_start = ty * TM;
    const int thread_col_start = tx * TN;
    
    #pragma unroll
    for (int i = 0; i < TM; ++i) {
        int global_row = c_row_start + thread_row_start + i;
        if (global_row < M) {
            #pragma unroll
            for (int j = 0; j < TN; ++j) {
                int global_col = c_col_start + thread_col_start + j;
                if (global_col < N) {
                    // Add residual in the final store
                    half result = __hadd(C_local[i * TN + j], 
                                         Residual[global_row * N + global_col]);
                    C[global_row * N + global_col] = result;
                }
            }
        }
    }
}

torch::Tensor gemm_residual_add_cuda(
    torch::Tensor a,
    torch::Tensor b,
    torch::Tensor residual
) {
    int M = a.size(0);
    int K = a.size(1);
    int N = b.size(1);
    
    auto output = torch::empty({M, N}, a.options());
    
    const half* a_ptr = reinterpret_cast<const half*>(a.data_ptr<at::Half>());
    const half* b_ptr = reinterpret_cast<const half*>(b.data_ptr<at::Half>());
    const half* res_ptr = reinterpret_cast<const half*>(residual.data_ptr<at::Half>());
    half* c_ptr = reinterpret_cast<half*>(output.data_ptr<at::Half>());
    
    // Block configuration:
    // BM=128, BN=128, TM=8, TN=8
    // Threads per block: (128/8) * (128/8) = 16 * 16 = 256
    const int BM = 128;
    const int BN = 128;
    const int TM = 8;
    const int TN = 8;
    
    dim3 grid((N + BN - 1) / BN, (M + BM - 1) / BM);
    dim3 block(BN / TN, BM / TM);
    
    // Shared memory: A tile (128*64) + B tile (64*128) = 16384 elements
    // Each half is 2 bytes = 32KB
    const int BK = 64;
    size_t smem_size = (BM * BK + BK * BN) * sizeof(half);
    
    gemm_residual_kernel<<<grid, block, smem_size>>>(
        a_ptr, b_ptr, res_ptr, c_ptr, M, N, K
    );
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    cudaDeviceSynchronize();
    
    return output;
}
'''

ext = load_inline(
    name='gemm_residual_ext',
    cpp_sources='torch::Tensor gemm_residual_add_cuda(torch::Tensor, torch::Tensor, torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['gemm_residual_add_cuda'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self):
        super().__init__()
    
    def forward(self, a: torch.Tensor, b: torch.Tensor, residual: torch.Tensor) -> torch.Tensor:
        # Ensure tensors are on CUDA
        if not a.is_cuda:
            a = a.cuda()
        if not b.is_cuda:
            b = b.cuda()
        if not residual.is_cuda:
            residual = residual.cuda()
        
        # Use the optimized CUDA kernel
        return ext.gemm_residual_add_cuda(a, b, residual)


def get_inputs():
    m = 2048
    n = 4096
    k = 2048
    a = torch.randn(m, k, dtype=torch.float16)
    b = torch.randn(k, n, dtype=torch.float16)
    residual = torch.randn(m, n, dtype=torch.float16)
    return [a, b, residual]


def get_init_inputs():
    return []
