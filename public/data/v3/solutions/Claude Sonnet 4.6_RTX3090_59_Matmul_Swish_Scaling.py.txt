import torch
import torch.nn as nn
import triton
import triton.language as tl

# Fused GEMM (with bias) + Swish + Scale using Triton
# Uses tensor core-friendly tile sizes for FP16/BF16

@triton.jit
def matmul_swish_scale_kernel(
    # Pointers
    A_ptr, B_ptr, bias_ptr, C_ptr,
    # Matrix dims
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Scalar
    scaling_factor,
    # Tile sizes (compile-time constants)
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    """Fused kernel: C = (A @ B.T + bias) * sigmoid(A @ B.T + bias) * scaling_factor"""
    # Program ID
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    # Block pointers
    offs_am = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_bn = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    a_ptrs = A_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = B_ptr + (offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn)

    # Accumulate in FP32
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_remaining = K - k * BLOCK_K
        mask_k = offs_k < k_remaining
        a = tl.load(a_ptrs, mask=mask_k[None, :], other=0.0)
        b = tl.load(b_ptrs, mask=mask_k[:, None], other=0.0)
        acc = tl.dot(a, b, acc)
        a_ptrs += BLOCK_K * stride_ak
        b_ptrs += BLOCK_K * stride_bk

    # Add bias
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    bias = tl.load(bias_ptr + offs_n, mask=offs_n < N, other=0.0)
    acc = acc + bias[None, :]

    # Swish activation: x * sigmoid(x)
    # sigmoid(x) = 1 / (1 + exp(-x))
    sig = tl.sigmoid(acc)
    acc = acc * sig

    # Scale
    acc = acc * scaling_factor

    # Write output
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    c_ptrs = C_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, acc.to(C_ptr.dtype.element_ty), mask=c_mask)


def matmul_swish_scale(x, weight, bias, scaling_factor):
    """
    x: [M, K]  (input)
    weight: [N, K]  (Linear weight, transposed for matmul)
    bias: [N]
    output: [M, N]
    Computes: swish(x @ weight.T + bias) * scaling_factor
    """
    # Make contiguous
    x = x.contiguous()
    weight = weight.contiguous()
    bias = bias.contiguous()

    M, K = x.shape
    N, K2 = weight.shape
    assert K == K2

    # Output in float32
    out = torch.empty((M, N), device=x.device, dtype=torch.float32)

    # Choose tile sizes based on problem size
    BLOCK_M = 128
    BLOCK_N = 128
    BLOCK_K = 32
    GROUP_M = 8

    grid = lambda META: (
        triton.cdiv(M, META['BLOCK_M']) * triton.cdiv(N, META['BLOCK_N']),
    )

    matmul_swish_scale_kernel[grid](
        x, weight, bias, out,
        M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(1), weight.stride(0),  # B is stored as [N, K], access as [K, N]
        out.stride(0), out.stride(1),
        scaling_factor,
        BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
        GROUP_M=GROUP_M,
    )
    return out


# Also create a CUDA extension version for comparison/fallback
from torch.utils.cpp_extension import load_inline

swish_scale_cuda_src = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void swish_scale_kernel_fp32(
    float* __restrict__ x,
    const float scale,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        float v = x[idx];
        float sig = 1.0f / (1.0f + expf(-v));
        x[idx] = v * sig * scale;
    }
}

void swish_scale_inplace(torch::Tensor x, float scale) {
    int n = x.numel();
    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    swish_scale_kernel_fp32<<<blocks, threads>>>(
        x.data_ptr<float>(), scale, n
    );
}
'''

swish_ext = load_inline(
    name='swish_scale_ext',
    cpp_sources='void swish_scale_inplace(torch::Tensor x, float scale);',
    cuda_sources=swish_scale_cuda_src,
    functions=['swish_scale_inplace'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    Fused model: Linear + Swish + Scale
    Uses Triton kernel for fused GEMM + Swish + Scale
    """
    def __init__(self, in_features, out_features, scaling_factor):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.scaling_factor = scaling_factor
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        # Use fused Triton kernel: GEMM + bias + Swish + Scale
        result = matmul_swish_scale(
            x.float() if x.dtype != torch.float32 else x,
            self.matmul.weight.float() if self.matmul.weight.dtype != torch.float32 else self.matmul.weight,
            self.matmul.bias.float() if self.matmul.bias.dtype != torch.float32 else self.matmul.bias,
            self.scaling_factor
        )
        return result


batch_size = 128
in_features = 4096
out_features = 4096
scaling_factor = 2.0


def get_inputs():
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [in_features, out_features, scaling_factor]
