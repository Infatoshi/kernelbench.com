import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_matmul_div_gelu_kernel(
    # Pointers
    A_ptr, B_ptr, bias_ptr, C_ptr,
    # Dimensions
    M, N, K,
    # Strides
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    # Scalar
    divisor,
    # Meta
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_bn = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    a_ptrs = A_ptr + (offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak)
    b_ptrs = B_ptr + (offs_k[:, None] * stride_bk + offs_bn[None, :] * stride_bn)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_K)):
        a = tl.load(a_ptrs, mask=offs_k[None, :] < K - k * BLOCK_K, other=0.0)
        b = tl.load(b_ptrs, mask=offs_k[:, None] < K - k * BLOCK_K, other=0.0)
        acc = tl.dot(a, b, acc)
        a_ptrs += BLOCK_K * stride_ak
        b_ptrs += BLOCK_K * stride_bk

    # Add bias
    bias = tl.load(bias_ptr + offs_bn)
    acc = acc + bias[None, :]

    # Divide
    acc = acc / divisor

    # GELU (tanh approximation): 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
    sqrt_2_over_pi: tl.constexpr = 0.7978845608028654
    x = acc
    x3 = x * x * x
    inner = sqrt_2_over_pi * (x + 0.044715 * x3)
    tanh_inner = tl.extra.libdevice.tanh(inner)
    gelu = 0.5 * x * (1.0 + tanh_inner)

    # Store
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    c_ptrs = C_ptr + stride_cm * offs_cm[:, None] + stride_cn * offs_cn[None, :]
    mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, gelu.to(tl.float16), mask=mask)


def fused_matmul_div_gelu(x, weight, bias, divisor):
    # x: (M, K), weight: (N, K) -> we need x @ weight.T
    # Triton: A=x (M,K), B=weight.T (K,N)
    M, K = x.shape
    N = weight.shape[0]
    
    # Convert to fp16 for tensor core usage
    x_fp16 = x.half()
    weight_fp16 = weight.half()
    
    # B = weight.T, shape (K, N)
    B = weight_fp16.t().contiguous()
    
    C = torch.empty((M, N), device=x.device, dtype=torch.float16)
    
    grid = lambda META: (triton.cdiv(M, META['BLOCK_M']) * triton.cdiv(N, META['BLOCK_N']),)
    
    fused_matmul_div_gelu_kernel[grid](
        x_fp16, B, bias,
        C,
        M, N, K,
        x_fp16.stride(0), x_fp16.stride(1),
        B.stride(0), B.stride(1),
        C.stride(0), C.stride(1),
        divisor,
        BLOCK_M=128, BLOCK_N=128, BLOCK_K=32,
        GROUP_SIZE_M=8,
        num_warps=8,
        num_stages=3,
    )
    
    return C.float()


class Model(nn.Module):
    def __init__(self, input_size, output_size, divisor):
        super(Model, self).__init__()
        self.linear = nn.Linear(input_size, output_size)
        self.divisor = divisor

    def forward(self, x):
        return fused_matmul_div_gelu(x, self.linear.weight, self.linear.bias, self.divisor)


batch_size = 128
input_size = 4096
output_size = 4096
divisor = 10.0

def get_inputs():
    return [torch.randn(batch_size, input_size)]

def get_init_inputs():
    return [input_size, output_size, divisor]
