import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def fused_scale_pool_clamp_kernel(
    x_ptr,          # Input after conv+gn (B, C, H, W)
    y_ptr,          # Output (B, C, H_out, W_out)
    scale_ptr,      # Scale parameters (C,)
    B: tl.constexpr,        # Batch size
    C: tl.constexpr,        # Number of channels
    H: tl.constexpr,        # Input height
    W: tl.constexpr,        # Input width
    H_out: tl.constexpr,    # Output height
    W_out: tl.constexpr,    # Output width
    clamp_min: tl.constexpr,
    clamp_max: tl.constexpr,
):
    """
    Fused kernel for Scale + MaxPool2d + Clamp
    
    Each program instance handles one output element (b, c, h_out, w_out)
    """
    pid = tl.program_id(0)
    
    # Decode position: pid -> (b, c, h_out, w_out)
    tmp = pid
    w_out_pos = tmp % W_out
    tmp = tmp // W_out
    h_out_pos = tmp % H_out
    tmp = tmp // H_out
    c = tmp % C
    b = tmp // C
    
    # MaxPool: get 2x2 window input positions
    h_in_start = h_out_pos * 2
    w_in_start = w_out_pos * 2
    
    # Compute max over 2x2 window, applying scale during the process
    max_val = float('-inf')
    scale_val = tl.load(scale_ptr + c)
    
    for dh in range(2):
        for dw in range(2):
            h_in_pos = h_in_start + dh
            w_in_pos = w_in_start + dw
            if h_in_pos < H and w_in_pos < W:
                idx = b * C * H * W + c * H * W + h_in_pos * W + w_in_pos
                val = tl.load(x_ptr + idx)
                # Apply scale
                scaled_val = val * scale_val
                max_val = tl.maximum(max_val, scaled_val)
    
    # Clamp
    result = tl.minimum(tl.maximum(max_val, clamp_min), clamp_max)
    
    # Store
    out_idx = b * C * H_out * W_out + c * H_out * W_out + h_out_pos * W_out + w_out_pos
    tl.store(y_ptr + out_idx, result)


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, scale_shape, maxpool_kernel_size, clamp_min, clamp_max):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.group_norm = nn.GroupNorm(num_groups, out_channels)
        self.scale = nn.Parameter(torch.ones(scale_shape))
        self.maxpool = nn.MaxPool2d(kernel_size=maxpool_kernel_size)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.num_groups = num_groups
        self.scale_shape = scale_shape
        self.maxpool_kernel_size = maxpool_kernel_size
        self.clamp_min_val = clamp_min
        self.clamp_max_val = clamp_max

    def forward(self, x):
        # Conv2d
        x = self.conv(x)
        
        # GroupNorm
        x = self.group_norm(x)
        
        # Get dimensions
        batch_size, channels, h, w = x.shape
        
        # Compute output size after maxpool
        h_out = h // self.maxpool_kernel_size
        w_out = w // self.maxpool_kernel_size
        
        # Allocate output
        y = torch.empty(batch_size, channels, h_out, w_out, device=x.device, dtype=x.dtype)
        
        # Fused kernel launch for Scale + MaxPool + Clamp
        total_out = batch_size * channels * h_out * w_out
        grid = (total_out,)
        
        fused_scale_pool_clamp_kernel[grid](
            x, y, self.scale,
            batch_size, channels, h, w, h_out, w_out,
            self.clamp_min_val, self.clamp_max_val,
        )
        
        return y


def get_inputs():
    return [torch.randn(16, 3, 256, 256)]


def get_init_inputs():
    return [3, 16, 3, 8, (16, 1, 1), 2, 0.0, 1.0]
