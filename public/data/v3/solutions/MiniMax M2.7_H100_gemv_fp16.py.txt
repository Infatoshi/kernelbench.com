"""
FP16 GEMV optimized for H100 with vectorized memory operations.

The operation y = x @ W.t() where:
- x: (batch, 4096) FP16
- W: (14336, 4096) FP16
- y: (batch, 14336) FP16

y[b, n] = sum over k of x[b, k] * W[n, k]
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized kernel with half2 vectorization
// w is (n, k), y[b, n_col] = sum_k x[b, k] * w[n_col, k]
__global__ void gemv_fast_kernel(
    const __half2* __restrict__ x,
    const __half2* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k_halves,
    int n
) {
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    float sum = 0.0f;
    
    // y[b, n_col] = sum over k of x[b, k] * w[n_col, k]
    // w has shape (n, k) = (n, k_halves*2), so w[n_col, k] = w[n_col * k + k]
    // But we have w as (n, k_halves) in half2, so:
    // w[n_col, 2*ki] = w[ki * n + n_col] (low half)
    // w[n_col, 2*ki+1] = w[ki * n + n_col] (high half)
    
    for (int ki = 0; ki < k_halves; ++ki) {
        __half2 x_v = x[batch_idx * k_halves + ki];
        __half2 w_v = w[ki * n + n_col];
        
        float x0 = __half2float(__low2half(x_v));
        float x1 = __half2float(__high2half(x_v));
        float w0 = __half2float(__low2half(w_v));
        float w1 = __half2float(__high2half(w_v));
        
        sum += x0 * w0 + x1 * w1;
    }
    
    y[batch_idx * n + n_col] = __float2half(sum);
}

// Kernel with loop unrolling for better ILP
__global__ void gemv_unrolled_kernel(
    const __half2* __restrict__ x,
    const __half2* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k_halves,
    int n
) {
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    float sum = 0.0f;
    
    // Unroll by 8 for better instruction-level parallelism
    const int UNROLL = 8;
    int k_rounded = (k_halves / UNROLL) * UNROLL;
    
    // Main loop
    for (int ki = 0; ki < k_rounded; ki += UNROLL) {
        float sum_local = 0.0f;
        
        #pragma unroll
        for (int i = 0; i < UNROLL; ++i) {
            __half2 x_v = x[batch_idx * k_halves + ki + i];
            __half2 w_v = w[(ki + i) * n + n_col];
            
            float x0 = __half2float(__low2half(x_v));
            float x1 = __half2float(__high2half(x_v));
            float w0 = __half2float(__low2half(w_v));
            float w1 = __half2float(__high2half(w_v));
            
            sum_local += x0 * w0 + x1 * w1;
        }
        sum += sum_local;
    }
    
    // Handle remainder
    for (int ki = k_rounded; ki < k_halves; ++ki) {
        __half2 x_v = x[batch_idx * k_halves + ki];
        __half2 w_v = w[ki * n + n_col];
        
        float x0 = __half2float(__low2half(x_v));
        float x1 = __half2float(__high2half(x_v));
        float w0 = __half2float(__low2half(w_v));
        float w1 = __half2float(__high2half(w_v));
        
        sum += x0 * w0 + x1 * w1;
    }
    
    y[batch_idx * n + n_col] = __float2half(sum);
}

// Kernel with software pipelining via register blocking
__global__ void gemv_pipelined_kernel(
    const __half2* __restrict__ x,
    const __half2* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k_halves,
    int n
) {
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    // Use 4 accumulators for better pipelining
    float sum0 = 0.0f;
    float sum1 = 0.0f;
    float sum2 = 0.0f;
    float sum3 = 0.0f;
    
    // Prefetch x values
    __half2 x0, x1, x2, x3;
    __half2 w0, w1, w2, w3;
    
    int k_iterations = k_halves / 4;
    
    for (int i = 0; i < k_iterations; ++i) {
        int ki = i * 4;
        
        // Load x values
        x0 = x[batch_idx * k_halves + ki];
        x1 = x[batch_idx * k_halves + ki + 1];
        x2 = x[batch_idx * k_halves + ki + 2];
        x3 = x[batch_idx * k_halves + ki + 3];
        
        // Load w values
        w0 = w[ki * n + n_col];
        w1 = w[(ki + 1) * n + n_col];
        w2 = w[(ki + 2) * n + n_col];
        w3 = w[(ki + 3) * n + n_col];
        
        // Compute
        sum0 += __half2float(__low2half(x0)) * __half2float(__low2half(w0));
        sum0 += __half2float(__high2half(x0)) * __half2float(__high2half(w0));
        
        sum1 += __half2float(__low2half(x1)) * __half2float(__low2half(w1));
        sum1 += __half2float(__high2half(x1)) * __half2float(__high2half(w1));
        
        sum2 += __half2float(__low2half(x2)) * __half2float(__low2half(w2));
        sum2 += __half2float(__high2half(x2)) * __half2float(__high2half(w2));
        
        sum3 += __half2float(__low2half(x3)) * __half2float(__low2half(w3));
        sum3 += __half2float(__high2half(x3)) * __half2float(__high2half(w3));
    }
    
    // Handle remainder
    for (int ki = k_iterations * 4; ki < k_halves; ++ki) {
        __half2 x_v = x[batch_idx * k_halves + ki];
        __half2 w_v = w[ki * n + n_col];
        
        float x0 = __half2float(__low2half(x_v));
        float x1 = __half2float(__high2half(x_v));
        float w0 = __half2float(__low2half(w_v));
        float w1 = __half2float(__high2half(w_v));
        
        sum0 += x0 * w0 + x1 * w1;
    }
    
    y[batch_idx * n + n_col] = __float2half(sum0 + sum1 + sum2 + sum3);
}

// Shared memory version for better memory efficiency
__global__ void gemv_smem_kernel(
    const __half2* __restrict__ x,
    const __half2* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k_halves,
    int n
) {
    extern __shared__ __half2 smem_w[];
    
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    float sum = 0.0f;
    
    // Process K in tiles
    const int TILE_K = 128;
    
    for (int tile = 0; tile < k_halves; tile += TILE_K) {
        // Load weight tile
        int tile_k = min(TILE_K, k_halves - tile);
        
        for (int ki = threadIdx.x; ki < tile_k; ki += blockDim.x) {
            smem_w[ki * n + n_col] = w[(tile + ki) * n + n_col];
        }
        __syncthreads();
        
        // Compute
        for (int ki = 0; ki < tile_k; ++ki) {
            __half2 x_v = x[batch_idx * k_halves + tile + ki];
            __half2 w_v = smem_w[ki * n + n_col];
            
            float x0 = __half2float(__low2half(x_v));
            float x1 = __half2float(__high2half(x_v));
            float w0 = __half2float(__low2half(w_v));
            float w1 = __half2float(__high2half(w_v));
            
            sum += x0 * w0 + x1 * w1;
        }
        
        __syncthreads();
    }
    
    y[batch_idx * n + n_col] = __float2half(sum);
}

// Shared memory with double buffering
__global__ void gemv_double_buffer_kernel(
    const __half2* __restrict__ x,
    const __half2* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k_halves,
    int n
) {
    extern __shared__ __half2 smem[][2];
    
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    float sum = 0.0f;
    const int TILE_K = 64;
    
    // Load first tile
    for (int ki = threadIdx.x; ki < TILE_K; ki += blockDim.x) {
        if (ki < k_halves) {
            smem[ki][0] = w[ki * n + n_col];
        }
    }
    __syncthreads();
    
    // Process remaining tiles
    for (int tile = TILE_K; tile < k_halves; tile += TILE_K) {
        // Load next tile
        for (int ki = threadIdx.x; ki < TILE_K; ki += blockDim.x) {
            if (tile + ki < k_halves) {
                smem[ki][1] = w[(tile + ki) * n + n_col];
            }
        }
        
        // Compute current tile
        int tile_start = tile - TILE_K;
        int tile_k = min(TILE_K, k_halves - tile_start);
        
        for (int ki = 0; ki < tile_k; ++ki) {
            __half2 x_v = x[batch_idx * k_halves + tile_start + ki];
            __half2 w_v = smem[ki][0];
            
            float x0 = __half2float(__low2half(x_v));
            float x1 = __half2float(__high2half(x_v));
            float w0 = __half2float(__low2half(w_v));
            float w1 = __half2float(__high2half(w_v));
            
            sum += x0 * w0 + x1 * w1;
        }
        
        __syncthreads();
        
        // Swap buffers
        #pragma unroll
        for (int ki = threadIdx.x; ki < TILE_K; ki += blockDim.x) {
            smem[ki][0] = smem[ki][1];
        }
        __syncthreads();
    }
    
    // Compute last tile
    int tile_start = (k_halves / TILE_K) * TILE_K;
    for (int ki = 0; ki < TILE_K && tile_start + ki < k_halves; ++ki) {
        __half2 x_v = x[batch_idx * k_halves + tile_start + ki];
        __half2 w_v = smem[ki][0];
        
        float x0 = __half2float(__low2half(x_v));
        float x1 = __half2float(__high2half(x_v));
        float w0 = __half2float(__low2half(w_v));
        float w1 = __half2float(__high2half(w_v));
        
        sum += x0 * w0 + x1 * w1;
    }
    
    y[batch_idx * n + n_col] = __float2half(sum);
}

// Simple kernel for small K
__global__ void gemv_simple_kernel(
    const __half* __restrict__ x,
    const __half* __restrict__ w,
    __half* __restrict__ y,
    int batch,
    int k,
    int n
) {
    int batch_idx = blockIdx.z;
    int n_col = blockIdx.y * blockDim.x + threadIdx.x;
    
    if (batch_idx >= batch || n_col >= n) return;
    
    float sum = 0.0f;
    
    for (int ki = 0; ki < k; ++ki) {
        __half x_v = x[batch_idx * k + ki];
        __half w_v = w[n_col * k + ki];
        sum += __half2float(x_v) * __half2float(w_v);
    }
    
    y[batch_idx * n + n_col] = __float2half(sum);
}

torch::Tensor gemv_forward(torch::Tensor x, torch::Tensor w) {
    auto batch = x.size(0);
    auto k = x.size(1);
    auto n = w.size(0);  // w is (n, k), output is (batch, n)
    
    auto y = torch::empty({batch, n}, x.options());
    
    if (batch == 0 || k == 0 || n == 0) {
        return y;
    }
    
    const __half* x_ptr = reinterpret_cast<const __half*>(x.data_ptr<c10::Half>());
    const __half* w_ptr = reinterpret_cast<const __half*>(w.data_ptr<c10::Half>());
    __half* y_ptr = reinterpret_cast<__half*>(y.data_ptr<c10::Half>());
    
    int k_halves = k / 2;
    
    if (k_halves > 0) {
        const __half2* x_ptr2 = reinterpret_cast<const __half2*>(x_ptr);
        const __half2* w_ptr2 = reinterpret_cast<const __half2*>(w_ptr);
        
        // Grid: (1, n_blocks, batch)
        int n_threads = 128;
        int n_blocks = (n + n_threads - 1) / n_threads;
        
        dim3 grid(1, n_blocks, batch);
        
        gemv_unrolled_kernel<<<grid, n_threads>>>(x_ptr2, w_ptr2, y_ptr, batch, k_halves, n);
    } else {
        // Fallback for k=1
        int n_threads = 256;
        int n_blocks = (n + n_threads - 1) / n_threads;
        
        gemv_simple_kernel<<<dim3(1, n_blocks, batch), n_threads>>>(
            x_ptr, w_ptr, y_ptr, batch, k, n
        );
    }
    
    return y;
}
'''

ext = load_inline(
    name='gemv_ext',
    cpp_sources='''
    torch::Tensor gemv_forward(torch::Tensor x, torch::Tensor w);
    ''',
    cuda_sources=cuda_source,
    functions=['gemv_forward'],
    extra_include_paths=['/usr/local/cuda/include'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16) * 0.02)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_fp16 = x.to(dtype=torch.float16)
        # x: (batch, k), w: (n, k), output: (batch, n)
        # Need to ensure w has shape (n, k) for the kernel
        return ext.gemv_forward(x_fp16, self.weight)


def get_inputs():
    return [torch.randn(32, 4096, dtype=torch.float16)]


def get_init_inputs():
    return []
