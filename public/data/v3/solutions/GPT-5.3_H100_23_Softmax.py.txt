import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def softmax_kernel(X, Y, stride, n_cols, BLOCK: tl.constexpr):
    row = tl.program_id(0)
    X_ptr = X + row * stride
    Y_ptr = Y + row * stride

    offs = tl.arange(0, BLOCK)

    # compute max
    max_val = -1e30
    for i in range(0, n_cols, BLOCK):
        cols = i + offs
        mask = cols < n_cols
        x = tl.load(X_ptr + cols, mask=mask, other=-1e30)
        max_val = tl.maximum(max_val, tl.max(x, axis=0))

    # compute sum exp
    sum_val = 0.0
    for i in range(0, n_cols, BLOCK):
        cols = i + offs
        mask = cols < n_cols
        x = tl.load(X_ptr + cols, mask=mask, other=-1e30)
        x = tl.exp(x - max_val)
        sum_val += tl.sum(x, axis=0)

    # write output
    for i in range(0, n_cols, BLOCK):
        cols = i + offs
        mask = cols < n_cols
        x = tl.load(X_ptr + cols, mask=mask, other=-1e30)
        x = tl.exp(x - max_val) / sum_val
        tl.store(Y_ptr + cols, x, mask=mask)


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_cuda
        n_rows, n_cols = x.shape
        y = torch.empty_like(x)

        BLOCK = 1024
        grid = (n_rows,)

        softmax_kernel[grid](x, y, x.stride(0), n_cols, BLOCK=BLOCK)
        return y

    return [torch.randn(256, 16384, device='cuda')]
