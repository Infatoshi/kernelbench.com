import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__device__ float silu(float x){ return x / (1.0f + expf(-x)); }

__global__ void moe_kernel(
    const float* __restrict__ x,
    const int64_t* __restrict__ expert_idx,
    const float* __restrict__ expert_w,
    const float* __restrict__ gate,
    const float* __restrict__ up,
    const float* __restrict__ down,
    float* __restrict__ out,
    int N, int H, int I, int topk
){
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    if (tid >= N*topk) return;

    int token = tid / topk;
    int k = tid % topk;
    int e = expert_idx[tid];
    float w = expert_w[tid];

    const float* x_ptr = x + token * H;
    const float* gate_w = gate + e * I * H;
    const float* up_w = up + e * I * H;
    const float* down_w = down + e * H * I;

    for(int h=0;h<H;h++){
        float val = 0.f;
        for(int i=0;i<I;i++){
            float g=0.f, u=0.f;
            for(int hh=0;hh<H;hh++){
                float xv = x_ptr[hh];
                g += xv * gate_w[i*H + hh];
                u += xv * up_w[i*H + hh];
            }
            float inter = silu(g) * u;
            val += inter * down_w[h*I + i];
        }
        atomicAdd(&out[token*H + h], val * w);
    }
}

void launch(torch::Tensor x, torch::Tensor expert_idx, torch::Tensor expert_w,
            torch::Tensor gate, torch::Tensor up, torch::Tensor down,
            torch::Tensor out, int topk) {
    int N = x.size(0);
    int H = x.size(1);
    int I = gate.size(1);

    int total = N * topk;
    int threads = 128;
    int blocks = (total + threads - 1) / threads;

    moe_kernel<<<blocks, threads, I*sizeof(float)>>>(
        x.data_ptr<float>(),
        expert_idx.data_ptr<int64_t>(),
        expert_w.data_ptr<float>(),
        gate.data_ptr<float>(),
        up.data_ptr<float>(),
        down.data_ptr<float>(),
        out.data_ptr<float>(),
        N, H, I, topk
    );
}
'''

ext = load_inline(name='moe_ext', cpp_sources='void launch(torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor,torch::Tensor,int);', cuda_sources=cuda_src, functions=['launch'], verbose=False)

class Model(nn.Module):
    def __init__(self, hidden_size, intermediate_size, num_experts):
        super().__init__()
        self.hidden_size = hidden_size
        self.intermediate_size = intermediate_size
        self.num_experts = num_experts

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.gate_proj = nn.Parameter(torch.randn(num_experts, intermediate_size, hidden_size) * 0.02)
        self.up_proj = nn.Parameter(torch.randn(num_experts, intermediate_size, hidden_size) * 0.02)
        self.down_proj = nn.Parameter(torch.randn(num_experts, hidden_size, intermediate_size) * 0.02)
        torch.random.set_rng_state(rng_state)

    def forward(self, x, expert_indices, expert_weights):
        B,S,H = x.shape
        topk = expert_indices.shape[-1]
        N = B*S

        x_flat = x.reshape(N, H).contiguous()
        idx = expert_indices.reshape(N*topk).contiguous()
        w = expert_weights.reshape(N*topk).contiguous()

        out = torch.zeros_like(x_flat)

        # follow reference grouping but fuse gate/up projection
        indices_flat = expert_indices.view(N * topk)
        weights_flat = expert_weights.view(N * topk)

        token_ids = torch.arange(N, device=x.device).unsqueeze(1).expand(-1, topk).reshape(-1)

        sorted_expert_idx, sort_order = indices_flat.sort()
        sorted_token_ids = token_ids[sort_order]
        sorted_weights = weights_flat[sort_order]

        expert_counts = torch.bincount(sorted_expert_idx, minlength=self.num_experts)
        expert_offsets = torch.cat([torch.zeros(1, dtype=torch.long, device=x.device), expert_counts.cumsum(0)])

        sorted_x = x_flat[sorted_token_ids]
        sorted_output = torch.empty_like(sorted_x)

        for e in range(self.num_experts):
            start = expert_offsets[e].item()
            end = expert_offsets[e+1].item()
            if start == end:
                continue

            expert_x = sorted_x[start:end]

            # fuse gate + up via concat
            w_cat = torch.cat([self.gate_proj[e], self.up_proj[e]], dim=0)  # (2I,H)
            proj = torch.einsum('nh,kh->nk', expert_x, w_cat)
            gate, up = proj.chunk(2, dim=1)

            gate = gate * torch.sigmoid(gate)
            inter = gate * up
            y = torch.einsum('ni,hi->nh', inter, self.down_proj[e])

            sorted_output[start:end] = y

        weighted = sorted_output * sorted_weights.unsqueeze(-1)

        out.index_add_(0, sorted_token_ids, weighted)

        return out.view(B,S,H)
