import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "layernorm"
SUPPORTED_PRECISIONS = ['fp16', 'bf16', 'fp32']
HARDWARE_REQUIRED = ['RTX3090', 'H100', 'B200']


@triton.jit
def _rmsnorm_kernel(
    x_ptr,
    y_ptr,
    outer_size,
    feat_size,
    eps,
    BLOCK_FEAT: tl.constexpr,
):
    pid = tl.program_id(0)
    if pid >= outer_size:
        return

    offs = tl.arange(0, BLOCK_FEAT)
    mask = offs < feat_size

    base = pid * feat_size
    x = tl.load(x_ptr + base + offs, mask=mask, other=0.0)
    x_f32 = x.to(tl.float32)
    mean_sq = tl.sum(x_f32 * x_f32, axis=0) / feat_size
    inv_rms = tl.rsqrt(mean_sq + eps)
    y = x_f32 * inv_rms
    tl.store(y_ptr + base + offs, y.to(x.dtype), mask=mask)


class Model(nn.Module):
    """Simple model that performs RMS Normalization."""

    def __init__(self, num_features: int, eps: float = 1e-5):
        super(Model, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert x.is_cuda, "Input must be CUDA tensor"
        assert x.shape[1] == self.num_features
        if not x.is_contiguous():
            x = x.contiguous()

        b, f, d1, d2 = x.shape
        x_view = x.permute(0, 2, 3, 1).contiguous().view(-1, f)
        y_view = torch.empty_like(x_view)

        outer_size = x_view.shape[0]
        block_feat = triton.next_power_of_2(f)
        _rmsnorm_kernel[(outer_size,)](
            x_view,
            y_view,
            outer_size,
            f,
            self.eps,
            BLOCK_FEAT=block_feat,
            num_warps=1,
            num_stages=2,
        )
        return y_view.view(b, d1, d2, f).permute(0, 3, 1, 2).contiguous()


batch_size = 16
features = 64
dim1 = 256
dim2 = 256


def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [features]
