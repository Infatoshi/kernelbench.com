import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// GELU approximation (fast)
__device__ __forceinline__ half gelu_half(half x) {
    float fx = __half2float(x);
    float t = tanhf(0.7978845608f * (fx + 0.044715f * fx * fx * fx));
    return __float2half(0.5f * fx * (1.0f + t));
}

// GELU for float
__device__ __forceinline__ float gelu_float(float x) {
    float t = tanhf(0.7978845608f * (x + 0.044715f * x * x * x));
    return 0.5f * x * (1.0f + t);
}

// Best version: Uses tensor core style computation with better memory access
__launch_bounds__(256, 2)
__global__ void best_fused_kernel(
    const half* __restrict__ weight,
    const half* __restrict__ bias,
    const half* __restrict__ input,
    float* __restrict__ output,
    int batch, int K, int N,
    float scale_factor
) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    
    // Each thread handles multiple output elements
    float max_val = -1e10f;
    
    // Process in chunks for better cache utilization
    const int ELEMS_PER_THREAD = (N + 255) / 256;
    
    #pragma unroll 4
    for (int e = 0; e < ELEMS_PER_THREAD; e++) {
        int col = tid + e * 256;
        if (col >= N) continue;
        
        half result = __float2half(0.0f);
        
        // Dot product with unrolling
        int k = 0;
        for (; k + 3 < K; k += 4) {
            half a0 = input[b * K + k];
            half a1 = input[b * K + k + 1];
            half a2 = input[b * K + k + 2];
            half a3 = input[b * K + k + 3];
            
            half w0 = __ldg(&weight[(k + 0) * N + col]);
            half w1 = __ldg(&weight[(k + 1) * N + col]);
            half w2 = __ldg(&weight[(k + 2) * N + col]);
            half w3 = __ldg(&weight[(k + 3) * N + col]);
            
            result = __hfma(a0, w0, result);
            result = __hfma(a1, w1, result);
            result = __hfma(a2, w2, result);
            result = __hfma(a3, w3, result);
        }
        
        // Remainder
        for (; k < K; k++) {
            result = __hfma(input[b * K + k], __ldg(&weight[k * N + col]), result);
        }
        
        // Add bias
        if (bias != nullptr) {
            result = __hadd(result, __ldg(&bias[col]));
        }
        
        // GELU
        result = gelu_half(result);
        
        // Scale
        result = __hmul(result, __float2half(scale_factor));
        
        float f = __half2float(result);
        max_val = fmaxf(max_val, f);
    }
    
    // Shared memory for final reduction
    extern __shared__ float smem[];
    smem[tid] = max_val;
    __syncthreads();
    
    // Parallel reduction
    for (int s = blockDim.x / 2; s >= 32; s >>= 1) {
        if (tid < s) {
            smem[tid] = fmaxf(smem[tid], smem[tid + s]);
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[b] = smem[0];
    }
}

// Float version for fp32 precision
__launch_bounds__(256, 2)
__global__ void best_fused_kernel_float(
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ input,
    float* __restrict__ output,
    int batch, int K, int N,
    float scale_factor
) {
    int b = blockIdx.x;
    int tid = threadIdx.x;
    
    float max_val = -1e10f;
    
    const int ELEMS_PER_THREAD = (N + 255) / 256;
    
    #pragma unroll 4
    for (int e = 0; e < ELEMS_PER_THREAD; e++) {
        int col = tid + e * 256;
        if (col >= N) continue;
        
        float result = 0.0f;
        
        int k = 0;
        for (; k + 3 < K; k += 4) {
            float a0 = input[b * K + k];
            float a1 = input[b * K + k + 1];
            float a2 = input[b * K + k + 2];
            float a3 = input[b * K + k + 3];
            
            float w0 = __ldg(&weight[(k + 0) * N + col]);
            float w1 = __ldg(&weight[(k + 1) * N + col]);
            float w2 = __ldg(&weight[(k + 2) * N + col]);
            float w3 = __ldg(&weight[(k + 3) * N + col]);
            
            result += a0 * w0;
            result += a1 * w1;
            result += a2 * w2;
            result += a3 * w3;
        }
        
        for (; k < K; k++) {
            result += input[b * K + k] * __ldg(&weight[k * N + col]);
        }
        
        if (bias != nullptr) {
            result += __ldg(&bias[col]);
        }
        
        result = gelu_float(result);
        result *= scale_factor;
        
        max_val = fmaxf(max_val, result);
    }
    
    extern __shared__ float smem[];
    smem[tid] = max_val;
    __syncthreads();
    
    for (int s = blockDim.x / 2; s >= 32; s >>= 1) {
        if (tid < s) {
            smem[tid] = fmaxf(smem[tid], smem[tid + s]);
        }
    }
    
    if (tid == 0) {
        output[b] = smem[0];
    }
}

torch::Tensor fused_forward_half(
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor input,
    float scale_factor
) {
    int batch = input.size(0);
    int K = input.size(1);
    int N = weight.size(0);
    
    auto output = torch::empty({batch}, input.options().dtype(torch::kFloat32));
    
    const int threads = 256;
    int shared_mem = threads * sizeof(float);
    
    best_fused_kernel<<<batch, threads, shared_mem>>>(
        reinterpret_cast<const half*>(weight.data_ptr<at::Half>()),
        bias.defined() ? reinterpret_cast<const half*>(bias.data_ptr<at::Half>()) : nullptr,
        reinterpret_cast<const half*>(input.data_ptr<at::Half>()),
        output.data_ptr<float>(),
        batch, K, N,
        scale_factor
    );
    
    return output;
}

torch::Tensor fused_forward_float(
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor input,
    float scale_factor
) {
    int batch = input.size(0);
    int K = input.size(1);
    int N = weight.size(0);
    
    auto output = torch::empty({batch}, input.options());
    
    const int threads = 256;
    int shared_mem = threads * sizeof(float);
    
    best_fused_kernel_float<<<batch, threads, shared_mem>>>(
        weight.data_ptr<float>(),
        bias.defined() ? bias.data_ptr<float>() : nullptr,
        input.data_ptr<float>(),
        output.data_ptr<float>(),
        batch, K, N,
        scale_factor
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources='''
torch::Tensor fused_forward_half(torch::Tensor weight, torch::Tensor bias, torch::Tensor input, float scale_factor);
torch::Tensor fused_forward_float(torch::Tensor weight, torch::Tensor bias, torch::Tensor input, float scale_factor);
''',
    cuda_sources=cuda_source,
    functions=['fused_forward_half', 'fused_forward_float'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor, precision='fp16'):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor
        self.precision = precision
        
        # Create weight and bias tensors directly
        if precision == 'fp16':
            self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float16))
            self.bias = nn.Parameter(torch.randn(out_features, dtype=torch.float16))
        elif precision == 'bf16':
            self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.bfloat16))
            self.bias = nn.Parameter(torch.randn(out_features, dtype=torch.bfloat16))
        else:
            self.weight = nn.Parameter(torch.randn(out_features, in_features, dtype=torch.float32))
            self.bias = nn.Parameter(torch.randn(out_features, dtype=torch.float32))
        
        # Initialize like nn.Linear
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        nn.init.zeros_(self.bias)
    
    def forward(self, x):
        if self.precision == 'fp16':
            x = x.to(torch.float16)
            weight = self.weight
            bias = self.bias
            return ext.fused_forward_half(weight, bias, x, self.scale_factor)
        elif self.precision == 'bf16':
            x = x.to(torch.bfloat16)
            weight = self.weight.to(torch.bfloat16)
            bias = self.bias.to(torch.bfloat16)
            # For bf16, convert to fp16 kernel
            return ext.fused_forward_half(weight.to(torch.float16), bias.to(torch.float16), x.to(torch.float16), self.scale_factor)
        else:
            weight = self.weight
            bias = self.bias
            return ext.fused_forward_float(weight, bias, x, self.scale_factor)

def get_inputs():
    return [torch.randn(128, 4096)]

def get_init_inputs():
    return [4096, 4096, 4, 2.0]
