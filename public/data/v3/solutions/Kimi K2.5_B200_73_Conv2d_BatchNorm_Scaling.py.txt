import torch
import torch.nn as nn
import triton
import triton.language as tl

# Module-level config
_IN_CHANNELS = 3
_OUT_CHANNELS = 16
_HEIGHT = 256
_WIDTH = 256
_KERNEL_SIZE = 3
_SCALING_FACTOR = 2.0
_BATCH_SIZE = 16

@triton.jit
def conv3x3_kernel(
    input_ptr, weight_ptr, bias_ptr, output_ptr,
    N, Co, Ho, Wo, H, W,
    BLOCK_H: tl.constexpr, BLOCK_W: tl.constexpr
):
    # Compute output coordinates
    n = tl.program_id(2) // Co
    co = tl.program_id(2) % Co
    h_out = tl.program_id(0) * BLOCK_H + tl.arange(0, BLOCK_H)
    w_out = tl.program_id(1) * BLOCK_W + tl.arange(0, BLOCK_W)
    
    # Mask for valid output pixels
    h_mask = h_out < Ho
    w_mask = w_out < Wo
    
    # Accumulator
    acc = tl.zeros((BLOCK_H, BLOCK_W), dtype=tl.float32)
    
    # Convolution computation
    for ci in range(3):
        for kh in range(3):
            for kw in range(3):
                h_in = h_out + kh
                w_in = w_out + kw
                
                # Load input
                in_offs = ((n * 3 + ci) * H + h_in[:, None]) * W + w_in[None, :]
                in_mask = (h_in[:, None] < H) & (w_in[None, :] < W) & h_mask[:, None] & w_mask[None, :]
                ival = tl.load(input_ptr + in_offs, mask=in_mask, other=0.0)
                
                # Load weight
                wval = tl.load(weight_ptr + ((co * 3 + ci) * 3 + kh) * 3 + kw)
                
                acc += ival * wval
    
    # Add bias
    bval = tl.load(bias_ptr + co)
    acc += bval
    
    # Store output
    out_offs = ((n * Co + co) * Ho + h_out[:, None]) * Wo + w_out[None, :]
    out_mask = h_mask[:, None] & w_mask[None, :]
    tl.store(output_ptr + out_offs, acc, mask=out_mask)


def conv3x3_triton(input, weight, bias):
    N, C, H, W = input.shape
    Co = weight.shape[0]
    Ho, Wo = H - 2, W - 2
    
    output = torch.empty(N, Co, Ho, Wo, device=input.device, dtype=input.dtype)
    
    BLOCK_H = 16
    BLOCK_W = 16
    
    grid = (triton.cdiv(Ho, BLOCK_H), triton.cdiv(Wo, BLOCK_W), N * Co)
    
    conv3x3_kernel[grid](
        input, weight, bias, output,
        N, Co, Ho, Wo, H, W,
        BLOCK_H=BLOCK_H, BLOCK_W=BLOCK_W
    )
    
    return output


class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor
        
        # Register buffers for fused params
        self.register_buffer('w_fused', torch.empty(out_channels, in_channels, kernel_size, kernel_size))
        self.register_buffer('b_fused', torch.empty(out_channels))
        self._weights_fused = False
        
    def _fuse_weights(self):
        if self._weights_fused:
            return
        with torch.no_grad():
            std = torch.sqrt(self.bn.running_var + self.bn.eps)
            s = self.bn.weight / std * self.scaling_factor
            self.w_fused.copy_(self.conv.weight * s.view(-1, 1, 1, 1))
            b = (self.conv.bias - self.bn.running_mean) / std * self.bn.weight + self.bn.bias
            self.b_fused.copy_(b * self.scaling_factor)
        self._weights_fused = True
        
    def forward(self, x):
        if self.training:
            x = self.conv(x)
            x = self.bn(x)
            return x * self.scaling_factor
        
        self._fuse_weights()
        
        # Triton kernel expects float32 for better precision
        if x.dtype == torch.float32:
            return conv3x3_triton(x, self.w_fused, self.b_fused)
        else:
            # Convert to float32 and back
            orig_dtype = x.dtype
            out_f32 = conv3x3_triton(x.float(), self.w_fused.float(), self.b_fused.float())
            return out_f32.to(orig_dtype)
    
    def train(self, mode=True):
        super().train(mode)
        if mode:
            self._weights_fused = False
        return self


def get_inputs():
    return [torch.randn(_BATCH_SIZE, _IN_CHANNELS, _HEIGHT, _WIDTH)]


def get_init_inputs():
    return [_IN_CHANNELS, _OUT_CHANNELS, _KERNEL_SIZE, _SCALING_FACTOR]
