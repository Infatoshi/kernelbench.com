import torch
import torch.nn as nn
import triton
import triton.language as tl

# Conv2D as implicit GEMM:
# Input: (N, C_in, H, W), Weight: (C_out, C_in, K, K)
# Output: (N, C_out, H_out, W_out)
# Reshape as: M = N * H_out * W_out, K_inner = C_in * K * K, N_inner = C_out
# A[m, k] = im2col(input)[m, k], B[k, n] = weight_reshaped[k, n]
# C[m, n] = sum_k A[m,k] * B[k,n]

@triton.jit
def conv2d_implicit_gemm_kernel(
    input_ptr, weight_ptr, output_ptr,
    # Input dimensions
    N, C_in, H, W,
    # Output dimensions
    C_out, H_out, W_out,
    # Kernel params
    K, stride, padding, dilation,
    # GEMM dimensions: M = N*H_out*W_out, K_inner = C_in*K*K, N_inner = C_out
    M, K_inner, N_inner,
    # Block sizes
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N_inner, BLOCK_N)
    
    pid_m = pid % num_pid_m
    pid_n = pid // num_pid_m
    
    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    # Accumulator
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    # Loop over K_inner dimension
    for k_start in range(0, K_inner, BLOCK_K):
        k_offs = k_start + offs_k
        
        # Load A (im2col): need to compute input indices from (m, k)
        # m -> (batch_idx, oh, ow)
        # k -> (c_in, kh, kw)
        m_idx = offs_m  # [BLOCK_M]
        
        batch_idx = m_idx // (H_out * W_out)
        rem = m_idx % (H_out * W_out)
        oh = rem // W_out
        ow = rem % W_out
        
        # For each k in k_offs
        c_in_idx = k_offs // (K * K)
        rem_k = k_offs % (K * K)
        kh = rem_k // K
        kw = rem_k % K
        
        # Input coordinates: ih = oh * stride - padding + kh * dilation
        # Compute for each (m, k) pair
        ih = oh[:, None] * stride - padding + kh[None, :] * dilation  # [BLOCK_M, BLOCK_K]
        iw = ow[:, None] * stride - padding + kw[None, :] * dilation  # [BLOCK_M, BLOCK_K]
        
        # Input pointer: input[batch, c_in, ih, iw]
        input_offset = batch_idx[:, None] * (C_in * H * W) + c_in_idx[None, :] * (H * W) + ih * W + iw
        
        # Mask for valid indices
        mask_m = offs_m[:, None] < M
        mask_k = k_offs[None, :] < K_inner
        mask_h = (ih >= 0) & (ih < H)
        mask_w = (iw >= 0) & (iw < W)
        mask_a = mask_m & mask_k & mask_h & mask_w
        
        a = tl.load(input_ptr + input_offset, mask=mask_a, other=0.0)
        
        # Load B (weight): weight[c_out, c_in, kh, kw] -> weight_reshaped[k, n]
        # k = c_in * K * K + kh * K + kw, n = c_out
        weight_offset = offs_n[None, :] * K_inner + k_offs[:, None]
        mask_b = (k_offs[:, None] < K_inner) & (offs_n[None, :] < N_inner)
        b = tl.load(weight_ptr + weight_offset, mask=mask_b, other=0.0)
        
        acc += tl.dot(a, b)
    
    # Store output
    # output[batch, c_out, oh, ow] = C[m, n]
    # m = batch * H_out * W_out + oh * W_out + ow, n = c_out
    out_batch = offs_m // (H_out * W_out)
    out_rem = offs_m % (H_out * W_out)
    out_oh = out_rem // W_out
    out_ow = out_rem % W_out
    
    out_offset = out_batch[:, None] * (C_out * H_out * W_out) + offs_n[None, :] * (H_out * W_out) + out_oh[:, None] * W_out + out_ow[:, None]
    
    mask_out = (offs_m[:, None] < M) & (offs_n[None, :] < N_inner)
    tl.store(output_ptr + out_offset, acc, mask=mask_out)


def conv2d_triton(input, weight, stride=1, padding=0, dilation=1):
    N, C_in, H, W = input.shape
    C_out, C_in_w, K, K2 = weight.shape
    assert K == K2
    assert C_in == C_in_w
    
    H_out = (H + 2 * padding - dilation * (K - 1) - 1) // stride + 1
    W_out = (W + 2 * padding - dilation * (K - 1) - 1) // stride + 1
    
    output = torch.empty(N, C_out, H_out, W_out, device=input.device, dtype=input.dtype)
    
    M = N * H_out * W_out
    K_inner = C_in * K * K
    N_inner = C_out
    
    # Choose block sizes
    BLOCK_M = 128
    BLOCK_N = 64
    BLOCK_K = 32  # K_inner = 27, so we need BLOCK_K >= 27, use 32
    
    grid = (triton.cdiv(M, BLOCK_M) * triton.cdiv(N_inner, BLOCK_N),)
    
    conv2d_implicit_gemm_kernel[grid](
        input, weight, output,
        N, C_in, H, W,
        C_out, H_out, W_out,
        K, stride, padding, dilation,
        M, K_inner, N_inner,
        BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=BLOCK_K,
    )
    
    return output


class Model(nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(Model, self).__init__()
        self.conv2d = nn.Conv2d(in_channels, out_channels, (kernel_size, kernel_size), stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return conv2d_triton(x, self.conv2d.weight, self.stride, self.padding, self.dilation)
