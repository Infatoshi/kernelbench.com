"""
FP4 GEMM optimized for B200 (Blackwell SM100) using CUDA.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
import os

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Optimized FP4 GEMM kernel for B200
// Uses tiled approach with shared memory
__global__ void fp4_gemm_kernel(
    const int8_t* __restrict__ A,
    const int8_t* __restrict__ B,
    half* __restrict__ C,
    float scale_a,
    float scale_b,
    int M, int N, int K
) {
    const float scale = scale_a * scale_b;
    
    constexpr int BLOCK_M = 64;
    constexpr int BLOCK_N = 64;
    constexpr int BLOCK_K = 32;
    constexpr int THREADS = 256;
    
    int tile_m = blockIdx.y;
    int tile_n = blockIdx.x;
    
    int tid = threadIdx.x;
    
    extern __shared__ __align__(4096) unsigned char smem_raw[];
    half* As = reinterpret_cast<half*>(smem_raw);
    half* Bs = reinterpret_cast<half*>(&As[BLOCK_M * BLOCK_K]);
    
    // Each thread computes one or more output elements
    // BLOCK_M x BLOCK_N = 64x64 = 4096 outputs
    // 256 threads -> 16 outputs per thread
    
    float acc[16] = {0.0f};
    
    // Thread's output positions
    int outputs_per_thread = (BLOCK_M * BLOCK_N) / THREADS;  // 16
    
    for (int kb = 0; kb < K; kb += BLOCK_K) {
        // Load A tile: BLOCK_M x BLOCK_K
        for (int idx = tid; idx < BLOCK_M * BLOCK_K; idx += THREADS) {
            int i = idx / BLOCK_K;
            int k = idx % BLOCK_K;
            int gm = tile_m * BLOCK_M + i;
            int gk = kb + k;
            if (gm < M && gk < K) {
                As[idx] = __float2half((float)A[gm * K + gk] * scale_a);
            } else {
                As[idx] = __float2half(0.0f);
            }
        }
        
        // Load B tile: BLOCK_K x BLOCK_N
        for (int idx = tid; idx < BLOCK_K * BLOCK_N; idx += THREADS) {
            int k = idx / BLOCK_N;
            int j = idx % BLOCK_N;
            int gk = kb + k;
            int gn = tile_n * BLOCK_N + j;
            if (gk < K && gn < N) {
                Bs[idx] = __float2half((float)B[gk * N + gn] * scale_b);
            } else {
                Bs[idx] = __float2half(0.0f);
            }
        }
        
        __syncthreads();
        
        // Compute: each thread computes its assigned outputs
        for (int out_idx = 0; out_idx < outputs_per_thread; out_idx++) {
            int global_out = tid * outputs_per_thread + out_idx;
            int row = global_out / BLOCK_N;
            int col = global_out % BLOCK_N;
            
            float local_acc = 0.0f;
            for (int k = 0; k < BLOCK_K; k++) {
                half a = As[row * BLOCK_K + k];
                half b = Bs[k * BLOCK_N + col];
                local_acc += __half2float(a) * __half2float(b);
            }
            acc[out_idx] = local_acc;
        }
        
        __syncthreads();
    }
    
    // Store results
    for (int out_idx = 0; out_idx < outputs_per_thread; out_idx++) {
        int global_out = tid * outputs_per_thread + out_idx;
        int row = global_out / BLOCK_N;
        int col = global_out % BLOCK_N;
        
        int gm = tile_m * BLOCK_M + row;
        int gn = tile_n * BLOCK_N + col;
        
        if (gm < M && gn < N) {
            C[gm * N + gn] = __float2half(acc[out_idx] * scale);
        }
    }
}

torch::Tensor fp4_gemm(
    torch::Tensor a_q,
    torch::Tensor b_q,
    torch::Tensor scale_a,
    torch::Tensor scale_b
) {
    int M = a_q.size(0);
    int K = a_q.size(1);
    int N = b_q.size(1);
    
    auto options = torch::TensorOptions()
        .dtype(torch::kFloat16)
        .device(a_q.device())
        .layout(torch::kStrided);
    
    torch::Tensor c = torch::empty({M, N}, options);
    
    float sa = scale_a.item<float>();
    float sb = scale_b.item<float>();
    
    // Configure kernel launch
    constexpr int BLOCK_M = 64;
    constexpr int BLOCK_N = 64;
    constexpr int BLOCK_K = 32;
    constexpr int THREADS = 256;
    
    dim3 grid((N + BLOCK_N - 1) / BLOCK_N, (M + BLOCK_M - 1) / BLOCK_M);
    dim3 block(THREADS);
    
    int smem_size = (BLOCK_M * BLOCK_K + BLOCK_K * BLOCK_N) * sizeof(half);
    
    fp4_gemm_kernel<<<grid, block, smem_size>>>(
        a_q.data_ptr<int8_t>(),
        b_q.data_ptr<int8_t>(),
        reinterpret_cast<half*>(c.data_ptr<c10::Half>()),
        sa, sb, M, N, K
    );
    
    return c;
}
'''

cpp_source = r'''
#include <torch/extension.h>
torch::Tensor fp4_gemm(torch::Tensor a_q, torch::Tensor b_q, torch::Tensor scale_a, torch::Tensor scale_b);
'''

# Compile the CUDA extension
ext = load_inline(
    name='fp4_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fp4_gemm'],
    extra_cuda_cflags=[
        '-arch=sm_100',
        '-O3',
        '-std=c++17',
        '--use_fast_math'
    ],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, m: int = 2048, n: int = 2048, k: int = 2048):
        super().__init__()
        self.m = m
        self.n = n
        self.k = k

    def forward(
        self,
        a_q: torch.Tensor,
        b_q: torch.Tensor,
        scale_a: torch.Tensor,
        scale_b: torch.Tensor,
    ) -> torch.Tensor:
        return ext.fp4_gemm(a_q, b_q, scale_a, scale_b)


def get_inputs():
    m = 2048
    n = 2048
    k = 2048
    a_q = torch.randint(-8, 8, (m, k), dtype=torch.int8)
    b_q = torch.randint(-8, 8, (k, n), dtype=torch.int8)
    scale_a = torch.tensor(0.08, dtype=torch.float32)
    scale_b = torch.tensor(0.08, dtype=torch.float32)
    return [a_q, b_q, scale_a, scale_b]


def get_init_inputs():
    return []
