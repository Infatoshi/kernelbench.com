import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// -----------------------------------------------------------------------
// FP32 kernel: processes 2 particles (4 floats pos + 4 floats vel) per thread
// using float4 loads/stores for maximum memory bandwidth.
// Each particle has 2D position + 2D velocity = 4 floats.
// We pack [pos.x, pos.y, vel.x, vel.y] per particle.
// Output: [new_pos.x, new_pos.y, new_vel.x, new_vel.y] per particle.
// -----------------------------------------------------------------------
__global__ void particle_step_fp32(
    const float* __restrict__ pos,    // [N, 2]
    const float* __restrict__ vel,    // [N, 2]
    float* __restrict__ out,          // [N, 4]  interleaved [pos, vel]
    float dt,
    float gx, float gy,
    float damping,
    int N
) {
    // Process 2 particles per thread using float4
    int tid = blockIdx.x * blockDim.x + threadIdx.x;
    int base = tid * 2;  // 2 particles per thread
    
    if (base >= N) return;
    
    // Load positions: 2 particles = 4 floats
    float4 p;
    if (base + 1 < N) {
        p = *reinterpret_cast<const float4*>(pos + base * 2);
    } else {
        // Last odd particle
        p.x = pos[base * 2 + 0];
        p.y = pos[base * 2 + 1];
        p.z = 0.0f; p.w = 0.0f;
    }
    
    // Load velocities: 2 particles = 4 floats
    float4 v;
    if (base + 1 < N) {
        v = *reinterpret_cast<const float4*>(vel + base * 2);
    } else {
        v.x = vel[base * 2 + 0];
        v.y = vel[base * 2 + 1];
        v.z = 0.0f; v.w = 0.0f;
    }
    
    // Integrate particle 0
    float vx0 = v.x + gx * dt;
    float vy0 = v.y + gy * dt;
    float px0 = p.x + vx0 * dt;
    float py0 = p.y + vy0 * dt;
    
    // Bounce particle 0
    if (px0 > 1.0f || px0 < -1.0f) { vx0 = -vx0 * damping; }
    if (py0 > 1.0f || py0 < -1.0f) { vy0 = -vy0 * damping; }
    px0 = fmaxf(-1.0f, fminf(1.0f, px0));
    py0 = fmaxf(-1.0f, fminf(1.0f, py0));
    
    // Write output particle 0: [px, py, vx, vy]
    // Output layout: first N*2 floats = positions, next N*2 floats = velocities
    // Actually reference uses torch.cat([new_pos, new_vel], dim=-1)
    // so output[i] = [pos_x, pos_y, vel_x, vel_y] for particle i
    out[base * 4 + 0] = px0;
    out[base * 4 + 1] = py0;
    out[base * 4 + 2] = vx0;
    out[base * 4 + 3] = vy0;
    
    if (base + 1 < N) {
        // Integrate particle 1
        float vx1 = v.z + gx * dt;
        float vy1 = v.w + gy * dt;
        float px1 = p.z + vx1 * dt;
        float py1 = p.w + vy1 * dt;
        
        // Bounce particle 1
        if (px1 > 1.0f || px1 < -1.0f) { vx1 = -vx1 * damping; }
        if (py1 > 1.0f || py1 < -1.0f) { vy1 = -vy1 * damping; }
        px1 = fmaxf(-1.0f, fminf(1.0f, px1));
        py1 = fmaxf(-1.0f, fminf(1.0f, py1));
        
        out[(base + 1) * 4 + 0] = px1;
        out[(base + 1) * 4 + 1] = py1;
        out[(base + 1) * 4 + 2] = vx1;
        out[(base + 1) * 4 + 3] = vy1;
    }
}

// -----------------------------------------------------------------------
// Scalar FP32 kernel for when N is small or alignment issues
// -----------------------------------------------------------------------
__global__ void particle_step_fp32_scalar(
    const float* __restrict__ pos,
    const float* __restrict__ vel,
    float* __restrict__ out,
    float dt,
    float gx, float gy,
    float damping,
    int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    
    float px = pos[i * 2 + 0];
    float py = pos[i * 2 + 1];
    float vx = vel[i * 2 + 0];
    float vy = vel[i * 2 + 1];
    
    vx += gx * dt;
    vy += gy * dt;
    px += vx * dt;
    py += vy * dt;
    
    if (px > 1.0f || px < -1.0f) { vx = -vx * damping; }
    if (py > 1.0f || py < -1.0f) { vy = -vy * damping; }
    px = fmaxf(-1.0f, fminf(1.0f, px));
    py = fmaxf(-1.0f, fminf(1.0f, py));
    
    out[i * 4 + 0] = px;
    out[i * 4 + 1] = py;
    out[i * 4 + 2] = vx;
    out[i * 4 + 3] = vy;
}


// -----------------------------------------------------------------------
// FP16 kernel
// -----------------------------------------------------------------------
__global__ void particle_step_fp16(
    const __half* __restrict__ pos,
    const __half* __restrict__ vel,
    __half* __restrict__ out,
    float dt,
    float gx, float gy,
    float damping,
    int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    
    float px = __half2float(pos[i * 2 + 0]);
    float py = __half2float(pos[i * 2 + 1]);
    float vx = __half2float(vel[i * 2 + 0]);
    float vy = __half2float(vel[i * 2 + 1]);
    
    vx += gx * dt;
    vy += gy * dt;
    px += vx * dt;
    py += vy * dt;
    
    if (px > 1.0f || px < -1.0f) { vx = -vx * damping; }
    if (py > 1.0f || py < -1.0f) { vy = -vy * damping; }
    px = fmaxf(-1.0f, fminf(1.0f, px));
    py = fmaxf(-1.0f, fminf(1.0f, py));
    
    out[i * 4 + 0] = __float2half(px);
    out[i * 4 + 1] = __float2half(py);
    out[i * 4 + 2] = __float2half(vx);
    out[i * 4 + 3] = __float2half(vy);
}

// -----------------------------------------------------------------------
// BF16 kernel
// -----------------------------------------------------------------------
__global__ void particle_step_bf16(
    const __nv_bfloat16* __restrict__ pos,
    const __nv_bfloat16* __restrict__ vel,
    __nv_bfloat16* __restrict__ out,
    float dt,
    float gx, float gy,
    float damping,
    int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    
    float px = __bfloat162float(pos[i * 2 + 0]);
    float py = __bfloat162float(pos[i * 2 + 1]);
    float vx = __bfloat162float(vel[i * 2 + 0]);
    float vy = __bfloat162float(vel[i * 2 + 1]);
    
    vx += gx * dt;
    vy += gy * dt;
    px += vx * dt;
    py += vy * dt;
    
    if (px > 1.0f || px < -1.0f) { vx = -vx * damping; }
    if (py > 1.0f || py < -1.0f) { vy = -vy * damping; }
    px = fmaxf(-1.0f, fminf(1.0f, px));
    py = fmaxf(-1.0f, fminf(1.0f, py));
    
    out[i * 4 + 0] = __float2bfloat16(px);
    out[i * 4 + 1] = __float2bfloat16(py);
    out[i * 4 + 2] = __float2bfloat16(vx);
    out[i * 4 + 3] = __float2bfloat16(vy);
}


torch::Tensor particle_step(
    torch::Tensor positions,
    torch::Tensor velocities,
    torch::Tensor dt_tensor,
    torch::Tensor gravity_tensor,
    float damping
) {
    int N = positions.size(0);
    float dt = dt_tensor.item<float>();
    float gx = gravity_tensor[0].item<float>();
    float gy = gravity_tensor[1].item<float>();
    
    auto dtype = positions.dtype();
    // Output: [N, 4] with [pos_x, pos_y, vel_x, vel_y]
    auto out = torch::empty({N, 4}, positions.options());
    
    const int BLOCK = 256;
    
    if (dtype == torch::kFloat32) {
        // Use float4 kernel (2 particles per thread)
        int threads_needed = (N + 1) / 2;
        int blocks = (threads_needed + BLOCK - 1) / BLOCK;
        particle_step_fp32<<<blocks, BLOCK>>>(
            positions.data_ptr<float>(),
            velocities.data_ptr<float>(),
            out.data_ptr<float>(),
            dt, gx, gy, damping, N
        );
    } else if (dtype == torch::kFloat16) {
        int blocks = (N + BLOCK - 1) / BLOCK;
        particle_step_fp16<<<blocks, BLOCK>>>(
            reinterpret_cast<const __half*>(positions.data_ptr<at::Half>()),
            reinterpret_cast<const __half*>(velocities.data_ptr<at::Half>()),
            reinterpret_cast<__half*>(out.data_ptr<at::Half>()),
            dt, gx, gy, damping, N
        );
    } else if (dtype == torch::kBFloat16) {
        int blocks = (N + BLOCK - 1) / BLOCK;
        particle_step_bf16<<<blocks, BLOCK>>>(
            reinterpret_cast<const __nv_bfloat16*>(positions.data_ptr<at::BFloat16>()),
            reinterpret_cast<const __nv_bfloat16*>(velocities.data_ptr<at::BFloat16>()),
            reinterpret_cast<__nv_bfloat16*>(out.data_ptr<at::BFloat16>()),
            dt, gx, gy, damping, N
        );
    } else {
        // Fallback: scalar fp32
        int blocks = (N + BLOCK - 1) / BLOCK;
        particle_step_fp32_scalar<<<blocks, BLOCK>>>(
            positions.data_ptr<float>(),
            velocities.data_ptr<float>(),
            out.data_ptr<float>(),
            dt, gx, gy, damping, N
        );
    }
    
    return out;
}
'''

cpp_source = '''
torch::Tensor particle_step(
    torch::Tensor positions,
    torch::Tensor velocities,
    torch::Tensor dt_tensor,
    torch::Tensor gravity_tensor,
    float damping
);
'''

ext = load_inline(
    name='particle_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['particle_step'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
)


class Model(nn.Module):
    """Optimized GPU particle integration step with boundary collisions."""

    def __init__(self, damping: float = 0.9):
        super().__init__()
        self.damping = damping

    def forward(
        self,
        positions: torch.Tensor,
        velocities: torch.Tensor,
        dt: torch.Tensor,
        gravity: torch.Tensor,
    ) -> torch.Tensor:
        # Ensure gravity is on the same device
        if not gravity.is_cuda:
            gravity = gravity.to(positions.device)
        if not dt.is_cuda:
            dt = dt.to(positions.device)
        
        # Returns [N, 4] with columns [pos_x, pos_y, vel_x, vel_y]
        # Reference returns torch.cat([new_pos, new_vel], dim=-1) which is same layout
        out = ext.particle_step(positions, velocities, dt, gravity, self.damping)
        return out


def get_inputs():
    n = 1_048_576
    positions = torch.rand(n, 2) * 2.0 - 1.0
    velocities = torch.randn(n, 2) * 0.05
    dt = torch.tensor(1.0 / 60.0)
    gravity = torch.tensor([0.0, -9.81])
    return [positions, velocities, dt, gravity]


def get_init_inputs():
    return [0.9]
