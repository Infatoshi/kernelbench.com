import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cmath>

// GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
__device__ __forceinline__ float fast_gelu(float x) {
    const float sqrt_2_over_pi = 0.7978845608f;
    const float coef = 0.044715f;
    float x3 = x * x * x;
    float inner = sqrt_2_over_pi * (x + coef * x3);
    float tanh_inner = tanhf(inner);
    return 0.5f * x * (1.0f + tanh_inner);
}

// Optimized GEMM with shared memory tiling - each block computes one output element
__global__ void optimized_gemm_bias_kernel(
    const float* __restrict__ input,    // [batch, in_features]
    const float* __restrict__ weight,   // [out_features, in_features]
    const float* __restrict__ bias,     // [out_features]
    float* __restrict__ output,         // [batch, out_features]
    int batch_size,
    int in_features,
    int out_features
) {
    int batch_idx = blockIdx.y;
    int out_idx = blockIdx.x;
    
    if (batch_idx >= batch_size || out_idx >= out_features) return;
    
    // Use shared memory for input row
    extern __shared__ float shared_in[];
    
    float sum = 0.0f;
    int num_tiles = (in_features + 256 - 1) / 256;
    
    for (int t = 0; t < num_tiles; ++t) {
        int k = t * 256 + threadIdx.x;
        if (k < in_features) {
            shared_in[threadIdx.x] = input[batch_idx * in_features + k];
        } else {
            shared_in[threadIdx.x] = 0.0f;
        }
        __syncthreads();
        
        // Each thread accumulates partial sums
        for (int i = 0; i < 256; ++i) {
            int k_idx = t * 256 + i;
            if (k_idx < in_features) {
                sum += shared_in[i] * weight[out_idx * in_features + k_idx];
            }
        }
        __syncthreads();
    }
    
    // Add bias and write
    if (threadIdx.x == 0) {
        float val = sum;
        if (bias != nullptr) {
            val += bias[out_idx];
        }
        output[batch_idx * out_features + out_idx] = val;
    }
}

// Better: 2D block tiling for GEMM
__global__ void tiled_gemm_kernel(
    const float* __restrict__ A,  // [batch, K]
    const float* __restrict__ B,  // [N, K] - row major
    const float* __restrict__ bias,
    float* __restrict__ C,        // [batch, N]
    int M,  // batch
    int K,  // in_features
    int N   // out_features
) {
    const int TILE_SIZE = 32;
    __shared__ float As[TILE_SIZE][TILE_SIZE];
    __shared__ float Bs[TILE_SIZE][TILE_SIZE];
    
    int bx = blockIdx.x;  // output column block
    int by = blockIdx.y;  // output row block (batch)
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    
    int row = by * TILE_SIZE + ty;  // batch index
    int col = bx * TILE_SIZE + tx;  // output feature index
    
    if (row >= M || col >= N) return;
    
    float sum = 0.0f;
    int num_tiles = (K + TILE_SIZE - 1) / TILE_SIZE;
    
    for (int t = 0; t < num_tiles; ++t) {
        int a_col = t * TILE_SIZE + tx;
        int b_row = t * TILE_SIZE + ty;
        
        // Load A[row, a_col]
        As[ty][tx] = (row < M && a_col < K) ? A[row * K + a_col] : 0.0f;
        
        // Load B[b_row, col] - B is [N, K]
        Bs[ty][tx] = (b_row < N && col < K) ? B[b_row * K + col] : 0.0f;
        
        __syncthreads();
        
        // Compute dot product
        for (int k = 0; k < TILE_SIZE; ++k) {
            sum += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    // Add bias and write
    if (row < M && col < N) {
        float val = sum;
        if (bias != nullptr) {
            val += bias[col];
        }
        C[row * N + col] = val;
    }
}

// Kernel 2: AvgPool + GELU + Scale + Max (fused)
__global__ void fused_postprocess_kernel(
    const float* __restrict__ input,  // [batch, out_features]
    float* __restrict__ output,      // [batch]
    int batch_size,
    int out_features,
    int pool_kernel_size,
    float scale_factor
) {
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    int pooled_size = (out_features + pool_kernel_size - 1) / pool_kernel_size;
    float max_val = -FLT_MAX;
    
    for (int i = threadIdx.x; i < pooled_size; i += blockDim.x) {
        // AvgPool
        float sum = 0.0f;
        int count = 0;
        for (int j = 0; j < pool_kernel_size; ++j) {
            int idx = i * pool_kernel_size + j;
            if (idx < out_features) {
                sum += input[batch_idx * out_features + idx];
                count++;
            }
        }
        float avg = sum / count;
        
        // GELU
        float gelu_val = fast_gelu(avg);
        
        // Scale
        float scaled = gelu_val * scale_factor;
        
        // Max reduction
        if (scaled > max_val) {
            max_val = scaled;
        }
    }
    
    // Block-level reduction for max
    extern __shared__ float shared_max[];
    shared_max[threadIdx.x] = max_val;
    __syncthreads();
    
    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            shared_max[threadIdx.x] = fmaxf(shared_max[threadIdx.x], shared_max[threadIdx.x + stride]);
        }
        __syncthreads();
    }
    
    if (threadIdx.x == 0) {
        output[batch_idx] = shared_max[0];
    }
}

torch::Tensor fused_matmul_avgpool_gelu_scale_max(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size,
    float scale_factor
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    
    // Allocate intermediate and output
    auto float_options = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    auto matmul_out = torch::empty({batch_size, out_features}, float_options);
    auto output = torch::empty({batch_size}, float_options);
    
    // Launch tiled GEMM kernel
    const int TILE_SIZE = 32;
    dim3 grid((out_features + TILE_SIZE - 1) / TILE_SIZE, 
              (batch_size + TILE_SIZE - 1) / TILE_SIZE);
    dim3 block(TILE_SIZE, TILE_SIZE);
    
    tiled_gemm_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.data_ptr<float>(),
        matmul_out.data_ptr<float>(),
        batch_size, in_features, out_features
    );
    
    // Launch fused postprocess kernel
    int threads2 = 256;
    int shared_mem2 = threads2 * sizeof(float);
    fused_postprocess_kernel<<<batch_size, threads2, shared_mem2>>>(
        matmul_out.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, out_features, pool_kernel_size, scale_factor
    );
    
    return output;
}
'''

ext = load_inline(
    name='fused_ext',
    cpp_sources='''
#include <torch/extension.h>
torch::Tensor fused_matmul_avgpool_gelu_scale_max(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size,
    float scale_factor
);
''',
    cuda_sources=cuda_source,
    functions=['fused_matmul_avgpool_gelu_scale_max'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)


class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor

    def forward(self, x):
        weight = self.matmul.weight
        bias = self.matmul.bias
        return ext.fused_matmul_avgpool_gelu_scale_max(
            x, weight, bias, self.pool_kernel_size, self.scale_factor
        )


def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features)]


def get_init_inputs():
    return [4096, 4096, 4, 2.0]
