import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Two-pass LayerNorm for large normalized dimensions
// Pass 1: Compute partial sums and partial sum-of-squares per block
// Pass 2: Normalize using the computed mean and variance

// We launch multiple blocks per instance (row).
// Each block processes a chunk of the normalized dimension.

// Warp reduce
__device__ __forceinline__ float warpReduceSum(float val) {
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1)
        val += __shfl_xor_sync(0xffffffff, val, offset);
    return val;
}

// Block reduce using shared memory
__device__ float blockReduceSum(float val) {
    __shared__ float shared[32];
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    val = warpReduceSum(val);
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    int nwarps = (blockDim.x + 31) >> 5;
    val = (threadIdx.x < nwarps) ? shared[threadIdx.x] : 0.0f;
    if (wid == 0) val = warpReduceSum(val);
    return val;
}

// Kernel 1: Compute partial sums for mean and variance
// Grid: (blocks_per_row, batch_size)
// Each block processes a chunk of elements
__global__ void layernorm_partial_sums_kernel(
    const float* __restrict__ input,
    float* __restrict__ partial_sum,
    float* __restrict__ partial_sum_sq,
    int N,  // normalized dimension size
    int blocks_per_row
) {
    int row = blockIdx.y;
    int block_in_row = blockIdx.x;
    
    const float* row_input = input + (size_t)row * N;
    
    int start = block_in_row * blockDim.x + threadIdx.x;
    int stride = blocks_per_row * blockDim.x;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = start; i < N; i += stride) {
        float val = row_input[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    if (threadIdx.x == 0) {
        int idx = row * blocks_per_row + block_in_row;
        partial_sum[idx] = local_sum;
        partial_sum_sq[idx] = local_sum_sq;
    }
}

// Kernel 2: Reduce partial sums and compute mean/invstd
__global__ void layernorm_reduce_stats_kernel(
    const float* __restrict__ partial_sum,
    const float* __restrict__ partial_sum_sq,
    float* __restrict__ mean_out,
    float* __restrict__ invstd_out,
    int blocks_per_row,
    int N,
    float eps
) {
    int row = blockIdx.x;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = threadIdx.x; i < blocks_per_row; i += blockDim.x) {
        int idx = row * blocks_per_row + i;
        local_sum += partial_sum[idx];
        local_sum_sq += partial_sum_sq[idx];
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    if (threadIdx.x == 0) {
        float mean = local_sum / N;
        float var = local_sum_sq / N - mean * mean;
        float invstd = rsqrtf(var + eps);
        mean_out[row] = mean;
        invstd_out[row] = invstd;
    }
}

// Kernel 3: Apply normalization with weight and bias
// Uses vectorized loads (float4)
__global__ void layernorm_apply_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ mean,
    const float* __restrict__ invstd,
    float* __restrict__ output,
    int N,
    int blocks_per_row
) {
    int row = blockIdx.y;
    int block_in_row = blockIdx.x;
    
    float m = mean[row];
    float s = invstd[row];
    
    const float* row_input = input + (size_t)row * N;
    float* row_output = output + (size_t)row * N;
    
    // Vectorized processing with float4
    int N4 = N >> 2;
    const float4* input4 = (const float4*)row_input;
    const float4* weight4 = (const float4*)weight;
    const float4* bias4 = (const float4*)bias;
    float4* output4 = (float4*)row_output;
    
    int start = block_in_row * blockDim.x + threadIdx.x;
    int stride = blocks_per_row * blockDim.x;
    
    for (int i = start; i < N4; i += stride) {
        float4 in_val = input4[i];
        float4 w_val = weight4[i];
        float4 b_val = bias4[i];
        
        float4 out_val;
        out_val.x = (in_val.x - m) * s * w_val.x + b_val.x;
        out_val.y = (in_val.y - m) * s * w_val.y + b_val.y;
        out_val.z = (in_val.z - m) * s * w_val.z + b_val.z;
        out_val.w = (in_val.w - m) * s * w_val.w + b_val.w;
        
        output4[i] = out_val;
    }
    
    // Handle remainder
    int rem_start = N4 * 4 + (block_in_row * blockDim.x + threadIdx.x);
    for (int i = rem_start; i < N; i += stride) {
        row_output[i] = (row_input[i] - m) * s * weight[i] + bias[i];
    }
}

// Also vectorized partial sums kernel
__global__ void layernorm_partial_sums_vec4_kernel(
    const float* __restrict__ input,
    float* __restrict__ partial_sum,
    float* __restrict__ partial_sum_sq,
    int N,
    int blocks_per_row
) {
    int row = blockIdx.y;
    int block_in_row = blockIdx.x;
    
    const float4* row_input4 = (const float4*)(input + (size_t)row * N);
    int N4 = N >> 2;
    
    int start = block_in_row * blockDim.x + threadIdx.x;
    int stride = blocks_per_row * blockDim.x;
    
    float local_sum = 0.0f;
    float local_sum_sq = 0.0f;
    
    for (int i = start; i < N4; i += stride) {
        float4 val = row_input4[i];
        local_sum += val.x + val.y + val.z + val.w;
        local_sum_sq += val.x * val.x + val.y * val.y + val.z * val.z + val.w * val.w;
    }
    
    // Handle remainder
    int rem_start = N4 * 4;
    for (int i = rem_start + start; i < N; i += stride) {
        float val = (input + (size_t)row * N)[i];
        local_sum += val;
        local_sum_sq += val * val;
    }
    
    local_sum = blockReduceSum(local_sum);
    local_sum_sq = blockReduceSum(local_sum_sq);
    
    if (threadIdx.x == 0) {
        int idx = row * blocks_per_row + block_in_row;
        partial_sum[idx] = local_sum;
        partial_sum_sq[idx] = local_sum_sq;
    }
}


torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, double eps) {
    // input shape: (batch_size, *normalized_shape) but flattened as (batch_size, N)
    // Weight and bias shape: (N,) where N = prod(normalized_shape)
    
    auto input_contig = input.contiguous();
    auto weight_contig = weight.contiguous();
    auto bias_contig = bias.contiguous();
    
    int64_t batch_size = input_contig.size(0);
    int64_t N = input_contig.numel() / batch_size;
    
    auto output = torch::empty_like(input_contig);
    
    // Choose blocks_per_row
    const int BLOCK_SIZE = 256;
    // We want enough blocks to keep GPU busy but not too many
    // N = 4194304, so we need many blocks
    int blocks_per_row = 128;  // 128 blocks per row, each block handles ~32768 elements
    
    // Allocate temporary buffers for partial sums
    auto options = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    auto partial_sum = torch::empty({batch_size * blocks_per_row}, options);
    auto partial_sum_sq = torch::empty({batch_size * blocks_per_row}, options);
    auto mean_tensor = torch::empty({batch_size}, options);
    auto invstd_tensor = torch::empty({batch_size}, options);
    
    // Launch kernel 1: partial sums (vectorized)
    dim3 grid1(blocks_per_row, batch_size);
    layernorm_partial_sums_vec4_kernel<<<grid1, BLOCK_SIZE>>>(
        input_contig.data_ptr<float>(),
        partial_sum.data_ptr<float>(),
        partial_sum_sq.data_ptr<float>(),
        N,
        blocks_per_row
    );
    
    // Launch kernel 2: reduce stats
    int reduce_threads = min(blocks_per_row, 256);
    layernorm_reduce_stats_kernel<<<batch_size, reduce_threads>>>(
        partial_sum.data_ptr<float>(),
        partial_sum_sq.data_ptr<float>(),
        mean_tensor.data_ptr<float>(),
        invstd_tensor.data_ptr<float>(),
        blocks_per_row,
        N,
        (float)eps
    );
    
    // Launch kernel 3: apply normalization (vectorized)
    dim3 grid3(blocks_per_row, batch_size);
    layernorm_apply_kernel<<<grid3, BLOCK_SIZE>>>(
        input_contig.data_ptr<float>(),
        weight_contig.data_ptr<float>(),
        bias_contig.data_ptr<float>(),
        mean_tensor.data_ptr<float>(),
        invstd_tensor.data_ptr<float>(),
        output.data_ptr<float>(),
        N,
        blocks_per_row
    );
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, double eps);
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    extra_cuda_cflags=['-O3', '--use_fast_math', '-std=c++17'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)
    
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Reshape: flatten batch dims, keep normalized dims flat
        orig_shape = x.shape
        normalized_numel = 1
        for s in self.ln.normalized_shape:
            normalized_numel *= s
        batch_size = x.numel() // normalized_numel
        
        x_flat = x.contiguous().view(batch_size, normalized_numel)
        
        weight = self.ln.weight.contiguous().view(-1)
        bias = self.ln.bias.contiguous().view(-1)
        
        out = ext.layernorm_forward(x_flat, weight, bias, self.ln.eps)
        return out.view(orig_shape)

batch_size = 16
features = 64
dim1 = 256
dim2 = 256

def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(features, dim1, dim2)]
