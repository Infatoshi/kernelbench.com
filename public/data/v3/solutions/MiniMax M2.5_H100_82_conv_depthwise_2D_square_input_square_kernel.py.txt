import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Depthwise 2D convolution optimized kernel
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <stdio.h>

// Depthwise convolution kernel optimized for H100
// Each channel is processed independently - depthwise property
template <typename scalar_t>
__global__ void depthwise_conv2d_kernel(
    const scalar_t* __restrict__ input,
    const scalar_t* __restrict__ weight,
    scalar_t* __restrict__ output,
    const int batch_size,
    const int channels,
    const int height,
    const int width,
    const int kernel_size,
    const int stride,
    const int padding,
    const int out_height,
    const int out_width) {
    
    // Use block index to determine which output to compute
    // grid: (batch * out_h * out_w, channels)
    const int batch_out_idx = blockIdx.x;
    const int channel_idx = blockIdx.y;
    const int tid = threadIdx.x;
    
    if (channel_idx >= channels) return;
    
    // Calculate batch, out_y, out_x from flat index
    const int out_elements = out_height * out_width;
    const int batch_idx = batch_out_idx / out_elements;
    const int out_flat = batch_out_idx % out_elements;
    const int out_y = out_flat / out_width;
    const int out_x = out_flat % out_width;
    
    if (batch_idx >= batch_size) return;
    
    // Compute the corresponding input region
    const int in_y = out_y * stride - padding;
    const int in_x = out_x * stride - padding;
    const int kernel_offset = kernel_size / 2;
    
    scalar_t sum = 0;
    
    // Each thread computes one output
    // Unrolled loop for kernel_size=3
    if (kernel_size == 3) {
        // ky = -1, 0, 1
        // kx = -1, 0, 1
        const int iy0 = in_y - 1;
        const int ix0 = in_x - 1;
        
        // Weight index base for this channel
        // PyTorch depthwise weight: (channels, 1, kh, kw)
        // So weight[channel_idx] gives filter for channel idx
        const int w_base = channel_idx * kernel_size * kernel_size;
        
        // Position (ky=-1, kx=-1)
        if (iy0 >= 0 && iy0 < height && ix0 >= 0 && ix0 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy0) * width + ix0;
            sum += input[in_idx] * weight[w_base + 0];
        }
        
        // Position (ky=-1, kx=0)
        const int ix1 = in_x;
        if (iy0 >= 0 && iy0 < height && ix1 >= 0 && ix1 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy0) * width + ix1;
            sum += input[in_idx] * weight[w_base + 1];
        }
        
        // Position (ky=-1, kx=1)
        const int ix2 = in_x + 1;
        if (iy0 >= 0 && iy0 < height && ix2 >= 0 && ix2 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy0) * width + ix2;
            sum += input[in_idx] * weight[w_base + 2];
        }
        
        // Position (ky=0, kx=-1)
        const int iy1 = in_y;
        if (iy1 >= 0 && iy1 < height && ix0 >= 0 && ix0 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy1) * width + ix0;
            sum += input[in_idx] * weight[w_base + 3];
        }
        
        // Position (ky=0, kx=0)
        if (iy1 >= 0 && iy1 < height && ix1 >= 0 && ix1 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy1) * width + ix1;
            sum += input[in_idx] * weight[w_base + 4];
        }
        
        // Position (ky=0, kx=1)
        if (iy1 >= 0 && iy1 < height && ix2 >= 0 && ix2 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy1) * width + ix2;
            sum += input[in_idx] * weight[w_base + 5];
        }
        
        // Position (ky=1, kx=-1)
        const int iy2 = in_y + 1;
        if (iy2 >= 0 && iy2 < height && ix0 >= 0 && ix0 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy2) * width + ix0;
            sum += input[in_idx] * weight[w_base + 6];
        }
        
        // Position (ky=1, kx=0)
        if (iy2 >= 0 && iy2 < height && ix1 >= 0 && ix1 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy2) * width + ix1;
            sum += input[in_idx] * weight[w_base + 7];
        }
        
        // Position (ky=1, kx=1)
        if (iy2 >= 0 && iy2 < height && ix2 >= 0 && ix2 < width) {
            const int in_idx = ((batch_idx * channels + channel_idx) * height + iy2) * width + ix2;
            sum += input[in_idx] * weight[w_base + 8];
        }
    } else {
        // General kernel for other sizes
        for (int ky = -kernel_offset; ky <= kernel_offset; ky++) {
            const int iy = in_y + ky;
            if (iy >= 0 && iy < height) {
                for (int kx = -kernel_offset; kx <= kernel_offset; kx++) {
                    const int ix = in_x + kx;
                    if (ix >= 0 && ix < width) {
                        const int w_idx = channel_idx * kernel_size * kernel_size + 
                                          (ky + kernel_offset) * kernel_size + (kx + kernel_offset);
                        const int in_idx = ((batch_idx * channels + channel_idx) * height + iy) * width + ix;
                        sum += input[in_idx] * weight[w_idx];
                    }
                }
            }
        }
    }
    
    const int out_idx = ((batch_idx * channels + channel_idx) * out_height + out_y) * out_width + out_x;
    output[out_idx] = sum;
}

torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding) {
    
    const int batch_size = input.size(0);
    const int channels = input.size(1);
    const int height = input.size(2);
    const int width = input.size(3);
    const int kernel_size = weight.size(2);
    
    const int out_height = (height + 2 * padding - kernel_size) / stride + 1;
    const int out_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                                input.options());
    
    // Grid: (batch * out_h * out_w, channels)
    const int out_elements = out_height * out_width;
    const int total_blocks = batch_size * out_elements;
    
    dim3 blocks(total_blocks, channels);
    const int block_size = 1;  // Each block computes one output
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "depthwise_conv2d_cuda", ([&] {
        depthwise_conv2d_kernel<scalar_t><<<blocks, block_size>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            batch_size, channels, height, width,
            kernel_size, stride, padding,
            out_height, out_width);
    }));
    
    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error: %s\n", cudaGetErrorString(err));
    }
    
    return output;
}
'''

ext = load_inline(
    name='depthwise_conv2d_ext',
    cpp_sources='''
torch::Tensor depthwise_conv2d_forward(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding);
''',
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_forward'],
    verbose=False,
)

class Model(nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel.
    Uses optimized CUDA kernel for depthwise convolution.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        
        # Initialize convolution weight matching PyTorch format: (out_channels, 1, kh, kw)
        self.weight = nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        
        if bias:
            self.bias = nn.Parameter(torch.zeros(in_channels))
        else:
            self.register_parameter('bias', None)
        
        # Initialize with Kaiming uniform for better convergence
        nn.init.kaiming_uniform_(self.weight, a=0, mode='fan_in')
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Performs the depthwise 2D convolution using custom CUDA kernel.
        """
        # Weight shape is (in_channels, 1, kh, kw)
        # Reshape to (in_channels, kh, kw) for our kernel
        w = self.weight.squeeze(1)  # (in_channels, kh, kw)
        
        # Use the optimized CUDA kernel
        output = ext.depthwise_conv2d_forward(x, w, self.stride, self.padding)
        
        if self.bias is not None:
            output = output + self.bias.view(1, -1, 1, 1)
        
        return output
