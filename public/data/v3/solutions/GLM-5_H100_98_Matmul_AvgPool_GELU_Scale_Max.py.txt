# CUTLASS GEMM + fused post-processing kernel for H100
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cfloat>

// Exact GELU using error function (matches PyTorch default)
__device__ __forceinline__ float gelu_exact(float x) {
    // GELU(x) = x * 0.5 * (1 + erf(x / sqrt(2)))
    return x * 0.5f * (1.0f + erff(x * 0.7071067811865475f));
}

// Fused kernel: AvgPool -> GELU -> Scale -> Max
// Input: (batch, out_features) from GEMM
// Output: (batch,) max values
template<typename T>
__global__ void fused_postprocess_kernel(
    const T* __restrict__ gemm_output,
    float* __restrict__ output,
    int batch_size,
    int out_features,
    int pool_kernel_size,
    int pooled_features,
    float scale_factor
) {
    extern __shared__ float shared_data[];
    
    int batch_idx = blockIdx.x;
    if (batch_idx >= batch_size) return;
    
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    // Each block processes one batch element
    const T* row_input = gemm_output + batch_idx * out_features;
    
    // Initialize max value
    float max_val = -FLT_MAX;
    
    // Process pooled features
    // Each thread handles multiple pooled positions
    for (int pool_idx = tid; pool_idx < pooled_features; pool_idx += block_size) {
        // Compute average pool
        float sum = 0.0f;
        int start = pool_idx * pool_kernel_size;
        for (int k = 0; k < pool_kernel_size; k++) {
            sum += static_cast<float>(row_input[start + k]);
        }
        float pooled = sum / static_cast<float>(pool_kernel_size);
        
        // Apply GELU
        float activated = gelu_exact(pooled);
        
        // Scale
        float scaled = activated * scale_factor;
        
        // Update max
        max_val = fmaxf(max_val, scaled);
    }
    
    // Store in shared memory for reduction
    shared_data[tid] = max_val;
    __syncthreads();
    
    // Parallel reduction for max
    for (int s = block_size / 2; s > 0; s >>= 1) {
        if (tid < s) {
            shared_data[tid] = fmaxf(shared_data[tid], shared_data[tid + s]);
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[batch_idx] = shared_data[0];
    }
}

// Simple GEMM kernel optimized for this specific case
template<typename T, typename ACC_T>
__global__ void simple_gemm_kernel(
    const T* __restrict__ A,
    const T* __restrict__ B,
    T* __restrict__ C,
    int M, int N, int K
) {
    // Use shared memory for tile-based GEMM
    __shared__ ACC_T As[32][32];
    __shared__ ACC_T Bs[32][32];
    
    int tx = threadIdx.x;
    int ty = threadIdx.y;
    int bx = blockIdx.x;
    int by = blockIdx.y;
    
    int row = by * 32 + ty;
    int col = bx * 32 + tx;
    
    ACC_T acc = 0;
    
    for (int t = 0; t < (K + 31) / 32; t++) {
        // Load tiles to shared memory
        int a_col = t * 32 + tx;
        int b_row = t * 32 + ty;
        
        if (row < M && a_col < K) {
            As[ty][tx] = static_cast<ACC_T>(A[row * K + a_col]);
        } else {
            As[ty][tx] = 0;
        }
        
        if (b_row < K && col < N) {
            Bs[ty][tx] = static_cast<ACC_T>(B[b_row * N + col]);
        } else {
            Bs[ty][tx] = 0;
        }
        
        __syncthreads();
        
        // Compute
        #pragma unroll
        for (int k = 0; k < 32; k++) {
            acc += As[ty][k] * Bs[k][tx];
        }
        
        __syncthreads();
    }
    
    if (row < M && col < N) {
        C[row * N + col] = static_cast<T>(acc);
    }
}

// Fused GEMM + postprocess - complete pipeline
torch::Tensor fused_matmul_pool_gelu_scale_max(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size,
    double scale_factor
) {
    int batch_size = input.size(0);
    int in_features = input.size(1);
    int out_features = weight.size(0);
    int pooled_features = out_features / pool_kernel_size;
    
    auto options = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    
    // Allocate intermediate for GEMM output
    auto gemm_output = torch::empty({batch_size, out_features}, input.options());
    auto output = torch::empty({batch_size}, options);
    
    dim3 blockDim(32, 32);
    dim3 gridDim((out_features + 31) / 32, (batch_size + 31) / 32);
    
    // Dispatch based on dtype
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(input.scalar_type(), "gemm_kernel", ([&] {
        simple_gemm_kernel<scalar_t, float><<<gridDim, blockDim>>>(
            input.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            gemm_output.data_ptr<scalar_t>(),
            batch_size, out_features, in_features
        );
    }));
    
    // Add bias
    if (bias.defined() && bias.numel() > 0) {
        gemm_output = gemm_output + bias;
    }
    
    // Fused postprocess kernel
    int threads = min(512, pooled_features);
    int shared_mem = threads * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES_AND_HALF(gemm_output.scalar_type(), "postprocess_kernel", ([&] {
        fused_postprocess_kernel<scalar_t><<<batch_size, threads, shared_mem>>>(
            gemm_output.data_ptr<scalar_t>(),
            output.data_ptr<float>(),
            batch_size,
            out_features,
            pool_kernel_size,
            pooled_features,
            static_cast<float>(scale_factor)
        );
    }));
    
    return output;
}

// Fused single kernel version for better memory locality
template<typename T>
__global__ void fused_complete_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const float* __restrict__ bias,
    float* __restrict__ output,
    int M, int N, int K,
    int pool_kernel_size,
    float scale_factor
) {
    // Each block processes one output row (batch element)
    int batch_idx = blockIdx.x;
    if (batch_idx >= M) return;
    
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    int pooled_features = N / pool_kernel_size;
    
    // Load weight column tiles into shared memory
    extern __shared__ float shared_mem[];
    float* weight_tile = shared_mem;
    float* partial_results = shared_mem + N * sizeof(float) / sizeof(float);
    
    const T* input_row = input + batch_idx * K;
    
    // Compute GEMM for this row and do fused operations
    // Each thread computes multiple output elements
    
    float max_val = -FLT_MAX;
    
    // Process output elements in chunks
    for (int out_col_base = 0; out_col_base < N; out_col_base += block_size) {
        int out_col = out_col_base + tid;
        float gemm_acc = 0.0f;
        
        if (out_col < N) {
            // Compute dot product for this column
            for (int k = 0; k < K; k++) {
                gemm_acc += static_cast<float>(input_row[k]) * static_cast<float>(weight[out_col * K + k]);
            }
            
            // Add bias
            if (bias != nullptr) {
                gemm_acc += bias[out_col];
            }
        }
        
        // Store partial result
        if (out_col < N) {
            partial_results[out_col] = gemm_acc;
        }
        __syncthreads();
    }
    
    // Now apply pooling, GELU, scale, and max
    for (int pool_idx = tid; pool_idx < pooled_features; pool_idx += block_size) {
        float sum = 0.0f;
        for (int k = 0; k < pool_kernel_size; k++) {
            sum += partial_results[pool_idx * pool_kernel_size + k];
        }
        float pooled = sum / static_cast<float>(pool_kernel_size);
        float activated = gelu_exact(pooled);
        float scaled = activated * scale_factor;
        max_val = fmaxf(max_val, scaled);
    }
    
    // Reduce max across threads
    partial_results[tid] = max_val;
    __syncthreads();
    
    for (int s = block_size / 2; s > 0; s >>= 1) {
        if (tid < s) {
            partial_results[tid] = fmaxf(partial_results[tid], partial_results[tid + s]);
        }
        __syncthreads();
    }
    
    if (tid == 0) {
        output[batch_idx] = partial_results[0];
    }
}
'''

cpp_source = r'''
#include <torch/extension.h>

torch::Tensor fused_matmul_pool_gelu_scale_max(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    int pool_kernel_size,
    double scale_factor
);
'''

# Load the extension
ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_matmul_pool_gelu_scale_max'],
    extra_cuda_cflags=['-O3', '-arch=sm_90'],
    verbose=False,
)

class Model(nn.Module):
    """
    Optimized fused Matmul_AvgPool_GELU_Scale_Max implementation for H100.
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.pool_kernel_size = pool_kernel_size
        self.scale_factor = scale_factor
        
        # Store weight and bias as parameters
        self.weight = nn.Parameter(torch.empty(out_features, in_features))
        self.bias = nn.Parameter(torch.empty(out_features))
        
        # Initialize weights (same as nn.Linear)
        nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / (fan_in ** 0.5) if fan_in > 0 else 0
        nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        """
        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_features).

        Returns:
            torch.Tensor: Output tensor of shape (batch_size,).
        """
        return ext.fused_matmul_pool_gelu_scale_max(
            x,
            self.weight,
            self.bias,
            self.pool_kernel_size,
            self.scale_factor
        )

# Hyperparameters
batch_size = 128
in_features = 4096
out_features = 4096
pool_kernel_size = 4
scale_factor = 2.0

def get_inputs():
    return [torch.randn(batch_size, in_features)]

def get_init_inputs():
    return [in_features, out_features, pool_kernel_size, scale_factor]
