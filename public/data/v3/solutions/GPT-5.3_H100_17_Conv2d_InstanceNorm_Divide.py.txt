import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    const float* __restrict__ b,
    float* __restrict__ out,
    int N, int C, int H, int W,
    int OC, int KH, int KW,
    int OH, int OW)
{
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * OC * OH * OW;
    if (idx >= total) return;

    int ow = idx % OW;
    int oh = (idx / OW) % OH;
    int oc = (idx / (OW*OH)) % OC;
    int n = idx / (OW*OH*OC);

    float val = b ? b[oc] : 0.0f;

    for (int ic = 0; ic < C; ++ic) {
        for (int kh = 0; kh < KH; ++kh) {
            for (int kw = 0; kw < KW; ++kw) {
                int ih = oh + kh;
                int iw = ow + kw;
                int x_idx = ((n*C + ic)*H + ih)*W + iw;
                int w_idx = ((oc*C + ic)*KH + kh)*KW + kw;
                val += x[x_idx] * w[w_idx];
            }
        }
    }

    out[idx] = val;
}

__global__ void inst_norm_div_kernel(
    float* x,
    int N, int C, int H, int W,
    float eps, float divv)
{
    int nc = blockIdx.x;
    int n = nc / C;
    int c = nc % C;

    int HW = H*W;

    float mean = 0.f;
    if (threadIdx.x == 0) {
        for (int i = 0; i < HW; ++i) {
            int h = i / W;
            int w = i % W;
            int idx = ((n*C + c)*H + h)*W + w;
            mean += x[idx];
        }
        mean /= HW;

        float var = 0.f;
        for (int i = 0; i < HW; ++i) {
            int h = i / W;
            int w = i % W;
            int idx = ((n*C + c)*H + h)*W + w;
            float v = x[idx] - mean;
            var += v*v;
        }
        var /= HW;
        float inv_std = rsqrtf(var + eps);

        for (int i = 0; i < HW; ++i) {
            int h = i / W;
            int w = i % W;
            int idx = ((n*C + c)*H + h)*W + w;
            float v = x[idx];
            v = (v - mean) * inv_std;
            x[idx] = v / divv;
        }
    }
}

torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b, float divv) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int OC = w.size(0);
    int KH = w.size(2);
    int KW = w.size(3);

    int OH = H - KH + 1;
    int OW = W - KW + 1;

    auto out = torch::empty({N, OC, OH, OW}, x.options());

    int total = N*OC*OH*OW;
    conv2d_kernel<<<(total+255)/256, 256>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        b.numel()? b.data_ptr<float>() : nullptr,
        out.data_ptr<float>(),
        N,C,H,W,OC,KH,KW,OH,OW);

    inst_norm_div_kernel<<<N*OC, 256>>>(
        out.data_ptr<float>(), N, OC, OH, OW, 1e-5f, divv);

    return out;
}
'''

ext = load_inline(
    name='fused_conv_instnorm_div',
    cpp_sources='torch::Tensor fused_op(torch::Tensor x, torch::Tensor w, torch::Tensor b, float divv);',
    cuda_sources=cuda_src,
    functions=['fused_op'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divide_by = divide_by

    def forward(self, x):
        return ext.fused_op(x.contiguous(), self.conv.weight.contiguous(), self.conv.bias.contiguous(), float(self.divide_by))


def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]


def get_init_inputs():
    return [3, 16, 3, 2.0]
