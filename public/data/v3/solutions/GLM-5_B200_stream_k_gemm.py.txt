"""Optimized GEMM using cuda.tile for B200 GPU.

Uses tiled MMA approach with tensor core operations.
"""

import torch
import torch.nn as nn
import cuda.tile as ct

# Tile dimensions optimized for tensor cores
TM = 64
TN = 64
TK = 32

@ct.kernel
def gemm_kernel(
    A: ct.Array,  # [M, K]
    B: ct.Array,  # [K, N]
    C: ct.Array,  # [M, N]
    M: ct.Constant[int],
    N: ct.Constant[int],
    K: ct.Constant[int],
):
    """Tiled GEMM kernel using tensor core MMA."""
    # Get block indices
    pid_m = ct.bid(0)
    pid_n = ct.bid(1)
    
    # Number of tiles
    num_m_tiles = ct.cdiv(M, TM)
    num_n_tiles = ct.cdiv(N, TN)
    num_k_tiles = ct.cdiv(K, TK)
    
    # Check bounds
    if pid_m >= num_m_tiles or pid_n >= num_n_tiles:
        return
    
    # Accumulator in float32 for precision
    acc = ct.full((TM, TN), 0.0, dtype=ct.float32)
    
    # Loop over K tiles
    for k in range(num_k_tiles):
        # Load tiles with zero-padding for edge cases
        a = ct.load(A, (pid_m * TM, k * TK), (TM, TK), padding_mode=ct.PaddingMode.ZERO)
        b = ct.load(B, (k * TK, pid_n * TN), (TK, TN), padding_mode=ct.PaddingMode.ZERO)
        
        # MMA accumulation
        acc = ct.mma(a, b, acc)
    
    # Convert accumulator to output dtype and store
    ct.store(C, (pid_m * TM, pid_n * TN), ct.astype(acc, C.dtype))


class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
        M, K = a.shape
        K2, N = b.shape
        
        assert K == K2, f"K dimension mismatch: {K} vs {K2}"
        
        # Create output tensor
        C = torch.empty(M, N, dtype=a.dtype, device=a.device)
        
        # Compute grid dimensions
        grid_m = (M + TM - 1) // TM
        grid_n = (N + TN - 1) // TN
        
        # Launch kernel with CUDA stream
        stream = torch.cuda.current_stream().cuda_stream
        ct.launch(
            stream,
            (grid_m, grid_n),
            gemm_kernel,
            (a, b, C, M, N, K),
        )
        
        return C


OP_TYPE = "gemm"

HARDWARE_REQUIRED = ["B200"]
