
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>

__global__ void conv_bn_scaling_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ running_mean,
    const float* __restrict__ running_var,
    const float* __restrict__ bn_weight,
    const float* __restrict__ bn_bias,
    float* __restrict__ output,
    int batch_size, int in_c, int in_h, int in_w,
    int out_c, int out_h, int out_w,
    int k_h, int k_w,
    float eps, float scaling_factor) {

    int batch_idx = blockIdx.z / out_c;
    int oc = blockIdx.z % out_c;
    int oh = blockIdx.y * blockDim.y + threadIdx.y;
    int ow = blockIdx.x * blockDim.x + threadIdx.x;

    if (oh < out_h && ow < out_w && batch_idx < batch_size) {
        float inv_std = 1.0f / sqrtf(running_var[oc] + eps);
        float combined_scale = (bn_weight[oc] * inv_std) * scaling_factor;
        float combined_bias = (bn_bias[oc] - bn_weight[oc] * running_mean[oc] * inv_std) * scaling_factor;
        
        if (bias != nullptr) {
            combined_bias += bias[oc] * combined_scale;
        }

        float val = 0.0f;
        for (int ic = 0; ic < in_c; ++ic) {
            for (int kh = 0; kh < k_h; ++kh) {
                for (int kw = 0; kw < k_w; ++kw) {
                    int ih = oh + kh;
                    int iw = ow + kw;
                    float in_val = input[((batch_idx * in_c + ic) * in_h + ih) * in_w + iw];
                    float w_val = weight[((oc * in_c + ic) * k_h + kh) * k_w + kw];
                    val += in_val * w_val;
                }
            }
        }
        output[((batch_idx * out_c + oc) * out_h + oh) * out_w + ow] = val * combined_scale + combined_bias;
    }
}

torch::Tensor conv_bn_scaling_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    float eps,
    float scaling_factor) {

    int batch_size = input.size(0);
    int in_c = input.size(1);
    int in_h = input.size(2);
    int in_w = input.size(3);
    int out_c = weight.size(0);
    int k_h = weight.size(2);
    int k_w = weight.size(3);
    int out_h = in_h - k_h + 1;
    int out_w = in_w - k_w + 1;

    auto output = torch::empty({batch_size, out_c, out_h, out_w}, input.options());

    dim3 block(16, 8, 1);
    dim3 grid((out_w + block.x - 1) / block.x, (out_h + block.y - 1) / block.y, out_c * batch_size);

    conv_bn_scaling_kernel<<<grid, block>>>(
        input.data_ptr<float>(),
        weight.data_ptr<float>(),
        bias.has_value() ? bias.value().data_ptr<float>() : nullptr,
        running_mean.data_ptr<float>(),
        running_var.data_ptr<float>(),
        bn_weight.data_ptr<float>(),
        bn_bias.data_ptr<float>(),
        output.data_ptr<float>(),
        batch_size, in_c, in_h, in_w,
        out_c, out_h, out_w,
        k_h, k_w,
        eps, scaling_factor
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor conv_bn_scaling_forward(
    torch::Tensor input,
    torch::Tensor weight,
    at::optional<torch::Tensor> bias,
    torch::Tensor running_mean,
    torch::Tensor running_var,
    torch::Tensor bn_weight,
    torch::Tensor bn_bias,
    float eps,
    float scaling_factor);
'''

ext = load_inline(
    name='fused_conv_bn_scaling',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['conv_bn_scaling_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels)
        self.scaling_factor = scaling_factor

    def forward(self, x):
        # Ensure variables are on the same device and type
        return ext.conv_bn_scaling_forward(
            x.contiguous(),
            self.conv.weight.contiguous(),
            self.conv.bias.contiguous() if self.conv.bias is not None else None,
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            self.bn.eps,
            float(self.scaling_factor)
        )

def get_inputs():
    return [torch.randn(16, 3, 256, 256, device='cuda')]

def get_init_inputs():
    return [3, 16, 3, 2.0]
