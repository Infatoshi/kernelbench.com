import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def gemm_dropout_mean_kernel(
    x_ptr,           # Input: (batch_size, in_features)
    weight_ptr,      # Weight: (out_features, in_features)
    bias_ptr,        # Bias: (out_features,)
    output_ptr,      # Output: (1, out_features)
    batch_size: tl.constexpr,
    in_features: tl.constexpr,
    out_features: tl.constexpr,
    dropout_p: tl.constexpr,
    BLOCK_M: tl.constexpr,  # Block size for output features
    BLOCK_N: tl.constexpr,  # Block size for input features
    HAS_BIAS: tl.constexpr,
    training: tl.constexpr,
    seed: tl.constexpr,
):
    """
    Fused kernel: GEMM -> Dropout -> Mean(dim=0)
    Each program block computes a tile of output features.
    """
    pid = tl.program_id(0)
    num_blocks_m = tl.cdiv(out_features, BLOCK_M)
    
    # Compute row offset for this program
    row_start = pid * BLOCK_M
    row_offs = row_start + tl.arange(0, BLOCK_M)
    row_mask = row_offs < out_features
    
    # Initialize accumulator for mean (sum over batch)
    acc = tl.zeros([BLOCK_M], dtype=tl.float32)
    
    if training:
        rng_offset = row_start
    
    # Loop over batch dimension
    for batch_idx in range(batch_size):
        # Compute dot product for this batch item
        batch_acc = tl.zeros([BLOCK_M], dtype=tl.float32)
        
        # Loop over input features
        for col_start in range(0, in_features, BLOCK_N):
            col_offs = col_start + tl.arange(0, BLOCK_N)
            col_mask = col_offs < in_features
            
            # Load input: x[batch_idx, col_offs]
            x_ptr_row = x_ptr + batch_idx * in_features
            x_vals = tl.load(x_ptr_row + col_offs, mask=col_mask, other=0.0).to(tl.float32)
            
            # Load weights: weight[row_offs[:, None], col_offs[None, :]]
            w_ptr_base = weight_ptr + row_offs[:, None] * in_features + col_offs[None, :]
            w_vals = tl.load(w_ptr_base, mask=row_mask[:, None] & col_mask[None, :], other=0.0).to(tl.float32)
            
            # Compute partial dot product
            batch_acc += tl.sum(x_vals[None, :] * w_vals, axis=1)
        
        # Add bias
        if HAS_BIAS:
            bias_vals = tl.load(bias_ptr + row_offs, mask=row_mask, other=0.0).to(tl.float32)
            batch_acc += bias_vals
        
        # Apply dropout
        if training:
            # Generate random numbers for dropout
            rand_vals = tl.rand(seed + rng_offset + batch_idx + tl.arange(0, BLOCK_M), 0)
            keep_prob = 1.0 - dropout_p
            batch_acc = tl.where(rand_vals < keep_prob, batch_acc / keep_prob, 0.0)
        
        # Accumulate for mean
        acc += batch_acc
    
    # Compute mean
    mean_vals = acc / batch_size
    
    # Store result
    tl.store(output_ptr + row_offs, mean_vals, mask=row_mask)


@triton.jit
def softmax_kernel(
    input_ptr,       # Input: (1, out_features)
    output_ptr,      # Output: (1, out_features)
    out_features: tl.constexpr,
    BLOCK_OUT: tl.constexpr,
):
    """
    Softmax kernel: computes exp(x - max(x)) / sum(exp(x - max(x)))
    """
    out_start = tl.program_id(0) * BLOCK_OUT
    
    offs = out_start + tl.arange(0, BLOCK_OUT)
    mask = offs < out_features
    
    # Load values
    vals = tl.load(input_ptr + offs, mask=mask, other=-float('inf')).to(tl.float32)
    
    # Compute max
    max_val = tl.max(vals, 0)
    
    # Compute exp(x - max)
    exp_vals = tl.exp(vals - max_val)
    
    # Compute sum
    sum_exp = tl.sum(exp_vals, 0)
    
    # Compute softmax
    softmax_vals = exp_vals / sum_exp
    
    # Store
    tl.store(output_ptr + offs, softmax_vals, mask=mask)


class FusedGEMMDropoutMeanSoftmax(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, weight, bias, dropout_p, training):
        batch_size, in_features = x.shape
        out_features = weight.shape[0]
        
        # Ensure contiguous
        x = x.contiguous()
        weight = weight.contiguous()
        if bias is not None:
            bias = bias.contiguous()
        
        # Allocate output for mean values (1, out_features)
        mean_output = torch.empty((1, out_features), device=x.device, dtype=torch.float32)
        
        # Choose block sizes based on problem size
        BLOCK_M = 64  # Output features per block
        BLOCK_N = 128  # Input features per block
        BLOCK_OUT = 256  # Softmax block size
        
        # Fused GEMM -> Dropout -> Mean kernel
        num_blocks_m = triton.cdiv(out_features, BLOCK_M)
        grid_mean = (num_blocks_m,)
        
        gemm_dropout_mean_kernel[grid_mean](
            x, weight, bias if bias is not None else x,
            mean_output,
            batch_size, in_features, out_features,
            dropout_p,
            BLOCK_M=BLOCK_M,
            BLOCK_N=BLOCK_N,
            HAS_BIAS=bias is not None,
            training=training,
            seed=42,
        )
        
        # Softmax kernel
        output = torch.empty((1, out_features), device=x.device, dtype=torch.float32)
        num_programs_softmax = triton.cdiv(out_features, BLOCK_OUT)
        grid_softmax = (num_programs_softmax,)
        softmax_kernel[grid_softmax](
            mean_output, output, out_features,
            BLOCK_OUT=BLOCK_OUT,
        )
        
        # Convert back to original dtype
        output = output.to(x.dtype)
        
        ctx.save_for_backward(x, weight, bias)
        ctx.dropout_p = dropout_p
        ctx.training = training
        
        return output


class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features, bias=True)
        self.dropout_p = dropout_p
        self.out_features = out_features
        
    def forward(self, x):
        training = self.training and self.dropout_p > 0
        return FusedGEMMDropoutMeanSoftmax.apply(
            x, 
            self.matmul.weight, 
            self.matmul.bias, 
            self.dropout_p, 
            training
        )
