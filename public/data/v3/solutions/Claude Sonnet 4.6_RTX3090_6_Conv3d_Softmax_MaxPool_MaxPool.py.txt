import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Fused Softmax + MaxPool + MaxPool kernel
# After conv: (N, C, D, H, W) = (16, 16, 14, 126, 126)
# Softmax along dim=1 (C channels)
# MaxPool3d(2) twice => single 4x4x4 spatial pooling

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>
#include <math.h>

// Fused kernel: Softmax(dim=1) + MaxPool3d(2) + MaxPool3d(2)
// Input: (N, C, D, H, W) float
// Output: (N, C, Dout, Hout, Wout) float
// Dout = D/4, Hout = H/4, Wout = W/4 (pool_kernel=2 twice = stride 4)
// 
// Strategy:
// Each output element (n, c, od, oh, ow) = max over 4x4x4 spatial window
// of softmax(conv_out[n, :, d, h, w])[c]
//
// For softmax + maxpool fusion:
// For each spatial position (d,h,w) in the 4x4x4 window, we need softmax over C channels
// then take max over spatial positions for channel c
//
// Since C=16 is small, we can process each (n, od, oh, ow) block with one thread block
// Each thread handles one channel c
// We compute softmax for all 64 spatial positions in the window, then take max per channel

__global__ void fused_softmax_maxpool4_kernel(
    const float* __restrict__ input,  // (N, C, D, H, W)
    float* __restrict__ output,       // (N, C, Dout, Hout, Wout)
    int N, int C, int D, int H, int W,
    int Dout, int Hout, int Wout
) {
    // Each block handles one (n, od, oh, ow) spatial output position
    // blockDim.x = C (up to 32)
    int idx = blockIdx.x;
    int total_out = N * Dout * Hout * Wout;
    if (idx >= total_out) return;

    int ow = idx % Wout;
    int tmp = idx / Wout;
    int oh = tmp % Hout;
    tmp /= Hout;
    int od = tmp % Dout;
    int n  = tmp / Dout;

    int c = threadIdx.x;
    if (c >= C) return;

    // We pool over 4x4x4 spatial window starting at (od*4, oh*4, ow*4)
    int d_start = od * 4;
    int h_start = oh * 4;
    int w_start = ow * 4;

    float max_val = -FLT_MAX;

    // For each of the 64 spatial positions in the window
    for (int dd = 0; dd < 4; dd++) {
        int d = d_start + dd;
        if (d >= D) continue;
        for (int dh = 0; dh < 4; dh++) {
            int h = h_start + dh;
            if (h >= H) continue;
            for (int dw = 0; dw < 4; dw++) {
                int w = w_start + dw;
                if (w >= W) continue;

                // Compute softmax for this spatial position
                // Load all C values for position (n, :, d, h, w)
                // The input is in (N, C, D, H, W) layout
                // input[n, c, d, h, w] = input[((n*C + c)*D + d)*H*W + h*W + w]
                
                // Find max for numerical stability
                float local_max = -FLT_MAX;
                for (int cc = 0; cc < C; cc++) {
                    float v = input[((n * C + cc) * D + d) * (H * W) + h * W + w];
                    if (v > local_max) local_max = v;
                }
                
                // Compute exp sum
                float exp_sum = 0.0f;
                for (int cc = 0; cc < C; cc++) {
                    float v = input[((n * C + cc) * D + d) * (H * W) + h * W + w];
                    exp_sum += expf(v - local_max);
                }
                
                // Compute softmax for channel c
                float v_c = input[((n * C + c) * D + d) * (H * W) + h * W + w];
                float softmax_val = expf(v_c - local_max) / exp_sum;
                
                if (softmax_val > max_val) max_val = softmax_val;
            }
        }
    }

    // Write output
    output[((n * C + c) * Dout + od) * (Hout * Wout) + oh * Wout + ow] = max_val;
}

// Optimized version using shared memory
// Each block handles one (n, od, oh, ow) output with C threads
// Shared memory stores the input values for the 4x4x4 window
__global__ void fused_softmax_maxpool4_smem_kernel(
    const float* __restrict__ input,  // (N, C, D, H, W)
    float* __restrict__ output,       // (N, C, Dout, Hout, Wout)
    int N, int C, int D, int H, int W,
    int Dout, int Hout, int Wout
) {
    extern __shared__ float smem[]; // C * 64 floats (e.g., 16*64=1024)

    int idx = blockIdx.x;
    int total_out = N * Dout * Hout * Wout;
    if (idx >= total_out) return;

    int ow = idx % Wout;
    int tmp = idx / Wout;
    int oh = tmp % Hout;
    tmp /= Hout;
    int od = tmp % Dout;
    int n  = tmp / Dout;

    int c = threadIdx.x;
    if (c >= C) return;

    int d_start = od * 4;
    int h_start = oh * 4;
    int w_start = ow * 4;

    // Load all C * 64 values into shared memory
    // smem layout: [spatial_idx * C + c] or [c * 64 + spatial_idx]
    // Use [spatial_pos * C + c] for easy per-position softmax
    
    // Each thread c loads all 64 spatial values for its channel
    for (int sp = 0; sp < 64; sp++) {
        int dd = sp / 16;
        int dh = (sp % 16) / 4;
        int dw = sp % 4;
        int d = d_start + dd;
        int h = h_start + dh;
        int w = w_start + dw;
        
        float val = 0.0f;
        if (d < D && h < H && w < W) {
            val = input[((n * C + c) * D + d) * (H * W) + h * W + w];
        } else {
            val = -FLT_MAX; // padding
        }
        smem[sp * C + c] = val;
    }
    __syncthreads();

    float max_val = -FLT_MAX;

    for (int sp = 0; sp < 64; sp++) {
        // Compute softmax for this spatial position
        // Find max across channels
        float local_max = -FLT_MAX;
        for (int cc = 0; cc < C; cc++) {
            float v = smem[sp * C + cc];
            if (v > local_max) local_max = v;
        }
        
        if (local_max == -FLT_MAX) continue; // padding position
        
        float exp_sum = 0.0f;
        for (int cc = 0; cc < C; cc++) {
            exp_sum += expf(smem[sp * C + cc] - local_max);
        }
        
        float softmax_val = expf(smem[sp * C + c] - local_max) / exp_sum;
        if (softmax_val > max_val) max_val = softmax_val;
    }

    output[((n * C + c) * Dout + od) * (Hout * Wout) + oh * Wout + ow] = max_val;
}

// Better version: warp-level softmax using shuffle
// Each warp handles one spatial position, computes softmax, then accumulates max per channel
// Block: (C, W_spatial) where W_spatial = number of spatial positions per block
// For C=16, use warp size 32 with 16 threads per warp doing C channels

// Most optimized: one warp per spatial output position, but batch spatial positions
// Use one thread block for multiple (od, oh, ow) positions
// blockDim.x = 32 (warp), blockDim.y = SPATIAL_BLOCK  
// Each warp computes one output pixel

template<int C_VAL>
__global__ void fused_softmax_maxpool4_warp_kernel(
    const float* __restrict__ input,  // (N, C, D, H, W)
    float* __restrict__ output,       // (N, C, Dout, Hout, Wout)
    int N, int D, int H, int W,
    int Dout, int Hout, int Wout
) {
    // Each warp (32 threads) handles one output position
    // threads 0..C-1 handle channels, others are idle (for C<=32)
    int warp_id = (blockIdx.x * blockDim.x + threadIdx.x) / 32;
    int lane = threadIdx.x % 32;
    
    int total_out = N * Dout * Hout * Wout;
    if (warp_id >= total_out) return;

    int ow = warp_id % Wout;
    int tmp = warp_id / Wout;
    int oh = tmp % Hout;
    tmp /= Hout;
    int od = tmp % Dout;
    int n  = tmp / Dout;

    int d_start = od * 4;
    int h_start = oh * 4;
    int w_start = ow * 4;

    float max_val = (lane < C_VAL) ? -FLT_MAX : -FLT_MAX;

    for (int dd = 0; dd < 4; dd++) {
        int d = d_start + dd;
        if (d >= D) continue;
        for (int dh = 0; dh < 4; dh++) {
            int h = h_start + dh;
            if (h >= H) continue;
            for (int dw = 0; dw < 4; dw++) {
                int w = w_start + dw;
                if (w >= W) continue;

                // Lane c reads channel c value
                float v_c = (lane < C_VAL) ? 
                    input[((n * C_VAL + lane) * D + d) * (H * W) + h * W + w] : -FLT_MAX;
                
                // Warp-level max reduction for numerical stability
                float local_max = v_c;
                #pragma unroll
                for (int offset = 16; offset >= 1; offset >>= 1) {
                    float other = __shfl_xor_sync(0xffffffff, local_max, offset);
                    local_max = fmaxf(local_max, other);
                }
                // local_max is now max across all lanes (including non-channel lanes with -FLT_MAX)
                // We only want max over C_VAL channels
                // Since non-channel lanes have -FLT_MAX, max is correct if C_VAL<=32
                
                float exp_v = (lane < C_VAL) ? expf(v_c - local_max) : 0.0f;
                
                // Sum reduction
                float exp_sum = exp_v;
                #pragma unroll
                for (int offset = 16; offset >= 1; offset >>= 1) {
                    exp_sum += __shfl_xor_sync(0xffffffff, exp_sum, offset);
                }
                
                float softmax_val = (lane < C_VAL) ? (exp_v / exp_sum) : -FLT_MAX;
                if (softmax_val > max_val) max_val = softmax_val;
            }
        }
    }

    // Write output for each channel
    if (lane < C_VAL) {
        output[((n * C_VAL + lane) * Dout + od) * (Hout * Wout) + oh * Wout + ow] = max_val;
    }
}

torch::Tensor fused_softmax_maxpool(torch::Tensor input, int pool_kernel_size) {
    // input: (N, C, D, H, W) float32
    // pool_kernel_size applied twice (so effectively pool 4x)
    TORCH_CHECK(input.is_cuda(), "Input must be on CUDA");
    TORCH_CHECK(input.is_contiguous(), "Input must be contiguous");
    TORCH_CHECK(input.dtype() == torch::kFloat32, "Input must be float32");
    
    int N = input.size(0);
    int C = input.size(1);
    int D = input.size(2);
    int H = input.size(3);
    int W = input.size(4);
    
    int pool_factor = pool_kernel_size * pool_kernel_size; // 4 for pool_kernel=2 twice
    // Wait: pool_kernel_size is 2, applied twice = 4 spatial factor
    int Dout = D / pool_kernel_size / pool_kernel_size;
    int Hout = H / pool_kernel_size / pool_kernel_size;
    int Wout = W / pool_kernel_size / pool_kernel_size;
    
    auto output = torch::empty({N, C, Dout, Hout, Wout}, input.options());
    
    int total_out = N * Dout * Hout * Wout;
    
    if (C == 16) {
        // Use warp-level kernel: one warp per output spatial position
        int warps_per_block = 4;
        int threads = warps_per_block * 32;
        int blocks = (total_out + warps_per_block - 1) / warps_per_block;
        
        fused_softmax_maxpool4_warp_kernel<16><<<blocks, threads>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            N, D, H, W, Dout, Hout, Wout
        );
    } else {
        // General case with shared memory
        int smem_size = 64 * C * sizeof(float);
        fused_softmax_maxpool4_smem_kernel<<<total_out, C, smem_size>>>(
            input.data_ptr<float>(),
            output.data_ptr<float>(),
            N, C, D, H, W, Dout, Hout, Wout
        );
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_softmax_maxpool(torch::Tensor input, int pool_kernel_size);
'''

ext = load_inline(
    name='fused_ops_v1',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_softmax_maxpool'],
    verbose=True,
    extra_cuda_cflags=['-O3', '-arch=sm_86', '--use_fast_math'],
)


class Model(nn.Module):
    """
    Optimized model: Conv3d + fused (Softmax + MaxPool + MaxPool)
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        x = self.conv(x)
        x = x.contiguous()
        x = ext.fused_softmax_maxpool(x, self.pool_kernel_size)
        return x


def get_inputs():
    return [torch.randn(16, 3, 16, 128, 128)]

def get_init_inputs():
    return [3, 16, 3, 2]
