import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

// LayerNorm forward kernel optimized for B200
// Each block handles one batch element with full warp-level reduction
template<typename T>
__global__ void layernorm_forward_kernel(
    const T* __restrict__ x,
    T* __restrict__ y,
    const T* __restrict__ gamma,
    const T* __restrict__ beta,
    const float eps,
    const int n
) {
    extern __shared__ float sdata[];
    
    const int tid = threadIdx.x;
    const int total_threads = blockDim.x;
    
    // Phase 1: Compute sum and squared sum
    float thread_sum = 0.0f;
    float thread_sq_sum = 0.0f;
    
    for (int i = tid; i < n; i += total_threads) {
        float val = static_cast<float>(x[i]);
        thread_sum += val;
        thread_sq_sum += val * val;
    }
    
    // Warp reduction using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        thread_sum += __shfl_down_sync(0xffffffff, thread_sum, offset);
        thread_sq_sum += __shfl_down_sync(0xffffffff, thread_sq_sum, offset);
    }
    
    // Store warp results to shared memory
    const int num_warps = (blockDim.x + 31) / 32;
    if (tid % 32 == 0) {
        sdata[tid / 32] = thread_sum;
        sdata[num_warps + tid / 32] = thread_sq_sum;
    }
    __syncthreads();
    
    // Final reduction in thread 0
    if (tid == 0) {
        float block_sum = 0.0f;
        float block_sq_sum = 0.0f;
        for (int i = 0; i < num_warps; i++) {
            block_sum += sdata[i];
            block_sq_sum += sdata[num_warps + i];
        }
        sdata[0] = block_sum / n;  // mean
        sdata[1] = rsqrtf(block_sq_sum / n + eps);  // rstd
    }
    __syncthreads();
    
    float mean = sdata[0];
    float rstd = sdata[1];
    
    // Phase 2: Normalize and apply gamma/beta
    for (int i = tid; i < n; i += total_threads) {
        float x_val = static_cast<float>(x[i]);
        float norm_val = (x_val - mean) * rstd;
        
        float g_val = gamma ? static_cast<float>(gamma[i]) : 1.0f;
        float b_val = beta ? static_cast<float>(beta[i]) : 0.0f;
        
        y[i] = static_cast<T>(norm_val * g_val + b_val);
    }
}

torch::Tensor layernorm_forward(
    torch::Tensor x,
    torch::Tensor gamma,
    torch::Tensor beta,
    double eps
) {
    const int batch_size = x.size(0);
    const int inner_size = x.size(1);
    const int normalized_size = x.size(2) * x.size(3);
    const int n = inner_size * normalized_size;
    
    auto y = torch::empty_like(x);
    
    const int threads = 256;
    const int smem_size = threads * 2 * sizeof(float);
    
    AT_DISPATCH_FLOATING_TYPES_AND2(at::ScalarType::Half, at::ScalarType::BFloat16, 
        x.scalar_type(), "layernorm_forward", ([&] {
            layernorm_forward_kernel<scalar_t><<<batch_size, threads, smem_size, 0>>>(
                x.data_ptr<scalar_t>(),
                y.data_ptr<scalar_t>(),
                gamma.data_ptr<scalar_t>(),
                beta.data_ptr<scalar_t>(),
                static_cast<float>(eps),
                n
            );
        }));
    
    return y;
}
'''

ext = load_inline(
    name='layernorm_ext',
    cpp_sources='''
torch::Tensor layernorm_forward(
    torch::Tensor x,
    torch::Tensor gamma,
    torch::Tensor beta,
    double eps
);
''',
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
)


class Model(nn.Module):
    """
    Optimized LayerNorm model using custom CUDA kernel.
    """
    def __init__(self, normalized_shape: tuple):
        super().__init__()
        self.normalized_shape = normalized_shape
        self.register_parameter('gamma', torch.nn.Parameter(torch.ones(normalized_shape)))
        self.register_parameter('beta', torch.nn.Parameter(torch.zeros(normalized_shape)))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.layernorm_forward(x, self.gamma, self.beta, self.eps)


def get_inputs():
    batch_size = 16
    features = 64
    dim1 = 256
    dim2 = 256
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]

def get_init_inputs():
    return [(64, 256, 256)]
