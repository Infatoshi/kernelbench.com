import os
os.environ.setdefault('TORCH_CUDA_ARCH_LIST', '8.6')

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# LayerNorm: input shape (16, 64, 256, 256), normalized_shape=(64,256,256)
# Each row has N = 64*256*256 = 4,194,304 elements.
# We normalize each of the 16 rows independently.
# Use Welford online algorithm for numerically stable mean/variance.

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Welford online algorithm for numerically stable mean/variance
// Each block handles one row. We use 512 threads per block.
// For N=4M elements, each thread processes N/512 = 8192 elements.

template <typename scalar_t, int BLOCK_SIZE>
__global__ void layernorm_kernel(
    const scalar_t* __restrict__ input,
    const float*    __restrict__ weight,
    const float*    __restrict__ bias,
    scalar_t*       __restrict__ output,
    int N,
    float eps
) {
    // Each block processes one row
    int row = blockIdx.x;
    const scalar_t* row_input = input + (long long)row * N;
    scalar_t*       row_output = output + (long long)row * N;

    int tid = threadIdx.x;

    // --- Pass 1: Compute mean and variance using Welford ---
    float local_mean  = 0.0f;
    float local_m2    = 0.0f;
    int   local_count = 0;

    for (int i = tid; i < N; i += BLOCK_SIZE) {
        float val = (float)row_input[i];
        local_count++;
        float delta  = val - local_mean;
        local_mean  += delta / (float)local_count;
        float delta2 = val - local_mean;
        local_m2    += delta * delta2;
    }

    // Warp-level reduction using shuffle
    unsigned mask = 0xffffffff;
    #pragma unroll
    for (int offset = 16; offset >= 1; offset >>= 1) {
        float other_mean  = __shfl_down_sync(mask, local_mean, offset);
        float other_m2    = __shfl_down_sync(mask, local_m2, offset);
        int   other_count = __shfl_down_sync(mask, local_count, offset);

        if (other_count > 0) {
            int   combined_count = local_count + other_count;
            float delta          = other_mean - local_mean;
            local_mean = (local_mean * (float)local_count + other_mean * (float)other_count)
                          / (float)combined_count;
            local_m2  = local_m2 + other_m2
                        + delta * delta * (float)local_count * (float)other_count
                          / (float)combined_count;
            local_count = combined_count;
        }
    }

    // Block-level reduction via shared memory
    // BLOCK_SIZE = 512 => 16 warps
    const int NUM_WARPS = BLOCK_SIZE / 32;
    __shared__ float s_mean[NUM_WARPS];
    __shared__ float s_m2[NUM_WARPS];
    __shared__ int   s_count[NUM_WARPS];

    int warp_id = tid / 32;
    int lane_id = tid % 32;

    if (lane_id == 0) {
        s_mean[warp_id]  = local_mean;
        s_m2[warp_id]    = local_m2;
        s_count[warp_id] = local_count;
    }
    __syncthreads();

    // First warp reduces across all warps
    if (warp_id == 0) {
        if (lane_id < NUM_WARPS) {
            local_mean  = s_mean[lane_id];
            local_m2    = s_m2[lane_id];
            local_count = s_count[lane_id];
        } else {
            local_mean  = 0.0f;
            local_m2    = 0.0f;
            local_count = 0;
        }

        #pragma unroll
        for (int offset = 16; offset >= 1; offset >>= 1) {
            float other_mean  = __shfl_down_sync(mask, local_mean, offset);
            float other_m2    = __shfl_down_sync(mask, local_m2, offset);
            int   other_count = __shfl_down_sync(mask, local_count, offset);

            if (lane_id + offset < 32 && other_count > 0) {
                int   combined_count = local_count + other_count;
                float delta          = other_mean - local_mean;
                local_mean = (local_mean * (float)local_count + other_mean * (float)other_count)
                              / (float)combined_count;
                local_m2  = local_m2 + other_m2
                            + delta * delta * (float)local_count * (float)other_count
                              / (float)combined_count;
                local_count = combined_count;
            }
        }
    }

    __shared__ float s_final_mean;
    __shared__ float s_final_rstd;

    if (tid == 0) {
        s_final_mean = local_mean;
        float variance = local_m2 / (float)N;
        s_final_rstd = rsqrtf(variance + eps);
    }
    __syncthreads();

    float mean = s_final_mean;
    float rstd = s_final_rstd;

    // --- Pass 2: Normalize and apply weight/bias ---
    // Use float4 vectorized loads when possible
    for (int i = tid; i < N; i += BLOCK_SIZE) {
        float val  = (float)row_input[i];
        float norm = (val - mean) * rstd;
        row_output[i] = (scalar_t)(norm * weight[i] + bias[i]);
    }
}

torch::Tensor layernorm_forward(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    double eps
) {
    TORCH_CHECK(input.is_contiguous(), "input must be contiguous");
    auto output = torch::empty_like(input);

    int batch = input.size(0);
    int N = input.numel() / batch;

    // Weight/bias are always float32
    auto weight_f = weight.to(torch::kFloat32).contiguous();
    auto bias_f   = bias.to(torch::kFloat32).contiguous();

    const int BLOCK_SIZE = 512;
    dim3 grid(batch);
    dim3 block(BLOCK_SIZE);

    AT_DISPATCH_FLOATING_TYPES_AND2(
        at::ScalarType::Half, at::ScalarType::BFloat16,
        input.scalar_type(), "layernorm_forward",
        [&]() {
            layernorm_kernel<scalar_t, BLOCK_SIZE><<<grid, block>>>(
                input.data_ptr<scalar_t>(),
                weight_f.data_ptr<float>(),
                bias_f.data_ptr<float>(),
                output.data_ptr<scalar_t>(),
                N,
                (float)eps
            );
        }
    );

    return output;
}
'''

cpp_source = '''
torch::Tensor layernorm_forward(torch::Tensor input, torch::Tensor weight, torch::Tensor bias, double eps);
'''

ext = load_inline(
    name='layernorm_ext2',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['layernorm_forward'],
    verbose=False,
    extra_cuda_cflags=['-O3', '--use_fast_math', '-arch=sm_86'],
)


class Model(nn.Module):
    """
    Optimized LayerNorm using custom CUDA kernel with Welford algorithm.
    """
    def __init__(self, normalized_shape: tuple):
        super(Model, self).__init__()
        self.normalized_shape = normalized_shape
        self.eps = 1e-5
        self.weight = nn.Parameter(torch.ones(normalized_shape))
        self.bias   = nn.Parameter(torch.zeros(normalized_shape))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        orig_shape = x.shape
        norm_size = 1
        for s in self.normalized_shape:
            norm_size *= s
        batch = x.numel() // norm_size

        x_2d = x.contiguous().view(batch, norm_size)
        w_1d = self.weight.contiguous().view(-1)
        b_1d = self.bias.contiguous().view(-1)

        out = ext.layernorm_forward(x_2d, w_1d, b_1d, self.eps)
        return out.view(orig_shape)


batch_size = 16
features = 64
dim1 = 256
dim2 = 256


def get_inputs():
    x = torch.randn(batch_size, features, dim1, dim2)
    return [x]


def get_init_inputs():
    return [(features, dim1, dim2)]
