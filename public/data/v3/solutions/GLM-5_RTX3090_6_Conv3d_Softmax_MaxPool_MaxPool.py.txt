import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <cmath>

// Fused Softmax + MaxPool3d + MaxPool3d kernel
// Input: (N, C, D, H, W) - output of Conv3d
// Output: (N, C, D', H', W') - after softmax + 2 maxpools

template <typename scalar_t, int POOL_SIZE>
__global__ void fused_softmax_maxpool_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int N, int C, int D, int H, int W,
    int out_D, int out_H, int out_W) {
    
    // Each thread handles one (n, c, d_out, h_out, w_out) output element
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_outputs = N * C * out_D * out_H * out_W;
    
    if (idx >= total_outputs) return;
    
    // Decode output index
    int w_out = idx % out_W;
    int h_out = (idx / out_W) % out_H;
    int d_out = (idx / (out_W * out_H)) % out_D;
    int c = (idx / (out_W * out_H * out_D)) % C;
    int n = idx / (out_W * out_H * out_D * C);
    
    // Compute the input region for double pooling
    // First pool: d_in = [d_out*2, d_out*2+1], etc
    // Second pool: d_conv = [d_in*2, d_in*2+1], etc
    // Combined: d_conv = [d_out*4, d_out*4+3]
    
    int d_start = d_out * POOL_SIZE * POOL_SIZE;
    int h_start = h_out * POOL_SIZE * POOL_SIZE;
    int w_start = w_out * POOL_SIZE * POOL_SIZE;
    
    // Find max value for this channel across the pooled region
    // We need to apply softmax first, then maxpool
    
    // Step 1: Find max for softmax stability (per spatial position)
    // For softmax across channels, we need to process all channels together
    // But maxpool operates independently on each channel
    
    // Actually, softmax is across channels (dim=1), so each spatial position
    // has its own softmax. Then maxpool operates on each channel independently.
    
    // Strategy: For each position in the double-pool window:
    //   1. Compute softmax across all channels
    //   2. Take max across the pool window for this channel
    
    scalar_t max_val = -FLT_MAX;
    
    // Iterate over the double-pooling window
    for (int pd = 0; pd < POOL_SIZE * POOL_SIZE && (d_start + pd) < D; pd++) {
        for (int ph = 0; ph < POOL_SIZE * POOL_SIZE && (h_start + ph) < H; ph++) {
            for (int pw = 0; pw < POOL_SIZE * POOL_SIZE && (w_start + pw) < W; pw++) {
                int d = d_start + pd;
                int h = h_start + ph;
                int w = w_start + pw;
                
                if (d >= D || h >= H || w >= W) continue;
                
                // This spatial position (d, h, w) contributes to output
                // We need softmax across channels, then this channel's value
                
                // For now, just compute the value for this channel at this position
                // after softmax would be applied
                
                // Softmax for position (n, d, h, w) across all channels:
                // softmax(c) = exp(x[n,c,d,h,w] - max_x) / sum(exp(x - max_x))
                
                // This requires reading all channels - expensive!
                // Let's compute softmax value for this channel
                
                // Find max across channels for this spatial position
                scalar_t local_max = -FLT_MAX;
                for (int cc = 0; cc < C; cc++) {
                    int idx_in = ((n * C + cc) * D + d) * H * W + h * W + w;
                    scalar_t val = input[idx_in];
                    local_max = max(local_max, val);
                }
                
                // Compute sum of exp(x - local_max)
                scalar_t sum_exp = 0;
                for (int cc = 0; cc < C; cc++) {
                    int idx_in = ((n * C + cc) * D + d) * H * W + h * W + w;
                    scalar_t val = input[idx_in];
                    sum_exp += exp(val - local_max);
                }
                
                // Get softmax value for this channel
                int idx_in = ((n * C + c) * D + d) * H * W + h * W + w;
                scalar_t softmax_val = exp(input[idx_in] - local_max) / sum_exp;
                
                max_val = max(max_val, softmax_val);
            }
        }
    }
    
    // Write output
    output[idx] = max_val;
}

// Optimized version using shared memory for softmax computation
template <typename scalar_t, int POOL_SIZE>
__global__ void fused_softmax_maxpool_optimized_kernel(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int N, int C, int D, int H, int W,
    int out_D, int out_H, int out_W) {
    
    // Each block handles one (n, d_out, h_out, w_out) position
    // All threads in block cooperate on softmax across channels
    
    int w_out = blockIdx.x % out_W;
    int h_out = (blockIdx.x / out_W) % out_H;
    int d_out = (blockIdx.x / (out_W * out_H)) % out_D;
    int n = blockIdx.x / (out_W * out_H * out_D);
    
    int tid = threadIdx.x;
    
    // Shared memory for channel values and reduction
    extern __shared__ char shared_mem[];
    scalar_t* channel_vals = (scalar_t*)shared_mem;
    scalar_t* max_vals = (scalar_t*)(shared_mem + C * sizeof(scalar_t));
    scalar_t* sum_vals = (scalar_t*)(shared_mem + 2 * C * sizeof(scalar_t));
    
    scalar_t final_max = -FLT_MAX;
    
    // Iterate over double-pooling window
    int d_start = d_out * POOL_SIZE * POOL_SIZE;
    int h_start = h_out * POOL_SIZE * POOL_SIZE;
    int w_start = w_out * POOL_SIZE * POOL_SIZE;
    
    for (int pd = 0; pd < POOL_SIZE * POOL_SIZE && (d_start + pd) < D; pd++) {
        for (int ph = 0; ph < POOL_SIZE * POOL_SIZE && (h_start + ph) < H; ph++) {
            for (int pw = 0; pw < POOL_SIZE * POOL_SIZE && (w_start + pw) < W; pw++) {
                int d = d_start + pd;
                int h = h_start + ph;
                int w = w_start + pw;
                
                if (d >= D || h >= H || w >= W) continue;
                
                // Load channel values for this spatial position
                if (tid < C) {
                    int idx_in = ((n * C + tid) * D + d) * H * W + h * W + w;
                    channel_vals[tid] = input[idx_in];
                }
                __syncthreads();
                
                // Find max across channels
                scalar_t local_max = -FLT_MAX;
                for (int cc = 0; cc < C; cc++) {
                    local_max = max(local_max, channel_vals[cc]);
                }
                
                // Compute sum of exp
                scalar_t local_sum = 0;
                for (int cc = 0; cc < C; cc++) {
                    local_sum += exp(channel_vals[cc] - local_max);
                }
                
                // Each thread computes softmax for its channel and updates max
                if (tid < C) {
                    scalar_t softmax_val = exp(channel_vals[tid] - local_max) / local_sum;
                    final_max = max(final_max, softmax_val);
                }
            }
        }
    }
    
    // Write output
    if (tid < C) {
        int idx_out = ((n * C + tid) * out_D + d_out) * out_H * out_W + h_out * out_W + w_out;
        output[idx_out] = final_max;
    }
}

// Fused softmax + maxpool kernel - processes all channels together
template <typename scalar_t, int POOL_SIZE>
__global__ void fused_kernel_v2(
    const scalar_t* __restrict__ input,
    scalar_t* __restrict__ output,
    int N, int C, int D, int H, int W,
    int out_D, int out_H, int out_W) {
    
    // Each block handles one (n, d_out, h_out, w_out)
    // Threads cooperate on computing softmax and maxpool
    
    int w_out = blockIdx.x % out_W;
    int h_out = (blockIdx.x / out_W) % out_H;
    int d_out = (blockIdx.x / (out_W * out_H)) % out_D;
    int n = blockIdx.x / (out_W * out_H * out_D);
    
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    // Shared memory
    extern __shared__ char smem[];
    scalar_t* s_data = (scalar_t*)smem;
    scalar_t* s_max = s_data + C;  // for max reduction
    
    // Double pooling: output looks at input range [out*4, out*4+3] if valid
    // But we need to handle boundaries properly
    // pool1: in range [out*2*2, out*2*2 + 2*2-1] = [out*4, out*4+3]
    
    int d_start = d_out * POOL_SIZE * POOL_SIZE;
    int h_start = h_out * POOL_SIZE * POOL_SIZE;
    int w_start = w_out * POOL_SIZE * POOL_SIZE;
    
    // Final output for each channel
    scalar_t channel_max = -FLT_MAX;
    
    // Iterate over pooling window (POOL_SIZE * POOL_SIZE in each dimension for double pool)
    for (int pd = 0; pd < POOL_SIZE * POOL_SIZE; pd++) {
        int d = d_start + pd;
        if (d >= D) continue;
        
        for (int ph = 0; ph < POOL_SIZE * POOL_SIZE; ph++) {
            int h = h_start + ph;
            if (h >= H) continue;
            
            for (int pw = 0; pw < POOL_SIZE * POOL_SIZE; pw++) {
                int w = w_start + pw;
                if (w >= W) continue;
                
                // Load all channel values for this spatial position
                if (tid < C) {
                    int idx_in = ((n * C + tid) * D + d) * H * W + h * W + w;
                    s_data[tid] = input[idx_in];
                }
                __syncthreads();
                
                // Find max for softmax (only need one thread to do this)
                scalar_t smax = -FLT_MAX;
                if (tid == 0) {
                    for (int cc = 0; cc < C; cc++) {
                        smax = max(smax, s_data[cc]);
                    }
                    s_max[0] = smax;
                }
                __syncthreads();
                smax = s_max[0];
                
                // Compute sum of exp (parallel reduction)
                scalar_t local_sum = 0;
                for (int cc = tid; cc < C; cc += block_size) {
                    local_sum += exp(s_data[cc] - smax);
                }
                
                // Reduce sum
                __shared__ scalar_t s_sums[256];
                s_sums[tid] = local_sum;
                __syncthreads();
                
                for (int s = block_size / 2; s > 0; s >>= 1) {
                    if (tid < s) {
                        s_sums[tid] += s_sums[tid + s];
                    }
                    __syncthreads();
                }
                scalar_t total_sum = s_sums[0];
                
                // Compute softmax for this thread's channel(s)
                if (tid < C) {
                    scalar_t softmax_val = exp(s_data[tid] - smax) / total_sum;
                    channel_max = max(channel_max, softmax_val);
                }
                __syncthreads();
            }
        }
    }
    
    // Write output
    if (tid < C) {
        int idx_out = ((n * C + tid) * out_D + d_out) * out_H * out_W + h_out * out_W + w_out;
        output[idx_out] = channel_max;
    }
}

torch::Tensor fused_softmax_maxpool_cuda(
    torch::Tensor input,
    int pool_size) {
    
    int N = input.size(0);
    int C = input.size(1);
    int D = input.size(2);
    int H = input.size(3);
    int W = input.size(4);
    
    // Output dimensions after double pooling with pool_size
    // First pool: floor(in / pool_size), second pool: floor(floor(in / pool_size) / pool_size)
    int out_D = (D / pool_size) / pool_size;
    int out_H = (H / pool_size) / pool_size;
    int out_W = (W / pool_size) / pool_size;
    
    auto output = torch::empty({N, C, out_D, out_H, out_W}, input.options());
    
    int total_blocks = N * out_D * out_H * out_W;
    int block_size = max(C, 32);
    block_size = min(block_size, 256);
    
    size_t shared_mem = (C + 1) * sizeof(float);
    
    if (pool_size == 2) {
        if (input.scalar_type() == torch::kFloat32) {
            fused_kernel_v2<float, 2><<<total_blocks, block_size, shared_mem>>>(
                input.data_ptr<float>(),
                output.data_ptr<float>(),
                N, C, D, H, W, out_D, out_H, out_W);
        } else if (input.scalar_type() == torch::kHalf) {
            fused_kernel_v2<at::Half, 2><<<total_blocks, block_size, shared_mem>>>(
                input.data_ptr<at::Half>(),
                output.data_ptr<at::Half>(),
                N, C, D, H, W, out_D, out_H, out_W);
        }
    }
    
    return output;
}
'''

cpp_source = r'''
torch::Tensor fused_softmax_maxpool_cuda(torch::Tensor input, int pool_size);
'''

ext = load_inline(
    name='fused_ops',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_softmax_maxpool_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3'],
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(Model, self).__init__()
        self.conv = nn.Conv3d(in_channels, out_channels, kernel_size)
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        # Conv3d
        x = self.conv(x)
        
        # Fused softmax + double maxpool
        x = ext.fused_softmax_maxpool_cuda(x, self.pool_kernel_size)
        
        return x

def get_inputs():
    batch_size = 16
    in_channels = 3
    depth, height, width = 16, 128, 128
    return [torch.randn(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    in_channels = 3
    out_channels = 16
    kernel_size = 3
    pool_kernel_size = 2
    return [in_channels, out_channels, kernel_size, pool_kernel_size]
