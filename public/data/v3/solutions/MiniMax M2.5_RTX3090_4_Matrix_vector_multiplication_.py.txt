import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Optimized CUDA kernel for matrix-vector multiplication
cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

#define THREADS_PER_BLOCK 256

// Optimized kernel with shared memory tiling for B vector
// Each block computes one output element (one row of the matrix)
__global__ void matvec_kernel(const float* __restrict__ A, 
                               const float* __restrict__ B, 
                               float* __restrict__ C, 
                               int M, int K) {
    // Shared memory for B vector tile
    extern __shared__ float s_B[];
    
    const int TILE_SIZE = 512;
    
    const int row = blockIdx.x;
    if (row >= M) return;
    
    const float* A_row = A + row * K;
    const int tid = threadIdx.x;
    
    float sum = 0.0f;
    
    // Process K in tiles
    for (int tile_start = 0; tile_start < K; tile_start += TILE_SIZE) {
        int tile_end = min(tile_start + TILE_SIZE, K);
        int tile_size = tile_end - tile_start;
        
        // Load tile of B into shared memory (coalesced)
        for (int i = tid; i < tile_size; i += THREADS_PER_BLOCK) {
            s_B[i] = B[tile_start + i];
        }
        __syncthreads();
        
        // Compute partial dot product - each thread computes elements at stride THREADS_PER_BLOCK
        for (int i = 0; i < tile_size; i += THREADS_PER_BLOCK) {
            int idx = tile_start + i + tid;
            if (idx < tile_end) {
                sum += A_row[idx] * s_B[idx - tile_start];
            }
        }
        
        __syncthreads();
    }
    
    // First thread writes result
    if (tid == 0) {
        C[row] = sum;
    }
}

// Launch wrapper
torch::Tensor matvec_op(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    
    auto C = torch::empty({M, 1}, A.options());
    
    const int block_size = 256;
    const int shared_mem = 512 * sizeof(float);
    
    dim3 grid(M);
    dim3 block(block_size);
    
    matvec_kernel<<<grid, block, shared_mem>>>(
        A.data_ptr<float>(),
        B.data_ptr<float>(),
        C.data_ptr<float>(),
        M, K
    );
    
    cudaDeviceSynchronize();
    
    return C;
}
'''

ext = load_inline(
    name='matvec_ext',
    cpp_sources='torch::Tensor matvec_op(torch::Tensor A, torch::Tensor B);',
    cuda_sources=cuda_source,
    functions=['matvec_op'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, A: torch.Tensor, B: torch.Tensor) -> torch.Tensor:
        return ext.matvec_op(A, B)

M = 256
K = 131072

def get_inputs():
    A = torch.randn(M, K)
    B = torch.randn(K, 1)
    return [A, B]

def get_init_inputs():
    return []
