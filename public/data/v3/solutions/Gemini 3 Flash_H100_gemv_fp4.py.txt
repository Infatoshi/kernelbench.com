import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

__global__ void gemv_fp4_kernel(
    const int8_t* __restrict__ x_q,
    const int8_t* __restrict__ weight_q,
    const float x_scale,
    const float weight_scale,
    half* __restrict__ output,
    int batch_size,
    int in_features,
    int out_features) {

    int batch_idx = blockIdx.y;
    int out_idx = blockIdx.x * blockDim.x + threadIdx.x;

    if (out_idx < out_features && batch_idx < batch_size) {
        float sum = 0.0f;
        const int8_t* x_ptr = x_q + batch_idx * in_features;
        const int8_t* w_ptr = weight_q + out_idx * in_features;

        // Use int4 vectorization or just better loop unrolling
        const float4* x_vec = (const float4*)x_ptr;
        const float4* w_vec = (const float4*)w_ptr;
        int n_vec = in_features / 4;

        #pragma unroll 4
        for (int i = 0; i < n_vec; ++i) {
            char4 xv = *(char4*)&x_vec[i]; // This is wrong, char4 is 4 bytes.
            char4 wv = *(char4*)&w_vec[i];
            sum += (float)xv.x * (float)wv.x;
            sum += (float)xv.y * (float)wv.y;
            sum += (float)xv.z * (float)wv.z;
            sum += (float)xv.w * (float)wv.w;
        }
        output[batch_idx * out_features + out_idx] = __float2half(sum * x_scale * weight_scale);
    }
}

torch::Tensor gemv_fp4(torch::Tensor x_q, float x_scale, torch::Tensor weight_q, float weight_scale) {
    int batch_size = x_q.size(0);
    int in_features = x_q.size(1);
    int out_features = weight_q.size(0);

    auto options = torch::TensorOptions().dtype(torch::kHalf).device(x_q.device());
    auto output = torch::empty({batch_size, out_features}, options);

    // Optimize block size for H100
    dim3 block(128);
    dim3 grid((out_features + block.x - 1) / block.x, batch_size);

    gemv_fp4_kernel<<<grid, block>>>(
        (const int8_t*)x_q.data_ptr<int8_t>(),
        (const int8_t*)weight_q.data_ptr<int8_t>(),
        x_scale,
        weight_scale,
        (half*)output.data_ptr<at::Half>(),
        batch_size,
        in_features,
        out_features
    );

    return output;
}
'''

cpp_source = r'''
torch::Tensor gemv_fp4(torch::Tensor x_q, float x_scale, torch::Tensor weight_q, float weight_scale);
'''

ext = load_inline(
    name='gemv_fp4_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['gemv_fp4'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_features: int = 4096, out_features: int = 14336):
        super().__init__()
        self.register_buffer("weight_q", torch.randint(-8, 8, (out_features, in_features), dtype=torch.int8))
        self.register_buffer("weight_scale", torch.tensor(0.08, dtype=torch.float32))

    def forward(self, x_q: torch.Tensor, x_scale: torch.Tensor) -> torch.Tensor:
        # x_q: [batch, in_features], weight_q: [out_features, in_features]
        return ext.gemv_fp4(x_q, x_scale.item(), self.weight_q, self.weight_scale.item())

def get_inputs():
    x_q = torch.randint(-8, 8, (32, 4096), dtype=torch.int8).cuda()
    x_scale = torch.tensor(0.08, dtype=torch.float32).cuda()
    return [x_q, x_scale]

def get_init_inputs():
    return []
