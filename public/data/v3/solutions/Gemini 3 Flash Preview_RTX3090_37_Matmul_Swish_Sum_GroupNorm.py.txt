import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.autotune(
    configs=[
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=3, num_warps=8),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 128, 'BLOCK_SIZE_N': 64, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
        triton.Config({'BLOCK_SIZE_M': 64, 'BLOCK_SIZE_N': 128, 'BLOCK_SIZE_K': 32, 'GROUP_SIZE_M': 8}, num_stages=4, num_warps=4),
    ],
    key=['M', 'N', 'K'],
)
@triton.jit
def matmul_swish_bias_kernel(
    X_ptr, W_ptr, MB_ptr, SB_ptr, Out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_om, stride_on,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    # 2D swizzle
    width = GROUP_SIZE_M * num_pid_n
    group_id = pid // width
    group_size = min(num_pid_m - group_id * GROUP_SIZE_M, GROUP_SIZE_M)
    pid_m = group_id * GROUP_SIZE_M + (pid % group_size)
    pid_n = (pid % width) // group_size

    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)
    
    a_ptr = X_ptr + (offs_am[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    b_ptr = W_ptr + (offs_bn[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        k_mask = (offs_k < K - k * BLOCK_SIZE_K)
        mask_a = (offs_am[:, None] < M) & k_mask[None, :]
        mask_b = (offs_bn[None, :] < N) & k_mask[:, None]
        a = tl.load(a_ptr, mask=mask_a, other=0.0)
        b = tl.load(b_ptr, mask=mask_b, other=0.0)
        accumulator += tl.dot(a, b)
        a_ptr += BLOCK_SIZE_K * stride_xk
        b_ptr += BLOCK_SIZE_K * stride_wk

    bias_m = tl.load(MB_ptr + offs_bn, mask=offs_bn < N, other=0.0)[None, :]
    bias_s = tl.load(SB_ptr + offs_bn, mask=offs_bn < N, other=0.0)[None, :]
    
    x = accumulator + bias_m
    x = x * tl.sigmoid(x)
    x = x + bias_s
    
    offs_om = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_on = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    out_ptr = Out_ptr + (offs_om[:, None] * stride_om + offs_on[None, :] * stride_on)
    mask = (offs_om[:, None] < M) & (offs_on[None, :] < N)
    tl.store(out_ptr, x.to(Out_ptr.dtype.element_ty), mask=mask)

@triton.jit
def group_norm_kernel(
    X_ptr, G_ptr, B_ptr, Out_ptr,
    M, N, groups, C_G,
    stride_xm, stride_xn,
    stride_om, stride_on,
    BLOCK_SIZE_N: tl.constexpr,
):
    pid_m = tl.program_id(0)
    pid_g = tl.program_id(1)
    
    offs_n = pid_g * C_G + tl.arange(0, BLOCK_SIZE_N)
    mask = (offs_n < (pid_g + 1) * C_G) & (offs_n < N)
    
    x_ptr = X_ptr + pid_m * stride_xm + offs_n * stride_xn
    x = tl.load(x_ptr, mask=mask, other=0.0).to(tl.float32)
    
    mean = tl.sum(x, axis=0) / C_G
    var = tl.sum((x - mean) * (x - mean), axis=0) / C_G
    rstd = 1.0 / tl.sqrt(var + 1e-5)
    
    gamma = tl.load(G_ptr + offs_n, mask=mask, other=0.0).to(tl.float32)
    beta = tl.load(B_ptr + offs_n, mask=mask, other=0.0).to(tl.float32)
    
    res = (x - mean) * rstd * gamma + beta
    
    out_ptr = Out_ptr + pid_m * stride_om + offs_n * stride_on
    tl.store(out_ptr, res.to(Out_ptr.dtype.element_ty), mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(Model, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups
        self.matmul = nn.Linear(in_features, out_features)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        M, K = x.shape
        N = self.out_features
        
        mid_out = torch.empty((M, N), device=x.device, dtype=x.dtype)
        
        grid = lambda META: (triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),)
        matmul_swish_bias_kernel[grid](
            x, self.matmul.weight, self.matmul.bias, self.bias, mid_out,
            M, N, K,
            x.stride(0), x.stride(1),
            self.matmul.weight.stride(0), self.matmul.weight.stride(1),
            mid_out.stride(0), mid_out.stride(1)
        )
        
        output = torch.empty((M, N), device=x.device, dtype=x.dtype)
        C_G = N // self.num_groups
        BLOCK_SIZE_N_GN = triton.next_power_of_2(C_G)
        
        gn_grid = (M, self.num_groups)
        group_norm_kernel[gn_grid](
            mid_out, self.group_norm.weight, self.group_norm.bias, output,
            M, N, self.num_groups, C_G,
            mid_out.stride(0), mid_out.stride(1),
            output.stride(0), output.stride(1),
            BLOCK_SIZE_N=BLOCK_SIZE_N_GN,
            num_warps=4
        )
        
        return output

def get_inputs():
    return [torch.randn(128, 4096, device='cuda')]

def get_init_inputs():
    return [4096, 4096, 32, (4096,)]
