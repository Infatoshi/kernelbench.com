import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Simple INT4 GEMM kernel without WMMA first to verify correctness
__global__ void int4_gemm_simple(
    const half* __restrict__ X,
    const uint8_t* __restrict__ W_packed,
    const half* __restrict__ scales,
    half* __restrict__ Y,
    int M, int N, int K, int group_size
) {
    int row = blockIdx.y * blockDim.y + threadIdx.y;
    int col = blockIdx.x * blockDim.x + threadIdx.x;
    
    if (row >= M || col >= N) return;
    
    int num_groups = K / group_size;
    float sum = 0.0f;
    
    for (int k = 0; k < K; k++) {
        // Load X value
        half x_val = X[row * K + k];
        
        // Unpack and dequantize W value
        int packed_idx = k / 2;
        uint8_t packed = W_packed[col * (K / 2) + packed_idx];
        
        int w_int;
        if (k % 2 == 0) {
            w_int = packed & 0x0F;
        } else {
            w_int = (packed >> 4) & 0x0F;
        }
        
        int group_idx = k / group_size;
        half scale = scales[col * num_groups + group_idx];
        
        float w_val = __half2float(scale) * (float)(w_int - 8);
        sum += __half2float(x_val) * w_val;
    }
    
    Y[row * N + col] = __float2half(sum);
}

torch::Tensor int4_gemm(
    torch::Tensor X,
    torch::Tensor W_packed,
    torch::Tensor scales,
    int group_size
) {
    int M = X.size(0);
    int K = X.size(1);
    int N = W_packed.size(0);
    
    auto Y = torch::zeros({M, N}, X.options());
    
    dim3 block(16, 16);
    dim3 grid((N + 15) / 16, (M + 15) / 16);
    
    int4_gemm_simple<<<grid, block>>>(
        reinterpret_cast<const half*>(X.data_ptr<at::Half>()),
        W_packed.data_ptr<uint8_t>(),
        reinterpret_cast<const half*>(scales.data_ptr<at::Half>()),
        reinterpret_cast<half*>(Y.data_ptr<at::Half>()),
        M, N, K, group_size
    );
    
    return Y;
}
'''

cpp_source = r'''
torch::Tensor int4_gemm(
    torch::Tensor X,
    torch::Tensor W_packed,
    torch::Tensor scales,
    int group_size
);
'''

ext = load_inline(
    name='int4_gemm_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['int4_gemm'],
    extra_cuda_cflags=['-arch=sm_86', '-O3', '--use_fast_math'],
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, K: int, N: int, group_size: int = 128):
        super().__init__()
        self.K = K
        self.N = N
        self.group_size = group_size
        self.num_groups = K // group_size

        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even for INT4 packing"

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        self.register_buffer(
            "weight_packed",
            torch.randint(0, 256, (N, K // 2), dtype=torch.uint8)
        )
        self.register_buffer(
            "scales",
            torch.randn(N, self.num_groups, dtype=torch.float16).abs() * 0.1
        )
        torch.random.set_rng_state(rng_state)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        batch_size, seq_len, _ = x.shape
        x_2d = x.view(-1, self.K)
        out = ext.int4_gemm(x_2d, self.weight_packed, self.scales, self.group_size)
        return out.view(batch_size, seq_len, self.N)


def get_inputs():
    return [torch.randn(batch_size, seq_len, K, dtype=torch.float16)]


def get_init_inputs():
    return [K, N, group_size]


batch_size = 4
seq_len = 2048
K = 4096
N = 11008
group_size = 128
