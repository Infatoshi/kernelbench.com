import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

#define TILE_SIZE 16
#define KERNEL_MAX 7

// Generic depthwise convolution kernel for any kernel size
template <typename T>
__global__ void depthwise_conv_kernel(
    const T* __restrict__ input,
    T* __restrict__ output,
    const T* __restrict__ weight,
    int batch_size,
    int channels,
    int height,
    int width,
    int out_height,
    int out_width,
    int stride,
    int padding,
    int kernel_size) {
    
    // Shared memory for kernel weights
    __shared__ T weight_smem[KERNEL_MAX * KERNEL_MAX];
    
    // Output tile position
    int out_x = blockIdx.x * TILE_SIZE + threadIdx.x;
    int out_y = blockIdx.y * TILE_SIZE + threadIdx.y;
    int bc = blockIdx.z;
    
    int b = bc / channels;
    int c = bc % channels;
    
    if (b >= batch_size || c >= channels) return;
    if (out_x >= out_width || out_y >= out_height) return;
    
    // Load kernel weights into shared memory
    if (threadIdx.x < kernel_size && threadIdx.y < kernel_size) {
        weight_smem[threadIdx.y * kernel_size + threadIdx.x] = 
            weight[c * kernel_size * kernel_size + threadIdx.y * kernel_size + threadIdx.x];
    }
    __syncthreads();
    
    // Compute input position
    int in_x = out_x * stride - padding;
    int in_y = out_y * stride - padding;
    
    // Accumulate convolution result
    T result = static_cast<T>(0);
    
    // Loop over kernel
    for (int ky = 0; ky < kernel_size; ky++) {
        for (int kx = 0; kx < kernel_size; kx++) {
            int ix = in_x + kx;
            int iy = in_y + ky;
            if (iy >= 0 && iy < height && ix >= 0 && ix < width) {
                int input_idx = ((b * channels + c) * height + iy) * width + ix;
                int weight_idx = ky * kernel_size + kx;
                result += input[input_idx] * weight_smem[weight_idx];
            }
        }
    }
    
    int out_idx = ((b * channels + c) * out_height + out_y) * out_width + out_x;
    output[out_idx] = result;
}

// Kernel wrapper
torch::Tensor depthwise_conv2d_fwd(
    torch::Tensor input,
    torch::Tensor weight,
    int stride,
    int padding) {
    
    int batch_size = input.size(0);
    int channels = input.size(1);
    int height = input.size(2);
    int width = input.size(3);
    int kernel_size = weight.size(2);
    
    int out_height = (height + 2 * padding - kernel_size) / stride + 1;
    int out_width = (width + 2 * padding - kernel_size) / stride + 1;
    
    auto output = torch::empty({batch_size, channels, out_height, out_width}, 
                               input.options());
    
    const dim3 block(TILE_SIZE, TILE_SIZE);
    dim3 grid;
    grid.x = (out_width + TILE_SIZE - 1) / TILE_SIZE;
    grid.y = (out_height + TILE_SIZE - 1) / TILE_SIZE;
    grid.z = batch_size * channels;
    
    AT_DISPATCH_FLOATING_TYPES(input.scalar_type(), "depthwise_conv2d_cuda", ([&] {
        depthwise_conv_kernel<scalar_t><<<grid, block>>>(
            input.data_ptr<scalar_t>(),
            output.data_ptr<scalar_t>(),
            weight.data_ptr<scalar_t>(),
            batch_size, channels, height, width,
            out_height, out_width, stride, padding, kernel_size);
    }));
    
    return output;
}
'''

ext = load_inline(
    name='depthwise_conv_ext',
    cpp_sources='''
        torch::Tensor depthwise_conv2d_fwd(torch::Tensor input, torch::Tensor weight, int stride, int padding);
    ''',
    cuda_sources=cuda_source,
    functions=['depthwise_conv2d_fwd'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(Model, self).__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Use conv2d as submodule name to match reference state_dict keys
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, 
                                padding=padding, groups=in_channels, bias=bias)
        # Disable PyTorch's autotuning for convolution
        self.conv2d._reversed_padding_repeated_twice = (0, 0, 0, 0)
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Extract weight with shape (C, 1, K, K) for depthwise conv
        weight = self.conv2d.weight
        return ext.depthwise_conv2d_fwd(x, weight, self.stride, self.padding)

def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width)
    return [x]

def get_init_inputs():
    return [3, 3, 1, 0]  # in_channels, kernel_size, stride, padding
