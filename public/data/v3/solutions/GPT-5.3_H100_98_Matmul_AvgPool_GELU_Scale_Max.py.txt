import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def fused_kernel(X, W, B, OUT, K, N_GROUPS, SCALE, BLOCK_K: tl.constexpr):
    row = tl.program_id(0)

    x_ptr = X + row * K

    max_val = -1e30

    for g in range(0, N_GROUPS):
        acc0 = 0.0
        acc1 = 0.0
        acc2 = 0.0
        acc3 = 0.0

        w_row0 = (g*4 + 0) * K
        w_row1 = (g*4 + 1) * K
        w_row2 = (g*4 + 2) * K
        w_row3 = (g*4 + 3) * K

        for k in range(0, K, BLOCK_K):
            k_offsets = k + tl.arange(0, BLOCK_K)
            mask = k_offsets < K

            x = tl.load(x_ptr + k_offsets, mask=mask, other=0.0).to(tl.float32)

            w0 = tl.load(W + w_row0 + k_offsets, mask=mask, other=0.0).to(tl.float32)
            w1 = tl.load(W + w_row1 + k_offsets, mask=mask, other=0.0).to(tl.float32)
            w2 = tl.load(W + w_row2 + k_offsets, mask=mask, other=0.0).to(tl.float32)
            w3 = tl.load(W + w_row3 + k_offsets, mask=mask, other=0.0).to(tl.float32)

            acc0 += tl.sum(x * w0)
            acc1 += tl.sum(x * w1)
            acc2 += tl.sum(x * w2)
            acc3 += tl.sum(x * w3)

        b0 = tl.load(B + g*4 + 0)
        b1 = tl.load(B + g*4 + 1)
        b2 = tl.load(B + g*4 + 2)
        b3 = tl.load(B + g*4 + 3)

        acc0 += b0
        acc1 += b1
        acc2 += b2
        acc3 += b3

        avg = (acc0 + acc1 + acc2 + acc3) * 0.25

        # GELU exact via erf
        inv_sqrt2 = 0.70710678118
        avg = 0.5 * avg * (1.0 + tl.erf(avg * inv_sqrt2))

        avg = avg * SCALE

        max_val = tl.maximum(max_val, avg)

    tl.store(OUT + row, max_val)


class Model(nn.Module):
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features)
        self.scale = scale_factor
        self.pool_k = pool_kernel_size

    def forward(self, x):
        W = self.linear.weight
        B = self.linear.bias

        Bsz, K = x.shape
        N = W.shape[0]
        groups = N // self.pool_k

        out = torch.empty((Bsz,), device=x.device, dtype=x.dtype)

        fused_kernel[(Bsz,)](
            x, W, B, out,
            K, groups, self.scale,
            BLOCK_K=64
        )
        return out
