import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cfloat>
#include <cuda_fp16.h>

// Optimized softmax kernel - one block per row
// Uses float internally for numerical stability

template <int BLOCK_SIZE>
__device__ float block_reduce_max(float val) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    // Warp reduction using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    }
    
    return val;
}

template <int BLOCK_SIZE>
__device__ float block_reduce_sum(float val) {
    int lane = threadIdx.x & 31;
    int wid = threadIdx.x >> 5;
    
    // Warp reduction using shuffle
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    
    return val;
}

// FP32 kernel
template <int BLOCK_SIZE>
__global__ void softmax_kernel_fp32(const float* __restrict__ input, float* __restrict__ output, int dim) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    
    __shared__ float shared_max[BLOCK_SIZE / 32];
    __shared__ float shared_sum[BLOCK_SIZE / 32];
    
    const float* row_input = input + row * dim;
    float* row_output = output + row * dim;
    
    // Find max - parallel reduction across threads
    float local_max = -FLT_MAX;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        local_max = fmaxf(local_max, row_input[i]);
    }
    
    // Warp-level reduce
    local_max = block_reduce_max<BLOCK_SIZE>(local_max);
    
    // First thread in each warp writes to shared
    int lane = tid & 31;
    int wid = tid >> 5;
    if (lane == 0) {
        shared_max[wid] = local_max;
    }
    __syncthreads();
    
    // First warp does final reduction
    float max_val;
    if (wid == 0) {
        max_val = (tid < (BLOCK_SIZE >> 5)) ? shared_max[tid] : -FLT_MAX;
        max_val = block_reduce_max<BLOCK_SIZE>(max_val);
    }
    
    // Broadcast max from first thread
    __shared__ float final_max;
    if (tid == 0) {
        final_max = max_val;
    }
    __syncthreads();
    max_val = final_max;
    
    // Compute exp(x - max) and sum
    float local_sum = 0.0f;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float exp_val = expf(row_input[i] - max_val);
        row_output[i] = exp_val;
        local_sum += exp_val;
    }
    
    // Warp-level reduce sum
    local_sum = block_reduce_sum<BLOCK_SIZE>(local_sum);
    
    if (lane == 0) {
        shared_sum[wid] = local_sum;
    }
    __syncthreads();
    
    // First warp does final reduction
    float sum_val;
    if (wid == 0) {
        sum_val = (tid < (BLOCK_SIZE >> 5)) ? shared_sum[tid] : 0.0f;
        sum_val = block_reduce_sum<BLOCK_SIZE>(sum_val);
    }
    
    // Broadcast sum from first thread
    __shared__ float final_sum;
    if (tid == 0) {
        final_sum = sum_val;
    }
    __syncthreads();
    sum_val = final_sum;
    
    float inv_sum = 1.0f / sum_val;
    
    // Normalize
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        row_output[i] *= inv_sum;
    }
}

// FP16 kernel - reads as half, computes in float, writes as half
template <int BLOCK_SIZE>
__global__ void softmax_kernel_fp16(const at::Half* __restrict__ input, at::Half* __restrict__ output, int dim) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    
    __shared__ float shared_max[BLOCK_SIZE / 32];
    __shared__ float shared_sum[BLOCK_SIZE / 32];
    
    const at::Half* row_input = input + row * dim;
    at::Half* row_output = output + row * dim;
    
    // Find max
    float local_max = -FLT_MAX;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = __half2float(row_input[i]);
        local_max = fmaxf(local_max, val);
    }
    
    local_max = block_reduce_max<BLOCK_SIZE>(local_max);
    
    int lane = tid & 31;
    int wid = tid >> 5;
    if (lane == 0) {
        shared_max[wid] = local_max;
    }
    __syncthreads();
    
    float max_val;
    if (wid == 0) {
        max_val = (tid < (BLOCK_SIZE >> 5)) ? shared_max[tid] : -FLT_MAX;
        max_val = block_reduce_max<BLOCK_SIZE>(max_val);
    }
    
    __shared__ float final_max;
    if (tid == 0) {
        final_max = max_val;
    }
    __syncthreads();
    max_val = final_max;
    
    // Compute exp and sum, keep intermediate in float
    float local_sum = 0.0f;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = __half2float(row_input[i]);
        float exp_val = expf(val - max_val);
        row_output[i] = __float2half(exp_val);
        local_sum += exp_val;
    }
    
    local_sum = block_reduce_sum<BLOCK_SIZE>(local_sum);
    
    if (lane == 0) {
        shared_sum[wid] = local_sum;
    }
    __syncthreads();
    
    float sum_val;
    if (wid == 0) {
        sum_val = (tid < (BLOCK_SIZE >> 5)) ? shared_sum[tid] : 0.0f;
        sum_val = block_reduce_sum<BLOCK_SIZE>(sum_val);
    }
    
    __shared__ float final_sum;
    if (tid == 0) {
        final_sum = sum_val;
    }
    __syncthreads();
    sum_val = final_sum;
    
    float inv_sum = 1.0f / sum_val;
    
    // Normalize
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = __half2float(row_output[i]);
        row_output[i] = __float2half(val * inv_sum);
    }
}

// BF16 kernel - reads as bf16, computes in float, writes as bf16
template <int BLOCK_SIZE>
__global__ void softmax_kernel_bf16(const at::BFloat16* __restrict__ input, at::BFloat16* __restrict__ output, int dim) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    
    __shared__ float shared_max[BLOCK_SIZE / 32];
    __shared__ float shared_sum[BLOCK_SIZE / 32];
    
    const at::BFloat16* row_input = input + row * dim;
    at::BFloat16* row_output = output + row * dim;
    
    // Find max
    float local_max = -FLT_MAX;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = float(row_input[i]);
        local_max = fmaxf(local_max, val);
    }
    
    local_max = block_reduce_max<BLOCK_SIZE>(local_max);
    
    int lane = tid & 31;
    int wid = tid >> 5;
    if (lane == 0) {
        shared_max[wid] = local_max;
    }
    __syncthreads();
    
    float max_val;
    if (wid == 0) {
        max_val = (tid < (BLOCK_SIZE >> 5)) ? shared_max[tid] : -FLT_MAX;
        max_val = block_reduce_max<BLOCK_SIZE>(max_val);
    }
    
    __shared__ float final_max;
    if (tid == 0) {
        final_max = max_val;
    }
    __syncthreads();
    max_val = final_max;
    
    // Compute exp and sum
    float local_sum = 0.0f;
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = float(row_input[i]);
        float exp_val = expf(val - max_val);
        row_output[i] = at::BFloat16(exp_val);
        local_sum += exp_val;
    }
    
    local_sum = block_reduce_sum<BLOCK_SIZE>(local_sum);
    
    if (lane == 0) {
        shared_sum[wid] = local_sum;
    }
    __syncthreads();
    
    float sum_val;
    if (wid == 0) {
        sum_val = (tid < (BLOCK_SIZE >> 5)) ? shared_sum[tid] : 0.0f;
        sum_val = block_reduce_sum<BLOCK_SIZE>(sum_val);
    }
    
    __shared__ float final_sum;
    if (tid == 0) {
        final_sum = sum_val;
    }
    __syncthreads();
    sum_val = final_sum;
    
    float inv_sum = 1.0f / sum_val;
    
    // Normalize
    for (int i = tid; i < dim; i += BLOCK_SIZE) {
        float val = float(row_output[i]);
        row_output[i] = at::BFloat16(val * inv_sum);
    }
}

torch::Tensor softmax_cuda(torch::Tensor x) {
    auto output = torch::empty_like(x);
    int batch_size = x.size(0);
    int dim = x.size(1);
    
    int block_size = 256;
    
    if (x.scalar_type() == torch::kFloat32) {
        softmax_kernel_fp32<256><<<batch_size, block_size>>>(
            x.data_ptr<float>(), output.data_ptr<float>(), dim);
    } else if (x.scalar_type() == torch::kFloat16) {
        softmax_kernel_fp16<256><<<batch_size, block_size>>>(
            reinterpret_cast<const at::Half*>(x.data_ptr<at::Half>()), 
            reinterpret_cast<at::Half*>(output.data_ptr<at::Half>()), dim);
    } else if (x.scalar_type() == torch::kBFloat16) {
        softmax_kernel_bf16<256><<<batch_size, block_size>>>(
            x.data_ptr<at::BFloat16>(), output.data_ptr<at::BFloat16>(), dim);
    } else {
        AT_ERROR("Unsupported dtype for softmax");
    }
    
    return output;
}
'''

ext = load_inline(
    name='softmax_ext',
    cpp_sources='torch::Tensor softmax_cuda(torch::Tensor);',
    cuda_sources=cuda_source,
    functions=['softmax_cuda'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-use_fast_math'],
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x):
        return ext.softmax_cuda(x)

def get_inputs():
    return [torch.randn(256, 16384)]

def get_init_inputs():
    return []
