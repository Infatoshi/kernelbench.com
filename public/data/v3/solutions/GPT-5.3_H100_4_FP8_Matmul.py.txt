import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, M: int, K: int, N: int, use_e4m3: bool = True):
        super().__init__()
        self.M = M
        self.K = K
        self.N = N
        self.use_e4m3 = use_e4m3

        if use_e4m3:
            self.fp8_dtype = torch.float8_e4m3fn
            self.fp8_max = 448.0
        else:
            self.fp8_dtype = torch.float8_e5m2
            self.fp8_max = 57344.0

        rng_state = torch.random.get_rng_state()
        torch.manual_seed(1337)
        weight = torch.randn(K, N) * 0.02
        torch.random.set_rng_state(rng_state)

        self.weight = nn.Parameter(weight)

        # Pre-quantized buffers (initialized later)
        self.register_buffer("w_fp8", None, persistent=False)
        self.register_buffer("w_scale_inv", None, persistent=False)

    def compute_scale(self, x: torch.Tensor) -> torch.Tensor:
        amax = x.abs().max()
        return self.fp8_max / amax.clamp(min=1e-12)

    def quantize_to_fp8(self, x: torch.Tensor, scale: torch.Tensor) -> torch.Tensor:
        x_scaled = x * scale
        x_clamped = x_scaled.clamp(-self.fp8_max, self.fp8_max)
        return x_clamped.to(self.fp8_dtype)

    def _prepare_weight(self):
        # Precompute FP8 weight + scale once
        if self.w_fp8 is not None:
            return
        w_t = self.weight.t().contiguous()  # (N, K)
        w_scale = self.compute_scale(w_t)
        w_fp8 = self.quantize_to_fp8(w_t, w_scale)
        self.w_fp8 = w_fp8
        self.w_scale_inv = (1.0 / w_scale).to(torch.float32)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        input_dtype = x.dtype
        B, S, _ = x.shape

        self._prepare_weight()

        x_2d = x.view(-1, self.K)

        x_scale = self.compute_scale(x_2d)
        x_fp8 = self.quantize_to_fp8(x_2d, x_scale)
        x_scale_inv = (1.0 / x_scale).to(torch.float32)

        out = torch._scaled_mm(
            x_fp8,
            self.w_fp8.t(),
            scale_a=x_scale_inv,
            scale_b=self.w_scale_inv,
            out_dtype=input_dtype,
        )

        return out.view(B, S, self.N)


def get_inputs():
    return [torch.randn(8, 2048, 4096, dtype=torch.float16, device='cuda')]


def get_init_inputs():
    return [8*2048, 4096, 4096, True]
