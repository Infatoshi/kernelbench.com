import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

__inline__ __device__ float warp_reduce_sum(float val) {
    for (int offset = 16; offset > 0; offset /= 2)
        val += __shfl_down_sync(0xffffffff, val, offset);
    return val;
}

__inline__ __device__ float warp_reduce_max(float val) {
    for (int offset = 16; offset > 0; offset /= 2)
        val = fmaxf(val, __shfl_down_sync(0xffffffff, val, offset));
    return val;
}

__global__ void cross_entropy_kernel(
    const float* __restrict__ logits,
    const int64_t* __restrict__ targets,
    float* __restrict__ losses,
    int B, int C
) {
    int b = blockIdx.x;
    if (b >= B) return;

    const float* row = logits + b * C;
    int tid = threadIdx.x;

    __shared__ float sdata[256];

    // compute max
    float max_val = -1e20f;
    for (int i = tid; i < C; i += blockDim.x) {
        max_val = fmaxf(max_val, row[i]);
    }
    sdata[tid] = max_val;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] = fmaxf(sdata[tid], sdata[tid + s]);
        }
        __syncthreads();
    }
    max_val = sdata[0];

    // compute sum exp
    float sum = 0.f;
    for (int i = tid; i < C; i += blockDim.x) {
        sum += expf(row[i] - max_val);
    }
    sdata[tid] = sum;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s >>= 1) {
        if (tid < s) {
            sdata[tid] += sdata[tid + s];
        }
        __syncthreads();
    }
    sum = sdata[0];

    if (tid == 0) {
        int t = targets[b];
        float log_prob = row[t] - max_val - logf(sum);
        losses[b] = -log_prob;
    }
}


torch::Tensor cross_entropy(torch::Tensor logits, torch::Tensor targets) {
    int B = logits.size(0);
    int C = logits.size(1);

    auto losses = torch::empty({B}, logits.options());

    cross_entropy_kernel<<<B, 256>>>(
        logits.data_ptr<float>(),
        targets.data_ptr<int64_t>(),
        losses.data_ptr<float>(),
        B, C
    );

    return losses.mean();
}
'''

ext = load_inline(
    name='ce_ext',
    cpp_sources='torch::Tensor cross_entropy(torch::Tensor logits, torch::Tensor targets);',
    cuda_sources=cuda_src,
    functions=['cross_entropy'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, predictions, targets):
        if predictions.dtype != torch.float32:
            predictions = predictions.float()
        return ext.cross_entropy(predictions, targets)

    return [torch.randn(4096, 1024, device='cuda'), torch.randint(0, 1024, (4096,), device='cuda')]
