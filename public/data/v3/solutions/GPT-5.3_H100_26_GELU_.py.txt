import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

#define GELU_COEFF 0.7978845608028654f  // sqrt(2/pi)
#define GELU_CONST 0.044715f

__device__ __forceinline__ float gelu_float(float x) {
    float x3 = x * x * x;
    float t = GELU_COEFF * (x + GELU_CONST * x3);
    return 0.5f * x * (1.0f + tanhf(t));
}

__global__ void gelu_fp32(const float* __restrict__ x, float* __restrict__ y, int n) {
    for (int i = blockIdx.x * blockDim.x + threadIdx.x; i < n; i += blockDim.x * gridDim.x) {
        y[i] = gelu_float(x[i]);
    }
}

__global__ void gelu_fp16(const half* __restrict__ x, half* __restrict__ y, int n) {
    for (int i = blockIdx.x * blockDim.x + threadIdx.x; i < n; i += blockDim.x * gridDim.x) {
        float xf = __half2float(x[i]);
        float yf = gelu_float(xf);
        y[i] = __float2half(yf);
    }
}

__global__ void gelu_bf16(const nv_bfloat16* __restrict__ x, nv_bfloat16* __restrict__ y, int n) {
    for (int i = blockIdx.x * blockDim.x + threadIdx.x; i < n; i += blockDim.x * gridDim.x) {
        float xf = __bfloat162float(x[i]);
        float yf = gelu_float(xf);
        y[i] = __float2bfloat16(yf);
    }
}

torch::Tensor gelu_forward(torch::Tensor x) {
    auto y = torch::empty_like(x);
    int n = x.numel();

    int threads = 256;
    int blocks = (n + threads - 1) / threads;
    blocks = min(blocks, 4096);

    if (x.scalar_type() == torch::kFloat32) {
        gelu_fp32<<<blocks, threads>>>(x.data_ptr<float>(), y.data_ptr<float>(), n);
    } else if (x.scalar_type() == torch::kFloat16) {
        gelu_fp16<<<blocks, threads>>>(reinterpret_cast<const half*>(x.data_ptr<at::Half>()),
                                      reinterpret_cast<half*>(y.data_ptr<at::Half>()), n);
    } else if (x.scalar_type() == torch::kBFloat16) {
        gelu_bf16<<<blocks, threads>>>(reinterpret_cast<const nv_bfloat16*>(x.data_ptr<at::BFloat16>()),
                                      reinterpret_cast<nv_bfloat16*>(y.data_ptr<at::BFloat16>()), n);
    }

    return y;
}
'''

ext = load_inline(
    name='gelu_fast_ext',
    cpp_sources='torch::Tensor gelu_forward(torch::Tensor x);',
    cuda_sources=cuda_source,
    functions=['gelu_forward'],
    verbose=False,
)

class Model(nn.Module):
    def __init__(self):
        super().__init__()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ext.gelu_forward(x)


def get_inputs():
    batch_size = 256
    dim = 16384
    x = torch.randn(batch_size, dim, device='cuda')
    return [x]


def get_init_inputs():
    return []
