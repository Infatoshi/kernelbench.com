from torch.utils.cpp_extension import load_inline

cuda_source = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_fp16.h>

// Fused Conv2d + Activation + BatchNorm kernel
// Activation: tanh(softplus(x)) * x where softplus(x) = log(1 + exp(x))
// BatchNorm: (x - mean) / sqrt(var + eps) * gamma + beta

__device__ __forceinline__ float compute_activation(float x) {
    // softplus(x) = log(1 + exp(x))
    // To avoid overflow, use: log(1 + exp(x)) = x + log(1 + exp(-x)) for x > 0
    float sp;
    if (x > 20.0f) {
        sp = x;
    } else if (x < -20.0f) {
        sp = expf(x);
    } else {
        sp = logf(1.0f + expf(x));
    }
    // tanh(softplus) * x
    return tanhf(sp) * x;
}

// Thread block processes a tile of output in one output channel
// Uses shared memory for input tile and weights
template <int BLOCK_SIZE, int TILE_H, int TILE_W, int KSIZE>
__global__ void fused_conv_activation_bn_tiled_kernel(
    const float* __restrict__ input,
    const float* __restrict__ weight,
    const float* __restrict__ bias,
    const float* __restrict__ bn_mean,
    const float* __restrict__ bn_var,
    const float* __restrict__ bn_gamma,
    const float* __restrict__ bn_beta,
    float* __restrict__ output,
    const int batch_size,
    const int in_channels,
    const int out_channels,
    const int height,
    const int width,
    const int out_height,
    const int out_width,
    const float eps) {
    
    const int n = blockIdx.z;
    const int oc = blockIdx.y;
    const int tile_id = blockIdx.x;
    
    const int tiles_per_row = (out_width + TILE_W - 1) / TILE_W;
    const int tile_oh = (tile_id / tiles_per_row) * TILE_H;
    const int tile_ow = (tile_id % tiles_per_row) * TILE_W;
    
    const int th = threadIdx.y;
    const int tw = threadIdx.x;
    
    const int oh = tile_oh + th;
    const int ow = tile_ow + tw;
    
    // Shared memory for weights of this output channel
    // Weight shape: [in_channels, KSIZE, KSIZE]
    __shared__ float weight_shared[3 * 3 * 3];  // 27 floats for 3x3 kernel with 3 input channels
    
    // Load weights
    const int tid = th * BLOCK_SIZE + tw;
    const int weight_size = in_channels * KSIZE * KSIZE;
    for (int i = tid; i < weight_size; i += BLOCK_SIZE * BLOCK_SIZE) {
        weight_shared[i] = weight[oc * weight_size + i];
    }
    
    // Load BN params and bias to registers
    __shared__ float bn_params[4];  // mean, var, gamma, beta
    __shared__ float bias_val;
    
    if (tid == 0) {
        bn_params[0] = bn_mean[oc];
        bn_params[1] = bn_var[oc];
        bn_params[2] = bn_gamma[oc];
        bn_params[3] = bn_beta[oc];
        bias_val = bias != nullptr ? bias[oc] : 0.0f;
    }
    
    __syncthreads();
    
    if (oh >= out_height || ow >= out_width) return;
    
    // Compute convolution
    float sum = bias_val;
    
    for (int ic = 0; ic < in_channels; ++ic) {
        for (int kh = 0; kh < KSIZE; ++kh) {
            for (int kw = 0; kw < KSIZE; ++kw) {
                const int ih = oh + kh;
                const int iw = ow + kw;
                
                const int input_idx = n * in_channels * height * width + 
                                     ic * height * width + 
                                     ih * width + iw;
                
                const int weight_idx = ic * KSIZE * KSIZE + kh * KSIZE + kw;
                
                sum += input[input_idx] * weight_shared[weight_idx];
            }
        }
    }
    
    // Apply activation
    float activated = compute_activation(sum);
    
    // Apply batch normalization
    float normalized = (activated - bn_params[0]) * rsqrtf(bn_params[1] + eps);
    float result = bn_params[2] * normalized + bn_params[3];
    
    output[n * out_channels * out_height * out_width + 
           oc * out_height * out_width + 
           oh * out_width + ow] = result;
}

    }

    __shared__ float bn_params[4];

    if (tid == 0) {

    if (oh >= out_height || ow >= out_width) return;

    float sum = bias_val;
    
    for (int ic = 0; ic < in_channels; ++ic) {
        for (int kh = 0; kh < KSIZE; ++kh) {
            for (int kw = 0; kw < KSIZE; ++kw) {
                const int ih = oh + kh;

                sum += __half2float(input[input_idx]) * weight_shared[weight_idx];
            }
        }
    }
    
    // Apply activation
    float activated = compute_activation(sum);
    
    // Apply batch normalization
    float normalized = (activated - bn_params[0]) * rsqrtf(bn_params[1] + eps);
    float result = bn_params[2] * normalized + bn_params[3];
    
    output[n * out_channels * out_height * out_width + 
           oc * out_height * out_width + 
           oh * out_width + ow] = __float2half(result);
}

torch::Tensor fused_conv_activation_bn(
    torch::Tensor input,
    torch::Tensor weight,
    torch::Tensor bias,
    torch::Tensor bn_mean,
    torch::Tensor bn_var,
    torch::Tensor bn_gamma,
    torch::Tensor bn_beta,
    float eps) {
    
    const int batch_size = input.size(0);
    const int in_channels = input.size(1);
    const int height = input.size(2);
    const int width = input.size(3);
    const int out_channels = weight.size(0);
    const int kernel_size = weight.size(2);
    
    const int out_height = height - kernel_size + 1;
    const int out_width = width - kernel_size + 1;
    
    auto output = torch::empty({batch_size, out_channels, out_height, out_width}, 
                               input.options());
    
    const int TILE_H = 16;
    const int TILE_W = 16;
    const int BLOCK_SIZE = 16;
    
    const int tiles_per_row = (out_width + TILE_W - 1) / TILE_W;
    const int tiles_per_col = (out_height + TILE_H - 1) / TILE_H;
    const int total_tiles = tiles_per_row * tiles_per_col;
    
    dim3 grid(total_tiles, out_channels, batch_size);
    dim3 block(BLOCK_SIZE, BLOCK_SIZE);
    
    if (input.scalar_type() == torch::kHalf) {
        fused_conv_activation_bn_tiled_kernel_half<BLOCK_SIZE, TILE_H, TILE_W, 3><<<grid, block>>>(
            input.data_ptr<at::Half>(),
            weight.data_ptr<at::Half>(),
            bias.defined() ? bias.data_ptr<at::Half>() : nullptr,
            bn_mean.data_ptr<float>(),
            bn_var.data_ptr<float>(),
            bn_gamma.data_ptr<float>(),
            bn_beta.data_ptr<float>(),
            output.data_ptr<at::Half>(),
            batch_size,
            in_channels,
            out_channels,
            height,
            width,
            out_height,
            out_width,
            eps);
    } else {
        fused_conv_activation_bn_tiled_kernel<BLOCK_SIZE, TILE_H, TILE_W, 3><<<grid, block>>>(
            input.data_ptr<float>(),
            weight.data_ptr<float>(),
            bias.defined() ? bias.data_ptr<float>() : nullptr,
            bn_mean.data_ptr<float>(),
            bn_var.data_ptr<float>(),
            bn_gamma.data_ptr<float>(),
            bn_beta.data_ptr<float>(),
            output.data_ptr<float>(),
            batch_size,
            in_channels,
            out_channels,
            height,
            width,
            out_height,
            out_width,
            eps);
    }
    
    return output;
}

ext = load_inline(
    name='fused_conv_bn_ext',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['fused_conv_activation_bn'],
    verbose=False,
    extra_cuda_cflags=['-arch=sm_86'],

class Model(nn.Module):
    """
    Fused Conv2d + Activation + BatchNorm model.
    Activation: tanh(softplus(x)) * x
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(Model, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)
        self.eps = eps

    def forward(self, x):
        return ext.fused_conv_activation_bn(
            x,
            self.conv.weight,
            self.conv.bias,
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            self.eps
        )

def get_inputs():
    batch_size = 16
