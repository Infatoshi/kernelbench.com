import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_src = r'''
#include <torch/extension.h>
#include <cuda_runtime.h>

__global__ void conv2d_kernel(
    const float* __restrict__ x,
    const float* __restrict__ w,
    float* __restrict__ y,
    int N, int C, int H, int W,
    int K, int R, int S,
    int H_out, int W_out
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * K * H_out * W_out;
    if (idx >= total) return;

    int w_out = idx % W_out;
    int h_out = (idx / W_out) % H_out;
    int k = (idx / (W_out * H_out)) % K;
    int n = idx / (W_out * H_out * K);

    float acc = 0.0f;

    for (int c = 0; c < C; ++c) {
        for (int r = 0; r < R; ++r) {
            for (int s = 0; s < S; ++s) {
                int h_in = h_out + r;
                int w_in = w_out + s;
                int x_idx = ((n*C + c)*H + h_in)*W + w_in;
                int w_idx = ((k*C + c)*R + r)*S + s;
                acc += x[x_idx] * w[w_idx];
            }
        }
    }

    int y_idx = ((n*K + k)*H_out + h_out)*W_out + w_out;
    y[y_idx] = acc;
}

torch::Tensor conv2d_forward(torch::Tensor x, torch::Tensor w) {
    int N = x.size(0);
    int C = x.size(1);
    int H = x.size(2);
    int W = x.size(3);

    int K = w.size(0);
    int R = w.size(2);
    int S = w.size(3);

    int H_out = H - R + 1;
    int W_out = W - S + 1;

    auto y = torch::empty({N, K, H_out, W_out}, x.options());

    int total = N * K * H_out * W_out;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;

    conv2d_kernel<<<blocks, threads>>>(
        x.data_ptr<float>(),
        w.data_ptr<float>(),
        y.data_ptr<float>(),
        N, C, H, W,
        K, R, S,
        H_out, W_out
    );

    return y;
}
'''

ext = load_inline(
    name='conv2d_ext',
    cpp_sources='torch::Tensor conv2d_forward(torch::Tensor x, torch::Tensor w);',
    cuda_sources=cuda_src,
    functions=['conv2d_forward'],
    verbose=False
)

class Model(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0, dilation=1, groups=1, bias=False):
        super().__init__()
        self.conv2d = nn.Conv2d(in_channels, out_channels, kernel_size, stride=1, padding=0, bias=False)

    def forward(self, x):
        return ext.conv2d_forward(x.contiguous(), self.conv2d.weight.contiguous())


def get_inputs():
    batch_size = 16
    in_channels = 3
    height = 256
    width = 256
    x = torch.randn(batch_size, in_channels, height, width, device='cuda', dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [3, 64, 3]
