
import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def matmul_dropout_mean_softmax_kernel(
    X_ptr, W_ptr, B_ptr, Out_ptr,
    BATCH, IN_FEAT, OUT_FEAT,
    dropout_p, scale, seed,
    stride_xm, stride_xk,
    stride_wk, stride_wn,
    stride_om, stride_on,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    pid = tl.program_id(0)
    col_idx = pid * BLOCK_SIZE_N
    offsets_n = col_idx + tl.arange(0, BLOCK_SIZE_N)
    
    acc_sum = tl.zeros((BLOCK_SIZE_N,), dtype=tl.float32)
    
    b_ptrs = B_ptr + offsets_n
    bias = tl.load(b_ptrs, mask=offsets_n < OUT_FEAT, other=0.0)

    for m in range(0, BATCH, BLOCK_SIZE_M):
        offsets_m = m + tl.arange(0, BLOCK_SIZE_M)
        acc = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
        
        for k in range(0, IN_FEAT, BLOCK_SIZE_K):
            offsets_k = k + tl.arange(0, BLOCK_SIZE_K)
            x_ptrs = X_ptr + (offsets_m[:, None] * stride_xm + offsets_k[None, :] * stride_xk)
            w_ptrs = W_ptr + (offsets_k[:, None] * stride_wk + offsets_n[None, :] * stride_wn)
            mask_x = (offsets_m[:, None] < BATCH) & (offsets_k[None, :] < IN_FEAT)
            mask_w = (offsets_k[:, None] < IN_FEAT) & (offsets_n[None, :] < OUT_FEAT)
            x = tl.load(x_ptrs, mask=mask_x, other=0.0)
            w = tl.load(w_ptrs, mask=mask_w, other=0.0)
            acc += tl.dot(x, w)
            
        acc += bias[None, :]
        
        if dropout_p > 0:
            random_offsets = offsets_m[:, None] * OUT_FEAT + offsets_n[None, :]
            rv = tl.rand(seed, random_offsets.to(tl.int32))
            keep_mask = rv >= dropout_p
            acc = tl.where(keep_mask, acc * scale, 0.0)

        row_mask = offsets_m < BATCH
        col_mask = offsets_n < OUT_FEAT
        masked_acc = tl.where(row_mask[:, None] & col_mask[None, :], acc, 0.0)
        acc_sum += tl.sum(masked_acc, axis=0)

    mean_val = acc_sum / BATCH
    out_ptrs = Out_ptr + offsets_n
    tl.store(out_ptrs, mean_val, mask=offsets_n < OUT_FEAT)

@triton.jit
def softmax_kernel(
    Out_ptr, 
    OUT_FEAT,
    BLOCK_SIZE: tl.constexpr
):
    offsets = tl.arange(0, BLOCK_SIZE)
    mask = offsets < OUT_FEAT
    
    val = tl.load(Out_ptr + offsets, mask=mask, other=-float('inf'))
    
    max_val = tl.max(val, axis=0)
    exp_val = tl.exp(val - max_val)
    sum_exp = tl.sum(exp_val, axis=0)
    
    out = exp_val / sum_exp
    tl.store(Out_ptr + offsets, out, mask=mask)

class Model(nn.Module):
    def __init__(self, in_features, out_features, dropout_p):
        super(Model, self).__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.dropout_p = dropout_p
        self.in_features = in_features
        self.out_features = out_features

    def forward(self, x):
        curr_p = self.dropout_p if self.training else 0.0
        batch_size = x.shape[0]
        device = x.device
        
        weight = self.matmul.weight.t().contiguous()
        bias = self.matmul.bias
        
        out = torch.empty((1, self.out_features), device=device, dtype=torch.float32)
        
        # Increased block sizes for better occupancy/performance
        BLOCK_SIZE_N = 128
        BLOCK_SIZE_M = 32
        BLOCK_SIZE_K = 32
        
        grid = (triton.cdiv(self.out_features, BLOCK_SIZE_N),)
        
        seed = torch.randint(0, 2**31 - 1, (1,)).item()
        scale = 1.0 / (1.0 - curr_p) if curr_p < 1.0 else 0.0
        
        matmul_dropout_mean_softmax_kernel[grid](
            x, weight, bias, out,
            batch_size, self.in_features, self.out_features,
            curr_p, scale, seed,
            x.stride(0), x.stride(1),
            weight.stride(0), weight.stride(1),
            out.stride(0), out.stride(1),
            BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K
        )
        
        BLOCK_SIZE_SOFTMAX = triton.next_power_of_2(self.out_features)
        softmax_kernel[(1,)](
            out, self.out_features, BLOCK_SIZE=BLOCK_SIZE_SOFTMAX
        )
        
        return out

def get_inputs():
    batch_size = 128
    in_features = 4096
    return [torch.randn(batch_size, in_features, device='cuda')]

def get_init_inputs():
    in_features = 4096
    out_features = 4096
    dropout_p = 0.2
    return [in_features, out_features, dropout_p]
