"""Triton softmax — fused for single-block rows, 2D-grid for multi-block.

fp32 accumulation throughout.  The online-softmax rescaling errors are
O(log n · eps), well below 1e-5 for all tested shapes.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl

BATCH = 8
VOCAB = 32768


# ---------------------------------------------------------------------------
# Single-block kernel — one load + one store when the full row fits in one
# block (vocab <= BLOCK_SIZE).  2x fewer loads than the online-softmax path.
# ---------------------------------------------------------------------------

@triton.jit
def _single_block_kernel(x_ptr, y_ptr, stride_x, stride_y,
                         n_cols,
                         BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    x_ptr += pid * stride_x
    y_ptr += pid * stride_y

    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(x_ptr + offs, mask=mask, other=-float('inf'), cache_modifier=".ca")

    m = tl.max(x)
    e = tl.exp(x - m)
    e = tl.where(mask, e, 0.0)
    d = tl.sum(e)
    y = e / d

    tl.store(y_ptr + offs, y, mask=mask, cache_modifier=".cs")


# ---------------------------------------------------------------------------
# Multi-block fused kernel — online softmax for rows that span many blocks
# (used when vocab > BLOCK_SIZE but we're not using the 2D path).
# ---------------------------------------------------------------------------

@triton.jit
def _fused_kernel(x_ptr, y_ptr, stride_x, stride_y,
                  n_cols,
                  BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)

    x_ptr += pid * stride_x
    y_ptr += pid * stride_y

    m = -float('inf')
    d = 0.0

    for start in range(0, n_cols, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < n_cols
        x = tl.load(x_ptr + offs, mask=mask, other=-float('inf'), cache_modifier=".ca")

        blk_max = tl.max(x)
        new_m = tl.maximum(m, blk_max)
        e = tl.exp(x - new_m)
        e = tl.where(mask, e, 0.0)
        d = d * tl.exp(m - new_m) + tl.sum(e)
        m = new_m

    for start in range(0, n_cols, BLOCK_SIZE):
        offs = start + tl.arange(0, BLOCK_SIZE)
        mask = offs < n_cols
        x = tl.load(x_ptr + offs, mask=mask, other=0.0, cache_modifier=".ca")
        y = tl.exp(x - m) / d
        tl.store(y_ptr + offs, y, mask=mask, cache_modifier=".cs")


# ---------------------------------------------------------------------------
# Two-kernel path — 2D grid for high parallelism on large vocabs
# ---------------------------------------------------------------------------

@triton.jit
def _reduce_kernel(x_ptr, block_max_ptr, block_sum_ptr,
                   stride_x,
                   n_cols,
                   BLOCK_SIZE: tl.constexpr):
    row_id = tl.program_id(0)
    col_id = tl.program_id(1)
    num_blocks = tl.cdiv(n_cols, BLOCK_SIZE)

    x_ptr += row_id * stride_x

    offs = col_id * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(x_ptr + offs, mask=mask, other=-float('inf'), cache_modifier=".ca")

    blk_max = tl.max(x)
    e = tl.exp(x - blk_max)
    e = tl.where(mask, e, 0.0)
    blk_sum = tl.sum(e)

    idx = row_id * num_blocks + col_id
    tl.store(block_max_ptr + idx, blk_max)
    tl.store(block_sum_ptr + idx, blk_sum)


@triton.jit
def _softmax_kernel(x_ptr, y_ptr, block_max_ptr, block_sum_ptr,
                    stride_x, stride_y,
                    n_cols,
                    BLOCK_SIZE: tl.constexpr,
                    num_blocks: tl.constexpr):
    row_id = tl.program_id(0)
    col_id = tl.program_id(1)

    x_ptr += row_id * stride_x
    y_ptr += row_id * stride_y

    base = row_id * num_blocks

    m = -float('inf')
    for k in range(num_blocks):
        bm = tl.load(block_max_ptr + base + k)
        m = tl.maximum(m, bm)

    s = 0.0
    for k in range(num_blocks):
        bm = tl.load(block_max_ptr + base + k)
        bs = tl.load(block_sum_ptr + base + k)
        s += bs * tl.exp(bm - m)

    offs = col_id * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(x_ptr + offs, mask=mask, other=0.0, cache_modifier=".ca")
    y = tl.exp(x - m) / s
    tl.store(y_ptr + offs, y, mask=mask, cache_modifier=".cs")


# ---------------------------------------------------------------------------
# Model  — dispatches to fused or two-kernel path based on vocab
# ---------------------------------------------------------------------------

class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = batch
        self.vocab = vocab

        # single-block threshold: rows below this use the fused single-kernel.
        # Multi-block path uses BLOCK_SIZE_BLK for its blocks.
        self._block_size_fused: int = 4096
        self._block_size_blk: int = 8192 if vocab >= 32768 else 4096

        self._num_blocks: int = (vocab + self._block_size_blk - 1) // self._block_size_blk
        self._num_warps_single: int = Model._pick_num_warps(vocab, 0)
        self._num_warps_fused: int = Model._pick_num_warps(vocab, 1)
        self._num_warps_blk: int = Model._pick_num_warps(vocab, 2)
        self._num_warps_reduce: int = Model._pick_num_warps(vocab, 3)

    @staticmethod
    def _pick_num_warps(vocab: int, which: int) -> int:
        """which=0 → single-block, 1 → fused, 2 → 2nd-pass, 3 → reduce."""
        if which == 0:        # single-block kernel (small vocabs)
            if vocab >= 32768:
                return 32
            return 4
        if which == 1:        # fused (multi-block, online softmax)
            if vocab >= 32768:
                return 32
            return 16
        if which == 2:        # second-pass (normalize) kernel
            if vocab >= 131072:
                return 32
            if vocab >= 32768:
                return 16
            return 8
        # reduce kernel
        if vocab >= 131072:
            return 8
        return 4

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        y = torch.empty_like(x)
        device = x.device
        nb = self._num_blocks

        # ---- single-block path (full row fits in one block, 1 load + 1 store)
        if nb == 1:
            _single_block_kernel[(self.batch,)](
                x, y, x.stride(0), y.stride(0), self.vocab,
                BLOCK_SIZE=self._block_size_fused,
                num_warps=self._num_warps_single,
            )
            return y

        # ---- multi-block 2D path ------------------------------------------
        buf = torch.empty(2, self.batch, nb, dtype=torch.float32, device=device)
        grid = (self.batch, nb)

        _reduce_kernel[grid](
            x, buf[0], buf[1], x.stride(0), self.vocab,
            BLOCK_SIZE=self._block_size_blk,
            num_warps=self._num_warps_reduce,
        )

        _softmax_kernel[grid](
            x, y, buf[0], buf[1],
            x.stride(0), y.stride(0), self.vocab,
            BLOCK_SIZE=self._block_size_blk,
            num_blocks=nb,
            num_warps=self._num_warps_blk,
        )

        return y
