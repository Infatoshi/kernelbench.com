"""Kimi Delta Attention (KDA) forward, chunk form -- custom Triton kernels.

Decomposition (implemented from scratch, no FLA imports):
  1. intra kernel  (parallel over B*H*NT): per chunk do the in-chunk cumsum of the
     log-decay g, build the decayed K-K matrix, invert (I - M) by a log-depth
     matmul doubling (M is strictly-lower / nilpotent), form A = (I-M)^{-1} * beta,
     then  w = A @ (exp(g)*k),  u = A @ v,  the intra q-k decay matrix Aqk, and
     pre-scale the elementwise quantities the serial scan needs:
       qg    = scale*q*exp(g)            (for q@S in the output)
       kdec  = k*exp(g[-1]-g)            (for the state update)
       gamma = exp(g[-1])               (per-chunk state decay)
  2. scan kernel (parallel over B*H*NV, sequential over chunks): keeps the
     inter-chunk state S, computes v_new = u - w@S, o = qg@S + Aqk@v_new, and
     updates S = gamma*S + kdec^T@v_new.  All elementwise work is already done, so
     the serial loop is purely loads + tensor-core matmuls.  V is tiled so each
     program is self-contained (no inter-chunk state stored to HBM).
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _intra_kernel(
    q, k, v, g, beta,
    w_ptr, u_ptr, Aqk_ptr, qg_ptr, kdec_ptr, gamma_ptr,
    scale,
    T,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    LOG_BT: tl.constexpr,
    NT: tl.constexpr,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    base_k = (i_b * T * H + i_h) * K
    base_v = (i_b * T * H + i_h) * V
    q += base_k
    k += base_k
    g += base_k
    v += base_v
    w_ptr += base_k
    u_ptr += base_v
    qg_ptr += base_k
    kdec_ptr += base_k
    Aqk_ptr += (i_b * T * H + i_h) * BT
    gamma_ptr += ((i_b * NT + i_t) * H + i_h) * K
    beta += i_b * T * H + i_h

    p_q = tl.make_block_ptr(q, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_k = tl.make_block_ptr(k, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_g = tl.make_block_ptr(g, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_v = tl.make_block_ptr(v, (T, V), (H * V, 1), (i_t * BT, 0), (BT, V), (1, 0))
    p_beta = tl.make_block_ptr(beta, (T,), (H,), (i_t * BT,), (BT,), (0,))

    b_q = tl.load(p_q, boundary_check=(0, 1)).to(tl.float32) * scale
    b_k = tl.load(p_k, boundary_check=(0, 1)).to(tl.float32)
    b_g_raw = tl.load(p_g, boundary_check=(0, 1)).to(tl.float32)
    b_v = tl.load(p_v, boundary_check=(0, 1))
    b_beta = tl.load(p_beta, boundary_check=(0,)).to(tl.float32)

    # cumulative decay within chunk
    b_g = tl.cumsum(b_g_raw, axis=0)            # [BT, K]
    oi = tl.arange(0, BT)
    gr = tl.sum(tl.where(oi[:, None] == (BT // 2), b_g, 0.0), 0)   # midpoint ref [K]
    gn = tl.sum(tl.where(oi[:, None] == (BT - 1), b_g, 0.0), 0)    # last row    [K]
    b_adj = b_g - gr[None, :]
    b_eg_a = tl.exp(b_adj)        # exp(g - gr)
    b_eng_a = tl.exp(-b_adj)      # exp(gr - g)

    b_kc_a = b_k * b_eg_a
    b_kn_a = b_k * b_eng_a
    b_qc_a = b_q * b_eg_a

    b_eg = tl.exp(b_g)            # real exp(g) (<=1, safe)
    b_kc = (b_k * b_eg).to(v.dtype.element_ty)

    # decayed K-K and Q-K matrices
    b_Akk = tl.dot(b_kc_a, tl.trans(b_kn_a))     # exp(g[c]-g[i]) k k
    b_Aqk = tl.dot(b_qc_a, tl.trans(b_kn_a))

    m_strict = oi[:, None] > oi[None, :]
    m_incl = oi[:, None] >= oi[None, :]
    b_Akk = tl.where(m_strict, b_Akk, 0.0)
    b_Aqk = tl.where(m_incl, b_Aqk, 0.0)

    # M = -(beta[c] * Akk), strict lower (nilpotent, M^BT = 0)
    b_M = -(b_Akk * b_beta[:, None])

    # (I - M)^{-1} = prod_{i=0..log2(BT)-1} (I + M^{2^i})
    b_eye = (oi[:, None] == oi[None, :]).to(tl.float32)
    b_Tinv = b_eye + b_M
    b_Mp = b_M
    for _ in tl.static_range(1, LOG_BT):
        b_Mp = tl.dot(b_Mp, b_Mp, input_precision="tf32x3")
        b_Tinv = b_Tinv + tl.dot(b_Tinv, b_Mp, input_precision="tf32x3")

    b_A = (b_Tinv * b_beta[None, :]).to(v.dtype.element_ty)

    b_w = tl.dot(b_A, b_kc)                       # [BT, K]
    b_u = tl.dot(b_A, b_v)                        # [BT, V]

    # elementwise quantities the serial scan needs (precomputed here, in parallel)
    b_qg = (b_q * b_eg).to(v.dtype.element_ty)            # scale*q*exp(g)
    b_kdec = (b_k * tl.exp(gn[None, :] - b_g)).to(v.dtype.element_ty)  # k*exp(g[-1]-g)

    p_w = tl.make_block_ptr(w_ptr, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_u = tl.make_block_ptr(u_ptr, (T, V), (H * V, 1), (i_t * BT, 0), (BT, V), (1, 0))
    p_qg = tl.make_block_ptr(qg_ptr, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_kdec = tl.make_block_ptr(kdec_ptr, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_Aqk = tl.make_block_ptr(Aqk_ptr, (T, BT), (H * BT, 1), (i_t * BT, 0), (BT, BT), (1, 0))
    tl.store(p_w, b_w.to(w_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_u, b_u.to(u_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_qg, b_qg.to(qg_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_kdec, b_kdec.to(kdec_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_Aqk, b_Aqk.to(Aqk_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(gamma_ptr + tl.arange(0, K), tl.exp(gn))


@triton.jit
def _scan_kernel(
    qg_ptr, kdec_ptr, gamma_ptr, w_ptr, u_ptr, Aqk_ptr, o_ptr,
    T,
    NT,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
):
    i_v = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    base_k = (i_b * T * H + i_h) * K
    base_v = (i_b * T * H + i_h) * V
    qg_ptr += base_k
    kdec_ptr += base_k
    w_ptr += base_k
    u_ptr += base_v
    Aqk_ptr += (i_b * T * H + i_h) * BT
    o_ptr += base_v
    gamma_ptr += (i_b * NT * H + i_h) * K

    b_S = tl.zeros([K, BV], dtype=tl.float32)
    ok = tl.arange(0, K)

    for i_t in range(0, NT):
        row0 = i_t * BT
        p_w = tl.make_block_ptr(w_ptr, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
        p_qg = tl.make_block_ptr(qg_ptr, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
        p_kdec = tl.make_block_ptr(kdec_ptr, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
        p_u = tl.make_block_ptr(u_ptr, (T, V), (H * V, 1), (row0, i_v * BV), (BT, BV), (1, 0))
        p_Aqk = tl.make_block_ptr(Aqk_ptr, (T, BT), (H * BT, 1), (row0, 0), (BT, BT), (1, 0))
        p_o = tl.make_block_ptr(o_ptr, (T, V), (H * V, 1), (row0, i_v * BV), (BT, BV), (1, 0))

        b_w = tl.load(p_w, boundary_check=(0, 1))
        b_qg = tl.load(p_qg, boundary_check=(0, 1))
        b_kdec = tl.load(p_kdec, boundary_check=(0, 1))
        b_u = tl.load(p_u, boundary_check=(0, 1)).to(tl.float32)
        b_Aqk = tl.load(p_Aqk, boundary_check=(0, 1))
        b_gamma = tl.load(gamma_ptr + i_t * H * K + ok)

        b_S_bf = b_S.to(w_ptr.dtype.element_ty)
        b_vnew = b_u - tl.dot(b_w, b_S_bf)        # [BT, BV]
        b_vnew_bf = b_vnew.to(w_ptr.dtype.element_ty)

        b_o = tl.dot(b_qg, b_S_bf) + tl.dot(b_Aqk, b_vnew_bf)
        tl.store(p_o, b_o.to(o_ptr.dtype.element_ty), boundary_check=(0, 1))

        b_S = b_S * b_gamma[:, None]
        b_S = b_S + tl.dot(tl.trans(b_kdec), b_vnew_bf)


class Model(nn.Module):
    def __init__(self, B, T, H, K, V, chunk_size=64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        B, T, H, K = q.shape
        V = v.shape[-1]
        BT = self.chunk_size
        NT = T // BT
        scale = self.scale
        dtype = v.dtype

        q = q.contiguous(); k = k.contiguous(); v = v.contiguous()
        g = g.contiguous(); beta = beta.contiguous()

        w = torch.empty(B, T, H, K, device=q.device, dtype=dtype)
        u = torch.empty(B, T, H, V, device=q.device, dtype=dtype)
        qg = torch.empty(B, T, H, K, device=q.device, dtype=dtype)
        kdec = torch.empty(B, T, H, K, device=q.device, dtype=dtype)
        Aqk = torch.empty(B, T, H, BT, device=q.device, dtype=dtype)
        gamma = torch.empty(B, NT, H, K, device=q.device, dtype=torch.float32)
        o = torch.empty(B, T, H, V, device=q.device, dtype=dtype)

        log_bt = int(BT).bit_length() - 1
        _intra_kernel[(NT, B * H)](
            q, k, v, g, beta, w, u, Aqk, qg, kdec, gamma, scale, T,
            H=H, K=K, V=V, BT=BT, LOG_BT=log_bt, NT=NT, num_warps=4,
        )

        BV = 32
        NV = V // BV
        _scan_kernel[(NV, B * H)](
            qg, kdec, gamma, w, u, Aqk, o, T, NT,
            H=H, K=K, V=V, BT=BT, BV=BV, num_warps=4, num_stages=3,
        )
        return o


# Module-level shape shims (overridden by harness).
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
