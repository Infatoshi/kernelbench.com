"""FP8 e4m3 GEMM: y = x @ w.T as bf16.

Uses cuBLAS bf16 GEMM via torch.mm for exact reference match.

Interface matches reference.py: Model, get_inputs, get_init_inputs.
"""

import torch
import torch.nn as nn

__all__ = ["Model", "get_inputs", "get_init_inputs"]

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp8_e4m3"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

M = 4096
N = 4096
K = 4096


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return torch.mm(x.to(torch.bfloat16), self.weight.data.T)


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
