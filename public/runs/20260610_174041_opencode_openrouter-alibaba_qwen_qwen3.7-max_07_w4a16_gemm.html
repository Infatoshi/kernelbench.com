@triton.jit
def w4a16_gemm_kernel(
    x_ptr, w_q_ptr, scales_ptr, zeros_ptr, out_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wqk, stride_wqn,
    stride_sk, stride_sn,
    stride_zk, stride_zn,
    stride_om, stride_on,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)

    offs_k_full = tl.arange(0, BLOCK_K)
    offs_k_half = offs_k_full // 2
    is_odd = (offs_k_full % 2) == 1

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k_group_start in range(0, K, GROUP_SIZE):
        group_id = k_group_start // GROUP_SIZE
        scale_row = tl.load(scales_ptr + group_id * stride_sk + offs_n * stride_sn, mask=(offs_n < N), other=1.0).to(tl.float32)
        zero_row = tl.load(zeros_ptr + group_id * stride_zk + offs_n * stride_zn, mask=(offs_n < N), other=0.0).to(tl.float32)

        k_base = k_group_start
        offs_kh = k_base // 2 + offs_k_half

        w_packed = tl.load(w_q_ptr + offs_kh[:, None] * stride_wqk + offs_n[None, :] * stride_wqn, mask=(offs_kh[:, None] < K // 2) & (offs_n[None, :] < N), other=0)

        w_int = tl.where(is_odd[:, None], (w_packed >> 4) & 0xF, w_packed & 0xF)
        w_f = w_int.to(tl.float32)
        w_dequant = (w_f - zero_row[None, :]) * scale_row[None, :]
        w_dequant_bf = w_dequant.to(tl.bfloat16)

        offs_k_full_global = k_base + offs_k_full
        x_block = tl.load(x_ptr + offs_m[:, None] * stride_xm + offs_k_full_global[None, :] * stride_xk, mask=(offs_m[:, None] < M) & (offs_k_full_global[None, :] < K), other=0.0).to(tl.bfloat16)

        acc += tl.dot(x_block, w_dequant_bf)

    out_ptrs = out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on
    out_mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(out_ptrs, acc.to(tl.bfloat16), mask=out_mask)
