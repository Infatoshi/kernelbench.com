"""Custom top-k kernel: bitonic sort network in shared memory.

Strategy:
  - For k=1: specialized argmax reduction kernel.
  - For k>1, n<=TILE_SIZE: single-block bitonic sort, extract top-k.
  - For k>1, n>TILE_SIZE: two-pass — chunked kernel finds per-tile top-k,
    merge kernel combines them via bitonic sort.
  - TILE_SIZE=2048 elements per tile, 1024 threads per block (each thread
    handles 2 elements during the sort).
"""

import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cuda_src = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <float.h>
#include <stdint.h>

// Sort a shared-memory buffer of size N (must be power of 2) using bitonic sort.
// NTHREADS threads cooperate. Each thread owns N/NTHREADS elements initially.
// We use a register-based approach: each thread holds multiple elements.
// For simplicity, we use a sequential bitonic sort approach where all threads
// participate in compare-swap phases via shared memory.

// Actually, let's use a simpler approach: load into smem, then do
// bitonic network with all NTHREADS threads, where the network has
// log2(N) stages and each stage does N/2 compare-swaps distributed
// across NTHREADS threads (each thread does ceil(N/(2*NTHREADS)) swaps).

__device__ __forceinline__ void swap_f(float& a, float& b) { float t=a; a=b; b=t; }
__device__ __forceinline__ void swap_i(int64_t& a, int64_t& b) { int64_t t=a; a=b; b=t; }

__global__
void topk_chunked_kernel(
    const float* __restrict__ input,
    float* __restrict__ intermediate_v,
    int64_t* __restrict__ intermediate_i,
    int n, int k, int num_chunks, int TILE_SZ)
{
    extern __shared__ char smem_raw[];
    float* s_val = (float*)smem_raw;
    int64_t* s_idx = (int64_t*)(s_val + TILE_SZ);

    int chunk_id = blockIdx.x;
    int row      = chunk_id / num_chunks;
    int chunk    = chunk_id % num_chunks;
    int tid      = threadIdx.x;
    int nthreads = blockDim.x;

    int offset = row * n + chunk * TILE_SZ;
    int valid  = n - chunk * TILE_SZ;
    if (valid > TILE_SZ) valid = TILE_SZ;

    // Each thread loads multiple elements (coalesced)
    for (int i = tid; i < TILE_SZ; i += nthreads) {
        if (i < valid) {
            s_val[i] = __ldg(&input[offset + i]);
            s_idx[i] = (int64_t)(chunk * TILE_SZ + i);
        } else {
            s_val[i] = -FLT_MAX;
            s_idx[i] = (int64_t)(chunk * TILE_SZ + i);
        }
    }
    __syncthreads();

    // Bitonic sort in shared memory
    for (int step = 1; step <= TILE_SZ; step <<= 1) {
        for (int sub = step >> 1; sub > 0; sub >>= 1) {
            // Each thread handles multiple compare-swaps
            for (int i = tid; i < (TILE_SZ >> 1); i += nthreads) {
                // Map work index to the two elements to compare
                int block_id = i / sub;
                int block_start = block_id * 2 * sub;
                int offset_in_block = i % sub;
                int a = block_start + offset_in_block;
                int b = block_start + offset_in_block + sub;
                if (b < TILE_SZ) {
                    int dir = (a & step) ? 1 : 0;
                    float va = s_val[a], vb = s_val[b];
                    int64_t ia = s_idx[a], ib = s_idx[b];
                    int swap = (dir == 0) ? (va > vb) : (va < vb);
                    if (swap) {
                        s_val[a] = vb; s_val[b] = va;
                        s_idx[a] = ib; s_idx[b] = ia;
                    }
                }
            }
            __syncthreads();
        }
    }

    // Extract top-k (last k elements, reversed for descending)
    int out_base = row * num_chunks * k + chunk * k;
    for (int i = tid; i < k; i += nthreads) {
        int src = TILE_SZ - 1 - i;
        intermediate_v[out_base + i] = s_val[src];
        intermediate_i[out_base + i] = s_idx[src];
    }
}

__global__
void topk_merge_kernel(
    const float* __restrict__ intermediate_v,
    const int64_t* __restrict__ intermediate_i,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int k, int num_chunks)
{
    extern __shared__ char smem_raw[];
    int total = num_chunks * k;
    // Pad total to next power of 2 for bitonic sort
    int sort_size = 1;
    while (sort_size < total) sort_size <<= 1;

    float* s_val = (float*)smem_raw;
    int64_t* s_idx = (int64_t*)(s_val + sort_size);
    int row = blockIdx.x;
    int tid = threadIdx.x;
    int nthreads = blockDim.x;

    int in_base = row * total;
    // Load actual data
    for (int i = tid; i < total; i += nthreads) {
        s_val[i] = __ldg(&intermediate_v[in_base + i]);
        s_idx[i] = __ldg(&intermediate_i[in_base + i]);
    }
    // Pad with -inf
    for (int i = total + tid; i < sort_size; i += nthreads) {
        s_val[i] = -FLT_MAX;
        s_idx[i] = (int64_t)0;
    }
    __syncthreads();

    // Bitonic sort
    for (int step = 1; step <= sort_size; step <<= 1) {
        for (int sub = step >> 1; sub > 0; sub >>= 1) {
            for (int i = tid; i < (sort_size >> 1); i += nthreads) {
                int block_id = i / sub;
                int block_start = block_id * 2 * sub;
                int offset_in_block = i % sub;
                int a = block_start + offset_in_block;
                int b = block_start + offset_in_block + sub;
                if (b < sort_size) {
                    int dir = (a & step) ? 1 : 0;
                    float va = s_val[a], vb = s_val[b];
                    int64_t ia = s_idx[a], ib = s_idx[b];
                    int swap = (dir == 0) ? (va > vb) : (va < vb);
                    if (swap) {
                        s_val[a] = vb; s_val[b] = va;
                        s_idx[a] = ib; s_idx[b] = ia;
                    }
                }
            }
            __syncthreads();
        }
    }

    // Write top-k (descending)
    int out_base = row * k;
    for (int i = tid; i < k; i += nthreads) {
        int src = sort_size - 1 - i;
        out_values[out_base + i]  = s_val[src];
        out_indices[out_base + i] = s_idx[src];
    }
}

__global__
void topk_argmax_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int n)
{
    extern __shared__ char smem_raw[];
    float* s_val = (float*)smem_raw;
    int64_t* s_idx = (int64_t*)(s_val + blockDim.x);

    int row = blockIdx.x;
    int tid = threadIdx.x;
    const float* row_ptr = input + row * n;

    float best_val = -FLT_MAX;
    int64_t best_idx = 0;
    for (int i = tid; i < n; i += blockDim.x) {
        float v = __ldg(&row_ptr[i]);
        if (v > best_val) { best_val = v; best_idx = (int64_t)i; }
    }
    s_val[tid] = best_val;
    s_idx[tid] = best_idx;
    __syncthreads();

    for (int half = blockDim.x >> 1; half > 0; half >>= 1) {
        if (tid < half) {
            if (s_val[tid + half] > s_val[tid]) {
                s_val[tid] = s_val[tid + half];
                s_idx[tid] = s_idx[tid + half];
            }
        }
        __syncthreads();
    }

    if (tid == 0) {
        out_values[row]  = s_val[0];
        out_indices[row] = s_idx[0];
    }
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("run_argmax", [](torch::Tensor input, int n) {
        int batch = input.size(0);
        auto out_v = torch::empty({batch, 1}, input.options());
        auto out_i = torch::empty({batch, 1}, input.options().dtype(torch::kInt64));
        int block = 1024;
        int smem = block * (4 + 8);
        topk_argmax_kernel<<<batch, block, smem>>>(
            input.data_ptr<float>(), out_v.data_ptr<float>(),
            out_i.data_ptr<int64_t>(), n);
        return std::vector<torch::Tensor>{out_v, out_i};
    }, "argmax top-1");

    m.def("run_chunked", [](torch::Tensor input, torch::Tensor inter_v,
                             torch::Tensor inter_i, int n, int k,
                             int num_chunks, int tile_size) {
        int batch = input.size(0);
        int smem = tile_size * (4 + 8);
        int threads = 1024;
        topk_chunked_kernel<<<batch * num_chunks, threads, smem>>>(
            input.data_ptr<float>(), inter_v.data_ptr<float>(),
            inter_i.data_ptr<int64_t>(), n, k, num_chunks, tile_size);
    }, "chunked top-k");

    m.def("run_merge", [](torch::Tensor inter_v, torch::Tensor inter_i,
                           torch::Tensor out_v, torch::Tensor out_i,
                           int k, int num_chunks) {
        int batch = inter_v.size(0);

            inter_v.data_ptr<float>(), inter_i.data_ptr<int64_t>(),
            out_v.data_ptr<float>(), out_i.data_ptr<int64_t>(),
            k, num_chunks);

}

_topk_module = None

            cpp_sources="",
            cuda_sources=_cuda_src,

class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        batch, n = x.shape
        k = self.k
        mod = _get_module()
        x = x.contiguous()

        if k == 1:
            result = mod.run_argmax(x, n)
            return result[0], result[1]

        num_chunks = (n + TILE_SIZE - 1) // TILE_SIZE

        if num_chunks == 1:
            inter_v = torch.empty(batch, k, dtype=torch.float32, device=x.device)
            inter_i = torch.empty(batch, k, dtype=torch.int64, device=x.device)
            mod.run_chunked(x, inter_v, inter_i, n, k, 1, TILE_SIZE)
            return inter_v, inter_i

        inter_v = torch.empty(batch, num_chunks, k, dtype=torch.float32, device=x.device)
        inter_i = torch.empty(batch, num_chunks, k, dtype=torch.int64, device=x.device)
        mod.run_chunked(x, inter_v, inter_i, n, k, num_chunks, TILE_SIZE)

        out_v = torch.empty(batch, k, dtype=torch.float32, device=x.device)
        out_i = torch.empty(batch, k, dtype=torch.int64, device=x.device)
        mod.run_merge(inter_v, inter_i, out_v, out_i, k, num_chunks)
        return out_v, out_i
