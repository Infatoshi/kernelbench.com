"""Custom top-k kernel using hierarchical per-chunk register top-k + warp/block merge via iterative reductions.

Implements the required Model interface. Uses CUDA C++ load_inline with a two-phase
approach (local top-k per block chunk, then merge of partials) to achieve high DRAM
bandwidth utilization while supporting all shapes including non-power-of-2 n and k=1.
"""
import os
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


# --------------------------------------------------------------------------------------
# CUDA kernel: register-thread local top-k (small chunks) + shared reduction merge
# for block-level then row-level top-k. Single-pass vectorized read of input.
# --------------------------------------------------------------------------------------

CUDA_SRC = r'''
#include <cuda_runtime.h>
#include <cfloat>
#include <cstdint>

__device__ __forceinline__ float neg_inf() {
  return -1.0e30f;
}

__device__ __forceinline__ void do_insert(float v, int64_t gi, float* myv, int64_t* myi, int& mycnt, int kk) {
  if (kk <= 0) return;
  if (mycnt < kk) {
    int pos = mycnt;
    while (pos > 0 && v > myv[pos - 1]) {
      myv[pos] = myv[pos - 1];
      myi[pos] = myi[pos - 1];
      pos--;
    }
    myv[pos] = v;
    myi[pos] = gi;
    mycnt++;
  } else if (v > myv[kk - 1]) {
    int pos = kk - 1;
    while (pos > 0 && v > myv[pos - 1]) {
      myv[pos] = myv[pos - 1];
      myi[pos] = myi[pos - 1];
      pos--;
    }
    myv[pos] = v;
    myi[pos] = gi;
  }
}

__global__ __launch_bounds__(256)
void topk_phase1(const float* __restrict__ inp,
                 float* __restrict__ partial_v,
                 int64_t* __restrict__ partial_i,
                 int batch, int n, int k, int bpr) {
  extern __shared__ char smem[];
  float*   red_val = (float*)smem;
  int*     red_tid = (int*)(red_val + 256);
  int*     red_lj  = red_tid + 256;
  int64_t* red_gi  = (int64_t*)(red_lj + 256);
  float*   btopv   = (float*)(red_gi + 256);
  int64_t* btopi   = (int64_t*)(btopv + 64);

  int row = blockIdx.x / bpr;
  int br  = blockIdx.x % bpr;
  if (row >= batch) return;

  size_t row_base = (size_t)row * n;
  int epb = (n + bpr - 1) / bpr;
  epb = ((epb + 3) / 4) * 4;  // align chunk size to 4 for float4
  int c0 = br * epb;
  int c1 = (c0 + epb < n ? c0 + epb : n);
  if (c0 >= n) {
    // still need to participate in writes; fall through with empty
  }

  float   myv[64];
  int64_t myi[64];
  int mycnt = 0;
  int curj  = 0;

  // Vectorized loads when chunk aligned
  if (c0 < c1) {
    size_t chunk_base = row_base + c0;
    const float4* ptr4 = reinterpret_cast<const float4*>(inp + chunk_base);
    int num4 = (c1 - c0) / 4;
    for (int vi = threadIdx.x; vi < num4; vi += blockDim.x) {
      float4 v4 = ptr4[vi];
      size_t gbase = (size_t)c0 + vi * 4ULL;
      do_insert(v4.x, (int64_t)(gbase + 0), myv, myi, mycnt, k);
      do_insert(v4.y, (int64_t)(gbase + 1), myv, myi, mycnt, k);
      do_insert(v4.z, (int64_t)(gbase + 2), myv, myi, mycnt, k);
      do_insert(v4.w, (int64_t)(gbase + 3), myv, myi, mycnt, k);
    }
    // tail (0-3 elems)
    int tail0 = c0 + num4 * 4;
    for (int p = tail0 + threadIdx.x; p < c1; p += blockDim.x) {
      float v = inp[row_base + (size_t)p];
      do_insert(v, (int64_t)p, myv, myi, mycnt, k);
    }
  }

  // Iterative k-way merge using block reduction (k times)
  for (int kk = 0; kk < k; kk++) {
    float cand = (curj < mycnt) ? myv[curj] : neg_inf();
    int64_t gi = (curj < mycnt) ? myi[curj] : -1LL;
    int lj = curj;
    red_val[threadIdx.x] = cand;
    red_tid[threadIdx.x] = threadIdx.x;
    red_lj[threadIdx.x] = lj;
    red_gi[threadIdx.x] = gi;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s /= 2) {
      if (threadIdx.x < s) {
        int o = threadIdx.x + s;
        if (red_val[o] > red_val[threadIdx.x]) {
          red_val[threadIdx.x] = red_val[o];
          red_tid[threadIdx.x] = red_tid[o];
          red_lj[threadIdx.x] = red_lj[o];
          red_gi[threadIdx.x] = red_gi[o];
        }
      }
      __syncthreads();
    }

    if (threadIdx.x == 0) {
      btopv[kk] = red_val[0];
      btopi[kk] = red_gi[0];
    }
    __syncthreads();

    int wt = red_tid[0];
    if (threadIdx.x == wt) {
      curj++;
    }
  }

  if (threadIdx.x == 0) {
    int pbase = (row * bpr + br) * k;
    for (int j = 0; j < k; j++) {
      partial_v[pbase + j] = btopv[j];
      partial_i[pbase + j] = btopi[j];
    }
  }
}

__global__ __launch_bounds__(256)
void topk_phase2(const float* __restrict__ p_v,
                 const int64_t* __restrict__ p_i,
                 float* __restrict__ out_v,
                 int64_t* __restrict__ out_i,
                 int batch, int n, int k, int bpr) {
  extern __shared__ char smem[];
  float*   red_val = (float*)smem;
  int*     red_tid = (int*)(red_val + 256);
  int*     red_lj  = red_tid + 256;
  int64_t* red_gi  = (int64_t*)(red_lj + 256);
  float*   btopv   = (float*)(red_gi + 256);
  int64_t* btopi   = (int64_t*)(btopv + 64);
  int*     curj    = (int*)(btopi + 64);

  int row = blockIdx.x;
  if (row >= batch) return;

  if (threadIdx.x < 256) {
    curj[threadIdx.x] = 0;
  }
  __syncthreads();

  for (int kk = 0; kk < k; kk++) {
    float cand = neg_inf();
    int64_t gi = -1LL;
    if (threadIdx.x < bpr) {
      int t = threadIdx.x;
      int j = curj[t];
      if (j < k) {
        int pbase = row * bpr * k + t * k + j;
        cand = p_v[pbase];
        gi = p_i[pbase];
      }
    }
    red_val[threadIdx.x] = cand;
    red_tid[threadIdx.x] = threadIdx.x;
    red_gi[threadIdx.x] = gi;
    __syncthreads();

    for (int s = blockDim.x / 2; s > 0; s /= 2) {
      if (threadIdx.x < s) {
        int o = threadIdx.x + s;
        if (red_val[o] > red_val[threadIdx.x]) {
          red_val[threadIdx.x] = red_val[o];
          red_tid[threadIdx.x] = red_tid[o];
          red_gi[threadIdx.x] = red_gi[o];
        }
      }
      __syncthreads();
    }

    if (threadIdx.x == 0) {
      btopv[kk] = red_val[0];
      btopi[kk] = red_gi[0];
    }
    __syncthreads();

    int wt = red_tid[0];
    if (threadIdx.x == wt && wt < bpr) {
      curj[wt]++;
    }
  }

  if (threadIdx.x == 0) {
    for (int j = 0; j < k; j++) {
      out_v[row * k + j] = btopv[j];
      out_i[row * k + j] = btopi[j];
    }
  }
}

std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int k,
                                     torch::Tensor partial_v,
                                     torch::Tensor partial_i,
                                     int bpr) {
  TORCH_CHECK(x.is_cuda(), "x must be CUDA");
  TORCH_CHECK(x.dim() == 2, "x must be 2D");
  TORCH_CHECK(k > 0 && k <= 64, "k must be in (0,64]");
  int batch = x.size(0);
  int n = x.size(1);
  TORCH_CHECK(partial_v.numel() >= batch * bpr * k, "partial_v too small");
  TORCH_CHECK(partial_i.numel() >= batch * bpr * k, "partial_i too small");

  auto out_v = torch::empty({batch, k}, x.options().dtype(torch::kFloat32));
  auto out_i = torch::empty({batch, k}, x.options().dtype(torch::kInt64));

  int threads = 256;
  dim3 grid1(batch * bpr);
  dim3 block(threads);

  // smem layout: 256*(float+3*int+int64) + 64*(float+int64) + 256*int
  size_t smem = 256ULL * (sizeof(float) + 3 * sizeof(int) + sizeof(int64_t)) +
                64ULL * (sizeof(float) + sizeof(int64_t)) +
                256ULL * sizeof(int);

  topk_phase1<<<grid1, block, smem, 0>>>(
      x.data_ptr<float>(),
      partial_v.data_ptr<float>(),
      partial_i.data_ptr<int64_t>(),
      batch, n, k, bpr);

  dim3 grid2(batch);
  topk_phase2<<<grid2, block, smem, 0>>>(
      partial_v.data_ptr<float>(),
      partial_i.data_ptr<int64_t>(),
      out_v.data_ptr<float>(),
      out_i.data_ptr<int64_t>(),
      batch, n, k, bpr);

  cudaDeviceSynchronize();
  return {out_v, out_i};
}
'''

CPP_SRC = r'''
#include <torch/extension.h>
#include <vector>

std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int k,
                                     torch::Tensor partial_v,
                                     torch::Tensor partial_i,
                                     int bpr);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("topk_cuda", &topk_cuda, "Custom top-k (bitonic-style hierarchical)");
}
'''

# Load the extension (done once at import)
_ext = None

def _get_ext():
  global _ext
  if _ext is None:
    _ext = load_inline(
        name="topk_bitonic_custom",
        cpp_sources=CPP_SRC,
        cuda_sources=CUDA_SRC,
        functions=["topk_cuda"],
        extra_cuda_cflags=["-O3", "--use_fast_math", "-lineinfo"],
        verbose=False,
    )
  return _ext


class Model(nn.Module):
  """Top-k over the last dim using custom CUDA kernel (register + reduction merge)."""

  def __init__(self, batch: int, n: int, k: int):
    super().__init__()
    self.batch = int(batch)
    self.n = int(n)
    self.k = int(k)
    # blocks per row chosen for good DRAM parallelism on large-n and balanced work
    self.bpr = max(1, min(256, (n + 127) // 128))
    max_part = self.batch * self.bpr * self.k
    # Persistent buffers for partial results (reused across forward calls)
    self.register_buffer("_partial_v", torch.empty(max_part, dtype=torch.float32, device="cuda"))
    self.register_buffer("_partial_i", torch.empty(max_part, dtype=torch.int64, device="cuda"))
    self.register_buffer("_dummy", torch.zeros(1))  # match reference state_dict shape
    self._ext = _get_ext()

  def forward(self, x: torch.Tensor):
    # x: (batch, n) fp32 on CUDA
    outs = self._ext.topk_cuda(x, self.k, self._partial_v, self._partial_i, self.bpr)
    return outs[0], outs[1]
