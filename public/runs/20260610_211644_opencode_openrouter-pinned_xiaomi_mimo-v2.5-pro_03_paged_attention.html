"""Triton paged-attention decode kernel for RTX PRO 6000 (SM120).

Single-query decode: each batch element has a query (num_heads, head_dim) and
attends over a KV cache stored as fixed-size pages. The KV cache layout is
(num_blocks, page_size, num_kv_heads, head_dim * 2) with [K|V] packed on the
last dim.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _paged_attn_kernel(
    Q,
    KVC,
    BT,
    SL,
    Out,
    stride_q_b, stride_q_h,
    stride_kvc_nb, stride_kvc_p, stride_kvc_h,
    stride_bt_b,
    stride_o_b, stride_o_h,
    scale,
    Hkv,
    PAGE_SIZE: tl.constexpr,
    D: tl.constexpr,
    BLOCK_L: tl.constexpr,
    G: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)

    seq_len = tl.load(SL + pid_b)
    kv_head = pid_h // G

    q_offs = tl.arange(0, D)
    q_ptr = Q + pid_b * stride_q_b + pid_h * stride_q_h + q_offs
    q = tl.load(q_ptr).to(tl.float32)

    m_prev = tl.full([], -1e9, dtype=tl.float32)
    l_prev = tl.full([], 0.0, dtype=tl.float32)
    acc = tl.zeros([D], dtype=tl.float32)

    for l_start in range(0, seq_len, BLOCK_L):
        l_offs = l_start + tl.arange(0, BLOCK_L)
        l_mask = l_offs < seq_len

        page_idx = l_offs // PAGE_SIZE
        page_off = l_offs % PAGE_SIZE

        bt_ptr = BT + pid_b * stride_bt_b
        physical_pages = tl.load(bt_ptr + page_idx, mask=l_mask, other=0)

        kv_offset = (physical_pages * stride_kvc_nb
                     + page_off * stride_kvc_p
                     + kv_head * stride_kvc_h)

        k_ptrs = KVC + kv_offset[:, None] + q_offs[None, :]
        v_ptrs = KVC + kv_offset[:, None] + D + q_offs[None, :]

        kv_mask = l_mask[:, None]
        k = tl.load(k_ptrs, mask=kv_mask, other=0.0).to(tl.float32)
        v = tl.load(v_ptrs, mask=kv_mask, other=0.0).to(tl.float32)

        scores = tl.sum(q[None, :] * k, axis=1) * scale
        scores = tl.where(l_mask, scores, -1e9)

        m_new = tl.maximum(m_prev, tl.max(scores, axis=0))
        alpha = tl.exp(m_prev - m_new)
        exp_scores = tl.exp(scores - m_new)
        l_new = alpha * l_prev + tl.sum(exp_scores, axis=0)

        acc = alpha * acc + tl.sum(exp_scores[:, None] * v, axis=0)

        m_prev = m_new
        l_prev = l_new

    out = acc / tl.maximum(l_prev, 1e-9)
    out_ptr = Out + pid_b * stride_o_b + pid_h * stride_o_h + q_offs
    tl.store(out_ptr, out.to(Out.dtype.element_ty))


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        # Choose BLOCK_L based on head_dim to balance compute vs memory
        BLOCK_L = 64

        grid = (B, H)

        _paged_attn_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0),
            out.stride(0), out.stride(1),
            self.scale,
            Hkv,
            PAGE_SIZE=P,
            D=D,
            BLOCK_L=BLOCK_L,
            G=G,
        )

        return out
