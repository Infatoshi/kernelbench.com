+    D: tl.constexpr,

+    pid_kv = tl.program_id(1)
+    pid_s = tl.program_id(2)
+    kv_head = pid_kv

+    split_start = pid_s * SPLIT_SIZE
+    split_end = tl.minimum(split_start + SPLIT_SIZE, L)

+    q = (tl.load(q_ptrs).to(tl.float32) * scale).to(tl.bfloat16)

+    l_i = tl.zeros([G], dtype=tl.float32)
+    acc = tl.zeros([G, D], dtype=tl.float32)

+    token_stride = Hkv * TWO_D
+    head_off = kv_head * TWO_D

+    n_iters = SPLIT_SIZE // BLOCK_N
+    for i in range(n_iters):
