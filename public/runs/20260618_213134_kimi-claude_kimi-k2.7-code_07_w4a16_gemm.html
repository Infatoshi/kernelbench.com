+@triton.jit
+def w4a16_gemm_kernel(
+    x_ptr,
+    w_ptr,
+    s_ptr,
+    z_ptr,
+    out_ptr,

+def gemv_kernel(
+    x_ptr,
+    w_ptr,
+    s_ptr,
+    z_ptr,
+    out_ptr,
+    K,
+    N,
+    GROUP_SIZE: tl.constexpr,
+    BLOCK_K: tl.constexpr,
+    BLOCK_N: tl.constexpr,
+):
+    pid_n = tl.program_id(0)
+    n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
+    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
+    num_pk = K // 2
+    for k in range(0, K, BLOCK_K):
+        rk = k + tl.arange(0, BLOCK_K)
+        pk = k // 2 + tl.arange(0, BLOCK_K // 2)
+        # x as (PK,2)
+        idx2 = tl.arange(0, 2)
+        x2 = tl.load(

+            other=0.0,
+        ).to(tl.bfloat16)
+        x_even = x2[:, 0]
+        x_odd = x2[:, 1]
+        wb = tl.load(

+        hi = ((wb >> 4) & 0xF).to(tl.bfloat16)

+        hi = (hi - zero[None, :]) * scale[None, :]
