"""Paged-attention decode kernel (FlashDecoding + GQA KV sharing) in Triton.

Single-query decode over a paged KV cache. One CTA per (batch, kv_head, kv-split)
streams its slice of the sequence ONCE and reuses K/V across all G query heads
that share the kv_head (GQA), so each KV element is read from DRAM exactly once.
An online (FlashAttention-style) softmax produces per-split partials; a second
trivial kernel merges them with the standard rescale/weighted-sum/normalize.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

# H100 PCIe: 114 SMs
NUM_SMS = 114

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _decode_split_kernel(
    Q_ptr,
    KV_ptr,
    BLOCK_TABLE_ptr,
    SEQ_LENS_ptr,
    O_PART_ptr,
    ML_ptr,
    stride_qb,
    stride_qh,
    stride_bt_b,
    stride_op_b,
    stride_op_h,
    stride_op_s,
    stride_ml_b,
    stride_ml_h,
    stride_ml_s,
    scale,
    G: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    Hkv: tl.constexpr,
    TWO_D: tl.constexpr,
    SPLIT_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)
    pid_s = tl.program_id(2)
    kv_head = pid_kv

    L = tl.load(SEQ_LENS_ptr + pid_b)
    split_start = pid_s * SPLIT_SIZE
    split_end = tl.minimum(split_start + SPLIT_SIZE, L)

    d = tl.arange(0, D)
    g = tl.arange(0, G)
    h0 = kv_head * G

    q_ptrs = Q_ptr + pid_b * stride_qb + (h0 + g)[:, None] * stride_qh + d[None, :]
    q = (tl.load(q_ptrs).to(tl.float32) * scale).to(tl.bfloat16)

    m_i = tl.full([G], -1.0e38, dtype=tl.float32)
    l_i = tl.zeros([G], dtype=tl.float32)
    acc = tl.zeros([G, D], dtype=tl.float32)

    page_stride = P * Hkv * TWO_D
    token_stride = Hkv * TWO_D
    head_off = kv_head * TWO_D

    n = tl.arange(0, BLOCK_N)
    n_iters = SPLIT_SIZE // BLOCK_N
    for i in range(n_iters):
        n_start = split_start + i * BLOCK_N
        tok_global = n_start + n
        mask = tok_global < split_end
        tok_safe = tl.where(mask, tok_global, 0)
        page_idx = tok_safe // P
        tok_in_page = tok_safe % P
        block_ids = tl.load(BLOCK_TABLE_ptr + pid_b * stride_bt_b + page_idx)
        base = block_ids * page_stride + tok_in_page * token_stride + head_off
        k = tl.load(KV_ptr + base[:, None] + d[None, :], mask=mask[:, None], other=0.0)
        v = tl.load(KV_ptr + base[:, None] + (D + d)[None, :], mask=mask[:, None], other=0.0)
        qk = tl.dot(q, tl.trans(k))
        qk = tl.where(mask[None, :], qk, -float("inf"))
        m_new = tl.maximum(m_i, tl.max(qk, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(qk - m_new[:, None])
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None]
        acc = tl.dot(p.to(tl.bfloat16), v, acc=acc)
        m_i = m_new

    op_ptrs = (
        O_PART_ptr
        + pid_b * stride_op_b
        + (h0 + g)[:, None] * stride_op_h
        + pid_s * stride_op_s
        + d[None, :]
    )
    tl.store(op_ptrs, acc)
    ml_off = pid_b * stride_ml_b + (h0 + g) * stride_ml_h + pid_s * stride_ml_s
    tl.store(ML_ptr + ml_off + 0, m_i)
    tl.store(ML_ptr + ml_off + 1, l_i)


@triton.jit
def _reduce_kernel(
    O_PART_ptr,
    ML_ptr,
    OUT_ptr,
    stride_op_b,
    stride_op_h,
    stride_op_s,
    stride_ml_b,
    stride_ml_h,
    stride_ml_s,
    stride_out_b,
    stride_out_h,
    num_splits,
    D: tl.constexpr,
    BLOCK_S: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)

    d = tl.arange(0, D)
    s = tl.arange(0, BLOCK_S)
    mask_s = s < num_splits

    ml_b = pid_b * stride_ml_b + pid_h * stride_ml_h + s * stride_ml_s
    m = tl.load(ML_ptr + ml_b + 0, mask=mask_s, other=-float("inf"))
    l = tl.load(ML_ptr + ml_b + 1, mask=mask_s, other=0.0)

    M = tl.max(m)
    scale_s = tl.exp(m - M)
    denom = tl.sum(scale_s * l)

    o_ptrs = (
        O_PART_ptr
        + pid_b * stride_op_b
        + pid_h * stride_op_h
        + s[:, None] * stride_op_s
        + d[None, :]
    )
    o = tl.load(o_ptrs, mask=mask_s[:, None], other=0.0)
    weighted = scale_s[:, None] * o
    out = tl.sum(weighted, axis=0) / denom
    tl.store(OUT_ptr + pid_b * stride_out_b + pid_h * stride_out_h + d, out.to(tl.bfloat16))


def _next_pow2(x: int) -> int:
    p = 1
    while p < x:
        p <<= 1
    return p


def _choose_config(B: int, Hkv: int, L: int, D: int, G: int):
    """Pick (num_splits, split_size, BLOCK_N, num_warps, num_stages)."""
    base = B * Hkv
    target_ctas = 8 * NUM_SMS
    ns = max(1, (target_ctas + base - 1) // base)
    max_ns = max(1, L // 64)
    ns = min(ns, max_ns)

    BLOCK_N = 128 if D >= 128 else 64
    split_size = ((L + ns - 1) // ns + BLOCK_N - 1) // BLOCK_N * BLOCK_N
    num_warps = 4 if D <= 64 else 8
    num_stages = 2
    return ns, split_size, BLOCK_N, num_warps, num_stages


# --- Shape knobs (overridden by check.py / benchmark.py from shapes.py) ----
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        P = self.page_size
        G = self.group_size

        L = int(seq_lens.max().item())
        ns, split_size, BLOCK_N, num_warps, num_stages = _choose_config(B, Hkv, L, D, G)

        device = query.device
        out = torch.empty(B, H, D, dtype=torch.bfloat16, device=device)
        o_part = torch.empty(B, H, ns, D, dtype=torch.float32, device=device)
        ml = torch.empty(B, H, ns, 2, dtype=torch.float32, device=device)

        grid = (B, Hkv, ns)
        _decode_split_kernel[grid](
            query,
            kv_cache,
            block_table,
            seq_lens,
            o_part,
            ml,
            query.stride(0),
            query.stride(1),
            block_table.stride(0),
            o_part.stride(0),
            o_part.stride(1),
            o_part.stride(2),
            ml.stride(0),
            ml.stride(1),
            ml.stride(2),
            self.scale,
            G=G,
            D=D,
            P=P,
            Hkv=Hkv,
            TWO_D=2 * D,
            SPLIT_SIZE=split_size,
            BLOCK_N=BLOCK_N,
            num_warps=num_warps,
            num_stages=num_stages,
        )

        grid2 = (B, H)
        _reduce_kernel[grid2](
            o_part,
            ml,
            out,
            o_part.stride(0),
            o_part.stride(1),
            o_part.stride(2),
            ml.stride(0),
            ml.stride(1),
            ml.stride(2),
            out.stride(0),
            out.stride(1),
            ns,
            D=D,
            BLOCK_S=_next_pow2(ns),
            num_warps=4,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
