"""Grouped GEMM + fused SwiGLU up-projection for top-K MoE FFN (B200 / SM100).

Per expert e with token rows [offsets[e]:offsets[e+1]):
    out = silu(x @ W_gate[e]) * (x @ W_up[e])

Triton grouped GEMM with dual-B accumulation sharing activations across K-tiles.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "grouped_gemm_swiglu"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _silu(x):
    x32 = x.to(tl.float32)
    return (x32 * tl.sigmoid(x32)).to(tl.bfloat16)


@triton.jit
def _grouped_swiglu_kernel(
    x_ptr,
    gate_ptr,
    up_ptr,
    out_ptr,
    expert_offsets_ptr,
    stride_xm,
    stride_xk,
    stride_we,
    stride_wh,
    stride_wi,
    stride_om,
    stride_on,
    H,
    I,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_e = tl.program_id(0)
    pid_m = tl.program_id(1)
    pid_n = tl.program_id(2)

    start = tl.load(expert_offsets_ptr + pid_e).to(tl.int64)
    end = tl.load(expert_offsets_ptr + pid_e + 1).to(tl.int64)
    m_size = end - start
    if m_size <= 0 or pid_m * BLOCK_M >= m_size:
        return

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    mask_m = offs_m < m_size
    mask_n = offs_n < I

    x_base = x_ptr + start * stride_xm
    gate_base = gate_ptr + pid_e * stride_we
    up_base = up_ptr + pid_e * stride_we

    acc_gate = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc_up = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k in range(0, H, BLOCK_K):
        offs_k_cur = k + offs_k
        mask_k = offs_k_cur < H

        a_ptrs = x_base + (offs_m[:, None] * stride_xm + offs_k_cur[None, :] * stride_xk)
        a = tl.load(a_ptrs, mask=mask_m[:, None] & mask_k[None, :], other=0.0)

        b_gate_ptrs = gate_base + (offs_k_cur[:, None] * stride_wh + offs_n[None, :] * stride_wi)
        b_gate = tl.load(b_gate_ptrs, mask=mask_k[:, None] & mask_n[None, :], other=0.0)

        b_up_ptrs = up_base + (offs_k_cur[:, None] * stride_wh + offs_n[None, :] * stride_wi)
        b_up = tl.load(b_up_ptrs, mask=mask_k[:, None] & mask_n[None, :], other=0.0)

        acc_gate += tl.dot(a, b_gate)
        acc_up += tl.dot(a, b_up)

    out = _silu(acc_gate.to(tl.bfloat16)) * acc_up.to(tl.bfloat16)

    out_base = out_ptr + start * stride_om
    out_ptrs = out_base + (offs_m[:, None] * stride_om + offs_n[None, :] * stride_on)
    tl.store(out_ptrs, out, mask=mask_m[:, None] & mask_n[None, :])


def _pick_config(H: int, I: int, max_m: int) -> dict:
    if I >= 3072:
        return dict(BLOCK_M=64, BLOCK_N=256, BLOCK_K=64, num_warps=8, num_stages=4)
    if H >= 4096:
        return dict(BLOCK_M=128, BLOCK_N=256, BLOCK_K=64, num_warps=8, num_stages=4)
    if max_m <= 512:
        return dict(BLOCK_M=64, BLOCK_N=128, BLOCK_K=64, num_warps=4, num_stages=3)
    return dict(BLOCK_M=128, BLOCK_N=128, BLOCK_K=64, num_warps=8, num_stages=4)


def grouped_swiglu(
    hidden_states: torch.Tensor,
    W_gate: torch.Tensor,
    W_up: torch.Tensor,
    expert_offsets: torch.Tensor,
) -> torch.Tensor:
    T_perm, H = hidden_states.shape
    E, H_w, I = W_gate.shape
    assert H == H_w and W_up.shape == W_gate.shape
    assert expert_offsets.shape == (E + 1,)

    hidden_states = hidden_states.contiguous()
    W_gate = W_gate.contiguous()
    W_up = W_up.contiguous()
    expert_offsets = expert_offsets.contiguous()

    out = torch.empty(T_perm, I, dtype=torch.bfloat16, device=hidden_states.device)

    max_m = (T_perm + E - 1) // E
    cfg = _pick_config(H, I, max_m)
    bm, bn = cfg["BLOCK_M"], cfg["BLOCK_N"]
    grid = (
        E,
        triton.cdiv(max_m, bm),
        triton.cdiv(I, bn),
    )

    _grouped_swiglu_kernel[grid](
        hidden_states,
        W_gate,
        W_up,
        out,
        expert_offsets,
        hidden_states.stride(0),
        hidden_states.stride(1),
        W_gate.stride(0),
        W_gate.stride(1),
        W_gate.stride(2),
        out.stride(0),
        out.stride(1),
        H,
        I,
        **{k: v for k, v in cfg.items()},
    )
    return out


class Model(nn.Module):
    """Up-projection of a top-K MoE FFN with fused SwiGLU."""

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(
        self,
        hidden_states: torch.Tensor,
        expert_offsets: torch.Tensor,
    ) -> torch.Tensor:
        return grouped_swiglu(hidden_states, self.W_gate, self.W_up, expert_offsets)


T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
