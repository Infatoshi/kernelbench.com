"""Fused W4A16 weight-only quantized GEMM (AWQ/GPTQ-style asymmetric int4).

M=1: CUDA split-K GEMV with coalesced uint4 weight layout.
M>1: Triton fused dequant + tensor-core GEMM.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl
from torch.utils.cpp_extension import load_inline

GROUP_SIZE = 128
WARP = 32
PACK = GROUP_SIZE // 2  # 64
NV = PACK // 16  # 4 uint4 chunks per group per column

_CUDA_SRC = r'''
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_bf16.h>
#include <cstdint>

// Weight layout as uint4 array: (n_groups, NV=4, n_tiles, 32)
// linear index: ((g * NV + v) * n_tiles + tile) * 32 + lane
// Warp-coalesced 16-byte loads; each thread owns one N column.
template <int GROUP>
__global__ __launch_bounds__(128, 8)
void w4a16_gemv_m1_u4(
    const __nv_bfloat16* __restrict__ x,
    const uint4* __restrict__ w,
    const float* __restrict__ sf,
    const float* __restrict__ bf,
    float* __restrict__ partial,
    int N, int n_tiles, int gps, int n_groups
) {
    constexpr int PACK_ = GROUP / 2;
    constexpr int NV_ = PACK_ / 16;
    constexpr int WARP_ = 32;
    const int warp_id = threadIdx.x / WARP_;
    const int lane = threadIdx.x % WARP_;
    const int tile = blockIdx.x * 4 + warp_id;
    const int split = blockIdx.y;
    const int n = tile * WARP_ + lane;
    const bool ok = tile < n_tiles && n < N;
    const int g0 = split * gps;
    const int g1 = min(g0 + gps, n_groups);

    __shared__ float sx[GROUP];
    float acc = 0.f;

    for (int g = g0; g < g1; ++g) {
        for (int i = threadIdx.x; i < GROUP; i += 128)
            sx[i] = __bfloat162float(x[g * GROUP + i]);
        __syncthreads();
        if (ok) {
            const float s = __ldg(&sf[g * N + n]);
            const float b = __ldg(&bf[g * N + n]);
            #pragma unroll
            for (int v = 0; v < NV_; ++v) {
                const size_t idx =
                    ((size_t)(g * NV_ + v) * n_tiles + tile) * WARP_ + lane;
                const uint4 u = __ldg(&w[idx]);
                const uint8_t* pk = reinterpret_cast<const uint8_t*>(&u);
                #pragma unroll
                for (int j = 0; j < 16; ++j) {
                    const int p = v * 16 + j;
                    acc = fmaf(sx[2 * p],
                               (float)(pk[j] & 0xF) * s + b, acc);
                    acc = fmaf(sx[2 * p + 1],
                               (float)((pk[j] >> 4) & 0xF) * s + b, acc);
                }
            }
        }
        __syncthreads();
    }
    if (ok) partial[(size_t)split * N + n] = acc;
}

__global__ void w4a16_reduce(
    const float* __restrict__ partial,
    __nv_bfloat16* __restrict__ out,
    int N, int splits
) {
    const int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (n >= N) return;
    float acc = 0.f;
    for (int s = 0; s < splits; ++s)
        acc += partial[(size_t)s * N + n];
    out[n] = __float2bfloat16(acc);
}

void w4a16_gemv_m1(
    torch::Tensor x, torch::Tensor w_u4, torch::Tensor sf, torch::Tensor bf,
    torch::Tensor out, torch::Tensor partial, int n_tiles
) {
    const int N = sf.size(1);
    const int n_groups = sf.size(0);
    const int splits = partial.size(0);
    constexpr int GROUP = 128;
    const int gps = (n_groups + splits - 1) / splits;
    auto stream = at::cuda::getCurrentCUDAStream();
    dim3 grid((n_tiles + 3) / 4, splits);
    w4a16_gemv_m1_u4<GROUP><<<grid, 128, 0, stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        reinterpret_cast<const uint4*>(w_u4.data_ptr()),
        sf.data_ptr<float>(), bf.data_ptr<float>(),
        partial.data_ptr<float>(), N, n_tiles, gps, n_groups);
    w4a16_reduce<<<(N + 255) / 256, 256, 0, stream>>>(
        partial.data_ptr<float>(),
        reinterpret_cast<__nv_bfloat16*>(out.data_ptr()), N, splits);
}
'''

_CPP_SRC = r'''
#include <torch/extension.h>
void w4a16_gemv_m1(torch::Tensor, torch::Tensor, torch::Tensor,
    torch::Tensor, torch::Tensor, torch::Tensor, int);
'''

_EXT = None


def _get_ext():
    global _EXT
    if _EXT is not None:
        return _EXT
    flags = ["-O3", "--use_fast_math", "-std=c++17"]
    if torch.cuda.get_device_capability()[0] >= 10:
        flags.append("-gencode=arch=compute_100a,code=sm_100a")
    _EXT = load_inline(
        name="w4a16_u4_coalesced_v1",
        cpp_sources=_CPP_SRC,
        cuda_sources=_CUDA_SRC,
        functions=["w4a16_gemv_m1"],
        extra_cuda_cflags=flags,
        verbose=False,
    )
    return _EXT


@triton.autotune(
    configs=[
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 128}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 256}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 128}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 256}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 64}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 128}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 128}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 256}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 64}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 128}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 128}, num_warps=8, num_stages=4),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 256}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 256, "BLOCK_N": 64}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 256, "BLOCK_N": 128}, num_warps=8, num_stages=2),
    ],
    key=["M", "N", "K"],
)
@triton.jit
def _w4a16_gemm_kernel(
    x_ptr, w_ptr, s_ptr, z_ptr, o_ptr,
    M, N, K,
    stride_xm, stride_xk, stride_wk, stride_wn,
    stride_sg, stride_sn, stride_zg, stride_zn,
    stride_om, stride_on,
    GROUP: tl.constexpr, BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    group = 8
    pid_grp = pid // (group * num_pid_n)
    first_m = pid_grp * group
    group_m = min(num_pid_m - first_m, group)
    pid_m = first_m + ((pid % (group * num_pid_n)) % group_m)
    pid_n = (pid % (group * num_pid_n)) // group_m

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_m = offs_m < M
    mask_n = offs_n < N
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for g in range(tl.cdiv(K, GROUP)):
        k_g = g * GROUP
        s = tl.load(s_ptr + g * stride_sg + offs_n * stride_sn, mask=mask_n, other=0.0).to(tl.float32)
        z = tl.load(z_ptr + g * stride_zg + offs_n * stride_zn, mask=mask_n, other=0.0).to(tl.float32)
        offs_k = k_g + tl.arange(0, GROUP)
        x = tl.load(
            x_ptr + offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk,
            mask=mask_m[:, None], other=0.0,
        )
        pack = (k_g // 2) + tl.arange(0, GROUP // 2)
        wp = tl.load(
            w_ptr + pack[:, None] * stride_wk + offs_n[None, :] * stride_wn,
            mask=mask_n[None, :], other=0,
        )
        lo = ((wp & 0xF).to(tl.float32) - z[None, :]) * s[None, :]
        hi = (((wp >> 4) & 0xF).to(tl.float32) - z[None, :]) * s[None, :]
        j = tl.join(lo, hi)
        w_deq = tl.reshape(tl.permute(j, (0, 2, 1)), (GROUP, BLOCK_N)).to(tl.bfloat16)
        acc += tl.dot(x, w_deq)

    tl.store(
        o_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on,
        acc.to(tl.bfloat16),
        mask=mask_m[:, None] & mask_n[None, :],
    )


_PARTIAL_CACHE: dict = {}
_PREP_CACHE: dict = {}


def _get_partial(splits, N, device):
    key = (splits, N, str(device))
    t = _PARTIAL_CACHE.get(key)
    if t is None or t.shape != (splits, N):
        t = torch.empty((splits, N), device=device, dtype=torch.float32)
        _PARTIAL_CACHE[key] = t
    return t


def _prepare(w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor):
    """Build coalesced uint4 weight layout + float scale/bias."""
    key = (w_q.data_ptr(), scales.data_ptr(), zeros.data_ptr(), tuple(w_q.shape))
    cached = _PREP_CACHE.get(key)
    if cached is not None:
        return cached

    K_half, N = w_q.shape
    n_groups = scales.shape[0]
    n_tiles = (N + WARP - 1) // WARP
    N_pad = n_tiles * WARP

    if N_pad != N:
        w_pad = torch.zeros(K_half, N_pad, dtype=torch.uint8, device=w_q.device)
        w_pad[:, :N] = w_q
    else:
        w_pad = w_q

    # (n_groups, PACK, n_tiles, WARP) -> (n_groups, NV, 16, n_tiles, WARP)
    # -> (n_groups, NV, n_tiles, WARP, 16) contiguous for uint4 loads
    w4 = w_pad.view(n_groups, PACK, n_tiles, WARP)
    w4 = w4.view(n_groups, NV, 16, n_tiles, WARP)
    w4 = w4.permute(0, 1, 3, 4, 2).contiguous()  # (ng, NV, n_tiles, WARP, 16)
    w_u4 = w4.view(-1)  # flat uint8, interpreted as uint4* in kernel

    sf = scales.float().contiguous()
    bf = (-zeros.float() * sf).contiguous()
    result = (w_u4, sf, bf, n_tiles)
    _PREP_CACHE[key] = result
    if len(_PREP_CACHE) > 16:
        _PREP_CACHE.pop(next(iter(_PREP_CACHE)))
    return result


def _launch_w4a16(x, w_q, scales, zeros, group_size=GROUP_SIZE):
    M, K = x.shape
    N = w_q.shape[1]
    x = x.contiguous()
    w_q = w_q.contiguous()
    scales = scales.contiguous()
    zeros = zeros.contiguous()
    out = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)

    if M == 1:
        w_u4, sf, bf, n_tiles = _prepare(w_q, scales, zeros)
        splits = 32
        partial = _get_partial(splits, N, x.device)
        _get_ext().w4a16_gemv_m1(
            x.view(-1), w_u4, sf, bf, out.view(-1), partial, n_tiles
        )
    else:
        grid = lambda META: (
            triton.cdiv(M, META["BLOCK_M"]) * triton.cdiv(N, META["BLOCK_N"]),
        )
        _w4a16_gemm_kernel[grid](
            x, w_q, scales, zeros, out,
            M, N, K,
            x.stride(0), x.stride(1),
            w_q.stride(0), w_q.stride(1),
            scales.stride(0), scales.stride(1),
            zeros.stride(0), zeros.stride(1),
            out.stride(0), out.stride(1),
            GROUP=group_size,
        )
    return out


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0 and K % 2 == 0
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size
        self.register_buffer("w_q", torch.zeros(K // 2, N, dtype=torch.uint8))
        self.register_buffer("scales", torch.ones(n_groups, N, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n_groups, N, dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return _launch_w4a16(
            x.to(torch.bfloat16), self.w_q, self.scales, self.zeros, self.group_size
        )


M, N, K = 1, 12288, 4096


def get_inputs():
    return [torch.randn(M, K, dtype=torch.bfloat16)]


def get_init_inputs():
    return [M, N, K]
