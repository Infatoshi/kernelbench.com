import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _moe_swiglu_kernel(
    x_ptr,
    offsets_ptr,
    wg_ptr,
    wu_ptr,
    out_ptr,
    H: tl.constexpr,
    I: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)
    expert = tl.program_id(2)

    start = tl.load(offsets_ptr + expert)
    end = tl.load(offsets_ptr + expert + 1)
    rows = end - start
    m_offsets = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    n_offsets = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    k_offsets = tl.arange(0, BLOCK_K)

    gate_acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)
    up_acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)

    for k0 in range(0, H, BLOCK_K):
        k = k0 + k_offsets
        x = tl.load(
            x_ptr + (start + m_offsets[:, None]) * H + k[None, :],
            mask=(m_offsets[:, None] < rows),
            other=0.0,
        )
        wg = tl.load(
            wg_ptr + expert * H * I + k[:, None] * I + n_offsets[None, :],
            mask=n_offsets[None, :] < I,
            other=0.0,
        )
        wu = tl.load(
            wu_ptr + expert * H * I + k[:, None] * I + n_offsets[None, :],
            mask=n_offsets[None, :] < I,
            other=0.0,
        )
        gate_acc = tl.dot(x, wg, gate_acc)
        up_acc = tl.dot(x, wu, up_acc)

    gate = gate_acc
    sig = 1.0 / (1.0 + tl.exp(-gate))
    y = gate * sig * up_acc
    tl.store(
        out_ptr + (start + m_offsets[:, None]) * I + n_offsets[None, :],
        y,
        mask=(m_offsets[:, None] < rows) & (n_offsets[None, :] < I),
    )

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets
