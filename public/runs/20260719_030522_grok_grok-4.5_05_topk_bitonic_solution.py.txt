"""High-performance top-k for B200 (SM100) — custom CUDA hybrid.

Paths:
  k==1   : multi-block argmax (ILP scan + warp/block reduce)
  k<=16  : hierarchical per-thread sorted insert + butterfly block merge
  k>16   : tile bitonic select (shared-mem sort of 1024-element tiles with a
           running top-k; multi-block for large n, then candidate merge)

Does not call library selection/sort entry points (see problem.yaml forbidden list).
"""
from __future__ import annotations

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>
#include <cfloat>
#include <algorithm>
#include <type_traits>

struct alignas(8) Pair {
  float val;
  int   idx;
};

__device__ __forceinline__ bool better(const Pair& a, const Pair& b) {
  if (a.idx < 0) return false;
  if (b.idx < 0) return true;
  return (a.val > b.val) || (a.val == b.val && a.idx < b.idx);
}

__device__ __forceinline__ Pair pmax(const Pair& a, const Pair& b) {
  return better(a, b) ? a : b;
}

__device__ __forceinline__ void swap_pair(Pair& a, Pair& b) {
  Pair t = a; a = b; b = t;
}

__device__ __forceinline__ void cas_desc(Pair& a, Pair& b) {
  if (better(b, a)) swap_pair(a, b);
}
__device__ __forceinline__ void cas_asc(Pair& a, Pair& b) {
  if (better(a, b)) swap_pair(a, b);
}

__device__ __forceinline__ void insert_topk(Pair* heap, int& fill, int k, Pair cand) {
  if (fill < k) {
    int pos = fill;
    while (pos > 0 && better(cand, heap[pos - 1])) {
      heap[pos] = heap[pos - 1];
      --pos;
    }
    heap[pos] = cand;
    ++fill;
    return;
  }
  if (!better(cand, heap[k - 1])) return;
  int pos = k - 1;
  while (pos > 0 && better(cand, heap[pos - 1])) {
    heap[pos] = heap[pos - 1];
    --pos;
  }
  heap[pos] = cand;
}

__device__ __forceinline__ Pair warp_reduce_max(Pair p) {
  #pragma unroll
  for (int off = 16; off > 0; off >>= 1) {
    Pair o;
    o.val = __shfl_down_sync(0xffffffff, p.val, off);
    o.idx = __shfl_down_sync(0xffffffff, p.idx, off);
    p = pmax(p, o);
  }
  return p;
}

__device__ void bitonic_sort_desc(Pair* s, int n, int tid, int nthreads) {
  for (int size = 2; size <= n; size <<= 1) {
    for (int stride = size >> 1; stride > 0; stride >>= 1) {
      __syncthreads();
      for (int i = tid; i < (n >> 1); i += nthreads) {
        int low = ((i / stride) * (stride << 1)) + (i % stride);
        int high = low + stride;
        bool ascending = ((low & size) != 0);
        if (ascending) cas_asc(s[low], s[high]);
        else           cas_desc(s[low], s[high]);
      }
    }
  }
  __syncthreads();
}

__device__ __forceinline__ void merge_lists(const Pair* a, const Pair* b, Pair* out, int k) {
  int i = 0, j = 0;
  for (int t = 0; t < k; ++t) {
    bool take_a;
    if (i >= k) take_a = false;
    else if (j >= k) take_a = true;
    else take_a = !better(b[j], a[i]);
    out[t] = take_a ? a[i++] : b[j++];
  }
}

// ============================== k==1 ==============================
__global__ void argmax_stage1(
    const float* __restrict__ x,
    float* __restrict__ bv, int* __restrict__ bi,
    int n, int bpr)
{
  const int row = blockIdx.y, b = blockIdx.x, tid = threadIdx.x, NT = blockDim.x;
  const int chunk = (n + bpr - 1) / bpr;
  const int start = b * chunk, end = min(start + chunk, n);
  Pair local{-FLT_MAX, 0};
  if (start < n) {
    const float* rowp = x + (size_t)row * n;
    float v0 = -FLT_MAX, v1 = -FLT_MAX, v2 = -FLT_MAX, v3 = -FLT_MAX;
    int i0 = start, i1 = start, i2 = start, i3 = start;
    int i = start + tid;
    for (; i + 3 * NT < end; i += 4 * NT) {
      float a = rowp[i], b2 = rowp[i + NT], c = rowp[i + 2 * NT], d = rowp[i + 3 * NT];
      if (a > v0) { v0 = a; i0 = i; }
      if (b2 > v1) { v1 = b2; i1 = i + NT; }
      if (c > v2) { v2 = c; i2 = i + 2 * NT; }
      if (d > v3) { v3 = d; i3 = i + 3 * NT; }
    }
    for (; i < end; i += NT) {
      float a = rowp[i];
      if (a > v0) { v0 = a; i0 = i; }
    }
    local = pmax(Pair{v0, i0}, pmax(Pair{v1, i1}, pmax(Pair{v2, i2}, Pair{v3, i3})));
  }
  __shared__ Pair sm[32];
  Pair w = warp_reduce_max(local);
  int lane = tid & 31, warp = tid >> 5;
  if (lane == 0) sm[warp] = w;
  __syncthreads();
  Pair bm{-FLT_MAX, 0};
  if (warp == 0) {
    Pair p = (lane < (NT + 31) / 32) ? sm[lane] : Pair{-FLT_MAX, 0};
    bm = warp_reduce_max(p);
  }
  if (tid == 0) { bv[row * bpr + b] = bm.val; bi[row * bpr + b] = bm.idx; }
}

__global__ void argmax_stage2(
    const float* __restrict__ bv, const int* __restrict__ bi,
    float* __restrict__ ov, int64_t* __restrict__ oi, int bpr)
{
  const int row = blockIdx.x, tid = threadIdx.x, NT = blockDim.x;
  Pair local{-FLT_MAX, 0};
  for (int i = tid; i < bpr; i += NT)
    local = pmax(local, Pair{bv[row * bpr + i], bi[row * bpr + i]});
  __shared__ Pair sm[32];
  Pair w = warp_reduce_max(local);
  int lane = tid & 31, warp = tid >> 5;
  if (lane == 0) sm[warp] = w;
  __syncthreads();
  if (warp == 0) {
    Pair p = (lane < (NT + 31) / 32) ? sm[lane] : Pair{-FLT_MAX, 0};
    w = warp_reduce_max(p);
  }
  if (tid == 0) { ov[row] = w.val; oi[row] = (int64_t)w.idx; }
}

// ===================== small-k hierarchical insert =====================
template <int KMAX, int BLOCK>
__device__ void block_merge_insert(Pair* smem, Pair* local, int k, int tid) {
  #pragma unroll
  for (int j = 0; j < KMAX; ++j) smem[tid * KMAX + j] = local[j];
  __syncthreads();
  for (int offset = 1; offset < BLOCK; offset <<= 1) {
    __syncthreads();
    if ((tid & (2 * offset - 1)) == 0) {
      Pair a[KMAX], b[KMAX], out[KMAX];
      #pragma unroll
      for (int j = 0; j < KMAX; ++j) {
        a[j] = smem[tid * KMAX + j];
        b[j] = smem[(tid + offset) * KMAX + j];
      }
      int ia = 0, ib = 0;
      #pragma unroll
      for (int t = 0; t < KMAX; ++t) {
        if (t >= k) { out[t] = Pair{-FLT_MAX, -1}; continue; }
        bool take_a;
        if (ia >= k) take_a = false;
        else if (ib >= k) take_a = true;
        else take_a = !better(b[ib], a[ia]);
        out[t] = take_a ? a[ia++] : b[ib++];
      }
      #pragma unroll
      for (int j = 0; j < KMAX; ++j) smem[tid * KMAX + j] = out[j];
    }
  }
  __syncthreads();
}

template <int KMAX, int BLOCK>
__global__ void topk_insert_single(
    const float* __restrict__ x,
    float* __restrict__ ov, int64_t* __restrict__ oi,
    int n, int k)
{
  const int row = blockIdx.x, tid = threadIdx.x;
  Pair local[KMAX];
  #pragma unroll
  for (int j = 0; j < KMAX; ++j) local[j] = Pair{-FLT_MAX, -1};
  int fill = 0;
  const float* rowp = x + (size_t)row * n;
  int i = tid;
  for (; i + 3 * BLOCK < n; i += 4 * BLOCK) {
    float a0 = rowp[i], a1 = rowp[i + BLOCK], a2 = rowp[i + 2 * BLOCK], a3 = rowp[i + 3 * BLOCK];
    float thr = (fill >= k) ? local[k - 1].val : -FLT_MAX;
    if (a0 > thr) insert_topk(local, fill, k, Pair{a0, i});
    thr = (fill >= k) ? local[k - 1].val : -FLT_MAX;
    if (a1 > thr) insert_topk(local, fill, k, Pair{a1, i + BLOCK});
    thr = (fill >= k) ? local[k - 1].val : -FLT_MAX;
    if (a2 > thr) insert_topk(local, fill, k, Pair{a2, i + 2 * BLOCK});
    thr = (fill >= k) ? local[k - 1].val : -FLT_MAX;
    if (a3 > thr) insert_topk(local, fill, k, Pair{a3, i + 3 * BLOCK});
  }
  for (; i < n; i += BLOCK) {
    float a0 = rowp[i];
    float thr = (fill >= k) ? local[k - 1].val : -FLT_MAX;
    if (a0 > thr) insert_topk(local, fill, k, Pair{a0, i});
  }
  extern __shared__ char dyn[];
  Pair* smem = reinterpret_cast<Pair*>(dyn);
  block_merge_insert<KMAX, BLOCK>(smem, local, k, tid);
  if (tid < k) {
    ov[row * k + tid] = smem[tid].val;
    oi[row * k + tid] = (int64_t)smem[tid].idx;
  }
}

// ===================== large-k tile bitonic =====================
template <int TILE, int BLOCK>
__global__ void topk_bitonic_single(
    const float* __restrict__ x,
    float* __restrict__ ov, int64_t* __restrict__ oi,
    int n, int k)
{
  const int row = blockIdx.x, tid = threadIdx.x;
  extern __shared__ char dyn[];
  Pair* s = reinterpret_cast<Pair*>(dyn);
  Pair* run = s + TILE;
  if (tid < 64) run[tid] = Pair{-FLT_MAX, -1};
  __syncthreads();

  const float* rowp = x + (size_t)row * n;
  for (int base = 0; base < n; base += TILE) {
    for (int i = tid; i < TILE; i += BLOCK) {
      int idx = base + i;
      s[i] = (idx < n) ? Pair{rowp[idx], idx} : Pair{-FLT_MAX, -1};
    }
    __syncthreads();
    bitonic_sort_desc(s, TILE, tid, BLOCK);
    if (tid == 0) {
      Pair a[64], b[64], out[64];
      for (int t = 0; t < k; ++t) { a[t] = s[t]; b[t] = run[t]; }
      merge_lists(a, b, out, k);
      for (int t = 0; t < k; ++t) run[t] = out[t];
    }
    __syncthreads();
  }
  if (tid < k) {
    ov[row * k + tid] = run[tid].val;
    oi[row * k + tid] = (int64_t)run[tid].idx;
  }
}

template <int TILE, int BLOCK>
__global__ void topk_bitonic_block(
    const float* __restrict__ x,
    float* __restrict__ cv, int64_t* __restrict__ ci,
    int n, int k, int bpr)
{
  const int row = blockIdx.y, blk = blockIdx.x, tid = threadIdx.x;
  const int chunk = (n + bpr - 1) / bpr;
  const int start = blk * chunk, end = min(start + chunk, n);

  extern __shared__ char dyn[];
  Pair* s = reinterpret_cast<Pair*>(dyn);
  const float* rowp = x + (size_t)row * n;

  // Fast path: chunk fits in one tile
  if (end - start <= TILE) {
    for (int i = tid; i < TILE; i += BLOCK) {
      int idx = start + i;
      s[i] = (idx < end) ? Pair{rowp[idx], idx} : Pair{-FLT_MAX, -1};
    }
    __syncthreads();
    bitonic_sort_desc(s, TILE, tid, BLOCK);
    if (tid < k) {
      size_t off = ((size_t)row * bpr + blk) * (size_t)k + tid;
      cv[off] = s[tid].val;
      ci[off] = (int64_t)s[tid].idx;
    }
    return;
  }

  Pair* run = s + TILE;
  if (tid < 64) run[tid] = Pair{-FLT_MAX, -1};
  __syncthreads();

  for (int base = start; base < end; base += TILE) {
    for (int i = tid; i < TILE; i += BLOCK) {
      int idx = base + i;
      s[i] = (idx < end) ? Pair{rowp[idx], idx} : Pair{-FLT_MAX, -1};
    }
    __syncthreads();
    bitonic_sort_desc(s, TILE, tid, BLOCK);
    if (tid == 0) {
      Pair a[64], b[64], out[64];
      for (int t = 0; t < k; ++t) { a[t] = s[t]; b[t] = run[t]; }
      merge_lists(a, b, out, k);
      for (int t = 0; t < k; ++t) run[t] = out[t];
    }
    __syncthreads();
  }
  if (tid < k) {
    size_t off = ((size_t)row * bpr + blk) * (size_t)k + tid;
    cv[off] = run[tid].val;
    ci[off] = (int64_t)run[tid].idx;
  }
}

template <int BLOCK>
__global__ void topk_bitonic_merge(
    const float* __restrict__ cv, const int64_t* __restrict__ ci,
    float* __restrict__ ov, int64_t* __restrict__ oi,
    int n_cand, int k)
{
  const int row = blockIdx.x, tid = threadIdx.x;
  extern __shared__ char dyn[];
  Pair* s = reinterpret_cast<Pair*>(dyn);

  int n_pow2 = 1;
  while (n_pow2 < n_cand) n_pow2 <<= 1;
  const int MAX_SINGLE = 8192;

  if (n_pow2 <= MAX_SINGLE) {
    for (int i = tid; i < n_pow2; i += BLOCK) {
      s[i] = (i < n_cand)
        ? Pair{cv[row * n_cand + i], (int)ci[row * n_cand + i]}
        : Pair{-FLT_MAX, -1};
    }
    __syncthreads();
    bitonic_sort_desc(s, n_pow2, tid, BLOCK);
    if (tid < k) {
      ov[row * k + tid] = s[tid].val;
      oi[row * k + tid] = (int64_t)s[tid].idx;
    }
    return;
  }

  const int WS = 2048;
  Pair* run = s + WS;
  if (tid < 64) run[tid] = Pair{-FLT_MAX, -1};
  __syncthreads();
  for (int base = 0; base < n_cand; base += WS) {
    for (int i = tid; i < WS; i += BLOCK) {
      int idx = base + i;
      s[i] = (idx < n_cand)
        ? Pair{cv[row * n_cand + idx], (int)ci[row * n_cand + idx]}
        : Pair{-FLT_MAX, -1};
    }
    __syncthreads();
    bitonic_sort_desc(s, WS, tid, BLOCK);
    if (tid == 0) {
      Pair a[64], b[64], out[64];
      for (int t = 0; t < k; ++t) { a[t] = s[t]; b[t] = run[t]; }
      merge_lists(a, b, out, k);
      for (int t = 0; t < k; ++t) run[t] = out[t];
    }
    __syncthreads();
  }
  if (tid < k) {
    ov[row * k + tid] = run[tid].val;
    oi[row * k + tid] = (int64_t)run[tid].idx;
  }
}

void topk_cuda(torch::Tensor x, torch::Tensor values, torch::Tensor indices, int64_t k_in) {
  TORCH_CHECK(x.is_cuda() && x.scalar_type() == torch::kFloat32, "x");
  TORCH_CHECK(x.dim() == 2, "2d");
  const int batch = (int)x.size(0);
  const int n = (int)x.size(1);
  const int k = (int)k_in;
  TORCH_CHECK(k >= 1 && k <= 64 && k <= n, "k");

  auto stream = at::cuda::getCurrentCUDAStream();
  const float* xp = x.data_ptr<float>();
  float* vp = values.data_ptr<float>();
  int64_t* ip = indices.data_ptr<int64_t>();

  if (k == 1) {
    int bpr = std::max(1, std::min(64, (n + 511) / 512));
    if (n <= 4096) bpr = std::max(1, std::min(8, (n + 511) / 512));
    auto bv = torch::empty({batch, bpr}, x.options());
    auto bi = torch::empty({batch, bpr}, x.options().dtype(torch::kInt32));
    argmax_stage1<<<dim3(bpr, batch), 256, 0, stream>>>(
        xp, bv.data_ptr<float>(), bi.data_ptr<int>(), n, bpr);
    argmax_stage2<<<batch, 128, 0, stream>>>(
        bv.data_ptr<float>(), bi.data_ptr<int>(), vp, ip, bpr);
    return;
  }

  if (k <= 16) {
    if (k <= 8) {
      constexpr int KMAX = 8, BLOCK = 256;
      size_t smem = (size_t)BLOCK * KMAX * sizeof(Pair);
      cudaFuncSetAttribute((const void*)topk_insert_single<KMAX, BLOCK>,
                           cudaFuncAttributeMaxDynamicSharedMemorySize, (int)smem);
      topk_insert_single<KMAX, BLOCK><<<batch, BLOCK, smem, stream>>>(xp, vp, ip, n, k);
    } else {
      constexpr int KMAX = 16, BLOCK = 256;
      size_t smem = (size_t)BLOCK * KMAX * sizeof(Pair);
      cudaFuncSetAttribute((const void*)topk_insert_single<KMAX, BLOCK>,
                           cudaFuncAttributeMaxDynamicSharedMemorySize, (int)smem);
      topk_insert_single<KMAX, BLOCK><<<batch, BLOCK, smem, stream>>>(xp, vp, ip, n, k);
    }
    return;
  }

  // Large k: tile bitonic select
  constexpr int TILE = 1024;
  constexpr int BLOCK = 256;
  size_t smem = (size_t)(TILE + 64) * sizeof(Pair);

  int bpr = 1;
  if (n >= 16384) {
    int tiles = (n + TILE - 1) / TILE;
    bpr = std::max(1, std::min(64, (tiles + 1) / 2));
  }

  if (bpr <= 1) {
    topk_bitonic_single<TILE, BLOCK><<<batch, BLOCK, smem, stream>>>(xp, vp, ip, n, k);
    return;
  }

  auto cv = torch::empty({batch, bpr * k}, x.options());
  auto ci = torch::empty({batch, bpr * k}, x.options().dtype(torch::kInt64));
  topk_bitonic_block<TILE, BLOCK><<<dim3(bpr, batch), BLOCK, smem, stream>>>(
      xp, cv.data_ptr<float>(), ci.data_ptr<int64_t>(), n, k, bpr);

  int n_cand = bpr * k;
  int n_pow2 = 1;
  while (n_pow2 < n_cand) n_pow2 <<= 1;
  n_pow2 = std::min(n_pow2, 8192);
  size_t smem2 = (size_t)(n_pow2 + 64) * sizeof(Pair);
  cudaFuncSetAttribute((const void*)topk_bitonic_merge<BLOCK>,
                       cudaFuncAttributeMaxDynamicSharedMemorySize, (int)smem2);
  topk_bitonic_merge<BLOCK><<<batch, BLOCK, smem2, stream>>>(
      cv.data_ptr<float>(), ci.data_ptr<int64_t>(), vp, ip, n_cand, k);
}

        _mod = load_inline(

class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        values = torch.empty(self.batch, self.k, device=x.device, dtype=torch.float32)
        indices = torch.empty(self.batch, self.k, device=x.device, dtype=torch.int64)
        _get_mod().topk_cuda(x.contiguous(), values, indices, int(self.k))
        return values, indices
