"""Paged attention decode — Triton kernel, RTX PRO 6000 (sm_120).

Page-based approach: each thread block (batch, head) loads whole pages of
K/V from the paged cache, computes Q·K in a single tl.dot call per page,
and accumulates attention output with online softmax.

This reduces the sequential loop depth by PAGE_SIZE× compared with
position-by-position iteration.
"""
from __future__ import annotations

import math
from typing import List

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _paged_attention_decode_kernel(
    Q, KV_CACHE, BLOCK_TABLE, SEQ_LEN_PTR, OUT,
    stride_qb: tl.constexpr, stride_qh: tl.constexpr, stride_qd: tl.constexpr,
    stride_kvb, stride_kvps, stride_kvkh, stride_kvd,
    stride_bt,
    HEAD_DIM: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    NUM_PAGES: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)
    num_q_heads = tl.num_programs(1)
    kv_head = h // (num_q_heads // NUM_KV_HEADS)

    off_d = tl.arange(0, HEAD_DIM)

    q = tl.load(Q + b * stride_qb + h * stride_qh + off_d * stride_qd)
    q_row = q.reshape((1, HEAD_DIM))

    att_max = tl.full([1], float("-inf"), dtype=tl.bfloat16)
    att_sum = tl.full([1], 1.0, dtype=tl.bfloat16)
    out = tl.zeros([PAGE_SIZE], dtype=tl.bfloat16)

    for p in tl.static_range(NUM_PAGES):
        block_idx = tl.load(BLOCK_TABLE + b * stride_bt + p)

        k_block = tl.load(
            KV_CACHE + block_idx * stride_kvb
            + tl.arange(0, PAGE_SIZE) * stride_kvps
            + kv_head * stride_kvkh
            + off_d * stride_kvd
        )

        sim = tl.dot(q_row, tl.trans(k_block))
        sim = (sim * (1.0 / math.sqrt(HEAD_DIM))).to(tl.bfloat16)

        seq_len = tl.load(SEQ_LEN_PTR + b)
        pos_offsets = tl.arange(0, PAGE_SIZE) + p * PAGE_SIZE
        valid = pos_offsets < seq_len
        sim = tl.where(valid, sim, tl.cast(float("-inf"), tl.bfloat16))

        new_max = tl.max(sim)
        att_probs = (tl.exp(sim - new_max) * 0.5).to(tl.bfloat16)

        alpha = (tl.exp(att_max - new_max) * 0.5).to(tl.bfloat16)
        out = out * alpha + tl.where(
            tl.static_range(PAGE_SIZE) < 1,
            tl.zeros([PAGE_SIZE], dtype=tl.bfloat16),
            tl.zeros([1], dtype=tl.bfloat16)
        )
        out = (out * alpha).to(tl.bfloat16)
        out += att_probs.reshape((PAGE_SIZE,)) * tl.load(
            KV_CACHE + block_idx * stride_kvb
            + tl.arange(0, PAGE_SIZE) * stride_kvps
            + kv_head * stride_kvkh
            + (HEAD_DIM + off_d) * stride_kvd
        )
        out = tl.where(valid.reshape((PAGE_SIZE,)), out, tl.zeros([1], dtype=tl.bfloat16))

        att_max = new_max
        att_sum = att_sum * alpha + tl.sum(att_probs)

    out = out / att_sum
    tl.store(OUT + b * stride_qb + h * stride_qh + off_d * stride_qd,
             out.to(tl.bfloat16))

+"""Paged attention decode kernel for RTX PRO 6000 Blackwell (sm_120).
+
+Triton-based single-query decode: each (batch, head) pair is one thread block.
+Loops over sequence positions, gathering K/V from the paged cache via the
+block_table on each iteration to keep SM register pressure bounded.
+
+Online softmax is used to handle variable-length sequences with a single pass.
+
+Layout:
+  query:       (batch, num_heads, head_dim)              bf16
+  kv_cache:    (num_blocks, page_size, num_kv_heads, 2*D) bf16  [K|V]
+  block_table: (batch, max_blocks_per_seq)               int32
+  seq_lens:    (batch,)                                   int32
+  output:      (batch, num_heads, head_dim)              bf16
+"""
+from __future__ import annotations
+
+import math
+from typing import List
+
+import torch
+import torch.nn as nn
+import triton
+import triton.language as tl
+
+# ---------------------------------------------------------------------------
+# Triton kernel
+# ---------------------------------------------------------------------------
+
+@triton.jit
+def _paged_attention_decode_kernel(
+    # Pointers
+    Q,                  # (batch, num_heads, head_dim)  bf16
+    KV_CACHE,           # (num_blocks, page_size, num_kv_heads, 2*head_dim) bf16
+    BLOCK_TABLE,        # (batch, max_blocks)           int32
+    SEQ_LEN_PTR,        # (batch,)                      int32
+    OUT,                # (batch, num_heads, head_dim)  bf16
+    # Query strides
+    stride_qb: tl.constexpr,
+    stride_qh: tl.constexpr,
+    stride_qd: tl.constexpr,
+    # KV cache strides
+    stride_kvb,        # stride for num_blocks axis
+    stride_kvps,       # stride for page_size axis
+    stride_kvkh,       # stride for num_kv_heads axis
+    stride_kvd: tl.constexpr,  # stride for last axis
+    # Block table strides
+    stride_bt: tl.constexpr,
+    # Shape parameters (runtime)
+    head_dim: tl.constexpr,
+    num_kv_heads: tl.constexpr,
+    page_size: tl.constexpr,
+    kv_cache_num_blocks,  # num_blocks (runtime scalar)
+    kv_cache_shape_ps,    # page_size (runtime scalar from shape)
+    kv_cache_shape_kh,    # num_kv_heads (runtime scalar from shape)
+    kv_cache_shape_d2,    # 2*head_dim (runtime scalar from shape)
+):
+    """Single-query paged-attention decode.
+
+    Each program_id(0) = batch element, program_id(1) = query head.
+    We loop over sequence positions in the KV cache, gather from the
+    paged block-table layout, and accumulate attention output with
+    online softmax.
+    """
+    b_idx = tl.program_id(0)
+    h_idx = tl.program_id(1)
+
+    # GQA: map query head -> kv head
+    num_query_heads = tl.num_programs(1)
+    group_size = num_query_heads // num_kv_heads
+    kv_head = h_idx // group_size
+
+    head_dim_f = float(head_dim)
+
+    # Load seq_len for this batch element
+    seq_len = tl.load(SEQ_LEN_PTR + b_idx)
+
+    # Compute max_num_pages for this batch element
+    max_num_pages = tl.cdiv(seq_len, page_size)
+
+    # ---- Load Q once ----
+    head_offsets = tl.arange(0, head_dim)
+    q = tl.load(
+        Q + b_idx * stride_qb + h_idx * stride_qh + head_offsets * stride_qd,
+    )
+
+    # ---- Online softmax accumulators ----
+    att_max = float("-inf")
+    att_sum: tl.float32 = 1.0
+    out_accum = tl.zeros([head_dim], dtype=tl.float32)
+
+    # ---- Loop over sequence positions ----
+    # For each position, compute the page index and intra-page offset,
+    # then gather K and V from the paged cache.
+    pos = 0
+    while pos < seq_len:
+        # Page index and intra-page offset for this position
+        page_idx = pos // page_size
+        page_off = pos % page_size
+
+        # Look up the block index from block_table[b, page_idx]
+        block_idx = tl.load(BLOCK_TABLE + b_idx * stride_bt + page_idx)
+
+        # ---- Load K from kv_cache[block_idx, page_off, kv_head, :] ----
+        k = tl.load(
+            KV_CACHE
+            + block_idx * stride_kvb
+            + page_off * stride_kvps
+            + kv_head * stride_kvkh
+            + head_offsets * stride_kvd
+        )
+
+        # ---- Compute Q·K / sqrt(D) (fp32) ----
+        sim = tl.sum(q * k) / tl.sqrt(head_dim_f)
+
+        # Mask: positions >= seq_len are already excluded by the loop bound
+
+        # ---- Online softmax update ----
+        new_max: tl.float32 = tl.maximum(att_max, sim)
+        alpha = tl.exp(att_max - new_max)
+        beta = tl.exp(sim - new_max)
+
+        out_accum = out_accum * alpha + beta * tl.to(v, tl.float32)
+        out_accum = out_accum * alpha + beta * tl.to(tl.load(
+            KV_CACHE
+            + block_idx * stride_kvb
+            + page_off * stride_kvps
+            + kv_head * stride_kvkh
+            + head_dim * stride_kvd
+            + head_offsets * stride_kvd
+        ), tl.float32)
+
+        att_max = new_max
+        att_sum = att_sum * alpha + beta
+
+        pos += 1
+
+    # Final normalization
+    out_final = out_accum / att_sum
+
+    # Store result (cast to bf16)
+    tl.store(
+        OUT + b_idx * stride_qb + h_idx * stride_qh + head_offsets * stride_qd,
+        out_final.to(tl.bfloat16),
+    )
