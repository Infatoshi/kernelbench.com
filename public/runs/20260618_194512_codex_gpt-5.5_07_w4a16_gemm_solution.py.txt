from __future__ import annotations

import os

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["B200"]

GROUP_SIZE = 128

_GEMV_EXT = None
_GEMV_EXT_FAILED = False


def _get_gemv_ext():
    global _GEMV_EXT, _GEMV_EXT_FAILED
    if _GEMV_EXT is not None or _GEMV_EXT_FAILED:
        return _GEMV_EXT

    try:
        os.environ.setdefault("TORCH_CUDA_ARCH_LIST", "10.0")
        from torch.utils.cpp_extension import load_inline

        cpp_src = """
void gemv8(torch::Tensor x, torch::Tensor wq, torch::Tensor scales,
           torch::Tensor zeros, torch::Tensor out, int N, int K);
void gemv16(torch::Tensor x, torch::Tensor wq, torch::Tensor scales,
            torch::Tensor zeros, torch::Tensor out, int N, int K);
"""
        cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_bf16.h>
#include <cuda_runtime.h>

template<int NTILE, int KPARTS>
__global__ __launch_bounds__(256, 2)
void gemv_kernel(const __nv_bfloat16* __restrict__ x,
                 const unsigned char* __restrict__ wq,
                 const __nv_bfloat16* __restrict__ scales,
                 const __nv_bfloat16* __restrict__ zeros,
                 __nv_bfloat16* __restrict__ out,
                 int N, int K) {
    int tid = threadIdx.x;
    int n_local = tid % NTILE;
    int kpart = tid / NTILE;
    int n = blockIdx.x * NTILE + n_local;
    float acc = 0.0f;
    constexpr int GROUP = 128;
    int groups = K / GROUP;

    for (int g = 0; g < groups; ++g) {
        float s = 0.0f;
        float z = 0.0f;
        if (n < N) {
            s = __bfloat162float(scales[g * N + n]);
            z = __bfloat162float(zeros[g * N + n]);
        }

        #pragma unroll
        for (int p = kpart; p < 64; p += KPARTS) {
            float xe = __bfloat162float(x[g * GROUP + 2 * p]);
            float xo = __bfloat162float(x[g * GROUP + 2 * p + 1]);
            unsigned char b = 0;
            if (n < N) {
                b = wq[(g * 64 + p) * N + n];
            }
            float lo = float(b & 0x0f);
            float hi = float((b >> 4) & 0x0f);
            acc = fmaf((lo - z) * s, xe, acc);
            acc = fmaf((hi - z) * s, xo, acc);
        }
    }

    extern __shared__ float smem[];
    smem[kpart * NTILE + n_local] = acc;
    __syncthreads();

    if (tid < NTILE) {
        int nn = blockIdx.x * NTILE + tid;
        if (nn < N) {
            float sum = 0.0f;
            #pragma unroll
            for (int kp = 0; kp < KPARTS; ++kp) {
                sum += smem[kp * NTILE + tid];
            }
            out[nn] = __float2bfloat16(sum);
        }
    }
}

void gemv8(torch::Tensor x, torch::Tensor wq, torch::Tensor scales,
           torch::Tensor zeros, torch::Tensor out, int N, int K) {
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    gemv_kernel<8, 32><<<(N + 7) / 8, 256, 8 * 32 * sizeof(float), stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        wq.data_ptr<unsigned char>(),
        reinterpret_cast<const __nv_bfloat16*>(scales.data_ptr()),
        reinterpret_cast<const __nv_bfloat16*>(zeros.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(out.data_ptr()),
        N, K);
}

void gemv16(torch::Tensor x, torch::Tensor wq, torch::Tensor scales,
            torch::Tensor zeros, torch::Tensor out, int N, int K) {
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();
    gemv_kernel<16, 16><<<(N + 15) / 16, 256, 16 * 16 * sizeof(float), stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        wq.data_ptr<unsigned char>(),
        reinterpret_cast<const __nv_bfloat16*>(scales.data_ptr()),
        reinterpret_cast<const __nv_bfloat16*>(zeros.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(out.data_ptr()),
        N, K);
}
"""
        _GEMV_EXT = load_inline(
            name="w4a16_decode_gemv_ext_v2",
            cpp_sources=cpp_src,
            cuda_sources=cuda_src,
            functions=["gemv8", "gemv16"],
            extra_cflags=["-O3"],
            extra_cuda_cflags=["-O3", "--use_fast_math"],
            verbose=False,
        )
    except Exception:
        _GEMV_EXT_FAILED = True
        _GEMV_EXT = None
    return _GEMV_EXT


@triton.jit
def _w4a16_gemm_kernel(
    x_ptr,
    wq_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K_PACKED: tl.constexpr,
    GROUP: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_p = tl.arange(0, BLOCK_K_PACKED)

    acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)

    for g in tl.range(0, K // GROUP):
        k_base = g * GROUP
        packed_rows = g * (GROUP // 2) + offs_p

        x_even = tl.load(
            x_ptr + offs_m[:, None] * K + (k_base + 2 * offs_p)[None, :],
            mask=offs_m[:, None] < M,
            other=0.0,
        )
        x_odd = tl.load(
            x_ptr + offs_m[:, None] * K + (k_base + 2 * offs_p + 1)[None, :],
            mask=offs_m[:, None] < M,
            other=0.0,
        )

        w_packed = tl.load(
            wq_ptr + packed_rows[:, None] * N + offs_n[None, :],
            mask=offs_n[None, :] < N,
            other=0,
        )
        scale = tl.load(scales_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        zero = tl.load(zeros_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)

        w_lo_i = w_packed & 0x0F
        w_hi_i = (w_packed >> 4) & 0x0F

        w_lo = ((w_lo_i.to(tl.float32) - zero[None, :]) * scale[None, :]).to(tl.bfloat16)
        w_hi = ((w_hi_i.to(tl.float32) - zero[None, :]) * scale[None, :]).to(tl.bfloat16)

        acc = tl.dot(x_even, w_lo, acc, out_dtype=tl.float32)
        acc = tl.dot(x_odd, w_hi, acc, out_dtype=tl.float32)

    tl.store(
        out_ptr + offs_m[:, None] * N + offs_n[None, :],
        acc.to(tl.bfloat16),
        mask=(offs_m[:, None] < M) & (offs_n[None, :] < N),
    )


@triton.jit
def _w4a16_gemm_group_scaled_kernel(
    x_ptr,
    wq_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K_PACKED: tl.constexpr,
    GROUP: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_p = tl.arange(0, BLOCK_K_PACKED)

    acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)

    for g in tl.range(0, K // GROUP):
        k_base = g * GROUP
        packed_rows = g * (GROUP // 2) + offs_p

        x_even = tl.load(
            x_ptr + offs_m[:, None] * K + (k_base + 2 * offs_p)[None, :],
            mask=offs_m[:, None] < M,
            other=0.0,
        )
        x_odd = tl.load(
            x_ptr + offs_m[:, None] * K + (k_base + 2 * offs_p + 1)[None, :],
            mask=offs_m[:, None] < M,
            other=0.0,
        )

        w_packed = tl.load(
            wq_ptr + packed_rows[:, None] * N + offs_n[None, :],
            mask=offs_n[None, :] < N,
            other=0,
        )

        q_lo = (w_packed & 0x0F).to(tl.bfloat16)
        q_hi = ((w_packed >> 4) & 0x0F).to(tl.bfloat16)

        qacc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)
        qacc = tl.dot(x_even, q_lo, qacc, out_dtype=tl.float32)
        qacc = tl.dot(x_odd, q_hi, qacc, out_dtype=tl.float32)

        x_sum = tl.sum(x_even.to(tl.float32), axis=1) + tl.sum(x_odd.to(tl.float32), axis=1)
        scale = tl.load(scales_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        zero = tl.load(zeros_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        acc += (qacc - x_sum[:, None] * zero[None, :]) * scale[None, :]

    tl.store(
        out_ptr + offs_m[:, None] * N + offs_n[None, :],
        acc.to(tl.bfloat16),
        mask=(offs_m[:, None] < M) & (offs_n[None, :] < N),
    )


@triton.jit
def _w4a16_gemm_dup128_kernel(
    x_ptr,
    wq_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    GROUP: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)

    for g in tl.range(0, K // GROUP):
        k_base = g * GROUP
        x_vals = tl.load(
            x_ptr + offs_m[:, None] * K + (k_base + offs_k)[None, :],
            mask=offs_m[:, None] < M,
            other=0.0,
        )

        packed_rows = g * (GROUP // 2) + (offs_k // 2)
        w_packed = tl.load(
            wq_ptr + packed_rows[:, None] * N + offs_n[None, :],
            mask=offs_n[None, :] < N,
            other=0,
        )
        w_lo = w_packed & 0x0F
        w_hi = (w_packed >> 4) & 0x0F
        use_hi = (offs_k & 1)[:, None] != 0
        w_i = tl.where(use_hi, w_hi, w_lo)

        scale = tl.load(scales_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        zero = tl.load(zeros_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        w_vals = ((w_i.to(tl.float32) - zero[None, :]) * scale[None, :]).to(tl.bfloat16)

        acc = tl.dot(x_vals, w_vals, acc, out_dtype=tl.float32)

    tl.store(
        out_ptr + offs_m[:, None] * N + offs_n[None, :],
        acc.to(tl.bfloat16),
        mask=(offs_m[:, None] < M) & (offs_n[None, :] < N),
    )


@triton.jit
def _w4a16_gemv_kernel(
    x_ptr,
    wq_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K_PACKED: tl.constexpr,
    GROUP: tl.constexpr,
):
    pid_n = tl.program_id(0)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_p = tl.arange(0, BLOCK_K_PACKED)

    acc = tl.zeros((BLOCK_N,), tl.float32)

    for g in tl.range(0, K // GROUP):
        k_base = g * GROUP
        packed_rows = g * (GROUP // 2) + offs_p

        x_even = tl.load(x_ptr + k_base + 2 * offs_p).to(tl.float32)
        x_odd = tl.load(x_ptr + k_base + 2 * offs_p + 1).to(tl.float32)

        w_packed = tl.load(
            wq_ptr + packed_rows[:, None] * N + offs_n[None, :],
            mask=offs_n[None, :] < N,
            other=0,
        )
        scale = tl.load(scales_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
        zero = tl.load(zeros_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)

        w_lo = (w_packed & 0x0F).to(tl.float32)
        w_hi = ((w_packed >> 4) & 0x0F).to(tl.float32)
        vals = ((w_lo - zero[None, :]) * x_even[:, None] + (w_hi - zero[None, :]) * x_odd[:, None])
        acc += tl.sum(vals, axis=0) * scale

    tl.store(out_ptr + offs_n, acc.to(tl.bfloat16), mask=offs_n < N)


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert group_size == GROUP_SIZE
        assert K % GROUP_SIZE == 0
        assert K % 2 == 0
        self.M = M
        self.N = N
        self.K = K
        self.group_size = group_size

        self.register_buffer("w_q", torch.empty((K // 2, N), dtype=torch.uint8))
        self.register_buffer("scales", torch.empty((K // group_size, N), dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.empty((K // group_size, N), dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        M = self.M
        N = self.N
        K = self.K
        out = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)

        if M == 1:
            ext = _get_gemv_ext()
            if ext is not None:
                if N <= 4096:
                    ext.gemv8(x, self.w_q, self.scales, self.zeros, out, N, K)
                else:
                    ext.gemv16(x, self.w_q, self.scales, self.zeros, out, N, K)
                return out

            if N > 4096:
                block_m = 8
                block_n = 64
                grid = (triton.cdiv(N, block_n), triton.cdiv(M, block_m))
                _w4a16_gemm_kernel[grid](
                    x,
                    self.w_q,
                    self.scales,
                    self.zeros,
                    out,
                    M,
                    N,
                    K,
                    BLOCK_M=block_m,
                    BLOCK_N=block_n,
                    BLOCK_K_PACKED=GROUP_SIZE // 2,
                    GROUP=GROUP_SIZE,
                    num_warps=4,
                    num_stages=3,
                )
                return out
            block_m = 16
            block_n = 64
            warps = 4
            stages = 3
        elif M <= 16:
            block_m = 16
            block_n = 64
            warps = 4
            stages = 3
        elif M <= 32:
            block_m = 32
            block_n = 32
            warps = 4
            stages = 3
        else:
            block_m = 128
            block_n = 32
            warps = 4
            stages = 3

        grid = (triton.cdiv(N, block_n), triton.cdiv(M, block_m))
        _w4a16_gemm_dup128_kernel[grid](
            x,
            self.w_q,
            self.scales,
            self.zeros,
            out,
            M,
            N,
            K,
            BLOCK_M=block_m,
            BLOCK_N=block_n,
            BLOCK_K=GROUP_SIZE,
            GROUP=GROUP_SIZE,
            num_warps=warps,
            num_stages=stages,
        )
        return out


M = 1
N = 12288
K = 4096


def get_inputs():
    x = torch.randn(M, K, dtype=torch.bfloat16)
    return [x]


def get_init_inputs():
    return [M, N, K]
