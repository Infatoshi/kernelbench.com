+def _grouped_swiglu(
+    A, Wg, Wu, Out, offsets,
+    H, I, E, Tperm,

+    pid_n = tl.program_id(1)
+
+    # Find which expert this m-tile belongs to (BM-padded grouping).
+    found = -1
+    cum = 0
+    row_start = 0
+    e_end = 0
+    for ee in range(0, E):
+        off0 = tl.load(offsets + ee)
+        off1 = tl.load(offsets + ee + 1)
+        nt = (off1 - off0 + BM - 1) // BM
+        take = (found < 0) & (pid_m < cum + nt)
+        found = tl.where(take, ee, found)
+        local = pid_m - cum

+        e_end = tl.where(take, off1, e_end)
+        cum += nt

+        return

+    offs_n = pid_n * BN + tl.arange(0, BN)
