Phase 1: grid = (batch, num_segs) — parallel bitonic sort per segment
Phase 2: grid = (batch,) — parallel bitonic sort of merged candidates

import torch.nn as nn

#include <cuda_fp16.h>
#include <cstdint>
#include <algorithm>

static constexpr float NEG_INF = -1e30f;

// =========================================================================
// Phase 1: segment-level bitonic sort
// =========================================================================
// Bitonic sort kernel:  sorts N elements (power-of-2) in shared memory
// =========================================================================
// =========================================================================
// Phase 1: segment-level bitonic sort
// =========================================================================
template <int SEG_N, int BLOCK1, int ELEM1, int K_OUT>
__global__ void seg_bitonic(
    const float* __restrict__ x,
    float*       __restrict__ seg_vals,
    int*         __restrict__ seg_idxs,
    int64_t row_stride,
    int num_segs,
    int n
)
{
    int bid_row = blockIdx.x;
    int bid_seg = blockIdx.y;
    int tid     = threadIdx.x;
    int seg_off = bid_seg * SEG_N;

    __shared__ float sv[SEG_N];
    __shared__ int   si[SEG_N];

    // Coalesced load
    #pragma unroll
    for (int e = 0; e < ELEM1; ++e) {
        int g = seg_off + tid * ELEM1 + e;
        int s = tid * ELEM1 + e;
        if (g < n) {
            sv[s] = x[(int64_t)bid_row * row_stride + g];
            si[s] = g;
        } else {
            sv[s] = NEG_INF;
            si[s] = -1;
        }
    }
    __syncthreads();

    // Bitonic sort (descending)
    for (int stage = 2; stage <= SEG_N; stage <<= 1) {
        for (int d = stage >> 1; d > 0; d >>= 1) {
            #pragma unroll
            for (int e = 0; e < ELEM1; ++e) {
                int mi = tid * ELEM1 + e;
                int jx = mi ^ d;
                if (jx > mi && ((mi & stage) != 0) != ((jx & stage) != 0)) {
                    bool dir_up = (mi & (stage >> 1)) == 0;
                    bool swap_ = (dir_up && sv[mi] < sv[jx]) || (!dir_up && sv[mi] > sv[jx]);
                    if (swap_) {
                        float tv = sv[mi]; sv[mi] = sv[jx]; sv[jx] = tv;
                        int   ti2= si[mi]; si[mi] = si[jx]; si[jx]   = ti2;
                    }
                }
            }
            __syncthreads();
        }
    }

    // Output top K_OUT
    int ob = (bid_row * num_segs + bid_seg) * K_OUT;
    #pragma unroll
    for (int ki = 0; ki < K_OUT; ++ki) {
        if (tid == 0) {
            seg_vals[ob + ki] = sv[ki];
            seg_idxs[ob + ki] = si[ki];
        }
    }
}

// =========================================================================
// Phase 2: merge all segment results via bitonic sort
// =========================================================================
template <int N_CAND, int B2, int E2>
__global__ void merge_bitonic(
    const float* __restrict__ seg_vals,
    const int*   __restrict__ seg_idxs,
    float*       __restrict__ values,
    int64_t*     __restrict__ indices,
    int k_out
)
{
    int bid = blockIdx.x;
    int tid = threadIdx.x;

    __shared__ float sv[N_CAND];
    __shared__ int   si[N_CAND];

    // Load candidates (bid row)
    int rb = bid * N_CAND;
    #pragma unroll
    for (int e = 0; e < E2; ++e) {
        int g = tid * E2 + e;
        sv[g] = seg_vals[rb + g];
        si[g] = seg_idxs[rb + g];
    }
    __syncthreads();

    // Bitonic sort (descending)
    for (int stage = 2; stage <= N_CAND; stage <<= 1) {
        for (int d = stage >> 1; d > 0; d >>= 1) {
            #pragma unroll
            for (int e = 0; e < E2; ++e) {
                int mi = tid * E2 + e;
                int jx = mi ^ d;
                if (jx > mi && ((mi & stage) != 0) != ((jx & stage) != 0)) {
                    bool dir_up = (mi & (stage >> 1)) == 0;
                    bool swap_ = (dir_up && sv[mi] < sv[jx]) || (!dir_up && sv[mi] > sv[jx]);
                    if (swap_) {
                        float tv = sv[mi]; sv[mi] = sv[jx]; sv[jx] = tv;
                        int   ti2= si[mi]; si[mi] = si[jx]; si[jx]   = ti2;
                    }
                }
            }
            __syncthreads();
        }
    }

    // Write top k_out
    if (tid < k_out) {
        values[(int64_t)bid * k_out + tid]  = sv[tid];
        indices[(int64_t)bid * k_out + tid] = static_cast<int64_t>(si[tid]);
    }
}

// Dispatcher
// =========================================================================
#include <torch/extension.h>

static int next_pow2(int v) {
    int p = 1;
    while (p < v) p <<= 1;
    return p;
}

// Phase 2 dispatcher: generate merge_bitonic instantiations
template <int N_CAND>
void dispatch_p2(int batch, int k_out,
    float* seg_v, int* seg_i, float* out_v, int64_t* out_i)
{
    constexpr int B2 = (N_CAND >= 256) ? 256 : N_CAND;
    constexpr int E2 = N_CAND / B2;
    merge_bitonic<N_CAND, B2, E2><<<batch, B2>>>(seg_v, seg_i, out_v, out_i, k_out);
}

    dispatch_p2<NC>(batch, k_out, seg_vals.data_ptr<float>(), seg_idxs.data_ptr<int>(), \
                    values.data_ptr<float>(), indices.data_ptr<int64_t>())

template <int K>
void do_p1(int batch, int num_segs, dim3 grid, int block,
    float* x, float* sv, int* si, int64_t rs, int n)
{

    constexpr int B1  = 256;
    constexpr int E1  = SN / B1;   // =8
    seg_bitonic<SN, B1, E1, K><<<grid, B1>>>(x, sv, si, rs, num_segs, n);
}

    int batch = x.size(0);
