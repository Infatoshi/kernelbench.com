GROUP_SIZE = 128

@triton.jit
def gemm_w4a16_kernel(
    a_ptr, b_ptr, c_ptr,
    scales_ptr, zeros_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    stride_scales_g, stride_scales_n,
    stride_zeros_g, stride_zeros_n,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_bn = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    a_ptrs = a_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    
    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_start = k * BLOCK_K
        
        a_mask = (offs_am[:, None] < M) & ((k_start + offs_k[None, :]) < K)
        a_val = tl.load(a_ptrs, mask=a_mask, other=0.0)

        group_idx = k_start // 128

        scales_ptrs = scales_ptr + group_idx * stride_scales_g + offs_bn * stride_scales_n
        zeros_ptrs = zeros_ptr + group_idx * stride_zeros_g + offs_bn * stride_zeros_n
        
        scales_mask = offs_bn < N
        scales_val = tl.load(scales_ptrs, mask=scales_mask, other=0.0)
        zeros_val = tl.load(zeros_ptrs, mask=scales_mask, other=0.0)

        k_packed_idx = (k_start + offs_k) // 2
        b_ptrs = b_ptr + k_packed_idx[:, None] * stride_bk + offs_bn[None, :] * stride_bn
        
        b_mask = (k_packed_idx[:, None] < (K // 2)) & (offs_bn[None, :] < N)
        b_packed = tl.load(b_ptrs, mask=b_mask, other=0)

        shifter = ((k_start + offs_k)[:, None] % 2) * 4
        b_unpacked = (b_packed >> shifter) & 0xF

        b_bf = (b_unpacked.to(tl.bfloat16) - zeros_val[None, :]) * scales_val[None, :]

        accumulator = tl.dot(a_val.to(tl.bfloat16), b_bf.to(tl.bfloat16), accumulator)

        a_ptrs += BLOCK_K * stride_ak

    c_val = accumulator.to(tl.bfloat16)
    
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    c_ptrs = c_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, c_val, mask=c_mask)

    stride_bk, stride_bn,
    stride_cm, stride_cn,

    stride_zeros_g, stride_zeros_n,

    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)

    offs_bn = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N

    offs_k_even = tl.arange(0, BLOCK_K // 2) * 2
    offs_k_odd = tl.arange(0, BLOCK_K // 2) * 2 + 1

    a_ptrs_odd = a_ptr + offs_am[:, None] * stride_am + offs_k_odd[None, :] * stride_ak
