import torch
import torch.nn as nn
import triton
import triton.language as tl
import reference

GROUP_SIZE = 128

@triton.jit
def gemm_w4a16_kernel(
    a_ptr, b_ptr, c_ptr,
    scales_ptr, zeros_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_cm, stride_cn,
    stride_scales_g, stride_scales_n,
    stride_zeros_g, stride_zeros_n,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_bn = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    a_ptrs = a_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    
    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k in range(0, tl.cdiv(K, BLOCK_K)):
        k_start = k * BLOCK_K
        
        a_mask = (offs_am[:, None] < M) & ((k_start + offs_k[None, :]) < K)
        a_val = tl.load(a_ptrs, mask=a_mask, other=0.0)

        group_idx = k_start // 128

        scales_ptrs = scales_ptr + group_idx * stride_scales_g + offs_bn * stride_scales_n
        zeros_ptrs = zeros_ptr + group_idx * stride_zeros_g + offs_bn * stride_zeros_n
        
        scales_mask = offs_bn < N
        scales_val = tl.load(scales_ptrs, mask=scales_mask, other=0.0)
        zeros_val = tl.load(zeros_ptrs, mask=scales_mask, other=0.0)

        k_packed_idx = (k_start + offs_k) // 2
        b_ptrs = b_ptr + k_packed_idx[:, None] * stride_bk + offs_bn[None, :] * stride_bn
        
        b_mask = (k_packed_idx[:, None] < (K // 2)) & (offs_bn[None, :] < N)
        b_packed = tl.load(b_ptrs, mask=b_mask, other=0)

        shifter = ((k_start + offs_k)[:, None] % 2) * 4
        b_unpacked = (b_packed >> shifter) & 0xF

        b_bf = (b_unpacked.to(tl.bfloat16) - zeros_val[None, :]) * scales_val[None, :]

        accumulator = tl.dot(a_val.to(tl.bfloat16), b_bf.to(tl.bfloat16), accumulator)

        a_ptrs += BLOCK_K * stride_ak

    c_val = accumulator.to(tl.bfloat16)
    
    offs_cm = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_cn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    c_ptrs = c_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    c_mask = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, c_val, mask=c_mask)


def gemm_w4a16(x, w_q, scales, zeros, group_size=128):
    M, K = x.shape
    Kh, N = w_q.shape
    assert Kh * 2 == K, f"Dimension mismatch: {Kh}*2 != {K}"
    
    out = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)
    
    stride_am, stride_ak = x.stride()
    stride_bk, stride_bn = w_q.stride()
    stride_cm, stride_cn = out.stride()
    
    stride_scales_g, stride_scales_n = scales.stride()
    stride_zeros_g, stride_zeros_n = zeros.stride()
    
    # Check shape to use tuned configurations
    if M == 1 and N == 12288 and K == 4096:
        BLOCK_M = 16
        BLOCK_N = 128
        BLOCK_K = 64
        num_warps = 8
        GROUP_SIZE_M = 4
    elif M == 32 and N == 12288 and K == 4096:
        BLOCK_M = 32
        BLOCK_N = 64
        BLOCK_K = 64
        num_warps = 4
        GROUP_SIZE_M = 4
    elif M == 256 and N == 12288 and K == 4096:
        BLOCK_M = 128
        BLOCK_N = 64
        BLOCK_K = 32
        num_warps = 4
        GROUP_SIZE_M = 8
    elif M == 1 and N == 4096 and K == 4096:
        BLOCK_M = 16
        BLOCK_N = 64
        BLOCK_K = 128
        num_warps = 4
        GROUP_SIZE_M = 4
    elif M == 16 and N == 14336 and K == 4096:
        BLOCK_M = 16
        BLOCK_N = 64
        BLOCK_K = 128
        num_warps = 4
        GROUP_SIZE_M = 1
    else:
        # Fallback configuration
        if M <= 16:
            BLOCK_M = 16
            BLOCK_N = 64
            BLOCK_K = 64
            num_warps = 4
            GROUP_SIZE_M = 8
        elif M <= 64:
            BLOCK_M = 32
            BLOCK_N = 128
            BLOCK_K = 64
            num_warps = 4
            GROUP_SIZE_M = 8
        else:
            BLOCK_M = 64
            BLOCK_N = 128
            BLOCK_K = 64
            num_warps = 8
            GROUP_SIZE_M = 8
        
    grid = lambda meta: (
        triton.cdiv(M, meta['BLOCK_M']) * triton.cdiv(N, meta['BLOCK_N']),
    )
    
    gemm_w4a16_kernel[grid](
        x, w_q, out,
        scales, zeros,
        M, N, K,
        stride_am, stride_ak,
        stride_bk, stride_bn,
        stride_cm, stride_cn,
        stride_scales_g, stride_scales_n,
        stride_zeros_g, stride_zeros_n,
        BLOCK_M=BLOCK_M,
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
        num_warps=num_warps,
    )
    return out


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even (int4 packing)"
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size

        # Register the buffers with the exact same name
        self.register_buffer("w_q", torch.zeros((K // 2, N), dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros((n_groups, N), dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros((n_groups, N), dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return gemm_w4a16(x, self.w_q, self.scales, self.zeros, self.group_size)
