def _load_extension():
    cuda_src = r"""
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <torch/torch.h>

constexpr int MMA_M = 16;
constexpr int THREADS_X = 256;

__device__ __forceinline__ unsigned lane_id() {
    unsigned reg;
    asm volatile("mov.u32 %0, %laneid;" : "=r"(reg));
    return reg;
}

__device__ __forceinline__ void barrier_sync() {
    asm volatile("bar.sync 0, %0;" : : "r"(THREADS_X));
}

__global__
void patch_embed_kernel_impl(
    const __nv_bfloat16* __restrict__ x,
    const __nv_bfloat16* __restrict__ w,
    __nv_bfloat16* __restrict__ y,
    int B, int C, int T, int H, int W,
    int kT, int kH, int kW,
    int OT, int OH, int OW,
    int embed_dim,
    int K
) {
    const int bx = blockIdx.x;
    const int by = blockIdx.y;

    const int num_patches = OT * OH * OW;
    const int M = B * num_patches;
    const int N = embed_dim;

    const int m_idx = bx * MMA_M;
    const int n_idx = by * MMA_M;

    const int thread_idx = threadIdx.x;
    const int patch_dim = kT * kH * kW;

    int row = thread_idx / MMA_M;
    int col = thread_idx % MMA_M;

    int m_pos = m_idx + row;
    int n_pos = n_idx + col;

    if (m_pos >= M || n_pos >= N) return;

    float sum = 0.0f;

    for (int k = 0; k < K; ++k) {
        int c = k / patch_dim;
        int rem = k % patch_dim;
        int kt = rem / (kH * kW);
        rem = rem % (kH * kW);
        int kh = rem / kW;
        int kw = rem % kW;

        int x_b = (m_pos / num_patches);
        int patch_idx = m_pos % num_patches;
        int x_t = patch_idx / (OH * OW);
        int rem2 = patch_idx % (OH * OW);
        int x_h = rem2 / OW;
        int x_w = rem2 % OW;

        int x_t_idx = x_t * kT + kt;
        int x_h_idx = x_h * kH + kh;
        int x_w_idx = x_w * kW + kw;

        int x_linear = ((x_b * C + c) * T + x_t_idx) * H + x_h_idx;
        x_linear = x_linear * W + x_w_idx;

        __nv_bfloat16 x_val = __ldg(&x[x_linear]);
        float x_f = __bfloat162float(x_val);

        int w_linear = n_pos * K + k;
        __nv_bfloat16 w_val = __ldg(&w[w_linear]);
        float w_f = __bfloat162float(w_val);

        sum += x_f * w_f;
    }

    int b = m_pos / num_patches;
    int patch_idx = m_pos % num_patches;
    int t_idx = patch_idx / (OH * OW);
    int rem2 = patch_idx % (OH * OW);
    int h_idx = rem2 / OW;
    int w_idx = rem2 % OW;

    int y_linear = ((b * embed_dim + n_pos) * OT + t_idx) * OH + h_idx;
    y_linear = y_linear * OW + w_idx;

    y[y_linear] = __float2bfloat16(sum);
}

torch::Tensor patch_embed_cuda(torch::Tensor x, torch::Tensor weight) {
    auto B = x.size(0);
    auto C = x.size(1);
    auto T = x.size(2);
    auto H = x.size(3);
    auto W = x.size(4);
    auto kT = weight.size(2);
    auto kH = weight.size(3);
    auto kW = weight.size(4);
    auto embed_dim = weight.size(0);

    auto OT = T / kT;
    auto OH = H / kH;
    auto OW = W / kW;
    auto K = C * kT * kH * kW;

    auto num_patches = OT * OH * OW;
    auto M = B * num_patches;
    auto N = embed_dim;

    auto y = torch::empty({B, embed_dim, OT, OH, OW}, x.options());

    const int threads_x = 256;
    int blocks_x = (M + MMA_M - 1) / MMA_M;
    int blocks_y = (N + MMA_M - 1) / MMA_M;

    patch_embed_kernel_impl<<<dim3(blocks_x, blocks_y, 1), dim3(threads_x, 1, 1)>>>(
        (const __nv_bfloat16*)x.data_ptr(),
        (const __nv_bfloat16*)weight.data_ptr(),
        (__nv_bfloat16*)y.data_ptr(),
        B, C, T, H, W,
        kT, kH, kW,
        OT, OH, OW,
        embed_dim,
        K
    );

    return y;
}

class Model(torch.nn.Module):
    def __init__(self, B: int, C: int, T: int, H: int, W: int,

            C, embed_dim,
