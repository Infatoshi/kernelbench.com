+def _paged_decode_kernel(
+    Q_ptr,
+    KV_ptr,
+    BT_ptr,
+    SL_ptr,
+    O_ptr,
+    stride_qb,
+    stride_qh,

+    num_kv_heads,

+    BLOCK_T: tl.constexpr,

+    head_idx = tl.program_id(1)
