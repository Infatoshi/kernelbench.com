@triton.jit
def _w4a16_gemv_splitk_kernel(
    x_ptr,
    w_q_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    N,
    K,
    stride_xk,
    stride_wk,
    stride_wn,
    stride_sg,
    stride_sn,
    stride_zg,
    stride_zn,
    GROUP_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_k_blocks = tl.cdiv(K, BLOCK_K)
    pid_n = pid // num_k_blocks
    k_idx = pid % num_k_blocks

    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = offs_n < N

    k_start = k_idx * BLOCK_K
    HALF_K: tl.constexpr = BLOCK_K // 2

    k_even = k_start + tl.arange(0, HALF_K) * 2
    k_odd = k_start + tl.arange(0, HALF_K) * 2 + 1

    x_even = tl.load(
        x_ptr + k_even[None, :] * stride_xk,
        mask=k_even[None, :] < K,
        other=0.0,
    )
    x_odd = tl.load(
        x_ptr + k_odd[None, :] * stride_xk,
        mask=k_odd[None, :] < K,
        other=0.0,
    )

    packed_k = k_start // 2 + tl.arange(0, HALF_K)
    w_packed = tl.load(
        w_q_ptr + packed_k[:, None] * stride_wk + offs_n[None, :] * stride_wn,
        mask=(packed_k[:, None] < K // 2) & mask_n[None, :],
        other=0,
        eviction_policy="evict_last",
    )

    w_lo = (w_packed & 0xF).to(tl.float32)
    w_hi = (w_packed >> 4).to(tl.float32)

    g = k_start // GROUP_SIZE
    s = tl.load(
        scales_ptr + g * stride_sg + offs_n * stride_sn,
        mask=mask_n,
        other=1.0,
    ).to(tl.float32)
    z = tl.load(
        zeros_ptr + g * stride_zg + offs_n * stride_zn,
        mask=mask_n,
        other=0.0,
    ).to(tl.float32)

    w_lo_dq = ((w_lo - z[None, :]) * s[None, :]).to(tl.bfloat16)
    w_hi_dq = ((w_hi - z[None, :]) * s[None, :]).to(tl.bfloat16)

    partial = tl.dot(x_even, w_lo_dq) + tl.dot(x_odd, w_hi_dq)
    acc = partial.reshape(BLOCK_N)

    tl.atomic_add(out_ptr + offs_n, acc, mask=mask_n)

@triton.jit
def _w4a16_gemm_kernel(
    x_ptr,
    w_q_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M,
    N,
    K,
    stride_xm,
    stride_xk,
    stride_wk,
    stride_wn,
    stride_sg,
    stride_sn,
    stride_zg,
    stride_zn,
    stride_om,
    stride_on,
    GROUP_SIZE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_m = offs_m < M
    mask_n = offs_n < N

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    HALF_K: tl.constexpr = BLOCK_K // 2

    for k_start in range(0, K, BLOCK_K):
        k_even = k_start + tl.arange(0, HALF_K) * 2
        k_odd = k_start + tl.arange(0, HALF_K) * 2 + 1

        x_even = tl.load(
            x_ptr + offs_m[:, None] * stride_xm + k_even[None, :] * stride_xk,
            mask=mask_m[:, None] & (k_even[None, :] < K),
            other=0.0,
        )
        x_odd = tl.load(
            x_ptr + offs_m[:, None] * stride_xm + k_odd[None, :] * stride_xk,
            mask=mask_m[:, None] & (k_odd[None, :] < K),
            other=0.0,
        )

        packed_k = k_start // 2 + tl.arange(0, HALF_K)
        w_packed = tl.load(
            w_q_ptr + packed_k[:, None] * stride_wk + offs_n[None, :] * stride_wn,
            mask=(packed_k[:, None] < K // 2) & mask_n[None, :],
            other=0,
            eviction_policy="evict_last",

        w_hi = (w_packed >> 4).to(tl.float32)
