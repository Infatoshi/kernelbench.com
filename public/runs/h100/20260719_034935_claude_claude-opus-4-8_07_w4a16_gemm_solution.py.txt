"""W4A16 weight-only int4 quantized GEMM (AWQ/GPTQ-style asymmetric), fused.

Weights are repacked once (cached) into a (K//8, N) int32 layout so both paths
do wide coalesced loads and a clean nibble unpack.

  * M == 1 (decode): a hand-written CUDA GEMV. Each thread owns one output
    column; scale-folding keeps the inner loop to ~1 FMA / nibble
    (partial = sum x*nib; out = s*partial - s*z*xsum, xsum reduced in smem).
    Split-K for occupancy + a fused reduce/cast. Memory-bound (~0.5 of HBM).
  * M >= 8: a Triton fused-dequant tensor-core GEMM (split-K for the skinny
    memory-bound cases, single pass for the compute regime).

If the CUDA extension fails to build, everything transparently falls back to
the Triton path.

Scheme:
  x:(M,K) bf16  w_q:(K//2,N) uint8 (lo=even-K, hi=odd-K)
  scales,zeros:(K//128,N) bf16 ; out = x @ dequant(w_q,scales,zeros) : (M,N) bf16
  dequant: w[k,n] = (unpack(w_q)[k,n] - zeros[k//128,n]) * scales[k//128,n]
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

GROUP_SIZE = 128


# --------------------------------------------------------------------------- #
# Triton fused-dequant GEMM (int32-repacked weights)
# --------------------------------------------------------------------------- #
@triton.jit
def _w4a16_kernel(
    x_ptr, wq_ptr, s_ptr, z_ptr, out_ptr,
    M, N, K,
    sxm, sxk, swk, swn, ssg, ssn, sok, som, son,
    BM: tl.constexpr, BN: tl.constexpr, BK: tl.constexpr,
    SPLIT_K: tl.constexpr, WBF16: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)
    pid_k = tl.program_id(2)
    offs_m = pid_m * BM + tl.arange(0, BM)
    offs_n = pid_n * BN + tl.arange(0, BN)
    m_mask = offs_m < M
    KW: tl.constexpr = BK // 8
    kw = tl.arange(0, KW)
    num_ktiles = K // BK
    per_split = num_ktiles // SPLIT_K
    kt0 = pid_k * per_split

    wq_ptrs = wq_ptr + (kt0 * KW + kw)[:, None] * swk + offs_n[None, :] * swn
    kcol = tl.arange(0, BK)
    x_ptrs = x_ptr + offs_m[:, None] * sxm + (kt0 * BK + kcol)[None, :] * sxk
    shifts = tl.arange(0, 8) * 4

    acc = tl.zeros((BM, BN), dtype=tl.float32)
    for i in range(per_split):
        kt = kt0 + i
        g = (kt * BK) // 128
        b = tl.load(wq_ptrs)  # [KW, BN] int32
        w = (b[:, None, :] >> shifts[None, :, None]) & 0xF  # [KW,8,BN]
        w = tl.reshape(w, [BK, BN]).to(tl.bfloat16)
        s = tl.load(s_ptr + g * ssg + offs_n * ssn)
        z = tl.load(z_ptr + g * ssg + offs_n * ssn)
        wdeq = (w - z[None, :]) * s[None, :]
        xt = tl.load(x_ptrs, mask=m_mask[:, None], other=0.0)
        acc = tl.dot(xt, wdeq, acc)
        wq_ptrs += KW * swk
        x_ptrs += BK * sxk

    out_ptrs = out_ptr + pid_k * sok + offs_m[:, None] * som + offs_n[None, :] * son
    if WBF16:
        tl.store(out_ptrs, acc.to(tl.bfloat16), mask=m_mask[:, None])
    else:
        tl.store(out_ptrs, acc, mask=m_mask[:, None])


def _triton_run(x, w32, scales, zeros, N, K, cfg):
    M = x.shape[0]
    BM, BN, BK, SK, warps, stages = cfg
    grid = (triton.cdiv(N, BN), triton.cdiv(M, BM), SK)
    if SK == 1:
        out = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)
        buf = out.view(1, M, N)
        wbf = True
    else:
        buf = torch.empty((SK, M, N), dtype=torch.float32, device=x.device)
        wbf = False
    _w4a16_kernel[grid](
        x, w32, scales, zeros, buf, M, N, K,
        x.stride(0), x.stride(1), w32.stride(0), w32.stride(1),
        scales.stride(0), scales.stride(1), buf.stride(0), buf.stride(1), buf.stride(2),
        BM=BM, BN=BN, BK=BK, SPLIT_K=SK, WBF16=wbf,
        num_warps=warps, num_stages=stages,
    )
    if SK == 1:
        return out
    if _CUDA is not None:
        return _CUDA.reduce_smn(buf, SK, M, N)
    return buf.sum(0).to(torch.bfloat16)


def _triton_candidates(M, N, K):
    cfgs = []
    if M <= 8:
        for bn in (64, 128):
            for bk in (32, 64):
                for sk in (4, 8, 16):
                    if (K // bk) % sk == 0:
                        cfgs.append((16, bn, bk, sk, 4, 3))
                        cfgs.append((16, bn, bk, sk, 4, 4))
    elif M <= 64:
        bm = 16 if M <= 16 else 32
        for bn in (64, 128):
            for bk in (32, 64):
                for sk in (1, 4, 8):
                    if (K // bk) % sk == 0:
                        cfgs.append((bm, bn, bk, sk, 4, 3))
                        cfgs.append((bm, bn, bk, sk, 4, 4))
    else:
        for bm in (64, 128):
            for bn in (64, 128):
                for bk in (32, 64):
                    for warps in (4, 8):
                        cfgs.append((bm, bn, bk, 1, warps, 3))
    return cfgs


# --------------------------------------------------------------------------- #
# CUDA GEMV (M == 1)
# --------------------------------------------------------------------------- #
_CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>

template<int KSLICE>
__global__ void gemv2_kernel(
    const __nv_bfloat16* __restrict__ x, const int* __restrict__ w32,
    const __nv_bfloat16* __restrict__ scales, const __nv_bfloat16* __restrict__ zeros,
    float* __restrict__ buf, int N, int K)
{
    const int GS = (KSLICE * 8) / 128;
    const int krange = KSLICE * 8;
    const int kw0 = blockIdx.y * KSLICE;
    const int kbase = kw0 * 8;
    const int g0 = kbase / 128;
    extern __shared__ float sm[];
    float* xs = sm;
    float* xsum = sm + krange;
    for (int j = threadIdx.x; j < krange; j += blockDim.x)
        xs[j] = __bfloat162float(x[kbase + j]);
    __syncthreads();
    if (threadIdx.x < GS) {
        float t = 0.f;
        #pragma unroll
        for (int i = 0; i < 128; i++) t += xs[threadIdx.x * 128 + i];
        xsum[threadIdx.x] = t;
    }
    __syncthreads();
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (n >= N) return;
    float acc = 0.f;
    #pragma unroll
    for (int lg = 0; lg < GS; lg++) {
        int g = g0 + lg;
        float s = __bfloat162float(scales[g * N + n]);
        float z = __bfloat162float(zeros[g * N + n]);
        float partial = 0.f;
        #pragma unroll 4
        for (int w = 0; w < 16; w++) {
            unsigned int packed = (unsigned int) w32[(kw0 + lg * 16 + w) * N + n];
            int koff = lg * 128 + w * 8;
            #pragma unroll
            for (int i = 0; i < 8; i++)
                partial += xs[koff + i] * (float)((packed >> (4 * i)) & 0xF);
        }
        acc += s * partial - s * z * xsum[lg];
    }
    buf[blockIdx.y * N + n] = acc;
}

__global__ void reduce_kernel(const float* __restrict__ buf, __nv_bfloat16* __restrict__ out,
                              int N, int SK) {
    int n = blockIdx.x * blockDim.x + threadIdx.x;
    if (n >= N) return;
    float acc = 0.f;
    for (int k = 0; k < SK; k++) acc += buf[k * N + n];
    out[n] = __float2bfloat16(acc);
}

// [SK, M, N] fp32 partials -> [M, N] bf16.
__global__ void reduce_smn_kernel(const float* __restrict__ buf, __nv_bfloat16* __restrict__ out,
                                  long MN, int SK) {
    long idx = (long)blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= MN) return;
    float acc = 0.f;
    for (int k = 0; k < SK; k++) acc += buf[(long)k * MN + idx];
    out[idx] = __float2bfloat16(acc);
}

torch::Tensor reduce_smn(torch::Tensor buf, int SK, int M, int N) {
    auto out = torch::empty({M, N}, torch::dtype(torch::kBFloat16).device(buf.device()));
    long MN = (long)M * N;
    int tb = 256;
    reduce_smn_kernel<<<(MN + tb - 1) / tb, tb>>>(buf.data_ptr<float>(),
        (__nv_bfloat16*)out.data_ptr(), MN, SK);
    return out;
}

torch::Tensor gemv(torch::Tensor x, torch::Tensor w32, torch::Tensor scales,
                   torch::Tensor zeros, int N, int K, int block_n, int split_k) {
    auto buf = torch::empty({split_k, N}, torch::dtype(torch::kFloat32).device(x.device()));
    auto out = torch::empty({1, N}, torch::dtype(torch::kBFloat16).device(x.device()));
    dim3 grid((N + block_n - 1) / block_n, split_k);
    dim3 block(block_n);
    int kslice = (K / 8) / split_k;
    int krange = kslice * 8;
    size_t smem = (krange + krange / 128) * sizeof(float);
    auto xp = (const __nv_bfloat16*)x.data_ptr();
    auto wp = (const int*)w32.data_ptr();
    auto sp = (const __nv_bfloat16*)scales.data_ptr();
    auto zp = (const __nv_bfloat16*)zeros.data_ptr();
    auto bp = buf.data_ptr<float>();
    #define LK(KS) gemv2_kernel<KS><<<grid, block, smem>>>(xp, wp, sp, zp, bp, N, K)
    switch (kslice) {
        case 512: LK(512); break; case 256: LK(256); break; case 128: LK(128); break;
        case 64: LK(64); break; case 32: LK(32); break; case 16: LK(16); break;
        default: TORCH_CHECK(false, "bad kslice ", kslice);
    }
    int tb = 256;
    reduce_kernel<<<(N + tb - 1) / tb, tb>>>(bp, (__nv_bfloat16*)out.data_ptr(), N, split_k);
    return out;
}
"""
_CPP_SRC = ("torch::Tensor gemv(torch::Tensor x, torch::Tensor w32, torch::Tensor scales, "
            "torch::Tensor zeros, int N, int K, int block_n, int split_k);\n"
            "torch::Tensor reduce_smn(torch::Tensor buf, int SK, int M, int N);")

_CUDA = None
try:
    from torch.utils.cpp_extension import load_inline
    _CUDA = load_inline(
        name="w4a16_gemv_cuda", cpp_sources=_CPP_SRC, cuda_sources=_CUDA_SRC,
        functions=["gemv", "reduce_smn"], extra_cuda_cflags=["-O3", "--use_fast_math"], verbose=False,
    )
except Exception:
    _CUDA = None


def _cuda_gemv_candidates(N, K):
    cfgs = []
    for bn in (128, 256):
        for sk in (8, 16, 32):
            if (K // 8) % sk == 0 and ((K // 8) // sk) % 16 == 0:
                cfgs.append((bn, sk))
    return cfgs


# --------------------------------------------------------------------------- #
# Dispatch + per-shape autotune (runs once per (M,N,K), during warmup)
# --------------------------------------------------------------------------- #
# Best configs found by offline min-of-many sweeps (robust to shared-GPU
# contention). ("cuda", block_n, split_k) or ("triton", (BM,BN,BK,SK,warps,st)).
_HARDCODED: dict = {
    (1, 12288, 4096): ("cuda", 128, 16),
    (1, 4096, 4096): ("cuda", 128, 32),
    (16, 14336, 4096): ("triton", (16, 64, 64, 4, 4, 3)),
    (32, 12288, 4096): ("triton", (32, 128, 32, 8, 4, 3)),
    (256, 12288, 4096): ("triton", (256, 64, 64, 1, 4, 3)),
}

_CFG_CACHE: dict = {}

_l2_scratch = None


def _l2flush():
    global _l2_scratch
    if _l2_scratch is None:
        _l2_scratch = torch.empty(128 * 1024 * 1024 // 4, dtype=torch.float32, device="cuda")
    _l2_scratch.zero_()


def _time_cfg(fn, iters=12):
    # min over trials with an L2 flush between each -> robust to contention from
    # other jobs sharing the GPU and measures the HBM-load path.
    for _ in range(3):
        fn()
    torch.cuda.synchronize()
    best = 1e30
    for _ in range(iters):
        _l2flush()
        torch.cuda.synchronize()
        s = torch.cuda.Event(enable_timing=True)
        e = torch.cuda.Event(enable_timing=True)
        s.record()
        fn()
        e.record()
        torch.cuda.synchronize()
        best = min(best, s.elapsed_time(e))
    return best


def _autotune(x, w32, scales, zeros, N, K):
    M = x.shape[0]
    best, best_t = None, 1e30
    if M == 1 and _CUDA is not None:
        for bn, sk in _cuda_gemv_candidates(N, K):
            try:
                t = _time_cfg(lambda: _CUDA.gemv(x, w32, scales, zeros, N, K, bn, sk))
            except Exception:
                continue
            if t < best_t:
                best_t, best = t, ("cuda", bn, sk)
    for cfg in _triton_candidates(M, N, K):
        try:
            t = _time_cfg(lambda: _triton_run(x, w32, scales, zeros, N, K, cfg))
        except Exception:
            continue
        if t < best_t:
            best_t, best = t, ("triton", cfg)
    if best is None:
        best = ("triton", (16, 64, 64, 1, 4, 3))
    return best


def w4a16_gemm(x, w32, scales, zeros, N, K):
    M = x.shape[0]
    key = (M, N, K)
    plan = _CFG_CACHE.get(key)
    if plan is None:
        plan = _HARDCODED.get(key)
        if plan is not None and plan[0] == "cuda" and _CUDA is None:
            plan = None  # CUDA unavailable -> autotune the Triton path
        if plan is None:
            plan = _autotune(x, w32, scales, zeros, N, K)
        _CFG_CACHE[key] = plan
    if plan[0] == "cuda":
        return _CUDA.gemv(x, w32, scales, zeros, N, K, plan[1], plan[2])
    return _triton_run(x, w32, scales, zeros, N, K, plan[1])


def _repack(w_q, K, N):
    """(K//2,N) uint8 [lo=even,hi=odd] -> (K//8,N) int32, nibble i = K-value 8kw+i."""
    unp = torch.empty((K, N), dtype=torch.uint8, device=w_q.device)
    unp[0::2] = w_q & 0xF
    unp[1::2] = (w_q >> 4) & 0xF
    unp = unp.to(torch.int64).view(K // 8, 8, N)
    shifts = (torch.arange(8, device=w_q.device, dtype=torch.int64) * 4).view(1, 8, 1)
    return (unp << shifts).sum(1).to(torch.int32).contiguous()


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0 and K % 2 == 0
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size
        self.register_buffer("w_q", torch.zeros((K // 2, N), dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros((n_groups, N), dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros((n_groups, N), dtype=torch.bfloat16))
        self._w32 = None

    def _prep(self):
        if self._w32 is None:
            self._w32 = _repack(self.w_q, self.K, self.N)
        return self._w32

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w32 = self._prep()
        return w4a16_gemm(x, w32, self.scales, self.zeros, self.N, self.K)


M = 1
N = 12288

    x = torch.randn(M, K, dtype=torch.bfloat16)
    return [x]


def get_init_inputs():
    return [M, N, K]
