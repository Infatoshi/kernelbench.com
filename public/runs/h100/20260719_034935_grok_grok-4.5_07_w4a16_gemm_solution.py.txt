"""Fused W4A16 weight-only quantized GEMM (AWQ/GPTQ asymmetric int4).

Hybrid:
  * M==1: CUDA multi-warp GEMV (streams packed int4, SMEM reduce over K-splits)
  * M>1 : Triton group-strided GEMM (dequant + tl.dot tensor cores)

Falls back to Triton for M==1 if the CUDA extension fails to build.
Never materializes the full bf16 weight matrix.
"""
from __future__ import annotations

import os

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

GROUP_SIZE = 128

# Ensure nvcc is findable for load_inline even under the GPU-lock wrapper.
os.environ.setdefault("CUDA_HOME", "/usr/local/cuda-12.8")
os.environ.setdefault("TORCH_CUDA_ARCH_LIST", "9.0")
if not os.environ.get("REAL_NVCC"):
    for cand in (
        "/usr/local/cuda-12.8/bin/nvcc",
        "/usr/local/cuda/bin/nvcc",
        "/usr/local/cuda-12/bin/nvcc",
    ):
        if os.path.isfile(cand):
            os.environ["REAL_NVCC"] = cand
            break

# ---------------------------------------------------------------------------
# CUDA GEMV
# ---------------------------------------------------------------------------

CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_bf16.h>
#include <cstdint>

__device__ __forceinline__ float bf2f(__nv_bfloat16 v) { return __bfloat162float(v); }
__device__ __forceinline__ __nv_bfloat16 f2bf(float v) { return __float2bfloat16(v); }

// Multi-warp GEMV, 4 columns per thread (vectorized weight/scale loads).
// Threads per block = (COLS_PER_BLOCK/4) * N_WARPS.
// Each warp-row owns a slice of quant-groups; reduce in SMEM.
template <int COLS_PER_BLOCK, int N_WARPS>
__global__ void gemv_mw_v4(
    const __nv_bfloat16* __restrict__ x,
    const uint8_t* __restrict__ w_q,
    const __nv_bfloat16* __restrict__ scales,
    const __nv_bfloat16* __restrict__ zeros,
    __nv_bfloat16* __restrict__ y,
    int N, int K, int group_size)
{
  constexpr int THREADS_N = COLS_PER_BLOCK / 4;
  const int tid = threadIdx.x;
  const int t   = tid % THREADS_N;   // column-thread index
  const int w   = tid / THREADS_N;   // warp-row (K-split)
  const int n0  = blockIdx.x * COLS_PER_BLOCK + t * 4;
  const bool full = (n0 + 3) < N;
  const bool any  = n0 < N;

  const int n_groups = K / group_size;
  const int gps = (n_groups + N_WARPS - 1) / N_WARPS;
  const int g0 = w * gps;
  const int g1 = min(g0 + gps, n_groups);

  float a0 = 0.f, a1 = 0.f, a2 = 0.f, a3 = 0.f;
  if (any && g0 < n_groups) {
    for (int g = g0; g < g1; ++g) {
      float s0, s1, s2, s3, z0, z1, z2, z3;
      if (full) {
        uint64_t sb = *reinterpret_cast<const uint64_t*>(
            scales + (int64_t)g * N + n0);
        uint64_t zb = *reinterpret_cast<const uint64_t*>(
            zeros + (int64_t)g * N + n0);
        auto* sp = reinterpret_cast<const __nv_bfloat16*>(&sb);
        auto* zp = reinterpret_cast<const __nv_bfloat16*>(&zb);
        s0 = bf2f(sp[0]); s1 = bf2f(sp[1]); s2 = bf2f(sp[2]); s3 = bf2f(sp[3]);
        z0 = bf2f(zp[0]); z1 = bf2f(zp[1]); z2 = bf2f(zp[2]); z3 = bf2f(zp[3]);
      } else {
        s0 = s1 = s2 = s3 = z0 = z1 = z2 = z3 = 0.f;
        if (n0 + 0 < N) { s0 = bf2f(scales[(int64_t)g * N + n0]);     z0 = bf2f(zeros[(int64_t)g * N + n0]); }
        if (n0 + 1 < N) { s1 = bf2f(scales[(int64_t)g * N + n0 + 1]); z1 = bf2f(zeros[(int64_t)g * N + n0 + 1]); }
        if (n0 + 2 < N) { s2 = bf2f(scales[(int64_t)g * N + n0 + 2]); z2 = bf2f(zeros[(int64_t)g * N + n0 + 2]); }
        if (n0 + 3 < N) { s3 = bf2f(scales[(int64_t)g * N + n0 + 3]); z3 = bf2f(zeros[(int64_t)g * N + n0 + 3]); }
      }
      const int k0 = g * group_size;
      const int kh0 = k0 >> 1;
      #pragma unroll 8
      for (int tt = 0; tt < (group_size >> 1); ++tt) {
        const float x0 = bf2f(__ldg(x + k0 + (tt << 1)));
        const float x1 = bf2f(__ldg(x + k0 + (tt << 1) + 1));
        uint8_t p0, p1, p2, p3;
        if (full) {
          uint32_t pk = *reinterpret_cast<const uint32_t*>(
              w_q + (int64_t)(kh0 + tt) * N + n0);
          p0 = pk & 0xFF; p1 = (pk >> 8) & 0xFF;
          p2 = (pk >> 16) & 0xFF; p3 = (pk >> 24) & 0xFF;
        } else {
          p0 = (n0 + 0 < N) ? w_q[(int64_t)(kh0 + tt) * N + n0] : 0;
          p1 = (n0 + 1 < N) ? w_q[(int64_t)(kh0 + tt) * N + n0 + 1] : 0;
          p2 = (n0 + 2 < N) ? w_q[(int64_t)(kh0 + tt) * N + n0 + 2] : 0;
          p3 = (n0 + 3 < N) ? w_q[(int64_t)(kh0 + tt) * N + n0 + 3] : 0;
        }
        a0 = fmaf(x0, (float(p0 & 0xF) - z0) * s0, a0);
        a0 = fmaf(x1, (float(p0 >> 4)  - z0) * s0, a0);
        a1 = fmaf(x0, (float(p1 & 0xF) - z1) * s1, a1);
        a1 = fmaf(x1, (float(p1 >> 4)  - z1) * s1, a1);
        a2 = fmaf(x0, (float(p2 & 0xF) - z2) * s2, a2);
        a2 = fmaf(x1, (float(p2 >> 4)  - z2) * s2, a2);
        a3 = fmaf(x0, (float(p3 & 0xF) - z3) * s3, a3);
        a3 = fmaf(x1, (float(p3 >> 4)  - z3) * s3, a3);
      }
    }
  }

  // Reduce across N_WARPS for each of the 4 columns
  __shared__ float sm[N_WARPS][COLS_PER_BLOCK];
  sm[w][t * 4 + 0] = a0;
  sm[w][t * 4 + 1] = a1;
  sm[w][t * 4 + 2] = a2;
  sm[w][t * 4 + 3] = a3;
  __syncthreads();

  if (w == 0) {
    // Each of the THREADS_N threads writes 4 outputs
    #pragma unroll
    for (int c = 0; c < 4; ++c) {
      int nn = n0 + c;
      if (nn >= N) continue;
      float sum = 0.f;
      #pragma unroll
      for (int i = 0; i < N_WARPS; ++i) sum += sm[i][t * 4 + c];
      y[nn] = f2bf(sum);
    }
  }
}

// Split-K GEMV for small N (more CTAs)
template <int BLOCK_N>
__global__ void gemv_splitk(
    const __nv_bfloat16* __restrict__ x,
    const uint8_t* __restrict__ w_q,
    const __nv_bfloat16* __restrict__ scales,
    const __nv_bfloat16* __restrict__ zeros,
    float* __restrict__ partials,
    int N, int K, int group_size, int n_splits)
{
  const int n = blockIdx.x * BLOCK_N + threadIdx.x;
  const int split = blockIdx.y;
  if (n >= N) return;

  const int n_groups = K / group_size;
  const int gps = (n_groups + n_splits - 1) / n_splits;
  const int g0 = split * gps;
  const int g1 = min(g0 + gps, n_groups);
  float acc = 0.f;
  if (g0 < n_groups) {
    for (int g = g0; g < g1; ++g) {
      const float s = bf2f(scales[(int64_t)g * N + n]);
      const float z = bf2f(zeros[(int64_t)g * N + n]);
      const int k0 = g * group_size;
      const int kh0 = k0 >> 1;
      #pragma unroll 8
      for (int t = 0; t < (group_size >> 1); ++t) {
        const uint8_t p = __ldg(w_q + ((int64_t)(kh0 + t) * N + n));
        const float x0 = bf2f(__ldg(x + k0 + (t << 1)));
        const float x1 = bf2f(__ldg(x + k0 + (t << 1) + 1));
        acc = fmaf(x0, (float(p & 0xF) - z) * s, acc);
        acc = fmaf(x1, (float(p >> 4) - z) * s, acc);
      }
    }
  }
  partials[(int64_t)split * N + n] = acc;
}

__global__ void reduce_splits(
    const float* __restrict__ partials,
    __nv_bfloat16* __restrict__ y,
    int N, int n_splits)
{
  int n = blockIdx.x * blockDim.x + threadIdx.x;
  if (n >= N) return;
  float acc = 0.f;
  for (int s = 0; s < n_splits; ++s) acc += partials[(int64_t)s * N + n];
  y[n] = f2bf(acc);
}

void w4a16_gemv_cuda(
    torch::Tensor x, torch::Tensor w_q, torch::Tensor scales,
    torch::Tensor zeros, torch::Tensor y, torch::Tensor partials,
    int group_size, int n_splits, int mode)
{
  const int N = (int)w_q.size(1);
  const int K = (int)x.numel();
  auto stream = at::cuda::getCurrentCUDAStream();

  if (mode == 0) {
    // 128 cols/block, 8 K-splits, 32 threads x 8 warps = 256 threads
    constexpr int COLS = 128, NW = 8;
    constexpr int THREADS = (COLS / 4) * NW;
    gemv_mw_v4<COLS, NW><<<(N + COLS - 1) / COLS, THREADS, 0, stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        w_q.data_ptr<uint8_t>(),
        reinterpret_cast<const __nv_bfloat16*>(scales.data_ptr()),
        reinterpret_cast<const __nv_bfloat16*>(zeros.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(y.data_ptr()),
        N, K, group_size);
  } else {
    constexpr int BN = 128;
    dim3 block(BN);
    dim3 grid((N + BN - 1) / BN, n_splits);
    gemv_splitk<BN><<<grid, block, 0, stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(x.data_ptr()),
        w_q.data_ptr<uint8_t>(),
        reinterpret_cast<const __nv_bfloat16*>(scales.data_ptr()),
        reinterpret_cast<const __nv_bfloat16*>(zeros.data_ptr()),
        partials.data_ptr<float>(),
        N, K, group_size, n_splits);
    int rt = 256;
    reduce_splits<<<(N + rt - 1) / rt, rt, 0, stream>>>(
        partials.data_ptr<float>(),
        reinterpret_cast<__nv_bfloat16*>(y.data_ptr()),
        N, n_splits);
  }
}
"""

CPP_SRC = r"""
#include <torch/extension.h>
void w4a16_gemv_cuda(torch::Tensor x, torch::Tensor w_q, torch::Tensor scales,
                     torch::Tensor zeros, torch::Tensor y, torch::Tensor partials,
                     int group_size, int n_splits, int mode);
"""

_cuda_ext = None
_cuda_failed = False
_partials_cache: dict = {}


def _load_cuda():
    global _cuda_ext, _cuda_failed
    if _cuda_failed:
        return None
    if _cuda_ext is not None:
        return _cuda_ext
    try:
        from torch.utils.cpp_extension import load_inline

        # Prefer run-local cache; fall back to /dev/shm if disk is full.
        if "TORCH_EXTENSIONS_DIR" not in os.environ:
            for d in (
                os.path.join(os.path.dirname(__file__), "..", "..", "..", "cache", "torch_extensions"),
                "/dev/shm/torch_ext_w4a16",
            ):
                d = os.path.abspath(d)
                try:
                    os.makedirs(d, exist_ok=True)
                    os.environ["TORCH_EXTENSIONS_DIR"] = d
                    break
                except OSError:
                    continue

        _cuda_ext = load_inline(
            name="w4a16_gemv_v4mw",
            cpp_sources=CPP_SRC,
            cuda_sources=CUDA_SRC,
            functions=["w4a16_gemv_cuda"],
            extra_cuda_cflags=[
                "-O3",
                "--use_fast_math",
                "-std=c++17",
                "--expt-relaxed-constexpr",
            ],
            verbose=False,
        )
        return _cuda_ext
    except Exception:
        _cuda_failed = True
        return None


def _get_partials(n_splits, N, device):
    key = (n_splits, N, str(device))
    buf = _partials_cache.get(key)
    if buf is None:
        buf = torch.empty((n_splits, N), device=device, dtype=torch.float32)
        _partials_cache[key] = buf
    return buf


# ---------------------------------------------------------------------------
# Triton GEMM (also M=1 fallback)
# ---------------------------------------------------------------------------


@triton.autotune(
    configs=[
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 64}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 128}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 128}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 256}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 64}, num_warps=4, num_stages=4),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 128}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 128}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 64}, num_warps=4, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 64}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 128}, num_warps=8, num_stages=2),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 64}, num_warps=8, num_stages=3),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 64}, num_warps=4, num_stages=3),
    ],
    key=["M", "N", "K"],
)
@triton.jit
def w4a16_gemm_kernel(
    A_ptr, B_ptr, S_ptr, Z_ptr, C_ptr,
    M, N, K,
    stride_am, stride_ak, stride_bk, stride_bn,
    stride_sg, stride_sn, stride_cm, stride_cn,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    GROUP_M: tl.constexpr = 8
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_m = offs_m < M
    mask_n = offs_n < N
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    n_groups = K // GROUP_SIZE
    for g in range(n_groups):
        k_base = g * GROUP_SIZE
        offs_k = k_base + tl.arange(0, GROUP_SIZE)

        a = tl.load(
            A_ptr + offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak,
            mask=mask_m[:, None], other=0.0,
        )

        offs_kh = (k_base // 2) + tl.arange(0, GROUP_SIZE // 2)
        b_pack = tl.load(
            B_ptr + offs_kh[:, None] * stride_bk + offs_n[None, :] * stride_bn,
            mask=mask_n[None, :], other=0,
        ).to(tl.int32)
        lo = (b_pack & 0xF).to(tl.float32)
        hi = ((b_pack >> 4) & 0xF).to(tl.float32)
        b_u = tl.trans(tl.interleave(tl.trans(lo), tl.trans(hi)))

        s = tl.load(S_ptr + g * stride_sg + offs_n * stride_sn, mask=mask_n, other=0.0).to(tl.float32)
        z = tl.load(Z_ptr + g * stride_sg + offs_n * stride_sn, mask=mask_n, other=0.0).to(tl.float32)
        b_bf = (b_u - z[None, :]) * s[None, :]

        acc = tl.dot(a.to(tl.bfloat16), b_bf.to(tl.bfloat16), acc)

    tl.store(
        C_ptr + offs_m[:, None] * stride_cm + offs_n[None, :] * stride_cn,
        acc.to(tl.bfloat16),
        mask=mask_m[:, None] & mask_n[None, :],
    )


def _w4a16_gemm_triton(x, w_q, scales, zeros, out):
    M, K = x.shape
    N = w_q.shape[1]
    grid = lambda meta: (
        triton.cdiv(M, meta["BLOCK_M"]) * triton.cdiv(N, meta["BLOCK_N"]),
    )
    w4a16_gemm_kernel[grid](
        x, w_q, scales, zeros, out,
        M, N, K,
        x.stride(0), x.stride(1),
        w_q.stride(0), w_q.stride(1),
        scales.stride(0), scales.stride(1),
        out.stride(0), out.stride(1),
        GROUP_SIZE=GROUP_SIZE,
    )


def _w4a16_gemv_cuda(x, w_q, scales, zeros, out):
    ext = _load_cuda()
    if ext is None:
        return False
    N = w_q.shape[1]
    K = x.numel()
    if N >= 8192:
        dummy = _get_partials(1, 1, x.device)
        ext.w4a16_gemv_cuda(
            x.view(-1), w_q, scales, zeros, out.view(-1),
            dummy, GROUP_SIZE, 1, 0,
        )
    else:
        n_groups = K // GROUP_SIZE
        n_splits = min(16, n_groups)
        while n_groups % n_splits != 0 and n_splits > 1:
            n_splits -= 1
        partials = _get_partials(n_splits, N, x.device)
        ext.w4a16_gemv_cuda(
            x.view(-1), w_q, scales, zeros, out.view(-1),
            partials, GROUP_SIZE, n_splits, 1,
        )
    return True


def _w4a16_gemm(x, w_q, scales, zeros):
    M, K = x.shape
    N = w_q.shape[1]
    if not x.is_contiguous():
        x = x.contiguous()
    out = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)

    if M == 1:
        if not _w4a16_gemv_cuda(x, w_q, scales, zeros, out):
            # Triton fallback (BLOCK_M=16 with masking)
            _w4a16_gemm_triton(x, w_q, scales, zeros, out)
    else:
        _w4a16_gemm_triton(x, w_q, scales, zeros, out)
    return out


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0 and K % 2 == 0
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size
        self.register_buffer("w_q", torch.empty(K // 2, N, dtype=torch.uint8))
        self.register_buffer("scales", torch.empty(n_groups, N, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.empty(n_groups, N, dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return _w4a16_gemm(x, self.w_q, self.scales, self.zeros)


M = 1
N = 12288
K = 4096


def get_inputs():
    return [torch.randn(M, K, dtype=torch.bfloat16)]


def get_init_inputs():
    return [M, N, K]
