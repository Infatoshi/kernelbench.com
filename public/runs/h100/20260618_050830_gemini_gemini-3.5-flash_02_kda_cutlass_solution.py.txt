import torch
import torch.nn as nn
from einops import rearrange

@torch.compile(fullgraph=True, dynamic=False)
def vectorized_kda_fwd(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0, f"T={T} must be a multiple of chunk_size={BT}"
    NT = T // BT

    # Convert to fp32 for calculation
    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale

    # Rearrange tensors to chunk form
    q = rearrange(q, "b (n c) h d -> b h n c d", c=BT)
    k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
    v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
    g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
    beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)

    # In-chunk cumsum for g
    g = g.cumsum(-2)

    # ---- Build A_init via single batched matrix multiply with max-scaling ----
    # Find max of g along the chunk step dimension for stability
    max_g = g.max(dim=-2, keepdim=True).values  # (B, H, N, 1, K)
    g_scaled = g - max_g                       # (B, H, N, BT, K)
    ke = k * g_scaled.exp()                     # (B, H, N, BT, K)
    kem = k * (-g_scaled).exp()                 # (B, H, N, BT, K)
    
    # Batched matrix multiply: (B, H, N, BT, K) @ (B, H, N, K, BT) -> (B, H, N, BT, BT)
    A = ke @ kem.transpose(-1, -2)
    A = A * beta.unsqueeze(-1)                  # Multiply row c by beta_c
    
    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    A_init = -A.masked_fill(mask_diag_upper, 0) # Strictly lower triangular

    # ---- Solve for w and u using batch solve_triangular ----
    I_minus_A = torch.eye(BT, dtype=q.dtype, device=q.device) - A_init
    Y_w = beta.unsqueeze(-1) * (g.exp() * k)
    Y_u = beta.unsqueeze(-1) * v
    
    w = torch.linalg.solve_triangular(I_minus_A, Y_w, upper=False)
    u = torch.linalg.solve_triangular(I_minus_A, Y_u, upper=False)

    # ---- Compute Aqk for all chunks ----
    qe = q * g_scaled.exp()
    Aqk = qe @ kem.transpose(-1, -2)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)
    Aqk = Aqk.masked_fill(mask_strict_upper, 0)

    # ---- Recurrent inter-chunk pass ----
    decay = g[..., -1, :].exp()                                    # (B, H, N, K)
    v_weight = ((g[..., -1:, :] - g).exp() * k).transpose(-1, -2)  # (B, H, N, K, BT)

    S = q.new_zeros(B, H, K, V)
    o = torch.zeros_like(v)
    q_exp = q * g.exp()                                            # (B, H, N, BT, K)

    for i in range(NT):
        q_exp_i = q_exp[:, :, i]         # (B, H, BT, K)
        u_i = u[:, :, i]                 # (B, H, BT, V)
        w_i = w[:, :, i]                 # (B, H, BT, K)
        Aqk_i = Aqk[:, :, i]             # (B, H, BT, BT)
        decay_i = decay[:, :, i].unsqueeze(-1)  # (B, H, K, 1)
        v_weight_i = v_weight[:, :, i]   # (B, H, K, BT)

        v_i = u_i - w_i @ S              # (B, H, BT, V)
        o[:, :, i] = q_exp_i @ S + Aqk_i @ v_i
        S = S * decay_i + v_weight_i @ v_i

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o.to(dtype)

class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        return vectorized_kda_fwd(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)
