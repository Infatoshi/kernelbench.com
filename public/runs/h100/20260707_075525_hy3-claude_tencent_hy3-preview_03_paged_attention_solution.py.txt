"""Paged attention decode kernel using CUDA C++ - debug version."""

import torch
import math
from torch.utils.cpp_extension import load_inline


# CUDA kernel code - debug version with checks
cuda_code = r"""
#include <cuda_runtime.h>
#include <cstdint>
#include <math.h>
#include <stdio.h>

#define MAX_HEAD_DIM 128

__global__ void paged_attention_decode_kernel(
    const float* __restrict__ query,
    const float* __restrict__ kv_cache,
    float* __restrict__ output,
    const int32_t* __restrict__ block_table,
    const int32_t* __restrict__ seq_lens,
    int batch, int num_heads, int num_kv_heads, int head_dim,
    int page_size, int group_size, float scale, int max_blocks_per_seq
) {
    int pid = blockIdx.x;
    int batch_idx = pid / num_heads;
    int head_idx = pid % num_heads;
    int kv_head_idx = head_idx / group_size;

    int seq_len = seq_lens[batch_idx];
    if (seq_len <= 0) return;

    // Compute strides
    int q_batch_stride = num_heads * head_dim;
    int q_head_stride = head_dim;
    int o_batch_stride = num_heads * head_dim;
    int o_head_stride = head_dim;
    int kv_block_stride = page_size * num_kv_heads * head_dim * 2;
    int kv_page_stride = num_kv_heads * head_dim * 2;
    int kv_kv_head_stride = head_dim * 2;

    // Pointers
    const float* q_ptr = query + batch_idx * q_batch_stride + head_idx * q_head_stride;
    float* o_ptr = output + batch_idx * o_batch_stride + head_idx * o_head_stride;

    // Load query into registers
    float q[128];
    for (int d = 0; d < head_dim; d++) {
        q[d] = q_ptr[d];
    }

    // First pass: compute max score
    float max_score = -INFINITY;
    for (int i = 0; i < seq_len; i++) {
        int page_idx = i / page_size;
        int page_offset = i % page_size;
        int block_id = block_table[batch_idx * max_blocks_per_seq + page_idx];

        const float* k_ptr = kv_cache +
            block_id * kv_block_stride +
            page_offset * kv_page_stride +
            kv_head_idx * kv_kv_head_stride;

        float k_dot_q = 0.0f;
        for (int d = 0; d < head_dim; d++) {
            k_dot_q += q[d] * k_ptr[d];
        }
        k_dot_q *= scale;

        if (k_dot_q > max_score) {
            max_score = k_dot_q;
        }
    }

    // Second pass: compute softmax and accumulate output
    float sum_exp = 0.0f;
    float out[128] = {0.0f};

    for (int i = 0; i < seq_len; i++) {
        int page_idx = i / page_size;
        int page_offset = i % page_size;
        int block_id = block_table[batch_idx * max_blocks_per_seq + page_idx];

        const float* k_ptr = kv_cache +
            block_id * kv_block_stride +
            page_offset * kv_page_stride +
            kv_head_idx * kv_kv_head_stride;

        float k_dot_q = 0.0f;
        for (int d = 0; d < head_dim; d++) {
            k_dot_q += q[d] * k_ptr[d];
        }
        k_dot_q *= scale;

        float exp_score = expf(k_dot_q - max_score);
        sum_exp += exp_score;

        const float* v_ptr = kv_cache +
            block_id * kv_block_stride +
            page_offset * kv_page_stride +
            kv_head_idx * kv_kv_head_stride +
            head_dim;  // V is stored after K

        for (int d = 0; d < head_dim; d++) {
            out[d] += exp_score * v_ptr[d];
        }
    }

    // Normalize and store output
    for (int d = 0; d < head_dim; d++) {
        out[d] /= sum_exp;
        o_ptr[d] = out[d];
    }
}

extern "C" void launch_paged_attention(
    const float* query,
    const float* kv_cache,
    float* output,
    const int32_t* block_table,
    const int32_t* seq_lens,
    int batch, int num_heads, int num_kv_heads, int head_dim,
    int page_size, float scale, int max_blocks_per_seq
) {
    int total_programs = batch * num_heads;

    // Debug: print arguments
    printf("Launching kernel: total_programs=%d, max_blocks_per_seq=%d\n", total_programs, max_blocks_per_seq);
    printf("  query=%p, kv_cache=%p, output=%p\n", (void*)query, (void*)kv_cache, (void*)output);
    printf("  block_table=%p, seq_lens=%p\n", (void*)block_table, (void*)seq_lens);

    paged_attention_decode_kernel<<<total_programs, 1>>>(
        query, kv_cache, output, block_table, seq_lens,
        batch, num_heads, num_kv_heads, head_dim, page_size,
        num_heads / num_kv_heads, scale, max_blocks_per_seq
    );

    cudaError_t err = cudaGetLastError();
    if (err != cudaSuccess) {
        printf("CUDA error after kernel launch: %s\n", cudaGetErrorString(err));
    }
}
"""


# C++ wrapper code
cpp_code = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <vector>
#include <algorithm>

extern "C" void launch_paged_attention(
    const float* query,
    const float* kv_cache,
    float* output,
    const int32_t* block_table,
    const int32_t* seq_lens,
    int batch, int num_heads, int num_kv_heads, int head_dim,
    int page_size, float scale, int max_blocks_per_seq
);

torch::Tensor paged_attention_decode(
    torch::Tensor query,
    torch::Tensor kv_cache,
    torch::Tensor block_table,
    torch::Tensor seq_lens,
    int num_heads, int num_kv_heads,
    int head_dim, int page_size, double scale
) {
    int batch = query.size(0);

    // Convert to float32
    auto query_f32 = query.to(torch::kFloat32);
    auto kv_cache_f32 = kv_cache.to(torch::kFloat32);
    auto output = torch::empty_like(query_f32);

    // Compute max_blocks_per_seq on host
    auto seq_lens_cpu = seq_lens.to(torch::kCPU);
    const int32_t* seq_lens_cpu_ptr = seq_lens_cpu.data_ptr<int32_t>();
    int max_seq_len = 0;
    for (int b = 0; b < batch; b++) {
        max_seq_len = std::max(max_seq_len, (int)seq_lens_cpu_ptr[b]);
    }
    int max_blocks_per_seq = (max_seq_len + page_size - 1) / page_size;

    printf("In C++ wrapper: batch=%d, num_heads=%d, head_dim=%d, max_blocks_per_seq=%d\n",
           batch, num_heads, head_dim, max_blocks_per_seq);
    printf("  query_f32 ptr=%p, kv_cache_f32 ptr=%p\n",
           (void*)query_f32.data_ptr<float>(), (void*)kv_cache_f32.data_ptr<float>());

    // Launch kernel
    launch_paged_attention(
        query_f32.data_ptr<float>(),
        kv_cache_f32.data_ptr<float>(),
        output.data_ptr<float>(),
        block_table.data_ptr<int32_t>(),
        seq_lens.data_ptr<int32_t>(),
        batch, num_heads, num_kv_heads, head_dim, page_size,
        static_cast<float>(scale), max_blocks_per_seq
    );

    return output.to(torch::kBFloat16);
}
"""


# Build the CUDA extension
paged_attn_cuda = load_inline(
    name="paged_attn_cuda",
    cpp_sources=[cpp_code],
    cuda_sources=[cuda_code],
    functions=["paged_attention_decode"],
    with_cuda=True,
    extra_cuda_cflags=["-arch=sm_90"],
)


class Model(torch.nn.Module):

    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(self, query, kv_cache, block_table, seq_lens):
        return paged_attn_cuda.paged_attention_decode(
            query, kv_cache, block_table, seq_lens,
            self.num_heads, self.num_kv_heads, self.head_dim, self.page_size, self.scale,
        )


def get_inputs():
    import reference
    B = reference.BATCH
    H = reference.NUM_HEADS
    Hkv = reference.NUM_KV_HEADS
    D = reference.HEAD_DIM
    L = reference.SEQ_LEN
    P = reference.PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[:B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    import reference
    return [reference.BATCH, reference.NUM_HEADS, reference.NUM_KV_HEADS,
            reference.HEAD_DIM, reference.SEQ_LEN, reference.PAGE_SIZE]
