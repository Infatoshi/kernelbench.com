"""TopK via bitonic sort — custom CUDA kernel.

Uses a two-pass approach:
  1. chunk_topk: divides each row into 1024-element chunks, sorts each chunk
     in shared memory via bitonic network, outputs the local top-k.
  2. merge_topk: merges all per-chunk candidate sets into the final top-k,
     also via bitonic sort in shared memory.

Handles non-power-of-2 n by padding chunks with -INF. Returns values and
int64 indices sorted descending.
"""

from __future__ import annotations

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

_cpp_src = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cstdint>
#include <vector>

#define CHUNK_SIZE 1024

// Forward declarations of launch wrappers (defined in cuda_sources).
void launch_chunk_topk(
    const float* input, float* cand_vals, int* cand_idxs,
    int batch, int n, int k, int num_chunks, cudaStream_t stream);

void launch_merge_topk(
    const float* cand_vals, const int* cand_idxs,
    float* out_vals, int64_t* out_idxs,
    int num_chunks, int k, int padded_size, int total_cands,
    int merge_threads, int smem_bytes, int batch, cudaStream_t stream);

static int next_pow2(int x) {
    int p = 1;
    while (p < x) p <<= 1;
    return p;
}

std::vector<torch::Tensor> topk_forward(torch::Tensor x, int64_t k64) {
    TORCH_CHECK(x.dim() == 2, "x must be 2D");
    TORCH_CHECK(x.scalar_type() == torch::kFloat32, "x must be float32");
    TORCH_CHECK(x.is_cuda(), "x must be on CUDA");

    int batch = static_cast<int>(x.size(0));
    int n     = static_cast<int>(x.size(1));
    int k     = static_cast<int>(k64);
    int num_chunks = (n + CHUNK_SIZE - 1) / CHUNK_SIZE;

    auto opts_f32 = torch::TensorOptions().dtype(torch::kFloat32).device(x.device());
    auto opts_i32 = torch::TensorOptions().dtype(torch::kInt32).device(x.device());
    auto opts_i64 = torch::TensorOptions().dtype(torch::kInt64).device(x.device());

    auto cand_vals = torch::empty({batch, num_chunks, k}, opts_f32);
    auto cand_idxs = torch::empty({batch, num_chunks, k}, opts_i32);

    cudaStream_t stream = 0;  // default stream
    launch_chunk_topk(
        x.const_data_ptr<float>(),
        cand_vals.data_ptr<float>(),
        cand_idxs.data_ptr<int>(),
        batch, n, k, num_chunks, stream);

    int total_cands  = num_chunks * k;
    int padded_cands = next_pow2(total_cands);
    int merge_threads = std::min(padded_cands, 1024);
    int smem_bytes    = padded_cands * 8;   // float val + int idx per element

    auto out_vals = torch::empty({batch, k}, opts_f32);
    auto out_idxs = torch::empty({batch, k}, opts_i64);

    launch_merge_topk(
        cand_vals.const_data_ptr<float>(),
        cand_idxs.const_data_ptr<int>(),
        out_vals.data_ptr<float>(),
        out_idxs.data_ptr<int64_t>(),
        num_chunks, k, padded_cands, total_cands,
        merge_threads, smem_bytes, batch, stream);

    return {out_vals, out_idxs};
}
"""


# ── CUDA kernels (compiled with nvcc) ───────────────────────────────────
_cuda_src = r"""
#include <cuda_runtime.h>
#include <cstdint>
#include <float.h>

#define CHUNK_SIZE   1024
#define CHUNK_THREADS 1024
#define NEG_INF __int_as_float(0xff800000)

// ── Device helper: bitonic compare-and-swap for (val, idx) pair ─────────
__device__ __forceinline__ void bitonic_cas(
    float &a_val, int &a_idx,
    float &b_val, int &b_idx,
    bool ascending)
{
    if (ascending) {
        if (a_val > b_val) {
            float t_val = a_val; a_val = b_val; b_val = t_val;
            int   t_idx = a_idx; a_idx = b_idx; b_idx = t_idx;
        }
    } else {
        if (a_val < b_val) {
            float t_val = a_val; a_val = b_val; b_val = t_val;
            int   t_idx = a_idx; a_idx = b_idx; b_idx = t_idx;
        }
    }
}


// ── Kernel 1: per-chunk top-k ───────────────────────────────────────────
extern "C" __global__ void chunk_topk_kernel(
    const float* __restrict__ input,
    float*       __restrict__ cand_vals,
    int*         __restrict__ cand_idxs,
    int batch, int n, int k, int num_chunks)
{
    int batch_idx = blockIdx.y;
    int chunk_idx = blockIdx.x;
    int tid = threadIdx.x;

    __shared__ float svals[CHUNK_SIZE];
    __shared__ int   sidxs[CHUNK_SIZE];

    int chunk_start = chunk_idx * CHUNK_SIZE;
    int chunk_limit = min(CHUNK_SIZE, max(0, n - chunk_start));

    // Load into shared memory; pad unused slots with NEG_INF.
    int global_idx = batch_idx * n + chunk_start + tid;
    if (tid < chunk_limit) {
        svals[tid] = __ldg(input + global_idx);
        sidxs[tid] = chunk_start + tid;
    } else if (tid < CHUNK_SIZE) {
        svals[tid] = NEG_INF;
        sidxs[tid] = -1;
    }
    __syncthreads();

    // Bitonic sort — ascending order.
    #pragma unroll 1
    for (int size = 2; size <= CHUNK_SIZE; size <<= 1) {
        #pragma unroll 1
        for (int stride = size >> 1; stride > 0; stride >>= 1) {
            int partner = tid ^ stride;
            if (tid < partner) {
                bool asc = ((tid & size) == 0);
                bitonic_cas(svals[tid], sidxs[tid],
                            svals[partner], sidxs[partner], asc);
            }
            __syncthreads();
        }
    }

    // Write top-k in descending order (largest first).
    if (tid < k) {
        int src = CHUNK_SIZE - 1 - tid;
        int dst = batch_idx * num_chunks * k + chunk_idx * k + tid;
        cand_vals[dst] = svals[src];
        cand_idxs[dst] = sidxs[src];
    }
}


// ── Kernel 2: merge per-chunk candidates into final top-k ────────────────
extern "C" __global__ void merge_topk_kernel(
    const float* __restrict__ cand_vals,
    const int*   __restrict__ cand_idxs,
    float*       __restrict__ out_vals,
    int64_t*     __restrict__ out_idxs,
    int num_chunks, int k, int padded_size, int total_cands)
{
    int batch_idx = blockIdx.x;
    int tid       = threadIdx.x;
    int elems_per_thread = padded_size / blockDim.x;

    extern __shared__ char raw[];
    float* svals = reinterpret_cast<float*>(raw);
    int*   sidxs = reinterpret_cast<int*>(raw + padded_size * sizeof(float));

    // Load candidates into shared memory; pad with NEG_INF.
    for (int i = tid; i < padded_size; i += blockDim.x) {
        if (i < total_cands) {
            svals[i] = __ldg(cand_vals + batch_idx * total_cands + i);
            sidxs[i] = __ldg(cand_idxs + batch_idx * total_cands + i);
        } else {
            svals[i] = NEG_INF;
            sidxs[i] = -1;
        }
    }
    __syncthreads();

    // Bitonic sort — ascending order, multi-element-per-thread.
    for (int size = 2; size <= padded_size; size <<= 1) {
        for (int stride = size >> 1; stride > 0; stride >>= 1) {
            for (int e = 0; e < elems_per_thread; ++e) {
                int pos     = tid + e * blockDim.x;
                int partner = pos ^ stride;
                if (pos < partner) {
                    bool asc = ((pos & size) == 0);
                    bitonic_cas(svals[pos], sidxs[pos],
                                svals[partner], sidxs[partner], asc);
                }
            }
            __syncthreads();
        }
    }

    // Output top-k in descending order.
    if (tid < k) {
        int src = padded_size - 1 - tid;
        out_vals[batch_idx * k + tid] = svals[src];
        out_idxs[batch_idx * k + tid] = static_cast<int64_t>(sidxs[src]);
    }
}


// ── Launch wrappers ─────────────────────────────────────────────────────
void launch_chunk_topk(
    const float* input, float* cand_vals, int* cand_idxs,
    int batch, int n, int k, int num_chunks, cudaStream_t stream)
{
    dim3 grid(num_chunks, batch, 1);
    dim3 block(CHUNK_THREADS, 1, 1);
    chunk_topk_kernel<<<grid, block, 0, stream>>>(
        input, cand_vals, cand_idxs, batch, n, k, num_chunks);
}

void launch_merge_topk(
    const float* cand_vals, const int* cand_idxs,
    float* out_vals, int64_t* out_idxs,
    int num_chunks, int k, int padded_size, int total_cands,
    int merge_threads, int smem_bytes, int batch, cudaStream_t stream)
{
    // Opt-in to larger dynamic shared memory if needed.
    int device = 0;
    cudaGetDevice(&device);
    int max_optin = 0;
    cudaDeviceGetAttribute(&max_optin, cudaDevAttrMaxSharedMemoryPerBlockOptin, device);
    if (smem_bytes > 49152 && smem_bytes <= max_optin) {
        cudaFuncSetAttribute(merge_topk_kernel,
                             cudaFuncAttributeMaxDynamicSharedMemorySize,
                             smem_bytes);
    }

    dim3 grid(batch, 1, 1);
    dim3 block(merge_threads, 1, 1);
    merge_topk_kernel<<<grid, block, smem_bytes, stream>>>(
        cand_vals, cand_idxs, out_vals, out_idxs,
        num_chunks, k, padded_size, total_cands);
}
"""
