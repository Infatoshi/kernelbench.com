"""Custom top-k via segmented bitonic selection (CUDA C++ via load_inline).

Strategy (bandwidth-bound on B200 HBM3e 8 TB/s):
  - Kernel A: split each row into `nbpr` contiguous segments; one block per
    segment finds the top-k of its slice using per-thread register buffering +
    a shared-memory bitonic sort over (threads * k) candidates. Writes the
    segment top-k to a scratch buffer.
  - Kernel B: merge the `nbpr` segment top-k arrays per row with another
    shared-memory bitonic sort, taking the global top-k (sorted descending).

No PyTorch selection routine is used; the selection is implemented by hand.
"""
import math
import torch
import torch.nn as nn

from torch.utils.cpp_extension import load_inline

_CUDA_SRC = r"""
#include <torch/extension.h>
#include <c10/cuda/CUDAStream.h>
#include <cstdio>
#include <cfloat>

#define WARP 32

// next power of two
__host__ __device__ static inline int npow2(int x) {
    int p = 1;
    while (p < x) p <<= 1;
    return p;
}

// ---------- Kernel A: per-segment top-k ----------
// Each block scatters its `slice` real elements into shared memory and runs a
// bitonic sort over exactly `slice` elements (padded to a power of two with
// -inf). This keeps shared memory ~ slice*8 bytes and only does O(slice log^2
// slice) work instead of sorting T*k padded candidates.
__global__ void seg_topk_kernel(
    const float* __restrict__ x,
    const int n,
    const int k,
    const int slice,
    const int nbpr,
    float* __restrict__ out_vals,   // (batch, nbpr, k)
    int64_t* __restrict__ out_idx)  // (batch, nbpr, k)
{
    const int row = blockIdx.y;
    const int seg = blockIdx.x;
    const int T = blockDim.x;
    const int M = npow2(slice);

    extern __shared__ char shmem[];
    float* sval = reinterpret_cast<float*>(shmem);
    int*   sidx = reinterpret_cast<int*>(shmem + M * sizeof(float));

    const int t = threadIdx.x;
    const int chunk = (slice + T - 1) / T;

    const int seg_start = seg * slice;
    const int xbase = row * n;

    // Scatter this thread's chunk into shared; invalid slots get -inf.
    for (int c = 0; c < chunk; c++) {
        int slot = t * chunk + c;
        int gi = seg_start + slot;
        if (slot < slice && gi < n) {
            sval[slot] = x[xbase + gi];
            sidx[slot] = gi;
        } else if (slot < M) {
            sval[slot] = -FLT_MAX;
            sidx[slot] = 0;
        }
    }
    // Fill any shared slots beyond what threads covered with -inf.
    for (int s = t; s < M; s += T) {
        if (s >= slice) { sval[s] = -FLT_MAX; sidx[s] = 0; }
    }
    __syncthreads();

    // bitonic sort ascending over M
    for (int kk = 2; kk <= M; kk <<= 1) {
        for (int jj = kk >> 1; jj > 0; jj >>= 1) {
            for (int i = t; i < M; i += T) {
                int l = i ^ jj;
                if (l > i) {
                    bool asc = ((i & kk) == 0);
                    float vi = sval[i], vl = sval[l];
                    if ((asc && vi > vl) || (!asc && vi < vl)) {
                        sval[i] = vl; sval[l] = vi;
                        int ti = sidx[i]; sidx[i] = sidx[l]; sidx[l] = ti;
                    }
                }
            }
            __syncthreads();
        }
    }

    // ascending: largest is at M-1. write top-k descending.
    int out_base = (row * nbpr + seg) * k;
    for (int j = 0; j < k; j++) {
        out_vals[out_base + j] = sval[M - 1 - j];
        out_idx[out_base + j]  = sidx[M - 1 - j];
    }
}

// ---------- Merge stage: combine MG*k candidate pairs into top-k ----------
// Each block merges MG contiguous input groups (each of k pairs) into a
// single top-k group. When `write_out` is set, the result is written to the
// final (batch, k) output (sorted descending); otherwise to the ping-pong
// buffer. With MG up to 64 the whole reduction collapses to very few kernel
// launches, keeping launch overhead small. Shared memory is npow2(MG*k)*8,
// bounded by MG*k <= 4096 -> <= 32 KB (within the default budget).
__global__ void merge_stage_kernel(
    const float* __restrict__ in_vals,   // (batch, gin, k)
    const int64_t* __restrict__ in_idx,  // (batch, gin, k)
    const int gin,
    const int MG,
    const int k,
    float* __restrict__ out_vals,        // (batch, gin/MG, k) or (batch, k)
    int64_t* __restrict__ out_idx,
    bool write_out)
{
    const int row = blockIdx.y;
    const int gi = blockIdx.x;
    const int Tb = blockDim.x;

    const int N = MG * k;
    const int M = npow2(N);

    extern __shared__ char shmem[];
    float* sval = reinterpret_cast<float*>(shmem);
    int*   sidx = reinterpret_cast<int*>(shmem + M * sizeof(float));

    for (int i = threadIdx.x; i < N; i += Tb) {
        int g = i / k;
        int j = i % k;
        int src = (row * gin + (gi * MG + g)) * k + j;
        sval[i] = in_vals[src];
        sidx[i] = in_idx[src];
    }
    for (int p = N + threadIdx.x; p < M; p += Tb) {
        sval[p] = -FLT_MAX; sidx[p] = 0;
    }
    __syncthreads();

    for (int kk = 2; kk <= M; kk <<= 1) {
        for (int jj = kk >> 1; jj > 0; jj >>= 1) {
            for (int i = threadIdx.x; i < M; i += Tb) {
                int l = i ^ jj;
                if (l > i) {
                    bool asc = ((i & kk) == 0);
                    float vi = sval[i], vl = sval[l];
                    if ((asc && vi > vl) || (!asc && vi < vl)) {
                        sval[i] = vl; sval[l] = vi;
                        int ti = sidx[i]; sidx[i] = sidx[l]; sidx[l] = ti;
                    }
                }
            }
            __syncthreads();
        }
    }

    for (int j = 0; j < k; j++) {
        if (write_out) {
            out_vals[row * k + j] = sval[M - 1 - j];
            out_idx[row * k + j]  = (int64_t)sidx[M - 1 - j];
        } else {
            int o = (row * (gin / MG) + gi) * k + j;
            out_vals[o] = sval[M - 1 - j];
            out_idx[o]  = (int64_t)sidx[M - 1 - j];
        }
    }
}

void topk_cuda(
    torch::Tensor x,
    int k,
    int nbpr,
    torch::Tensor scratch_vals,
    torch::Tensor scratch_idx,
    torch::Tensor scratch2_vals,
    torch::Tensor scratch2_idx,
    torch::Tensor out_vals,
    torch::Tensor out_idx,
    int run_b)
{
    const int batch = x.size(0);
    const int n = x.size(1);
    const int slice = (n + nbpr - 1) / nbpr;

    // Block size: a power of two in [64, 256]. Keep it <= slice so the scatter
    // cleanly covers all `slice` shared slots (each thread handles 1 element
    // when T == slice, or a few when T < slice).
    int T = 256;
    while (T > slice && T > 64) T >>= 1;
    if (T < 64) T = 64;
    const int M_A = npow2(slice);
    size_t sh_A = M_A * (sizeof(float) + sizeof(int));

    cudaStream_t stream = c10::cuda::getCurrentCUDAStream();

    dim3 gridA(nbpr, batch);
    seg_topk_kernel<<<gridA, T, sh_A, stream>>>(
        x.data_ptr<float>(), n, k, slice, nbpr,
        scratch_vals.data_ptr<float>(), scratch_idx.data_ptr<int64_t>());

    if (!run_b) return;

    // Tree merge: repeatedly combine MG groups into one until a single
    // group remains. MG is chosen as large as possible (<= 64) so each
    // kernel launch merges many groups, collapsing the reduction to very
    // few launches (low launch overhead) while keeping shared memory
    // (npow2(MG*k)*8) within the 48 KB default budget.
    int gin = nbpr;
    float* cur_v = scratch_vals.data_ptr<float>();
    int64_t* cur_i = scratch_idx.data_ptr<int64_t>();
    float* nxt_v = scratch2_vals.data_ptr<float>();
    int64_t* nxt_i = scratch2_idx.data_ptr<int64_t>();

    int Tb = 256;

    while (gin > 1) {
        int MG = 1;
        while ((MG << 1) <= 4 && (MG << 1) * k <= 4096 && (MG << 1) <= gin) MG <<= 1;
        int gout = gin / MG;
        dim3 grid(gout, batch);
        const int M_stage = npow2(MG * k);
        size_t sh_stage = M_stage * (sizeof(float) + sizeof(int));
        if (gout == 1) {
            merge_stage_kernel<<<grid, Tb, sh_stage, stream>>>(
                cur_v, cur_i, gin, MG, k,
                out_vals.data_ptr<float>(), out_idx.data_ptr<int64_t>(), true);
        } else {
            merge_stage_kernel<<<grid, Tb, sh_stage, stream>>>(
                cur_v, cur_i, gin, MG, k,
                nxt_v, nxt_i, false);
        }
        float* tv = cur_v; cur_v = nxt_v; nxt_v = tv;
        int64_t* ti = cur_i; cur_i = nxt_i; nxt_i = ti;
        gin = gout;
    }
}
"""

_MODULE = load_inline(
    name="topk_bitonic",
    cpp_sources=[
        "void topk_cuda(torch::Tensor x, int k, int nbpr, "
        "torch::Tensor scratch_vals, torch::Tensor scratch_idx, "
        "torch::Tensor scratch2_vals, torch::Tensor scratch2_idx, "
        "torch::Tensor out_vals, torch::Tensor out_idx, int run_b);"
    ],
    cuda_sources=_CUDA_SRC,
    functions=["topk_cuda"],
    verbose=False,
    extra_cuda_cflags=[],
    build_directory="/workspace/problems/05_topk_bitonic/build_topk",
)


def _choose_nbpr(n: int, k: int) -> int:
    # Power-of-two number of segments, targeting small segments (~128
    # elements) for many concurrent blocks (good latency hiding). Capped so
    # the scratch buffers and tree-merge depth stay reasonable.
    target = 128
    nbpr = 1
    while (nbpr << 1) <= 1024 and (nbpr << 1) * target <= n:
        nbpr <<= 1
    nbpr = max(1, nbpr)
    return nbpr


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))
        self.nbpr = _choose_nbpr(n, k)
        self.scratch_vals = torch.empty(batch, self.nbpr, k, dtype=torch.float32, device="cuda")
        self.scratch_idx = torch.empty(batch, self.nbpr, k, dtype=torch.int64, device="cuda")
        self.scratch2_vals = torch.empty(batch, self.nbpr, k, dtype=torch.float32, device="cuda")
        self.scratch2_idx = torch.empty(batch, self.nbpr, k, dtype=torch.int64, device="cuda")

    def forward(self, x: torch.Tensor):
        out_vals = torch.empty(self.batch, self.k, dtype=torch.float32, device=x.device)
        out_idx = torch.empty(self.batch, self.k, dtype=torch.int64, device=x.device)
        import os as _os
        run_b = int(_os.environ.get("RUN_B", "1"))
        _MODULE.topk_cuda(
            x, self.k, self.nbpr,
            self.scratch_vals, self.scratch_idx,
            self.scratch2_vals, self.scratch2_idx,
            out_vals, out_idx, run_b,
        )
        return out_vals, out_idx


batch = 64
n = 8192
k = 8


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
