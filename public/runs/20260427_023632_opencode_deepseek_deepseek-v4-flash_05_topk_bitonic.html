"""Optimized top-k kernel for RTX PRO 6000 Blackwell (SM120).

Strategy:
  Phase 1: Each thread scans strided elements, maintains sorted top-k in registers.
  Phase 2: Copy per-thread top-k to shared memory (ordered-uint32 encoding).
  Phase 3: Radix selection (4-byte radix) across all shared-memory candidates
           to find the exact k-th largest threshold value.
  Phase 4: Gather elements >= threshold from shared memory to global output.
  Phase 5: Bitonic sort of k output elements into descending order.
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cc = torch.cuda.get_device_capability(0)
_SM = _cc[0] * 10 + _cc[1]
_ARCH_FLAG = f"-arch=sm_{_SM}"

CUDA_SOURCE = r"""
#include <cuda_runtime.h>
#include <cstdint>

#define WARP 32
#define MAX_K 64
#define THREADS 256

__device__ __forceinline__ uint32_t f32u(float x) {
    uint32_t u = __float_as_uint(x);
    return (u & 0x80000000) ? ~u : (u | 0x80000000);
}

__device__ __forceinline__ float u2f(uint32_t u) {
    u = (u & 0x80000000) ? (u & 0x7FFFFFFF) : ~u;
    return __uint_as_float(u);
}

__global__ void __launch_bounds__(THREADS)
topk_kernel(const float* __restrict__ x,
            float* __restrict__ values,
            int64_t* __restrict__ indices,
            int batch, int n, int k) {
    int row = blockIdx.x;
    if (row >= batch) return;

    const float* row_x = x + (int64_t)row * n;
    float* row_val = values + (int64_t)row * k;
    int64_t* row_idx = indices + (int64_t)row * k;

    int tid = threadIdx.x;

    /* ---- Phase 1: per-thread register top-k --------------------------- */
    float lv[MAX_K];
    int   li[MAX_K];
    for (int i = 0; i < MAX_K; i++) {
        lv[i] = -__uint_as_float(0xFF7FFFFF);
        li[i] = 0;
    }
    int cnt = 0;

    for (int i = tid; i < n; i += THREADS) {
        float val = row_x[i];
        if (cnt < k) {
            int pos = cnt;
            while (pos > 0 && lv[pos - 1] < val) {
                lv[pos] = lv[pos - 1];
                li[pos] = li[pos - 1];
                --pos;
            }
            lv[pos] = val;
            li[pos] = i;
            ++cnt;
        } else if (val > lv[k - 1]) {
            int pos = k - 1;
            while (pos > 0 && lv[pos - 1] < val) {
                lv[pos] = lv[pos - 1];
                li[pos] = li[pos - 1];
                --pos;
            }
            lv[pos] = val;
            li[pos] = i;
        }
    }

    __syncthreads();

    /* ---- Phase 2: copy local top-k to shared memory ------------------ */
    extern __shared__ char smem[];
    uint32_t* all_u = (uint32_t*)smem;
    int*      all_i = (int*)(smem + THREADS * k * 4);

    int base = tid * k;
    for (int i = 0; i < k; i++) {
        all_u[base + i] = f32u(lv[i]);
        all_i[base + i] = li[i];
    }
    __syncthreads();

    /* ---- Phase 3: radix selection over shared-memory candidates ------ */
    uint32_t prefix = 0;
    int remaining = k;

    __shared__ uint32_t hist[256];

    int total_cand = THREADS * k;

    for (int byte = 3; byte >= 0; --byte) {
        if (tid < 256) hist[tid] = 0;
        __syncthreads();

        for (int i = tid; i < total_cand; i += THREADS) {
            uint32_t u = all_u[i];
            int shift = (byte + 1) * 8;
            uint32_t higher = (byte < 3) ? (u >> shift) : 0;
            if (higher == prefix) {
                uint32_t bval = (u >> (byte * 8)) & 0xFF;
                atomicAdd(&hist[bval], 1);
            }
        }
        __syncthreads();

        if (tid == 0) {
            uint32_t counted = 0;
            for (int b = 255; b >= 0; --b) {
                if (counted + hist[b] >= (uint32_t)remaining) {
                    remaining -= (int)counted;
                    prefix = (prefix << 8) | (uint32_t)b;
                    break;
                }
                counted += hist[b];
            }
        }
        __syncthreads();
    }

    uint32_t thresh = prefix;
    __syncthreads();

    /* ---- Phase 4: gather from shared memory into global output ------- */
    int local_above = 0;
    for (int i = tid; i < total_cand; i += THREADS) {
        if (all_u[i] > thresh) ++local_above;
    }

    int warp_id = tid / WARP;
    int lane = tid % WARP;
    for (int offset = 16; offset > 0; offset /= 2)
        local_above += __shfl_xor_sync(0xFFFFFFFF, local_above, offset);

    __shared__ int warp_sums[8];
    if (lane == 0) warp_sums[warp_id] = local_above;
    __syncthreads();

    __shared__ int above_total;
    if (tid == 0) {
        int total = 0;
        for (int w = 0; w < (THREADS / WARP); w++) total += warp_sums[w];
        above_total = total;
    }
    __syncthreads();

    __shared__ int gpos;
    if (tid == 0) gpos = 0;
    __syncthreads();

    for (int i = tid; i < total_cand; i += THREADS) {
        if (all_u[i] > thresh) {
            int pos = atomicAdd(&gpos, 1);
            if (pos < k) {
                row_val[pos] = u2f(all_u[i]);
                row_idx[pos] = (int64_t)all_i[i];
            }
        }
    }
    __syncthreads();

    int need = k - above_total;
    if (need > 0 && tid == 0) {
        int written = 0;
        for (int i = 0; i < total_cand && written < need; i++) {
            if (all_u[i] == thresh) {
                row_val[above_total + written] = u2f(all_u[i]);
                row_idx[above_total + written] = (int64_t)all_i[i];
                ++written;
            }
        }
    }
    __syncthreads();

    /* ---- Phase 5: sort k output elements (descending) ---------------- */
    if (k > 1) {
        __shared__ float sv[MAX_K];
        __shared__ int64_t si[MAX_K];

        for (int i = tid; i < k; i += THREADS) {
            sv[i] = row_val[i];
            si[i] = row_idx[i];
        }
        __syncthreads();

        for (int size = 2; size <= k; size *= 2) {
            for (int stride = size / 2; stride > 0; stride /= 2) {
                for (int i = tid; i < k; i += THREADS) {
                    int j = i ^ stride;
                    if (j > i) {
                        bool asc = (i & size) == 0;
                        if ((asc && sv[i] > sv[j]) || (!asc && sv[i] < sv[j])) {
                            float tv = sv[i]; sv[i] = sv[j]; sv[j] = tv;
                            int64_t ti = si[i]; si[i] = si[j]; si[j] = ti;
                        }
                    }
                }
                __syncthreads();
            }
        }

        for (int i = tid; i < k / 2; i += THREADS) {
            int j = k - 1 - i;
            float tv = sv[i]; sv[i] = sv[j]; sv[j] = tv;
            int64_t ti = si[i]; si[i] = si[j]; si[j] = ti;
        }
        __syncthreads();

        for (int i = tid; i < k; i += THREADS) {
            row_val[i] = sv[i];
            row_idx[i] = si[i];
        }
    }
}

extern "C" void launch_topk(
    float* x,
    float* values,
    int64_t* indices,
    int batch, int n, int k
) {
    const int threads = 256;
    const int blocks = batch;
    size_t smem = (size_t)threads * (size_t)k * 4 * 2;
    topk_kernel<<<blocks, threads, smem>>>(
        x, values, indices, batch, n, k
    );
}
"""
