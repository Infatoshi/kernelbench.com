        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:

        B, T, H, K = q.shape
        V = v.shape[-1]
        BT = self.chunk_size

        q = q * self.scale

        k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
        v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
        g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
        beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)

        g = g.cumsum(-2)

@torch.compile(fullgraph=True, mode="max-autotune")
def _forward_compiled(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0, f"T={T} must be a multiple of chunk_size={BT}"
    NT = T // BT

    # Cast to fp32 for internal computation
    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale

    q = rearrange(q, "b (n c) h d -> b h n c d", c=BT)
    k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
    v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
    g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
    beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)

    g = g.cumsum(-2)

    # ---- Build A_kk (intra-chunk K-K interaction, lower-triangular w/ diag masked) ----
    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    
    # Vectorized A_ref
    temp1 = k * g.exp()
    temp2 = k * (-g).exp()
    A_ref = temp1 @ temp2.transpose(-2, -1)
    A_ref = A_ref * beta[..., None]
    A_ref = -A_ref.masked_fill(mask_diag_upper, 0)

    # Matrix inversion via solve_triangular
    M = torch.eye(BT, device=q.device).expand_as(A_ref) - A_ref
    M_inv = torch.linalg.solve_triangular(M, torch.eye(BT, device=q.device).expand_as(M), upper=False)
    A_final = M_inv * beta[..., None, :]

    w = A_final @ (g.exp() * k)
    u = A_final @ v

    # ---- Recurrent inter-chunk pass ----
    S = q.new_zeros(B, H, K, V)
    o = torch.zeros_like(v)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)
    
    for i in range(NT):
        q_i, k_i, u_i, g_i, w_i = q[:, :, i], k[:, :, i], u[:, :, i], g[:, :, i], w[:, :, i]
        
        temp_q = q_i * g_i.exp()
        temp_k = k_i * (-g_i).exp()
        Aqk = temp_q @ temp_k.transpose(-2, -1)
        Aqk = Aqk.masked_fill(mask_strict_upper, 0)
