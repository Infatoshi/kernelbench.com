from __future__ import annotations

from dataclasses import dataclass, field
import torch
import torch.nn as nn
import torch.nn.functional as F
import triton
import triton.language as tl

OP_TYPE = "kimi_linear_w4a16_decode"
HARDWARE_REQUIRED = ["RTX_PRO_6000"]
EPS = 1.0e-6
GROUP_SIZE = 128


@dataclass(frozen=True)
class Config:
    hidden: int = 2304
    kda_heads: int = 32
    kda_head_dim: int = 128
    short_conv: int = 4
    mla_heads: int = 32
    kv_lora: int = 512
    qk_nope: int = 128
    qk_rope: int = 64
    v_head: int = 128
    rope_theta: float = 10000.0
    n_experts: int = 64
    n_active: int = 8
    n_shared: int = 1
    moe_inter: int = 1024
    routed_scaling: float = 2.446
    group: int = 128
    pattern: tuple = ("K", "K", "K", "M")
    dtype: torch.dtype = field(default=torch.bfloat16)


def build_config(shape: dict) -> Config:
    return Config(n_experts=int(shape.get("n_experts", 64)))


# --------------------------------------------------------------------------- #
# Triton Kernels for W4A16 GEMV
# --------------------------------------------------------------------------- #

@triton.jit
def gemv_w4a16_kernel(
    x_ptr, w_q_ptr, scales_ptr, zeros_ptr, out_ptr,
    K, N,
    stride_w_q_k, stride_w_q_n,
    stride_scales_k, stride_scales_n,
    stride_zeros_k, stride_zeros_n,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    col_offsets = pid * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = col_offsets < N
    
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    
    for k_start in range(0, K, BLOCK_K):
        # Load x_even and x_odd
        k_even_offsets = k_start + 2 * tl.arange(0, BLOCK_K // 2)
        k_odd_offsets = k_even_offsets + 1
        
        x_even = tl.load(x_ptr + k_even_offsets).to(tl.bfloat16)
        x_odd = tl.load(x_ptr + k_odd_offsets).to(tl.bfloat16)
        
        # Load w_q packed
        packed_row_offsets = (k_start // 2) + tl.arange(0, BLOCK_K // 2)
        w_q_offsets = packed_row_offsets[:, None] * stride_w_q_k + col_offsets[None, :] * stride_w_q_n
        w_q_packed = tl.load(w_q_ptr + w_q_offsets, mask=mask_n[None, :], other=0)
        
        # Unpack lo and hi
        lo = (w_q_packed & 0xF).to(tl.bfloat16)
        hi = ((w_q_packed >> 4) & 0xF).to(tl.bfloat16)
        
        # Load scale and zero
        group_idx = k_start // 128
        scale_offsets = group_idx * stride_scales_k + col_offsets * stride_scales_n
        zero_offsets = group_idx * stride_zeros_k + col_offsets * stride_zeros_n
        
        scale = tl.load(scales_ptr + scale_offsets, mask=mask_n, other=0.0).to(tl.bfloat16)
        zero = tl.load(zeros_ptr + zero_offsets, mask=mask_n, other=0.0).to(tl.bfloat16)
        
        diff_even = lo - zero[None, :]
        diff_odd = hi - zero[None, :]
        
        dot_even = tl.sum(x_even[:, None] * diff_even, axis=0)
        dot_odd = tl.sum(x_odd[:, None] * diff_odd, axis=0)
        
        acc += (dot_even.to(tl.float32) + dot_odd.to(tl.float32)) * scale.to(tl.float32)
        
    tl.store(out_ptr + col_offsets, acc.to(tl.bfloat16), mask=mask_n)


def gemv_w4a16(x: torch.Tensor, w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor) -> torch.Tensor:
    K = x.shape[0]
    N = w_q.shape[1]
    out = torch.empty(N, dtype=torch.bfloat16, device=x.device)
    
    # RTX PRO 6000 GDDR7 tuned block size
    BLOCK_K = 128
    BLOCK_N = 64 if N % 64 == 0 else 32
    
    grid = (triton.cdiv(N, BLOCK_N),)
    
    gemv_w4a16_kernel[grid](
        x, w_q, scales, zeros, out,
        K, N,
        w_q.stride(0), w_q.stride(1),
        scales.stride(0), scales.stride(1),
        zeros.stride(0), zeros.stride(1),
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
    )
    return out


@triton.jit
def gemv_w4a16_batched_kernel(
    x_ptr, w_q_ptr, scales_ptr, zeros_ptr, out_ptr, idx_ptr,
    K, N,
    stride_x_e, stride_x_k,
    stride_w_q_e, stride_w_q_k, stride_w_q_n,
    stride_scales_e, stride_scales_k, stride_scales_n,
    stride_zeros_e, stride_zeros_k, stride_zeros_n,
    stride_out_e, stride_out_n,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_e = tl.program_id(1)
    
    expert_idx = tl.load(idx_ptr + pid_e)
    
    col_offsets = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = col_offsets < N
    
    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    
    x_expert_ptr = x_ptr + pid_e * stride_x_e
    w_q_expert_ptr = w_q_ptr + expert_idx * stride_w_q_e
    scales_expert_ptr = scales_ptr + expert_idx * stride_scales_e
    zeros_expert_ptr = zeros_ptr + expert_idx * stride_zeros_e
    
    for k_start in range(0, K, BLOCK_K):
        k_even_offsets = k_start * stride_x_k + 2 * tl.arange(0, BLOCK_K // 2) * stride_x_k
        k_odd_offsets = k_even_offsets + stride_x_k
        
        x_even = tl.load(x_expert_ptr + k_even_offsets).to(tl.bfloat16)
        x_odd = tl.load(x_expert_ptr + k_odd_offsets).to(tl.bfloat16)
        
        packed_row_offsets = (k_start // 2) + tl.arange(0, BLOCK_K // 2)
        w_q_offsets = packed_row_offsets[:, None] * stride_w_q_k + col_offsets[None, :] * stride_w_q_n
        w_q_packed = tl.load(w_q_expert_ptr + w_q_offsets, mask=mask_n[None, :], other=0)
        
        lo = (w_q_packed & 0xF).to(tl.bfloat16)
        hi = ((w_q_packed >> 4) & 0xF).to(tl.bfloat16)
        
        group_idx = k_start // 128
        scale_offsets = group_idx * stride_scales_k + col_offsets * stride_scales_n
        zero_offsets = group_idx * stride_zeros_k + col_offsets * stride_zeros_n
        
        scale = tl.load(scales_expert_ptr + scale_offsets, mask=mask_n, other=0.0).to(tl.bfloat16)
        zero = tl.load(zeros_expert_ptr + zero_offsets, mask=mask_n, other=0.0).to(tl.bfloat16)
        
        diff_even = lo - zero[None, :]
        diff_odd = hi - zero[None, :]
        
        dot_even = tl.sum(x_even[:, None] * diff_even, axis=0)
        dot_odd = tl.sum(x_odd[:, None] * diff_odd, axis=0)
        
        acc += (dot_even.to(tl.float32) + dot_odd.to(tl.float32)) * scale.to(tl.float32)
        
    out_expert_ptr = out_ptr + pid_e * stride_out_e
    tl.store(out_expert_ptr + col_offsets * stride_out_n, acc.to(tl.bfloat16), mask=mask_n)


def gemv_w4a16_batched(x: torch.Tensor, w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor, idx: torch.Tensor) -> torch.Tensor:
    E_active = idx.shape[0]
    K = x.shape[-1]
    N = w_q.shape[2]
    out = torch.empty((E_active, N), dtype=torch.bfloat16, device=x.device)
    
    BLOCK_K = 128
    BLOCK_N = 64 if N % 64 == 0 else 32
    
    grid = (triton.cdiv(N, BLOCK_N), E_active)
    
    stride_x_e = x.stride(0) if x.ndim > 1 else 0
    stride_x_k = x.stride(-1)
    
    gemv_w4a16_batched_kernel[grid](
        x, w_q, scales, zeros, out, idx,
        K, N,
        stride_x_e, stride_x_k,
        w_q.stride(0), w_q.stride(1), w_q.stride(2),
        scales.stride(0), scales.stride(1), scales.stride(2),
        zeros.stride(0), zeros.stride(1), zeros.stride(2),
        out.stride(0), out.stride(1),
        BLOCK_N=BLOCK_N,
        BLOCK_K=BLOCK_K,
    )
    return out

# Quantized Layers

def _unpack_int4(w_packed: torch.Tensor, K: int) -> torch.Tensor:
    out = torch.empty((K, w_packed.shape[1]), dtype=torch.uint8, device=w_packed.device)
    out[0::2] = w_packed & 0xF
    out[1::2] = (w_packed >> 4) & 0xF
    return out


def dequant(w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor, K: int, group: int) -> torch.Tensor:
    wu = _unpack_int4(w_q, K).to(torch.bfloat16)
    s = scales.repeat_interleave(group, dim=0)
    z = zeros.repeat_interleave(group, dim=0)
    return (wu - z) * s


class QuantLinear(nn.Module):
    def __init__(self, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        assert in_f % group == 0 and in_f % 2 == 0
        self.in_f, self.out_f, self.group = in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(ng, out_f, dtype=torch.bfloat16))

    def weight_bf(self) -> torch.Tensor:
        return dequant(self.w_q, self.scales, self.zeros, self.in_f, self.group)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Falls back to our fast fused path if 1D (batch=1 decode)
        if x.ndim == 1:
            return gemv_w4a16(x, self.w_q, self.scales, self.zeros)
        return (x.float() @ self.weight_bf().float()).to(torch.bfloat16)


class QuantExperts(nn.Module):
    def __init__(self, n: int, in_f: int, out_f: int, group: int = GROUP_SIZE):
        super().__init__()
        self.n, self.in_f, self.out_f, self.group = n, in_f, out_f, group
        ng = in_f // group
        self.register_buffer("w_q", torch.zeros(n, in_f // 2, out_f, dtype=torch.uint8))
        self.register_buffer("scales", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.zeros(n, ng, out_f, dtype=torch.bfloat16))

    def weight_bf(self, e: int) -> torch.Tensor:
        return dequant(self.w_q[e], self.scales[e], self.zeros[e], self.in_f, self.group)

# Help helpers

def _rmsnorm(x: torch.Tensor, w: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    xf = xf * torch.rsqrt(xf.pow(2).mean(-1, keepdim=True) + EPS)
    return (xf * w.float()).to(x.dtype)


def _rope_cossin(pos: int, dim: int, theta: float, device):
    inv = 1.0 / (theta ** (torch.arange(0, dim, 2, device=device, dtype=torch.float32) / dim))
    ang = pos * inv
    return torch.cos(ang), torch.sin(ang)


def _apply_rope(x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor:
    xf = x.float()
    even, odd = xf[..., 0::2], xf[..., 1::2]
    out = torch.empty_like(xf)
    out[..., 0::2] = even * cos - odd * sin
    out[..., 1::2] = odd * cos + even * sin
    return out.to(x.dtype)


def _wbf(ql) -> torch.Tensor:
    return ql.weight_bf()

class KDA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, Dk, d = cfg.kda_heads, cfg.kda_head_dim, cfg.hidden
        self.q_proj = QuantLinear(d, H * Dk, cfg.group)
        self.k_proj = QuantLinear(d, H * Dk, cfg.group)
        self.v_proj = QuantLinear(d, H * Dk, cfg.group)
        self.g_proj = QuantLinear(d, H * Dk, cfg.group)
        self.beta_proj = nn.Linear(d, H, bias=False, dtype=cfg.dtype)
        self.conv_w = nn.Parameter(torch.empty(3, H * Dk, cfg.short_conv, dtype=cfg.dtype))
        self.o_proj = QuantLinear(H * Dk, d, cfg.group)
        self.scale = Dk ** -0.5

    def _short_conv(self, val, prev, idx):
        win = torch.cat([prev, val[None]], dim=0)
        w = self.conv_w[idx].float().transpose(0, 1)
        out = (win.float() * w).sum(0)
        return F.silu(out).to(val.dtype), win[1:]

    def step(self, x, st):
        H, Dk = self.cfg.kda_heads, self.cfg.kda_head_dim
        q = gemv_w4a16(x, self.q_proj.w_q, self.q_proj.scales, self.q_proj.zeros)
        k = gemv_w4a16(x, self.k_proj.w_q, self.k_proj.scales, self.k_proj.zeros)
        v = gemv_w4a16(x, self.v_proj.w_q, self.v_proj.scales, self.v_proj.zeros)
        
        q, st["cq"] = self._short_conv(q, st["cq"], 0)
        k, st["ck"] = self._short_conv(k, st["ck"], 1)
        v, st["cv"] = self._short_conv(v, st["cv"], 2)
        q = q.view(H, Dk).float() * self.scale
        k = k.view(H, Dk).float()
        v = v.view(H, Dk).float()
        
        g_proj_out = gemv_w4a16(x, self.g_proj.w_q, self.g_proj.scales, self.g_proj.zeros)
        g = (-F.softplus(g_proj_out.float())).view(H, Dk)
        
        beta = torch.sigmoid(self.beta_proj(x).float())
        S = st["S"] * g.exp()[:, :, None]
        pred = (S * k[:, :, None]).sum(1)
        S = S + beta[:, None, None] * k[:, :, None] * (v - pred)[:, None, :]
        o = (S * q[:, :, None]).sum(1)
        st["S"] = S
        
        o_bf = o.reshape(H * Dk).to(torch.bfloat16)
        return gemv_w4a16(o_bf, self.o_proj.w_q, self.o_proj.scales, self.o_proj.zeros)


class MLA(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        H, d = cfg.mla_heads, cfg.hidden
        self.q_proj = QuantLinear(d, H * (cfg.qk_nope + cfg.qk_rope), cfg.group)
        self.kv_a = QuantLinear(d, cfg.kv_lora + cfg.qk_rope, cfg.group)
        self.kv_b = QuantLinear(cfg.kv_lora, H * (cfg.qk_nope + cfg.v_head), cfg.group)
        self.o_proj = QuantLinear(H * cfg.v_head, d, cfg.group)
        self.scale = (cfg.qk_nope + cfg.qk_rope) ** -0.5

    def step(self, x, st):
        cfg = self.cfg
        H = cfg.mla_heads
        pos = st["c_kv"].shape[0]
        
        q = gemv_w4a16(x, self.q_proj.w_q, self.q_proj.scales, self.q_proj.zeros).view(H, cfg.qk_nope + cfg.qk_rope)
        q_nope = q[:, : cfg.qk_nope]
        q_rope = q[:, cfg.qk_nope :]
        
        kv = gemv_w4a16(x, self.kv_a.w_q, self.kv_a.scales, self.kv_a.zeros)
        c_kv = kv[: cfg.kv_lora]
        k_rope = kv[cfg.kv_lora :]
        
        cos, sin = _rope_cossin(pos, cfg.qk_rope, cfg.rope_theta, x.device)
        q_rope = _apply_rope(q_rope, cos, sin)
        k_rope = _apply_rope(k_rope, cos, sin)
        
        st["c_kv"] = torch.cat([st["c_kv"], c_kv[None]], 0)
        st["k_rope"] = torch.cat([st["k_rope"], k_rope[None]], 0)
        
        # Incrementally update the latent projection cache
        if "kvb_cache" in st:
            new_kvb = gemv_w4a16(c_kv, self.kv_b.w_q, self.kv_b.scales, self.kv_b.zeros)
            st["kvb_cache"] = torch.cat([st["kvb_cache"], new_kvb[None, :]], 0)
            kvb = st["kvb_cache"].view(-1, H, cfg.qk_nope + cfg.v_head)
        else:
            # First token/step: materialize kv_b and project full context
            W_bf = _wbf(self.kv_b)
            st["kvb_cache"] = st["c_kv"] @ W_bf
            kvb = st["kvb_cache"].view(-1, H, cfg.qk_nope + cfg.v_head)
            
        k_nope = kvb[..., : cfg.qk_nope]
        v = kvb[..., cfg.qk_nope :]
        
        scores = (torch.einsum("hd,lhd->lh", q_nope, k_nope).float()
                  + torch.einsum("hd,ld->lh", q_rope, st["k_rope"]).float()) * self.scale
        p = torch.softmax(scores, dim=0).to(x.dtype)
        o = torch.einsum("lh,lhd->hd", p, v)
        
        o_bf = o.reshape(H * cfg.v_head)
        return gemv_w4a16(o_bf, self.o_proj.w_q, self.o_proj.scales, self.o_proj.zeros)


class MoE(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        d, m, E = cfg.hidden, cfg.moe_inter, cfg.n_experts
        self.router = nn.Linear(d, E, bias=False, dtype=cfg.dtype)
        self.gate = QuantExperts(E, d, m, cfg.group)
        self.up = QuantExperts(E, d, m, cfg.group)
        self.down = QuantExperts(E, m, d, cfg.group)
        self.s_gate = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_up = QuantExperts(cfg.n_shared, d, m, cfg.group)
        self.s_down = QuantExperts(cfg.n_shared, m, d, cfg.group)

    def step(self, x):
        cfg = self.cfg
        probs = torch.softmax(self.router(x).float(), dim=-1)
        w, idx = torch.topk(probs, cfg.n_active)
        w = (w / (w.sum() + 1e-9) * cfg.routed_scaling).to(x.dtype)
        
        # Batched gemv for active experts
        g = gemv_w4a16_batched(x, self.gate.w_q, self.gate.scales, self.gate.zeros, idx)
        u = gemv_w4a16_batched(x, self.up.w_q, self.up.scales, self.up.zeros, idx)
        
        hh = F.silu(g) * u
        
        routed = gemv_w4a16_batched(hh, self.down.w_q, self.down.scales, self.down.zeros, idx)
        out = (w[:, None] * routed).sum(0)
        
        # Shared expert
        sidx = torch.arange(cfg.n_shared, device=x.device)
        sg = gemv_w4a16_batched(x, self.s_gate.w_q, self.s_gate.scales, self.s_gate.zeros, sidx)
        su = gemv_w4a16_batched(x, self.s_up.w_q, self.s_up.scales, self.s_up.zeros, sidx)
        sh = F.silu(sg) * su
        
        srouted = gemv_w4a16_batched(sh, self.s_down.w_q, self.s_down.scales, self.s_down.zeros, sidx)
        out = out + srouted.sum(0)
        return out


class Block(nn.Module):
    def __init__(self, cfg: Config, kind: str):
        super().__init__()
        self.kind = kind
        self.attn_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.moe_norm = nn.Parameter(torch.ones(cfg.hidden, dtype=cfg.dtype))
        self.attn = KDA(cfg) if kind == "K" else MLA(cfg)
        self.moe = MoE(cfg)

    def step(self, x, st):
        h = x + self.attn.step(_rmsnorm(x, self.attn_norm), st)
        return h + self.moe.step(_rmsnorm(h, self.moe_norm))


class Model(nn.Module):
    def __init__(self, cfg: Config):
        super().__init__()
        self.cfg = cfg
        self.blocks = nn.ModuleList(Block(cfg, k) for k in cfg.pattern)

    def step(self, hidden, state):
        for i, blk in enumerate(self.blocks):
            hidden = blk.step(hidden, state[i])
        return hidden, state


def init_state(cfg: Config, context_len: int, seed: int) -> list:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed)
    H, Dk = cfg.kda_heads, cfg.kda_head_dim
    C = H * Dk
    state = []
    for kind in cfg.pattern:
        if kind == "K":
            state.append({
                "S": torch.randn(H, Dk, Dk, device=dev, generator=g) * 0.05,
                "cq": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "ck": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "cv": torch.randn(cfg.short_conv - 1, C, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
            })
        else:
            state.append({
                "c_kv": torch.randn(context_len, cfg.kv_lora, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
                "k_rope": torch.randn(context_len, cfg.qk_rope, device=dev, generator=g, dtype=cfg.dtype) * 0.1,
            })
    return state


def init_token(cfg: Config, seed: int) -> torch.Tensor:
    dev = torch.device("cuda:0")
    g = torch.Generator(device=dev).manual_seed(seed + 1)
    return torch.randn(cfg.hidden, device=dev, generator=g, dtype=cfg.dtype) * 0.25
