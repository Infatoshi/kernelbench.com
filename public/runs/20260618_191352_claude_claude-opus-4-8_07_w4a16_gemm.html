+"""Fused GEMV microbench: one group per CTA, atomic accumulate."""
+import torch, triton, triton.language as tl, statistics
+import reference
+
+dev="cuda"
+M,N,K = 1,12288,4096
+ref = reference.Model(M,N,K).to(dev).eval()
+wq = ref.w_q; scales = ref.scales; zeros = ref.zeros
+x = torch.randn(M,K,dtype=torch.bfloat16,device=dev)
+n_groups = K//128
+# reference output
+w_un = torch.empty((K,N),dtype=torch.uint8,device=dev)
+w_un[0::2]=wq&0xF; w_un[1::2]=(wq>>4)&0xF
+s_full=scales.repeat_interleave(128,0); z_full=zeros.repeat_interleave(128,0)
+w_bf=(w_un.to(torch.bfloat16)-z_full)*s_full
+ref_out=(x@w_bf).float()
+
+total_bytes = M*K*2+(K//2)*N+(K//128)*N*2*2+M*N*2
+
+def timeit(fn,iters=50):
+    for _ in range(20): fn()
+    torch.cuda.synchronize()
+    flush=torch.empty(256*1024*1024//4,dtype=torch.float32,device=dev)
+    ts=[]
+    for _ in range(iters):
+        flush.zero_();torch.cuda.synchronize()
+        s=torch.cuda.Event(True);e=torch.cuda.Event(True)
+        s.record();fn();e.record();torch.cuda.synchronize()
+        ts.append(s.elapsed_time(e))
+    return statistics.median(ts)
+
+@triton.jit
+def gemv(x_ptr, wq_ptr, s_ptr, z_ptr, out_ptr, N, BLOCK_N: tl.constexpr):
+    pid_n = tl.program_id(0)
+    g = tl.program_id(1)
+    n_off = pid_n*BLOCK_N + tl.arange(0,BLOCK_N)
+    nm = n_off < N
+    GH: tl.constexpr = 64
+    i = tl.arange(0,GH)
+    s = tl.load(s_ptr+g*N+n_off, mask=nm).to(tl.float32)
+    z = tl.load(z_ptr+g*N+n_off, mask=nm).to(tl.float32)
+    wq = tl.load(wq_ptr+(g*GH+i)[:,None]*N+n_off[None,:], mask=nm[None,:], other=0)
+    w_lo=(wq&0xF).to(tl.float32); w_hi=((wq>>4)&0xF).to(tl.float32)
+    xe = tl.load(x_ptr+g*128+2*i).to(tl.float32)
+    xo = tl.load(x_ptr+g*128+2*i+1).to(tl.float32)
+    # A = sum x*w_int ; B = sum x
+    A = tl.sum(xe[:,None]*w_lo + xo[:,None]*w_hi, axis=0)
+    B = tl.sum(xe+xo, axis=0)
+    part = s*(A - z*B)
+    tl.atomic_add(out_ptr+n_off, part, mask=nm)
+
