@triton.jit
def _gemv_splitk_kernel(
    x_ptr, wq_ptr, s_ptr, z_ptr, partial_ptr,
    N,
    stride_wq_k, stride_s_g,
    BN: tl.constexpr,
    HALF_GROUP: tl.constexpr,
    GROUPS_PER_SPLIT: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_k = tl.program_id(1)
    rn = pid_n * BN + tl.arange(0, BN)
    mask_n = rn < N
    acc = tl.zeros((BN,), dtype=tl.float32)
    rk = tl.arange(0, HALF_GROUP)

    g_start = pid_k * GROUPS_PER_SPLIT
    for g_off in range(GROUPS_PER_SPLIT):
        g = g_start + g_off
        k0 = g * HALF_GROUP * 2
        x_e = tl.load(x_ptr + k0 + rk * 2).to(tl.float32)
        x_o = tl.load(x_ptr + k0 + rk * 2 + 1).to(tl.float32)
        xsum = tl.sum(x_e) + tl.sum(x_o)

        wq = tl.load(
            wq_ptr + (g * HALF_GROUP + rk[:, None]) * stride_wq_k + rn[None, :],
            mask=mask_n[None, :],
            other=0,
        )
        lo = (wq & 0xF).to(tl.float32)
        hi = ((wq >> 4) & 0xF).to(tl.float32)
        wdot = (
            tl.sum(x_e[:, None] * lo, axis=0)
            + tl.sum(x_o[:, None] * hi, axis=0)
        )

        s = tl.load(s_ptr + g * stride_s_g + rn, mask=mask_n).to(tl.float32)
        z = tl.load(z_ptr + g * stride_s_g + rn, mask=mask_n).to(tl.float32)
        acc += s * (wdot - z * xsum)

    tl.store(partial_ptr + pid_k * N + rn, acc, mask=mask_n)


@triton.jit
def _gemv_reduce_kernel(
    partial_ptr, out_ptr,
    N,
    NUM_SPLITS: tl.constexpr,
    BN: tl.constexpr,
):
    pid = tl.program_id(0)
    rn = pid * BN + tl.arange(0, BN)
    mask_n = rn < N
    acc = tl.zeros((BN,), dtype=tl.float32)
    for k in range(NUM_SPLITS):
        acc += tl.load(partial_ptr + k * N + rn, mask=mask_n)
    tl.store(out_ptr + rn, acc.to(tl.bfloat16), mask=mask_n)


@triton.jit
def _gemm_kernel(
    x_ptr, wq_ptr, s_ptr, z_ptr, out_ptr,
    M, N,
    stride_x_m, stride_wq_k, stride_s_g, stride_out_m,
    BM: tl.constexpr,
    BN: tl.constexpr,
    HALF_GROUP: tl.constexpr,
    NUM_GROUPS: tl.constexpr,
):
    pid = tl.program_id(0)
    num_n_tiles = tl.cdiv(N, BN)
    pid_m = pid // num_n_tiles
    pid_n = pid % num_n_tiles

    rm = pid_m * BM + tl.arange(0, BM)
    rn = pid_n * BN + tl.arange(0, BN)
    mask_m = rm < M
    mask_n = rn < N

    acc = tl.zeros((BM, BN), dtype=tl.float32)
    rk = tl.arange(0, HALF_GROUP)

    for g in range(NUM_GROUPS):
        k0 = g * HALF_GROUP * 2

        wq = tl.load(
            wq_ptr + (g * HALF_GROUP + rk[:, None]) * stride_wq_k + rn[None, :],
            mask=mask_n[None, :],
            other=0,
        )
        s = tl.load(s_ptr + g * stride_s_g + rn, mask=mask_n)
        z = tl.load(z_ptr + g * stride_s_g + rn, mask=mask_n)

        lo = (wq & 0xF).to(tl.bfloat16)
        hi = ((wq >> 4) & 0xF).to(tl.bfloat16)
        w_lo = (lo - z[None, :]) * s[None, :]
        w_hi = (hi - z[None, :]) * s[None, :]

        x_e = tl.load(
            x_ptr + rm[:, None] * stride_x_m + k0 + rk[None, :] * 2,
            mask=mask_m[:, None],
            other=0.0,
        ).to(tl.bfloat16)
        x_o = tl.load(
            x_ptr + rm[:, None] * stride_x_m + k0 + rk[None, :] * 2 + 1,
            mask=mask_m[:, None],
            other=0.0,
        ).to(tl.bfloat16)

        acc = tl.dot(x_e, w_lo, acc)
        acc = tl.dot(x_o, w_hi, acc)

    tl.store(
        out_ptr + rm[:, None] * stride_out_m + rn[None, :],
        acc.to(tl.bfloat16),
        mask=mask_m[:, None] & mask_n[None, :],
    )


def _torch_fallback(x, w_q, scales, zeros, K, group_size):
    w = torch.empty(K, w_q.shape[1], dtype=torch.uint8, device=w_q.device)
    w[0::2] = w_q & 0xF
    w[1::2] = (w_q >> 4) & 0xF
    w_bf = w.to(torch.bfloat16)
    s = scales.repeat_interleave(group_size, dim=0)
    z = zeros.repeat_interleave(group_size, dim=0)
    return x.to(torch.bfloat16) @ ((w_bf - z) * s)
