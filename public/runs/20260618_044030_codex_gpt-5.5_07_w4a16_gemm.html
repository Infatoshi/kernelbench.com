+    x_ptr,
+    w_ptr,
+    scales_ptr,
+    zeros_ptr,
+    out_ptr,
+    N: tl.constexpr,
+    K: tl.constexpr,
+    BLOCK_N: tl.constexpr,

+    pid_n = tl.program_id(0)

+    offs_h = tl.arange(0, 64)

+        z = tl.load(zeros_ptr + g * N + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
