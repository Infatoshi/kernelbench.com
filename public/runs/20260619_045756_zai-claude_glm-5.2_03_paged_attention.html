"""Paged-attention decode kernel (FlashDecoding) for RTX 3090 (sm_86, 936 GB/s).

Memory-bound: the KV cache must be streamed once per token. Two Triton kernels:

  1. `_decode_kernel`  -- one program per (batch, kv_head, seq-split). Each
     program loads the G query heads that share its kv_head once and reuses them
     across the whole split (GQA reuse), gathers paged K/V via block_table, and
     runs an online-softmax FlashAttention pass writing a partial (O, m, l).
     Splitting the sequence gives enough programs to fill the 82 SMs even at
     small batch.

  2. `_reduce_kernel`  -- one program per (batch, query_head); merges the
     `num_splits` partials with the standard flash reduction into the final
     bf16 output.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# --- Shape knobs (overridden by check.py / benchmark.py from shapes.py) ----
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

_NUM_SMS = 82  # GA102 on the RTX 3090


def _block_n(head_dim: int, l_max: int, ns: int) -> int:
    # Per tuning on sm_86: D=128 moves 2x bytes/token so a 64-wide tile is the
    # sweet spot; D=64 prefers a narrower 32-wide tile (nw=8 hurts occupancy).
    return 32 if head_dim <= 64 else 64


def _num_warps(head_dim: int, block_n: int) -> int:
    return 4


@triton.jit
def _decode_kernel(
    Q, KV, BT, SL,
    O_part, LM_part, OUT,
    stride_qb, stride_qh,
    stride_kvblk, stride_kvp, stride_kvh,
    stride_btb,
    H: tl.constexpr, Hkv: tl.constexpr, G: tl.constexpr,
    SCALE: tl.constexpr,
    L_max,
    P: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    TOKENS_PER_SPLIT: tl.constexpr,
):
    pid_bh = tl.program_id(0)
    pid_split = tl.program_id(1)
    b = pid_bh // Hkv
    kvh = pid_bh % Hkv
    h0 = kvh * G

    seq_len_b = tl.load(SL + b)

    offs_m = tl.arange(0, BLOCK_M)
    offs_d = tl.arange(0, HEAD_DIM)
    m_valid = offs_m < G

    # Load Q (BLOCK_M, HEAD_DIM) -- padded rows zeroed, ignored downstream.
    q_ptrs = Q + b * stride_qb + (h0 + offs_m[:, None]) * stride_qh + offs_d[None, :]
    q = tl.load(q_ptrs, mask=m_valid[:, None], other=0.0)

    m_i = tl.full([BLOCK_M], -float("inf"), dtype=tl.float32)
    l_i = tl.zeros([BLOCK_M], dtype=tl.float32)
    acc = tl.zeros([BLOCK_M, HEAD_DIM], dtype=tl.float32)

    split_start = pid_split * TOKENS_PER_SPLIT
    split_end = tl.minimum(split_start + TOKENS_PER_SPLIT, L_max)

    for start_n in range(split_start, split_end, BLOCK_N):
        offs_n = start_n + tl.arange(0, BLOCK_N)
        n_valid = (offs_n < seq_len_b) & (offs_n < split_end)

        # Paged gather: logical token -> physical page via block_table.
        logical_page = offs_n // P
        intra = offs_n % P
        phys = tl.load(BT + b * stride_btb + logical_page, mask=n_valid, other=0)
        kv_off = phys * stride_kvblk + intra * stride_kvp + kvh * stride_kvh

        k_ptrs = KV + kv_off[:, None] + offs_d[None, :]
        k = tl.load(k_ptrs, mask=n_valid[:, None], other=0.0)
        v_ptrs = KV + kv_off[:, None] + (HEAD_DIM + offs_d[None, :])
        v = tl.load(v_ptrs, mask=n_valid[:, None], other=0.0)

        qk = tl.dot(q, tl.trans(k)) * SCALE  # (BLOCK_M, BLOCK_N) fp32
        qk = tl.where(n_valid[None, :], qk, -float("inf"))
        m_new = tl.maximum(m_i, tl.max(qk, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(qk - m_new[:, None])
        p_bf = p.to(v.dtype)
        l_i = alpha * l_i + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p_bf, v)
        m_i = m_new

    # Single-split fast path: normalize and write the final bf16 output
    # directly, skipping the reduce kernel and its launch.
    if NUM_SPLITS == 1:
        acc = acc / l_i[:, None]
        out_ptrs = OUT + (b * H + h0 + offs_m[:, None]) * HEAD_DIM + offs_d[None, :]
        tl.store(out_ptrs, acc.to(OUT.dtype.element_ty), mask=m_valid[:, None])
    else:
        # Write partials: O_part (B*H, NUM_SPLITS, HEAD_DIM); LM_part (B*H, NUM_SPLITS, 2)
        row = b * H + h0 + offs_m
        o_ptrs = O_part + row[:, None] * (NUM_SPLITS * HEAD_DIM) + pid_split * HEAD_DIM + offs_d[None, :]
        tl.store(o_ptrs, acc, mask=m_valid[:, None])
        lm_ptrs = LM_part + row * (NUM_SPLITS * 2) + pid_split * 2
        tl.store(lm_ptrs, m_i, mask=m_valid)
        tl.store(lm_ptrs + 1, l_i, mask=m_valid)


@triton.jit
def _reduce_kernel(
    O_part, LM_part, OUT,
    H: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    NS_PAD: tl.constexpr,
):
    pid = tl.program_id(0)  # B*H
    b = pid // H
    h = pid % H

    offs_d = tl.arange(0, HEAD_DIM)
    s_off = tl.arange(0, NS_PAD)
    s_valid = s_off < NUM_SPLITS

    base = (b * H + h)
    m_s = tl.load(LM_part + base * (NUM_SPLITS * 2) + s_off * 2 + 0,
                  mask=s_valid, other=-float("inf"))
    l_s = tl.load(LM_part + base * (NUM_SPLITS * 2) + s_off * 2 + 1,
                  mask=s_valid, other=0.0)
    m_max = tl.max(m_s, axis=0)
    alpha = tl.exp(m_s - m_max)

    o_ptrs = (O_part + base * (NUM_SPLITS * HEAD_DIM)
              + s_off[:, None] * HEAD_DIM + offs_d[None, :])
    o_s = tl.load(o_ptrs, mask=s_valid[:, None], other=0.0)
    acc = tl.sum(o_s * alpha[:, None], axis=0)
    out = acc / tl.sum(alpha * l_s, axis=0)
    tl.store(OUT + base * HEAD_DIM + offs_d, out)


def _next_pow2(x: int) -> int:
    p = 1
    while p < x:
        p <<= 1
    return p


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        self.block_n = _block_n(head_dim, seq_len, 4)
        self.num_warps = _num_warps(head_dim, self.block_n)
        # Best num_warps varies by shape (small/single-pass likes 8, large likes 4);
        # autotune picks. D=64 never benefits from >4.
        self._nw_candidates = [4, 8] if head_dim >= 128 else [2, 4]

        # One-time autotune of the sequence split count (cached afterwards).
        # KV (16-268 MB) >> the 3090's 6 MB L2, so warm-timing ranks ns the
        # same as the benchmark's L2-flushed cold timing.
        self.ns = 1
        self._autotuned = False
        ns_max = min(16, max(1, seq_len // page_size))
        self._ns_candidates = [n for n in (1, 2, 3, 4, 6, 8) if n <= ns_max]

        B, H, D = batch, num_heads, head_dim
        # Over-allocate staging for the largest candidate; the kernels index
        # with NUM_SPLITS as the stride, which stays in bounds.
        self.register_buffer("O_part", torch.empty((B * H, ns_max, D), dtype=torch.float32), persistent=False)
        self.register_buffer("LM_part", torch.empty((B * H, ns_max, 2), dtype=torch.float32), persistent=False)
        self.register_buffer("out", torch.empty((B, H, D), dtype=torch.bfloat16), persistent=False)

    def _launch(self, query, kv_cache, block_table, seq_lens, ns, num_warps):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size
        tokens_per_split = (self.seq_len + ns - 1) // ns
        _decode_kernel[(B * Hkv, ns)](
            query, kv_cache, block_table, seq_lens,
            self.O_part, self.LM_part, self.out,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0),
            H, Hkv, G, self.scale, self.seq_len,
            P, D, 16, self.block_n, ns, tokens_per_split,
            num_warps=num_warps,
        )
        if ns == 1:
            return
        _reduce_kernel[(B * H,)](
            self.O_part, self.LM_part, self.out,
            H, D, ns, _next_pow2(ns),
            num_warps=2,
        )

    def _autotune(self, query, kv_cache, block_table, seq_lens):
        best_cfg, best_t = (1, self._nw_candidates[0]), float("inf")
        for ns in self._ns_candidates:
            for nw in self._nw_candidates:
                try:
                    for _ in range(6):
                        self._launch(query, kv_cache, block_table, seq_lens, ns, nw)
                    torch.cuda.synchronize()
                    s = torch.cuda.Event(enable_timing=True)
                    e = torch.cuda.Event(enable_timing=True)
                    s.record()
                    for _ in range(16):
                        self._launch(query, kv_cache, block_table, seq_lens, ns, nw)
                    e.record()
                    torch.cuda.synchronize()
                    t = s.elapsed_time(e) / 16
                except Exception:
                    continue
                if t < best_t:
                    best_t, best_cfg = t, (ns, nw)
        self.ns, self.num_warps = best_cfg
        self._autotuned = True

    def forward(self, query, kv_cache, block_table, seq_lens):
        if not self._autotuned:
            self._autotune(query, kv_cache, block_table, seq_lens)
        self._launch(query, kv_cache, block_table, seq_lens, self.ns, self.num_warps)
        return self.out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]
