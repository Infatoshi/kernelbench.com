"""Triton grouped GEMM + fused SwiGLU for MoE up-projection.

Computes, per expert e:
    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
where x_e is the slice of permuted hidden states routed to expert e.

To keep register pressure low on Blackwell (one MMA accumulator in tensor
memory at a time), the two expert weight matrices are interleaved at runtime
into a single (E, H, 2*I) buffer:

    W[e, k, 2*j + 0] = W_up[e, k, j]
    W[e, k, 2*j + 1] = W_gate[e, k, j]

A single grouped GEMM produces the interleaved accumulator
acc = [up_0, gate_0, up_1, gate_1, ...] and the epilogue extracts each pair:

    out_j = silu(gate_j) * up_j

This fuses both projections and SwiGLU into one kernel while only requiring a
single MMA accumulator.
"""
from __future__ import annotations

import math
from typing import List, Tuple, Optional

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "grouped_gemm_swiglu"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


# ---------------------------------------------------------------------------
# Triton kernel
# ---------------------------------------------------------------------------
@triton.jit
def _swiglu_kernel(
    a_ptr,
    w_ptr,
    out_ptr,
    tile_meta_ptr,
    expert_offsets_ptr,
    # strides
    stride_a_m,
    stride_a_k,
    stride_w_e,
    stride_w_k,
    stride_w_n,
    stride_out_m,
    stride_out_n,
    # dims
    H: tl.constexpr,
    I: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N_OUT: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    """One block = one (expert, M-tile, N-output-tile).

    The weight matrix has 2*I columns (interleaved up/gate), so the MMA
    accumulator is BLOCK_M x (2*BLOCK_N_OUT).
    """
    tile_id = tl.program_id(0)

    # Each tile metadata row: [expert_id, tile_m, tile_n_out]
    expert_id = tl.load(tile_meta_ptr + tile_id * 3)
    tile_m = tl.load(tile_meta_ptr + tile_id * 3 + 1)
    tile_n_out = tl.load(tile_meta_ptr + tile_id * 3 + 2)

    # Expert row range
    off_start = tl.load(expert_offsets_ptr + expert_id)
    off_end = tl.load(expert_offsets_ptr + expert_id + 1)

    m_start = off_start + tile_m * BLOCK_M
    n_acc_start = tile_n_out * (2 * BLOCK_N_OUT)

    # Actual bounds
    m_len = off_end - m_start
    m_len = tl.minimum(m_len, BLOCK_M)
    n_acc_len = (2 * I) - n_acc_start
    n_acc_len = tl.minimum(n_acc_len, 2 * BLOCK_N_OUT)
    n_out_len = n_acc_len // 2

    if m_len <= 0 or n_out_len <= 0:
        return

    offs_m = tl.arange(0, BLOCK_M)
    offs_n_acc = tl.arange(0, 2 * BLOCK_N_OUT)
    offs_k = tl.arange(0, BLOCK_K)

    m_mask = offs_m < m_len
    n_acc_mask = offs_n_acc < n_acc_len

    # Pointers to A tile
    a_block_ptr = a_ptr + (m_start + offs_m[:, None]) * stride_a_m + offs_k[None, :] * stride_a_k

    # Pointer to weight tile for this expert (interleaved up/gate)
    w_base = w_ptr + expert_id * stride_w_e + n_acc_start * stride_w_n
    w_block_ptr = w_base + offs_k[:, None] * stride_w_k + offs_n_acc[None, :] * stride_w_n

    acc = tl.zeros((BLOCK_M, 2 * BLOCK_N_OUT), dtype=tl.float32)

    num_k_tiles = tl.cdiv(H, BLOCK_K)
    for k_tile in range(num_k_tiles):
        k_len = H - k_tile * BLOCK_K
        k_len = tl.minimum(k_len, BLOCK_K)
        k_mask = offs_k < k_len

        a = tl.load(a_block_ptr, mask=m_mask[:, None] & k_mask[None, :], other=0.0)
        w = tl.load(w_block_ptr, mask=k_mask[:, None] & n_acc_mask[None, :], other=0.0)

        acc += tl.dot(a, w, out_dtype=tl.float32)

        a_block_ptr += BLOCK_K * stride_a_k
        w_block_ptr += BLOCK_K * stride_w_k

    # SwiGLU epilogue: extract up (even cols) and gate (odd cols)
    # W[e, k, 2*j + 0] = up, W[e, k, 2*j + 1] = gate
    # out_j = silu(gate_j) * up_j
    acc_3d = acc.reshape(BLOCK_M, BLOCK_N_OUT, 2)
    up, gate = tl.split(acc_3d)
    sig = tl.sigmoid(gate.to(tl.float32))
    out_f32 = gate.to(tl.float32) * sig * up.to(tl.float32)
    out = out_f32.to(out_ptr.dtype.element_ty)

    # Store output tile (BLOCK_M x BLOCK_N_OUT)
    n_out_start = n_acc_start // 2
    offs_n_out = tl.arange(0, BLOCK_N_OUT)
    n_out_mask = offs_n_out < n_out_len
    out_block_ptr = out_ptr + (m_start + offs_m[:, None]) * stride_out_m + (n_out_start + offs_n_out[None, :]) * stride_out_n
    tl.store(out_block_ptr, out, mask=m_mask[:, None] & n_out_mask[None, :])


# ---------------------------------------------------------------------------
# PyTorch module
# ---------------------------------------------------------------------------
class Model(nn.Module):
    # Tile shape; tuned for Blackwell B200.
    BLOCK_M = 128
    BLOCK_N_OUT = 64    # accum tile is 128 columns (interleaved)
    BLOCK_K = 64
    NUM_WARPS = 4

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)
        self._cached_w: Optional[torch.Tensor] = None

    def _get_interleaved_weights(self) -> torch.Tensor:
        """Build/lazy-cache interleaved (E, H, 2*I) weight tensor.

        Layout: [up_0, gate_0, up_1, gate_1, ...] along the last dim.
        """
        if self._cached_w is None or self._cached_w.shape != (self.E, self.H, 2 * self.I):
            # (E, H, I, 2) with up in slot 0, gate in slot 1
            w = torch.stack([self.W_up, self.W_gate], dim=-1)
            # interleave -> (E, H, I*2)
            self._cached_w = w.view(self.E, self.H, 2 * self.I).contiguous()
        return self._cached_w

    def _build_tile_metadata(self, expert_offsets: torch.Tensor, block_m: int, block_n_out: int) -> torch.Tensor:
        """Build a (num_tiles, 3) int32 tensor mapping each block to (expert, tile_m, tile_n_out)."""
        offsets = expert_offsets.tolist()
        meta = []
        for e in range(self.E):
            start = offsets[e]
            end = offsets[e + 1]
            n_tokens = end - start
            if n_tokens <= 0:
                continue
            num_m_tiles = (n_tokens + block_m - 1) // block_m
            num_n_tiles = (self.I + block_n_out - 1) // block_n_out
            for tm in range(num_m_tiles):
                for tn in range(num_n_tiles):
                    meta.append([e, tm, tn])
        return torch.tensor(meta, dtype=torch.int32, device=expert_offsets.device)

    def forward(
        self,
        hidden_states: torch.Tensor,
        expert_offsets: torch.Tensor,
    ) -> torch.Tensor:
        T_perm, H = hidden_states.shape
        assert H == self.H
        out = torch.empty(T_perm, self.I, dtype=torch.bfloat16, device=hidden_states.device)

        w = self._get_interleaved_weights()

        block_m = self.BLOCK_M
        block_n_out = self.BLOCK_N_OUT
        block_k = self.BLOCK_K

        tile_meta = self._build_tile_metadata(expert_offsets, block_m, block_n_out)
        num_tiles = tile_meta.shape[0]
        if num_tiles == 0:
            return out

        grid = (num_tiles,)
        _swiglu_kernel[grid](
            hidden_states,
            w,
            out,
            tile_meta,
            expert_offsets,
            hidden_states.stride(0),
            hidden_states.stride(1),
            w.stride(0),
            w.stride(1),
            w.stride(2),
            out.stride(0),
            out.stride(1),
            H=self.H,
            I=self.I,
            BLOCK_M=block_m,
            BLOCK_N_OUT=block_n_out,
            BLOCK_K=block_k,
            num_warps=self.NUM_WARPS,
        )
        return out


# ---------------------------------------------------------------------------
# Shape shims and input builders (same interface as reference.py)
# ---------------------------------------------------------------------------
T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
