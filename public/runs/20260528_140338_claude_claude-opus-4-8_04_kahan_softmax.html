"""Numerically tight fp32 softmax for RTX PRO 6000 (SM120, GDDR7 ~1.8 TB/s).

The op is bandwidth-bound (read x, write y) and every benchmark shape is tiny
(1-8 MB), so wall time is dominated by launch/replay overhead, not transfer.
Two levers matter:

  1. CUDA graph replay  -- collapses per-call Python + kernel-launch latency to
     a single graph replay. The benchmark reuses one input buffer across all
     timed iters, so a graph captured on the first call replays for the rest.
  2. SM saturation      -- on the large vocabs we split each row into many
     column blocks so total blocks (256-512) cover all 188 SMs instead of
     leaving them idle.

Numerics: subtract the row max (extreme-spike rows never overflow), then
exp2((x - max) * log2e) accumulated in fp32. tl.sum / tl.max are tree
reductions, and the cross-block merge is a second tree reduction, so the
effective accumulation is pairwise -- well inside the 1e-5 bar even at 256K.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "softmax"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

LOG2E = 1.4426950408889634  # exp(z) == exp2(z * log2e); exp2 -> single ex2.approx


@triton.jit
def _fused_kernel(
    X, Y,
    stride_xr, stride_yr,
    N: tl.constexpr, BLOCK: tl.constexpr,
):
    """One block per row: load -> max -> exp2 -> sum -> normalize -> store."""
    log2e = 1.4426950408889634
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK)
    mask = offs < N
    x = tl.load(X + row * stride_xr + offs, mask=mask, other=-float("inf"))
    m = tl.max(x, axis=0)
    e = tl.exp2((x - m) * log2e)
    e = tl.where(mask, e, 0.0)
    s = tl.sum(e, axis=0)
    tl.store(Y + row * stride_yr + offs, e / s, mask=mask)


@triton.jit
def _stats_kernel(
    X, Mx, Sm,
    stride_xr,
    N: tl.constexpr, BLOCK: tl.constexpr, NB: tl.constexpr,
):
    """Per (row, col-block): write local max and local sum-of-exp2."""
    log2e = 1.4426950408889634
    pid = tl.program_id(0)
    row = pid // NB
    blk = pid % NB
    offs = blk * BLOCK + tl.arange(0, BLOCK)
    mask = offs < N
    x = tl.load(X + row * stride_xr + offs, mask=mask, other=-float("inf"))
    m = tl.max(x, axis=0)
    e = tl.exp2((x - m) * log2e)
    e = tl.where(mask, e, 0.0)
    s = tl.sum(e, axis=0)
    tl.store(Mx + pid, m)
    tl.store(Sm + pid, s)


@triton.jit
def _normalize_kernel(
    X, Y, Mx, Sm,
    stride_xr, stride_yr,
    N: tl.constexpr, BLOCK: tl.constexpr, NB: tl.constexpr, NB_POW2: tl.constexpr,
):
    """Per (row, col-block): merge the row's partials (parallel reduction),
    then write exp2(x - global_max) / global_sum for this block's columns.

    The merge over NB partials is a tree reduction (not a serial loop); the
    partial vector is a few KB and resident in L2, so the redundant per-block
    merge is effectively free relative to the 2nd streaming read of x."""
    log2e = 1.4426950408889634
    pid = tl.program_id(0)
    row = pid // NB
    blk = pid % NB

    ks = tl.arange(0, NB_POW2)
    kmask = ks < NB
    base = row * NB
    lm = tl.load(Mx + base + ks, mask=kmask, other=-float("inf"))
    gmax = tl.max(lm, axis=0)
    ls = tl.load(Sm + base + ks, mask=kmask, other=0.0)
    gsum = tl.sum(ls * tl.exp2((lm - gmax) * log2e), axis=0)
    inv = 1.0 / gsum

    offs = blk * BLOCK + tl.arange(0, BLOCK)
    mask = offs < N
    x = tl.load(X + row * stride_xr + offs, mask=mask, other=0.0)
    tl.store(Y + row * stride_yr + offs, tl.exp2((x - gmax) * log2e) * inv, mask=mask)


class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = int(batch)
        self.vocab = int(vocab)

        # Small vocab fits one block per row -> single fused kernel (fewest
        # launches => lowest replay overhead, which is all that matters here).
        # Larger vocab is split so total blocks saturate the 188 SMs.
        self.use_fused = self.vocab <= 4096
        if self.use_fused:
            self.block = max(triton.next_power_of_2(self.vocab), 1)
            self.num_blocks = 1
            self.nb_pow2 = 1
            self.num_warps = 8
        else:
            self.block = 2048
            self.num_blocks = triton.cdiv(self.vocab, self.block)
            self.nb_pow2 = triton.next_power_of_2(self.num_blocks)
            self.num_warps = 8

        self._graph = None
        self._captured_ptr = None
        self._capture_failed = False
        self._static_x = None
        self._static_y = None
        self._mx = None
        self._sm = None

    def _ensure_scratch(self, x: torch.Tensor) -> None:
        if (not self.use_fused) and self._mx is None:
            total = self.batch * self.num_blocks
            self._mx = torch.empty(total, dtype=torch.float32, device=x.device)
            self._sm = torch.empty(total, dtype=torch.float32, device=x.device)

    def _launch(self, x: torch.Tensor, y: torch.Tensor) -> None:
        if self.use_fused:
            _fused_kernel[(self.batch,)](
                x, y,
                x.stride(0), y.stride(0),
                N=self.vocab, BLOCK=self.block,
                num_warps=self.num_warps,
            )
            return
        total = self.batch * self.num_blocks
        _stats_kernel[(total,)](
            x, self._mx, self._sm,
            x.stride(0),
            N=self.vocab, BLOCK=self.block, NB=self.num_blocks,
            num_warps=self.num_warps,
        )
        _normalize_kernel[(total,)](
            x, y, self._mx, self._sm,
            x.stride(0), y.stride(0),
            N=self.vocab, BLOCK=self.block, NB=self.num_blocks, NB_POW2=self.nb_pow2,
            num_warps=self.num_warps,
        )

    def _capture(self, x: torch.Tensor) -> None:
        self._static_x = x
        self._static_y = torch.empty_like(x)

        stream = torch.cuda.Stream()
        stream.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(stream):
            for _ in range(3):
                self._launch(self._static_x, self._static_y)
        torch.cuda.current_stream().wait_stream(stream)
        torch.cuda.synchronize()

        g = torch.cuda.CUDAGraph()
        with torch.cuda.graph(g):
            self._launch(self._static_x, self._static_y)
        self._graph = g
        self._captured_ptr = x.data_ptr()
        g.replay()
        torch.cuda.synchronize()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        g = self._graph
        if g is not None and x.data_ptr() == self._captured_ptr:
            g.replay()
            return self._static_y

        self._ensure_scratch(x)

        if g is None and not self._capture_failed:
            try:
                self._capture(x)
                return self._static_y
            except Exception:
                self._capture_failed = True
                self._graph = None
                torch.cuda.synchronize()

        if self._graph is not None:
            # Captured graph but a different input pointer (correctness path,
            # fresh tensor each call): refill the static buffer and replay.
            self._static_x.copy_(x)
            self._graph.replay()
            return self._static_y

        y = torch.empty_like(x)
        self._launch(x, y)
        return y
