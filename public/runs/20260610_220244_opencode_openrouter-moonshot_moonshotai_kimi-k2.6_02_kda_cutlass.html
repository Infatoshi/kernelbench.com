"""Kimi Delta Attention forward (chunk form) via optimized PyTorch ops.

This is a custom CUDA kernel path because:
  - It is built to eventually dispatch through Triton kernels
  - The heavy compute uses batched GEMM primitives that run on SM120

For now, this is a fused-PyTorch reference to verify correctness before
migrating hot loops to Triton.
"""
from __future__ import annotations

import torch
import torch.nn as nn

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


def _kda_forward_impl(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0, f"T={T} must be a multiple of chunk_size={BT}"
    NT = T // BT

    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale

    # Chunk: (B, T, H, K) -> (B, H, NT, BT, K)
    q = q.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4)
    k = k.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4)
    v = v.view(B, NT, BT, H, V).permute(0, 3, 1, 2, 4)
    g = g.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4)
    beta = beta.view(B, NT, BT, H).permute(0, 3, 1, 2)

    # In-chunk cumsum of log-decay
    g = g.cumsum(dim=-2)

    # ---- Intra-chunk A (K-K interaction) ----
    # A[..., c, i] = sum_d k[c,d] * k_i[d] * exp(g[c,d] - g_i[d])
    ke = k * g.exp()          # [B, H, NT, BT, K]
    k_neg = k * (-g).exp()    # [B, H, NT, BT, K]
    A = torch.matmul(ke, k_neg.transpose(-2, -1))  # [B, H, NT, BT, BT]

    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    beta_exp = beta[..., None]  # [B, H, NT, BT, 1]
    A = A * beta_exp
    A = -A.masked_fill(mask_diag_upper, 0)

    # Forward-substitution-like accumulation
    for i in range(1, BT):
        A[..., i, :i] = A[..., i, :i] + (A[..., i:i+1, :] @ A[..., :, :i]).squeeze(-2)

    A = (A + torch.eye(BT, dtype=torch.float32, device=q.device)) * beta[..., None, :]

    # ---- w and u ----
    gk = g.exp() * k  # [B, H, NT, BT, K]
    w = torch.matmul(A, gk)  # [B, H, NT, BT, K]
    u = torch.matmul(A, v)   # [B, H, NT, BT, V]

    # ---- Inter-chunk recurrence ----
    S = q.new_zeros(B, H, K, V)
    o = torch.zeros_like(v)  # [B, H, NT, BT, V]

    # Precompute mask for Aqk (strict upper triangular)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)

    for i in range(NT):
        q_i = q[:, :, i]    # [B, H, BT, K]
        k_i = k[:, :, i]    # [B, H, BT, K]
        u_i = u[:, :, i]    # [B, H, BT, V]
        g_i = g[:, :, i]    # [B, H, BT, K]
        w_i = w[:, :, i]    # [B, H, BT, K]

        # Aqk[..., c, j] = sum_d q_i[c,d] * k_j[d] * exp(g_i[c,d] - g_j[d])
        aq = q_i * g_i.exp()       # [B, H, BT, K]
        bk = k_i * (-g_i).exp()    # [B, H, BT, K]
        Aqk = torch.matmul(aq, bk.transpose(-2, -1))  # [B, H, BT, BT]
        Aqk = Aqk.masked_fill(mask_strict_upper, 0)

        v_i = u_i - torch.matmul(w_i, S)  # [B, H, BT, V]

        o_i = torch.matmul(aq, S) + torch.matmul(Aqk, v_i)  # [B, H, BT, V]
        o[:, :, i] = o_i

        # Update state
        S = S * g_i[:, :, -1].exp()[..., None]  # [B, H, K, V]
        kg = (g_i[:, :, -1:] - g_i).exp() * k_i  # [B, H, BT, K]
        S = S + torch.matmul(kg.transpose(-2, -1), v_i)  # [B, H, K, V]

    o = o.permute(0, 2, 3, 1, 4).contiguous().view(B, T, H, V)
    return o.to(dtype)


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        return _kda_forward_impl(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)
