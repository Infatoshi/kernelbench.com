"""KDA forward (chunk form) — CUDA intra correction + optimized PyTorch inter.

Reduces inter-chunk loop to 2 bmm calls per chunk (from 4) by pre-computing
Aqk@u and (qp - Aqk@w) in batched bmms over all chunks.
"""
from __future__ import annotations

import torch
import torch.nn as nn
from einops import rearrange

_CUDA_SRC = r"""
#include <cuda_runtime.h>
#include <cuda_bf16.h>

#define C 64

__global__ void intra_correction_kernel(
    float* __restrict__ A, int stride_A,
    const float* __restrict__ beta, int stride_beta,
    int total_chunks
) {
    extern __shared__ float sA[];
    int chunk = blockIdx.x;
    if (chunk >= total_chunks) return;
    int tid = threadIdx.x;
    float* cA = A + chunk * stride_A;
    const float* cB = beta + chunk * stride_beta;

    for (int j = 0; j < C; j++) sA[tid * C + j] = cA[tid * C + j];
    __syncthreads();
    for (int i = 1; i < C; i++) {
        if (tid < i) {
            float s = sA[i * C + tid];
            for (int k = tid + 1; k < i; k++)
                s += sA[i * C + k] * sA[k * C + tid];
            sA[i * C + tid] = s;
        }
        __syncthreads();
    }
    for (int j = 0; j < C; j++) {
        float v = sA[tid * C + j];
        if (tid == j) v += 1.0f;
        v *= cB[j];
        cA[tid * C + j] = v;
    }
}

extern "C" void launch_intra_correction(
    float* A, int stride_A,
    const float* beta, int stride_beta,
    int total_chunks
) {
    int shmem = C * C * sizeof(float);
    intra_correction_kernel<<<total_chunks, C, shmem>>>(
        A, stride_A, beta, stride_beta, total_chunks
    );
}
"""

def _intra_correction(A, beta):
    if _cuda_module is not None:
        return _cuda_module.kda_correction(A.contiguous(), beta.contiguous())
    C = A.size(-1)
    for i in range(1, C):
        p = A[:, i].clone()
        A[:, i, :i] = A[:, i, :i].clone() + (p.unsqueeze(-1) * A[:, :, :i]).sum(-2)
    A = (A + torch.eye(C, dtype=A.dtype, device=A.device)) * beta.unsqueeze(1)
    return A


def _kda_chunk_forward(q, k, v, g, beta, scale, chunk_size=64):
    dtype = v.dtype
    B, T, H, K_ = q.shape
    V = v.shape[-1]
    BT = chunk_size
    NT = T // BT

    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale
    q = rearrange(q, "b (n c) h d -> b h n c d", c=BT)
    k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
    v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
    g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
    beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)
    g = g.cumsum(-2)

    k_plus = torch.exp(g) * k
    k_minus = torch.exp(-g) * k

    A = torch.matmul(k_plus, k_minus.transpose(-2, -1))
    A = A * beta.unsqueeze(-1)
    mask = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=A.device), diagonal=0)
    A = A.masked_fill(mask, 0.0)
    A = -A
    A_flat = A.reshape(B * H * NT, BT, BT)
    A_flat = _intra_correction(A_flat, beta.reshape(B * H * NT, BT))
    A = A_flat.reshape(B, H, NT, BT, BT)

    w = torch.matmul(A, k_plus)
    u = torch.matmul(A, v)
    q_plus = torch.exp(g) * q

    # Pre-compute Aqk + derived matrices
    Aqk_all = torch.matmul(q_plus, k_minus.transpose(-2, -1))
    mask2 = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=Aqk_all.device), diagonal=1)
    Aqk_all = Aqk_all.masked_fill(mask2, 0.0)

    g_last_all = g[:, :, :, -1:, :]
    decay_all = g_last_all.exp()
    km_decayed = k_minus * decay_all
    decay_all = decay_all.squeeze(-2).unsqueeze(-1)

    # q_eff = qp - Aqk@w,  pre_u = Aqk@u  (batched over all chunks)
    bh_nt = B * H * NT
    q_eff = q_plus.reshape(bh_nt, BT, K_) - \
        torch.bmm(Aqk_all.reshape(bh_nt, BT, BT), w.reshape(bh_nt, BT, K_))
    pre_u = torch.bmm(Aqk_all.reshape(bh_nt, BT, BT), u.reshape(bh_nt, BT, V))

    q_eff = q_eff.reshape(B, H, NT, BT, K_)
    pre_u = pre_u.reshape(B, H, NT, BT, V)
    o = torch.zeros_like(v)

    # Inter-chunk: 2 bmm/iter (was 4)
    S = torch.zeros(B, H, K_, V, dtype=torch.float32, device=q.device)
    for n in range(NT):
        qe_n = q_eff[:, :, n]
        w_n = w[:, :, n]
        u_n = u[:, :, n]
        pu_n = pre_u[:, :, n]
        dec_n = decay_all[:, :, n]
        kd_n = km_decayed[:, :, n]

        v_bar = u_n - torch.matmul(w_n, S)
        o[:, :, n] = torch.matmul(qe_n, S) + pu_n
        S = S * dec_n + torch.matmul(kd_n.transpose(-2, -1), v_bar)

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o.to(dtype)
