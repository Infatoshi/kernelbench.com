+
+Theory: w bf16 has 8 sig bits. u = w * s (s = power of 2) -> u_hi = e4m3(u) captures

+from pathlib import Path

+    wf = w_bf16.float()
+    amax = wf.abs().max()
+    # scale to put max at <= 448 (e4m3 max), power of 2
+    s = torch.pow(2.0, torch.floor(torch.log2(448.0 / amax)))
+    u = wf * s  # exact in fp32 (bf16 mantissa * pow2)
+    u_hi = u.to(torch.float8_e4m3fn)
+    r = u - u_hi.float()  # exact in fp32
+    u_lo = r.to(torch.float8_e4m3fn)

+    diff = (ref_f - sol_f).abs()
