        cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <c10/cuda/CUDAException.h>
#include <cuda.h>
#include <cuda_bf16.h>
#include <cuda_runtime.h>

namespace {

constexpr int BT = 64;
constexpr int D = 128;
constexpr int VT = 4;

__device__ __forceinline__ float bf16_to_float(const __nv_bfloat16 x) {
  return __bfloat162float(x);
}

__device__ __forceinline__ __nv_bfloat16 float_to_bf16(const float x) {
  return __float2bfloat16(x);
}

__global__ void prepare_kernel(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const __nv_bfloat16* __restrict__ v,
    const float* __restrict__ g,
    const __nv_bfloat16* __restrict__ beta,
    float* __restrict__ elast,
    float* __restrict__ qe,
    float* __restrict__ dk,
    float* __restrict__ u,
    float* __restrict__ w,
    float* __restrict__ aqk,
    int B,
    int T,
    int H,
    float scale) {
  extern __shared__ float smem[];
  float* ksh = smem;
  float* esh = ksh + BT * D;
  float* qsh = esh + BT * D;
  float* ash = qsh + BT * D;
  float* bsh = ash + BT * BT;
  float* row = bsh + BT;

  const int NT = T / BT;
  const int cid = blockIdx.x;
  const int n = cid % NT;
  const int h = (cid / NT) % H;
  const int b = cid / (NT * H);
  const int tid = threadIdx.x;

  for (int idx = tid; idx < BT * D; idx += blockDim.x) {
    const int r = idx / D;
    const int d = idx - r * D;
    const int t = n * BT + r;
    const int off = ((b * T + t) * H + h) * D + d;
    ksh[idx] = bf16_to_float(k[off]);
    qsh[idx] = bf16_to_float(q[off]) * scale;
  }

  for (int d = tid; d < D; d += blockDim.x) {
    float acc = 0.0f;
    for (int r = 0; r < BT; ++r) {
      const int t = n * BT + r;
      const int off = ((b * T + t) * H + h) * D + d;
      acc += g[off];
      const float ev = expf(acc);
      esh[r * D + d] = ev;
      if (r == BT - 1) {
        const int eoff = ((b * H + h) * NT + n) * D + d;
        elast[eoff] = ev;
      }
    }
  }

  for (int r = tid; r < BT; r += blockDim.x) {
    const int t = n * BT + r;
    const int off = (b * T + t) * H + h;
    bsh[r] = bf16_to_float(beta[off]);
  }
  __syncthreads();

  for (int idx = tid; idx < BT * D; idx += blockDim.x) {
    const int r = idx / D;
    const int d = idx - r * D;
    const int t = n * BT + r;
    const int off = ((b * T + t) * H + h) * D + d;
    qe[off] = qsh[idx] * esh[idx];
    dk[off] = (esh[(BT - 1) * D + d] / esh[idx]) * ksh[idx];
  }

  for (int idx = tid; idx < BT * BT; idx += blockDim.x) {
    const int r = idx / BT;
    const int c = idx - r * BT;
    float val = 0.0f;
    if (c < r) {
      float sum = 0.0f;
      #pragma unroll 4
      for (int d = 0; d < D; ++d) {
        const float ratio = esh[r * D + d] / esh[c * D + d];
        sum += ksh[r * D + d] * ratio * ksh[c * D + d];
      }
      val = -bsh[r] * sum;
    }
    ash[idx] = val;
  }
  __syncthreads();

  for (int r = 1; r < BT; ++r) {
    for (int j = tid; j < BT; j += blockDim.x) {
      row[j] = ash[r * BT + j];
    }
    __syncthreads();
    for (int c = tid; c < r; c += blockDim.x) {
      float acc = 0.0f;
      #pragma unroll
      for (int j = 0; j < BT; ++j) {
        acc += row[j] * ash[j * BT + c];
      }
      ash[r * BT + c] = row[c] + acc;
    }
    __syncthreads();
  }

  for (int idx = tid; idx < BT * BT; idx += blockDim.x) {
    const int r = idx / BT;
    const int c = idx - r * BT;
    float val = ash[idx];
    if (r == c) {
      val += 1.0f;
    }
    ash[idx] = val * bsh[c];
  }
  __syncthreads();

  for (int idx = tid; idx < BT * D; idx += blockDim.x) {
    const int r = idx / D;
    const int d = idx - r * D;
    const int t = n * BT + r;
    float su = 0.0f;
    float sw = 0.0f;
    for (int j = 0; j <= r; ++j) {
      const float a = ash[r * BT + j];
      const int tj = n * BT + j;
      const int voff = ((b * T + tj) * H + h) * D + d;
      su += a * bf16_to_float(v[voff]);
      sw += a * esh[j * D + d] * ksh[j * D + d];
    }
    const int out = ((b * T + t) * H + h) * D + d;
    u[out] = su;
    w[out] = sw;
  }

  const int aq_base = ((b * H + h) * NT + n) * BT * BT;
  for (int idx = tid; idx < BT * BT; idx += blockDim.x) {
    const int r = idx / BT;
    const int c = idx - r * BT;
    float val = 0.0f;
    if (c <= r) {
      float sum = 0.0f;
      #pragma unroll 4
      for (int d = 0; d < D; ++d) {
        const float ratio = esh[r * D + d] / esh[c * D + d];
        sum += qsh[r * D + d] * ratio * ksh[c * D + d];
      }
      val = sum;
    }
    aqk[aq_base + idx] = val;
  }
}

__global__ void recur_kernel(
    const float* __restrict__ elast,
    const float* __restrict__ qe,
    const float* __restrict__ dk,
    const float* __restrict__ u,
    const float* __restrict__ w,
    const float* __restrict__ aqk,
    __nv_bfloat16* __restrict__ out,
    int B,
    int T,
    int H) {
  extern __shared__ float smem[];
  float* S = smem;
  float* vi = S + D * VT;

  const int NT = T / BT;
  const int tile = blockIdx.x % (D / VT);
  const int h = (blockIdx.x / (D / VT)) % H;
  const int b = blockIdx.x / ((D / VT) * H);
  const int v0 = tile * VT;
  const int tid = threadIdx.x;

  for (int idx = tid; idx < D * VT; idx += blockDim.x) {
    S[idx] = 0.0f;
  }
  __syncthreads();

  for (int n = 0; n < NT; ++n) {
    for (int idx = tid; idx < BT * VT; idx += blockDim.x) {
      const int r = idx / VT;
      const int vv = idx - r * VT;
      const int t = n * BT + r;
      const int voff = ((b * T + t) * H + h) * D + (v0 + vv);
      float acc = u[voff];
      #pragma unroll 4
      for (int d = 0; d < D; ++d) {
        const int woff = ((b * T + t) * H + h) * D + d;
        acc -= w[woff] * S[d * VT + vv];
      }
      vi[idx] = acc;
    }
    __syncthreads();

    const int aq_base = ((b * H + h) * NT + n) * BT * BT;
    for (int idx = tid; idx < BT * VT; idx += blockDim.x) {
      const int r = idx / VT;
      const int vv = idx - r * VT;
      const int t = n * BT + r;
      float term1 = 0.0f;
      #pragma unroll 4
      for (int d = 0; d < D; ++d) {
        const int qoff = ((b * T + t) * H + h) * D + d;
        term1 += qe[qoff] * S[d * VT + vv];
      }
      float term2 = 0.0f;
      #pragma unroll
      for (int j = 0; j <= r; ++j) {
        term2 += aqk[aq_base + r * BT + j] * vi[j * VT + vv];
      }
      const int ooff = ((b * T + t) * H + h) * D + (v0 + vv);
      out[ooff] = float_to_bf16(term1 + term2);
    }
    __syncthreads();

    for (int idx = tid; idx < D * VT; idx += blockDim.x) {
      const int d = idx / VT;
      const int vv = idx - d * VT;
      const int last_off = ((b * H + h) * NT + n) * D + d;
      const float last_decay = elast[last_off];
      float acc = S[idx] * last_decay;
      #pragma unroll
      for (int r = 0; r < BT; ++r) {
        const int t = n * BT + r;
        const int off = ((b * T + t) * H + h) * D + d;
        acc += dk[off] * vi[r * VT + vv];
      }
      S[idx] = acc;
    }
    __syncthreads();
  }
}

}  // namespace

torch::Tensor kda_forward_cuda(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v,
    torch::Tensor g,
    torch::Tensor beta,
    double scale) {
  TORCH_CHECK(q.is_cuda(), "q must be CUDA");
  TORCH_CHECK(q.scalar_type() == at::kBFloat16, "q must be bf16");
  TORCH_CHECK(k.scalar_type() == at::kBFloat16, "k must be bf16");
  TORCH_CHECK(v.scalar_type() == at::kBFloat16, "v must be bf16");
  TORCH_CHECK(beta.scalar_type() == at::kBFloat16, "beta must be bf16");
  TORCH_CHECK(g.scalar_type() == at::kFloat, "g must be fp32");
  const int B = q.size(0);
  const int T = q.size(1);
  const int H = q.size(2);
  TORCH_CHECK(q.size(3) == D && k.size(3) == D && v.size(3) == D, "D must be 128");
  TORCH_CHECK(T % BT == 0, "T must be divisible by 64");

  auto fopts = torch::TensorOptions().device(q.device()).dtype(torch::kFloat32);
  auto elast = torch::empty({B, H, T / BT, D}, fopts);
  auto qe = torch::empty({B, T, H, D}, fopts);
  auto dk = torch::empty({B, T, H, D}, fopts);
  auto u = torch::empty({B, T, H, D}, fopts);
  auto w = torch::empty({B, T, H, D}, fopts);
  auto aqk = torch::empty({B, H, T / BT, BT, BT}, fopts);
  auto out = torch::empty_like(v);

  const int blocks_prepare = B * H * (T / BT);
  const int threads_prepare = 256;
  const int threads_recur = 512;
  const size_t smem_prepare = (BT * D * 3 + BT * BT + BT + BT) * sizeof(float);
  const size_t smem_recur = (D * VT + BT * VT) * sizeof(float);

  cudaStream_t stream = at::cuda::getCurrentCUDAStream();
  cudaFuncSetAttribute(prepare_kernel, cudaFuncAttributeMaxDynamicSharedMemorySize, smem_prepare);
  cudaFuncSetAttribute(recur_kernel, cudaFuncAttributeMaxDynamicSharedMemorySize, smem_recur);

  prepare_kernel<<<blocks_prepare, threads_prepare, smem_prepare, stream>>>(
      reinterpret_cast<const __nv_bfloat16*>(q.data_ptr<at::BFloat16>()),
      reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>()),
      reinterpret_cast<const __nv_bfloat16*>(v.data_ptr<at::BFloat16>()),
      g.data_ptr<float>(),
      reinterpret_cast<const __nv_bfloat16*>(beta.data_ptr<at::BFloat16>()),
      elast.data_ptr<float>(),
      qe.data_ptr<float>(),
      dk.data_ptr<float>(),
      u.data_ptr<float>(),
      w.data_ptr<float>(),
      aqk.data_ptr<float>(),
      B,
      T,
      H,
      static_cast<float>(scale));
  C10_CUDA_KERNEL_LAUNCH_CHECK();

  const int blocks_recur = B * H * (D / VT);
  recur_kernel<<<blocks_recur, threads_recur, smem_recur, stream>>>(
      elast.data_ptr<float>(),
      qe.data_ptr<float>(),
      dk.data_ptr<float>(),
      u.data_ptr<float>(),
      w.data_ptr<float>(),
      aqk.data_ptr<float>(),
      reinterpret_cast<__nv_bfloat16*>(out.data_ptr<at::BFloat16>()),
      B,
      T,
      H);
  C10_CUDA_KERNEL_LAUNCH_CHECK();
  return out;
}

def _torch_forward(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    BT = chunk_size
    NT = T // BT

    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale

    q = rearrange(q, "b (n c) h d -> b h n c d", c=BT)
    k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
    v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
    g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
    beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)

    g = g.cumsum(-2)

    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    A = torch.zeros(*q.shape[:-1], BT, dtype=torch.float32, device=q.device)
    for i in range(BT):
        k_i = k[..., i, :]
        g_i = g[..., i:i + 1, :]
        A[..., i] = torch.einsum("... c d, ... d -> ... c", k * (g - g_i).exp(), k_i)
    A = A * beta[..., None]
    A = -A.masked_fill(mask_diag_upper, 0)

    for i in range(1, BT):
        A[..., i, :i] = A[..., i, :i].clone() + (A[..., i, :, None].clone() * A[..., :, :i].clone()).sum(-2)
    A = (A + torch.eye(BT, dtype=torch.float32, device=q.device)) * beta[..., None, :]

    w = A @ (g.exp() * k)
    u = A @ v

    S = q.new_zeros(B, H, K, v.shape[-1])
    o = torch.zeros_like(v)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)
    for i in range(NT):
        q_i, k_i, u_i, g_i, w_i = q[:, :, i], k[:, :, i], u[:, :, i], g[:, :, i], w[:, :, i]
        Aqk = torch.zeros(B, H, BT, BT, dtype=torch.float32, device=q.device)
        for j in range(BT):
            k_j = k[:, :, i, j]
            g_j = g[:, :, i, j:j + 1, :]
            Aqk[..., j] = torch.einsum("... c d, ... d -> ... c", q_i * (g_i - g_j).exp(), k_j)
        Aqk = Aqk.masked_fill(mask_strict_upper, 0)
        v_i = u_i - w_i @ S
        o[:, :, i] = (q_i * g_i.exp()) @ S + Aqk @ v_i
        S = S * rearrange(g_i[:, :, -1].exp(), "b h k -> b h k 1")
        S = S + rearrange((g_i[:, :, -1:] - g_i).exp() * k_i, "b h c k -> b h k c") @ v_i

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o.to(dtype)


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        if q.is_cuda and q.dtype is torch.bfloat16 and q.shape[-1] == 128 and v.shape[-1] == 128:
            try:
                ext = _load_ext()
                return ext.kda_forward(
                    q.contiguous(),
                    k.contiguous(),
                    v.contiguous(),
                    g.contiguous(),
                    beta.contiguous(),
                    self.scale,
                )
            except Exception:
                pass
        return _torch_forward(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)

def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
