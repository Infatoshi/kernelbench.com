"""Fused variable-length grouped GEMM + SwiGLU for MoE up-projection.

Per expert e, with x_e = permuted hidden states for expert e (rows
[offsets[e]:offsets[e+1]]):

    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])

Strategy: a single Triton grouped GEMM whose tiles are assigned to experts via
a precomputed per-tile (expert, row-start, row-count) table built from the
contiguous expert_offsets. Each program owns one (M-tile, N-block) and computes
BOTH the gate and up partial products against the same loaded x tile (x read
once, two weight tiles), accumulating into two fp32 accumulators, then fuses
silu(gate)*up in the epilogue and writes the final bf16 output directly.

@triton.jit
def _fused_swiglu_grouped_gemm_kernel(
    x_ptr,
    wg_ptr,
    wu_ptr,
    out_ptr,
    m_start_ptr,
    m_n_ptr,
    m_expert_ptr,
    H: tl.constexpr,
    N: tl.constexpr,
    stride_xm,
    stride_xk,
    stride_we,
    stride_wk,
    stride_wn,
    stride_om,
    stride_on,
    M_tiles,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(M_tiles - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + (pid % num_pid_in_group) % group_size_m
    pid_n = (pid % num_pid_in_group) // group_size_m

    m_start = tl.load(m_start_ptr + pid_m)
    m_n = tl.load(m_n_ptr + pid_m)
    expert = tl.load(m_expert_ptr + pid_m)

    offs_m = tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    row_mask = offs_m < m_n

    x_ptrs = x_ptr + (m_start + offs_m[:, None]) * stride_xm + offs_k[None, :] * stride_xk
    w_off = expert * stride_we
    wg_ptrs = wg_ptr + w_off + offs_k[:, None] * stride_wk + offs_n[None, :] * stride_wn
    wu_ptrs = wu_ptr + w_off + offs_k[:, None] * stride_wk + offs_n[None, :] * stride_wn

    acc_g = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc_u = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    rm = row_mask[:, None]
    for _k in range(0, tl.cdiv(H, BLOCK_K)):
        a = tl.load(x_ptrs, mask=rm, other=0.0)
        bg = tl.load(wg_ptrs)
        bu = tl.load(wu_ptrs)
        acc_g = tl.dot(a, bg, acc=acc_g)
        acc_u = tl.dot(a, bu, acc=acc_u)
        x_ptrs += BLOCK_K * stride_xk
        wg_ptrs += BLOCK_K * stride_wk
        wu_ptrs += BLOCK_K * stride_wk

    res = (acc_g * tl.sigmoid(acc_g)) * acc_u
    out_ptrs = out_ptr + (m_start + offs_m[:, None]) * stride_om + offs_n[None, :] * stride_on
    tl.store(out_ptrs, res.to(tl.bfloat16), mask=rm)


def _build_tile_meta(offsets_cpu, E, bm, device):
    """Per-tile (row_start, row_count, expert_id) for the contiguous grouped layout."""
    starts, ns, experts = [], [], []
    for e in range(E):
        s = int(offsets_cpu[e])
        en = int(offsets_cpu[e + 1])
        n = en - s
        if n <= 0:
            continue
        n_tiles = (n + bm - 1) // bm
        for t in range(n_tiles):
            ts = s + t * bm
            starts.append(ts)
            ns.append(min(bm, en - ts))
            experts.append(e)
    if not starts:
        starts = [0]
        ns = [0]
        experts = [0]
    m_start = torch.tensor(starts, dtype=torch.int32, device=device)
    m_n = torch.tensor(ns, dtype=torch.int32, device=device)
    m_expert = torch.tensor(experts, dtype=torch.int32, device=device)
    return m_start, m_n, m_expert, len(starts)

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets
