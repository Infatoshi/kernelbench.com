"""
Kimi Delta Attention (KDA) forward, chunk form.

Intra-chunk computation via forward substitution (M/N recurrence, no
cumulative inverse or shared memory needed).  Inter-chunk recurrence
via a custom Triton kernel.

Target: NVIDIA RTX PRO 6000 (SM120 Blackwell, compute capability 12.0).
"""

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


# =========================================================================
# Inter-chunk Triton kernel
# =========================================================================
# Grid: (B * H,) — one program per (batch, head) pair.
# Keeps state S (K × V) in registers across the sequential scan over N chunks.
#
# For each chunk n:
#   1. Aqk = tril(Q_exp @ K_decay^T)
#   2. v_corrected = u - w @ S
#   3. o_chunk = Q_exp @ S + Aqk @ v_corrected
#   4. S = S * diag(exp(g_last)) + k_decayed^T @ v_corrected

@triton.jit
def _inter_chunk_kernel(
    q_ptr, k_ptr, g_ptr, w_ptr, u_ptr, o_ptr,
    stride_q0, stride_q1, stride_q2, stride_q3,
    stride_g0, stride_g1, stride_g2, stride_g3,
    stride_w0, stride_w1, stride_w2, stride_w3,
    stride_u0, stride_u1, stride_u2, stride_u3,
    stride_o0, stride_o1, stride_o2, stride_o3,
    H,
    N: tl.constexpr,
    BT: tl.constexpr, K: tl.constexpr, V: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // H
    h = pid % H

    # Flattened tensor: dim0 = B*H, dim1 = N, dim2 = BT, dim3 = K (or V)
    # stride0 = stride for dim0, stride1 = for dim1, stride2 = for dim2, stride3 = for dim3
    bh = b * H + h
    bh_base = bh * stride_q0

    offs_c = tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = tl.arange(0, V)

    # State S (K × V) — lives in registers throughout the scan
    S_reg = tl.zeros((K, V), dtype=tl.float32)

    for n_val in tl.static_range(N):
        bn_base = bh_base + n_val * stride_q1

        q_n = q_ptr + bn_base
        k_n = k_ptr + bn_base
        g_n = g_ptr + bn_base
        w_n = w_ptr + bn_base
        u_n = u_ptr + bn_base
        o_n = o_ptr + bn_base

        # --- load chunk data (BT × K  /  BT × V) ---
        # All tensors are fp32 in Python. Load as fp32.
        # offset = c * stride2 + k * stride3  (stride2=BT stride, stride3=last-dim stride)
        q_f32 = tl.load(
            q_n + offs_c[:, None] * stride_q2 + offs_k[None, :] * stride_q3,
            mask=(offs_c[:, None] < BT) & (offs_k[None, :] < K),
            other=0.0,
        ).to(tl.float32)

        k_f32 = tl.load(
            k_n + offs_c[:, None] * stride_q2 + offs_k[None, :] * stride_q3,
            mask=(offs_c[:, None] < BT) & (offs_k[None, :] < K),
            other=0.0,
        ).to(tl.float32)

        g_f32 = tl.load(
            g_n + offs_c[:, None] * stride_g2 + offs_k[None, :] * stride_g3,
            mask=(offs_c[:, None] < BT) & (offs_k[None, :] < K),
            other=0.0,
        ).to(tl.float32)

        w_f32 = tl.load(
            w_n + offs_c[:, None] * stride_w2 + offs_k[None, :] * stride_w3,
            mask=(offs_c[:, None] < BT) & (offs_k[None, :] < K),
            other=0.0,
        ).to(tl.float32)

        u_f32 = tl.load(
            u_n + offs_c[:, None] * stride_u2 + offs_v[None, :] * stride_u3,
            mask=(offs_c[:, None] < BT) & (offs_v[None, :] < V),
            other=0.0,
        ).to(tl.float32)

        # ---- 1. Aqk = tril(Q_exp @ K_decay^T) ----
        q_exp   = q_f32 * tl.exp(g_f32)            # (BT, K)
        k_decay = k_f32 * tl.exp(-g_f32)            # (BT, K)
        Aqk_f   = tl.dot(q_exp, tl.trans(k_decay))  # (BT, BT)
        Aqk     = tl.where(offs_c[None, :] > offs_c[:, None], 0.0, Aqk_f)

        # ---- 2. v_corrected = u - w @ S_reg ----
        wS     = tl.dot(w_f32, S_reg)               # (BT, V)
        v_corr = u_f32 - wS

        # ---- 3. o_chunk = q_exp @ S_reg + Aqk @ v_corr ----
        qS      = tl.dot(q_exp, S_reg)              # (BT, V)
        Av      = tl.dot(Aqk, v_corr)               # (BT, V)
        o_chunk = qS + Av

        # store as fp32
        tl.store(o_n + offs_c[:, None] * stride_o2 + offs_v[None, :] * stride_o3,
                 o_chunk,
                 mask=(offs_c[:, None] < BT) & (offs_v[None, :] < V))

        # ---- 4. update S_reg ----
        # Load g_last = g[BT-1, :] from global memory directly
        g_last  = tl.load(
            g_n + (BT - 1) * stride_g2 + offs_k * stride_g3,
            mask=offs_k < K, other=0.0,
        ).to(tl.float32)                            # (K,)
        gl_exp  = tl.exp(g_last)

        S_reg = S_reg * gl_exp[:, None]             # row-wise scaling

        k_decayed = k_f32 * tl.exp(g_last[None, :] - g_f32)   # (BT, K)
        S_reg += tl.dot(tl.trans(k_decayed), v_corr)          # (K, BT) @ (BT, V)


# =========================================================================
# Model
# =========================================================================

class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        B, T, H = q.shape[0], q.shape[1], q.shape[2]
        Kdim = q.shape[3]
        Vdim = v.shape[3]
        BT = self.chunk_size
        NT = T // BT

        device = q.device
        dtype_out = v.dtype

        # upcast
        q = q.float() * self.scale
        k = k.float()
        v = v.float()
        beta = beta.float()
        g = g.float()

        # Reshape: (B, T, H, D) -> (B, H, N, BT, D)
        def _r5(x, D):
            return x.reshape(B, NT, BT, H, D).permute(0, 3, 1, 2, 4).contiguous()

        def _r4(x):
            return x.reshape(B, NT, BT, H).permute(0, 3, 1, 2).contiguous()

        q_r   = _r5(q, Kdim)
        k_r   = _r5(k, Kdim)
        v_r   = _r5(v, Vdim)
        g_r   = _r5(g, Kdim)
        beta_r = _r4(beta)

        # In-chunk cumsum on g
        g_r = g_r.cumsum(dim=-2)

        # ---------------------------------------------------------------
        # Intra-chunk: build A matrix and compute w = A @ K_exp, u = A @ v
        #
        # A = (I - L)^{-1} * diag(beta)  where L = -tril(K_decay @ K_exp^T) * beta_row
        # ---------------------------------------------------------------

        B_flat = B * H * NT
        k_flat = k_r.reshape(B_flat, BT, Kdim)     # (B_flat, BT, K)
        v_flat = v_r.reshape(B_flat, BT, Vdim)
        g_flat = g_r.reshape(B_flat, BT, Kdim)
        beta_flat = beta_r.reshape(B_flat, BT)

        K_decay = k_flat * torch.exp(-g_flat)       # (B_flat, BT, K)
        K_exp   = k_flat * torch.exp(g_flat)        # (B_flat, BT, K)

        # S = K_decay @ K_exp^T   (B_flat, BT, BT)
        S = torch.bmm(K_decay, K_exp.transpose(1, 2))

        # A_init = -tril(S) * beta[i]  (row scaling)
        # Use strict lower triangular mask (j < i)
        mask_tril = torch.tril(torch.ones(BT, BT, device=device), diagonal=-1).bool()
        mask_tril = mask_tril.unsqueeze(0)  # (1, BT, BT) for broadcasting
        beta_row = beta_flat.unsqueeze(-1)  # (B_flat, BT, 1)
        A = -S.masked_fill(~mask_tril, 0.0) * beta_row

        # Cumulative inverse: A += A^2 + A^3 + ...  (sequential over rows)
        for i in range(1, BT):
            # A[i, :i] += (A[i, :] @ A[:, :i])
            contrib = torch.bmm(A[:, i:i+1, :], A[:, :, :i])  # (B_flat, 1, i)
            A[:, i, :i] += contrib[:, 0, :]

        # A = (A + I) * beta[j]  (column scaling)
        A = A + torch.eye(BT, device=device).unsqueeze(0)     # add identity
        beta_col = beta_flat.unsqueeze(1)                      # (B_flat, 1, BT)
        A = A * beta_col

        # w = A @ K_exp,  u = A @ v
        w_flat = torch.bmm(A, K_exp)          # (B_flat, BT, K)
        u_flat = torch.bmm(A, v_flat)          # (B_flat, BT, V)

        # Reshape w, u back to (B, H, NT, BT, D)
        w = w_flat.reshape(B, H, NT, BT, Kdim)
        u = u_flat.reshape(B, H, NT, BT, Vdim)

        # ---------------------------------------------------------------
        # Inter-chunk: Triton kernel
        # ---------------------------------------------------------------

        q_f = q_r.reshape(B * H, NT, BT, Kdim).contiguous()
        k_f = k_r.reshape(B * H, NT, BT, Kdim).contiguous()
        g_f = g_r.reshape(B * H, NT, BT, Kdim).contiguous()
        w_f = w.reshape(B * H, NT, BT, Kdim).contiguous()
        u_f = u.reshape(B * H, NT, BT, Vdim).contiguous()
        o_f = torch.empty(B * H, NT, BT, Vdim, dtype=torch.float32, device=device)

        grid_inter = (B * H,)
        _inter_chunk_kernel[grid_inter](
            q_f, k_f, g_f, w_f, u_f, o_f,
            q_f.stride(0), q_f.stride(1), q_f.stride(2), q_f.stride(3),
            g_f.stride(0), g_f.stride(1), g_f.stride(2), g_f.stride(3),
            w_f.stride(0), w_f.stride(1), w_f.stride(2), w_f.stride(3),
            u_f.stride(0), u_f.stride(1), u_f.stride(2), u_f.stride(3),
            o_f.stride(0), o_f.stride(1), o_f.stride(2), o_f.stride(3),
            H, NT, BT=BT, K=Kdim, V=Vdim,
            num_warps=4, num_stages=1,
        )

        # Reshape output  (B*H, NT, BT, V) -> (B, T, H, V)
        o = o_f.reshape(B, H, NT, BT, Vdim)
        o = o.permute(0, 2, 3, 1, 4).contiguous()
        o = o.reshape(B, T, H, Vdim)

        return o.to(dtype_out)


# =========================================================================
# Module-level shape shims  (overridden per shape by check.py / benchmark.py)
# =========================================================================

B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
