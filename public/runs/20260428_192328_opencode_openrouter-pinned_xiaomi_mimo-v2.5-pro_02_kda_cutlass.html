Uses concatenated matmuls and torch.compile to reduce kernel launch overhead.
No FLA imports.
"""
from __future__ import annotations

import torch
import torch.nn as nn

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@torch.compile(mode="reduce-overhead")
def _fused_inter_chunk(q_i, k_i, g_i, u_i, w_i, S, mask_strict_upper):
    """Fused inter-chunk body. Compiled to reduce kernel launches."""
    q_pos_i = q_i * g_i.exp()
    k_neg_i = k_i * (-g_i).exp()
    Aqk = torch.matmul(q_pos_i, k_neg_i.transpose(-1, -2))
    Aqk = Aqk.masked_fill(mask_strict_upper, 0)

    v_i = u_i - torch.matmul(w_i, S)

    q_exp = q_i * g_i.exp()
    o_i = torch.matmul(q_exp, S) + torch.matmul(Aqk, v_i)

    g_last = g_i[:, -1, :]
    S = S * g_last.unsqueeze(-1).exp()
    k_decayed = (g_last.unsqueeze(-2) - g_i).exp() * k_i
    S = S + torch.matmul(k_decayed.transpose(-1, -2), v_i)

    return o_i, S


def _kda_fwd(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    """KDA forward, chunk form. No initial/final state. Returns o in v's dtype."""
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0
    NT = T // BT

    # Work in fp32
    q = q.float() * scale
    k = k.float()
    v = v.float()
    g = g.float()
    beta = beta.float()

    # Reshape to chunk form: (B*H, NT, BT, D)
    q = q.reshape(B, T, H, K).permute(0, 2, 1, 3).reshape(B * H, NT, BT, K)
    k = k.reshape(B, T, H, K).permute(0, 2, 1, 3).reshape(B * H, NT, BT, K)
    v = v.reshape(B, T, H, V).permute(0, 2, 1, 3).reshape(B * H, NT, BT, V)
    g = g.reshape(B, T, H, K).permute(0, 2, 1, 3).reshape(B * H, NT, BT, K)
    beta = beta.reshape(B, T, H).permute(0, 2, 1).reshape(B * H, NT, BT)

    # In-chunk cumsum of g
    g = g.cumsum(-2)

    # ---- Build A_kk ----
    k_pos = k * g.exp()
    k_neg = k * (-g).exp()
    A = torch.matmul(k_pos, k_neg.transpose(-1, -2))

    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    A = A * beta.unsqueeze(-1)
    A_neg = -A.masked_fill(mask_diag_upper, 0)

    # Newton iteration -> solve_triangular
    I_mat = torch.eye(BT, dtype=torch.float32, device=q.device).expand_as(A_neg).contiguous()
    A_inv = torch.linalg.solve_triangular(I_mat - A_neg, I_mat, upper=False, unitriangular=False)
    A = A_inv * beta.unsqueeze(-2)

    # Compute w = A @ (exp(g) * k) and u = A @ v
    g_exp_k = g.exp() * k
    w = torch.matmul(A, g_exp_k)
    u = torch.matmul(A, v)

    # ---- Recurrent inter-chunk pass ----
    BH = B * H
    S = torch.zeros(BH, K, V, dtype=torch.float32, device=q.device)
    o = torch.zeros(BH, NT, BT, V, dtype=torch.float32, device=q.device)

    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)

    for ic in range(NT):
        o_i, S = _fused_inter_chunk(
            q[:, ic], k[:, ic], g[:, ic], u[:, ic], w[:, ic],
            S, mask_strict_upper,
        )
        o[:, ic] = o_i

    # Reshape back
    o = o.reshape(B, H, NT, BT, V).permute(0, 2, 3, 1, 4).reshape(B, T, H, V)
    return o.to(dtype)
