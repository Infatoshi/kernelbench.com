"""FP8 e4m3 GEMM - optimized BF16 matmul with CUDA graph capture.

The reference does x_bf16 @ w_bf16.T using cuBLAS BF16 mma. To match the
reference within the bf16 tolerance (0.01), we use the same algorithm.

We capture the cast + matmul as a CUDA graph on the first call and replay
it on subsequent calls, reducing kernel launch overhead.
"""
import torch
import torch.nn as nn


class Model(nn.Module):
    """y = (x @ w.T).to(bf16), where x is fp8_e4m3 (M, K), w is fp8_e4m3 (N, K)."""

    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        # Store weight as bf16 (matches reference)
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)
        # Pre-allocate buffers (reused across calls)
        self.register_buffer('_x_bf', torch.empty(M, K, dtype=torch.bfloat16),
                             persistent=False)
        self.register_buffer('_out', torch.empty(M, N, dtype=torch.bfloat16),
                             persistent=False)
        # CUDA graph state
        self._graph = None
        self._graph_x_ptr = None
        self._graph_w_ptr = None

    def _capture_graph(self, x: torch.Tensor):
        """Capture the cast + matmul as a CUDA graph."""
        # Warmup on a side stream (required for CUDA graph capture)
        s = torch.cuda.Stream()
        s.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(s):
            for _ in range(3):
                self._x_bf.copy_(x)
                torch.mm(self._x_bf, self.weight.T, out=self._out)
        torch.cuda.current_stream().wait_stream(s)

        self._graph = torch.cuda.CUDAGraph()
        with torch.cuda.graph(self._graph):
            self._x_bf.copy_(x)
            torch.mm(self._x_bf, self.weight.T, out=self._out)
        self._graph_x_ptr = x.data_ptr()
        self._graph_w_ptr = self.weight.data_ptr()

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Re-capture graph if input or weight pointer changed
        if (self._graph is None
                or self._graph_x_ptr != x.data_ptr()
                or self._graph_w_ptr != self.weight.data_ptr()):
            self._capture_graph(x)
        self._graph.replay()
        return self._out


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
