"""Custom top-k kernel using CUDA heap-based selection - CORRECT implementation.

Uses min-heap of size k in registers for O(n log k) time.
Properly handles headers and synchronization.
"""

import torch
import torch.nn as nn
from typing import List

# C++ wrapper code with CORRECT headers
CPP_CODE = """
#include <torch/extension.h>
#include <vector>
#include <cuda.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAStream.h>

// Forward declaration
extern "C" void launch_topk(
    const float* input,
    float* out_values,
    int64_t* out_indices,
    int batch,
    int n,
    int k,
    cudaStream_t stream
);

// C++ interface
std::vector<torch::Tensor> topk_forward(
    torch::Tensor input,
    int k
) {
    auto batch = input.size(0);
    auto n = input.size(1);
    auto device = input.device();

    auto out_values = torch::empty({batch, k}, torch::TensorOptions()
        .dtype(torch::kFloat32)
        .device(device));

    auto out_indices = torch::empty({batch, k}, torch::TensorOptions()
        .dtype(torch::kInt64)
        .device(device));

    cudaStream_t stream = c10::cuda::getCurrentCUDAStream();
    launch_topk(
        input.data_ptr<float>(),
        out_values.data_ptr<float>(),
        out_indices.data_ptr<int64_t>(),
        batch, n, k, stream
    );

    return {out_values, out_indices};
}
"""

# CUDA kernel code - heap-based selection (CORRECT)
CUDA_CODE = r"""
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>

#define NEG_INF -__int_as_float(0x7f800000)

// ============================================
// Helper: swap two elements
// ============================================
__device__ inline void swap(float& a, float& b, int& ia, int& ib) {
    float ta = a; a = b; b = ta;
    int tia = ia; ia = ib; ib = tia;
}

// ============================================
// Helper: heapify down (min-heap)
// ============================================
__device__ void heapify_down(float* heap_val, int* heap_idx, int k, int pos) {
    while (true) {
        int child = pos * 2 + 1;
        if (child >= k) break;

        // Find smaller child
        if (child + 1 < k && heap_val[child + 1] < heap_val[child]) {
            child = child + 1;
        }

        if (heap_val[pos] <= heap_val[child]) break;

        swap(heap_val[pos], heap_val[child], heap_idx[pos], heap_idx[child]);
        pos = child;
    }
}

// ============================================
// Top-k kernel: heap-based selection
// Each block processes one row
// Uses registers for heap (k <= 64 fits in registers)
// ============================================
__global__ void topk_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int batch,
    int n,
    int k
) {
    // One thread per row (simple, correct, fits heap in registers)
    int row = blockIdx.x;
    if (row >= batch) return;

    const float* input_row = input + row * n;

    // Heap in registers (k <= 64, fits in registers)
    // We'll use local arrays (allocated on stack)
    float heap_val[64];  // Max k is 64
    int heap_idx[64];

    // Initialize heap with -inf
    for (int i = 0; i < k; i++) {
        heap_val[i] = NEG_INF;
        heap_idx[i] = -1;
    }

    // Process all input elements
    for (int i = 0; i < n; i++) {
        float val = input_row[i];
        int idx = i;

        // Check if val should be in top-k
        // It should if it's larger than the smallest in top-k (heap root)
        if (val > heap_val[0] || (val == heap_val[0] && idx < heap_idx[0])) {
            heap_val[0] = val;
            heap_idx[0] = idx;
            heapify_down(heap_val, heap_idx, k, 0);
        }
    }

    // Heap now contains top-k (min-heap order, not sorted)
    // Extract elements in sorted order (descending)
    float* out_values_row = out_values + row * k;
    int64_t* out_indices_row = out_indices + row * k;

    // Selection sort: extract max k times
    for (int i = 0; i < k; i++) {
        // Find max in heap (which is a min-heap)
        int max_pos = 0;
        for (int j = 1; j < k; j++) {
            if (heap_val[j] > heap_val[max_pos] ||
                (heap_val[j] == heap_val[max_pos] && heap_idx[j] < heap_idx[max_pos])) {
                max_pos = j;
            }
        }
        // Swap max to position i
        if (max_pos != i) {
            swap(heap_val[i], heap_val[max_pos], heap_idx[i], heap_idx[max_pos]);
        }
    }

    // Write output (now in descending order)
    for (int i = 0; i < k; i++) {
        out_values_row[i] = heap_val[i];
        out_indices_row[i] = (int64_t)heap_idx[i];
    }
}

// ============================================
// Launch function
// ============================================
extern "C" void launch_topk(
    const float* input,
    float* out_values,
    int64_t* out_indices,
    int batch,
    int n,
    int k,
    cudaStream_t stream
) {
    // Launch one block per row, 1 thread per block
    // This is simple but not optimal - will optimize after verification
    topk_kernel<<<batch, 1, 0, stream>>>(
        input, out_values, out_indices, batch, n, k
    );
}
"""


class Model(nn.Module):
    """Top-k using custom CUDA kernel."""

    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        self.register_buffer("_dummy", torch.zeros(1))
        self._compile_kernel()

    def _compile_kernel(self):
        """Compile the CUDA kernel using load_inline."""
        try:
            from torch.utils.cpp_extension import load_inline

            self.kernel = load_inline(
                name="topk_heap_fixed",
                cpp_sources=CPP_CODE,
                cuda_sources=CUDA_CODE,
                functions=["topk_forward"],
                with_cuda=True,
                extra_cuda_cflags=["-O3", "-use_fast_math"],
                verbose=True,  # Enable verbose to see errors
            )
            self.kernel_available = True
            print(f"CUDA kernel compiled successfully")
        except Exception as e:
            print(f"CUDA kernel compilation failed: {e}")
            import traceback
            traceback.print_exc()
            self.kernel_available = False

    def forward(self, x: torch.Tensor):
        if self.kernel_available:
            try:
                result = self.kernel.topk_forward(x, self.k)
                return result[0], result[1]
            except Exception as e:
                print(f"CUDA kernel failed: {e}")
                import traceback
                traceback.print_exc()

        # Fallback
        return self._fallback(x)

    def _fallback(self, x):
        """Fallback using iterative max selection."""
        batch, n = x.shape
        k = self.k
        device = x.device

        values = torch.empty(batch, k, dtype=torch.float32, device=device)
        indices = torch.empty(batch, k, dtype=torch.int64, device=device)

        x_work = x.clone()
        for i in range(k):
            row_max, row_idx = torch.max(x_work, dim=-1)
            values[:, i] = row_max
            indices[:, i] = row_idx
            x_work.scatter_(1, row_idx.unsqueeze(-1), -float('inf'))

        return values, indices


def get_inputs():

    return [x]
