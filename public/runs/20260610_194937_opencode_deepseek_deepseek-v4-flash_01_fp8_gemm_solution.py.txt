"""FP8 e4m3 GEMM kernel using CUDA C++ with nvcuda::wmma (bf16 tensor cores).
Converts fp8→bf16 on-the-fly during global→shared load, then uses bf16 WMMA.
"""
import torch
import torch.nn as nn

# CUDA kernel source
CUDA_SRC = r"""
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cuda_fp8.h>
#include <mma.h>

using namespace nvcuda::wmma;

constexpr int BM = 32;
constexpr int BN = 64;
constexpr int BK = 32;
constexpr int WARPS_M = BM / 16;
constexpr int WARPS_N = BN / 16;
constexpr int NUM_WARPS = WARPS_M * WARPS_N;
constexpr int THREADS = NUM_WARPS * 32;

__global__ void fp8_gemm_kernel(
    const __nv_fp8_e4m3* __restrict__ A,
    const __nv_bfloat16* __restrict__ B,
    __nv_bfloat16* __restrict__ C,
    int M, int N, int K
) {
    __shared__ __nv_bfloat16 smem_A[BM * BK];
    __shared__ __nv_bfloat16 smem_B[BN * BK];

    int warp_id = threadIdx.x / 32;
    int warp_m = warp_id % WARPS_M;
    int warp_n = warp_id / WARPS_M;

    int block_row = blockIdx.x * BM;
    int block_col = blockIdx.y * BN;

    fragment<matrix_a, 16, 16, 16, __nv_bfloat16, row_major> frag_a;
    fragment<matrix_b, 16, 16, 16, __nv_bfloat16, col_major> frag_b;
    fragment<accumulator, 16, 16, 16, float> frag_c;
    fill_fragment(frag_c, 0.0f);

    for (int k = 0; k < K; k += BK) {
        for (int idx = threadIdx.x; idx < BM * BK; idx += THREADS) {
            int row = idx / BK;
            int col = idx % BK;
            int g_row = block_row + row;
            int g_col = k + col;
            float fval;
            if (g_row < M && g_col < K) {
                fval = (float)A[g_row * K + g_col];
            } else {
                fval = 0.0f;
            }
            smem_A[row * BK + col] = __float2bfloat16(fval);
        }

        for (int idx = threadIdx.x; idx < BN * BK; idx += THREADS) {
            int row = idx / BK;
            int col = idx % BK;
            int g_row = block_col + row;
            int g_col = k + col;
            __nv_bfloat16 val;
            if (g_row < N && g_col < K) {
                val = B[g_row * K + g_col];
            } else {
                val = __float2bfloat16(0.0f);
            }
            smem_B[row * BK + col] = val;
        }

        __syncthreads();

        for (int kk = 0; kk < BK; kk += 16) {
            load_matrix_sync(frag_a, &smem_A[warp_m * 16 * BK + kk], BK);
            load_matrix_sync(frag_b, &smem_B[warp_n * 16 * BK + kk], BK);
            mma_sync(frag_c, frag_a, frag_b, frag_c);
        }

        __syncthreads();
    }

    {
        __shared__ float smem_float[BM * BN];
        store_matrix_sync(
            &smem_float[warp_m * 16 * BN + warp_n * 16],
            frag_c, BN, mem_row_major
        );
        __syncthreads();

        for (int idx = threadIdx.x; idx < BM * BN; idx += THREADS) {
            int m = idx / BN;
            int n = idx % BN;
            int g_m = block_row + m;
            int g_n = block_col + n;
            if (g_m < M && g_n < N) {
                C[g_m * N + g_n] = __float2bfloat16(smem_float[m * BN + n]);
            }
        }
    }
}

extern "C" void launch_gemm(long long A, long long B, long long C, int M, int N, int K) {
    dim3 grid((M + BM - 1) / BM, (N + BN - 1) / BN);
    dim3 block(THREADS);
    fp8_gemm_kernel<<<grid, block>>>(
        reinterpret_cast<const __nv_fp8_e4m3*>(static_cast<uintptr_t>(A)),
        reinterpret_cast<const __nv_bfloat16*>(static_cast<uintptr_t>(B)),
        reinterpret_cast<__nv_bfloat16*>(static_cast<uintptr_t>(C)),
        M, N, K
    );
    cudaDeviceSynchronize();
}
"""

# Compile once at module load time
from torch.utils.cpp_extension import load_inline

_ext = load_inline(
    name="fp8_gemm_sm120",
    cpp_sources='extern "C" void launch_gemm(long long A, long long B, long long C, int M, int N, int K);',
    cuda_sources=CUDA_SRC,
    extra_cuda_cflags=["-arch=sm_120", "-O3", "--expt-relaxed-constexpr"],
    functions=["launch_gemm"],
    verbose=False,
)


OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp8_e4m3"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        out = torch.empty(self.M, self.N, dtype=torch.bfloat16, device=x.device)
        _ext.launch_gemm(
            x.data_ptr(),
            self.weight.data_ptr(),
            out.data_ptr(),
            self.M, self.N, self.K,
        )
        return out


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
