"""Kimi Delta Attention (KDA) forward, chunk form.

Strategy:
  - Use PyTorch for small operations (per-chunk 64x64 matrix inverse, cumsum)
  - Use a single big Triton kernel for the heavy inter+output work
  - This keeps the implementation small and correct
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


# ======================================================================================
# Kernel: inter-chunk recurrence + output (the heavy kernel)
# ======================================================================================
@triton.jit
def _inter_out_kernel(
    q_ptr, k_ptr, g_ptr,
    Aqk_ptr, w_ptr, u_ptr,
    o_ptr,
    scale,
    T,
    NT,
    H: tl.constexpr,
    HV: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
    NTW: tl.constexpr,  # number of chunks per program
):
    """One program per (V_chunk, NTW_group, B*H).

    Loops over NTW chunks. State h is (K, BV) kept in registers.
    """
    pid_v = tl.program_id(0)
    pid_ntw = tl.program_id(1)
    pid_bh = tl.program_id(2)
    i_b = pid_bh // HV
    i_hv = pid_bh % HV
    i_h = i_hv

    # Determine chunk range for this group
    chunk_start = pid_ntw * NTW
    chunk_end = tl.minimum(chunk_start + NTW, NT)

    # Pointers
    q_base = q_ptr + (i_b * T * H + i_h) * K
    k_base = k_ptr + (i_b * T * H + i_h) * K
    g_base = g_ptr + (i_b * T * HV + i_hv) * K
    Aqk_base = Aqk_ptr + (i_b * T * HV + i_hv) * BT
    w_base = w_ptr + (i_b * T * HV + i_hv) * K
    u_base = u_ptr + (i_b * T * HV + i_hv) * V
    o_base = o_ptr + (i_b * T * HV + i_hv) * V

    o_t = tl.arange(0, BT)
    o_k = tl.arange(0, K)
    o_v = pid_v * BV + tl.arange(0, BV)

    # Initialize h to zero
    b_h = tl.zeros((K, BV), dtype=tl.float32)

    # Main loop over chunks
    for i_t in range(chunk_start, chunk_end):
        chunk_off = i_t * BT

        # Load u, w, Aqk, k, g, q for this chunk
        b_u = tl.load(u_base + (chunk_off + o_t[:, None]) * HV * V + o_v[None, :]).to(tl.float32)
        b_w = tl.load(w_base + (chunk_off + o_t[:, None]) * HV * K + o_k[None, :]).to(tl.float32)
        b_k = tl.load(k_base + (chunk_off + o_t[:, None]) * H * K + o_k[None, :]).to(tl.float32)
        b_g = tl.load(g_base + (chunk_off + o_t[:, None]) * HV * K + o_k[None, :]).to(tl.float32)
        b_q = tl.load(q_base + (chunk_off + o_t[:, None]) * H * K + o_k[None, :]).to(tl.float32) * scale
        b_Aqk = tl.load(Aqk_base + (chunk_off + o_t[:, None]) * HV * BT + o_t[None, :]).to(tl.float32)

        # v_new = u - w @ h
        b_v_new = b_u - tl.dot(b_w.to(tl.bfloat16), b_h.to(tl.bfloat16), out_dtype=tl.float32)
        # o_inter = (q * g.exp()) @ h
        b_qg = b_q * tl.exp(b_g)  # (BT, K)
        b_o_inter = tl.dot(b_qg.to(tl.bfloat16), b_h.to(tl.bfloat16), out_dtype=tl.float32)
        # o = o_inter + Aqk @ v_new
        b_o = b_o_inter + tl.dot(b_Aqk.to(tl.bfloat16), b_v_new.to(tl.bfloat16), out_dtype=tl.float32)

        # Store o
        tl.store(o_base + (chunk_off + o_t[:, None]) * HV * V + o_v[None, :],
                 b_o.to(tl.bfloat16))

        # Update h: h_new = h * exp(g_n) + (k * exp(g_n - g)).T @ v_new
        b_gn = tl.load(g_base + (chunk_off + BT - 1) * HV * K + o_k).to(tl.float32)  # (K,)
        b_h = b_h * tl.exp(b_gn[:, None])  # (K, BV)
        b_kg = b_k * tl.exp(b_gn[None, :] - b_g)  # (BT, K)
        b_h = b_h + tl.dot(tl.trans(b_kg.to(tl.bfloat16)), b_v_new.to(tl.bfloat16),
                           out_dtype=tl.float32)


# ======================================================================================
# Wrapper
# ======================================================================================

class Model(nn.Module):
    def __init__(self, B, T, H, K, V, chunk_size=64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        B, T, H, K = q.shape
        V = v.shape[-1]
        BT = self.chunk_size
        NT = T // BT
        device = q.device

        # Step 1: per-chunk cumsum of g
        g_cumsum = chunk_local_cumsum(g, BT)

        # Step 2: intra in PyTorch (small operations)
        Aqk, u, w = _intra_pytorch(q, k, v, g_cumsum, beta, self.scale, BT)

        # Step 3: inter+output in Triton
        o = torch.empty_like(v)
        # Choose NTW to keep good occupancy; aim for ~16+ blocks total
        # Default: 1 chunk per program (full wave per B*H*V_chunk)
        BV = 64
        NTW = NT  # process all chunks in one program
        grid = (V // BV, 1, B * H)
        _inter_out_kernel[grid](
            q, k, g_cumsum,
            Aqk, w, u,
            o,
            self.scale,
            T, NT, H, H, K, V, BT, BV, NTW,
            num_warps=4, num_stages=2,
        )

        return o


def chunk_local_cumsum(g, chunk_size):
    """Per-chunk cumsum of g along the time axis. g is (B, T, H, K)."""
    B, T, H, K = g.shape
    NT = T // chunk_size
    BT = chunk_size
    g = g.view(B, NT, BT, H, K)
    g_cumsum = g.cumsum(dim=2)
    g_cumsum = g_cumsum.view(B, T, H, K)
    return g_cumsum


def _intra_pytorch(q, k, v, g, beta, scale, BT):
    """Compute Aqk, u, w using PyTorch (the small operations)."""
    from einops import rearrange
    B, T, H, K = q.shape
    V = v.shape[-1]
    NT = T // BT

    q_f = q.to(torch.float32) * scale
    k_f = k.to(torch.float32)
    v_f = v.to(torch.float32)
    g_f = g.to(torch.float32)
    beta_f = beta.to(torch.float32)

    # Reshape to (B, H, NT, BT, D)
    q_r = rearrange(q_f, "b (n c) h d -> b h n c d", c=BT)
    k_r = rearrange(k_f, "b (n c) h d -> b h n c d", c=BT)
    v_r = rearrange(v_f, "b (n c) h d -> b h n c d", c=BT)
    g_r = rearrange(g_f, "b (n c) h d -> b h n c d", c=BT)
    beta_r = rearrange(beta_f, "b (n c) h -> b h n c", c=BT)

    # Compute A_inv = (I + A_kk)^{-1} for each chunk
    # A_kk[i, c] = beta[i] * <k_c, k_i>_decay for c < i
    # Use the kg_pos, kg_neg trick
    kg_pos = k_r * g_r.exp()  # (B, H, NT, BT, K)
    kg_neg = k_r * (-g_r).exp()  # (B, H, NT, BT, K)
    # Akk_raw[c, i] = sum_d kg_pos[c, d] * kg_neg[i, d]
    # bmm: (B*H*NT, BT, K) @ (B*H*NT, K, BT) -> (B*H*NT, BT, BT)
    Akk_raw = torch.bmm(
        kg_pos.reshape(-1, BT, K),
        kg_neg.reshape(-1, K, BT)
    ).reshape(B, H, NT, BT, BT)
    # A[i, c] = beta[i] * Akk_raw[c, i] for c < i
    A = Akk_raw.transpose(-1, -2) * beta_r[..., None]  # (B, H, NT, BT, BT)
    # Mask to strict lower
    mask_lower = torch.tril(torch.ones(BT, BT, device=q.device, dtype=torch.bool), diagonal=-1)
    A = A * mask_lower

    # Compute A_inv = (I + A)^{-1} via forward sub (or torch.linalg.inv)
    I = torch.eye(BT, device=q.device, dtype=torch.float32)
    A_plus_I = A + I  # unit lower triangular
    # Use torch.linalg.solve_triangular for efficiency
    A_inv = torch.linalg.solve_triangular(
        A_plus_I, I.unsqueeze(0).unsqueeze(0).unsqueeze(0).expand(B, H, NT, BT, BT),
        upper=False, unitriangular=True
    )

    # u = A_inv @ (beta * v) where v is (B, H, NT, BT, V)
    bv = v_r * beta_r[..., None]  # (B, H, NT, BT, V)
    u = torch.bmm(
        A_inv.reshape(-1, BT, BT),
        bv.reshape(-1, BT, V)
    ).reshape(B, H, NT, BT, V)

    # w = A_inv @ (beta * g.exp() * k)
    bgek = (k_r * g_r.exp()) * beta_r[..., None]  # (B, H, NT, BT, K)
    w = torch.bmm(
        A_inv.reshape(-1, BT, BT),
        bgek.reshape(-1, BT, K)
    ).reshape(B, H, NT, BT, K)

    # Aqk: q @ k.T with decay, lower triangular
    qg = q_r * g_r.exp()  # (B, H, NT, BT, K)
    kg = k_r * (-g_r).exp()  # (B, H, NT, BT, K)
    # bmm: (B*H*NT, BT, K) @ (B*H*NT, K, BT) -> (B*H*NT, BT, BT)
    Aqk = torch.bmm(
        qg.reshape(-1, BT, K),
        kg.reshape(-1, K, BT)
    ).reshape(B, H, NT, BT, BT)
    mask_lower_eq = torch.tril(torch.ones(BT, BT, device=q.device, dtype=torch.bool), diagonal=0)
    Aqk = Aqk * mask_lower_eq

    # Convert back to bf16 and reshape to (B, T, H, D)
    Aqk = Aqk.to(torch.bfloat16).permute(0, 3, 1, 2, 4).reshape(B, T, H, BT)
    u = u.to(torch.bfloat16).permute(0, 3, 1, 2, 4).reshape(B, T, H, V)
    w = w.to(torch.bfloat16).permute(0, 3, 1, 2, 4).reshape(B, T, H, K)
    return Aqk, u, w
