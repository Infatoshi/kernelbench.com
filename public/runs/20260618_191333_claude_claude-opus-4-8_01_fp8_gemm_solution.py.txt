"""FP8 e4m3 GEMM for B200 (SM100). y = (x @ w.T) * weight_scale -> bf16.

x: fp8_e4m3 (M, K)   w: fp8_e4m3 (N, K)   weight_scale: (N,) fp32.
Real fp8 x fp8 tensor-core MMA via Triton tl.dot (fp32 accumulate).

Two kernels:
  * _fp8_gemm_kernel    : standard tiled GEMM, predicated K-tail (fast main loop).
  * _fp8_gemm_splitk    : split-K for skinny M (decode) to keep all SMs busy.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl

E4M3_MAX = 448.0


def _configs():
    cfgs = []
    for BM, BN, BK, s, w in [
        (256, 256, 128, 3, 8),
        (256, 256, 64, 4, 8),
        (128, 256, 128, 4, 8),
        (128, 128, 128, 3, 8),
        (64, 256, 128, 4, 8),
        (128, 64, 128, 4, 4),
    ]:
        cfgs.append(triton.Config({"BM": BM, "BN": BN, "BK": BK, "GROUP_M": 8},
                                  num_stages=s, num_warps=w))
    return cfgs


@triton.autotune(configs=_configs(), key=["M", "N", "K"])
@triton.jit
def _fp8_gemm_kernel(
    x_ptr, w_ptr, s_ptr, y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_ym, stride_yn,
    BM: tl.constexpr, BN: tl.constexpr, BK: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BM)
    num_pid_n = tl.cdiv(N, BN)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = (pid_m * BM + tl.arange(0, BM)) % M
    offs_n = (pid_n * BN + tl.arange(0, BN)) % N
    offs_k = tl.arange(0, BK)

    x_ptrs = x_ptr + (offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    w_ptrs = w_ptr + (offs_k[:, None] * stride_wk + offs_n[None, :] * stride_wn)

    acc = tl.zeros((BM, BN), dtype=tl.float32)
    k_full = (K // BK) * BK
    for k in range(0, k_full, BK):
        a = tl.load(x_ptrs)
        b = tl.load(w_ptrs)
        acc = tl.dot(a, b, acc)
        x_ptrs += BK * stride_xk
        w_ptrs += BK * stride_wk
    if k_full < K:
        kmask = offs_k < K - k_full
        a = tl.load(x_ptrs, mask=kmask[None, :], other=0.0)
        b = tl.load(w_ptrs, mask=kmask[:, None], other=0.0)
        acc = tl.dot(a, b, acc)

    scale = tl.load(s_ptr + offs_n).to(tl.float32)
    acc = acc * scale[None, :]
    y = acc.to(tl.bfloat16)

    offs_ym = pid_m * BM + tl.arange(0, BM)
    offs_yn = pid_n * BN + tl.arange(0, BN)
    y_ptrs = y_ptr + offs_ym[:, None] * stride_ym + offs_yn[None, :] * stride_yn
    ymask = (offs_ym[:, None] < M) & (offs_yn[None, :] < N)
    tl.store(y_ptrs, y, mask=ymask)


def _splitk_configs():
    cfgs = []
    for BN, BK, SK, s, w in [
        (128, 128, 16, 3, 4),
        (128, 128, 8, 4, 4),
        (128, 128, 8, 3, 4),
        (256, 128, 8, 3, 4),
        (64, 256, 8, 3, 4),
        (128, 64, 16, 4, 4),
    ]:
        cfgs.append(triton.Config({"BN": BN, "BK": BK, "SPLIT_K": SK},
                                  num_stages=s, num_warps=w))
    return cfgs


@triton.autotune(configs=_splitk_configs(), key=["M", "N", "K"], reset_to_zero=["y_ptr"])
@triton.jit
def _fp8_gemm_splitk(
    x_ptr, w_ptr, s_ptr, y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_ym, stride_yn,
    BM: tl.constexpr, BN: tl.constexpr, BK: tl.constexpr,
    SPLIT_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    pid_k = tl.program_id(1)

    offs_m = tl.arange(0, BM)
    offs_n = (pid_n * BN + tl.arange(0, BN)) % N
    offs_k = pid_k * BK + tl.arange(0, BK)

    # coalesced loads: K (contiguous, stride 1) is the last axis of each tile
    x_ptrs = x_ptr + (offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    w_ptrs = w_ptr + (offs_n[:, None] * stride_wn + offs_k[None, :] * stride_wk)

    acc = tl.zeros((BM, BN), dtype=tl.float32)
    step = BK * SPLIT_K
    for k0 in range(pid_k * BK, K, step):
        kmask = offs_k < K
        a = tl.load(x_ptrs, mask=(offs_m[:, None] < M) & kmask[None, :], other=0.0)
        wt = tl.load(w_ptrs, mask=kmask[None, :], other=0.0)  # (BN, BK)
        acc = tl.dot(a, tl.trans(wt), acc)
        x_ptrs += step * stride_xk
        w_ptrs += step * stride_wk
        offs_k += step

    scale = tl.load(s_ptr + offs_n).to(tl.float32)
    acc = acc * scale[None, :]

    offs_ym = tl.arange(0, BM)
    offs_yn = pid_n * BN + tl.arange(0, BN)
    y_ptrs = y_ptr + offs_ym[:, None] * stride_ym + offs_yn[None, :] * stride_yn
    ymask = (offs_ym[:, None] < M) & (offs_yn[None, :] < N)
    tl.atomic_add(y_ptrs, acc, mask=ymask, sem="relaxed")


def _fp8_gemm(x, w, s):
    M, K = x.shape
    N, _ = w.shape
    if M <= 64:
        BM = 32 if M <= 32 else 64
        y = torch.zeros((M, N), dtype=torch.float32, device=x.device)
        grid = lambda meta: (triton.cdiv(N, meta["BN"]), meta["SPLIT_K"])
        _fp8_gemm_splitk[grid](
            x, w, s, y, M, N, K,
            x.stride(0), x.stride(1),
            w.stride(0), w.stride(1),
            y.stride(0), y.stride(1),
            BM=BM,
        )
        return y.to(torch.bfloat16)
    y = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)
    grid = lambda meta: (triton.cdiv(M, meta["BM"]) * triton.cdiv(N, meta["BN"]),)
    _fp8_gemm_kernel[grid](
        x, w, s, y, M, N, K,
        x.stride(0), x.stride(1),
        w.stride(0), w.stride(1),
        y.stride(0), y.stride(1),
    )
    return y


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.register_buffer("weight", torch.empty(N, K, dtype=torch.float8_e4m3fn))
        self.register_buffer("weight_scale", torch.empty(N, dtype=torch.float32))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.weight
        K = w.shape[1]
        if K % 16 != 0:
            Kp = (K + 15) // 16 * 16
            key = (w.data_ptr(), w._version)
            if getattr(self, "_wp_key", None) != key:
                wp = torch.zeros(w.shape[0], Kp, dtype=w.dtype, device=w.device)
                wp[:, :K] = w
                self._wp = wp
                self._wp_key = key
            M = x.shape[0]
            xbuf = getattr(self, "_xp", None)
            if xbuf is None or xbuf.shape[0] != M or xbuf.shape[1] != Kp:
                xbuf = torch.zeros(M, Kp, dtype=x.dtype, device=x.device)
                self._xp = xbuf
            xbuf[:, :K] = x  # pad columns stay zero
            return _fp8_gemm(xbuf, self._wp, self.weight_scale)
        return _fp8_gemm(x, w, self.weight_scale)


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
