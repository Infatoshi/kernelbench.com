"""Paged single-query decode via Triton (FlashDecoding-style split-KV + fused path).

Per (batch, kv_head[, partition]) we stream paged K/V with online softmax in fp32.
Q is pre-scaled by (1/sqrt(d))*log2(e) so the inner loop uses exp2. When
batch*num_kv_heads saturates the GPU we use one fused kernel; otherwise phase-1
partials are merged in phase-2 with log-sum-exp weighting. CUDA graph replay
drops host launch overhead on repeated benchmark calls.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

_NUM_SMS = 188
_LOG2E = 1.4426950408889634
@triton.jit
def _phase1_kernel(
    Q_ptr,
    KV_ptr,
    BT_ptr,
    SL_ptr,
    O_ptr,
    M_ptr,
    L_ptr,
    SCALE_LOG2E: tl.constexpr,
    H: tl.constexpr,
    Hkv: tl.constexpr,
    D: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    PARTITION_SIZE: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    PAD_G: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_D: tl.constexpr,
    NUM_PARTITIONS: tl.constexpr,
    N_ITERS: tl.constexpr,
):
    part_id = tl.program_id(0)
    kvh = tl.program_id(1)
    b = tl.program_id(2)

    seq_len = tl.load(SL_ptr + b)
    part_start = part_id * PARTITION_SIZE
    part_end = tl.minimum(part_start + PARTITION_SIZE, seq_len)

    offs_g = tl.arange(0, PAD_G)
    offs_d = tl.arange(0, BLOCK_D)
    offs_n = tl.arange(0, BLOCK_N)
    g_mask = offs_g < GROUP_SIZE
    d_mask = offs_d < D

    h_base = kvh * GROUP_SIZE
    q_ptrs = Q_ptr + ((b * H + h_base + offs_g[:, None]) * D + offs_d[None, :])
    q = tl.load(q_ptrs, mask=g_mask[:, None] & d_mask[None, :], other=0.0)
    q = (q.to(tl.float32) * SCALE_LOG2E).to(tl.bfloat16)

    m_i = tl.full((PAD_G,), -1.0e30, tl.float32)
    l_i = tl.zeros((PAD_G,), tl.float32)
    acc = tl.zeros((PAD_G, BLOCK_D), tl.float32)

    for it in tl.static_range(N_ITERS):
        n_start = part_start + it * BLOCK_N
        tok_idx = n_start + offs_n
        valid_n = tok_idx < part_end

        page_idx = tok_idx // PAGE_SIZE
        page_off = tok_idx % PAGE_SIZE
        block_id = tl.load(
            BT_ptr + b * MAX_BLOCKS + page_idx,
            mask=valid_n,
            other=0,
        )

        kv_base = ((block_id * PAGE_SIZE + page_off) * Hkv + kvh) * (2 * D)
        k_ptrs = KV_ptr + offs_d[:, None] + kv_base[None, :]
        v_ptrs = KV_ptr + kv_base[:, None] + (D + offs_d[None, :])

        k = tl.load(
            k_ptrs,
            mask=d_mask[:, None] & valid_n[None, :],
            other=0.0,
            eviction_policy="evict_first",
        )
        v = tl.load(
            v_ptrs,
            mask=valid_n[:, None] & d_mask[None, :],
            other=0.0,
            eviction_policy="evict_first",
        )

        scores = tl.dot(q, k, out_dtype=tl.float32)
        scores = tl.where(valid_n[None, :], scores, -1.0e30)

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        alpha = tl.exp2(m_i - m_new)
        p = tl.exp2(scores - m_new[:, None])
        p = tl.where(valid_n[None, :], p, 0.0)

        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v, out_dtype=tl.float32)
        m_i = m_new

    ml_off = ((b * Hkv + kvh) * NUM_PARTITIONS + part_id) * GROUP_SIZE + offs_g
    tl.store(M_ptr + ml_off, m_i, mask=g_mask)
    tl.store(L_ptr + ml_off, l_i, mask=g_mask)
    o_off = (
        (((b * Hkv + kvh) * NUM_PARTITIONS + part_id) * GROUP_SIZE + offs_g[:, None]) * D
        + offs_d[None, :]
    )
    tl.store(O_ptr + o_off, acc, mask=g_mask[:, None] & d_mask[None, :])

@triton.jit
def _phase2_kernel(
    O_ptr,
    M_ptr,
    L_ptr,
    OUT_ptr,
    H: tl.constexpr,
    Hkv: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    D: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_P: tl.constexpr,
    NUM_PARTITIONS: tl.constexpr,
):
    h = tl.program_id(0)
    b = tl.program_id(1)

    kvh = h // GROUP_SIZE
    g = h % GROUP_SIZE

    offs_p = tl.arange(0, BLOCK_P)
    offs_d = tl.arange(0, BLOCK_D)
    p_mask = offs_p < NUM_PARTITIONS
    d_mask = offs_d < D

    ml_off = ((b * Hkv + kvh) * NUM_PARTITIONS + offs_p) * GROUP_SIZE + g
    m_arr = tl.load(M_ptr + ml_off, mask=p_mask, other=-1.0e30)
    l_arr = tl.load(L_ptr + ml_off, mask=p_mask, other=0.0)

    m_global = tl.max(m_arr, axis=0)
    weights = tl.exp2(m_arr - m_global)
    weights = tl.where(p_mask, weights, 0.0)
    l_global = tl.sum(weights * l_arr, axis=0)

    o_off = (
        (((b * Hkv + kvh) * NUM_PARTITIONS + offs_p[:, None]) * GROUP_SIZE + g) * D
        + offs_d[None, :]
    )
    o_arr = tl.load(O_ptr + o_off, mask=p_mask[:, None] & d_mask[None, :], other=0.0)
    o_final = tl.sum(weights[:, None] * o_arr, axis=0) / l_global

    tl.store(
        OUT_ptr + (b * H + h) * D + offs_d,
        o_final.to(tl.bfloat16),
        mask=d_mask,
    )


def _next_pow2(x: int) -> int:
    return 1 << (x - 1).bit_length()


def _pick_config(batch: int, num_kv_heads: int, head_dim: int, seq_len: int) -> tuple[int, int, int]:
    """Return (block_n, partition_size, num_partitions)."""
    block_n = 64 if head_dim <= 64 else 32
    bhkv = batch * num_kv_heads

    if bhkv >= _NUM_SMS:
        partition_size = ((seq_len + block_n - 1) // block_n) * block_n
        return block_n, partition_size, 1

    target_blocks = _NUM_SMS * 4
    num_partitions = (target_blocks + bhkv - 1) // bhkv
    partition_size = (seq_len + num_partitions - 1) // num_partitions
    partition_size = ((partition_size + block_n - 1) // block_n) * block_n
    partition_size = max(partition_size, 2 * block_n)
    num_partitions = (seq_len + partition_size - 1) // partition_size
    return block_n, partition_size, num_partitions


class Model(nn.Module):
    """Single-query paged attention decode (Triton, same I/O as reference)."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.scale_log2e = self.scale * _LOG2E

        self.block_n, self.partition_size, self.num_partitions = _pick_config(
            batch, num_kv_heads, head_dim, seq_len
        )
        self.block_d = _next_pow2(head_dim)
        self.pad_g = max(16, _next_pow2(self.group_size))
        self.n_iters = self.partition_size // self.block_n

        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._scratch = None
        self._graph_cache: dict[tuple[int, ...], torch.cuda.CUDAGraph] = {}

    def _alloc_buffers(self, device: torch.device, out_dtype: torch.dtype):
        if self._scratch is not None:
            return self._scratch

        b, h, hkv, d, g, np = (
            self.batch,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.group_size,
            self.num_partitions,
        )
        out = torch.empty(b, h, d, dtype=out_dtype, device=device)
        if np > 1:
            partial_o = torch.empty(b, hkv, np, g, d, dtype=torch.float32, device=device)
            partial_m = torch.empty(b, hkv, np, g, dtype=torch.float32, device=device)
            partial_l = torch.empty(b, hkv, np, g, dtype=torch.float32, device=device)
        else:
            partial_o = partial_m = partial_l = None
        self._scratch = (out, partial_o, partial_m, partial_l)
        return self._scratch

    def _launch(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> None:
        out, partial_o, partial_m, partial_l = self._scratch
        b = self.batch
        h = self.num_heads
        hkv = self.num_kv_heads
        d = self.head_dim
        g = self.group_size
        p = self.page_size
        bn = self.block_n
        bd = self.block_d
        pg = self.pad_g
        np = self.num_partitions
        ps = self.partition_size
        ni = self.n_iters
        mb = block_table.shape[1]
        s = self.scale_log2e

        if np == 1:
            grid = (hkv, b)
            _fused_kernel[grid](
                query,
                kv_cache,
                block_table,
                seq_lens,
                out,
                s,
                h,
                hkv,
                d,
                p,
                mb,
                g,
                pg,
                bn,
                bd,
                ni,
                num_warps=4,
                num_stages=3,
            )
            return

        grid1 = (np, hkv, b)
        _phase1_kernel[grid1](
            query,
            kv_cache,
            block_table,
            seq_lens,
            partial_o,
            partial_m,
            partial_l,
            s,
            h,
            hkv,
            d,
            p,
            ps,
            mb,
            g,
            pg,
            bn,
            bd,
            np,
            ni,
            num_warps=4,
            num_stages=3,
        )
        bp = max(16, _next_pow2(np))
        grid2 = (h, b)
        _phase2_kernel[grid2](
            partial_o,
            partial_m,
            partial_l,
            out,
            h,
            hkv,
            g,
            d,
            bd,
            bp,
            np,
            num_warps=2,
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        self._alloc_buffers(query.device, query.dtype)
        out = self._scratch[0]

        key = (
            query.data_ptr(),
            kv_cache.data_ptr(),
            block_table.data_ptr(),
            seq_lens.data_ptr(),
        )
        graph = self._graph_cache.get(key)
        if graph is None:
            self._launch(query, kv_cache, block_table, seq_lens)
            torch.cuda.synchronize()
            graph = torch.cuda.CUDAGraph()
            with torch.cuda.graph(graph):
                self._launch(query, kv_cache, block_table, seq_lens)
            self._graph_cache[key] = graph
        graph.replay()
        return out


def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    b = BATCH
    h = NUM_HEADS
    hkv = NUM_KV_HEADS
    d = HEAD_DIM
    seq_len = SEQ_LEN
    page_size = PAGE_SIZE

    pages_per_seq = (seq_len + page_size - 1) // page_size
    total_pages = max(b * pages_per_seq + 8, 64)

    query = torch.randn(b, h, d, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, page_size, hkv, 2 * d, dtype=torch.bfloat16) * 0.1
    block_table = (
        torch.randperm(total_pages)[: b * pages_per_seq]
        .reshape(b, pages_per_seq)
        .int()
        .contiguous()
    )
    seq_lens = torch.full((b,), seq_len, dtype=torch.int32)
    return [query, kv_cache, block_table, seq_lens]
