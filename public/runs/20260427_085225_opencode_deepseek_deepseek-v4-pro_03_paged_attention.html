"""Paged attention decode kernel for RTX PRO 6000 (SM120 Blackwell).

Single-query decode with grouped-query attention (GQA).
One 32-thread warp-block per query head for maximum parallelism.
Each lane handles D/32 dimensions; all reduction via __shfl_xor_sync.
Online softmax processes KV cache in a single streaming pass.
"""
from __future__ import annotations

import math

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

# ---------------------------------------------------------------------------
# CUDA kernel + wrapper
# ---------------------------------------------------------------------------

_cpp_src = r"""
torch::Tensor launch_paged_attention_decode(
    torch::Tensor query,
    torch::Tensor kv_cache,
    torch::Tensor block_table,
    torch::Tensor seq_lens,
    int B, int H, int Hkv, int D, int P, int max_blocks, float scale);
"""

_cuda_src = r"""
#include <cuda_runtime.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

#define WARP_SIZE 32
#define MAX_STRIDE 8

extern "C" __global__ void paged_attention_decode_kernel(
    const __nv_bfloat16* __restrict__ query,
    const __nv_bfloat16* __restrict__ kv_cache,
    const int* __restrict__ block_table,
    const int* __restrict__ seq_lens,
    __nv_bfloat16* __restrict__ output,
    int B, int H, int Hkv, int D, int P, int max_blocks, float scale,
    int G)
{
    // One block per query head
    int linear_bid = blockIdx.x;
    int b      = linear_bid / H;
    int q_head = linear_bid % H;
    if (b >= B) return;

    int L = seq_lens[b];
    if (L <= 0) return;

    int kv_h = q_head / G;   // which KV head this query head maps to
    int num_pages = (L + P - 1) / P;
    int lane_id = threadIdx.x;
    int stride  = D / WARP_SIZE;
    unsigned mask = __activemask();

    // Load this query head (fp32)
    float q_val[MAX_STRIDE];
    int q_base = b * H * D + q_head * D;
    #pragma unroll
    for (int s = 0; s < stride; s++) {
        q_val[s] = __bfloat162float(query[q_base + lane_id + s * WARP_SIZE]);
    }

    // Online softmax state for this single query head
    float m = -1e10f;
    float l = 0.0f;
    float acc[MAX_STRIDE];
    #pragma unroll
    for (int s = 0; s < stride; s++) {
        acc[s] = 0.0f;
    }

    // Offsets into kv_cache for this KV head
    const int* bt_base = block_table + b * max_blocks;
    int page_stride  = P * Hkv * 2 * D;
    int token_stride = Hkv * 2 * D;
    int k_off = kv_h * 2 * D;
    int v_off = k_off + D;

    // Main loop over pages and tokens
    for (int pg = 0; pg < num_pages; pg++) {
        int block_idx = bt_base[pg];
        int page_base = block_idx * page_stride;
        int tokens = (pg == num_pages - 1) ? (L - pg * P) : P;

        for (int t = 0; t < tokens; t++) {
            int base = page_base + t * token_stride;

            // Load K and V values for this token (strided across lanes)
            float k_vals[MAX_STRIDE];
            float v_vals[MAX_STRIDE];
            #pragma unroll
            for (int s = 0; s < stride; s++) {
                int idx = lane_id + s * WARP_SIZE;
                k_vals[s] = __bfloat162float(kv_cache[base + k_off + idx]);
                v_vals[s] = __bfloat162float(kv_cache[base + v_off + idx]);
            }

            // Dot product: Q @ K
            float dot = 0.0f;
            #pragma unroll
            for (int s = 0; s < stride; s++) {
                dot += q_val[s] * k_vals[s];
            }
            // Warp-level butterfly reduction
            #pragma unroll
            for (int offset = WARP_SIZE / 2; offset > 0; offset >>= 1) {
                dot += __shfl_xor_sync(mask, dot, offset);
            }
            dot *= scale;

            // Online softmax — lane 0 computes, then broadcast
            float weight_val = 0.0f;
            float rescale_val = 0.0f;
            if (lane_id == 0) {
                float m_new = fmaxf(m, dot);
                float old_scale = __expf(m - m_new);
                float exp_val   = __expf(dot - m_new);
                weight_val  = exp_val;
                rescale_val = old_scale;
                l = old_scale * l + exp_val;
                m = m_new;
            }
            weight_val  = __shfl_sync(mask, weight_val,  0);
            rescale_val = __shfl_sync(mask, rescale_val, 0);

            // Update output accumulator
            #pragma unroll
            for (int s = 0; s < stride; s++) {
                acc[s] = rescale_val * acc[s] + weight_val * v_vals[s];
            }
        }
    }

    // Normalise and write output
    float l_final = __shfl_sync(mask, l, 0);
    int out_base = b * H * D + q_head * D;
    #pragma unroll
    for (int s = 0; s < stride; s++) {
        output[out_base + lane_id + s * WARP_SIZE] = __float2bfloat16(acc[s] / l_final);
    }
}

// -----------------------------------------------------------------------
// C++ wrapper
// -----------------------------------------------------------------------
torch::Tensor launch_paged_attention_decode(
    torch::Tensor query,
    torch::Tensor kv_cache,
    torch::Tensor block_table,
    torch::Tensor seq_lens,
    int B, int H, int Hkv, int D, int P, int max_blocks, float scale)
{
    auto output = torch::empty({B, H, D}, query.options());

    constexpr int THREADS = 32;
    int G = H / Hkv;
    int num_blocks = B * H;

    paged_attention_decode_kernel<<<num_blocks, THREADS>>>(
        reinterpret_cast<const __nv_bfloat16*>(query.data_ptr<torch::BFloat16>()),
        reinterpret_cast<const __nv_bfloat16*>(kv_cache.data_ptr<torch::BFloat16>()),
        block_table.data_ptr<int>(),
        seq_lens.data_ptr<int>(),
        reinterpret_cast<__nv_bfloat16*>(output.data_ptr<torch::BFloat16>()),
        B, H, Hkv, D, P, max_blocks, scale, G);

    return output;
}
"""

_module = None

def _get_module():
    global _module
    if _module is None:
        _module = load_inline(
            name="paged_attention_decode_ext",
            cpp_sources=[_cpp_src],
            cuda_sources=[_cuda_src],
            functions=["launch_paged_attention_decode"],
            extra_cuda_cflags=[
                "-std=c++17",
                "-O3",
                "--use_fast_math",
                "-gencode", "arch=compute_120,code=sm_120",
            ],
            extra_cflags=["-O3"],
            verbose=False,
        )
    return _module


# ---------------------------------------------------------------------------
# Model
# ---------------------------------------------------------------------------

class Model(nn.Module):
    """Single-query paged attention decode -- custom CUDA kernel."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        max_blocks = block_table.shape[1]

        mod = _get_module()
        return mod.launch_paged_attention_decode(
            query.contiguous(),
            kv_cache.contiguous(),
            block_table.contiguous(),
            seq_lens.contiguous(),
            B, H, self.num_kv_heads, D, self.page_size, max_blocks, self.scale,
        )


# ---------------------------------------------------------------------------
# Input factories
# ---------------------------------------------------------------------------

def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]

    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
