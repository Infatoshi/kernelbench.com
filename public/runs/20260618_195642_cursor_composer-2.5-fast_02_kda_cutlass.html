def chunk_kda_vectorized(q, k, v, g, beta, scale, chunk_size=64):
    dtype = v.dtype

    BT = chunk_size
    NT = T_ // BT
    q = rearrange(q.float() * scale, 'b (n c) h d -> b h n c d', c=BT)
    k = rearrange(k.float(), 'b (n c) h d -> b h n c d', c=BT)
    v = rearrange(v.float(), 'b (n c) h d -> b h n c d', c=BT)
    g = rearrange(g.float(), 'b (n c) h d -> b h n c d', c=BT)
    beta = rearrange(beta.float(), 'b (n c) h -> b h n c', c=BT)
    g = g.cumsum(dim=-2)
    decay = (g.unsqueeze(-2) - g.unsqueeze(-3)).exp()

    A = A * beta.unsqueeze(-1)
    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    A = -A.masked_fill(mask_diag_upper, 0)
    for i in range(1, BT):
        A[..., i, :i] = A[..., i, :i] + (A[..., i, :, None] * A[..., :, :i]).sum(-2)
    A = (A + torch.eye(BT, device=q.device, dtype=A.dtype)) * beta.unsqueeze(-2)
    gk = g.exp() * k
    w = A @ gk
    u = A @ v
    Aqk = (q.unsqueeze(-2) * decay * k.unsqueeze(-3)).sum(-1)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)
    Aqk = Aqk.masked_fill(mask_strict_upper, 0)
    S = q.new_zeros(B_, H_, K_, V_)
    o = torch.zeros_like(v)
    for i in range(NT):

        u_i, g_i, w_i = u[:, :, i], g[:, :, i], w[:, :, i]
        Aqk_i = Aqk[:, :, i]
        v_i = u_i - w_i @ S
