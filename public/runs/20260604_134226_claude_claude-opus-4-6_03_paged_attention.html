import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

_LOG2E = 1.4426950408889634
_NUM_SMS = 170  # RTX PRO 6000 Blackwell GB202

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _splitk_kernel(
    Q, KV, BT, SL, PM, PL, PA,
    stride_qb, stride_qh, stride_qd,
    stride_kv0, stride_kv1, stride_kv2, stride_kv3,
    stride_bt0, stride_bt1,
    stride_pb, stride_ph, stride_pp,
    stride_ab, stride_ah, stride_ap, stride_ad,
    scale_log2e,
    PGSZ: tl.constexpr,
    GRP: tl.constexpr,
    D: tl.constexpr,
    TPP: tl.constexpr,
    BN: tl.constexpr,
    BPP: tl.constexpr,
):
    b = tl.program_id(0)
    kvh = tl.program_id(1)
    part = tl.program_id(2)

    seqlen = tl.load(SL + b)
    t0 = part * TPP
    t1 = tl.minimum(t0 + TPP, seqlen)

    h0 = kvh * GRP
    og = tl.arange(0, GRP)
    od = tl.arange(0, D)

    q = tl.load(Q + b * stride_qb + (h0 + og[:, None]) * stride_qh + od[None, :] * stride_qd)
    q = (q.to(tl.float32) * scale_log2e).to(tl.bfloat16)

    m_i = tl.full([GRP], float("-inf"), dtype=tl.float32)
    l_i = tl.zeros([GRP], dtype=tl.float32)
    acc = tl.zeros([GRP, D], dtype=tl.float32)

    for blk in range(BPP):
        on = t0 + blk * BN + tl.arange(0, BN)
        mask = on < t1

        pn = on // PGSZ
        sn = on % PGSZ
        bi = tl.load(BT + b * stride_bt0 + pn * stride_bt1, mask=mask, other=0)

        base = bi.to(tl.int64) * stride_kv0 + sn.to(tl.int64) * stride_kv1 + kvh * stride_kv2

        k = tl.load(KV + base[:, None] + od[None, :] * stride_kv3, mask=mask[:, None], other=0.0)
        v = tl.load(KV + base[:, None] + (D + od[None, :]) * stride_kv3, mask=mask[:, None], other=0.0)

        s = tl.dot(q, tl.trans(k))
        s = tl.where(mask[None, :], s, float("-inf"))

        m_blk = tl.max(s, axis=1)
        m_new = tl.maximum(m_i, m_blk)
        ms = tl.where(m_new == float("-inf"), 0.0, m_new)

        alpha = tl.exp2(m_i - ms)
        p = tl.exp2(s - ms[:, None])

        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v)
        m_i = m_new

    heads = h0 + og
    tl.store(PM + b * stride_pb + heads * stride_ph + part * stride_pp, m_i)
    tl.store(PL + b * stride_pb + heads * stride_ph + part * stride_pp, l_i)
    tl.store(
        PA + b * stride_ab + heads[:, None] * stride_ah + part * stride_ap + od[None, :] * stride_ad,
        acc,
    )


@triton.jit
def _combine_kernel(
    PM, PL, PA, Out,
    stride_pb, stride_ph, stride_pp,
    stride_ab, stride_ah, stride_ap, stride_ad,
    stride_ob, stride_oh, stride_od,
    D: tl.constexpr,
    NP: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)
    od = tl.arange(0, D)

    mg = tl.full([], float("-inf"), dtype=tl.float32)
    for p in range(NP):
        mg = tl.maximum(mg, tl.load(PM + b * stride_pb + h * stride_ph + p * stride_pp))
    ms = tl.where(mg == float("-inf"), 0.0, mg)

    lg = tl.zeros([], dtype=tl.float32)
    acc = tl.zeros([D], dtype=tl.float32)
    for p in range(NP):
        mp = tl.load(PM + b * stride_pb + h * stride_ph + p * stride_pp)
        lp = tl.load(PL + b * stride_pb + h * stride_ph + p * stride_pp)
        c = tl.exp2(mp - ms)
        lg += lp * c
        ap = tl.load(PA + b * stride_ab + h * stride_ah + p * stride_ap + od * stride_ad)
        acc += ap * c

    tl.store(Out + b * stride_ob + h * stride_oh + od * stride_od, (acc / lg).to(tl.bfloat16))


def _pick_np(B, Hkv, L, bn):
    base = B * Hkv
    target = 3 * _NUM_SMS
    np_ = max(1, -(-target // base))
    tpp = -(-L // np_)
    while tpp < 2 * bn and np_ > 1:
        np_ -= 1
        tpp = -(-L // np_)
    return np_


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale_log2e = (1.0 / math.sqrt(head_dim)) * _LOG2E
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._bufs_ready = False
        self._graph = None
        self._graph_key = None

    def _ensure_bufs(self, dev):
        if self._bufs_ready:
            return
        B, H, D = self.batch, self.num_heads, self.head_dim
        bn = 64
        self._np = _pick_np(B, self.num_kv_heads, self.seq_len, bn)
        self._tpp = -(-self.seq_len // self._np)
        self._bpp = -(-self._tpp // bn)
        self._pm = torch.empty(B, H, self._np, dtype=torch.float32, device=dev)
        self._pl = torch.empty(B, H, self._np, dtype=torch.float32, device=dev)
        self._pa = torch.empty(B, H, self._np, D, dtype=torch.float32, device=dev)
        self._out = torch.empty(B, H, D, dtype=torch.bfloat16, device=dev)
        self._bufs_ready = True

    def _launch(self, query, kv_cache, block_table, seq_lens):
        B, H, D = self.batch, self.num_heads, self.head_dim
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size

        _splitk_kernel[(B, Hkv, self._np)](
            query, kv_cache, block_table, seq_lens,
            self._pm, self._pl, self._pa,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0), block_table.stride(1),
            self._pm.stride(0), self._pm.stride(1), self._pm.stride(2),
            self._pa.stride(0), self._pa.stride(1), self._pa.stride(2), self._pa.stride(3),
            self.scale_log2e,
            PGSZ=P, GRP=G, D=D, TPP=self._tpp, BN=64, BPP=self._bpp,
            num_warps=4, num_stages=1,
        )
        _combine_kernel[(B, H)](
            self._pm, self._pl, self._pa, self._out,
            self._pm.stride(0), self._pm.stride(1), self._pm.stride(2),
            self._pa.stride(0), self._pa.stride(1), self._pa.stride(2), self._pa.stride(3),
            self._out.stride(0), self._out.stride(1), self._out.stride(2),
            D=D, NP=self._np,
            num_warps=2, num_stages=1,
        )

    def forward(self, query, kv_cache, block_table, seq_lens):
        self._ensure_bufs(query.device)
        key = (query.data_ptr(), kv_cache.data_ptr(), block_table.data_ptr(), seq_lens.data_ptr())

        if self._graph is not None and self._graph_key == key:
            self._graph.replay()
            return self._out

        self._launch(query, kv_cache, block_table, seq_lens)

        try:
            g = torch.cuda.CUDAGraph()
            with torch.cuda.graph(g):
                self._launch(query, kv_cache, block_table, seq_lens)
            self._graph = g
            self._graph_key = key
        except Exception:
            self._graph = None
            self._graph_key = None

        return self._out


def get_inputs():
    B, H, Hkv, D, L, P = BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE
    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)
    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1
    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)
    return [query, kv_cache, block_table, seq_lens]
