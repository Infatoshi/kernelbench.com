"""Multi-axis RoPE pre-attention prep fused kernel (Triton).

Fuses gather + per-axis mixing + rotate-half + transpose into a single
kernel with direct (B, H, T, D) output.

Key optimization:
- Group by (batch, time): one program loads all 6 cos/sin rows once and
  processes ALL heads at that (b,t). Eliminates H_total-way redundant
  cos/sin cache reads — the main bandwidth bottleneck.
- Process head_dim in two halves to avoid double-loading q/k.
- All rotation ops are element-local, no cross-thread communication.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "rope"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000"]


@triton.jit
def _mrope_kernel(
    q_ptr, k_ptr, out_q_ptr, out_k_ptr,
    cos_cache_ptr, sin_cache_ptr,
    pos_t_ptr, pos_h_ptr, pos_w_ptr,
    stride_q_b, stride_q_t, stride_q_h,
    stride_k_b, stride_k_t, stride_k_h,
    stride_out_q_b, stride_out_q_h, stride_out_q_t,
    stride_out_k_b, stride_out_k_h, stride_out_k_t,
    B, T, D,
    s0, s1, s2,
    half_D: tl.constexpr,
    Hq_: tl.constexpr,
    Hkv_: tl.constexpr,
):
    b = tl.program_id(0)
    t = tl.program_id(1)

    # --- Load position IDs once per (b, t) ---
    pos_offs = b * T + t
    pt = tl.load(pos_t_ptr + pos_offs)
    ph = tl.load(pos_h_ptr + pos_offs)
    pw = tl.load(pos_w_ptr + pos_offs)

    d_offs = tl.arange(0, half_D)

    # --- Load all 6 cos/sin rows once (shared across all heads) ---
    cos_t = tl.load(cos_cache_ptr + pt * D + d_offs)
    sin_t = tl.load(sin_cache_ptr + pt * D + d_offs)
    cos_h = tl.load(cos_cache_ptr + ph * D + d_offs)
    sin_h = tl.load(sin_cache_ptr + ph * D + d_offs)
    cos_w = tl.load(cos_cache_ptr + pw * D + d_offs)
    sin_w = tl.load(sin_cache_ptr + pw * D + d_offs)

    cos_t_2 = tl.load(cos_cache_ptr + pt * D + d_offs + half_D)
    sin_t_2 = tl.load(sin_cache_ptr + pt * D + d_offs + half_D)
    cos_h_2 = tl.load(cos_cache_ptr + ph * D + d_offs + half_D)
    sin_h_2 = tl.load(sin_cache_ptr + ph * D + d_offs + half_D)
    cos_w_2 = tl.load(cos_cache_ptr + pw * D + d_offs + half_D)
    sin_w_2 = tl.load(sin_cache_ptr + pw * D + d_offs + half_D)

    # --- Per-element axis mixing ---
    is_t = d_offs < s0
    is_h = (d_offs >= s0) & (d_offs < s0 + s1)
    cos_first = tl.where(is_t, cos_t, tl.where(is_h, cos_h, cos_w))
    sin_first = tl.where(is_t, sin_t, tl.where(is_h, sin_h, sin_w))
    cos_second = tl.where(is_t, cos_t_2, tl.where(is_h, cos_h_2, cos_w_2))
    sin_second = tl.where(is_t, sin_t_2, tl.where(is_h, sin_h_2, sin_w_2))

    # --- Process all q heads ---
    for h in range(Hq_):
        base_x = b * stride_q_b + t * stride_q_t + h * stride_q_h
        x_first = tl.load(q_ptr + base_x + d_offs)
        x_second = tl.load(q_ptr + base_x + d_offs + half_D)

        out_first = x_first * cos_first + (-x_second) * sin_first
        out_second = x_second * cos_second + x_first * sin_second

        base_out = b * stride_out_q_b + h * stride_out_q_h + t * stride_out_q_t
        tl.store(out_q_ptr + base_out + d_offs, out_first)
        tl.store(out_q_ptr + base_out + d_offs + half_D, out_second)

    # --- Process all k heads ---
    for h in range(Hkv_):
        base_x = b * stride_k_b + t * stride_k_t + h * stride_k_h
        x_first = tl.load(k_ptr + base_x + d_offs)
        x_second = tl.load(k_ptr + base_x + d_offs + half_D)

        out_first = x_first * cos_first + (-x_second) * sin_first
        out_second = x_second * cos_second + x_first * sin_second

        base_out = b * stride_out_k_b + h * stride_out_k_h + t * stride_out_k_t
        tl.store(out_k_ptr + base_out + d_offs, out_first)
        tl.store(out_k_ptr + base_out + d_offs + half_D, out_second)


class Model(nn.Module):
    def __init__(self, B: int, T: int, Hq: int, Hkv: int, D: int,
                 mrope_section: tuple[int, int, int], max_pos: int):
        super().__init__()
        self.B, self.T = B, T
        self.Hq, self.Hkv = Hq, Hkv
        self.D = D
        self.mrope_section = tuple(mrope_section)
        self.max_pos = max_pos

        inv_freq = 1.0 / (10000.0 ** (torch.arange(0, D, 2, dtype=torch.float32) / D))
        pos = torch.arange(max_pos, dtype=torch.float32)
        freqs = torch.outer(pos, inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        self.register_buffer("cos_cache", emb.cos().to(torch.bfloat16))
        self.register_buffer("sin_cache", emb.sin().to(torch.bfloat16))

    def forward(
        self,
        q: torch.Tensor, k: torch.Tensor,
        pos_t: torch.Tensor, pos_h: torch.Tensor, pos_w: torch.Tensor,
    ) -> tuple[torch.Tensor, torch.Tensor]:
        B, T, Hq, D = q.shape
        _, _, Hkv, _ = k.shape

        out_q = torch.empty(B, Hq, T, D, dtype=q.dtype, device=q.device)
        out_k = torch.empty(B, Hkv, T, D, dtype=k.dtype, device=k.device)

        s0, s1, s2 = self.mrope_section
        half_D = D // 2

        grid = (B, T)
        _mrope_kernel[grid](
            q, k, out_q, out_k,
            self.cos_cache, self.sin_cache,
            pos_t, pos_h, pos_w,
            q.stride(0), q.stride(1), q.stride(2),
            k.stride(0), k.stride(1), k.stride(2),
            out_q.stride(0), out_q.stride(1), out_q.stride(2),
            out_k.stride(0), out_k.stride(1), out_k.stride(2),
            B, T, D,
            s0, s1, s2,
            half_D=half_D,
            Hq_=Hq, Hkv_=Hkv,
            num_warps=2,
        )

        return out_q, out_k
