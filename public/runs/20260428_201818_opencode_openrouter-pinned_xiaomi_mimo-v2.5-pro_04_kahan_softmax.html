"""Custom Triton softmax with multi-block for large vocabs.

Small vocabs (<=32K): single-block 2-pass online softmax.
Large vocabs (>32K): 3-phase multi-block.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _softmax_single_kernel(
    X, Y,
    stride_x_row,
    stride_y_row,
    vocab: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """Single block per row, 2-pass online softmax."""
    row_idx = tl.program_id(0)
    x_row_ptr = X + row_idx * stride_x_row
    y_row_ptr = Y + row_idx * stride_y_row

    row_max = -float("inf")
    acc = 0.0

    for off in tl.range(0, vocab, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < vocab
        vals = tl.load(x_row_ptr + cols, mask=mask, other=-float("inf"),
                        cache_modifier=".ca")
        block_max = tl.max(vals, axis=0)
        new_max = tl.maximum(row_max, block_max)
        acc = acc * tl.exp(row_max - new_max) + tl.sum(tl.exp(vals - new_max), axis=0)
        row_max = new_max

    inv_sum = 1.0 / acc

    for off in tl.range(0, vocab, BLOCK_SIZE):
        cols = off + tl.arange(0, BLOCK_SIZE)
        mask = cols < vocab
        vals = tl.load(x_row_ptr + cols, mask=mask, other=0.0,
                        cache_modifier=".ca")
        out = tl.exp(vals - row_max) * inv_sum
        tl.store(y_row_ptr + cols, out, mask=mask, cache_modifier=".cs")


@triton.jit
def _softmax_phase1(
    X, ROW_MAX, ROW_SUM,
    stride_x_row,
    vocab: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
    N_SPLITS: tl.constexpr,
):
    """Phase 1: per-partition max and exp-sum."""
    pid = tl.program_id(0)
    row_idx = pid // N_SPLITS
    split_idx = pid % N_SPLITS

    x_row_ptr = X + row_idx * stride_x_row
    part_start = split_idx * BLOCK_SIZE
    part_end = tl.minimum((split_idx + 1) * BLOCK_SIZE, vocab)

    cols = part_start + tl.arange(0, BLOCK_SIZE)
    mask = cols < part_end
    vals = tl.load(x_row_ptr + cols, mask=mask, other=-float("inf"),
                    cache_modifier=".ca")
    m = tl.max(vals, axis=0)
    e = tl.exp(vals - m)
    s = tl.sum(e, axis=0)

    tl.store(ROW_MAX + row_idx * N_SPLITS + split_idx, m)
    tl.store(ROW_SUM + row_idx * N_SPLITS + split_idx, s)


@triton.jit
def _softmax_phase2(
    ROW_MAX, ROW_SUM, GLOBAL_MAX, INV_SUM,
    N_SPLITS: tl.constexpr,
):
    """Phase 2: reduce across partitions."""
    row_idx = tl.program_id(0)

    max_ptrs = ROW_MAX + row_idx * N_SPLITS + tl.arange(0, N_SPLITS)
    part_maxes = tl.load(max_ptrs)
    g_max = tl.max(part_maxes, axis=0)

    sum_ptrs = ROW_SUM + row_idx * N_SPLITS + tl.arange(0, N_SPLITS)
    part_sums = tl.load(sum_ptrs)
    rescaled = part_sums * tl.exp(part_maxes - g_max)
    g_sum = tl.sum(rescaled, axis=0)

    tl.store(GLOBAL_MAX + row_idx, g_max)
    tl.store(INV_SUM + row_idx, 1.0 / g_sum)


@triton.jit
def _softmax_phase3(
    X, Y, GLOBAL_MAX, INV_SUM,
    stride_x_row,
    stride_y_row,
    vocab: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
    N_SPLITS: tl.constexpr,
):
    """Phase 3: normalize each partition."""
    pid = tl.program_id(0)
    row_idx = pid // N_SPLITS
    split_idx = pid % N_SPLITS

    x_row_ptr = X + row_idx * stride_x_row
    y_row_ptr = Y + row_idx * stride_y_row
    g_max = tl.load(GLOBAL_MAX + row_idx)
    inv_s = tl.load(INV_SUM + row_idx)

    part_start = split_idx * BLOCK_SIZE
    part_end = tl.minimum((split_idx + 1) * BLOCK_SIZE, vocab)

    cols = part_start + tl.arange(0, BLOCK_SIZE)
    mask = cols < part_end
    vals = tl.load(x_row_ptr + cols, mask=mask, other=0.0,
                    cache_modifier=".ca")
    out = tl.exp(vals - g_max) * inv_s
    tl.store(y_row_ptr + cols, out, mask=mask, cache_modifier=".cs")
