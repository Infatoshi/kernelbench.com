import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_attn_decode_grouped_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    NUM_PAGES: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
    PAGES_PER_TILE: tl.constexpr,
    SCALE: tl.constexpr,
):
    b = tl.program_id(0)
    kv_h = tl.program_id(1)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, HEAD_DIM)
    offs_n = tl.arange(0, BLOCK_N)
    q_heads = kv_h * GROUP_SIZE + offs_g

    q = tl.load(
        query + (b * NUM_HEADS + q_heads[:, None]) * HEAD_DIM + offs_d[None, :]
    )
    seqlen = tl.load(seq_lens + b)

    m = tl.full((GROUP_SIZE,), -float("inf"), tl.float32)
    l = tl.zeros((GROUP_SIZE,), tl.float32)
    acc = tl.zeros((GROUP_SIZE, HEAD_DIM), tl.float32)

    for page_base in range(0, NUM_PAGES, PAGES_PER_TILE):
        page_ids = page_base + offs_n // PAGE_SIZE
        page_offs = offs_n % PAGE_SIZE
        token_idx = page_ids * PAGE_SIZE + page_offs
        valid_n = (page_ids < NUM_PAGES) & (token_idx < seqlen)
        block_ids = tl.load(
            block_table + b * NUM_PAGES + page_ids,
            mask=page_ids < NUM_PAGES,
            other=0,
        )

        k_token = block_ids[None, :] * PAGE_SIZE + page_offs[None, :]
        v_token = block_ids[:, None] * PAGE_SIZE + page_offs[:, None]
        k_base = ((k_token * NUM_KV_HEADS + kv_h) * (HEAD_DIM * 2)) + offs_d[:, None]
        v_base = (
            ((v_token * NUM_KV_HEADS + kv_h) * (HEAD_DIM * 2))
            + HEAD_DIM
            + offs_d[None, :]
        )

        k = tl.load(kv_cache + k_base, mask=valid_n[None, :], other=0.0)
        v = tl.load(kv_cache + v_base, mask=valid_n[:, None], other=0.0)
        scores = tl.dot(q, k, out_dtype=tl.float32) * SCALE
        scores = tl.where(valid_n[None, :], scores, -float("inf"))

        tile_m = tl.max(scores, axis=1)
        m_new = tl.maximum(m, tile_m)
        alpha = tl.exp(m - m_new)
        p = tl.exp(scores - m_new[:, None])
        p = tl.where(valid_n[None, :], p, 0.0)
        l_new = l * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v, out_dtype=tl.float32)
        m = m_new
        l = l_new

    acc = acc / l[:, None]
    tl.store(
        out + (b * NUM_HEADS + q_heads[:, None]) * HEAD_DIM + offs_d[None, :],
        acc,
    )


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.num_pages = triton.cdiv(seq_len, page_size)
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer(
            "_dummy",
            torch.zeros(1, dtype=torch.bfloat16),
            persistent=False,
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        out = torch.empty_like(query)
        grid = (self.batch, self.num_kv_heads)
        _paged_attn_decode_grouped_kernel[grid](
            query,
            kv_cache,
            block_table,
            seq_lens,
            out,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
            self.num_pages,
            self.group_size,
            self.page_size * 8,
            8,
            self.scale,
            num_warps=4,
        )
        return out


def get_inputs():

    h = NUM_HEADS
    hkv = NUM_KV_HEADS
    d = HEAD_DIM
    length = SEQ_LEN
    page = PAGE_SIZE

    pages_per_seq = (length + page - 1) // page
