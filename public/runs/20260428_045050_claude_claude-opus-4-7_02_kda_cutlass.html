"""Custom KDA forward (chunk form) using Triton kernels for SM120.

Implementation notes:
  - g is rescaled by RCP_LN2 so we can use exp2 throughout (faster than exp).
  - In-chunk cumsum on g is fused into the intra kernel via tl.cumsum.
  - Phase 1 (intra): per-chunk forward substitution on the (I + L) factor and
    materialize the chunk-local Akk_inv, w, u tensors.
  - Phase 2 (inter): sequential per-(b, h) loop over chunks computing the state
    h and the output o, with the V dim split into BV-tiles for parallelism.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

RCP_LN2 = 1.4426950408889634  # log2(e), so exp(x) = exp2(x * RCP_LN2)


@triton.jit
def _kda_intra_kernel(
    k_ptr,
    v_ptr,
    g_in_ptr,
    g_out_ptr,
    beta_ptr,
    A_ptr,
    w_ptr,
    u_ptr,
    T,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    SCALE_LOG: tl.constexpr,
):
    """Per-chunk preparation: cumsum g; build Akk_inv*beta_col; w, u.

    Grid: (B*H, NT)
    """
    i_bh = tl.program_id(0)
    i_n = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H
    NT = T // BT

    offs_t = tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = tl.arange(0, V)

    # ---------- Pointers ----------
    # k, g: (B, T, H, K) — strides (T*H*K, H*K, K, 1)
    k_base = i_b * T * H * K + i_n * BT * H * K + i_h * K
    g_base = k_base
    # v: (B, T, H, V)
    v_base = i_b * T * H * V + i_n * BT * H * V + i_h * V
    # beta: (B, T, H)
    beta_base = i_b * T * H + i_n * BT * H + i_h
    # A: (B, NT, H, BT, BT) contiguous on the last two dims
    A_base = i_b * NT * H * BT * BT + i_n * H * BT * BT + i_h * BT * BT

    # ---------- Load k, g, beta ----------
    kg_ptrs_t = offs_t[:, None] * H * K + offs_k[None, :]
    b_k = tl.load(k_ptr + k_base + kg_ptrs_t)            # (BT, K) bf16
    b_g_raw = tl.load(g_in_ptr + g_base + kg_ptrs_t).to(tl.float32) * SCALE_LOG
    b_g = tl.cumsum(b_g_raw, axis=0)                      # (BT, K) fp32

    # Save the cumsumed g back to global (used by inter kernel).
    tl.store(g_out_ptr + g_base + kg_ptrs_t, b_g)

    b_beta = tl.load(beta_ptr + beta_base + offs_t * H).to(tl.float32)  # (BT,)

    # ---------- Akk_raw = -beta_row * (k*exp(g)) @ (k*exp(-g)).T ----------
    b_eg = tl.exp2(b_g)             # (BT, K) fp32
    b_eng = tl.exp2(-b_g)           # (BT, K) fp32
    b_kf = b_k.to(tl.float32) * b_eg
    b_kg = b_k.to(tl.float32) * b_eng
    b_A = tl.dot(b_kf.to(b_k.dtype), tl.trans(b_kg).to(b_k.dtype))  # (BT, BT) fp32
    mask_strict_lower = offs_t[:, None] > offs_t[None, :]
    b_A = tl.where(mask_strict_lower, -b_A * b_beta[:, None], 0.0)

    # ---------- Forward substitute (BT iterations) ----------
    # for i in 1..BT-1: A[i, :i] += sum_m A[i, m] * A[m, :i]   for m < i
    for i in tl.static_range(1, BT):
        is_row = (offs_t == i)
        # row i of A (BT,)
        row_i = tl.sum(tl.where(is_row[:, None], b_A, 0.0), axis=0)
        # (row_i @ A)[j] = sum_m row_i[m] * A[m, j]
        update = tl.sum(row_i[:, None] * b_A, axis=0)
        # write only the lower part of row i
        write_mask = is_row[:, None] & (offs_t[None, :] < i)
        b_A = tl.where(write_mask, b_A + update[None, :], b_A)

    # Add identity on the diagonal
    b_A = tl.where(offs_t[:, None] == offs_t[None, :], b_A + 1.0, b_A)
    # Multiply by beta_col: A *= beta[None, :]
    b_A = b_A * b_beta[None, :]

    # ---------- Store A_inv (fp32 for downstream precision) ----------
    A_ptrs = offs_t[:, None] * BT + offs_t[None, :]
    tl.store(A_ptr + A_base + A_ptrs, b_A)

    # ---------- w = A @ (g.exp() * k) ----------
    b_w = tl.dot(b_A.to(b_k.dtype), b_kf.to(b_k.dtype))   # (BT, K) fp32
    tl.store(w_ptr + k_base + kg_ptrs_t, b_w.to(b_k.dtype))

    # ---------- u = A @ v ----------
    v_ptrs_t = offs_t[:, None] * H * V + offs_v[None, :]
    b_v = tl.load(v_ptr + v_base + v_ptrs_t)              # (BT, V) bf16
    b_u = tl.dot(b_A.to(b_v.dtype), b_v)                  # (BT, V) fp32
    tl.store(u_ptr + v_base + v_ptrs_t, b_u.to(b_v.dtype))


@triton.jit
def _kda_inter_kernel(
    q_ptr,
    k_ptr,
    g_ptr,
    w_ptr,
    u_ptr,
    o_ptr,
    scale,
    T,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
    SCALE_LOG: tl.constexpr,
):
    """Inter-chunk recurrence: maintain state S, compute o.

    Grid: (B*H, V/BV)
    Each program handles one (b, h) pair and one V-slice. Sequential over chunks.
    """
    i_bh = tl.program_id(0)
    i_v = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H
    NT = T // BT

    offs_t = tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = tl.arange(0, BV)

    v_lo = i_v * BV

    # Strict-upper mask for Aqk (zero out strict-upper, keep diag and below)
    mask_su = offs_t[:, None] < offs_t[None, :]

    # State S: (K, BV) fp32
    b_S = tl.zeros((K, BV), dtype=tl.float32)

    for i_n in range(NT):
        # Pointers
        k_base = i_b * T * H * K + i_n * BT * H * K + i_h * K
        v_base = i_b * T * H * V + i_n * BT * H * V + i_h * V + v_lo

        kg_ptrs_t = offs_t[:, None] * H * K + offs_k[None, :]
        v_ptrs_t = offs_t[:, None] * H * V + offs_v[None, :]

        # Load q, k, g, w, u_slice
        b_q = tl.load(q_ptr + k_base + kg_ptrs_t)              # (BT, K) bf16
        b_k = tl.load(k_ptr + k_base + kg_ptrs_t)              # (BT, K) bf16
        b_g = tl.load(g_ptr + k_base + kg_ptrs_t)              # (BT, K) fp32
        b_w = tl.load(w_ptr + k_base + kg_ptrs_t)              # (BT, K) bf16
        b_u = tl.load(u_ptr + v_base + v_ptrs_t)               # (BT, BV) bf16

        # Aqk[c, c'] = q[c] * exp(g[c] - g[c']) * k[c']
        # = (q * exp(g))[c] @ (k * exp(-g))[c'].T
        b_eg = tl.exp2(b_g)
        b_eng = tl.exp2(-b_g)
        # apply scale to q here so q @ S also gets scaled
        b_qe = (b_q.to(tl.float32) * b_eg * scale).to(b_q.dtype)
        b_kf = (b_k.to(tl.float32) * b_eng).to(b_k.dtype)
        b_Aqk = tl.dot(b_qe, tl.trans(b_kf))                   # (BT, BT) fp32
        b_Aqk = tl.where(mask_su, 0.0, b_Aqk)

        # v_new = u - w @ S        (BT, BV)
        b_v_new = b_u.to(tl.float32) - tl.dot(b_w, b_S.to(b_w.dtype))

        # o = (q*exp(g)*scale) @ S + Aqk @ v_new
        b_o = tl.dot(b_qe, b_S.to(b_qe.dtype))
        b_o += tl.dot(b_Aqk.to(b_w.dtype), b_v_new.to(b_w.dtype))

        # Store o
        tl.store(o_ptr + v_base + v_ptrs_t, b_o.to(b_q.dtype))

        # Update S: S = exp(g_last) * S + (k * exp(g_last - g)).T @ v_new
        # g_last = g[BT-1, :]
        g_last_off = (BT - 1) * H * K
        b_g_last = tl.load(g_ptr + k_base + g_last_off + offs_k)  # (K,) fp32
        b_S = b_S * tl.exp2(b_g_last)[:, None]
        b_state_term = (b_k.to(tl.float32) * tl.exp2(b_g_last[None, :] - b_g)).to(b_k.dtype)
        b_S = b_S + tl.dot(tl.trans(b_state_term), b_v_new.to(b_state_term.dtype))


def kda_forward(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    """KDA forward (chunk form). Returns o of shape (B, T, H, V) with v's dtype."""
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0
    NT = T // BT
    assert q.is_contiguous() and k.is_contiguous() and v.is_contiguous() and beta.is_contiguous()

    # Workspaces
    g_cum = torch.empty_like(g)                                       # fp32, (B, T, H, K)
    A_inv = torch.empty(B, NT, H, BT, BT, device=q.device, dtype=torch.float32)
    w = torch.empty_like(k)                                           # bf16, (B, T, H, K)
    u = torch.empty_like(v)                                           # bf16, (B, T, H, V)
    o = torch.empty_like(v)                                           # bf16, (B, T, H, V)

    # Phase 1: intra-chunk preparation
    grid_intra = (B * H, NT)
    _kda_intra_kernel[grid_intra](
        k, v, g, g_cum, beta, A_inv, w, u,
        T,
        H=H, K=K, V=V, BT=BT,
        SCALE_LOG=RCP_LN2,
        num_warps=4,
        num_stages=2,
    )

    # Phase 2: inter-chunk recurrence
    BV = 32  # split V into BV-sized tiles
    assert V % BV == 0
    grid_inter = (B * H, V // BV)
    _kda_inter_kernel[grid_inter](
        q, k, g_cum, w, u, o, scale,
        T,
        H=H, K=K, V=V, BT=BT, BV=BV,
        SCALE_LOG=RCP_LN2,
        num_warps=4,
        num_stages=2,
    )
    return o
