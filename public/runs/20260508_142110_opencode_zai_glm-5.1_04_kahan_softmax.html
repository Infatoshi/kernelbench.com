@triton.jit
def _softmax_single_kernel(
    X_ptr, Y_ptr,
    batch, vocab,
    stride_xb, stride_xv,
    stride_yb, stride_yv,
    BLOCK: tl.constexpr,
):
    row_idx = tl.program_id(0)
    if row_idx >= batch:
        return

    x_row = X_ptr + row_idx * stride_xb
    y_row = Y_ptr + row_idx * stride_yb

    row_max = -float('inf')
    for block_start in range(0, vocab, BLOCK):
        offs = block_start + tl.arange(0, BLOCK)
        mask = offs < vocab
        x = tl.load(x_row + offs, mask=mask, other=-float('inf'))
        row_max = tl.maximum(row_max, tl.max(x, axis=0))

    total = 0.0
    comp = 0.0
    for block_start in range(0, vocab, BLOCK):
        offs = block_start + tl.arange(0, BLOCK)
        mask = offs < vocab
        x = tl.load(x_row + offs, mask=mask, other=-float('inf'))
        e = tl.exp(x - row_max)
        tl.store(y_row + offs, e, mask=mask)
        block_sum = tl.sum(e, axis=0)
        y = block_sum - comp
        t = total + y
        comp = (t - total) - y
        total = t

    inv_total = 1.0 / total
    for block_start in range(0, vocab, BLOCK):
        offs = block_start + tl.arange(0, BLOCK)
        mask = offs < vocab
        e = tl.load(y_row + offs, mask=mask, other=0.0)
        tl.store(y_row + offs, e * inv_total, mask=mask)


@triton.jit
def _find_max_sum_kernel(
    X_ptr, buf_max_ptr, buf_sum_ptr,
    batch, vocab, num_chunks,
    stride_xb,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    row_idx = pid // num_chunks
    chunk_idx = pid % num_chunks
    if row_idx >= batch:
        return

    chunk_start = chunk_idx * BLOCK
    x_base = X_ptr + row_idx * stride_xb

    offs = chunk_start + tl.arange(0, BLOCK)
    mask = offs < vocab
    x = tl.load(x_base + offs, mask=mask, other=-float('inf'))
    chunk_max = tl.max(x, axis=0)
    e = tl.exp(x - chunk_max)
    chunk_sum = tl.sum(e, axis=0)

    tl.store(buf_max_ptr + row_idx * num_chunks + chunk_idx, chunk_max)
    tl.store(buf_sum_ptr + row_idx * num_chunks + chunk_idx, chunk_sum)


@triton.jit
def _finalize_kernel(
    X_ptr, Y_ptr, buf_max_ptr, buf_sum_ptr,
    batch, vocab, num_chunks,
    stride_xb, stride_yb,
    BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    row_idx = pid // num_chunks
    chunk_idx = pid % num_chunks
    if row_idx >= batch:
        return

    global_max = -float('inf')
    for i in range(num_chunks):
        m = tl.load(buf_max_ptr + row_idx * num_chunks + i)
        global_max = tl.maximum(global_max, m)

    total = 0.0
    comp = 0.0
    for i in range(num_chunks):
        cm = tl.load(buf_max_ptr + row_idx * num_chunks + i)
        cs = tl.load(buf_sum_ptr + row_idx * num_chunks + i)
        adjusted = cs * tl.exp(cm - global_max)
        y = adjusted - comp
        t = total + y
        comp = (t - total) - y
        total = t

    inv_total = 1.0 / total
    chunk_start = chunk_idx * BLOCK
    x_base = X_ptr + row_idx * stride_xb
    y_base = Y_ptr + row_idx * stride_yb

    offs = chunk_start + tl.arange(0, BLOCK)
    mask = offs < vocab
    x = tl.load(x_base + offs, mask=mask, other=-float('inf'))
    e = tl.exp(x - global_max)
    tl.store(y_base + offs, e * inv_total, mask=mask)


class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = batch
        self.vocab = vocab

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        y = torch.empty_like(x)
        BLOCK = 4096
        num_chunks = triton.cdiv(self.vocab, BLOCK)

        if self.batch >= 32:
            _softmax_single_kernel[(self.batch,)](
                x, y,
                self.batch, self.vocab,
                x.stride(0), x.stride(1),
                y.stride(0), y.stride(1),
                BLOCK=BLOCK,
            )
        else:
            total_programs = self.batch * num_chunks
            buf_max = torch.empty(self.batch, num_chunks, dtype=torch.float32, device=x.device)
            buf_sum = torch.empty(self.batch, num_chunks, dtype=torch.float32, device=x.device)

            _find_max_sum_kernel[(total_programs,)](
                x, buf_max, buf_sum,
                self.batch, self.vocab, num_chunks,
                x.stride(0),
                BLOCK=BLOCK,
            )
            _finalize_kernel[(total_programs,)](
                x, y, buf_max, buf_sum,
                self.batch, self.vocab, num_chunks,
                x.stride(0), y.stride(0),
                BLOCK=BLOCK,
            )
        return y
