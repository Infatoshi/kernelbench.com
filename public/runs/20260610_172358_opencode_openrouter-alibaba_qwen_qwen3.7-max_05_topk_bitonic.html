"""Top-k via Triton: packed int64 sort with tl.topk."""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def argmax_kernel(x_ptr, out_vals, out_idxs, n, batch_stride, out_stride,
                  BLOCK_N: tl.constexpr):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_N)
    mask = offs < n
    x = tl.load(x_ptr + row * batch_stride + offs, mask=mask, other=float('-inf'))
    idx = tl.argmax(x, axis=0)
    val = tl.load(x_ptr + row * batch_stride + idx)
    tl.store(out_vals + row * out_stride, val)
    tl.store(out_idxs + row * out_stride, idx.to(tl.int64))


@triton.jit
def topk_kernel(x_ptr, out_vals, out_idxs, batch_stride, out_stride,
                n, K: tl.constexpr, BLOCK_N: tl.constexpr):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_N)
    mask = offs < n
    x = tl.load(x_ptr + row * batch_stride + offs, mask=mask, other=float('-inf'))

    bits = x.to(tl.uint32, bitcast=True)
    neg = bits >= 0x80000000
    transformed = tl.where(neg, ~bits, bits | 0x80000000)
    packed = (transformed.to(tl.int64) << 32) | offs.to(tl.int64)
    packed = packed ^ (tl.full((), 1, tl.int64) << 63)

    top_packed = tl.topk(packed, K)

    top_flipped = top_packed ^ (tl.full((), 1, tl.int64) << 63)
    top_transformed = (top_flipped >> 32).to(tl.uint32)
    top_idx = (top_flipped & 0xFFFFFFFF).to(tl.int32)
    was_pos = ((top_transformed >> 31) & 1).to(tl.int1)
    orig_bits = tl.where(was_pos,
                         top_transformed ^ 0x80000000,
                         ~top_transformed)
    vals = orig_bits.to(tl.float32, bitcast=True)

    k_offs = tl.arange(0, K)
    tl.store(out_vals + row * out_stride + k_offs, vals)
    tl.store(out_idxs + row * out_stride + k_offs, top_idx.to(tl.int64))


@triton.jit
def local_topk_kernel(x_ptr, inter_vals, inter_idxs,
                      batch_stride, chunk_size, chunks_per_row, n,
                      K: tl.constexpr, BLOCK_N: tl.constexpr):
    pid = tl.program_id(0)
    row = pid // chunks_per_row
    chunk = pid % chunks_per_row

    chunk_start = chunk * chunk_size
    offs = tl.arange(0, BLOCK_N)
    global_offs = chunk_start + offs
    mask = global_offs < n
    x = tl.load(x_ptr + row * batch_stride + global_offs, mask=mask,
                other=float('-inf'))

    bits = x.to(tl.uint32, bitcast=True)
    neg = bits >= 0x80000000
    transformed = tl.where(neg, ~bits, bits | 0x80000000)
    packed = (transformed.to(tl.int64) << 32) | global_offs.to(tl.int64)
    packed = packed ^ (tl.full((), 1, tl.int64) << 63)

    top_packed = tl.topk(packed, K)

    top_flipped = top_packed ^ (tl.full((), 1, tl.int64) << 63)
    top_transformed = (top_flipped >> 32).to(tl.uint32)
    top_idx = (top_flipped & 0xFFFFFFFF).to(tl.int32)
    was_pos = ((top_transformed >> 31) & 1).to(tl.int1)

                         top_transformed ^ 0x80000000,
                         ~top_transformed)
    vals = orig_bits.to(tl.float32, bitcast=True)

    k_offs = tl.arange(0, K)

@triton.jit
def merge_kernel(inter_vals, inter_idxs, out_vals, out_idxs,
                 chunks_per_row,
                 K: tl.constexpr, TOTAL: tl.constexpr, TPAD: tl.constexpr):
    row = tl.program_id(0)
    offs = tl.arange(0, TPAD)
    base = row * chunks_per_row * K

    mask = offs < TOTAL
    vals = tl.load(inter_vals + base + offs, mask=mask, other=float('-inf'))
    idxs = tl.load(inter_idxs + base + offs, mask=mask, other=0)

    bits = vals.to(tl.uint32, bitcast=True)
    neg = bits >= 0x80000000
    transformed = tl.where(neg, ~bits, bits | 0x80000000)
    packed = (transformed.to(tl.int64) << 32) | idxs.to(tl.int64)
    packed = packed ^ (tl.full((), 1, tl.int64) << 63)

    top_packed = tl.topk(packed, K)

    top_flipped = top_packed ^ (tl.full((), 1, tl.int64) << 63)
    top_transformed = (top_flipped >> 32).to(tl.uint32)

def _next_pow2(x):
    p = 1
    while p < x:
        p <<= 1
    return p
