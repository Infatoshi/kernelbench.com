+
+Fused unpack + GEMM in Triton. The packed weight w_q is (K//2, N) uint8 with two
+int4 weights per byte: low nibble = even-K row, high nibble = odd-K row. We never
+materialize the full bf16 weight; we unpack and dequant inside the GEMM main loop.
+
+Because the dot product over K is order-invariant (as long as x and w use the same
+K permutation), we split each packed byte tile into its even-K (low-nibble) and
+odd-K (high-nibble) halves and accumulate two sub-dots. This avoids an expensive
+nibble-interleave.
+"""

+import torch
+import torch.nn as nn
+import triton
+import triton.language as tl
+
+GROUP_SIZE = 128
+

+@triton.jit
+def _w4a16_kernel(
+    x_ptr, wq_ptr, scl_ptr, zero_ptr, out_ptr,
+    M, N, K,
+    stride_xm, stride_xk,
+    stride_wq_kh, stride_wq_n,
+    stride_scl_g, stride_scl_n,
+    stride_zero_g, stride_zero_n,
+    stride_om, stride_on,

+    pid_n = tl.program_id(1)
+

+    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
+    m_mask = offs_m < M
+    n_mask = offs_n < N
+
+    # BLOCK_K is along the *unpacked* K; each packed byte covers 2 K. We process
+    # BLOCK_K2 = BLOCK_K // 2 packed rows per iteration.
+    BLOCK_K2: tl.constexpr = BLOCK_K // 2
+    offs_k2 = tl.arange(0, BLOCK_K2)
+
+    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
+
+    n_groups = K // GROUP_SIZE
+    # number of packed-row tiles
+    n_kh_tiles = (K // 2) // BLOCK_K2
+
+    for t in range(0, n_kh_tiles):
+        kh0 = t * BLOCK_K2
+        kh = kh0 + offs_k2  # packed row indices
+

+        hi = ((packed >> 4) & 0xF).to(tl.float32)
+

+        g = kh // (GROUP_SIZE // 2)
+        scl_ptrs = scl_ptr + g[:, None] * stride_scl_g + offs_n[None, :] * stride_scl_n
+        zero_ptrs = zero_ptr + g[:, None] * stride_zero_g + offs_n[None, :] * stride_zero_n
+        scl = tl.load(scl_ptrs, mask=n_mask[None, :], other=0).to(tl.float32)
+        zero = tl.load(zero_ptrs, mask=n_mask[None, :], other=0).to(tl.float32)
+
+        w_lo = ((lo - zero) * scl).to(tl.bfloat16)   # even-K weights
+        w_hi = ((hi - zero) * scl).to(tl.bfloat16)   # odd-K weights
+
+        # x even/odd columns: even k = 2*kh, odd k = 2*kh+1
+        k_even = 2 * kh0 + 2 * offs_k2

+        x_odd = tl.load(x_odd_ptrs, mask=m_mask[:, None], other=0.0)
+

+        acc += tl.dot(x_odd, w_hi, out_dtype=tl.float32)
+

+    out_ptrs = out_ptr + offs_m[:, None] * stride_om + offs_n[None, :] * stride_on
