"""Kimi Delta Attention (KDA) forward, chunk form - correct implementation.

This implements the chunk-parallel KDA forward pass following the
same algorithm as reference.py (which follows fla/ops/kda/naive.py).
"""

from __future__ import annotations

import torch
import torch.nn as nn
from einops import rearrange

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


def _chunk_kda_forward(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    """KDA forward implementation matching reference.py algorithm exactly."""
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0, f"T={T} must be a multiple of chunk_size={BT}"
    NT = T // BT

    # Convert to float32 for computation (more stable)
    q = q.to(torch.float32)
    k = k.to(torch.float32)
    v = v.to(torch.float32)
    g = g.to(torch.float32)
    beta = beta.to(torch.float32)

    q = q * scale

    # Reshape to (B, H, NT, BT, ...)
    q = rearrange(q, "b (n c) h d -> b h n c d", c=BT)
    k = rearrange(k, "b (n c) h d -> b h n c d", c=BT)
    v = rearrange(v, "b (n c) h d -> b h n c d", c=BT)
    g = rearrange(g, "b (n c) h d -> b h n c d", c=BT)
    beta = rearrange(beta, "b (n c) h -> b h n c", c=BT)

    # Cumsum g along chunk dimension
    g = g.cumsum(-2)

    # ---- Build A_kk (intra-chunk K-K interaction) ----
    mask_diag_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=0)
    A = torch.zeros(*q.shape[:-1], BT, dtype=torch.float32, device=q.device)

    for i in range(BT):
        k_i = k[..., i, :]  # (B, H, NT, K)
        g_i = g[..., i:i+1, :]  # (B, H, NT, 1, K)
        # A[..., i] = (k * (g - g_i).exp()) @ k_i
        # Implement einsum manually: sum over K dimension
        diff = g - g_i  # (B, H, NT, BT, K)
        exp_diff = torch.exp(diff)  # (B, H, NT, BT, K)
        weighted_k = k * exp_diff  # (B, H, NT, BT, K)
        # Dot product: (weighted_k * k_i.unsqueeze(-2)).sum(dim=-1)
        A[..., i] = (weighted_k * k_i.unsqueeze(-2)).sum(dim=-1)

    A = A * beta[..., None]
    A = -A.masked_fill(mask_diag_upper, 0)

    for i in range(1, BT):
        A[..., i, :i] = A[..., i, :i].clone() + (A[..., i, :, None].clone() * A[..., :, :i].clone()).sum(-2)

    A = (A + torch.eye(BT, dtype=torch.float32, device=q.device)) * beta[..., None, :]

    # ---- Compute w and u ----
    w = A @ (torch.exp(g) * k)
    u = A @ v

    # ---- Recurrent inter-chunk pass ----
    S = torch.zeros(B, H, K, V, dtype=torch.float32, device=q.device)
    o = torch.zeros_like(v)
    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=q.device), diagonal=1)

    for i in range(NT):
        q_i = q[:, :, i]  # (B, H, BT, K)
        k_i = k[:, :, i]  # (B, H, BT, K)
        u_i = u[:, :, i]  # (B, H, BT, V)
        g_i = g[:, :, i]  # (B, H, BT, K)
        w_i = w[:, :, i]  # (B, H, BT, K)

        # Build Aqk: (B, H, BT, BT)
        Aqk = torch.zeros(B, H, BT, BT, dtype=torch.float32, device=q.device)
        for j in range(BT):
            k_j = k_i[:, :, j]  # (B, H, K)
            g_j = g_i[:, :, j:j+1, :]  # (B, H, 1, K)
            # Aqk[..., j] = (q_i * (g_i - g_j).exp()) @ k_j
            diff = g_i - g_j  # (B, H, BT, K)
            exp_diff = torch.exp(diff)  # (B, H, BT, K)
            weighted_q = q_i * exp_diff  # (B, H, BT, K)
            # Dot product with k_j
            Aqk[..., j] = (weighted_q * k_j.unsqueeze(-2)).sum(dim=-1)

        Aqk = Aqk.masked_fill(mask_strict_upper, 0)

        # v_i = u_i - w_i @ S
        # w_i: (B, H, BT, K), S: (B, H, K, V)
        wS = torch.bmm(w_i.reshape(-1, BT, K), S.reshape(-1, K, V)).reshape(B, H, BT, V)
        v_i = u_i - wS

        # o[:, :, i] = (q_i * exp(g_i)) @ S + Aqk @ v_i
        qS = torch.bmm((q_i * torch.exp(g_i)).reshape(-1, BT, K), S.reshape(-1, K, V)).reshape(B, H, BT, V)
        Av = torch.bmm(Aqk.reshape(-1, BT, BT), v_i.reshape(-1, BT, V)).reshape(B, H, BT, V)
        o[:, :, i] = qS + Av

        # Update S: S = S * exp(g_i[:, :, -1]) + (exp(g_i[:, :, -1:] - g_i) * k_i) @ v_i
        g_last = g_i[:, :, -1, :]  # (B, H, K)
        S = S * g_last.unsqueeze(-1).exp()  # (B, H, K, V)

        # k_i @ v_i with decay
        decay = torch.exp(g_last.unsqueeze(-2) - g_i)  # (B, H, BT, K)
        weighted_k = k_i * decay  # (B, H, BT, K)
        # S_update = weighted_k^T @ v_i
        S_update = torch.bmm(weighted_k.reshape(-1, BT, K).transpose(-2, -1), v_i.reshape(-1, BT, V)).reshape(B, H, K, V)
        S = S + S_update

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o.to(dtype)


class Model(nn.Module):
    """KDA forward (chunk form). No learned parameters; all inputs are activations."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        # No learned params; declare a dummy buffer so state_dict is well-defined.
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        return _chunk_kda_forward(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)


# Module-level shape shims (overridden by check.py / benchmark.py per shape).
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    """Return a list of activations for one forward call."""
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    # log-decay: small negative numbers so exp(g) is in (0, 1).
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
