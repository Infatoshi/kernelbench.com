import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _paged_attention_kernel(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr, Out_ptr,
    stride_q_b, stride_q_h,
    stride_kv_b, stride_kv_p, stride_kv_h,
    stride_bt_b, stride_bt_p,
    stride_o_b, stride_o_h,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    SCALE: tl.constexpr,
    PAGES_PER_ITER: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // NUM_HEADS
    head_idx = pid % NUM_HEADS
    kv_head = head_idx // (NUM_HEADS // NUM_KV_HEADS)
    seq_len = tl.load(SL_ptr + batch_idx)
    N_TOKENS: tl.constexpr = PAGES_PER_ITER * PAGE_SIZE

    q_ptrs = Q_ptr + batch_idx * stride_q_b + head_idx * stride_q_h
    q_ptrs = q_ptrs + tl.arange(0, HEAD_DIM)
    q = tl.load(q_ptrs).to(tl.float32) * SCALE

    m = float('-inf')
    l = 0.0
    acc = tl.zeros([HEAD_DIM], tl.float32)

    for block_iter in range(0, MAX_BLOCKS, PAGES_PER_ITER):
        page_start = block_iter * PAGE_SIZE
        if page_start < seq_len:
            n_valid_req = seq_len - page_start
            n_valid_this = tl.where(n_valid_req < N_TOKENS, n_valid_req, N_TOKENS)

            tok_offs = tl.arange(0, N_TOKENS)
            valid = tok_offs < n_valid_this

            if PAGES_PER_ITER == 1:
                blk_id = tl.load(BT_ptr + batch_idx * stride_bt_b + block_iter * stride_bt_p)
            else:
                pg_off = tok_offs // PAGE_SIZE
                bt_base = BT_ptr + batch_idx * stride_bt_b + block_iter * stride_bt_p
                blk_id = tl.load(bt_base + pg_off, mask=valid, other=0)

            tok_in_page = tok_offs % PAGE_SIZE
            kv_base = KV_ptr + kv_head * stride_kv_h
            s_off = tok_in_page[:, None] * stride_kv_p + blk_id[:, None] * stride_kv_b

            k_ptrs = kv_base + s_off + tl.arange(0, HEAD_DIM)[None, :]
            k = tl.load(k_ptrs, mask=valid[:, None], other=0.0).to(tl.float32)

            v_ptrs = kv_base + s_off + (HEAD_DIM + tl.arange(0, HEAD_DIM))[None, :]
            v = tl.load(v_ptrs, mask=valid[:, None], other=0.0).to(tl.float32)

            s = tl.sum(q[None, :] * k, axis=1)
            s = tl.where(valid, s, float('-inf'))

            s_max = tl.max(s)
            m_new = tl.maximum(m, s_max)
            rescale = tl.exp(m - m_new)
            p = tl.exp(s - m_new)
            p = tl.where(valid, p, 0.0)

            l = rescale * l + tl.sum(p)
            acc = rescale * acc + tl.sum(p[:, None] * v, axis=0)
            m = m_new

    acc = acc / l

    o_ptrs = Out_ptr + batch_idx * stride_o_b + head_idx * stride_o_h
    o_ptrs = o_ptrs + tl.arange(0, HEAD_DIM)
    tl.store(o_ptrs, acc.to(tl.bfloat16))


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.max_blocks = (seq_len + page_size - 1) // page_size
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

        total_heads = batch * num_heads
        if head_dim <= 64:
            self._num_warps = 2
            self._num_stages = 2
            self._ppi = 16
        elif total_heads >= 1024:
            self._num_warps = 2
            self._num_stages = 2
            self._ppi = 1
        else:
            self._num_warps = 2
            self._num_stages = 2
            self._ppi = 4

    def forward(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        _paged_attention_kernel[(B * H, 1)](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0), block_table.stride(1),
            out.stride(0), out.stride(1),
            NUM_HEADS=H, NUM_KV_HEADS=self.num_kv_heads, HEAD_DIM=D,
            PAGE_SIZE=self.page_size, MAX_BLOCKS=self.max_blocks,
            SCALE=self.scale, PAGES_PER_ITER=self._ppi,
            num_warps=self._num_warps, num_stages=self._num_stages,
        )
        return out
