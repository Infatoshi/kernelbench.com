import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cuda_source = """
#include <torch/extension.h>
#include <cuda_fp16.h>
#include <cuda_bf16.h>

__global__ void fwd_prep_kernel(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const __nv_bfloat16* __restrict__ v,
    const float* __restrict__ g,
    const __nv_bfloat16* __restrict__ beta,
    float* __restrict__ w,
    float* __restrict__ u,
    float* __restrict__ aqk,
    float* __restrict__ g_cum_out,
    float* __restrict__ g_last,
    int B, int T, int H, int K, int V, int NT, float scale
) {
    int pid_c = blockIdx.x; 
    int pid_bh = blockIdx.y; 
    int tid = threadIdx.y * blockDim.x + threadIdx.x; 
    
    int chunk_offset_k = pid_bh * T * K + pid_c * 64 * K;
    int chunk_offset_v = pid_bh * T * V + pid_c * 64 * V;
    int chunk_offset_b = pid_bh * T + pid_c * 64;
    
    extern __shared__ float smem[];
    float* g_cum = smem; 
    float* A = g_cum + 64 * 128; 
    
    if (tid < 128) {
        float sum = 0.0f;
        for (int t = 0; t < 64; ++t) {
            sum += g[chunk_offset_k + t * 128 + tid];
            g_cum[t * 128 + tid] = sum;
            g_cum_out[(pid_bh * T + pid_c * 64 + t) * 128 + tid] = sum;
            if (t == 63) {
                g_last[pid_bh * NT * 128 + pid_c * 128 + tid] = sum;
            }
        }
    }
    __syncthreads();
    
    for (int idx = tid; idx < 4096; idx += blockDim.x * blockDim.y) {
        int i = idx / 64;
        int j = idx % 64;
        if (i > j) {
            float sum = 0.0f;
            for (int d = 0; d < 128; ++d) {
                float k_i = __bfloat162float(k[chunk_offset_k + i * 128 + d]);
                float k_j = __bfloat162float(k[chunk_offset_k + j * 128 + d]);
                sum += k_i * expf(g_cum[i * 128 + d] - g_cum[j * 128 + d]) * k_j;
            }
            float b_i = __bfloat162float(beta[chunk_offset_b + i]);
            A[i * 64 + j] = -sum * b_i;
        } else {
            A[i * 64 + j] = 0.0f;
        }
    }
    __syncthreads();
    
    for (int i = 1; i < 64; ++i) {
        for (int j = tid; j < i; j += 128) {
            float sum = 0.0f;
            for (int k_idx = j + 1; k_idx < i; ++k_idx) {
                sum += A[i * 64 + k_idx] * A[k_idx * 64 + j];
            }
            A[i * 64 + j] += sum;
        }
        __syncthreads();
    }
    
    for (int idx = tid; idx < 4096; idx += 128) {
        int i = idx / 64;
        int j = idx % 64;
        float b_j = __bfloat162float(beta[chunk_offset_b + j]);
        if (i == j) {
            A[i * 64 + j] = (A[i * 64 + j] + 1.0f) * b_j;
        } else {
            A[i * 64 + j] = A[i * 64 + j] * b_j;
        }
    }
    __syncthreads();
    
    if (tid < 128) {
        for (int i = 0; i < 64; ++i) {
            float sum_w = 0.0f;
            float sum_u = 0.0f;
            for (int j = 0; j <= i; ++j) {
                float a_ij = A[i * 64 + j];
                float k_j = __bfloat162float(k[chunk_offset_k + j * 128 + tid]);
                float g_j = g_cum[j * 128 + tid];
                float v_j = __bfloat162float(v[chunk_offset_v + j * 128 + tid]);
                
                sum_w += a_ij * expf(g_j) * k_j;
                sum_u += a_ij * v_j;
            }
            w[(pid_bh * NT + pid_c) * 64 * 128 + i * 128 + tid] = sum_w;
            u[(pid_bh * NT + pid_c) * 64 * 128 + i * 128 + tid] = sum_u;
        }
    }
    
    for (int idx = tid; idx < 4096; idx += 128) {
        int i = idx / 64;
        int j = idx % 64;
        if (i >= j) {
            float sum = 0.0f;
            for (int d = 0; d < 128; ++d) {
                float q_id = __bfloat162float(q[chunk_offset_k + i * 128 + d]) * scale;
                float k_jd = __bfloat162float(k[chunk_offset_k + j * 128 + d]);
                sum += q_id * expf(g_cum[i * 128 + d] - g_cum[j * 128 + d]) * k_jd;
            }
            aqk[(pid_bh * NT + pid_c) * 4096 + i * 64 + j] = sum;
        } else {
            aqk[(pid_bh * NT + pid_c) * 4096 + i * 64 + j] = 0.0f;
        }
    }
}

__global__ void fwd_recurrent_kernel(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const float* __restrict__ u,
    const float* __restrict__ w,
    const float* __restrict__ aqk,
    const float* __restrict__ g_cum_out,
    const float* __restrict__ g_last,
    __nv_bfloat16* __restrict__ o,
    int B, int T, int H, int K, int V, int NT, float scale
) {
    int pid_bh = blockIdx.x; 
    int v_idx = threadIdx.x; 
    
    float S[128];
    for(int i=0; i<128; i++) S[i] = 0.0f;
    
    for (int c = 0; c < NT; ++c) {
        int chunk_offset_k = pid_bh * T * K + c * 64 * K;
        int chunk_offset_v = pid_bh * T * V + c * 64 * V;
        
        float v_i[64];
        for (int i = 0; i < 64; ++i) {
            float sum = 0.0f;
            for (int d = 0; d < 128; ++d) {
                sum += w[(pid_bh * NT + c) * 64 * 128 + i * 128 + d] * S[d];
            }
            v_i[i] = u[(pid_bh * NT + c) * 64 * 128 + i * 128 + v_idx] - sum;
        }
        
        float o_i[64];
        for (int i = 0; i < 64; ++i) {
            float sum_q = 0.0f;
            for (int d = 0; d < 128; ++d) {
                float q_id = __bfloat162float(q[chunk_offset_k + i * 128 + d]) * scale;
                float g_cum_id = g_cum_out[(pid_bh * T + c * 64) * 128 + i * 128 + d];
                sum_q += q_id * expf(g_cum_id) * S[d];
            }
            
            float sum_aqk = 0.0f;
            for (int j = 0; j <= i; ++j) {
                sum_aqk += aqk[(pid_bh * NT + c) * 4096 + i * 64 + j] * v_i[j];
            }
            
            o_i[i] = sum_q + sum_aqk;
            o[chunk_offset_v + i * 128 + v_idx] = __float2bfloat16(o_i[i]);
        }
        
        for (int d = 0; d < 128; ++d) {
            float g_last_d = g_last[pid_bh * NT * 128 + c * 128 + d];
            float sum_k = 0.0f;
            for (int i = 0; i < 64; ++i) {
                float k_id = __bfloat162float(k[chunk_offset_k + i * 128 + d]);
                float g_cum_id = g_cum_out[(pid_bh * T + c * 64) * 128 + i * 128 + d];
                sum_k += k_id * expf(g_last_d - g_cum_id) * v_i[i];
            }
            S[d] = S[d] * expf(g_last_d) + sum_k;
        }
    }
}

void launch_kda_fwd_cuda(
    torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor g, torch::Tensor beta,
    torch::Tensor w, torch::Tensor u, torch::Tensor aqk, torch::Tensor g_cum_out, torch::Tensor g_last, torch::Tensor o,
    int B, int T, int H, int K, int V, int NT, float scale
) {
    dim3 grid1(NT, B * H);
    dim3 block1(128);
    int smem_size = (64 * 128 + 64 * 64) * sizeof(float);
    
    fwd_prep_kernel<<<grid1, block1, smem_size>>>(
        reinterpret_cast<const __nv_bfloat16*>(q.data_ptr<at::BFloat16>()),
        reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>()),
        reinterpret_cast<const __nv_bfloat16*>(v.data_ptr<at::BFloat16>()),
        g.data_ptr<float>(),
        reinterpret_cast<const __nv_bfloat16*>(beta.data_ptr<at::BFloat16>()),
        w.data_ptr<float>(),
        u.data_ptr<float>(),
        aqk.data_ptr<float>(),
        g_cum_out.data_ptr<float>(),
        g_last.data_ptr<float>(),
        B, T, H, K, V, NT, scale
    );
    
    dim3 grid2(B * H);
    dim3 block2(V);
    
    fwd_recurrent_kernel<<<grid2, block2>>>(
        reinterpret_cast<const __nv_bfloat16*>(q.data_ptr<at::BFloat16>()),
        reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>()),
        u.data_ptr<float>(),
        w.data_ptr<float>(),
        aqk.data_ptr<float>(),
        g_cum_out.data_ptr<float>(),
        g_last.data_ptr<float>(),
        reinterpret_cast<__nv_bfloat16*>(o.data_ptr<at::BFloat16>()),
        B, T, H, K, V, NT, scale
    );
}
"""

cpp_source = """
void launch_kda_fwd_cuda(
    torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor g, torch::Tensor beta,
    torch::Tensor w, torch::Tensor u, torch::Tensor aqk, torch::Tensor g_cum_out, torch::Tensor g_last, torch::Tensor o,
    int B, int T, int H, int K, int V, int NT, float scale
);

void launch_kda_fwd(
    torch::Tensor q, torch::Tensor k, torch::Tensor v, torch::Tensor g, torch::Tensor beta,
    torch::Tensor w, torch::Tensor u, torch::Tensor aqk, torch::Tensor g_cum_out, torch::Tensor g_last, torch::Tensor o,
    int B, int T, int H, int K, int V, int NT, float scale
) {
    launch_kda_fwd_cuda(q, k, v, g, beta, w, u, aqk, g_cum_out, g_last, o, B, T, H, K, V, NT, scale);
}
"""

kda_module = load_inline(
    name='custom_kda_fwd',
    cpp_sources=cpp_source,
    cuda_sources=cuda_source,
    functions=['launch_kda_fwd'],
    verbose=False,
    extra_cuda_cflags=['-O3', '-std=c++17', '-gencode=arch=compute_86,code=sm_86']
)

class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        dtype = v.dtype
        device = q.device
        B, T, H, K = q.shape
        V = v.shape[-1]
        BT = self.chunk_size
        NT = T // BT
        
        q = q.contiguous()
        k = k.contiguous()
        v = v.contiguous()
        g = g.contiguous()
        beta = beta.contiguous()
        
        aqk = torch.empty(B, H, NT, BT, BT, device=device, dtype=torch.float32)
        g_cum_out = torch.empty(B, H, T, K, device=device, dtype=torch.float32)
        g_last = torch.empty(B, H, NT, K, device=device, dtype=torch.float32)
        
        u = torch.empty(B, H, NT, BT, V, device=device, dtype=torch.float32)
        w = torch.empty(B, H, NT, BT, K, device=device, dtype=torch.float32)
        o = torch.empty_like(v)
        
        kda_module.launch_kda_fwd(
            q, k, v, g, beta, w, u, aqk, g_cum_out, g_last, o, B, T, H, K, V, NT, self.scale
        )
        
        return o.to(dtype)
