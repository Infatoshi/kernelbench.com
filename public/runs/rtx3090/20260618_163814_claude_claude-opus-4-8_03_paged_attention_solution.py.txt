"""Paged-attention decode kernel for RTX 3090 (SM86 Ampere, GDDR6X, 936 GB/s).

Single-query decode is bandwidth-bound: the KV cache is streamed once per token.
The design is FlashDecoding:

  * Grid (num_splits, batch, num_kv_heads). Each program loads the KV slice for
    ONE kv head and computes ALL grouped-query heads that share it, so the KV
    cache is read from DRAM exactly once (the bytes_formula counts it once).
  * Online fp32 softmax (numerically stable max-subtraction) over the sequence
    slice assigned to this split.
  * When num_splits == 1 the kernel normalizes in-register and writes the final
    bf16 output directly (no second launch). When num_splits > 1 it writes
    per-split unnormalized accumulators + per-split max m and sum l, and a small
    reduction kernel merges them.

Splitting fills the 82 SMs of the 3090 when batch*num_kv_heads is small
(e.g. the 4x64x8 long-context shape only has 32 (batch,kv_head) pairs).
"""
from __future__ import annotations

import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


# --------------------------------------------------------------------------- #
# Split kernel: one program == (split, batch, kv_head). Streams its KV slice,
# runs online softmax for all G query heads of this kv head.
# --------------------------------------------------------------------------- #
@triton.jit
def _split_kernel(
    Q, KV, BT, SEQLEN,
    Acc, Mp, Lp, Out,
    scale,
    stride_qb, stride_qh,
    stride_kv_blk, stride_kv_slot, stride_kv_head,
    stride_bt_b,
    stride_acc_b, stride_acc_h, stride_acc_s,
    stride_mp_b, stride_mp_h,
    stride_ob, stride_oh,
    tokens_per_split,
    G: tl.constexpr, D: tl.constexpr, P: tl.constexpr,
    BLOCK_G: tl.constexpr, BLOCK_N: tl.constexpr,
    SINGLE_SPLIT: tl.constexpr,
):
    split_id = tl.program_id(0)
    b = tl.program_id(1)
    kh = tl.program_id(2)

    L = tl.load(SEQLEN + b)
    tok_start = split_id * tokens_per_split
    tok_end = tl.minimum(tok_start + tokens_per_split, L)

    offs_g = tl.arange(0, BLOCK_G)
    offs_d = tl.arange(0, D)
    g_mask = offs_g < G
    q_head = kh * G + offs_g

    # Q for the G heads of this kv head: (BLOCK_G, D), bf16.
    q_ptrs = Q + b * stride_qb + q_head[:, None] * stride_qh + offs_d[None, :]
    q = tl.load(q_ptrs, mask=g_mask[:, None], other=0.0)

    m_i = tl.full((BLOCK_G,), float("-inf"), tl.float32)
    l_i = tl.zeros((BLOCK_G,), tl.float32)
    acc = tl.zeros((BLOCK_G, D), tl.float32)

    offs_n = tl.arange(0, BLOCK_N)
    for start in range(tok_start, tok_end, BLOCK_N):
        tok = start + offs_n
        n_mask = tok < tok_end
        page_idx = tok // P
        slot = tok % P
        page_id = tl.load(BT + b * stride_bt_b + page_idx, mask=n_mask, other=0)
        kv_base = page_id * stride_kv_blk + slot * stride_kv_slot + kh * stride_kv_head

        k_ptrs = KV + kv_base[:, None] + offs_d[None, :]
        v_ptrs = KV + kv_base[:, None] + (D + offs_d)[None, :]
        k = tl.load(k_ptrs, mask=n_mask[:, None], other=0.0)        # (BLOCK_N, D) bf16
        v = tl.load(v_ptrs, mask=n_mask[:, None], other=0.0)

        qk = tl.dot(q, tl.trans(k)) * scale                        # (BLOCK_G, BLOCK_N) fp32
        qk = tl.where(n_mask[None, :], qk, float("-inf"))

        m_new = tl.maximum(m_i, tl.max(qk, axis=1))
        p = tl.exp(qk - m_new[:, None])
        alpha = tl.exp(m_i - m_new)
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v)
        m_i = m_new

    if SINGLE_SPLIT:
        out = acc / l_i[:, None]
        o_ptrs = Out + b * stride_ob + q_head[:, None] * stride_oh + offs_d[None, :]
        tl.store(o_ptrs, out.to(tl.bfloat16), mask=g_mask[:, None])
    else:
        acc_ptrs = (Acc + b * stride_acc_b + q_head[:, None] * stride_acc_h
                    + split_id * stride_acc_s + offs_d[None, :])
        tl.store(acc_ptrs, acc, mask=g_mask[:, None])
        m_ptrs = Mp + b * stride_mp_b + q_head * stride_mp_h + split_id
        l_ptrs = Lp + b * stride_mp_b + q_head * stride_mp_h + split_id
        tl.store(m_ptrs, m_i, mask=g_mask)
        tl.store(l_ptrs, l_i, mask=g_mask)


# --------------------------------------------------------------------------- #
# Reduction kernel: merge per-split partials for one (batch, head) into the
# final normalized bf16 output.
# --------------------------------------------------------------------------- #
@triton.jit
def _reduce_kernel(
    Acc, Mp, Lp, Out,
    stride_acc_b, stride_acc_h, stride_acc_s,
    stride_mp_b, stride_mp_h,
    stride_ob, stride_oh,
    num_splits,
    D: tl.constexpr, BLOCK_S: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)

    offs_s = tl.arange(0, BLOCK_S)
    offs_d = tl.arange(0, D)
    s_mask = offs_s < num_splits

    m_s = tl.load(Mp + b * stride_mp_b + h * stride_mp_h + offs_s, mask=s_mask, other=float("-inf"))
    l_s = tl.load(Lp + b * stride_mp_b + h * stride_mp_h + offs_s, mask=s_mask, other=0.0)

    global_m = tl.max(m_s, axis=0)
    scale_s = tl.exp(m_s - global_m)                                # (BLOCK_S,)
    l_total = tl.sum(l_s * scale_s, axis=0)

    acc_ptrs = (Acc + b * stride_acc_b + h * stride_acc_h
                + offs_s[:, None] * stride_acc_s + offs_d[None, :])
    acc = tl.load(acc_ptrs, mask=s_mask[:, None], other=0.0)        # (BLOCK_S, D) fp32
    out = tl.sum(acc * scale_s[:, None], axis=0) / l_total          # (D,)

    o_ptrs = Out + b * stride_ob + h * stride_oh + offs_d
    tl.store(o_ptrs, out.to(tl.bfloat16))


_SM_COUNT = 82  # RTX 3090 (GA102)


def _pick_splits(base: int, seq_pages: int, target: int = 256, min_pages: int = 4) -> int:
    """Choose number of sequence splits so total programs ~= `target`.

    When `base = batch * num_kv_heads` already fills the SMs we use a single
    pass (num_splits == 1): the kernel then normalizes in-register and writes
    the final output directly, avoiding the extra reduction-kernel launch whose
    fixed overhead measurably hurts the short, SM-filling shapes.
    """
    if base >= int(_SM_COUNT * 0.75):
        return 1
    splits = max(1, (target + base - 1) // base)
    max_splits = max(1, seq_pages // min_pages)
    return max(1, min(splits, max_splits, seq_pages))


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        seq_pages = (self.seq_len + P - 1) // P
        base = B * Hkv
        # D=128 moves 2x the bytes/token of D=64, so each program does more work
        # and fewer splits already fill the SMs; narrow heads need more splits.
        split_target = 128 if D >= 128 else 256
        num_splits = _pick_splits(base, seq_pages, target=split_target)
        pages_per_split = (seq_pages + num_splits - 1) // num_splits
        tokens_per_split = pages_per_split * P
        num_splits = (seq_pages + pages_per_split - 1) // pages_per_split

        BLOCK_G = max(16, triton.next_power_of_2(G))
        if D >= 128:
            BLOCK_N, num_warps, num_stages = 64, 4, (3 if num_splits == 1 else 2)
        else:
            BLOCK_N, num_warps, num_stages = 32, 4, 2

        stride_kv_blk = kv_cache.stride(0)
        stride_kv_slot = kv_cache.stride(1)
        stride_kv_head = kv_cache.stride(2)

        single = num_splits == 1
        if single:
            acc = mp = lp = out  # dummies; unused under SINGLE_SPLIT
            stride_acc_b = stride_acc_h = stride_acc_s = 0
            stride_mp_b = stride_mp_h = 0
        else:
            acc = torch.empty(B, H, num_splits, D, dtype=torch.float32, device=query.device)
            mp = torch.empty(B, H, num_splits, dtype=torch.float32, device=query.device)
            lp = torch.empty(B, H, num_splits, dtype=torch.float32, device=query.device)
            stride_acc_b, stride_acc_h, stride_acc_s = acc.stride(0), acc.stride(1), acc.stride(2)
            stride_mp_b, stride_mp_h = mp.stride(0), mp.stride(1)

        grid = (num_splits, B, Hkv)
        _split_kernel[grid](
            query, kv_cache, block_table, seq_lens,
            acc, mp, lp, out,
            self.scale,
            query.stride(0), query.stride(1),
            stride_kv_blk, stride_kv_slot, stride_kv_head,
            block_table.stride(0),
            stride_acc_b, stride_acc_h, stride_acc_s,
            stride_mp_b, stride_mp_h,
            out.stride(0), out.stride(1),
            tokens_per_split,
            G=G, D=D, P=P,
            BLOCK_G=BLOCK_G, BLOCK_N=BLOCK_N,
            SINGLE_SPLIT=single,
            num_warps=num_warps, num_stages=num_stages,
        )

        if not single:
            BLOCK_S = triton.next_power_of_2(num_splits)
            _reduce_kernel[(B, H)](
                acc, mp, lp, out,
                stride_acc_b, stride_acc_h, stride_acc_s,
                stride_mp_b, stride_mp_h,
                out.stride(0), out.stride(1),
                num_splits,
                D=D, BLOCK_S=BLOCK_S,
                num_warps=4,
            )

        return out
