import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _paged_attention_kernel(
    q_ptr, kv_ptr, block_table_ptr, seq_lens_ptr, out_ptr,
    stride_q_b, stride_q_h, stride_q_d,
    stride_kv_block, stride_kv_page, stride_kv_h, stride_kv_d,
    stride_bt_b, stride_bt_block,
    stride_out_b, stride_out_h, stride_out_d,
    scale,
    GROUP_SIZE: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    BLOCK_SEQ: tl.constexpr,
):
    b = tl.program_id(0)
    h_kv = tl.program_id(1)

    seq_len = tl.load(seq_lens_ptr + b)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, HEAD_DIM)
    
    q_ptrs = q_ptr + b * stride_q_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_q_h + offs_d[None, :] * stride_q_d
    q = tl.load(q_ptrs) # (GROUP_SIZE, HEAD_DIM)
    
    m_i = tl.zeros([GROUP_SIZE], dtype=tl.float32) - float('inf')
    l_i = tl.zeros([GROUP_SIZE], dtype=tl.float32)
    acc = tl.zeros([GROUP_SIZE, HEAD_DIM], dtype=tl.float32)
    
    num_pages = (seq_len + PAGE_SIZE - 1) // PAGE_SIZE
    PAGES_PER_BLOCK = BLOCK_SEQ // PAGE_SIZE
    
    offs_seq = tl.arange(0, BLOCK_SEQ)
    logical_page_idx = offs_seq // PAGE_SIZE
    page_offset = offs_seq % PAGE_SIZE
    
    for start_page_idx in range(0, num_pages, PAGES_PER_BLOCK):
        start_seq = start_page_idx * PAGE_SIZE
        seq_idx = start_seq + offs_seq
        
        page_idx = start_page_idx + logical_page_idx
        bt_mask = page_idx < num_pages
        block_ids_seq = tl.load(block_table_ptr + b * stride_bt_b + page_idx * stride_bt_block, mask=bt_mask, other=0)
        
        k_ptrs = kv_ptr + block_ids_seq[:, None] * stride_kv_block + page_offset[:, None] * stride_kv_page + h_kv * stride_kv_h + offs_d[None, :] * stride_kv_d
        v_ptrs = kv_ptr + block_ids_seq[:, None] * stride_kv_block + page_offset[:, None] * stride_kv_page + h_kv * stride_kv_h + (offs_d[None, :] + HEAD_DIM) * stride_kv_d
        
        seq_mask = seq_idx < seq_len
        load_mask = bt_mask & seq_mask
        
        k = tl.load(k_ptrs, mask=load_mask[:, None], other=0.0)
        v = tl.load(v_ptrs, mask=load_mask[:, None], other=0.0)
        
        qk = tl.dot(q, tl.trans(k)) * scale
        qk = tl.where(load_mask[None, :], qk, float('-inf'))
        
        m_i_new = tl.maximum(m_i, tl.max(qk, axis=1))
        alpha = tl.exp(m_i - m_i_new)
        p = tl.exp(qk - m_i_new[:, None])
        
        l_i_new = l_i * alpha + tl.sum(p, axis=1)
        
        p = p.to(v.dtype)
        acc = acc * alpha[:, None] + tl.dot(p, v)
        
        m_i = m_i_new
        l_i = l_i_new
        
    acc = acc / l_i[:, None]
    
    out_ptrs = out_ptr + b * stride_out_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_out_h + offs_d[None, :] * stride_out_d
    tl.store(out_ptrs, acc.to(out_ptr.dtype.element_ty))


@triton.jit
def _paged_attention_split_kernel(
    q_ptr, kv_ptr, block_table_ptr, seq_lens_ptr, 
    acc_ptr, m_ptr, l_ptr,
    stride_q_b, stride_q_h, stride_q_d,
    stride_kv_block, stride_kv_page, stride_kv_h, stride_kv_d,
    stride_bt_b, stride_bt_block,
    stride_acc_b, stride_acc_h, stride_acc_s, stride_acc_d,
    stride_m_b, stride_m_h, stride_m_s,
    scale,
    GROUP_SIZE: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    BLOCK_SEQ: tl.constexpr,
):
    split_idx = tl.program_id(0)
    b = tl.program_id(1)
    h_kv = tl.program_id(2)
    num_splits = tl.num_programs(0)

    seq_len = tl.load(seq_lens_ptr + b)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, HEAD_DIM)
    
    q_ptrs = q_ptr + b * stride_q_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_q_h + offs_d[None, :] * stride_q_d
    q = tl.load(q_ptrs) # (GROUP_SIZE, HEAD_DIM)
    
    m_i = tl.zeros([GROUP_SIZE], dtype=tl.float32) - float('inf')
    l_i = tl.zeros([GROUP_SIZE], dtype=tl.float32)
    acc = tl.zeros([GROUP_SIZE, HEAD_DIM], dtype=tl.float32)
    
    num_pages = (seq_len + PAGE_SIZE - 1) // PAGE_SIZE
    
    pages_per_split = (num_pages + num_splits - 1) // num_splits
    start_page_idx = split_idx * pages_per_split
    end_page_idx = tl.minimum((split_idx + 1) * pages_per_split, num_pages)
    
    PAGES_PER_BLOCK = BLOCK_SEQ // PAGE_SIZE
    
    offs_seq = tl.arange(0, BLOCK_SEQ)
    logical_page_idx = offs_seq // PAGE_SIZE
    page_offset = offs_seq % PAGE_SIZE
    
    for start_p in range(start_page_idx, end_page_idx, PAGES_PER_BLOCK):
        start_seq = start_p * PAGE_SIZE
        seq_idx = start_seq + offs_seq
        
        page_idx = start_p + logical_page_idx
        bt_mask = page_idx < end_page_idx
        block_ids_seq = tl.load(block_table_ptr + b * stride_bt_b + page_idx * stride_bt_block, mask=bt_mask, other=0)
        
        k_ptrs = kv_ptr + block_ids_seq[:, None] * stride_kv_block + page_offset[:, None] * stride_kv_page + h_kv * stride_kv_h + offs_d[None, :] * stride_kv_d
        v_ptrs = kv_ptr + block_ids_seq[:, None] * stride_kv_block + page_offset[:, None] * stride_kv_page + h_kv * stride_kv_h + (offs_d[None, :] + HEAD_DIM) * stride_kv_d
        
        seq_mask = seq_idx < seq_len
        load_mask = bt_mask & seq_mask
        
        k = tl.load(k_ptrs, mask=load_mask[:, None], other=0.0)
        v = tl.load(v_ptrs, mask=load_mask[:, None], other=0.0)
        
        qk = tl.dot(q, tl.trans(k)) * scale
        qk = tl.where(load_mask[None, :], qk, float('-inf'))
        
        m_i_new = tl.maximum(m_i, tl.max(qk, axis=1))
        alpha = tl.exp(m_i - m_i_new)
        p = tl.exp(qk - m_i_new[:, None])
        
        l_i_new = l_i * alpha + tl.sum(p, axis=1)
        
        p = p.to(v.dtype)
        acc = acc * alpha[:, None] + tl.dot(p, v)
        
        m_i = m_i_new
        l_i = l_i_new
        
    acc_ptrs = acc_ptr + b * stride_acc_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_acc_h + split_idx * stride_acc_s + offs_d[None, :] * stride_acc_d
    tl.store(acc_ptrs, acc)
    
    m_ptrs = m_ptr + b * stride_m_b + (h_kv * GROUP_SIZE + offs_g) * stride_m_h + split_idx * stride_m_s
    tl.store(m_ptrs, m_i)
    
    l_ptrs = l_ptr + b * stride_m_b + (h_kv * GROUP_SIZE + offs_g) * stride_m_h + split_idx * stride_m_s
    tl.store(l_ptrs, l_i)


@triton.jit
def _paged_attention_reduce_kernel(
    acc_ptr, m_ptr, l_ptr, out_ptr,
    stride_acc_b, stride_acc_h, stride_acc_s, stride_acc_d,
    stride_m_b, stride_m_h, stride_m_s,
    stride_out_b, stride_out_h, stride_out_d,
    num_splits: tl.constexpr,
    HEAD_DIM: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)
    
    offs_d = tl.arange(0, HEAD_DIM)
    offs_s = tl.arange(0, num_splits)
    
    m_ptrs = m_ptr + b * stride_m_b + h * stride_m_h + offs_s * stride_m_s
    m = tl.load(m_ptrs)
    
    l_ptrs = l_ptr + b * stride_m_b + h * stride_m_h + offs_s * stride_m_s
    l = tl.load(l_ptrs)
    
    m_max = tl.max(m, axis=0)
    alpha = tl.exp(m - m_max)
    
    acc_ptrs = acc_ptr + b * stride_acc_b + h * stride_acc_h + offs_s[:, None] * stride_acc_s + offs_d[None, :] * stride_acc_d
    acc = tl.load(acc_ptrs)
    
    l_scaled = l * alpha
    l_sum = tl.sum(l_scaled, axis=0)
    
    acc_scaled = acc * alpha[:, None]
    acc_sum = tl.sum(acc_scaled, axis=0)
    
    out = acc_sum / l_sum
    
    out_ptrs = out_ptr + b * stride_out_b + h * stride_out_h + offs_d * stride_out_d
    tl.store(out_ptrs, out.to(out_ptr.dtype.element_ty))


class Model(nn.Module):
    """Single-query paged attention decode.

    Forward inputs (all on device):
      query:       (batch, num_heads, head_dim)               bf16
      kv_cache:    (num_blocks, page_size, num_kv_heads, head_dim * 2)
      block_table: (batch, max_blocks)                        int32
      seq_lens:    (batch,)                                   int32

    Output:
      attn_out:    (batch, num_heads, head_dim)               bf16
    """

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        out = torch.empty_like(query)
        
        BLOCK_SEQ = 64
        
        # Use split-K for small batches to increase SM occupancy
        if B * self.num_kv_heads >= 64:
            num_splits = 1
        else:
            num_splits = 4
            
        num_warps = 2 if D <= 64 else 4
        
        if num_splits == 1:
            grid = (B, self.num_kv_heads)
            _paged_attention_kernel[grid](
                query, kv_cache, block_table, seq_lens, out,
                query.stride(0), query.stride(1), query.stride(2),
                kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
                block_table.stride(0), block_table.stride(1),
                out.stride(0), out.stride(1), out.stride(2),
                self.scale,
                GROUP_SIZE=self.group_size,
                HEAD_DIM=D,
                PAGE_SIZE=self.page_size,
                BLOCK_SEQ=BLOCK_SEQ,
                num_warps=num_warps
            )
        else:
            acc_buf = torch.empty(B, H, num_splits, D, dtype=torch.float32, device=query.device)
            m_buf = torch.empty(B, H, num_splits, dtype=torch.float32, device=query.device)
            l_buf = torch.empty(B, H, num_splits, dtype=torch.float32, device=query.device)
            
            grid_split = (num_splits, B, self.num_kv_heads)
            _paged_attention_split_kernel[grid_split](
                query, kv_cache, block_table, seq_lens,
                acc_buf, m_buf, l_buf,
                query.stride(0), query.stride(1), query.stride(2),
                kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
                block_table.stride(0), block_table.stride(1),
                acc_buf.stride(0), acc_buf.stride(1), acc_buf.stride(2), acc_buf.stride(3),
                m_buf.stride(0), m_buf.stride(1), m_buf.stride(2),
                self.scale,
                GROUP_SIZE=self.group_size,
                HEAD_DIM=D,
                PAGE_SIZE=self.page_size,
                BLOCK_SEQ=BLOCK_SEQ,
                num_warps=num_warps
            )
            
            grid_reduce = (B, H)
            _paged_attention_reduce_kernel[grid_reduce](
                acc_buf, m_buf, l_buf, out,
                acc_buf.stride(0), acc_buf.stride(1), acc_buf.stride(2), acc_buf.stride(3),
                m_buf.stride(0), m_buf.stride(1), m_buf.stride(2),
                out.stride(0), out.stride(1), out.stride(2),
                num_splits=num_splits,
                HEAD_DIM=D,
                num_warps=num_warps
            )
            
        return out
