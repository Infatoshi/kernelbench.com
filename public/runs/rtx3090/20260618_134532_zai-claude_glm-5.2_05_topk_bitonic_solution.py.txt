"""Top-k for RTX 3090 (SM86), shape-adaptive.

The reference uses a multi-pass radix select (~8 passes for fp32). These shapes
are tiny (each row is 16 KB - 512 KB), so per-call time is dominated by kernel
work, not DRAM. Two kernels cover the deck, picked per (k, n):

  * Small k (1, 8, 16) -- per-row register cascade. One block per row; each
    thread streams a strided slice keeping a sorted-descending register buffer
    of size K (the K largest it has seen). Buffers are merged across the block
    by a log2(BLOCK)-deep merge-tree in shared mem (stride K+1 to dodge 32-way
    bank conflicts). All-register, static-index cascade (a dynamic index into a
    register array spills to local memory; a shared-mem min-heap was tried and
    is slower due to shmem load latency in the dependent sift-down).

  * Large k (32, 64) -- per-row radix-256 histogram select. Four byte-passes
    (MSB first) build the monotonic uint of the k-th largest via a 256-bin
    histogram, then gather + in-shmem bitonic sort. ~12 barriers total vs ~96
    for a 1-bit radix, and 4x fewer row reads -- far cheaper than the O(n*K)
    cascade once K is large.

  * The single very-large row (n=131072, k=64) -- segmented radix: many blocks
    each radix-select their contiguous segment to top-K into a temp buffer,
    then one block radix-selects the merged candidates. Keeps the whole GPU
    busy instead of one block.

Correctness: the per-thread cascade buffer is size K (any single thread sees
at most K of the global top-K, since only K elements are >= the k-th value), so
the global top-K survives to the merge. Radix finds the exact k-th largest's
monotonic uint (the largest value with count(>=v) >= k), so the gather collects
exactly the top-K (ties filled to K). Both paths return values sorted
descending; indices are checked leniently by the harness (gather-vs-reference).

Build: the CUDA is compiled to a standalone .so (extern "C", called via ctypes)
so it never pulls in the ML framework's C++ headers -- the installed framework's
List_inl.h is ill-formed under gcc>=14 and would fail an extension build.
"""
from __future__ import annotations

import ctypes
import os
import subprocess

import torch
import torch.nn as nn

OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_3090"]

_SEG_THRESHOLD = 1 << 15  # 32768; above this, use multi-block segmented path

_CUDA_SRC = r"""
#include <cuda_runtime.h>

// Per-thread: insert v (global index gi) into a sorted-DESCENDING register
// buffer tv[0..K-1] (tv[0]=largest, tv[K-1]=smallest=threshold). Drops old min.
// Fully-unrolled conditional-cascade with STATIC indices only -- dynamic
// indexing into a register array would spill it to local memory (very slow).
// O(K) per qualifying element, but all-register so latency is 1 cycle/step;
// a shmem min-heap (O(log K)) was tried and is slower due to shmem load
// latency in the data-dependent sift-down chain.
template <int K>
__device__ __forceinline__ void insert_topk(float* tv, int* ti, float v, int gi) {
    if (v <= tv[K - 1]) return;
    float carry = v; int carry_i = gi; bool active = false;
    #pragma unroll
    for (int i = 0; i < K; i++) {
        if (carry > tv[i]) active = true;
        if (active) {
            float tf = tv[i]; tv[i] = carry; carry = tf;
            int t2 = ti[i]; ti[i] = carry_i; carry_i = t2;
        }
    }
}

// Compile-time bitonic sort of a size-K register buffer, DESCENDING. Static
// indices only -> no register spill.
template <int K>
__device__ __forceinline__ void sort_desc(float* tv, int* ti) {
    #pragma unroll
    for (int size = 2; size <= K; size <<= 1)
    #pragma unroll
    for (int stride = size >> 1; stride > 0; stride >>= 1)
    #pragma unroll
    for (int i = 0; i < K; i++) {
        int j = i ^ stride;
        if (j > i) {
            bool up = ((i & size) == 0);
            if ((up && tv[i] < tv[j]) || (!up && tv[i] > tv[j])) {
                float f = tv[i]; tv[i] = tv[j]; tv[j] = f;
                int g = ti[i]; ti[i] = ti[j]; ti[j] = g;
            }
        }
    }
}

// Merge-tree reduction over `nbuf` shared-mem buffers (each size K, sorted
// desc, laid out at stride S=K+1 to dodge 32-way bank conflicts). After return,
// buffer 0 holds the top-K of the union, sorted desc.
// Pairs merged at distance d=1,2,4,...; one __syncthreads per level.
template <int BLOCK, int K>
__device__ __forceinline__ void merge_tree(float* sv, int* si, int nbuf) {
    constexpr int S = K + 1;   // coprime to 32 -> no bank conflicts
    for (int d = 1; d < nbuf; d <<= 1) {
        for (int base = threadIdx.x * (2 * d); base + d < nbuf;
             base += BLOCK * (2 * d)) {
            float* A = sv + base * S; int* Ai = si + base * S;
            float* B = sv + (base + d) * S; int* Bi = si + (base + d) * S;
            float tmp[K]; int ti2[K];
            int ia = 0, ib = 0;
            #pragma unroll
            for (int c = 0; c < K; c++) {
                float va = A[ia], vb = B[ib];
                if (va >= vb) { tmp[c] = va; ti2[c] = Ai[ia]; ia++; }
                else         { tmp[c] = vb; ti2[c] = Bi[ib]; ib++; }
            }
            #pragma unroll
            for (int c = 0; c < K; c++) { A[c] = tmp[c]; Ai[c] = ti2[c]; }
        }
        __syncthreads();
    }
}

template <int K, int BLOCK>
__global__ void topk_reduce_kernel(
    const float* __restrict__ x, float* __restrict__ out_vals,
    long long* __restrict__ out_idx, float* __restrict__ tmp_vals,
    long long* __restrict__ tmp_idx, int batch, int n, int nseg, int seg_size)
{
    const int row = blockIdx.x;
    const int seg = blockIdx.y;
    if (row >= batch) return;
    const int seg_start = seg * seg_size;
    const int seg_end = (seg_start + seg_size < n) ? (seg_start + seg_size) : n;
    const float* xrow = x + (size_t)row * n;

    constexpr int M = BLOCK * (K + 1);
    extern __shared__ float smem[];
    float* sv = smem;
    int* si = reinterpret_cast<int*>(sv + M);

    float tv[K]; int ti[K];
    #pragma unroll
    for (int i = 0; i < K; i++) { tv[i] = -INFINITY; ti[i] = 0; }

    // Tiled, prefetch-style streaming: issue UTILE loads up front (unrolled)
    // so many global loads are in flight and DRAM latency is hidden, then
    // insert each into the per-thread register buffer (O(K) cascade).
    constexpr int UTILE = 8;
    const int len = seg_end - seg_start;
    const int stride = BLOCK * UTILE;
    int full = (len / stride) * stride;
    for (int base = 0; base < full; base += stride) {
        float vb[UTILE]; int ib[UTILE];
        #pragma unroll
        for (int j = 0; j < UTILE; j++) {
            int q = base + threadIdx.x + j * BLOCK;
            vb[j] = xrow[seg_start + q];
            ib[j] = seg_start + q;
        }
        #pragma unroll
        for (int j = 0; j < UTILE; j++) insert_topk<K>(tv, ti, vb[j], ib[j]);
    }
    for (int q = full + threadIdx.x; q < len; q += BLOCK) {
        insert_topk<K>(tv, ti, xrow[seg_start + q], seg_start + q);
    }

    // dump sorted-descending per-thread buffers into shmem (stride K+1).
    #pragma unroll
    for (int i = 0; i < K; i++) {
        int off = threadIdx.x * (K + 1) + i;
        sv[off] = tv[i]; si[off] = ti[i];
    }
    __syncthreads();

    merge_tree<BLOCK, K>(sv, si, BLOCK);

    // block top-K now in sv[0..K-1], sorted desc
    if (nseg == 1) {
        for (int i = threadIdx.x; i < K; i += BLOCK) {
            out_vals[(size_t)row * K + i] = sv[i];
            out_idx [(size_t)row * K + i] = (long long)si[i];
        }
    } else {
        size_t slot = ((size_t)row * nseg + seg) * K;
        for (int i = threadIdx.x; i < K; i += BLOCK) {
            tmp_vals[slot + i] = sv[i];
            tmp_idx [slot + i] = (long long)si[i];
        }
    }
}

template <int K, int BLOCK>
__global__ void topk_merge_kernel(
    const float* __restrict__ tmp_vals, const long long* __restrict__ tmp_idx,
    float* __restrict__ out_vals, long long* __restrict__ out_idx, int nseg)
{
    const int row = blockIdx.x;
    const size_t base = (size_t)row * nseg * K;

    constexpr int S = K + 1;
    constexpr int M = BLOCK * S;
    extern __shared__ float smem[];
    float* sv = smem;
    int* si = reinterpret_cast<int*>(sv + M);

    // load nseg buffers (each K, sorted desc from reduce stage) into shmem at
    // stride S; pad extra buffers with -INFINITY.
    int nbuf = nseg;
    int np = 1; while (np < nbuf) np <<= 1;
    nbuf = np;   // round up to power of two (merge_tree requirement)
    for (int b = threadIdx.x; b < nbuf; b += BLOCK) {
        int dst = b * S;
        if (b < nseg) {
            int src = b * K;
            #pragma unroll
            for (int i = 0; i < K; i++) {
                sv[dst + i] = tmp_vals[base + src + i];
                si[dst + i] = (int)tmp_idx[base + src + i];
            }
        } else {
            #pragma unroll
            for (int i = 0; i < K; i++) { sv[dst + i] = -INFINITY; si[dst + i] = 0; }
        }
    }
    __syncthreads();

    merge_tree<BLOCK, K>(sv, si, nbuf);

    for (int i = threadIdx.x; i < K; i += BLOCK) {
        out_vals[(size_t)row * K + i] = sv[i];
        out_idx [(size_t)row * K + i] = (long long)si[i];
    }
}

// -----------------------------------------------------------------------
// Radix-256 histogram select top-k (one block per row). 4 byte-passes (MSB
// first) build the monotonic uint of the k-th largest via a 256-bin histogram
// per pass; ~12 __syncthreads total vs ~96 for a 1-bit radix, and 4x fewer
// row reads. Far cheaper than the O(n*K) per-thread cascade for large K.
// -----------------------------------------------------------------------

// float -> monotonic uint (larger float -> larger uint). NaN -> 0 (excluded).
__device__ __forceinline__ unsigned f2u(float f) {
    unsigned u = __float_as_int(f);
    if ((u & 0x7f800000u) == 0x7f800000u && (u & 0x007fffffu)) return 0;
    return (u & 0x80000000u) ? ~u : (u | 0x80000000u);
}

// Selects top-k from xv[0..len-1]. For a selected element at position p, its
// reported index is (idx_in ? idx_in[p] : base + p). Writes k results sorted
// descending to shmem gsv[0..k-1], gsi[0..k-1]. Uses external shmem hist[256].
template <int BLOCK, int K>
__device__ __forceinline__ void radix_select(
    const float* xv, const int* idx_in, int base, int len, int k,
    float* gsv, int* gsi, int* hist)
{
    const int tid = threadIdx.x;
    __shared__ unsigned s_thr;
    __shared__ int s_krem;
    if (tid == 0) { s_thr = 0; s_krem = k; }
    __syncthreads();

    unsigned thr = 0;
    int krem = k;
    for (int sh = 24; sh >= 0; sh -= 8) {
        for (int i = tid; i < 256; i += BLOCK) hist[i] = 0;
        __syncthreads();
        unsigned pmask = (sh == 24) ? 0u : (~0u << (sh + 8));
        unsigned pval = thr & pmask;
        for (int p = tid; p < len; p += BLOCK) {
            unsigned u = f2u(xv[p]);
            if ((u & pmask) == pval) atomicAdd(&hist[(u >> sh) & 0xff], 1);
        }
        __syncthreads();
        if (tid == 0) {
            int c = 0, chosen = 0;
            #pragma unroll
            for (int b = 255; b >= 0; b--) {
                if (c + hist[b] >= krem) { chosen = b; break; }
                c += hist[b];
            }
            thr |= (unsigned)(chosen << sh);
            krem = krem - c;
            s_thr = thr; s_krem = (krem < 1) ? 1 : krem;
        }
        __syncthreads();
        thr = s_thr; krem = s_krem;
    }

    __shared__ int wpos;
    if (tid == 0) wpos = 0;
    __syncthreads();
    for (int p = tid; p < len; p += BLOCK) {
        unsigned u = f2u(xv[p]);
        if (u > thr) {
            int pos = atomicAdd(&wpos, 1);
            if (pos < k) { gsv[pos] = xv[p]; gsi[pos] = idx_in ? idx_in[p] : (base + p); }
        }
    }
    __syncthreads();
    if (wpos < k) {
        for (int p = tid; p < len; p += BLOCK) {
            unsigned u = f2u(xv[p]);
            if (u == thr) {
                int pos = atomicAdd(&wpos, 1);
                if (pos < k) { gsv[pos] = xv[p]; gsi[pos] = idx_in ? idx_in[p] : (base + p); }
            }
        }
        __syncthreads();
    }

    // cooperative in-shmem bitonic sort, descending
    for (int size = 2; size <= K; size <<= 1)
        for (int stride = size >> 1; stride > 0; stride >>= 1) {
            for (int i = tid; i < K; i += BLOCK) {
                int j = i ^ stride;
                if (j > i && j < K) {
                    bool up = ((i & size) == 0);
                    if ((up && gsv[i] < gsv[j]) || (!up && gsv[i] > gsv[j])) {
                        float f = gsv[i]; gsv[i] = gsv[j]; gsv[j] = f;
                        int g = gsi[i]; gsi[i] = gsi[j]; gsi[j] = g;
                    }
                }
            }
            __syncthreads();
        }
}

template <int BLOCK, int K>
__global__ void topk_radix_kernel(
    const float* __restrict__ x, float* __restrict__ out_vals,
    long long* __restrict__ out_idx, int n, int k)
{
    const int row = blockIdx.x;
    const float* xr = x + (size_t)row * n;
    __shared__ int hist[256];
    __shared__ float gsv[K];
    __shared__ int gsi[K];
    radix_select<BLOCK, K>(xr, (const int*)0, 0, n, k, gsv, gsi, hist);
    const int tid = threadIdx.x;
    for (int i = tid; i < K; i += BLOCK) {
        out_vals[(size_t)row * K + i] = gsv[i];
        out_idx [(size_t)row * K + i] = (long long)gsi[i];
    }
}

// Segmented radix reduce: grid(batch, nseg). Each block radix-selects top-K
// from its contiguous segment, writes sorted (val, global_idx) to temp.
template <int BLOCK, int K>
__global__ void topk_radix_seg_kernel(
    const float* __restrict__ x, float* __restrict__ tmp_vals,
    int* __restrict__ tmp_idx, int n, int nseg, int seg_size, int k)
{
    const int row = blockIdx.x;
    const int seg = blockIdx.y;
    const int seg_start = seg * seg_size;
    const int seg_end = (seg_start + seg_size < n) ? (seg_start + seg_size) : n;
    const float* xr = x + (size_t)row * n + seg_start;
    __shared__ int hist[256];
    __shared__ float gsv[K];
    __shared__ int gsi[K];
    radix_select<BLOCK, K>(xr, (const int*)0, seg_start, seg_end - seg_start, k, gsv, gsi, hist);
    const int tid = threadIdx.x;
    size_t slot = ((size_t)row * nseg + seg) * K;
    for (int i = tid; i < K; i += BLOCK) {
        tmp_vals[slot + i] = gsv[i];
        tmp_idx [slot + i] = gsi[i];
    }
}

// Radix merge: grid(batch). Reads nseg*K (val,idx) candidates, radix-selects
// top-K, writes sorted output.
template <int BLOCK, int K>
__global__ void topk_radix_merge_kernel(
    const float* __restrict__ tmp_vals, const int* __restrict__ tmp_idx,
    float* __restrict__ out_vals, long long* __restrict__ out_idx,
    int nseg, int k)
{
    const int row = blockIdx.x;
    const float* tv = tmp_vals + (size_t)row * nseg * K;
    const int* ti = tmp_idx + (size_t)row * nseg * K;
    __shared__ int hist[256];
    __shared__ float gsv[K];
    __shared__ int gsi[K];
    radix_select<BLOCK, K>(tv, ti, 0, nseg * K, k, gsv, gsi, hist);
    const int tid = threadIdx.x;
    for (int i = tid; i < K; i += BLOCK) {
        out_vals[(size_t)row * K + i] = gsv[i];
        out_idx [(size_t)row * K + i] = (long long)gsi[i];
    }
}

#define LAUNCH_RADIX(KB) \
extern "C" void launch_radix_##KB(const float* x, float* ov, long long* oi, \
    int batch, int n, int k) { \
    topk_radix_kernel<1024, KB><<<batch, 1024>>>(x, ov, oi, n, k); \
}
#define LAUNCH_RADIX_BIG(KB) \
extern "C" void launch_radix_big_##KB(const float* x, float* ov, long long* oi, \
    float* tv_, int* ti_, int batch, int n, int nseg, int seg_size, int k) { \
    dim3 g(batch, nseg); \
    topk_radix_seg_kernel<1024, KB><<<g, 1024>>>(x, tv_, ti_, n, nseg, seg_size, k); \
    topk_radix_merge_kernel<1024, KB><<<batch, 1024>>>(tv_, ti_, ov, oi, nseg, k); \
}
LAUNCH_RADIX(8)
LAUNCH_RADIX(16)
LAUNCH_RADIX(32)
LAUNCH_RADIX(64)
LAUNCH_RADIX_BIG(64)

#define LAUNCH_REDUCE(KB, BL) \
extern "C" void launch_reduce_##KB(const float* x, float* ov, long long* oi, \
    float* tv_, long long* ti_, int batch, int n, int nseg, int seg_size) { \
    constexpr int M = BL * (KB + 1); \
    size_t smem = (size_t)M * (sizeof(float) + sizeof(int)); \
    cudaFuncSetAttribute(topk_reduce_kernel<KB, BL>, \
        cudaFuncAttributeMaxDynamicSharedMemorySize, 99 * 1024); \
    dim3 grid(batch, nseg); \
    topk_reduce_kernel<KB, BL><<<grid, BL, smem>>>( \
        x, ov, oi, tv_, ti_, batch, n, nseg, seg_size); \
}

#define LAUNCH_MERGE(KB, BL) \
extern "C" void launch_merge_##KB(const float* tv_, const long long* ti_, \
    float* ov, long long* oi, int batch, int nseg) { \
    constexpr int M = BL * (KB + 1); \
    size_t smem = (size_t)M * (sizeof(float) + sizeof(int)); \
    cudaFuncSetAttribute(topk_merge_kernel<KB, BL>, \
        cudaFuncAttributeMaxDynamicSharedMemorySize, 99 * 1024); \
    topk_merge_kernel<KB, BL><<<batch, BL, smem>>>(tv_, ti_, ov, oi, nseg); \
}

LAUNCH_REDUCE(1, 256)
LAUNCH_REDUCE(8, 256)
LAUNCH_REDUCE(16, 256)
LAUNCH_REDUCE(32, 256)
LAUNCH_REDUCE(64, 128)
LAUNCH_MERGE(1, 256)
LAUNCH_MERGE(8, 256)
LAUNCH_MERGE(16, 256)
LAUNCH_MERGE(32, 256)
LAUNCH_MERGE(64, 128)

// Fused big-row path: enqueue reduce + merge back-to-back in ONE host call so
// there is no Python round-trip (and GPU idle gap) between the two launches.
#define FUSED_BIG(KB, RBL, MBL) \
extern "C" void launch_big_##KB(const float* x, float* ov, long long* oi, \
    float* tv_, long long* ti_, int batch, int n, int nseg, int seg_size) { \
    constexpr int RM = RBL * (KB + 1); \
    size_t rsmem = (size_t)RM * (sizeof(float) + sizeof(int)); \
    cudaFuncSetAttribute(topk_reduce_kernel<KB, RBL>, \
        cudaFuncAttributeMaxDynamicSharedMemorySize, 99 * 1024); \
    dim3 grid(batch, nseg); \
    topk_reduce_kernel<KB, RBL><<<grid, RBL, rsmem>>>( \
        x, ov, oi, tv_, ti_, batch, n, nseg, seg_size); \
    constexpr int MM = MBL * (KB + 1); \
    size_t msmem = (size_t)MM * (sizeof(float) + sizeof(int)); \
    cudaFuncSetAttribute(topk_merge_kernel<KB, MBL>, \
        cudaFuncAttributeMaxDynamicSharedMemorySize, 99 * 1024); \
    topk_merge_kernel<KB, MBL><<<batch, MBL, msmem>>>(tv_, ti_, ov, oi, nseg); \
}
FUSED_BIG(64, 128, 128)
"""

_SO = None


def _build_and_load():
    here = os.path.dirname(os.path.abspath(__file__))
    cache_dir = os.path.join(here, "_kb_topk_cache")
    os.makedirs(cache_dir, exist_ok=True)
    so_path = os.path.join(cache_dir, "kb_topk_sm86.so")

    if not os.path.exists(so_path):
        cu_path = os.path.join(cache_dir, "topk.cu")
        with open(cu_path, "w") as f:
            f.write(_CUDA_SRC)
        last = None
        for ccbin in ("/usr/bin/g++-14", "/usr/bin/g++-13", "/usr/bin/g++-12", "g++"):
            cmd = [
                "nvcc", "-O3", "--use_fast_math", "-std=c++17",
                "-arch=sm_86", "-Xcompiler=-fPIC",
                "-ccbin", ccbin, "--shared", cu_path, "-o", so_path,
            ]
            try:
                subprocess.run(cmd, check=True)
                break
            except subprocess.CalledProcessError as e:
                last = e
        else:
            raise RuntimeError(f"nvcc build failed: {last}")

    lib = ctypes.CDLL(so_path)
    for fn in ("launch_reduce_1", "launch_reduce_8", "launch_reduce_16",
               "launch_reduce_32", "launch_reduce_64",
               "launch_merge_1", "launch_merge_8", "launch_merge_16",
               "launch_merge_32", "launch_merge_64",
               "launch_big_64",
               "launch_radix_8", "launch_radix_16", "launch_radix_32",
               "launch_radix_64", "launch_radix_big_64"):
        getattr(lib, fn).restype = None
    return lib


def _lib():
    global _SO
    if _SO is None:
        _SO = _build_and_load()
    return _SO


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        b, n, k = self.batch, self.n, self.k
        x = x.contiguous()
        out_vals = torch.empty((b, k), dtype=torch.float32, device=x.device)
        out_idx = torch.empty((b, k), dtype=torch.int64, device=x.device)

        lib = _lib()
        xp = ctypes.c_void_p(x.data_ptr())
        ovp = ctypes.c_void_p(out_vals.data_ptr())
        oip = ctypes.c_void_p(out_idx.data_ptr())

        # K<=16: the per-thread cascade is cheaper than radix's fixed pass
        # overhead. K>=32: radix-256 histogram select wins. The single very-large
        # row uses a segmented radix (many blocks reduce, one merges) so the row
        # is reduced with full GPU parallelism instead of one block.
        if k in (1, 8, 16):
            getattr(lib, f"launch_reduce_{k}")(xp, ovp, oip, xp, oip, b, n, 1, n)
        elif n <= _SEG_THRESHOLD:
            getattr(lib, f"launch_radix_{k}")(xp, ovp, oip, b, n, k)
        else:
            seg_size = 2048
            nseg = (n + seg_size - 1) // seg_size
            tmp_vals = torch.empty((b, nseg, k), dtype=torch.float32, device=x.device)
            tmp_idx = torch.empty((b, nseg, k), dtype=torch.int32, device=x.device)
            tvp = ctypes.c_void_p(tmp_vals.data_ptr())
            tip = ctypes.c_void_p(tmp_idx.data_ptr())
            getattr(lib, f"launch_radix_big_{k}")(
                xp, ovp, oip, tvp, tip, b, n, nseg, seg_size, k)

        return out_vals, out_idx
