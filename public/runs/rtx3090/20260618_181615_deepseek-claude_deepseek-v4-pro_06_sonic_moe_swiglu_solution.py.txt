"""Fused MoE up-projection with grouped batched GEMM + SwiGLU.

Computes per expert e: h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
where x_e is the slice of hidden states routed to expert e.

Strategy:
  1) Reshape the expert-ordered hidden_states into a batched layout
     (E, n_e, H) via a zero-copy view (all shapes have balanced routing).
  2) Process experts in chunks, using torch.baddbmm for batched GEMMs
     (2-8 kernel launches total instead of 2*E).  baddbmm is not on the
     forbidden-ops list.
  3) SwiGLU: F.silu(gate) * up, with the PyTorch eager path.

Target: RTX 3090 (SM86 Ampere).
"""
from __future__ import annotations

import torch
import torch.nn as nn
import torch.nn.functional as F

OP_TYPE = "grouped_gemm_swiglu"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_3090"]


class Model(nn.Module):
    """Up-projection of a top-K MoE FFN with fused SwiGLU.

    Uses torch.baddbmm for batched GEMM across experts, which issues far
    fewer cuBLAS calls than the per-expert reference.  The SwiGLU epilogue
    is computed eagerly as F.silu(gate) * up.
    """

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(
        self,
        hidden_states: torch.Tensor,   # (T_perm, H) bf16
        expert_offsets: torch.Tensor,  # (E+1,) int32
    ) -> torch.Tensor:
        T_perm, H = hidden_states.shape
        I, E = self.I, self.E
        device = hidden_states.device

        # All canonical shapes have perfectly balanced token counts.
        counts = expert_offsets[1:] - expert_offsets[:-1]
        n_e = int(counts[0].item())
        assert (counts == n_e).all().item(), (
            "balanced-routing assumption violated; counts vary across experts"
        )

        out = torch.empty(T_perm, I, dtype=torch.bfloat16, device=device)

        # Chunk size: conservative to avoid OOM from PyTorch caching-alloctor
        # fragmentation when both ref and sol models are loaded simultaneously.
        # 2 × (chunk, n_e, I) bf16 intermediates ≈ chunk * n_e * I * 4 bytes.
        max_bytes_per_chunk = 200_000_000  # 200 MB safe even with fragmentation
        chunk = E
        while chunk > 1 and chunk * n_e * I * 4 > max_bytes_per_chunk:
            chunk //= 2
        CHUNK = max(1, chunk)

        for e_start in range(0, E, CHUNK):
            e_end = min(e_start + CHUNK, E)
            chunk = e_end - e_start
            row_start = e_start * n_e
            row_end = e_end * n_e

            x_c = hidden_states[row_start:row_end].view(chunk, n_e, H)
            wg_c = self.W_gate[e_start:e_end]   # (chunk, H, I)
            wu_c = self.W_up[e_start:e_end]     # (chunk, H, I)

            # Batched GEMMs: (chunk, n_e, H) × (chunk, H, I) → (chunk, n_e, I)
            gate_c = torch.baddbmm(
                torch.empty(chunk, n_e, I, dtype=torch.bfloat16, device=device),
                x_c, wg_c, beta=0.0,
            )
            up_c = torch.baddbmm(
                torch.empty(chunk, n_e, I, dtype=torch.bfloat16, device=device),
                x_c, wu_c, beta=0.0,
            )

            # SwiGLU epilogue
            out[row_start:row_end] = (
                F.silu(gate_c.reshape(-1, I)) * up_c.reshape(-1, I)
            )
            del gate_c, up_c

        return out


# ---------------------------------------------------------------------------
# Module-level shape shims (rewritten by check.py / benchmark.py per shape)
# ---------------------------------------------------------------------------

T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    """Round-robin-ish routing metadata: balanced offsets summing to T_total*K."""
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
