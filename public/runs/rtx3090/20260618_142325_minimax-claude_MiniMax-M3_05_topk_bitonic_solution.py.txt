"""TopK via Triton kernels for RTX 3090 (SM86, GDDR6X, 936 GB/s).

Two kernel strategies picked by shape:
  - Single-block repeated argmax: K iterations of argmax over the
    loaded row. Used for multi-row shapes where 1 block per row
    already covers most SMs. Read the data once, do K reductions
    in registers/shmem.
  - Two-stage chunked: chunk kernel produces per-chunk top-K, merge
    kernel collapses per-row candidates. Used for batch=1 with huge
    n (where single-block would burn the read on one SM), and for
    K > 8 (where K iterations of full argmax is too slow).

K=1 always uses a direct argmax path because Triton's tl.topk has an
internal issue at K=1 ('dtype' object has no attribute 'numel').

The packed-value trick: float -> sortable uint32 -> upper-32 of a uint64,
index -> lower-32. tl.topk on uint64 sorts first by value then by index
for ties, and the unsigned comparison naturally gives float order once
we apply the standard sign-flip transform.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _to_sortable_u32(x_int32):
    """Float bit pattern -> uint32 with the same total order under unsigned compare."""
    sign_mask = (x_int32 >> 31).to(tl.uint32)
    return x_int32.to(tl.uint32) ^ (0x80000000 | sign_mask)


@triton.jit
def _from_sortable_u32(u32):
    """Inverse of _to_sortable_u32."""
    sign_mask = (u32 >> 31)
    return (u32 ^ (0x80000000 | sign_mask)).to(tl.int32)


@triton.jit
def _topk_argmax_loop_kernel(
    X_ptr, V_ptr, I_ptr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    """One CTA per row. Repeated argmax over the loaded row.

    The row is loaded once (BLOCK_N = next_pow2(N) >= N, padded with
    -inf). Then K iterations of argmax + mask-out. Each iteration is
    a tree reduction in registers/shmem, no DRAM re-read.
    """
    pid = tl.program_id(0)
    row_offset = pid.to(tl.int64) * N

    offsets = tl.arange(0, BLOCK_N)
    mask = offsets < N
    x = tl.load(X_ptr + row_offset + offsets, mask=mask, other=float("-inf"))

    out_base = pid.to(tl.int64) * K
    for i in tl.static_range(K):
        idx = tl.argmax(x, axis=0)
        val = tl.max(x, axis=0)
        tl.store(V_ptr + out_base + i, val)
        tl.store(I_ptr + out_base + i, idx.to(tl.int64))
        x = tl.where(offsets == idx, float("-inf"), x)


@triton.jit
def _topk_chunk_kernel(
    X_ptr, V_ptr, I_ptr,
    N: tl.constexpr,
    K: tl.constexpr,
    BLOCK_N: tl.constexpr,
    NUM_CHUNKS: tl.constexpr,
):
    """One CTA per (row, chunk). Bitonic top-K of the chunk."""
    pid_b = tl.program_id(0)
    pid_c = tl.program_id(1)

    row_offset = pid_b.to(tl.int64) * N
    chunk_start = pid_c * BLOCK_N

    offsets = chunk_start + tl.arange(0, BLOCK_N)
    mask = offsets < N
    x = tl.load(X_ptr + row_offset + offsets, mask=mask, other=float("-inf"))

    if K == 1:
        idx = tl.argmax(x, axis=0)
        val = tl.max(x, axis=0)
        out_offset = (pid_b * NUM_CHUNKS + pid_c)
        tl.store(V_ptr + out_offset, val)
        tl.store(I_ptr + out_offset, (idx + chunk_start).to(tl.int64))
    else:
        x_int = x.to(tl.int32, bitcast=True)
        sign = (x_int >> 31).to(tl.uint32)
        x_sort = x_int.to(tl.uint32) ^ (0x80000000 | sign)
        idx = offsets.to(tl.uint32)
        packed = (x_sort.to(tl.uint64) << 32) | idx.to(tl.uint64)
        packed = tl.where(mask, packed, tl.zeros_like(packed))
        packed_topk = tl.topk(packed, k=K)

        val_sort = (packed_topk >> 32).to(tl.uint32)
        idx_topk = (packed_topk & 0xFFFFFFFF).to(tl.uint32)
        sign_top = (val_sort >> 31)
        val_int = (val_sort ^ (0x80000000 | sign_top)).to(tl.int32)
        val = val_int.to(tl.float32, bitcast=True)

        out_offsets = (pid_b * NUM_CHUNKS + pid_c) * K + tl.arange(0, K)
        tl.store(V_ptr + out_offsets, val)
        tl.store(I_ptr + out_offsets, idx_topk.to(tl.int64))


@triton.jit
def _topk_merge_kernel(
    CV_ptr, CI_ptr, V_ptr, I_ptr,
    K: tl.constexpr,
    CANDIDATES: tl.constexpr,
):
    """One CTA per row. Merges per-chunk candidates into the final top-K."""
    pid = tl.program_id(0)
    cand_offset = pid.to(tl.int64) * CANDIDATES

    offsets = tl.arange(0, CANDIDATES)
    cv = tl.load(CV_ptr + cand_offset + offsets)
    ci = tl.load(CI_ptr + cand_offset + offsets)

    if K == 1:
        idx = tl.argmax(cv, axis=0)
        val = tl.max(cv, axis=0)
        out_idx = tl.sum(tl.where(offsets == idx, ci, 0))
        out_offset = pid.to(tl.int64)
        tl.store(V_ptr + out_offset, val)
        tl.store(I_ptr + out_offset, out_idx)
    else:
        cv_int = cv.to(tl.int32, bitcast=True)
        sign = (cv_int >> 31).to(tl.uint32)
        cv_sort = cv_int.to(tl.uint32) ^ (0x80000000 | sign)
        ci_u32 = ci.to(tl.int32).to(tl.uint32)
        packed = (cv_sort.to(tl.uint64) << 32) | ci_u32.to(tl.uint64)
        packed_topk = tl.topk(packed, k=K)

        val_sort = (packed_topk >> 32).to(tl.uint32)
        idx_topk = (packed_topk & 0xFFFFFFFF).to(tl.uint32)
        sign_top = (val_sort >> 31)
        val_int = (val_sort ^ (0x80000000 | sign_top)).to(tl.int32)
        val = val_int.to(tl.float32, bitcast=True)

        out_offsets = pid.to(tl.int64) * K + tl.arange(0, K)
        tl.store(V_ptr + out_offsets, val)
        tl.store(I_ptr + out_offsets, idx_topk.to(tl.int64))


def _next_pow2(x: int) -> int:
    p = 1
    while p < x:
        p *= 2
    return p


class Model(nn.Module):
    """Top-k over the last dim of a 2D fp32 tensor."""

    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        # dummy buffer so state_dict is non-empty and load_state_dict(strict=True)
        # is meaningful (matches reference.py).
        self.register_buffer("_dummy", torch.zeros(1))
        self._configure()

    def _configure(self):
        n = self.n
        k = self.k
        # Strategy selection:
        # - batch == 1 with n > 16384: chunked (single block would use 1/82 SMs)
        # - k > 8: chunked (single-block K iterations of full argmax is O(K*N))
        # - otherwise: single-block repeated argmax (one block per row)
        if (self.batch == 1 and n > 16384) or k > 8:
            self.strategy = "chunked"
            # Sweet spot: 32 chunks of 4096 wins for (1, 131072, 64).
            # CANDIDATES = num_chunks * K. Cap CANDIDATES at 2048.
            max_candidates = 2048
            max_chunks = max_candidates // k
            mc = 1
            while mc * 2 <= max_chunks:
                mc *= 2
            max_chunks = mc
            num_chunks = max_chunks
            # BLOCK_N in [1024, 16384]
            while num_chunks > 1 and (n + num_chunks - 1) // num_chunks > 16384:
                num_chunks //= 2
            while num_chunks < max_chunks and (n + num_chunks - 1) // num_chunks >= 1024:
                num_chunks *= 2
            self.num_chunks = num_chunks
            self.block_n = _next_pow2((n + num_chunks - 1) // num_chunks)
            if self.block_n > 16384:
                self.block_n = 16384
                nc2 = (n + self.block_n - 1) // self.block_n
                self.num_chunks = 1
                while self.num_chunks < nc2:
                    self.num_chunks *= 2
            self.candidates = self.num_chunks * k
        else:
            self.strategy = "single"
            self.block_n = _next_pow2(n)

    def forward(self, x: torch.Tensor):
        v = torch.empty(self.batch, self.k, dtype=torch.float32, device=x.device)
        i = torch.empty(self.batch, self.k, dtype=torch.int64, device=x.device)

        if self.strategy == "single":
            _topk_argmax_loop_kernel[(self.batch,)](
                x, v, i, self.n, self.k, self.block_n,
                num_warps=4,
            )
        else:
            cv = torch.empty(
                self.batch, self.num_chunks, self.k,
                dtype=torch.float32, device=x.device,
            )
            ci = torch.empty(
                self.batch, self.num_chunks, self.k,
                dtype=torch.int64, device=x.device,
            )
            _topk_chunk_kernel[(self.batch, self.num_chunks)](
                x, cv, ci, self.n, self.k, self.block_n, self.num_chunks,
                num_warps=4,
            )
            _topk_merge_kernel[(self.batch,)](
                cv, ci, v, i, self.k, self.candidates,
                num_warps=4,
            )

        return v, i


batch = 64
n = 8192
k = 8


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
