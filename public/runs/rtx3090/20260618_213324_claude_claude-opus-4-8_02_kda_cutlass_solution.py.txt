"""Kimi Delta Attention (KDA) forward, chunk form, custom Triton kernels.

Hand-written chunk-parallel KDA for RTX 3090 (SM86). Independent Triton
reimplementation of the standard chunked delta-rule decomposition:

  stage 1 (intra, parallel over b,h,chunk):
      gc = cumsum(g) within chunk;
      Aqk[c,j] = scale*<q_c e^{gc_c-gc_j}, k_j>   (j<=c)
      Akk[c,i] = beta_c*<k_c e^{gc_c-gc_i}, k_i>  (i<c, strict lower)
      Ainv = (I + Akk)^{-1}    (forward substitution, fp32)
      w = Ainv @ (beta * k * e^{gc});  u = Ainv @ (beta * v)
      kg = k * e^{gc_last - gc}
  stage 2 (scan, parallel over b,h,v-block; sequential over chunks):
      h[i] = S;  v_new = u - w@S;  S = diag(e^{gc_last})S + kg^T @ v_new
  stage 3 (output, parallel over b,h,chunk,v-block):
      o = scale*(q e^{gc}) @ h + tril(Aqk) @ v_new
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


# ----------------------------------------------------------------------------
# stage 1: intra-chunk (cumsum, Aqk, Akk, inverse, w, u, kg)
# ----------------------------------------------------------------------------
@triton.jit
def _intra_kernel(
    q, k, v, g, beta,
    qg_out, gclast, w, u, kg, Aqk,
    T, NT,
    H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr,
    NSTEP: tl.constexpr,
    scale,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    qk_base = (i_b * T * H + i_h) * K
    v_base = (i_b * T * H + i_h) * V
    b_base = (i_b * T * H + i_h)
    a_base = (i_b * T * H + i_h) * BT

    o_t = tl.arange(0, BT)

    # ---- cumsum g within chunk ----
    p_g = tl.make_block_ptr(g + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    b_g = tl.load(p_g, boundary_check=(0, 1)).to(tl.float32)
    b_gc = tl.cumsum(b_g, axis=0)  # (BT, K)

    # last-row cumsum (chunk total decay) -> tiny per-chunk buffer for the scan
    b_gc_last = tl.sum(tl.where((o_t == (BT - 1))[:, None], b_gc, 0.0), 0)  # (K,)
    gclast_off = ((i_b * NT + i_t) * H + i_h) * K
    tl.store(gclast + gclast_off + tl.arange(0, K), b_gc_last)

    eg = tl.exp(b_gc)         # e^{gc}
    egn = tl.exp(-b_gc)       # e^{-gc}

    p_q = tl.make_block_ptr(q + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_k = tl.make_block_ptr(k + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_v = tl.make_block_ptr(v + v_base, (T, V), (H * V, 1), (i_t * BT, 0), (BT, V), (1, 0))
    b_q = tl.load(p_q, boundary_check=(0, 1)).to(tl.float32)
    b_k = tl.load(p_k, boundary_check=(0, 1)).to(tl.float32)
    b_v = tl.load(p_v, boundary_check=(0, 1)).to(tl.float32)

    p_beta = tl.make_block_ptr(beta + b_base, (T,), (H,), (i_t * BT,), (BT,), (0,))
    b_beta = tl.load(p_beta, boundary_check=(0,)).to(tl.float32)

    # decayed factors as bf16 operands for the BTxBT tensor-core dots
    b_qg = (b_q * eg).to(tl.bfloat16)
    b_kgp = (b_k * eg).to(tl.bfloat16)
    b_kgm = (b_k * egn).to(tl.bfloat16)

    # store q*e^{gc} (decayed query) for the output kernel -> it reads this
    # directly instead of re-loading q and gc and recomputing the exp.
    p_qg = tl.make_block_ptr(qg_out + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    tl.store(p_qg, b_qg, boundary_check=(0, 1))

    # Aqk (BT,BT), lower incl diag, scaled
    b_Aqk = tl.dot(b_qg, tl.trans(b_kgm)) * scale
    m_le = o_t[:, None] >= o_t[None, :]
    b_Aqk = tl.where(m_le, b_Aqk, 0.0)
    p_Aqk = tl.make_block_ptr(Aqk + a_base, (T, BT), (H * BT, 1), (i_t * BT, 0), (BT, BT), (1, 0))
    tl.store(p_Aqk, b_Aqk.to(Aqk.dtype.element_ty), boundary_check=(0, 1))

    # Akk strict lower (L), fp32
    b_Akk = tl.dot(b_kgp, tl.trans(b_kgm)) * b_beta[:, None]
    m_lt = o_t[:, None] > o_t[None, :]
    b_L = tl.where(m_lt, b_Akk, 0.0)

    # ---- inverse of (I + L) via Neumann doubling ----
    # N = -L is strictly lower (nilpotent, N^BT = 0), so
    # (I+L)^{-1} = sum_i N^i = prod_j (I + N^{2^j}).  log2(BT) doubling steps,
    # each two BTxBT tensor-core matmuls instead of BT serial scalar steps.
    b_eye = (o_t[:, None] == o_t[None, :]).to(tl.float32)
    b_N = -b_L
    b_Ai = b_eye + b_N            # S_1 = sum_{i<2} N^i  (fp32 accumulator)
    b_P = b_N.to(tl.bfloat16)
    for _ in tl.static_range(NSTEP):   # each step doubles the term count
        b_P = tl.dot(b_P, b_P).to(tl.bfloat16)
        b_Ai = b_Ai + tl.dot(b_P, b_Ai.to(tl.bfloat16))

    # ---- w, u ----
    b_rhs_w = (b_beta[:, None] * b_k * eg).to(w.dtype.element_ty)
    b_rhs_u = (b_beta[:, None] * b_v).to(u.dtype.element_ty)
    b_Ai_c = b_Ai.to(w.dtype.element_ty)
    b_w = tl.dot(b_Ai_c, b_rhs_w)
    b_u = tl.dot(b_Ai_c, b_rhs_u)
    p_w = tl.make_block_ptr(w + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    p_u = tl.make_block_ptr(u + v_base, (T, V), (H * V, 1), (i_t * BT, 0), (BT, V), (1, 0))
    tl.store(p_w, b_w.to(w.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_u, b_u.to(u.dtype.element_ty), boundary_check=(0, 1))

    # ---- kg = k * e^{gc_last - gc} ----
    b_kg = b_k * tl.exp(b_gc_last[None, :] - b_gc)
    p_kg = tl.make_block_ptr(kg + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
    tl.store(p_kg, b_kg.to(kg.dtype.element_ty), boundary_check=(0, 1))


# ----------------------------------------------------------------------------
# stage 2: sequential state scan -> h (per chunk state), v_new
# ----------------------------------------------------------------------------
@triton.jit
def _scan_kernel(
    w, u, kg, gclast, h, v_new,
    T, NT,
    H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr,
    BV: tl.constexpr, BK: tl.constexpr,
):
    i_v = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    o_k = tl.arange(0, K)

    # state accumulator (K, BV) fp32, K split into BK-blocks handled as full K here
    b_h = tl.zeros([K, BV], dtype=tl.float32)

    for i_t in range(NT):
        qk_base = (i_b * T * H + i_h) * K
        v_base = (i_b * T * H + i_h) * V
        h_base = ((i_b * NT + i_t) * H + i_h) * K * V

        # store current state to h[i_t]
        p_h = tl.make_block_ptr(h + h_base, (K, V), (V, 1), (0, i_v * BV), (K, BV), (1, 0))
        tl.store(p_h, b_h.to(h.dtype.element_ty), boundary_check=(0, 1))

        # v_new = u - w @ S
        p_w = tl.make_block_ptr(w + qk_base, (T, K), (H * K, 1), (i_t * BT, 0), (BT, K), (1, 0))
        b_w = tl.load(p_w, boundary_check=(0, 1))
        b_wh = tl.dot(b_w, b_h.to(b_w.dtype))  # (BT, BV)
        p_u = tl.make_block_ptr(u + v_base, (T, V), (H * V, 1), (i_t * BT, i_v * BV), (BT, BV), (1, 0))
        b_vn = tl.load(p_u, boundary_check=(0, 1)) - b_wh
        p_vn = tl.make_block_ptr(v_new + v_base, (T, V), (H * V, 1), (i_t * BT, i_v * BV), (BT, BV), (1, 0))
        tl.store(p_vn, b_vn.to(v_new.dtype.element_ty), boundary_check=(0, 1))

        # decay state by e^{gc_last} (read tiny per-chunk gc_last buffer)
        gclast_off = ((i_b * NT + i_t) * H + i_h) * K
        b_gc_last = tl.load(gclast + gclast_off + o_k).to(tl.float32)  # (K,)
        b_h = b_h * tl.exp(b_gc_last)[:, None]

        # S += kg^T @ v_new
        p_kg = tl.make_block_ptr(kg + qk_base, (K, T), (1, H * K), (0, i_t * BT), (K, BT), (0, 1))
        b_kg = tl.load(p_kg, boundary_check=(0, 1))  # (K, BT)
        b_h += tl.dot(b_kg, b_vn.to(b_kg.dtype))


# ----------------------------------------------------------------------------
# stage 3: output o = scale*(q e^{gc}) @ h + tril(Aqk) @ v_new
# ----------------------------------------------------------------------------
@triton.jit
def _o_kernel(
    qg, h, Aqk, v_new, o,
    T, NT,
    H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr,
    BV: tl.constexpr, BK: tl.constexpr,
    scale,
):
    i_v = tl.program_id(0)
    i_t = tl.program_id(1)
    i_bh = tl.program_id(2)
    i_b = i_bh // H
    i_h = i_bh % H

    qk_base = (i_b * T * H + i_h) * K
    v_base = (i_b * T * H + i_h) * V
    a_base = (i_b * T * H + i_h) * BT
    h_base = ((i_b * NT + i_t) * H + i_h) * K * V

    o_t = tl.arange(0, BT)

    b_o = tl.zeros([BT, BV], dtype=tl.float32)
    for i_k in range(tl.cdiv(K, BK)):
        p_qg = tl.make_block_ptr(qg + qk_base, (T, K), (H * K, 1), (i_t * BT, i_k * BK), (BT, BK), (1, 0))
        p_h = tl.make_block_ptr(h + h_base, (K, V), (V, 1), (i_k * BK, i_v * BV), (BK, BV), (1, 0))
        b_qg = tl.load(p_qg, boundary_check=(0, 1))
        b_h = tl.load(p_h, boundary_check=(0, 1))
        b_o += tl.dot(b_qg, b_h)
    b_o *= scale

    p_A = tl.make_block_ptr(Aqk + a_base, (T, BT), (H * BT, 1), (i_t * BT, 0), (BT, BT), (1, 0))
    b_A = tl.load(p_A, boundary_check=(0, 1))
    m_le = o_t[:, None] >= o_t[None, :]
    b_A = tl.where(m_le, b_A, 0.0).to(v_new.dtype.element_ty)
    p_vn = tl.make_block_ptr(v_new + v_base, (T, V), (H * V, 1), (i_t * BT, i_v * BV), (BT, BV), (1, 0))
    b_vn = tl.load(p_vn, boundary_check=(0, 1))
    b_o += tl.dot(b_A, b_vn)

    p_o = tl.make_block_ptr(o + v_base, (T, V), (H * V, 1), (i_t * BT, i_v * BV), (BT, BV), (1, 0))
    tl.store(p_o, b_o.to(o.dtype.element_ty), boundary_check=(0, 1))


def kda_forward(q, k, v, g, beta, scale, chunk_size=64):
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0
    NT = T // BT

    q = q.contiguous()
    k = k.contiguous()
    v = v.contiguous()
    g = g.contiguous()
    beta = beta.contiguous()

    dev = q.device
    qg = torch.empty(B, T, H, K, device=dev, dtype=torch.bfloat16)   # q * e^{gc} (for output)
    gclast = torch.empty(B, NT, H, K, device=dev, dtype=torch.float32)  # per-chunk total decay (for scan)
    w = torch.empty(B, T, H, K, device=dev, dtype=torch.bfloat16)
    u = torch.empty(B, T, H, V, device=dev, dtype=torch.bfloat16)
    kg = torch.empty(B, T, H, K, device=dev, dtype=torch.bfloat16)
    Aqk = torch.empty(B, T, H, BT, device=dev, dtype=torch.bfloat16)
    h = torch.empty(B, NT, H, K, V, device=dev, dtype=torch.bfloat16)
    v_new = torch.empty(B, T, H, V, device=dev, dtype=torch.bfloat16)
    o = torch.empty(B, T, H, V, device=dev, dtype=torch.bfloat16)

    # stage 1
    nstep = max(0, (BT - 1).bit_length() - 1)  # doubling steps to cover BT terms
    _intra_kernel[(NT, B * H)](
        q, k, v, g, beta, qg, gclast, w, u, kg, Aqk,
        T, NT, H, K, V, BT, nstep, scale,
        num_warps=8, num_stages=1,
    )

    # stage 2 (scan) -- small BV maximizes program count for the sequential scan.
    # Deeper software pipeline (ns=3) helps when occupancy is low (few programs)
    # or the chunk loop is long; otherwise ns=2 keeps it lean.
    BV = 16
    BK = 64
    scan_ns = 3 if (B * H <= 4 or NT >= 64) else 2
    _scan_kernel[(triton.cdiv(V, BV), B * H)](
        w, u, kg, gclast, h, v_new,
        T, NT, H, K, V, BT, BV, BK,
        num_warps=4, num_stages=scan_ns,
    )

    # stage 3 (output)
    BVo = 64
    BKo = 64
    _o_kernel[(triton.cdiv(V, BVo), NT, B * H)](
        qg, h, Aqk, v_new, o,
        T, NT, H, K, V, BT, BVo, BKo, scale,
        num_warps=4, num_stages=2,
    )

    return o


class Model(nn.Module):
    """KDA forward (chunk form). No learned parameters; all inputs are activations."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        return kda_forward(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)


# Module-level shape shims (overridden by check.py / benchmark.py per shape).
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
