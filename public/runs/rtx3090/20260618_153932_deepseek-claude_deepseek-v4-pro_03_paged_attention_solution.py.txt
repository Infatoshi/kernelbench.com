"""Paged attention decode kernel for RTX 3090 (SM86 Ampere).

Custom CUDA kernel (loaded from kernel.so, compiled from kernel.cu):
one threadblock per (batch, head), cooperative KV-cache page loads
into shared memory, warp-shuffle dot-product reduction, two-pass
batch-softmax for numerical stability. bf16 in/out, fp32 compute.

Compiled as a standalone .so to avoid PyTorch header / GCC 16
template incompatibilities.  __global__ void paged_attention_kernel.
"""

import ctypes
import math
import os
from pathlib import Path

import torch
import torch.nn as nn

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# Module-level shape knobs (overridden by check.py / benchmark.py)
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

# ── load the pre-compiled kernel ──────────────────────────────────────────
_kernel_lib = None


def _get_kernel_lib():
    global _kernel_lib
    if _kernel_lib is not None:
        return _kernel_lib

    so_path = Path(__file__).parent / "kernel.so"
    if not so_path.exists():
        raise FileNotFoundError(
            f"kernel.so not found at {so_path}. "
            "Compile with: /opt/cuda/bin/nvcc -O3 --use_fast_math -arch=sm_86 "
            "-shared -Xcompiler -fPIC -std=c++17 -o kernel.so kernel.cu"
        )

    _kernel_lib = ctypes.CDLL(str(so_path))

    # int launch_paged_attention(void* query, void* kv_cache,
    #     void* block_table, void* seq_lens, void* output,
    #     int batch, int num_heads, int num_kv_heads, int head_dim,
    #     int page_size, int max_blocks, float scale, cudaStream_t stream)
    _kernel_lib.launch_paged_attention.argtypes = [
        ctypes.c_void_p,  # query
        ctypes.c_void_p,  # kv_cache
        ctypes.c_void_p,  # block_table
        ctypes.c_void_p,  # seq_lens
        ctypes.c_void_p,  # output
        ctypes.c_int,     # batch
        ctypes.c_int,     # num_heads
        ctypes.c_int,     # num_kv_heads
        ctypes.c_int,     # head_dim
        ctypes.c_int,     # page_size
        ctypes.c_int,     # max_blocks
        ctypes.c_float,   # scale
        ctypes.c_void_p,  # stream (cudaStream_t)
    ]
    _kernel_lib.launch_paged_attention.restype = ctypes.c_int

    return _kernel_lib


# ── Model ──────────────────────────────────────────────────────────────────


class Model(nn.Module):
    """Single-query paged attention decode.

    Forward inputs (all on device):
      query:       (batch, num_heads, head_dim)               bf16
      kv_cache:    (num_blocks, page_size, num_kv_heads, 2*head_dim)
      block_table: (batch, max_blocks)                        int32
      seq_lens:    (batch,)                                   int32

    Output:
      attn_out:    (batch, num_heads, head_dim)               bf16
    """

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        # No learned parameters; empty buffer for state_dict compatibility.
        self.register_buffer(
            "_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        lib = _get_kernel_lib()

        B, H, D = query.shape
        out = torch.empty(B, H, D, dtype=torch.bfloat16, device=query.device)

        err = lib.launch_paged_attention(
            ctypes.c_void_p(query.data_ptr()),
            ctypes.c_void_p(kv_cache.data_ptr()),
            ctypes.c_void_p(block_table.data_ptr()),
            ctypes.c_void_p(seq_lens.data_ptr()),
            ctypes.c_void_p(out.data_ptr()),
            B,
            H,
            self.num_kv_heads,
            D,
            self.page_size,
            block_table.shape[1],
            ctypes.c_float(self.scale),
            ctypes.c_void_p(0),  # default stream
        )
        if err != 0:
            torch.cuda.synchronize()
            raise RuntimeError(f"CUDA kernel launch failed with error {err}")

        return out


def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = (
        torch.randperm(total_pages)[: B * pages_per_seq]
        .reshape(B, pages_per_seq)
        .int()
    )
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
