"""Sonic-MoE up-projection: grouped GEMM + fused SwiGLU on RTX 3090 (SM86).

Implements per-expert variable-length grouped GEMM (gate + up) with SwiGLU
fused into the epilogue. Triton kernel, bf16 in/out, fp32 accumulate.

For each expert e (with tokens [offsets[e]:offsets[e+1]] in the permuted
hidden_states), computes:
    gate = x_e @ W_gate[e]    # (n_e, I)
    up   = x_e @ W_up[e]      # (n_e, I)
    out  = silu(gate) * up    # (n_e, I)

Kernel layout: 2D grid (E, N_TILES), each block loops over its expert's M
tiles inside the kernel. Per K iter we load X once and use it for two
independent mma sequences (gate and up) sharing the same X tile.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


# ---------- Triton kernel ------------------------------------------------------
@triton.jit
def _grouped_swiglu_kernel(
    X_ptr,            # (T_perm, H) bf16
    Wg_ptr, Wu_ptr,   # (E, H, I) bf16
    Offsets_ptr,      # (E+1,) int32
    Out_ptr,          # (T_perm, I) bf16
    H: tl.constexpr,
    I: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    stride_xm,
    stride_xk,
    stride_we,
    stride_wk,
    stride_wn,
    stride_om,
    stride_on,
):
    pid_e = tl.program_id(0)
    pid_n = tl.program_id(1)

    # Per-expert row range
    start = tl.load(Offsets_ptr + pid_e).to(tl.int64)
    end = tl.load(Offsets_ptr + pid_e + 1).to(tl.int64)
    n_tokens = end - start

    n_offsets = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)

    # W bases (constant across M tiles)
    wg_base = Wg_ptr + pid_e.to(tl.int64) * stride_we + n_offsets[None, :] * stride_wn
    wu_base = Wu_ptr + pid_e.to(tl.int64) * stride_we + n_offsets[None, :] * stride_wn
    o_row_base = Out_ptr + n_offsets[None, :] * stride_on

    for m_start in tl.range(0, n_tokens, BLOCK_M):
        m_offsets = m_start + tl.arange(0, BLOCK_M)
        m_mask = m_offsets < n_tokens
        m_global = start + m_offsets

        x_row = X_ptr + m_global[:, None] * stride_xm
        o_base = o_row_base + m_global[:, None] * stride_om

        acc_g = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
        acc_u = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

        for k_start in tl.range(0, H, BLOCK_K):
            k_offsets = k_start + tl.arange(0, BLOCK_K)
            x_blk = tl.load(
                x_row + k_offsets[None, :] * stride_xk,
                mask=m_mask[:, None],
                other=0.0,
            )
            wg_blk = tl.load(wg_base + k_offsets[:, None] * stride_wk)
            wu_blk = tl.load(wu_base + k_offsets[:, None] * stride_wk)
            acc_g = tl.dot(x_blk, wg_blk, acc_g)
            acc_u = tl.dot(x_blk, wu_blk, acc_u)

        # Fused SwiGLU epilogue
        silu_g = acc_g * tl.sigmoid(acc_g)
        out_val = (silu_g * acc_u).to(tl.bfloat16)

        tl.store(o_base, out_val, mask=m_mask[:, None])


# ---------- Tile config -------------------------------------------------------
# Empirically best on RTX 3090 (SM86) for the three target shapes:
# BM=128 BN=128 BK=64 NW=8 NS=3 hits ~96% of compute peak for shape 0/2
# and ~88% for the smaller shape 1.
_BM = 128
_BN = 128
_BK = 64
_NW = 8
_NS = 3


def _grouped_swiglu(hidden_states, expert_offsets, W_gate, W_up):
    T_perm, H = hidden_states.shape
    E, _, I = W_gate.shape

    out = torch.empty(T_perm, I, dtype=hidden_states.dtype, device=hidden_states.device)

    n_tiles = I // _BN
    grid = (E, n_tiles)

    _grouped_swiglu_kernel[grid](
        hidden_states, W_gate, W_up, expert_offsets, out,
        H, I, _BM, _BN, _BK,
        hidden_states.stride(0), hidden_states.stride(1),
        W_gate.stride(0), W_gate.stride(1), W_gate.stride(2),
        out.stride(0), out.stride(1),
        num_warps=_NW,
        num_stages=_NS,
    )
    return out


# ---------- Model -------------------------------------------------------------
class Model(nn.Module):
    """Up-projection of a top-K MoE FFN with fused SwiGLU (Triton)."""

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(
        self,
        hidden_states: torch.Tensor,   # (T_perm, H) bf16
        expert_offsets: torch.Tensor,  # (E+1,) int32
    ) -> torch.Tensor:
        return _grouped_swiglu(hidden_states, expert_offsets, self.W_gate, self.W_up)


# Module-level shape shims (rewritten by check.py / benchmark.py per shape).
T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
