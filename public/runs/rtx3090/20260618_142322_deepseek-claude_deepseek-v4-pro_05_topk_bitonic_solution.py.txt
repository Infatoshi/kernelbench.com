"""Top-k via register-based insertion scan + shared-memory warp merge + block bitonic sort.

Design:
  - Per-thread: sorted-descending buffer of k (value, index) pairs in registers.
    Insertion uses binary search + shift.  Only 2k registers needed per thread.
  - Warp merge: warps take turns using shared memory.  Each warp copies its
    per-thread buffers to shared memory, does a butterfly merge (reads partner
    data from the stable shared-memory copy, insertion-sorts each element
    into its register buffer), writes back for the next mask round.
    8 barriers between warps (~1 µs total).
  - Block merge: lane-0 of each warp writes its result to the block merge
    buffer; a bitonic sort extracts the block-wide top-k.
"""

from __future__ import annotations

import os
import tempfile

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load

OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


CUDA_SOURCE = r"""
#include <torch/extension.h>
#include <c10/cuda/CUDAStream.h>

#define MAX_K 64
#define WARP_SIZE 32
#define BLOCK_SIZE 256
#define NUM_WARPS (BLOCK_SIZE / WARP_SIZE)
#define MERGE_BUF_SIZE 512
#define NEG_INF __int_as_float(0xff7fffff)

// ---------------------------------------------------------------------------
__global__ void __launch_bounds__(BLOCK_SIZE, 1)
topk_scan_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    long* __restrict__ out_indices,
    float* __restrict__ inter_values,
    int* __restrict__   inter_indices,
    int batch_size, int n, int k, int blocks_per_row)
{
    int row       = blockIdx.x / blocks_per_row;
    int block_col = blockIdx.x % blocks_per_row;
    if (row >= batch_size) return;

    int tid     = threadIdx.x;
    int lane    = tid & 31;
    int warp_id = tid >> 5;

    // Shared memory: warp_sm (WARP_SIZE*MAX_K elements, 16 KB) + block merge (4 KB)
    __shared__ float warp_sm_val[WARP_SIZE * MAX_K];
    __shared__ int   warp_sm_idx[WARP_SIZE * MAX_K];
    __shared__ float block_val[MERGE_BUF_SIZE];
    __shared__ int   block_idx[MERGE_BUF_SIZE];

    #define WSM_V(l) (warp_sm_val + (l) * MAX_K)
    #define WSM_I(l) (warp_sm_idx + (l) * MAX_K)

    // ---- per-thread sorted buffer (registers) ----
    float my_val[MAX_K];
    int   my_idx[MAX_K];
    #pragma unroll
    for (int i = 0; i < MAX_K; i++) { my_val[i] = NEG_INF; my_idx[i] = -1; }

    int chunk_start = block_col * (n / blocks_per_row);
    int chunk_end   = (block_col == blocks_per_row - 1) ? n
                     : (block_col + 1) * (n / blocks_per_row);

    // ---- Phase 1: per-thread scan (insertion sort, all threads concurrent) ----
    for (int idx = chunk_start + tid; idx < chunk_end; idx += BLOCK_SIZE) {
        float val = __ldg(input + row * n + idx);
        if (val <= my_val[k - 1]) continue;

        int lo = 0, hi = k - 1;
        #pragma unroll 1
        while (lo < hi) {
            int mid = (lo + hi) >> 1;
            if (my_val[mid] < val) hi = mid;
            else                   lo = mid + 1;
        }
        int pos = lo;
        #pragma unroll
        for (int i = k - 1; i > pos; i--) {
            my_val[i] = my_val[i - 1];
            my_idx[i] = my_idx[i - 1];
        }
        my_val[pos] = val;
        my_idx[pos] = idx;
    }

    // ---- Phase 2: warp merge (sequential warps, shared memory) ----
    for (int w = 0; w < NUM_WARPS; w++) {
        if (warp_id == w) {
            // 2a. Copy my buffer → shared memory (pad with NEG_INF)
            for (int i = 0; i < MAX_K; i++) {
                int off = lane * MAX_K + i;
                if (i < k) {
                    WSM_V(lane)[i] = my_val[i];
                    WSM_I(lane)[i] = my_idx[i];
                } else {
                    WSM_V(lane)[i] = NEG_INF;
                    WSM_I(lane)[i] = -1;
                }
            }

            // 2b. Butterfly merge reading partner data from stable shared memory
            for (int mask = 1; mask < WARP_SIZE; mask <<= 1) {
                int partner = lane ^ mask;
                #pragma unroll 1
                for (int i = 0; i < k; i++) {
                    float p_val = WSM_V(partner)[i];
                    int   p_idx = WSM_I(partner)[i];

                    if (p_val <= my_val[k - 1]) continue;

                    int lo = 0, hi = k - 1;
                    #pragma unroll 1
                    while (lo < hi) {
                        int mid = (lo + hi) >> 1;
                        if (my_val[mid] < p_val) hi = mid;
                        else                     lo = mid + 1;
                    }
                    int pos = lo;
                    #pragma unroll
                    for (int j = k - 1; j > pos; j--) {
                        my_val[j] = my_val[j - 1];
                        my_idx[j] = my_idx[j - 1];
                    }
                    my_val[pos] = p_val;
                    my_idx[pos] = p_idx;
                }

                // Write merged result back to shared memory for next mask round
                for (int i = 0; i < k; i++) {
                    WSM_V(lane)[i] = my_val[i];
                    WSM_I(lane)[i] = my_idx[i];
                }
            }

            // 2c. Lane 0 writes warp result to block merge buffer
            int base = w * k;
            for (int i = 0; i < k; i++) {
                block_val[base + i] = WSM_V(0)[i];
                block_idx[base + i] = WSM_I(0)[i];
            }
        }
        __syncthreads();  // hand off shared memory to next warp
    }

    // Init unwritten portion of block merge buffer
    for (int i = tid + NUM_WARPS * k; i < MERGE_BUF_SIZE; i += BLOCK_SIZE) {
        block_val[i] = NEG_INF;
        block_idx[i] = -1;
    }
    __syncthreads();

    // ---- Phase 3: block merge (bitonic sort descending) ----
    #pragma unroll 1
    for (int stage = 2; stage <= MERGE_BUF_SIZE; stage <<= 1) {
        #pragma unroll 1
        for (int step = stage >> 1; step > 0; step >>= 1) {
            #pragma unroll
            for (int e = 0; e < MERGE_BUF_SIZE / BLOCK_SIZE; e++) {
                int i = tid * (MERGE_BUF_SIZE / BLOCK_SIZE) + e;
                int partner = i ^ step;
                if (i < partner && i < MERGE_BUF_SIZE) {
                    bool ascending = ((i & stage) != 0);
                    float av = block_val[i]; int ai = block_idx[i];
                    float bv = block_val[partner]; int bi = block_idx[partner];
                    bool swap = ascending ? (av > bv) : (av < bv);
                    if (swap) {
                        block_val[i] = bv; block_idx[i] = bi;
                        block_val[partner] = av; block_idx[partner] = ai;
                    }
                }
            }
            __syncthreads();
        }
    }

    // ---- write output ----
    if (blocks_per_row == 1) {
        if (tid < k) {
            out_values[row * k + tid]  = block_val[tid];
            out_indices[row * k + tid] = (long)block_idx[tid];
        }
    } else {
        if (tid < k) {
            int off = row * blocks_per_row * k + block_col * k + tid;
            inter_values[off] = block_val[tid];
            inter_indices[off] = block_idx[tid];
        }
    }
}

// ---------------------------------------------------------------------------
__global__ void topk_merge_kernel(
    const float* __restrict__ inter_values,
    const int*   __restrict__ inter_indices,
    float* __restrict__ out_values,
    long* __restrict__ out_indices,
    int batch_size, int blocks_per_row, int k)
{
    int row = blockIdx.x;
    if (row >= batch_size) return;

    int tid = threadIdx.x;
    int total_elems = blocks_per_row * k;

    __shared__ float merge_val[MERGE_BUF_SIZE];
    __shared__ int   merge_idx[MERGE_BUF_SIZE];

    for (int i = tid; i < MERGE_BUF_SIZE; i += BLOCK_SIZE) {
        merge_val[i] = NEG_INF;
        merge_idx[i] = -1;
    }
    __syncthreads();

    int row_off = row * blocks_per_row * k;
    for (int i = tid; i < total_elems; i += BLOCK_SIZE) {
        merge_val[i] = inter_values[row_off + i];
        merge_idx[i] = inter_indices[row_off + i];
    }
    __syncthreads();

    #pragma unroll 1
    for (int stage = 2; stage <= MERGE_BUF_SIZE; stage <<= 1) {
        #pragma unroll 1
        for (int step = stage >> 1; step > 0; step >>= 1) {
            #pragma unroll
            for (int e = 0; e < MERGE_BUF_SIZE / BLOCK_SIZE; e++) {
                int i = tid * (MERGE_BUF_SIZE / BLOCK_SIZE) + e;
                int partner = i ^ step;
                if (i < partner && i < MERGE_BUF_SIZE) {
                    bool ascending = ((i & stage) != 0);
                    float av = merge_val[i]; int ai = merge_idx[i];
                    float bv = merge_val[partner]; int bi = merge_idx[partner];
                    bool swap = ascending ? (av > bv) : (av < bv);
                    if (swap) {
                        merge_val[i] = bv; merge_idx[i] = bi;
                        merge_val[partner] = av; merge_idx[partner] = ai;
                    }
                }
            }
            __syncthreads();
        }
    }

    if (tid < k) {
        out_values[row * k + tid]  = merge_val[tid];
        out_indices[row * k + tid] = (long)merge_idx[tid];
    }
}

// ---------------------------------------------------------------------------
torch::Tensor launch_topk_scan(
    torch::Tensor input,
    torch::Tensor out_values,
    torch::Tensor out_indices,
    torch::Tensor inter_values,
    torch::Tensor inter_indices,
    int64_t batch_size, int64_t n, int64_t k, int64_t blocks_per_row)
{
    auto stream = at::cuda::getCurrentCUDAStream(input.device().index());
    int grid = (int)batch_size * (int)blocks_per_row;
    dim3 block(BLOCK_SIZE);
    topk_scan_kernel<<<grid, block, 0, stream>>>(
        input.data_ptr<float>(),
        out_values.data_ptr<float>(),
        out_indices.data_ptr<long>(),
        inter_values.data_ptr<float>(),
        inter_indices.data_ptr<int>(),
        (int)batch_size, (int)n, (int)k, (int)blocks_per_row);
    return out_values;
}

torch::Tensor launch_topk_merge(
    torch::Tensor inter_values,
    torch::Tensor inter_indices,
    torch::Tensor out_values,
    torch::Tensor out_indices,
    int64_t batch_size, int64_t blocks_per_row, int64_t k)
{
    auto stream = at::cuda::getCurrentCUDAStream(inter_values.device().index());
    dim3 block(BLOCK_SIZE);
    topk_merge_kernel<<<(int)batch_size, block, 0, stream>>>(
        inter_values.data_ptr<float>(),
        inter_indices.data_ptr<int>(),
        out_values.data_ptr<float>(),
        out_indices.data_ptr<long>(),
        (int)batch_size, (int)blocks_per_row, (int)k);
    return out_values;
}
"""

CPP_SOURCE = r"""
#include <torch/extension.h>

torch::Tensor launch_topk_scan(
    torch::Tensor input,
    torch::Tensor out_values,
    torch::Tensor out_indices,
    torch::Tensor inter_values,
    torch::Tensor inter_indices,
    int64_t batch_size, int64_t n, int64_t k, int64_t blocks_per_row);

torch::Tensor launch_topk_merge(
    torch::Tensor inter_values,
    torch::Tensor inter_indices,
    torch::Tensor out_values,
    torch::Tensor out_indices,
    int64_t batch_size, int64_t blocks_per_row, int64_t k);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("launch_topk_scan", &launch_topk_scan, "launch_topk_scan");
    m.def("launch_topk_merge", &launch_topk_merge, "launch_topk_merge");
}

# ---------------------------------------------------------------------------
_topk_module = None


def _get_module():
    global _topk_module
    if _topk_module is not None:
        return _topk_module
    build_dir = tempfile.mkdtemp(prefix="topk_build_")
    cuda_path = os.path.join(build_dir, "kernels.cu")
    cpp_path = os.path.join(build_dir, "bindings.cpp")
    with open(cuda_path, "w") as f:
        f.write(CUDA_SOURCE)
    with open(cpp_path, "w") as f:
        f.write(CPP_SOURCE)
    _topk_module = load(
        name="topk_bitonic_ops",
        sources=[cpp_path, cuda_path],
        extra_cflags=["-fpermissive", "-Wno-error"],
        extra_cuda_cflags=["-arch=sm_86", "--use_fast_math"],
        verbose=False,
    )
    return _topk_module


def _topk_cuda(x: torch.Tensor, k: int) -> tuple[torch.Tensor, torch.Tensor]:
    mod = _get_module()
    batch, n = x.shape
    device = x.device
    out_values = torch.empty(batch, k, dtype=torch.float32, device=device)
    out_indices = torch.empty(batch, k, dtype=torch.int64, device=device)
    mod.launch_topk_scan(
        x, out_values, out_indices,
        torch.empty(1, device=device),
        torch.empty(1, dtype=torch.int32, device=device),
        batch, n, k, 1,
    )
    return out_values, out_indices


# ---------------------------------------------------------------------------
batch = 64
n = 8192
k = 8


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        return _topk_cuda(x, self.k)


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
