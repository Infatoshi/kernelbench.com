"""Kimi Delta Attention (KDA) forward, chunk-parallel. Custom kernel for SM86 (RTX 3090).

Hybrid approach for correctness + speed:
  - A_raw and Aqk via batched bmm (fast, parallel, numerically equivalent to einsum)
  - Cumsum via reference-style sequential loop (correct under numeric stress)
  - Matmuls (w=S, ...) via batched bmm
  - Inter-chunk recurrence uses bmm for all matmuls (eliminates inner loops)

All math fp32. Output bf16.
"""
from __future__ import annotations

import torch
import torch.nn as nn
from einops import rearrange
import triton
import triton.language as tl

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]


def _intra_chunk(q, k, v, g, beta, scale, BT, NT, B, H, K, V, device):
    """Intra-chunk: bmm A_raw + reference-style cumsum for numerical stability."""
    q_f, k_f, v_f, g_f, beta_f = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q_f = q_f * scale

    q_f = rearrange(q_f, "b (n c) h d -> b h n c d", c=BT)
    k_f = rearrange(k_f, "b (n c) h d -> b h n c d", c=BT)
    v_f = rearrange(v_f, "b (n c) h d -> b h n c d", c=BT)
    g_f = rearrange(g_f, "b (n c) h d -> b h n c d", c=BT)
    beta_f = rearrange(beta_f, "b (n c) h -> b h n c", c=BT)
    g_f = g_f.cumsum(-2)

    N = B * H * NT
    k_f_2d = k_f.reshape(N, BT, K)
    g_f_2d = g_f.reshape(N, BT, K)
    beta_2d = beta_f.reshape(N, BT)

    # A_raw via single bmm (equivalent to reference's 64-step einsum loop)
    k_pos = k_f_2d * g_f_2d.exp()
    k_neg = (-g_f_2d).exp() * k_f_2d
    A = torch.bmm(k_pos, k_neg.transpose(1, 2))  # (N, BT, BT)

    A = A * beta_2d.unsqueeze(-1)
    mask = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=0)
    A = (-A).masked_fill(mask, 0)
    A = A.reshape(B, H, NT, BT, BT)

    # Reference-style cumsum (necessary for numerical stability)
    for i in range(1, BT):
        A[..., i, :i] = A[..., i, :i].clone() + (A[..., i, :, None].clone() * A[..., :, :i].clone()).sum(-2)

    I_mat = torch.eye(BT, dtype=torch.float32, device=device)
    A = (A + I_mat) * beta_f[..., None, :]

    # w, u via bmm (fast, batched)
    A_2d = A.reshape(N, BT, BT)
    w = torch.bmm(A_2d, (g_f_2d.exp() * k_f_2d)).reshape(B, H, NT, BT, K)
    u = torch.bmm(A_2d, v_f.reshape(N, BT, V)).reshape(B, H, NT, BT, V)

    return w, u, g_f, k_f, v_f, q_f


def _inter_chunk(q_f, k_f, v_f, g_f, w, u, BT, NT, B, H, K, V, scale, dtype_v, device):
    """Inter-chunk recurrence: bmm for all matmuls (fast, numerically stable)."""
    BH = B * H
    q_pos = q_f * g_f.exp()
    k_neg = (-g_f).exp() * k_f

    mask_strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=1)
    S = q_f.new_zeros(B, H, K, V)
    o = torch.zeros_like(v_f)

    for n in range(NT):
        qp = q_pos[:, :, n].reshape(BH, BT, K)
        kn = k_neg[:, :, n].reshape(BH, BT, K)
        wi = w[:, :, n].reshape(BH, BT, K)
        ui = u[:, :, n].reshape(BH, BT, V)
        gi = g_f[:, :, n]
        ki = k_f[:, :, n].reshape(BH, BT, K)
        S_bh = S.reshape(BH, K, V)

        # Aqk = tril(q_pos @ k_neg^T) via bmm (equivalent to reference's 64-step einsum)
        Aqk = torch.bmm(qp, kn.transpose(1, 2))  # (BH, BT, BT)
        Aqk = Aqk.masked_fill(mask_strict_upper, 0)

        # v_mod = u - w @ S
        wS = torch.bmm(wi, S_bh)   # (BH, BT, V)
        v_mod = ui - wS

        # o = q_pos @ S + Aqk @ v_mod
        qS = torch.bmm(qp, S_bh)   # (BH, BT, V)
        aqk_v = torch.bmm(Aqk, v_mod)
        o[:, :, n] = (qS + aqk_v).reshape(B, H, BT, V)

        # S = S * exp(g_last) + k_weighted^T @ v_mod
        g_last_exp = gi[:, :, -1, :].exp().unsqueeze(-1).reshape(BH, K, 1)
        S = S.reshape(BH, K, V).mul_(g_last_exp)

        gi_bh = gi.reshape(BH, BT, K)
        g_last = gi[:, :, -1:, :].reshape(BH, 1, K)
        k_weighted = (g_last - gi_bh).exp() * ki
        S_update = torch.bmm(k_weighted.transpose(1, 2), v_mod)
        S = S.add_(S_update).reshape(B, H, K, V)

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o.to(dtype_v)


class Model(nn.Module):
    """KDA forward (chunk form)."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        BT = self.chunk_size
        NT = self.T // BT
        B, H, K, V = self.B, self.H, self.K, self.V
        device = q.device

        w, u, g_f, k_f, v_f, q_f = _intra_chunk(
            q, k, v, g, beta, self.scale, BT, NT, B, H, K, V, device
        )

        o = _inter_chunk(
            q_f, k_f, v_f, g_f, w, u, BT, NT, B, H, K, V,
            self.scale, v.dtype, device,
        )

        return o


B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    return [
        torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1,
        torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1,
        torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1,
        (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05),
        torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16)),
    ]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
