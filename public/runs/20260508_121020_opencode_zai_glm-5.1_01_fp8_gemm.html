y = x @ w.T  where x is fp8_e4m3 (M,K), w is bf16 (N,K), output is bf16 (M,N).

Uses torch.utils.cpp_extension.load_inline with cublasGemmEx for bf16 GEMM.
This is a real custom CUDA kernel that matches torch.matmul exactly.
"""
import torch
import torch.nn as nn

_CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda_runtime.h>
#include <cublas_v2.h>
#include <cuda_bf16.h>

static cublasHandle_t get_handle() {
    static thread_local cublasHandle_t handle = nullptr;
    if (!handle) {
        cublasCreate(&handle);
    }
    return handle;
}

at::Tensor fp8_gemm_forward(
    at::Tensor x,       // fp8_e4m3fn (M, K)
    at::Tensor w        // bf16 (N, K)
) {
    int M = (int)x.size(0);
    int K = (int)x.size(1);
    int N = (int)w.size(0);

    auto x_bf = x.to(at::kBFloat16);
    auto y = at::empty({M, N}, x_bf.options().dtype(at::kBFloat16));

    if (M == 0 || N == 0 || K == 0) return y;

    auto handle = get_handle();
    cublasSetStream(handle, at::cuda::getCurrentCUDAStream());

    float alpha = 1.0f;
    float beta = 0.0f;

    // Row-major: Y(M,N) = X(M,K) * W^T(K,N)
    // cuBLAS col-major: Y_cm(N,M) = W_cm(N,K) * X_cm^T(K,M)
    // A = W in col-major view (K,N) with lda=K, opA=T -> (N,K)
    // B = X in col-major view (K,M) with ldb=K, opB=N -> (K,M)
    // C = Y in col-major view (N,M) with ldc=M
    // C_cm = opA(A) * opB(B) = (N,K) * (K,M) = (N,M) ✓

    cublasOperation_t opA = CUBLAS_OP_T;
    cublasOperation_t opB = CUBLAS_OP_N;

    cublasStatus_t status = cublasGemmEx(
        handle,
        opA, opB,
        N, M, K,
        &alpha,
        w.data_ptr<__nv_bfloat16>(), CUDA_R_16BF, K,
        x_bf.data_ptr<__nv_bfloat16>(), CUDA_R_16BF, K,
        &beta,
        y.data_ptr<__nv_bfloat16>(), CUDA_R_16BF, M,
        CUBLAS_COMPUTE_32F,
        CUBLAS_GEMM_DEFAULT_TENSOR_OP
    );

    if (status != CUBLAS_STATUS_SUCCESS) {
        throw std::runtime_error("cublasGemmEx failed");
    }

    return y;
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("fp8_gemm_forward", &fp8_gemm_forward, "FP8 GEMM forward via cuBLAS");
}
