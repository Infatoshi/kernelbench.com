"""Fused multi-axis RoPE pre-attention kernel for RTX PRO 6000 (Blackwell, sm_120).

Fuses: per-axis cos/sin cache gather + mrope_section mixing + rotate-half
rotation + (B,T,H,D)->(B,H,T,D) transpose into a single CUDA kernel.

Each thread handles one RoPE pair (d, d+D/2) and reads cos/sin ONLY from
the axis assigned to its slice by mrope_section — cutting cache reads by 3x.
One block per (b,t) pair. No shared memory, no syncthreads.
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "rope"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000"]

B = 1
T = 4096
Hq = 32
Hkv = 8
D = 128
MROPE_SECTION = (16, 24, 24)
MAX_POS = 32768


_cuda_src = r"""
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <stdint.h>

extern "C" __global__ void
mrope_fused_kernel(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const int64_t* __restrict__ pos_t,
    const int64_t* __restrict__ pos_h,
    const int64_t* __restrict__ pos_w,
    const __nv_bfloat16* __restrict__ cos_cache,
    const __nv_bfloat16* __restrict__ sin_cache,
    __nv_bfloat16* __restrict__ q_rot,
    __nv_bfloat16* __restrict__ k_rot,
    int B, int T, int Hq, int Hkv, int D,
    int s0, int s1, int s2
) {
    int blk = blockIdx.x;
    if (blk >= B * T) return;

    int b = blk / T;
    int t = blk % T;
    int d0 = threadIdx.x;
    int d1 = d0 + D / 2;

    int off_bt = b * T + t;
    unsigned int d0u = d0;
    unsigned int boundary1 = (unsigned int)s0;
    unsigned int boundary2 = (unsigned int)(s0 + s1);

    // Each thread reads cos/sin only from the axis that owns its d0 slice.
    // cos_val/sin_val for d0 also apply to d1 (mirrored by mrope mixing).
    float cos_val, sin_val;
    if (d0u < boundary1) {
        cos_val = __bfloat162float(cos_cache[pos_t[off_bt] * D + d0]);
        sin_val = __bfloat162float(sin_cache[pos_t[off_bt] * D + d0]);
    } else if (d0u < boundary2) {
        cos_val = __bfloat162float(cos_cache[pos_h[off_bt] * D + d0]);
        sin_val = __bfloat162float(sin_cache[pos_h[off_bt] * D + d0]);
    } else {
        cos_val = __bfloat162float(cos_cache[pos_w[off_bt] * D + d0]);
        sin_val = __bfloat162float(sin_cache[pos_w[off_bt] * D + d0]);
    }

    // --- strides ---
    int q_stride_H  = D;
    int q_stride_T  = Hq * D;
    int q_base      = b * T * q_stride_T + t * q_stride_T;

    int qo_stride_T = D;
    int qo_stride_H = T * D;
    int qo_stride_B = Hq * T * D;
    int qo_base     = b * qo_stride_B + t * qo_stride_T;

    int k_stride_H  = D;
    int k_stride_T  = Hkv * D;
    int k_base      = b * T * k_stride_T + t * k_stride_T;

    int ko_stride_T = D;
    int ko_stride_H = T * D;
    int ko_stride_B = Hkv * T * D;
    int ko_base     = b * ko_stride_B + t * ko_stride_T;

    // --- rotate q heads ---
    for (int h = 0; h < Hq; ++h) {
        int q_h_off  = q_base  + h * q_stride_H;
        int qo_h_off = qo_base + h * qo_stride_H;

        float q0 = __bfloat162float(q[q_h_off + d0]);
        float q1 = __bfloat162float(q[q_h_off + d1]);

        q_rot[qo_h_off + d0] = __float2bfloat16(q0 * cos_val - q1 * sin_val);
        q_rot[qo_h_off + d1] = __float2bfloat16(q1 * cos_val + q0 * sin_val);
    }

    // --- rotate k heads ---
    for (int h = 0; h < Hkv; ++h) {
        int k_h_off  = k_base  + h * k_stride_H;
        int ko_h_off = ko_base + h * ko_stride_H;

        float k0 = __bfloat162float(k[k_h_off + d0]);
        float k1 = __bfloat162float(k[k_h_off + d1]);

        k_rot[ko_h_off + d0] = __float2bfloat16(k0 * cos_val - k1 * sin_val);
        k_rot[ko_h_off + d1] = __float2bfloat16(k1 * cos_val + k0 * sin_val);
    }
}

torch::Tensor launch_mrope(torch::Tensor q, torch::Tensor k,
                           torch::Tensor pos_t, torch::Tensor pos_h, torch::Tensor pos_w,
                           torch::Tensor cos_cache, torch::Tensor sin_cache,
                           torch::Tensor q_rot, torch::Tensor k_rot,
                           int B, int T, int Hq, int Hkv, int D,
                           int s0, int s1, int s2) {
    int blocks = B * T;
    int threads = D / 2;
    mrope_fused_kernel<<<blocks, threads>>>(
        (const __nv_bfloat16*)q.data_ptr(),
        (const __nv_bfloat16*)k.data_ptr(),
        (const int64_t*)pos_t.data_ptr(),
        (const int64_t*)pos_h.data_ptr(),
        (const int64_t*)pos_w.data_ptr(),
        (const __nv_bfloat16*)cos_cache.data_ptr(),
        (const __nv_bfloat16*)sin_cache.data_ptr(),
        (__nv_bfloat16*)q_rot.data_ptr(),
        (__nv_bfloat16*)k_rot.data_ptr(),
        B, T, Hq, Hkv, D, s0, s1, s2
    );
    return q_rot;
}
"""

def _build_inv_freq(D: int, base: float = 10000.0) -> torch.Tensor:
    return 1.0 / (base ** (torch.arange(0, D, 2, dtype=torch.float32) / D))


class Model(nn.Module):
    def __init__(self, B: int, T: int, Hq: int, Hkv: int, D: int,
                 mrope_section: tuple[int, int, int], max_pos: int):
        super().__init__()
        assert sum(mrope_section) == D // 2, \
            f"mrope_section must sum to D/2, got {mrope_section} sum={sum(mrope_section)} D/2={D//2}"
        self.B, self.T = B, T
        self.Hq, self.Hkv, self.D = Hq, Hkv, D
        self.mrope_section = tuple(mrope_section)
        self.max_pos = max_pos

        inv_freq = _build_inv_freq(D)
        pos = torch.arange(max_pos, dtype=torch.float32)
        freqs = torch.outer(pos, inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        self.register_buffer("cos_cache", emb.cos().to(torch.bfloat16))
        self.register_buffer("sin_cache", emb.sin().to(torch.bfloat16))

        self._mrope_module = load_inline(
            name="mrope_fused",
            cpp_sources=_cpp_src,
            cuda_sources=_cuda_src,
            functions=["launch_mrope"],
            extra_cuda_cflags=["-arch=sm_120", "-O3", "--use_fast_math"],
            verbose=False,
        )

    def forward(
        self,
        q: torch.Tensor, k: torch.Tensor,
        pos_t: torch.Tensor, pos_h: torch.Tensor, pos_w: torch.Tensor,
    ) -> tuple[torch.Tensor, torch.Tensor]:
        B, T, Hq, D = q.shape
        Hkv = k.shape[2]
        s0, s1, s2 = self.mrope_section

        q_rot = torch.empty(B, Hq, T, D, dtype=torch.bfloat16, device=q.device)
        k_rot = torch.empty(B, Hkv, T, D, dtype=torch.bfloat16, device=k.device)

        self._mrope_module.launch_mrope(
            q, k, pos_t, pos_h, pos_w,
            self.cos_cache, self.sin_cache,
            q_rot, k_rot,
            B, T, Hq, Hkv, D, s0, s1, s2,
        )

        return q_rot, k_rot


def get_inputs():
    q = torch.randn(B, T, Hq, D, dtype=torch.bfloat16) * 0.5
    k = torch.randn(B, T, Hkv, D, dtype=torch.bfloat16) * 0.5

    pos_w = torch.randint(0, MAX_POS, (B, T), dtype=torch.int64)
    return [q, k, pos_t, pos_h, pos_w]


def get_init_inputs():
    return [B, T, Hq, Hkv, D, MROPE_SECTION, MAX_POS]
