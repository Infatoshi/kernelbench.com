"""Fused W4A16 weight-only quantized GEMM using Triton.

AWQ/GPTQ-style asymmetric int4 with explicit zero-points and per-group bf16
scales. The kernel fuses int4 unpack, dequant, and matrix multiply in a single
Triton kernel, avoiding the bandwidth waste of a separate dequant step.
"""

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

GROUP_SIZE = 128


def _prune_configs(configs, named_args, **kwargs):
    M_val = named_args.get("M", None)
    if M_val is None:
        return configs

    pruned = []
    for cfg in configs:
        bm = cfg.kwargs.get("BLOCK_M", 16)
        # Don't use BLOCK_M too large relative to M
        if bm > max(M_val * 4, 32):
            continue
        # For small M, don't use huge BLOCK_M (waste)
        if M_val <= 4 and bm > 16:
            continue
        pruned.append(cfg)
    return pruned if pruned else configs[:1]


@triton.autotune(
    configs=[
        # Small M configs
        triton.Config({"BLOCK_M": 1, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 1, "BLOCK_N": 256, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 1, "BLOCK_N": 512, "BLOCK_K": 128}, num_stages=4, num_warps=8),
        triton.Config({"BLOCK_M": 8, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 8, "BLOCK_N": 256, "BLOCK_K": 128}, num_stages=4, num_warps=8),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 64, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 16, "BLOCK_N": 256, "BLOCK_K": 128}, num_stages=3, num_warps=8),
        # Medium M configs
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 64, "BLOCK_K": 128}, num_stages=4, num_warps=4),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=4, num_warps=8),
        triton.Config({"BLOCK_M": 32, "BLOCK_N": 256, "BLOCK_K": 128}, num_stages=3, num_warps=8),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 64, "BLOCK_K": 128}, num_stages=4, num_warps=8),
        triton.Config({"BLOCK_M": 64, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=3, num_warps=8),
        # Large M configs
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 64, "BLOCK_K": 128}, num_stages=3, num_warps=8),
        triton.Config({"BLOCK_M": 128, "BLOCK_N": 128, "BLOCK_K": 128}, num_stages=3, num_warps=8),
        triton.Config({"BLOCK_M": 256, "BLOCK_N": 64, "BLOCK_K": 128}, num_stages=3, num_warps=8),
    ],
    key=["M", "N", "K"],
    prune_configs_by={
        "early_config_prune": _prune_configs,
        "perf_model": None,
        "top_k": None,
    },
)
@triton.jit
def _w4a16_gemm_kernel(
    x_ptr,
    wq_ptr,
    scales_ptr,
    zeros_ptr,
    y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wqkh, stride_wqn,
    stride_sg, stride_sn,
    stride_zg, stride_zn,
    stride_ym, stride_yn,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    """Fused W4A16 GEMM: unpack int4, dequant in bf16, multiply."""
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    x_ptrs = x_ptr + offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    HALF_K: tl.constexpr = BLOCK_K // 2
    K_GROUPS = K // BLOCK_K

    for kg in range(K_GROUPS):
        k_start = kg * BLOCK_K

        x = tl.load(
            x_ptrs + k_start * stride_xk,
            mask=(offs_m[:, None] < M) & (offs_k[None, :] + k_start < K),
            other=0.0,
        )

        offs_hk = tl.arange(0, HALF_K)
        wq_packed_ptrs = (
            wq_ptr
            + offs_hk[:, None] * stride_wqkh
            + offs_n[None, :] * stride_wqn
            + (k_start // 2) * stride_wqkh
        )
        wq_mask = (offs_n[None, :] < N) & ((k_start // 2 + offs_hk[:, None]) < (K // 2))
        wq_packed = tl.load(wq_packed_ptrs, mask=wq_mask, other=0)

        wq_lo = wq_packed & 0x0F
        wq_hi = (wq_packed >> 4) & 0x0F

        wq_lo_t = tl.trans(wq_lo)
        wq_hi_t = tl.trans(wq_hi)
        wq_t = tl.interleave(wq_lo_t, wq_hi_t)
        wq = tl.trans(wq_t)

        g_idx = k_start // 128
        s_ptrs = scales_ptr + g_idx * stride_sg + offs_n[None, :] * stride_sn
        z_ptrs = zeros_ptr + g_idx * stride_zg + offs_n[None, :] * stride_zn
        scale = tl.load(s_ptrs, mask=offs_n[None, :] < N, other=0.0)
        zero = tl.load(z_ptrs, mask=offs_n[None, :] < N, other=0.0)

        wq_bf16 = wq.to(tl.bfloat16)
        zero_bf16 = zero.to(tl.bfloat16)
        scale_bf16 = scale.to(tl.bfloat16)
        w_deq = (wq_bf16 - zero_bf16) * scale_bf16

        x_bf16 = x.to(tl.bfloat16)
        acc += tl.dot(x_bf16, w_deq, input_precision="ieee")

    y_ptrs = y_ptr + offs_m[:, None] * stride_ym + offs_n[None, :] * stride_yn
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=(offs_m[:, None] < M) & (offs_n[None, :] < N))


class Model(nn.Module):
    """W4A16 GEMM: y = x @ dequant(w_q, scales, zeros)."""

    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0, "K must be even (int4 packing)"
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size

        self.register_buffer("w_q", torch.empty(K // 2, N, dtype=torch.uint8))
        self.register_buffer("scales", torch.empty(n_groups, N, dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.empty(n_groups, N, dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        M_actual = x.shape[0]
        assert x.shape[1] == self.K
        assert x.dtype == torch.bfloat16

        y = torch.empty(M_actual, self.N, dtype=torch.bfloat16, device=x.device)

        grid = lambda meta: (
            triton.cdiv(M_actual, meta["BLOCK_M"]) * triton.cdiv(self.N, meta["BLOCK_N"]),
        )

        _w4a16_gemm_kernel[grid](
            x, self.w_q, self.scales, self.zeros, y,
            M_actual, self.N, self.K,
            x.stride(0), x.stride(1),
            self.w_q.stride(0), self.w_q.stride(1),
            self.scales.stride(0), self.scales.stride(1),
            self.zeros.stride(0), self.zeros.stride(1),
            y.stride(0), y.stride(1),
        )

        return y
