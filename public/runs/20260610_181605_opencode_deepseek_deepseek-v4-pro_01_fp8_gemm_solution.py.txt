"""FP8 e4m3 GEMM for RTX PRO 6000 (Blackwell SM120).

Custom CUDA C++ kernel using Warp Matrix Multiply-Accumulate (WMMA)
to match torch.matmul numerical behavior exactly.
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

OP_TYPE = "gemm"
SUPPORTED_PRECISIONS = ["fp8_e4m3"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


_cuda_src = r"""
#include <cuda_bf16.h>
#include <mma.h>

#define WARP_SIZE 32

// WMMA tile dimensions for bf16
#define WMMA_M 16
#define WMMA_N 16
#define WMMA_K 16

__global__ void gemm_kernel(
    const __nv_bfloat16* __restrict__ A,
    const __nv_bfloat16* __restrict__ B,
    __nv_bfloat16* __restrict__ C,
    int M, int N, int K,
    int lda, int ldb, int ldc)
{
    // Each thread block computes one 16x16 output element
    // (simplest approach: one WMMA tile per block)
    int row = blockIdx.y * 16;
    int col = blockIdx.x * 16;

    if (row >= M || col >= N) return;

    // Accumulator
    float acc[8] = {0.0f};

    for (int k = 0; k < K; k += 16) {
        // Load A fragment from global memory
        // A is MxK, row-major. We need the 16x16 tile at (row, k)
        nvcuda::wmma::fragment<nvcuda::wmma::matrix_a, WMMA_M, WMMA_N, WMMA_K,
                                __nv_bfloat16, nvcuda::wmma::row_major> a_frag;
        nvcuda::wmma::fragment<nvcuda::wmma::matrix_b, WMMA_M, WMMA_N, WMMA_K,
                                __nv_bfloat16, nvcuda::wmma::col_major> b_frag;
        nvcuda::wmma::fragment<nvcuda::wmma::accumulator, WMMA_M, WMMA_N, WMMA_K,
                                float> acc_frag;

        // Load accumulator values
        for (int i = 0; i < 8; i++)
            acc_frag.x[i] = acc[i];

        // B is NxK, stored row-major. We need B[col:col+16, k:k+16] in col-major.
        // Since B is (N,K) row-major and we need B[K]x[N] for the MMA,
        // we load B as NxK but interpret as col-major (B^T).
        const __nv_bfloat16* a_tile = A + row * lda + k;
        const __nv_bfloat16* b_tile = B + col * ldb + k;

        // Handle edge cases: pad with zeros if out of bounds
        bool a_edge = (row + 16 > M) || (k + 16 > K);
        bool b_edge = (col + 16 > N) || (k + 16 > K);

        if (!a_edge && !b_edge) {
            nvcuda::wmma::load_matrix_sync(a_frag, a_tile, lda);
            nvcuda::wmma::load_matrix_sync(b_frag, b_tile, ldb);
        } else {
            // Edge handling: load with padding
            __nv_bfloat16 a_pad[16 * 16] = {};
            __nv_bfloat16 b_pad[16 * 16] = {};
            for (int r = 0; r < 16 && row + r < M; r++)
                for (int c = 0; c < 16 && k + c < K; c++)
                    a_pad[r * 16 + c] = A[(row + r) * lda + (k + c)];
            for (int r = 0; r < 16 && col + r < N; r++)
                for (int c = 0; c < 16 && k + c < K; c++)
                    b_pad[r * 16 + c] = B[(col + r) * ldb + (k + c)];
            nvcuda::wmma::load_matrix_sync(a_frag, a_pad, 16);
            nvcuda::wmma::load_matrix_sync(b_frag, b_pad, 16);
        }

        nvcuda::wmma::mma_sync(acc_frag, a_frag, b_frag, acc_frag);

        for (int i = 0; i < 8; i++)
            acc[i] = acc_frag.x[i];
    }

    // Store result (convert from float to bf16)
    float c_tmp[16 * 16];
    nvcuda::wmma::fragment<nvcuda::wmma::accumulator, WMMA_M, WMMA_N, WMMA_K,
                            float> acc_frag;
    for (int i = 0; i < 8; i++)
        acc_frag.x[i] = acc[i];

    // store_matrix_sync for float accumulator with 16x16x16 expects float* output
    nvcuda::wmma::store_matrix_sync(c_tmp, acc_frag, 16, nvcuda::wmma::mem_row_major);

    for (int r = 0; r < 16 && row + r < M; r++) {
        for (int c = 0; c < 16 && col + c < N; c++) {
            C[(row + r) * ldc + (col + c)] = __float2bfloat16(c_tmp[r * 16 + c]);
        }
    }
}


torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int N = B.size(0);
    int K = A.size(1);

    auto C = torch::empty({M, N}, A.options().dtype(torch::kBFloat16));

    dim3 grid((N + 15) / 16, (M + 15) / 16);
    dim3 block(32);  // one warp per block

    gemm_kernel<<<grid, block>>>(
        reinterpret_cast<__nv_bfloat16*>(A.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(B.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(C.data_ptr()),
        M, N, K,
        A.stride(0), B.stride(0), C.stride(0)
    );

    return C;
}
"""

_cuda_fn = load_inline(
    name="fp8_gemm_cuda",
    cpp_sources="torch::Tensor gemm_forward(torch::Tensor A, torch::Tensor B);",
    cuda_sources=_cuda_src,
    functions=["gemm_forward"],
    extra_cuda_cflags=["-arch=sm_120a", "-std=c++17"],
    verbose=False,
)


class Model(nn.Module):
    """y = (x @ w.T).to(bf16), where x is fp8_e4m3 (M, K), w is fp8_e4m3 (N, K)."""

    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        M, N, K = self.M, self.N, self.K
        x_bf = x.to(torch.bfloat16).contiguous()
        w_bf = self.weight.to(torch.bfloat16).contiguous()
        return _cuda_fn.gemm_forward(x_bf, w_bf)


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
