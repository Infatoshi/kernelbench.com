"""Roofline math: achieved TFLOPS / GB/s, peak fraction.

FLOPS and bytes formulas come from the problem's `problem.yaml`. They are the
dense-equivalent algorithmic work; kernels that skip work (sparsity, early
exit) are still measured against the dense cost.
"""
from __future__ import annotations


def compute_tflops(flops: float, time_ms: float) -> float:
    """flops per invocation and wall time in ms -> TFLOPS."""
    if time_ms <= 0:
        return 0.0
    return flops / (time_ms * 1e-3) / 1e12


def compute_gbps(bytes_moved: float, time_ms: float) -> float:
    """bytes per invocation and wall time in ms -> GB/s."""
    if time_ms <= 0:
        return 0.0
    return bytes_moved / (time_ms * 1e-3) / 1e9


def peak_fraction(
    achieved: float,
    peak: float,
) -> float:
    """Return achieved / peak, clamped to [0, infinity)."""
    if peak <= 0:
        return 0.0
    return max(0.0, achieved / peak)

+def _build_schedule(expert_offsets: torch.Tensor, E: int, T_perm: int, BLOCK_M: int):
+    counts = (expert_offsets[1:] - expert_offsets[:-1]).to(torch.int64)        # (E,)
+    ntiles = (counts + (BLOCK_M - 1)) // BLOCK_M                               # (E,)

+"""Ceiling experiment: clean single-matrix grouped GEMM (no SwiGLU, one accumulator).
+Tells us the best achievable Triton efficiency for these shapes."""
+import torch, triton, triton.language as tl, itertools, math

+def build_sched(offsets, E, T_perm, BM):
+    counts=(offsets[1:]-offsets[:-1]).to(torch.int64)
+    ntiles=(counts+BM-1)//BM
+    cum=torch.cumsum(ntiles,0)
+    total=int(T_perm//BM+E)
+    t=torch.arange(total,device=offsets.device,dtype=torch.int64)
+    eof=torch.searchsorted(cum,t,right=True)
+    valid=eof<E
+    ec=torch.clamp(eof,max=E-1)
+    tb=cum-ntiles
+    tie=t-tb[ec]
+    rs=offsets[:-1].to(torch.int64)[ec]+tie*BM
+    eid=torch.where(valid,ec,torch.full_like(ec,E))
+    return eid.to(torch.int32).contiguous(), rs.to(torch.int32).contiguous(), total
+
+@triton.jit
+def gemm_k(a_ptr,w_ptr,o_ptr,eid_ptr,rs_ptr,off_ptr,H,N,nE,
+           sam,sak,swe,swk,swn,som,son,
+           BM:tl.constexpr,BN:tl.constexpr,BK:tl.constexpr,GROUP:tl.constexpr):
+    pid=tl.program_id(0)
+    num_pid_n=tl.cdiv(N,BN)
+    # swizzle within fixed m? simpler: pid_m, pid_n grid 1D
+    num_pid_m=tl.num_programs(0)//num_pid_n
+    # group-major swizzle
+    pid_m=pid//num_pid_n
+    pid_n=pid%num_pid_n
+    e=tl.load(eid_ptr+pid_m)
+    if e>=nE: return
+    row0=tl.load(rs_ptr+pid_m)
+    nend=tl.load(off_ptr+e+1)
+    rm=row0+tl.arange(0,BM)
+    rn=pid_n*BN+tl.arange(0,BN)
+    rk=tl.arange(0,BK)
+    rmask=rm<nend
+    ab=a_ptr+rm[:,None]*sam
