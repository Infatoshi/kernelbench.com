+"""Paged-attention decode kernel (Triton flash-decoding) for SM120 Blackwell.
+
+Single-query decode. Memory-bound: the KV cache must be streamed exactly once,
+reused across each GQA group. Small-batch shapes are parallelized with split-K
+(flash-decoding) to fill all 188 SMs, then a cheap combine kernel reduces the
+per-split partial softmaxes.
+"""
+import math
+
+import torch
+import torch.nn as nn
+import triton
+import triton.language as tl
+
+
+@triton.jit
+def _paged_decode_phase1(
+    q_ptr,            # (B, H, D) bf16
+    kv_ptr,           # (num_blocks, P, Hkv, 2D) bf16
+    bt_ptr,           # (B, max_blocks) int32
+    sl_ptr,           # (B,) int32
+    o_ptr,            # (B, H, S, D) fp32   partial unnormalized output
+    m_ptr,            # (B, H, S) fp32      partial max
+    l_ptr,            # (B, H, S) fp32      partial sum
+    scale,
+    H: tl.constexpr, Hkv: tl.constexpr, G: tl.constexpr, D: tl.constexpr,
+    P: tl.constexpr, max_blocks: tl.constexpr,
+    S: tl.constexpr,                 # num_splits
+    BLOCK_N: tl.constexpr,           # tokens per inner iteration
+    GP: tl.constexpr,                # padded group (>=16 for tl.dot)
+    PAGES_PER_BLOCK: tl.constexpr,
+):
+    pid = tl.program_id(0)
+    split = pid % S
+    bkv = pid // S
+    b = bkv // Hkv
+    kvh = bkv % Hkv
+
+    L = tl.load(sl_ptr + b)
+    num_pages = (L + P - 1) // P
+    pages_per_split = (num_pages + S - 1) // S
+    page_start = split * pages_per_split
+    page_end = tl.minimum(page_start + pages_per_split, num_pages)
+
+    # Load Q for the group: (GP, D), rows >= G are zero-padded.
+    grow = tl.arange(0, GP)
+    dcol = tl.arange(0, D)
+    q_head = kvh * G + grow
+    q_off = b * (H * D) + q_head[:, None] * D + dcol[None, :]
+    q_mask = (grow[:, None] < G)
+    q = tl.load(q_ptr + q_off, mask=q_mask, other=0.0).to(tl.float32)
+
+    m_i = tl.full((GP,), -float("inf"), dtype=tl.float32)
+    l_i = tl.zeros((GP,), dtype=tl.float32)
+    acc = tl.zeros((GP, D), dtype=tl.float32)
+
+    row = tl.arange(0, BLOCK_N)
+    local_page = row // P
+    within = row % P
+
+    kv2d = 2 * D
+    page_stride = P * Hkv * kv2d
+
+    for pblk in range(page_start, page_end, PAGES_PER_BLOCK):
+        gpage = pblk + local_page                       # (BLOCK_N,)
+        valid_page = gpage < page_end
+        page_id = tl.load(bt_ptr + b * max_blocks + gpage, mask=valid_page, other=0)
+        token = gpage * P + within
+        valid = token < L
+
+        base = page_id * page_stride + within * (Hkv * kv2d) + kvh * kv2d
+        k_off = base[:, None] + dcol[None, :]
+        v_off = base[:, None] + (D + dcol[None, :])
+        k = tl.load(kv_ptr + k_off, mask=valid[:, None], other=0.0).to(tl.float32)
+        v = tl.load(kv_ptr + v_off, mask=valid[:, None], other=0.0).to(tl.float32)
+
+        # scores: (GP, BLOCK_N)
+        s = tl.dot(q, tl.trans(k)) * scale
+        s = tl.where(valid[None, :], s, -float("inf"))
+
+        m_new = tl.maximum(m_i, tl.max(s, axis=1))
+        alpha = tl.exp(m_i - m_new)
+        p = tl.exp(s - m_new[:, None])
+        l_i = l_i * alpha + tl.sum(p, axis=1)
+        acc = acc * alpha[:, None] + tl.dot(p.to(v.dtype), v)
+        m_i = m_new
+
+    # Store partials: (B, H, S, D)
+    out_head = kvh * G + grow
+    valid_g = grow < G
+    o_base = b * (H * S * D) + out_head[:, None] * (S * D) + split * D + dcol[None, :]
+    tl.store(o_ptr + o_base, acc, mask=valid_g[:, None])
+    ml_off = b * (H * S) + out_head * S + split
+    tl.store(m_ptr + ml_off, m_i, mask=valid_g)
+    tl.store(l_ptr + ml_off, l_i, mask=valid_g)
+

+@triton.jit
+def _paged_decode_combine(
+    o_ptr,            # (B, H, S, D) fp32
+    m_ptr,            # (B, H, S) fp32
+    l_ptr,            # (B, H, S) fp32
+    out_ptr,          # (B, H, D) bf16
+    H: tl.constexpr, D: tl.constexpr, S: tl.constexpr,
+):
+    pid = tl.program_id(0)          # over B*H
+    b = pid // H
+    h = pid % H
+
+    sidx = tl.arange(0, S)
+    dcol = tl.arange(0, D)
+
+    m = tl.load(m_ptr + (b * H + h) * S + sidx)        # (S,)
+    l = tl.load(l_ptr + (b * H + h) * S + sidx)
+    m_g = tl.max(m, axis=0)
+    scale = tl.exp(m - m_g)
+    l_g = tl.sum(l * scale, axis=0)
+
+    o = tl.load(o_ptr + (b * H + h) * (S * D) + sidx[:, None] * D + dcol[None, :])  # (S, D)
+    o = o * scale[:, None]
+    out = tl.sum(o, axis=0) / l_g
+    tl.store(out_ptr + (b * H + h) * D + dcol, out.to(tl.bfloat16))

+def _pick_num_splits(batch, num_kv_heads, num_pages):
+    groups = batch * num_kv_heads
+    target = 2048
+    s = max(1, target // groups)
+    s = min(s, num_pages)
+    return s
