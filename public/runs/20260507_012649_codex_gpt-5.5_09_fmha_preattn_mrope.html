def _build_inv_freq(D: int, base: float = 10000.0) -> torch.Tensor:
    return 1.0 / (base ** (torch.arange(0, D, 2, dtype=torch.float32) / D))

def _mrope_qk_kernel(
    q,
    k,
    pos_t,
    pos_h,
    pos_w,
    cos_cache,
    sin_cache,
    q_out,
    k_out,
    T: tl.constexpr,
    Hq: tl.constexpr,
    Hkv: tl.constexpr,
    D: tl.constexpr,
    S0: tl.constexpr,
    S1: tl.constexpr,
    BLOCK_T: tl.constexpr,
    BLOCK_H: tl.constexpr,
    HALF: tl.constexpr,
):
    pid_t = tl.program_id(0)
    pid_h = tl.program_id(1)
    b = tl.program_id(2)

    offs_t = pid_t * BLOCK_T + tl.arange(0, BLOCK_T)
    offs_h = pid_h * BLOCK_H + tl.arange(0, BLOCK_H)
    offs_p = tl.arange(0, HALF)
    t_mask = offs_t < T
    hq_mask = offs_h < Hq
    cache_mask = t_mask[:, None]

    pos_base = b * T + offs_t
    pt = tl.load(pos_t + pos_base, mask=t_mask, other=0)
    ph = tl.load(pos_h + pos_base, mask=t_mask, other=0)
    pw = tl.load(pos_w + pos_base, mask=t_mask, other=0)

    pair = offs_p[None, :]
    axis_pos = tl.where(pair < S0, pt[:, None], tl.where(pair < S0 + S1, ph[:, None], pw[:, None]))
    cache_off = axis_pos * D + pair
    c = tl.load(cos_cache + cache_off, mask=cache_mask, other=0.0)
    s = tl.load(sin_cache + cache_off, mask=cache_mask, other=0.0)

    q_mask = t_mask[:, None, None] & hq_mask[None, :, None]
    q_in0 = ((b * T + offs_t[:, None, None]) * Hq + offs_h[None, :, None]) * D + offs_p[None, None, :]
    q0 = tl.load(q + q_in0, mask=q_mask, other=0.0)
    q1 = tl.load(q + q_in0 + HALF, mask=q_mask, other=0.0)
    q_out0 = q0 * c[:, None, :] - q1 * s[:, None, :]
    q_out1 = q1 * c[:, None, :] + q0 * s[:, None, :]

    q_dst0 = ((b * Hq + offs_h[None, :, None]) * T + offs_t[:, None, None]) * D + offs_p[None, None, :]
    tl.store(q_out + q_dst0, q_out0, mask=q_mask)
    tl.store(q_out + q_dst0 + HALF, q_out1, mask=q_mask)

    hkv_mask = offs_h < Hkv
    k_mask = t_mask[:, None, None] & hkv_mask[None, :, None]
    k_in0 = ((b * T + offs_t[:, None, None]) * Hkv + offs_h[None, :, None]) * D + offs_p[None, None, :]
    k0 = tl.load(k + k_in0, mask=k_mask, other=0.0)
    k1 = tl.load(k + k_in0 + HALF, mask=k_mask, other=0.0)
    k_out0 = k0 * c[:, None, :] - k1 * s[:, None, :]
    k_out1 = k1 * c[:, None, :] + k0 * s[:, None, :]

    k_dst0 = ((b * Hkv + offs_h[None, :, None]) * T + offs_t[:, None, None]) * D + offs_p[None, None, :]
    tl.store(k_out + k_dst0, k_out0, mask=k_mask)
    tl.store(k_out + k_dst0 + HALF, k_out1, mask=k_mask)
