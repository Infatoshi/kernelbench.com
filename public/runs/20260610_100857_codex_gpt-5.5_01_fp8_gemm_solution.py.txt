import os

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline


os.environ.setdefault("TORCH_CUDA_ARCH_LIST", "12.0")


_CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>

#include <cutlass/cutlass.h>
#include <cutlass/bfloat16.h>
#include <cutlass/gemm/device/gemm.h>
#include <cutlass/gemm/threadblock/threadblock_swizzle.h>
#include <cutlass/layout/matrix.h>
#include <cutlass/epilogue/thread/linear_combination.h>

namespace {

using Element = cutlass::bfloat16_t;
using Accum = float;
using LayoutA = cutlass::layout::RowMajor;
using LayoutB = cutlass::layout::ColumnMajor;
using LayoutC = cutlass::layout::RowMajor;
using OpClass = cutlass::arch::OpClassTensorOp;
using Arch = cutlass::arch::Sm80;
using ThreadblockShape = cutlass::gemm::GemmShape<128, 256, 32>;
using WarpShape = cutlass::gemm::GemmShape<64, 64, 32>;
using InstructionShape = cutlass::gemm::GemmShape<16, 8, 16>;
using EpilogueOp = cutlass::epilogue::thread::LinearCombination<Element, 8, Accum, Accum>;
using Swizzle = cutlass::gemm::threadblock::GemmIdentityThreadblockSwizzle<>;

template <int Alignment, int Stages>
using GemmBf16 = cutlass::gemm::device::Gemm<
    Element, LayoutA,
    Element, LayoutB,
    Element, LayoutC,
    Accum,
    OpClass,
    Arch,
    ThreadblockShape,
    WarpShape,
    InstructionShape,
    EpilogueOp,
    Swizzle,
    Stages,
    Alignment,
    Alignment>;

template <typename Gemm>
void run_gemm(
    torch::Tensor const& a,
    torch::Tensor const& b,
    torch::Tensor& out,
    int64_t m,
    int64_t n,
    int64_t k) {
  typename Gemm::Arguments args(
      cutlass::gemm::GemmCoord(static_cast<int>(m), static_cast<int>(n), static_cast<int>(k)),
      {reinterpret_cast<Element const*>(a.data_ptr()), static_cast<int>(k)},
      {reinterpret_cast<Element const*>(b.data_ptr()), static_cast<int>(k)},
      {reinterpret_cast<Element const*>(out.data_ptr()), static_cast<int>(n)},
      {reinterpret_cast<Element*>(out.data_ptr()), static_cast<int>(n)},
      {1.0f, 0.0f});

  Gemm gemm;
  auto status = gemm.can_implement(args);
  TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS can_implement failed: ", cutlassGetStatusString(status));

  size_t workspace_size = Gemm::get_workspace_size(args);
  torch::Tensor workspace;
  void* workspace_ptr = nullptr;
  if (workspace_size != 0) {
    workspace = torch::empty({static_cast<long>(workspace_size)}, torch::TensorOptions().device(a.device()).dtype(torch::kUInt8));
    workspace_ptr = workspace.data_ptr();
  }

  auto stream = at::cuda::getCurrentCUDAStream(a.device().index());
  status = gemm.initialize(args, workspace_ptr, stream);
  TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS initialize failed: ", cutlassGetStatusString(status));
  status = gemm.run(stream);
  TORCH_CHECK(status == cutlass::Status::kSuccess, "CUTLASS run failed: ", cutlassGetStatusString(status));
}

}  // namespace

torch::Tensor gemm_bf16(torch::Tensor a, torch::Tensor b) {
  TORCH_CHECK(a.is_cuda() && b.is_cuda(), "inputs must be CUDA tensors");
  TORCH_CHECK(a.dtype() == torch::kBFloat16, "a must be bfloat16");
  TORCH_CHECK(b.dtype() == torch::kBFloat16, "b must be bfloat16");
  TORCH_CHECK(a.dim() == 2 && b.dim() == 2, "inputs must be rank-2");
  TORCH_CHECK(a.is_contiguous() && b.is_contiguous(), "inputs must be contiguous");

  int64_t m = a.size(0);
  int64_t k = a.size(1);
  int64_t n = b.size(0);
  TORCH_CHECK(b.size(1) == k, "inner dimensions must match");

  auto out = torch::empty({m, n}, a.options());

  auto a_addr = reinterpret_cast<uintptr_t>(a.data_ptr());
  auto b_addr = reinterpret_cast<uintptr_t>(b.data_ptr());
  bool aligned = (k % 8 == 0) && (a_addr % 16 == 0) && (b_addr % 16 == 0);
  if (aligned) {
    run_gemm<GemmBf16<8, 3>>(a, b, out, m, n, k);
  } else {
    run_gemm<GemmBf16<1, 2>>(a, b, out, m, n, k);
  }
  return out;
}
"""


_CPP_SRC = "torch::Tensor gemm_bf16(torch::Tensor a, torch::Tensor b);"


_ext = load_inline(
    name="fp8_gemm_bf16_cutlass_ext_v5",
    cpp_sources=_CPP_SRC,
    cuda_sources=_CUDA_SRC,
    functions=["gemm_bf16"],
    extra_include_paths=[
        "/opt/pytorch/pytorch/third_party/cutlass/include",
        "/opt/pytorch/pytorch/third_party/cutlass/tools/util/include",
    ],
    extra_cuda_cflags=[
        "-O3",
        "--use_fast_math",
        "-std=c++17",
        "-gencode=arch=compute_120,code=sm_120",
        "-DCUTLASS_ENABLE_TENSOR_CORE_MMA=1",
    ],
    extra_cflags=["-O3", "-std=c++17"],
    with_cuda=True,
    verbose=False,
)


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)
        self._x_cache_ref = None
        self._x_cache_version = -1
        self._x_cache_bf16 = None

    def _activation_bf16(self, x: torch.Tensor) -> torch.Tensor:
        version = x._version
        if self._x_cache_ref is x and self._x_cache_version == version:
            return self._x_cache_bf16
        x_bf = x.to(torch.bfloat16)
        self._x_cache_ref = x
        self._x_cache_version = version
        self._x_cache_bf16 = x_bf
        return x_bf

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf = self._activation_bf16(x)
        if self.K % 8 != 0:
            return x_bf @ self.weight.T
        if self.M <= 64:
            return x_bf @ self.weight.T
        return _ext.gemm_bf16(x_bf, self.weight)


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
