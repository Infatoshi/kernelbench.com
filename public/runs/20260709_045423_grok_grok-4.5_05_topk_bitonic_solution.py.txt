"""High-performance top-k for B200 (custom CUDA).

largest=True, sorted=True over the last dim of a 2D fp32 tensor.
No ATen selection/sort ops.

Paths:
  k == 1 : vectorized multi-warp argmax
  else   : CUB BlockRadixSort over tiles (partial top-k per tile),
           then CUB BlockRadixSort merge of candidates.
           Single-tile rows write the final result directly.
"""
from __future__ import annotations

import os
from pathlib import Path

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cub/block/block_radix_sort.cuh>
#include <cuda_runtime.h>
#include <float.h>
#include <cstdint>
#include <mutex>
#include <type_traits>

// =====================================================================
// k = 1 : argmax
// =====================================================================
template <int BLOCK>
__global__ void argmax_kernel(
    const float* __restrict__ x,
    float* __restrict__ values,
    int64_t* __restrict__ indices,
    int n, int batch)
{
  const int row = blockIdx.x;
  if (row >= batch) return;
  const float* row_x = x + (size_t)row * n;

  float best_v = -FLT_MAX;
  int   best_i = INT_MAX;

  const int n4 = n >> 2;
  const float4* row4 = reinterpret_cast<const float4*>(row_x);
  for (int j = threadIdx.x; j < n4; j += BLOCK) {
    float4 v4 = __ldg(&row4[j]);
    int base = j << 2;
    if (v4.x > best_v || (v4.x == best_v && base     < best_i)) { best_v = v4.x; best_i = base; }
    if (v4.y > best_v || (v4.y == best_v && base + 1 < best_i)) { best_v = v4.y; best_i = base + 1; }
    if (v4.z > best_v || (v4.z == best_v && base + 2 < best_i)) { best_v = v4.z; best_i = base + 2; }
    if (v4.w > best_v || (v4.w == best_v && base + 3 < best_i)) { best_v = v4.w; best_i = base + 3; }
  }
  for (int j = (n4 << 2) + threadIdx.x; j < n; j += BLOCK) {
    float v = __ldg(&row_x[j]);
    if (v > best_v || (v == best_v && j < best_i)) { best_v = v; best_i = j; }
  }

  const unsigned mask = 0xffffffffu;
  #pragma unroll
  for (int off = 16; off > 0; off >>= 1) {
    float ov = __shfl_down_sync(mask, best_v, off);
    int   oi = __shfl_down_sync(mask, best_i, off);
    if (ov > best_v || (ov == best_v && oi < best_i)) { best_v = ov; best_i = oi; }
  }

  constexpr int NW = BLOCK / 32;
  __shared__ float wb_v[NW];
  __shared__ int   wb_i[NW];
  int lane = threadIdx.x & 31;
  int wid  = threadIdx.x >> 5;
  if (lane == 0) { wb_v[wid] = best_v; wb_i[wid] = best_i; }
  __syncthreads();

  if (wid == 0) {
    best_v = (lane < NW) ? wb_v[lane] : -FLT_MAX;
    best_i = (lane < NW) ? wb_i[lane] : INT_MAX;
    #pragma unroll
    for (int off = 16; off > 0; off >>= 1) {
      float ov = __shfl_down_sync(mask, best_v, off);
      int   oi = __shfl_down_sync(mask, best_i, off);
      if (ov > best_v || (ov == best_v && oi < best_i)) { best_v = ov; best_i = oi; }
    }
    if (lane == 0) {
      values[row]  = best_v;
      indices[row] = (int64_t)best_i;
    }
  }
}

// =====================================================================
// CUB BlockRadixSort tile kernels
// Arrangement: thread t holds elements t, t+BLOCK, ... (blocked).
// SortDescendingBlockedToStriped → rank r at thread (r%BLOCK), item (r/BLOCK).
// =====================================================================
template <int BLOCK, int ITEMS, int K>
__global__ void partial_topk_kernel(
    const float* __restrict__ x,
    float* __restrict__ partial_vals,
    int* __restrict__ partial_idxs,
    int n, int batch, int num_blocks_per_row)
{
  constexpr int TILE = BLOCK * ITEMS;
  using SortT = cub::BlockRadixSort<float, BLOCK, ITEMS, int>;
  __shared__ typename SortT::TempStorage sort_tmp;

  const int row = blockIdx.y;
  const int b   = blockIdx.x;
  if (row >= batch || b >= num_blocks_per_row) return;

  const int start = b * TILE;
  float keys[ITEMS];
  int   idxs[ITEMS];

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int idx = start + threadIdx.x + i * BLOCK;
    if (idx < n) {
      keys[i] = __ldg(x + (size_t)row * n + idx);
      idxs[i] = idx;
    } else {
      keys[i] = -FLT_MAX;
      idxs[i] = INT_MAX;
    }
  }

  SortT(sort_tmp).SortDescendingBlockedToStriped(keys, idxs);

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int rank = threadIdx.x + i * BLOCK;
    if (rank < K) {
      size_t off = ((size_t)row * num_blocks_per_row + b) * K + rank;
      partial_vals[off] = keys[i];
      partial_idxs[off] = idxs[i];
    }
  }
}

template <int BLOCK, int ITEMS, int K>
__global__ void single_tile_topk_kernel(
    const float* __restrict__ x,
    float* __restrict__ values,
    int64_t* __restrict__ indices,
    int n, int batch)
{
  using SortT = cub::BlockRadixSort<float, BLOCK, ITEMS, int>;
  __shared__ typename SortT::TempStorage sort_tmp;

  const int row = blockIdx.x;
  if (row >= batch) return;

  float keys[ITEMS];
  int   idxs[ITEMS];

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int idx = threadIdx.x + i * BLOCK;
    if (idx < n) {
      keys[i] = __ldg(x + (size_t)row * n + idx);
      idxs[i] = idx;
    } else {
      keys[i] = -FLT_MAX;
      idxs[i] = INT_MAX;
    }
  }

  SortT(sort_tmp).SortDescendingBlockedToStriped(keys, idxs);

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int rank = threadIdx.x + i * BLOCK;
    if (rank < K) {
      values[(size_t)row * K + rank]  = keys[i];
      indices[(size_t)row * K + rank] = (int64_t)idxs[i];
    }
  }
}

// n_cand <= BLOCK * ITEMS
template <int BLOCK, int ITEMS>
__global__ void merge_radix_kernel(
    const float* __restrict__ in_vals,
    const int* __restrict__ in_idxs,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int n_cand, int k)
{
  using SortT = cub::BlockRadixSort<float, BLOCK, ITEMS, int>;
  __shared__ typename SortT::TempStorage sort_tmp;

  const int row = blockIdx.x;
  if (row >= batch) return;

  float keys[ITEMS];
  int   idxs[ITEMS];

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int idx = threadIdx.x + i * BLOCK;
    if (idx < n_cand) {
      keys[i] = in_vals[(size_t)row * n_cand + idx];
      idxs[i] = in_idxs[(size_t)row * n_cand + idx];
    } else {
      keys[i] = -FLT_MAX;
      idxs[i] = INT_MAX;
    }
  }

  SortT(sort_tmp).SortDescendingBlockedToStriped(keys, idxs);

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int rank = threadIdx.x + i * BLOCK;
    if (rank < k) {
      out_vals[(size_t)row * k + rank] = keys[i];
      out_idxs[(size_t)row * k + rank] = (int64_t)idxs[i];
    }
  }
}

// Hierarchical window merge when n_cand > 4096
template <int BLOCK, int ITEMS>
__global__ void merge_window_kernel(
    const float* __restrict__ in_vals,
    const int* __restrict__ in_idxs,
    float* __restrict__ out_vals,
    int* __restrict__ out_idxs,
    int batch, int n_cand, int window, int k, int num_windows)
{
  using SortT = cub::BlockRadixSort<float, BLOCK, ITEMS, int>;
  __shared__ typename SortT::TempStorage sort_tmp;

  const int row = blockIdx.y;
  const int w   = blockIdx.x;
  if (row >= batch || w >= num_windows) return;

  const int start = w * window;
  int cnt = n_cand - start;
  if (cnt > window) cnt = window;
  if (cnt < 0) cnt = 0;

  float keys[ITEMS];
  int   idxs[ITEMS];
  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int li = threadIdx.x + i * BLOCK;
    if (li < cnt) {
      keys[i] = in_vals[(size_t)row * n_cand + start + li];
      idxs[i] = in_idxs[(size_t)row * n_cand + start + li];
    } else {
      keys[i] = -FLT_MAX;
      idxs[i] = INT_MAX;
    }
  }
  SortT(sort_tmp).SortDescendingBlockedToStriped(keys, idxs);

  #pragma unroll
  for (int i = 0; i < ITEMS; i++) {
    int rank = threadIdx.x + i * BLOCK;
    if (rank < k) {
      size_t off = ((size_t)row * num_windows + w) * k + rank;
      out_vals[off] = keys[i];
      out_idxs[off] = idxs[i];
    }
  }
}

// =====================================================================
// Workspace (persistent device buffers)
// =====================================================================
struct WS {
  float* v = nullptr;
  int*   i = nullptr;
  size_t cap = 0;
};
static std::mutex g_mu;
static WS g_ws;

static void ensure_ws(size_t n) {
  if (g_ws.cap >= n) return;
  std::lock_guard<std::mutex> lk(g_mu);
  if (g_ws.cap >= n) return;
  if (g_ws.v) cudaFree(g_ws.v);
  if (g_ws.i) cudaFree(g_ws.i);
  size_t cap = n + n / 2 + 4096;
  cudaMalloc(&g_ws.v, cap * sizeof(float));
  cudaMalloc(&g_ws.i, cap * sizeof(int));
  g_ws.cap = cap;
}

// Prefer more tiles for SM occupancy when batch is large; prefer larger tiles
// (cheaper merge) for huge single-row inputs. Keep num_blocks * k <= 4096.
static int choose_items(int n, int k, int batch) {
  const int* order;
  int n_order;
  int pref_large[] = {16, 8, 4};
  int pref_small[] = {4, 8, 16};
  if (batch <= 4 && n >= 65536) {
    order = pref_large; n_order = 3;
  } else {
    order = pref_small; n_order = 3;
  }
  for (int oi = 0; oi < n_order; oi++) {
    int items = order[oi];
    int tile = 256 * items;
    int nb = (n + tile - 1) / tile;
    if ((int64_t)nb * k <= 4096) return items;
  }
  return 16;
}

template <int K>
static void launch_partial(
    const float* x, float* pv, int* pi,
    int batch, int n, int num_blocks, int items, cudaStream_t stream)
{
  dim3 grid(num_blocks, batch);
  if (items == 4)
    partial_topk_kernel<256, 4, K><<<grid, 256, 0, stream>>>(x, pv, pi, n, batch, num_blocks);
  else if (items == 8)
    partial_topk_kernel<256, 8, K><<<grid, 256, 0, stream>>>(x, pv, pi, n, batch, num_blocks);
  else
    partial_topk_kernel<256, 16, K><<<grid, 256, 0, stream>>>(x, pv, pi, n, batch, num_blocks);
}

template <int K>
static void launch_single(
    const float* x, float* values, int64_t* indices,
    int batch, int n, int items, cudaStream_t stream)
{
  if (items == 4)
    single_tile_topk_kernel<256, 4, K><<<batch, 256, 0, stream>>>(x, values, indices, n, batch);
  else if (items == 8)
    single_tile_topk_kernel<256, 8, K><<<batch, 256, 0, stream>>>(x, values, indices, n, batch);
  else
    single_tile_topk_kernel<256, 16, K><<<batch, 256, 0, stream>>>(x, values, indices, n, batch);
}

static void launch_merge_radix(
    const float* pv, const int* pi,
    float* values, int64_t* indices,
    int batch, int n_cand, int k, cudaStream_t stream)
{
  if (n_cand <= 1024)
    merge_radix_kernel<256, 4><<<batch, 256, 0, stream>>>(pv, pi, values, indices, batch, n_cand, k);
  else if (n_cand <= 2048)
    merge_radix_kernel<256, 8><<<batch, 256, 0, stream>>>(pv, pi, values, indices, batch, n_cand, k);
  else
    merge_radix_kernel<256, 16><<<batch, 256, 0, stream>>>(pv, pi, values, indices, batch, n_cand, k);
}

template <int K>
static void launch_topk(
    const float* x, float* values, int64_t* indices,
    int batch, int n, cudaStream_t stream)
{
  // Fast path: whole row fits in one BlockRadixSort
  if (n <= 4096) {
    int items = 4;
    while (256 * items < n && items < 16) items *= 2;
    if (256 * items >= n) {
      launch_single<K>(x, values, indices, batch, n, items, stream);
      return;
    }
  }

  int items = choose_items(n, K, batch);
  int tile = 256 * items;
  int num_blocks = (n + tile - 1) / tile;
  if (num_blocks < 1) num_blocks = 1;
  int n_cand = num_blocks * K;

  size_t need = (size_t)batch * n_cand;
  if (n_cand > 4096) {
    int nw = (n_cand + 4095) / 4096;
    need += (size_t)batch * nw * K;
  }
  ensure_ws(need);

  launch_partial<K>(x, g_ws.v, g_ws.i, batch, n, num_blocks, items, stream);

  if (n_cand <= 4096) {
    launch_merge_radix(g_ws.v, g_ws.i, values, indices, batch, n_cand, K, stream);
    return;
  }

  // Hierarchical: windows of 4096 candidates → top-K each → final merge
  constexpr int WINDOW = 4096;
  int num_windows = (n_cand + WINDOW - 1) / WINDOW;
  float* pv2 = g_ws.v + (size_t)batch * n_cand;
  int*   pi2 = g_ws.i + (size_t)batch * n_cand;

  dim3 grid(num_windows, batch);
  merge_window_kernel<256, 16><<<grid, 256, 0, stream>>>(
      g_ws.v, g_ws.i, pv2, pi2, batch, n_cand, WINDOW, K, num_windows);

  int n_cand2 = num_windows * K;
  launch_merge_radix(pv2, pi2, values, indices, batch, n_cand2, K, stream);
}

std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int64_t k_in) {
  TORCH_CHECK(x.is_cuda(), "x must be CUDA");
  TORCH_CHECK(x.scalar_type() == torch::kFloat32, "x must be fp32");
  TORCH_CHECK(x.dim() == 2, "x must be 2D");
  TORCH_CHECK(k_in >= 1 && k_in <= x.size(1), "invalid k");

  x = x.contiguous();
  const int batch = (int)x.size(0);
  const int n     = (int)x.size(1);
  const int k     = (int)k_in;

  auto values  = torch::empty({batch, k}, x.options());
  auto indices = torch::empty(
      {batch, k},
      torch::TensorOptions().dtype(torch::kInt64).device(x.device()));

  cudaStream_t stream = at::cuda::getCurrentCUDAStream();
  const float* x_ptr = x.data_ptr<float>();
  float* v_ptr = values.data_ptr<float>();
  int64_t* i_ptr = indices.data_ptr<int64_t>();

  if (k == 1) {
    argmax_kernel<256><<<batch, 256, 0, stream>>>(x_ptr, v_ptr, i_ptr, n, batch);
  } else if (k == 2) {
    launch_topk<2>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else if (k == 4) {
    launch_topk<4>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else if (k == 8) {
    launch_topk<8>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else if (k == 16) {
    launch_topk<16>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else if (k == 32) {
    launch_topk<32>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else if (k == 64) {
    launch_topk<64>(x_ptr, v_ptr, i_ptr, batch, n, stream);
  } else {
    TORCH_CHECK(false, "unsupported k (use power-of-two k in 2..64)");
  }

  C10_CUDA_CHECK(cudaGetLastError());
  return {values, indices};
}
"""

CPP_SRC = r"""
#include <torch/extension.h>
#include <vector>

std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int64_t k);

std::vector<torch::Tensor> topk_forward(torch::Tensor x, int64_t k) {
  return topk_cuda(x, k);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
  m.def("topk_forward", &topk_forward, "Custom top-k (CUDA)");
}

_EXT = None


def _load_ext():
    global _EXT
    if _EXT is not None:
        return _EXT
    build_dir = Path(__file__).resolve().parent / ".cuda_build_topk"
    build_dir.mkdir(exist_ok=True)
    os.environ.setdefault("TORCH_CUDA_ARCH_LIST", "10.0")
    _EXT = load_inline(
        name="topk_bitonic_b200_v8",
        cpp_sources=CPP_SRC,
        cuda_sources=CUDA_SRC,
        extra_cuda_cflags=["-O3", "--use_fast_math", "-lineinfo"],
        extra_cflags=["-O3"],
        build_directory=str(build_dir),
        verbose=False,
    )
    return _EXT


batch = 64
n = 8192
k = 8


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))
        _load_ext()

    def forward(self, x: torch.Tensor):
        return _load_ext().topk_forward(x, int(self.k))


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
