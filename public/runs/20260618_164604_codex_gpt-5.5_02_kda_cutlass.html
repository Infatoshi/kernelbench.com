def _local_cumsum_kernel(
    g,
    go,
    scale: tl.constexpr,
    T: tl.constexpr,
    H: tl.constexpr,
    K: tl.constexpr,
    BT: tl.constexpr,
    BS: tl.constexpr,
):
    i_s = tl.program_id(0)
    i_t = tl.program_id(1)
    i_bh = tl.program_id(2)
    i_b = i_bh // H
    i_h = i_bh - i_b * H

    offs_t = i_t * BT + tl.arange(0, BT)
    offs_s = i_s * BS + tl.arange(0, BS)
    p = ((i_b * T + offs_t[:, None]) * H + i_h) * K + offs_s[None, :]
    mask = offs_s[None, :] < K
    x = tl.load(g + p, mask=mask, other=0.0).to(tl.float32)
    y = tl.cumsum(x, axis=0) * scale
    tl.store(go + p, y, mask=mask)

    b_gp = _exp2(b_g - b_mid[None, :])
    b_gn = _exp2(b_mid[None, :] - b_g)
    b_gp = tl.where(mask_t[:, None], b_gp, 0.0)
    b_gn = tl.where(mask_t[:, None], b_gn, 0.0)

    tl.store(ad + p_ad, b_ak, mask=mask_t[:, None])
