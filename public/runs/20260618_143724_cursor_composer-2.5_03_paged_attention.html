"""Custom paged-attention decode kernel for RTX 3090 (SM86 Ampere).

Memory-bound single-query decode via FlashDecoding-style Triton kernels:
  * Main kernel grid: (num_splits, batch, num_kv_heads). Each program owns one
    contiguous chunk of the sequence for one (batch, kv_head) pair and computes
    attention for ALL G query heads in that GQA group, sharing the K/V load.
  * Reduction kernel grid: (batch, num_heads). Merges per-split partial outputs
    via log-sum-exp combination.

KV cache layout: (num_blocks, page_size, num_kv_heads, head_dim*2) with [K|V]
packed on the last dim. Logical token n of batch b maps to physical block
block_table[b, n//P], offset n%P.

All softmax math runs in fp32 (online softmax) for numeric-stress tolerance.
"""

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _decode_split_kernel(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr,
    OP_ptr, MP_ptr, LP_ptr,
    sm_scale,
    q_b_stride, q_h_stride,
    kv_blk_stride, kv_pg_stride, kv_h_stride,
    bt_b_stride,
    op_b_stride, op_h_stride, op_split_stride,
    ml_b_stride, ml_h_stride, ml_split_stride,
    seq_len_max,
    G: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    BLOCK_N: tl.constexpr,
    G_PAD: tl.constexpr,
    NORMALIZE: tl.constexpr,
):
    sp = tl.program_id(0)
    b = tl.program_id(1)
    kvi = tl.program_id(2)

    seq_len = tl.load(SL_ptr + b)

    tokens_per_split = tl.cdiv(seq_len, NUM_SPLITS)
    start = sp * tokens_per_split
    end = tl.minimum(start + tokens_per_split, seq_len)

    g_offs = tl.arange(0, G_PAD)
    d_offs = tl.arange(0, D)
    q_ptrs = (Q_ptr + b * q_b_stride
              + (kvi * G + g_offs[:, None]) * q_h_stride
              + d_offs[None, :])
    q = tl.load(q_ptrs, mask=g_offs[:, None] < G, other=0.0)

    m_i = tl.full([G_PAD], -float("inf"), dtype=tl.float32)
    l_i = tl.zeros([G_PAD], dtype=tl.float32)
    acc = tl.zeros([G_PAD, D], dtype=tl.float32)

    n_offs = tl.arange(0, BLOCK_N)

    for start_n in range(start, end, BLOCK_N):
        n = start_n + n_offs
        valid = n < end

        page_idx = n // P
        off_in_page = n % P
        block_ids = tl.load(BT_ptr + b * bt_b_stride + page_idx,
                            mask=valid, other=0)

        kv_base = (KV_ptr + block_ids[:, None] * kv_blk_stride
                   + off_in_page[:, None] * kv_pg_stride
                   + kvi * kv_h_stride)
        k = tl.load(kv_base + d_offs[None, :], mask=valid[:, None], other=0.0,
                    cache_modifier=".cg")
        v = tl.load(kv_base + (d_offs + D)[None, :], mask=valid[:, None], other=0.0,
                    cache_modifier=".cg")

        s = tl.dot(q, tl.trans(k)) * sm_scale
        s = tl.where(valid[None, :], s, -float("inf"))

        m_ij = tl.maximum(m_i, tl.max(s, axis=1))
        alpha = tl.exp(m_i - m_ij)
        p = tl.exp(s - m_ij[:, None])
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None]
        acc += tl.dot(p.to(tl.bfloat16), v)

        m_i = m_ij

    head_off = kvi * G + g_offs
    if NORMALIZE:
        acc = acc / l_i[:, None]
        out_ptrs = (OP_ptr + b * op_b_stride + head_off[:, None] * op_h_stride
                    + d_offs[None, :])
        tl.store(out_ptrs, acc.to(tl.bfloat16), mask=g_offs[:, None] < G)
    else:
        out_ptrs = (OP_ptr + b * op_b_stride + head_off[:, None] * op_h_stride
                    + sp * op_split_stride + d_offs[None, :])
        tl.store(out_ptrs, acc.to(tl.bfloat16), mask=g_offs[:, None] < G)
        m_ptrs = (MP_ptr + b * ml_b_stride + head_off * ml_h_stride
                  + sp * ml_split_stride)
        tl.store(m_ptrs, m_i, mask=g_offs < G)
        l_ptrs = (LP_ptr + b * ml_b_stride + head_off * ml_h_stride
                  + sp * ml_split_stride)
        tl.store(l_ptrs, l_i, mask=g_offs < G)


@triton.jit
def _reduce_kernel(
    OP_ptr, MP_ptr, LP_ptr, O_ptr,
    op_b_stride, op_h_stride, op_split_stride,
    ml_b_stride, ml_h_stride,
    o_b_stride, o_h_stride,
    H,
    NUM_SPLITS: tl.constexpr,
    D: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)

    sp_offs = tl.arange(0, NUM_SPLITS)
    d_offs = tl.arange(0, D)

    m_ptrs = MP_ptr + b * ml_b_stride + h * ml_h_stride + sp_offs
    l_ptrs = LP_ptr + b * ml_b_stride + h * ml_h_stride + sp_offs
    m_s = tl.load(m_ptrs)
    l_s = tl.load(l_ptrs)

    m_global = tl.max(m_s, axis=0)
    w = tl.exp(m_s - m_global)
    denom = tl.sum(w * l_s, axis=0)

    op_ptrs = (OP_ptr + b * op_b_stride + h * op_h_stride
               + sp_offs[:, None] * op_split_stride + d_offs[None, :])
    op = tl.load(op_ptrs)
    o = tl.sum(op.to(tl.float32) * w[:, None], axis=0) / denom
    tl.store(O_ptr + b * o_b_stride + h * o_h_stride + d_offs,
             o.to(tl.bfloat16))


def _next_pow2(x: int) -> int:
    p = 1
    while p < x:
        p <<= 1
    return p

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):

        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._buf_cache = {}

        # Per-shape tuning from micro-benchmark sweep on RTX 3090 (SM86).
        if head_dim == 64:
            self._block_n, self._num_warps, self._num_stages = 32, 2, 4
        elif batch >= 32:
            self._block_n, self._num_warps, self._num_stages = 32, 4, 4
        elif batch >= 16:
            self._block_n, self._num_warps, self._num_stages = 64, 8, 2
        elif seq_len >= 4096:
            self._block_n, self._num_warps, self._num_stages = 32, 4, 4
        else:
            self._block_n, self._num_warps, self._num_stages = 64, 4, 4

    def _pick_splits(self, B, Hkv, BLOCK_N, num_sms):
        if B * Hkv >= int(num_sms * 0.75):
            return 1
        target = 2 * num_sms
        ns = max(1, target // (B * Hkv))
        ns = _next_pow2(ns)
        max_ns = max(1, self.seq_len // (BLOCK_N * 2))
        while ns > max_ns and ns > 1:
            ns //= 2
        return max(1, int(ns))

        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:

        G = self.group_size
        P = self.page_size
        scale = self.scale

        num_warps = self._num_warps
        num_stages = self._num_stages

        G_PAD = max(16, _next_pow2(G))

                NUM_SPLITS=1,
                BLOCK_N=BLOCK_N,
                G_PAD=G_PAD,
                NORMALIZE=True,
                num_warps=num_warps,
                num_stages=num_stages,
            )
            return out

        key = (str(device), B, H, D, num_splits)
        if key not in cache:
            cache[key] = (
                torch.empty(B, H, num_splits, D, dtype=query.dtype, device=device),
                torch.empty(B, H, num_splits, dtype=torch.float32, device=device),
                torch.empty(B, H, num_splits, dtype=torch.float32, device=device),
            )
        op_partial, m_partial, l_partial = cache[key]

        grid = (num_splits, B, Hkv)
        _decode_split_kernel[grid](
            query, kv_cache, block_table, seq_lens,
            op_partial, m_partial, l_partial,
            scale,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0),
            op_partial.stride(0), op_partial.stride(1), op_partial.stride(2),
            m_partial.stride(0), m_partial.stride(1), m_partial.stride(2),
            self.seq_len,
            G=G, D=D, P=P,
            NUM_SPLITS=num_splits,
            BLOCK_N=BLOCK_N,
            G_PAD=G_PAD,
            NORMALIZE=False,
            num_warps=num_warps,
            num_stages=num_stages,
        )

        _reduce_kernel[(B, H)](
            op_partial, m_partial, l_partial, out,
            op_partial.stride(0), op_partial.stride(1), op_partial.stride(2),
            m_partial.stride(0), m_partial.stride(1),
            out.stride(0), out.stride(1),
            H,
            NUM_SPLITS=num_splits,
            D=D,
            num_warps=2,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]

    """Time `fn(*inputs)` and return the median wall time in milliseconds.

    Each timed call is preceded by an L2 flush.
    """

            fn(*inputs)
