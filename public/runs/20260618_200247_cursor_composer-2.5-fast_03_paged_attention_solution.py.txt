"""Triton paged-attention decode for B200 (SM100).

GQA grouped decode with KV-split parallelism, autotuned tile sizes,
and page-table indirection over packed [K|V] KV cache.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

# --- Shape knobs (overridden by check.py / benchmark.py from shapes.py) ----
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

_AUTOTUNE_CONFIGS = [
    triton.Config({"BLOCK_N": 16}, num_warps=4, num_stages=2),
    triton.Config({"BLOCK_N": 32}, num_warps=4, num_stages=3),
    triton.Config({"BLOCK_N": 64}, num_warps=4, num_stages=3),
    triton.Config({"BLOCK_N": 64}, num_warps=8, num_stages=3),
]


@triton.autotune(
    configs=_AUTOTUNE_CONFIGS,
    key=["Lk", "NUM_KV_SPLITS", "BLOCK_H"],
)
@triton.jit
def _fwd_grouped_kernel_stage1(
    Q,
    K_Buffer,
    V_Buffer,
    sm_scale,
    Block_table,
    B_Seqlen,
    Att_Out,
    Att_Lse,
    stride_bt_b,
    stride_qbs,
    stride_qh,
    stride_buf_kbs,
    stride_buf_kh,
    stride_buf_vbs,
    stride_buf_vh,
    stride_mid_ob,
    stride_mid_oh,
    stride_mid_os,
    kv_group_num: tl.constexpr,
    q_head_num: tl.constexpr,
    BLOCK_DMODEL: tl.constexpr,
    BLOCK_DV: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_H: tl.constexpr,
    NUM_KV_SPLITS: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    Lk: tl.constexpr,
    Lv: tl.constexpr,
):
    cur_batch = tl.program_id(0)
    cur_head_id = tl.program_id(1)
    cur_kv_head = cur_head_id // tl.cdiv(kv_group_num, BLOCK_H)
    split_kv_id = tl.program_id(2)

    if BLOCK_H < kv_group_num:
        VALID_BLOCK_H: tl.constexpr = BLOCK_H
    else:
        VALID_BLOCK_H: tl.constexpr = kv_group_num
    cur_head = cur_head_id * VALID_BLOCK_H + tl.arange(0, BLOCK_H)
    mask_h = cur_head < (cur_head_id + 1) * VALID_BLOCK_H
    mask_h = mask_h & (cur_head < q_head_num)

    offs_d = tl.arange(0, BLOCK_DMODEL)
    offs_dv = tl.arange(0, BLOCK_DV)
    mask_d = offs_d < Lk
    mask_dv = offs_dv < Lv

    seq_len = tl.load(B_Seqlen + cur_batch)
    kv_len_per_split = tl.cdiv(seq_len, NUM_KV_SPLITS)
    split_start = kv_len_per_split * split_kv_id
    split_end = tl.minimum(split_start + kv_len_per_split, seq_len)

    e_max = tl.zeros([BLOCK_H], dtype=tl.float32) - float("inf")
    e_sum = tl.zeros([BLOCK_H], dtype=tl.float32)
    acc = tl.zeros([BLOCK_H, BLOCK_DV], dtype=tl.float32)
    base_offs_k = cur_kv_head * stride_buf_kh + offs_d[:, None]
    base_offs_v = cur_kv_head * stride_buf_vh + offs_dv[None, :]

    if split_end > split_start:
        offs_q = cur_batch * stride_qbs + cur_head[:, None] * stride_qh + offs_d[None, :]
        q = tl.load(Q + offs_q, mask=(mask_h[:, None]) & (mask_d[None, :]), other=0.0, cache_modifier=".ca")
        q_k = q.to(K_Buffer.dtype.element_ty)

        for start_n in tl.range(split_start, split_end, BLOCK_N, num_stages=2):
            offs_n = start_n + tl.arange(0, BLOCK_N)
            kv_page_number = tl.load(
                Block_table + stride_bt_b * cur_batch + offs_n // PAGE_SIZE,
                mask=offs_n < split_end,
                other=0,
                cache_modifier=".ca",
            )
            kv_loc = kv_page_number * PAGE_SIZE + offs_n % PAGE_SIZE
            offs_buf_k = kv_loc[None, :] * stride_buf_kbs + base_offs_k
            k = tl.load(
                K_Buffer + offs_buf_k,
                mask=(offs_n[None, :] < split_end) & (mask_d[:, None]),
                other=0.0,
                cache_modifier=".cg",
            )
            qk = tl.dot(q_k, k) * sm_scale
            qk = tl.where(
                mask_h[:, None] & (offs_n[None, :] < split_end), qk, float("-inf")
            )

            offs_buf_v = kv_loc[:, None] * stride_buf_vbs + base_offs_v
            v = tl.load(
                V_Buffer + offs_buf_v,
                mask=(offs_n[:, None] < split_end) & (mask_dv[None, :]),
                other=0.0,
                cache_modifier=".cg",
            )

            n_e_max = tl.maximum(tl.max(qk, 1), e_max)
            re_scale = tl.exp(e_max - n_e_max)
            p = tl.exp(qk - n_e_max[:, None])
            acc *= re_scale[:, None]
            acc += tl.dot(p.to(v.dtype), v)
            e_sum = e_sum * re_scale + tl.sum(p, 1)
            e_max = n_e_max

        offs_mid_o = (
            cur_batch * stride_mid_ob
            + cur_head[:, None] * stride_mid_oh
            + split_kv_id * stride_mid_os
            + offs_dv[None, :]
        )
        tl.store(
            Att_Out + offs_mid_o,
            acc / e_sum[:, None],
            mask=(mask_h[:, None]) & (mask_dv[None, :]),
        )
        offs_lse = (
            cur_batch * stride_mid_ob + cur_head * stride_mid_oh + split_kv_id * stride_mid_os
        ) // Lv
        tl.store(Att_Lse + offs_lse, e_max + tl.log(e_sum), mask=mask_h)


@triton.jit
def _fwd_kernel_stage2(
    Mid_O,
    Mid_Lse,
    O,
    B_Seqlen,
    stride_mid_ob,
    stride_mid_oh,
    stride_mid_os,
    stride_obs,
    stride_oh,
    NUM_KV_SPLITS: tl.constexpr,
    BLOCK_DV: tl.constexpr,
    Lv: tl.constexpr,
):
    cur_batch = tl.program_id(0)
    cur_head = tl.program_id(1)

    seq_len = tl.load(B_Seqlen + cur_batch)
    offs_d = tl.arange(0, BLOCK_DV)
    mask_d = offs_d < Lv

    e_sum = 0.0
    e_max = -float("inf")
    acc = tl.zeros([BLOCK_DV], dtype=tl.float32)

    offs_v = cur_batch * stride_mid_ob + cur_head * stride_mid_oh + offs_d
    offs_logic = (cur_batch * stride_mid_ob + cur_head * stride_mid_oh) // Lv
    kv_len_per_split = tl.cdiv(seq_len, NUM_KV_SPLITS)

    for split_id in tl.range(0, NUM_KV_SPLITS, num_stages=2):
        split_start = kv_len_per_split * split_id
        split_end = tl.minimum(split_start + kv_len_per_split, seq_len)
        if split_end > split_start:
            tv = tl.load(Mid_O + offs_v + split_id * stride_mid_os, mask=mask_d, other=0.0)
            tlogic = tl.load(Mid_Lse + offs_logic + split_id * stride_mid_os // Lv)
            n_e_max = tl.maximum(tlogic, e_max)
            old_scale = tl.exp(e_max - n_e_max)
            acc *= old_scale
            exp_logic = tl.exp(tlogic - n_e_max)
            acc += exp_logic * tv
            e_sum = e_sum * old_scale + exp_logic
            e_max = n_e_max

    tl.store(
        O + cur_batch * stride_obs + cur_head * stride_oh + offs_d,
        acc / e_sum,
        mask=mask_d,
    )


def _pick_num_kv_splits(max_seq_len: int, batch: int, num_heads: int, kv_group: int) -> int:
    parallel = batch * triton.cdiv(num_heads, min(16, kv_group))
    target_programs = 4096
    splits = triton.cdiv(target_programs, max(parallel, 1))
    splits = max(4, min(64, splits))
    # Also ensure each split covers enough tokens to amortize overhead
    min_splits = triton.cdiv(max_seq_len, 512)
    return max(splits, min(min_splits, 64))


def _run_grouped_stage1(
    q, k_cache, v_cache, block_table, seq_lens, att_out, att_lse, num_kv_splits, sm_scale, page_size
):
    Lk = k_cache.shape[-1]
    Lv = v_cache.shape[-1]
    BLOCK_DMODEL = triton.next_power_of_2(Lk)
    BLOCK_DV = triton.next_power_of_2(Lv)
    batch, head_num = q.shape[0], q.shape[1]
    kv_group_num = head_num // k_cache.shape[-2]
    BLOCK_H = min(16, kv_group_num)

    grid = (batch, triton.cdiv(head_num, BLOCK_H), num_kv_splits)
    _fwd_grouped_kernel_stage1[grid](
        q,
        k_cache,
        v_cache,
        sm_scale,
        block_table,
        seq_lens,
        att_out,
        att_lse,
        block_table.stride(0),
        q.stride(0),
        q.stride(1),
        k_cache.stride(-3),
        k_cache.stride(-2),
        v_cache.stride(-3),
        v_cache.stride(-2),
        att_out.stride(0),
        att_out.stride(1),
        att_out.stride(2),
        kv_group_num=kv_group_num,
        q_head_num=head_num,
        BLOCK_DMODEL=BLOCK_DMODEL,
        BLOCK_DV=BLOCK_DV,
        BLOCK_H=BLOCK_H,
        NUM_KV_SPLITS=num_kv_splits,
        PAGE_SIZE=page_size,
        Lk=Lk,
        Lv=Lv,
    )


def _run_stage2(att_out, att_lse, out, seq_lens, num_kv_splits):
    batch, head_num = out.shape[0], out.shape[1]
    Lv = out.shape[-1]
    BLOCK_DV = triton.next_power_of_2(Lv)
    grid = (batch, head_num)
    _fwd_kernel_stage2[grid](
        att_out,
        att_lse,
        out,
        seq_lens,
        att_out.stride(0),
        att_out.stride(1),
        att_out.stride(2),
        out.stride(0),
        out.stride(1),
        NUM_KV_SPLITS=num_kv_splits,
        BLOCK_DV=BLOCK_DV,
        Lv=Lv,
        num_warps=4,
        num_stages=2,
    )


def paged_attention_decode(
    query: torch.Tensor,
    kv_cache: torch.Tensor,
    block_table: torch.Tensor,
    seq_lens: torch.Tensor,
    num_kv_heads: int,
    head_dim: int,
    page_size: int,
    sm_scale: float,
    workspace: dict | None = None,
) -> torch.Tensor:
    D = head_dim
    k_cache = kv_cache[..., :D]
    v_cache = kv_cache[..., D:]

    B, H, _ = query.shape
    max_seq = int(seq_lens.max().item())
    kv_group = H // num_kv_heads
    num_kv_splits = _pick_num_kv_splits(max_seq, B, H, kv_group)

    if workspace is None:
        workspace = {}

    key = (B, H, D, num_kv_splits)
    if key not in workspace:
        workspace[key] = {
            "att_out": torch.empty(B, H, num_kv_splits, D, dtype=torch.float32, device=query.device),
            "att_lse": torch.empty(B, H, num_kv_splits, dtype=torch.float32, device=query.device),
            "out": torch.empty(B, H, D, dtype=torch.float32, device=query.device),
        }
    ws = workspace[key]

    _run_grouped_stage1(
        query, k_cache, v_cache, block_table, seq_lens,
        ws["att_out"], ws["att_lse"], num_kv_splits, sm_scale, page_size,
    )
    _run_stage2(ws["att_out"], ws["att_lse"], ws["out"], seq_lens, num_kv_splits)
    return ws["out"].to(query.dtype)


class Model(nn.Module):
    """Single-query paged attention decode."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._workspace: dict = {}

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        return paged_attention_decode(
            query,
            kv_cache,
            block_table,
            seq_lens,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
            self.scale,
            self._workspace,
        )


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
