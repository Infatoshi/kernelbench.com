"""Triton paged-attention decode - tl.dot with nw=2 for RTX PRO 6000 Blackwell."""

import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_attn_kernel(
    q_ptr, kv_cache_ptr, block_table_ptr, seq_lens_ptr, out_ptr,
    stride_qb, stride_qh, stride_qd,
    stride_kcb, stride_kcp, stride_kch,
    stride_btb,
    stride_ob, stride_oh, stride_od,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    kv_group_size: tl.constexpr,
    scale,
):
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)
    kv_head = pid_h // kv_group_size

    seq_len = tl.load(seq_lens_ptr + pid_b)
    num_pages = (seq_len + PAGE_SIZE - 1) // PAGE_SIZE

    offs_d = tl.arange(0, HEAD_DIM)
    offs_p = tl.arange(0, PAGE_SIZE)

    q = tl.load(q_ptr + pid_b * stride_qb + pid_h * stride_qh + offs_d).to(tl.float16)
    q_2d = q[None, :]

    m_i = -float("inf")
    d_i = 0.0
    acc = tl.zeros([1, HEAD_DIM], tl.float32)

    for i in range(num_pages):
        block_id = tl.load(block_table_ptr + pid_b * stride_btb + i)
        remaining = seq_len - i * PAGE_SIZE
        num_tokens = tl.minimum(PAGE_SIZE, remaining)
        base_off = block_id * stride_kcb + kv_head * stride_kch

        k_ptrs = kv_cache_ptr + base_off + offs_p[:, None] * stride_kcp + offs_d[None, :]
        k_mask = offs_p[:, None] < num_tokens
        k = tl.load(k_ptrs, mask=k_mask).to(tl.float16)
        k_T = tl.trans(k)
        scores_2d = tl.dot(q_2d, k_T) * scale

        scores_1d = tl.sum(scores_2d.to(tl.float32), axis=0)
        scores_1d = tl.where(offs_p < num_tokens, scores_1d, -float("inf"))

        m_new = tl.maximum(m_i, tl.max(scores_1d, axis=0))
        alpha = tl.exp(m_i - m_new)
        beta = tl.exp(scores_1d - m_new)
        d_new = d_i * alpha + tl.sum(beta, axis=0)

        v_ptrs = kv_cache_ptr + base_off + offs_p[:, None] * stride_kcp + (HEAD_DIM + offs_d[None, :])
        v = tl.load(v_ptrs, mask=k_mask).to(tl.float16)
        beta_2d = beta[None, :].to(tl.float16)
        weighted_v = tl.dot(beta_2d, v)

        acc = acc * alpha + weighted_v.to(tl.float32)
        m_i = m_new
        d_i = d_new

    acc_1d = tl.sum(acc, axis=0)
    out = acc_1d / d_i
    tl.store(out_ptr + pid_b * stride_ob + pid_h * stride_oh + offs_d, out.to(tl.bfloat16))


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = float(1.0 / math.sqrt(head_dim))
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        P = self.page_size
        G = self.group_size

        out = torch.empty_like(query)
        grid = (B, H)

        _paged_attn_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0),
            out.stride(0), out.stride(1), out.stride(2),
            D, P, G, self.scale,
            num_warps=2, num_stages=4,
        )

        return out


def get_inputs():
    from reference import BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE as P
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1
    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    from reference import BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]

        o = m(*ref_inps)
