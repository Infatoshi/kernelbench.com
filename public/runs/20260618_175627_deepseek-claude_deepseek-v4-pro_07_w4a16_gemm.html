"""W4A16 weight-only quantized GEMM — fused Triton kernel for RTX 3090 (SM86).

AWQ/GPTQ-style asymmetric int4 with explicit zero-points and per-group bf16
scales (group_size=128 along K).  Fuses int4 unpack, asymmetric dequant, and
GEMM in a single kernel pass — no materialized bf16 weight matrix.
"""

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["RTX_3090"]

GROUP_SIZE = 128


@triton.jit
def _w4a16_gemm_kernel(
    x_ptr,           # (M, K) bf16
    w_ptr,           # (K // 2, N) uint8 — two int4 weights packed per byte
    scales_ptr,      # (K // group, N) bf16
    zeros_ptr,       # (K // group, N) bf16
    out_ptr,         # (M, N) bf16
    M,
    N,
    K,
    group_size: tl.constexpr,
    stride_xm,
    stride_xk,
    stride_wk,
    stride_wn,
    stride_sg,
    stride_sn,
    stride_om,
    stride_on,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    """Fused W4A16 GEMM: unpack int4 → dequant → matmul in one pass."""
    pid = tl.program_id(0)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for k in range(0, K, BLOCK_K):
        # ── load activations: (BLOCK_M, BLOCK_K) bf16 ────────────────────
        offs_k_full = k + tl.arange(0, BLOCK_K)
        x_ptrs = (
            x_ptr
            + offs_m[:, None] * stride_xm
            + offs_k_full[None, :] * stride_xk
        )
        x = tl.load(
            x_ptrs,
            mask=(offs_m[:, None] < M) & (offs_k_full[None, :] < K),
            other=0.0,
        )

        # ── load packed weights: (BLOCK_K//2, BLOCK_N) uint8 ─────────────
        w_k_packed = k // 2
        offs_k_p = w_k_packed + tl.arange(0, BLOCK_K // 2)
        w_ptrs = (
            w_ptr
            + offs_k_p[:, None] * stride_wk
            + offs_n[None, :] * stride_wn
        )
        w_bytes = tl.load(
            w_ptrs,
            mask=(offs_k_p[:, None] < K // 2) & (offs_n[None, :] < N),
            other=0,
        )

        # ── unpack int4 → interleave even/odd back to (BLOCK_K, BLOCK_N) ─
        w_lo = w_bytes & 0xF             # even K rows (0, 2, 4, …)
        w_hi = (w_bytes >> 4) & 0xF      # odd  K rows (1, 3, 5, …)
        # tl.interleave acts on the last dim, so transpose→interleave→untranspose
        w_lo_t = tl.trans(w_lo)          # (BLOCK_N, BLOCK_K//2)
        w_hi_t = tl.trans(w_hi)
        w_interleaved_t = tl.interleave(w_lo_t, w_hi_t)  # (BLOCK_N, BLOCK_K)
        w_unpacked = tl.trans(w_interleaved_t)           # (BLOCK_K, BLOCK_N)

        # ── load group-wise scales and zeros: (BLOCK_N,) bf16 ───────────
        group_idx = k // group_size
        s_ptrs = scales_ptr + group_idx * stride_sg + offs_n * stride_sn
        scales_k = tl.load(s_ptrs, mask=offs_n < N, other=1.0)
        z_ptrs = zeros_ptr + group_idx * stride_sg + offs_n * stride_sn
        zeros_k = tl.load(z_ptrs, mask=offs_n < N, other=0.0)

        # ── dequant: w_bf = (int4 - zero) * scale ────────────────────────
        w_int_bf16 = w_unpacked.to(tl.bfloat16)          # (BLOCK_K, BLOCK_N)
        w_deq = (w_int_bf16 - zeros_k[None, :]) * scales_k[None, :]

        # ── tensor-core GEMM tile (bf16×bf16 → fp32 accumulate) ──────────
        acc += tl.dot(x, w_deq)

    # ── store output: (BLOCK_M, BLOCK_N) bf16 ──────────────────────────
    out_ptrs = (
        out_ptr
        + offs_m[:, None] * stride_om
        + offs_n[None, :] * stride_on
    )
    out = acc.to(tl.bfloat16)
    tl.store(out_ptrs, out, mask=(offs_m[:, None] < M) & (offs_n[None, :] < N))

    Fused Triton kernel that unpacks int4 weights, applies group-wise
    asymmetric dequant, and computes the GEMM in a single pass.
    """

    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0, "K must be divisible by group_size"
        assert K % 2 == 0,         "K must be even (int4 packing)"
        self.M = M
        self.N = N
        self.K = K
        self.group_size = group_size
        n_groups = K // group_size

        torch.manual_seed(0xC0DE ^ (M * 1315423911 + N * 2654435761 + K))
        w_full = torch.randn(K, N, dtype=torch.float32) * 0.02

        w_g = w_full.view(n_groups, group_size, N)
        w_min = w_g.amin(dim=1, keepdim=True)
        w_max = w_g.amax(dim=1, keepdim=True)
        scales = (w_max - w_min).clamp_min(1e-8) / 15.0
        zeros = (-w_min / scales).round().clamp(0, 15)
        w_q = ((w_g / scales) + zeros).round().clamp(0, 15).to(torch.uint8)
        w_q = w_q.view(K, N)

        scales_2d = scales.squeeze(1).to(torch.bfloat16)
        zeros_2d = zeros.squeeze(1).to(torch.bfloat16)
        w_packed = _pack_int4(w_q)

        self.register_buffer("w_q", w_packed)
        self.register_buffer("scales", scales_2d)
        self.register_buffer("zeros", zeros_2d)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        M_batch, K_in = x.shape
        assert K_in == self.K, f"Input K={K_in} != model K={self.K}"
        N = self.N
        K = self.K

        # ── tile-size dispatch ─────────────────────────────────────────
        if M_batch == 1:
            BLOCK_M, BLOCK_N, warps = 1, 64, 8
        elif M_batch <= 32:
            BLOCK_M, BLOCK_N, warps = 16, 64, 4
        else:
            BLOCK_M, BLOCK_N, warps = 32, 64, 4

        BLOCK_K = 128  # = group_size

        grid = (
            triton.cdiv(M_batch, BLOCK_M) * triton.cdiv(N, BLOCK_N),
        )

        out = torch.empty(M_batch, N, dtype=torch.bfloat16, device=x.device)

        _w4a16_gemm_kernel[grid](
            x,
            self.w_q,
            self.scales,
            self.zeros,
            out,
            M_batch,
            N,
            K,
            group_size=self.group_size,
            stride_xm=x.stride(0),
            stride_xk=x.stride(1),
            stride_wk=self.w_q.stride(0),
            stride_wn=self.w_q.stride(1),
            stride_sg=self.scales.stride(0),
            stride_sn=self.scales.stride(1),
            stride_om=out.stride(0),
            stride_on=out.stride(1),
            BLOCK_M=BLOCK_M,
            BLOCK_N=BLOCK_N,
            BLOCK_K=BLOCK_K,
            num_warps=warps,
        )

        return out


def _pack_int4(w_q: torch.Tensor) -> torch.Tensor:
    K, N = w_q.shape
    assert K % 2 == 0
    lo = w_q[0::2].to(torch.uint8) & 0xF
    hi = w_q[1::2].to(torch.uint8) & 0xF
    return (lo | (hi << 4)).contiguous()
