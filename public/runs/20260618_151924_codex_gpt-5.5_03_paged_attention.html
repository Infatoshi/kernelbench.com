import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_attention_stage1(
    query,
    kv_cache,
    block_table,
    seq_lens,
    partial_acc,
    partial_m,
    partial_l,
    sm_scale: tl.constexpr,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE_: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    BLOCK_TABLE_STRIDE: tl.constexpr,
    NUM_PARTS: tl.constexpr,
    PARTITION_SIZE: tl.constexpr,
    BLOCK_M: tl.constexpr,
):
    b = tl.program_id(0)
    kv_head = tl.program_id(1)
    part = tl.program_id(2)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, HEAD_DIM)
    q_heads = kv_head * GROUP_SIZE + offs_g

    q_offsets = (b * NUM_HEADS + q_heads[:, None]) * HEAD_DIM + offs_d[None, :]
    q = tl.load(query + q_offsets).to(tl.float32)

    m_i = tl.full((GROUP_SIZE,), -float("inf"), tl.float32)
    l_i = tl.zeros((GROUP_SIZE,), tl.float32)
    acc = tl.zeros((GROUP_SIZE, HEAD_DIM), tl.float32)

    seq_len = tl.load(seq_lens + b)
    part_start = part * PARTITION_SIZE
    token_offsets = tl.arange(0, BLOCK_M)

    for block_start in range(0, PARTITION_SIZE, BLOCK_M):
        logical_tokens = part_start + block_start + token_offsets
        valid_tokens = logical_tokens < seq_len
        page_ids = logical_tokens // PAGE_SIZE_
        page_offsets = logical_tokens - page_ids * PAGE_SIZE_
        block_ids = tl.load(
            block_table + b * BLOCK_TABLE_STRIDE + page_ids,
            mask=valid_tokens,
            other=0,
        ).to(tl.int64)

        kv_offsets = (
            ((block_ids[:, None] * PAGE_SIZE_ + page_offsets[:, None]) * NUM_KV_HEADS + kv_head)
            * (2 * HEAD_DIM)
            + offs_d[None, :]
        )
        k = tl.load(kv_cache + kv_offsets, mask=valid_tokens[:, None], other=0.0).to(tl.float32)
        v = tl.load(
            kv_cache + kv_offsets + HEAD_DIM,
            mask=valid_tokens[:, None],
            other=0.0,
        ).to(tl.float32)

        scores = tl.dot(q, tl.trans(k), input_precision="tf32", out_dtype=tl.float32)
        scores = scores * sm_scale
        scores = tl.where(valid_tokens[None, :], scores, -float("inf"))

        m_block = tl.max(scores, axis=1)
        m_new = tl.maximum(m_i, m_block)
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new[:, None])

        acc = acc * alpha[:, None] + tl.dot(p, v, input_precision="tf32", out_dtype=tl.float32)
        l_i = l_i * alpha + tl.sum(p, axis=1)
        m_i = m_new

    stats_base = ((b * NUM_KV_HEADS + kv_head) * NUM_PARTS + part) * GROUP_SIZE
    tl.store(partial_m + stats_base + offs_g, m_i)
    tl.store(partial_l + stats_base + offs_g, l_i)

    acc_base = stats_base * HEAD_DIM
    acc_offsets = acc_base + offs_g[:, None] * HEAD_DIM + offs_d[None, :]
    tl.store(partial_acc + acc_offsets, acc)


@triton.jit
def _paged_attention_stage2(
    partial_acc,
    partial_m,
    partial_l,
    output,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    NUM_PARTS: tl.constexpr,
):
    b = tl.program_id(0)
    kv_head = tl.program_id(1)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, HEAD_DIM)
    q_heads = kv_head * GROUP_SIZE + offs_g

    m_i = tl.full((GROUP_SIZE,), -float("inf"), tl.float32)
    l_i = tl.zeros((GROUP_SIZE,), tl.float32)
    acc = tl.zeros((GROUP_SIZE, HEAD_DIM), tl.float32)

    for part in range(0, NUM_PARTS):
        stats_base = ((b * NUM_KV_HEADS + kv_head) * NUM_PARTS + part) * GROUP_SIZE
        m_part = tl.load(partial_m + stats_base + offs_g)
        l_part = tl.load(partial_l + stats_base + offs_g)

        m_new = tl.maximum(m_i, m_part)
        alpha = tl.exp(m_i - m_new)
        beta = tl.exp(m_part - m_new)

        acc_base = stats_base * HEAD_DIM
        acc_offsets = acc_base + offs_g[:, None] * HEAD_DIM + offs_d[None, :]
        acc_part = tl.load(partial_acc + acc_offsets)

        acc = acc * alpha[:, None] + acc_part * beta[:, None]
        l_i = l_i * alpha + l_part * beta
        m_i = m_new

    out = acc / l_i[:, None]
    out_offsets = (b * NUM_HEADS + q_heads[:, None]) * HEAD_DIM + offs_d[None, :]
    tl.store(output + out_offsets, out)


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._scratch_key = None
        self._partial_acc = None
        self._partial_m = None
        self._partial_l = None

    def _scratch(self, query: torch.Tensor, num_parts: int):
        key = (
            query.device.type,
            query.device.index,
            self.batch,
            self.num_kv_heads,
            num_parts,
            self.group_size,
            self.head_dim,
        )
        if self._scratch_key != key:
            shape_acc = (
                self.batch,
                self.num_kv_heads,
                num_parts,
                self.group_size,
                self.head_dim,
            )
            shape_stats = (self.batch, self.num_kv_heads, num_parts, self.group_size)
            self._partial_acc = torch.empty(shape_acc, device=query.device, dtype=torch.float32)
            self._partial_m = torch.empty(shape_stats, device=query.device, dtype=torch.float32)
            self._partial_l = torch.empty(shape_stats, device=query.device, dtype=torch.float32)
            self._scratch_key = key
        return self._partial_acc, self._partial_m, self._partial_l

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        out = torch.empty_like(query)
        partition_size = 256
        block_m = 64
        num_parts = triton.cdiv(self.seq_len, partition_size)
        partial_acc, partial_m, partial_l = self._scratch(query, num_parts)

        grid_stage1 = (self.batch, self.num_kv_heads, num_parts)
        _paged_attention_stage1[grid_stage1](
            query,
            kv_cache,
            block_table,
            seq_lens,
            partial_acc,
            partial_m,
            partial_l,
            self.scale,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
            self.group_size,
            block_table.stride(0),
            num_parts,
            partition_size,
            block_m,
            num_warps=4,
        )

        grid_stage2 = (self.batch, self.num_kv_heads)
        _paged_attention_stage2[grid_stage2](
            partial_acc,
            partial_m,
            partial_l,
            out,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.group_size,
            num_parts,
            num_warps=4,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]
