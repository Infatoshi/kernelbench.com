The kernel iterates over pages from the block table, gathers K/V from the paged
KV cache, computes QK^T scores, applies online softmax, and accumulates the
attention output.

Memory layout:
  query:       (batch, num_heads, head_dim)     bf16, contiguous
  kv_cache:    (num_blocks, page_size, num_kv_heads, 2*head_dim)  bf16

# Re-do the kernel with a cleaner design
@triton.jit
def _paged_attention_kernel(
    Q,
    KV_CACHE,
    BLOCK_TABLE,
    SEQ_LENS,
    OUT,
    stride_qb,
    stride_qh,
    stride_kvb,
    stride_kvp,
    stride_kvh,
    stride_bt,
    LOG2_HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
):
    """Single-query paged attention kernel.

    Each program handles one (batch, head) pair. Iterates over all slots
    of the sequence by walking pages from the block table.

    Grid dimension 0 = num_heads, dimension 1 = batch.
    """
    # Program indices
    pid_h = tl.program_id(0)
    pid_b = tl.program_id(1)

    # Head dimension is always a power of 2 (64, 128, 256)
    HEAD_DIM = 1 << LOG2_HEAD_DIM

    # Sequence length for this batch element
    seq_len = tl.load(SEQ_LENS + pid_b)

    # KV head index for GQA: each group of (num_heads/num_kv_heads) Q heads
    # shares the same KV head
    kv_head = pid_h % NUM_KV_HEADS

    # Scale factor for attention: 1/sqrt(HEAD_DIM)
    scale = 1.0 / float(HEAD_DIM).__sqrt__()
    # Triton doesn't have float.__sqrt__ in constexpr context easily;
    # use tl.math.rsq for reciprocal sqrt
    # Actually, tl.exp and tl.log are available. For rsqrt:
    scale = tl.exp(-0.5 * tl.log(float(HEAD_DIM)))

    # Load query: one row of shape (HEAD_DIM,) per program
    q_offsets = tl.arange(0, HEAD_DIM)
    q_ptr = Q + pid_b * stride_qb + pid_h * stride_qh + q_offsets
    q = tl.load(q_ptr, mask=q_offsets < HEAD_DIM, other=0.0).to(tl.float32)

    # Online softmax accumulators
    m = float("-inf")  # running max of logits
    d = 1.0             # running sum of exp(logit - m)
    acc = tl.zeros([HEAD_DIM], dtype=tl.float32)  # accumulated output

    # Number of pages for this sequence
    n_pages = (seq_len + PAGE_SIZE - 1) // PAGE_SIZE

    # Iterate over pages
    page_off = 0  # cumulative page_slot offset
    for page_idx in tl.range(0, n_pages, loops_unroll_min=4):
        # Load page block id from block table
        blk_id = tl.load(BLOCK_TABLE + pid_b * stride_bt + page_idx)

        # Number of valid slots in this page
        remaining = seq_len - page_off
        page_count = min(remaining, PAGE_SIZE)

        # Iterate over slots within this page
        for slot in tl.range(0, page_count):
            # Global slot index
            slot_idx = page_off + slot

            # Compute pointers into KV cache
            # KV layout: [blk_id, slot, kv_head, :]
            # Stride from base: blk_id * stride_kvb + slot * stride_kvp + kv_head * stride_kvh
            base_off = blk_id * stride_kvb + slot * stride_kvp + kv_head * stride_kvh

            # Load key (first HEAD_DIM elements)
            k_ptr = KV_CACHE + base_off + q_offsets
            k = tl.load(k_ptr, mask=q_offsets < HEAD_DIM, other=0.0).to(tl.float32)

            # Load value (next HEAD_DIM elements, offset by HEAD_DIM)
            v_ptr = KV_CACHE + base_off + HEAD_DIM + q_offsets
            v = tl.load(v_ptr, mask=q_offsets < HEAD_DIM, other=0.0).to(tl.float32)

            # QK^T dot product
            logit = tl.sum(q * k, axis=0) * scale

            # Online softmax update
            new_m = max(m, logit)
            # Correction: exp(old_m - new_m)
            old_exp_m = tl.exp(m - new_m)
            d = d * old_exp_m + tl.exp(logit - new_m)
            acc = acc * old_exp_m + v * tl.exp(logit - new_m)
            m = new_m

        page_off += PAGE_SIZE

    # Normalize accumulator and store output
    if seq_len > 0:
        acc = acc / d
        out_ptr = OUT + pid_b * stride_qb + pid_h * stride_qh + q_offsets
        tl.store(out_ptr, acc.to(tl.bfloat16), mask=q_offsets < HEAD_DIM)


def paged_attention_fwd(
    query: torch.Tensor,
    kv_cache: torch.Tensor,
    block_table: torch.Tensor,
    seq_lens: torch.Tensor,
    out: torch.Tensor,
    page_size: int,
):
    """Launch the paged attention Triton kernel."""
    B, H, D = query.shape
    NUM_KV_HEADS = kv_cache.shape[2]

    # HEAD_DIM must be a power of 2 for LOG2_HEAD_DIM
    assert D > 0 and (D & (D - 1)) == 0, f"HEAD_DIM={D} must be a power of 2"
    LOG2_HEAD_DIM = int(math.log2(D))

    assert page_size > 0 and page_size <= 64, f"PAGE_SIZE={page_size} out of range"

    grid = (H, B)

    _paged_attention_kernel[grid](
        query,
        kv_cache,
        block_table,
        seq_lens,
        out,
        query.stride(0),
        query.stride(1),
        kv_cache.stride(0),
        kv_cache.stride(1),
        kv_cache.stride(2),
        block_table.stride(0),
        LOG2_HEAD_DIM=LOG2_HEAD_DIM,
        PAGE_SIZE=page_size,
        NUM_KV_HEADS=NUM_KV_HEADS,
    )


class Model(torch.nn.Module):
    """Paged attention decode using Triton kernel."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)
        paged_attention_fwd(query, kv_cache, block_table, seq_lens, out, self.page_size)
        return out

+    # shares the same KV head
