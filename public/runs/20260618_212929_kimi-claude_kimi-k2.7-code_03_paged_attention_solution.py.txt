"""Triton paged-attention decode kernel for Blackwell (SM100).

Single-query decode with grouped-query attention (GQA), paged KV-cache gather,
and online softmax.  One kernel block processes one query head; the same physical
KV pages are re-loaded for the query heads that share a KV head, which is the
natural GQA behaviour for a memory-bound decode kernel.
"""
import math
from typing import List

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_decode_kernel(
    out_ptr,
    q_ptr,
    kv_cache_ptr,
    block_table_ptr,
    seq_lens_ptr,
    stride_ob,
    stride_oh,
    stride_od,
    stride_qb,
    stride_qh,
    stride_qd,
    stride_kvn,
    stride_kvp,
    stride_kvh,
    stride_btb,
    stride_btp,
    scale,
    num_heads,
    num_kv_heads,
    head_dim,
    GROUP_SIZE: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    BLOCK_D: tl.constexpr,
    MAX_NUM_PAGES: tl.constexpr,
):
    """One block = one (batch, query head)."""
    pid = tl.program_id(0)
    b = pid // num_heads
    h = pid % num_heads
    kv_head = h // GROUP_SIZE

    d_offs = tl.arange(0, BLOCK_D)
    dim_mask = d_offs < head_dim

    # Load query vector for this head.
    q = tl.load(q_ptr + b * stride_qb + h * stride_qh + d_offs, mask=dim_mask, other=0.0)
    q = q[None, :].to(tl.float32)

    L = tl.load(seq_lens_ptr + b)

    # Online-softmax accumulators.
    m = -float('inf')
    l_sum = 0.0
    acc = tl.zeros((1, BLOCK_D), dtype=tl.float32)
    page_offs = tl.arange(0, PAGE_SIZE)

    for p in range(MAX_NUM_PAGES):
        token_start = p * PAGE_SIZE
        token_idx = token_start + page_offs
        valid = token_idx < L

        # Translate logical page -> physical page.
        phys_page = tl.load(block_table_ptr + b * stride_btb + p)
        kv_base = kv_cache_ptr + phys_page * stride_kvn + kv_head * stride_kvh

        # Load K/V tile for the whole page.  Invalid tail tokens are zeroed.
        k = tl.load(
            kv_base + page_offs[:, None] * stride_kvp + d_offs[None, :],
            mask=dim_mask[None, :] & valid[:, None],
            other=0.0,
        ).to(tl.float32)
        v = tl.load(
            kv_base + page_offs[:, None] * stride_kvp + (head_dim + d_offs[None, :]),
            mask=dim_mask[None, :] & valid[:, None],
            other=0.0,
        ).to(tl.float32)

        # scores = q @ K^T  -> (1, PAGE_SIZE)
        scores = tl.dot(q, tl.trans(k), allow_tf32=False) * scale
        scores = tl.where(valid[None, :], scores, -float('inf'))

        # Online softmax update.
        m_prev = m
        m = tl.maximum(m, tl.max(scores))
        alpha = tl.exp(m_prev - m)
        l_sum = l_sum * alpha + tl.sum(tl.exp(scores - m))
        p_softmax = tl.exp(scores - m)

        # Weighted sum of V rows -> (1, BLOCK_D).
        acc = acc * alpha + tl.dot(p_softmax, v, allow_tf32=False)

    out = tl.reshape(acc / l_sum, (BLOCK_D,))
    tl.store(
        out_ptr + b * stride_ob + h * stride_oh + d_offs,
        out.to(tl.bfloat16),
        mask=dim_mask,
    )


class Model(nn.Module):
    """Single-query paged attention decode.

    Forward inputs (all on device):
      query:       (batch, num_heads, head_dim)               bf16
      kv_cache:    (num_blocks, page_size, num_kv_heads, head_dim * 2)
                   Layout: last dim packs [K | V] so a single gather pulls both.
                   Stored as bf16.
      block_table: (batch, max_blocks)                        int32
      seq_lens:    (batch,)                                   int32

    Output:
      attn_out:    (batch, num_heads, head_dim)               bf16
    """

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        self.max_num_pages = (seq_len + page_size - 1) // page_size
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        block_d = triton.next_power_of_2(D)
        grid = (B * H,)
        num_warps = max(2, block_d // 32)

        _paged_decode_kernel[grid](
            out,
            query,
            kv_cache,
            block_table,
            seq_lens,
            out.stride(0),
            out.stride(1),
            out.stride(2),
            query.stride(0),
            query.stride(1),
            query.stride(2),
            kv_cache.stride(0),
            kv_cache.stride(1),
            kv_cache.stride(2),
            block_table.stride(0),
            block_table.stride(1),
            scale=self.scale,
            num_heads=H,
            num_kv_heads=self.num_kv_heads,
            head_dim=D,
            GROUP_SIZE=self.group_size,
            PAGE_SIZE=self.page_size,
            BLOCK_D=block_d,
            MAX_NUM_PAGES=self.max_num_pages,
            num_warps=num_warps,
        )
        return out


def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
