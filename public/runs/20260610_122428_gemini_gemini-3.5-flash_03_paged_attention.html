BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def paged_attention_kernel(
    Q_ptr, KV_cache_ptr, Block_table_ptr, Seq_lens_ptr, Out_ptr,
    stride_q_b, stride_q_h, stride_q_d,
    stride_block, stride_page, stride_h_kv, stride_dim,
    stride_bt_b, stride_bt_p,
    stride_out_b, stride_out_h, stride_out_d,
    scale,
    num_heads, num_kv_heads, max_blocks,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    # Map a 1D grid of size B * H
    pid = tl.program_id(0)
    b = pid // num_heads
    h = pid % num_heads
    
    # h_kv:
    group_size = num_heads // num_kv_heads
    h_kv = h // group_size
    
    # Load seq_len for this batch element:
    seq_len = tl.load(Seq_lens_ptr + b)
    
    cols = tl.arange(0, HEAD_DIM)
    
    if seq_len <= 0:
        out_ptrs = Out_ptr + b * stride_out_b + h * stride_out_h + cols * stride_out_d
        tl.store(out_ptrs, tl.zeros((HEAD_DIM,), dtype=tl.bfloat16))
        return

    # Load query:
    q_ptrs = Q_ptr + b * stride_q_b + h * stride_q_h + cols * stride_q_d
    q = tl.load(q_ptrs).to(tl.float32)
    
    # Initialize online softmax:
    m = tl.full((1,), -3.40282347e+38, dtype=tl.float32)
    d = tl.full((1,), 0.0, dtype=tl.float32)
    acc = tl.zeros((HEAD_DIM,), dtype=tl.float32)
    
    # Loop over seq_len in blocks of BLOCK_N:
    for start_l in range(0, seq_len, BLOCK_N):
        offsets = start_l + tl.arange(0, BLOCK_N)
        mask = offsets < seq_len
        
        # Load block numbers from block_table:
        page_indices = offsets // PAGE_SIZE
        bt_ptrs = Block_table_ptr + b * stride_bt_b + page_indices * stride_bt_p
        block_nums = tl.load(bt_ptrs, mask=mask, other=0)
        
        # page_offsets:
        page_offsets = offsets % PAGE_SIZE
        
        # Load K chunk:
        k_ptrs = KV_cache_ptr + block_nums[:, None] * stride_block + page_offsets[:, None] * stride_page + h_kv * stride_h_kv + cols[None, :] * stride_dim
        K_chunk = tl.load(k_ptrs, mask=mask[:, None], other=0.0).to(tl.float32)
        
        # Compute scores:
        scores = tl.sum(K_chunk * q[None, :], axis=1) * scale # [BLOCK_N]
        scores = tl.where(mask, scores, -3.40282347e+38)
        
        # Update online softmax:
        m_new = tl.maximum(m, tl.max(scores, axis=0)) # [1]
        alpha = tl.exp(m - m_new) # [1]
        exp_scores = tl.exp(scores - m_new) # [BLOCK_N]
        
        # Load V chunk (only after K_chunk is no longer needed):
        v_ptrs = KV_cache_ptr + block_nums[:, None] * stride_block + page_offsets[:, None] * stride_page + h_kv * stride_h_kv + (HEAD_DIM + cols[None, :]) * stride_dim
        V_chunk = tl.load(v_ptrs, mask=mask[:, None], other=0.0).to(tl.float32)
        
        # Update acc and d:
        acc = acc * alpha + tl.sum(exp_scores[:, None] * V_chunk, axis=0) # [HEAD_DIM]
        d = d * alpha + tl.sum(exp_scores, axis=0) # [1]
        m = m_new

    # Final normalization:
    out = acc / d
    
    # Store output:
    out_ptrs = Out_ptr + b * stride_out_b + h * stride_out_h + cols * stride_out_d
    tl.store(out_ptrs, out.to(tl.bfloat16))


class Model(nn.Module):
    """Single-query paged attention decode using Triton.

    Forward inputs (all on device):
      query:       (batch, num_heads, head_dim)               bf16
      kv_cache:    (num_blocks, page_size, num_kv_heads, head_dim * 2)
                   Layout: last dim packs [K | V] so a single gather pulls both.
                   Stored as bf16.
      block_table: (batch, max_blocks)                        int32
      seq_lens:    (batch,)                                   int32

    Output:
      attn_out:    (batch, num_heads, head_dim)               bf16
    """

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        grid = (B * H,)
        
        # Dynamically set BLOCK_N based on concurrent threads to balance latency hiding vs registers
        if B * H >= 512:
            BLOCK_N = 128
        else:
            BLOCK_N = 256

        paged_attention_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0), block_table.stride(1),
            out.stride(0), out.stride(1), out.stride(2),
            self.scale,
            self.num_heads, self.num_kv_heads, block_table.shape[1],
            HEAD_DIM=self.head_dim,
            PAGE_SIZE=self.page_size,
            BLOCK_N=BLOCK_N,
        )

        return out


def get_inputs():

    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    # Keep the global pool larger than strictly needed and shuffle assignments
    # so the block_table actually exercises non-contiguous gather.
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    # Pad to pages_per_seq columns; for fixed-seq-len shapes this is exact.
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]
