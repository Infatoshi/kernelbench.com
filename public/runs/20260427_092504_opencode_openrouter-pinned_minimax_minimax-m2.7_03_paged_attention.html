 Implements single-query paged attention decode using CUDA C++ via load_inline.
"""

SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024

CUDA_SOURCE = r"""
#include <torch/extension.h>
#include <cuda_bf16.h>
#include <cuda_runtime.h>
#include <device_launch_parameters.h>

using bfloat16 = __nv_bfloat16;

__device__ inline float bf16_to_float(bfloat16 x) {
    return __bfloat162float(x);
}

__device__ inline bfloat16 float_to_bf16(float x) {
    return __float2bfloat16(x);
}

template <int THREADS>
__device__ inline float warp_reduce_sum(float val) {
    #pragma unroll
    for (int mask = THREADS >> 1; mask > 0; mask >>= 1) {
        val += __shfl_xor_sync(0xffffffff, val, mask);
    }
    return val;
}

template <int HEAD_DIM, int MAX_PAGE_SIZE>
__global__ void paged_attention_kernel(
    const bfloat16* __restrict__ query,
    const bfloat16* __restrict__ kv_cache,
    const int* __restrict__ block_table,
    const int* __restrict__ seq_lens,
    bfloat16* __restrict__ output,
    float scale,
    int batch,
    int num_heads,
    int num_kv_heads,
    int group_size,
    int max_blocks_per_seq,
    int page_size
) {
    int batch_idx = blockIdx.x;
    int head_idx = blockIdx.y * blockDim.y + threadIdx.y;

    if (batch_idx >= batch || head_idx >= num_heads) return;

    int kv_head_idx = head_idx / group_size;
    int seq_len = seq_lens[batch_idx];
    int num_pages = (seq_len + page_size - 1) / page_size;

    extern __shared__ char smem[];
    bfloat16* smem_kv = reinterpret_cast<bfloat16*>(smem);
    bfloat16* smem_scores = reinterpret_cast<bfloat16*>(smem + 2 * MAX_PAGE_SIZE * HEAD_DIM * sizeof(bfloat16));

    const bfloat16* q_ptr = query + batch_idx * num_heads * HEAD_DIM + head_idx * HEAD_DIM;

    float q_f[HEAD_DIM];
    for (int i = 0; i < HEAD_DIM; i++) {
        q_f[i] = bf16_to_float(q_ptr[i]);
    }

    float max_score = -1e20f;
    float sum_exp = 0.0f;
    float output_f[HEAD_DIM];
    for (int i = 0; i < HEAD_DIM; i++) {
        output_f[i] = 0.0f;
    }

    for (int page_idx = 0; page_idx < num_pages; page_idx++) {
        int page_num = block_table[batch_idx * max_blocks_per_seq + page_idx];
        int tokens_in_page = (page_idx == num_pages - 1) ? ((seq_len - 1) % page_size + 1) : page_size;

        for (int tid = threadIdx.x; tid < tokens_in_page * HEAD_DIM; tid += blockDim.x) {
            int token_in_page = tid / HEAD_DIM;
            int dim = tid % HEAD_DIM;
            int cache_idx = page_num * page_size * num_kv_heads * HEAD_DIM * 2 +
                           token_in_page * num_kv_heads * HEAD_DIM * 2 +
                           kv_head_idx * HEAD_DIM * 2 +
                           dim;
            smem_kv[token_in_page * HEAD_DIM + dim] = kv_cache[cache_idx];
            smem_kv[MAX_PAGE_SIZE * HEAD_DIM + token_in_page * HEAD_DIM + dim] = kv_cache[cache_idx + HEAD_DIM];
        }

        __syncthreads();

        for (int token_local = threadIdx.y; token_local < tokens_in_page; token_local += blockDim.y) {
            int token_abs = page_idx * page_size + token_local;
            if (token_abs >= seq_len) continue;

            bfloat16* k_vec = smem_kv + token_local * HEAD_DIM;

            float score = 0.0f;
            for (int dim = threadIdx.x; dim < HEAD_DIM; dim += blockDim.x) {
                score += q_f[dim] * bf16_to_float(k_vec[dim]);
            }
            score = warp_reduce_sum<32>(score);

            float exp_s = 0.0f;
            if (threadIdx.x == 0) {
                exp_s = expf(score * scale - max_score);
                smem_scores[token_local] = float_to_bf16(exp_s);
            }

            float weight = warp_reduce_sum<32>(exp_s);
            if (threadIdx.x == 0) {
                sum_exp += weight;
            }

            bfloat16* v_vec = smem_kv + MAX_PAGE_SIZE * HEAD_DIM + token_local * HEAD_DIM;
            for (int dim = threadIdx.x; dim < HEAD_DIM; dim += blockDim.x) {
                output_f[dim] += weight * bf16_to_float(v_vec[dim]);
            }
        }

        __syncthreads();
    }

    if (sum_exp > 0.0f) {
        float inv_sum_exp = 1.0f / sum_exp;
        for (int dim = threadIdx.x; dim < HEAD_DIM; dim += blockDim.x) {
            output_f[dim] *= inv_sum_exp;
        }
    }

    if (threadIdx.x == 0 && threadIdx.y == 0) {
        bfloat16* out_ptr = output + batch_idx * num_heads * HEAD_DIM + head_idx * HEAD_DIM;
        for (int dim = 0; dim < HEAD_DIM; dim++) {
            out_ptr[dim] = float_to_bf16(output_f[dim]);
        }
    }
}

torch::Tensor paged_attention_forward_impl(
    torch::Tensor query,
    torch::Tensor kv_cache,
    torch::Tensor block_table,
    torch::Tensor seq_lens,
    int num_heads,
    int num_kv_heads,
    int head_dim,
    int page_size
) {
    int batch = query.size(0);
    int group_size = num_heads / num_kv_heads;
    int max_blocks_per_seq = block_table.size(1);

    auto output = torch::empty_like(query);

    const int BLOCK_SIZE = 128;
    const int NUM_THREADS_Y = 4;
    const int MAX_PAGE_SIZE = 16;

    dim3 block(BLOCK_SIZE, NUM_THREADS_Y);
    dim3 grid(batch, (num_heads + NUM_THREADS_Y - 1) / NUM_THREADS_Y);

    int smem_size = 2 * MAX_PAGE_SIZE * head_dim * sizeof(bfloat16) + MAX_PAGE_SIZE * sizeof(bfloat16);

    float scale = 1.0f / sqrtf(static_cast<float>(head_dim));

    paged_attention_kernel<128, 16><<<grid, block, smem_size>>>(
        static_cast<bfloat16*>(query.data_ptr()),
        static_cast<bfloat16*>(kv_cache.data_ptr()),
        static_cast<int*>(block_table.data_ptr()),
        static_cast<int*>(seq_lens.data_ptr()),
        static_cast<bfloat16*>(output.data_ptr()),
        scale,
        batch,
        num_heads,
        num_kv_heads,
        group_size,
        max_blocks_per_seq,
        page_size
    );

    return output;
}

def _get_cuda_module():
    global _cached_module
    if _cached_module is None:
        _cached_module = load_inline(
            name="paged_attention_cuda",
            cpp_sources=CPP_SOURCE,

class Model(nn.Module):
    """Single-query paged attention decode.

    Forward inputs (all on device):
      query:       (batch, num_heads, head_dim)               bf16
      kv_cache:    (num_blocks, page_size, num_kv_heads, head_dim * 2)
                   Layout: last dim packs [K | V] so a single gather pulls both.
                   Stored as bf16.
      block_table: (batch, max_blocks)                        int32
      seq_lens:    (batch,)                                   int32

    Output:
      attn_out:    (batch, num_heads, head_dim)               bf16
    """

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads

        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        cuda_mod = _get_cuda_module()
        output = cuda_mod.paged_attention_forward(
            query,
            kv_cache,
            block_table,
            seq_lens,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
        )
        return output
