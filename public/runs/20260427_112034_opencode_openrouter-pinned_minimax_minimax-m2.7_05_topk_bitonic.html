"""Bitonic sort top-k CUDA kernel.

Implements top-k using a bitonic sort network in shared memory.
- k=1: warp reduction argmax (very fast)
- k>1: bitonic sort network, pad non-power-of-2 n to next power of 2
- One block per row for good SM utilization on large-n cases
"""
import math
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

cpp_source = r"""
#include <torch/extension.h>
#include <cuda/std/atomic>

#define INF_NEG __int_as_float(0xff800000)  // -infinity

__device__ inline float warp_reduce_max_idx(float val, int idx, int* max_idx) {
    for (int offset = 16; offset > 0; offset /= 2) {
        float other_val = __shfl_down_sync(0xffffffff, val, offset);
        int other_idx = __shfl_down_sync(0xffffffff, idx, offset);
        if (other_val > val || (other_val == val && other_idx < *max_idx)) {
            val = other_val;
            *max_idx = other_idx;
        }
    }
    return val;
}

__global__ void topk_argmax_kernel(
    const float* __restrict__ input,
    float* __restrict__ output_val,
    int64_t* __restrict__ output_idx,
    int batch, int n, int k
) {
    int row = blockIdx.x;
    if (row >= batch) return;

    const float* row_data = input + row * n;
    int tid = threadIdx.x;
    int num_threads = blockDim.x;

    __shared__ float shared[256];
    __shared__ int indices[256];

    if (tid < 256 && tid < n) {
        shared[tid] = row_data[tid];
        indices[tid] = tid;
    } else if (tid < 256) {
        shared[tid] = INF_NEG;
        indices[tid] = -1;
    }
    __syncthreads();

    for (int i = tid + 256; i < n; i += 256) {
        if (row_data[i] > shared[0] || (row_data[i] == shared[0] && i < indices[0])) {
            shared[0] = row_data[i];
            indices[0] = i;
        }
    }
    __syncthreads();

    if (n <= 256) {
        if (tid == 0) {
            float val = shared[0];
            int idx = indices[0];
            for (int i = 1; i < n && i < 256; i++) {
                if (shared[i] > val || (shared[i] == val && indices[i] < idx)) {
                    val = shared[i];
                    idx = indices[i];
                }
            }
            output_val[0] = val;
            output_idx[0] = idx;
        }
        return;
    }

    for (int base = 0; base < n; base += 256) {
        int local_n = min(256, n - base);
        if (tid < local_n) {
            shared[tid] = row_data[base + tid];
            indices[tid] = base + tid;
        }
        __syncthreads();

        for (int offset = 128; offset > 32; offset /= 2) {
            if (tid + offset < local_n) {
                if (shared[tid + offset] > shared[tid] ||
                    (shared[tid + offset] == shared[tid] && indices[tid + offset] < indices[tid])) {
                    shared[tid] = shared[tid + offset];
                    indices[tid] = indices[tid + offset];
                }
            }
            __syncthreads();
        }

        if (tid < 32) {
            int my_idx = indices[tid];
            float my_val = shared[tid];
            if (tid + 32 < local_n) {
                if (shared[tid + 32] > my_val ||
                    (shared[tid + 32] == my_val && indices[tid + 32] < my_idx)) {
                    my_val = shared[tid + 32];
                    my_idx = indices[tid + 32];
                }
            }
            my_val = warp_reduce_max_idx(my_val, my_idx, &my_idx);
            if (tid == 0) {
                shared[0] = my_val;
                indices[0] = my_idx;
            }
        }
        __syncthreads();

        if (tid == 0) {
            if (shared[0] > row_data[indices[0]] ||
                (shared[0] == row_data[indices[0]] && indices[0] < indices[0])) {
            }
            if (base == 0 || shared[0] > output_val[0] ||
                (shared[0] == output_val[0] && indices[0] < output_idx[0])) {
                output_val[0] = shared[0];
                output_idx[0] = indices[0];
            }
        }
    }

    if (tid == 0 && n > 256) {
        float val = shared[0];
        int idx = indices[0];
        val = warp_reduce_max_idx(val, idx, &idx);
        output_val[0] = val;
        output_idx[0] = idx;
    }
}

template <typename T>
__device__ inline void swap_if_greater(T* a, T* b, int* idx_a, int* idx_b, bool asc) {
    if (asc) {
        if (**a > **b || (**a == **b && *idx_a > *idx_b)) {
            T tmp = **a; **a = **b; **b = tmp;
            int tmp_idx = *idx_a; *idx_a = *idx_b; *idx_b = tmp_idx;
        }
    } else {
        if (**a < **b || (**a == **b && *idx_a > *idx_b)) {
            T tmp = **a; **a = **b; **b = tmp;
            int tmp_idx = *idx_a; *idx_a = *idx_b; *idx_b = tmp_idx;
        }
    }
}

__device__ inline void bitonic_step(float* a, float* b, int* idx_a, int* idx_b, int d, int j) {
    bool ascending = ((j & d) == 0);
    if (ascending) {
        if (*a < *b || (*a == *b && *idx_a > *idx_b)) {
            float tmp = *a; *a = *b; *b = tmp;
            int tmp_idx = *idx_a; *idx_a = *idx_b; *idx_b = tmp_idx;
        }
    } else {
        if (*a > *b || (*a == *b && *idx_a > *idx_b)) {
            float tmp = *a; *a = *b; *b = tmp;
            int tmp_idx = *idx_a; *idx_a = *idx_b; *idx_b = tmp_idx;
        }
    }
}

__global__ void topk_bitonic_kernel(
    const float* __restrict__ input,
    float* __restrict__ output_val,
    int64_t* __restrict__ output_idx,
    int batch, int n, int k
) {
    int row = blockIdx.x;
    if (row >= batch) return;

    int padded_n = gridDim.y;
    int original_n = n;

    extern __shared__ float shared[];
    float* s_values = shared;
    int* s_indices = (int*)(s_values + padded_n);

    int tid = threadIdx.x;
    int num_threads = blockDim.x;

    for (int i = tid; i < padded_n; i += num_threads) {
        if (i < original_n) {
            s_values[i] = input[row * original_n + i];
            s_indices[i] = i;
        } else {
            s_values[i] = INF_NEG;
            s_indices[i] = -1;
        }
    }
    __syncthreads();

    for (int d = padded_n / 2; d > 0; d /= 2) {
        for (int dd = d; dd > 0; dd /= 2) {
            int j = tid;
            int i = j ^ dd;
            if (i > j && i < padded_n && j < padded_n) {
                float a = s_values[j];
                float b = s_values[i];
                int idx_a = s_indices[j];
                int idx_b = s_indices[i];
                bitonic_step(&s_values[j], &s_values[i], &s_indices[j], &s_indices[i], d, j);
            }
            __syncthreads();
        }
    }

    if (tid < k) {
        output_val[row * k + tid] = s_values[tid];
        output_idx[row * k + tid] = s_indices[tid];
    }
}

__global__ void topk_small_n_kernel(
    const float* __restrict__ input,
    float* __restrict__ output_val,
    int64_t* __restrict__ output_idx,
    int batch, int n, int k
) {
    int row = blockIdx.x;
    if (row >= batch) return;
    if (k == 1) {
        int tid = threadIdx.x;
        int num_threads = blockDim.x;
        __shared__ float s_val;
        __shared__ int s_idx;

        if (tid == 0) {
            s_val = INF_NEG;
            s_idx = -1;
        }
        __syncthreads();

        for (int i = tid; i < n; i += num_threads) {
            float val = input[row * n + i];
            int idx = i;
            float expected = s_val;
            int expected_idx = s_idx;

            if (val > expected || (val == expected && idx < expected_idx)) {
                s_val = val;
                s_idx = idx;
            }
        }
        __syncthreads();

        if (tid == 0) {
            output_val[row * k] = s_val;
            output_idx[row * k] = s_idx;
        }
        return;
    }

    extern __shared__ float shared[];
    float* s_values = shared;
    int* s_indices = (int*)(shared + n);

    int tid = threadIdx.x;

    if (tid < n) {
        s_values[tid] = input[row * n + tid];
        s_indices[tid] = tid;
    }
    __syncthreads();

    for (int d = n / 2; d > 0; d /= 2) {
        for (int dd = d; dd > 0; dd /= 2) {
            int j = tid;
            int i = j ^ dd;
            if (i > j && i < n && j < n) {
                float a = s_values[j];
                float b = s_values[i];
                bitonic_step(&s_values[j], &s_values[i], &s_indices[j], &s_indices[i], d, j);
            }
            __syncthreads();
        }
    }

    if (tid < k) {
        output_val[row * k + tid] = s_values[tid];
        output_idx[row * k + tid] = s_indices[tid];
    }
}

void topk_launcher(
    const float* input,
    float* output_val,
    int64_t* output_idx,
    int batch, int n, int k,
    cudaStream_t stream
) {
    if (k == 1) {
        int threads = n > 256 ? 256 : 32;
        while (threads < n && threads < 1024) threads *= 2;
        threads = min(threads, 1024);
        topk_argmax_kernel<<<batch, threads, 0, stream>>>(
            input, output_val, output_idx, batch, n, k);
    } else if (n <= 512) {
        topk_small_n_kernel<<<batch, 512, 512 * (sizeof(float) + sizeof(int)), stream>>>(
            input, output_val, output_idx, batch, n, k);
    } else {
        int padded_n = 1;
        while (padded_n < n) padded_n *= 2;
        int threads = 256;
        while (threads < padded_n && threads < 1024) threads *= 2;
        topk_bitonic_kernel<<<batch, threads, padded_n * (sizeof(float) + sizeof(int)), stream>>>(
            input, output_val, output_idx, batch, n, k);
    }
}

    int batch = input.size(0);

    auto output_val = torch::empty({batch, k}, input.options());
    auto output_idx = torch::empty({batch, k}, input.options().dtype(torch::kInt64));

        output_val.data_ptr<float>(),
        output_idx.data_ptr<int64_t>(),

    );

}
