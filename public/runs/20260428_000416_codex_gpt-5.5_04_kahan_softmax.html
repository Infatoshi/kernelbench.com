@triton.jit
def _single_block_kernel(
    x_ptr,
    y_ptr,
    stride_x: tl.constexpr,
    stride_y: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(x_ptr + row * stride_x + offs, mask=mask, other=-float("inf"), cache_modifier=".ca")
    m = tl.max(x, axis=0)
    e = tl.exp(x - m)
    e = tl.where(mask, e, 0.0)
    s = tl.sum(e, axis=0)
    y = e / s

    tl.store(y_ptr + row * stride_y + offs, y, mask=mask, cache_modifier=".cs")


@triton.jit
def _block_stats_kernel(
    x_ptr,
    block_max_ptr,
    block_sum_ptr,
    stride_x: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    block = tl.program_id(1)
    num_blocks = tl.cdiv(n_cols, BLOCK_SIZE)
    offs = block * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols

    x = tl.load(x_ptr + row * stride_x + offs, mask=mask, other=-float("inf"), cache_modifier=".ca")
    m = tl.max(x, axis=0)
    e = tl.exp(x - m)
    e = tl.where(mask, e, 0.0)
    s = tl.sum(e, axis=0)

    out = row * num_blocks + block
    tl.store(block_max_ptr + out, m)
    tl.store(block_sum_ptr + out, s)


@triton.jit
def _normalize_kernel(
    x_ptr,
    y_ptr,
    block_max_ptr,
    block_sum_ptr,
    stride_x: tl.constexpr,
    stride_y: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
    NUM_BLOCKS: tl.constexpr,
):
    row = tl.program_id(0)
    block = tl.program_id(1)
    base = row * NUM_BLOCKS

    m = -float("inf")
    for k in tl.static_range(0, NUM_BLOCKS):
        m = tl.maximum(m, tl.load(block_max_ptr + base + k))

    s = 0.0
    for k in tl.static_range(0, NUM_BLOCKS):
        bm = tl.load(block_max_ptr + base + k)
        bs = tl.load(block_sum_ptr + base + k) * tl.exp(bm - m)
        s += bs

    offs = block * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols
    x = tl.load(x_ptr + row * stride_x + offs, mask=mask, other=0.0, cache_modifier=".ca")
    y = tl.exp(x - m) / s
    tl.store(y_ptr + row * stride_y + offs, y, mask=mask, cache_modifier=".cs")


class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = batch
        self.vocab = vocab
        self._graph = None
        self._graph_key = None
        self._graph_y = None
        self._graph_stats = None
        self._block_size = 4096 if vocab <= 32768 else 8192
        self._num_blocks = triton.cdiv(vocab, self._block_size)
        if vocab <= 4096:
            self._warps_single = 4
        else:
            self._warps_single = 8
        if vocab >= 131072:
            self._warps_block = 4
            self._warps_stats = 8
        elif vocab >= 32768:
            self._warps_block = 8
            self._warps_stats = 4
        else:
            self._warps_block = 8
            self._warps_stats = 4

    def _launch(self, x: torch.Tensor, y: torch.Tensor, stats: torch.Tensor | None) -> None:
        if self._num_blocks == 1:
            _single_block_kernel[(self.batch,)](
                x,
                y,
                stride_x=x.stride(0),
                stride_y=y.stride(0),
                n_cols=self.vocab,
                BLOCK_SIZE=self._block_size,
                num_warps=self._warps_single,
            )
            return

        grid = (self.batch, self._num_blocks)
        _block_stats_kernel[grid](
            x,
            stats[0],
            stats[1],
            stride_x=x.stride(0),
            n_cols=self.vocab,
            BLOCK_SIZE=self._block_size,
            num_warps=self._warps_stats,
        )
        _normalize_kernel[grid](
            x,
            y,
            stats[0],
            stats[1],
            stride_x=x.stride(0),
            stride_y=y.stride(0),
            n_cols=self.vocab,
            BLOCK_SIZE=self._block_size,
            NUM_BLOCKS=self._num_blocks,
            num_warps=self._warps_block,
        )

    def _allocate_outputs(self, x: torch.Tensor) -> tuple[torch.Tensor, torch.Tensor | None]:
        y = torch.empty_like(x)
        if self._num_blocks == 1:
            return y, None
        stats = torch.empty((2, self.batch, self._num_blocks), dtype=torch.float32, device=x.device)
        return y, stats

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        key = (x.data_ptr(), x.stride(0), x.stride(1), self.batch, self.vocab)
        if self._graph_key != key:
            y, stats = self._allocate_outputs(x)
            self._launch(x, y, stats)
            torch.cuda.synchronize()
            graph = torch.cuda.CUDAGraph()
            with torch.cuda.graph(graph):
                self._launch(x, y, stats)
            self._graph = graph
            self._graph_key = key
            self._graph_y = y
            self._graph_stats = stats
            return y

        self._graph.replay()
        return self._graph_y
