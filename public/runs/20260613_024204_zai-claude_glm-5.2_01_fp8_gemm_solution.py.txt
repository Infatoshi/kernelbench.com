"""Custom FP8 e4m3 GEMM for RTX PRO 6000 (SM120 Blackwell).

y = x @ w.T  with x fp8_e4m3 (M,K), w fp8_e4m3 (N,K), output bf16 (M,N).

Implemented as a Triton kernel. Inputs hit the tensor cores via tl.dot on fp8
tiles, accumulating in fp32. Because the bf16 reference weight has values that
fall in fp8's subnormal region, we quantize the weight per-tensor (amax / 448)
and rescale the fp32 accumulator before the bf16 store. The weight quantization
is cached and invalidated via Parameter._version (free in steady state, robust
to the correctness harness's in-place weight rescaling).
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


def _configs():
    cfgs = []
    # Compute-bound tiles (square / rectangular). fp8 MMA on Blackwell wants
    # num_warps=8 for these tile sizes.
    for (bm, bn, bk) in [
        (128, 128, 128),
        (128, 256, 128),
        (128, 256, 64),
        (256, 128, 128),
        (128, 128, 64),
    ]:
        for st in (3, 4):
            cfgs.append(triton.Config({"BM": bm, "BN": bn, "BK": bk, "GROUP": 8},
                                      num_warps=8, num_stages=st))
    # Skinny-M tiles (decode-style). Smaller M tile, can use fewer warps.
    for (bm, bn, bk) in [(16, 128, 128), (32, 128, 128), (32, 256, 128), (64, 128, 128)]:
        for nw in (4, 8):
            cfgs.append(triton.Config({"BM": bm, "BN": bn, "BK": bk, "GROUP": 8},
                                      num_warps=nw, num_stages=3))
    return cfgs


@triton.autotune(configs=_configs(), key=["M", "N", "K"])
@triton.jit
def _gemm_kernel(
    a_ptr, b_ptr, c_ptr, scale_w,
    M, N, K,
    s_am, s_ak, s_bn, s_bk, s_cm, s_cn,
    BM: tl.constexpr, BN: tl.constexpr, BK: tl.constexpr, GROUP: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BM)
    num_pid_n = tl.cdiv(N, BN)
    num_pid_in_group = GROUP * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP
    group_size_m = min(num_pid_m - first_pid_m, GROUP)
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = pid_m * BM + tl.arange(0, BM)
    offs_n = pid_n * BN + tl.arange(0, BN)
    offs_k = tl.arange(0, BK)
    a_ptrs = a_ptr + offs_m[:, None] * s_am + offs_k[None, :] * s_ak
    b_ptrs = b_ptr + offs_k[:, None] * s_bk + offs_n[None, :] * s_bn

    acc = tl.zeros((BM, BN), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BK)):
        krem = K - k * BK
        a = tl.load(a_ptrs, mask=offs_k[None, :] < krem, other=0.0)
        b = tl.load(b_ptrs, mask=offs_k[:, None] < krem, other=0.0)
        acc = tl.dot(a, b, acc)
        a_ptrs += BK * s_ak
        b_ptrs += BK * s_bk

    acc = acc * scale_w
    c = acc.to(tl.bfloat16)
    c_ptrs = c_ptr + offs_m[:, None] * s_cm + offs_n[None, :] * s_cn
    mask = (offs_m[:, None] < M) & (offs_n[None, :] < N)
    tl.store(c_ptrs, c, mask=mask)


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        # Match the reference parameter exactly (bf16, (N,K)) for state_dict.
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        self._cache = None  # (version, w_fp8, scale_w)

    def _prepare(self):
        w = self.weight
        v = w._version
        if self._cache is None or self._cache[0] != v:
            with torch.no_grad():
                amax = w.abs().amax().clamp_min(1e-12)
                scale = (amax / 448.0).item()
                wq = (w.float() / scale).clamp(-448.0, 448.0).to(torch.float8_e4m3fn)
            self._cache = (v, wq, scale)
        return self._cache[1], self._cache[2]

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w_fp8, scale_w = self._prepare()
        M, N = x.shape[0], self.N
        out = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)
        grid = lambda meta: (triton.cdiv(M, meta["BM"]) * triton.cdiv(N, meta["BN"]),)
        _gemm_kernel[grid](
            x, w_fp8, out, scale_w,
            M, N, self.K,
            x.stride(0), x.stride(1),
            w_fp8.stride(0), w_fp8.stride(1),
            out.stride(0), out.stride(1),
        )
        return out


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
