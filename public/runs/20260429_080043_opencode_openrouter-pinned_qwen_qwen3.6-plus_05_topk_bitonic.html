"""Fast top-k CUDA kernel for RTX PRO 6000 Blackwell (SM120).

Strategy: per-row block reduction. Each block handles one row. Threads find
local top-k in registers, then one thread merges all local top-k via shared mem.
"""
import torch
import torch.nn as nn
import torch.utils.cpp_extension

CPP_SOURCE = """
#include <torch/extension.h>

void topk_launch(
    const float* input,
    float* values,
    int64_t* indices,
    int batch,
    int n,
    int k,
    int threads_per_block
);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("topk_launch", &topk_launch, "topk_launch");
}
"""

CUDA_SOURCE = r"""
#include <cuda_runtime.h>
#include <c10/cuda/CUDAStream.h>
#include <torch/extension.h>

#define INF_NEG -3.4028235e+38f

template <int K>
__global__ void topk_kernel(
    const float* __restrict__ input,
    float* __restrict__ values,
    int64_t* __restrict__ indices,
    int batch,
    int n
)
{
    extern __shared__ char smem[];
    float* s_vals = (float*)smem;
    int* s_idxs = (int*)(s_vals + blockDim.x * K);

    int row = blockIdx.x;
    if (row >= batch) return;

    const float* row_ptr = input + row * n;
    int tid = threadIdx.x;
    int nthreads = blockDim.x;

    // Thread-local top-k in registers
    float t_vals[K];
    int t_idxs[K];
    #pragma unroll
    for (int i = 0; i < K; i++) {
        t_vals[i] = INF_NEG;
        t_idxs[i] = -1;
    }

    // Grid-stride loop over elements
    for (int idx = tid; idx < n; idx += nthreads) {
        float val = row_ptr[idx];
        #pragma unroll
        for (int j = 0; j < K; j++) {
            if (val > t_vals[j]) {
                #pragma unroll
                for (int l = K-1; l > j; l--) {
                    t_vals[l] = t_vals[l-1];
                    t_idxs[l] = t_idxs[l-1];
                }
                t_vals[j] = val;
                t_idxs[j] = idx;
                break;
            }
        }
    }

    // Write thread's top-k to shared memory
    #pragma unroll
    for (int i = 0; i < K; i++) {
        s_vals[tid * K + i] = t_vals[i];
        s_idxs[tid * K + i] = t_idxs[i];
    }
    __syncthreads();

    // Thread 0 merges all threads' top-k into final answer
    if (tid == 0) {
        float g_vals[K];
        int g_idxs[K];
        #pragma unroll
        for (int i = 0; i < K; i++) {
            g_vals[i] = INF_NEG;
            g_idxs[i] = -1;
        }

        for (int t = 0; t < nthreads; t++) {
            #pragma unroll
            for (int i = 0; i < K; i++) {
                float val = s_vals[t * K + i];
                int idx = s_idxs[t * K + i];
                #pragma unroll
                for (int j = 0; j < K; j++) {
                    if (val > g_vals[j]) {
                        #pragma unroll
                        for (int l = K-1; l > j; l--) {
                            g_vals[l] = g_vals[l-1];
                            g_idxs[l] = g_idxs[l-1];
                        }
                        g_vals[j] = val;
                        g_idxs[j] = idx;
                        break;
                    }
                }
            }
        }

        for (int i = 0; i < K; i++) {
            values[row * K + i] = g_vals[i];
            indices[row * K + i] = (int64_t)g_idxs[i];
        }
    }
}

// Launcher with K dispatch
void topk_launch(
    const float* input,
    float* values,
    int64_t* indices,
    int batch,
    int n,
    int k,
    int threads_per_block
)
{
    // Round k up to next power of 2
    int k_pow2 = 1;
    while (k_pow2 < k) k_pow2 *= 2;

    size_t smem_sz = k_pow2 * (sizeof(float) + sizeof(int)) * threads_per_block;

    auto stream = c10::cuda::getCurrentCUDAStream();

    switch (k_pow2) {
    case 1:
        topk_kernel<1><<<batch, threads_per_block, smem_sz, stream>>>(
            input, values, indices, batch, n);
        break;
    case 2:
        topk_kernel<2><<<batch, threads_per_block, smem_sz, stream>>>(
            input, values, indices, batch, n);
        break;
    case 4:
        topk_kernel<4><<<batch, threads_per_block, smem_sz, stream>>>(
            input, values, indices, batch, n);
        break;
    case 8:
        topk_kernel<8><<<batch, threads_per_block, smem_sz, stream>>>(
            input, values, indices, batch, n);
        break;
    case 16:
        topk_kernel<16><<<batch, threads_per_block, smem_sz, stream>>>(

        break;
    case 32:

        break;
    case 64:

        break;
    default:
        break;
    }
}
"""
