@triton.jit
def _grouped_gemm_swiglu_kernel(
    hidden_ptr,
    w_gate_ptr,
    w_up_ptr,
    offsets_ptr,
    out_ptr,
    T_perm,
    H,
    I,
    E,
    max_tokens,
    stride_hs_m,
    stride_hs_k,
    stride_wg_e,
    stride_wg_k,
    stride_wg_n,
    stride_wu_e,
    stride_wu_k,
    stride_wu_n,
    stride_out_m,
    stride_out_n,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_m_blocks = tl.cdiv(max_tokens, BLOCK_M)
    num_n_blocks = tl.cdiv(I, BLOCK_N)

    e = pid // (num_m_blocks * num_n_blocks)
    rest = pid % (num_m_blocks * num_n_blocks)
    m_block = rest // num_n_blocks
    n_block = rest % num_n_blocks

    if e >= E:
        return

    off_e_start = tl.load(offsets_ptr + e)
    off_e_end = tl.load(offsets_ptr + e + 1)
    n_e = off_e_end - off_e_start

    m_start = m_block * BLOCK_M
    if m_start >= n_e:
        return

    n_start = n_block * BLOCK_N
    if n_start >= I:
        return

    m_offs = m_start + tl.arange(0, BLOCK_M)
    mask_m = m_offs < n_e
    n_offs = n_start + tl.arange(0, BLOCK_N)
    mask_n = n_offs < I

    gate_acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    up_acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    hs_base = off_e_start * stride_hs_m
    wg_e_base = e * stride_wg_e
    wu_e_base = e * stride_wu_e

    for k_start in range(0, H, BLOCK_K):
        k_offs = k_start + tl.arange(0, BLOCK_K)
        mask_k = k_offs < H

        a = tl.load(
            hidden_ptr + hs_base + m_offs[:, None] * stride_hs_m + k_offs[None, :] * stride_hs_k,
            mask=mask_m[:, None] & mask_k[None, :],
            other=0.0,
        )
        b_gate = tl.load(
            w_gate_ptr + wg_e_base + k_offs[:, None] * stride_wg_k + n_offs[None, :] * stride_wg_n,
            mask=mask_k[:, None] & mask_n[None, :],
            other=0.0,
        )
        b_up = tl.load(
            w_up_ptr + wu_e_base + k_offs[:, None] * stride_wu_k + n_offs[None, :] * stride_wu_n,
            mask=mask_k[:, None] & mask_n[None, :],
            other=0.0,
        )

        gate_acc = tl.dot(a, b_gate, gate_acc)
        up_acc = tl.dot(a, b_up, up_acc)

    result = gate_acc * tl.sigmoid(gate_acc) * up_acc

    tl.store(
        out_ptr + off_e_start * stride_out_m + m_offs[:, None] * stride_out_m + n_offs[None, :] * stride_out_n,
        result.to(tl.bfloat16),
        mask=mask_m[:, None] & mask_n[None, :],
    )
