"""Paged attention decode – GQA-aware Triton kernel with split+reduce and CUDA graph."""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _paged_attn_split_kernel(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr,
    MI_ptr, LI_ptr, ACC_ptr,
    scale,
    H: tl.constexpr, Hkv: tl.constexpr,
    D: tl.constexpr, P: tl.constexpr,
    S: tl.constexpr,
    max_pages,
    sq_b, sq_h, sq_d,
    skv_blk, skv_p, skv_hkv, skv_d,
    sbt_b, sbt_p,
    smi_stride, sli_stride,
    sacc_b, sacc_kvh, sacc_s, sacc_d,
    NUM_STAGES: tl.constexpr,
):
    pid = tl.program_id(0)
    total_splits = Hkv * S
    bid = pid // total_splits
    rem = pid % total_splits
    kv_hid = rem // S
    sid = rem % S

    G: tl.constexpr = H // Hkv

    seq_len_b = tl.load(SL_ptr + bid).to(tl.int32)
    total_pages_seq = (seq_len_b + P - 1) // P
    pages_per_split = (total_pages_seq + S - 1) // S
    start_page = sid * pages_per_split
    end_page = min(start_page + pages_per_split, total_pages_seq)

    d_offs = tl.arange(0, D)
    g_offs = tl.arange(0, G)
    h_offs = kv_hid * G + g_offs

    q = tl.load(
        Q_ptr + bid * sq_b + h_offs[:, None] * sq_h + d_offs[None, :] * sq_d,
    ).to(tl.float32)

    m_val = tl.full((G,), value=-1.0e10, dtype=tl.float32)
    l_val = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, D), dtype=tl.float32)

    p_offs = tl.arange(0, P)

    for page_idx in tl.range(start_page, max_pages, num_stages=NUM_STAGES):
        if page_idx < end_page:
            phys = tl.load(BT_ptr + bid * sbt_b + page_idx * sbt_p).to(tl.int64)

            k_base = KV_ptr + phys * skv_blk + kv_hid * skv_hkv
            k = tl.load(
                k_base + p_offs[:, None] * skv_p + d_offs[None, :] * skv_d,
            ).to(tl.float32)

            scores = tl.dot(q, tl.trans(k)) * scale

            token_offs = page_idx * P + p_offs
            valid_mask = (token_offs[None, :] < seq_len_b)
            g_valid = (g_offs[:, None] < G)
            scores = tl.where(valid_mask & g_valid, scores, -1.0e10)

            page_max = tl.max(scores, axis=1)
            new_m = tl.maximum(m_val, page_max)
            alpha = tl.exp(m_val - new_m)
            exp_scores = tl.exp(scores - new_m[:, None])
            page_l = tl.sum(exp_scores, axis=1)
            new_l = l_val * alpha + page_l

            v_base = KV_ptr + phys * skv_blk + kv_hid * skv_hkv + D * skv_d
            v = tl.load(
                v_base + p_offs[:, None] * skv_p + d_offs[None, :] * skv_d,
            ).to(tl.float32)

            exp_v = exp_scores.to(v.dtype)
            acc = acc * alpha[:, None] + tl.dot(exp_v, v)
            m_val = new_m
            l_val = new_l

    mi_off = bid * smi_stride + kv_hid * S * G + sid * G + g_offs
    tl.store(MI_ptr + mi_off, m_val, mask=(g_offs < G))
    tl.store(LI_ptr + mi_off, l_val, mask=(g_offs < G))

    acc_off = bid * sacc_b + kv_hid * sacc_kvh + sid * sacc_s + g_offs[:, None] * sacc_d + d_offs[None, :] * 1
    tl.store(ACC_ptr + acc_off, acc, mask=(g_offs[:, None] < G) & (d_offs[None, :] < D))


@triton.jit
def _paged_attn_combine_kernel(
    MI_ptr, LI_ptr, ACC_ptr,
    O_ptr,
    H: tl.constexpr, Hkv: tl.constexpr,
    D: tl.constexpr, S: tl.constexpr,
    smi_stride, sli_stride,
    sacc_b, sacc_kvh, sacc_s, sacc_d,
    so_b, so_h, so_d,
):
    pid = tl.program_id(0)
    G: tl.constexpr = H // Hkv
    bid = pid // H
    hid = pid % H
    kv_hid = hid // G
    g_idx = hid % G

    d_offs = tl.arange(0, D)

    mi_base = MI_ptr + bid * smi_stride + kv_hid * S * G + g_idx
    li_base = LI_ptr + bid * smi_stride + kv_hid * S * G + g_idx
    acc_base = ACC_ptr + bid * sacc_b + kv_hid * sacc_kvh + g_idx * sacc_d

    global_max = -1.0e10
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        global_max = tl.maximum(global_max, mi_s)

    global_l = 0.0
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        li_s = tl.load(li_base + s * G)
        global_l += li_s * tl.exp(mi_s - global_max)

    acc = tl.zeros((D,), dtype=tl.float32)
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        a_s = tl.exp(mi_s - global_max)
        partial = tl.load(acc_base + s * sacc_s + d_offs)
        acc += a_s * partial

    acc = acc / global_l

    out_base = O_ptr + bid * so_b + hid * so_h
    tl.store(out_base + d_offs * so_d, acc.to(tl.bfloat16))


def next_pow2(n: int) -> int:
    if n <= 1:
        return 1
    return 1 << (n - 1).bit_length()


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._graph = None
        self._graph_ready = False
        self._sq = None

    def _forward_impl(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        P = self.page_size
        G = H // Hkv

        max_pages = (self.seq_len + P - 1) // P

        pages_per_split_target = max(2, max_pages // 16)
        raw_splits = max(max_pages // pages_per_split_target, 1)
        S = next_pow2(raw_splits)
        S = min(S, 64)

        output = torch.empty(B, H, D, dtype=torch.bfloat16, device=query.device)

        mi = torch.empty(B * Hkv * S * G, dtype=torch.float32, device=query.device)
        li = torch.empty(B * Hkv * S * G, dtype=torch.float32, device=query.device)
        acc_buf = torch.empty(B, Hkv, S, G, D, dtype=torch.float32, device=query.device)

        grid1 = (B * Hkv * S,)
        _paged_attn_split_kernel[grid1](
            query, kv_cache, block_table, seq_lens,
            mi, li, acc_buf,
            self.scale,
            H=H, Hkv=Hkv, D=D, P=P, S=S,
            max_pages=max_pages,
            sq_b=query.stride(0), sq_h=query.stride(1), sq_d=query.stride(2),
            skv_blk=kv_cache.stride(0), skv_p=kv_cache.stride(1),
            skv_hkv=kv_cache.stride(2), skv_d=kv_cache.stride(3),
            sbt_b=block_table.stride(0), sbt_p=block_table.stride(1),
            smi_stride=Hkv * S * G, sli_stride=Hkv * S * G,
            sacc_b=acc_buf.stride(0), sacc_kvh=acc_buf.stride(1),
            sacc_s=acc_buf.stride(2), sacc_d=acc_buf.stride(3),
            NUM_STAGES=4,
            num_warps=4,
        )

        grid2 = (B * H,)
        _paged_attn_combine_kernel[grid2](
            mi, li, acc_buf, output,
            H=H, Hkv=Hkv, D=D, S=S,
            smi_stride=Hkv * S * G, sli_stride=Hkv * S * G,
            sacc_b=acc_buf.stride(0), sacc_kvh=acc_buf.stride(1),
            sacc_s=acc_buf.stride(2), sacc_d=acc_buf.stride(3),
            so_b=output.stride(0), so_h=output.stride(1), so_d=output.stride(2),
            num_warps=1,
        )

        return output

    def forward(self, query, kv_cache, block_table, seq_lens):
        if not self._graph_ready:
            result = self._forward_impl(query, kv_cache, block_table, seq_lens)
            torch.cuda.synchronize()
            self._sq = query
            self._skv = kv_cache
            self._sbt = block_table
            self._ssl = seq_lens
            g = torch.cuda.CUDAGraph()
            with torch.cuda.graph(g):
                self._so = self._forward_impl(self._sq, self._skv, self._sbt, self._ssl)
            self._graph = g
            self._graph_ready = True
            return result

        if (query.data_ptr() == self._sq.data_ptr() and
            kv_cache.data_ptr() == self._skv.data_ptr() and
            block_table.data_ptr() == self._sbt.data_ptr() and
            seq_lens.data_ptr() == self._ssl.data_ptr()):
            self._graph.replay()
            return self._so
        else:
            self._sq.copy_(query)
            self._skv.copy_(kv_cache)
            self._sbt.copy_(block_table)
            self._ssl.copy_(seq_lens)
            self._graph.replay()
            return self._so
