import torch
import torch.nn as nn
import triton
import triton.language as tl


E4M3_MAX = 448.0


@triton.jit
def _fp8_gemm_scale_kernel(
    x_ptr,
    w_ptr,
    s_ptr,
    y_ptr,
    M: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    W_STRIDE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    K_FULL: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    group_size = GROUP_M * num_pid_n
    group_id = pid // group_size
    first_pid_m = group_id * GROUP_M
    group_m = tl.minimum(num_pid_m - first_pid_m, GROUP_M)
    pid_in_group = pid - group_id * group_size
    pid_n = pid_in_group // group_m
    pid_m = first_pid_m + (pid_in_group - pid_n * group_m)

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)
    for k0 in range(0, K_FULL, BLOCK_K):
        k = k0 + offs_k
        a = tl.load(x_ptr + offs_m[:, None] * K + k[None, :])
        b = tl.load(w_ptr + offs_n[None, :] * W_STRIDE + k[:, None])
        acc += tl.dot(a, b, out_dtype=tl.float32)

    if K_FULL < K:
        k = K_FULL + offs_k
        tail_mask = k < K
        a = tl.load(
            x_ptr + offs_m[:, None] * K + k[None, :],
            mask=tail_mask[None, :],
            other=0.0,
        )
        b = tl.load(w_ptr + offs_n[None, :] * W_STRIDE + k[:, None])
        acc += tl.dot(a, b, out_dtype=tl.float32)

    scale = tl.load(s_ptr + offs_n, mask=offs_n < N, other=0.0)
    out = acc * scale[None, :]
    tl.store(
        y_ptr + offs_m[:, None] * N + offs_n[None, :],
        out,
        mask=(offs_m[:, None] < M) & (offs_n[None, :] < N),
    )


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        w = torch.empty(N, K, dtype=torch.bfloat16)
        nn.init.normal_(w, std=0.02)
        s = (w.float().abs().amax(dim=1, keepdim=True) / E4M3_MAX).clamp(min=1e-12)
        w_fp8 = (w.float() / s).to(torch.float8_e4m3fn)
        self.register_buffer("weight", w_fp8)
        self.register_buffer("weight_scale", s.squeeze(1).to(torch.float32))
        self.register_buffer("_weight_packed", torch.empty(0, dtype=torch.float8_e4m3fn), persistent=False)
        self._packed_version = -1
        self._packed_stride = K

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        y = torch.empty((self.M, self.N), device=x.device, dtype=torch.bfloat16)
        weight, w_stride = self._weight_for_gemm()
        if self.M <= 64:
            bm, bn, bk, gm, warps, stages = 32, 128, 256, 1, 4, 3
        elif self.K % 128 != 0:
            bm, bn, bk, gm, warps, stages = 128, 256, 128, 8, 8, 4
        elif self.N > 8192:
            bm, bn, bk, gm, warps, stages = 128, 256, 128, 4, 8, 4
        else:
            bm, bn, bk, gm, warps, stages = 128, 256, 128, 4, 8, 4

        grid = (triton.cdiv(self.M, bm) * triton.cdiv(self.N, bn),)
        _fp8_gemm_scale_kernel[grid](
            x,
            weight,
            self.weight_scale,
            y,
            self.M,
            self.N,
            self.K,
            w_stride,
            BLOCK_M=bm,
            BLOCK_N=bn,
            BLOCK_K=bk,
            K_FULL=(self.K // bk) * bk,
            GROUP_M=gm,
            num_warps=warps,
            num_stages=stages,
        )
        return y

    def _weight_for_gemm(self):
        if self.K % 128 == 0:
            return self.weight, self.K

        version = self.weight._version
        k_pad = triton.cdiv(self.K, 128) * 128
        if (
            self._packed_version != version
            or self._weight_packed.device != self.weight.device
            or self._weight_packed.shape != (self.N, k_pad)
        ):
            packed = torch.empty((self.N, k_pad), device=self.weight.device, dtype=self.weight.dtype)
            packed.zero_()
            packed[:, : self.K].copy_(self.weight)
            self._weight_packed = packed
            self._packed_version = version
            self._packed_stride = k_pad
        return self._weight_packed, self._packed_stride


M = 4096
N = 4096
K = 4096


def get_inputs():
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn)
    return [x]


def get_init_inputs():
    return [M, N, K]
