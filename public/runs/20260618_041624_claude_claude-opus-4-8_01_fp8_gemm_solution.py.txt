"""FP8 e4m3 GEMM for H100 (SM90 Hopper).

y = (x @ weight.T) * weight_scale, with x fp8_e4m3 (M,K), weight fp8_e4m3 (N,K),
weight_scale (N,). Output bf16 (M,N).

Real fp8 x fp8 tensor-core MMA (fp32 accumulate) via Triton tl.dot on fp8e4nv.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl

E4M3_MAX = 448.0


def _configs():
    cfgs = []
    for BM, BN, BK, w, s in [
        (128, 128, 256, 8, 3),   # square-aligned winner
        (128, 256, 128, 8, 3),
        (128, 256, 64, 8, 4),
        (256, 128, 128, 8, 3),
        (128, 128, 128, 8, 4),
        (64, 256, 128, 4, 4),
        (64, 128, 256, 4, 4),    # skinny-M winner
        (64, 128, 128, 4, 4),
        (64, 64, 256, 4, 4),
        (128, 64, 256, 4, 4),
    ]:
        cfgs.append(triton.Config(
            {"BLOCK_M": BM, "BLOCK_N": BN, "BLOCK_K": BK, "GROUP_M": 8},
            num_warps=w, num_stages=s))
    return cfgs


@triton.autotune(configs=_configs(), key=["M", "N", "K"])
@triton.jit
def _fp8_gemm_kernel(
    x_ptr, w_ptr, scale_ptr, y_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_ym, stride_yn,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_M: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(M, BLOCK_M)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    num_pid_in_group = GROUP_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_n = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    x_ptrs = x_ptr + (offs_m[:, None] * stride_xm + offs_k[None, :] * stride_xk)
    # w is (N,K); we want w.T tile (BK, BN): element[k,n] = w[n,k]
    w_ptrs = w_ptr + (offs_n[None, :] * stride_wn + offs_k[:, None] * stride_wk)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    k_full = K // BLOCK_K
    for k in range(0, k_full):
        x = tl.load(x_ptrs)
        w = tl.load(w_ptrs)
        acc = tl.dot(x, w, acc)
        x_ptrs += BLOCK_K * stride_xk
        w_ptrs += BLOCK_K * stride_wk
    # masked tail (K not a multiple of BLOCK_K)
    k_rem = K - k_full * BLOCK_K
    if k_rem > 0:
        x = tl.load(x_ptrs, mask=offs_k[None, :] < k_rem, other=0.0)
        w = tl.load(w_ptrs, mask=offs_k[:, None] < k_rem, other=0.0)
        acc = tl.dot(x, w, acc)

    scale = tl.load(scale_ptr + offs_n, mask=offs_n < N, other=0.0).to(tl.float32)
    acc = acc * scale[None, :]

    offs_ym = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_yn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    y_ptrs = y_ptr + offs_ym[:, None] * stride_ym + offs_yn[None, :] * stride_yn
    ymask = (offs_ym[:, None] < M) & (offs_yn[None, :] < N)
    tl.store(y_ptrs, acc.to(tl.bfloat16), mask=ymask)


def _sk_configs():
    cfgs = []
    for BM, BN, BK, w, s in [
        (16, 128, 128, 4, 4),
        (16, 128, 64, 4, 4),
        (32, 128, 128, 4, 4),
        (16, 256, 128, 4, 3),
        (32, 256, 128, 4, 3),
        (16, 64, 128, 2, 4),
        (32, 64, 128, 2, 4),
        (16, 128, 256, 4, 3),
    ]:
        cfgs.append(triton.Config(
            {"BLOCK_M": BM, "BLOCK_N": BN, "BLOCK_K": BK},
            num_warps=w, num_stages=s))
    return cfgs


@triton.autotune(configs=_sk_configs(), key=["M", "N", "K", "SPLIT_K"])
@triton.jit
def _fp8_gemm_splitk_kernel(
    x_ptr, w_ptr, yp_ptr,
    M, N, K,
    stride_xm, stride_xk,
    stride_wn, stride_wk,
    stride_yp_s, stride_yp_m, stride_yp_n,
    SPLIT_K: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    pid_sk = tl.program_id(1)
    num_pid_n = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    offs_m = (pid_m * BLOCK_M + tl.arange(0, BLOCK_M)) % M
    offs_n = (pid_n * BLOCK_N + tl.arange(0, BLOCK_N)) % N
    offs_k = tl.arange(0, BLOCK_K)

    x_ptrs = x_ptr + (offs_m[:, None] * stride_xm
                      + (pid_sk * BLOCK_K + offs_k)[None, :] * stride_xk)
    w_ptrs = w_ptr + (offs_n[None, :] * stride_wn
                      + (pid_sk * BLOCK_K + offs_k)[:, None] * stride_wk)

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    k = pid_sk * BLOCK_K
    step = SPLIT_K * BLOCK_K
    while k < K:
        kmask = (k + offs_k) < K
        x = tl.load(x_ptrs, mask=kmask[None, :], other=0.0)
        w = tl.load(w_ptrs, mask=kmask[:, None], other=0.0)
        acc = tl.dot(x, w, acc)
        x_ptrs += step * stride_xk
        w_ptrs += step * stride_wk
        k += step

    # write partial for this split into yp[pid_sk] (no atomics)
    offs_ym = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_yn = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    yp_ptrs = (yp_ptr + pid_sk * stride_yp_s
               + offs_ym[:, None] * stride_yp_m + offs_yn[None, :] * stride_yp_n)
    ymask = (offs_ym[:, None] < M) & (offs_yn[None, :] < N)
    tl.store(yp_ptrs, acc, mask=ymask)


@triton.jit
def _reduce_scale_kernel(yp_ptr, scale_ptr, yo_ptr, M, N, SPLIT_K,
                         stride_yp_s, stride_yp_m, stride_yp_n,
                         stride_ym, stride_yn,
                         BLOCK: tl.constexpr, SK: tl.constexpr):
    pid = tl.program_id(0)
    offs = pid * BLOCK + tl.arange(0, BLOCK)
    mask = offs < M * N
    m = offs // N
    n = offs % N
    acc = tl.zeros((BLOCK,), dtype=tl.float32)
    for s in range(0, SK):
        v = tl.load(yp_ptr + s * stride_yp_s + m * stride_yp_m + n * stride_yp_n,
                    mask=mask, other=0.0)
        acc += v
    sc = tl.load(scale_ptr + n, mask=mask, other=0.0).to(tl.float32)
    tl.store(yo_ptr + m * stride_ym + n * stride_yn, (acc * sc).to(tl.bfloat16), mask=mask)


def fp8_gemm_splitk(x, weight, weight_scale, split_k=8):
    M, K = x.shape
    N, _ = weight.shape
    yp = torch.empty((split_k, M, N), device=x.device, dtype=torch.float32)
    grid = lambda META: (triton.cdiv(M, META["BLOCK_M"]) * triton.cdiv(N, META["BLOCK_N"]),
                         split_k)
    _fp8_gemm_splitk_kernel[grid](
        x, weight, yp, M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        yp.stride(0), yp.stride(1), yp.stride(2),
        SPLIT_K=split_k,
    )
    y = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)
    n_elem = M * N
    grid2 = lambda META: (triton.cdiv(n_elem, META["BLOCK"]),)
    _reduce_scale_kernel[grid2](yp, weight_scale, y, M, N, split_k,
                                yp.stride(0), yp.stride(1), yp.stride(2),
                                y.stride(0), y.stride(1),
                                BLOCK=1024, SK=split_k)
    return y


def fp8_gemm(x, weight, weight_scale):
    M, K = x.shape
    N, Kw = weight.shape
    assert K == Kw
    y = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)
    grid = lambda META: (triton.cdiv(M, META["BLOCK_M"]) * triton.cdiv(N, META["BLOCK_N"]),)
    _fp8_gemm_kernel[grid](
        x, weight, weight_scale, y,
        M, N, K,
        x.stride(0), x.stride(1),
        weight.stride(0), weight.stride(1),
        y.stride(0), y.stride(1),
    )
    return y


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        w = torch.empty(N, K, dtype=torch.bfloat16)
        nn.init.normal_(w, std=0.02)
        s = (w.float().abs().amax(dim=1, keepdim=True) / E4M3_MAX).clamp(min=1e-12)
        w_fp8 = (w.float() / s).to(torch.float8_e4m3fn)
        self.register_buffer("weight", w_fp8)
        self.register_buffer("weight_scale", s.squeeze(1).to(torch.float32))
        self._wpad = None
        self._wpad_ver = -1

    def _aligned_weight(self, Kpad):
        """Return a (N, K) view of weight whose row stride is 16-aligned."""
        ver = self.weight._version
        if (self._wpad is None or self._wpad.shape[1] != Kpad
                or self._wpad_ver != ver):
            wp = torch.empty((self.N, Kpad), dtype=torch.float8_e4m3fn,
                             device=self.weight.device)
            wp[:, :self.K] = self.weight
            self._wpad = wp
            self._wpad_ver = ver
        return self._wpad[:, :self.K]

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        M, K = x.shape
        if K % 16 != 0:
            Kpad = (K + 15) // 16 * 16
            xp = torch.empty((M, Kpad), dtype=x.dtype, device=x.device)
            xp[:, :K] = x
            xa = xp[:, :K]
            w, sc = self._aligned_weight(Kpad), self.weight_scale
        else:
            xa, w, sc = x, self.weight, self.weight_scale
        return fp8_gemm(xa, w, sc)
