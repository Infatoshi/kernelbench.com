+"""Test if CUB BlockRadixSort compiles and works"""
+import torch

+
+cpp_src = """
+#include <torch/extension.h>
+"""
+
+cuda_src = r"""
+#include <torch/extension.h>
+#include <cub/cub.cuh>
+#include <cuda_runtime.h>
+#include <limits>
+#include <cmath>
+
+template<int BLOCK_THREADS, int ITEMS_PER_THREAD>
+__global__ void test_sort_kernel(const float* input, float* out_vals, int* out_idx, int n, int k) {
+    using BlockRadixSort = cub::BlockRadixSort<float, BLOCK_THREADS, ITEMS_PER_THREAD, int>;
+    __shared__ typename BlockRadixSort::TempStorage temp_storage;
+    
+    float keys[ITEMS_PER_THREAD];
+    int values[ITEMS_PER_THREAD];
+    
+    #pragma unroll
+    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
+        int idx = threadIdx.x + i * BLOCK_THREADS;
+        if (idx < n) {
+            keys[i] = input[idx];
+            values[i] = idx;
+        } else {
+            keys[i] = -std::numeric_limits<float>::infinity();
+            values[i] = -1;
+        }
+    }
+    
+    BlockRadixSort(temp_storage).SortDescendingBlockedToStriped(keys, values);
+    
+    #pragma unroll
+    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
+        int global_pos = threadIdx.x + i * BLOCK_THREADS;
+        if (global_pos < k) {
+            out_vals[global_pos] = keys[i];
+            out_idx[global_pos] = values[i];
+        }
+    }
+}
+
+void run_test(torch::Tensor input, torch::Tensor out_vals, torch::Tensor out_idx, int n, int k) {
+    test_sort_kernel<256, 16><<<1, 256>>>(
+        input.data_ptr<float>(),
+        out_vals.data_ptr<float>(),
+        out_idx.data_ptr<int>(),
+        n, k
+    );
+}
+"""
