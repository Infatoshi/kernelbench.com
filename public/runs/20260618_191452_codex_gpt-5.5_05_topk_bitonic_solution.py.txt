_CUDA_SRC = r"""
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <math_constants.h>
#include <torch/extension.h>

static constexpr int BLOCK_THREADS = 256;
static constexpr int CHUNK = 1024;

__device__ __forceinline__ bool better_pair(float av, int ai, float bv, int bi) {
    return (av > bv) || ((av == bv) && (ai < bi));
}

__device__ __forceinline__ bool below_boundary(float v, int idx, float prev_v, int prev_i, int rank) {
    return rank == 0 || (v < prev_v) || ((v == prev_v) && (idx > prev_i));
}

__device__ __forceinline__ void warp_reduce_best(float &v, int &idx) {
    unsigned mask = 0xffffffffu;
    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        float ov = __shfl_down_sync(mask, v, offset);
        int oi = __shfl_down_sync(mask, idx, offset);
        if (better_pair(ov, oi, v, idx)) {
            v = ov;
            idx = oi;
        }
    }
}

template<int THREADS>
__device__ __forceinline__ void block_reduce_best(float &v, int &idx) {
    __shared__ float warp_vals[THREADS / 32];
    __shared__ int warp_idxs[THREADS / 32];

    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;

    warp_reduce_best(v, idx);
    if (lane == 0) {
        warp_vals[warp] = v;
        warp_idxs[warp] = idx;
    }
    __syncthreads();

    if (warp == 0) {
        v = lane < (THREADS / 32) ? warp_vals[lane] : -CUDART_INF_F;
        idx = lane < (THREADS / 32) ? warp_idxs[lane] : 0x3fffffff;
        warp_reduce_best(v, idx);
        if (lane == 0) {
            warp_vals[0] = v;
            warp_idxs[0] = idx;
        }
    }
    __syncthreads();

    v = warp_vals[0];
    idx = warp_idxs[0];
    __syncthreads();
}

template<int K>
__global__ __launch_bounds__(BLOCK_THREADS, 2)
void stage_kernel(const float *__restrict__ x,
                  float *__restrict__ tmp_v,
                  int *__restrict__ tmp_i,
                  int n,
                  int chunks) {
    __shared__ float s_vals[CHUNK];

    int bid = blockIdx.x;
    int row = bid / chunks;
    int chunk = bid - row * chunks;
    int base_col = chunk * CHUNK;
    int row_base = row * n;

    for (int p = threadIdx.x; p < CHUNK; p += BLOCK_THREADS) {
        int col = base_col + p;
        s_vals[p] = col < n ? x[row_base + col] : -CUDART_INF_F;
    }
    __syncthreads();

    float prev_v = CUDART_INF_F;
    int prev_i = base_col - 1;
    int out_base = bid * K;

    #pragma unroll
    for (int rank = 0; rank < K; ++rank) {
        float local_v = -CUDART_INF_F;
        int local_i = 0x3fffffff;

        for (int p = threadIdx.x; p < CHUNK; p += BLOCK_THREADS) {
            float v = s_vals[p];
            int idx = base_col + p;
            if (below_boundary(v, idx, prev_v, prev_i, rank) &&
                better_pair(v, idx, local_v, local_i)) {
                local_v = v;
                local_i = idx;
            }
        }

        block_reduce_best<BLOCK_THREADS>(local_v, local_i);
        if (threadIdx.x == 0) {
            tmp_v[out_base + rank] = local_v;
            tmp_i[out_base + rank] = local_i;
        }
        prev_v = local_v;
        prev_i = local_i;
    }
}

template<int K>
__global__ __launch_bounds__(BLOCK_THREADS, 2)
void stage_sort_kernel(const float *__restrict__ x,
                       float *__restrict__ tmp_v,
                       int *__restrict__ tmp_i,
                       int n,
                       int chunks) {
    __shared__ float s_vals[CHUNK];
    __shared__ int s_idxs[CHUNK];

    int bid = blockIdx.x;
    int row = bid / chunks;
    int chunk = bid - row * chunks;
    int base_col = chunk * CHUNK;
    int row_base = row * n;

    for (int p = threadIdx.x; p < CHUNK; p += BLOCK_THREADS) {
        int col = base_col + p;
        s_vals[p] = col < n ? x[row_base + col] : -CUDART_INF_F;
        s_idxs[p] = col < n ? col : 0x3fffffff;
    }
    __syncthreads();

    for (int size = 2; size <= CHUNK; size <<= 1) {
        for (int stride = size >> 1; stride > 0; stride >>= 1) {
            for (int i = threadIdx.x; i < CHUNK; i += BLOCK_THREADS) {
                int j = i ^ stride;
                if (j > i) {
                    float vi = s_vals[i];
                    float vj = s_vals[j];
                    int ii = s_idxs[i];
                    int ij = s_idxs[j];
                    bool first_half = (i & size) == 0;
                    bool do_swap = first_half
                        ? better_pair(vj, ij, vi, ii)
                        : better_pair(vi, ii, vj, ij);
                    if (do_swap) {
                        s_vals[i] = vj;
                        s_vals[j] = vi;
                        s_idxs[i] = ij;
                        s_idxs[j] = ii;
                    }
                }
            }
            __syncthreads();
        }
    }

    int out_base = bid * K;
    for (int p = threadIdx.x; p < K; p += BLOCK_THREADS) {
        tmp_v[out_base + p] = s_vals[p];
        tmp_i[out_base + p] = s_idxs[p];
    }
}

template<int K, int CAND>
__global__ __launch_bounds__(BLOCK_THREADS, 2)
void merge_kernel(const float *__restrict__ tmp_v,
                  const int *__restrict__ tmp_i,
                  float *__restrict__ out_v,
                  int64_t *__restrict__ out_i) {
    int row = blockIdx.x;
    int row_base = row * CAND;

    float prev_v = CUDART_INF_F;
    int prev_i = -1;
    int out_base = row * K;

    #pragma unroll
    for (int rank = 0; rank < K; ++rank) {
        float local_v = -CUDART_INF_F;
        int local_i = 0x3fffffff;

        for (int p = threadIdx.x; p < CAND; p += BLOCK_THREADS) {
            float v = tmp_v[row_base + p];
            int idx = tmp_i[row_base + p];
            if (below_boundary(v, idx, prev_v, prev_i, rank) &&
                better_pair(v, idx, local_v, local_i)) {
                local_v = v;
                local_i = idx;
            }
        }

        block_reduce_best<BLOCK_THREADS>(local_v, local_i);
        if (threadIdx.x == 0) {
            out_v[out_base + rank] = local_v;
            out_i[out_base + rank] = static_cast<int64_t>(local_i);
        }
        prev_v = local_v;
        prev_i = local_i;
    }
}

__global__ __launch_bounds__(BLOCK_THREADS, 2)
void argmax_kernel(const float *__restrict__ x,
                   float *__restrict__ out_v,
                   int64_t *__restrict__ out_i,
                   int n) {
    __shared__ float s_vals[4096];

    int row = blockIdx.x;
    int row_base = row * n;
    for (int p = threadIdx.x; p < 4096; p += BLOCK_THREADS) {
        s_vals[p] = p < n ? x[row_base + p] : -CUDART_INF_F;
    }
    __syncthreads();

    float local_v = -CUDART_INF_F;
    int local_i = 0x3fffffff;
    for (int p = threadIdx.x; p < 4096; p += BLOCK_THREADS) {
        float v = s_vals[p];
        if (better_pair(v, p, local_v, local_i)) {
            local_v = v;
            local_i = p;
        }
    }
    block_reduce_best<BLOCK_THREADS>(local_v, local_i);
    if (threadIdx.x == 0) {
        out_v[row] = local_v;
        out_i[row] = static_cast<int64_t>(local_i);
    }
}

template<int K>
__global__ __launch_bounds__(BLOCK_THREADS, 2)
void row_kernel(const float *__restrict__ x,
                float *__restrict__ out_v,
                int64_t *__restrict__ out_i,
                int n) {
    int row = blockIdx.x;
    int row_base = row * n;
    float prev_v = CUDART_INF_F;
    int prev_i = -1;
    int out_base = row * K;

    #pragma unroll
    for (int rank = 0; rank < K; ++rank) {
        float local_v = -CUDART_INF_F;
        int local_i = 0x3fffffff;
        for (int col = threadIdx.x; col < n; col += BLOCK_THREADS) {
            float v = x[row_base + col];
            if (below_boundary(v, col, prev_v, prev_i, rank) &&
                better_pair(v, col, local_v, local_i)) {
                local_v = v;
                local_i = col;
            }
        }
        block_reduce_best<BLOCK_THREADS>(local_v, local_i);
        if (threadIdx.x == 0) {
            out_v[out_base + rank] = local_v;
            out_i[out_base + rank] = static_cast<int64_t>(local_i);
        }
        prev_v = local_v;
        prev_i = local_i;
    }
}

void topk_forward_cuda(torch::Tensor x,
                       torch::Tensor tmp_v,
                       torch::Tensor tmp_i,
                       torch::Tensor out_v,
                       torch::Tensor out_i,
                       int64_t n64,
                       int64_t k64,
                       int64_t chunks64) {
    int n = static_cast<int>(n64);
    int k = static_cast<int>(k64);
    int chunks = static_cast<int>(chunks64);
    int batch = static_cast<int>(x.size(0));
    cudaStream_t stream = at::cuda::getCurrentCUDAStream();

    const float *x_ptr = x.data_ptr<float>();
    float *tmp_v_ptr = tmp_v.numel() ? tmp_v.data_ptr<float>() : nullptr;
    int *tmp_i_ptr = tmp_i.numel() ? tmp_i.data_ptr<int>() : nullptr;
    float *out_v_ptr = out_v.data_ptr<float>();
    int64_t *out_i_ptr = out_i.data_ptr<int64_t>();

    if (k == 1) {
        argmax_kernel<<<batch, BLOCK_THREADS, 0, stream>>>(x_ptr, out_v_ptr, out_i_ptr, n);
    } else if (k == 8) {
        row_kernel<8><<<batch, BLOCK_THREADS, 0, stream>>>(x_ptr, out_v_ptr, out_i_ptr, n);
    } else if (k == 16) {
        stage_kernel<16><<<batch * chunks, BLOCK_THREADS, 0, stream>>>(x_ptr, tmp_v_ptr, tmp_i_ptr, n, chunks);
        merge_kernel<16, 192><<<batch, BLOCK_THREADS, 0, stream>>>(tmp_v_ptr, tmp_i_ptr, out_v_ptr, out_i_ptr);
    } else if (k == 32) {
        stage_sort_kernel<32><<<batch * chunks, BLOCK_THREADS, 0, stream>>>(x_ptr, tmp_v_ptr, tmp_i_ptr, n, chunks);
        merge_kernel<32, 512><<<batch, BLOCK_THREADS, 0, stream>>>(tmp_v_ptr, tmp_i_ptr, out_v_ptr, out_i_ptr);
    } else {
        stage_sort_kernel<64><<<batch * chunks, BLOCK_THREADS, 0, stream>>>(x_ptr, tmp_v_ptr, tmp_i_ptr, n, chunks);
        merge_kernel<64, 8192><<<batch, BLOCK_THREADS, 0, stream>>>(tmp_v_ptr, tmp_i_ptr, out_v_ptr, out_i_ptr);
    }
}
"""


_EXT = load_inline(
    name="topk_b200_shared_select_v9",
    cpp_sources=_CPP_SRC,
    cuda_sources=_CUDA_SRC,
    extra_cuda_cflags=["-O3", "--use_fast_math"],
    with_cuda=True,
    verbose=False,
)


@triton.jit
def _triton_stage_kernel(
    x_ptr,
    tmp_v_ptr,
    tmp_i_ptr,
    n: tl.constexpr,
    k: tl.constexpr,
    chunks: tl.constexpr,
    block: tl.constexpr,
):
    pid = tl.program_id(0)
    row = pid // chunks
    chunk = pid - row * chunks
    offs = tl.arange(0, block)
    cols = chunk * block + offs
    mask = cols < n

    vals = tl.load(x_ptr + row * n + cols, mask=mask, other=-float("inf"))
    top_vals = tl.topk(vals, k)
    out_offs = tl.arange(0, k)
    tl.store(tmp_v_ptr + (row * chunks + chunk) * k + out_offs, top_vals)

    big = n + block
    for j in tl.static_range(0, k):
        tv = tl.max(tl.where(out_offs == j, top_vals, -float("inf")), axis=0)
        found = tl.min(tl.where((vals == tv) & mask, cols, big), axis=0)
        tl.store(tmp_i_ptr + (row * chunks + chunk) * k + j, found)


@triton.jit
def _triton_merge_kernel(
    tmp_v_ptr,
    tmp_i_ptr,
    out_v_ptr,
    out_i_ptr,
    k: tl.constexpr,
    chunks: tl.constexpr,
    cand_block: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, cand_block)
    mask = offs < (chunks * k)
    vals = tl.load(tmp_v_ptr + row * chunks * k + offs, mask=mask, other=-float("inf"))
    top_vals = tl.topk(vals, k)

    out_offs = tl.arange(0, k)
    tl.store(out_v_ptr + row * k + out_offs, top_vals)

    big = cand_block + 1
    for j in tl.static_range(0, k):
        tv = tl.max(tl.where(out_offs == j, top_vals, -float("inf")), axis=0)
        pos = tl.min(tl.where((vals == tv) & mask, offs, big), axis=0)
        src = tl.load(tmp_i_ptr + row * chunks * k + pos, mask=pos < (chunks * k), other=0)
        tl.store(out_i_ptr + row * k + j, src.to(tl.int64))


@triton.jit
def _triton_direct_kernel(
    x_ptr,
    out_v_ptr,
    out_i_ptr,
    n: tl.constexpr,
    k: tl.constexpr,
    block: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, block)
    mask = offs < n
    vals = tl.load(x_ptr + row * n + offs, mask=mask, other=-float("inf"))

    if k == 1:
        best = tl.max(vals, axis=0)
        idx = tl.min(tl.where((vals == best) & mask, offs, n), axis=0)
        tl.store(out_v_ptr + row, best)
        tl.store(out_i_ptr + row, idx.to(tl.int64))
    else:
        top_vals = tl.topk(vals, k)
        out_offs = tl.arange(0, k)
        tl.store(out_v_ptr + row * k + out_offs, top_vals)

        big = block + 1
        for j in tl.static_range(0, k):
            tv = tl.max(tl.where(out_offs == j, top_vals, -float("inf")), axis=0)
            idx = tl.min(tl.where((vals == tv) & mask, offs, big), axis=0)
            tl.store(out_i_ptr + row * k + j, idx.to(tl.int64))


def _ceil_div(a: int, b: int) -> int:
    return (a + b - 1) // b


def _next_power_of_2(x: int) -> int:
    return 1 << (x - 1).bit_length()


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        self.block = 2048 if k == 16 else (4096 if k >= 32 else 1024)
        self.stage_warps = 8 if k == 16 else 16
        self.merge_warps = 16 if k == 64 else 8
        self.chunks = _ceil_div(n, self.block)
        self.cand_block = _next_power_of_2(self.chunks * k)
        self.register_buffer("_dummy", torch.zeros(1))

        out_v = torch.empty((batch, k), dtype=torch.float32)
        out_i = torch.empty((batch, k), dtype=torch.int64)
        if k == 1:
            tmp_v = torch.empty((0,), dtype=torch.float32)
            tmp_i = torch.empty((0,), dtype=torch.int32)
        else:
            tmp_v = torch.empty((batch, self.chunks, k), dtype=torch.float32)
            tmp_i = torch.empty((batch, self.chunks, k), dtype=torch.int32)
        self.register_buffer("_out_v", out_v, persistent=False)
        self.register_buffer("_out_i", out_i, persistent=False)
        self.register_buffer("_tmp_v", tmp_v, persistent=False)
        self.register_buffer("_tmp_i", tmp_i, persistent=False)

    def forward(self, x: torch.Tensor):
        if self.k >= 16:
            _triton_stage_kernel[(self.batch * self.chunks,)](
                x,
                self._tmp_v,
                self._tmp_i,
                self.n,
                self.k,
                self.chunks,
                self.block,
                num_warps=self.stage_warps,
            )
            _triton_merge_kernel[(self.batch,)](
                self._tmp_v,
                self._tmp_i,
                self._out_v,
                self._out_i,
                self.k,
                self.chunks,
                self.cand_block,
                num_warps=self.merge_warps,
            )
            return self._out_v, self._out_i

        if self.k == 1:
            _triton_direct_kernel[(self.batch,)](
                x,
                self._out_v,
                self._out_i,
                self.n,
                self.k,
                _next_power_of_2(self.n),
                num_warps=16,
            )
            return self._out_v, self._out_i

        _EXT.topk_forward_cuda(
            x,
            self._tmp_v,
            self._tmp_i,
            self._out_v,
            self._out_i,
            self.n,
            self.k,
            self.chunks,
        )
        return self._out_v, self._out_i


batch = 64
n = 8192
k = 8


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
