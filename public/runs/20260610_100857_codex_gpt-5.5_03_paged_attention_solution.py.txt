import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_decode_one_head_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    max_blocks: tl.constexpr,
    num_heads: tl.constexpr,
    num_kv_heads: tl.constexpr,
    head_dim: tl.constexpr,
    page_size: tl.constexpr,
    scale: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // num_heads
    h = pid - b * num_heads
    group_size: tl.constexpr = num_heads // num_kv_heads
    kv_h = h // group_size

    offs_d = tl.arange(0, BLOCK_D)
    d_mask = offs_d < head_dim
    q = tl.load(
        query + (b * num_heads + h) * head_dim + offs_d,
        mask=d_mask,
        other=0.0,
    ).to(tl.float32)

    m_i = tl.full((), -float("inf"), tl.float32)
    l_i = tl.full((), 0.0, tl.float32)
    acc = tl.zeros((BLOCK_D,), tl.float32)
    L = tl.load(seq_lens + b)

    offs_m = tl.arange(0, BLOCK_M)
    for start in range(0, L, BLOCK_M):
        tok = start + offs_m
        tok_mask = tok < L
        page_idx = tok // page_size
        page_off = tok - page_idx * page_size
        block_id = tl.load(
            block_table + b * max_blocks + page_idx,
            mask=tok_mask,
            other=0,
        )

        base = ((block_id * page_size + page_off) * num_kv_heads + kv_h) * (head_dim * 2)
        k = tl.load(
            kv_cache + base[:, None] + offs_d[None, :],
            mask=tok_mask[:, None] & d_mask[None, :],
            other=0.0,
        ).to(tl.float32)
        scores = tl.sum(k * q[None, :], axis=1) * scale
        scores = tl.where(tok_mask, scores, -float("inf"))

        m_new = tl.maximum(m_i, tl.max(scores, axis=0))
        p = tl.exp(scores - m_new)
        alpha = tl.exp(m_i - m_new)

        v = tl.load(
            kv_cache + base[:, None] + head_dim + offs_d[None, :],
            mask=tok_mask[:, None] & d_mask[None, :],
            other=0.0,
        ).to(tl.float32)
        acc = acc * alpha + tl.sum(p[:, None] * v, axis=0)
        l_i = l_i * alpha + tl.sum(p, axis=0)
        m_i = m_new

    result = acc / l_i
    tl.store(
        out + (b * num_heads + h) * head_dim + offs_d,
        result,
        mask=d_mask,
    )


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.scale = 1.0 / math.sqrt(head_dim)
        self.max_blocks = triton.cdiv(seq_len, page_size)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        out = torch.empty_like(query)
        if self.head_dim == 64:
            block_m = 512
        elif self.batch >= 16:
            block_m = 128
        else:
            block_m = 256
        grid = (self.batch * self.num_heads,)
        _paged_decode_one_head_kernel[grid](
            query,
            kv_cache,
            block_table,
            seq_lens,
            out,
            self.max_blocks,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
            self.scale,
            BLOCK_M=block_m,
            BLOCK_D=triton.next_power_of_2(self.head_dim),
            num_warps=4,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1
    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)
    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
