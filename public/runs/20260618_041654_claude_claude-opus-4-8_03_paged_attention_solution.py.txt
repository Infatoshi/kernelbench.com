"""Paged-attention decode kernel (FlashDecoding split-KV) in Triton.

Single-query GQA decode. KV cache is paged:
  kv_cache: (num_blocks, page_size, num_kv_heads, 2*head_dim)  bf16, [K|V] packed.

Memory-bound: split each (batch, kv_head) KV sequence into chunks across CTAs,
compute partial online-softmax outputs, then combine. For shapes with enough
base parallelism (num_splits==1) the partial kernel writes the final output
directly and the combine launch is skipped.

Launch configs (BLOCK_N, num_warps, num_stages, num_splits) are selected from a
per-shape lookup table tuned offline with an L2-flushed benchmark.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _attn_partial(
    Q, KV, BlockTable, SeqLens,
    OutAcc, OutLse, Out,
    scale,
    stride_qb, stride_qh, stride_qd,
    stride_kv_page, stride_kv_pos, stride_kv_head, stride_kv_d,
    stride_bt_b, stride_bt_p,
    stride_oa_b, stride_oa_h, stride_oa_s, stride_oa_d,
    stride_ol_b, stride_ol_h, stride_ol_s,
    stride_ob, stride_oh, stride_od,
    num_kv_heads, group_size,
    PAGE_SIZE: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    BLOCK_H: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    WRITE_FINAL: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    pid = tl.program_id(0)
    split = tl.program_id(1)
    b = pid // num_kv_heads
    kvh = pid % num_kv_heads

    seqlen = tl.load(SeqLens + b)

    offs_h = tl.arange(0, BLOCK_H)
    offs_d = tl.arange(0, HEAD_DIM)
    offs_n = tl.arange(0, BLOCK_N)

    q_heads = kvh * group_size + offs_h
    q_mask = offs_h < group_size
    q_ptrs = Q + b * stride_qb + q_heads[:, None] * stride_qh + offs_d[None, :] * stride_qd
    q = tl.load(q_ptrs, mask=q_mask[:, None], other=0.0)
    q = (q * scale).to(q.dtype)

    m_i = tl.full([BLOCK_H], float("-inf"), dtype=tl.float32)
    l_i = tl.zeros([BLOCK_H], dtype=tl.float32)
    acc = tl.zeros([BLOCK_H, HEAD_DIM], dtype=tl.float32)

    start = split * BLOCK_N
    step = NUM_SPLITS * BLOCK_N
    for n_start in range(start, seqlen, step):
        toks = n_start + offs_n
        mask_n = toks < seqlen
        logical_page = toks // PAGE_SIZE
        page_off = toks % PAGE_SIZE
        phys = tl.load(BlockTable + b * stride_bt_b + logical_page * stride_bt_p,
                       mask=mask_n, other=0)
        kv_base = phys * stride_kv_page + page_off * stride_kv_pos + kvh * stride_kv_head

        k_ptrs = KV + kv_base[None, :] + offs_d[:, None] * stride_kv_d
        k = tl.load(k_ptrs, mask=mask_n[None, :], other=0.0)
        v_ptrs = KV + kv_base[:, None] + (HEAD_DIM + offs_d)[None, :] * stride_kv_d
        v = tl.load(v_ptrs, mask=mask_n[:, None], other=0.0)

        scores = tl.dot(q, k)
        scores = tl.where(mask_n[None, :], scores, float("-inf"))

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        p = tl.exp(scores - m_new[:, None])
        alpha = tl.exp(m_i - m_new)
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(v.dtype), v)
        m_i = m_new

    l_safe = tl.where(l_i > 0, l_i, 1.0)
    acc = acc / l_safe[:, None]

    if WRITE_FINAL:
        o_ptrs = Out + b * stride_ob + q_heads[:, None] * stride_oh + offs_d[None, :] * stride_od
        tl.store(o_ptrs, acc.to(Out.dtype.element_ty), mask=q_mask[:, None])
    else:
        lse = tl.where(l_i > 0, m_i + tl.log(l_i), float("-inf"))
        oa_ptrs = (OutAcc + b * stride_oa_b + q_heads[:, None] * stride_oa_h
                   + split * stride_oa_s + offs_d[None, :] * stride_oa_d)
        tl.store(oa_ptrs, acc, mask=q_mask[:, None])
        ol_ptrs = OutLse + b * stride_ol_b + q_heads * stride_ol_h + split * stride_ol_s
        tl.store(ol_ptrs, lse, mask=q_mask)


@triton.jit
def _combine(
    OutAcc, OutLse, Out,
    stride_oa_b, stride_oa_h, stride_oa_s, stride_oa_d,
    stride_ol_b, stride_ol_h, stride_ol_s,
    stride_ob, stride_oh, stride_od,
    num_splits,
    HEAD_DIM: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
):
    b = tl.program_id(0)
    h = tl.program_id(1)

    offs_d = tl.arange(0, HEAD_DIM)
    offs_s = tl.arange(0, NUM_SPLITS)
    s_mask = offs_s < num_splits

    lse = tl.load(OutLse + b * stride_ol_b + h * stride_ol_h + offs_s * stride_ol_s,
                  mask=s_mask, other=float("-inf"))
    m = tl.max(lse, axis=0)
    p = tl.where(s_mask, tl.exp(lse - m), 0.0)
    denom = tl.sum(p, axis=0)

    acc_ptrs = (OutAcc + b * stride_oa_b + h * stride_oa_h
                + offs_s[:, None] * stride_oa_s + offs_d[None, :] * stride_oa_d)
    acc = tl.load(acc_ptrs, mask=s_mask[:, None], other=0.0)
    out = tl.sum(acc * p[:, None], axis=0) / denom

    o_ptrs = Out + b * stride_ob + h * stride_oh + offs_d * stride_od
    tl.store(o_ptrs, out.to(Out.dtype.element_ty))


# Per-shape tuned launch configs: (B, H, Hkv, D, L, P) -> (BLOCK_N, num_warps, num_stages, num_splits)
_CONFIG = {
    (8, 32, 8, 128, 1024, 16): (64, 4, 3, 8),
    (32, 32, 8, 128, 2048, 16): (64, 2, 4, 2),
    (4, 64, 8, 128, 4096, 16): (64, 4, 2, 8),
    (16, 32, 8, 128, 1535, 16): (64, 2, 4, 4),
    (8, 16, 4, 64, 2000, 16): (64, 4, 2, 16),
}


def _default_config(B, H, Hkv, D, L, P):
    base = B * Hkv
    max_useful = max(1, (L + 31) // 32)
    ns = max(1, min((512 + base - 1) // base, max_useful, 32))
    return (64, 4, 3, ns)


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

        key = (batch, num_heads, num_kv_heads, head_dim, seq_len, page_size)
        self.cfg = _CONFIG.get(key) or _default_config(*key)
        self.BLOCK_H = max(16, triton.next_power_of_2(self.group_size))

    def forward(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size
        BLOCK_N, num_warps, num_stages, ns = self.cfg
        BLOCK_H = self.BLOCK_H

        out = torch.empty((B, H, D), dtype=query.dtype, device=query.device)

        write_final = ns == 1
        if write_final:
            out_acc = out
            out_lse = out
            oa_s = (0, 0, 0, 0)
            ol_s = (0, 0, 0)
        else:
            out_acc = torch.empty((B, H, ns, D), dtype=torch.float32, device=query.device)
            out_lse = torch.empty((B, H, ns), dtype=torch.float32, device=query.device)
            oa_s = (out_acc.stride(0), out_acc.stride(1), out_acc.stride(2), out_acc.stride(3))
            ol_s = (out_lse.stride(0), out_lse.stride(1), out_lse.stride(2))

        grid = (B * Hkv, ns)
        _attn_partial[grid](
            query, kv_cache, block_table, seq_lens,
            out_acc, out_lse, out,
            self.scale,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0), block_table.stride(1),
            oa_s[0], oa_s[1], oa_s[2], oa_s[3],
            ol_s[0], ol_s[1], ol_s[2],
            out.stride(0), out.stride(1), out.stride(2),
            Hkv, G,
            PAGE_SIZE=P, HEAD_DIM=D, BLOCK_H=BLOCK_H, NUM_SPLITS=ns,
            WRITE_FINAL=write_final, BLOCK_N=BLOCK_N,
            num_warps=num_warps, num_stages=num_stages,
        )

        if not write_final:
            _combine[(B, H)](
                out_acc, out_lse, out,
                out_acc.stride(0), out_acc.stride(1), out_acc.stride(2), out_acc.stride(3),
                out_lse.stride(0), out_lse.stride(1), out_lse.stride(2),
                out.stride(0), out.stride(1), out.stride(2),
                ns,
                HEAD_DIM=D, NUM_SPLITS=triton.next_power_of_2(ns),
            )
        return out
