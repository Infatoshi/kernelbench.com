@triton.jit
def bsearch(pid_m, m_blocks_cumsum, E):
    """
    Find which expert owns the m-th block using binary search over the
    precomputed cumulative sum of blocks per expert.
    """
    left = 0
    right = E
    while left < right - 1:
        mid = (left + right) // 2
        if pid_m < tl.load(m_blocks_cumsum + mid):
            right = mid
        else:
            left = mid
    return left

@triton.jit
def grouped_gemm_swiglu_kernel(
    x_ptr, w_gate_ptr, w_up_ptr, out_ptr,
    expert_offsets, m_blocks_cumsum,
    stride_xm, stride_we, stride_wk, stride_outm,
    E: tl.constexpr, H: tl.constexpr, I: tl.constexpr,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
    GROUP_M: tl.constexpr
):
    pid = tl.program_id(0)
    num_m_blocks = tl.load(m_blocks_cumsum + E)
    num_n_blocks = tl.cdiv(I, BLOCK_N)
    
    # Grid Swizzling to improve L2 cache hit rate
    num_m_groups = tl.cdiv(num_m_blocks, GROUP_M)
    group_id = pid // (GROUP_M * num_n_blocks)
    first_pid_m = group_id * GROUP_M
    group_size_m = min(num_m_blocks - first_pid_m, GROUP_M)
    
    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % (group_size_m * num_n_blocks)) // group_size_m
    
    expert_id = bsearch(pid_m, m_blocks_cumsum, E)
    first_m_block = tl.load(m_blocks_cumsum + expert_id)
    m_idx = pid_m - first_m_block
    
    expert_start = tl.load(expert_offsets + expert_id)
    expert_end = tl.load(expert_offsets + expert_id + 1)
    expert_n = expert_end - expert_start
    
    start_m = m_idx * BLOCK_M
    start_n = pid_n * BLOCK_N
    
    offs_m = start_m + tl.arange(0, BLOCK_M)
    offs_n = start_n + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)
    
    x_ptrs = x_ptr + (expert_start + offs_m[:, None]) * stride_xm + offs_k[None, :]
    w_gate_ptrs = w_gate_ptr + expert_id * stride_we + offs_k[:, None] * stride_wk + offs_n[None, :]
    w_up_ptrs = w_up_ptr + expert_id * stride_we + offs_k[:, None] * stride_wk + offs_n[None, :]
    
    acc_gate = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc_up = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    
    mask_m = offs_m < expert_n
    
    for k in range(0, H, BLOCK_K):
        x = tl.load(x_ptrs, mask=mask_m[:, None], other=0.0)
        wg = tl.load(w_gate_ptrs)
        wu = tl.load(w_up_ptrs)
        
        acc_gate += tl.dot(x, wg)
        acc_up += tl.dot(x, wu)
        
        x_ptrs += BLOCK_K
        w_gate_ptrs += BLOCK_K * stride_wk
        w_up_ptrs += BLOCK_K * stride_wk
        
    gate = acc_gate.to(tl.float32)
    up = acc_up.to(tl.float32)
    
    # SwiGLU: silu(gate) * up
    silu_gate = gate * tl.sigmoid(gate)
    out = silu_gate * up
    
    out_ptrs = out_ptr + (expert_start + offs_m[:, None]) * stride_outm + offs_n[None, :]
    tl.store(out_ptrs, out.to(out_ptr.dtype.element_ty), mask=mask_m[:, None])

    reference_out: torch.Tensor,
    solution_out: torch.Tensor,
    dtype: torch.dtype | None = None,
    override: dict | None = None,
) -> tuple[bool, str]:

        return False, f"shape mismatch: ref={tuple(reference_out.shape)} sol={tuple(solution_out.shape)}"

    if torch.isnan(solution_out).any():
        return False, "solution contains NaN"

        return False, "solution contains Inf"

    tol = tolerance_for_dtype(dtype, override)

    if tol["atol"] == 0 and tol["rtol"] == 0:
        if torch.equal(reference_out, solution_out):
            return True, "ok (exact)"

    ref_f = reference_out.float()
    sol_f = solution_out.float()

        return True, f"ok (atol={tol['atol']}, rtol={tol['rtol']})"
