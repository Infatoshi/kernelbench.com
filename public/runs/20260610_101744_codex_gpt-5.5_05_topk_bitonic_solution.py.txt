_CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cub/block/block_radix_sort.cuh>
#include <cub/warp/warp_merge_sort.cuh>

#include <stdint.h>
#include <vector>

static constexpr float NEG_INF = -3.4028234663852886e38f;

struct FloatGreater {
    __device__ __forceinline__ bool operator()(const float& a, const float& b) const {
        return a > b;
    }
};

__device__ __forceinline__ bool better_pair(float av, int ai, float bv, int bi) {
    return (av > bv) || ((av == bv) && (ai >= 0) && ((bi < 0) || (ai < bi)));
}

__device__ __forceinline__ void swap_pair(float &av, int &ai, float &bv, int &bi) {
    float tv = av;
    int ti = ai;
    av = bv;
    ai = bi;
    bv = tv;
    bi = ti;
}

template<int N, int THREADS>
__device__ __forceinline__ void bitonic_desc(float* vals, int* inds) {
    const int tid = threadIdx.x;
    #pragma unroll
    for (int size = 2; size <= N; size <<= 1) {
        for (int stride = size >> 1; stride > 0; stride >>= 1) {
            for (int i = tid; i < N; i += THREADS) {
                int j = i ^ stride;
                if (j > i) {
                    bool desc = ((i & size) == 0);
                    bool do_swap = desc
                        ? better_pair(vals[j], inds[j], vals[i], inds[i])
                        : better_pair(vals[i], inds[i], vals[j], inds[j]);
                    if (do_swap) {
                        swap_pair(vals[i], inds[i], vals[j], inds[j]);
                    }
                }
            }
            __syncthreads();
        }
    }
}

template<int K>
__device__ __forceinline__ void insert_local(float v, int idx, float (&vals)[K], int (&inds)[K]) {
    if (!better_pair(v, idx, vals[K - 1], inds[K - 1])) {
        return;
    }
    int pos = K - 1;
    while (pos > 0 && better_pair(v, idx, vals[pos - 1], inds[pos - 1])) {
        vals[pos] = vals[pos - 1];
        inds[pos] = inds[pos - 1];
        --pos;
    }
    vals[pos] = v;
    inds[pos] = idx;
}

template<int K, int SORT_N, int THREADS>
__global__ void segment_sort_kernel(
    const float* __restrict__ x,
    float* __restrict__ part_vals,
    int* __restrict__ part_inds,
    int batch,
    int n,
    int segments) {
    int seg = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    int seg_len = (n + segments - 1) / segments;
    int begin = seg * seg_len;
    int end = min(n, begin + seg_len);

    extern __shared__ unsigned char smem[];
    float* vals = reinterpret_cast<float*>(smem);
    int* inds = reinterpret_cast<int*>(vals + SORT_N);

    const float* row_ptr = x + static_cast<int64_t>(row) * n;
    for (int i = tid; i < SORT_N; i += THREADS) {
        int col = begin + i;
        if (col < end) {
            vals[i] = row_ptr[col];
            inds[i] = col;
        } else {
            vals[i] = NEG_INF;
            inds[i] = -1;
        }
    }
    __syncthreads();

    bitonic_desc<SORT_N, THREADS>(vals, inds);

    int64_t out_base = (static_cast<int64_t>(row) * segments + seg) * K;
    for (int i = tid; i < K; i += THREADS) {
        part_vals[out_base + i] = vals[i];
        part_inds[out_base + i] = inds[i];
    }
}

template<int K, int THREADS>
__global__ void scan_final_kernel(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_inds,
    int n) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    constexpr int CAND = K * THREADS;

    float local_vals[K];
    int local_inds[K];
    #pragma unroll
    for (int i = 0; i < K; ++i) {
        local_vals[i] = NEG_INF;
        local_inds[i] = -1;
    }

    const float* row_ptr = x + static_cast<int64_t>(row) * n;
    for (int col = tid; col < n; col += THREADS) {
        insert_local<K>(row_ptr[col], col, local_vals, local_inds);
    }

    extern __shared__ unsigned char smem[];
    float* vals = reinterpret_cast<float*>(smem);
    int* inds = reinterpret_cast<int*>(vals + CAND);
    int dst = tid * K;
    #pragma unroll
    for (int i = 0; i < K; ++i) {
        vals[dst + i] = local_vals[i];
        inds[dst + i] = local_inds[i];
    }
    __syncthreads();

    bitonic_desc<CAND, THREADS>(vals, inds);

    int64_t out_base = static_cast<int64_t>(row) * K;
    for (int i = tid; i < K; i += THREADS) {
        out_vals[out_base + i] = vals[i];
        out_inds[out_base + i] = static_cast<int64_t>(inds[i]);
    }
}

template<int K, int THREADS>
__global__ void merge_local_kernel(
    const float* __restrict__ part_vals,
    const int* __restrict__ part_inds,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_inds,
    int segments) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    constexpr int CAND = K * THREADS;

    float local_vals[K];
    int local_inds[K];
    #pragma unroll
    for (int i = 0; i < K; ++i) {
        local_vals[i] = NEG_INF;
        local_inds[i] = -1;
    }

    int total = segments * K;
    int64_t row_base = static_cast<int64_t>(row) * segments * K;
    for (int c = tid; c < total; c += THREADS) {
        float v = part_vals[row_base + c];
        int idx = part_inds[row_base + c];
        insert_local<K>(v, idx, local_vals, local_inds);
    }

    extern __shared__ unsigned char smem[];
    float* vals = reinterpret_cast<float*>(smem);
    int* inds = reinterpret_cast<int*>(vals + CAND);
    int dst = tid * K;
    #pragma unroll
    for (int i = 0; i < K; ++i) {
        vals[dst + i] = local_vals[i];
        inds[dst + i] = local_inds[i];
    }
    __syncthreads();

    bitonic_desc<CAND, THREADS>(vals, inds);

    int64_t out_base = static_cast<int64_t>(row) * K;
    for (int i = tid; i < K; i += THREADS) {
        out_vals[out_base + i] = vals[i];
        out_inds[out_base + i] = static_cast<int64_t>(inds[i]);
    }
}

template<int K, int GROUP, int SORT_N, int THREADS>
__global__ void group_merge_kernel(
    const float* __restrict__ in_vals,
    const int* __restrict__ in_inds,
    float* __restrict__ out_vals,
    int* __restrict__ out_inds,
    int segments,
    int groups) {
    int group = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    int start_seg = group * GROUP;
    int group_segments = min(GROUP, segments - start_seg);
    int total = group_segments * K;

    extern __shared__ unsigned char smem[];
    float* vals = reinterpret_cast<float*>(smem);
    int* inds = reinterpret_cast<int*>(vals + SORT_N);

    int64_t in_base = (static_cast<int64_t>(row) * segments + start_seg) * K;
    for (int i = tid; i < SORT_N; i += THREADS) {
        if (i < total) {
            vals[i] = in_vals[in_base + i];
            inds[i] = in_inds[in_base + i];
        } else {
            vals[i] = NEG_INF;
            inds[i] = -1;
        }
    }
    __syncthreads();

    bitonic_desc<SORT_N, THREADS>(vals, inds);

    int64_t out_base = (static_cast<int64_t>(row) * groups + group) * K;
    for (int i = tid; i < K; i += THREADS) {
        out_vals[out_base + i] = vals[i];
        out_inds[out_base + i] = inds[i];
    }
}

template<int THREADS>
__global__ void argmax_kernel(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_inds,
    int n) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    const float* row_ptr = x + static_cast<int64_t>(row) * n;
    float best_v = NEG_INF;
    int best_i = -1;
    for (int col = tid; col < n; col += THREADS) {
        float v = row_ptr[col];
        if (better_pair(v, col, best_v, best_i)) {
            best_v = v;
            best_i = col;
        }
    }

    __shared__ float vals[THREADS];
    __shared__ int inds[THREADS];
    vals[tid] = best_v;
    inds[tid] = best_i;
    __syncthreads();

    for (int stride = THREADS >> 1; stride > 0; stride >>= 1) {
        if (tid < stride && better_pair(vals[tid + stride], inds[tid + stride], vals[tid], inds[tid])) {
            vals[tid] = vals[tid + stride];
            inds[tid] = inds[tid + stride];
        }
        __syncthreads();
    }

    if (tid == 0) {
        out_vals[row] = vals[0];
        out_inds[row] = static_cast<int64_t>(inds[0]);
    }
}

template<int K, int THREADS>
void launch_scan_one(torch::Tensor x, torch::Tensor vals, torch::Tensor inds, cudaStream_t stream) {
    int batch = static_cast<int>(x.size(0));
    int n = static_cast<int>(x.size(1));
    size_t smem = (K * THREADS) * (sizeof(float) + sizeof(int));
    scan_final_kernel<K, THREADS><<<batch, THREADS, smem, stream>>>(
        x.data_ptr<float>(), vals.data_ptr<float>(), inds.data_ptr<int64_t>(), n);
}

template<int K, int SEGMENTS, int THREADS>
void launch_scan_two(torch::Tensor x, torch::Tensor vals, torch::Tensor inds, cudaStream_t stream) {
    int batch = static_cast<int>(x.size(0));
    int n = static_cast<int>(x.size(1));
    auto part_vals = torch::empty({batch, SEGMENTS, K}, x.options());
    auto part_inds = torch::empty({batch, SEGMENTS, K}, x.options().dtype(torch::kInt32));

    size_t smem1 = (K * THREADS) * (sizeof(float) + sizeof(int));
    scan_part_kernel<K, THREADS><<<dim3(SEGMENTS, batch), THREADS, smem1, stream>>>(
        x.data_ptr<float>(), part_vals.data_ptr<float>(), part_inds.data_ptr<int>(),
        batch, n, SEGMENTS);

    if constexpr (K == 16) {
        constexpr int MT = 32;
        size_t smem2 = (K * MT) * (sizeof(float) + sizeof(int));
        merge_local_kernel<K, MT><<<batch, MT, smem2, stream>>>(
            part_vals.data_ptr<float>(), part_inds.data_ptr<int>(),
            vals.data_ptr<float>(), inds.data_ptr<int64_t>(), SEGMENTS);
    } else {
        constexpr int MT = 64;
        size_t smem2 = (K * MT) * (sizeof(float) + sizeof(int));
        merge_local_kernel<K, MT><<<batch, MT, smem2, stream>>>(
            part_vals.data_ptr<float>(), part_inds.data_ptr<int>(),
            vals.data_ptr<float>(), inds.data_ptr<int64_t>(), SEGMENTS);
    }
}

template<int K, int BLOCK_THREADS, int ITEMS_PER_THREAD>
__global__ void cub_segment_kernel(
    const float* __restrict__ x,
    float* __restrict__ part_vals,
    int* __restrict__ part_inds,
    int batch,
    int n,
    int segments) {
    int seg = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    constexpr int SORT_N = BLOCK_THREADS * ITEMS_PER_THREAD;
    int seg_len = (n + segments - 1) / segments;
    int begin = seg * seg_len;
    int end = min(n, begin + seg_len);

    float keys[ITEMS_PER_THREAD];
    int vals[ITEMS_PER_THREAD];
    const float* row_ptr = x + static_cast<int64_t>(row) * n;
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        int col = begin + linear;
        if (linear < SORT_N && col < end) {
            keys[i] = row_ptr[col];
            vals[i] = col;
        } else {
            keys[i] = NEG_INF;
            vals[i] = -1;
        }
    }

    using Sort = cub::BlockRadixSort<float, BLOCK_THREADS, ITEMS_PER_THREAD, int>;
    __shared__ typename Sort::TempStorage temp;
    Sort(temp).SortDescending(keys, vals);

    int64_t out_base = (static_cast<int64_t>(row) * segments + seg) * K;
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < K) {
            part_vals[out_base + linear] = keys[i];
            part_inds[out_base + linear] = vals[i];
        }
    }
}

template<int K, int BLOCK_THREADS, int ITEMS_PER_THREAD>
__global__ void warp_segment_kernel(
    const float* __restrict__ x,
    float* __restrict__ part_vals,
    int* __restrict__ part_inds,
    int batch,
    int n,
    int segments) {
    int seg = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    constexpr int WARP_THREADS = 32;
    constexpr int WARPS = BLOCK_THREADS / WARP_THREADS;
    constexpr int WARP_ITEMS = WARP_THREADS * ITEMS_PER_THREAD;
    constexpr int CAND = WARPS * K;
    constexpr int FINAL_ITEMS = (CAND + BLOCK_THREADS - 1) / BLOCK_THREADS;
    constexpr int FINAL_N = FINAL_ITEMS * BLOCK_THREADS;
    int warp_id = tid / WARP_THREADS;
    int lane = tid & (WARP_THREADS - 1);

    int seg_len = (n + segments - 1) / segments;
    int begin = seg * seg_len;
    int end = min(n, begin + seg_len);
    const float* row_ptr = x + static_cast<int64_t>(row) * n;

    float keys[ITEMS_PER_THREAD];
    int vals[ITEMS_PER_THREAD];
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int offset = warp_id * WARP_ITEMS + lane * ITEMS_PER_THREAD + i;
        int col = begin + offset;
        if (col < end) {
            keys[i] = row_ptr[col];
            vals[i] = col;
        } else {
            keys[i] = NEG_INF;
            vals[i] = -1;
        }
    }

    using WarpSort = cub::WarpMergeSort<float, ITEMS_PER_THREAD, WARP_THREADS, int>;
    __shared__ typename WarpSort::TempStorage warp_temp[WARPS];
    __shared__ float cand_vals[CAND];
    __shared__ int cand_inds[CAND];
    WarpSort(warp_temp[warp_id]).Sort(keys, vals, FloatGreater());

    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = lane * ITEMS_PER_THREAD + i;
        if (linear < K) {
            int dst = warp_id * K + linear;
            cand_vals[dst] = keys[i];
            cand_inds[dst] = vals[i];
        }
    }
    __syncthreads();

    float bkeys[FINAL_ITEMS];
    int bvals[FINAL_ITEMS];
    #pragma unroll
    for (int i = 0; i < FINAL_ITEMS; ++i) {
        int linear = tid * FINAL_ITEMS + i;
        if (linear < CAND) {
            bkeys[i] = cand_vals[linear];
            bvals[i] = cand_inds[linear];
        } else {
            bkeys[i] = NEG_INF;
            bvals[i] = -1;
        }
    }

    using BlockSort = cub::BlockRadixSort<float, BLOCK_THREADS, FINAL_ITEMS, int>;
    __shared__ typename BlockSort::TempStorage block_temp;
    BlockSort(block_temp).SortDescending(bkeys, bvals);

    int64_t out_base = (static_cast<int64_t>(row) * segments + seg) * K;
    #pragma unroll
    for (int i = 0; i < FINAL_ITEMS; ++i) {
        int linear = tid * FINAL_ITEMS + i;
        if (linear < K) {
            part_vals[out_base + linear] = bkeys[i];
            part_inds[out_base + linear] = bvals[i];
        }
    }
}

template<int K, int BLOCK_THREADS, int ITEMS_PER_THREAD>
__global__ void cub_final_kernel(
    const float* __restrict__ part_vals,
    const int* __restrict__ part_inds,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_inds,
    int segments) {
    int row = blockIdx.x;
    int tid = threadIdx.x;
    constexpr int SORT_N = BLOCK_THREADS * ITEMS_PER_THREAD;
    int total = segments * K;
    int64_t row_base = static_cast<int64_t>(row) * segments * K;

    float keys[ITEMS_PER_THREAD];
    int vals[ITEMS_PER_THREAD];
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < total) {
            keys[i] = part_vals[row_base + linear];
            vals[i] = part_inds[row_base + linear];
        } else {
            keys[i] = NEG_INF;
            vals[i] = -1;
        }
    }

    using Sort = cub::BlockRadixSort<float, BLOCK_THREADS, ITEMS_PER_THREAD, int>;
    __shared__ typename Sort::TempStorage temp;
    Sort(temp).SortDescending(keys, vals);

    int64_t out_base = static_cast<int64_t>(row) * K;
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < K) {
            out_vals[out_base + linear] = keys[i];
            out_inds[out_base + linear] = static_cast<int64_t>(vals[i]);
        }
    }
}

template<int K, int GROUP, int BLOCK_THREADS, int ITEMS_PER_THREAD>
__global__ void cub_group_kernel(
    const float* __restrict__ in_vals,
    const int* __restrict__ in_inds,
    float* __restrict__ out_vals,
    int* __restrict__ out_inds,
    int segments,
    int groups) {
    int group = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    constexpr int SORT_N = BLOCK_THREADS * ITEMS_PER_THREAD;
    int start_seg = group * GROUP;
    int group_segments = min(GROUP, segments - start_seg);
    int total = group_segments * K;
    int64_t in_base = (static_cast<int64_t>(row) * segments + start_seg) * K;

    float keys[ITEMS_PER_THREAD];
    int vals[ITEMS_PER_THREAD];
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < total) {
            keys[i] = in_vals[in_base + linear];
            vals[i] = in_inds[in_base + linear];
        } else {
            keys[i] = NEG_INF;
            vals[i] = -1;
        }
    }

    using Sort = cub::BlockRadixSort<float, BLOCK_THREADS, ITEMS_PER_THREAD, int>;
    __shared__ typename Sort::TempStorage temp;
    Sort(temp).SortDescending(keys, vals);

    int64_t out_base = (static_cast<int64_t>(row) * groups + group) * K;
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < K) {
            out_vals[out_base + linear] = keys[i];
            out_inds[out_base + linear] = vals[i];
        }
    }
}

template<int K, int BLOCK_THREADS, int ITEMS_PER_THREAD>
__global__ void cub_atomic_kernel(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_inds,
    int* __restrict__ locks,
    int* __restrict__ ready,
    int n,
    int segments) {
    int seg = blockIdx.x;
    int row = blockIdx.y;
    int tid = threadIdx.x;
    constexpr int SORT_N = BLOCK_THREADS * ITEMS_PER_THREAD;

    if (seg == 0) {
        int64_t out_base = static_cast<int64_t>(row) * K;
        for (int i = tid; i < K; i += BLOCK_THREADS) {
            out_vals[out_base + i] = NEG_INF;
            out_inds[out_base + i] = -1;
        }
        if (tid == 0) {
            locks[row] = 0;
        }
        __syncthreads();
        if (tid == 0) {
            __threadfence();
            atomicExch(ready + row, 1);
        }
    }

    int seg_len = (n + segments - 1) / segments;
    int begin = seg * seg_len;
    int end = min(n, begin + seg_len);
    const float* row_ptr = x + static_cast<int64_t>(row) * n;

    float keys[ITEMS_PER_THREAD];
    int vals[ITEMS_PER_THREAD];
    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        int col = begin + linear;
        if (linear < SORT_N && col < end) {
            keys[i] = row_ptr[col];
            vals[i] = col;
        } else {
            keys[i] = NEG_INF;
            vals[i] = -1;
        }
    }

    using Sort = cub::BlockRadixSort<float, BLOCK_THREADS, ITEMS_PER_THREAD, int>;
    __shared__ typename Sort::TempStorage temp;
    __shared__ float cand_vals[K];
    __shared__ int cand_inds[K];
    Sort(temp).SortDescending(keys, vals);

    #pragma unroll
    for (int i = 0; i < ITEMS_PER_THREAD; ++i) {
        int linear = tid * ITEMS_PER_THREAD + i;
        if (linear < K) {
            cand_vals[linear] = keys[i];
            cand_inds[linear] = vals[i];
        }
    }
    __syncthreads();

    if (tid == 0) {
        while (atomicAdd(ready + row, 0) == 0) {
        }
        while (atomicCAS(locks + row, 0, 1) != 0) {
        }

        float cur_vals[K];
        int cur_inds[K];
        int64_t out_base = static_cast<int64_t>(row) * K;
        #pragma unroll
        for (int i = 0; i < K; ++i) {
            cur_vals[i] = out_vals[out_base + i];
            cur_inds[i] = static_cast<int>(out_inds[out_base + i]);
        }
        #pragma unroll
        for (int c = 0; c < K; ++c) {
            float v = cand_vals[c];
            int idx = cand_inds[c];
            if (idx >= 0 && better_pair(v, idx, cur_vals[K - 1], cur_inds[K - 1])) {
                int pos = K - 1;
                while (pos > 0 && better_pair(v, idx, cur_vals[pos - 1], cur_inds[pos - 1])) {
                    cur_vals[pos] = cur_vals[pos - 1];
                    cur_inds[pos] = cur_inds[pos - 1];
                    --pos;
                }
                cur_vals[pos] = v;
                cur_inds[pos] = idx;
            }
        }
        #pragma unroll
        for (int i = 0; i < K; ++i) {
            out_vals[out_base + i] = cur_vals[i];
            out_inds[out_base + i] = static_cast<int64_t>(cur_inds[i]);
        }
        __threadfence();
        atomicExch(locks + row, 0);
    }
}
