+def _matmul_fp8_kernel(
+    x_ptr,

+    y_ptr,
+    M_: tl.constexpr,
+    N_: tl.constexpr,
+    K_: tl.constexpr,

+    BLOCK_N: tl.constexpr,
+    BLOCK_K: tl.constexpr,
+    GROUP_M: tl.constexpr,
+):

+    num_pid_m = tl.cdiv(M_, BLOCK_M)
+    num_pid_n = tl.cdiv(N_, BLOCK_N)
+    num_pid_in_group = GROUP_M * num_pid_n
+    group_id = pid // num_pid_in_group
+    first_pid_m = group_id * GROUP_M
+    group_size_m = tl.minimum(num_pid_m - first_pid_m, GROUP_M)
+    pid_in_group = pid - group_id * num_pid_in_group
+    pid_m = first_pid_m + (pid_in_group % group_size_m)
+    pid_n = pid_in_group // group_size_m
+
+    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
+    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
+    offs_k = tl.arange(0, BLOCK_K)
+
+    a_ptrs = x_ptr + offs_m[:, None] * K_ + offs_k[None, :]
+    b_ptrs = w_ptr + offs_n[None, :] * K_ + offs_k[:, None]
+    acc = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)
+
+    for k0 in range(0, K_, BLOCK_K):
