import torch
import torch.nn as nn
import triton
import triton.language as tl

@triton.jit
def gemm_kernel(
    a_ptr, b_ptr, c_ptr, scale_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_wn, stride_wk,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    a_ptrs = a_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    b_ptrs = b_ptr + offs_k[:, None] * stride_wk + offs_bn[None, :] * stride_wn

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        a = tl.load(a_ptrs)
        b = tl.load(b_ptrs)
        accumulator = tl.dot(a, b, accumulator)
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_wk

    # Load scales
    scale = tl.load(scale_ptr + offs_bn)
    accumulator = accumulator * scale[None, :]

    # Store output
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    mask_c = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    tl.store(c_ptrs, accumulator.to(tl.bfloat16), mask=mask_c)

def triton_gemm(x, w, scale, M, N, K, BLOCK_SIZE_M, BLOCK_SIZE_N, BLOCK_SIZE_K, GROUP_SIZE_M, num_warps, num_stages):
    # Padding if K is not a multiple of BLOCK_SIZE_K
    if K % BLOCK_SIZE_K != 0:
        K_padded = triton.cdiv(K, BLOCK_SIZE_K) * BLOCK_SIZE_K
        x_padded = torch.zeros((M, K_padded), dtype=x.dtype, device=x.device)
        x_padded[:, :K].copy_(x)
        w_padded = torch.zeros((w.shape[0], K_padded), dtype=w.dtype, device=w.device)
        w_padded[:, :K].copy_(w)
        K_val = K_padded
    else:
        x_padded = x
        w_padded = w
        K_val = K

    y = torch.empty((M, N), device=x.device, dtype=torch.bfloat16)
    grid = lambda META: (triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),)
    
    gemm_kernel[grid](
        x_padded, w_padded, y, scale,
        M, N, K_val,
        x_padded.stride(0), x_padded.stride(1),
        w_padded.stride(0), w_padded.stride(1),
        y.stride(0), y.stride(1),
        BLOCK_SIZE_M=BLOCK_SIZE_M, BLOCK_SIZE_N=BLOCK_SIZE_N, BLOCK_SIZE_K=BLOCK_SIZE_K,
        GROUP_SIZE_M=GROUP_SIZE_M,
        num_warps=num_warps,
        num_stages=num_stages,
    )
    return y

class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.register_buffer("weight", torch.empty(N, K, dtype=torch.float8_e4m3fn))
        self.register_buffer("weight_scale", torch.empty(N, dtype=torch.float32))
        
        self.g = None
        self.static_x = None
        self.static_y = None

    def _forward_impl(self, x: torch.Tensor) -> torch.Tensor:
        # Configuration mapping for high-performance tuning on B200
        if self.M == 4096 and self.N == 4096 and self.K == 4096:
            return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                               256, 256, 128, 4, 8, 3)
        elif self.M == 4096 and self.N == 4096 and self.K == 4127:
            return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                               256, 256, 128, 1, 8, 3)
        elif self.M == 32 and self.N == 8192 and self.K == 8192:
            return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                               64, 64, 256, 4, 4, 4)
        elif self.M == 4096 and self.N == 14336 and self.K == 4096:
            return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                               256, 256, 128, 8, 8, 3)
        else:
            # Fallback configuration for generic shapes
            if self.M <= 64:
                return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                                   64, 64, 256, 4, 4, 4)
            else:
                return triton_gemm(x, self.weight, self.weight_scale, self.M, self.N, self.K,
                                   128, 128, 128, 8, 4, 3)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # CUDA Graph replay to eliminate Python launch overhead and boost geomean performance
        if self.g is None:
            # Warm up Triton and CUDA
            for _ in range(3):
                _ = self._forward_impl(x)
            
            self.static_x = torch.empty_like(x)
            self.static_x.copy_(x)
            
            # Run once to preallocate buffers correctly within the graph context
            self.static_y = self._forward_impl(self.static_x)
            
            self.g = torch.cuda.CUDAGraph()
            with torch.cuda.graph(self.g):
                self.static_y = self._forward_impl(self.static_x)
                
        self.static_x.copy_(x)
        self.g.replay()
        return self.static_y.clone()
