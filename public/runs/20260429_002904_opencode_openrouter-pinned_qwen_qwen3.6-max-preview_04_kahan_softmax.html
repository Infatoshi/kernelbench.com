"""Triton softmax with online max-reduce + Kahan-compensated summation.

Optimized for RTX PRO 6000 Blackwell (SM120, GDDR7, 1.8 TB/s).

Strategy: fused 2-pass online kernel with constexpr N for full loop unrolling.
Sweep-validated optimal config: BLOCK_N=4096, 8 warps for large shapes;
BLOCK_N=2048, 4 warps for small shapes.

Numerical approach:
  - Online max tracking with scale adjustment (avoids overflow on extreme logits)
  - Kahan-compensated summation of exp(x - running_max) for tight fp32 accuracy
  - Matches fp64 reference within atol=rtol=1e-5 on all shapes including 256K vocab
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "softmax"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _softmax_kernel(
    X_ptr,
    Y_ptr,
    stride_xr,
    stride_yr,
    N: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    """One-row-per-program online softmax.

    Pass 1: single loop tracking running_max and Kahan-sum of exp(x-max).
    Pass 2: re-read X, compute exp(x-final_max)/total_sum, write Y.

    2 memory passes (optimal for softmax), full fp32 Kahan-compensated.
    """
    row_id = tl.program_id(0)
    x_base = X_ptr + row_id * stride_xr
    y_base = Y_ptr + row_id * stride_yr

    running_max = tl.full([1], float('-inf'), dtype=tl.float32)
    running_sum = tl.full([1], 0.0, dtype=tl.float32)
    comp = tl.full([1], 0.0, dtype=tl.float32)

    for off in range(0, N, BLOCK_N):
        idx = off + tl.arange(0, BLOCK_N)
        mask = idx < N
        x = tl.load(x_base + idx, mask=mask, other=float('-inf')).to(tl.float32)

        chunk_max = tl.max(x, keep_dims=True)
        new_max = tl.maximum(running_max, chunk_max)
        scale = tl.exp(running_max - new_max)

        running_sum = running_sum * scale
        comp = comp * scale

        e = tl.exp(x - new_max)
        e_masked = tl.where(mask, e, 0.0)
        e_chunk = tl.sum(e_masked, keep_dims=True).to(tl.float32)

        y_k = e_chunk - comp
        new_sum = running_sum + y_k
        comp = (new_sum - running_sum) - y_k
        running_sum = new_sum
        running_max = new_max

    final_max = running_max.broadcast_to([BLOCK_N])
    total_sum_vec = running_sum.broadcast_to([BLOCK_N])

    for off in range(0, N, BLOCK_N):
        idx = off + tl.arange(0, BLOCK_N)
        mask = idx < N
        x = tl.load(x_base + idx, mask=mask, other=0.0).to(tl.float32)
        e = tl.exp(x - final_max)
        y = tl.where(mask, e / total_sum_vec, 0.0)
        tl.store(y_base + idx, y, mask=mask)


def get_launch_config(vocab):
    """Return (BLOCK_N, num_warps, num_stages) validated by sweep."""
    if vocab <= 2048:
        return vocab, 4, 3
    if vocab <= 4096:
        return 2048, 4, 3
    blk = min(4096, vocab)
    while vocab % blk != 0 and blk > 32:
        blk //= 2
    return blk, 8, 2


def triton_softmax(x):
    assert x.is_cuda and x.dtype == torch.float32 and x.ndim == 2
    batch, vocab = x.shape
    y = torch.empty_like(x)
    block_n, num_warps, num_stages = get_launch_config(vocab)
    _softmax_kernel[(batch,)](
        x, y, x.stride(0), y.stride(0), vocab,
        BLOCK_N=block_n,
        num_warps=num_warps,
        num_stages=num_stages,
    )
    return y

+"""Triton softmax with fp32 Kahan-compensated summation.
+
+Produces results matching the fp64 reference within atol=rtol=1e-5
+even for the 256K-vocab and extreme-logit shapes.
+"""
+import torch
+import torch.nn as nn
+import triton
+import triton.language as tl
+
+
+OP_TYPE = "softmax"
+SUPPORTED_PRECISIONS = ["fp32"]
+HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]
+
+
+# ---------------------------------------------------------------------------
+# Triton kernel
+# ---------------------------------------------------------------------------
+
+@triton.jit
+def _softmax_kernel_fwd(
+    X_ptr,
+    Y_ptr,
+    stride_xr,
+    stride_yr,
+    N: tl.constexpr,
+    BLOCK_N: tl.constexpr,
+):
+    """One-row-per-program softmax with fp32 Kahan-compensated summation.
+
+    Block-N must divide N (N is power of 2, BLOCK_N is power of 2 -> fine).
+    """
+    row_id = tl.program_id(0)
+    x_base = X_ptr + row_id * stride_xr
+    y_base = Y_ptr + row_id * stride_yr
+
+    # ---- Pass 1: find row-max ------------------------------------------
+    row_max = tl.full([1], -float('inf'), dtype=tl.float32)
+    for off0 in range(0, N, BLOCK_N):
+        idx = off0 + tl.arange(0, BLOCK_N)
+        x = tl.load(x_base + idx).to(tl.float32)
+        local_max = tl.max(x)
+        row_max = tl.maximum(row_max, local_max)
+
+    row_max = row_max.reshape(()).to(tl.float32)
+
+    # ---- Pass 2: Kahan-sum exp(x - max) --------------------------------
+    #  Kahan sum:  running_sum and c (compensation), both fp32.
+    running_sum = tl.zeros([1], dtype=tl.float32)
+    comp = tl.zeros([1], dtype=tl.float32)
+
+    for off0 in range(0, N, BLOCK_N):
+        idx = off0 + tl.arange(0, BLOCK_N)
+        x = tl.load(x_base + idx).to(tl.float32)
+        e = tl.exp(x - row_max)
+        # Kahan step:  y = e - c;  new_sum = running_sum + y;  c = (new_sum - running_sum) - y
+        y_kahan = e - comp
+        tmp_sum = running_sum + y_kahan
+        comp = (tmp_sum - running_sum) - y_kahan
+        running_sum = tmp_sum
+
+    total_sum = running_sum.reshape(()).to(tl.float32)
+
+    # ---- Pass 3: write out exp(x - max) / total_sum --------------------
+    for off0 in range(0, N, BLOCK_N):
+        idx = off0 + tl.arange(0, BLOCK_N)
+        x = tl.load(x_base + idx).to(tl.float32)
+        e = tl.exp(x - row_max)
+        y = e / total_sum
+        tl.store(y_base + idx, y.to(tl.float32))
+
+
+def triton_softmax(x: torch.Tensor) -> torch.Tensor:
+    """x: (batch, vocab) fp32 on CUDA -> y: (batch, vocab) fp32 on CUDA."""
+    assert x.is_cuda and x.dtype == torch.float32 and x.ndim == 2
+    batch, vocab = x.shape
+    y = torch.empty_like(x)
+
+    # Pick BLOCK_N that divides vocab.  For small vocab use vocab itself.
+    # For large vocab use a reasonable block that keeps registers in check.
+    if vocab <= 2048:
+        BLOCK_N = vocab
+    elif vocab <= 8192:
+        BLOCK_N = 2048
+    elif vocab <= 32768:
+        BLOCK_N = 4096
+    else:
+        BLOCK_N = 4096
+
+    # Ensure BLOCK_N divides N
+    while vocab % BLOCK_N != 0 and BLOCK_N > 64:
+        BLOCK_N //= 2
+
+    grid = (batch,)
+    _softmax_kernel_fwd[grid](
+        x, y,
+        x.stride(0), y.stride(0),
+        vocab,
+        BLOCK_N=BLOCK_N,
+    )
+    return y
+

+        self.batch = batch

+

+        return triton_softmax(x)
