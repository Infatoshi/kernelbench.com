"""Triton-based grouped GEMM + fused SwiGLU kernel for MoE up-projection."""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _swiglu_forward_kernel(
    hidden_states,   # (T_perm, H)
    expert_offsets,  # (E+1,)
    W_gate,          # (E, H, I)
    W_up,            # (E, H, I)
    output,          # (T_perm, I)
    H: tl.constexpr,
    I: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_K: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    pid_e = tl.program_id(0)
    pid_n = tl.program_id(1)

    start = tl.load(expert_offsets + pid_e)
    end = tl.load(expert_offsets + pid_e + 1)
    M = end - start
    if M <= 0:
        return

    hs_base = hidden_states + start * H
    wg_base = W_gate + pid_e * H * I
    wu_base = W_up + pid_e * H * I
    out_base = output + start * I

    col_offsets = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    col_mask = col_offsets < I

    for off_m in range(0, M, BLOCK_M):
        m_offsets = off_m + tl.arange(0, BLOCK_M)
        m_mask = m_offsets < M

        acc_gate = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
        acc_up = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

        for off_k in range(0, H, BLOCK_K):
            k_offsets = off_k + tl.arange(0, BLOCK_K)
            k_mask = k_offsets < H

            a_ptrs = hs_base + m_offsets[:, None] * H + k_offsets[None, :]
            a = tl.load(a_ptrs, mask=m_mask[:, None] & k_mask[None, :], other=0.0).to(tl.float32)

            wg_ptrs = wg_base + k_offsets[:, None] * I + col_offsets[None, :]
            wg_tile = tl.load(wg_ptrs, mask=k_mask[:, None] & col_mask[None, :], other=0.0).to(tl.float32)
            acc_gate = tl.dot(a, wg_tile, acc=acc_gate)

            wu_ptrs = wu_base + k_offsets[:, None] * I + col_offsets[None, :]
            wu_tile = tl.load(wu_ptrs, mask=k_mask[:, None] & col_mask[None, :], other=0.0).to(tl.float32)
            acc_up = tl.dot(a, wu_tile, acc=acc_up)

        silu_val = acc_gate * tl.sigmoid(acc_gate)
        final_val = silu_val * acc_up

        o_ptrs = out_base + m_offsets[:, None] * I + col_offsets[None, :]
        tl.store(o_ptrs, final_val, mask=m_mask[:, None] & col_mask[None, :])


def _select_tiling(H, I):
    """Select tiling based on problem shape."""
    if H >= 4096 and I <= 1536:
        return 128, 64, 64  # HEADLINE: many K steps, moderate N
    elif I >= 4096:
        return 64, 64, 128  # WIDE: more N tiling
    else:
        return 128, 64, 64  # DEFAULT


def _forward(hidden_states, expert_offsets, W_gate, W_up, E, H, I):
    T_perm, _ = hidden_states.shape
    output = torch.empty(T_perm, I, dtype=torch.bfloat16, device=hidden_states.device)

    BLOCK_M, BLOCK_K, BLOCK_N = _select_tiling(H, I)
    num_n_tiles = triton.cdiv(I, BLOCK_N)

    # Grid: (E, num_n_tiles) -> E * num_n_tiles blocks
    grid = (E, max(1, num_n_tiles), 1)

    _swiglu_forward_kernel[grid](
        hidden_states, expert_offsets, W_gate, W_up,
        output, H, I, BLOCK_M, BLOCK_K, BLOCK_N,
    )
    return output
