"""Paged-attention decode kernel (Triton) for B200 / SM100.

Single-query decode. Each batch element has a query (num_heads, head_dim) and
attends over a KV cache of seq_len[b] tokens stored as fixed-size pages in a
global pool; block_table[b] lists the pages for batch element b.

KV cache layout: (num_blocks, page_size, num_kv_heads, head_dim*2), last dim
packs [K | V] so one gather pulls both. Output: (batch, num_heads, head_dim).

Strategy
--------
Decode attention is memory-bound: the KV cache is streamed once. The dominant
cost is therefore KV bandwidth, and GQA means multiple query heads share the
same KV head. To avoid re-streaming the KV cache once per query head, we launch
one program per (batch, kv_head) and process all G = num_heads/num_kv_heads
query heads of that group together, sharing a single KV stream.

Within a program we loop over pages (PAGE_SIZE tokens each). For each page we
load the packed [K|V] tile once and compute, for all G query heads at once:
    S    = Q @ K^T * scale      (G, P)
    P    = softmax(S)           (online, across pages)
    O   += P @ V                (G, D)
Online softmax rescales the running accumulator so numerical behaviour matches
the fp32 reference within the 0.02 tolerance.
"""
from __future__ import annotations

import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# Module-level shape knobs (overridden by check.py / benchmark.py).
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16

MAX_G = 8  # max num_heads // num_kv_heads handled per program (masked at runtime)


@triton.jit
def paged_decode_kernel(
    Q, KV, BLOCK_TABLE, SEQ_LENS, OUT,
    stride_qb, stride_qh, stride_qd,
    stride_kvb, stride_kvp, stride_kvh, stride_kvd,
    stride_bt,
    stride_ob, stride_oh, stride_od,
    B, Hkv, G, D, P, scale: tl.constexpr,
    MAX_G: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_P: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // Hkv
    hkv = pid % Hkv

    L = tl.load(SEQ_LENS + b)
    gv = tl.arange(0, MAX_G)
    gmask = gv < G

    q_offs = (b * stride_qb
              + (hkv * G + gv[:, None]) * stride_qh
              + tl.arange(0, BLOCK_D)[None, :] * stride_qd)
    q = tl.load(Q + q_offs, mask=gmask[:, None], other=0.0).to(tl.float32)

    m = tl.full((MAX_G,), float("-inf"), dtype=tl.float32)
    l = tl.full((MAX_G,), 0.0, dtype=tl.float32)
    o = tl.zeros((MAX_G, BLOCK_D), dtype=tl.float32)

    num_pages = tl.cdiv(L, P)
    pidx = tl.arange(0, BLOCK_P)
    didx = tl.arange(0, BLOCK_D)

    for p in range(0, 512):
        if p >= num_pages:
            break
        blk = tl.load(BLOCK_TABLE + b * stride_bt + p)
        kv_base = (blk * stride_kvb + hkv * stride_kvh)
        # Load packed [K|V] for this page: (P, 2D).
        pos = pidx
        tok_idx = p * P + pos
        valid = tok_idx < L
        packed_offs = (kv_base
                       + pos[:, None] * stride_kvp
                       + didx[None, :] * stride_kvd)
        packed = tl.load(KV + packed_offs, mask=valid[:, None], other=0.0).to(tl.float32)
        k = packed[:, :D]
        v = packed[:, D:]

        # S = Q @ K^T * scale  -> (G, P)
        s = tl.dot(q, tl.trans(k)) * scale  # (G, P)
        s = tl.where(valid[None, :], s, float("-inf"))

        m_cur = tl.max(s, axis=1)            # (G,)
        m_new = tl.maximum(m, m_cur)         # (G,)
        alpha = tl.exp(m - m_new)            # (G,)
        p_soft = tl.exp(s - m_new[:, None])  # (G, P)
        l_cur = tl.sum(p_soft, axis=1)       # (G,)

        l = l * alpha + l_cur
        o = o * alpha[:, None] + tl.dot(p_soft, v)  # (G, D)
        m = m_new

    o = o / l[:, None]
    out_offs = (b * stride_ob
                + (hkv * G + gv[:, None]) * stride_oh
                + tl.arange(0, BLOCK_D)[None, :] * stride_od)
    tl.store(OUT + out_offs, o.to(tl.bfloat16), mask=gmask[:, None])


class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size
        assert D in (64, 128) and P == 16

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        grid = (B * Hkv,)
        paged_decode_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0),
            out.stride(0), out.stride(1), out.stride(2),
            B, Hkv, G, D, P, self.scale,
            MAX_G=MAX_G,
            BLOCK_D=D,
            BLOCK_P=P,
            num_warps=4,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
