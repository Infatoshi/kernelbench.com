"""Numerically tight softmax with Kahan compensated summation.

Single-block path for smaller vocabs (V ≤ 32768) where one-kernel-launch
simplicity wins.  Multi-block map-reduce for large vocabs where parallelism
across blocks is needed to saturate GPU bandwidth.

Map:    each block computes local (max, Kahan-sum-of-exp) for its chunk.
Reduce: GPU-side Kahan combination of per-block results (num_warps=1).
Norm:   each block normalizes its chunk using global (max, sum).
"""

import torch
import torch.nn as nn
import triton
import triton.language as tl

# ---------------------------------------------------------------------------
# Single-block kernel
# ---------------------------------------------------------------------------

@triton.jit
def _single_kernel(x_ptr, y_ptr, V: tl.constexpr, BLOCK_VOCAB: tl.constexpr):
    """Online softmax + Kahan sum, then normalize.  One block per row."""
    pid = tl.program_id(0)
    offs = tl.arange(0, BLOCK_VOCAB)

    m = float('-inf')
    s = 0.0
    c = 0.0

    for start in range(0, V, BLOCK_VOCAB):
        mask = (start + offs) < V
        addr = pid * V + start + offs
        x = tl.load(x_ptr + addr, mask=mask, other=float('-inf'))

        m_local = tl.max(tl.where(mask, x, float('-inf')))
        m_new = tl.maximum(m, m_local)
        scale = tl.exp(m - m_new)

        s = s * scale
        c = c * scale

        exp_vals = tl.exp(x - m_new)
        chunk_sum = tl.sum(tl.where(mask, exp_vals, 0.0))

        t = s
        y_k = chunk_sum - c
        s = t + y_k
        c = (s - t) - y_k

        m = m_new

    inv_sum = 1.0 / s

    for start in range(0, V, BLOCK_VOCAB):
        mask = (start + offs) < V
        addr = pid * V + start + offs
        x = tl.load(x_ptr + addr, mask=mask, other=0.0)
        y = tl.exp(x - m) * inv_sum
        tl.store(y_ptr + addr, y, mask=mask)


# ---------------------------------------------------------------------------
# Multi-block kernels
# ---------------------------------------------------------------------------

@triton.jit
def _map_kernel(x_ptr, max_buf, sum_buf,
                elems_per_block: tl.constexpr, BLOCK_VOCAB: tl.constexpr):
    pid = tl.program_id(0)
    offs = tl.arange(0, BLOCK_VOCAB)
    base = pid * elems_per_block

    m = float('-inf')
    s = 0.0
    c = 0.0

    for start in range(0, elems_per_block, BLOCK_VOCAB):
        mask = (start + offs) < elems_per_block
        addr = base + start + offs
        x = tl.load(x_ptr + addr, mask=mask, other=float('-inf'))

        m_local = tl.max(tl.where(mask, x, float('-inf')))
        m_new = tl.maximum(m, m_local)
        scale = tl.exp(m - m_new)

        s = s * scale
        c = c * scale

        exp_vals = tl.exp(x - m_new)
        chunk_sum = tl.sum(tl.where(mask, exp_vals, 0.0))

        t = s
        y_k = chunk_sum - c
        s = t + y_k
        c = (s - t) - y_k

        m = m_new

    tl.store(max_buf + pid, m)
    tl.store(sum_buf + pid, s)


@triton.jit
def _reduce_kernel(max_in, sum_in, max_out, sum_out,
                   blocks_per_row: tl.constexpr):
    rid = tl.program_id(0)
    base = rid * blocks_per_row

    m = float('-inf')
    s = 0.0
    c = 0.0

    for k in range(blocks_per_row):
        mk = tl.load(max_in + (base + k))
        sk = tl.load(sum_in + (base + k))

        m_new = tl.maximum(m, mk)
        scale = tl.exp(m - m_new)
        s = s * scale
        c = c * scale

        if mk < m_new:
            sk = sk * tl.exp(mk - m_new)

        t = s
        y_k = sk - c
        s = t + y_k
        c = (s - t) - y_k

        m = m_new

    tl.store(max_out + rid, m)
    tl.store(sum_out + rid, s)


@triton.jit
def _norm_kernel(x_ptr, y_ptr, max_per_row, sum_per_row, row_ids,
                 elems_per_block: tl.constexpr, BLOCK_VOCAB: tl.constexpr):
    pid = tl.program_id(0)
    offs = tl.arange(0, BLOCK_VOCAB)

    row = tl.load(row_ids + pid)
    m_global = tl.load(max_per_row + row)
    s_global = tl.load(sum_per_row + row)
    inv_sum = 1.0 / s_global

    base = pid * elems_per_block
    for start in range(0, elems_per_block, BLOCK_VOCAB):
        mask = (start + offs) < elems_per_block
        addr = base + start + offs
        x = tl.load(x_ptr + addr, mask=mask, other=0.0)
        y = tl.exp(x - m_global) * inv_sum
        tl.store(y_ptr + addr, y, mask=mask)


# ---------------------------------------------------------------------------
# Tuning
# ---------------------------------------------------------------------------

def _single_config(vocab: int) -> tuple:
    """Return (BLOCK_VOCAB, num_warps) for single-block path."""
    if vocab <= 4096:
        return (1024, 4)
    elif vocab <= 16384:
        return (2048, 8)
    else:
        return (4096, 8)


def _multi_config(vocab: int) -> tuple:
    """Return (BLOCK_VOCAB, num_warps) for multi-block path."""
    if vocab <= 32768:
        return (2048, 8)
    elif vocab <= 131072:
        return (4096, 8)
    else:
        return (8192, 8)


# ---------------------------------------------------------------------------
# Model
# ---------------------------------------------------------------------------

class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = batch
        self.vocab = vocab

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, V = x.shape
        device = x.device

        # Single-block path for small vocabs or high-batch cases
        if V <= 32768:
            y = torch.empty_like(x)
            BLOCK_VOCAB, num_warps = _single_config(V)
            _single_kernel[(B,)](
                x, y,
                V=V,
                BLOCK_VOCAB=BLOCK_VOCAB,
                num_warps=num_warps,
            )
            return y

        # ---- Multi-block path for large vocabs (> 32768) ----
        BLOCK_VOCAB, num_warps = _multi_config(V)

        # Choose blocks_per_row as a power-of-2 dividing V
        target = max(1, min(64 // max(B, 1), V // 4096))
        blocks_per_row = triton.next_power_of_2(target)
        while V % blocks_per_row != 0 and blocks_per_row > 1:
            blocks_per_row //= 2
        elems_per_block = V // blocks_per_row
        total_blocks = B * blocks_per_row

        row_ids = torch.arange(B, device=device, dtype=torch.int32).repeat_interleave(blocks_per_row)

        # 1) Map
        max_buf = torch.empty(total_blocks, dtype=torch.float32, device=device)
        sum_buf = torch.empty(total_blocks, dtype=torch.float32, device=device)

        _map_kernel[(total_blocks,)](
            x, max_buf, sum_buf,
            elems_per_block=elems_per_block,
            BLOCK_VOCAB=BLOCK_VOCAB,
            num_warps=num_warps,
        )

        # 2) Reduce
        global_max = torch.empty(B, dtype=torch.float32, device=device)
        global_sum = torch.empty(B, dtype=torch.float32, device=device)

        _reduce_kernel[(B,)](
            max_buf, sum_buf, global_max, global_sum,
            blocks_per_row=blocks_per_row,
            num_warps=1,
        )

        # 3) Norm
        y = torch.empty_like(x)
        _norm_kernel[(total_blocks,)](
            x, y, global_max, global_sum, row_ids,
            elems_per_block=elems_per_block,
            BLOCK_VOCAB=BLOCK_VOCAB,
            num_warps=num_warps,
        )

        return y

+
+Two-pass Triton kernel:
+  Pass 1 (forward): online max-finding + Kahan-compensated summation of exp(x - max)
+  Pass 2 (backward): re-read input, compute exp(x - max) / sum, write output
+

+which is where naive fp32 summation over 256K elements drifts past 1e-5.
