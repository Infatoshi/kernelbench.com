+"""Calibrate achievable DRAM bandwidth & cuBLAS compute for the W4A16 shapes."""
+import torch, triton, triton.language as tl, reference, shapes
+import statistics
+
+dev = torch.device("cuda:0")
+PEAK_BW = 2039.0
+
+@triton.jit
+def _read_reduce(wq_ptr, out_ptr, N, K, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr):
+    # Pure streaming read of the (K//2, N) uint8 weight, reduce to bandwidth probe.
+    pid = tl.program_id(0)
+    offs_n = pid * BLOCK_N + tl.arange(0, BLOCK_N)
+    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
+    for kh in range(0, K // 2, BLOCK_K):
+        offs_k = kh + tl.arange(0, BLOCK_K)
+        w = tl.load(wq_ptr + offs_k[:, None] * N + offs_n[None, :]).to(tl.float32)
+        acc += tl.sum(w, axis=0)
+    tl.store(out_ptr + offs_n, acc)
+
+
+def t_ms(fn, n=50, warm=20):
+    for _ in range(warm): fn()
+    torch.cuda.synchronize()
+    ts = []
+    for _ in range(n):
+        a, b = torch.cuda.Event(enable_timing=True), torch.cuda.Event(enable_timing=True)
+        a.record(); fn(); b.record(); torch.cuda.synchronize()
+        ts.append(a.elapsed_time(b))
+    return statistics.median(ts)
+
