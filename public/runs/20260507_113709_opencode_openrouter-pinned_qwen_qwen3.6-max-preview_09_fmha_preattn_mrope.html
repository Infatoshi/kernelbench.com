"""Fused multi-axis RoPE pre-attention prep via Triton kernels.

Two-stage with separate q and k rotation kernels:
  Stage 1: grid=(B*T,). Gather+mix cos/sin once per position => (B*T, D).
  Stage 2: grid=(B*T, Hq) for q, (B*T, Hkv) for k.
    Each block loads cos/sin ONCE, applies rotate-half, writes transposed output.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "rope"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000"]


@triton.jit
def _mix_cos_sin_kernel(
    cos_cache_ptr, sin_cache_ptr,
    pos_t_ptr, pos_h_ptr, pos_w_ptr,
    cos_out_ptr, sin_out_ptr,
    B, T, D, max_pos,
    s0, s1, s2, half_D,
    stride_pos,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // T
    t = pid % T

    pos_base = b * stride_pos + t
    pt = tl.load(pos_t_ptr + pos_base)
    ph = tl.load(pos_h_ptr + pos_base)
    pw = tl.load(pos_w_ptr + pos_base)

    d_offs = tl.arange(0, BLOCK_D)
    mask = d_offs < D

    row_t = pt.to(tl.int64) * D
    row_h = ph.to(tl.int64) * D
    row_w = pw.to(tl.int64) * D

    cos_t = tl.load(cos_cache_ptr + row_t + d_offs, mask=mask)
    cos_h = tl.load(cos_cache_ptr + row_h + d_offs, mask=mask)
    cos_w = tl.load(cos_cache_ptr + row_w + d_offs, mask=mask)

    sin_t = tl.load(sin_cache_ptr + row_t + d_offs, mask=mask)
    sin_h = tl.load(sin_cache_ptr + row_h + d_offs, mask=mask)
    sin_w = tl.load(sin_cache_ptr + row_w + d_offs, mask=mask)

    cut0_s1 = s0 + s1

    cos_mixed = tl.where(
        d_offs < s0, cos_t,
        tl.where(d_offs < cut0_s1, cos_h,
            tl.where(d_offs < half_D, cos_w,
                tl.where(d_offs < half_D + s0, cos_t,
                    tl.where(d_offs < half_D + cut0_s1, cos_h, cos_w)))))

    sin_mixed = tl.where(
        d_offs < s0, sin_t,
        tl.where(d_offs < cut0_s1, sin_h,
            tl.where(d_offs < half_D, sin_w,
                tl.where(d_offs < half_D + s0, sin_t,
                    tl.where(d_offs < half_D + cut0_s1, sin_h, sin_w)))))

    out_base = pid * D
    tl.store(cos_out_ptr + out_base + d_offs, cos_mixed, mask=mask)
    tl.store(sin_out_ptr + out_base + d_offs, sin_mixed, mask=mask)


@triton.jit
def _rope_transpose_kernel(
    x_ptr, cos_ptr, sin_ptr, out_ptr,
    T, H, D, half_D,
    stride_x_bt, stride_x_h,
    stride_cs,
    stride_out_b, stride_out_h, stride_out_t,
    BLOCK_D: tl.constexpr,
):
    """Rotate + transpose kernel. Writes directly to (B,H,T,D).
    Grid: (1, B*T, H).
    x: (B*T, H, D) flat view.
    cos/sin: (B*T, D) flat.
    out: (B, H, T, D)."""
    pid_bt = tl.program_id(1)
    pid_h  = tl.program_id(2)

    b = pid_bt // T
    t = pid_bt % T

    d_offs = tl.arange(0, BLOCK_D)
    mask_d = d_offs < D
    mask_half = d_offs < half_D
    mask_second = (d_offs + half_D) < D

    # Load cos/sin ONCE (shared across all heads for this position)
    cs_base = pid_bt * stride_cs
    cos_vals = tl.load(cos_ptr + cs_base + d_offs, mask=mask_d)
    sin_vals = tl.load(sin_ptr + cs_base + d_offs, mask=mask_d)

    # Load x for this head
    x_base = pid_bt * stride_x_bt + pid_h * stride_x_h
    x_d = tl.load(x_ptr + x_base + d_offs, mask=mask_half)
    x_dh = tl.load(x_ptr + x_base + d_offs + half_D, mask=mask_second)

    # Rotate
    out_first = x_d * cos_vals - x_dh * sin_vals
    out_second = x_dh * cos_vals + x_d * sin_vals

    # Write to (B,H,T,D) layout directly
    out_base = b * stride_out_b + pid_h * stride_out_h + t * stride_out_t
    tl.store(out_ptr + out_base + d_offs, out_first, mask=mask_half)
    tl.store(out_ptr + out_base + d_offs + half_D, out_second, mask=mask_second)


def _next_power_of_2(n):
    return 1 << (n - 1).bit_length()


class Model(nn.Module):
    def __init__(self, B: int, T: int, Hq: int, Hkv: int, D: int,
                 mrope_section: tuple[int, int, int], max_pos: int):
        super().__init__()
        assert sum(mrope_section) == D // 2
        self.B, self.T = B, T
        self.Hq, self.Hkv, self.D = Hq, Hkv, D
        self.mrope_section = tuple(mrope_section)
        self.max_pos = max_pos

        inv_freq = 1.0 / (10000.0 ** (torch.arange(0, D, 2, dtype=torch.float32) / D))
        pos = torch.arange(max_pos, dtype=torch.float32)
        freqs = torch.outer(pos, inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        self.register_buffer("cos_cache", emb.cos().to(torch.bfloat16))
        self.register_buffer("sin_cache", emb.sin().to(torch.bfloat16))
        self._BLOCK_D = _next_power_of_2(D)

    def forward(
        self,
        q: torch.Tensor, k: torch.Tensor,
        pos_t: torch.Tensor, pos_h: torch.Tensor, pos_w: torch.Tensor,
    ) -> tuple[torch.Tensor, torch.Tensor]:

        _, _, Hkv, _ = k.shape
        half_D = D // 2
        s0, s1, s2 = self.mrope_section
        BT = B * T
        BLOCK_D = self._BLOCK_D

        # Stage 1: gather + mix cos/sin
        cos_mixed = torch.empty(BT, D, dtype=torch.bfloat16, device=q.device)
        sin_mixed = torch.empty(BT, D, dtype=torch.bfloat16, device=q.device)

        grid_mix = (BT,)
        _mix_cos_sin_kernel[grid_mix](
            self.cos_cache, self.sin_cache,
            pos_t, pos_h, pos_w,
            cos_mixed, sin_mixed,
            B, T, D, self.max_pos,
            s0, s1, s2, half_D,
            pos_t.stride(0),
            BLOCK_D=BLOCK_D,
            num_warps=4,
        )

        # Stage 2q: rotate + transpose q
        q_flat = q.view(BT, Hq, D)
        q_rot = torch.empty(B, Hq, T, D, dtype=torch.bfloat16, device=q.device)
        _rope_transpose_kernel[(1, BT, Hq)](
            q_flat, cos_mixed, sin_mixed, q_rot,
            T, Hq, D, half_D,
            q_flat.stride(0), q_flat.stride(1),
            cos_mixed.stride(0),
            q_rot.stride(0), q_rot.stride(1), q_rot.stride(2),
            BLOCK_D=BLOCK_D,
            num_warps=4,
        )

        # Stage 2k: rotate + transpose k
        k_flat = k.view(BT, Hkv, D)

            k_flat, cos_mixed, sin_mixed, k_rot,
            T, Hkv, D, half_D,
            k_flat.stride(0), k_flat.stride(1),
            cos_mixed.stride(0),
            k_rot.stride(0), k_rot.stride(1), k_rot.stride(2),
            BLOCK_D=BLOCK_D,
            num_warps=4,
        )

        return q_rot, k_rot
