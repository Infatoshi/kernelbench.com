"""Triton grouped GEMM + fused SwiGLU for MoE up-projection.

Per expert e, compute:
    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
where x_e is the slice of permuted hidden states for expert e.

This uses a single fused kernel where each block binary-searches a compact
per-expert tile-offset table to discover its (expert, M-tile, N-tile).
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


# ---------------------------------------------------------------------------
# Triton kernel
# ---------------------------------------------------------------------------

@triton.jit
def grouped_gemm_swiglu_kernel(
    a_ptr,
    gate_ptr,
    up_ptr,
    out_ptr,
    expert_offsets_ptr,
    expert_tile_offsets_ptr,
    total_tiles,
    E: tl.constexpr,
    N: tl.constexpr,
    K: tl.constexpr,
    stride_am,
    stride_ak,
    stride_gate_e,
    stride_gate_k,
    stride_gate_n,
    stride_up_e,
    stride_up_k,
    stride_up_n,
    stride_out_m,
    stride_out_n,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    """Compute silu(A @ W_gate) * (A @ W_up) for one (expert, M-tile, N-tile)."""
    tile_idx = tl.program_id(axis=0)

    # Locate the expert for this tile by finding the largest e with
    # expert_tile_offsets[e] <= tile_idx.  E <= 128, so a vectorized scan is
    # simple and avoids tricky scalar binary-search bookkeeping in Triton.
    idx = tl.arange(0, E)
    offsets_vec = tl.load(expert_tile_offsets_ptr + idx)
    mask = offsets_vec <= tile_idx
    e = tl.sum(mask, dtype=tl.int32) - 1

    start = tl.load(expert_offsets_ptr + e)
    end = tl.load(expert_offsets_ptr + e + 1)

    tile_offset = tl.load(expert_tile_offsets_ptr + e)
    local_tile_idx = tile_idx - tile_offset
    num_n_tiles = N // BLOCK_N
    m_block = local_tile_idx // num_n_tiles
    n_block = local_tile_idx % num_n_tiles

    m_start = start + m_block * BLOCK_M
    n_start = n_block * BLOCK_N

    offs_m = m_start + tl.arange(0, BLOCK_M)
    offs_n = n_start + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    # Pointers for A (slice of hidden states) and B weights for this expert.
    a_ptrs = a_ptr + (offs_m[:, None] * stride_am + offs_k[None, :] * stride_ak)
    gate_ptrs = (
        gate_ptr
        + e * stride_gate_e
        + (offs_k[:, None] * stride_gate_k + offs_n[None, :] * stride_gate_n)
    )
    up_ptrs = (
        up_ptr
        + e * stride_up_e
        + (offs_k[:, None] * stride_up_k + offs_n[None, :] * stride_up_n)
    )

    acc_gate = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc_up = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    # K is the hidden dimension H; it is always a multiple of small BLOCK_K choices.
    for k in range(0, K, BLOCK_K):
        a = tl.load(a_ptrs, mask=offs_m[:, None] < end, other=0.0)
        gate_b = tl.load(gate_ptrs)
        up_b = tl.load(up_ptrs)

        acc_gate = tl.dot(a, gate_b, acc_gate)
        acc_up = tl.dot(a, up_b, acc_up)

        a_ptrs += BLOCK_K * stride_ak
        gate_ptrs += BLOCK_K * stride_gate_k
        up_ptrs += BLOCK_K * stride_up_k

    # Fused SwiGLU epilogue: silu(gate) * up.
    gate_f32 = acc_gate  # already fp32
    up_f32 = acc_up
    out = gate_f32 * tl.sigmoid(gate_f32) * up_f32
    out = out.to(tl.bfloat16)

    # Store output tile.
    out_ptrs = out_ptr + (offs_m[:, None] * stride_out_m + offs_n[None, :] * stride_out_n)
    out_mask = (offs_m[:, None] < end) & (offs_n[None, :] < N)
    tl.store(out_ptrs, out, mask=out_mask)


# ---------------------------------------------------------------------------
# Host-side tile-offset construction
# ---------------------------------------------------------------------------

def _build_expert_tile_offsets(
    expert_offsets: torch.Tensor,
    N: int,
    BLOCK_M: int,
    BLOCK_N: int,
) -> tuple[torch.Tensor, int]:
    """Build prefix-sum of per-expert tile counts (length E+1) on the same device."""
    device = expert_offsets.device
    E = expert_offsets.shape[0] - 1
    num_n_tiles = (N + BLOCK_N - 1) // BLOCK_N

    offsets_cpu = expert_offsets.cpu().tolist()
    tile_offsets_l = [0]
    total = 0
    for e in range(E):
        m_e = offsets_cpu[e + 1] - offsets_cpu[e]
        nm = (m_e + BLOCK_M - 1) // BLOCK_M
        total += nm * num_n_tiles
        tile_offsets_l.append(total)

    return torch.tensor(tile_offsets_l, dtype=torch.int32, device=device), total


# ---------------------------------------------------------------------------
# Model
# ---------------------------------------------------------------------------

OP_TYPE = "grouped_gemm_swiglu"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


class Model(nn.Module):
    """Up-projection of a top-K MoE FFN with fused SwiGLU.

    Inputs at call time:
      hidden_states:    (T_perm, H)  bf16, already permuted to expert order
      expert_offsets:   (E+1,)       int32, prefix sums of token counts per expert

    Output:
      gated_up:         (T_perm, I)  bf16
    """

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)
        # Cache compact tile-offset tables per (expert_offsets signature, tile shape).
        # In inference/benchmarking the same routing metadata is reused many times.
        self._tile_cache: dict[tuple[tuple[int, ...], int, int, int], tuple] = {}

    def _get_tile_offsets(
        self,
        expert_offsets: torch.Tensor,
        N: int,
        BLOCK_M: int,
        BLOCK_N: int,
    ) -> tuple[torch.Tensor, int]:
        key_offsets = tuple(expert_offsets.cpu().tolist())
        key = (key_offsets, N, BLOCK_M, BLOCK_N)
        cached = self._tile_cache.get(key)
        if cached is not None:
            return cached
        offsets, total = _build_expert_tile_offsets(expert_offsets, N, BLOCK_M, BLOCK_N)
        self._tile_cache[key] = (offsets, total)
        return offsets, total

    def forward(
        self,
        hidden_states: torch.Tensor,
        expert_offsets: torch.Tensor,
    ) -> torch.Tensor:
        T_perm, H = hidden_states.shape
        N = self.I
        K = H
        device = hidden_states.device
        E = self.E

        # Tuned tile configuration: 256x128 tiles with 3 stages gives the best
        # balance of occupancy, tensor-core utilization, and shared-memory usage
        # across all three target shapes on B200.
        BLOCK_M, BLOCK_N, BLOCK_K = 256, 128, 64
        num_warps, num_stages = 4, 3

        expert_tile_offsets, total_tiles = self._get_tile_offsets(
            expert_offsets, N, BLOCK_M, BLOCK_N
        )

        out = torch.empty(T_perm, N, dtype=torch.bfloat16, device=device)

        grouped_gemm_swiglu_kernel[(total_tiles,)](
            hidden_states,
            self.W_gate,
            self.W_up,
            out,
            expert_offsets,
            expert_tile_offsets,
            total_tiles,
            E,
            N,
            K,
            hidden_states.stride(0),
            hidden_states.stride(1),
            self.W_gate.stride(0),
            self.W_gate.stride(1),
            self.W_gate.stride(2),
            self.W_up.stride(0),
            self.W_up.stride(1),
            self.W_up.stride(2),
            out.stride(0),
            out.stride(1),
            BLOCK_M=BLOCK_M,
            BLOCK_N=BLOCK_N,
            BLOCK_K=BLOCK_K,
            num_warps=num_warps,
            num_stages=num_stages,
        )
        return out


# Module-level shape shims rewritten by check.py / benchmark.py per shape.
T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    """Round-robin-ish routing metadata: balanced offsets summing to T_total*K."""
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
