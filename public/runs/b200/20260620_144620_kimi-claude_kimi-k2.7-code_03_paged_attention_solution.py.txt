"""Paged-attention decode kernel implemented in Triton.

Single-query decode with GQA and a packed [K|V] paged KV cache.

Two launch strategies are used:
  * For small batch*kv_head counts we use split-KV over the sequence: each
    (batch, kv_head) pair is processed by S independent blocks, each reading a
    contiguous chunk of pages, and a tiny reduction kernel merges the partial
    softmax states.
  * For large batch*kv_head counts there is already enough parallelism, so a
    single block per (batch, kv_head) processes the whole sequence.

Inside each block the G = num_heads//num_kv_heads query heads sharing the same
KV head are processed together as rows of a [G, head_dim] tile, so each KV
token is read once per block.  Both QK^T and the softmax-weighted V reduction
use tensor-core `tl.dot`.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# Default shape knobs; overridden by check.py / benchmark.py via shapes.py.
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _decode_split_kernel(
    q_ptr,
    kv_cache_ptr,
    block_table_ptr,
    seq_lens_ptr,
    part_m_ptr,
    part_l_ptr,
    part_o_ptr,
    stride_q_b,
    stride_q_h,
    stride_q_d,
    stride_kv_page,
    stride_kv_hkv,
    stride_kv_kv,
    stride_bt_b,
    stride_bt_p,
    stride_pm_b,
    stride_pm_h,
    stride_pm_s,
    stride_pm_g,
    stride_pl_b,
    stride_pl_h,
    stride_pl_s,
    stride_pl_g,
    stride_po_b,
    stride_po_h,
    stride_po_s,
    stride_po_g,
    stride_po_d,
    num_heads: tl.constexpr,
    num_kv_heads: tl.constexpr,
    head_dim: tl.constexpr,
    page_size: tl.constexpr,
    scale: tl.constexpr,
    S: tl.constexpr,
    G: tl.constexpr,
):
    pid = tl.program_id(0)
    split = pid % S
    kv_head_idx = (pid // S) % num_kv_heads
    batch_idx = pid // (S * num_kv_heads)

    seq_len = tl.load(seq_lens_ptr + batch_idx)
    if seq_len <= 0:
        return
    num_pages = (seq_len + page_size - 1) // page_size
    pages_per_split = (num_pages + S - 1) // S
    start_page = split * pages_per_split
    end_page = tl.minimum(start_page + pages_per_split, num_pages)

    # Query tile [G, head_dim] (bf16 for tl.dot).
    g_offs = tl.arange(0, G)[:, None]
    d_offs = tl.arange(0, head_dim)[None, :]
    q_head_base = kv_head_idx * G
    q_ptrs = (
        q_ptr
        + batch_idx * stride_q_b
        + (q_head_base + g_offs) * stride_q_h
        + d_offs * stride_q_d
    )
    q_bf16 = tl.load(q_ptrs, mask=(g_offs < G) & (d_offs < head_dim), other=0.0)

    m = tl.full((G,), value=float("-inf"), dtype=tl.float32)
    l = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, head_dim), dtype=tl.float32)

    d_offs_page = tl.arange(0, head_dim)

    page_idx = start_page
    while page_idx < end_page:
        block_id = tl.load(block_table_ptr + batch_idx * stride_bt_b + page_idx * stride_bt_p)
        tokens_in_this_page = page_size
        if page_idx == num_pages - 1:
            tokens_in_this_page = seq_len - page_idx * page_size

        phys_base = block_id * page_size
        token_offs = tl.arange(0, page_size)[:, None]
        kv_ptrs = (
            kv_cache_ptr
            + (phys_base + token_offs) * stride_kv_page
            + kv_head_idx * stride_kv_hkv
            + d_offs_page[None, :] * stride_kv_kv
        )
        kv_mask = (token_offs < tokens_in_this_page) & (d_offs_page[None, :] < head_dim)
        k_page = tl.load(kv_ptrs, mask=kv_mask, other=0.0)
        v_page = tl.load(kv_ptrs + head_dim, mask=kv_mask, other=0.0)

        # Full-page QK^T: [G, page_size].
        scores = tl.dot(q_bf16, tl.trans(k_page)).to(tl.float32) * scale
        valid = tl.arange(0, page_size)[None, :] < tokens_in_this_page
        scores = tl.where(valid, scores, float("-inf"))

        m_tile = tl.max(scores, axis=1)
        m_new = tl.maximum(m, m_tile)
        alpha = tl.exp(m - m_new)
        beta = tl.exp(m_tile - m_new)
        p = tl.exp(scores - m_tile[:, None]).to(q_ptr.dtype.element_ty)
        l = l * alpha + tl.sum(p.to(tl.float32), axis=1) * beta
        acc = acc * alpha[:, None] + beta[:, None] * tl.dot(p, v_page)
        m = m_new

        page_idx += 1

    # Write partials (one softmax state per query head).
    pm_g = tl.arange(0, G)
    pm_ptrs = (
        part_m_ptr
        + batch_idx * stride_pm_b
        + kv_head_idx * stride_pm_h
        + split * stride_pm_s
        + pm_g * stride_pm_g
    )
    pl_ptrs = (
        part_l_ptr
        + batch_idx * stride_pl_b
        + kv_head_idx * stride_pl_h
        + split * stride_pl_s
        + pm_g * stride_pl_g
    )
    tl.store(pm_ptrs, m, mask=pm_g < G)
    tl.store(pl_ptrs, l, mask=pm_g < G)

    po_g = tl.arange(0, G)[:, None]
    po_d = tl.arange(0, head_dim)[None, :]
    po_ptrs = (
        part_o_ptr
        + batch_idx * stride_po_b
        + kv_head_idx * stride_po_h
        + split * stride_po_s
        + po_g * stride_po_g
        + po_d * stride_po_d
    )
    tl.store(po_ptrs, acc, mask=(po_g < G) & (po_d < head_dim))


@triton.jit
def _decode_reduce_kernel(
    part_m_ptr,
    part_l_ptr,
    part_o_ptr,
    out_ptr,
    stride_pm_b,
    stride_pm_h,
    stride_pm_s,
    stride_pm_g,
    stride_pl_b,
    stride_pl_h,
    stride_pl_s,
    stride_pl_g,
    stride_po_b,
    stride_po_h,
    stride_po_s,
    stride_po_g,
    stride_po_d,
    stride_o_b,
    stride_o_h,
    stride_o_d,
    num_kv_heads: tl.constexpr,
    head_dim: tl.constexpr,
    S: tl.constexpr,
    G: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // num_kv_heads
    kv_head_idx = pid % num_kv_heads

    d_offs = tl.arange(0, head_dim)
    g_offs = tl.arange(0, G)[:, None]
    pm_g = tl.arange(0, G)

    m = tl.full((G,), value=float("-inf"), dtype=tl.float32)
    l = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, head_dim), dtype=tl.float32)

    for split in tl.static_range(0, S):
        m_ptrs = (
            part_m_ptr
            + batch_idx * stride_pm_b
            + kv_head_idx * stride_pm_h
            + split * stride_pm_s
            + pm_g * stride_pm_g
        )
        l_ptrs = (
            part_l_ptr
            + batch_idx * stride_pl_b
            + kv_head_idx * stride_pl_h
            + split * stride_pl_s
            + pm_g * stride_pl_g
        )
        m_s = tl.load(m_ptrs, mask=pm_g < G, other=float("-inf"))
        l_s = tl.load(l_ptrs, mask=pm_g < G, other=0.0)
        o_ptrs = (
            part_o_ptr
            + batch_idx * stride_po_b
            + kv_head_idx * stride_po_h
            + split * stride_po_s
            + g_offs * stride_po_g
            + d_offs * stride_po_d
        )
        o_s = tl.load(o_ptrs, mask=(g_offs < G) & (d_offs < head_dim), other=0.0).to(tl.float32)

        m_new = tl.maximum(m, m_s)
        alpha = tl.exp(m - m_new)
        beta = tl.exp(m_s - m_new)
        l = l * alpha + l_s * beta
        acc = acc * alpha[:, None] + o_s * beta[:, None]
        m = m_new

    out = (acc / (l[:, None] + 1e-6)).to(out_ptr.dtype.element_ty)
    q_head_base = kv_head_idx * G
    out_ptrs = (
        out_ptr
        + batch_idx * stride_o_b
        + (q_head_base + g_offs) * stride_o_h
        + d_offs[None, :] * stride_o_d
    )
    tl.store(out_ptrs, out, mask=(g_offs < G) & (d_offs < head_dim))


@triton.jit
def _decode_nosplit_kernel(
    q_ptr,
    kv_cache_ptr,
    block_table_ptr,
    seq_lens_ptr,
    out_ptr,
    stride_q_b,
    stride_q_h,
    stride_q_d,
    stride_kv_page,
    stride_kv_hkv,
    stride_kv_kv,
    stride_bt_b,
    stride_bt_p,
    stride_o_b,
    stride_o_h,
    stride_o_d,
    num_heads: tl.constexpr,
    num_kv_heads: tl.constexpr,
    head_dim: tl.constexpr,
    page_size: tl.constexpr,
    scale: tl.constexpr,
    G: tl.constexpr,
):
    pid = tl.program_id(0)
    batch_idx = pid // num_kv_heads
    kv_head_idx = pid % num_kv_heads

    seq_len = tl.load(seq_lens_ptr + batch_idx)
    if seq_len <= 0:
        return
    num_pages = (seq_len + page_size - 1) // page_size

    g_offs = tl.arange(0, G)[:, None]
    d_offs = tl.arange(0, head_dim)[None, :]
    q_head_base = kv_head_idx * G
    q_ptrs = (
        q_ptr
        + batch_idx * stride_q_b
        + (q_head_base + g_offs) * stride_q_h
        + d_offs * stride_q_d
    )
    q_bf16 = tl.load(q_ptrs, mask=(g_offs < G) & (d_offs < head_dim), other=0.0)

    m = tl.full((G,), value=float("-inf"), dtype=tl.float32)
    l = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, head_dim), dtype=tl.float32)

    d_offs_page = tl.arange(0, head_dim)

    page_idx = 0
    while page_idx < num_pages:
        block_id = tl.load(block_table_ptr + batch_idx * stride_bt_b + page_idx * stride_bt_p)
        tokens_in_this_page = page_size
        if page_idx == num_pages - 1:
            tokens_in_this_page = seq_len - page_idx * page_size

        phys_base = block_id * page_size
        token_offs = tl.arange(0, page_size)[:, None]
        kv_ptrs = (
            kv_cache_ptr
            + (phys_base + token_offs) * stride_kv_page
            + kv_head_idx * stride_kv_hkv
            + d_offs_page[None, :] * stride_kv_kv
        )
        kv_mask = (token_offs < tokens_in_this_page) & (d_offs_page[None, :] < head_dim)
        k_page = tl.load(kv_ptrs, mask=kv_mask, other=0.0)
        v_page = tl.load(kv_ptrs + head_dim, mask=kv_mask, other=0.0)

        scores = tl.dot(q_bf16, tl.trans(k_page)).to(tl.float32) * scale
        valid = tl.arange(0, page_size)[None, :] < tokens_in_this_page
        scores = tl.where(valid, scores, float("-inf"))

        m_tile = tl.max(scores, axis=1)
        m_new = tl.maximum(m, m_tile)
        alpha = tl.exp(m - m_new)
        beta = tl.exp(m_tile - m_new)
        p = tl.exp(scores - m_tile[:, None]).to(q_ptr.dtype.element_ty)
        l = l * alpha + tl.sum(p.to(tl.float32), axis=1) * beta
        acc = acc * alpha[:, None] + beta[:, None] * tl.dot(p, v_page)
        m = m_new
        page_idx += 1

    out = (acc / (l[:, None] + 1e-6)).to(q_ptr.dtype.element_ty)
    out_ptrs = (
        out_ptr
        + batch_idx * stride_o_b
        + (q_head_base + g_offs) * stride_o_h
        + d_offs * stride_o_d
    )
    tl.store(out_ptrs, out, mask=(g_offs < G) & (d_offs < head_dim))


class Model(nn.Module):
    """Single-query paged attention decode.  See reference.py for the interface."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        # Decide launch strategy based on inherent parallelism.
        # Non-split is faster when there are already many (batch, kv_head) blocks.
        self.use_split = batch * num_kv_heads < 256

        if self.use_split:
            num_pages = (seq_len + page_size - 1) // page_size
            # Small head_dim has less work per block; use more splits to create
            # enough parallelism while keeping the tiny tile well-occupied.
            if head_dim <= 64:
                max_S = 32
                base_nw = 4
            else:
                max_S = 16
                base_nw = 2
            self.S = min(num_pages, max_S)
            self.num_warps_split = base_nw
            self.register_buffer(
                "_part_m",
                torch.empty(batch, num_kv_heads, max_S, self.group_size, dtype=torch.float32),
                persistent=False,
            )
            self.register_buffer(
                "_part_l",
                torch.empty(batch, num_kv_heads, max_S, self.group_size, dtype=torch.float32),
                persistent=False,
            )
            self.register_buffer(
                "_part_o",
                torch.empty(batch, num_kv_heads, max_S, self.group_size, head_dim, dtype=torch.float32),
                persistent=False,
            )

        self.register_buffer("_out", torch.empty(batch, num_heads, head_dim, dtype=torch.bfloat16), persistent=False)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size

        out = self._out

        if not self.use_split:
            grid = (B * Hkv,)
            _decode_nosplit_kernel[grid](
                query,
                kv_cache,
                block_table,
                seq_lens,
                out,
                query.stride(0),
                query.stride(1),
                query.stride(2),
                kv_cache.stride(1),
                kv_cache.stride(2),
                kv_cache.stride(3),
                block_table.stride(0),
                block_table.stride(1),
                out.stride(0),
                out.stride(1),
                out.stride(2),
                num_heads=H,
                num_kv_heads=Hkv,
                head_dim=D,
                page_size=P,
                scale=float(self.scale),
                G=G,
                num_warps=max(G, 4),
                num_stages=1,
            )
            return out

        S = self.S
        part_m = self._part_m
        part_l = self._part_l
        part_o = self._part_o

        grid_split = (B * Hkv * S,)
        grid_reduce = (B * Hkv,)

        _decode_split_kernel[grid_split](
            query,
            kv_cache,
            block_table,
            seq_lens,
            part_m,
            part_l,
            part_o,
            query.stride(0),
            query.stride(1),
            query.stride(2),
            kv_cache.stride(1),
            kv_cache.stride(2),
            kv_cache.stride(3),
            block_table.stride(0),
            block_table.stride(1),
            part_m.stride(0),
            part_m.stride(1),
            part_m.stride(2),
            part_m.stride(3),
            part_l.stride(0),
            part_l.stride(1),
            part_l.stride(2),
            part_l.stride(3),
            part_o.stride(0),
            part_o.stride(1),
            part_o.stride(2),
            part_o.stride(3),
            part_o.stride(4),
            num_heads=H,
            num_kv_heads=Hkv,
            head_dim=D,
            page_size=P,
            scale=float(self.scale),
            S=S,
            G=G,
            num_warps=self.num_warps_split,
            num_stages=2,
        )

        _decode_reduce_kernel[grid_reduce](
            part_m,
            part_l,
            part_o,
            out,
            part_m.stride(0),
            part_m.stride(1),
            part_m.stride(2),
            part_m.stride(3),
            part_l.stride(0),
            part_l.stride(1),
            part_l.stride(2),
            part_l.stride(3),
            part_o.stride(0),
            part_o.stride(1),
            part_o.stride(2),
            part_o.stride(3),
            part_o.stride(4),
            out.stride(0),
            out.stride(1),
            out.stride(2),
            num_kv_heads=Hkv,
            head_dim=D,
            S=S,
            G=G,
            num_warps=max(G, 4),
            num_stages=1,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
