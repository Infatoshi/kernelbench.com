"""Custom grouped GEMM + fused SwiGLU kernel (Triton) for top-K MoE up-projection.

For each expert e, the M dimension (n_e token rows) is partitioned across CTAs so
the GPU stays saturated; within a CTA the (m, n) output tile is one program that
reduces over K, reusing the input tile x exactly once per K step.

Per expert e (one CTA per (m, n) tile):
    gate = x_e @ W_gate[e]   # (n_e, I)
    up   = x_e @ W_up[e]     # (n_e, I)
    out  = silu(gate) * up   # (n_e, I)

x_e occupies rows [offsets[e]:offsets[e+1]] of `hidden_states`. Weights
W_gate[e], W_up[e] are (H, I) slices of the (E, H, I) tensors.

No vendor GEMM dispatch is used: the matmuls are manual tl.dot with blocking and
the SwiGLU activation is fused into the epilogue. Block sizes are chosen per shape
to balance CTA count and MMA efficiency on B200 (SM100).
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "grouped_gemm_swiglu"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _grouped_swiglu_kernel(
    x_ptr, gate_w_ptr, up_w_ptr, out_ptr,
    offsets_ptr,
    stride_xrow, stride_xh,        # hidden_states: (T_perm, H)
    stride_w_h, stride_w_i,        # per-expert weight: (H, I)
    stride_out,                    # out: (T_perm, I)
    H: tl.constexpr, I: tl.constexpr,
    BLOCK_M: tl.constexpr = 256, BLOCK_N: tl.constexpr = 128, BLOCK_K: tl.constexpr = 64,
):
    # 2D grid over experts and output tiles: axis0 = expert, axis1 = M tile,
    # axis2 = N tile. Each CTA owns one (m, n) output tile and reduces over K,
    # so the input tile x is loaded exactly once per K step (reused across the
    # K reduction), keeping memory traffic minimal.
    e = tl.program_id(0)
    m_tile = tl.program_id(1)
    n_tile = tl.program_id(2)

    start = tl.load(offsets_ptr + e)
    end = tl.load(offsets_ptr + e + 1)
    n = end - start
    if n <= 0:
        return

    offs_m = start + m_tile * BLOCK_M + tl.arange(0, BLOCK_M)
    mask_m = offs_m < end
    offs_n = n_tile * BLOCK_N + tl.arange(0, BLOCK_N)
    n_mask = offs_n < I
    offs_k = tl.arange(0, BLOCK_K)

    gate_acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    up_acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    x_ptrs = x_ptr + offs_m[:, None] * stride_xrow + offs_k[None, :] * stride_xh
    gw_ptrs = gate_w_ptr + e * (H * I) + offs_k[:, None] * stride_w_h + offs_n[None, :] * stride_w_i
    uw_ptrs = up_w_ptr + e * (H * I) + offs_k[:, None] * stride_w_h + offs_n[None, :] * stride_w_i
    for k in range(0, H, BLOCK_K):
        k_mask = offs_k < (H - k)
        x = tl.load(x_ptrs, mask=mask_m[:, None] & k_mask[None, :], other=0.0)
        gw = tl.load(gw_ptrs, mask=k_mask[:, None] & n_mask[None, :], other=0.0)
        uw = tl.load(uw_ptrs, mask=k_mask[:, None] & n_mask[None, :], other=0.0)
        gate_acc = tl.dot(x, gw, acc=gate_acc)
        up_acc = tl.dot(x, uw, acc=up_acc)
        x_ptrs += BLOCK_K * stride_xh
        gw_ptrs += BLOCK_K * stride_w_h
        uw_ptrs += BLOCK_K * stride_w_h
    # Fused SwiGLU epilogue (numerically stable sigmoid).
    sig = tl.where(gate_acc >= 0,
                   1.0 / (1.0 + tl.exp(-gate_acc)),
                   tl.exp(gate_acc) / (1.0 + tl.exp(gate_acc)))
    out = gate_acc * sig * up_acc
    out_ptrs = out_ptr + offs_m[:, None] * stride_out + offs_n[None, :]
    tl.store(out_ptrs, out.to(tl.bfloat16), mask=mask_m[:, None] & n_mask[None, :])


class Model(nn.Module):
    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(self, hidden_states, expert_offsets):
        T_perm, H = hidden_states.shape
        out = torch.empty(T_perm, self.I, dtype=torch.bfloat16, device=hidden_states.device)

        counts = expert_offsets[1:] - expert_offsets[:-1]
        max_m = int(counts.max().item())

        # Heuristic tile selection: large tiles for big FFNs (saturate the GPU
        # with fewer, fatter CTAs), smaller tiles for small shapes (more CTAs).
        expert_flops = max_m * H * self.I
        if expert_flops >= 2_000_000_000:
            BLOCK_M, BLOCK_N = 256, 128
        else:
            BLOCK_M, BLOCK_N = 128, 128

        def grid(meta):
            return (self.E,
                    triton.cdiv(max_m, meta["BLOCK_M"]),
                    triton.cdiv(self.I, meta["BLOCK_N"]))

        _grouped_swiglu_kernel[grid](
            hidden_states, self.W_gate, self.W_up, out,
            expert_offsets,
            hidden_states.stride(0), hidden_states.stride(1),
            self.W_gate.stride(1), self.W_gate.stride(2),
            out.stride(0),
            H=H, I=self.I,
            BLOCK_M=BLOCK_M, BLOCK_N=BLOCK_N, BLOCK_K=64,
            num_warps=8, num_stages=3,
        )
        return out


# Module-level shape shims rewritten by check.py / benchmark.py per shape.
T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
