"""Custom CUDA top-k: radix select (large n / large k), streaming select, warp argmax (k=1)."""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CPP_SRC = r"""
#include <torch/extension.h>

void launch_topk(
    const torch::Tensor& input,
    torch::Tensor& out_values,
    torch::Tensor& out_indices,
    torch::Tensor& temp_buf,
    int k);
"""

CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>
#include <cfloat>
#include <cmath>

#ifndef CUDART_INF_F
#define CUDART_INF_F INFINITY
#endif

__device__ __forceinline__ uint32_t float_to_sortable(uint32_t f) {
    uint32_t mask = (-(f >> 31)) | 0x80000000u;
    return f ^ mask;
}

// ---------- k=1 argmax ----------
__device__ void warp_reduce_argmax(float val, int idx, float& out_val, int& out_idx) {
    for (int offset = 16; offset > 0; offset >>= 1) {
        float ov = __shfl_down_sync(0xffffffff, val, offset);
        int oi = __shfl_down_sync(0xffffffff, idx, offset);
        if (ov > val || (ov == val && oi < idx)) {
            val = ov;
            idx = oi;
        }
    }
    out_val = val;
    out_idx = idx;
}

__global__ void topk1_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int n, int in_stride
) {
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int bsize = blockDim.x;
    const float* row_in = input + (int64_t)row * in_stride;

    float local_max = -CUDART_INF_F;
    int local_idx = 0;

    const int n4 = n & ~3;
    for (int i = tid * 4; i < n4; i += bsize * 4) {
        float4 v4 = *reinterpret_cast<const float4*>(row_in + i);
        float vs[4] = {v4.x, v4.y, v4.z, v4.w};
        for (int j = 0; j < 4; j++) {
            int idx = i + j;
            if (vs[j] > local_max || (vs[j] == local_max && idx < local_idx)) {
                local_max = vs[j];
                local_idx = idx;
            }
        }
    }
    for (int i = n4 + tid; i < n; i += bsize) {
        float v = row_in[i];
        if (v > local_max || (v == local_max && i < local_idx)) {
            local_max = v;
            local_idx = i;
        }
    }

    float warp_max;
    int warp_idx;
    warp_reduce_argmax(local_max, local_idx, warp_max, warp_idx);

    __shared__ float smax[32];
    __shared__ int sidx[32];
    const int lane = tid & 31;
    const int warp_id = tid >> 5;
    if (lane == 0) {
        smax[warp_id] = warp_max;
        sidx[warp_id] = warp_idx;
    }
    __syncthreads();

    if (warp_id == 0) {
        float val = (lane < (bsize + 31) / 32) ? smax[lane] : -CUDART_INF_F;
        int idx = (lane < (bsize + 31) / 32) ? sidx[lane] : 0;
        warp_reduce_argmax(val, idx, warp_max, warp_idx);
        if (lane == 0) {
            out_values[row] = warp_max;
            out_indices[row] = warp_idx;
        }
    }
}

// ---------- streaming top-k ----------
__device__ void insert_topk(float val, int idx, float* vals, int* idxs, int k) {
    if (val <= vals[k - 1]) return;
    int pos = k - 1;
    while (pos > 0 && val > vals[pos - 1]) {
        vals[pos] = vals[pos - 1];
        idxs[pos] = idxs[pos - 1];
        pos--;
    }
    vals[pos] = val;
    idxs[pos] = idx;
}

__device__ void merge_topk(
    const float* a, const int* ai,
    const float* b, const int* bi,
    float* c, int* ci, int k
) {
    int ia = 0, ib = 0, ic = 0;
    while (ic < k) {
        if (ia < k && (ib >= k || a[ia] > b[ib] || (a[ia] == b[ib] && ai[ia] < ai[ib]))) {
            c[ic] = a[ia]; ci[ic] = ai[ia]; ia++;
        } else if (ib < k) {
            c[ic] = b[ib]; ci[ic] = bi[ib]; ib++;
        } else {
            break;
        }
        ic++;
    }
}

template<int MAXK>
__global__ void topk_stream_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int n, int k, int in_stride
) {
    extern __shared__ char smem_raw[];
    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int bsize = blockDim.x;
    const float* row_in = input + (int64_t)row * in_stride;

    float* tvals = reinterpret_cast<float*>(smem_raw);
    int* tidxs = reinterpret_cast<int*>(tvals + bsize * MAXK);

    float* my_vals = tvals + tid * MAXK;
    int* my_idxs = tidxs + tid * MAXK;

    for (int j = 0; j < k; j++) {
        my_vals[j] = -CUDART_INF_F;
        my_idxs[j] = 0;
    }

    const int n4 = n & ~3;
    for (int base = tid * 4; base < n4; base += bsize * 4) {
        float4 v4 = *reinterpret_cast<const float4*>(row_in + base);
        insert_topk(v4.x, base + 0, my_vals, my_idxs, k);
        insert_topk(v4.y, base + 1, my_vals, my_idxs, k);
        insert_topk(v4.z, base + 2, my_vals, my_idxs, k);
        insert_topk(v4.w, base + 3, my_vals, my_idxs, k);
    }
    for (int i = n4 + tid; i < n; i += bsize) {
        insert_topk(row_in[i], i, my_vals, my_idxs, k);
    }
    __syncthreads();

    for (int stride = bsize >> 1; stride > 0; stride >>= 1) {
        if (tid < stride) {
            float rv[MAXK];
            int ri[MAXK];
            merge_topk(
                tvals + tid * MAXK, tidxs + tid * MAXK,
                tvals + (tid + stride) * MAXK, tidxs + (tid + stride) * MAXK,
                rv, ri, k
            );
            for (int j = 0; j < k; j++) {
                tvals[tid * MAXK + j] = rv[j];
                tidxs[tid * MAXK + j] = ri[j];
            }
        }
        __syncthreads();
    }

    if (tid == 0) {
        float* out_v = out_values + (int64_t)row * k;
        int64_t* out_i = out_indices + (int64_t)row * k;
        for (int j = 0; j < k; j++) {
            out_v[j] = tvals[j];
            out_i[j] = tidxs[j];
        }
    }
}

// ---------- radix select ----------
__device__ int find_split_bin(const int* hist, int k_rem) {
    int cum = 0;
    for (int b = 255; b >= 0; b--) {
        cum += hist[b];
        if (cum >= k_rem) return b;
    }
    return 0;
}

__device__ void sort_pairs_desc(float* vals, int* idxs, int count) {
    for (int i = 1; i < count; i++) {
        float v = vals[i];
        int ix = idxs[i];
        int j = i - 1;
        while (j >= 0 && (vals[j] < v || (vals[j] == v && idxs[j] > ix))) {
            vals[j + 1] = vals[j];
            idxs[j + 1] = idxs[j];
            j--;
        }
        vals[j + 1] = v;
        idxs[j + 1] = ix;
    }
}

__global__ void topk_radix_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int32_t* __restrict__ temp_buf,
    int n, int k, int in_stride
) {
    extern __shared__ int smem[];
    int* hist = smem;

    const int row = blockIdx.x;
    const int tid = threadIdx.x;
    const int bsize = blockDim.x;
    const float* row_in = input + (int64_t)row * in_stride;
    int32_t* cand = temp_buf + (int64_t)row * n;
    int32_t* cand2 = cand + n;

    __shared__ int cand_count;
    __shared__ int out_count;
    __shared__ int remaining_k;
    __shared__ int split_bin;
    __shared__ int count_gt;
    __shared__ float partial_vals[64];
    __shared__ int partial_idxs[64];

    if (tid == 0) {
        cand_count = n;
        out_count = 0;
    }
    __syncthreads();

    bool use_all = true;
    int cur_count = n;

    for (int pass = 0; pass < 4; pass++) {
        if (tid == 0) remaining_k = k - out_count;
        __syncthreads();
        if (remaining_k <= 0) break;

        for (int i = tid; i < 256; i += bsize) hist[i] = 0;
        __syncthreads();

        const int shift = 24 - pass * 8;

        if (use_all) {
            const int n4 = n & ~3;
            for (int base = tid * 4; base < n4; base += bsize * 4) {
                float4 v4 = *reinterpret_cast<const float4*>(row_in + base);
                uint32_t ks[4] = {
                    float_to_sortable(__float_as_uint(v4.x)),
                    float_to_sortable(__float_as_uint(v4.y)),
                    float_to_sortable(__float_as_uint(v4.z)),
                    float_to_sortable(__float_as_uint(v4.w)),
                };
                for (int j = 0; j < 4; j++) {
                    int digit = (ks[j] >> shift) & 0xFF;
                    atomicAdd(&hist[digit], 1);
                }
            }
            for (int i = n4 + tid; i < n; i += bsize) {
                uint32_t key = float_to_sortable(__float_as_uint(row_in[i]));
                int digit = (key >> shift) & 0xFF;
                atomicAdd(&hist[digit], 1);
            }
        } else {
            for (int ci = tid; ci < cur_count; ci += bsize) {
                int i = cand[ci];
                uint32_t key = float_to_sortable(__float_as_uint(row_in[i]));
                int digit = (key >> shift) & 0xFF;
                atomicAdd(&hist[digit], 1);
            }
        }
        __syncthreads();

        if (tid == 0) {
            split_bin = find_split_bin(hist, remaining_k);
            count_gt = 0;
            for (int b = 255; b > split_bin; b--) count_gt += hist[b];
        }
        __syncthreads();

        __shared__ int new_count;
        __shared__ int gt_written;
        if (tid == 0) {
            new_count = 0;
            gt_written = 0;
        }
        __syncthreads();

        if (use_all) {
            for (int i = tid; i < n; i += bsize) {
                float val = row_in[i];
                uint32_t key = float_to_sortable(__float_as_uint(val));
                int digit = (key >> shift) & 0xFF;
                if (digit > split_bin) {
                    int pos = atomicAdd(&gt_written, 1);
                    if (pos < 64) {
                        partial_vals[out_count + pos] = val;
                        partial_idxs[out_count + pos] = i;
                    }
                } else if (digit == split_bin) {
                    int pos = atomicAdd(&new_count, 1);
                    cand2[pos] = i;
                }
            }
        } else {
            for (int ci = tid; ci < cur_count; ci += bsize) {
                int i = cand[ci];
                float val = row_in[i];
                uint32_t key = float_to_sortable(__float_as_uint(val));
                int digit = (key >> shift) & 0xFF;
                if (digit > split_bin) {
                    int pos = atomicAdd(&gt_written, 1);
                    if (pos < 64) {
                        partial_vals[out_count + pos] = val;
                        partial_idxs[out_count + pos] = i;
                    }
                } else if (digit == split_bin) {
                    int pos = atomicAdd(&new_count, 1);
                    cand2[pos] = i;
                }
            }
        }
        __syncthreads();

        if (tid == 0) {
            out_count += gt_written;
            remaining_k -= gt_written;
            cand_count = new_count;
        }
        __syncthreads();

        cur_count = cand_count;
        use_all = false;

        for (int ci = tid; ci < cur_count; ci += bsize) {
            cand[ci] = cand2[ci];
        }
        __syncthreads();

        if (cur_count <= remaining_k) break;
    }

    __shared__ float final_vals[64];
    __shared__ int final_idxs[64];
    __shared__ int final_count;

    if (tid == 0) {
        final_count = out_count;
        for (int i = 0; i < out_count; i++) {
            final_vals[i] = partial_vals[i];
            final_idxs[i] = partial_idxs[i];
        }
    }
    __syncthreads();

    for (int ci = tid; ci < cur_count; ci += bsize) {
        int i = cand[ci];
        int pos = atomicAdd(&final_count, 1);
        if (pos < 64) {
            final_vals[pos] = row_in[i];
            final_idxs[pos] = i;
        }
    }
    __syncthreads();

    if (tid == 0) {
        int fc = final_count;
        if (fc > 64) fc = 64;
        sort_pairs_desc(final_vals, final_idxs, fc);
        float* out_v = out_values + (int64_t)row * k;
        int64_t* out_i = out_indices + (int64_t)row * k;
        for (int i = 0; i < k; i++) {
            out_v[i] = final_vals[i];
            out_i[i] = final_idxs[i];
        }
    }
}

template<int MAXK, int THREADS>
void launch_stream(
    const float* in_ptr, float* val_ptr, int64_t* idx_ptr,
    int batch, int n, int k, int in_stride
) {
    const size_t smem = THREADS * MAXK * (sizeof(float) + sizeof(int));
    topk_stream_kernel<MAXK><<<batch, THREADS, smem>>>(
        in_ptr, val_ptr, idx_ptr, n, k, in_stride);
}

void launch_topk(
    const torch::Tensor& input,
    torch::Tensor& out_values,
    torch::Tensor& out_indices,
    torch::Tensor& temp_buf,
    int k
) {
    const int batch = input.size(0);
    const int n = input.size(1);
    const int in_stride = input.stride(0);
    const float* in_ptr = input.data_ptr<float>();
    float* val_ptr = out_values.data_ptr<float>();
    int64_t* idx_ptr = out_indices.data_ptr<int64_t>();

    if (k == 1) {
        topk1_kernel<<<batch, 256>>>(in_ptr, val_ptr, idx_ptr, n, in_stride);
    } else if (n >= 65536 || k > 32) {
        const size_t smem = 256 * sizeof(int);
        int32_t* tmp = temp_buf.data_ptr<int32_t>();
        topk_radix_kernel<<<batch, 256, smem>>>(
            in_ptr, val_ptr, idx_ptr, tmp, n, k, in_stride);
    } else if (k <= 8) {
        launch_stream<8, 256>(in_ptr, val_ptr, idx_ptr, batch, n, k, in_stride);
    } else if (k <= 16) {
        launch_stream<16, 256>(in_ptr, val_ptr, idx_ptr, batch, n, k, in_stride);
    } else {
        launch_stream<32, 128>(in_ptr, val_ptr, idx_ptr, batch, n, k, in_stride);
    }
}
"""

_cpp_ext = load_inline(
    name="topk_cuda_ext_v2",
    cpp_sources=CPP_SRC,
    cuda_sources=[CUDA_SRC],
    functions=["launch_topk"],
    with_cuda=True,
    extra_cuda_cflags=["-O3", "--use_fast_math", "-gencode=arch=compute_100,code=sm_100"],
)

OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

batch = 64
n = 8192
k = 8


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))
        max_n = max(s["n"] for s in __import__("shapes").SHAPES)
        self.register_buffer(
            "_temp",
            torch.empty(batch, 2 * max_n, dtype=torch.int32, device="cuda"),
            persistent=False,
        )

    def forward(self, x: torch.Tensor):
        b, n_in = x.shape
        k_out = self.k
        values = torch.empty(b, k_out, dtype=torch.float32, device=x.device)
        indices = torch.empty(b, k_out, dtype=torch.int64, device=x.device)
        temp = self._temp[:b, : 2 * n_in]
        _cpp_ext.launch_topk(x.contiguous(), values, indices, temp, k_out)
        return values, indices


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
