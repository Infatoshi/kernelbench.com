"""Fast DeepSeek NSA-inspired sparse attention (CUDA).

Bench-faithful semantics matching reference.nsa_attend:
  block_size=64, top_n_blocks=8, sliding_window=64, causal bf16.

Design:
  * Full-block importance = scale * q · mean(K_block) (exact identity).
  * 4 warps/CTA, one query per warp; Q + online-softmax acc in registers.
  * Warp-cooperative dots for coalesced K/V/mean loads.
  * Top-n ∪ sliding window → merged intervals → online sparse softmax.
  * Real CUDA via load_inline (no Triton / flash-attn / SDPA).
"""
from __future__ import annotations

import os

import torch
import torch.nn as nn

BLOCK_SIZE = 64
TOP_N_BLOCKS = 8
SLIDING_WINDOW = 64

_ext = None

_CUDA_SRC = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cuda_bf16.h>
#include <cmath>
#include <vector>

constexpr int BS = 64;
constexpr int TOPN = 8;
constexpr int WIN = 64;
constexpr int MAX_BLOCKS = 256;
constexpr int WARP = 32;
constexpr int WARPS_PER_CTA = 4;

__device__ __forceinline__ float warp_sum(float v) {
#pragma unroll
  for (int o = 16; o > 0; o >>= 1)
    v += __shfl_xor_sync(0xffffffff, v, o);
  return v;
}

__device__ __forceinline__ float ld_bf(const __nv_bfloat16* p) {
  return __bfloat162float(__ldg(p));
}

__device__ __forceinline__ float ld_f(const float* p) {
  return __ldg(p);
}

// ---------------------------------------------------------------------------
// Block mean of K — one warp per (bh, block)
// ---------------------------------------------------------------------------
__global__ void block_mean_k_kernel(
    const __nv_bfloat16* __restrict__ k,
    float* __restrict__ means,
    int S, int D, int n_blocks, int BH
) {
  int bi = (int)blockIdx.x;
  int bh = (int)blockIdx.y;
  if (bi >= n_blocks || bh >= BH) return;
  int s0 = bi * BS;
  int s1 = min(s0 + BS, S);
  int n = s1 - s0;
  float inv_n = 1.f / float(n);
  const __nv_bfloat16* kb = k + (size_t)bh * S * D;
  int lane = threadIdx.x;
  for (int d = lane; d < D; d += WARP) {
    float acc = 0.f;
#pragma unroll 8
    for (int j = s0; j < s1; ++j)
      acc += ld_bf(kb + (size_t)j * D + d);
    means[(size_t)bh * n_blocks * D + (size_t)bi * D + d] = acc * inv_n;
  }
}

// ---------------------------------------------------------------------------
// Warp-cooperative NSA
// blockDim = (32, WARPS_PER_CTA)
// ---------------------------------------------------------------------------
template<int D>
__global__ void nsa_multiwarp_kernel(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const __nv_bfloat16* __restrict__ v,
    const float* __restrict__ block_means,
    __nv_bfloat16* __restrict__ out,
    int S, int n_blocks, float scale, int n_queries
) {
  static_assert(D % WARP == 0, "D multiple of 32");
  constexpr int VPT = D / WARP;  // 2 or 4

  const int warp_id = threadIdx.y;
  const int lane = threadIdx.x;
  const int qid = (int)blockIdx.x * WARPS_PER_CTA + warp_id;
  if (qid >= n_queries) return;

  const int t = qid % S;
  const int bh = qid / S;

  const __nv_bfloat16* q_row = q + (size_t)bh * S * D + (size_t)t * D;
  const __nv_bfloat16* k_base = k + (size_t)bh * S * D;
  const __nv_bfloat16* v_base = v + (size_t)bh * S * D;
  const float* means_base = block_means + (size_t)bh * n_blocks * D;
  __nv_bfloat16* o_row = out + (size_t)bh * S * D + (size_t)t * D;

  // Per-warp shared: imps[n_blocks] + top/intervals workspace
  extern __shared__ float smem[];
  const int slab = n_blocks + TOPN + 48;
  float* imps = smem + warp_id * slab;
  int* top_bi = reinterpret_cast<int*>(imps + n_blocks);
  int* iv_lo = top_bi + TOPN;
  int* iv_hi = iv_lo + 16;
  int* n_iv_ptr = iv_hi + 16;

  // ---- Q registers ----
  float q_reg[VPT];
#pragma unroll
  for (int i = 0; i < VPT; ++i)
    q_reg[i] = ld_bf(q_row + lane + i * WARP);

  const int t_block = t / BS;
  const int n_causal = t_block + 1;

  // ---- full-block importances (cooperative warp dot) ----
  for (int bi = 0; bi < t_block; ++bi) {
    const float* mk = means_base + (size_t)bi * D;
    float partial = 0.f;
#pragma unroll
    for (int i = 0; i < VPT; ++i)
      partial += q_reg[i] * ld_f(mk + lane + i * WARP);
    float dot = warp_sum(partial);
    if (lane == 0) imps[bi] = scale * dot;
  }

  // Partial block t_block
  if (t_block < n_blocks) {
    int s0 = t_block * BS;
    int n_valid = t - s0 + 1;
    float local = 0.f;
    for (int j = s0; j <= t; ++j) {
      const __nv_bfloat16* kj = k_base + (size_t)j * D;
      float partial = 0.f;
#pragma unroll
      for (int i = 0; i < VPT; ++i)
        partial += q_reg[i] * ld_bf(kj + lane + i * WARP);
      local += warp_sum(partial);
    }
    if (lane == 0)
      imps[t_block] = scale * (local / float(n_valid));
  }
  for (int bi = t_block + 1 + lane; bi < n_blocks; bi += WARP)
    imps[bi] = -1e9f;
  __syncwarp();

  // ---- top-N + merge intervals (lane 0) ----
  if (lane == 0) {
    float top_imp[TOPN];
#pragma unroll
    for (int i = 0; i < TOPN; ++i) {
      top_bi[i] = -1;
      top_imp[i] = -1e30f;
    }
    for (int bi = 0; bi < n_causal; ++bi) {
      float imp = imps[bi];
      int place = -1;
#pragma unroll
      for (int i = 0; i < TOPN; ++i) {
        if (top_bi[i] < 0) { place = i; break; }
        if (imp > top_imp[i] || (imp == top_imp[i] && bi > top_bi[i])) {
          place = i; break;
        }
      }
      if (place < 0) continue;
      for (int j = TOPN - 1; j > place; --j) {
        top_imp[j] = top_imp[j - 1];
        top_bi[j] = top_bi[j - 1];
      }
      top_imp[place] = imp;
      top_bi[place] = bi;
    }

    int niv = 0;
    int tmp_lo[16], tmp_hi[16];
#pragma unroll
    for (int i = 0; i < TOPN; ++i) {
      int bi = top_bi[i];
      if (bi < 0) continue;
      int s0 = bi * BS;
      if (s0 > t) continue;
      int s1 = min(s0 + BS, t + 1);
      if (s0 < s1) {
        tmp_lo[niv] = s0;
        tmp_hi[niv] = s1;
        ++niv;
      }
    }
    {
      int w0 = max(0, t + 1 - WIN);
      int w1 = t + 1;
      if (w0 < w1) {
        tmp_lo[niv] = w0;
        tmp_hi[niv] = w1;
        ++niv;
      }
    }
    if (niv == 0) {
      tmp_lo[0] = t;
      tmp_hi[0] = t + 1;
      niv = 1;
    }
    // sort by lo
    for (int i = 1; i < niv; ++i) {
      int lo = tmp_lo[i], hi = tmp_hi[i];
      int j = i - 1;
      while (j >= 0 && tmp_lo[j] > lo) {
        tmp_lo[j + 1] = tmp_lo[j];
        tmp_hi[j + 1] = tmp_hi[j];
        --j;
      }
      tmp_lo[j + 1] = lo;
      tmp_hi[j + 1] = hi;
    }
    // merge overlapping/adjacent
    int m = 0;
    for (int i = 0; i < niv; ++i) {
      if (m == 0 || tmp_lo[i] > tmp_hi[m - 1]) {
        tmp_lo[m] = tmp_lo[i];
        tmp_hi[m] = tmp_hi[i];
        ++m;
      } else {
        tmp_hi[m - 1] = max(tmp_hi[m - 1], tmp_hi[i]);
      }
    }
    for (int i = 0; i < m; ++i) {
      iv_lo[i] = tmp_lo[i];
      iv_hi[i] = tmp_hi[i];
    }
    *n_iv_ptr = m;
  }
  __syncwarp();
  const int n_iv = *n_iv_ptr;

  // ---- online sparse attention ----
  float acc[VPT];
#pragma unroll
  for (int i = 0; i < VPT; ++i) acc[i] = 0.f;
  float m_i = -1e30f;
  float l_i = 0.f;

  for (int iv = 0; iv < n_iv; ++iv) {
    const int lo = iv_lo[iv];
    const int hi = iv_hi[iv];
    for (int j = lo; j < hi; ++j) {
      // score
      const __nv_bfloat16* kj = k_base + (size_t)j * D;
      float partial = 0.f;
#pragma unroll
      for (int i = 0; i < VPT; ++i)
        partial += q_reg[i] * ld_bf(kj + lane + i * WARP);
      const float s = scale * warp_sum(partial);

      const float m_new = fmaxf(m_i, s);
      const float e_old = __expf(m_i - m_new);
      const float e_s = __expf(s - m_new);
#pragma unroll
      for (int i = 0; i < VPT; ++i)
        acc[i] *= e_old;

      // weighted V
      const __nv_bfloat16* vj = v_base + (size_t)j * D;
#pragma unroll
      for (int i = 0; i < VPT; ++i)
        acc[i] += e_s * ld_bf(vj + lane + i * WARP);

      l_i = l_i * e_old + e_s;
      m_i = m_new;
    }
  }

  const float inv_l = 1.f / l_i;
#pragma unroll
  for (int i = 0; i < VPT; ++i)
    o_row[lane + i * WARP] = __float2bfloat16_rn(acc[i] * inv_l);
}

torch::Tensor nsa_forward(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor v
) {
  TORCH_CHECK(q.is_cuda() && k.is_cuda() && v.is_cuda(), "CUDA tensors required");
  TORCH_CHECK(q.scalar_type() == at::kBFloat16, "bf16 required");
  TORCH_CHECK(q.dim() == 4, "q must be (B,H,S,D)");

  q = q.contiguous();
  k = k.contiguous();
  v = v.contiguous();

  const int B = (int)q.size(0);
  const int H = (int)q.size(1);
  const int S = (int)q.size(2);
  const int D = (int)q.size(3);
  TORCH_CHECK(D == 64 || D == 128, "D must be 64 or 128");
  TORCH_CHECK(k.sizes() == q.sizes() && v.sizes() == q.sizes(), "shape mismatch");

  const int n_blocks = (S + BS - 1) / BS;
  TORCH_CHECK(n_blocks <= MAX_BLOCKS, "S too large");

  auto means = torch::empty({B, H, n_blocks, D}, q.options().dtype(at::kFloat));
  auto out = torch::empty_like(q);
  auto stream = at::cuda::getCurrentCUDAStream();

  {
    dim3 grid(n_blocks, B * H);
    block_mean_k_kernel<<<grid, WARP, 0, stream>>>(
        reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>()),
        means.data_ptr<float>(),
        S, D, n_blocks, B * H);
  }

  float scale = 1.f / sqrtf(float(D));
  int n_queries = B * H * S;
  int grid = (n_queries + WARPS_PER_CTA - 1) / WARPS_PER_CTA;
  dim3 block(WARP, WARPS_PER_CTA);
  size_t smem = sizeof(float) * (size_t)(n_blocks + TOPN + 48) * WARPS_PER_CTA;

  auto q_ptr = reinterpret_cast<const __nv_bfloat16*>(q.data_ptr<at::BFloat16>());
  auto k_ptr = reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>());
  auto v_ptr = reinterpret_cast<const __nv_bfloat16*>(v.data_ptr<at::BFloat16>());
  auto o_ptr = reinterpret_cast<__nv_bfloat16*>(out.data_ptr<at::BFloat16>());
  auto m_ptr = means.data_ptr<float>();

  if (D == 64) {
    nsa_multiwarp_kernel<64><<<grid, block, smem, stream>>>(
        q_ptr, k_ptr, v_ptr, m_ptr, o_ptr, S, n_blocks, scale, n_queries);
  } else {
    nsa_multiwarp_kernel<128><<<grid, block, smem, stream>>>(
        q_ptr, k_ptr, v_ptr, m_ptr, o_ptr, S, n_blocks, scale, n_queries);
  }

  return out;
}
"""

_CPP_SRC = r"""
#include <torch/extension.h>
torch::Tensor nsa_forward(torch::Tensor q, torch::Tensor k, torch::Tensor v);
"""


def _get_ext():
    global _ext
    if _ext is not None:
        return _ext
    from torch.utils.cpp_extension import load_inline

    os.environ.setdefault("TORCH_CUDA_ARCH_LIST", "")
    major, minor = torch.cuda.get_device_capability(0)
    extra_cuda_cflags = [
        "-O3",
        "--use_fast_math",
        "-lineinfo",
        "-U__CUDA_NO_HALF_OPERATORS__",
        "-U__CUDA_NO_HALF_CONVERSIONS__",
        "-U__CUDA_NO_BFLOAT16_CONVERSIONS__",
        "-U__CUDA_NO_HALF2_OPERATORS__",
        f"-gencode=arch=compute_{major}{minor},code=sm_{major}{minor}",
    ]
    _ext = load_inline(
        name="deepseek_nsa_cuda_v4f",
        cpp_sources=_CPP_SRC,
        cuda_sources=_CUDA_SRC,
        functions=["nsa_forward"],
        extra_cuda_cflags=extra_cuda_cflags,
        verbose=False,
    )
    return _ext


class Model(nn.Module):
    def __init__(self, B: int, H: int, S: int, D: int):
        super().__init__()
        self.B, self.H, self.S, self.D = B, H, S, D
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16))

    def forward(self, q: torch.Tensor, k: torch.Tensor, v: torch.Tensor) -> torch.Tensor:
        return _get_ext().nsa_forward(q, k, v)


def get_init_inputs():
    return [1, 16, 1024, 64]


def get_inputs():
    B, H, S, D = 1, 16, 1024, 64
    q = torch.randn(B, H, S, D, dtype=torch.bfloat16)
    k = torch.randn(B, H, S, D, dtype=torch.bfloat16)
    v = torch.randn(B, H, S, D, dtype=torch.bfloat16)
    return [q, k, v]
