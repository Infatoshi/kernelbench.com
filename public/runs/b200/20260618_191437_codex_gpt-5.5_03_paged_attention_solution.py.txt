import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _decode_one_head_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    H: tl.constexpr,
    HKV: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    GROUP: tl.constexpr,
    MAX_SEQ: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // H
    h = pid - b * H
    kv_h = h // GROUP

    offs_d = tl.arange(0, BLOCK_D)
    mask_d = offs_d < D
    q = tl.load(query + (b * H + h) * D + offs_d, mask=mask_d, other=0.0).to(tl.float32)

    m = tl.full((), -float("inf"), tl.float32)
    denom = tl.full((), 0.0, tl.float32)
    acc = tl.zeros((BLOCK_D,), tl.float32)
    seqlen = tl.load(seq_lens + b)

    offs_m = tl.arange(0, BLOCK_M)
    for start in range(0, MAX_SEQ, BLOCK_M):
        tok = start + offs_m
        valid_m = tok < seqlen
        page = tok // P
        page_off = tok - page * P
        block = tl.load(block_table + b * MAX_BLOCKS + page, mask=valid_m, other=0)

        kv_base = ((block * P + page_off) * HKV + kv_h) * (D + D)
        k = tl.load(
            kv_cache + kv_base[:, None] + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        ).to(tl.float32)
        score = tl.sum(k * q[None, :], axis=1) * SCALE
        score = tl.where(valid_m, score, -float("inf"))

        tile_m = tl.max(score, axis=0)
        new_m = tl.maximum(m, tile_m)
        alpha = tl.exp(m - new_m)
        p = tl.exp(score - new_m)

        v = tl.load(
            kv_cache + kv_base[:, None] + D + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        ).to(tl.float32)
        acc = acc * alpha + tl.sum(p[:, None] * v, axis=0)
        denom = denom * alpha + tl.sum(p, axis=0)
        m = new_m

    result = acc / denom
    tl.store(out + (b * H + h) * D + offs_d, result, mask=mask_d)


@triton.jit
def _decode_group_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    H: tl.constexpr,
    HKV: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    GROUP: tl.constexpr,
    MAX_SEQ: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_G: tl.constexpr,
):
    b = tl.program_id(0)
    kv_h = tl.program_id(1)
    g_block = tl.program_id(2)
    g_start = g_block * BLOCK_G

    offs_d = tl.arange(0, BLOCK_D)
    offs_g = tl.arange(0, BLOCK_G)
    offs_m = tl.arange(0, BLOCK_M)
    head = kv_h * GROUP + g_start + offs_g
    mask_d = offs_d < D
    mask_g = (g_start + offs_g) < GROUP

    q = tl.load(
        query + (b * H + head[None, :]) * D + offs_d[:, None],
        mask=mask_d[:, None] & mask_g[None, :],
        other=0.0,
    )

    m = tl.full((BLOCK_G,), -float("inf"), tl.float32)
    denom = tl.zeros((BLOCK_G,), tl.float32)
    acc = tl.zeros((BLOCK_G, BLOCK_D), tl.float32)
    seqlen = tl.load(seq_lens + b)

    for start in range(0, MAX_SEQ, BLOCK_M):
        tok = start + offs_m
        valid_m = tok < seqlen
        page = tok // P
        page_off = tok - page * P
        block = tl.load(block_table + b * MAX_BLOCKS + page, mask=valid_m, other=0)

        kv_base = ((block * P + page_off) * HKV + kv_h) * (D + D)
        k = tl.load(
            kv_cache + kv_base[:, None] + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        )
        score = tl.dot(k, q, out_dtype=tl.float32) * SCALE
        score = tl.where(valid_m[:, None] & mask_g[None, :], score, -float("inf"))

        tile_m = tl.max(score, axis=0)
        new_m = tl.maximum(m, tile_m)
        alpha = tl.exp(m - new_m)
        p = tl.exp(score - new_m[None, :])

        v = tl.load(
            kv_cache + kv_base[:, None] + D + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        )
        acc = acc * alpha[:, None] + tl.dot(tl.trans(p), v.to(tl.float32), out_dtype=tl.float32)
        denom = denom * alpha + tl.sum(p, axis=0)
        m = new_m

    result = acc / denom[:, None]
    tl.store(
        out + (b * H + head[:, None]) * D + offs_d[None, :],
        result,
        mask=mask_g[:, None] & mask_d[None, :],
    )


@triton.jit
def _partial_group_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    partial_acc,
    partial_m,
    partial_l,
    H: tl.constexpr,
    HKV: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    GROUP: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    SCALE: tl.constexpr,
    PARTITION_SIZE: tl.constexpr,
    NUM_PARTS: tl.constexpr,
    NUM_G_BLOCKS: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_G: tl.constexpr,
):
    b = tl.program_id(0)
    kv_h = tl.program_id(1)
    part_and_g = tl.program_id(2)
    part = part_and_g // NUM_G_BLOCKS
    g_block = part_and_g - part * NUM_G_BLOCKS
    g_start = g_block * BLOCK_G

    offs_d = tl.arange(0, BLOCK_D)
    offs_g = tl.arange(0, BLOCK_G)
    offs_m = tl.arange(0, BLOCK_M)
    head = kv_h * GROUP + g_start + offs_g
    mask_d = offs_d < D
    mask_g = (g_start + offs_g) < GROUP

    q = tl.load(
        query + (b * H + head[None, :]) * D + offs_d[:, None],
        mask=mask_d[:, None] & mask_g[None, :],
        other=0.0,
    )

    m = tl.full((BLOCK_G,), -float("inf"), tl.float32)
    denom = tl.zeros((BLOCK_G,), tl.float32)
    acc = tl.zeros((BLOCK_G, BLOCK_D), tl.float32)
    seqlen = tl.load(seq_lens + b)
    part_start = part * PARTITION_SIZE

    for rel_start in range(0, PARTITION_SIZE, BLOCK_M):
        tok = part_start + rel_start + offs_m
        valid_m = tok < seqlen
        page = tok // P
        page_off = tok - page * P
        block = tl.load(block_table + b * MAX_BLOCKS + page, mask=valid_m, other=0)

        kv_base = ((block * P + page_off) * HKV + kv_h) * (D + D)
        k = tl.load(
            kv_cache + kv_base[:, None] + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        )
        score = tl.dot(k, q, out_dtype=tl.float32) * SCALE
        score = tl.where(valid_m[:, None] & mask_g[None, :], score, -float("inf"))

        tile_m = tl.max(score, axis=0)
        new_m = tl.maximum(m, tile_m)
        alpha = tl.exp(m - new_m)
        p = tl.exp(score - new_m[None, :])

        v = tl.load(
            kv_cache + kv_base[:, None] + D + offs_d[None, :],
            mask=valid_m[:, None] & mask_d[None, :],
            other=0.0,
        )
        acc = acc * alpha[:, None] + tl.dot(tl.trans(p), v.to(tl.float32), out_dtype=tl.float32)
        denom = denom * alpha + tl.sum(p, axis=0)
        m = new_m

    stats_off = ((b * HKV + kv_h) * NUM_PARTS + part) * GROUP + g_start + offs_g
    tl.store(partial_m + stats_off, m, mask=mask_g)
    tl.store(partial_l + stats_off, denom, mask=mask_g)

    acc_off = (
        ((b * HKV + kv_h) * NUM_PARTS + part) * GROUP + g_start + offs_g[:, None]
    ) * D + offs_d[None, :]
    tl.store(partial_acc + acc_off, acc, mask=mask_g[:, None] & mask_d[None, :])


@triton.jit
def _finalize_group_kernel(
    partial_acc,
    partial_m,
    partial_l,
    out,
    H: tl.constexpr,
    HKV: tl.constexpr,
    D: tl.constexpr,
    GROUP: tl.constexpr,
    NUM_PARTS: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_G: tl.constexpr,
):
    b = tl.program_id(0)
    kv_h = tl.program_id(1)
    g_block = tl.program_id(2)
    g_start = g_block * BLOCK_G

    offs_d = tl.arange(0, BLOCK_D)
    offs_g = tl.arange(0, BLOCK_G)
    head = kv_h * GROUP + g_start + offs_g
    mask_d = offs_d < D
    mask_g = (g_start + offs_g) < GROUP

    m = tl.full((BLOCK_G,), -float("inf"), tl.float32)
    denom = tl.zeros((BLOCK_G,), tl.float32)
    acc = tl.zeros((BLOCK_G, BLOCK_D), tl.float32)

    for part in range(0, NUM_PARTS):
        stats_off = ((b * HKV + kv_h) * NUM_PARTS + part) * GROUP + g_start + offs_g
        part_m = tl.load(partial_m + stats_off, mask=mask_g, other=-float("inf"))
        part_l = tl.load(partial_l + stats_off, mask=mask_g, other=0.0)

        new_m = tl.maximum(m, part_m)
        alpha = tl.exp(m - new_m)
        beta = tl.exp(part_m - new_m)
        acc_off = (
            ((b * HKV + kv_h) * NUM_PARTS + part) * GROUP + g_start + offs_g[:, None]
        ) * D + offs_d[None, :]
        part_acc = tl.load(
            partial_acc + acc_off,
            mask=mask_g[:, None] & mask_d[None, :],
            other=0.0,
        )
        acc = acc * alpha[:, None] + part_acc * beta[:, None]
        denom = denom * alpha + part_l * beta
        m = new_m

    result = acc / denom[:, None]
    tl.store(
        out + (b * H + head[:, None]) * D + offs_d[None, :],
        result,
        mask=mask_g[:, None] & mask_d[None, :],
    )


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.max_blocks = (seq_len + page_size - 1) // page_size
        self.partition_size = 128 if head_dim == 64 or seq_len <= 1024 else 256
        self.num_partitions = triton.cdiv(seq_len, self.partition_size)
        self._workspace_key = None
        self._partial_acc = None
        self._partial_m = None
        self._partial_l = None
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def _ensure_workspace(self, query: torch.Tensor):
        key = (query.device, self.batch, self.num_kv_heads, self.num_partitions, self.group_size, self.head_dim)
        if key != self._workspace_key:
            self._partial_acc = torch.empty(
                (self.batch, self.num_kv_heads, self.num_partitions, self.group_size, self.head_dim),
                device=query.device,
                dtype=torch.float32,
            )
            self._partial_m = torch.empty(
                (self.batch, self.num_kv_heads, self.num_partitions, self.group_size),
                device=query.device,
                dtype=torch.float32,
            )
            self._partial_l = torch.empty_like(self._partial_m)
            self._workspace_key = key
        return self._partial_acc, self._partial_m, self._partial_l

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        out = torch.empty_like(query)
        block_m = 32
        block_d = triton.next_power_of_2(self.head_dim)
        heads_per_program = self.group_size
        num_g_blocks = triton.cdiv(self.group_size, heads_per_program)
        partial_acc, partial_m, partial_l = self._ensure_workspace(query)
        _partial_group_kernel[(self.batch, self.num_kv_heads, self.num_partitions * num_g_blocks)](
            query,
            kv_cache,
            block_table,
            seq_lens,
            partial_acc,
            partial_m,
            partial_l,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.page_size,
            self.group_size,
            self.max_blocks,
            self.scale,
            PARTITION_SIZE=self.partition_size,
            NUM_PARTS=self.num_partitions,
            NUM_G_BLOCKS=num_g_blocks,
            BLOCK_M=block_m,
            BLOCK_D=block_d,
            BLOCK_G=heads_per_program,
            num_warps=4,
            num_stages=3,
        )
        _finalize_group_kernel[(self.batch, self.num_kv_heads, num_g_blocks)](
            partial_acc,
            partial_m,
            partial_l,
            out,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.group_size,
            self.num_partitions,
            BLOCK_D=block_d,
            BLOCK_G=heads_per_program,
            num_warps=4,
            num_stages=3,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
