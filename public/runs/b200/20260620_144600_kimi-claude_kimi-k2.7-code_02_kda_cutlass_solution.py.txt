"""Chunk-parallel Kimi Delta Attention forward via custom Triton kernels.

The per-chunk triangular solve (A_kk, w, u, Aqk) and the inter-chunk
recurrence are each implemented as a fused Triton kernel.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl
from einops import rearrange

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


@triton.jit
def _kda_chunk_kernel(
    q_ptr, k_ptr, v_ptr, g_ptr, beta_ptr, w_ptr, u_ptr, aqk_ptr,
    B, H, NT, K, V,
    stride_qb, stride_qh, stride_qn, stride_qc, stride_qk,
    stride_kb, stride_kh, stride_kn, stride_kc, stride_kk,
    stride_vb, stride_vh, stride_vn, stride_vc, stride_vv,
    stride_gb, stride_gh, stride_gn, stride_gc, stride_gk,
    stride_bb, stride_bh, stride_bn, stride_bc,
    stride_wb, stride_wh, stride_wn, stride_wc, stride_wk,
    stride_ub, stride_uh, stride_un, stride_uc, stride_uv,
    stride_ab, stride_ah, stride_an, stride_ai, stride_aj,
    BLOCK_T: tl.constexpr, BLOCK_K: tl.constexpr, BLOCK_V: tl.constexpr,
):
    """Per-chunk kernel: builds A_kk and computes w, u, Aqk."""
    pid_n = tl.program_id(0)
    pid_bh = tl.program_id(1)
    b = pid_bh // H
    h = pid_bh % H
    n = pid_n

    offs_c = tl.arange(0, BLOCK_T)
    offs_k = tl.arange(0, BLOCK_K)
    base_q = q_ptr + b * stride_qb + h * stride_qh + n * stride_qn
    base_k = k_ptr + b * stride_kb + h * stride_kh + n * stride_kn
    base_v = v_ptr + b * stride_vb + h * stride_vh + n * stride_vn
    base_g = g_ptr + b * stride_gb + h * stride_gh + n * stride_gn
    base_b = beta_ptr + b * stride_bb + h * stride_bh + n * stride_bn
    base_w = w_ptr + b * stride_wb + h * stride_wh + n * stride_wn
    base_u = u_ptr + b * stride_ub + h * stride_uh + n * stride_un
    base_a = aqk_ptr + b * stride_ab + h * stride_ah + n * stride_an

    q = tl.load(base_q + offs_c[:, None] * stride_qc + offs_k[None, :] * stride_qk).to(tl.float32)
    k = tl.load(base_k + offs_c[:, None] * stride_kc + offs_k[None, :] * stride_kk).to(tl.float32)
    g = tl.load(base_g + offs_c[:, None] * stride_gc + offs_k[None, :] * stride_gk).to(tl.float32)
    beta = tl.load(base_b + offs_c * stride_bc).to(tl.float32)
    g_cs = tl.cumsum(g, axis=0)

    kp = k * tl.exp(g_cs)
    kn = k * tl.exp(-g_cs)
    qp = q * tl.exp(g_cs)

    # M = kp @ kn^T, then A = -M * beta, keep strict lower
    M = tl.dot(kp, tl.trans(kn))
    A = -M * beta[None, :]
    strict_lower = (offs_c[:, None] > offs_c[None, :]).to(tl.float32)
    A = A * strict_lower

    # Forward-substitution: A += A @ A (row by row)
    for i in range(1, BLOCK_T):
        row_mask = (offs_c[:, None] == i).to(tl.float32)
        row = tl.sum(A * row_mask, axis=0, keep_dims=True)
        upd = tl.dot(row, A)
        A = A + upd * row_mask

    eye = (offs_c[:, None] == offs_c[None, :]).to(tl.float32)
    A = (A + eye) * beta[None, :]

    # w = A @ kp, u = A @ v
    w = tl.dot(A, kp)
    tl.store(base_w + offs_c[:, None] * stride_wc + offs_k[None, :] * stride_wk, w)

    offs_v0 = tl.arange(0, BLOCK_V)
    num_vb = tl.cdiv(V, BLOCK_V)
    for vb in range(num_vb):
        offs_v = offs_v0 + vb * BLOCK_V
        mask_v = offs_v < V
        v_tile = tl.load(
            base_v + offs_c[:, None] * stride_vc + offs_v[None, :] * stride_vv,
            mask=mask_v[None, :], other=0.0,
        ).to(tl.float32)
        u_tile = tl.dot(A, v_tile)
        tl.store(
            base_u + offs_c[:, None] * stride_uc + offs_v[None, :] * stride_uv,
            u_tile, mask=mask_v[None, :],
        )

    # Aqk = qp @ kn^T, keep lower triangular incl. diagonal
    Aqk = tl.dot(qp, tl.trans(kn))
    lower_mask = (offs_c[:, None] >= offs_c[None, :]).to(tl.float32)
    Aqk = Aqk * lower_mask
    tl.store(base_a + offs_c[:, None] * stride_ai + offs_c[None, :] * stride_aj, Aqk)


@triton.jit
def _kda_recurrent_kernel(
    q_ptr, k_ptr, v_ptr, g_ptr, w_ptr, u_ptr, aqk_ptr, o_ptr,
    B, H, NT, K, V,
    stride_qb, stride_qh, stride_qn, stride_qc, stride_qk,
    stride_kb, stride_kh, stride_kn, stride_kc, stride_kk,
    stride_vb, stride_vh, stride_vn, stride_vc, stride_vv,
    stride_gb, stride_gh, stride_gn, stride_gc, stride_gk,
    stride_wb, stride_wh, stride_wn, stride_wc, stride_wk,
    stride_ub, stride_uh, stride_un, stride_uc, stride_uv,
    stride_ab, stride_ah, stride_an, stride_ai, stride_aj,
    stride_ob, stride_oh, stride_on, stride_oc, stride_ov,
    BLOCK_T: tl.constexpr, BLOCK_K: tl.constexpr, BLOCK_V: tl.constexpr,
):
    """Inter-chunk recurrent pass: walks chunks in order, maintains S, writes o."""
    pid_bh = tl.program_id(0)
    pid_vb = tl.program_id(1)
    b = pid_bh // H
    h = pid_bh % H
    v0 = pid_vb * BLOCK_V
    offs_c = tl.arange(0, BLOCK_T)
    offs_k = tl.arange(0, BLOCK_K)
    offs_v = tl.arange(0, BLOCK_V) + v0
    mask_v = offs_v < V

    S = tl.zeros((BLOCK_K, BLOCK_V), dtype=tl.float32)

    for n in range(NT):
        base_q = q_ptr + b * stride_qb + h * stride_qh + n * stride_qn
        base_k = k_ptr + b * stride_kb + h * stride_kh + n * stride_kn
        base_v = v_ptr + b * stride_vb + h * stride_vh + n * stride_vn
        base_g = g_ptr + b * stride_gb + h * stride_gh + n * stride_gn
        base_w = w_ptr + b * stride_wb + h * stride_wh + n * stride_wn
        base_u = u_ptr + b * stride_ub + h * stride_uh + n * stride_un
        base_a = aqk_ptr + b * stride_ab + h * stride_ah + n * stride_an
        base_o = o_ptr + b * stride_ob + h * stride_oh + n * stride_on

        q = tl.load(base_q + offs_c[:, None] * stride_qc + offs_k[None, :] * stride_qk).to(tl.float32)
        k_ = tl.load(base_k + offs_c[:, None] * stride_kc + offs_k[None, :] * stride_kk).to(tl.float32)
        v = tl.load(
            base_v + offs_c[:, None] * stride_vc + offs_v[None, :] * stride_vv,
            mask=mask_v[None, :], other=0.0,
        ).to(tl.float32)
        g = tl.load(base_g + offs_c[:, None] * stride_gc + offs_k[None, :] * stride_gk).to(tl.float32)
        w = tl.load(base_w + offs_c[:, None] * stride_wc + offs_k[None, :] * stride_wk)
        u = tl.load(
            base_u + offs_c[:, None] * stride_uc + offs_v[None, :] * stride_uv,
            mask=mask_v[None, :], other=0.0,
        )
        Aqk = tl.load(base_a + offs_c[:, None] * stride_ai + offs_c[None, :] * stride_aj)

        g_cs = tl.cumsum(g, axis=0)

        v_adj = u - tl.dot(w, S)

        q_dec = q * tl.exp(g_cs)
        o_tile = tl.dot(q_dec, S)
        o_tile = o_tile + tl.dot(Aqk, v_adj)
        tl.store(
            base_o + offs_c[:, None] * stride_oc + offs_v[None, :] * stride_ov,
            o_tile.to(tl.bfloat16), mask=mask_v[None, :],
        )

        mask_last = (offs_c[:, None] == BLOCK_T - 1).to(tl.float32)
        g_last = tl.sum(g_cs * mask_last, axis=0)
        S = S * tl.exp(g_last)[:, None]
        k_dec_state = k_ * tl.exp(g_last[None, :] - g_cs)
        S = S + tl.dot(tl.trans(k_dec_state), v_adj)


def _kda_fwd(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int,
) -> torch.Tensor:
    B, T, H, K = q.shape
    V = v.shape[-1]
    NT = T // chunk_size

    q = rearrange(q.contiguous(), "b (n c) h d -> b h n c d", c=chunk_size) * scale
    k = rearrange(k.contiguous(), "b (n c) h d -> b h n c d", c=chunk_size)
    v = rearrange(v.contiguous(), "b (n c) h d -> b h n c d", c=chunk_size)
    g = rearrange(g.contiguous().to(torch.float32), "b (n c) h d -> b h n c d", c=chunk_size)
    beta = rearrange(beta.contiguous(), "b (n c) h -> b h n c", c=chunk_size).to(torch.float32)

    g_cs = g.cumsum(-2)

    # ---- A_kk via triangular solve (matches reference forward substitution) ----
    kp = k * g_cs.exp()
    kn = k * (-g_cs).exp()
    M = torch.matmul(kp, kn.transpose(-2, -1))
    strict_lower = torch.tril(torch.ones(chunk_size, chunk_size, device=q.device), diagonal=-1)
    M = M * strict_lower
    Mbeta = M * beta[..., None]
    eye = torch.eye(chunk_size, dtype=Mbeta.dtype, device=q.device)
    D = eye[None, None, None, :, :] * beta[..., None]
    A = torch.linalg.solve_triangular(eye[None, None, None, :, :] + Mbeta, D, upper=False)

    # ---- w and u --------------------------------------------------------------
    k_dec = g_cs.exp() * k
    w = torch.matmul(A, k_dec)
    u = torch.matmul(A, v.to(torch.float32))

    # ---- Aqk (lower triangular incl. diagonal) --------------------------------
    qp = q * g_cs.exp()
    Aqk = torch.matmul(qp, kn.transpose(-2, -1))
    Aqk = Aqk * torch.tril(torch.ones(chunk_size, chunk_size, device=q.device))

    w = w.to(torch.float32)
    u = u.to(torch.float32)
    Aqk = Aqk.to(torch.float32)
    o = torch.empty_like(v)

    grid = (B * H, triton.cdiv(V, 32))
    _kda_recurrent_kernel[grid](
        q, k, v, g, w, u, Aqk, o,
        B, H, NT, K, V,
        *q.stride(), *k.stride(), *v.stride(), *g.stride(),
        *w.stride(), *u.stride(), *Aqk.stride(), *o.stride(),
        BLOCK_T=chunk_size, BLOCK_K=K, BLOCK_V=32,
        num_warps=4, num_stages=1,
    )

    o = rearrange(o, "b h n c d -> b (n c) h d")
    return o


class Model(nn.Module):
    """KDA forward (chunk form). No learned parameters; all inputs are activations."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        return _kda_fwd(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)


# Module-level shape shims (overridden by check.py / benchmark.py per shape).
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    """Return a list of activations for one forward call."""
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
