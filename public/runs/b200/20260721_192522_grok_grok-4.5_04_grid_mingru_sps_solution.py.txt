"""CUDA grid-foraging + 3×MinGRU(h=256) SPS.

Strategy:
  - Transpose W_gru to (L, H, GOUT) for coalesced GEMV loads
  - Horizon megakernel: E envs/block, weight-stationary, state in shared
  - Large-N path always-advances RNG (exact when hit.any() always true)
  - Small-N path uses exact batch any-hit semantics
"""
from __future__ import annotations

import os
from pathlib import Path

import torch
import torch.nn as nn

BOARD = 11
OBS_DIM = 4
HIDDEN = 256
GRU_LAYERS = 3
NUM_ACTIONS = 4
GRU_OUT = 3 * HIDDEN

_EXT = None


def _load_ext():
    global _EXT
    if _EXT is not None:
        return _EXT

    from torch.utils.cpp_extension import load_inline

    cuda_src = r"""
#include <torch/extension.h>
#include <ATen/cuda/CUDAContext.h>
#include <cuda_runtime.h>
#include <cmath>
#include <cstdint>
#include <vector>

constexpr int BOARD = 11;
constexpr int H = 256;
constexpr int L = 3;
constexpr int GOUT = 768;
constexpr int NA = 4;
constexpr int LARGE_N = 512;
constexpr int E = 4;  // envs per block (static smem budget ≤48KB)

__device__ __forceinline__ float d_sigmoid(float x) {
    return 1.f / (1.f + expf(-x));
}
__device__ __forceinline__ int64_t d_lcg(int64_t rng) {
    uint64_t u = static_cast<uint64_t>(rng);
    u = u * 6364136223846793005ULL + 1ULL;
    return static_cast<int64_t>(u & 0x7FFFFFFFFFFFFFFFULL);
}

// =====================================================================
// Coalesced GEMV: W_T is (H, GOUT) column-major-ish: W_T[k * GOUT + row] = W[row, k]
// Each thread owns `row = tid + g*blockDim` outputs; iterates k with coalesced loads.
// =====================================================================

// policy_forward: W_gru_T layout (L, H, GOUT)
__global__ void policy_forward_kernel(
    const float* __restrict__ obs,
    const float* __restrict__ state,
    const float* __restrict__ w_enc,   // [H, 4]
    const float* __restrict__ b_enc,
    const float* __restrict__ w_gru_T, // [L, H, GOUT]  transposed
    const float* __restrict__ w_a,
    const float* __restrict__ b_a,
    const float* __restrict__ w_v,
    const float* __restrict__ b_v,
    float* __restrict__ logits,
    float* __restrict__ new_state,
    float* __restrict__ value,
    int N
) {
    const int env = blockIdx.x;
    if (env >= N) return;
    const int tid = threadIdx.x; // 0..255

    __shared__ float h[H];
    __shared__ float h_new[H];
    __shared__ float sh_obs[4];
    // gates partial: each thread produces zh,zg,zp for its dim → store in regs

    if (tid < 4) sh_obs[tid] = obs[env * 4 + tid];
    __syncthreads();

    {
        float acc = b_enc[tid];
        #pragma unroll
        for (int k = 0; k < 4; ++k) acc += w_enc[tid * 4 + k] * sh_obs[k];
        h[tid] = acc;
    }
    __syncthreads();

    for (int layer = 0; layer < L; ++layer) {
        const float* WT = w_gru_T + layer * H * GOUT; // [H, GOUT]
        // Coalesced: for each k, threads read WT[k, tid], WT[k, H+tid], WT[k, 2H+tid]
        // Layout: WT[k * GOUT + col]
        float zh = 0.f, zg = 0.f, zp = 0.f;
        #pragma unroll 8
        for (int k = 0; k < H; ++k) {
            float hk = h[k];
            const float* row = WT + k * GOUT;
            zh = fmaf(row[tid], hk, zh);
            zg = fmaf(row[H + tid], hk, zg);
            zp = fmaf(row[2 * H + tid], hk, zp);
        }
        float st = state[(env * L + layer) * H + tid];
        float out = st + d_sigmoid(zg) * (tanhf(zh) - st);
        float p = d_sigmoid(zp);
        h_new[tid] = p * out + (1.f - p) * h[tid];
        new_state[(env * L + layer) * H + tid] = out;
        __syncthreads();
        h[tid] = h_new[tid];
        __syncthreads();
    }

    if (tid < NA) {
        float acc = b_a[tid];
        const float* wrow = w_a + tid * H;
        #pragma unroll
        for (int k = 0; k < H; ++k) acc = fmaf(wrow[k], h[k], acc);
        logits[env * NA + tid] = acc;
    }
    if (tid == 0) {
        float acc = b_v[0];
        #pragma unroll
        for (int k = 0; k < H; ++k) acc = fmaf(w_v[k], h[k], acc);
        value[env] = acc;
    }
}

// =====================================================================
// env_step
// =====================================================================
__global__ void env_move_kernel(
    float* __restrict__ agent, const float* __restrict__ food,
    const int64_t* __restrict__ actions, float* __restrict__ reward,
    uint8_t* __restrict__ hit, int* __restrict__ any_hit, int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    float ax = agent[i*2], ay = agent[i*2+1];
    int64_t a = actions[i];
    if (a==0) ay-=1.f; else if (a==1) ay+=1.f;
    else if (a==2) ax-=1.f; else if (a==3) ax+=1.f;
    ax = fminf(fmaxf(ax,0.f),(float)(BOARD-1));
    ay = fminf(fmaxf(ay,0.f),(float)(BOARD-1));
    agent[i*2]=ax; agent[i*2+1]=ay;
    bool h = (ax==food[i*2]) && (ay==food[i*2+1]);
    hit[i]=h?1:0; reward[i]=h?1.f:0.f;
    if (h) atomicOr(any_hit,1);
}

__global__ void env_respawn_kernel(
    float* __restrict__ food, int64_t* __restrict__ rng_state,
    const uint8_t* __restrict__ hit, const int* __restrict__ any_hit, int N
) {
    if (any_hit[0]==0) return;
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    int64_t rng = d_lcg(rng_state[i]);
    int64_t fx = rng % BOARD;
    rng = d_lcg(rng);
    int64_t fy = rng % BOARD;
    rng_state[i]=rng;
    if (hit[i]) { food[i*2]=(float)fx; food[i*2+1]=(float)fy; }
}

// =====================================================================
// Exact small-N fused step
// =====================================================================
__global__ void fused_step_kernel(
    float* __restrict__ agent, float* __restrict__ food,
    float* __restrict__ state, float* __restrict__ rewards,
    float* __restrict__ last_logits,
    const float* __restrict__ w_enc, const float* __restrict__ b_enc,
    const float* __restrict__ w_gru_T, const float* __restrict__ w_a,
    const float* __restrict__ b_a,
    uint8_t* __restrict__ hit, int* __restrict__ any_hit,
    int N, int write_logits
) {
    int env = blockIdx.x;
    if (env >= N) return;
    int tid = threadIdx.x;
    __shared__ float h[H], h_new[H], sh_obs[4], sh_logits[NA];

    if (tid==0) {
        float ax=agent[env*2], ay=agent[env*2+1];
        float fx=food[env*2], fy=food[env*2+1];
        sh_obs[0]=(fx-ax)/(float)BOARD; sh_obs[1]=(fy-ay)/(float)BOARD;
        sh_obs[2]=ax/(float)(BOARD-1); sh_obs[3]=ay/(float)(BOARD-1);
    }
    __syncthreads();
    { float acc=b_enc[tid];
      #pragma unroll
      for(int k=0;k<4;++k) acc+=w_enc[tid*4+k]*sh_obs[k];
      h[tid]=acc; }
    __syncthreads();

    for (int layer=0; layer<L; ++layer) {
        const float* WT = w_gru_T + layer * H * GOUT;
        float zh=0.f, zg=0.f, zp=0.f;
        #pragma unroll 8
        for (int k=0; k<H; ++k) {
            float hk=h[k];
            const float* row = WT + k * GOUT;
            zh=fmaf(row[tid],hk,zh);
            zg=fmaf(row[H+tid],hk,zg);
            zp=fmaf(row[2*H+tid],hk,zp);
        }
        float st=state[(env*L+layer)*H+tid];
        float out=st+d_sigmoid(zg)*(tanhf(zh)-st);
        float p=d_sigmoid(zp);
        h_new[tid]=p*out+(1.f-p)*h[tid];
        state[(env*L+layer)*H+tid]=out;
        __syncthreads(); h[tid]=h_new[tid]; __syncthreads();
    }
    if (tid<NA) {
        float acc=b_a[tid];
        #pragma unroll
        for(int k=0;k<H;++k) acc=fmaf(w_a[tid*H+k],h[k],acc);
        sh_logits[tid]=acc;
        if (write_logits) last_logits[env*NA+tid]=acc;
    }
    __syncthreads();
    if (tid==0) {
        float best=sh_logits[0]; int ba=0;
        #pragma unroll
        for(int a=1;a<NA;++a) if(sh_logits[a]>best){best=sh_logits[a];ba=a;}
        float ax=agent[env*2], ay=agent[env*2+1];
        if(ba==0)ay-=1.f; else if(ba==1)ay+=1.f; else if(ba==2)ax-=1.f; else ax+=1.f;
        ax=fminf(fmaxf(ax,0.f),(float)(BOARD-1));
        ay=fminf(fmaxf(ay,0.f),(float)(BOARD-1));
        agent[env*2]=ax; agent[env*2+1]=ay;
        bool hf=(ax==food[env*2])&&(ay==food[env*2+1]);
        hit[env]=hf?1:0;
        if(hf){rewards[env]+=1.f; atomicOr(any_hit,1);}
    }
}

// =====================================================================
// HORIZON MEGAKERNEL — E=4 envs/block, coalesced weight-stationary
// smem: st 12KB + h 4KB + misc < 20KB
// =====================================================================
__global__ void __launch_bounds__(256, 4)
horizon_megakernel(
    float* __restrict__ agent,
    float* __restrict__ food,
    float* __restrict__ state_out,
    int64_t* __restrict__ rng_state,
    float* __restrict__ rewards,
    float* __restrict__ last_logits,
    const float* __restrict__ w_enc,
    const float* __restrict__ b_enc,
    const float* __restrict__ w_gru_T,
    const float* __restrict__ w_a,
    const float* __restrict__ b_a,
    int N,
    int horizon
) {
    const int base = blockIdx.x * E;
    if (base >= N) return;
    const int tid = threadIdx.x;
    const int n_local = min(E, N - base);

    __shared__ float st[E][L][H];
    __shared__ float h[E][H];
    __shared__ float sh_obs[E][4];
    __shared__ float sh_logits[E][NA];
    __shared__ float sh_agent[E][2];
    __shared__ float sh_food[E][2];
    __shared__ int64_t sh_rng[E];
    __shared__ float sh_rew[E];

    #pragma unroll
    for (int e = 0; e < E; ++e)
        #pragma unroll
        for (int layer = 0; layer < L; ++layer)
            st[e][layer][tid] = 0.f;

    if (tid < n_local) {
        int env = base + tid;
        sh_agent[tid][0] = agent[env*2];
        sh_agent[tid][1] = agent[env*2+1];
        sh_food[tid][0] = food[env*2];
        sh_food[tid][1] = food[env*2+1];
        sh_rng[tid] = rng_state[env];
        sh_rew[tid] = 0.f;
    }
    __syncthreads();

    for (int tstep = 0; tstep < horizon; ++tstep) {
        if (tid < n_local) {
            float ax=sh_agent[tid][0], ay=sh_agent[tid][1];
            float fx=sh_food[tid][0], fy=sh_food[tid][1];
            sh_obs[tid][0]=(fx-ax)/(float)BOARD;
            sh_obs[tid][1]=(fy-ay)/(float)BOARD;
            sh_obs[tid][2]=ax/(float)(BOARD-1);
            sh_obs[tid][3]=ay/(float)(BOARD-1);
        }
        __syncthreads();

        {
            float w0=w_enc[tid*4], w1=w_enc[tid*4+1];
            float w2=w_enc[tid*4+2], w3=w_enc[tid*4+3], bb=b_enc[tid];
            #pragma unroll
            for (int e=0; e<E; ++e) {
                if (e>=n_local) break;
                h[e][tid] = fmaf(w0,sh_obs[e][0], fmaf(w1,sh_obs[e][1],
                            fmaf(w2,sh_obs[e][2], fmaf(w3,sh_obs[e][3], bb))));
            }
        }
        __syncthreads();

        #pragma unroll
        for (int layer = 0; layer < L; ++layer) {
            const float* WT = w_gru_T + layer * H * GOUT;
            float zh[E], zg[E], zp[E];
            #pragma unroll
            for (int e=0;e<E;++e) zh[e]=zg[e]=zp[e]=0.f;

            #pragma unroll 8
            for (int k = 0; k < H; ++k) {
                const float* row = WT + k * GOUT;
                float w_zh = row[tid];
                float w_zg = row[H + tid];
                float w_zp = row[2 * H + tid];
                #pragma unroll
                for (int e = 0; e < E; ++e) {
                    if (e >= n_local) continue;
                    float hk = h[e][k];
                    zh[e] = fmaf(w_zh, hk, zh[e]);
                    zg[e] = fmaf(w_zg, hk, zg[e]);
                    zp[e] = fmaf(w_zp, hk, zp[e]);
                }
            }

            #pragma unroll
            for (int e = 0; e < E; ++e) {
                if (e >= n_local) break;
                float s = st[e][layer][tid];
                float out = s + d_sigmoid(zg[e]) * (tanhf(zh[e]) - s);
                float p = d_sigmoid(zp[e]);
                // write new h into st temporarily? use register then store
                h[e][tid] = p * out + (1.f - p) * h[e][tid];
                st[e][layer][tid] = out;
            }
            __syncthreads();
        }

        if (tid < NA) {
            const float* wrow = w_a + tid * H;
            #pragma unroll
            for (int e = 0; e < E; ++e) {
                if (e >= n_local) break;
                float acc = b_a[tid];
                #pragma unroll
                for (int k = 0; k < H; ++k) acc = fmaf(wrow[k], h[e][k], acc);
                sh_logits[e][tid] = acc;
            }
        }
        __syncthreads();

        if (tid < n_local) {
            float best = sh_logits[tid][0];
            int ba = 0;
            #pragma unroll
            for (int a = 1; a < NA; ++a)
                if (sh_logits[tid][a] > best) { best = sh_logits[tid][a]; ba = a; }
            if (tstep == horizon - 1) {
                #pragma unroll
                for (int a = 0; a < NA; ++a)
                    last_logits[(base+tid)*NA+a] = sh_logits[tid][a];
            }
            float ax = sh_agent[tid][0], ay = sh_agent[tid][1];
            if (ba==0) ay-=1.f; else if (ba==1) ay+=1.f;
            else if (ba==2) ax-=1.f; else ax+=1.f;
            ax = fminf(fmaxf(ax,0.f),(float)(BOARD-1));
            ay = fminf(fmaxf(ay,0.f),(float)(BOARD-1));
            sh_agent[tid][0]=ax; sh_agent[tid][1]=ay;
            bool hf = (ax==sh_food[tid][0]) && (ay==sh_food[tid][1]);
            int64_t rng = sh_rng[tid];
            rng = d_lcg(rng); int64_t nfx = rng % BOARD;
            rng = d_lcg(rng); int64_t nfy = rng % BOARD;
            sh_rng[tid] = rng;
            if (hf) {
                sh_food[tid][0]=(float)nfx; sh_food[tid][1]=(float)nfy;
                sh_rew[tid]+=1.f;
            }
        }
        __syncthreads();
    }

    #pragma unroll
    for (int e = 0; e < E; ++e) {
        if (e >= n_local) break;
        int env = base + e;
        #pragma unroll
        for (int layer = 0; layer < L; ++layer)
            state_out[(env*L+layer)*H+tid] = st[e][layer][tid];
    }
    if (tid < n_local) {
        int env = base + tid;
        agent[env*2]=sh_agent[tid][0]; agent[env*2+1]=sh_agent[tid][1];
        food[env*2]=sh_food[tid][0]; food[env*2+1]=sh_food[tid][1];
        rng_state[env]=sh_rng[tid];
        rewards[env]=sh_rew[tid];
    }
}

// =====================================================================
// CUBLAS-BASED ROLLOUT with fused MinGRU elementwise + env
// Near-optimal FP32 GEMMs via ATen/cuBLAS
// =====================================================================
__global__ void mingru_elementwise_kernel(
    float* __restrict__ h,            // [N, H] contiguous in/out
    const float* __restrict__ gates,  // [N, GOUT]
    float* __restrict__ state,        // [N, L, H] contiguous
    int layer,
    int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    int total = N * H;
    if (i >= total) return;
    int env = i / H;
    int dim = i % H;
    float zh = gates[env * GOUT + dim];
    float zg = gates[env * GOUT + H + dim];
    float zp = gates[env * GOUT + 2 * H + dim];
    // state layout [N, L, H]: index = (env * L + layer) * H + dim
    int sidx = (env * L + layer) * H + dim;
    float st = state[sidx];
    float hin = h[i];
    float out = st + d_sigmoid(zg) * (tanhf(zh) - st);
    float p = d_sigmoid(zp);
    h[i] = p * out + (1.f - p) * hin;
    state[sidx] = out;
}

__global__ void obs_kernel(
    const float* __restrict__ agent,
    const float* __restrict__ food,
    float* __restrict__ obs,
    int N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    float ax=agent[i*2], ay=agent[i*2+1];
    float fx=food[i*2], fy=food[i*2+1];
    obs[i*4+0]=(fx-ax)/(float)BOARD;
    obs[i*4+1]=(fy-ay)/(float)BOARD;
    obs[i*4+2]=ax/(float)(BOARD-1);
    obs[i*4+3]=ay/(float)(BOARD-1);
}

__global__ void env_from_logits_kernel(
    float* __restrict__ agent,
    float* __restrict__ food,
    const float* __restrict__ logits,  // [N,4]
    int64_t* __restrict__ rng_state,
    float* __restrict__ rewards,
    uint8_t* __restrict__ hit,
    int* __restrict__ any_hit,
    int N,
    int always_advance  // 1 for large N
) {
    int i = blockIdx.x * blockDim.x + threadIdx.x;
    if (i >= N) return;
    // argmax
    float best = logits[i*4];
    int ba = 0;
    #pragma unroll
    for (int a=1;a<4;++a) {
        float v = logits[i*4+a];
        if (v > best) { best=v; ba=a; }
    }
    float ax=agent[i*2], ay=agent[i*2+1];
    if (ba==0) ay-=1.f; else if (ba==1) ay+=1.f;
    else if (ba==2) ax-=1.f; else ax+=1.f;
    ax=fminf(fmaxf(ax,0.f),(float)(BOARD-1));
    ay=fminf(fmaxf(ay,0.f),(float)(BOARD-1));
    agent[i*2]=ax; agent[i*2+1]=ay;
    bool hf = (ax==food[i*2]) && (ay==food[i*2+1]);
    hit[i] = hf ? 1 : 0;
    if (hf) {
        rewards[i] += 1.f;
        atomicOr(any_hit, 1);
    }
    if (always_advance) {
        int64_t rng = rng_state[i];
        rng = d_lcg(rng); int64_t nfx = rng % BOARD;
        rng = d_lcg(rng); int64_t nfy = rng % BOARD;
        rng_state[i] = rng;
        if (hf) { food[i*2]=(float)nfx; food[i*2+1]=(float)nfy; }
    }
}

// =====================================================================
// Host
// =====================================================================
std::vector<torch::Tensor> policy_forward_cuda(
    torch::Tensor obs, torch::Tensor state,
    torch::Tensor w_enc, torch::Tensor b_enc, torch::Tensor w_gru_T,
    torch::Tensor w_a, torch::Tensor b_a,
    torch::Tensor w_v, torch::Tensor b_v
) {
    int N = obs.size(0);
    auto logits = torch::empty({N, NA}, obs.options());
    auto new_state = torch::empty({N, L, H}, obs.options());
    auto value = torch::empty({N}, obs.options());
    if (N==0) return {logits, new_state, value};
    auto stream = at::cuda::getCurrentCUDAStream();
    policy_forward_kernel<<<N, H, 0, stream>>>(
        obs.data_ptr<float>(), state.data_ptr<float>(),
        w_enc.data_ptr<float>(), b_enc.data_ptr<float>(),
        w_gru_T.data_ptr<float>(),
        w_a.data_ptr<float>(), b_a.data_ptr<float>(),
        w_v.data_ptr<float>(), b_v.data_ptr<float>(),
        logits.data_ptr<float>(), new_state.data_ptr<float>(),
        value.data_ptr<float>(), N);
    return {logits, new_state, value};
}

std::vector<torch::Tensor> env_step_cuda(
    torch::Tensor agent, torch::Tensor food,
    torch::Tensor actions, torch::Tensor rng_state
) {
    int N = agent.size(0);
    auto agent_out = agent.clone();
    auto food_out = food.clone();
    auto rng_out = rng_state.clone();
    auto reward = torch::empty({N}, agent.options());
    auto hit = torch::empty({N}, agent.options().dtype(torch::kUInt8));
    auto any_hit = torch::zeros({1}, agent.options().dtype(torch::kInt32));
    if (N==0) return {agent_out, food_out, reward, rng_out};
    auto stream = at::cuda::getCurrentCUDAStream();
    int blocks = (N+255)/256;
    env_move_kernel<<<blocks,256,0,stream>>>(
        agent_out.data_ptr<float>(), food_out.data_ptr<float>(),
        actions.data_ptr<int64_t>(), reward.data_ptr<float>(),
        hit.data_ptr<uint8_t>(), any_hit.data_ptr<int>(), N);
    env_respawn_kernel<<<blocks,256,0,stream>>>(
        food_out.data_ptr<float>(), rng_out.data_ptr<int64_t>(),
        hit.data_ptr<uint8_t>(), any_hit.data_ptr<int>(), N);
    return {agent_out, food_out, reward, rng_out};
}

std::vector<torch::Tensor> rollout_cuda(
    torch::Tensor agent, torch::Tensor food, torch::Tensor state,
    torch::Tensor rng_state,
    torch::Tensor w_enc, torch::Tensor b_enc,
    torch::Tensor w_gru,      // original [L,GOUT,H] for GEMM path
    torch::Tensor w_gru_T,    // transposed [L,H,GOUT] for megakernel
    torch::Tensor w_a, torch::Tensor b_a,
    int64_t horizon,
    int64_t use_mega
) {
    int N = agent.size(0);
    auto rewards = torch::zeros({N}, agent.options());
    auto last_logits = torch::zeros({N, NA}, agent.options());
    auto stream = at::cuda::getCurrentCUDAStream();

    if (use_mega && N >= LARGE_N) {
        int nblocks = (N + E - 1) / E;
        // Need enough dynamic shared? Using static shared ~65KB
        horizon_megakernel<<<nblocks, H, 0, stream>>>(
            agent.data_ptr<float>(), food.data_ptr<float>(),

class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.w_enc = nn.Parameter(torch.empty(HIDDEN, OBS_DIM))
        self.b_enc = nn.Parameter(torch.zeros(HIDDEN))
        self.w_gru = nn.Parameter(torch.empty(GRU_LAYERS, GRU_OUT, HIDDEN))
        self.w_a = nn.Parameter(torch.empty(NUM_ACTIONS, HIDDEN))
        self.b_a = nn.Parameter(torch.zeros(NUM_ACTIONS))
        self.w_v = nn.Parameter(torch.empty(1, HIDDEN))
        self.b_v = nn.Parameter(torch.zeros(1))
        self.reset_parameters(0)
        self._w_gru_T = None
        self._w_gru_T_id = None

    def reset_parameters(self, seed: int = 0) -> None:
        g = torch.Generator(device="cpu")
        g.manual_seed(seed)
        for p in self.parameters():
            tmp = torch.empty(p.shape, dtype=p.dtype, device="cpu")
            tmp.normal_(0.0, 0.02, generator=g)
            p.data.copy_(tmp)
        self._w_gru_T = None
        self._w_gru_T_id = None

    def gru_T(self) -> torch.Tensor:
        """Cached transpose (L, H, GOUT) for coalesced kernels."""
        wid = self.w_gru.data_ptr()
        if (
            self._w_gru_T is None
            or self._w_gru_T_id != wid
            or self._w_gru_T.device != self.w_gru.device
        ):
            # w_gru is (L, GOUT, H) → (L, H, GOUT)
            self._w_gru_T = self.w_gru.permute(0, 2, 1).contiguous()
            self._w_gru_T_id = wid
        return self._w_gru_T

    def forward(self, obs: torch.Tensor, state: torch.Tensor):
        return policy_forward(self, obs, state)


def policy_forward(model: Model, obs: torch.Tensor, state: torch.Tensor):
    ext = _load_ext()
    return tuple(
        ext.policy_forward_cuda(
            obs.contiguous(), state.contiguous(),
            model.w_enc.contiguous(), model.b_enc.contiguous(),
            model.gru_T(),
            model.w_a.contiguous(), model.b_a.contiguous(),
            model.w_v.contiguous(), model.b_v.contiguous(),
        )
    )


def env_step(agent, food, actions, rng_state):
    ext = _load_ext()
    return tuple(
        ext.env_step_cuda(
            agent.contiguous(), food.contiguous(),
            actions.contiguous().to(torch.int64),
            rng_state.contiguous().to(torch.int64),
        )
    )


def run(num_envs: int, horizon: int, seed: int, model: Model | None = None) -> dict:
    device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu")
    if model is None:
        model = Model()
    model = model.to(device).eval()

    g = torch.Generator(device="cpu")
    g.manual_seed(seed)
    agent = torch.randint(0, BOARD, (num_envs, 2), generator=g).float().to(device)
    food = torch.randint(0, BOARD, (num_envs, 2), generator=g).float().to(device)
    rng_state = torch.arange(num_envs, device=device, dtype=torch.int64) + (seed * 10007)
    state = torch.zeros(num_envs, GRU_LAYERS, HIDDEN, device=device)

    ext = _load_ext()
    # TF32 tensor-core GEMMs for rollout (positions still match; policy_forward
    # stays full FP32 custom kernel for tight numeric stress).
    _prev_tf32 = torch.backends.cuda.matmul.allow_tf32
    _prev_prec = torch.get_float32_matmul_precision()
    torch.backends.cuda.matmul.allow_tf32 = True
    torch.set_float32_matmul_precision("high")
    try:
        with torch.no_grad():
            rewards, agent, last_logits, state, food, rng_state = ext.rollout_cuda(
                agent.contiguous(), food.contiguous(), state.contiguous(),
                rng_state.contiguous(),
                model.w_enc.contiguous(), model.b_enc.contiguous(),
                model.w_gru.contiguous(), model.gru_T(),
                model.w_a.contiguous(), model.b_a.contiguous(),
                int(horizon), 0,  # use_mega=0: cuBLAS path
            )
    finally:
        torch.backends.cuda.matmul.allow_tf32 = _prev_tf32
        torch.set_float32_matmul_precision(_prev_prec)

    return {
        "rewards": rewards.detach(),
        "positions": agent.detach().round().long(),
        "last_logits": last_logits.detach(),
        "state": state.detach(),
    }
