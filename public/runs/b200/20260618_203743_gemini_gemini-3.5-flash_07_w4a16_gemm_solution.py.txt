import torch
import torch.nn as nn
import triton
import triton.language as tl

GROUP_SIZE = 128

@triton.jit
def w4a16_gemm_interleave_kernel(
    a_ptr, b_ptr, scales_ptr, zeros_ptr, c_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_scales_k, stride_scales_n,
    stride_zeros_k, stride_zeros_n,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)

    for k in range(0, K, BLOCK_SIZE_K):
        a_offsets = offs_am[:, None] * stride_am + (k + offs_k)[None, :] * stride_ak
        a_mask = (offs_am[:, None] < M) & ((k + offs_k)[None, :] < K)
        a_block = tl.load(a_ptr + a_offsets, mask=a_mask, other=0.0)

        offs_bk = k // 2 + tl.arange(0, BLOCK_SIZE_K // 2)
        b_offsets = offs_bk[:, None] * stride_bk + offs_bn[None, :] * stride_bn
        b_mask = (offs_bk[:, None] < (K // 2)) & (offs_bn[None, :] < N)
        b_packed = tl.load(b_ptr + b_offsets, mask=b_mask, other=0)

        b_even = (b_packed & 0x0F).to(tl.bfloat16)
        b_odd = ((b_packed >> 4) & 0x0F).to(tl.bfloat16)

        b_even_T = tl.trans(b_even)
        b_odd_T = tl.trans(b_odd)
        b_joined_3d = tl.join(b_even_T, b_odd_T)
        b_interleaved_T = tl.reshape(b_joined_3d, (BLOCK_SIZE_N, BLOCK_SIZE_K))
        b_unpacked = tl.trans(b_interleaved_T)

        group_idx = (k + offs_k) // 128
        scales_offsets = group_idx[:, None] * stride_scales_k + offs_bn[None, :] * stride_scales_n
        scales_mask = (group_idx[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        scales_block = tl.load(scales_ptr + scales_offsets, mask=scales_mask, other=1.0)

        zeros_offsets = group_idx[:, None] * stride_zeros_k + offs_bn[None, :] * stride_zeros_n
        zeros_mask = (group_idx[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        zeros_block = tl.load(zeros_ptr + zeros_offsets, mask=zeros_mask, other=0.0)

        b_dequant = (b_unpacked - zeros_block) * scales_block

        accumulator = tl.dot(a_block, b_dequant, acc=accumulator)

    c_offsets = offs_am[:, None] * stride_cm + offs_bn[None, :] * stride_cn
    c_mask = (offs_am[:, None] < M) & (offs_bn[None, :] < N)
    tl.store(c_ptr + c_offsets, accumulator.to(tl.bfloat16), mask=c_mask)


@triton.jit
def w4a16_gemv_split_kernel(
    a_ptr, b_ptr, scales_ptr, zeros_ptr, c_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bk, stride_bn,
    stride_scales_k, stride_scales_n,
    stride_zeros_k, stride_zeros_n,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    num_pid_in_group = GROUP_SIZE_M * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * GROUP_SIZE_M
    group_size_m = min(num_pid_m - first_pid_m, GROUP_SIZE_M)
    pid_m = first_pid_m + ((pid % num_pid_in_group) % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    
    offs_k_even = 2 * tl.arange(0, BLOCK_SIZE_K // 2)
    offs_k_odd = 2 * tl.arange(0, BLOCK_SIZE_K // 2) + 1

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)

    for k in range(0, K, BLOCK_SIZE_K):
        a_offsets_even = offs_am[:, None] * stride_am + (k + offs_k_even)[None, :] * stride_ak
        a_mask_even = (offs_am[:, None] < M) & ((k + offs_k_even)[None, :] < K)
        a_even = tl.load(a_ptr + a_offsets_even, mask=a_mask_even, other=0.0)

        a_offsets_odd = offs_am[:, None] * stride_am + (k + offs_k_odd)[None, :] * stride_ak
        a_mask_odd = (offs_am[:, None] < M) & ((k + offs_k_odd)[None, :] < K)
        a_odd = tl.load(a_ptr + a_offsets_odd, mask=a_mask_odd, other=0.0)

        offs_bk = k // 2 + tl.arange(0, BLOCK_SIZE_K // 2)
        b_offsets = offs_bk[:, None] * stride_bk + offs_bn[None, :] * stride_bn
        b_mask = (offs_bk[:, None] < (K // 2)) & (offs_bn[None, :] < N)
        b_packed = tl.load(b_ptr + b_offsets, mask=b_mask, other=0)

        b_even = (b_packed & 0x0F).to(tl.bfloat16)
        b_odd = ((b_packed >> 4) & 0x0F).to(tl.bfloat16)

        group_idx_even = (k + offs_k_even) // 128
        scales_offsets_even = group_idx_even[:, None] * stride_scales_k + offs_bn[None, :] * stride_scales_n
        scales_mask_even = (group_idx_even[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        scales_block_even = tl.load(scales_ptr + scales_offsets_even, mask=scales_mask_even, other=1.0)

        zeros_offsets_even = group_idx_even[:, None] * stride_zeros_k + offs_bn[None, :] * stride_zeros_n
        zeros_mask_even = (group_idx_even[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        zeros_block_even = tl.load(zeros_ptr + zeros_offsets_even, mask=zeros_mask_even, other=0.0)

        group_idx_odd = (k + offs_k_odd) // 128
        scales_offsets_odd = group_idx_odd[:, None] * stride_scales_k + offs_bn[None, :] * stride_scales_n
        scales_mask_odd = (group_idx_odd[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        scales_block_odd = tl.load(scales_ptr + scales_offsets_odd, mask=scales_mask_odd, other=1.0)

        zeros_offsets_odd = group_idx_odd[:, None] * stride_zeros_k + offs_bn[None, :] * stride_zeros_n
        zeros_mask_odd = (group_idx_odd[:, None] < (K // 128)) & (offs_bn[None, :] < N)
        zeros_block_odd = tl.load(zeros_ptr + zeros_offsets_odd, mask=zeros_mask_odd, other=0.0)

        b_even_dequant = (b_even - zeros_block_even) * scales_block_even
        b_odd_dequant = (b_odd - zeros_block_odd) * scales_block_odd

        accumulator = tl.dot(a_even, b_even_dequant, acc=accumulator)
        accumulator = tl.dot(a_odd, b_odd_dequant, acc=accumulator)

    c_offsets = offs_am[:, None] * stride_cm + offs_bn[None, :] * stride_cn
    c_mask = (offs_am[:, None] < M) & (offs_bn[None, :] < N)
    tl.store(c_ptr + c_offsets, accumulator.to(tl.bfloat16), mask=c_mask)


def w4a16_gemm(x, w_q, scales, zeros):
    M, K = x.shape
    _, N = w_q.shape
    c = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)
    
    grid = lambda meta: (
        triton.cdiv(M, meta['BLOCK_SIZE_M']) * triton.cdiv(N, meta['BLOCK_SIZE_N']),
    )
    
    if M == 1:
        # Match Shape 0 and Shape 3, or other M=1 decodes
        if N == 12288:
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 256
            num_warps = 4
            num_stages = 2
        elif N == 4096:
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 256
            num_warps = 8
            num_stages = 2
        else:
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 256
            num_warps = 4
            num_stages = 2
            
        w4a16_gemv_split_kernel[grid](
            x, w_q, scales, zeros, c,
            M, N, K,
            x.stride(0), x.stride(1),
            w_q.stride(0), w_q.stride(1),
            scales.stride(0), scales.stride(1),
            zeros.stride(0), zeros.stride(1),
            c.stride(0), c.stride(1),
            BLOCK_SIZE_M=1,
            BLOCK_SIZE_N=BLOCK_SIZE_N,
            BLOCK_SIZE_K=BLOCK_SIZE_K,
            GROUP_SIZE_M=1,
            num_warps=num_warps,
            num_stages=num_stages,
        )
    else:
        # Match prefills and speculative-decode-ish shapes
        if M == 32 and N == 12288 and K == 4096:
            BLOCK_SIZE_M = 32
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 128
            GROUP_SIZE_M = 4
            num_warps = 4
            num_stages = 3
        elif M == 256 and N == 12288 and K == 4096:
            BLOCK_SIZE_M = 128
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 64
            GROUP_SIZE_M = 16
            num_warps = 4
            num_stages = 2
        elif M == 16 and N == 14336 and K == 4096:
            BLOCK_SIZE_M = 16
            BLOCK_SIZE_N = 64
            BLOCK_SIZE_K = 128
            GROUP_SIZE_M = 4
            num_warps = 4
            num_stages = 3
        else:
            # General fallback heuristics for M > 1
            if M <= 16:
                BLOCK_SIZE_M = 16
                BLOCK_SIZE_N = 64
                BLOCK_SIZE_K = 128
                GROUP_SIZE_M = 4
                num_warps = 4
                num_stages = 3
            elif M <= 32:
                BLOCK_SIZE_M = 32
                BLOCK_SIZE_N = 64
                BLOCK_SIZE_K = 128
                GROUP_SIZE_M = 4
                num_warps = 4
                num_stages = 3
            else:
                BLOCK_SIZE_M = 128
                BLOCK_SIZE_N = 64
                BLOCK_SIZE_K = 64
                GROUP_SIZE_M = 16
                num_warps = 4
                num_stages = 2

        w4a16_gemm_interleave_kernel[grid](
            x, w_q, scales, zeros, c,
            M, N, K,
            x.stride(0), x.stride(1),
            w_q.stride(0), w_q.stride(1),
            scales.stride(0), scales.stride(1),
            zeros.stride(0), zeros.stride(1),
            c.stride(0), c.stride(1),
            BLOCK_SIZE_M=BLOCK_SIZE_M,
            BLOCK_SIZE_N=BLOCK_SIZE_N,
            BLOCK_SIZE_K=BLOCK_SIZE_K,
            GROUP_SIZE_M=GROUP_SIZE_M,
            num_warps=num_warps,
            num_stages=num_stages,
        )
    return c


class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size
        self.register_buffer("w_q", torch.empty((K // 2, N), dtype=torch.uint8))
        self.register_buffer("scales", torch.empty((n_groups, N), dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.empty((n_groups, N), dtype=torch.bfloat16))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return w4a16_gemm(x, self.w_q, self.scales, self.zeros)


def get_inputs():
    x = torch.randn(M, K, dtype=torch.bfloat16)
    return [x]


def get_init_inputs():
    return [M, N, K]
