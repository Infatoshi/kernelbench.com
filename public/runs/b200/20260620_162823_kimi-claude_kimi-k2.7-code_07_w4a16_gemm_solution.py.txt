"""Fused W4A16 weight-only quantized GEMM (AWQ/GPTQ-style asymmetric int4).

Kernel: transposed-layout Triton blocked matmul with on-the-fly int4 unpack
and per-group bf16 scale/zero-point dequant.

By computing the GEMM as y^T = W^T @ x^T, the large N dimension becomes the
M dimension of the tensor-core MMA, giving much better utilization on the
M=1 decode path.

The packed weight tensor is stored as (N, K // 2) after init so that the
memory stream inside the kernel is contiguous along the K reduction.
Triton block pointers are used for the main memory loads.

For small M we process two K-groups per loop iteration (K=256 dot) to amortize
loop overhead; for large M we use one group per iteration to stay within
shared-memory limits.

Layout:
  x:      (M, K)               bf16
  w_q:    (K // 2, N)          uint8   -- low nibble = even-K, high = odd-K
  scales: (K // 128, N)        bf16
  zeros:  (K // 128, N)        bf16
  out:    (M, N)               bf16

Dequant formula per (k, n):
  w_bf[k, n] = (unpack(w_q)[k, n] - zeros[k // 128, n]) * scales[k // 128, n]
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

GROUP_SIZE = 128


def _kernel_config(M: int, N: int, K: int) -> dict:
    """Return the best manually-swept config."""
    if M == 1:
        if N >= 8192:
            return {"BLOCK_M": 1, "BLOCK_N": 32, "num_warps": 2, "num_stages": 3, "groups_per_iter": 2}
        return {"BLOCK_M": 1, "BLOCK_N": 8, "num_warps": 2, "num_stages": 3, "groups_per_iter": 2}
    if M <= 16:
        return {"BLOCK_M": 16, "BLOCK_N": 64, "num_warps": 4, "num_stages": 3, "groups_per_iter": 2}
    if M <= 64:
        return {"BLOCK_M": 32, "BLOCK_N": 64, "num_warps": 8, "num_stages": 3, "groups_per_iter": 1}
    return {"BLOCK_M": 128, "BLOCK_N": 128, "num_warps": 8, "num_stages": 3, "groups_per_iter": 1}


@triton.jit
def w4a16_gemm_kernel_1g(
    x_ptr,
    w_q_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M,
    N,
    K,
    stride_xm,
    stride_xk,
    stride_wq_n,
    stride_wq_k,
    stride_sg,
    stride_sn,
    stride_zg,
    stride_zn,
    stride_om,
    stride_on,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    GROUPS: tl.constexpr,
):
    """One K-group per loop iteration (large M)."""
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)

    offs_m = pid_m * BLOCK_M
    offs_n = pid_n * BLOCK_N

    x_block_ptr = tl.make_block_ptr(
        base=x_ptr,
        shape=(M, K),
        strides=(stride_xm, stride_xk),
        offsets=(offs_m, 0),
        block_shape=(BLOCK_M, 128),
        order=(1, 0),
    )
    wq_block_ptr = tl.make_block_ptr(
        base=w_q_ptr,
        shape=(N, K // 2),
        strides=(stride_wq_n, stride_wq_k),
        offsets=(offs_n, 0),
        block_shape=(BLOCK_N, 64),
        order=(1, 0),
    )

    acc = tl.zeros((BLOCK_N, BLOCK_M), dtype=tl.float32)

    for g in tl.range(0, GROUPS):
        k0 = g * 128
        x_group = tl.load(tl.advance(x_block_ptr, (0, k0))).to(tl.bfloat16)
        x_t = tl.trans(x_group)

        w_packed_t = tl.load(tl.advance(wq_block_ptr, (0, k0 // 2)))
        w_lo = (w_packed_t & 0xF).to(tl.bfloat16)
        w_hi = ((w_packed_t >> 4) & 0xF).to(tl.bfloat16)
        w_t = tl.interleave(w_lo, w_hi)

        offs_n_local = offs_n + tl.arange(0, BLOCK_N)
        mask_n = offs_n_local < N
        scale = tl.load(scales_ptr + g * stride_sg + offs_n_local * stride_sn, mask=mask_n, other=0.0).to(tl.bfloat16)
        zero = tl.load(zeros_ptr + g * stride_zg + offs_n_local * stride_zn, mask=mask_n, other=0.0).to(tl.bfloat16)
        w_t = (w_t - zero[:, None]) * scale[:, None]

        acc += tl.dot(w_t, x_t)

    offs_m_local = offs_m + tl.arange(0, BLOCK_M)
    offs_n_local = offs_n + tl.arange(0, BLOCK_N)
    mask_m = offs_m_local < M
    mask_n = offs_n_local < N
    out_ptrs = out_ptr + offs_m_local[None, :] * stride_om + offs_n_local[:, None] * stride_on
    tl.store(out_ptrs, acc.to(tl.bfloat16), mask=mask_m[None, :] & mask_n[:, None])


@triton.jit
def w4a16_gemm_kernel_2g(
    x_ptr,
    w_q_ptr,
    scales_ptr,
    zeros_ptr,
    out_ptr,
    M,
    N,
    K,
    stride_xm,
    stride_xk,
    stride_wq_n,
    stride_wq_k,
    stride_sg,
    stride_sn,
    stride_zg,
    stride_zn,
    stride_om,
    stride_on,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    GROUPS: tl.constexpr,
):
    """Two K-groups per loop iteration (small M)."""
    pid_n = tl.program_id(0)
    pid_m = tl.program_id(1)

    offs_m = pid_m * BLOCK_M
    offs_n = pid_n * BLOCK_N

    x_block_ptr = tl.make_block_ptr(
        base=x_ptr,
        shape=(M, K),
        strides=(stride_xm, stride_xk),
        offsets=(offs_m, 0),
        block_shape=(BLOCK_M, 128),
        order=(1, 0),
    )
    wq_block_ptr = tl.make_block_ptr(
        base=w_q_ptr,
        shape=(N, K // 2),
        strides=(stride_wq_n, stride_wq_k),
        offsets=(offs_n, 0),
        block_shape=(BLOCK_N, 64),
        order=(1, 0),
    )

    acc = tl.zeros((BLOCK_N, BLOCK_M), dtype=tl.float32)

    for g in tl.range(0, GROUPS // 2):
        g0 = g * 2
        g1 = g0 + 1

        x0 = tl.load(tl.advance(x_block_ptr, (0, g0 * 128))).to(tl.bfloat16)
        x1 = tl.load(tl.advance(x_block_ptr, (0, g1 * 128))).to(tl.bfloat16)
        x_t_0 = tl.trans(x0)
        x_t_1 = tl.trans(x1)

        w0 = tl.load(tl.advance(wq_block_ptr, (0, g0 * 64)))
        w1 = tl.load(tl.advance(wq_block_ptr, (0, g1 * 64)))
        w_lo0 = (w0 & 0xF).to(tl.bfloat16)
        w_hi0 = ((w0 >> 4) & 0xF).to(tl.bfloat16)
        w_lo1 = (w1 & 0xF).to(tl.bfloat16)
        w_hi1 = ((w1 >> 4) & 0xF).to(tl.bfloat16)

        offs_n_local = offs_n + tl.arange(0, BLOCK_N)
        mask_n = offs_n_local < N
        s0 = tl.load(scales_ptr + g0 * stride_sg + offs_n_local * stride_sn, mask=mask_n, other=0.0).to(tl.bfloat16)
        z0 = tl.load(zeros_ptr + g0 * stride_zg + offs_n_local * stride_zn, mask=mask_n, other=0.0).to(tl.bfloat16)
        s1 = tl.load(scales_ptr + g1 * stride_sg + offs_n_local * stride_sn, mask=mask_n, other=0.0).to(tl.bfloat16)
        z1 = tl.load(zeros_ptr + g1 * stride_zg + offs_n_local * stride_zn, mask=mask_n, other=0.0).to(tl.bfloat16)

        wt0 = (tl.interleave(w_lo0, w_hi0) - z0[:, None]) * s0[:, None]
        wt1 = (tl.interleave(w_lo1, w_hi1) - z1[:, None]) * s1[:, None]

        wt_combined = tl.reshape(tl.join(wt0, wt1), (BLOCK_N, 256))
        xt0_t = tl.trans(x_t_0)
        xt1_t = tl.trans(x_t_1)
        xt_combined_t = tl.reshape(tl.join(xt0_t, xt1_t), (BLOCK_M, 256))
        xt_combined = tl.trans(xt_combined_t)

        acc += tl.dot(wt_combined, xt_combined)

    offs_m_local = offs_m + tl.arange(0, BLOCK_M)
    offs_n_local = offs_n + tl.arange(0, BLOCK_N)
    mask_m = offs_m_local < M
    mask_n = offs_n_local < N
    out_ptrs = out_ptr + offs_m_local[None, :] * stride_om + offs_n_local[:, None] * stride_on
    tl.store(out_ptrs, acc.to(tl.bfloat16), mask=mask_m[None, :] & mask_n[:, None])


def w4a16_gemm(x: torch.Tensor, w_q: torch.Tensor, scales: torch.Tensor, zeros: torch.Tensor) -> torch.Tensor:
    M, K = x.shape
    N = w_q.shape[0]
    assert w_q.shape[1] * 2 == K
    assert scales.shape == (K // 128, N)
    assert zeros.shape == scales.shape

    out = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)

    cfg = _kernel_config(M, N, K)
    BLOCK_M = cfg["BLOCK_M"]
    BLOCK_N = cfg["BLOCK_N"]
    num_warps = cfg["num_warps"]
    num_stages = cfg["num_stages"]
    groups_per_iter = cfg["groups_per_iter"]

    grid = (triton.cdiv(N, BLOCK_N), triton.cdiv(M, BLOCK_M))

    if groups_per_iter == 2:
        assert K // 128 % 2 == 0
        w4a16_gemm_kernel_2g[grid](
            x,
            w_q,
            scales,
            zeros,
            out,
            M,
            N,
            K,
            x.stride(0),
            x.stride(1),
            w_q.stride(0),
            w_q.stride(1),
            scales.stride(0),
            scales.stride(1),
            zeros.stride(0),
            zeros.stride(1),
            out.stride(0),
            out.stride(1),
            BLOCK_M=BLOCK_M,
            BLOCK_N=BLOCK_N,
            GROUPS=K // 128,
            num_warps=num_warps,
            num_stages=num_stages,
        )
    else:
        w4a16_gemm_kernel_1g[grid](
            x,
            w_q,
            scales,
            zeros,
            out,
            M,
            N,
            K,
            x.stride(0),
            x.stride(1),
            w_q.stride(0),
            w_q.stride(1),
            scales.stride(0),
            scales.stride(1),
            zeros.stride(0),
            zeros.stride(1),
            out.stride(0),
            out.stride(1),
            BLOCK_M=BLOCK_M,
            BLOCK_N=BLOCK_N,
            GROUPS=K // 128,
            num_warps=num_warps,
            num_stages=num_stages,
        )
    return out


class Model(nn.Module):
    """W4A16 GEMM: y = x @ dequant(w_q, scales, zeros)."""

    def __init__(self, M: int, N: int, K: int, group_size: int = GROUP_SIZE):
        super().__init__()
        assert K % group_size == 0
        assert K % 2 == 0
        self.M, self.N, self.K = M, N, K
        self.group_size = group_size
        n_groups = K // group_size

        # w_q is stored transposed as (N, K // 2) for contiguous K streaming.
        self.register_buffer("w_q", torch.empty((N, K // 2), dtype=torch.uint8))
        self.register_buffer("scales", torch.empty((n_groups, N), dtype=torch.bfloat16))
        self.register_buffer("zeros", torch.empty((n_groups, N), dtype=torch.bfloat16))

    def _load_from_state_dict(self, state_dict, prefix, local_metadata, strict,
                              missing_keys, unexpected_keys, error_msgs):
        """Transpose the incoming (K//2, N) packed weight into our (N, K//2) layout."""
        key = prefix + "w_q"
        if key in state_dict:
            w = state_dict[key]
            if w.dim() == 2 and w.shape[0] * 2 == self.K and w.shape[1] == self.N:
                state_dict[key] = w.t().contiguous()
        super()._load_from_state_dict(
            state_dict, prefix, local_metadata, strict,
            missing_keys, unexpected_keys, error_msgs
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return w4a16_gemm(x, self.w_q, self.scales, self.zeros)


# Shims used by check.py / benchmark.py to build the model for the current shape.
M = 1
N = 12288
K = 4096


def get_inputs():
    x = torch.randn(M, K, dtype=torch.bfloat16)
    return [x]


def get_init_inputs():
    return [M, N, K]
