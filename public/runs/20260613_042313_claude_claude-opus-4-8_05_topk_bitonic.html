+
+Memory-bound top-k. Strategy: single-pass streaming bitonic top-k. Each block
+keeps a shared-memory buffer of size BUF (= 2*nextpow2(k), >=64); it streams the
+input in tiles of BUF/2, sorting BUF elements descending each tile and keeping
+the best BUF/2. Rows are split across multiple blocks (phase 1); a small second
+kernel merges the per-block partial top-k (phase 2).
+"""

+// Bitonic sort of BUF elements DESCENDING, blockDim.x == BUF threads.
+__device__ __forceinline__ void bitonic_desc(float* sv, int* si, int BUF) {
+    const int tid = threadIdx.x;
+    for (int k = 2; k <= BUF; k <<= 1) {
+        for (int j = k >> 1; j > 0; j >>= 1) {
+            int ixj = tid ^ j;
+            if (ixj > tid) {
+                bool up = ((tid & k) == 0);
+                float a = sv[tid], b = sv[ixj];
+                // descending: smallest pushed toward high index
+                bool swap = up ? (a < b) : (a > b);
+                if (swap) {
+                    sv[tid] = b; sv[ixj] = a;
+                    int ta = si[tid]; si[tid] = si[ixj]; si[ixj] = ta;
+                }
+            }
+            __syncthreads();
+        }
+    }
+}
