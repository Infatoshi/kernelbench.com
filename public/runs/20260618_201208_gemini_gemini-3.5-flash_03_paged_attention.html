import triton.language as tl
import math

@triton.jit
def paged_attn_decode_kernel(
    Q_ptr,             # (batch, num_heads, head_dim)
    KV_ptr,            # (num_blocks, page_size, num_kv_heads, head_dim * 2)
    BlockTable_ptr,    # (batch, max_blocks)
    SeqLens_ptr,       # (batch,)
    Out_ptr,           # (batch, num_heads, head_dim)
    stride_q_b, stride_q_h, stride_q_d,
    stride_kv_block, stride_kv_page, stride_kv_head, stride_kv_dim,
    stride_bt_b, stride_bt_idx,
    stride_out_b, stride_out_h, stride_out_d,
    scale,
    num_heads, num_kv_heads, head_dim, max_blocks, page_size,
    GROUP_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    # Each program handles one batch element and one head
    pid = tl.program_id(0)
    b = pid // num_heads
    h = pid % num_heads
    h_kv = h // GROUP_SIZE

    # Load sequence length
    L = tl.load(SeqLens_ptr + b)
    if L <= 0:
        return

    # Pointers for Q
    offs_d = tl.arange(0, BLOCK_D)
    q_ptrs = Q_ptr + b * stride_q_b + h * stride_q_h + offs_d * stride_q_d
    q_mask = offs_d < head_dim
    q = tl.load(q_ptrs, mask=q_mask, other=0.0)

    # Initialize online softmax variables
    m_i = -1e20
    l_i = 0.0
    acc = tl.zeros((BLOCK_D,), dtype=tl.float32)

    # Loop over the sequence in blocks of BLOCK_N
    num_blocks_seq = tl.cdiv(L, BLOCK_N)
    for start_block in range(0, num_blocks_seq):
        start_t = start_block * BLOCK_N
        offs_t = tl.arange(0, BLOCK_N)
        t = start_t + offs_t
        t_mask = t < L

        # Find block table entries
        page_idx = t // page_size
        page_offset = t % page_size

        # Load block IDs from block table
        bt_ptrs = BlockTable_ptr + b * stride_bt_b + page_idx * stride_bt_idx
        # Mask out-of-bounds page indices
        # max_pages is max_blocks, so we mask page_idx < max_blocks
        bt_mask = t_mask & (page_idx < max_blocks)
        block_ids = tl.load(bt_ptrs, mask=bt_mask, other=0)

        # Compute 2D pointers for K and V
        # block_ids is (BLOCK_N,), page_offset is (BLOCK_N,)
        # We expand to 2D
        block_ids_2d = tl.expand_dims(block_ids, 1)
        page_offset_2d = tl.expand_dims(page_offset, 1)
        offs_d_2d = tl.expand_dims(offs_d, 0)

        # Apply mask

        m_block = tl.max(qk, axis=0)
        m_next = tl.maximum(m_i, m_block)

        p = tl.exp(qk - m_next)
