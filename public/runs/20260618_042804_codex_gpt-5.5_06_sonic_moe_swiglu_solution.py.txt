from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _moe_swiglu_kernel(
    x,
    offsets,
    w_gate,
    w_up,
    out,
    H: tl.constexpr,
    I: tl.constexpr,
    MAX_M: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    m_block = tl.program_id(0)
    n_block = tl.program_id(1)
    expert = tl.program_id(2)

    start = tl.load(offsets + expert)
    end = tl.load(offsets + expert + 1)
    rows = start + m_block * BLOCK_M + tl.arange(0, BLOCK_M)
    cols = n_block * BLOCK_N + tl.arange(0, BLOCK_N)
    ks = tl.arange(0, BLOCK_K)

    acc_gate = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)
    acc_up = tl.zeros((BLOCK_M, BLOCK_N), tl.float32)

    for k0 in range(0, H, BLOCK_K):
        k_idxs = k0 + ks
        a = tl.load(
            x + rows[:, None] * H + k_idxs[None, :],
            mask=(rows[:, None] < end) & (k_idxs[None, :] < H),
            other=0.0,
        )
        wg = tl.load(
            w_gate + expert * H * I + k_idxs[:, None] * I + cols[None, :],
            mask=(k_idxs[:, None] < H) & (cols[None, :] < I),
            other=0.0,
        )
        wu = tl.load(
            w_up + expert * H * I + k_idxs[:, None] * I + cols[None, :],
            mask=(k_idxs[:, None] < H) & (cols[None, :] < I),
            other=0.0,
        )
        acc_gate += tl.dot(a, wg, out_dtype=tl.float32)
        acc_up += tl.dot(a, wu, out_dtype=tl.float32)

    gated = (acc_gate * tl.sigmoid(acc_gate)) * acc_up
    tl.store(
        out + rows[:, None] * I + cols[None, :],
        gated,
        mask=(rows[:, None] < end) & (cols[None, :] < I),
    )

class Model(nn.Module):
    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def _tile_config(self) -> tuple[int, int, int, int, int]:
        if self.H == 2048 and self.I == 1024:
            return 128, 128, 64, 8, 3
        if self.H == 2048 and self.I == 4096:
            return 64, 256, 64, 8, 3
        return 128, 128, 64, 8, 3

    def forward(
        self,
        hidden_states: torch.Tensor,
        expert_offsets: torch.Tensor,
    ) -> torch.Tensor:
        T_perm = hidden_states.shape[0]
        out = torch.empty((T_perm, self.I), dtype=torch.bfloat16, device=hidden_states.device)
        block_m, block_n, block_k, num_warps, num_stages = self._tile_config()
        max_m = triton.cdiv(self.T_total * self.K, self.E)
        grid = (
            triton.cdiv(max_m, block_m),
            triton.cdiv(self.I, block_n),
            self.E,
        )
        _moe_swiglu_kernel[grid](
            hidden_states,
            expert_offsets,
            self.W_gate,
            self.W_up,
            out,
            self.H,
            self.I,
            max_m,
            BLOCK_M=block_m,
            BLOCK_N=block_n,
            BLOCK_K=block_k,
            num_warps=num_warps,
            num_stages=num_stages,
        )
        return out


T_total = 32768
H = 4096
I = 1536  # noqa: E741
E = 128
K = 8


def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]


def get_init_inputs():
    return [T_total, H, I, E, K]
