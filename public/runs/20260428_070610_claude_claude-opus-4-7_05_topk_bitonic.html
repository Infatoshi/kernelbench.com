"""TopK custom kernel via load_inline (v5).

Strategy:
  - Stage 1: full bitonic sort of small chunks in shared memory; take
    chunk-local top-K. Stage 1 has many small chunks for SM parallelism.
  - Stage 2: each stage-1 chunk emits a sorted-descending run of K
    elements. We merge G sorted runs of length K into the global top-K
    using a tournament-style argmax: per iteration, one thread per run
    holds its current head value; warp+block argmax picks the run with
    the largest head; that run advances by 1. K iterations -> top-K.
    This is O(K * log G) instead of bitonic-sort's O(G*K * log^2(G*K)).
  - Shape (128, 4096, 1) is k=1 argmax — dedicated max-reduction kernel.
"""

// Block-wide bitonic sort, ascending. N must be a power of 2.
template<int N, int THREADS>
__device__ __forceinline__ void bitonic_sort_block(float* svals, int* sidxs) {
    #pragma unroll
    for (int sz = 2; sz <= N; sz <<= 1) {
        #pragma unroll
        for (int j = sz >> 1; j > 0; j >>= 1) {
            #pragma unroll
            for (int t0 = 0; t0 < N; t0 += THREADS) {
                int t = t0 + threadIdx.x;
                if (t < N) {
                    int p = t ^ j;
                    if (p > t) {
                        bool asc = (t & sz) == 0;
                        float vt = svals[t], vp = svals[p];
                        bool sw = asc ? (vt > vp) : (vt < vp);
                        if (sw) {
                            svals[t] = vp; svals[p] = vt;
                            int it = sidxs[t], ip = sidxs[p];
                            sidxs[t] = ip; sidxs[p] = it;
                        }
                    }
                }
            }
            __syncthreads();
        }
    }
}

// Stage 1 / single-stage chunk kernel: load CHUNK_SIZE into shared (pad
// to N_PAD with -inf), bitonic sort, write top-K to (out_v, out_i).
template<int K, int N_PAD, int THREADS, int WRITE_INT64>
__global__ void topk_chunk_kernel(
    const float* __restrict__ x,
    void* __restrict__ out_v,
    void* __restrict__ out_i,
    int n,
    int chunk_size,
    int chunks_per_row)
{
    extern __shared__ unsigned char smem_raw[];
    float* svals = reinterpret_cast<float*>(smem_raw);
    int* sidxs = reinterpret_cast<int*>(smem_raw + N_PAD * sizeof(float));

    int row = blockIdx.y;
    int blk = blockIdx.x;
    int chunk_start = blk * chunk_size;
    int chunk_end = chunk_start + chunk_size;
    if (chunk_end > n) chunk_end = n;
    const float* xrow = x + row * (size_t)n;

    #pragma unroll
    for (int i0 = 0; i0 < N_PAD; i0 += THREADS) {
        int i = i0 + threadIdx.x;
        if (i < N_PAD) {
            int gi = chunk_start + i;
            if (gi < chunk_end) {
                svals[i] = xrow[gi];
                sidxs[i] = gi;
            } else {
                svals[i] = NEG_INF;
                sidxs[i] = -1;
            }
        }
    }
    __syncthreads();

    bitonic_sort_block<N_PAD, THREADS>(svals, sidxs);

    int out_off = (row * chunks_per_row + blk) * K;
    if (WRITE_INT64) {
        float* ov = reinterpret_cast<float*>(out_v);
        int64_t* oi = reinterpret_cast<int64_t*>(out_i);
        for (int q = threadIdx.x; q < K; q += THREADS) {
            ov[out_off + q] = svals[N_PAD - 1 - q];
            oi[out_off + q] = (int64_t)sidxs[N_PAD - 1 - q];
        }
    } else {
        float* ov = reinterpret_cast<float*>(out_v);
        int* oi = reinterpret_cast<int*>(out_i);
        for (int q = threadIdx.x; q < K; q += THREADS) {
            ov[out_off + q] = svals[N_PAD - 1 - q];
            oi[out_off + q] = sidxs[N_PAD - 1 - q];
        }
    }
}

// Stage 2 (bitonic): merge candidates (already sorted in chunks of K)
// via full bitonic sort over N_PAD candidates.
template<int K_OUT, int N_PAD, int THREADS>
__global__ void topk_stage2_bitonic_kernel(
    const float* __restrict__ in_v,
    const int* __restrict__ in_i,
    float* __restrict__ vals,
    int64_t* __restrict__ idxs,
    int candidates_per_row)
{
    extern __shared__ unsigned char smem_raw[];
    float* svals = reinterpret_cast<float*>(smem_raw);
    int* sidxs = reinterpret_cast<int*>(smem_raw + N_PAD * sizeof(float));

    int row = blockIdx.x;
    const float* in_v_row = in_v + row * candidates_per_row;
    const int* in_i_row = in_i + row * candidates_per_row;

    #pragma unroll
    for (int i0 = 0; i0 < N_PAD; i0 += THREADS) {
        int i = i0 + threadIdx.x;
        if (i < N_PAD) {
            if (i < candidates_per_row) {
                svals[i] = in_v_row[i];
                sidxs[i] = in_i_row[i];
            } else {
                svals[i] = NEG_INF;
                sidxs[i] = -1;
            }
        }
    }
    __syncthreads();

    bitonic_sort_block<N_PAD, THREADS>(svals, sidxs);

    for (int q = threadIdx.x; q < K_OUT; q += THREADS) {
        vals[row * K_OUT + q] = svals[N_PAD - 1 - q];
        idxs[row * K_OUT + q] = (int64_t)sidxs[N_PAD - 1 - q];
    }
}

// Stage 2 (k-way merge tournament): NUM_RUNS sorted-desc runs of RUN_LEN
// each per row. THREADS == NUM_RUNS (one thread per run). K iterations of
// argmax + advance.
template<int K_OUT, int NUM_RUNS, int RUN_LEN, int THREADS>
__global__ void topk_stage2_merge_kernel(
    const float* __restrict__ in_v,
    const int* __restrict__ in_i,
    float* __restrict__ vals,
    int64_t* __restrict__ idxs)
{
    int row = blockIdx.x;
    int tid = threadIdx.x;

    // Each thread handles one run.
    int run_off = row * NUM_RUNS * RUN_LEN + tid * RUN_LEN;

    float head_v = (tid < NUM_RUNS) ? in_v[run_off] : NEG_INF;
    int head_i = (tid < NUM_RUNS) ? in_i[run_off] : -1;
    int head_ptr = 0;

    __shared__ float smax[(THREADS + 31) / 32];
    __shared__ int sidx[(THREADS + 31) / 32];
    __shared__ int s_winner;
    __shared__ float s_winner_v;
    __shared__ int s_winner_i;

    constexpr int WARP_SIZE = 32;
    constexpr int NUM_WARPS = THREADS / WARP_SIZE;

    for (int k_iter = 0; k_iter < K_OUT; ++k_iter) {
        // Warp-level argmax.
        unsigned mask = 0xFFFFFFFFu;
        float my_v = head_v;
        int my_t = tid;
        #pragma unroll
        for (int s = WARP_SIZE / 2; s > 0; s >>= 1) {
            float ov = __shfl_xor_sync(mask, my_v, s);
            int ot = __shfl_xor_sync(mask, my_t, s);
            // Tie-break: prefer smaller thread index (deterministic).
            bool take = (ov > my_v) || (ov == my_v && ot < my_t);
            if (take) { my_v = ov; my_t = ot; }
        }
        int lane = tid & (WARP_SIZE - 1);
        int warp = tid / WARP_SIZE;
        if (lane == 0) { smax[warp] = my_v; sidx[warp] = my_t; }
        __syncthreads();

        if (warp == 0) {
            float v = (lane < NUM_WARPS) ? smax[lane] : NEG_INF;
            int t = (lane < NUM_WARPS) ? sidx[lane] : -1;
            #pragma unroll
            for (int s = NUM_WARPS / 2; s > 0; s >>= 1) {
                float ov = __shfl_xor_sync(mask, v, s);
                int ot = __shfl_xor_sync(mask, t, s);
                bool take = (ov > v) || (ov == v && ot < t);
                if (take) { v = ov; t = ot; }
            }
            if (lane == 0) {
                s_winner = t;
                s_winner_v = v;
                // We need the index too — the winner thread will write it.
                vals[row * K_OUT + k_iter] = v;
            }
        }
        __syncthreads();

        int w = s_winner;
        if (tid == w) {
            idxs[row * K_OUT + k_iter] = (int64_t)head_i;
            head_ptr++;
            if (head_ptr < RUN_LEN) {
                head_v = in_v[run_off + head_ptr];
                head_i = in_i[run_off + head_ptr];
            } else {
                head_v = NEG_INF;
                head_i = -1;
            }
        }
        __syncthreads();
    }
}

// Argmax kernel for k=1.
template<int N, int THREADS>
__global__ void argmax_kernel(
    const float* __restrict__ x,
    float* __restrict__ vals,
    int64_t* __restrict__ idxs,
    int n)
{
    int row = blockIdx.x;
    const float* xrow = x + row * (size_t)n;

    float my_v = NEG_INF;
    int my_i = -1;
    for (int i = threadIdx.x; i < n; i += THREADS) {
        float v = xrow[i];
        if (v > my_v) { my_v = v; my_i = i; }
    }

    unsigned mask = 0xFFFFFFFFu;
    #pragma unroll
    for (int s = 16; s > 0; s >>= 1) {
        float ov = __shfl_xor_sync(mask, my_v, s);
        int oi = __shfl_xor_sync(mask, my_i, s);
        if (ov > my_v || (ov == my_v && oi < my_i)) { my_v = ov; my_i = oi; }
    }

    __shared__ float wv[THREADS / 32];
    __shared__ int wi[THREADS / 32];
    int lane = threadIdx.x & 31;
    int warp = threadIdx.x >> 5;
    if (lane == 0) { wv[warp] = my_v; wi[warp] = my_i; }
    __syncthreads();

    if (warp == 0) {
        my_v = (lane < THREADS / 32) ? wv[lane] : NEG_INF;
        my_i = (lane < THREADS / 32) ? wi[lane] : -1;
        #pragma unroll
        for (int s = 16; s > 0; s >>= 1) {
            float ov = __shfl_xor_sync(mask, my_v, s);
            int oi = __shfl_xor_sync(mask, my_i, s);
            if (ov > my_v || (ov == my_v && oi < my_i)) { my_v = ov; my_i = oi; }
        }
        if (lane == 0) {
            vals[row] = my_v;
            idxs[row] = (int64_t)my_i;
        }
    }
}

template<int K, int N_PAD, int THREADS>
inline void launch_chunk_int32(torch::Tensor x, torch::Tensor v_scratch,
                                torch::Tensor i_scratch,
                                int n, int chunk_size, int chunks_per_row,
                                int batch, cudaStream_t stream)
{
    size_t sm = (size_t)N_PAD * (sizeof(float) + sizeof(int));

"""Roofline math: achieved TFLOPS / GB/s, peak fraction.

FLOPS and bytes formulas come from the problem's `problem.yaml`. They are the
dense-equivalent algorithmic work; kernels that skip work (sparsity, early
exit) are still measured against the dense cost.
"""
from __future__ import annotations


def compute_tflops(flops: float, time_ms: float) -> float:
    """flops per invocation and wall time in ms -> TFLOPS."""
    if time_ms <= 0:
        return 0.0
    return flops / (time_ms * 1e-3) / 1e12


def compute_gbps(bytes_moved: float, time_ms: float) -> float:
    """bytes per invocation and wall time in ms -> GB/s."""
    if time_ms <= 0:
        return 0.0
    return bytes_moved / (time_ms * 1e-3) / 1e9


def peak_fraction(
    achieved: float,
    peak: float,
) -> float:
    """Return achieved / peak, clamped to [0, infinity)."""
    if peak <= 0:
        return 0.0
    return max(0.0, achieved / peak)
