_CUDA_SRC = r"""
#include <float.h>

#define HEAP_K 8
#define BS 1024

__device__ __forceinline__ void
heapi(float* v, int64_t* i, float val, int64_t idx, int& sz) {
    int p = sz++;
    v[p] = val; i[p] = idx;
    while (p > 0) {
        int q = (p - 1) >> 1;
        if (v[p] > v[q]) {
            float t1 = v[p]; v[p] = v[q]; v[q] = t1;
            int64_t t2 = i[p]; i[p] = i[q]; i[q] = t2;
            p = q;
        } else break;
    }
}

__device__ __forceinline__ void
heapsd(float* v, int64_t* i, int sz) {
    int p = 0;
    while (true) {
        int l = 2*p+1, r = 2*p+2, best = p;
        if (l < sz && v[l] < v[best]) best = l;
        if (r < sz && v[r] < v[best]) best = r;
        if (best == p) break;
        float t1 = v[p]; v[p] = v[best]; v[best] = t1;
        int64_t t2 = i[p]; i[p] = i[best]; i[best] = t2;
        p = best;
    }
}

extern "C" __global__ void
topk_kernel(const float* __restrict__ X,
            float*       __restrict__ OutV,
            int64_t*     __restrict__ OutI,
            int batch, int n, int k)
{
    int row = blockIdx.x;
    int tid = threadIdx.x;
    if (row >= batch) return;

    extern __shared__ char sbuf[];
    // Layout: [sv: 8192 floats][si: 8192 int64][rv: 1024 floats...pad][ri: 1024 int64]
    // sv:     0..32767      bytes
    // si:     32768..98303  bytes
    // rv:     98304..102399 bytes  (aligned to 8)
    // ri:     102400..110591 bytes
    float*   sv = (float*)sbuf;
    int64_t* si = (int64_t*)(sbuf + 8192 * 4);
    float*   rv = (float*)(sbuf + (8192 * 4 + 8192 * 8));
    int64_t* ri = (int64_t*)(sbuf + ((8192 * 4 + 8192 * 8 + 1024 * 4 + 7) & ~7));

    // Phase 1: each thread scans block-stride tile, maintains top-HEAP_K
    float lv[HEAP_K]; int64_t li[HEAP_K]; int lsz = 0;
    for (int i = tid; i < n; i += BS) {
        float v = X[row * n + i];
        if (lsz < HEAP_K) {
            heapi(lv, li, v, i, lsz);
        } else if (v > lv[0]) {
            lv[0] = v; li[0] = i;
            heapsd(lv, li, lsz);
        }
    }
    for (int p = lsz; p < HEAP_K; p++) {
        lv[p] = -FLT_MAX; li[p] = -1;
    }

    int gbase = tid * HEAP_K;
    for (int p = 0; p < HEAP_K; p++) {
        sv[gbase + p] = lv[p];
        si[gbase + p] = li[p];
    }
    __syncthreads();

    int total = BS * HEAP_K;

    // Phase 2: k-pass max extraction
    for (int kk = 0; kk < k; kk++) {
        float mymx = -FLT_MAX;
        int mymi = 0;
        for (int j = tid; j < total; j += BS) {
            if (sv[j] > mymx) { mymx = sv[j]; mymi = j; }
        }
        rv[tid] = mymx;
        ri[tid] = mymi;
        __syncthreads();

        for (int off = BS / 2; off > 0; off >>= 1) {
            __syncthreads();
            if (tid < off) {
                if (rv[tid + off] > rv[tid]) {
                    rv[tid] = rv[tid + off];
                    ri[tid] = ri[tid + off];
                }
            }
        }
        __syncthreads();

        float wval = rv[0];
        int wslot = (int)ri[0];

        if (tid == 0) {
            OutV[row * k + kk] = wval;
            OutI[row * k + kk] = si[wslot];
        }

        if (tid == wslot % BS) {
            sv[wslot] = -FLT_MAX;
        }
        __syncthreads();
    }
}

topk_launch(const float* X, float* OutV, int64_t* OutI,
            int batch, int n, int k, cudaStream_t stream) {
