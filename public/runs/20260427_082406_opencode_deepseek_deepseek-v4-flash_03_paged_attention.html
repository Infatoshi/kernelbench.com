"""Paged attention decode — Triton kernel, (B, H) grid for max parallelism.

Single-query decode: each batch element has a query (num_heads, head_dim) and
attends over a paged KV cache. Grid is (batch, num_heads). Each program handles
one query head, loads its KV data via page indirection, and computes attention
with online softmax in fp32.
"""

import math

import torch
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_attn_kernel(
    q_ptr, kv_ptr, bt_ptr, sl_ptr, out_ptr,
    stride_qb, stride_qh,
    stride_kvb, stride_kvps, stride_kvkh,
    stride_btb,
    stride_ob, stride_oh,
    scale,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
):
    batch = tl.program_id(0)
    head = tl.program_id(1)
    kv_head = head // GROUP_SIZE

    offs_d = tl.arange(0, HEAD_DIM)

    # Load query as bf16 once, reuse across all pages via Tensor-Core dot
    q_bf16 = tl.load(
        q_ptr + batch * stride_qb + head * stride_qh + offs_d
    )

    L = tl.load(sl_ptr + batch)
    num_pages = (L + PAGE_SIZE - 1) // PAGE_SIZE

    offs_t = tl.arange(0, PAGE_SIZE)

    # Online-softmax state across all pages
    m_i = -float('inf')
    d_i = 0.0
    acc = tl.zeros([HEAD_DIM], dtype=tl.float32)

    for p in range(num_pages):
        block_id = tl.load(bt_ptr + batch * stride_btb + p)
        remaining = L - p * PAGE_SIZE
        tokens_in_page = tl.where(remaining > PAGE_SIZE, PAGE_SIZE, remaining)
        mask = offs_t < tokens_in_page

        kv_base = kv_ptr + block_id * stride_kvb + kv_head * stride_kvkh

        # Load K and V as fp32 for elementwise dot
        k_fp32 = tl.load(
            kv_base + offs_t[:, None] * stride_kvps + offs_d[None, :],
            mask=mask[:, None],
        ).to(tl.float32)
        v_fp32 = tl.load(
            kv_base + offs_t[:, None] * stride_kvps + offs_d[None, :] + HEAD_DIM,
            mask=mask[:, None],
        ).to(tl.float32)

        # Elementwise dot, fp32 accumulate
        scores = tl.sum(q_bf16.to(tl.float32)[None, :] * k_fp32, axis=1) * scale
        scores = tl.where(mask, scores, float('-inf'))

        m_block = tl.max(scores, 0)
        m_new = tl.maximum(m_i, m_block)

        alpha = tl.exp(m_i - m_new)
        exp_scores = tl.exp(scores - m_new)
        block_d = tl.sum(exp_scores)

        acc = acc * alpha + tl.sum(exp_scores[:, None] * v_fp32, axis=0)
        d_i = d_i * alpha + block_d
        m_i = m_new

    out = acc / d_i
    tl.store(
        out_ptr + batch * stride_ob + head * stride_oh + offs_d,
        out.to(tl.bfloat16),
    )


class Model(torch.nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer(
            "_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        P = self.page_size
        G = self.group_size

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        grid = (B, H)
        _paged_attn_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2),
            block_table.stride(0),
            out.stride(0), out.stride(1),
            self.scale,
            HEAD_DIM=D,
            PAGE_SIZE=P,
            GROUP_SIZE=G,
            num_warps=2,
        )

        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE
    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)
    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1
    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(
        B, pages_per_seq
    ).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)
    return [query, kv_cache, block_table, seq_lens]
