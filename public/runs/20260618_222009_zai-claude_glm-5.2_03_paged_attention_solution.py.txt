"""Paged-attention decode kernel for B200 (SM100).

Memory-bound single-query decode. Strategy:
  * Split-K FlashDecoding: each (batch, kv_head_group, seq_split) CTA streams
    its chunk of the KV cache exactly once, sharing K/V loads across the G
    query heads in the GQA group (bandwidth-optimal for GQA).
  * Online (FlashAttention) softmax so we never materialize the full score
    matrix; per-split partials (unnormalized O + logsumexp) go to a temp.
  * A tiny reduction kernel merges the num_splits partials per (batch, head).
  * Everything is captured in a CUDA graph keyed on input data_ptr()s, so the
    repeated-call timed loop pays ~0 launch overhead (decode kernels are
    3-45us; a Python/Triton launch alone is ~9us).
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


# --------------------------------------------------------------------------- #
# Kernels
# --------------------------------------------------------------------------- #
@triton.jit
def _fwd_decode_kernel(
    # Pointers
    Q_ptr,            # (B, H, D) bf16
    KV_ptr,           # (num_blocks, P, Hkv, 2D) bf16
    BT_ptr,           # (B, max_blocks) int32
    SL_ptr,           # (B,) int32
    O_part_ptr,       # (B, H, NUM_SPLITS, D) fp32  -- unnormalized partial O
    LSE_part_ptr,     # (B, H, NUM_SPLITS) fp32
    # Strides (in elements)
    stride_q_b, stride_q_h,
    stride_kv_blk, stride_kv_p, stride_kv_hkv,
    stride_bt_b,
    stride_o_b, stride_o_h, stride_o_s,
    stride_lse_b, stride_lse_h,
    # Scalars
    scale,
    # Constexprs
    D: tl.constexpr,
    PAD_G: tl.constexpr,       # padded group size (>= G, power of two, >=16)
    G: tl.constexpr,           # real group size
    P: tl.constexpr,           # page size
    Hkv: tl.constexpr,
    PAGES_PER_SPLIT: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)
    pid_s = tl.program_id(2)

    L = tl.load(SL_ptr + pid_b)
    total_pages = (L + P - 1) // P
    page_start = pid_s * PAGES_PER_SPLIT
    page_end = tl.minimum(page_start + PAGES_PER_SPLIT, total_pages)

    # ---- load Q tile (PAD_G, D); rows >= G zeroed ----
    offs_g = tl.arange(0, PAD_G)
    offs_d = tl.arange(0, D)
    offs_p = tl.arange(0, P)
    q_base = Q_ptr + pid_b * stride_q_b + (pid_kv * G) * stride_q_h
    q_ptrs = q_base + offs_g[:, None] * stride_q_h + offs_d[None, :]
    q = tl.load(q_ptrs, mask=offs_g[:, None] < G, other=0.0)   # (PAD_G, D) bf16

    # ---- running online-softmax state (per padded head row) ----
    m_i = tl.full([PAD_G], -float("inf"), dtype=tl.float32)
    l_i = tl.zeros([PAD_G], dtype=tl.float32)
    acc = tl.zeros([PAD_G, D], dtype=tl.float32)

    kv_base = KV_ptr + pid_kv * stride_kv_hkv
    NEG_INF = -float("inf")

    for j in tl.range(PAGES_PER_SPLIT):
        p = page_start + j
        in_range = p < page_end
        # block id (guard the load; other=0 -> block 0 always exists)
        block_id = tl.load(BT_ptr + pid_b * stride_bt_b + p,
                           mask=in_range, other=0)
        # token global indices for this page
        tok = p * P + offs_p                       # (P,)
        valid_tok = tok < L                        # (P,)
        blk_ptr = KV_ptr + block_id.to(tl.int64) * stride_kv_blk
        k_ptrs = (blk_ptr + offs_p[:, None] * stride_kv_p
                  + pid_kv * stride_kv_hkv + offs_d[None, :])
        v_ptrs = (blk_ptr + offs_p[:, None] * stride_kv_p
                  + pid_kv * stride_kv_hkv + D + offs_d[None, :])
        k = tl.load(k_ptrs, mask=valid_tok[:, None], other=0.0)   # (P, D)
        v = tl.load(v_ptrs, mask=valid_tok[:, None], other=0.0)   # (P, D)

        # scores = (q @ k^T) * scale  : (PAD_G, P)  bf16 x bf16 -> fp32
        scores = tl.dot(q, tl.trans(k)) * scale    # (PAD_G, P)
        scores = tl.where(valid_tok[None, :], scores, NEG_INF)

        m_curr = tl.max(scores, axis=1)            # (PAD_G,)
        m_new = tl.maximum(m_i, m_curr)
        alpha = tl.exp(m_i - m_new)
        p_exp = tl.exp(scores - m_new[:, None])    # (PAD_G, P)
        l_i = l_i * alpha + tl.sum(p_exp, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p_exp.to(v.dtype), v)
        m_i = m_new

    lse = tl.where(l_i > 0.0, m_i + tl.log(l_i), NEG_INF)
    inv_l = tl.where(l_i > 0.0, 1.0 / l_i, 0.0)
    o_norm = acc * inv_l[:, None]                   # per-split normalized O

    # ---- write partials for the G real heads ----
    head_off = pid_kv * G + offs_g                 # (PAD_G,)
    o_ptrs = (O_part_ptr + pid_b * stride_o_b + head_off[:, None] * stride_o_h
              + pid_s * stride_o_s + offs_d[None, :])
    lse_ptrs = (LSE_part_ptr + pid_b * stride_lse_b
                + head_off * stride_lse_h + pid_s)
    tl.store(o_ptrs, o_norm, mask=offs_g[:, None] < G)
    tl.store(lse_ptrs, lse, mask=offs_g < G)


@triton.jit
def _reduce_kernel(
    O_part_ptr,       # (B, H, NUM_SPLITS, D) fp32
    LSE_part_ptr,     # (B, H, NUM_SPLITS) fp32
    O_ptr,            # (B, H, D) bf16 out
    stride_o_b, stride_o_h, stride_o_s,
    stride_lse_b, stride_lse_h,
    stride_ob, stride_oh,
    D: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    BLOCK_S: tl.constexpr,       # next pow2 >= NUM_SPLITS
    BLOCK_D: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)

    offs_s = tl.arange(0, BLOCK_S)
    offs_d = tl.arange(0, BLOCK_D)
    valid_s = offs_s < NUM_SPLITS

    lse_ptrs = (LSE_part_ptr + pid_b * stride_lse_b
                + pid_h * stride_lse_h + offs_s)
    NEG_INF = -float("inf")
    lse = tl.load(lse_ptrs, mask=valid_s, other=NEG_INF)     # (BLOCK_S,)
    m = tl.max(lse, axis=0)
    alpha = tl.exp(lse - m)
    alpha = tl.where(valid_s, alpha, 0.0)
    denom = tl.sum(alpha, axis=0)

    o_part_ptrs = (O_part_ptr + pid_b * stride_o_b + pid_h * stride_o_h
                   + offs_s[:, None] * stride_o_s + offs_d[None, :])
    o_part = tl.load(o_part_ptrs,
                     mask=valid_s[:, None] & (offs_d[None, :] < D),
                     other=0.0)                             # (BLOCK_S, BLOCK_D)
    o = tl.sum(alpha[:, None] * o_part, axis=0) / denom      # (BLOCK_D,)
    o = tl.where(offs_d < D, o, 0.0)

    out_ptrs = O_ptr + pid_b * stride_ob + pid_h * stride_oh + offs_d
    tl.store(out_ptrs, o.to(O_ptr.dtype.element_ty), mask=offs_d < D)


# --------------------------------------------------------------------------- #
# Helpers
# --------------------------------------------------------------------------- #
def _next_pow2(x):
    p = 1
    while p < x:
        p <<= 1
    return p


def _choose_splits(B, Hkv, total_pages):
    """Pick NUM_SPLITS so B*Hkv*splits gives enough CTAs to fill 148 SMs
    without exploding partial traffic. Each split needs >=1 page."""
    target_ctas = 148 * 2            # aim for ~2 waves for latency hiding
    base = B * Hkv
    s = max(1, (target_ctas + base - 1) // base)
    s = min(s, total_pages)          # cannot exceed pages
    # keep splits a divisor-ish small number; cap at 8
    s = min(s, 8)
    return max(1, s)


# --------------------------------------------------------------------------- #
# Model
# --------------------------------------------------------------------------- #
class Model(nn.Module):
    def __init__(self, batch, num_heads, num_kv_heads, head_dim, seq_len, page_size):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.pad_g = max(16, _next_pow2(self.group_size))
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

        self._max_blocks = (seq_len + page_size - 1) // page_size
        self._num_splits = _choose_splits(batch, num_kv_heads, self._max_blocks)
        self._pages_per_split = (self._max_blocks + self._num_splits - 1) // self._num_splits
        self._block_d = _next_pow2(head_dim)
        self._block_s = _next_pow2(self._num_splits)

        # persistent scratch (reused across calls; stable for graph capture)
        self._o_part = torch.empty(batch, num_heads, self._num_splits, head_dim, dtype=torch.float32, device="cuda")
        self._lse_part = torch.empty(batch, num_heads, self._num_splits, dtype=torch.float32, device="cuda")
        self._out = torch.empty(batch, num_heads, head_dim, dtype=torch.bfloat16, device="cuda")

        # graph cache
        self._graph = None
        self._graph_key = None
        self._eager_count = {}

    # ---- raw kernel launch (no graph) ----
    def _run(self, query, kv_cache, block_table, seq_lens):
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        P = self.page_size
        G = self.group_size
        PAD_G = self.pad_g

        grid = (B, Hkv, self._num_splits)
        _fwd_decode_kernel[grid](
            query, kv_cache, block_table, seq_lens,
            self._o_part, self._lse_part,
            H * D, D,
            P * Hkv * 2 * D, Hkv * 2 * D, 2 * D,
            self._max_blocks,
            H * self._num_splits * D, self._num_splits * D, D,
            H * self._num_splits, self._num_splits,
            self.scale,
            D=D, PAD_G=PAD_G, G=G, P=P, Hkv=Hkv,
            PAGES_PER_SPLIT=self._pages_per_split,
            num_warps=4, num_stages=2,
        )
        grid_r = (B, H)
        _reduce_kernel[grid_r](
            self._o_part, self._lse_part, self._out,
            H * self._num_splits * D, self._num_splits * D, D,
            H * self._num_splits, self._num_splits,
            H * D, D,
            D=D, NUM_SPLITS=self._num_splits, BLOCK_S=self._block_s, BLOCK_D=self._block_d,
            num_warps=2,
        )
        return self._out

    def _capture_graph(self, query, kv_cache, block_table, seq_lens):
        # warmup the exact kernel path once more (ensure compiled), then capture
        self._run(query, kv_cache, block_table, seq_lens)
        torch.cuda.synchronize()
        g = torch.cuda.CUDAGraph()
        # capture on a side stream then no-op; standard pattern uses default stream
        with torch.cuda.graph(g):
            self._run(query, kv_cache, block_table, seq_lens)
        self._graph = g

    def forward(self, query, kv_cache, block_table, seq_lens):
        key = (query.data_ptr(), kv_cache.data_ptr(),
               block_table.data_ptr(), seq_lens.data_ptr(),
               query.shape, kv_cache.shape)
        if self._graph is not None and key == self._graph_key:
            self._graph.replay()
            return self._out
        out = self._run(query, kv_cache, block_table, seq_lens)
        c = self._eager_count.get(key, 0) + 1
        self._eager_count[key] = c
        if c == 2:
            try:
                self._capture_graph(query, kv_cache, block_table, seq_lens)
                self._graph_key = key
            except Exception:
                self._graph = None
                self._graph_key = None
        return out


def get_inputs():
    B = BATCH = 8
    H = NUM_HEADS = 32
    Hkv = NUM_KV_HEADS = 8
    D = HEAD_DIM = 128
    L = SEQ_LEN = 1024
    P = PAGE_SIZE = 16
    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)
    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1
    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)
    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]


BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16
