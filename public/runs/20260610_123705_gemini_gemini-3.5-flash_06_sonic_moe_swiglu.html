    X_ptr,
    W_gate_ptr,
    W_up_ptr,
    Out_ptr,
    # Offsets and metadata
    expert_offsets_ptr,
    m_offsets_ptr,

    stride_ws, stride_wh, stride_wi,

    BLOCK_SIZE_N: tl.constexpr,
    BLOCK_SIZE_K: tl.constexpr,
):

    pid_n = tl.program_id(1)

    high = E

        val = tl.load(m_offsets_ptr + mid)

    token_end = tl.load(expert_offsets_ptr + e + 1)

    row_start = local_pid_m * BLOCK_SIZE_M

    offs_m = row_start + tl.arange(0, BLOCK_SIZE_M)
    offs_n = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    W_up_base = W_up_ptr + e * stride_ws

    acc_up = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)

    mask_n = offs_n[None, :] < I

    X_base_rows = X_base + tl.arange(0, BLOCK_SIZE_M)[:, None] * stride_xs
    W_gate_cols = W_gate_base + offs_n[None, :] * stride_wi
    W_up_cols = W_up_base + offs_n[None, :] * stride_wi

        mask_k_x = (offs_k + k)[None, :] < H
        x_tile = tl.load(
            X_base_rows + (offs_k + k)[None, :] * stride_xh,
            mask=mask_m & mask_k_x,
            other=0.0

        mask_k_w = (offs_k + k)[:, None] < H

            W_gate_cols + (offs_k + k)[:, None] * stride_wh,
            mask=mask_k_w & mask_n,
            other=0.0
        )

            W_up_cols + (offs_k + k)[:, None] * stride_wh,
            mask=mask_k_w & mask_n,
            other=0.0

        acc_up += tl.dot(x_tile, w_up_tile)

    # Apply SwiGLU: silu(gate) * up
    gate_val = acc_gate * tl.sigmoid(acc_gate)
    fused_val = gate_val * acc_up

    # Cast to bfloat16
    fused_val = fused_val.to(tl.bfloat16)

    )

    base = T_perm // E
