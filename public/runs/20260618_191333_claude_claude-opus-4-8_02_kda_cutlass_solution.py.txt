"""Kimi Delta Attention (KDA) forward, chunk-parallel form, custom Triton kernels.

Three kernels:
  1. prepare : per-chunk, parallel. cumsum(g), build intra-chunk K-K matrix,
               invert (I - M) via the matrix-power product (M nilpotent), and
               precompute everything the later kernels need:
                 w, u            (delta-rule WY factors)
                 kdecay = k*exp(gc_last-gc)   (state update factor)
                 qg = q*scale*exp(gc), kgn = k*exp(-gc)  (output factors)
                 gd = exp(gc_last)           (per-chunk state decay)
  2. scan    : sequential over chunks, per (b, h, V-block). Pure bf16 matmuls;
               carries state S (K, V); emits per-chunk init state h, v_new.
  3. output  : per-chunk, parallel. o = qg @ h + tril(qg@kgn^T) @ v_new.

All intra-chunk decays are bounded (g is a log-decay, gc monotone-ish), so the
math is numerically stable in fp32. Intermediates are bf16 to keep traffic low.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

LOG2E = tl.constexpr(1.4426950408889634)  # 1/ln(2): work in log2 units for exp2


# ----------------------------------------------------------------------------
# Kernel 1: per-chunk prepare
# ----------------------------------------------------------------------------
@triton.jit
def _prepare_kernel(
    q_ptr, k_ptr, v_ptr, g_ptr, beta_ptr,
    w_ptr, u_ptr, kdecay_ptr, qg_ptr, kgn_ptr, gd_ptr,
    scale,
    T, NT, H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    qkg_base = (i_b * T * H + i_h) * K
    v_base = (i_b * T * H + i_h) * V
    beta_base = i_b * T * H + i_h
    row0 = i_t * BT

    p_q = tl.make_block_ptr(q_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_k = tl.make_block_ptr(k_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_g = tl.make_block_ptr(g_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    b_q = tl.load(p_q, boundary_check=(0, 1)).to(tl.float32) * scale
    b_k = tl.load(p_k, boundary_check=(0, 1)).to(tl.float32)
    b_g = tl.load(p_g, boundary_check=(0, 1)).to(tl.float32)
    p_v = tl.make_block_ptr(v_ptr + v_base, (T, V), (H * V, 1), (row0, 0), (BT, V), (1, 0))
    b_v = tl.load(p_v, boundary_check=(0, 1))

    o_t = tl.arange(0, BT)
    b_beta = tl.load(beta_ptr + beta_base + (row0 + o_t) * H).to(tl.float32)

    # cumulative log-decay within chunk (inclusive), log2 units
    b_gc = tl.cumsum(b_g, axis=0) * LOG2E
    b_gc_last = tl.sum(b_g, axis=0) * LOG2E   # last row of cumsum, shape [K]

    b_exp = tl.exp2(b_gc)
    b_kg_pos = (b_k * b_exp).to(tl.bfloat16)
    b_kg_neg = (b_k * tl.exp2(-b_gc)).to(tl.bfloat16)

    # ---- store the elementwise outputs early to shorten tile liveness ----
    gd_base = ((i_b * NT + i_t) * H + i_h) * K
    tl.store(gd_ptr + gd_base + tl.arange(0, K), tl.exp2(b_gc_last))
    p_qg = tl.make_block_ptr(qg_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_kgn = tl.make_block_ptr(kgn_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_kd = tl.make_block_ptr(kdecay_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    tl.store(p_qg, (b_q * b_exp).to(qg_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_kgn, b_kg_neg, boundary_check=(0, 1))
    tl.store(p_kd, (b_k * tl.exp2(b_gc_last[None, :] - b_gc)).to(kdecay_ptr.dtype.element_ty),
             boundary_check=(0, 1))

    # KK[c,i] = sum_d k[c,d]*exp(gc[c,d]-gc[i,d])*k[i,d]
    b_KK = tl.dot(b_kg_pos, tl.trans(b_kg_neg))
    strict_lower = o_t[:, None] > o_t[None, :]
    b_M = tl.where(strict_lower, -b_beta[:, None] * b_KK, 0.0)

    # T = (I - M)^{-1} = prod_{j=0..5} (I + M^{2^j})   (M^{64} = 0)
    eye = (o_t[:, None] == o_t[None, :]).to(tl.float32)
    b_P = eye + b_M
    b_Mp = b_M
    for _ in range(0, 5):
        b_Mp = tl.dot(b_Mp.to(tl.bfloat16), b_Mp.to(tl.bfloat16))
        b_P = b_P + tl.dot(b_P.to(tl.bfloat16), b_Mp.to(tl.bfloat16))

    b_A = (b_P * b_beta[None, :]).to(tl.bfloat16)   # A_final[c,j] = T[c,j]*beta[j]
    b_w = tl.dot(b_A, b_kg_pos)
    b_u = tl.dot(b_A, b_v)

    p_w = tl.make_block_ptr(w_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_u = tl.make_block_ptr(u_ptr + v_base, (T, V), (H * V, 1), (row0, 0), (BT, V), (1, 0))
    tl.store(p_w, b_w.to(w_ptr.dtype.element_ty), boundary_check=(0, 1))
    tl.store(p_u, b_u.to(u_ptr.dtype.element_ty), boundary_check=(0, 1))


# ----------------------------------------------------------------------------
# Kernel 2: sequential inter-chunk state scan -> h, v_new  (bf16 matmuls)
# ----------------------------------------------------------------------------
@triton.jit
def _scan_kernel(
    w_ptr, u_ptr, kdecay_ptr, gd_ptr, h_ptr, vnew_ptr,
    T, NT, H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr, BV: tl.constexpr,
):
    i_v = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    qkg_base = (i_b * T * H + i_h) * K
    v_base = (i_b * T * H + i_h) * V
    o_k = tl.arange(0, K)
    b_h = tl.zeros([K, BV], dtype=tl.float32)

    for i_t in range(0, NT):
        row0 = i_t * BT
        p_w = tl.make_block_ptr(w_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
        p_kd = tl.make_block_ptr(kdecay_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
        p_u = tl.make_block_ptr(u_ptr + v_base, (T, V), (H * V, 1), (row0, i_v * BV), (BT, BV), (1, 0))

        b_w = tl.load(p_w, boundary_check=(0, 1))
        b_kd = tl.load(p_kd, boundary_check=(0, 1))
        b_u = tl.load(p_u, boundary_check=(0, 1)).to(tl.float32)

        h_base = ((i_b * NT + i_t) * H + i_h) * K * V
        p_h = tl.make_block_ptr(h_ptr + h_base, (K, V), (V, 1), (0, i_v * BV), (K, BV), (1, 0))
        tl.store(p_h, b_h.to(p_h.dtype.element_ty), boundary_check=(0, 1))

        b_vnew = b_u - tl.dot(b_w, b_h.to(tl.bfloat16))
        p_vnew = tl.make_block_ptr(vnew_ptr + v_base, (T, V), (H * V, 1), (row0, i_v * BV), (BT, BV), (1, 0))
        b_vnew16 = b_vnew.to(tl.bfloat16)
        tl.store(p_vnew, b_vnew16, boundary_check=(0, 1))

        b_gd = tl.load(gd_ptr + ((i_b * NT + i_t) * H + i_h) * K + o_k)
        b_h = b_h * b_gd[:, None] + tl.dot(tl.trans(b_kd), b_vnew16)


# ----------------------------------------------------------------------------
# Kernel 3: per-chunk output  o = qg @ h + tril(qg@kgn^T) @ v_new
# ----------------------------------------------------------------------------
@triton.jit
def _output_kernel(
    qg_ptr, kgn_ptr, h_ptr, vnew_ptr, o_ptr,
    T, NT, H: tl.constexpr, K: tl.constexpr, V: tl.constexpr, BT: tl.constexpr,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    qkg_base = (i_b * T * H + i_h) * K
    v_base = (i_b * T * H + i_h) * V
    row0 = i_t * BT

    p_qg = tl.make_block_ptr(qg_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    p_kgn = tl.make_block_ptr(kgn_ptr + qkg_base, (T, K), (H * K, 1), (row0, 0), (BT, K), (1, 0))
    b_qg = tl.load(p_qg, boundary_check=(0, 1))
    b_kgn = tl.load(p_kgn, boundary_check=(0, 1))

    b_Aqk = tl.dot(b_qg, tl.trans(b_kgn), input_precision="tf32")
    o_t = tl.arange(0, BT)
    b_Aqk = tl.where(o_t[:, None] >= o_t[None, :], b_Aqk, 0.0).to(tl.bfloat16)

    h_base = ((i_b * NT + i_t) * H + i_h) * K * V
    p_h = tl.make_block_ptr(h_ptr + h_base, (K, V), (V, 1), (0, 0), (K, V), (1, 0))
    b_h = tl.load(p_h, boundary_check=(0, 1))
    p_vnew = tl.make_block_ptr(vnew_ptr + v_base, (T, V), (H * V, 1), (row0, 0), (BT, V), (1, 0))
    b_vnew = tl.load(p_vnew, boundary_check=(0, 1))

    b_o = tl.dot(b_qg, b_h) + tl.dot(b_Aqk, b_vnew)

    p_o = tl.make_block_ptr(o_ptr + v_base, (T, V), (H * V, 1), (row0, 0), (BT, V), (1, 0))
    tl.store(p_o, b_o.to(o_ptr.dtype.element_ty), boundary_check=(0, 1))


def _kda_forward(q, k, v, g, beta, scale, chunk_size):
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0
    NT = T // BT

    q = q.contiguous(); k = k.contiguous(); v = v.contiguous()
    g = g.contiguous(); beta = beta.contiguous()

    bf16 = torch.bfloat16
    dev = q.device
    w = torch.empty(B, T, H, K, dtype=bf16, device=dev)
    u = torch.empty(B, T, H, V, dtype=bf16, device=dev)
    kdecay = torch.empty(B, T, H, K, dtype=bf16, device=dev)
    qg = torch.empty(B, T, H, K, dtype=bf16, device=dev)
    kgn = torch.empty(B, T, H, K, dtype=bf16, device=dev)
    gd = torch.empty(B, NT, H, K, dtype=torch.float32, device=dev)
    h = torch.empty(B, NT, H, K, V, dtype=bf16, device=dev)
    vnew = torch.empty(B, T, H, V, dtype=bf16, device=dev)
    o = torch.empty(B, T, H, V, dtype=bf16, device=dev)

    grid_chunk = (NT, B * H)
    _prepare_kernel[grid_chunk](
        q, k, v, g, beta, w, u, kdecay, qg, kgn, gd, scale,
        T, NT, H, K, V, BT, num_warps=8,
    )

    BV = 16
    _scan_kernel[(V // BV, B * H)](
        w, u, kdecay, gd, h, vnew,
        T, NT, H, K, V, BT, BV, num_warps=4, num_stages=1,
    )

    _output_kernel[grid_chunk](
        qg, kgn, h, vnew, o,
        T, NT, H, K, V, BT, num_warps=4,
    )
    return o


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        return _kda_forward(q, k, v, g, beta, self.scale, self.chunk_size)


# Module-level shape shims (overridden by check.py / benchmark.py per shape).
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
