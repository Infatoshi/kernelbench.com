import torch
import torch.nn as nn
from einops import rearrange
import triton
import triton.language as tl

@triton.jit
def kda_intra_u_w_kernel(
    k_ptr, v_ptr, g_ptr, g_cumsum_ptr, beta_ptr, u_ptr, w_ptr,
    stride_b, stride_h, stride_t, stride_d,
    stride_beta_b, stride_beta_t, stride_beta_h,
    B, H, T, K, V, BT: tl.constexpr
):
    pid = tl.program_id(0)
    b = pid // (H * (T // BT))
    rem = pid % (H * (T // BT))
    h = rem // (T // BT)
    n = rem % (T // BT)

    # Base pointer offsets
    chunk_offset = b * stride_b + h * stride_h + n * BT * stride_t

    # Define axes
    axis_k = tl.arange(0, 128)[:, None] # (128, 1)
    axis_v = tl.arange(0, 128)[None, :] # (1, 128)

    # Initialize states in registers/SRAM
    S_v = tl.zeros((128, 128), dtype=tl.float32)
    S_k = tl.zeros((128, 128), dtype=tl.float32)

    for t in range(BT):
        t_offset = chunk_offset + t * stride_t
        
        # Load inputs
        kt = tl.load(k_ptr + t_offset + axis_k * stride_d) # (128, 1)
        g_val = tl.load(g_ptr + t_offset + axis_k * stride_d) # (128, 1)
        decay = tl.exp(g_val) # (128, 1)

        beta_offset = b * stride_beta_b + (n * BT + t) * stride_beta_t + h * stride_beta_h
        beta_val = tl.load(beta_ptr + beta_offset)

        # Update u
        S_v = S_v * decay
        dot_v = tl.sum(S_v * kt, axis=0)[None, :] # (1, 128)
        vt = tl.load(v_ptr + t_offset + axis_v * stride_d) # (1, 128)
        ut = beta_val * (vt - dot_v) # (1, 128)
        tl.store(u_ptr + t_offset + axis_v * stride_d, ut)
        S_v = S_v + kt * ut

        # Update w
        g_cumsum_val = tl.load(g_cumsum_ptr + t_offset + axis_k * stride_d) # (128, 1)
        w_input = tl.exp(g_cumsum_val) * kt # (128, 1)
        S_k = S_k * decay
        dot_k = tl.sum(S_k * kt, axis=0)[:, None] # (128, 1)
        wt = beta_val * (w_input - dot_k) # (128, 1)
        tl.store(w_ptr + t_offset + axis_k * stride_d, wt)
        S_k = S_k + kt * tl.trans(wt)


@triton.jit
def kda_inter_chunk_kernel(
    q_ptr, k_ptr, u_ptr, g_cumsum_ptr, w_ptr, o_ptr,
    stride_b, stride_h, stride_t, stride_d,
    B, H, T, K, V, NT, BT: tl.constexpr
):
    pid = tl.program_id(0)
    b = pid // H
    h = pid % H

    # Thread/block dimensions
    axis_c = tl.arange(0, 64)[:, None] # (64, 1)
    axis_j = tl.arange(0, 64)[None, :] # (1, 64)
    axis_k = tl.arange(0, 128)
    axis_v = tl.arange(0, 128)

    # Initialize state S in registers: shape (128, 128)
    S = tl.zeros((128, 128), dtype=tl.float32)

    # Base pointers for this batch and head
    base_offset = b * stride_b + h * stride_h

    # Loop over chunks
    for i in range(NT):
        # Offset for the start of chunk i
        chunk_offset = base_offset + i * BT * stride_t

        # Load q_i, k_i, u_i, g_i, w_i
        q_i = tl.load(q_ptr + chunk_offset + axis_c * stride_t + axis_k[None, :] * stride_d)
        k_i = tl.load(k_ptr + chunk_offset + axis_c * stride_t + axis_k[None, :] * stride_d)
        u_i = tl.load(u_ptr + chunk_offset + axis_c * stride_t + axis_v[None, :] * stride_d)
        g_i = tl.load(g_cumsum_ptr + chunk_offset + axis_c * stride_t + axis_k[None, :] * stride_d)
        w_i = tl.load(w_ptr + chunk_offset + axis_c * stride_t + axis_k[None, :] * stride_d)

        # X_q = q_i * tl.exp(g_i)
        X_q = q_i * tl.exp(g_i)
        # Y_k = k_i * tl.exp(-g_i)
        Y_k = k_i * tl.exp(-g_i)

        # Compute Aqk = X_q @ Y_k.T
        Aqk = tl.dot(X_q, tl.trans(Y_k)) # (64, 64)
        
        # Mask Aqk to be lower triangular (strictly upper is masked to 0)
        mask = axis_c >= axis_j # (64, 64)
        Aqk = tl.where(mask, Aqk, 0.0)

        # v_i = u_i - w_i @ S
        w_S = tl.dot(w_i, S) # (64, 128)
        v_i = u_i - w_S

        # o_i = X_q @ S + Aqk @ v_i
        o_i = tl.dot(X_q, S) + tl.dot(Aqk, v_i)

        # Store o_i
        tl.store(o_ptr + chunk_offset + axis_c * stride_t + axis_v[None, :] * stride_d, o_i)

        # Update state S
        decay_last_val = tl.load(g_cumsum_ptr + chunk_offset + 63 * stride_t + axis_k * stride_d) # (128,)
        decay_last = tl.exp(decay_last_val)[:, None] # (128, 1)

        # S = S * decay_last
        S = S * decay_last

        # Z_k = exp(g_i[63] - g_i) * k_i
        Z_k = tl.exp(decay_last_val[None, :] - g_i) * k_i # (64, 128)

        # S = S + Z_k.T @ v_i
        S = S + tl.dot(tl.trans(Z_k), v_i)


class Model(nn.Module):
    """KDA forward (chunk form). Fused Triton implementation (two-kernel)."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        orig_dtype = v.dtype
        B, T, H, K = q.shape
        V = v.shape[-1]
        BT = self.chunk_size
        NT = T // BT

        # Cast/ensure float32 for Triton accumulation
        q_f32 = q.to(torch.float32)
        k_f32 = k.to(torch.float32)
        v_f32 = v.to(torch.float32)
        g_f32 = g.to(torch.float32)
        beta_f32 = beta.to(torch.float32)

        # Precompute g_cumsum
        g_cumsum = rearrange(rearrange(g_f32, "b (n bt) h d -> b h n bt d", bt=BT).cumsum(-2), "b h n bt d -> b (n bt) h d").contiguous()

        # Allocate u and w
        u = torch.empty_like(v_f32)
        w = torch.empty_like(k_f32)

        # Launch Triton kernel for the heavy intra-chunk recurrences
        grid_intra = (B * H * NT,)
        kda_intra_u_w_kernel[grid_intra](
            k_f32, v_f32, g_f32, g_cumsum, beta_f32, u, w,
            k_f32.stride(0), k_f32.stride(2), k_f32.stride(1), k_f32.stride(3),
            beta_f32.stride(0), beta_f32.stride(1), beta_f32.stride(2),
            B, H, T, K, V, BT
        )

        # Pre-scale q
        q_scaled = q_f32 * self.scale

        # Launch Triton inter-chunk kernel
        o = torch.empty_like(v_f32)
        grid_inter = (B * H,)
        kda_inter_chunk_kernel[grid_inter](
            q_scaled, k_f32, u, g_cumsum, w, o,
            q_scaled.stride(0), q_scaled.stride(2), q_scaled.stride(1), q_scaled.stride(3),
            B, H, T, K, V, NT, BT,
            num_stages=1
        )

        return o.to(orig_dtype)
