"""Custom top-k kernel for RTX PRO 6000 (SM120 Blackwell, 1.8 TB/s GDDR7).

Strategy
--------
This problem is *memory bound* and the inputs are tiny (0.5-2 MB), so kernel
runtime is dominated by launch overhead, not compute or even DRAM latency.
The stock PyTorch selection op only reaches ~2% of peak here because it
dispatches a chain of radix-select kernels.

Our kernel keeps the launch count minimal (1-3 kernels) and wraps the whole
sequence in a CUDA graph (captured lazily, replayed on the hot path) to
eliminate per-call host/driver overhead.

Algorithm:
  * Represent each element as a packed uint64 (sortable float-key << 32 | index),
    so a single integer comparison orders by value (ties broken by index).
  * Phase 1 (topk_reduce): grid (blocks_per_row, batch). Each block reads a
    contiguous tile of its row from DRAM, fully bitonic-sorts the tile in
    shared memory, and writes the tile's top-k as packed candidates.
  * Finalize (topk_merge): grid (batch). Each block bitonic-sorts its row's
    (blocks_per_row * k) candidates and emits sorted-descending values+indices.
  * For very large n / small batch (single huge row) the candidate fan-out is
    reduced by an intermediate merge level before finalize.

float->key packing makes ascending-uint order == ascending-float order, so the
bitonic network is exact (selection returns the original float values), which
satisfies the 1e-4 value tolerance trivially and is robust to the tiny/large
stress scales.
"""

__device__ __forceinline__ uint32_t f2k(float f){
    uint32_t u = __float_as_uint(f);
    uint32_t mask = -(u >> 31);                 // FFFFFFFF if negative, else 0
    return u ^ (mask | 0x80000000u);            // asc uint == asc float
}
__device__ __forceinline__ float k2f(uint32_t uk){
    uint32_t u = (uk >> 31) ? (uk ^ 0x80000000u) : (~uk);
    return __uint_as_float(u);
}
__device__ __forceinline__ uint64_t pack(uint32_t key, uint32_t idx){
    return ((uint64_t)key << 32) | (uint64_t)idx;
}

// Ascending in-place bitonic sort over P (power of 2) packed uint64 elements,
// T threads per block. smem[0] = smallest afterwards.
__device__ __forceinline__ void bitonic_asc(uint64_t* a, int P, int T, int tid){
    for (int size = 2; size <= P; size <<= 1){
        for (int stride = size >> 1; stride > 0; stride >>= 1){
            for (int i = tid; i < P; i += T){
                int j = i ^ stride;
                if (j > i){
                    bool dir = ((i & size) == 0);      // ascending segment
                    uint64_t ai = a[i], aj = a[j];
                    bool swap = dir ? (ai > aj) : (ai < aj);
                    if (swap){ a[i] = aj; a[j] = ai; }
                }
            }
            __syncthreads();
        }
    }
}

// Phase 1: read fp32 input tile -> packed top-k candidates. grid(bpr, batch).
__global__ void phase1_kernel(
    const float* __restrict__ in, uint64_t* __restrict__ cand,
    int n, int k, int bpr, int tile, int tile_pow2)
{
    int row = blockIdx.y, tblk = blockIdx.x, tid = threadIdx.x, T = blockDim.x;
    const float* row_in = in + (size_t)row * n;
    int start = tblk * tile;
    int end = start + tile; if (end > n) end = n;
    extern __shared__ uint64_t smem[];
    const uint64_t SENTINEL = pack(0u, 0xFFFFFFFFu);   // smallest possible key
    for (int i = tid; i < tile_pow2; i += T){
        int idx = start + i;
        smem[i] = (idx < end) ? pack(f2k(row_in[idx]), (uint32_t)idx) : SENTINEL;
    }
    __syncthreads();
    bitonic_asc(smem, tile_pow2, T, tid);
    uint64_t* out = cand + ((size_t)row * bpr + tblk) * k;
    for (int i = tid; i < k; i += T){
        out[i] = smem[tile_pow2 - k + i];
    }
}

// Finalize: merge bpr*k packed candidates -> descending values+indices. grid(batch).
__global__ void finalize_kernel(
    const uint64_t* __restrict__ cand, float* __restrict__ values, long* __restrict__ indices,
    int k, int bpr, int cpow2)
{
    int row = blockIdx.x, tid = threadIdx.x, T = blockDim.x;
    const uint64_t* row_cand = cand + (size_t)row * bpr * k;
    int C = bpr * k;
    extern __shared__ uint64_t smem[];
    const uint64_t SENTINEL = pack(0u, 0xFFFFFFFFu);
    for (int i = tid; i < cpow2; i += T){
        smem[i] = (i < C) ? row_cand[i] : SENTINEL;
    }
    __syncthreads();
    bitonic_asc(smem, cpow2, T, tid);
    for (int i = tid; i < k; i += T){
        uint64_t v = smem[cpow2 - 1 - i];
        values[(size_t)row * k + i]  = k2f((uint32_t)(v >> 32));
        indices[(size_t)row * k + i] = (long)(uint32_t)(v & 0xFFFFFFFFu);
    }
}

// Intermediate merge: packed candidates -> packed candidates. grid(groups, batch).
__global__ void merge_kernel(
    const uint64_t* __restrict__ cin, uint64_t* __restrict__ cout,
    int k, int bpr_in, int fan, int cpow2)
{
    int row = blockIdx.y, group = blockIdx.x, tid = threadIdx.x, T = blockDim.x;
    int bpr_out = gridDim.x;
    const uint64_t* base = cin + (size_t)row * bpr_in * k;
    int start_tile = group * fan;
    int C = fan * k;
    extern __shared__ uint64_t smem[];
    const uint64_t SENTINEL = pack(0u, 0xFFFFFFFFu);
    for (int i = tid; i < cpow2; i += T){

    }
    __syncthreads();

    for (int i = tid; i < k; i += T){

    }
}

class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()

        self.register_buffer("_dummy", torch.zeros(1))
        self._cfg = self._plan(batch, n, k)
        self._ready = False
        self._graph = None
        self._captured_ptr = None

    @staticmethod
    def _plan(batch, n, k):
        # Choose blocks-per-row: enough total blocks to cover the SM array,
        # bounded so that the single-block finalize candidate fan-out (bpr*k)
        # stays manageable; large fan-outs get an intermediate merge level.
        target_blocks = 512
        bpr = max(1, (target_blocks + batch - 1) // batch)
        # single-level finalize wants bpr*k <= ~8192; if larger, merge first
        merge_fan = 1
        if bpr * k > 8192:
            # add one merge level: group phase1 tiles so finalize input fits
            merge_fan = _next_pow2((bpr * k + 8191) // 8192)
            bpr1 = bpr
            groups = (bpr1 + merge_fan - 1) // merge_fan
            bpr = bpr1  # phase1 still uses bpr1 blocks
            tile = (n + bpr1 - 1) // bpr1
            return {
                "merge": True, "bpr1": bpr1, "groups": groups, "fan": merge_fan,
                "tile": tile, "tp": _next_pow2(tile),
                "cpow2_merge": _next_pow2(merge_fan * k),
                "cpow2_fin": _next_pow2(groups * k),
            }
        tile = (n + bpr - 1) // bpr
        return {
            "merge": False, "bpr": bpr, "tile": tile, "tp": _next_pow2(tile),
            "cpow2": _next_pow2(bpr * k),
        }

    def _setup(self, x):
        dev = x.device
        c = self._cfg
        if c["merge"]:
            self._cand1 = torch.empty(self.batch * c["bpr1"] * self.k, dtype=torch.int64, device=dev)

        else:
