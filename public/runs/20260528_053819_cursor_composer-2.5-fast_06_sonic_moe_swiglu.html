"""Grouped GEMM + fused SwiGLU for MoE up-projection (Triton, SM120).

Per expert e:
    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])

One K-loop fuses gate/up GEMMs on the same activation tile; SwiGLU runs in fp32
before storing bf16. Grid covers (expert, M-tile, N-tile) with early exit on
empty/padding tiles.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({"BM": 128, "BN": 128, "BK": 64}, num_warps=8, num_stages=3),
        triton.Config({"BM": 128, "BN": 128, "BK": 32}, num_warps=8, num_stages=4),
        triton.Config({"BM": 128, "BN": 64, "BK": 128}, num_warps=4, num_stages=3),
        triton.Config({"BM": 64, "BN": 128, "BK": 64}, num_warps=4, num_stages=3),
        triton.Config({"BM": 64, "BN": 64, "BK": 64}, num_warps=4, num_stages=3),
        triton.Config({"BM": 128, "BN": 256, "BK": 64}, num_warps=8, num_stages=3),
        triton.Config({"BM": 64, "BN": 256, "BK": 32}, num_warps=8, num_stages=4),
        triton.Config({"BM": 256, "BN": 128, "BK": 32}, num_warps=8, num_stages=4),
        triton.Config({"BM": 32, "BN": 128, "BK": 64}, num_warps=4, num_stages=3),
    ],
    key=["H", "I"],
    warmup=1,
    rep=1,
)
@triton.jit
def _grouped_gemm_swiglu_kernel(
    hidden_ptr,
    W_gate_ptr,
    W_up_ptr,
    out_ptr,
    offsets_ptr,
    stride_hm,
    stride_hh,
    stride_wge,
    stride_wgh,
    stride_wgi,
    stride_wue,
    stride_wuh,
    stride_wui,
    stride_om,
    stride_oi,
    H: tl.constexpr,
    I: tl.constexpr,
    BM: tl.constexpr,
    BN: tl.constexpr,
    BK: tl.constexpr,
):
    n_blocks_n: tl.constexpr = (I + BN - 1) // BN

    expert_idx = tl.program_id(0)
    tile_id = tl.program_id(1)

    n_block = tile_id % n_blocks_n
    m_block = tile_id // n_blocks_n

    start = tl.load(offsets_ptr + expert_idx).to(tl.int32)
    end = tl.load(offsets_ptr + expert_idx + 1).to(tl.int32)
    m_size = end - start

    m_off = m_block * BM
    if m_off >= m_size:
        return

    n_off = n_block * BN

    rows = start + m_off + tl.arange(0, BM)
    cols_n = n_off + tl.arange(0, BN)

    m_mask = (m_off + tl.arange(0, BM)) < m_size
    n_mask = cols_n < I

    acc = tl.zeros((BM, 2 * BN), dtype=tl.float32)

    for k_start in range(0, H, BK):
        k_offs = k_start + tl.arange(0, BK)

        x = tl.load(
            hidden_ptr + rows[:, None] * stride_hm + k_offs[None, :] * stride_hh,
            mask=m_mask[:, None] & (k_offs[None, :] < H),
            other=0.0,
        )

        wg = tl.load(
            W_gate_ptr
            + expert_idx * stride_wge
            + k_offs[:, None] * stride_wgh
            + cols_n[None, :] * stride_wgi,
            mask=(k_offs[:, None] < H) & n_mask[None, :],
            other=0.0,
        )

        wu = tl.load(
            W_up_ptr
            + expert_idx * stride_wue
            + k_offs[:, None] * stride_wuh
            + cols_n[None, :] * stride_wui,
            mask=(k_offs[:, None] < H) & n_mask[None, :],
            other=0.0,
        )

        w_join = tl.join(wg, wu)
        w_pair = tl.reshape(tl.permute(w_join, (0, 2, 1)), (BK, 2 * BN))
        acc = tl.dot(x, w_pair, acc)

    acc_gate = acc[:, :BN]
    acc_up = acc[:, BN:]
    out_val = acc_gate * tl.sigmoid(acc_gate) * acc_up

    tl.store(
        out_ptr + rows[:, None] * stride_om + cols_n[None, :] * stride_oi,
        out_val.to(tl.bfloat16),
        mask=m_mask[:, None] & n_mask[None, :],
    )


def _max_tokens_per_expert(t_perm: int, num_experts: int) -> int:
    """Upper bound on rows per expert for balanced routing (see reference._build_routing)."""
    return (t_perm + num_experts - 1) // num_experts


def _launch(
    hidden_states: torch.Tensor,
    W_gate: torch.Tensor,
    W_up: torch.Tensor,
    expert_offsets: torch.Tensor,
    max_tokens_per_expert: int,
) -> torch.Tensor:
    t_perm, H = hidden_states.shape
    E = W_gate.shape[0]
    I = W_gate.shape[2]

    out = torch.empty(t_perm, I, dtype=torch.bfloat16, device=hidden_states.device)

    def grid_fn(meta):
        bm = meta["BM"]
        bn = meta["BN"]
        nbn = triton.cdiv(I, bn)
        mbm = triton.cdiv(max_tokens_per_expert, bm)
        return (E, mbm * nbn)

    _grouped_gemm_swiglu_kernel[grid_fn](
        hidden_states,
        W_gate,
        W_up,
        out,
        expert_offsets,
        hidden_states.stride(0),
        hidden_states.stride(1),
        W_gate.stride(0),
        W_gate.stride(1),
        W_gate.stride(2),
        W_up.stride(0),
        W_up.stride(1),
        W_up.stride(2),
        out.stride(0),
        out.stride(1),
        H=H,
        I=I,
    )
    return out

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets
