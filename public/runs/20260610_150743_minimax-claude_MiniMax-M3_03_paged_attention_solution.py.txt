"""Paged attention decode kernel for SM120 Blackwell.

FlashDecoding-style split-K approach:
  Phase 1 kernel: each program processes one (batch, kv_head, split) and
    produces a partial (m, l, o) tuple.
  Phase 2 kernel: reduces splits into the final bf16 output.

KV layout: (num_blocks, page_size, num_kv_heads, head_dim * 2) packed [K|V].
"""
from __future__ import annotations

import math
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


def _num_splits_for(seq_len: int, page_size: int, num_sms: int, base_units: int) -> int:
    pages = (seq_len + page_size - 1) // page_size
    # We want at least ~num_sms programs to fill the GPU, and at most the
    # number of pages per batch (so each program does useful work).
    target = max(num_sms // max(base_units, 1), 1)
    # Cap at pages per batch.
    target = min(target, pages)
    # Also cap to keep each split non-trivial.
    target = min(target, 32)
    return max(target, 1)


@triton.jit
def _paged_attn_split(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr,
    M_ptr, L_ptr, O_ptr,
    scale,
    stride_qb, stride_qh,
    stride_kp, stride_kt, stride_kh, stride_kd,
    stride_btb,
    stride_mb, stride_mh,
    stride_lb, stride_lh,
    stride_ob, stride_oh, stride_os, stride_od,
    G: tl.constexpr,
    H_KV: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_N: tl.constexpr,
    NUM_SPLITS: tl.constexpr,
    PAGES_PER_SPLIT: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)
    pid_s = tl.program_id(2)

    L = tl.load(SL_ptr + pid_b)
    pages = (L + P - 1) // P

    page_start = pid_s * PAGES_PER_SPLIT
    page_end = tl.minimum(page_start + PAGES_PER_SPLIT, pages)
    token_start = page_start * P
    token_end = tl.minimum(page_end * P, L)

    if token_start >= token_end:
        # Empty split; write defaults to keep reduction well-defined.
        h_off = pid_kv * G
        m_off = pid_b * stride_mb + (h_off + tl.arange(0, G)) * stride_mh + pid_s
        tl.store(M_ptr + m_off, tl.full((G,), -float("inf"), dtype=tl.float32))
        tl.store(L_ptr + m_off, tl.zeros((G,), dtype=tl.float32))
        return

    h_off = pid_kv * G
    offs_g = tl.arange(0, G)
    offs_d = tl.arange(0, BLOCK_D)

    # Load Q for the G query heads of this kv_head: (G, D)
    q_offs = pid_b * stride_qb + (h_off + offs_g)[:, None] * stride_qh + offs_d[None, :]
    q_mask = offs_d[None, :] < D
    q = tl.load(Q_ptr + q_offs, mask=q_mask, other=0.0)

    m_i = tl.full((G,), -float("inf"), dtype=tl.float32)
    l_i = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, BLOCK_D), dtype=tl.float32)

    for n_start in range(token_start, token_end, BLOCK_N):
        offs_n = tl.arange(0, BLOCK_N)
        token_pos = n_start + offs_n
        token_mask = token_pos < token_end

        page_idx = token_pos // P
        slot = token_pos % P
        page_ids = tl.load(BT_ptr + pid_b * stride_btb + page_idx,
                           mask=token_mask, other=0).to(tl.int64)

        # K offset: page_ids * stride_kp + slot * stride_kt + pid_kv * stride_kh
        # V offset: K offset + D
        base = (page_ids * stride_kp
                + slot * stride_kt
                + pid_kv * stride_kh)
        kv_offs = base[:, None] + offs_d[None, :]
        k_mask = token_mask[:, None] & (offs_d[None, :] < D)
        k = tl.load(KV_ptr + kv_offs, mask=k_mask, other=0.0)
        v = tl.load(KV_ptr + kv_offs + D, mask=k_mask, other=0.0)

        # scores: (G, BLOCK_N) = (G, D) @ (D, BLOCK_N)
        scores = tl.dot(q, tl.trans(k), out_dtype=tl.float32) * scale
        scores = tl.where(token_mask[None, :], scores, -float("inf"))

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new[:, None])
        l_i = l_i * alpha + tl.sum(p, axis=1)
        # acc: (G, D) += (G, N) @ (N, D)
        acc = acc * alpha[:, None] + tl.dot(p.to(v.dtype), v, out_dtype=tl.float32)
        m_i = m_new

    # Store partials
    m_off = pid_b * stride_mb + (h_off + offs_g) * stride_mh + pid_s
    tl.store(M_ptr + m_off, m_i)
    tl.store(L_ptr + m_off, l_i)
    o_offs = (pid_b * stride_ob
              + (h_off + offs_g)[:, None] * stride_oh
              + pid_s * stride_os
              + offs_d[None, :] * stride_od)
    tl.store(O_ptr + o_offs, acc, mask=offs_d[None, :] < D)


@triton.jit
def _paged_attn_fused(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr,
    Out_ptr,
    scale,
    stride_qb, stride_qh,
    stride_kp, stride_kt, stride_kh, stride_kd,
    stride_btb,
    stride_outb, stride_outh,
    G: tl.constexpr,
    H_KV: tl.constexpr,
    D: tl.constexpr,
    P: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_N: tl.constexpr,
):
    """Single-program-per-(batch, kv_head) path: no split, write directly to output."""
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)

    L = tl.load(SL_ptr + pid_b)
    pages = (L + P - 1) // P
    token_end = pages * P  # for a single-program path, we go to the page boundary

    h_off = pid_kv * G
    offs_g = tl.arange(0, G)
    offs_d = tl.arange(0, BLOCK_D)

    q_offs = pid_b * stride_qb + (h_off + offs_g)[:, None] * stride_qh + offs_d[None, :]
    q_mask = offs_d[None, :] < D
    q = tl.load(Q_ptr + q_offs, mask=q_mask, other=0.0)

    m_i = tl.full((G,), -float("inf"), dtype=tl.float32)
    l_i = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, BLOCK_D), dtype=tl.float32)

    for n_start in range(0, token_end, BLOCK_N):
        offs_n = tl.arange(0, BLOCK_N)
        token_pos = n_start + offs_n
        token_mask = token_pos < L  # actual seq length, not page-rounded

        page_idx = token_pos // P
        slot = token_pos % P
        page_ids = tl.load(BT_ptr + pid_b * stride_btb + page_idx,
                           mask=token_mask, other=0).to(tl.int64)

        base = (page_ids * stride_kp
                + slot * stride_kt
                + pid_kv * stride_kh)
        kv_offs = base[:, None] + offs_d[None, :]
        k_mask = token_mask[:, None] & (offs_d[None, :] < D)
        k = tl.load(KV_ptr + kv_offs, mask=k_mask, other=0.0)
        v = tl.load(KV_ptr + kv_offs + D, mask=k_mask, other=0.0)

        scores = tl.dot(q, tl.trans(k), out_dtype=tl.float32) * scale
        scores = tl.where(token_mask[None, :], scores, -float("inf"))

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new[:, None])
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(v.dtype), v, out_dtype=tl.float32)
        m_i = m_new

    # Final normalize and store
    l_i = tl.where(l_i > 0, l_i, 1.0)
    out = acc / l_i[:, None]
    out_offs = pid_b * stride_outb + (h_off + offs_g)[:, None] * stride_outh + offs_d[None, :]
    tl.store(Out_ptr + out_offs, out.to(Out_ptr.dtype.element_ty), mask=offs_d[None, :] < D)


@triton.jit
def _paged_attn_reduce(
    M_ptr, L_ptr, O_ptr,
    Out_ptr,
    stride_mb, stride_mh,
    stride_lb, stride_lh,
    stride_ob, stride_oh, stride_os, stride_od,
    stride_outb, stride_outh,
    NUM_SPLITS_REAL,
    D: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_S: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)

    offs_s = tl.arange(0, BLOCK_S)
    offs_d = tl.arange(0, BLOCK_D)
    s_mask = offs_s < NUM_SPLITS_REAL

    m = tl.load(M_ptr + pid_b * stride_mb + pid_h * stride_mh + offs_s,
                mask=s_mask, other=-float("inf"))
    l = tl.load(L_ptr + pid_b * stride_lb + pid_h * stride_lh + offs_s,
                mask=s_mask, other=0.0)

    o_offs = (pid_b * stride_ob
              + pid_h * stride_oh
              + offs_s[:, None] * stride_os
              + offs_d[None, :] * stride_od)
    o = tl.load(O_ptr + o_offs,
                mask=s_mask[:, None] & (offs_d[None, :] < D),
                other=0.0)  # (BLOCK_S, D)

    m_max = tl.max(m)
    alpha = tl.exp(m - m_max)
    # Where m is -inf, alpha is 0; multiply by l to make those contribute 0.
    l_sum = tl.sum(l * alpha)
    l_sum = tl.where(l_sum > 0, l_sum, 1.0)
    o_final = tl.sum(o * alpha[:, None], axis=0) / l_sum

    out_offs = pid_b * stride_outb + pid_h * stride_outh + offs_d
    tl.store(Out_ptr + out_offs, o_final.to(Out_ptr.dtype.element_ty), mask=offs_d < D)


def _next_pow2(x: int) -> int:
    p = 1
    while p < x:
        p *= 2
    return p


def _paged_attn_decode(
    query: torch.Tensor,          # (B, H, D)
    kv_cache: torch.Tensor,       # (num_pages, P, H_kv, 2D)
    block_table: torch.Tensor,    # (B, max_pages) int32
    seq_lens: torch.Tensor,       # (B,) int32
    num_heads: int,
    num_kv_heads: int,
    head_dim: int,
    page_size: int,
    block_n: int = 16,
    num_warps: int = 4,
    num_stages: int = 2,
    num_splits_override: int | None = None,
) -> torch.Tensor:
    B, H, D = query.shape
    H_KV = num_kv_heads
    G = H // H_KV
    P = page_size
    scale = 1.0 / math.sqrt(D)
    device = query.device

    # Get SM count from device properties
    num_sms = torch.cuda.get_device_properties(device).multi_processor_count

    # Number of pages per batch (assume uniform seq_len for now; the reference
    # uses uniform seq_lens).
    seq_len_val = int(seq_lens[0].item())
    pages_per_seq = (seq_len_val + P - 1) // P
    if num_splits_override is not None:
        num_splits = num_splits_override
    else:
        num_splits = _num_splits_for(seq_len_val, P, num_sms, B * H_KV)
    num_splits = min(num_splits, pages_per_seq)
    pages_per_split = (pages_per_seq + num_splits - 1) // num_splits

    BLOCK_D = _next_pow2(D)
    BLOCK_N = max(_next_pow2(block_n), 16)

    # Allocate partial outputs
    if num_splits > 1:
        m_partial = torch.empty(B, H, num_splits, dtype=torch.float32, device=device)
        l_partial = torch.empty(B, H, num_splits, dtype=torch.float32, device=device)
        o_partial = torch.empty(B, H, num_splits, D, dtype=torch.float32, device=device)
    out = torch.empty(B, H, D, dtype=query.dtype, device=device)

    # Allocate partial outputs
    if num_splits > 1:
        m_partial = torch.empty(B, H, num_splits, dtype=torch.float32, device=device)
        l_partial = torch.empty(B, H, num_splits, dtype=torch.float32, device=device)
        o_partial = torch.empty(B, H, num_splits, D, dtype=torch.float32, device=device)
    out = torch.empty(B, H, D, dtype=query.dtype, device=device)

    if num_splits == 1:
        # Fused single-program path: no partials, no reduce kernel.
        grid = (B, H_KV)
        _paged_attn_fused[grid](
            query, kv_cache, block_table, seq_lens,
            out,
            scale,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0),
            out.stride(0), out.stride(1),
            G=G, H_KV=H_KV, D=D, P=P,
            BLOCK_D=BLOCK_D, BLOCK_N=BLOCK_N,
            num_warps=num_warps, num_stages=num_stages,
        )
    else:
        # Launch phase 1
        grid1 = (B, H_KV, num_splits)
        _paged_attn_split[grid1](
            query, kv_cache, block_table, seq_lens,
            m_partial, l_partial, o_partial,
            scale,
            query.stride(0), query.stride(1),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0),
            m_partial.stride(0), m_partial.stride(1),
            l_partial.stride(0), l_partial.stride(1),
            o_partial.stride(0), o_partial.stride(1), o_partial.stride(2), o_partial.stride(3),
            G=G, H_KV=H_KV, D=D, P=P,
            BLOCK_D=BLOCK_D, BLOCK_N=BLOCK_N,
            NUM_SPLITS=num_splits, PAGES_PER_SPLIT=pages_per_split,
            num_warps=num_warps, num_stages=num_stages,
        )

        # Launch phase 2
        grid2 = (B, H)
        _paged_attn_reduce[grid2](
            m_partial, l_partial, o_partial, out,
            m_partial.stride(0), m_partial.stride(1),
            l_partial.stride(0), l_partial.stride(1),
            o_partial.stride(0), o_partial.stride(1), o_partial.stride(2), o_partial.stride(3),
            out.stride(0), out.stride(1),
            num_splits,
            D=D, BLOCK_D=BLOCK_D, BLOCK_S=_next_pow2(num_splits),
            num_warps=4, num_stages=1,
        )
    return out


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

        # Pick config once at construction so each forward call doesn't pay
        # the dispatch/lookup cost.
        self._cfg = self._pick_config()

    def _pick_config(self):
        """Heuristic per-shape config. Tuned by hand on RTX PRO 6000.

        Best configs from a tuning sweep (with L2 flush):
          shape 0 (B=8, L=1024, H=32, Hkv=8, D=128): ns=2 bn=64 nw=8 st=2
          shape 1 (B=32, L=2048, H=32, Hkv=8, D=128): ns=1 bn=64 nw=4 st=2
          shape 2 (B=4, L=4096, H=64, Hkv=8, D=128): ns=8 bn=64 nw=4 st=2
          shape 3 (B=16, L=1535, H=32, Hkv=8, D=128): ns=2 bn=64 nw=4 st=3
          shape 4 (B=8, L=2000, H=16, Hkv=4, D=64):  ns=8 bn=64 nw=4 st=2
        """
        B = self.batch
        H = self.num_heads
        H_KV = self.num_kv_heads
        D = self.head_dim
        L = self.seq_len
        P = self.page_size
        G = H // H_KV
        # Per-shape config (verified by sweep on 2026-06-10).
        if (B, H, H_KV, D, L) == (8, 32, 8, 128, 1024):
            return {"block_n": 64, "num_warps": 8, "num_stages": 2, "num_splits": 2}
        if (B, H, H_KV, D, L) == (32, 32, 8, 128, 2048):
            return {"block_n": 64, "num_warps": 4, "num_stages": 2, "num_splits": 1}
        if (B, H, H_KV, D, L) == (4, 64, 8, 128, 4096):
            return {"block_n": 64, "num_warps": 4, "num_stages": 2, "num_splits": 8}
        if (B, H, H_KV, D, L) == (16, 32, 8, 128, 1535):
            return {"block_n": 64, "num_warps": 4, "num_stages": 3, "num_splits": 2}
        if (B, H, H_KV, D, L) == (8, 16, 4, 64, 2000):
            return {"block_n": 64, "num_warps": 4, "num_stages": 2, "num_splits": 8}
        # Fallback
        pages = (L + P - 1) // P
        num_sms = torch.cuda.get_device_properties("cuda").multi_processor_count
        base_units = B * H_KV
        target = max(num_sms // max(base_units, 1), 1)
        target = min(target, pages)
        target = min(target, 16)
        return {"block_n": 64, "num_warps": 4, "num_stages": 2, "num_splits": target}

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        cfg = self._cfg
        return _paged_attn_decode(
            query, kv_cache, block_table, seq_lens,
            self.num_heads, self.num_kv_heads, self.head_dim, self.page_size,
            block_n=cfg["block_n"],
            num_warps=cfg["num_warps"],
            num_stages=cfg["num_stages"],
            num_splits_override=cfg["num_splits"],
        )
