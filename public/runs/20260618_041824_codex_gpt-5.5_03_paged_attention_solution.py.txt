import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _paged_attention_decode_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    H: tl.constexpr,
    H_KV: tl.constexpr,
    D: tl.constexpr,
    L_MAX: tl.constexpr,
    P: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // H
    h = pid - b * H
    h_kv = h // GROUP_SIZE

    offs_d = tl.arange(0, BLOCK_D)
    mask_d = offs_d < D
    q = tl.load(query + (b * H + h) * D + offs_d, mask=mask_d, other=0.0).to(tl.float32)

    m_i = tl.full((), -float("inf"), tl.float32)
    l_i = tl.full((), 0.0, tl.float32)
    acc = tl.zeros((BLOCK_D,), tl.float32)
    tok_offsets = tl.arange(0, BLOCK_M)
    seq_len = tl.load(seq_lens + b)

    for start in tl.range(0, L_MAX, BLOCK_M):
        toks = start + tok_offsets
        tok_mask = toks < seq_len
        page_cols = toks // P
        page_offsets = toks - page_cols * P
        pages = tl.load(
            block_table + b * MAX_BLOCKS + page_cols,
            mask=tok_mask,
            other=0,
        )

        token_base = ((pages * P + page_offsets) * H_KV + h_kv) * (2 * D)
        k_ptrs = kv_cache + token_base[:, None] + offs_d[None, :]
        k = tl.load(k_ptrs, mask=tok_mask[:, None] & mask_d[None, :], other=0.0).to(tl.float32)
        scores = tl.sum(k * q[None, :], axis=1) * SCALE
        scores = tl.where(tok_mask, scores, -float("inf"))

        block_m = tl.max(scores, axis=0)
        m_new = tl.maximum(m_i, block_m)
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new)
        l_new = l_i * alpha + tl.sum(p, axis=0)

        v_ptrs = kv_cache + token_base[:, None] + D + offs_d[None, :]
        v = tl.load(v_ptrs, mask=tok_mask[:, None] & mask_d[None, :], other=0.0).to(tl.float32)
        acc = acc * alpha + tl.sum(p[:, None] * v, axis=0)

        m_i = m_new
        l_i = l_new

    result = acc / l_i
    tl.store(out + (b * H + h) * D + offs_d, result, mask=mask_d)


@triton.jit
def _paged_attention_decode_hblock_kernel(
    query,
    kv_cache,
    block_table,
    seq_lens,
    out,
    H: tl.constexpr,
    H_KV: tl.constexpr,
    D: tl.constexpr,
    L_MAX: tl.constexpr,
    P: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    SCALE: tl.constexpr,
    BLOCK_M: tl.constexpr,
    BLOCK_D: tl.constexpr,
    HEAD_BLOCK: tl.constexpr,
):
    pid = tl.program_id(0)
    blocks_per_batch = H // HEAD_BLOCK
    b = pid // blocks_per_batch
    head_block = pid - b * blocks_per_batch
    h0 = head_block * HEAD_BLOCK
    h_kv = h0 // GROUP_SIZE

    offs_h = tl.arange(0, HEAD_BLOCK)
    offs_d = tl.arange(0, BLOCK_D)
    offs_m = tl.arange(0, BLOCK_M)
    mask_d = offs_d < D

    q_ptrs = query + (b * H + h0 + offs_h[:, None]) * D + offs_d[None, :]
    q = tl.load(q_ptrs, mask=mask_d[None, :], other=0.0)

    m_i = tl.full((HEAD_BLOCK,), -float("inf"), tl.float32)
    l_i = tl.full((HEAD_BLOCK,), 0.0, tl.float32)
    acc = tl.zeros((HEAD_BLOCK, BLOCK_D), tl.float32)
    seq_len = tl.load(seq_lens + b)

    for start in tl.range(0, L_MAX, BLOCK_M):
        toks = start + offs_m
        tok_mask = toks < seq_len
        page_cols = toks // P
        page_offsets = toks - page_cols * P
        pages = tl.load(
            block_table + b * MAX_BLOCKS + page_cols,
            mask=tok_mask,
            other=0,
        )

        token_base = ((pages * P + page_offsets) * H_KV + h_kv) * (2 * D)
        k_ptrs = kv_cache + token_base[None, :] + offs_d[:, None]
        k_t = tl.load(k_ptrs, mask=mask_d[:, None] & tok_mask[None, :], other=0.0)
        scores = tl.dot(q, k_t, out_dtype=tl.float32) * SCALE
        scores = tl.where(tok_mask[None, :], scores, -float("inf"))

        block_m = tl.max(scores, axis=1)
        m_new = tl.maximum(m_i, block_m)
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new[:, None])
        l_new = l_i * alpha + tl.sum(p, axis=1)

        v_ptrs = kv_cache + token_base[:, None] + D + offs_d[None, :]
        v = tl.load(v_ptrs, mask=tok_mask[:, None] & mask_d[None, :], other=0.0)
        acc = acc * alpha[:, None] + tl.dot(p, v, out_dtype=tl.float32)

        m_i = m_new
        l_i = l_new

    result = acc / l_i[:, None]
    out_ptrs = out + (b * H + h0 + offs_h[:, None]) * D + offs_d[None, :]
    tl.store(out_ptrs, result, mask=mask_d[None, :])


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.max_blocks = triton.cdiv(seq_len, page_size)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        out = torch.empty_like(query)
        block_m = 32
        head_block = 2
        _paged_attention_decode_hblock_kernel[(self.batch * (self.num_heads // head_block),)](
            query,
            kv_cache,
            block_table,
            seq_lens,
            out,
            self.num_heads,
            self.num_kv_heads,
            self.head_dim,
            self.seq_len,
            self.page_size,
            self.max_blocks,
            self.group_size,
            self.scale,
            BLOCK_M=block_m,
            BLOCK_D=triton.next_power_of_2(self.head_dim),
            HEAD_BLOCK=head_block,
            num_warps=4,
        )
        return out


def get_inputs():
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
