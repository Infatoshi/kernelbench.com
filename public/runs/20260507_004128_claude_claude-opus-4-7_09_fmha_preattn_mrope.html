Streams q/k once, gathers per-axis cos/sin from cache, applies rotate-half,
and writes the rotated outputs directly in (B, H, T, D) layout. One CUDA
block per (B, T) token; threads parallelize along the head_dim and serially
loop over (Hq + Hkv) heads.

def _build_inv_freq(D: int, base: float = 10000.0) -> torch.Tensor:

// NTHR = HALF / 2 threads per block per token.
// Multiple tokens per block (TOK_PER_BLK) for better latency hiding.
// HQ and HKV are compile-time so the head loops fully unroll.
template <int D, int S0, int S1, int S2, int HQ, int HKV, int TOK_PER_BLK>
__global__ void mrope_fused_kernel_v3(
    const __nv_bfloat16* __restrict__ q,
    const __nv_bfloat16* __restrict__ k,
    const int64_t* __restrict__ pos_t,
    const int64_t* __restrict__ pos_h,
    const int64_t* __restrict__ pos_w,
    const __nv_bfloat16* __restrict__ cos_cache,
    const __nv_bfloat16* __restrict__ sin_cache,
    __nv_bfloat16* __restrict__ q_out,
    __nv_bfloat16* __restrict__ k_out,
    int B, int T)
{
    constexpr int HALF = D / 2;
    constexpr int VEC = 2;
    constexpr int NTHR = HALF / VEC;

    int b = blockIdx.y;
    int tok_base = blockIdx.x * TOK_PER_BLK;
    int tid = threadIdx.x;
    int tok_off = threadIdx.y;
    int t = tok_base + tok_off;
    if (t >= T) return;

    int d_low = tid * VEC;
    int d_high = d_low + HALF;
    int pair_idx = d_low;

    int bt = b * T + t;
    int64_t pos;
    if (pair_idx < S0) {
        pos = pos_t[bt];
    } else if (pair_idx < (S0 + S1)) {
        pos = pos_h[bt];
    } else {
        pos = pos_w[bt];

    __nv_bfloat162 c2 = *reinterpret_cast<const __nv_bfloat162*>(cos_cache + cache_off);
    __nv_bfloat162 s2 = *reinterpret_cast<const __nv_bfloat162*>(sin_cache + cache_off);
    float c0 = __bfloat162float(__low2bfloat16(c2));
    float c1 = __bfloat162float(__high2bfloat16(c2));
    float s0v = __bfloat162float(__low2bfloat16(s2));
    float s1v = __bfloat162float(__high2bfloat16(s2));

    int q_in_off = ((b * T + t) * HQ) * D;
    int q_out_b  = b * HQ * T * D + t * D;

    for (int h = 0; h < HQ; h++) {
        const __nv_bfloat16* qb = q + q_in_off + h * D;
        __nv_bfloat162 x_lo = *reinterpret_cast<const __nv_bfloat162*>(qb + d_low);
        __nv_bfloat162 x_hi = *reinterpret_cast<const __nv_bfloat162*>(qb + d_high);
        float xl0 = __bfloat162float(__low2bfloat16(x_lo));
        float xl1 = __bfloat162float(__high2bfloat16(x_lo));
        float xh0 = __bfloat162float(__low2bfloat16(x_hi));
        float xh1 = __bfloat162float(__high2bfloat16(x_hi));

        float ol0 = xl0 * c0 - xh0 * s0v;
        float ol1 = xl1 * c1 - xh1 * s1v;
        float oh0 = xh0 * c0 + xl0 * s0v;
        float oh1 = xh1 * c1 + xl1 * s1v;

        *reinterpret_cast<__nv_bfloat162*>(qo + d_high) = out_hi;
    }

std::vector<torch::Tensor> mrope_fused_forward(
    torch::Tensor q,
    torch::Tensor k,
    torch::Tensor pos_t,
    torch::Tensor pos_h,
    torch::Tensor pos_w,
    torch::Tensor cos_cache,
    torch::Tensor sin_cache,
    int64_t s0,
    int64_t s1,
    int64_t s2)
{
    int B   = q.size(0);
    int T   = q.size(1);
    int Hq  = q.size(2);
    int D   = q.size(3);
    int Hkv = k.size(2);

    auto opts = q.options();
    auto q_out = torch::empty({B, Hq, T, D}, opts);
    auto k_out = torch::empty({B, Hkv, T, D}, opts);

    auto qp   = reinterpret_cast<const __nv_bfloat16*>(q.data_ptr<at::BFloat16>());
    auto kp   = reinterpret_cast<const __nv_bfloat16*>(k.data_ptr<at::BFloat16>());
    auto cp   = reinterpret_cast<const __nv_bfloat16*>(cos_cache.data_ptr<at::BFloat16>());
    auto sp   = reinterpret_cast<const __nv_bfloat16*>(sin_cache.data_ptr<at::BFloat16>());
    auto qop  = reinterpret_cast<__nv_bfloat16*>(q_out.data_ptr<at::BFloat16>());
    auto kop  = reinterpret_cast<__nv_bfloat16*>(k_out.data_ptr<at::BFloat16>());
    auto ptp  = pos_t.data_ptr<int64_t>();
    auto php  = pos_h.data_ptr<int64_t>();
    auto pwp  = pos_w.data_ptr<int64_t>();

    auto stream = c10::cuda::getCurrentCUDAStream();

    // Dispatch on (D, mrope_section). NTHR = D/4 (each thread = 2 dims low + 2 dims high).
    #define LAUNCH_V4_D128(HQ, HKV, WPB) do { \
        dim3 block(32, WPB); \
        dim3 grid(T, B); \

class Model(nn.Module):
    def __init__(self, B: int, T: int, Hq: int, Hkv: int, D: int,
                 mrope_section: tuple[int, int, int], max_pos: int):
        super().__init__()
        assert sum(mrope_section) == D // 2
        self.B, self.T = B, T
        self.Hq, self.Hkv, self.D = Hq, Hkv, D
        self.mrope_section = tuple(mrope_section)
        self.max_pos = max_pos

        inv_freq = _build_inv_freq(D)
        pos = torch.arange(max_pos, dtype=torch.float32)
        freqs = torch.outer(pos, inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        self.register_buffer("cos_cache", emb.cos().to(torch.bfloat16))
        self.register_buffer("sin_cache", emb.sin().to(torch.bfloat16))

        # Build module eagerly so JIT compile doesn't pollute first timing
        _get_module()

    def forward(self, q, k, pos_t, pos_h, pos_w):
        s0, s1, s2 = self.mrope_section
        mod = _get_module()
        q_out, k_out = mod.mrope_fused_forward(
            q.contiguous(), k.contiguous(),
            pos_t.contiguous(), pos_h.contiguous(), pos_w.contiguous(),
            self.cos_cache, self.sin_cache,
            int(s0), int(s1), int(s2),
        )
        return q_out, k_out
