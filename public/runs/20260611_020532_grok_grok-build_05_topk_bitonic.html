"""Custom top-k kernel using per-block (per-row) local top-k heaps + block-level candidate merge via repeated argmax.
CUDA C++ implementation via torch.utils.cpp_extension.
"""
import os
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

# Max cand 8192 to fit ~98KB shared (for bdim*k <=8192)
CUDA_KERNEL = r"""
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>
#include <cfloat>
#include <ATen/ATen.h>
#include <ATen/cuda/CUDAContext.h>

constexpr int MAX_CAND = 8192;  // for seg cands in scan (bdim*k up to 128*64) and merge (segs*k)
constexpr int MAX_WARPS = 16;  // support up to 512 threads
constexpr int MAX_THREADS_BOUND = 512;

// SCAN kernel: each block handles one (row, seg) segment of a row, keeps local top-k of its segment via per-thread,
// then block-level extract top-k of segment, writes to group buffer (linear group_id = row*segs + seg)
__global__ __launch_bounds__(MAX_THREADS_BOUND)
void topk_scan_kernel(
    const float* __restrict__ input,
    float* __restrict__ group_vals,   // (batch * segs, k) flat
    int* __restrict__ group_idxs,     // (batch * segs, k) flat int32
    int batch,
    int n,
    int k,
    int segs,
    int smem_cand_floats
) {
  int bid = blockIdx.x;
  int row = bid / segs;
  int seg = bid % segs;
  if (row >= batch) return;

  int seg_size = (n + segs - 1) / segs;
  int seg_start = seg * seg_size;
  int seg_end = min(seg_start + seg_size, n);
  int seg_n = seg_end - seg_start;
  if (seg_n <= 0) {
    // write -inf or 0? but for safety fill with -inf idx=-1 but caller ensures
    return;
  }

  const float* row_in = input + (size_t)row * n + seg_start;

  extern __shared__ uint8_t smem_base[];
  float* cvals = reinterpret_cast<float*>(smem_base);
  int* cidxs = reinterpret_cast<int*>(smem_base + smem_cand_floats * sizeof(float));
  float* wmax = reinterpret_cast<float*>(smem_base + smem_cand_floats * (sizeof(float) + sizeof(int)));
  int* widx = reinterpret_cast<int*>(reinterpret_cast<uint8_t*>(wmax) + MAX_WARPS * sizeof(float));
  int* wpos = reinterpret_cast<int*>(reinterpret_cast<uint8_t*>(widx) + MAX_WARPS * sizeof(int));
  int* cand_cnt_ptr = reinterpret_cast<int*>(reinterpret_cast<uint8_t*>(wpos) + MAX_WARPS * sizeof(int));

  int tid = threadIdx.x;
  int bdim = blockDim.x;
  int num_warps = (bdim + 31) / 32;
  int warp = tid / 32;
  int lane = tid & 31;

  if (tid == 0) *cand_cnt_ptr = 0;
  __syncthreads();

  float my_vals[64];
  int my_idxs[64];
  int my_cnt = 0;
  float my_min = FLT_MAX;

  // vectorized over the seg
  const int seg_n4 = seg_n / 4;
  const float4* seg_in4 = reinterpret_cast<const float4*>(row_in);
  for (int i4 = tid; i4 < seg_n4; i4 += bdim) {
    float4 v4 = seg_in4[i4];
    int base = seg_start + i4 * 4;
    float vs[4] = {v4.x, v4.y, v4.z, v4.w};
    #pragma unroll
    for (int t = 0; t < 4; t++) {
      float v = vs[t];
      int idx = base + t;
      if (my_cnt < k) {
        my_vals[my_cnt] = v;
        my_idxs[my_cnt] = idx;
        if (v < my_min) my_min = v;
        my_cnt++;
      } else if (v > my_min) {
        int minp = 0; float cm = my_vals[0];
        for (int j=1; j<k; j++) if (my_vals[j] < cm) { cm=my_vals[j]; minp=j; }
        my_vals[minp] = v; my_idxs[minp] = idx;
        my_min = my_vals[0]; for (int j=1; j<k; j++) if (my_vals[j] < my_min) my_min = my_vals[j];
      }
    }
  }
  for (int ii = seg_n4 * 4 + tid; ii < seg_n; ii += bdim) {
    float v = row_in[ii];
    int idx = seg_start + ii;
    if (my_cnt < k) {
      my_vals[my_cnt] = v; my_idxs[my_cnt] = idx; if (v < my_min) my_min = v; my_cnt++;
    } else if (v > my_min) {
      int minp=0; float cm=my_vals[0];
      for(int j=1;j<k;j++) if(my_vals[j]<cm){cm=my_vals[j];minp=j;}
      my_vals[minp]=v; my_idxs[minp]=idx;
      my_min = my_vals[0]; for(int j=1;j<k;j++) if(my_vals[j]<my_min) my_min=my_vals[j];
    }
  }

  for (int j=0; j<my_cnt; j++) {
    int slot = atomicAdd(cand_cnt_ptr, 1);
    if (slot < smem_cand_floats && slot < MAX_CAND) {
      cvals[slot] = my_vals[j];
      cidxs[slot] = my_idxs[j];
    }
  }
  __syncthreads();

  int nc = *cand_cnt_ptr;
  if (nc > smem_cand_floats) nc = smem_cand_floats;
  if (nc > MAX_CAND) nc = MAX_CAND;

  for (int r=0; r<k; r++) {
    float lmax = -FLT_MAX; int lidx=-1; int lpos=-1;
    for (int c=tid; c<nc; c+=bdim) {
      float v = cvals[c];
      if (v > lmax) { lmax=v; lidx=cidxs[c]; lpos=c; }
    }
    for (int off=16; off>0; off/=2) {
      float ov = __shfl_down_sync(0xffffffffu, lmax, off);
      int oi = __shfl_down_sync(0xffffffffu, lidx, off);
      int op = __shfl_down_sync(0xffffffffu, lpos, off);
      if (ov > lmax || (ov == lmax && (oi < lidx || lidx < 0))) { lmax=ov; lidx=oi; lpos=op; }
    }
    if (lane==0 && warp < MAX_WARPS) {
      wmax[warp]=lmax; widx[warp]=lidx; wpos[warp]=lpos;
    }
    __syncthreads();
    if (warp == 0) {
      int nw = (num_warps < MAX_WARPS) ? num_warps : MAX_WARPS;
      if (tid == 0) {
        float best = wmax[0]; int bestj=0;
        for (int j=1; j<nw; j++) {
          if (wmax[j] > best || (wmax[j]==best && (widx[j] < widx[bestj] || widx[bestj]<0))) { best=wmax[j]; bestj=j; }
        }
        wmax[0]=best; widx[0]=widx[bestj]; wpos[0]=wpos[bestj];
      }
    }
    __syncthreads();
    float chosen_v = wmax[0];
    int chosen_i32 = widx[0];
    int chosen_p = wpos[0];
    if (tid == 0) {
      int gidx = (row * segs + seg) * k + r;
      group_vals[gidx] = chosen_v;
      group_idxs[gidx] = chosen_i32;
      if (chosen_p >=0 && chosen_p < nc) cvals[chosen_p] = -FLT_MAX;
    }
    __syncthreads();
  }
}

// MERGE kernel: per row, load segs*k candidates from group, extract top k sorted to final out
// Use dynamic smem to support large MAX_CAND without ptxas 48KB limit
__global__ __launch_bounds__(256)
void topk_merge_kernel(
    const float* __restrict__ group_vals,  // flat (batch*segs, k)
    const int* __restrict__ group_idxs,
    float* __restrict__ out_values,
    int64_t* __restrict__ out_indices,
    int batch,
    int k,
    int segs,
    int smem_cand_floats
) {
  int row = blockIdx.x;
  if (row >= batch) return;

  float* row_out_val = out_values + (size_t)row * k;
  int64_t* row_out_idx = out_indices + (size_t)row * k;

  extern __shared__ uint8_t smem_base[];
  float* cvals = reinterpret_cast<float*>(smem_base);
  int* cidxs = reinterpret_cast<int*>(smem_base + smem_cand_floats * sizeof(float));
  float* wmax = reinterpret_cast<float*>(smem_base + smem_cand_floats * (sizeof(float) + sizeof(int)));
  int* widx = reinterpret_cast<int*>(reinterpret_cast<uint8_t*>(wmax) + MAX_WARPS * sizeof(float));
  int* wpos = reinterpret_cast<int*>(reinterpret_cast<uint8_t*>(widx) + MAX_WARPS * sizeof(int));

  int tid = threadIdx.x;
  int bdim = blockDim.x;  // 256 or less
  int num_warps = (bdim + 31)/32;
  int warp = tid / 32;
  int lane = tid & 31;

  int nc = segs * k;
  if (nc > smem_cand_floats) nc = smem_cand_floats;
  if (nc > MAX_CAND) nc = MAX_CAND;

  // load the group cands for this row into smem (coalesced)
  for (int c = tid; c < nc; c += bdim) {
    int gidx = (row * segs) * k + c;
    cvals[c] = group_vals[gidx];
    cidxs[c] = group_idxs[gidx];
  }
  __syncthreads();

  for (int r = 0; r < k; r++) {
    float lmax = -FLT_MAX; int lidx=-1; int lpos=-1;
    for (int c=tid; c<nc; c+=bdim) {
      float v = cvals[c];
      if (v > lmax) { lmax = v; lidx = cidxs[c]; lpos = c; }
    }
    for (int off=16; off>0; off/=2) {
      float ov = __shfl_down_sync(0xffffffffu, lmax, off);
      int oi = __shfl_down_sync(0xffffffffu, lidx, off);
      int op = __shfl_down_sync(0xffffffffu, lpos, off);
      if (ov > lmax || (ov == lmax && (oi < lidx || lidx < 0))) { lmax=ov; lidx=oi; lpos=op; }
    }
    if (lane == 0 && warp < MAX_WARPS) {
      wmax[warp] = lmax; widx[warp] = lidx; wpos[warp] = lpos;
    }
    __syncthreads();
    if (warp == 0) {
      int nw = (num_warps < MAX_WARPS) ? num_warps : MAX_WARPS;
      if (tid == 0) {
        float best = wmax[0]; int bestj = 0;
        for (int j=1; j<nw; j++) {
          if (wmax[j] > best || (wmax[j]==best && (widx[j] < widx[bestj] || widx[bestj]<0))) { best=wmax[j]; bestj=j; }
        }
        wmax[0] = best; widx[0] = widx[bestj]; wpos[0] = wpos[bestj];
      }
    }
    __syncthreads();
    float chosen_v = wmax[0];
    int chosen_i32 = widx[0];
    int chosen_p = wpos[0];
    if (tid == 0) {
      row_out_val[r] = chosen_v;
      row_out_idx[r] = (int64_t)chosen_i32;
      if (chosen_p >=0 && chosen_p < nc) cvals[chosen_p] = -FLT_MAX;
    }
    __syncthreads();
  }
}

void launch_topk(
    const float* input,
    float* out_values,
    int64_t* out_indices,
    float* group_vals,
    int* group_idxs,
    int batch,
    int n,
    int k,
    int segs,
    cudaStream_t stream
) {
  if (batch == 0 || n == 0 || k == 0) return;

  // SCAN
  int threads = 256;
  if (k >= 64) threads = 128;  // 128*64=8192
  else if (k >= 8) threads = 256;
  else threads = 512;
  if (threads * k > MAX_CAND) threads = max(32, MAX_CAND / max(1,k));
  if (threads < 32) threads = 32;

  int num_cand = threads * k;
  if (num_cand > MAX_CAND) num_cand = MAX_CAND;

  size_t smem_size = (size_t)num_cand * sizeof(float) +
                     (size_t)num_cand * sizeof(int) +
                     (size_t)MAX_WARPS * sizeof(float) +
                     (size_t)MAX_WARPS * sizeof(int) +
                     (size_t)MAX_WARPS * sizeof(int) +
                     sizeof(int);

  int total_blocks = batch * segs;
  dim3 grid_scan(total_blocks);
  dim3 block_scan(threads);

  static bool attr_set = false;
  if (!attr_set) {
    cudaFuncSetAttribute(topk_scan_kernel, cudaFuncAttributeMaxDynamicSharedMemorySize, 101376);
    cudaFuncSetAttribute(topk_merge_kernel, cudaFuncAttributeMaxDynamicSharedMemorySize, 101376);
    attr_set = true;
  }

  topk_scan_kernel<<<grid_scan, block_scan, smem_size, stream>>>(
      input, group_vals, group_idxs, batch, n, k, segs, num_cand
  );

  // MERGE (always, handles segs=1 as sort of the k)
  dim3 grid_merge(batch);
  dim3 block_merge(256);
  int merge_cand = segs * k;
  if (merge_cand > MAX_CAND) merge_cand = MAX_CAND;
  size_t merge_smem = (size_t)merge_cand * sizeof(float) +
                      (size_t)merge_cand * sizeof(int) +
                      (size_t)MAX_WARPS * sizeof(float) +
                      (size_t)MAX_WARPS * sizeof(int) +
                      (size_t)MAX_WARPS * sizeof(int);
  if (merge_smem < 4096) merge_smem = 4096;
  topk_merge_kernel<<<grid_merge, block_merge, merge_smem, stream>>>(
      group_vals, group_idxs, out_values, out_indices, batch, k, segs, merge_cand
  );
}
"""

  TORCH_CHECK(x.is_cuda(), "input must be a CUDA tensor");
  TORCH_CHECK(x.dim() == 2, "input must be 2D (batch, n)");
  TORCH_CHECK(x.dtype() == torch::kFloat32, "input must be float32");
  TORCH_CHECK(k > 0, "k must be positive");

  int batch = static_cast<int>(x.size(0));
  int n = static_cast<int>(x.size(1));
  TORCH_CHECK(k <= n, "k cannot exceed n");
  TORCH_CHECK(segs >= 1, "segs >=1");

  auto val_opts = torch::TensorOptions().dtype(torch::kFloat32).device(x.device());
  auto idx_opts = torch::TensorOptions().dtype(torch::kInt64).device(x.device());

  auto indices = torch::empty({batch, k}, idx_opts);

  auto cuda_stream = at::cuda::getCurrentCUDAStream();
  cudaStream_t stream = cuda_stream.stream();

  launch_topk(
      x.data_ptr<float>(),
      values.data_ptr<float>(),
      indices.data_ptr<int64_t>(),
      group_vals.data_ptr<float>(),
      group_idxs.data_ptr<int>(),
      batch,
      n,
      static_cast<int>(k),
      static_cast<int>(segs),

# Load the extension (CUDA kernel + wrapper)
# Use a unique name to avoid cache collisions during iteration
_ext = load_inline(
    name="topk_bitonic_custom",
    cpp_sources=[CPP_WRAPPER],
    cuda_sources=[CUDA_KERNEL],
    functions=["topk_forward"],
    verbose=False,
    extra_cuda_cflags=[
        "-O3",
        "--use_fast_math",
        "-Xptxas=-O3",
        "-lineinfo",
    ],
)

class Model(nn.Module):
    """Top-k over the last dim of a 2D tensor using custom CUDA segmented scan + merge kernel."""

    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        # dummy buffer for state_dict compatibility
        self.register_buffer("_dummy", torch.zeros(1))
        # max segs for prealloc (lazy)
        self._max_segs = 64
        self._group_vals = None
        self._group_idxs = None

    def _get_groups(self, device):
        if (self._group_vals is None or self._group_vals.device != device or
            self._group_vals.shape[0] != self.batch or self._group_vals.shape[2] != self.k):
            self._group_vals = torch.empty((self.batch, self._max_segs, self.k), dtype=torch.float32, device=device)
            self._group_idxs = torch.empty((self.batch, self._max_segs, self.k), dtype=torch.int32, device=device)
        return self._group_vals, self._group_idxs

    def forward(self, x: torch.Tensor):
        device = x.device
        # choose segs for better parallelism on low-batch large-n cases
        if self.batch <= 1:
            segs = 32
        elif self.batch <= 4:
            segs = 16
        elif self.batch <= 16:
            segs = 4
        else:
            segs = 1
        segs = min(segs, max(1, (self.n + 255) // 256))  # don't over seg
        gv, gi = self._get_groups(device)
        # zero? not needed, kernel writes what it uses
        out = _ext.topk_forward(x, self.k, segs, gv, gi)
        return out[0], out[1]
