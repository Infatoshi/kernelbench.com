"""Kimi Delta Attention (KDA) forward, chunk form — custom Triton kernels for B200 (SM100).

Two-kernel design (matches the FLA chunk decomposition, reimplemented from scratch):
  1. kda_intra_kernel  — parallel over (chunk, b*h). Builds the intra-chunk
     delta-rule inverse Ainv=(I-A0)^{-1}, the causal intra-attention Aqk, and the
     per-chunk projections w,u,qg,kg + the chunk-end decay gd.
  2. kda_recur_kernel  — parallel over (v-tile, b*h). Fused state recurrence +
     output: sequential over chunks, S(K,BV) in registers, no h/v_new round-trip.

Math mirrors reference.py exactly (natural exp, in-chunk cumsum of g applied here).
"""
from __future__ import annotations
import torch
import triton
import triton.language as tl

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["B200"]


# ---------------------------------------------------------------------------
# Block forward-substitution inverse of (I - A0) for a 16x16 strictly-lower-tri
# diagonal block. b_P is positive, strict-lower-masked (= -A0_diag). Returns
# (I - A0_diag)^{-1}. 15 sequential row updates on a tiny [16,16] tile.
# ---------------------------------------------------------------------------
@triton.jit
def _invert_diag(b_P, BC: tl.constexpr):
    ar = tl.arange(0, BC)
    A0 = -b_P
    b_Ai = A0
    for i in tl.static_range(1, BC):
        m_i = (ar == i)
        row = tl.sum(A0 * m_i[:, None], axis=0)
        new_row = row + tl.sum(row[:, None] * b_Ai, axis=0)
        b_Ai = tl.where(m_i[:, None], new_row, b_Ai)
    b_Ai = b_Ai + (ar[:, None] == ar[None, :]).to(tl.float32)
    return b_Ai


@triton.jit
def kda_intra_kernel(
    q_ptr, k_ptr, v_ptr, g_ptr, beta_ptr,
    w_ptr, u_ptr, qg_ptr, kg_ptr, Aqk_ptr, gd_ptr,
    scale, T, NT,
    H: tl.constexpr, K: tl.constexpr, V: tl.constexpr,
    BT: tl.constexpr, BC: tl.constexpr,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H
    NK: tl.constexpr = BT // BC   # 4

    arc = tl.arange(0, BC)
    m_strict = arc[:, None] > arc[None, :]      # within-block strict-lower (row>col)
    m_lowinc = arc[:, None] >= arc[None, :]     # within-block lower-incl-diag

    q_b = q_ptr + i_b * T * H * K + i_h * K
    k_b = k_ptr + i_b * T * H * K + i_h * K
    g_b = g_ptr + i_b * T * H * K + i_h * K
    v_b = v_ptr + i_b * T * H * V + i_h * V
    beta_b = beta_ptr + i_b * T * H + i_h
    w_b = w_ptr + i_b * T * H * K + i_h * K
    u_b = u_ptr + i_b * T * H * V + i_h * V
    qg_b = qg_ptr + i_b * T * H * K + i_h * K
    kg_b = kg_ptr + i_b * T * H * K + i_h * K
    Aqk_b = Aqk_ptr + i_b * T * H * BT + i_h * BT
    gd_b = gd_ptr + (i_b * NT + i_t) * H * K + i_h * K

    t0 = i_t * BT

    # ---- load g in NC sub-blocks, cumsum across the whole chunk via running prefix
    b_g = []
    b_gc = []
    b_exp_g = []
    b_neg_g = []
    prefix = tl.zeros([K], dtype=tl.float32)
    for I in tl.static_range(NK):
        p_g = tl.make_block_ptr(g_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        gI = tl.load(p_g).to(tl.float32)
        b_g.append(gI)
        gcI = tl.cumsum(gI, axis=0) + prefix[None, :]
        b_gc.append(gcI)
        b_exp_g.append(tl.exp(gcI))
        b_neg_g.append(tl.exp(-gcI))
        prefix = prefix + tl.sum(gI, axis=0)
    gd = tl.exp(prefix)                          # exp(g_last), [K]

    # ---- load k, q, beta per sub-block; form kc,ki,kcq ----
    b_kc = []
    b_ki = []
    b_kcq = []
    b_beta = []
    for I in tl.static_range(NK):
        p_k = tl.make_block_ptr(k_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        p_q = tl.make_block_ptr(q_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        p_be = tl.make_block_ptr(beta_b, (T,), (H,), (t0 + I * BC,), (BC,), (0,))
        kI = tl.load(p_k).to(tl.float32)
        qI = tl.load(p_q).to(tl.float32) * scale
        b_kc.append(kI * b_exp_g[I])
        b_ki.append(kI * b_neg_g[I])
        b_kcq.append(qI * b_exp_g[I])
        b_beta.append(tl.load(p_be).to(tl.float32))

    # ---- P_IJ = beta_I[:,None] * (ki_I @ kc_J^T), positive (= -A0), for I>=J ----
    # diagonal blocks masked strict-lower; off-diagonal (I>J) full.
    b_P = [[None] * NK for _ in range(NK)]
    for I in tl.static_range(NK):
        for J in tl.static_range(NK):
            if I < J:
                continue
            blk = b_beta[I][:, None] * tl.dot(b_ki[I], tl.trans(b_kc[J]))
            if I == J:
                blk = tl.where(m_strict, blk, 0.0)
            b_P[I][J] = blk

    # ---- invert the 4 diagonal blocks, merge off-diagonals -> Ainv blocks ----
    b_Ai = [[None] * NK for _ in range(NK)]
    for I in tl.static_range(NK):
        b_Ai[I][I] = _invert_diag(b_P[I][I], BC)
    # 2x2 block-pair merges, then 3, then 4
    b_Ai[1][0] = -tl.dot(tl.dot(b_Ai[1][1], b_P[1][0]), b_Ai[0][0])
    b_Ai[2][1] = -tl.dot(tl.dot(b_Ai[2][2], b_P[2][1]), b_Ai[1][1])
    b_Ai[2][0] = -tl.dot(b_Ai[2][2], tl.dot(b_P[2][0], b_Ai[0][0]) + tl.dot(b_P[2][1], b_Ai[1][0]))
    b_Ai[3][2] = -tl.dot(tl.dot(b_Ai[3][3], b_P[3][2]), b_Ai[2][2])
    b_Ai[3][1] = -tl.dot(b_Ai[3][3], tl.dot(b_P[3][1], b_Ai[1][1]) + tl.dot(b_P[3][2], b_Ai[2][1]))
    b_Ai[3][0] = -tl.dot(
        b_Ai[3][3],
        tl.dot(b_P[3][0], b_Ai[0][0]) + tl.dot(b_P[3][1], b_Ai[1][0]) + tl.dot(b_P[3][2], b_Ai[2][0]),
    )

    # ---- precompute beta_J*v_J blocks (v loaded once each) ----
    b_bkv = []
    for J in tl.static_range(NK):
        p_vJ = tl.make_block_ptr(v_b, (T, V), (H * V, 1), (t0 + J * BC, 0), (BC, V), (1, 0))
        b_bkv.append(b_beta[J][:, None] * tl.load(p_vJ).to(tl.float32))
    b_bkc = [b_beta[J][:, None] * b_kc[J] for J in tl.static_range(NK)]

    # ---- w_I = sum_{J<=I} Ainv_IJ @ (beta_J*kc_J) ; u_I = sum_{J<=I} Ainv_IJ @ (beta_J*v_J)
    for I in tl.static_range(NK):
        bw = tl.dot(b_Ai[I][0], b_bkc[0])
        bu = tl.dot(b_Ai[I][0], b_bkv[0])
        for J in tl.static_range(1, NK):
            if J > I:
                continue
            bw += tl.dot(b_Ai[I][J], b_bkc[J])
            bu += tl.dot(b_Ai[I][J], b_bkv[J])
        p_w = tl.make_block_ptr(w_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        p_u = tl.make_block_ptr(u_b, (T, V), (H * V, 1), (t0 + I * BC, 0), (BC, V), (1, 0))
        tl.store(p_w, bw.to(p_w.dtype.element_ty))
        tl.store(p_u, bu.to(p_u.dtype.element_ty))

    # ---- qg = kcq ; kg = k*gd*neg_g = gd * ki  (since ki = k*neg_g) ----
    gd2 = gd[None, :]
    for I in tl.static_range(NK):
        p_qg = tl.make_block_ptr(qg_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        p_kg = tl.make_block_ptr(kg_b, (T, K), (H * K, 1), (t0 + I * BC, 0), (BC, K), (1, 0))
        tl.store(p_qg, b_kcq[I].to(p_qg.dtype.element_ty))
        tl.store(p_kg, (gd2 * b_ki[I]).to(p_kg.dtype.element_ty))

    # ---- Aqk_IJ = ki_I @ kcq_J^T, for I>=J (diag lower-incl); store into (T,BT) ----
    for I in tl.static_range(NK):
        for J in tl.static_range(NK):
            if I < J:
                continue
            blk = tl.dot(b_ki[I], tl.trans(b_kcq[J]))
            if I == J:
                blk = tl.where(m_lowinc, blk, 0.0)
            p_A = tl.make_block_ptr(Aqk_b, (T, BT), (H * BT, 1), (t0 + I * BC, J * BC), (BC, BC), (1, 0))
            tl.store(p_A, blk.to(p_A.dtype.element_ty))

    tl.store(gd_b + tl.arange(0, K), gd)


@triton.jit
def kda_recur_kernel(
    w_ptr, u_ptr, qg_ptr, kg_ptr, Aqk_ptr, gd_ptr, o_ptr,
    T, NT,
    H: tl.constexpr, K: tl.constexpr, V: tl.constexpr,
    BT: tl.constexpr, BV: tl.constexpr,
):
    i_v = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    w_b = w_ptr + i_b * T * H * K + i_h * K
    u_b = u_ptr + i_b * T * H * V + i_h * V
    qg_b = qg_ptr + i_b * T * H * K + i_h * K
    kg_b = kg_ptr + i_b * T * H * K + i_h * K
    Aqk_b = Aqk_ptr + i_b * T * H * BT + i_h * BT
    gd_b = gd_ptr + i_b * NT * H * K + i_h * K
    o_b = o_ptr + i_b * T * H * V + i_h * V

    # state S[K,BV] in registers, carried across chunks (no K-tiling: K=128 fits one tile)
    b_S = tl.zeros([K, BV], dtype=tl.float32)

    for n in range(NT):
        t0 = n * BT
        p_w = tl.make_block_ptr(w_b, (T, K), (H * K, 1), (t0, 0), (BT, K), (1, 0))
        p_qg = tl.make_block_ptr(qg_b, (T, K), (H * K, 1), (t0, 0), (BT, K), (1, 0))
        p_kg = tl.make_block_ptr(kg_b, (T, K), (H * K, 1), (t0, 0), (BT, K), (1, 0))
        p_Aqk = tl.make_block_ptr(Aqk_b, (T, BT), (H * BT, 1), (t0, 0), (BT, BT), (1, 0))
        p_u = tl.make_block_ptr(u_b, (T, V), (H * V, 1), (t0, i_v * BV), (BT, BV), (1, 0))
        p_o = tl.make_block_ptr(o_b, (T, V), (H * V, 1), (t0, i_v * BV), (BT, BV), (1, 0))

        b_w = tl.load(p_w)
        b_qg = tl.load(p_qg)
        b_kg = tl.load(p_kg)
        b_Aqk = tl.load(p_Aqk)
        b_u = tl.load(p_u)

        # v_new = u - w@S ; o = qg@S + Aqk@v_new
        b_vnew = (b_u - tl.dot(b_w, b_S)).to(b_Aqk.dtype)
        b_o = tl.dot(b_qg, b_S) + tl.dot(b_Aqk, b_vnew)
        tl.store(p_o, b_o.to(p_o.dtype.element_ty))

        # S = S*gd + kg^T @ v_new
        b_gd = tl.load(gd_b + n * H * K + tl.arange(0, K))      # [K]
        b_S = b_S * b_gd[:, None] + tl.dot(tl.trans(b_kg), b_vnew)


class Model(torch.nn.Module):
    def __init__(self, B, T, H, K, V, chunk_size=64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        B, T, H, K, V = self.B, self.T, self.H, self.K, self.V
        BT = self.chunk_size
        assert T % BT == 0
        NT = T // BT
        device = q.device

        w = torch.empty(B, T, H, K, dtype=torch.float32, device=device)
        u = torch.empty(B, T, H, V, dtype=torch.float32, device=device)
        qg = torch.empty(B, T, H, K, dtype=torch.float32, device=device)
        kg = torch.empty(B, T, H, K, dtype=torch.float32, device=device)
        Aqk = torch.empty(B, T, H, BT, dtype=torch.float32, device=device)
        gd = torch.empty(B, NT, H, K, dtype=torch.float32, device=device)
        o = torch.empty(B, T, H, V, dtype=v.dtype, device=device)

        grid_intra = (NT, B * H)
        kda_intra_kernel[grid_intra](
            q, k, v, g, beta, w, u, qg, kg, Aqk, gd,
            self.scale, T, NT, H=H, K=K, V=V, BT=BT, BC=16, num_warps=4,
        )

        BV = 64
        NV = V // BV
        grid_recur = (NV, B * H)
        kda_recur_kernel[grid_recur](
            w, u, qg, kg, Aqk, gd, o,
            T, NT, H=H, K=K, V=V, BT=BT, BV=BV, num_warps=4,
        )
        return o


# module-level shape shims (overridden by check.py / benchmark.py per shape)
B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
