"""Custom fp32-accumulated softmax for RTX PRO 6000 (SM120 Blackwell).

Strategy:
- Small / medium vocab (V <= 16384): single-pass single-program-per-row
  fused kernel. Whole row lives in registers; one global read, one global
  write.
- Large vocab (V > 16384): two-pass online-softmax with multi-program-per-row
  for SM saturation.
    Pass 1 (`stats_kernel`):    compute per-chunk (m_local, log-sum-exp)
                                base, write to small stats buffers.
    Pass 2 (`normalize_kernel`):reduce stats to the row's global m and s
                                (parallel tree across chunks), reload x,
                                emit y = exp(x - m_global) / s_global.

Numerical notes:
- All reductions accumulate in fp32 with Triton's tree-reduce, which gives
  O(log N * eps) error -- well below the 1e-5 tolerance even at V = 262144.
- Subtract the row max before exp on every block, then rescale block sums
  by exp(m_local - m_global) when combining. This keeps any logit, however
  large (e.g. 30.0 spike), from overflowing fp32.
- We never re-exp the input on pass 1 to write intermediates; pass 2
  recomputes exp(x - m_global) from a re-loaded x. The second read hits L2
  in practice (rows are <= 1MB and L2 on Blackwell consumer is 96 MB).

CUDAGraph capture in `forward()` removes per-call launch overhead; if the
input pointer changes (different x each call), we re-capture lazily.
"""

import torch.nn as nn
import triton
import triton.language as tl

SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# Kernels

@triton.jit
def softmax_single_kernel(
    x_ptr, y_ptr,
    stride_x: tl.constexpr,
    stride_y: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
):
    """One program per row; whole row in registers."""
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols
    x = tl.load(
        x_ptr + row * stride_x + offs,
        mask=mask,
        other=-float("inf"),
        cache_modifier=".ca",
    )
    m = tl.max(x, axis=0)
    e = tl.exp(x - m)
    e = tl.where(mask, e, 0.0)
    s = tl.sum(e, axis=0)
    y = e / s
    tl.store(
        y_ptr + row * stride_y + offs,
        y,
        mask=mask,
        cache_modifier=".cs",
    )


@triton.jit
def stats_kernel(
    x_ptr,
    m_out_ptr,
    s_out_ptr,
    stride_x: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
    NUM_BLOCKS: tl.constexpr,
):
    """Pass 1: per-chunk (m_local, sum_exp_local)."""
    row = tl.program_id(0)
    chunk = tl.program_id(1)
    offs = chunk * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols
    x = tl.load(
        x_ptr + row * stride_x + offs,
        mask=mask,
        other=-float("inf"),
        cache_modifier=".ca",
    )
    m_local = tl.max(x, axis=0)
    e = tl.exp(x - m_local)
    e = tl.where(mask, e, 0.0)
    s_local = tl.sum(e, axis=0)
    out = row * NUM_BLOCKS + chunk
    tl.store(m_out_ptr + out, m_local)
    tl.store(s_out_ptr + out, s_local)


@triton.jit
def normalize_kernel(
    x_ptr,
    y_ptr,
    m_in_ptr,
    s_in_ptr,
    stride_x: tl.constexpr,
    stride_y: tl.constexpr,
    n_cols: tl.constexpr,
    BLOCK_SIZE: tl.constexpr,
    NUM_BLOCKS: tl.constexpr,
):
    """Pass 2: read all per-chunk stats, fuse to global (m, s), write y."""
    row = tl.program_id(0)
    chunk = tl.program_id(1)
    base = row * NUM_BLOCKS

    # Load every chunk's (m_local, s_local) and reduce in registers using a
    # parallel tree (Triton's tl.max / tl.sum over a constexpr range).
    block_offs = tl.arange(0, NUM_BLOCKS)
    m_blocks = tl.load(m_in_ptr + base + block_offs)
    s_blocks = tl.load(s_in_ptr + base + block_offs)
    m_global = tl.max(m_blocks, axis=0)
    s_global = tl.sum(s_blocks * tl.exp(m_blocks - m_global), axis=0)

    offs = chunk * BLOCK_SIZE + tl.arange(0, BLOCK_SIZE)
    mask = offs < n_cols
    x = tl.load(
        x_ptr + row * stride_x + offs,
        mask=mask,
        other=0.0,
        cache_modifier=".ca",
    )
    y = tl.exp(x - m_global) / s_global
    tl.store(
        y_ptr + row * stride_y + offs,
        y,
        mask=mask,
        cache_modifier=".cs",
    )

class Model(nn.Module):
    def __init__(self, batch: int, vocab: int):
        super().__init__()
        self.batch = batch
        self.vocab = vocab
        self._cfg = _pick_config(vocab)
        if self._cfg["mode"] == "two_pass":
            self._num_blocks = triton.cdiv(vocab, self._cfg["BLOCK_SIZE"])
        else:
            self._num_blocks = 1
        self._graph: torch.cuda.CUDAGraph | None = None
        self._graph_ptr: int = 0
        self._graph_y: torch.Tensor | None = None
        self._graph_m: torch.Tensor | None = None
        self._graph_s: torch.Tensor | None = None

    def _launch(
        self,
        x: torch.Tensor,
        y: torch.Tensor,
        m_buf: torch.Tensor | None,
        s_buf: torch.Tensor | None,
    ) -> None:
        cfg = self._cfg
        if cfg["mode"] == "single":
            softmax_single_kernel[(self.batch,)](
                x, y,
                stride_x=x.stride(0),
                stride_y=y.stride(0),
                n_cols=self.vocab,
                BLOCK_SIZE=cfg["BLOCK_SIZE"],
                num_warps=cfg["num_warps"],
            )
            return

        grid = (self.batch, self._num_blocks)
        stats_kernel[grid](
            x, m_buf, s_buf,
            stride_x=x.stride(0),
            n_cols=self.vocab,
            BLOCK_SIZE=cfg["BLOCK_SIZE"],
            NUM_BLOCKS=self._num_blocks,
            num_warps=cfg["num_warps_stats"],
        )
        normalize_kernel[grid](
            x, y, m_buf, s_buf,
            stride_x=x.stride(0),
            stride_y=y.stride(0),
            n_cols=self.vocab,
            BLOCK_SIZE=cfg["BLOCK_SIZE"],
            NUM_BLOCKS=self._num_blocks,
            num_warps=cfg["num_warps_norm"],
        )

    def _make_buffers(self, x: torch.Tensor):
        y = torch.empty_like(x)
        if self._cfg["mode"] == "two_pass":
            n = self.batch * self._num_blocks
            m_buf = torch.empty(n, dtype=torch.float32, device=x.device)
            s_buf = torch.empty(n, dtype=torch.float32, device=x.device)
            return y, m_buf, s_buf
        return y, None, None

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Hot path: same input pointer => just replay the captured graph.
        # Bypass the broader nn.Module.__call__ hooks for the per-call overhead;
        # see __call__ override below.
        ptr = x.data_ptr()
        if self._graph is not None and ptr == self._graph_ptr:
            self._graph.replay()
            return self._graph_y
        # Slow path: first call OR input pointer changed (e.g. new test seed).
        y, m_buf, s_buf = self._make_buffers(x)
        self._launch(x, y, m_buf, s_buf)
        torch.cuda.synchronize()
        graph = torch.cuda.CUDAGraph()
        with torch.cuda.graph(graph):
            self._launch(x, y, m_buf, s_buf)
        self._graph = graph
        self._graph_ptr = ptr
        self._graph_y = y
        self._graph_m = m_buf
        self._graph_s = s_buf
        return y

    # Bypass nn.Module.__call__ overhead (hooks, etc.) on the hot path. The
    # benchmark calls model(x) -> __call__; we route directly to forward.
    __call__ = forward

def compute_tflops(flops: float, time_ms: float) -> float:
    """flops per invocation and wall time in ms -> TFLOPS."""

    """bytes per invocation and wall time in ms -> GB/s."""

+Strategy:
+- Small / medium vocab (V <= 16384): single-pass single-program-per-row
+  fused kernel. Whole row lives in registers; one global read, one global
+  write.
+- Large vocab (V > 16384): two-pass online-softmax with multi-program-per-row
+  for SM saturation.
+    Pass 1 (`stats_kernel`):    compute per-chunk (m_local, log-sum-exp)
+                                base, write to small stats buffers.
+    Pass 2 (`normalize_kernel`):reduce stats to the row's global m and s
+                                (parallel tree across chunks), reload x,
+                                emit y = exp(x - m_global) / s_global.
+
+Numerical notes:
+- All reductions accumulate in fp32 with Triton's tree-reduce, which gives
+  O(log N * eps) error -- well below the 1e-5 tolerance even at V = 262144.
+- Subtract the row max before exp on every block, then rescale block sums
+  by exp(m_local - m_global) when combining. This keeps any logit, however
+  large (e.g. 30.0 spike), from overflowing fp32.
+- We never re-exp the input on pass 1 to write intermediates; pass 2
+  recomputes exp(x - m_global) from a re-loaded x. The second read hits L2
+  in practice (rows are <= 1MB and L2 on Blackwell consumer is 96 MB).
