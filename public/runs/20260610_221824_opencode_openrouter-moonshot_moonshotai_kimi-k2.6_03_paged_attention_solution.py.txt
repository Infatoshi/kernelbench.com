"""Triton paged-attention decode kernel for Blackwell/SM120.

Strategy: one CUDA block per (batch element, query head).  Each block streams
through the paged KV cache token-by-token, carrying a single-query-head online
softmax accumulator in fp32 and writing the final bf16 result.  The loop is
kept simple so that Triton can vectorise the loads and the dominant cost
(rematerialising K/V from DRAM/L2) is overlapped with the tiny amount of math.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


# ---------------------------------------------------------------------------
# Triton kernel
# ---------------------------------------------------------------------------
@triton.jit
def _paged_decode_kernel(
    out_ptr,
    q_ptr,
    kv_cache_ptr,
    block_table_ptr,
    seq_lens_ptr,
    scale: tl.constexpr,
    num_heads: tl.constexpr,
    num_kv_heads: tl.constexpr,
    head_dim: tl.constexpr,
    page_size: tl.constexpr,
    max_blocks_per_seq: tl.constexpr,
):
    """Single-query-head paged decode."""
    batch_id = tl.program_id(0)
    head_id = tl.program_id(1)

    kv_head_id = head_id // (num_heads // num_kv_heads)
    seq_len = tl.load(seq_lens_ptr + batch_id)
    num_pages = tl.cdiv(seq_len, page_size)

    offs_d = tl.arange(0, head_dim)

    q_base = batch_id * num_heads * head_dim + head_id * head_dim
    q = tl.load(q_ptr + q_base + offs_d)
    qf = q.to(tl.float32)

    m = -float("inf")
    l = 0.0
    acc = tl.zeros((head_dim,), dtype=tl.float32)

    for p in range(num_pages):
        block_id = tl.load(block_table_ptr + batch_id * max_blocks_per_seq + p)

        for tok in tl.static_range(page_size):
            pos = p * page_size + tok
            if pos < seq_len:
                kv_off = (
                    block_id * page_size * num_kv_heads * head_dim * 2
                    + tok * num_kv_heads * head_dim * 2
                    + kv_head_id * head_dim * 2
                )

                k = tl.load(kv_cache_ptr + kv_off + offs_d)
                v = tl.load(kv_cache_ptr + kv_off + head_dim + offs_d)

                kf = k.to(tl.float32)
                vf = v.to(tl.float32)

                score = tl.sum(qf * kf) * scale

                m_new = tl.maximum(m, score)
                alpha = tl.exp(m - m_new)
                beta = tl.exp(score - m_new)

                acc = acc * alpha + beta * vf
                l = l * alpha + beta
                m = m_new

    acc = acc / l
    tl.store(out_ptr + q_base + offs_d, acc.to(tl.bfloat16))


# ---------------------------------------------------------------------------
# Model wrapper
# ---------------------------------------------------------------------------
class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "GQA required"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.scale = 1.0 / math.sqrt(head_dim)

        self.register_buffer(
            "_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        P = self.page_size
        max_blocks = block_table.shape[1]

        out = torch.empty_like(query)

        grid = (B, H)
        _paged_decode_kernel[grid](
            out,
            query,
            kv_cache,
            block_table,
            seq_lens,
            scale=self.scale,
            num_heads=H,
            num_kv_heads=Hkv,
            head_dim=D,
            page_size=P,
            max_blocks_per_seq=max_blocks,
            num_warps=max(1, D // 32),
        )
        return out


def get_inputs():
    """Build random paged inputs matching the reference shape knobs."""
    # We read the *module-level* knobs that check.py overwrites.
    import reference as _ref

    B = _ref.BATCH
    H = _ref.NUM_HEADS
    Hkv = _ref.NUM_KV_HEADS
    D = _ref.HEAD_DIM
    L = _ref.SEQ_LEN
    P = _ref.PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = (
        torch.randperm(total_pages)[: B * pages_per_seq]
        .reshape(B, pages_per_seq)
        .int()
    )
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    import reference as _ref

    return [
        _ref.BATCH,
        _ref.NUM_HEADS,
        _ref.NUM_KV_HEADS,
        _ref.HEAD_DIM,
        _ref.SEQ_LEN,
        _ref.PAGE_SIZE,
    ]
