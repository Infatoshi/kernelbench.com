"""Single-pass top-k for RTX 3090 (SM86).

Block-per-row kernel: each thread streams a strided slice of its row,
maintaining a sorted-descending register buffer of size K. A shared-memory
merge-tree combines per-thread buffers into the block top-K (sorted desc).

For very large rows (n > 32768) a segmented two-launch path uses multiple
blocks per row to keep enough warps in flight for DRAM bandwidth.
"""

template <int K>
__device__ __forceinline__ void insert_topk(float* tv, int* ti, float v, int gi) {
    if (v <= tv[K - 1]) return;
    float carry = v;
    int carry_i = gi;
    bool active = false;
    #pragma unroll
    for (int i = 0; i < K; i++) {
        if (carry > tv[i]) active = true;
        if (active) {
            float tf = tv[i];
            tv[i] = carry;
            carry = tf;
            int t2 = ti[i];
            ti[i] = carry_i;
            carry_i = t2;
        }
    }
}

template <int BLOCK, int K>
__device__ __forceinline__ void merge_tree(float* sv, int* si, int nbuf) {
    constexpr int S = K + 1;
    for (int d = 1; d < nbuf; d <<= 1) {
        for (int base = threadIdx.x * (2 * d); base + d < nbuf;
             base += BLOCK * (2 * d)) {
            float* A = sv + base * S;
            int* Ai = si + base * S;
            float* B = sv + (base + d) * S;
            int* Bi = si + (base + d) * S;
            float tmp[K];
            int ti2[K];
            int ia = 0, ib = 0;
            #pragma unroll
            for (int c = 0; c < K; c++) {
                float va = A[ia], vb = B[ib];
                if (va >= vb) {
                    tmp[c] = va;
                    ti2[c] = Ai[ia];
                    ia++;
                } else {
                    tmp[c] = vb;
                    ti2[c] = Bi[ib];
                    ib++;
                }
            }
            #pragma unroll
            for (int c = 0; c < K; c++) {
                A[c] = tmp[c];
                Ai[c] = ti2[c];
            }
        }
        __syncthreads();
    }
}

template <int K, int BLOCK>
__global__ void topk_reduce_kernel(
    const float* __restrict__ x, float* __restrict__ out_vals,
    long long* __restrict__ out_idx, float* __restrict__ tmp_vals,
    long long* __restrict__ tmp_idx, int batch, int n, int nseg, int seg_size)
{
    const int row = blockIdx.x;
    const int seg = blockIdx.y;
    if (row >= batch) return;
    const int seg_start = seg * seg_size;
    const int seg_end = (seg_start + seg_size < n) ? (seg_start + seg_size) : n;
    const float* xrow = x + (size_t)row * n;

    constexpr int M = BLOCK * (K + 1);
    extern __shared__ float smem[];
    float* sv = smem;
    int* si = reinterpret_cast<int*>(sv + M);

    float tv[K];
    int ti[K];
    #pragma unroll
    for (int i = 0; i < K; i++) {
        tv[i] = -INFINITY;
        ti[i] = 0;
    }

    constexpr int UTILE = 8;
    const int len = seg_end - seg_start;
    const int stride = BLOCK * UTILE;
    int full = (len / stride) * stride;
    for (int base = 0; base < full; base += stride) {
        float vb[UTILE];
        int ib[UTILE];
        #pragma unroll
        for (int j = 0; j < UTILE; j++) {
            int q = base + threadIdx.x + j * BLOCK;
            vb[j] = __ldg(&xrow[seg_start + q]);
            ib[j] = seg_start + q;
        }
        #pragma unroll
        for (int j = 0; j < UTILE; j++) {
            insert_topk<K>(tv, ti, vb[j], ib[j]);
        }
    }
    for (int q = full + threadIdx.x; q < len; q += BLOCK) {
        insert_topk<K>(tv, ti, xrow[seg_start + q], seg_start + q);
    }

    #pragma unroll
    for (int i = 0; i < K; i++) {
        int off = threadIdx.x * (K + 1) + i;
        sv[off] = tv[i];
        si[off] = ti[i];
    }
    __syncthreads();

    merge_tree<BLOCK, K>(sv, si, BLOCK);

            out_vals[(size_t)row * K + i] = sv[i];
            out_idx[(size_t)row * K + i] = (long long)si[i];
        }
    } else {

            tmp_vals[slot + i] = sv[i];
            tmp_idx[slot + i] = (long long)si[i];
        }
    }
}

template <int K, int BLOCK>
__global__ void topk_merge_kernel(
    const float* __restrict__ tmp_vals, const long long* __restrict__ tmp_idx,
    float* __restrict__ out_vals, long long* __restrict__ out_idx, int nseg)
{
    const int row = blockIdx.x;
    const size_t base = (size_t)row * nseg * K;

    constexpr int S = K + 1;
    constexpr int M = BLOCK * S;
    extern __shared__ float smem[];
    float* sv = smem;
    int* si = reinterpret_cast<int*>(sv + M);

    int nbuf = nseg;
    int np = 1;
    while (np < nbuf) np <<= 1;
    nbuf = np;
    for (int b = threadIdx.x; b < nbuf; b += BLOCK) {
        int dst = b * S;
        if (b < nseg) {
            int src = b * K;
            #pragma unroll
            for (int i = 0; i < K; i++) {
                sv[dst + i] = tmp_vals[base + src + i];
                si[dst + i] = (int)tmp_idx[base + src + i];
            }
        } else {
            #pragma unroll
            for (int i = 0; i < K; i++) {
                sv[dst + i] = -INFINITY;
                si[dst + i] = 0;
            }
        }
    }
    __syncthreads();

    merge_tree<BLOCK, K>(sv, si, nbuf);

    for (int i = threadIdx.x; i < K; i += BLOCK) {
        out_vals[(size_t)row * K + i] = sv[i];
        out_idx[(size_t)row * K + i] = (long long)si[i];
    }
}

template <int BLOCK>
__global__ void argmax_kernel(
    const float* __restrict__ x, float* __restrict__ out_vals,
    long long* __restrict__ out_idx, int batch, int n)
{
    const int row = blockIdx.x;
    if (row >= batch) return;
    const float* xr = x + (size_t)row * n;
    const int tid = threadIdx.x;
    const int lane = tid & 31;
    const int wid = tid >> 5;
    constexpr int NWARPS = BLOCK / 32;

    float best = -INFINITY;
    int best_i = 0;

    constexpr int UTILE = 16;
    const int stride = BLOCK * UTILE;
    int full = (n / stride) * stride;
    for (int base = 0; base < full; base += stride) {
        #pragma unroll
        for (int j = 0; j < UTILE; j++) {
            int p = base + tid + j * BLOCK;
            float v = __ldg(&xr[p]);
            if (v > best) {
                best = v;
                best_i = p;
            }
        }
    }
    for (int p = full + tid; p < n; p += BLOCK) {
        float v = __ldg(&xr[p]);
        if (v > best) {
            best = v;
            best_i = p;
        }
    }

    #pragma unroll
    for (int offset = 16; offset > 0; offset >>= 1) {
        float ov = __shfl_down_sync(0xffffffffu, best, offset);
        int oi = __shfl_down_sync(0xffffffffu, best_i, offset);
        if (ov > best) {
            best = ov;
            best_i = oi;
        }
    }

    __shared__ float sw_val[32];
    __shared__ int sw_idx[32];
    if (lane == 0) {
        sw_val[wid] = best;
        sw_idx[wid] = best_i;
    }
    __syncthreads();

    if (wid == 0) {
        float bv = (lane < NWARPS) ? sw_val[lane] : -INFINITY;
        int bi = (lane < NWARPS) ? sw_idx[lane] : 0;
        #pragma unroll
        for (int offset = 16; offset > 0; offset >>= 1) {
            float ov = __shfl_down_sync(0xffffffffu, bv, offset);
            int oi = __shfl_down_sync(0xffffffffu, bi, offset);
            if (ov > bv) {
                bv = ov;
                bi = oi;
            }
        }
        if (lane == 0) {
            out_vals[row] = bv;
            out_idx[row] = (long long)bi;
        }
    }
}
