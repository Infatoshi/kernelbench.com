"""Grouped GEMM + fused SwiGLU for MoE up-projection on Blackwell (SM120).

Implements: h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
using Triton kernel with grouped GEMM.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def triton_gemm_kernel(
    a_ptr, b_ptr, c_ptr,
    N_e, I, H,
    stride_a_row, stride_a_col,
    stride_b_row, stride_b_col,
    stride_c_row, stride_c_col,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(N_e, BLOCK_M)
    num_pid_n = tl.cdiv(I, BLOCK_N)
    num_pid_in_group = num_pid_m * num_pid_n
    group_id = pid // num_pid_in_group
    first_pid_m = group_id * num_pid_m
    group_size_m = min(num_pid_m, N_e - first_pid_m)

    pid_m = first_pid_m + (pid % group_size_m)
    pid_n = (pid % num_pid_in_group) // group_size_m

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    offs_k = tl.arange(0, BLOCK_K)

    a_ptrs = a_ptr + offs_m[:, None] * stride_a_row + offs_k[None, :] * stride_a_col
    b_ptrs = b_ptr + offs_k[:, None] * stride_b_row + offs_n[None, :] * stride_b_col

    accumulator = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(H, BLOCK_K)):
        mask_m = offs_m[:, None] < N_e
        mask_k = offs_k[None, :] < H
        a = tl.load(a_ptrs, mask=mask_m & mask_k, other=0.0)

        mask_k = offs_k[:, None] < H
        mask_n = offs_n[None, :] < I
        b = tl.load(b_ptrs, mask=mask_k & mask_n, other=0.0)

        accumulator += tl.dot(a, b)

        a_ptrs += BLOCK_K * stride_a_col
        b_ptrs += BLOCK_K * stride_b_row
        offs_k += BLOCK_K

    offs_m_out = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n_out = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    c_ptrs = c_ptr + offs_m_out[:, None] * stride_c_row + offs_n_out[None, :] * stride_c_col
    mask_m_out = offs_m_out[:, None] < N_e
    mask_n_out = offs_n_out[None, :] < I
    tl.store(c_ptrs, accumulator, mask=mask_m_out & mask_n_out)


def triton_grouped_gemm(hidden_states, W, expert_offsets, E):
    T_perm, H = hidden_states.shape
    I = W.shape[2]
    output = torch.empty(T_perm, I, device=hidden_states.device, dtype=hidden_states.dtype)

    for e in range(E):
        start = int(expert_offsets[e].item())
        end = int(expert_offsets[e + 1].item())
        if end == start:
            continue
        N_e = end - start

        x_e = hidden_states[start:end]
        w_e = W[e]
        out_e = output[start:end]

        grid_m = triton.cdiv(N_e, 64)
        grid_n = triton.cdiv(I, 64)
        grid = (grid_m * grid_n,)

        triton_gemm_kernel[grid](
            x_e, w_e, out_e,
            N_e, I, H,
            x_e.stride(0), 1,
            w_e.stride(0), w_e.stride(1),
            out_e.stride(0), 1,
            64, 64, 64,
        )
    return output


class Model(nn.Module):
    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(self, hidden_states: torch.Tensor, expert_offsets: torch.Tensor) -> torch.Tensor:
        gate_out = triton_grouped_gemm(hidden_states, self.W_gate, expert_offsets, self.E)
        up_out = triton_grouped_gemm(hidden_states, self.W_up, expert_offsets, self.E)
        silu_gate = torch.nn.functional.silu(gate_out)
        return silu_gate * up_out

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets
