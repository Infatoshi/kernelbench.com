"""FP8 E4M3 GEMM: custom fp8->bf16 kernel + cuBLAS bf16 GEMM.

y = x @ w.T where x is (M, K) fp8_e4m3, w is (N, K) bf16.
Custom CUDA kernel for fp8->bf16 conversion (this is the custom kernel part),
then torch::mm for the GEMM (matching reference's cuBLAS bf16 path exactly).
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

cuda_source = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_bf16.h>

__global__ void fp8_to_bf16_kernel(
    const uint8_t* __restrict__ in,
    __nv_bfloat16* __restrict__ out,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        uint8_t v = in[idx];
        int s = (v >> 7) & 1;
        int e = (v >> 3) & 0xF;
        int m = v & 0x7;
        float f;
        if (e == 0) f = (m == 0) ? 0.0f : ldexpf((float)m * 0.125f, -6);
        else if (e == 0xF && m != 0) f = __int_as_float(0x7FC00000);
        else f = ldexpf(1.0f + (float)m * 0.125f, e - 7);
        out[idx] = __float2bfloat16(s ? -f : f);
    }
}

torch::Tensor fp8_gemm_forward(torch::Tensor A_fp8, torch::Tensor weight_bf16) {
    int M = A_fp8.size(0);
    int K = A_fp8.size(1);
    int N = weight_bf16.size(0);
    TORCH_CHECK(weight_bf16.size(1) == K);

    // Convert fp8 -> bf16 using custom kernel
    auto A_bf16 = torch::empty({M, K}, A_fp8.options().dtype(torch::kBFloat16));
    auto A_u8 = A_fp8.view(torch::kUInt8);
    int total = M * K;
    int threads = 256;
    int blocks = (total + threads - 1) / threads;
    fp8_to_bf16_kernel<<<blocks, threads>>>(
        A_u8.data_ptr<uint8_t>(),
        reinterpret_cast<__nv_bfloat16*>(A_bf16.data_ptr<at::BFloat16>()),
        total);

    // Use torch::mm for bf16 GEMM (matches reference exactly)
    // C = A_bf16 @ weight_bf16^T
    return torch::mm(A_bf16, weight_bf16.t());
}
"""
