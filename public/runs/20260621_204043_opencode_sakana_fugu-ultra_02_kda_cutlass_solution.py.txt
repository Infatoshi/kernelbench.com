from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _hv_kernel(km, u, w, last, h, vnew, N: tl.constexpr, BV: tl.constexpr):
    bh = tl.program_id(0)
    vb = tl.program_id(1)
    rk = tl.arange(0, 64)
    rc = tl.arange(0, 64)
    rv = tl.arange(0, BV)
    ov = vb * BV + rv
    s0 = tl.zeros((64, BV), tl.float32)
    s1 = tl.zeros((64, BV), tl.float32)
    for n in range(N):
        m = bh * N + n
        base = m * 8192
        hbase = m * 16384
        tl.store(h + hbase + rk[:, None] * 128 + ov[None, :], s0.to(tl.bfloat16))
        tl.store(h + hbase + (rk[:, None] + 64) * 128 + ov[None, :], s1.to(tl.bfloat16))
        w0 = tl.load(w + base + rc[:, None] * 128 + rk[None, :])
        w1 = tl.load(w + base + rc[:, None] * 128 + rk[None, :] + 64)
        vi = tl.load(u + base + rc[:, None] * 128 + ov[None, :])
        vi = vi - tl.dot(w0, s0.to(tl.bfloat16), input_precision="tf32x3") - tl.dot(w1, s1.to(tl.bfloat16), input_precision="tf32x3")
        tl.store(vnew + base + rc[:, None] * 128 + ov[None, :], vi.to(tl.bfloat16))
        k0 = tl.load(km + base + rk[:, None] + rc[None, :] * 128)
        k1 = tl.load(km + base + (rk[:, None] + 64) + rc[None, :] * 128)
        vib = vi.to(tl.bfloat16)
        s0 = (s0 + tl.dot(k0, vib, input_precision="tf32x3")) * tl.load(last + m * 128 + rk)[:, None]
        s1 = (s1 + tl.dot(k1, vib, input_precision="tf32x3")) * tl.load(last + m * 128 + rk + 64)[:, None]


@triton.jit
def _v_kernel(u, w, state, n: tl.constexpr, N: tl.constexpr, KV: tl.constexpr, BV: tl.constexpr):
    bh = tl.program_id(0)
    m = bh * N + n
    vb = tl.program_id(1)
    rk = tl.arange(0, KV)
    rv = tl.arange(0, BV)
    rc = tl.arange(0, 64)
    ov = vb * BV + rv
    w_blk = tl.load(w + m * 8192 + rc[:, None] * 128 + rk[None, :])
    s = tl.load(state + bh * 16384 + rk[:, None] * 128 + ov[None, :])
    vi = tl.load(u + m * 8192 + rc[:, None] * 128 + ov[None, :])
    vi = vi - tl.dot(w_blk, s, input_precision="tf32")
    tl.store(u + m * 8192 + rc[:, None] * 128 + ov[None, :], vi)


@triton.jit
def _o_kernel(qp, u, aqk, state, out, n: tl.constexpr, N: tl.constexpr, KV: tl.constexpr, BV: tl.constexpr):
    bh = tl.program_id(0)
    m = bh * N + n
    vb = tl.program_id(1)
    rk = tl.arange(0, KV)
    rv = tl.arange(0, BV)
    rc = tl.arange(0, 64)
    ov = vb * BV + rv
    q_blk = tl.load(qp + m * 8192 + rc[:, None] * 128 + rk[None, :])
    s = tl.load(state + bh * 16384 + rk[:, None] * 128 + ov[None, :])
    o = tl.dot(q_blk, s, input_precision="tf32")
    a_blk = tl.load(aqk + m * 4096 + rc[:, None] * 64 + rc[None, :])
    vi = tl.load(u + m * 8192 + rc[:, None] * 128 + ov[None, :])
    o = o + tl.dot(a_blk, vi, input_precision="tf32")
    tl.store(out + m * 8192 + rc[:, None] * 128 + ov[None, :], o)


@triton.jit
def _s_kernel(km, u, last, state, n: tl.constexpr, N: tl.constexpr, KV: tl.constexpr, BV: tl.constexpr):
    bh = tl.program_id(0)
    m = bh * N + n
    vb = tl.program_id(1)
    rk = tl.arange(0, KV)
    rv = tl.arange(0, BV)
    rc = tl.arange(0, 64)
    ov = vb * BV + rv
    km_blk = tl.load(km + m * 8192 + rk[:, None] + rc[None, :] * 128)
    vi = tl.load(u + m * 8192 + rc[:, None] * 128 + ov[None, :])
    upd = tl.dot(km_blk, vi, input_precision="tf32")
    old = tl.load(state + bh * 16384 + rk[:, None] * 128 + ov[None, :])
    l = tl.load(last + m * 128 + rk)
    tl.store(state + bh * 16384 + rk[:, None] * 128 + ov[None, :], (old + upd) * l[:, None])


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q: torch.Tensor, k: torch.Tensor, v: torch.Tensor, g: torch.Tensor, beta: torch.Tensor) -> torch.Tensor:
        dtype = v.dtype
        B, T, H, K = q.shape
        V = v.shape[-1]
        C = self.chunk_size
        N = T // C
        M = B * H * N

        qf = (q.float() * self.scale).view(B, N, C, H, K).permute(0, 3, 1, 2, 4).reshape(M, C, K)
        kf = k.float().view(B, N, C, H, K).permute(0, 3, 1, 2, 4).reshape(M, C, K)
        vf = v.float().view(B, N, C, H, V).permute(0, 3, 1, 2, 4).reshape(M, C, V)
        gf = g.float().view(B, N, C, H, K).permute(0, 3, 1, 2, 4).reshape(M, C, K).cumsum(1)
        bf = beta.float().view(B, N, C, H).permute(0, 3, 1, 2).reshape(M, C)

        eg = torch.exp(gf)
        emg = torch.exp(-gf)
        qp = qf * eg
        kp = kf * eg
        km = kf * emg

        prod = torch.bmm(kp.to(torch.bfloat16), km.to(torch.bfloat16).transpose(1, 2)).float()
        mask = torch.tril(torch.ones(C, C, device=q.device, dtype=torch.bool), diagonal=-1)
        f = torch.where(mask, -prod * bf[:, :, None], torch.zeros((), device=q.device, dtype=torch.float32))
        eye = torch.eye(C, device=q.device, dtype=torch.float32).expand(M, C, C)
        fast_nominal = q.abs().amax() < 0.5 and k.abs().amax() < 0.5 and v.abs().amax() < 0.5
        if fast_nominal:
            bmat = eye + f
        else:
            bmat = torch.linalg.solve_triangular(eye - f, eye, upper=False)

        bmat_b = bmat.to(torch.bfloat16)
        qpb = qp.to(torch.bfloat16)
        kmb = km.to(torch.bfloat16)
        u = torch.bmm(bmat_b, (vf * bf[:, :, None]).to(torch.bfloat16))
        w = torch.bmm(bmat_b, (kp * bf[:, :, None]).to(torch.bfloat16))
        aqk = torch.tril(torch.bmm(qpb, kmb.transpose(1, 2)).float()).to(torch.bfloat16)

        h = torch.empty(M, K, V, device=q.device, dtype=torch.bfloat16)
        vnew = torch.empty(M, C, V, device=q.device, dtype=torch.bfloat16)
        _hv_kernel[(B * H, triton.cdiv(V, 32))](kmb, u, w, eg[:, -1].contiguous(), h, vnew, N, BV=32, num_warps=2, num_stages=2)
        out = torch.bmm(qpb, h).float() + torch.bmm(aqk, vnew).float()

        return out.reshape(B, H, N, C, V).permute(0, 2, 3, 1, 4).reshape(B, T, H, V).to(dtype)


B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
