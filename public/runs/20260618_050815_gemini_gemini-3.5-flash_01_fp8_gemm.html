import torch
import torch.nn as nn

class Model(nn.Module):
    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.register_buffer("weight", torch.empty(N, K, dtype=torch.float8_e4m3fn))
        self.register_buffer("weight_scale", torch.empty(N, dtype=torch.float32))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x_bf = x.to(torch.bfloat16)
        w_bf = self.weight.to(torch.bfloat16)
        y = (x_bf @ w_bf.T).float()
        y = y * self.weight_scale[None, :]
        return y.to(torch.bfloat16)

def get_inputs():
    x = (torch.rand(4096, 4096) * 8 - 4).to(torch.float8_e4m3fn)

import torch
import triton
import triton.language as tl

@triton.jit
def fp8_gemm_kernel(
    a_ptr, b_ptr, c_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bn, stride_bk,  # strides of weight (N, K)
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
):
    # Identifiers
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    pid_m = pid // num_pid_n
    pid_n = pid % num_pid_n

    # Compute blocks
    offs_am = (pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)) % M
    offs_bn = (pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)) % N
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    # Pointers
    a_ptrs = a_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    # w is (N, K). We want b of shape (K, N) which is w.T.
    # So b_ptrs[k, n] = w[n, k] = w_ptr + n * stride_bn + k * stride_bk
    b_ptrs = b_ptr + offs_bn[None, :] * stride_bn + offs_k[:, None] * stride_bk

    accumulator = tl.zeros((BLOCK_SIZE_M, BLOCK_SIZE_N), dtype=tl.float32)
    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):
        # Load a and b
        # For out-of-bounds K, we need masking
        k_remaining = K - k * BLOCK_SIZE_K
        if k_remaining < BLOCK_SIZE_K:
            mask_a = offs_k[None, :] < k_remaining
            mask_b = offs_k[:, None] < k_remaining
            a = tl.load(a_ptrs, mask=mask_a, other=0.0)
            b = tl.load(b_ptrs, mask=mask_b, other=0.0)
        else:
            a = tl.load(a_ptrs)
            b = tl.load(b_ptrs)

        # Dot product
        accumulator = tl.dot(a, b, accumulator, out_dtype=tl.float32)

        # Advance pointers
        a_ptrs += BLOCK_SIZE_K * stride_ak
        b_ptrs += BLOCK_SIZE_K * stride_bk

    # Store output c
    offs_cm = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_cn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    c_ptrs = c_ptr + offs_cm[:, None] * stride_cm + offs_cn[None, :] * stride_cn
    mask_c = (offs_cm[:, None] < M) & (offs_cn[None, :] < N)
    
    # Store as fp32 or bf16
    tl.store(c_ptrs, accumulator, mask=mask_c)

def test():
    M, N, K = 4096, 4096, 4096
    x = (torch.rand(M, K) * 8 - 4).to(torch.float8_e4m3fn).cuda()
    w = (torch.rand(N, K) * 8 - 4).to(torch.float8_e4m3fn).cuda()
    c = torch.empty((M, N), dtype=torch.float32, device="cuda")

    grid = lambda META: (triton.cdiv(M, META['BLOCK_SIZE_M']) * triton.cdiv(N, META['BLOCK_SIZE_N']),)
    
    fp8_gemm_kernel[grid](
        x, w, c,
        M, N, K,
        x.stride(0), x.stride(1),
        w.stride(0), w.stride(1),
        c.stride(0), c.stride(1),
        BLOCK_SIZE_M=128, BLOCK_SIZE_N=128, BLOCK_SIZE_K=64,
    )
    print("Kernel finished successfully!")
    ref = (x.to(torch.float16) @ w.to(torch.float16).T).to(torch.float32)
    diff = torch.abs(c - ref).max()
    print("Max diff:", diff.item())

if __name__ == "__main__":

import triton.language as tl
import math

@triton.jit
def fp8_gemm_kernel(
    a_ptr, b_ptr, c_ptr,
    weight_scale_ptr,
    M, N, K,
    stride_am, stride_ak,
    stride_bn, stride_bk,
    stride_cm, stride_cn,
    BLOCK_SIZE_M: tl.constexpr, BLOCK_SIZE_N: tl.constexpr, BLOCK_SIZE_K: tl.constexpr,
    GROUP_SIZE_M: tl.constexpr,
    EVEN_M: tl.constexpr, EVEN_N: tl.constexpr, EVEN_K: tl.constexpr,
):
    # Program ID
    pid = tl.program_id(axis=0)
    num_pid_m = tl.cdiv(M, BLOCK_SIZE_M)
    num_pid_n = tl.cdiv(N, BLOCK_SIZE_N)
    
    # L2 cache swizzling
    # Group of programs
    width = GROUP_SIZE_M * num_pid_n
    group_id = pid // width
    row_idx = group_id * GROUP_SIZE_M
    col_idx = pid % num_pid_n
    
    # Program coordinates
    pid_m = row_idx + (pid % width) // num_pid_n
    pid_n = col_idx
    
    if pid_m >= num_pid_m:
        return

    # Compute block offsets
    offs_am = pid_m * BLOCK_SIZE_M + tl.arange(0, BLOCK_SIZE_M)
    offs_bn = pid_n * BLOCK_SIZE_N + tl.arange(0, BLOCK_SIZE_N)
    offs_k = tl.arange(0, BLOCK_SIZE_K)

    # Initial pointers
    # a: (M, K)
    a_ptrs = a_ptr + offs_am[:, None] * stride_am + offs_k[None, :] * stride_ak
    # b: (K, N), which is w.T where w is (N, K)
    b_ptrs = b_ptr + offs_bn[None, :] * stride_bn + offs_k[:, None] * stride_bk

    for k in range(0, tl.cdiv(K, BLOCK_SIZE_K)):

            b = tl.load(b_ptrs, mask=mask_b, other=0.0)

                mask_a = offs_am[:, None] < M
                a = tl.load(a_ptrs, mask=mask_a, other=0.0)

                a = tl.load(a_ptrs)
