"""Optimized Triton kernel for multi-axis RoPE pre-attention prep.

Unified kernel: cos/sin loaded once per (b, t_tile, d_tile), reused across all H.
Tunable BLOCK_T, BLOCK_D, num_warps for each shape.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _rope_kernel(
    q_ptr, k_ptr,
    cos_cache, sin_cache,
    pos_t_ptr, pos_h_ptr, pos_w_ptr,
    q_out_ptr, k_out_ptr,
    B: tl.int64,
    T: tl.int64,
    Hq: tl.int64,
    Hkv: tl.int64,
    D: tl.int64,
    s0: tl.int64,
    s1: tl.int64,
    stride_qb, stride_qt, stride_qh, stride_qd,
    stride_kb, stride_kt, stride_kh, stride_kd,
    stride_pb, stride_pt,
    stride_ccb, stride_ccd,
    stride_qob, stride_qoh, stride_qot, stride_qod,
    stride_kob, stride_koh, stride_kot, stride_kod,
    BLOCK_T: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    pid_d = tl.program_id(1)

    R = tl.cdiv(T, BLOCK_T)
    D_tiles = tl.cdiv(D, BLOCK_D)

    r = pid % R
    b = pid // R
    d_off = pid_d * BLOCK_D

    if b >= B or pid_d >= D_tiles:
        return

    t_off = r * BLOCK_T
    half_d = D // 2

    _rt = tl.arange(0, BLOCK_T)
    _rd = tl.arange(0, BLOCK_D)
    t_rng = t_off + _rt
    d_rng = d_off + _rd
    t_mask = t_rng < T
    d_mask = d_rng < D
    td_mask = t_mask[:, None] & d_mask[None, :]

    # Positions
    po = b * stride_pb + t_rng * stride_pt
    p_t = tl.load(pos_t_ptr + po, mask=t_mask, other=0)
    p_h = tl.load(pos_h_ptr + po, mask=t_mask, other=0)
    p_w = tl.load(pos_w_ptr + po, mask=t_mask, other=0)

    # Axis select per d
    d_f = tl.where(d_rng < half_d, d_rng, d_rng - half_d)
    is_t = d_f < s0
    is_h = (d_f >= s0) & (d_f < s0 + s1)
    pos_sel = tl.where(is_t[None, :], p_t[:, None],
                 tl.where(is_h[None, :], p_h[:, None], p_w[:, None]))

    # Cos/sin cache — loaded ONCE, reused for all heads
    cache_off = pos_sel * stride_ccb + d_rng[None, :] * stride_ccd
    cos = tl.load(cos_cache + cache_off, mask=td_mask, other=1.0)
    sin = tl.load(sin_cache + cache_off, mask=td_mask, other=0.0)

    # Rotate-half mirror indices (same for all heads)
    d_mir = tl.where(d_rng < half_d, d_rng + half_d, d_rng - half_d)
    is_first_half = d_rng[None, :] < half_d

    # Process Q heads
    h = 0
    while h < Hq:
        q_base = b * stride_qb + h * stride_qh
        qx = tl.load(q_ptr + q_base + t_rng[:, None] * stride_qt + d_rng[None, :] * stride_qd,
                      mask=td_mask, other=0.0)
        qm = tl.load(q_ptr + q_base + t_rng[:, None] * stride_qt + d_mir[None, :] * stride_qd,
                      mask=td_mask, other=0.0)
        out = qx * cos + tl.where(is_first_half, -qm, qm) * sin
        tl.store(q_out_ptr + b * stride_qob + h * stride_qoh + t_rng[:, None] * stride_qot + d_rng[None, :] * stride_qod,
                 out, mask=td_mask)
        h += 1

    # Process K heads
    h = 0
    while h < Hkv:
        k_base = b * stride_kb + h * stride_kh
        kx = tl.load(k_ptr + k_base + t_rng[:, None] * stride_kt + d_rng[None, :] * stride_kd,
                      mask=td_mask, other=0.0)
        km = tl.load(k_ptr + k_base + t_rng[:, None] * stride_kt + d_mir[None, :] * stride_kd,
                      mask=td_mask, other=0.0)
        out = kx * cos + tl.where(is_first_half, -km, km) * sin
        tl.store(k_out_ptr + b * stride_kob + h * stride_koh + t_rng[:, None] * stride_kot + d_rng[None, :] * stride_kod,
                 out, mask=td_mask)
        h += 1


def apply_mrope_torch(q, k, cos_cache, sin_cache, pos_t, pos_h, pos_w, mrope_section):
    B, T, Hq, D = q.shape
    _, _, Hkv, _ = k.shape
    s0, s1, s2 = mrope_section

    q_rot = torch.empty(B, Hq, T, D, dtype=q.dtype, device=q.device)
    k_rot = torch.empty(B, Hkv, T, D, dtype=k.dtype, device=k.device)

    # Tunable configs — selected based on shape characteristics
    # Larger H benefits from smaller grid => larger BLOCK_T
    # Bandwidth-bound: want maximal memory throughput
    if D == 64:
        BLOCK_T, BLOCK_D, num_warps = 32, 64, 4
    elif T <= 4096:
        BLOCK_T, BLOCK_D, num_warps = 16, 64, 4
    else:
        BLOCK_T, BLOCK_D, num_warps = 16, 64, 8

    n_t = (T + BLOCK_T - 1) // BLOCK_T
    n_d = (D + BLOCK_D - 1) // BLOCK_D

    _rope_kernel[(B * n_t, n_d)](
        q, k, cos_cache, sin_cache,
        pos_t, pos_h, pos_w,
        q_rot, k_rot,
        B=B, T=T, Hq=Hq, Hkv=Hkv, D=D,
        s0=s0, s1=s1,
        stride_qb=q.stride(0), stride_qt=q.stride(1),
        stride_qh=q.stride(2), stride_qd=q.stride(3),
        stride_kb=k.stride(0), stride_kt=k.stride(1),
        stride_kh=k.stride(2), stride_kd=k.stride(3),
        stride_pb=pos_t.stride(0), stride_pt=pos_t.stride(1),
        stride_ccb=cos_cache.stride(0), stride_ccd=cos_cache.stride(1),
        stride_qob=q_rot.stride(0), stride_qoh=q_rot.stride(1),
        stride_qot=q_rot.stride(2), stride_qod=q_rot.stride(3),
        stride_kob=k_rot.stride(0), stride_koh=k_rot.stride(1),
        stride_kot=k_rot.stride(2), stride_kod=k_rot.stride(3),
        BLOCK_T=BLOCK_T, BLOCK_D=BLOCK_D,
        num_warps=num_warps,
    )

    return q_rot, k_rot


class Model(nn.Module):
    def __init__(self, B: int, T: int, Hq: int, Hkv: int, D: int,
                 mrope_section: tuple[int, int, int], max_pos: int):
        super().__init__()
        assert sum(mrope_section) == D // 2
        self.B, self.T = B, T
        self.Hq, self.Hkv, self.D = Hq, Hkv, D
        self.mrope_section = tuple(mrope_section)
        self.max_pos = max_pos

        inv_freq = 1.0 / (10000.0 ** (torch.arange(0, D, 2, dtype=torch.float32) / D))
        pos = torch.arange(max_pos, dtype=torch.float32)
        freqs = torch.outer(pos, inv_freq)
        emb = torch.cat([freqs, freqs], dim=-1)
        self.register_buffer("cos_cache", emb.cos().to(torch.bfloat16))
        self.register_buffer("sin_cache", emb.sin().to(torch.bfloat16))

    def forward(
        self,
        q: torch.Tensor, k: torch.Tensor,
        pos_t: torch.Tensor, pos_h: torch.Tensor, pos_w: torch.Tensor,
    ) -> tuple[torch.Tensor, torch.Tensor]:
        return apply_mrope_torch(
            q, k, self.cos_cache, self.sin_cache,
            pos_t, pos_h, pos_w, self.mrope_section,
        )
