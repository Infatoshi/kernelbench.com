"""Triton grouped GEMM + fused SwiGLU for MoE up-projection (SM120 Blackwell).

Each program instance handles ONE (expert, n_tile, m_tile) triple.
Loops over K tiles to accumulate the full GEMM.
SwiGLU is fused in the epilogue: silu(gate) * up.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


BM = 128
BK = 64
BN = 128


@triton.jit
def _grouped_gemm_swiglu_kernel(
    x_ptr, w_gate_ptr, w_up_ptr, out_ptr, offsets_ptr,
    H: tl.constexpr, I: tl.constexpr, E: tl.constexpr,
    num_n_tiles: tl.constexpr, max_m_tiles: tl.constexpr,
    BM: tl.constexpr, BK: tl.constexpr, BN: tl.constexpr,
    num_k_tiles: tl.constexpr,
):
    pid = tl.program_id(0)
    expert = pid // (num_n_tiles * max_m_tiles)
    rem = pid % (num_n_tiles * max_m_tiles)
    n_tile = rem // max_m_tiles
    m_tile_pid = rem % max_m_tiles

    start = tl.load(offsets_ptr + expert)
    end = tl.load(offsets_ptr + expert + 1)
    num_rows = end - start
    m_tiles_needed = tl.cdiv(num_rows, BM)
    if m_tile_pid >= m_tiles_needed:
        return

    m_start = start + m_tile_pid * BM
    n_start = n_tile * BN

    row_offs = m_start + tl.arange(0, BM)
    load_row_mask = row_offs < end

    n_offs = n_start + tl.arange(0, BN)
    n_mask = n_offs < I

    k_offs_base = tl.arange(0, BK)

    gate_acc = tl.zeros((BM, BN), dtype=tl.float32)
    up_acc = tl.zeros((BM, BN), dtype=tl.float32)

    expert_w_off = expert * H * I

    for k_tile in range(num_k_tiles):
        k_start = k_tile * BK
        k_offs = k_start + k_offs_base
        k_mask = k_offs < H

        x_ptrs = x_ptr + row_offs[:, None] * H + k_offs[None, :]
        x_mask = load_row_mask[:, None] & k_mask[None, :]
        x = tl.load(x_ptrs, mask=x_mask, other=0.0).to(tl.float32)

        w_ptrs = k_offs[:, None] * I + n_offs[None, :]
        wg_ptrs = w_gate_ptr + expert_w_off + w_ptrs
        wu_ptrs = w_up_ptr + expert_w_off + w_ptrs
        wg_mask = k_mask[:, None] & n_mask[None, :]
        wg = tl.load(wg_ptrs, mask=wg_mask, other=0.0).to(tl.float32)
        wu = tl.load(wu_ptrs, mask=wg_mask, other=0.0).to(tl.float32)

        gate_acc += tl.dot(x, wg)
        up_acc += tl.dot(x, wu)

    gate_silu = gate_acc * tl.sigmoid(gate_acc)
    result = (gate_silu * up_acc).to(tl.bfloat16)

    out_ptrs = out_ptr + row_offs[:, None] * I + n_offs[None, :]
    store_mask = load_row_mask[:, None] & n_mask[None, :]
    tl.store(out_ptrs, result, mask=store_mask)


class FusedSwiGLUGroupedGEMM(torch.autograd.Function):
    @staticmethod
    def forward(ctx, hidden_states, W_gate, W_up, expert_offsets, E, I):
        T_perm, H = hidden_states.shape
        out = torch.empty((T_perm, I), dtype=torch.bfloat16, device=hidden_states.device)

        num_k_tiles = triton.cdiv(H, BK)
        num_n_tiles = triton.cdiv(I, BN)

        max_tokens_per_expert = int(expert_offsets[1:].diff().max().item()) if T_perm > 0 else 0
        max_m_tiles = triton.cdiv(max_tokens_per_expert, BM)
        total_programs = E * num_n_tiles * max_m_tiles

        if total_programs > 0:
            _grouped_gemm_swiglu_kernel[(total_programs,)](
                hidden_states, W_gate, W_up, out, expert_offsets,
                H=H, I=I, E=E,
                num_n_tiles=num_n_tiles, max_m_tiles=max_m_tiles,
                BM=BM, BK=BK, BN=BN,
                num_k_tiles=num_k_tiles,
            )

        return out

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets

### Expert routing
- **Top-k routing**: each token selects k experts (typically k=2) based on gating scores
