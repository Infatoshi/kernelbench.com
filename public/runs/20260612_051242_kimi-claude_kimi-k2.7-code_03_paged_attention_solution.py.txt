import math

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _paged_attn_decode_split_kernel(
    out_partial_ptr,
    m_partial_ptr,
    l_partial_ptr,
    q_ptr,
    kv_cache_ptr,
    block_table_ptr,
    seq_lens_ptr,
    scale,
    stride_qb,
    stride_qh,
    stride_qd,
    stride_kvn,
    stride_kvp,
    stride_kvh,
    stride_bt_b,
    stride_bt_p,
    stride_op_b,
    stride_op_kv,
    stride_op_c,
    stride_mp_b,
    stride_mp_kv,
    stride_mp_c,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    BLOCK_D: tl.constexpr,
    CHUNK_PAGES: tl.constexpr,
    PAGES_PER_TILE: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)
    pid_chunk = tl.program_id(2)

    seq_len = tl.load(seq_lens_ptr + pid_b)
    num_pages = (seq_len + PAGE_SIZE - 1) // PAGE_SIZE
    chunk_start_page = pid_chunk * CHUNK_PAGES

    offs_d = tl.arange(0, BLOCK_D)
    mask_d = offs_d < HEAD_DIM
    offs_g = tl.arange(0, GROUP_SIZE)
    q_h = pid_kv * GROUP_SIZE + offs_g

    q_ptrs = q_ptr + pid_b * stride_qb + q_h[:, None] * stride_qh + offs_d[None, :]
    q = tl.load(q_ptrs, mask=mask_d[None, :], other=0.0).to(tl.bfloat16)

    base_out = pid_b * stride_op_b + pid_kv * stride_op_kv + pid_chunk * stride_op_c
    base_mp = pid_b * stride_mp_b + pid_kv * stride_mp_kv + pid_chunk * stride_mp_c

    if chunk_start_page >= num_pages:
        tl.store(out_partial_ptr + base_out + offs_g[:, None] * BLOCK_D + offs_d[None, :],
                 tl.zeros((GROUP_SIZE, BLOCK_D), tl.float32), mask=mask_d[None, :])
        tl.store(m_partial_ptr + base_mp + offs_g[:, None],
                 tl.full((GROUP_SIZE, 1), float("-inf"), tl.float32))
        tl.store(l_partial_ptr + base_mp + offs_g[:, None],
                 tl.zeros((GROUP_SIZE, 1), tl.float32))
        return

    chunk_end_page = tl.minimum((pid_chunk + 1) * CHUNK_PAGES, num_pages)

    m = tl.full((GROUP_SIZE, 1), float("-inf"), tl.float32)
    l = tl.zeros((GROUP_SIZE, 1), tl.float32)
    acc = tl.zeros((GROUP_SIZE, BLOCK_D), tl.float32)

    offs_n = tl.arange(0, PAGE_SIZE)
    offs_n_big = tl.arange(0, PAGES_PER_TILE * PAGE_SIZE)

    for p in range(chunk_start_page, chunk_end_page, PAGES_PER_TILE):
        page_idx0 = tl.load(block_table_ptr + pid_b * stride_bt_b + p * stride_bt_p)
        kv_base0 = kv_cache_ptr + page_idx0 * stride_kvn + pid_kv * stride_kvh
        remain0 = seq_len - p * PAGE_SIZE
        mask_n0 = offs_n < remain0

        k_ptrs0 = kv_base0 + offs_n[:, None] * stride_kvp + offs_d[None, :]
        k0 = tl.load(k_ptrs0, mask=mask_n0[:, None] & mask_d[None, :], other=0.0).to(tl.bfloat16)
        v_ptrs0 = kv_base0 + HEAD_DIM + offs_n[:, None] * stride_kvp + offs_d[None, :]
        v0 = tl.load(v_ptrs0, mask=mask_n0[:, None] & mask_d[None, :], other=0.0).to(tl.bfloat16)

        p1 = p + 1
        valid1 = p1 < chunk_end_page
        page_idx1 = tl.load(block_table_ptr + pid_b * stride_bt_b + p1 * stride_bt_p, mask=valid1, other=0)
        kv_base1 = kv_cache_ptr + page_idx1 * stride_kvn + pid_kv * stride_kvh
        remain1 = seq_len - p1 * PAGE_SIZE
        mask_n1 = offs_n < remain1

        k_ptrs1 = kv_base1 + offs_n[:, None] * stride_kvp + offs_d[None, :]
        k1 = tl.load(k_ptrs1, mask=mask_n1[:, None] & mask_d[None, :], other=0.0).to(tl.bfloat16)
        v_ptrs1 = kv_base1 + HEAD_DIM + offs_n[:, None] * stride_kvp + offs_d[None, :]
        v1 = tl.load(v_ptrs1, mask=mask_n1[:, None] & mask_d[None, :], other=0.0).to(tl.bfloat16)

        k = tl.cat(k0, k1, axis=0)
        v = tl.cat(v0, v1, axis=0)
        mask_n_big = tl.cat(mask_n0, mask_n1, axis=0)

        scores = scale * tl.dot(q, tl.trans(k)).to(tl.float32)
        scores = tl.where(mask_n_big[None, :], scores, float("-inf"))

        m_new = tl.maximum(m, tl.max(scores, axis=1)[:, None])
        alpha = tl.exp(m - m_new)
        p_acc = tl.exp(scores - m_new)
        l = l * alpha + tl.sum(p_acc, axis=1)[:, None]
        acc = acc * alpha + tl.dot(p_acc.to(tl.bfloat16), v).to(tl.float32)
        m = m_new

    tl.store(out_partial_ptr + base_out + offs_g[:, None] * BLOCK_D + offs_d[None, :],
             acc, mask=mask_d[None, :])
    tl.store(m_partial_ptr + base_mp + offs_g[:, None], m)
    tl.store(l_partial_ptr + base_mp + offs_g[:, None], l)


@triton.jit
def _paged_attn_decode_reduce_kernel(
    out_ptr,
    out_partial_ptr,
    m_partial_ptr,
    l_partial_ptr,
    stride_ob,
    stride_oh,
    stride_od,
    stride_op_b,
    stride_op_kv,
    stride_op_c,
    stride_mp_b,
    stride_mp_kv,
    stride_mp_c,
    NUM_HEADS: tl.constexpr,
    NUM_KV_HEADS: tl.constexpr,
    HEAD_DIM: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    BLOCK_D: tl.constexpr,
    NUM_CHUNKS: tl.constexpr,
):
    pid_b = tl.program_id(0)
    pid_kv = tl.program_id(1)

    offs_d = tl.arange(0, BLOCK_D)
    mask_d = offs_d < HEAD_DIM
    offs_g = tl.arange(0, GROUP_SIZE)
    q_h = pid_kv * GROUP_SIZE + offs_g

    m_global = tl.full((GROUP_SIZE, 1), float("-inf"), tl.float32)
    l_global = tl.zeros((GROUP_SIZE, 1), tl.float32)
    acc_global = tl.zeros((GROUP_SIZE, BLOCK_D), tl.float32)

    for c in range(NUM_CHUNKS):
        base_out = pid_b * stride_op_b + pid_kv * stride_op_kv + c * stride_op_c
        base_mp = pid_b * stride_mp_b + pid_kv * stride_mp_kv + c * stride_mp_c

        m_c = tl.load(m_partial_ptr + base_mp + offs_g[:, None])
        l_c = tl.load(l_partial_ptr + base_mp + offs_g[:, None])
        acc_c = tl.load(
            out_partial_ptr + base_out + offs_g[:, None] * BLOCK_D + offs_d[None, :],
            mask=mask_d[None, :], other=0.0
        ).to(tl.float32)

        m_new = tl.maximum(m_global, m_c)
        alpha_g = tl.exp(m_global - m_new)
        alpha_c = tl.exp(m_c - m_new)
        acc_global = acc_global * alpha_g + acc_c * alpha_c
        l_global = l_global * alpha_g + l_c * alpha_c
        m_global = m_new

    out = (acc_global / l_global).to(tl.bfloat16)
    out_ptrs = out_ptr + pid_b * stride_ob + q_h[:, None] * stride_oh + offs_d[None, :]
    tl.store(out_ptrs, out, mask=mask_d[None, :])


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        P = self.page_size
        G = self.group_size

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        block_d = max(64, 1 << (D - 1).bit_length()) if D > 0 else 64

        # Larger tiles help small head_dim cases utilise tensor cores better.
        if D <= 64:
            pages_per_tile = 2
        else:
            pages_per_tile = 1

        chunk_pages = 16
        chunk_tokens = chunk_pages * P
        max_seq_len = int(seq_lens.max().item())
        num_chunks = (max_seq_len + chunk_tokens - 1) // chunk_tokens
        if num_chunks < 1:
            num_chunks = 1

        partial_shape = (B, Hkv, num_chunks, G, block_d)
        stat_shape = (B, Hkv, num_chunks, G)
        out_partial = torch.empty(partial_shape, dtype=torch.float32, device=query.device)
        m_partial = torch.empty(stat_shape, dtype=torch.float32, device=query.device)
        l_partial = torch.empty(stat_shape, dtype=torch.float32, device=query.device)

        stride_op_b = Hkv * num_chunks * G * block_d
        stride_op_kv = num_chunks * G * block_d
        stride_op_c = G * block_d
        stride_mp_b = Hkv * num_chunks * G
        stride_mp_kv = num_chunks * G
        stride_mp_c = G

        grid_split = (B, Hkv, num_chunks)
        _paged_attn_decode_split_kernel[grid_split](
            out_partial,
            m_partial,
            l_partial,
            query,
            kv_cache,
            block_table,
            seq_lens,
            self.scale,
            query.stride(0),
            query.stride(1),
            query.stride(2),
            kv_cache.stride(0),
            kv_cache.stride(1),
            kv_cache.stride(2),
            block_table.stride(0),
            block_table.stride(1),
            stride_op_b,
            stride_op_kv,
            stride_op_c,
            stride_mp_b,
            stride_mp_kv,
            stride_mp_c,
            NUM_HEADS=H,
            NUM_KV_HEADS=Hkv,
            HEAD_DIM=D,
            PAGE_SIZE=P,
            GROUP_SIZE=G,
            BLOCK_D=block_d,
            CHUNK_PAGES=chunk_pages,
            PAGES_PER_TILE=pages_per_tile,
        )

        grid_reduce = (B, Hkv)
        _paged_attn_decode_reduce_kernel[grid_reduce](
            out,
            out_partial,
            m_partial,
            l_partial,
            out.stride(0),
            out.stride(1),
            out.stride(2),
            stride_op_b,
            stride_op_kv,
            stride_op_c,
            stride_mp_b,
            stride_mp_kv,
            stride_mp_c,
            NUM_HEADS=H,
            NUM_KV_HEADS=Hkv,
            HEAD_DIM=D,
            PAGE_SIZE=P,
            GROUP_SIZE=G,
            BLOCK_D=block_d,
            NUM_CHUNKS=num_chunks,
        )

        return out
