"""Fused patch-gather + tensor-core GEMM kernel for 3D patch embedding.

Uses Triton with tl.dot to fuse the strided patch gather into the GEMM K-loop,
avoiding materialization of the im2col buffer. The patch-gather offsets decompose
into a rank-1 sum of m-dependent and k-dependent terms, enabling efficient
2D pointer arithmetic.

Output layout: (B, embed_dim, T//kT, H//kH, W//kW). No bias.
"""
import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 128}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 128}, num_warps=8, num_stages=2),
        triton.Config({'BLOCK_M': 32, 'BLOCK_N': 128, 'BLOCK_K': 128}, num_warps=4, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 64, 'BLOCK_K': 128}, num_warps=4, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 128}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 256, 'BLOCK_K': 128}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 256, 'BLOCK_K': 128}, num_warps=8, num_stages=2),
        triton.Config({'BLOCK_M': 64, 'BLOCK_N': 128, 'BLOCK_K': 64}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 128, 'BLOCK_N': 128, 'BLOCK_K': 64}, num_warps=8, num_stages=3),
        triton.Config({'BLOCK_M': 32, 'BLOCK_N': 64, 'BLOCK_K': 64}, num_warps=4, num_stages=3),
    ],
    key=['K', 'N', 'M_total'],
)
@triton.jit
def patch_embed_kernel(
    x_ptr, w_ptr, out_ptr,
    nPT, nPH, nPW,
    stride_x_b, stride_x_c, stride_x_t, stride_x_h, stride_x_w,
    stride_out_b, stride_out_n, stride_out_pt, stride_out_ph, stride_out_pw,
    K, N,
    kT, kH, kW,
    M_total,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_n_blocks = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_n_blocks
    pid_n = pid % num_n_blocks

    m_start = pid_m * BLOCK_M
    n_start = pid_n * BLOCK_N

    m_off = m_start + tl.arange(0, BLOCK_M)
    n_off = n_start + tl.arange(0, BLOCK_N)

    patches_per_batch = nPT * nPH * nPW
    pw_m = m_off % nPW
    ph_m = (m_off // nPW) % nPH
    pt_m = (m_off // (nPW * nPH)) % nPT
    b_m = m_off // patches_per_batch

    m_mask = m_off < M_total
    n_mask = n_off < N

    m_part = b_m * stride_x_b + pt_m * kT * stride_x_t + ph_m * kH * stride_x_h + pw_m * kW * stride_x_w

    out_m_part = b_m * stride_out_b + pt_m * stride_out_pt + ph_m * stride_out_ph + pw_m * stride_out_pw

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    kTkHkW = kT * kH * kW
    kHkW = kH * kW

    for k_start in range(0, K, BLOCK_K):
        k_off = k_start + tl.arange(0, BLOCK_K)

        kw_k = k_off % kW
        kh_k = (k_off // kW) % kH
        kt_k = (k_off // kHkW) % kT
        c_k = k_off // kTkHkW

        k_mask = k_off < K

        k_part = c_k * stride_x_c + kt_k * stride_x_t + kh_k * stride_x_h + kw_k * stride_x_w

        a_ptrs = x_ptr + m_part[:, None] + k_part[None, :]
        a_mask = m_mask[:, None] * k_mask[None, :]
        a = tl.load(a_ptrs, mask=a_mask, other=0.0)

        w_ptrs = w_ptr + n_off[None, :] * K + k_off[:, None]
        w_mask = n_mask[None, :] * k_mask[:, None]
        w = tl.load(w_ptrs, mask=w_mask, other=0.0, eviction_policy='evict_last')

        acc += tl.dot(a, w)

    out_ptrs = out_ptr + out_m_part[:, None] + n_off[None, :] * stride_out_n
    out_mask = m_mask[:, None] * n_mask[None, :]
    tl.store(out_ptrs, acc.to(tl.bfloat16), mask=out_mask)


class Model(nn.Module):
    def __init__(self, B: int, C: int, T: int, H: int, W: int,
                 kT: int, kH: int, kW: int, embed_dim: int):
        super().__init__()
        self.B, self.C, self.T, self.H, self.W = B, C, T, H, W
        self.kT, self.kH, self.kW = kT, kH, kW
        self.embed_dim = embed_dim

        self.conv = nn.Conv3d(
            C, embed_dim,
            kernel_size=(kT, kH, kW),
            stride=(kT, kH, kW),
            bias=False,
            dtype=torch.bfloat16,
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, C, T, H, W = x.shape
        kT, kH, kW = self.kT, self.kH, self.kW
        embed_dim = self.embed_dim

        nPT = T // kT
        nPH = H // kH
        nPW = W // kW

        M_total = B * nPT * nPH * nPW
        K = C * kT * kH * kW
        N = embed_dim

        out = torch.empty(B, embed_dim, nPT, nPH, nPW, dtype=torch.bfloat16, device=x.device)

        stride_x = x.stride()
        stride_out = out.stride()

        grid = lambda META: (triton.cdiv(M_total, META['BLOCK_M']) * triton.cdiv(N, META['BLOCK_N']),)

        patch_embed_kernel[grid](
            x, self.conv.weight, out,
            nPT, nPH, nPW,
            stride_x[0], stride_x[1], stride_x[2], stride_x[3], stride_x[4],
            stride_out[0], stride_out[1], stride_out[2], stride_out[3], stride_out[4],
            K, N,
            kT, kH, kW,
            M_total,
        )

        return out
