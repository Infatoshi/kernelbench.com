"""Optimized KDA forward (chunk form) for RTX PRO 6000 Blackwell.

Uses CUDA graphs for intra-chunk + Triton kernel for inter-chunk recurrence.
"""
from __future__ import annotations

from typing import Optional
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
from einops import rearrange
import triton
import triton.language as tl


_FWD_SOLVE_SRC = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>

#define BT 64

__global__ void tri_solve_kernel(
    const float* __restrict__ M,
    const float* __restrict__ rhs,
    float* __restrict__ x,
    int batch, int D
) {
    int b = blockIdx.x;
    if (b >= batch) return;
    int d_base = blockIdx.y * blockDim.x;
    int d = d_base + threadIdx.x;
    if (d >= D) return;

    __shared__ float sM[BT][BT];
    for (int r = threadIdx.x; r < BT * BT; r += blockDim.x) {
        int row = r / BT;
        int col = r % BT;
        sM[row][col] = M[b * BT * BT + row * BT + col];
    }
    __syncthreads();

    const float* rhs_b = rhs + b * BT * D;
    float* x_b = x + b * BT * D;

    float acc[BT];
    for (int i = 0; i < BT; i++) {
        float val = rhs_b[i * D + d];
        for (int j = 0; j < i; j++) {
            val -= sM[i][j] * acc[j];
        }
        acc[i] = val;
    }

    for (int i = 0; i < BT; i++) {
        x_b[i * D + d] = acc[i];
    }
}

torch::Tensor tri_solve(torch::Tensor M, torch::Tensor rhs) {
    int batch = M.size(0);
    int D = rhs.size(2);
    auto x = torch::empty_like(rhs);
    int block_d = 256;
    dim3 grid(batch, (D + block_d - 1) / block_d);
    dim3 block(block_d);
    tri_solve_kernel<<<grid, block>>>(
        M.data_ptr<float>(), rhs.data_ptr<float>(),
        x.data_ptr<float>(), batch, D);
    return x;
}
"""

_fwd_solve_mod = load_inline(
    name="kda_solve3",
    cpp_sources="torch::Tensor tri_solve(torch::Tensor M, torch::Tensor rhs);",
    cuda_sources=[_FWD_SOLVE_SRC],
    functions=["tri_solve"],
    verbose=False,
)


@triton.jit
def _inter_fwd(
    w_ptr, u_ptr, qg_ptr, Aqk_ptr, wkT_ptr, g_last_ptr,
    o_ptr, S_ptr,
    NT: tl.constexpr, BT: tl.constexpr, K: tl.constexpr, V: tl.constexpr,
    BLOCK_V: tl.constexpr,
):
    pid = tl.program_id(0)
    v_tiles = V // BLOCK_V
    bh = pid // v_tiles
    v_tile = pid % v_tiles
    v_off = v_tile * BLOCK_V

    bt_off = tl.arange(0, BT)
    v_idx = v_off + tl.arange(0, BLOCK_V)
    k_off = tl.arange(0, K)

    S_bh = bh * K * V
    S_base = S_bh + k_off[:, None] * V + v_idx[None, :]
    S_tile = tl.load(S_ptr + S_base).to(tl.float32)

    BT_K = BT * K
    BT_V = BT * V
    BT_BT = BT * BT
    K_BT = K * BT

    for i in range(NT):
        i_bt = i * BT
        i_k = i * K

        w_chunk = w_ptr + bh * NT * BT_K + i_bt * K
        qg_chunk = qg_ptr + bh * NT * BT_K + i_bt * K
        u_chunk = u_ptr + bh * NT * BT_V + i_bt * V
        aqk_chunk = Aqk_ptr + bh * NT * BT_BT + i_bt * BT
        wkT_chunk = wkT_ptr + bh * NT * K_BT + i_k * BT
        gl_chunk = g_last_ptr + bh * NT * K + i_k
        o_chunk = o_ptr + bh * NT * BT_V + i_bt * V

        v_acc = tl.zeros([BT, BLOCK_V], dtype=tl.float32)
        for k_start in range(0, K, BT):
            k_inner = k_start + tl.arange(0, BT)
            w_tile = tl.load(w_chunk + bt_off[:, None] * K + k_inner[None, :]).to(tl.float32)
            s_tile = tl.load(S_ptr + S_bh + k_inner[:, None] * V + v_idx[None, :]).to(tl.float32)
            v_acc += tl.dot(w_tile, s_tile, input_precision="ieee")

        u_tile = tl.load(u_chunk + bt_off[:, None] * V + v_idx[None, :]).to(tl.float32)
        v_i = u_tile - v_acc

        qg_acc = tl.zeros([BT, BLOCK_V], dtype=tl.float32)
        for k_start in range(0, K, BT):
            k_inner = k_start + tl.arange(0, BT)
            qg_tile = tl.load(qg_chunk + bt_off[:, None] * K + k_inner[None, :]).to(tl.float32)
            s_tile2 = tl.load(S_ptr + S_bh + k_inner[:, None] * V + v_idx[None, :]).to(tl.float32)
            qg_acc += tl.dot(qg_tile, s_tile2, input_precision="ieee")

        aqk_tile = tl.load(aqk_chunk + bt_off[:, None] * BT + bt_off[None, :]).to(tl.float32)
        o_tile = qg_acc + tl.dot(aqk_tile, v_i, input_precision="ieee")
        tl.store(o_chunk + bt_off[:, None] * V + v_idx[None, :], o_tile.to(tl.bfloat16))

        gl_vals = tl.load(gl_chunk + k_off).to(tl.float32)
        S_tile = S_tile * gl_vals[:, None]

        wkT_tile = tl.load(wkT_chunk + k_off[:, None] * BT + bt_off[None, :]).to(tl.float32)
        S_tile = S_tile + tl.dot(wkT_tile, v_i, input_precision="ieee")

        tl.store(S_ptr + S_bh + k_off[:, None] * V + v_idx[None, :], S_tile.to(tl.float32))


class _GraphCache:
    def __init__(self):
        self.graphs = {}

    def _build_graph(self, B, T, H, K, V, BT, NT, BH, B_NT, scale, device):
        s_q = torch.randn(B, T, H, K, dtype=torch.bfloat16, device=device) * 0.1
        s_k = torch.randn(B, T, H, K, dtype=torch.bfloat16, device=device) * 0.1
        s_v = torch.randn(B, T, H, V, dtype=torch.bfloat16, device=device) * 0.1
        s_g = torch.randn(B, T, H, K, dtype=torch.float32, device=device) * 0.01
        s_beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16, device=device))

        warmup = torch.cuda.Stream()
        warmup.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(warmup):
            for _ in range(3):
                _kda_fwd_eager(s_q, s_k, s_v, s_g, s_beta, scale, BT)
        torch.cuda.current_stream().wait_stream(warmup)

        graph = torch.cuda.CUDAGraph()
        with torch.cuda.graph(graph):
            s_out = _kda_fwd_eager(s_q, s_k, s_v, s_g, s_beta, scale, BT)

        return graph, s_q, s_k, s_v, s_g, s_beta, s_out

    def run(self, q, k, v, g, beta, scale, BT):
        B, T, H, K = q.shape
        V = v.shape[-1]
        key = (B, T, H, K, V, BT)

        if key not in self.graphs:
            NT = T // BT
            BH = B * H
            B_NT = BH * NT
            device = q.device
            self.graphs[key] = self._build_graph(B, T, H, K, V, BT, NT, BH, B_NT, scale, device)

        graph, s_q, s_k, s_v, s_g, s_beta, s_out = self.graphs[key]
        s_q.copy_(q)
        s_k.copy_(k)
        s_v.copy_(v)
        s_g.copy_(g)
        s_beta.copy_(beta)
        graph.replay()
        return s_out.clone()


_graph_cache = _GraphCache()


def _kda_fwd_eager(q, k, v, g, beta, scale, chunk_size):
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    NT = T // BT
    device = q.device
    BH = B * H
    B_NT = BH * NT

    q, k, v, g, beta = (x.to(torch.float32) for x in (q, k, v, g, beta))
    q = q * scale

    q_r = rearrange(q, "b (n c) h d -> (b h) n c d", c=BT).contiguous()
    k_r = rearrange(k, "b (n c) h d -> (b h) n c d", c=BT).contiguous()
    v_r = rearrange(v, "b (n c) h d -> (b h) n c d", c=BT).contiguous()
    g_r = rearrange(g, "b (n c) h d -> (b h) n c d", c=BT).contiguous()
    beta_r = rearrange(beta, "b (n c) h -> (b h) n c", c=BT).contiguous()

    g_r = g_r.cumsum(dim=2)

    g_exp = g_r.exp()
    g_neg = (-g_r).exp()
    kg = k_r * g_exp
    kg_inv = k_r * g_neg
    qg = q_r * g_exp

    kg_b = kg.reshape(B_NT, BT, K)
    kg_inv_t = kg_inv.reshape(B_NT, BT, K).transpose(-1, -2).contiguous()

    Akk = torch.bmm(kg_b, kg_inv_t).view(BH, NT, BT, BT)
    strict_lower = torch.tril(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=-1)
    L = (Akk * strict_lower[None, None, :, :]) * beta_r[..., :, None]
    M = L + torch.eye(BT, dtype=torch.float32, device=device)

    lhs_wu = torch.cat([
        (beta_r[..., :, None] * kg).reshape(B_NT, BT, K),
        (beta_r[..., :, None] * v_r).reshape(B_NT, BT, V),
    ], dim=-1)

    M_b = M.reshape(B_NT, BT, BT).contiguous()
    lhs_wu_c = lhs_wu.contiguous()
    wu = _fwd_solve_mod.tri_solve(M_b, lhs_wu_c)
    w = wu[:, :, :K].view(BH, NT, BT, K)
    u = wu[:, :, K:].view(BH, NT, BT, V)

    qg_b = qg.reshape(B_NT, BT, K)
    Aqk = torch.bmm(qg_b, kg_inv_t).view(BH, NT, BT, BT)
    strict_upper = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=1)
    Aqk = Aqk * (~strict_upper[None, None, :, :])

    g_last = g_r[:, :, -1, :]
    decay = (g_last.unsqueeze(2) - g_r).exp() * k_r
    wkT = decay.permute(0, 1, 3, 2).contiguous()
    g_last_exp = g_last.exp().contiguous()

    BLOCK_V = 32
    grid = (BH * (V // BLOCK_V),)

    o_out = torch.zeros(BH, NT * BT * V, dtype=torch.bfloat16, device=device)
    S_state = torch.zeros(BH, K, V, dtype=torch.float32, device=device)

    _inter_fwd[grid](
        w.reshape(BH, -1), u.reshape(BH, -1), qg.reshape(BH, -1),
        Aqk.reshape(BH, -1), wkT.reshape(BH, -1),
        g_last_exp.reshape(BH, -1),
        o_out, S_state,
        NT=NT, BT=BT, K=K, V=V,
        BLOCK_V=BLOCK_V,
        num_warps=4,
    )

    o_out = o_out.view(B, H, NT, BT, V)
    o_out = rearrange(o_out, "b h n c d -> b (n c) h d")
    return o_out.to(dtype)


def _kda_fwd(q, k, v, g, beta, scale, chunk_size):
    return _kda_fwd_eager(q, k, v, g, beta, scale, chunk_size)


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(self, q, k, v, g, beta):
        return _kda_fwd(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size)


B = 2
T = 1024
H = 8
K = 128
V = 128
CHUNK_SIZE = 64


def get_inputs():
    torch.manual_seed(0)
    q = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    k = torch.randn(B, T, H, K, dtype=torch.bfloat16) * 0.1
    v = torch.randn(B, T, H, V, dtype=torch.bfloat16) * 0.1
    g = (torch.randn(B, T, H, K, dtype=torch.float32) * 0.1 - 0.05)
    beta = torch.sigmoid(torch.randn(B, T, H, dtype=torch.bfloat16))
    return [q, k, v, g, beta]


def get_init_inputs():
    return [B, T, H, K, V, CHUNK_SIZE]
