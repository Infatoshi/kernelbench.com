import torch
import triton
import triton.language as tl

@triton.jit
def index_kernel(Y, K: tl.constexpr, BLOCK_N: tl.constexpr, PAD_2K: tl.constexpr):
    # Construct vals of size BLOCK_N
    idx_block = tl.arange(0, BLOCK_N)
    vals = idx_block * 1.5
    
    # Construct run_values of size PAD_2K
    idx_pad = tl.arange(0, PAD_2K)
    run_values = idx_pad * 10.0
    
    # We want to merge run_values (first K) and vals (first K)
    # index map for vals: if K <= i < 2*K, index is i - K, else 0
    gather_idx = tl.where((idx_pad >= K) & (idx_pad < 2*K), idx_pad - K, 0)
    
    # Gather from vals using gather_idx
    gathered_vals = vals[gather_idx]
    
    # Now merge: if i < K, run_values, else gathered_vals (if i < 2*K)
    merged = tl.where(idx_pad < K, run_values, gathered_vals)
    # For i >= 2*K, set to -1e38
    merged = tl.where(idx_pad < 2*K, merged, -1e38)
    
    tl.store(Y + idx_pad, merged)

def main():
    y = torch.zeros(16, dtype=torch.float32, device="cuda")
    index_kernel[(1,)](y, K=4, BLOCK_N=8, PAD_2K=16)
    print("Gather from register worked, output:", y)

if __name__ == "__main__":
    main()

import torch
import triton
import triton.language as tl
import time

@triton.jit
def sort_large_kernel(X, Y, I, N: tl.constexpr):
    idx = tl.arange(0, N)
    x = tl.load(X + idx)
    
    # Sort values and indices
    sorted_x, sorted_idx = tl.sort((x, idx), dim=0, descending=True)
    
    tl.store(Y + idx, sorted_x)
    tl.store(I + idx, sorted_idx)

def main():
    for N in [4096, 8192, 16384, 32768, 65536]:
        x = torch.randn(N, dtype=torch.float32, device="cuda")
        y = torch.zeros(N, dtype=torch.float32, device="cuda")
        i = torch.zeros(N, dtype=torch.int32, device="cuda")
        
        try:
            # Warmup
            sort_large_kernel[(1,)](x, y, i, N=N)
            torch.cuda.synchronize()
            
            # Time
            t0 = time.time()
            for _ in range(100):
                sort_large_kernel[(1,)](x, y, i, N=N)
            torch.cuda.synchronize()
            t1 = time.time()
            ms = (t1 - t0) * 10.0
            print(f"N={N} worked, time={ms:.3f} ms")
        except Exception as e:
            print(f"N={N} failed: {e}")

if __name__ == "__main__":
    main()

import torch
import triton
import triton.language as tl

@triton.jit
def pack_sort_kernel(X, Y_val, Y_idx, N: tl.constexpr):
    idx = tl.arange(0, N)
    f = tl.load(X + idx)
    
    # 1. Float to sortable uint32
    f_int = f.to(tl.int32, bitcast=True)
    # Use shift right arithmetic or logical. Let's cast to int32 for shifting
    sign = f_int >> 31
    mask = tl.where(sign != 0, 0xFFFFFFFF, 0x80000000)
    sortable_key = f_int ^ mask
    
    # Cast to int64 and shift left by 32
    # In Triton, we need to be careful with unsigned/signed shifts or casting
    key_64 = sortable_key.to(tl.int64) & 0xFFFFFFFF
    packed = (key_64 << 32) | (idx.to(tl.int64) & 0xFFFFFFFF)
    
    # 2. Sort the packed 64-bit int
    sorted_packed = tl.sort(packed, dim=0, descending=True)
    
    # 3. Unpack
    sorted_key_u32 = (sorted_packed >> 32) & 0xFFFFFFFF
    sorted_idx = sorted_packed & 0xFFFFFFFF
    
    # Revert sortable_key to float
    is_pos = (sorted_key_u32 & 0x80000000) != 0
    revert_mask = tl.where(is_pos, 0x80000000, 0xFFFFFFFF)
    revert_f_int = sorted_key_u32 ^ revert_mask
    sorted_f = revert_f_int.to(tl.int32).to(tl.float32, bitcast=True)
    
    tl.store(Y_val + idx, sorted_f)
    tl.store(Y_idx + idx, sorted_idx)

def main():
    N = 16
    x = torch.tensor([1.5, -2.0, 0.0, 10.5, -0.5, 3.2, -1.2, 5.0, 5.0, -10.0, 0.1, 0.2, 0.3, -0.1, -0.2, -0.3], dtype=torch.float32, device="cuda")
    y_val = torch.zeros(N, dtype=torch.float32, device="cuda")
    y_idx = torch.zeros(N, dtype=torch.int64, device="cuda")
    pack_sort_kernel[(1,)](x, y_val, y_idx, N=N)
    
    # Compare with torch
    ref_val, ref_idx = torch.sort(x, descending=True)
    print("X:        ", x)
    print("Ref Val:  ", ref_val)
    print("TritonVal:", y_val)
    print("Ref Idx:  ", ref_idx)
    print("TritonIdx:", y_idx)
    
    assert torch.allclose(ref_val, y_val), "Values mismatch!"
    # Let's check gathered values match
    gathered = x[y_idx]
    assert torch.allclose(ref_val, gathered), "Gathered values mismatch!"
    print("ALL TESTS PASSED!")

if __name__ == "__main__":
    main()

@triton.jit
def join_kernel(Y, K: tl.constexpr):
    # Create two tensors of size K
    a = tl.arange(0, K) * 1.0
    b = tl.arange(0, K) * 10.0
    
    # Join them (creates shape [K, 2])
    joined = tl.join(a, b)
    
    # Reshape to [2 * K]
    reshaped = tl.reshape(joined, (2 * K,))
    
    # Store to Y
    idx = tl.arange(0, 2 * K)
    tl.store(Y + idx, reshaped)

def main():
