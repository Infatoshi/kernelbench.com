cpp_source = r"""
#include <torch/extension.h>
#include <cuda_fp8.h>
#include <cuda_bf16.h>

void launch_fp8_gemm(const __nv_fp8_e4m3* A, const __nv_fp8_e4m3* B, __nv_bfloat16* C, int M, int N, int K) {
    dim3 grid((N + 127) / 128, (M + 127) / 128);
    dim3 block(128);
    size_t smem_size = 128 * 32 * 2 * sizeof(__nv_fp8_e4m3) + 128 * 32 * 2 * sizeof(__nv_fp8_e4m3);
    fp8_gemm_kernel<<<grid, block, smem_size>>>(A, B, C, M, N, K);
}

torch::Tensor fp8_gemm(torch::Tensor A, torch::Tensor B) {
    int M = A.size(0);
    int K = A.size(1);
    int N = B.size(0);

    auto C = torch::empty({M, N}, torch::TensorOptions().dtype(torch::kBFloat16).device(A.device()));

    launch_fp8_gemm(
        reinterpret_cast<const __nv_fp8_e4m3*>(A.data_ptr()),
        reinterpret_cast<const __nv_fp8_e4m3*>(B.data_ptr()),
        reinterpret_cast<__nv_bfloat16*>(C.data_ptr()),
        M, N, K
    );
    return C;
}
"""
