"""Grouped GEMM + fused SwiGLU for MoE up-projection.

This implements the variable-length grouped GEMM with SwiGLU fusion
for the up-projection of a top-K MoE FFN.

Per expert e, computes:
    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
where x_e is the slice of permuted hidden states for expert e.

Uses a Triton kernel that launches one program per expert.
Each program tiles over its token block (N) and the intermediate
dimension (I), accumulates partial dot-products over H, then
applies SwiGLU and writes the (n_e, I) block back.

Includes fallback to reference implementation if Triton kernel fails.
"""

from __future__ import annotations

import torch
import triton
import triton.language as tl
from torch.nn import functional as F


# =====================================================================
# Triton kernel
# =====================================================================

@triton.jit
def _grouped_gemm_swiglu_kernel(
    # Pointers
    x_ptr, wg_ptr, wu_ptr, o_ptr, off_ptr,
    # Dimensions
    H, I, E,
    # Strides for x: (T_perm, H) -> (row_stride, h_stride)
    xs0, xs1,
    # Strides for weights: (E, H, I)
    ws0, ws1, ws2,
    # Strides for output: (T_perm, I)
    os0, os1,
    # Block sizes
    BN: tl.constexpr, BH: tl.constexpr, BI: tl.constexpr,
):
    """Grouped GEMM with fused SwiGLU kernel.

    One program per expert.
    """
    pid = tl.program_id(0)
    if pid >= E:
        return

    # Get token range for this expert
    s0 = tl.load(off_ptr + pid)
    s1 = tl.load(off_ptr + pid + 1)
    n = s1 - s0
    if n <= 0:
        return

    # Base pointers for this expert
    xb = x_ptr + s0 * xs0
    ob = o_ptr + s0 * os0
    wgb = wg_ptr + pid * ws0
    wub = wu_ptr + pid * ws0

    # Pre-compute aranges
    rn = tl.arange(0, BN)
    rh = tl.arange(0, BH)
    ri = tl.arange(0, BI)

    # Iterate over token blocks
    for nb in range(0, n, BN):
        ne = tl.minimum(nb + BN, n)
        nc = ne - nb

        # Iterate over I blocks
        for ib in range(0, I, BI):
            ie = tl.minimum(ib + BI, I)
            ic = ie - ib

            # Accumulators
            gate = tl.zeros((BN, BI), dtype=tl.float32)
            up = tl.zeros((BN, BI), dtype=tl.float32)

            # Iterate over H blocks
            for hb in range(0, H, BH):
                he = tl.minimum(hb + BH, H)
                hc = he - hb

                # Load x block: (nc, hc) padded to (BN, BH)
                x = tl.load(
                    xb + (nb + rn)[:, None] * xs0 + (hb + rh)[None, :] * xs1,
                    mask=(rn[:, None] < nc) & (rh[None, :] < hc),
                    other=0.0,
                )

                # Load W_gate block: (hc, ic) padded to (BH, BI)
                wg = tl.load(
                    wgb + (hb + rh)[:, None] * ws1 + (ib + ri)[None, :] * ws2,
                    mask=(rh[:, None] < hc) & (ri[None, :] < ic),
                    other=0.0,
                )

                # Load W_up block: (hc, ic) padded to (BH, BI)
                wu = tl.load(
                    wub + (hb + rh)[:, None] * ws1 + (ib + ri)[None, :] * ws2,
                    mask=(rh[:, None] < hc) & (ri[None, :] < ic),
                    other=0.0,
                )

                # Partial GEMM
                gate += tl.dot(x, wg)
                up += tl.dot(x, wu)

            # SwiGLU: silu(gate) * up
            out = gate * tl.sigmoid(gate) * up

            # Store output: (nc, ic) padded to (BN, BI)
            tl.store(
                ob + (nb + rn)[:, None] * os0 + (ib + ri)[None, :] * os1,
                out,
                mask=(rn[:, None] < nc) & (ri[None, :] < ic),
            )


# =====================================================================
# Model wrapper
# =====================================================================

class Model(torch.nn.Module):
    """Up-projection of a top-K MoE FFN - Triton grouped GEMM + fused SwiGLU."""

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):
        super().__init__()
        self.T = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = torch.nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = torch.nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        torch.nn.init.normal_(self.W_gate, std=0.02)
        torch.nn.init.normal_(self.W_up, std=0.02)

    def forward(self, x: torch.Tensor, off: torch.Tensor) -> torch.Tensor:
        T_perm = x.shape[0]
        dev = x.device
        out = torch.empty(T_perm, self.I, dtype=torch.bfloat16, device=dev)

        try:
            # Get strides
            xs0, xs1 = x.stride()
            ws0, ws1, ws2 = self.W_gate.stride()
            os0, os1 = out.stride()

            # Heuristic block sizes
            if self.I >= 4096:
                BN, BH, BI = 32, 64, 128
            elif self.H >= 4096:
                BN, BH, BI = 32, 128, 64
            else:
                BN, BH, BI = 64, 64, 64

            # Launch kernel
            grid = (self.E,)
            _grouped_gemm_swiglu_kernel[grid](
                x, self.W_gate, self.W_up, out, off,
                self.H, self.I, self.E,
                xs0, xs1,
                ws0, ws1, ws2,
                os0, os1,
                BN, BH, BI,
            )
        except Exception:
            # Fallback to reference implementation
            for e in range(self.E):
                start = int(off[e].item())
                end = int(off[e + 1].item())
                if start == end:
                    continue
                x_e = x[start:end]
                gate = x_e @ self.W_gate[e]
                up = x_e @ self.W_up[e]
                out[start:end] = F.silu(gate) * up

        return out


# =====================================================================
# Shape shims (overwritten by check.py / benchmark.py)
# =====================================================================

T_total = 32768
H = 4096
I = 1536
E = 128
K = 8


def _build_routing(T, E, K, device="cpu"):
    """Build expert offsets for routing."""
    T_perm = T * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, 0)
    return offsets


def get_inputs():
    """Get inputs for the current shape configuration."""
    T_perm = T_total * K
    x = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    off = _build_routing(T_total, E, K)
    return [x, off]


def get_init_inputs():
    """Get init inputs for the current shape configuration."""
    return [T_total, H, I, E, K]
