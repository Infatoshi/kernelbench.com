"""Top-k via custom CUDA: hierarchical multi-block + bitonic merge."""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <cstdint>
#include <cfloat>

struct Pair {
    float val;
    int idx;
};

__host__ __device__ int next_pow2(int x) {
    int p = 1;
    while (p < x) p <<= 1;
    return p;
}

__device__ __forceinline__ void swap_pair(Pair& a, Pair& b) {
    Pair t = a; a = b; b = t;
}

__device__ __forceinline__ void insert_topk(Pair* top, int k, float val, int idx) {
    if (val <= top[k - 1].val) return;
    int pos = k - 1;
    while (pos > 0 && val > top[pos - 1].val) {
        top[pos] = top[pos - 1];
        pos--;
    }
    top[pos].val = val;
    top[pos].idx = idx;
}

__device__ void bitonic_sort_desc(Pair* data, int n) {
    for (int size = 2; size <= n; size <<= 1) {
        for (int dia = size >> 1; dia > 0; dia >>= 1) {
            for (int i = threadIdx.x; i < n; i += blockDim.x) {
                int opp = i ^ dia;
                if (opp > i) {
                    bool up = (i & size) == 0;
                    float vi = data[i].val;
                    float vo = data[opp].val;
                    if (up) {
                        if (vi < vo) swap_pair(data[i], data[opp]);
                    } else {
                        if (vi > vo) swap_pair(data[i], data[opp]);
                    }
                }
            }
            __syncthreads();
        }
    }
}

__device__ void merge_topk_smem(Pair* smem, int total, int k) {
    int sort_n = next_pow2(total);
    for (int i = threadIdx.x + total; i < sort_n; i += blockDim.x) {
        smem[i].val = -FLT_MAX;
        smem[i].idx = -1;
    }
    __syncthreads();
    bitonic_sort_desc(smem, sort_n);
}

__device__ void scan_partition_topk(
    const float* __restrict__ row, int start, int len, int k, int scan_threads,
    Pair* __restrict__ merge_out
) {
    extern __shared__ Pair smem[];
    Pair* my_top = smem;
    Pair* merge = smem + (size_t)scan_threads * k;

    if (threadIdx.x < scan_threads) {
        Pair* top = my_top + threadIdx.x * k;
        for (int j = 0; j < k; j++) {
            top[j].val = -FLT_MAX;
            top[j].idx = -1;
        }
    }
    __syncthreads();

    if (threadIdx.x < scan_threads) {
        Pair* top = my_top + threadIdx.x * k;
        int tstart = start + (int)((int64_t)threadIdx.x * len / scan_threads);
        int tend   = start + (int)((int64_t)(threadIdx.x + 1) * len / scan_threads);

        int i = tstart;
        while (i < tend && (reinterpret_cast<uintptr_t>(row + i) & 15)) {
            insert_topk(top, k, row[i], i);
            i++;
        }
        for (; i + 3 < tend; i += 4) {
            float4 v = *reinterpret_cast<const float4*>(row + i);
            insert_topk(top, k, v.x, i);
            insert_topk(top, k, v.y, i + 1);
            insert_topk(top, k, v.z, i + 2);
            insert_topk(top, k, v.w, i + 3);
        }
        for (; i < tend; i++) {
            insert_topk(top, k, row[i], i);
        }
    }
    __syncthreads();

    int total = scan_threads * k;
    for (int i = threadIdx.x; i < total; i += blockDim.x) {
        merge[i] = smem[i];
    }
    __syncthreads();

    merge_topk_smem(merge, total, k);
    __syncthreads();

    if (threadIdx.x < k) {
        merge_out[threadIdx.x] = merge[threadIdx.x];
    }
}

__device__ void row_argmax(const float* __restrict__ row, int n, float* out_val, int64_t* out_idx) {
    float local_max = -FLT_MAX;
    int local_idx = 0;

    int i = threadIdx.x * 4;
    for (; i + 3 < n; i += blockDim.x * 4) {
        float4 v = *reinterpret_cast<const float4*>(row + i);
        if (v.x > local_max) { local_max = v.x; local_idx = i; }
        if (v.y > local_max) { local_max = v.y; local_idx = i + 1; }
        if (v.z > local_max) { local_max = v.z; local_idx = i + 2; }
        if (v.w > local_max) { local_max = v.w; local_idx = i + 3; }
    }
    for (; i < n; i += blockDim.x) {
        float v = row[i];
        if (v > local_max) { local_max = v; local_idx = i; }
    }

    __shared__ float smax[128];
    __shared__ int sidx[128];
    smax[threadIdx.x] = local_max;
    sidx[threadIdx.x] = local_idx;
    __syncthreads();

    for (int stride = blockDim.x / 2; stride > 0; stride >>= 1) {
        if (threadIdx.x < stride) {
            if (smax[threadIdx.x + stride] > smax[threadIdx.x]) {
                smax[threadIdx.x] = smax[threadIdx.x + stride];
                sidx[threadIdx.x] = sidx[threadIdx.x + stride];
            }
        }
        __syncthreads();
    }

    if (threadIdx.x == 0) {
        *out_val = smax[0];
        *out_idx = sidx[0];
    }
}

__global__ void topk_single_kernel(
    const float* __restrict__ input,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int n, int k, int scan_threads
) {
    int row = blockIdx.x;
    if (row >= batch) return;
    const float* row_ptr = input + (int64_t)row * n;
    float* out_v = out_vals + (int64_t)row * k;
    int64_t* out_i = out_idxs + (int64_t)row * k;

    if (k == 1) {
        row_argmax(row_ptr, n, out_v, out_i);
        return;
    }

    extern __shared__ Pair smem[];
    Pair* merge = smem + (size_t)scan_threads * k;
    scan_partition_topk(row_ptr, 0, n, k, scan_threads, merge);
    __syncthreads();
    if (threadIdx.x < k) {
        out_v[threadIdx.x] = merge[threadIdx.x].val;
        out_i[threadIdx.x] = merge[threadIdx.x].idx;
    }
}

__global__ void topk_tile_kernel(
    const float* __restrict__ input,
    float* __restrict__ partial_vals,
    int64_t* __restrict__ partial_idxs,
    int batch, int n, int k, int scan_threads, int num_tiles
) {
    int tile = blockIdx.x;
    int row = blockIdx.y;
    if (row >= batch || tile >= num_tiles) return;

    const float* row_ptr = input + (int64_t)row * n;
    int tile_len = n / num_tiles;
    int start = tile * tile_len;
    int len = (tile == num_tiles - 1) ? (n - start) : tile_len;

    extern __shared__ Pair smem[];
    Pair* merge = smem + (size_t)scan_threads * k;
    scan_partition_topk(row_ptr, start, len, k, scan_threads, merge);
    __syncthreads();

    float* pv = partial_vals + ((int64_t)row * num_tiles + tile) * k;
    int64_t* pi = partial_idxs + ((int64_t)row * num_tiles + tile) * k;
    if (threadIdx.x < k) {
        pv[threadIdx.x] = merge[threadIdx.x].val;
        pi[threadIdx.x] = merge[threadIdx.x].idx;
    }
}

__global__ void topk_merge_kernel(
    const float* __restrict__ partial_vals,
    const int64_t* __restrict__ partial_idxs,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int k, int num_tiles
) {
    int row = blockIdx.x;
    if (row >= batch) return;

    extern __shared__ Pair smem[];
    int total = num_tiles * k;

    const float* pv = partial_vals + (int64_t)row * num_tiles * k;
    const int64_t* pi = partial_idxs + (int64_t)row * num_tiles * k;

    for (int i = threadIdx.x; i < total; i += blockDim.x) {
        smem[i].val = pv[i];
        smem[i].idx = pi[i];
    }
    __syncthreads();

    merge_topk_smem(smem, total, k);
    __syncthreads();

    float* ov = out_vals + (int64_t)row * k;
    int64_t* oi = out_idxs + (int64_t)row * k;
    if (threadIdx.x < k) {
        ov[threadIdx.x] = smem[threadIdx.x].val;
        oi[threadIdx.x] = smem[threadIdx.x].idx;
    }
}

std::tuple<torch::Tensor, torch::Tensor> topk_cuda(torch::Tensor input, int k) {
    TORCH_CHECK(input.is_cuda(), "input must be CUDA");
    TORCH_CHECK(input.dtype() == torch::kFloat32, "input must be float32");
    TORCH_CHECK(input.dim() == 2, "input must be 2D");
    TORCH_CHECK(k >= 1 && k <= 64, "k must be in [1, 64]");
    int batch = input.size(0);
    int n = input.size(1);

    auto opts_f = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    auto opts_i = torch::TensorOptions().dtype(torch::kInt64).device(input.device());
    auto out_vals = torch::empty({batch, k}, opts_f);
    auto out_idxs = torch::empty({batch, k}, opts_i);

    int scan_threads = 128;
    if (k >= 16) scan_threads = 64;
    if (k >= 32) scan_threads = 32;

    int block_threads = 256;
    int total_pairs = scan_threads * k;
    int sort_n = next_pow2(total_pairs);
    int smem_bytes = sort_n * (int)sizeof(Pair) + total_pairs * (int)sizeof(Pair);

    if (k == 1) {
        topk_single_kernel<<<batch, 128>>>(
            input.data_ptr<float>(), out_vals.data_ptr<float>(), out_idxs.data_ptr<int64_t>(),
            batch, n, k, scan_threads);
        return std::make_tuple(out_vals, out_idxs);
    }

    int num_tiles = 1;
    if (n > 8192) num_tiles = (n + 8191) / 8192;
    if (num_tiles > 128) num_tiles = 128;

    if (num_tiles == 1) {
        topk_single_kernel<<<batch, block_threads, smem_bytes>>>(
            input.data_ptr<float>(), out_vals.data_ptr<float>(), out_idxs.data_ptr<int64_t>(),
            batch, n, k, scan_threads);
    } else {
        auto partial_vals = torch::empty({batch, num_tiles, k}, opts_f);
        auto partial_idxs = torch::empty({batch, num_tiles, k}, opts_i);

        dim3 grid(num_tiles, batch);
        topk_tile_kernel<<<grid, block_threads, smem_bytes>>>(
            input.data_ptr<float>(),
            partial_vals.data_ptr<float>(),
            partial_idxs.data_ptr<int64_t>(),
            batch, n, k, scan_threads, num_tiles);

        int merge_total = num_tiles * k;
        int merge_sort_n = next_pow2(merge_total);
        int merge_smem = merge_sort_n * (int)sizeof(Pair);

        topk_merge_kernel<<<batch, 256, merge_smem>>>(
            partial_vals.data_ptr<float>(),
            partial_idxs.data_ptr<int64_t>(),
            out_vals.data_ptr<float>(),
            out_idxs.data_ptr<int64_t>(),
            batch, k, num_tiles);
    }

    return std::make_tuple(out_vals, out_idxs);
}
"""

CPP_SRC = r"""
#include <torch/extension.h>
std::tuple<torch::Tensor, torch::Tensor> topk_cuda(torch::Tensor input, int k);
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("topk_cuda", &topk_cuda, "TopK CUDA");
}
"""

_topk_ext = None


def _get_ext():
    global _topk_ext
    if _topk_ext is None:
        _topk_ext = load_inline(
            name="topk_cuda_ext_hier",
            cpp_sources=CPP_SRC,
            cuda_sources=CUDA_SRC,
            functions=None,
            extra_cuda_cflags=["-O3", "--use_fast_math"],
            verbose=False,
        )
    return _topk_ext


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        ext = _get_ext()
        return ext.topk_cuda(x.contiguous(), self.k)


batch = 64
n = 8192
k = 8


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
