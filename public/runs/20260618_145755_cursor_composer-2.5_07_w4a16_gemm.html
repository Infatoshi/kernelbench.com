AWQ/GPTQ-style asymmetric int4 with per-group bf16 scales and zero-points.
Two kernels: GEMV (M==1, N-major repacked weights) and GEMM (M>1).

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "gemm_w4a16"
SUPPORTED_PRECISIONS = ["int4_bf16"]
HARDWARE_REQUIRED = ["RTX_3090", "RTX_PRO_6000", "H100", "B200"]

GROUP_SIZE = 128


@triton.jit
def _gemm_kernel(
    x_ptr, w_ptr, s_ptr, z_ptr, out_ptr,
    M, N, K,
    stride_xm,
    stride_wk,
    stride_sg,
    BLOCK_M: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_K: tl.constexpr,
    HALF_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_n_tiles = tl.cdiv(N, BLOCK_N)
    pid_m = pid // num_n_tiles
    pid_n = pid % num_n_tiles

    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_m = offs_m < M
    mask_n = offs_n < N

    idx_half = tl.arange(0, HALF_K)
    x_row = x_ptr + offs_m[:, None] * stride_xm

    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    for g in range(0, K // BLOCK_K):
        k0 = g * BLOCK_K
        k_even = k0 + 2 * idx_half
        k_odd = k0 + 2 * idx_half + 1
        x_lo = tl.load(x_row + k_even[None, :], mask=mask_m[:, None], other=0.0)
        x_hi = tl.load(x_row + k_odd[None, :], mask=mask_m[:, None], other=0.0)

        w_ptrs = w_ptr + (k0 // 2 + idx_half[:, None]) * stride_wk + offs_n[None, :]
        w_pack = tl.load(w_ptrs, mask=mask_n[None, :], other=0)
        w_lo = (w_pack & 0xF).to(tl.float32)
        w_hi = ((w_pack >> 4) & 0xF).to(tl.float32)

        s = tl.load(s_ptr + g * stride_sg + offs_n, mask=mask_n, other=0.0)
        z = tl.load(z_ptr + g * stride_sg + offs_n, mask=mask_n, other=0.0)
        w_lo_f = ((w_lo - z[None, :]) * s[None, :]).to(tl.bfloat16)
        w_hi_f = ((w_hi - z[None, :]) * s[None, :]).to(tl.bfloat16)

        acc = tl.dot(x_lo, w_lo_f, acc=acc)
        acc = tl.dot(x_hi, w_hi_f, acc=acc)

    out_ptrs = out_ptr + offs_m[:, None] * N + offs_n[None, :]
    tl.store(out_ptrs, acc.to(tl.bfloat16), mask=mask_m[:, None] & mask_n[None, :])


def _launch_gemm(x, w_q, scales, zeros, M, N, K):
    out = torch.empty((M, N), dtype=torch.bfloat16, device=x.device)
    # Tuned on RTX 3090 (SM86).
    if M >= 128:
        bm, bn, nw, ns = 32, 256, 4, 2
    elif M >= 32:
        bm, bn, nw, ns = 16, 128, 4, 2
    else:
        bm, bn, nw, ns = 16, 128, 8, 2
    bk = GROUP_SIZE
    grid = (triton.cdiv(M, bm) * triton.cdiv(N, bn),)
    _gemm_kernel[grid](
        x, w_q, scales, zeros, out,
        M, N, K,
        x.stride(0), w_q.stride(0), scales.stride(0),
        BLOCK_M=bm, BLOCK_N=bn, BLOCK_K=bk, HALF_K=bk // 2,
        num_warps=nw, num_stages=ns,
    )
    return out


@triton.jit
def _gemv_kernel(
    x_ptr, wt_ptr, st_ptr, zt_ptr, out_ptr,
    N, K, KP, G,
    stride_wn,
    stride_gn,
    BLOCK_N: tl.constexpr,
    HALF_K: tl.constexpr,
    BLOCK_K: tl.constexpr,
):
    pid_n = tl.program_id(0)
    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
    mask_n = offs_n < N

    acc = tl.zeros((BLOCK_N,), dtype=tl.float32)
    n_ptr_w = wt_ptr + offs_n * stride_wn
    n_ptr_s = st_ptr + offs_n * stride_gn
    n_ptr_z = zt_ptr + offs_n * stride_gn

    for g in range(0, G):
        k0 = g * BLOCK_K
        idx_half = tl.arange(0, HALF_K)
        k_even = k0 + 2 * idx_half
        k_odd = k0 + 2 * idx_half + 1
        x_lo = tl.load(x_ptr + k_even)
        x_hi = tl.load(x_ptr + k_odd)

        p = tl.load(
            n_ptr_w[:, None] + (k0 // 2 + idx_half)[None, :],
            mask=mask_n[:, None],
            other=0,
        )
        lo = (p & 0xF).to(tl.float32)
        hi = ((p >> 4) & 0xF).to(tl.float32)
        s = tl.load(n_ptr_s + g, mask=mask_n, other=0.0)
        z = tl.load(n_ptr_z + g, mask=mask_n, other=0.0)
        lo = (lo - z[:, None]) * s[:, None]
        hi = (hi - z[:, None]) * s[:, None]
        acc += tl.sum(x_lo[None, :] * lo, axis=1)
        acc += tl.sum(x_hi[None, :] * hi, axis=1)

    tl.store(out_ptr + offs_n, acc.to(tl.bfloat16), mask=mask_n)

    kp = K // 2
    g = K // GROUP_SIZE
    out = torch.empty((1, N), dtype=torch.bfloat16, device=x.device)
    if N >= 8192:

    else:
        bn, nw, ns = 8, 4, 2

        wt.stride(0), st.stride(0),

        num_warps=nw, num_stages=ns,
    )

+@triton.jit
+def _gemm_kernel(
+    x_ptr, w_ptr, s_ptr, z_ptr, out_ptr,
+    M, N, K,
+    stride_xm,
+    stride_wk,
+    stride_sg,
+    BLOCK_M: tl.constexpr,
+    BLOCK_N: tl.constexpr,
+    BLOCK_K: tl.constexpr,
+    HALF_K: tl.constexpr,
+):
+    pid = tl.program_id(0)
+    num_n_tiles = tl.cdiv(N, BLOCK_N)
+    pid_m = pid // num_n_tiles
+    pid_n = pid % num_n_tiles
+
+    offs_m = pid_m * BLOCK_M + tl.arange(0, BLOCK_M)
+    offs_n = pid_n * BLOCK_N + tl.arange(0, BLOCK_N)
+    mask_m = offs_m < M
+    mask_n = offs_n < N

+    idx_half = tl.arange(0, HALF_K)
+    x_row = x_ptr + offs_m[:, None] * stride_xm
+
+    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
+
+    for g in range(0, K // BLOCK_K):
+        k0 = g * BLOCK_K
+        k_even = k0 + 2 * idx_half
+        k_odd = k0 + 2 * idx_half + 1
+        x_lo = tl.load(x_row + k_even[None, :], mask=mask_m[:, None], other=0.0)
+        x_hi = tl.load(x_row + k_odd[None, :], mask=mask_m[:, None], other=0.0)
+

+        w_pack = tl.load(w_ptrs, mask=mask_n[None, :], other=0)
+        w_lo = (w_pack & 0xF).to(tl.float32)
+        w_hi = ((w_pack >> 4) & 0xF).to(tl.float32)
+

+        w_hi_f = ((w_hi - z[None, :]) * s[None, :]).to(tl.bfloat16)
+

+        acc = tl.dot(x_hi, w_hi_f, acc=acc)
