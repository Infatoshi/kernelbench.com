import math

@triton.jit
def _paged_attn_split_kernel(
    Q_ptr, KV_ptr, BT_ptr, SL_ptr,
    MI_ptr, LI_ptr, ACC_ptr,
    scale,
    H: tl.constexpr, Hkv: tl.constexpr,
    D: tl.constexpr, P: tl.constexpr,
    S: tl.constexpr,
    max_pages,
    sq_b, sq_h, sq_d,
    skv_blk, skv_p, skv_hkv, skv_d,
    sbt_b, sbt_p,
    smi_stride, sli_stride,
    sacc_b, sacc_kvh, sacc_s, sacc_d,
    NUM_STAGES: tl.constexpr,
):
    pid = tl.program_id(0)
    total_splits = Hkv * S
    bid = pid // total_splits
    rem = pid % total_splits
    kv_hid = rem // S
    sid = rem % S

    G: tl.constexpr = H // Hkv

    seq_len_b = tl.load(SL_ptr + bid).to(tl.int32)
    total_pages_seq = (seq_len_b + P - 1) // P
    pages_per_split = (total_pages_seq + S - 1) // S
    start_page = sid * pages_per_split
    end_page = min(start_page + pages_per_split, total_pages_seq)

    d_offs = tl.arange(0, D)
    g_offs = tl.arange(0, G)
    h_offs = kv_hid * G + g_offs

    q = tl.load(
        Q_ptr + bid * sq_b + h_offs[:, None] * sq_h + d_offs[None, :] * sq_d,
    ).to(tl.float32)

    m_val = tl.full((G,), value=-1.0e10, dtype=tl.float32)
    l_val = tl.zeros((G,), dtype=tl.float32)
    acc = tl.zeros((G, D), dtype=tl.float32)

    p_offs = tl.arange(0, P)

    for page_idx in tl.range(start_page, max_pages, num_stages=NUM_STAGES):
        if page_idx < end_page:
            phys = tl.load(BT_ptr + bid * sbt_b + page_idx * sbt_p).to(tl.int64)

            k_base = KV_ptr + phys * skv_blk + kv_hid * skv_hkv
            k = tl.load(
                k_base + p_offs[:, None] * skv_p + d_offs[None, :] * skv_d,
            ).to(tl.float32)

            scores = tl.dot(q, tl.trans(k)) * scale

            token_offs = page_idx * P + p_offs
            valid_mask = (token_offs[None, :] < seq_len_b)
            g_valid = (g_offs[:, None] < G)
            scores = tl.where(valid_mask & g_valid, scores, -1.0e10)

            page_max = tl.max(scores, axis=1)
            new_m = tl.maximum(m_val, page_max)
            alpha = tl.exp(m_val - new_m)
            exp_scores = tl.exp(scores - new_m[:, None])
            page_l = tl.sum(exp_scores, axis=1)
            new_l = l_val * alpha + page_l

            v_base = KV_ptr + phys * skv_blk + kv_hid * skv_hkv + D * skv_d
            v = tl.load(
                v_base + p_offs[:, None] * skv_p + d_offs[None, :] * skv_d,
            ).to(tl.float32)

            exp_v = exp_scores.to(v.dtype)
            acc = acc * alpha[:, None] + tl.dot(exp_v, v)
            m_val = new_m
            l_val = new_l

    mi_off = bid * smi_stride + kv_hid * S * G + sid * G + g_offs
    tl.store(MI_ptr + mi_off, m_val, mask=(g_offs < G))
    tl.store(LI_ptr + mi_off, l_val, mask=(g_offs < G))

    acc_off = bid * sacc_b + kv_hid * sacc_kvh + sid * sacc_s + g_offs[:, None] * sacc_d + d_offs[None, :] * 1
    tl.store(ACC_ptr + acc_off, acc, mask=(g_offs[:, None] < G) & (d_offs[None, :] < D))


@triton.jit
def _paged_attn_combine_kernel(
    MI_ptr, LI_ptr, ACC_ptr,
    O_ptr,
    H: tl.constexpr, Hkv: tl.constexpr,
    D: tl.constexpr, S: tl.constexpr,
    smi_stride, sli_stride,
    sacc_b, sacc_kvh, sacc_s, sacc_d,
    so_b, so_h, so_d,
):
    pid = tl.program_id(0)
    G: tl.constexpr = H // Hkv
    bid = pid // H
    hid = pid % H
    kv_hid = hid // G
    g_idx = hid % G

    d_offs = tl.arange(0, D)

    mi_base = MI_ptr + bid * smi_stride + kv_hid * S * G + g_idx
    li_base = LI_ptr + bid * smi_stride + kv_hid * S * G + g_idx
    acc_base = ACC_ptr + bid * sacc_b + kv_hid * sacc_kvh + g_idx * sacc_d

    global_max = -1.0e10
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        global_max = tl.maximum(global_max, mi_s)

    global_l = 0.0
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        li_s = tl.load(li_base + s * G)
        global_l += li_s * tl.exp(mi_s - global_max)

    acc = tl.zeros((D,), dtype=tl.float32)
    for s in tl.static_range(0, S):
        mi_s = tl.load(mi_base + s * G)
        a_s = tl.exp(mi_s - global_max)
        partial = tl.load(acc_base + s * sacc_s + d_offs)
        acc += a_s * partial

    acc = acc / global_l

    out_base = O_ptr + bid * so_b + hid * so_h
    tl.store(out_base + d_offs * so_d, acc.to(tl.bfloat16))


def next_pow2(n: int) -> int:
    if n <= 1:
        return 1
    return 1 << (n - 1).bit_length()
