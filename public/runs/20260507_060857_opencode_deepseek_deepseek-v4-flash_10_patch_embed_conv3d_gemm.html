def _im2col_3d(x_ptr, buf_ptr,
    stride_x_b, stride_x_c, stride_x_t, stride_x_h, stride_x_w,
    T_out, H_out, W_out, num_patches, K,
    kT, kH, kW,
    BLOCK_M: tl.constexpr, BLOCK_K: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pid_m = tl.cdiv(num_patches, BLOCK_M)
    pid_m = pid % num_pid_m
    pid_k = pid // num_pid_m
    m_start = pid_m * BLOCK_M
    m_offsets = m_start + tl.arange(0, BLOCK_M)
    k_start = pid_k * BLOCK_K
    k_offsets = k_start + tl.arange(0, BLOCK_K)
    m_mask = m_offsets < num_patches
    k_mask = k_offsets < K
    _m = m_offsets[:, None]
    tom = T_out * H_out * W_out
    how = H_out * W_out
    _b = _m // tom
    r1 = _m - _b * tom
    _out_t = r1 // how
    r2 = r1 - _out_t * how
    _out_h = r2 // W_out
    _out_w = r2 - _out_h * W_out
    _k2 = k_offsets[None, :]
    prod_khw = kH * kW
    prod_tkhw = kT * prod_khw
    _c = _k2 // prod_tkhw
    rk1 = _k2 - _c * prod_tkhw
    _dt = rk1 // prod_khw
    rk2 = rk1 - _dt * prod_khw
    _dh = rk2 // kW
    _dw = rk2 - _dh * kW
    x_ptrs = (x_ptr + _b * stride_x_b + _c * stride_x_c
        + (_out_t * kT + _dt) * stride_x_t
        + (_out_h * kH + _dh) * stride_x_h
        + (_out_w * kW + _dw) * stride_x_w)
    x = tl.load(x_ptrs, mask=m_mask[:, None] & k_mask[None, :], other=0.0)
    buf_ptrs = buf_ptr + _m * K + _k2
    tl.store(buf_ptrs, x.to(tl.bfloat16), mask=m_mask[:, None] & k_mask[None, :])

def _gemm_patch_5d(a_ptr, w_ptr, out_ptr,
    stride_out_b, stride_out_d, stride_out_t, stride_out_h, stride_out_w,
    T_out, H_out, W_out, num_patches, K, N,
    kT, kH, kW,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,

    num_pid_m = tl.cdiv(num_patches, BLOCK_M)

    m_offsets = m_start + tl.arange(0, BLOCK_M)
    n_start = pid_n * BLOCK_N
    n_offsets = n_start + tl.arange(0, BLOCK_N)
    m_mask = m_offsets < num_patches
    n_mask = n_offsets < N
    _m = m_offsets[:, None]
    _n_2d = n_offsets[None, :]
    tom = T_out * H_out * W_out
    how = H_out * W_out
    _b = _m // tom
    r1 = _m - _b * tom
    _out_t = r1 // how
    r2 = r1 - _out_t * how
    _out_h = r2 // W_out
    _out_w = r2 - _out_h * W_out
    acc = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    for k_start in range(0, K, BLOCK_K):

        + _out_t * stride_out_t + _out_h * stride_out_h + _out_w * stride_out_w)
