Uses PyTorch bmm (cuBLAS) for batch matmuls (intra-chunk A, w, u) +
Triton fused kernel for the inter-chunk recurrence.

# ---------------------------------------------------------------------------
# Intra-chunk: PyTorch bmm + recursive inversion via torch.compile
# ---------------------------------------------------------------------------

def _intra_chunk(k, g, v, beta):
    """Compute w, u using batched matmul (cuBLAS)."""
    B, H, NT, BT, Kdim = k.shape
    Vdim = v.shape[-1]
    device = k.device

    k_f32 = k.to(torch.float32)
    v_f32 = v.to(torch.float32)
    beta_f32 = beta.to(torch.float32)
    exp_g = torch.exp(g)
    exp_neg_g = torch.exp(-g)
    k_exp_g = k_f32 * exp_g
    k_exp_neg_g = k_f32 * exp_neg_g

    M = B * H * NT
    k_exp_g_bmm = k_exp_g.reshape(M, BT, Kdim)
    k_exp_neg_g_bmm = k_exp_neg_g.reshape(M, BT, Kdim)

    # A = k_exp_g @ k_exp_neg_g.T  via cuBLAS batched matmul
    A = torch.bmm(k_exp_g_bmm, k_exp_neg_g_bmm.transpose(1, 2))
    A_view = A.view(B, H, NT, BT, BT)

    # Mask: zero upper triangle + diagonal, negate lower
    mask_diag_upper = torch.triu(
        torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=0
    )
    A_lower = -A_view.masked_fill(mask_diag_upper, 0.0)

    # Recursive inversion
    A_inv = _recursive_inversion(A_lower)

    # final_A = (A_inv + I) * beta_j
    eye = torch.eye(BT, dtype=torch.float32, device=device)
    final_A = (A_inv + eye.view(1, 1, 1, BT, BT)) * beta_f32[..., None, :]
    final_A_bmm = final_A.reshape(M, BT, BT)

    # w = final_A @ k_exp_g,  u = final_A @ v
    w = torch.bmm(final_A_bmm, k_exp_g_bmm).view(B, H, NT, BT, Kdim)
    u = torch.bmm(final_A_bmm, v_f32.reshape(M, BT, Vdim)).view(B, H, NT, BT, Vdim)

    return w, u


@torch.compile(fullgraph=False, mode="reduce-overhead")
def _recursive_inversion(A_lower: torch.Tensor) -> torch.Tensor:
    """Compute (I - A_lower)^{-1} - I via triangular recursion."""
    BT = A_lower.shape[-1]
    A = A_lower.clone()
    for i in range(1, BT):
        left = A[..., i, :, None]
        right = A[..., :, :i]
        increment = (left * right).sum(-2)
        A[..., i, :i] = A[..., i, :i] + increment
    return A

def _launch_inter_chunk(
    q_exp_g, k_exp_neg_g, k_fp32, g, exp_g, w, u,

    Vdim = w.shape[-1]  # w has Vdim == u has Vdim, but actually w is (...,K)
    # w shape: (B, H, NT, BT, K)
    Vdim = u.shape[-1]
    device = q_exp_g.device

    S_buf = torch.zeros(B * H, Kdim, Vdim, dtype=torch.float32, device=device)

    # Output
    o = torch.empty(B, H, NT, BT, Vdim, dtype=torch.float32, device=device)

    # Ensure contiguous
    q_exp_g = q_exp_g.contiguous()
    k_exp_neg_g = k_exp_neg_g.contiguous()
    k_fp32 = k_fp32.contiguous()
    g = g.contiguous()
    exp_g = exp_g.contiguous()
    w = w.contiguous()
    u = u.contiguous()

+"""KDA forward (chunk form) — Triton implementation for SM120 Blackwell.
+
+Two-kernel design:
+  1. _intra_chunk_kernel: parallel over (B, H, NT) — computes w, u from k, g, v, beta
+  2. _inter_chunk_kernel: parallel over (B, H) — sequential chunk recurrence → o

+
+from __future__ import annotations
+
+import torch
+import torch.nn as nn
+import triton
+import triton.language as tl
+from einops import rearrange
+
+OP_TYPE = "linear_attention"
+SUPPORTED_PRECISIONS = ["bf16"]
+HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]
+
+
+# ---------------------------------------------------------------------------
+# Intra-chunk kernel: for each chunk compute w (BT,K) and u (BT,V)
+# ---------------------------------------------------------------------------
+
+@triton.jit
+def _intra_chunk_kernel(
+    k_ptr, g_ptr, v_ptr, beta_ptr,
+    w_ptr, u_ptr,
+    BT: tl.constexpr, K: tl.constexpr, V: tl.constexpr,
+    stride_b: tl.constexpr, stride_h: tl.constexpr,
+    stride_nt: tl.constexpr, stride_bt_k: tl.constexpr,
+    stride_k: tl.constexpr,
+    stride_bt_v: tl.constexpr, stride_v: tl.constexpr,
+    BLOCK_K_K: tl.constexpr, BLOCK_K_V: tl.constexpr,
+):
+    """
+    One program per chunk. Computes:
+      A = -(k * exp(g)) @ (k * exp(-g)).T   [lower-triangular part]
+      A_inv = (I - A)^(-1) - I               [recursive triangular inversion]
+      final_A = (A_inv + I) * beta[None, :]
+      w = final_A @ (k * exp(g))
+      u = final_A @ v
+    """
+    pid = tl.program_id(0)
+    NT = tl.num_programs(0) // (1)  # NT is inferred, but we don't have it directly
+    # We'll pass NT separately; for now compute B*H*NT as num_programs
+
+    # --- decayed keys ------------------------------------------------------
+    # Load k and g for this chunk: shape (BT, K)
+    offsets_bt = tl.arange(0, BT)  # [0, BT)
+    offsets_k = tl.arange(0, BLOCK_K_K)  # [0, BLOCK_K_K)
+
+    # k_exp_g = k * exp(g), stored in registers/shared: (BT, K) in fp32
+    # We compute A = (k_exp_g) @ (k_exp_neg_g).T tile by tile
+    # A shape: (BT, BT), lower-tri part needed
+
+    # We store A in a (BT, BT) fp32 scratch in smem
+    A_smem = tl.zeros((BT, BT), dtype=tl.float32)
+
+    # Compute A block by block over K
+    for k_start in range(0, K, BLOCK_K_K):
+        k_off = k_start + offsets_k
+        k_mask = k_off < K
+
+        # Load k tile: (BT, BLOCK_K_K)
+        k_tile = tl.load(
+            k_ptr + offsets_bt[:, None] * stride_bt_k + k_off[None, :] * stride_k,
+            mask=(offsets_bt[:, None] < BT) & k_mask[None, :],
+            other=0.0,
+        ).to(tl.float32)
+
+        # Load g tile: (BT, BLOCK_K_K)
+        g_tile = tl.load(
+            g_ptr + offsets_bt[:, None] * stride_bt_k + k_off[None, :] * stride_k,
+            mask=(offsets_bt[:, None] < BT) & k_mask[None, :],
+            other=0.0,
+        )
+

+        exp_g = tl.math.exp(g_tile)
