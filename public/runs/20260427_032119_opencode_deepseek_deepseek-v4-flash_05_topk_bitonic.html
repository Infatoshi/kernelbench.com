"""Top-k via per-thread insertion sort + block-wide bitonic merge in shared memory.

Each block handles one row:
  1. Per-thread: strided access, maintain sorted top-k in registers via insertion.
  2. Block-wide: write all candidates to shared memory, bitonic sort once.
  3. Write first k elements as the final result.

No separate tile pass or merge kernel needed — one kernel, one sort.
"""

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

CPP_SOURCE = """
void launch_topk(
    at::Tensor x,
    at::Tensor out_vals,
    at::Tensor out_idxs,
    int batch, int n, int k
);
"""

CUDA_SOURCE = r"""
#include <torch/extension.h>
#include <cstdint>
#include <cuda_runtime.h>

constexpr int BLOCK = 64;
constexpr int MAX_K = 64;
constexpr int MAX_CAND = BLOCK * MAX_K;  // 4096

__device__ __forceinline__ void swapf(float &a, float &b) {
    float t = a; a = b; b = t;
}
__device__ __forceinline__ void swapi(int &a, int &b) {
    int t = a; a = b; b = t;
}
__device__ __forceinline__ void swapll(int64_t &a, int64_t &b) {
    int64_t t = a; a = b; b = t;
}

__global__ void topk_global(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int n, int k
) {
    int row = blockIdx.x;
    if (row >= batch) return;

    const float* x_row = x + row * n;
    int tid = threadIdx.x;

    // ---------------------------------------------------------------
    // Phase 1: per-thread top-k via insertion sort in registers
    // ---------------------------------------------------------------
    float my_vals[MAX_K];
    int   my_idxs[MAX_K];
    for (int i = 0; i < k; ++i) {
        my_vals[i] = __int_as_float(0xff800000);  // -inf
        my_idxs[i] = 0;
    }

    for (int idx = tid; idx < n; idx += BLOCK) {
        float v = x_row[idx];
        if (v > my_vals[k - 1]) {
            int ins = k - 1;
            while (ins > 0 && v > my_vals[ins - 1]) {
                my_vals[ins] = my_vals[ins - 1];
                my_idxs[ins] = my_idxs[ins - 1];
                --ins;
            }
            my_vals[ins] = v;
            my_idxs[ins] = idx;
        }
    }

    // ---------------------------------------------------------------
    // Phase 2: write candidates to shared memory
    // ---------------------------------------------------------------
    __shared__ float s_vals[MAX_CAND];
    __shared__ int   s_idxs[MAX_CAND];

    for (int i = tid; i < MAX_CAND; i += BLOCK) {
        s_vals[i] = __int_as_float(0xff800000);
        s_idxs[i] = 0;
    }
    __syncthreads();

    int base = tid * k;
    for (int i = 0; i < k; ++i) {
        s_vals[base + i] = my_vals[i];
        s_idxs[base + i] = my_idxs[i];
    }
    __syncthreads();

    // ---------------------------------------------------------------
    // Phase 3: block-wide bitonic sort (descending)
    // ---------------------------------------------------------------
    int total = BLOCK * k;  // exact count, not necessarily power of 2
    int padded = 1;
    while (padded < total) padded <<= 1;

    for (int kk = 2; kk <= padded; kk <<= 1) {
        for (int j = kk >> 1; j > 0; j >>= 1) {
            for (int i = tid; i < padded; i += BLOCK) {
                int ixj = i ^ j;
                if (ixj > i) {
                    if ((i & kk) == 0) {
                        if (s_vals[i] < s_vals[ixj]) {
                            swapf(s_vals[i], s_vals[ixj]);
                            swapi(s_idxs[i], s_idxs[ixj]);
                        }
                    } else {
                        if (s_vals[i] > s_vals[ixj]) {
                            swapf(s_vals[i], s_vals[ixj]);
                            swapi(s_idxs[i], s_idxs[ixj]);
                        }
                    }
                }
            }
            if (j < BLOCK) __syncthreads();
        }
    }

    // ---------------------------------------------------------------
    // Phase 4: write result
    // ---------------------------------------------------------------
    if (tid < k) {
        out_vals[row * k + tid] = s_vals[tid];
        out_idxs[row * k + tid] = (int64_t)s_idxs[tid];
    }
}

void launch_topk(
    at::Tensor x,
    at::Tensor out_vals,
    at::Tensor out_idxs,
    int batch, int n, int k
) {
    dim3 grid(batch);
    dim3 block(BLOCK);

    topk_global<<<grid, block>>>(
        x.data_ptr<float>(),
        out_vals.data_ptr<float>(),
        out_idxs.data_ptr<int64_t>(),
        batch, n, k
    );
}
"""

class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch = batch
        self.n = n
        self.k = k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x):

        _kernel.launch_topk(x, out_vals, out_idxs, batch, n, k)

        return out_vals, out_idxs
