"""Top-k via bitonic sorting network in shared memory — optimized.

Key design:
- k=1 fast path: parallel warp-reduce (no bitonic sort).
- All other cases: one block per row, processing the row in tiles.
  Each tile is bitonic-sorted in shared memory, then its top-k is
  bitonic-merged with the running top-k.
- Non-power-of-2 n is handled by padding tiles with -inf.
- Single kernel launch per forward call.
"""
from __future__ import annotations

import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_cpp_src = r"""
#include <torch/extension.h>
#include <vector>
#include <cstdint>
std::vector<torch::Tensor> topk_forward(const torch::Tensor& input, int64_t k);
"""

_cuda_src = r"""
#include <cuda_runtime.h>
#include <float.h>
#include <stdint.h>

#define MAX_K 64

// ============================================================
// Bitonic sort in shared memory (ascending order)
// ============================================================
template <int N>
__device__ void bitonic_sort(float* smv, float* smi) {
    int tid = threadIdx.x, nt = blockDim.x, np = N / 2;
    for (int stage = 2; stage <= N; stage <<= 1) {
        for (int step = stage >> 1; step > 0; step >>= 1) {
            for (int p = tid; p < np; p += nt) {
                int a = (p / step) * (step << 1) + (p % step);
                int b = a + step;
                bool asc = ((a / stage) % 2) == 0;
                float va = smv[a], vb = smv[b], ia = smi[a], ib = smi[b];
                if ((asc && va > vb) || (!asc && va < vb)) {
                    smv[a] = vb; smv[b] = va; smi[a] = ib; smi[b] = ia;
                }
            }
            __syncthreads();
        }
    }
}

// ============================================================
// k=1 fast path: parallel reduction (max + argmax)
// ============================================================
__global__ void topk_argmax(
    const float* __restrict__ input,
    float* __restrict__ out_vals, int64_t* __restrict__ out_idx,
    int n) {
    int row = blockIdx.x, tid = threadIdx.x;
    const float* r = input + (int64_t)row * n;

    float bv = -FLT_MAX; int bi = -1;
    for (int i = tid; i < n; i += blockDim.x) {
        float v = r[i]; if (v > bv) { bv = v; bi = i; }
    }
    // warp reduce
    for (int o = 16; o > 0; o >>= 1) {
        float ov = __shfl_down_sync(0xffffffff, bv, o);
        int   oi = __shfl_down_sync(0xffffffff, bi, o);
        if (ov > bv || (ov == bv && oi < bi)) { bv = ov; bi = oi; }
    }
    __shared__ float sv[32]; __shared__ int si[32];
    int wid = tid >> 5, lid = tid & 31;
    if (lid == 0) { sv[wid] = bv; si[wid] = bi; }
    __syncthreads();
    if (wid == 0) {
        float wv = sv[lid]; int wi = si[lid];
        for (int o = 16; o > 0; o >>= 1) {
            float ov = __shfl_down_sync(0xffffffff, wv, o);
            int   oi = __shfl_down_sync(0xffffffff, wi, o);
            if (ov > wv || (ov == wv && oi < wi)) { wv = ov; wi = oi; }
        }
        if (lid == 0) { out_vals[row] = wv; out_idx[row] = wi; }
    }
}

// ============================================================
// Tiled top-k kernel: one block per row, processes in tiles
// ============================================================
template <int TILE, int K>
__global__ void topk_tiled(
    const float* __restrict__ input,
    float* __restrict__ out_vals, int64_t* __restrict__ out_idx,
    int n) {
    int row = blockIdx.x, tid = threadIdx.x, nt = blockDim.x;
    const float* row_in = input + (int64_t)row * n;

    // Shared memory layout (TILE = next_pow2 of n, up to 4096):
    // [0, TILE)       : tile_val
    // [TILE, 2*TILE)  : tile_idx
    //    (or if TILE < 2*MAX_K we need more space for merge)
    // After tile is processed:
    // [0, 2*K) : merge_buf_val
    // [2*K, 4*K) : merge_buf_idx
    //
    // We use the same region: tile occupies [0, 2*TILE), merge occupies [0, 4*K)
    // Since K <= 64 and TILE >= 32, 4*K <= 256 and 2*TILE >= 64, they can share.
    // But for TILE=32 and K=64, 2*TILE=64 < 4*K=256 — need separate regions.
    // Solution: always allocate 2*TILE space (for tile) and put merge at offset 2*TILE.
    //
    // For K<=64 and TILE<=4096:
    //   tile region: [0, 2*TILE)   (up to 8K floats = 32KB for TILE=4096)
    //   running region: [2*TILE, 2*TILE+2*K)  (values + indices)
    //   merge region: [2*TILE+2*K, 2*TILE+6*K) (2K values + 2K indices)

    extern __shared__ float smem[];
    float* tile_v = smem;
    float* tile_i = smem + TILE;
    float* run_v  = smem + 2 * TILE;
    float* run_i  = smem + 2 * TILE + K;
    float* mrg_v  = smem + 2 * TILE + 2 * K;
    float* mrg_i  = smem + 2 * TILE + 4 * K;

    bool first_tile = true;

    for (int t_start = 0; t_start < n; t_start += TILE) {
        int t_end = t_start + TILE;
        // Load tile
        for (int i = tid; i < TILE; i += nt) {
            int gi = t_start + i;
            if (gi < n) { tile_v[i] = row_in[gi]; tile_i[i] = (float)gi; }
            else        { tile_v[i] = -FLT_MAX;     tile_i[i] = -1.0f; }
        }
        __syncthreads();

        // Sort tile ascending
        bitonic_sort<TILE>(tile_v, tile_i);

        if (first_tile) {
            // Copy top K (descending) into running
            for (int i = tid; i < K; i += nt) {
                run_v[i] = tile_v[TILE - 1 - i];
                run_i[i] = tile_i[TILE - 1 - i];
            }
            first_tile = false;
        } else {
            // Merge running (desc) with tile top-K (desc) using bitonic merge
            // Build bitonic sequence: running (desc) + reversed(tile_topK)
            // running is descending; reversed tile_topK → tile_topK[K-1..0] is ascending
            // The concatenation [descending, ascending] is bitonic!
            for (int i = tid; i < K; i += nt) {
                mrg_v[i] = run_v[i];                  // running, descending
                mrg_i[i] = run_i[i];
                mrg_v[2*K - 1 - i] = tile_v[TILE - 1 - i]; // reversed tile top-K
                mrg_i[2*K - 1 - i] = tile_i[TILE - 1 - i];
            }
            __syncthreads();

            // Bitonic merge of size 2K (ascending)
            // Single stage: merge the bitonic sequence into ascending order
            {
                int N2 = 2 * K, np2 = N2 / 2;
                for (int step = N2 >> 1; step > 0; step >>= 1) {
                    for (int p = tid; p < np2; p += nt) {
                        int a = (p / step) * (step << 1) + (p % step);
                        int b = a + step;
                        float va = mrg_v[a], vb = mrg_v[b], ia = mrg_i[a], ib = mrg_i[b];
                        if (va < vb) { // descending sort for merge: larger at a
                            mrg_v[a] = vb; mrg_v[b] = va;
                            mrg_i[a] = ib; mrg_i[b] = ia;
                        }
                    }
                    __syncthreads();
                }
            }
            // Top K are now at mrg_v[0..K-1] (descending)
            for (int i = tid; i < K; i += nt) {
                run_v[i] = mrg_v[i];
                run_i[i] = mrg_i[i];
            }
        }
        __syncthreads();
    }

    // Write output
    float* rv = out_vals + (int64_t)row * K;
    int64_t* ri = out_idx + (int64_t)row * K;
    for (int i = tid; i < K; i += nt) {
        rv[i] = run_v[i];
        ri[i] = (int64_t)run_i[i];
    }
}

// ============================================================
// Helpers
// ============================================================
static inline int next_pow2(int v) {
    if (v <= 1) return 1;
    int p = 2; while (p < v) p <<= 1;
    return p;
}

// ============================================================
// Dispatch
// ============================================================

#define DISPATCH_TILE(T)                                               \
    if (k == 1)  topk_tiled<T,1><<<grid,block,shmem,stream>>>(inp,vals,idx,n); \
    else if (k == 4)  topk_tiled<T,4><<<grid,block,shmem,stream>>>(inp,vals,idx,n); \
    else if (k == 8)  topk_tiled<T,8><<<grid,block,shmem,stream>>>(inp,vals,idx,n); \
    else if (k == 16) topk_tiled<T,16><<<grid,block,shmem,stream>>>(inp,vals,idx,n); \
    else if (k == 32) topk_tiled<T,32><<<grid,block,shmem,stream>>>(inp,vals,idx,n); \
    else if (k == 64) topk_tiled<T,64><<<grid,block,shmem,stream>>>(inp,vals,idx,n)

#define DISPATCH_ALL_TILES \
    if (tile == 32) { DISPATCH_TILE(32); }       \
    else if (tile == 64) { DISPATCH_TILE(64); }    \
    else if (tile == 128) { DISPATCH_TILE(128); }  \
    else if (tile == 256) { DISPATCH_TILE(256); }  \
    else if (tile == 512) { DISPATCH_TILE(512); }  \
    else if (tile == 1024) { DISPATCH_TILE(1024); } \
    else if (tile == 2048) { DISPATCH_TILE(2048); } \
    else if (tile == 4096) { DISPATCH_TILE(4096); }

static void launch_tiled(const float* inp, float* vals, int64_t* idx,
                         int batch, int n, int k, int tile, cudaStream_t stream) {
    int block_dim = (tile <= 512) ? (tile / 2) : 256;
    if (block_dim < 32) block_dim = 32;
    // Shared memory: 2*TILE (tile vals+idx) + 2*K (running) + 4*K (merge)
    int shmem = (2 * tile + 0 + 6 * MAX_K) * (int)sizeof(float);
    dim3 grid(batch), block(block_dim);

    DISPATCH_ALL_TILES
}
#undef DISPATCH_ALL_TILES
#undef DISPATCH_TILE

// ============================================================
// Entry point
// ============================================================
std::vector<torch::Tensor> topk_forward(const torch::Tensor& input, int64_t k) {
    int batch = (int)input.size(0);
    int n = (int)input.size(1);

    auto opts_f32 = torch::TensorOptions().dtype(torch::kFloat32).device(input.device());
    auto opts_i64 = torch::TensorOptions().dtype(torch::kInt64).device(input.device());
    auto out_vals = torch::empty({batch, k}, opts_f32);
    auto out_idx  = torch::empty({batch, k}, opts_i64);
    auto x = input.contiguous();
    cudaStream_t stream = 0;

    if (k == 1) {
        int nt = 512;
        topk_argmax<<<dim3(batch), dim3(nt), 0, stream>>>(
            x.data_ptr<float>(), out_vals.data_ptr<float>(),
            out_idx.data_ptr<int64_t>(), n);
        return {out_vals, out_idx};
    }

    // Tile size = min(next_pow2(n), 4096).  One block per row.
    int n_padded = next_pow2(n);
    int tile = n_padded < 4096 ? n_padded : 4096;

    launch_tiled(x.data_ptr<float>(), out_vals.data_ptr<float>(),
                 out_idx.data_ptr<int64_t>(), batch, n, (int)k, tile, stream);

    return {out_vals, out_idx};
}
"""

_ext = load_inline(
    name="topk_bitonic_ext",
    cpp_sources=_cpp_src,
    cuda_sources=_cuda_src,
    functions=["topk_forward"],
    verbose=False,
    extra_cuda_cflags=["-O3", "-use_fast_math", "--expt-relaxed-constexpr"],
)

# ---- Python model ----
OP_TYPE = "topk"
SUPPORTED_PRECISIONS = ["fp32"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor):
        vals, idx = _ext.topk_forward(x.contiguous(), self.k)
        return vals, idx


batch = 64
n = 8192
k = 8


def get_inputs():
    x = torch.randn(batch, n, dtype=torch.float32)
    return [x]


def get_init_inputs():
    return [batch, n, k]
