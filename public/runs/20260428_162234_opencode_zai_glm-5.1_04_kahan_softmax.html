import torch
import torch.nn as nn
import triton
import triton.language as tl


def _next_power_of_2(n):
    return 1 << (n - 1).bit_length()


@triton.jit
def _softmax_single_block_kernel(
    X_ptr,
    Y_ptr,
    stride_xb,
    stride_xv,
    stride_yb,
    stride_yv,
    VOCAB,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    offs = tl.arange(0, BLOCK_SIZE)
    mask = offs < VOCAB

    x = tl.load(X_ptr + row * stride_xb + offs * stride_xv, mask=mask, other=float('-inf'), cache_modifier='.ca')
    m = tl.max(tl.where(mask, x, float('-inf')), axis=0)
    e = tl.exp(x - m)
    d = tl.sum(tl.where(mask, e, 0.0), axis=0)
    y = e / d
    tl.store(Y_ptr + row * stride_yb + offs * stride_yv, y, mask=mask, cache_modifier='.cs')


@triton.jit
def _softmax_partial_kernel(
    X_ptr,
    PARTIAL_MAX_ptr,
    PARTIAL_SUM_ptr,
    stride_xb,
    stride_xv,
    stride_pmb,
    stride_pmcb,
    stride_psb,
    stride_pscb,
    VOCAB,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    col_block = tl.program_id(1)

    x_base = X_ptr + row * stride_xb
    start = col_block * BLOCK_SIZE
    offs = start + tl.arange(0, BLOCK_SIZE)
    mask = offs < VOCAB

    x = tl.load(x_base + offs * stride_xv, mask=mask, other=float('-inf'), cache_modifier='.ca')

    local_max = tl.max(tl.where(mask, x, float('-inf')), axis=0)
    e = tl.exp(x - local_max)
    local_sum = tl.sum(tl.where(mask, e, 0.0), axis=0)

    tl.store(PARTIAL_MAX_ptr + row * stride_pmb + col_block * stride_pmcb, local_max)
    tl.store(PARTIAL_SUM_ptr + row * stride_psb + col_block * stride_pscb, local_sum)


@triton.jit
def _softmax_normalize_kernel(
    X_ptr,
    Y_ptr,
    PARTIAL_MAX_ptr,
    PARTIAL_SUM_ptr,
    stride_xb,
    stride_xv,
    stride_yb,
    stride_yv,
    stride_pmb,
    stride_pmcb,
    stride_psb,
    stride_pscb,
    VOCAB,
    NUM_BLOCKS,
    BLOCK_SIZE: tl.constexpr,
):
    row = tl.program_id(0)
    col_block = tl.program_id(1)

    row_max = float('-inf')
    for i in tl.range(0, NUM_BLOCKS):
        pm = tl.load(PARTIAL_MAX_ptr + row * stride_pmb + i * stride_pmcb)
        row_max = tl.maximum(row_max, pm)

    row_sum = 0.0
    comp = 0.0
    for i in tl.range(0, NUM_BLOCKS):
        pm = tl.load(PARTIAL_MAX_ptr + row * stride_pmb + i * stride_pmcb)
        ps = tl.load(PARTIAL_SUM_ptr + row * stride_psb + i * stride_pscb)
        adjusted = ps * tl.exp(pm - row_max)
        y = adjusted - comp
        t = row_sum + y
        comp = (t - row_sum) - y
        row_sum = t

    inv_sum = 1.0 / row_sum
    x_base = X_ptr + row * stride_xb
    y_base = Y_ptr + row * stride_yb
    start = col_block * BLOCK_SIZE
    offs = start + tl.arange(0, BLOCK_SIZE)
    mask = offs < VOCAB
    x = tl.load(x_base + offs * stride_xv, mask=mask, other=float('-inf'), cache_modifier='.ca')
    e = tl.exp(x - row_max)
    tl.store(y_base + offs * stride_yv, e * inv_sum, mask=mask, cache_modifier='.cs')

        x = x.contiguous()
        batch, vocab = x.shape
        y = torch.empty_like(x)

            BLOCK_SIZE = _next_power_of_2(vocab)
            num_warps = 8
            if BLOCK_SIZE >= 4096:
                num_warps = 16
            if BLOCK_SIZE >= 8192:
                num_warps = 32
            grid = (batch,)
            _softmax_single_block_kernel[grid](
                x, y,
                x.stride(0), x.stride(1),
                y.stride(0), y.stride(1),
                vocab,
                BLOCK_SIZE=BLOCK_SIZE,
                num_warps=num_warps,
            )
        else:
            BLOCK_SIZE = 2048
            num_blocks = (vocab + BLOCK_SIZE - 1) // BLOCK_SIZE

            partial_sum = torch.empty(batch, num_blocks, dtype=torch.float32, device=x.device)

                vocab,
                BLOCK_SIZE=BLOCK_SIZE,
                num_warps=num_warps,

                vocab,
                num_blocks,
                BLOCK_SIZE=BLOCK_SIZE,
                num_warps=num_warps,
            )

        return y
