"""Custom top-k kernel (CUDA C++ via load_inline) for SM120 Blackwell.

Returns (values fp32 [batch,k] sorted desc, indices int64 [batch,k]),
matching the largest/sorted selection contract over the last dim.
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline

_CUDA_SRC = r"""
#include <torch/extension.h>
#include <cuda.h>
#include <cuda_runtime.h>
#include <c10/cuda/CUDAStream.h>
#include <math_constants.h>

// One block per row. Each thread keeps a sorted (desc) top-k buffer of its
// strided slice in shared memory; then the whole candidate pool (T*k) is
// bitonic-sorted descending and the first k are written out.
__global__ void topk_block_kernel(
        const float* __restrict__ x,
        float* __restrict__ out_val,
        long*  __restrict__ out_idx,
        int n, int k) {
    extern __shared__ char smem[];
    float* sval = (float*)smem;
    int*   sidx = (int*)(sval + blockDim.x * k);

    const int row = blockIdx.x;
    const int t = threadIdx.x;
    const int T = blockDim.x;
    const int C = T * k;
    const float* xr = x + (long)row * n;

    // init this thread's k-block to -inf
    float* blk = sval + t * k;
    int*   iblk = sidx + t * k;
    for (int i = 0; i < k; ++i) { blk[i] = -CUDART_INF_F; iblk[i] = -1; }

    // scan strided slice, maintain sorted-desc top-k
    for (int j = t; j < n; j += T) {
        float e = xr[j];
        if (e > blk[k-1]) {
            int pos = k - 1;
            while (pos > 0 && blk[pos-1] < e) {
                blk[pos] = blk[pos-1];
                iblk[pos] = iblk[pos-1];
                --pos;
            }
            blk[pos] = e;
            iblk[pos] = j;
        }
    }
    __syncthreads();

    // bitonic sort descending over C elements
    for (int kk = 2; kk <= C; kk <<= 1) {
        for (int jj = kk >> 1; jj > 0; jj >>= 1) {
            for (int i = t; i < C; i += T) {
                int ixj = i ^ jj;
                if (ixj > i) {
                    bool up = ((i & kk) == 0);
                    float vi = sval[i], vj = sval[ixj];
                    // descending: swap when (vi < vj) == up
                    if ((vi < vj) == up) {
                        sval[i] = vj; sval[ixj] = vi;
                        int ti = sidx[i]; sidx[i] = sidx[ixj]; sidx[ixj] = ti;
                    }
                }
            }
            __syncthreads();
        }
    }

    for (int i = t; i < k; i += T) {
        out_val[(long)row * k + i] = sval[i];
        out_idx[(long)row * k + i] = (long)sidx[i];
    }
}

static int next_pow2(int v){ int p=1; while(p<v) p<<=1; return p; }

std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int64_t k) {
    TORCH_CHECK(x.is_cuda() && x.dim()==2 && x.scalar_type()==torch::kFloat32);
    x = x.contiguous();
    int batch = x.size(0);
    int n = x.size(1);
    auto opts_v = torch::TensorOptions().dtype(torch::kFloat32).device(x.device());
    auto opts_i = torch::TensorOptions().dtype(torch::kInt64).device(x.device());
    auto out_val = torch::empty({batch, (long)k}, opts_v);
    auto out_idx = torch::empty({batch, (long)k}, opts_i);

    // choose T so that C=T*k is a power of two and <= 8192
    int T = 8192 / (int)k;
    if (T > 1024) T = 1024;
    if (T < 32) T = 32;
    T = next_pow2(T);
    int C = T * (int)k;
    size_t shmem = (size_t)C * (sizeof(float) + sizeof(int));

    cudaFuncSetAttribute(topk_block_kernel,
        cudaFuncAttributeMaxDynamicSharedMemorySize, (int)shmem);

    auto stream = c10::cuda::getCurrentCUDAStream();
    topk_block_kernel<<<batch, T, shmem, stream>>>(
        x.data_ptr<float>(), out_val.data_ptr<float>(),
        out_idx.data_ptr<long>(), n, (int)k);
    return {out_val, out_idx};
}
"""

+"""Custom top-k kernel (CUDA C++ via load_inline) for SM120 Blackwell.
+
+Contract matches torch.topk(x, k, dim=-1, largest=True, sorted=True):
+returns (values fp32 [batch,k] sorted desc, indices int64 [batch,k]).
+"""
+import torch
+import torch.nn as nn
+from torch.utils.cpp_extension import load_inline
+
+_CUDA_SRC = r"""
+#include <torch/extension.h>
+#include <cuda.h>
+#include <cuda_runtime.h>
+#include <c10/cuda/CUDAStream.h>
+
+// One block per row. Each thread keeps a sorted (desc) top-k buffer of its
+// strided slice in shared memory; then the whole candidate pool (T*k) is
+// bitonic-sorted descending and the first k are written out.
+__global__ void topk_block_kernel(
+        const float* __restrict__ x,
+        float* __restrict__ out_val,
+        long*  __restrict__ out_idx,
+        int n, int k) {
+    extern __shared__ char smem[];
+    float* sval = (float*)smem;
+    int*   sidx = (int*)(sval + blockDim.x * k);
+
+    const int row = blockIdx.x;
+    const int t = threadIdx.x;
+    const int T = blockDim.x;
+    const int C = T * k;
+    const float* xr = x + (long)row * n;
+
+    // init this thread's k-block to -inf
+    float* blk = sval + t * k;
+    int*   iblk = sidx + t * k;
+    for (int i = 0; i < k; ++i) { blk[i] = -CUDART_INF_F; iblk[i] = -1; }
+
+    // scan strided slice, maintain sorted-desc top-k
+    for (int j = t; j < n; j += T) {
+        float e = xr[j];
+        if (e > blk[k-1]) {
+            int pos = k - 1;
+            while (pos > 0 && blk[pos-1] < e) {
+                blk[pos] = blk[pos-1];
+                iblk[pos] = iblk[pos-1];
+                --pos;
+            }
+            blk[pos] = e;
+            iblk[pos] = j;
+        }
+    }
+    __syncthreads();
+
+    // bitonic sort descending over C elements
+    for (int kk = 2; kk <= C; kk <<= 1) {
+        for (int jj = kk >> 1; jj > 0; jj >>= 1) {
+            for (int i = t; i < C; i += T) {
+                int ixj = i ^ jj;
+                if (ixj > i) {
+                    bool up = ((i & kk) == 0);
+                    float vi = sval[i], vj = sval[ixj];
+                    // descending: swap when (vi < vj) == up
+                    if ((vi < vj) == up) {
+                        sval[i] = vj; sval[ixj] = vi;
+                        int ti = sidx[i]; sidx[i] = sidx[ixj]; sidx[ixj] = ti;
+                    }
+                }
+            }
+            __syncthreads();
+        }
+    }
+
+    for (int i = t; i < k; i += T) {
+        out_val[(long)row * k + i] = sval[i];
+        out_idx[(long)row * k + i] = (long)sidx[i];
+    }
+}
+
+static int next_pow2(int v){ int p=1; while(p<v) p<<=1; return p; }
+
+std::vector<torch::Tensor> topk_cuda(torch::Tensor x, int64_t k) {
+    TORCH_CHECK(x.is_cuda() && x.dim()==2 && x.scalar_type()==torch::kFloat32);
+    x = x.contiguous();
+    int batch = x.size(0);
+    int n = x.size(1);
+    auto opts_v = torch::TensorOptions().dtype(torch::kFloat32).device(x.device());
+    auto opts_i = torch::TensorOptions().dtype(torch::kInt64).device(x.device());
+    auto out_val = torch::empty({batch, (long)k}, opts_v);
+    auto out_idx = torch::empty({batch, (long)k}, opts_i);
+
+    // choose T so that C=T*k is a power of two and <= 8192
+    int T = 8192 / (int)k;
+    if (T > 1024) T = 1024;
+    if (T < 32) T = 32;
+    T = next_pow2(T);
+    int C = T * (int)k;
+    size_t shmem = (size_t)C * (sizeof(float) + sizeof(int));
+
+    cudaFuncSetAttribute(topk_block_kernel,
+        cudaFuncAttributeMaxDynamicSharedMemorySize, (int)shmem);
+
+    auto stream = c10::cuda::getCurrentCUDAStream();
+    topk_block_kernel<<<batch, T, shmem, stream>>>(
+        x.data_ptr<float>(), out_val.data_ptr<float>(),
+        out_idx.data_ptr<long>(), n, (int)k);
+    return {out_val, out_idx};
+}
