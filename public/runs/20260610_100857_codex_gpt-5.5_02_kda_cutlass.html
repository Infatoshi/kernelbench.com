import torch
import torch.nn as nn

torch.set_float32_matmul_precision("high")


def _precompute_chunk_tensors(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
):
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    C = 64
    N = T // C

    qf = (q.float() * scale).reshape(B, N, C, H, K).permute(0, 3, 1, 2, 4)
    kf = k.float().reshape(B, N, C, H, K).permute(0, 3, 1, 2, 4)
    vf = v.float().reshape(B, N, C, H, V).permute(0, 3, 1, 2, 4)
    gf = g.float().reshape(B, N, C, H, K).permute(0, 3, 1, 2, 4).cumsum(-2)
    bf = beta.float().reshape(B, N, C, H).permute(0, 3, 1, 2)

    eg = torch.exp(gf)
    kg0 = kf * eg
    kig = kf * torch.exp(-gf)

    kk = torch.matmul(kg0, kig.transpose(-1, -2))
    strict = torch.tril(torch.ones(C, C, device=q.device, dtype=torch.bool), diagonal=-1)
    tri = torch.where(strict, kk * bf[..., :, None], torch.zeros_like(kk))

    rhs = torch.cat((vf, kg0), dim=-1) * bf[..., :, None]
    solved = torch.linalg.solve_triangular(tri, rhs, upper=False, unitriangular=True)
    u = solved[..., :V]
    w = solved[..., V:]

    qg = qf * eg
    full = torch.tril(torch.ones(C, C, device=q.device, dtype=torch.bool), diagonal=0)
    a = torch.where(full, torch.matmul(qg, kig.transpose(-1, -2)), torch.zeros_like(kk))
    kg = kf * torch.exp(gf[..., -1:, :] - gf)
    gn = eg[..., -1, :]

    return (
        u.to(dtype).contiguous(),
        w.to(dtype).contiguous(),
        qg.to(dtype).contiguous(),
        a.to(dtype).contiguous(),
        kg.to(dtype).contiguous(),
        gn.to(dtype).contiguous(),
    )


_compiled_precompute = torch.compile(_precompute_chunk_tensors, mode="reduce-overhead")


@triton.jit
def _state_scan_kernel(
    u_ptr,
    w_ptr,
    kg_ptr,
    gn_ptr,
    h_ptr,
    z_ptr,
    T: tl.constexpr,
    H: tl.constexpr,
    BV: tl.constexpr,
):
    pid_v = tl.program_id(0)
    pid_bh = tl.program_id(1)
    hidx = pid_bh % H
    bidx = pid_bh // H
    nt = T // 64

    offs_c = tl.arange(0, 64)
    offs_k = tl.arange(0, 64)
    offs_v = pid_v * BV + tl.arange(0, BV)

    s0 = tl.zeros((64, BV), dtype=tl.float32)
    s1 = tl.zeros((64, BV), dtype=tl.float32)

    bh_base = (bidx * H + hidx) * nt
    for n in tl.range(0, nt, 1):
        chunk_base = bh_base + n

        h_base = chunk_base * 128 * 128
        h0 = h_base + offs_k[:, None] * 128 + offs_v[None, :]
        h1 = h_base + (offs_k[:, None] + 64) * 128 + offs_v[None, :]
        tl.store(h_ptr + h0, s0)
        tl.store(h_ptr + h1, s1)

        w_base = chunk_base * 64 * 128
        w0 = w_base + offs_c[:, None] * 128 + offs_k[None, :]
        w1 = w_base + offs_c[:, None] * 128 + (offs_k[None, :] + 64)
        bw0 = tl.load(w_ptr + w0)
        bw1 = tl.load(w_ptr + w1)
        pred = tl.dot(bw0, s0.to(bw0.dtype)) + tl.dot(bw1, s1.to(bw1.dtype))

        z_base = chunk_base * 64 * 128
        zv = z_base + offs_c[:, None] * 128 + offs_v[None, :]
        bu = tl.load(u_ptr + zv).to(tl.float32)
        bz = bu - pred
        tl.store(z_ptr + zv, bz)

        gn_base = chunk_base * 128
        g0 = tl.load(gn_ptr + gn_base + offs_k).to(tl.float32)
        g1 = tl.load(gn_ptr + gn_base + offs_k + 64).to(tl.float32)
        s0 *= g0[:, None]
        s1 *= g1[:, None]

        kg_base = chunk_base * 64 * 128
        kg0 = tl.load(kg_ptr + kg_base + offs_c[None, :] * 128 + offs_k[:, None])
        kg1 = tl.load(kg_ptr + kg_base + offs_c[None, :] * 128 + (offs_k[:, None] + 64))
        bz_tc = bz.to(kg0.dtype)
        s0 += tl.dot(kg0, bz_tc)
        s1 += tl.dot(kg1, bz_tc)


@triton.jit
def _output_kernel(
    qg_ptr,
    a_ptr,
    h_ptr,
    z_ptr,
    o_ptr,
    T: tl.constexpr,
    H: tl.constexpr,
    BV: tl.constexpr,
):
    pid_v = tl.program_id(0)
    n = tl.program_id(1)
    pid_bh = tl.program_id(2)
    hidx = pid_bh % H
    bidx = pid_bh // H
    nt = T // 64

    offs_c = tl.arange(0, 64)
    offs_k = tl.arange(0, 64)
    offs_v = pid_v * BV + tl.arange(0, BV)
    chunk_base = ((bidx * H + hidx) * nt + n)

    q_base = chunk_base * 64 * 128
    q0 = q_base + offs_c[:, None] * 128 + offs_k[None, :]
    q1 = q_base + offs_c[:, None] * 128 + (offs_k[None, :] + 64)
    bq0 = tl.load(qg_ptr + q0)
    bq1 = tl.load(qg_ptr + q1)

    h_base = chunk_base * 128 * 128
    h0 = h_base + offs_k[:, None] * 128 + offs_v[None, :]
    h1 = h_base + (offs_k[:, None] + 64) * 128 + offs_v[None, :]
    bh0 = tl.load(h_ptr + h0)
    bh1 = tl.load(h_ptr + h1)
    out = tl.dot(bq0, bh0) + tl.dot(bq1, bh1)

    a_base = chunk_base * 64 * 64
    aoffs = a_base + offs_c[:, None] * 64 + offs_c[None, :]
    ba = tl.load(a_ptr + aoffs)
    z_base = chunk_base * 64 * 128
    zoffs = z_base + offs_c[:, None] * 128 + offs_v[None, :]
    bz = tl.load(z_ptr + zoffs)
    out += tl.dot(ba, bz)

    o_base = ((bidx * T) * H + hidx) * 128
    ooffs = o_base + (n * 64 + offs_c[:, None]) * H * 128 + offs_v[None, :]
    tl.store(o_ptr + ooffs, out)


class Model(nn.Module):
    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)

    def forward(
        self,
        q: torch.Tensor,
        k: torch.Tensor,
        v: torch.Tensor,
        g: torch.Tensor,
        beta: torch.Tensor,
    ) -> torch.Tensor:
        u, w, qg, a, kg, gn = _compiled_precompute(q, k, v, g, beta, self.scale)
        nt = self.T // 64
        h = torch.empty((self.B, self.H, nt, 128, 128), device=q.device, dtype=v.dtype)
        z = torch.empty((self.B, self.H, nt, 64, 128), device=q.device, dtype=v.dtype)
        o = torch.empty_like(v)
        _state_scan_kernel[(triton.cdiv(self.V, 32), self.B * self.H)](
            u,
            w,
            kg,
            gn,
            h,
            z,
            self.T,
            self.H,
            BV=32,
            num_warps=4,
        )
        _output_kernel[(triton.cdiv(self.V, 64), nt, self.B * self.H)](
            qg,
            a,
            h,
            z,
            o,
            self.T,
            self.H,
            BV=64,
            num_warps=4,
        )
        return o
