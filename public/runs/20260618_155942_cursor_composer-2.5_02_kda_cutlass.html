"""Kimi Delta Attention (KDA) forward, chunk form — custom Triton kernels.

Implements the chunk-parallel KDA forward (Songlin Yang et al., FLA library)
for RTX 3090 (SM86). Two Triton kernels:

  1. `_intra_kernel`  — parallel over (B, H, N_chunks). Builds the intra-chunk
     delta-rule transition C = (I + tril(kg @ kneg.T, -1))^{-1} (row-scaled
     by beta) and produces u = C @ (beta*v), w = C @ (beta*kg), plus the decayed
     qg = q*scale*exp(g), kneg = k*exp(-g), decay = exp(g_last).

  2. `_recurrence_kernel` — sequential over chunks (grid splits V for parallelism).
     Maintains state S[K,BV] and emits o per chunk.

Math mirrors reference.py exactly. Does NOT import or call fla.ops.*.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_3090"]


@triton.jit
def _intra_kernel(
    q_ptr, k_ptr, v_ptr, g_ptr, beta_ptr,
    u_ptr, w_ptr, qg_ptr, kneg_ptr, decay_ptr,
    scale,
    N, T,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
):
    pid = tl.program_id(0)
    n = pid % N
    bh = pid // N
    h = bh % H
    b = bh // H

    sTK = H * K
    sTV = H * V
    base_k = b * (T * H * K) + h * K
    base_v = b * (T * H * V) + h * V

    offs_t = n * BT + tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = tl.arange(0, V)

    tk = offs_t[:, None] * sTK + offs_k[None, :]
    tv = offs_t[:, None] * sTV + offs_v[None, :]

    q = tl.load(q_ptr + base_k + tk).to(tl.float32)
    k = tl.load(k_ptr + base_k + tk).to(tl.float32)
    v = tl.load(v_ptr + base_v + tv).to(tl.float32)
    g = tl.load(g_ptr + base_k + tk)
    beta = tl.load(beta_ptr + b * (T * H) + h + offs_t * H).to(tl.float32)

    g = tl.cumsum(g, axis=0)
    eg = tl.exp(g)
    kg = k * eg
    kneg = k * tl.exp(-g)
    qg = q * scale * eg

    last_row = (tl.arange(0, BT)[:, None] == (BT - 1))
    decay = tl.sum(tl.where(last_row, eg, 0.0), axis=0)

    M = tl.dot(kg, tl.trans(kneg))
    ri = tl.arange(0, BT)[:, None]
    ci = tl.arange(0, BT)[None, :]
    A = -tl.where(ri > ci, beta[:, None] * M, 0.0)
    for i in tl.range(1, BT):
        r = tl.sum(tl.where(ri == i, A, 0.0), axis=0)
        upd = tl.sum(r[:, None] * A, axis=0)
        A = tl.where((ri == i) & (ci < i), (r + upd)[None, :], A)
    C = A + tl.where(ri == ci, 1.0, 0.0)

    bv = beta[:, None] * v
    u = tl.dot(C, bv)
    bkg = beta[:, None] * kg
    w = tl.dot(C, bkg)

    tl.store(u_ptr + base_v + tv, u.to(tl.bfloat16))
    tl.store(w_ptr + base_k + tk, w.to(tl.bfloat16))
    tl.store(qg_ptr + base_k + tk, qg.to(tl.bfloat16))
    tl.store(kneg_ptr + base_k + tk, kneg.to(tl.bfloat16))
    tl.store(decay_ptr + bh * (N * K) + n * K + offs_k, decay)


@triton.jit
def _recurrence_kernel(
    qg_ptr, kneg_ptr, w_ptr, u_ptr, decay_ptr, o_ptr,
    NV, N, T,
    H: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
):
    pid = tl.program_id(0)
    iv = pid % NV
    bh = pid // NV
    h = bh % H
    b = bh // H

    sTK = H * K
    sTV = H * V
    base_k = b * (T * H * K) + h * K
    base_v = b * (T * H * V) + h * V

    offs_bt = tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = iv * BV + tl.arange(0, BV)
    ri = offs_bt[:, None]
    ci = offs_bt[None, :]

    S = tl.zeros([K, BV], dtype=tl.float32)

    for n in tl.range(0, N):
        t0 = n * BT
        tk = (t0 + offs_bt)[:, None] * sTK + offs_k[None, :]
        tv = (t0 + offs_bt)[:, None] * sTV + offs_v[None, :]

        w_bf = tl.load(w_ptr + base_k + tk)
        u = tl.load(u_ptr + base_v + tv).to(tl.float32)
        v_new = u - tl.dot(w_bf, S.to(tl.bfloat16))

        qg_bf = tl.load(qg_ptr + base_k + tk)
        kneg_bf = tl.load(kneg_ptr + base_k + tk)
        Aqk = tl.dot(qg_bf, tl.trans(kneg_bf))
        Aqk = tl.where(ci <= ri, Aqk, 0.0)
        o = tl.dot(qg_bf, S.to(tl.bfloat16)) + tl.dot(Aqk.to(tl.bfloat16), v_new.to(tl.bfloat16))
        tl.store(o_ptr + base_v + tv, o.to(tl.bfloat16))

        decay = tl.load(decay_ptr + bh * (N * K) + n * K + offs_k)
        S = decay[:, None] * (S + tl.dot(tl.trans(kneg_bf), v_new.to(tl.bfloat16)))


def _pick_bv(v: int, b: int, h: int) -> int:
    # Thin batch/head count: smaller V-tiles raise grid parallelism on SM86.
    if b * h < 6 and v % 16 == 0:
        return 16
    for bv in (32, 64, 16):
        if v % bv == 0:
            return bv
    return v


def kda_forward(q, k, v, g, beta, scale: float, chunk_size: int = 64, ws: dict | None = None):
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0
    N = T // BT

    if ws is None:
        u = torch.empty(B, T, H, V, dtype=torch.bfloat16, device=q.device)
        qg = torch.empty(B, T, H, K, dtype=torch.bfloat16, device=q.device)
        w = torch.empty(B, T, H, K, dtype=torch.bfloat16, device=q.device)
        kneg = torch.empty(B, T, H, K, dtype=torch.bfloat16, device=q.device)
        decay = torch.empty(B * H, N, K, dtype=torch.float32, device=q.device)
        o = torch.empty(B, T, H, V, dtype=torch.bfloat16, device=q.device)
    else:
        u, qg, w, kneg, decay, o = ws["u"], ws["qg"], ws["w"], ws["kneg"], ws["decay"], ws["o"]

    grid_intra = (B * H * N,)
    _intra_kernel[grid_intra](
        q, k, v, g, beta,
        u, w, qg, kneg, decay,
        scale,
        N, T,
        H=H, K=K, V=V, BT=BT,
        num_warps=8,
        num_stages=3,
    )

    BV = _pick_bv(V, B, H)
    NV = V // BV
    grid_rec = (B * H * NV,)
    _recurrence_kernel[grid_rec](
        qg, kneg, w, u, decay, o,
        NV, N, T,
        H=H, K=K, V=V, BT=BT, BV=BV,
        num_warps=4,
        num_stages=3,
    )
    return o


class Model(nn.Module):
    """KDA forward (chunk form). No learned parameters."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B, self.T, self.H, self.K, self.V = B, T, H, K, V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)
        self._ws: dict | None = None

    def _get_ws(self, device):
        if self._ws is None or self._ws["o"].device != device:
            B, T, H, K, V = self.B, self.T, self.H, self.K, self.V
            N = T // self.chunk_size
            self._ws = {
                "u": torch.empty(B, T, H, V, dtype=torch.bfloat16, device=device),
                "qg": torch.empty(B, T, H, K, dtype=torch.bfloat16, device=device),
                "w": torch.empty(B, T, H, K, dtype=torch.bfloat16, device=device),
                "kneg": torch.empty(B, T, H, K, dtype=torch.bfloat16, device=device),
                "decay": torch.empty(B * H, N, K, dtype=torch.float32, device=device),
                "o": torch.empty(B, T, H, V, dtype=torch.bfloat16, device=device),
            }
        return self._ws

    def forward(self, q, k, v, g, beta):
        ws = self._get_ws(q.device)
        return kda_forward(q, k, v, g, beta, scale=self.scale, chunk_size=self.chunk_size, ws=ws)
