import math
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# --- Shape knobs (overridden by check.py / benchmark.py from shapes.py) ----
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def paged_attn_decode_gqa_kernel(
    Q_ptr,             # (batch, num_heads, head_dim)
    KV_ptr,            # (num_blocks, page_size, num_kv_heads, head_dim * 2)
    BlockTable_ptr,    # (batch, max_blocks)
    SeqLens_ptr,       # (batch,)
    Out_ptr,           # (batch, num_heads, head_dim)
    stride_q_b, stride_q_h, stride_q_d,
    stride_kv_block, stride_kv_page, stride_kv_head, stride_kv_dim,
    stride_bt_b, stride_bt_idx,
    stride_out_b, stride_out_h, stride_out_d,
    scale,
    num_heads, num_kv_heads, head_dim, max_blocks, page_size,
    GROUP_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    pid = tl.program_id(0)
    b = pid // num_kv_heads
    h_kv = pid % num_kv_heads

    # Load sequence length
    L = tl.load(SeqLens_ptr + b)
    if L <= 0:
        return

    # Load all queries in the group
    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, BLOCK_D)
    
    q_ptrs = Q_ptr + b * stride_q_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_q_h + offs_d[None, :] * stride_q_d
    q_mask = (offs_g[:, None] < GROUP_SIZE) & (offs_d[None, :] < head_dim)
    q = tl.load(q_ptrs, mask=q_mask, other=0.0)  # (GROUP_SIZE, BLOCK_D)

    # Initialize online softmax variables
    m_i = tl.full((GROUP_SIZE,), -1e20, dtype=tl.float32)
    l_i = tl.zeros((GROUP_SIZE,), dtype=tl.float32)
    acc = tl.zeros((GROUP_SIZE, BLOCK_D), dtype=tl.float32)

    # Loop over the sequence in blocks of BLOCK_N
    num_blocks_seq = tl.cdiv(L, BLOCK_N)
    for start_block in range(0, num_blocks_seq):
        start_t = start_block * BLOCK_N
        offs_t = tl.arange(0, BLOCK_N)
        t = start_t + offs_t
        t_mask = t < L

        # Find block table entries using shift and mask
        page_idx = t >> 4
        page_offset = t & 15

        # Load block IDs from block table
        bt_ptrs = BlockTable_ptr + b * stride_bt_b + page_idx * stride_bt_idx
        bt_mask = t_mask & (page_idx < max_blocks)
        block_ids = tl.load(bt_ptrs, mask=bt_mask, other=0)

        # Compute 2D pointers for K and V
        block_ids_2d = tl.expand_dims(block_ids, 1)
        page_offset_2d = tl.expand_dims(page_offset, 1)
        offs_d_2d = tl.expand_dims(offs_d, 0)

        # Base pointer for each token
        base_ptrs = KV_ptr + block_ids_2d * stride_kv_block + page_offset_2d * stride_kv_page + h_kv * stride_kv_head

        # K pointers: shape (BLOCK_N, BLOCK_D)
        k_ptrs = base_ptrs + offs_d_2d * stride_kv_dim
        k_mask = tl.expand_dims(t_mask, 1) & tl.expand_dims(offs_d < head_dim, 0)
        k = tl.load(k_ptrs, mask=k_mask, other=0.0)

        # Compute dot product: (GROUP_SIZE, BLOCK_N)
        # q is (GROUP_SIZE, BLOCK_D), k is (BLOCK_N, BLOCK_D)
        qk = tl.sum(q[:, None, :] * k[None, :, :], axis=2) * scale
        # Apply mask
        qk = tl.where(t_mask[None, :], qk, -1e4)

        # Softmax update
        m_block = tl.max(qk, axis=1)  # (GROUP_SIZE,)
        m_next = tl.maximum(m_i, m_block)
        
        alpha = tl.exp(m_i - m_next)
        p = tl.exp(qk - m_next[:, None])
        l_block = tl.sum(tl.where(t_mask[None, :], p, 0.0), axis=1)
        l_next = l_i * alpha + l_block

        # Accumulator update
        acc = acc * alpha[:, None]

        # Load V: (BLOCK_N, BLOCK_D)
        v_ptrs = base_ptrs + (head_dim + offs_d_2d) * stride_kv_dim
        v = tl.load(v_ptrs, mask=k_mask, other=0.0)

        # Update acc with V
        p_mask = tl.where(t_mask[None, :], p, 0.0)
        acc += tl.sum(p_mask[:, :, None] * v[None, :, :], axis=1)

        m_i = m_next
        l_i = l_next

    # Final normalization
    acc = acc / l_i[:, None]

    # Write output
    out_ptrs = Out_ptr + b * stride_out_b + (h_kv * GROUP_SIZE + offs_g[:, None]) * stride_out_h + offs_d[None, :] * stride_out_d
    out_mask = (offs_g[:, None] < GROUP_SIZE) & (offs_d[None, :] < head_dim)
    tl.store(out_ptrs, acc, mask=out_mask)


class Model(nn.Module):
    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        out = torch.empty_like(query)

        # Meta-parameters
        BLOCK_N = 32
        BLOCK_D = triton.next_power_of_2(D)

        grid = (B * self.num_kv_heads,)
        paged_attn_decode_gqa_kernel[grid](
            query, kv_cache, block_table, seq_lens, out,
            query.stride(0), query.stride(1), query.stride(2),
            kv_cache.stride(0), kv_cache.stride(1), kv_cache.stride(2), kv_cache.stride(3),
            block_table.stride(0), block_table.stride(1),
            out.stride(0), out.stride(1), out.stride(2),
            self.scale,
            H, self.num_kv_heads, D, block_table.shape[1], self.page_size,
            GROUP_SIZE=self.group_size,
            BLOCK_N=BLOCK_N,
            BLOCK_D=BLOCK_D,
        )
        return out


def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
