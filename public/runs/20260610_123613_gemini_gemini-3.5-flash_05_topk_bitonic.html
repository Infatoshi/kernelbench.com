import triton.language as tl

def pack_sort_kernel(X_ptr, Y_values_ptr, Y_indices_ptr, N, BLOCK_SIZE: tl.constexpr):
    pid = tl.program_id(0)
    cols = tl.arange(0, BLOCK_SIZE)
    mask = cols < N
    
    # Load float values
    vals = tl.load(X_ptr + pid * N + cols, mask=mask, other=-float('inf'))
    
    # Interpret float32 as int32/uint32
    u = vals.to(tl.int32).to(tl.uint32, bitcast=True)
    
    # Convert float32 to sortable uint32
    # s = tl.where((u & 0x80000000) != 0, ~u, u ^ 0x80000000)
    # But wait, in Triton, bitwise NOT on uint32 is ~u. Let's see if ~u works.
    cond = (u & 0x80000000) != 0
    s = tl.where(cond, ~u, u ^ 0x80000000)
    
    # Pack sortable uint32 and index (cols) into uint64
    packed = (s.to(tl.uint64) << 32) | cols.to(tl.uint64)
    
    # Sort
    sorted_packed = tl.sort(packed, dim=0, descending=True)
    
    # Unpack
    sorted_s = (sorted_packed >> 32).to(tl.uint32)
    sorted_indices = (sorted_packed & 0xFFFFFFFF).to(tl.int64)

    cond_back = (sorted_s & 0x80000000) == 0
    sorted_u = tl.where(cond_back, ~sorted_s, sorted_s ^ 0x80000000)
    sorted_vals = sorted_u.to(tl.int32, bitcast=True).to(tl.float32, bitcast=True)
    
    # Store
    tl.store(Y_values_ptr + pid * BLOCK_SIZE + cols, sorted_vals, mask=mask)
    tl.store(Y_indices_ptr + pid * BLOCK_SIZE + cols, sorted_indices, mask=mask)

    # Scratch space for this block: Scratch_ptr + pid * BLOCK_SIZE_2K

    # -inf as uint32 is 0xff800000
    # Convert to sortable uint32:
    # u = 0xff800000
    # cond = (u & 0x80000000) != 0 -> True
    # s = ~u = 0x007fffff
    init_val = (tl.values_val(0x007fffff, cols_2k).to(tl.uint64) << 32) | 0xffffffff
