"""Paged attention decode – vectorized PyTorch implementation.

Strategy
========
1. Gather paged KV into a contiguous (B, S, Hkv, 2*D) tensor
   using one `index_select` call (highly optimised in PyTorch).
2. Split K / V, then broadcast for GQA.
3. Compute attention using PyTorch optimised kernels
   (matmul + softmax) – no custom Triton loop.
4. Handle variable sequence lengths with masking.

This avoids the sequential-per-position Triton kernel which
under-utilises DRAM bandwidth.
"""

import math
from typing import List

import torch
import torch.nn as nn


# ---------------------------------------------------------------------------
# Reference PyTorch implementation (correct, reasonably fast) – fallback
# ---------------------------------------------------------------------------

def _ref_pytorch(query, kv_cache, block_table, seq_lens,
                 num_kv_heads, page_size, scale):
    B, H, D = query.shape
    Hkv = num_kv_heads
    G = H // Hkv
    P = page_size
    dev = query.device
    max_seq = int(seq_lens.max())
    pages = (max_seq + P - 1) // P

    flat = block_table[:, :pages].contiguous().view(-1)
    kv = kv_cache.index_select(0, flat).reshape(
        B, pages * P, Hkv, 2 * D
    )[:, :max_seq]
    k = kv[..., :D]
    v = kv[..., D:]
    mask = (torch.arange(max_seq, device=dev) < seq_lens.unsqueeze(1)).view(B, max_seq, 1, 1)
    k = k.repeat_interleave(G, dim=2)
    v = v.repeat_interleave(G, dim=2)
    qf = query.float()
    kf = k.float()
    vf = v.float()
    scores = torch.matmul(qf, kf.transpose(-2, -1)) * scale
    scores = scores.masked_fill(~mask.squeeze(-1).squeeze(-1), float("-inf"))
    probs = torch.softmax(scores, dim=-1)
    out = torch.matmul(probs, vf)
    return out.to(query.dtype)


# ---------------------------------------------------------------------------
# Optimised forward – gathers KV then uses matrix multiplication
# ---------------------------------------------------------------------------

class Model(nn.Module):
    """Single-query paged attention decode."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, \
            "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        # No learned parameters: everything flows through get_inputs().
        # We keep an empty buffer so state_dict() round-trips trivially.
        self.register_buffer(
            "_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False
        )

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        Hkv = self.num_kv_heads
        G = self.group_size
        P = self.page_size

        # Max sequence length across the batch (all same in the test shapes,
        # but we handle the general case).
        max_seq = int(seq_lens.max().item())
        num_pages = (max_seq + P - 1) // P

        # ---- Step 1: gather paged KV into contiguous memory ----
        # Flatten the block table so we can index_select once.
        # block_table: (B, num_pages) → flat: (B * num_pages,)
        bt_flat = block_table[:, :num_pages].contiguous().view(-1)
        # (B * num_pages, P, Hkv, 2*D)
        kv = kv_cache.index_select(0, bt_flat)
        # (B, num_pages * P, Hkv, 2*D) → trim to max_seq
        kv = kv.reshape(B, num_pages * P, Hkv, 2 * D)[:, :max_seq]
        # Split K and V
        k = kv[..., :D]        # (B, max_seq, Hkv, D)
        v = kv[..., D:]        # (B, max_seq, Hkv, D)

        # ---- Step 2: Mask out padding positions  ---------------------------
        # arange: (max_seq,) → (1, max_seq, 1, 1) after unsqueeze
        arange = torch.arange(max_seq, device=query.device).unsqueeze(0)
        # (B, max_seq)
        len_mask = (arange < seq_lens.unsqueeze(1))
        # Expand to (B, max_seq, H, D) for broadcasting
        len_mask = len_mask.view(B, max_seq, 1, 1)

        # ---- Step 3: GQA broadcast  -------------------------------------------
        # (B, max_seq, Hkv, D) → (B, max_seq, H, D)
        k = k.repeat_interleave(G, dim=2)
        v = v.repeat_interleave(G, dim=2)

        # ---- Step 4: Compute attention in fp32  -----------------------
        qf = query.float()                     # (B, H, D)
        kf = k.float()                         # (B, max_seq, H, D)
        vf = v.float()                         # (B, max_seq, H, D)

        # scores: (B, H, max_seq) = Q @ K^T / sqrt(D)
        scores = torch.matmul(qf, kf.transpose(-2, -1)) * self.scale
        # Mask: set scores to -inf for positions beyond seq_len
        scores = scores.masked_fill(~len_mask.squeeze(-1).squeeze(-1), float("-inf"))
        probs = torch.softmax(scores, dim=-1)      # (B, H, max_seq)
        out = torch.matmul(probs, vf)             # (B, H, D)
        return out.to(query.dtype)


# ---------------------------------------------------------------------------
# Input builders – required interface
# ---------------------------------------------------------------------------

def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    import reference

    B = reference.BATCH
    H = reference.NUM_HEADS
    Hkv = reference.NUM_KV_HEADS
    D = reference.HEAD_DIM
    L = reference.SEQ_LEN
    P = reference.PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    # Keep the global pool larger than strictly needed and shuffle assignments
    # so the block_table actually exercises non-contiguous gather.
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    # Pad to pages_per_seq columns; for fixed-seq-len shapes this is exact.
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    """Return init arguments for the current shape."""
    import reference

    return [reference.BATCH, reference.NUM_HEADS, reference.NUM_KV_HEADS,
            reference.HEAD_DIM, reference.SEQ_LEN, reference.PAGE_SIZE]
