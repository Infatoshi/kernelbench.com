@triton.jit
def grouped_gemm_swiglu_kernel(
    hidden_ptr, hidden_ld,
    offsets_ptr,
    gate_ptr, gate_ld,
    up_ptr, up_ld,
    out_ptr, out_ld,
    H, I,
    max_n_e,
    BLOCK_M: tl.constexpr, BLOCK_N: tl.constexpr, BLOCK_K: tl.constexpr,
):
    """Grouped GEMM. Grid: (ceil(max_n_e/BLOCK_M), E, ceil(I/BLOCK_N)).

    M-tile is innermost so consecutive programs for the same (expert, N-tile)
    achieve L2 cache reuse of weight tiles.
    """
    pid_m = tl.program_id(0)
    expert_id = tl.program_id(1)
    pid_n = tl.program_id(2)

    m_start = tl.load(offsets_ptr + expert_id)
    m_end = tl.load(offsets_ptr + expert_id + 1)
    n_tokens = m_end - m_start

    m_off = pid_m * BLOCK_M
    if m_off >= n_tokens:
        return

    m_range_abs = m_start + m_off + tl.arange(0, BLOCK_M)
    mask_m = m_range_abs < m_end

    n_off = pid_n * BLOCK_N
    n_range = n_off + tl.arange(0, BLOCK_N)
    mask_n = n_range < I

    acc_gate = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)
    acc_up = tl.zeros((BLOCK_M, BLOCK_N), dtype=tl.float32)

    gate_base = gate_ptr + expert_id * H * I
    up_base = up_ptr + expert_id * H * I

    for k_off in range(0, H, BLOCK_K):
        k_range = k_off + tl.arange(0, BLOCK_K)
        mask_k = k_range < H

        x = tl.load(
            hidden_ptr + m_range_abs[:, None] * hidden_ld + k_range[None, :],
            mask=(mask_m[:, None] & mask_k[None, :]),
            other=0.0,
        )

        gate_w = tl.load(
            gate_base + k_range[:, None] * gate_ld + n_range[None, :],
            mask=(mask_k[:, None] & mask_n[None, :]),
            other=0.0,
        )

        up_w = tl.load(
            up_base + k_range[:, None] * up_ld + n_range[None, :],
            mask=(mask_k[:, None] & mask_n[None, :]),
            other=0.0,
        )

        acc_gate = tl.dot(x, gate_w, acc_gate)
        acc_up = tl.dot(x, up_w, acc_up)

    silu_gate = acc_gate * tl.sigmoid(acc_gate)
    result = (silu_gate * acc_up).to(tl.bfloat16)

    tl.store(
        out_ptr + m_range_abs[:, None] * out_ld + n_range[None, :],
        result,
        mask=(mask_m[:, None] & mask_n[None, :]),
    )

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets
