struct Pair {
    float val;
    int idx;
};

__host__ __device__ int next_pow2(int x) {
    int p = 1;
    while (p < x) p <<= 1;
    return p;
}

__device__ __forceinline__ void swap_pair(Pair& a, Pair& b) {
    Pair t = a; a = b; b = t;
}

__device__ __forceinline__ void insert_topk(Pair* top, int k, float val, int idx) {
    if (val <= top[k - 1].val) return;
    int pos = k - 1;
    while (pos > 0 && val > top[pos - 1].val) {
        top[pos] = top[pos - 1];
        pos--;
    }
    top[pos].val = val;
    top[pos].idx = idx;
}

__device__ void bitonic_sort_desc(Pair* data, int n) {
    for (int size = 2; size <= n; size <<= 1) {
        for (int dia = size >> 1; dia > 0; dia >>= 1) {
            for (int i = threadIdx.x; i < n; i += blockDim.x) {
                int opp = i ^ dia;
                if (opp > i) {
                    bool up = (i & size) == 0;
                    float vi = data[i].val;
                    float vo = data[opp].val;
                    if (up) {
                        if (vi < vo) swap_pair(data[i], data[opp]);
                    } else {
                        if (vi > vo) swap_pair(data[i], data[opp]);
                    }
                }
            }
            __syncthreads();
        }
    }
}

__device__ void merge_topk_smem(Pair* smem, int total, int k) {
    int sort_n = next_pow2(total);
    for (int i = threadIdx.x + total; i < sort_n; i += blockDim.x) {
        smem[i].val = -FLT_MAX;
        smem[i].idx = -1;
    }
    __syncthreads();
    bitonic_sort_desc(smem, sort_n);
}
