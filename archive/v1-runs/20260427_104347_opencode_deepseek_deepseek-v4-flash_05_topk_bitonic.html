"""Top-k solution: CUDA C++ bitonic sort + merge via load_inline.

Strategy:
  - For k == 1: block-wide argmax reduction (simple and fast).
  - For k > 1: tile-based bitonic sort in shared memory with a running top-k
    merge buffer. Each block processes one row; the row is processed in tiles
    of size TILE_SIZE (power of 2, up to 8192). Invalid positions beyond n
    are padded with -FLT_MAX so they don't pollute the top-k.
"""
import torch
import torch.nn as nn
from torch.utils.cpp_extension import load_inline
from pathlib import Path

# ---------------------------------------------------------------------------
# CUDA source
# ---------------------------------------------------------------------------
CUDA_SRC = r"""
#include <cuda_runtime.h>
#include <cstdint>
#include <cfloat>
#include <climits>

// ---------------------------------------------------------------------------
// Constants / helpers
// ---------------------------------------------------------------------------
#define TILE_SIZE 8192
#define BLOCK_SIZE 256
#define ELEMS_PER_THREAD (TILE_SIZE / BLOCK_SIZE)  // 32

// ---------------------------------------------------------------------------
// Argmax kernel (k == 1)
// ---------------------------------------------------------------------------
__global__ void argmax_kernel(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int n
) {
    // Shared memory for block reduction
    __shared__ float s_vals[BLOCK_SIZE];
    __shared__ int s_idxs[BLOCK_SIZE];

    int row = blockIdx.x;
    if (row >= batch) return;

    const float* row_x = x + row * n;
    int tid = threadIdx.x;

    float local_max = -FLT_MAX;
    int local_idx = -1;

    // Each thread scans its elements
    for (int i = tid; i < n; i += BLOCK_SIZE) {
        float v = row_x[i];
        if (v > local_max) {
            local_max = v;
            local_idx = i;
        }
    }

    s_vals[tid] = local_max;
    s_idxs[tid] = local_idx;
    __syncthreads();

    // Tree reduction (warp-level then block-level)
    for (int s = BLOCK_SIZE / 2; s > 0; s >>= 1) {
        if (tid < s) {
            if (s_vals[tid + s] > s_vals[tid]) {
                s_vals[tid] = s_vals[tid + s];
                s_idxs[tid] = s_idxs[tid + s];
            }
        }
        __syncthreads();
    }

    if (tid == 0) {
        out_vals[row] = s_vals[0];
        out_idxs[row] = (int64_t)s_idxs[0];
    }
}

// ---------------------------------------------------------------------------
// Top-k helper: merge a sorted pair list of length k into the running top-k
// ---------------------------------------------------------------------------
__device__ void merge_topk(
    float* running_vals, int* running_idxs, int k,
    const float* tile_vals, const int* tile_idxs
) {
    // Two sorted lists (descending): running and tile. Merge top-k into running.
    float tmp_vals[64];
    int tmp_idxs[64];
    int ri = 0, ti = 0, oi = 0;
    while (oi < k) {
        if (ti >= k || (ri < k && running_vals[ri] >= tile_vals[ti])) {
            tmp_vals[oi] = running_vals[ri];
            tmp_idxs[oi] = running_idxs[ri];
            ri++;
        } else {
            tmp_vals[oi] = tile_vals[ti];
            tmp_idxs[oi] = tile_idxs[ti];
            ti++;
        }
        oi++;
    }
    // Write back
    for (int i = 0; i < k; i++) {
        running_vals[i] = tmp_vals[i];
        running_idxs[i] = tmp_idxs[i];
    }
}

// ---------------------------------------------------------------------------
// Top-k kernel via bitonic sort in shared memory
// ---------------------------------------------------------------------------
template <int K>
__global__ void topk_kernel(
    const float* __restrict__ x,
    float* __restrict__ out_vals,
    int64_t* __restrict__ out_idxs,
    int batch, int n
) {
    // Dynamic shared memory: tile_vals[TILE_SIZE] + tile_idxs[TILE_SIZE]
    // We use the first K slots for the running top-k between tiles
    // (reuse the same memory to save space).
    extern __shared__ char smem[];
    float* tile_vals = (float*)smem;
    int* tile_idxs = (int*)(smem + TILE_SIZE * sizeof(float));

    // Running top-k occupies the same shared memory but we copy it out
    // before starting a new tile (conceptually it lives separately).
    float running_vals[64];
    int running_idxs[64];

    int row = blockIdx.x;
    if (row >= batch) return;

    // Initialize running top-k with -FLT_MAX
    #pragma unroll
    for (int i = 0; i < K; i++) {
        running_vals[i] = -FLT_MAX;
        running_idxs[i] = -1;
    }

    const float* row_x = x + row * n;
    int tid = threadIdx.x;  // 0..BLOCK_SIZE-1

    // Process tiles
    for (int tile_base = 0; tile_base < n; tile_base += TILE_SIZE) {
        int tile_end = min(tile_base + TILE_SIZE, n);
        int tile_actual = tile_end - tile_base;

        // ---- Load tile into shared memory (coalesced) ----
        #pragma unroll
        for (int i = 0; i < ELEMS_PER_THREAD; i++) {
            int pos = tile_base + i * BLOCK_SIZE + tid;
            int smem_pos = i * BLOCK_SIZE + tid;
            if (pos < n) {
                tile_vals[smem_pos] = row_x[pos];
                tile_idxs[smem_pos] = pos;
            } else {
                // Pad with -inf
                tile_vals[smem_pos] = -FLT_MAX;
                tile_idxs[smem_pos] = -1;
            }
        }
        __syncthreads();

        // ---- Bitonic sort (full TILE_SIZE in shared memory) ----
        int actual_tile_pow2 = 1;
        while (actual_tile_pow2 < tile_actual) actual_tile_pow2 <<= 1;
        // Use exactly TILE_SIZE (which is >= actual_tile_pow2) for the sort
        // since the padded -inf values don't affect top-K.

        for (int size = 2; size <= TILE_SIZE; size <<= 1) {
            for (int stride = size >> 1; stride > 0; stride >>= 1) {
                #pragma unroll
                for (int i = 0; i < ELEMS_PER_THREAD; i++) {
                    int pos = tid * ELEMS_PER_THREAD + i;
                    int partner = pos ^ stride;
                    if (partner > pos && partner < TILE_SIZE) {
                        float a = tile_vals[pos];
                        float b = tile_vals[partner];
                        bool asc = ((pos & size) == 0);
                        if (asc ? (a < b) : (a > b)) {
                            tile_vals[pos] = b;
                            tile_vals[partner] = a;
                            int tmp = tile_idxs[pos];
                            tile_idxs[pos] = tile_idxs[partner];
                            tile_idxs[partner] = tmp;
                        }
                    }
                }
                __syncthreads();
            }
        }

        // ---- Merge tile top-k into running top-k ----
        // Tile is sorted descending; top-K elements are at indices [0, K)
        if (K <= 64) {
            merge_topk(running_vals, running_idxs, K, &tile_vals[0], &tile_idxs[0]);
        }
    }

    // ---- Write final output ----
    if (tid < K) {
        out_vals[row * K + tid] = running_vals[tid];
        out_idxs[row * K + tid] = (int64_t)running_idxs[tid];
    }
}

// ---------------------------------------------------------------------------
// Host launcher
// ---------------------------------------------------------------------------
void launch_topk(
    const float* x, float* out_vals, int64_t* out_idxs,
    int batch, int n, int k
) {
    dim3 grid(batch);
    dim3 block(BLOCK_SIZE);

    if (k == 1) {
        argmax_kernel<<<grid, block>>>(x, out_vals, out_idxs, batch, n);
    } else {
        size_t smem_bytes = TILE_SIZE * (sizeof(float) + sizeof(int));
        // Choose template based on k
        switch (k) {
            case 8:   topk_kernel<8><<<grid, block, smem_bytes>>>(x, out_vals, out_idxs, batch, n); break;
            case 16:  topk_kernel<16><<<grid, block, smem_bytes>>>(x, out_vals, out_idxs, batch, n); break;
            case 32:  topk_kernel<32><<<grid, block, smem_bytes>>>(x, out_vals, out_idxs, batch, n); break;
            case 64:  topk_kernel<64><<<grid, block, smem_bytes>>>(x, out_vals, out_idxs, batch, n); break;
            default:  topk_kernel<8><<<grid, block, smem_bytes>>>(x, out_vals, out_idxs, batch, n); break;
        }
    }
    CUDA_CHECK(cudaGetLastError());
}
"""

class Model(nn.Module):
    def __init__(self, batch: int, n: int, k: int):
        super().__init__()
        self.batch, self.n, self.k = batch, n, k
        self.register_buffer("_dummy", torch.zeros(1))

    def forward(self, x: torch.Tensor) -> tuple[torch.Tensor, torch.Tensor]:
        assert x.is_cuda and x.dtype == torch.float32
        batch, n = x.shape
        k = self.k
        out_vals = torch.empty(batch, k, dtype=torch.float32, device=x.device)
        out_idxs = torch.empty(batch, k, dtype=torch.int64, device=x.device)
        _ext.launch_topk(x.data_ptr(), out_vals.data_ptr(), out_idxs.data_ptr(),
                         batch, n, k)
        return out_vals, out_idxs

+"""Top-k solution: CUDA C++ bitonic sort + merge via load_inline.
+
+Strategy:
+  - For k == 1: block-wide argmax reduction (simple and fast).
+  - For k > 1: tile-based bitonic sort in shared memory with a running top-k
+    merge buffer. Each block processes one row; the row is processed in tiles
+    of size TILE_SIZE (power of 2, up to 8192). Invalid positions beyond n
+    are padded with -FLT_MAX so they don't pollute the top-k.
+"""
+import torch
+import torch.nn as nn
+from torch.utils.cpp_extension import load_inline
+from pathlib import Path
+
+# ---------------------------------------------------------------------------
+# CUDA source
+# ---------------------------------------------------------------------------
+CUDA_SRC = r"""
+#include <cuda_runtime.h>
+#include <cstdint>
+#include <cfloat>
+#include <climits>
+
+// ---------------------------------------------------------------------------
+// Constants / helpers
+// ---------------------------------------------------------------------------
+#define TILE_SIZE 8192
+#define BLOCK_SIZE 256
+#define ELEMS_PER_THREAD (TILE_SIZE / BLOCK_SIZE)  // 32
+
+// ---------------------------------------------------------------------------
+// Argmax kernel (k == 1)
+// ---------------------------------------------------------------------------
+__global__ void argmax_kernel(
+    const float* __restrict__ x,
+    float* __restrict__ out_vals,
+    int64_t* __restrict__ out_idxs,
+    int batch, int n
+) {
+    // Shared memory for block reduction
+    __shared__ float s_vals[BLOCK_SIZE];
+    __shared__ int s_idxs[BLOCK_SIZE];
+
+    int row = blockIdx.x;
+    if (row >= batch) return;
+
+    const float* row_x = x + row * n;
+    int tid = threadIdx.x;
+
+    float local_max = -FLT_MAX;
+    int local_idx = -1;
+
+    // Each thread scans its elements
+    for (int i = tid; i < n; i += BLOCK_SIZE) {
+        float v = row_x[i];
+        if (v > local_max) {
+            local_max = v;
+            local_idx = i;
+        }
+    }
+
+    s_vals[tid] = local_max;
+    s_idxs[tid] = local_idx;
+    __syncthreads();
+
+    // Tree reduction (warp-level then block-level)
+    for (int s = BLOCK_SIZE / 2; s > 0; s >>= 1) {
+        if (tid < s) {
+            if (s_vals[tid + s] > s_vals[tid]) {
+                s_vals[tid] = s_vals[tid + s];
+                s_idxs[tid] = s_idxs[tid + s];
+            }
+        }
+        __syncthreads();
+    }
+
+    if (tid == 0) {
+        out_vals[row] = s_vals[0];
+        out_idxs[row] = (int64_t)s_idxs[0];
+    }
+}
+
+// ---------------------------------------------------------------------------
+// Top-k helper: merge a sorted pair list of length k into the running top-k
+// ---------------------------------------------------------------------------
+__device__ void merge_topk(
+    float* running_vals, int* running_idxs, int k,
+    const float* tile_vals, const int* tile_idxs
+) {
+    // Two sorted lists (descending): running and tile. Merge top-k into running.
+    float tmp_vals[64];
+    int tmp_idxs[64];
+    int ri = 0, ti = 0, oi = 0;
+    while (oi < k) {
+        if (ti >= k || (ri < k && running_vals[ri] >= tile_vals[ti])) {
+            tmp_vals[oi] = running_vals[ri];
+            tmp_idxs[oi] = running_idxs[ri];
+            ri++;
+        } else {
+            tmp_vals[oi] = tile_vals[ti];
+            tmp_idxs[oi] = tile_idxs[ti];
+            ti++;
+        }
+        oi++;
+    }
+    // Write back
+    for (int i = 0; i < k; i++) {
+        running_vals[i] = tmp_vals[i];
+        running_idxs[i] = tmp_idxs[i];
+    }
