"""Optimized Triton implementation of paged attention decode kernel.

Single-query decode: each batch element has a query of shape (num_heads, head_dim)
and attends over a KV cache of `seq_len[b]` tokens stored as fixed-size pages in
a global pool. Pages for batch element b are listed in `block_table[b]`.

The KV cache is laid out as (num_blocks, page_size, num_kv_heads, head_dim * 2)
where the last dim packs [K | V] so a single gather pulls both.
"""
import math
import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

# --- Shape knobs (matching reference.py) ----
BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def paged_attention_decode_kernel(
    Q_ptr,
    KV_ptr,
    BlockTable_ptr,
    SeqLens_ptr,
    Out_ptr,
    scale,
    stride_q_b, stride_q_h, stride_q_d,
    stride_kv_b, stride_kv_p, stride_kv_h, stride_kv_d,
    stride_bt_b, stride_bt_m,
    stride_sl_b,
    stride_out_b, stride_out_h, stride_out_d,
    num_heads,
    num_kv_heads,
    max_blocks,
    is_bf16: tl.constexpr,
    head_dim: tl.constexpr,
    page_size: tl.constexpr,
    group_size: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_D: tl.constexpr,
):
    # Map thread block to (batch, head)
    pid_b = tl.program_id(0)
    pid_h = tl.program_id(1)
    
    # Compute GQA KV head index
    h_kv = pid_h // group_size
    
    # Load sequence length for this batch element
    seq_len = tl.load(SeqLens_ptr + pid_b * stride_sl_b)
    
    if seq_len <= 0:
        # If sequence length is 0, write 0s to output and return
        offs_d = tl.arange(0, BLOCK_D)
        out_ptrs = Out_ptr + pid_b * stride_out_b + pid_h * stride_out_h + offs_d * stride_out_d
        tl.store(out_ptrs, 0.0, mask=offs_d < head_dim)
        return

    # Load query vector Q (shape [BLOCK_D], masked if head_dim < BLOCK_D)
    offs_d = tl.arange(0, BLOCK_D)
    q_ptrs = Q_ptr + pid_b * stride_q_b + pid_h * stride_q_h + offs_d * stride_q_d
    q = tl.load(q_ptrs, mask=offs_d < head_dim, other=0.0).to(tl.float32)
    
    # Initialize online softmax accumulators
    m_i = -float("inf")
    d_i = 0.0
    o_i = tl.zeros([BLOCK_D], dtype=tl.float32)
    
    # Loop over the sequence length in blocks of BLOCK_N
    num_iters = tl.cdiv(seq_len, BLOCK_N)
    
    for i in range(0, num_iters):
        start_l = i * BLOCK_N
        offs_n = start_l + tl.arange(0, BLOCK_N)
        
        # Compute block indices and offsets within the page
        block_indices = offs_n // page_size
        block_offsets = offs_n % page_size
        
        # Clamp block indices to avoid out-of-bound access in block table
        clamped_block_indices = tl.where(offs_n < seq_len, block_indices, 0)
        bt_ptrs = BlockTable_ptr + pid_b * stride_bt_b + clamped_block_indices * stride_bt_m
        physical_blocks = tl.load(bt_ptrs, mask=offs_n < seq_len, other=0)
        
        # Construct 2D pointer offsets for K and V
        p_blocks_2d = physical_blocks[:, None]
        p_offsets_2d = block_offsets[:, None]
        offs_d_2d = offs_d[None, :]
        
        # Compute K and V pointers
        # kv_cache layout: (num_blocks, page_size, num_kv_heads, head_dim * 2)
        k_offsets = (
            p_blocks_2d * stride_kv_b +
            p_offsets_2d * stride_kv_p +
            h_kv * stride_kv_h +
            offs_d_2d * stride_kv_d
        )
        
        v_offsets = k_offsets + head_dim * stride_kv_d
        
        k_ptrs = KV_ptr + k_offsets
        v_ptrs = KV_ptr + v_offsets
        
        # Load K and V blocks
        mask_2d = (offs_n[:, None] < seq_len) & (offs_d_2d < head_dim)
        k_block = tl.load(k_ptrs, mask=mask_2d, other=0.0).to(tl.float32)
        v_block = tl.load(v_ptrs, mask=mask_2d, other=0.0).to(tl.float32)
        
        # Compute QK^T for this block: (BLOCK_N,)
        q_broadcast = tl.broadcast_to(q[None, :], [BLOCK_N, BLOCK_D])
        scores = tl.sum(q_broadcast * k_block, axis=1) * scale
        
        # Mask out-of-bounds scores to -inf
        scores = tl.where(offs_n < seq_len, scores, -float("inf"))
        
        # Online softmax update
        m_block = tl.max(scores, axis=0)
        m_new = tl.maximum(m_i, m_block)
        
        # Compute scaling factors
        alpha = tl.math.exp(m_i - m_new)
        beta = tl.math.exp(scores - m_new)
        
        # Update sum of exponentials
        d_new = d_i * alpha + tl.sum(beta, axis=0)
        
        # Update running sum of values
        beta_broadcast = tl.broadcast_to(beta[:, None], [BLOCK_N, BLOCK_D])
        v_weighted = tl.sum(beta_broadcast * v_block, axis=0)
        o_i = o_i * alpha + v_weighted
        
        # Update running states
        m_i = m_new
        d_i = d_new

    # Compute final output
    out = o_i / d_i
    
    # Store output to global memory
    out_ptrs = Out_ptr + pid_b * stride_out_b + pid_h * stride_out_h + offs_d * stride_out_d
    if is_bf16:
        tl.store(out_ptrs, out.to(tl.bfloat16), mask=offs_d < head_dim)
    else:
        tl.store(out_ptrs, out.to(tl.float16), mask=offs_d < head_dim)


class Model(nn.Module):
    """Single-query paged attention decode kernel implemented in Triton."""

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,
    ):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be a multiple of num_kv_heads (GQA)"
        self.batch = batch
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.head_dim = head_dim
        self.seq_len = seq_len
        self.page_size = page_size
        self.group_size = num_heads // num_kv_heads
        self.scale = 1.0 / math.sqrt(head_dim)

        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        B, H, D = query.shape
        assert H == self.num_heads
        assert D == self.head_dim

        out = torch.empty(B, H, D, dtype=query.dtype, device=query.device)

        # Retrieve strides
        stride_q_b, stride_q_h, stride_q_d = query.stride()
        stride_kv_b, stride_kv_p, stride_kv_h, stride_kv_d = kv_cache.stride()
        stride_bt_b, stride_bt_m = block_table.stride()
        stride_sl_b = seq_lens.stride(0)
        stride_out_b, stride_out_h, stride_out_d = out.stride()

        max_blocks = block_table.shape[1]
        is_bf16 = (query.dtype == torch.bfloat16)

        # Decide BLOCK_N (32 or 64). Since minimum seq_len is 1024, 64 is highly efficient.
        BLOCK_N = 64
        # BLOCK_D must be a power of 2, set it to the smallest power of 2 >= head_dim
        BLOCK_D = 1 << (self.head_dim - 1).bit_length()

        grid = (B, H)

        paged_attention_decode_kernel[grid](
            query,
            kv_cache,
            block_table,
            seq_lens,
            out,
            self.scale,
            stride_q_b, stride_q_h, stride_q_d,
            stride_kv_b, stride_kv_p, stride_kv_h, stride_kv_d,
            stride_bt_b, stride_bt_m,
            stride_sl_b,
            stride_out_b, stride_out_h, stride_out_d,
            self.num_heads,
            self.num_kv_heads,
            max_blocks,
            is_bf16,
            head_dim=self.head_dim,
            page_size=self.page_size,
            group_size=self.group_size,
            BLOCK_N=BLOCK_N,
            BLOCK_D=BLOCK_D,
            num_warps=4,
        )

        return out


def get_inputs():
    """Build random paged inputs for the current module-level shape knobs."""
    B = BATCH
    H = NUM_HEADS
    Hkv = NUM_KV_HEADS
    D = HEAD_DIM
    L = SEQ_LEN
    P = PAGE_SIZE

    pages_per_seq = (L + P - 1) // P
    total_pages = max(B * pages_per_seq + 8, 64)

    query = torch.randn(B, H, D, dtype=torch.bfloat16) * 0.1
    kv_cache = torch.randn(total_pages, P, Hkv, 2 * D, dtype=torch.bfloat16) * 0.1

    perm = torch.randperm(total_pages)[: B * pages_per_seq].reshape(B, pages_per_seq).int()
    block_table = perm.contiguous()
    seq_lens = torch.full((B,), L, dtype=torch.int32)

    return [query, kv_cache, block_table, seq_lens]


def get_init_inputs():
    return [BATCH, NUM_HEADS, NUM_KV_HEADS, HEAD_DIM, SEQ_LEN, PAGE_SIZE]
