"""Grouped GEMM + fused SwiGLU for Sonic-MoE up-projection (Triton, SM120 Blackwell).

Implements variable-length grouped GEMM over experts using persistent strided
tile scheduler (no host sync, fully dynamic w.r.t. expert_offsets).
Fuses the two projections and SwiGLU (silu(gate) * up) in the epilogue.
Loads x once per K-tile; two W tiles (gate+up) per K-tile.
"""

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.jit
def _decode_tile(
    tile_id,
    offsets_ptr,
    E: tl.constexpr,
    BM: tl.constexpr,
    BN: tl.constexpr,
    I: tl.constexpr,
):
    """Decode flat tile_id -> (expert, m_start, m_size, n_start, valid)."""
    cur = 0
    for e in range(E):
        start = tl.load(offsets_ptr + e)
        end = tl.load(offsets_ptr + e + 1)
        m_size = end - start
        if m_size > 0:
            m_blocks = (m_size + BM - 1) // BM
            n_blocks = (I + BN - 1) // BN
            tiles_e = m_blocks * n_blocks
            nxt = cur + tiles_e
            if tile_id < nxt:
                loc = tile_id - cur
                m_block = loc // n_blocks
                n_block = loc % n_blocks
                m_start = start + m_block * BM
                n_start = n_block * BN
                return e, m_start, m_size, n_start, True
            cur = nxt
    return 0, 0, 0, 0, False


@triton.autotune(

    ],
    key=["H", "I", "E"],
)
@triton.jit
def _grouped_gemm_swiglu_persistent(
    hidden_ptr,  # (T_perm, H) bf16
    W_gate_ptr,  # (E, H, I) bf16
    W_up_ptr,    # (E, H, I) bf16
    out_ptr,     # (T_perm, I) bf16
    offsets_ptr, # (E+1,) int32
    stride_hm, stride_hh,
    stride_wge, stride_wgh, stride_wgi,
    stride_wue, stride_wuh, stride_wui,
    stride_om, stride_oi,
    T_perm: tl.constexpr,
    H: tl.constexpr,
    I: tl.constexpr,
    E: tl.constexpr,
    BM: tl.constexpr,
    BN: tl.constexpr,
    BK: tl.constexpr,
    MAX_TILES: tl.constexpr,
):
    pid = tl.program_id(0)
    num_pids = tl.num_programs(0)

    # Strided persistent scheduler: no atomics, no host sync, data-driven via offsets
    for tile_id in range(pid, MAX_TILES, num_pids):
        expert, m_start, m_size, n_start, valid = _decode_tile(
            tile_id, offsets_ptr, E, BM, BN, I
        )
        if not valid:
            break

        row_idx = m_start + tl.arange(0, BM)
        offs_n = n_start + tl.arange(0, BN)
        m_mask = (tl.arange(0, BM) < m_size) & (row_idx < T_perm)
        n_mask = offs_n < I
        offs_m = row_idx  # absolute rows for ptr math (masked accesses safe)

        acc_gate = tl.zeros((BM, BN), dtype=tl.float32)
        acc_up = tl.zeros((BM, BN), dtype=tl.float32)

        # K loop - load x once, both weights
        for k0 in range(0, H, BK):
            offs_k = k0 + tl.arange(0, BK)
            k_mask = offs_k < H

            # Load x tile (BM, BK) - masked
            x = tl.load(
                hidden_ptr
                + offs_m[:, None] * stride_hm
                + offs_k[None, :] * stride_hh,
                mask=m_mask[:, None] & k_mask[None, :],
                other=0.0,
            )

            # Load W_gate[e, k, n] (BK, BN)
            wg = tl.load(
                W_gate_ptr
                + expert * stride_wge
                + offs_k[:, None] * stride_wgh
                + offs_n[None, :] * stride_wgi,
                mask=k_mask[:, None] & n_mask[None, :],
                other=0.0,
            )

            # Load W_up[e, k, n]
            wu = tl.load(
                W_up_ptr
                + expert * stride_wue
                + offs_k[:, None] * stride_wuh
                + offs_n[None, :] * stride_wui,
                mask=k_mask[:, None] & n_mask[None, :],
                other=0.0,
            )

            acc_gate += tl.dot(x, wg)
            acc_up += tl.dot(x, wu)

        # Fused SwiGLU epilogue
        silu_gate = acc_gate * tl.sigmoid(acc_gate)
        result = silu_gate * acc_up

        # Store (BM, BN) bf16 with mask
        tl.store(
            out_ptr
            + offs_m[:, None] * stride_om
            + offs_n[None, :] * stride_oi,
            result.to(tl.bfloat16),
            mask=m_mask[:, None] & n_mask[None, :],
        )


class Model(nn.Module):
    """Up-projection of a top-K MoE FFN with fused SwiGLU via custom grouped GEMM.

    Custom Triton kernel: persistent strided tile scheduler over variable-length
    expert groups + fused dual-projection + SwiGLU in epilogue.
    """

    def __init__(self, T_total: int, H: int, I: int, E: int, K: int):  # noqa: E741
        super().__init__()
        self.T_total = T_total
        self.H = H
        self.I = I
        self.E = E
        self.K = K
        self.W_gate = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        self.W_up = nn.Parameter(torch.empty(E, H, I, dtype=torch.bfloat16))
        nn.init.normal_(self.W_gate, std=0.02)
        nn.init.normal_(self.W_up, std=0.02)

    def forward(
        self,
        hidden_states: torch.Tensor,   # (T_perm, H) bf16
        expert_offsets: torch.Tensor,  # (E+1,) int32
    ) -> torch.Tensor:
        T_perm, H = hidden_states.shape
        E = self.E
        I = self.I

        out = torch.empty(T_perm, I, dtype=torch.bfloat16, device=hidden_states.device)

        if T_perm == 0 or E == 0:
            return out

        # Conservative upper bound on total tiles (covers any routing, any tile size in autotune).
        # Uses smallest tile sizes from configs (BM=32, BN=64) so MAX_TILES is always sufficient.
        # Kernel decode will early exit once real tiles for chosen (BM,BN) are exhausted.
        MAX_TILES = ((T_perm + 31) // 32) * ((I + 63) // 64) + 1024

        # Fixed persistent grid - good occupancy across shapes, no syncs, CUDA-graph friendly
        NUM_CTAS = 2048

        grid = (NUM_CTAS,)

        _grouped_gemm_swiglu_persistent[grid](
            hidden_states,
            self.W_gate,
            self.W_up,
            out,
            expert_offsets,
            hidden_states.stride(0), hidden_states.stride(1),
            self.W_gate.stride(0), self.W_gate.stride(1), self.W_gate.stride(2),
            self.W_up.stride(0), self.W_up.stride(1), self.W_up.stride(2),
            out.stride(0), out.stride(1),
            T_perm=T_perm,
            H=H,
            I=I,
            E=E,
            MAX_TILES=MAX_TILES,
            # BM/BN/BK + warps/stages provided by autotune
        )
        return out

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    """Round-robin-ish routing metadata: balanced offsets summing to T_total*K."""
    T_perm = T_total * K
    base = T_perm // E
    rem = T_perm - base * E
    counts = torch.full((E,), base, dtype=torch.int32, device=device)
    counts[:rem] += 1
    offsets = torch.zeros(E + 1, dtype=torch.int32, device=device)
    offsets[1:] = torch.cumsum(counts, dim=0)
    return offsets


def get_inputs():
    T_perm = T_total * K
    hidden_states = torch.randn(T_perm, H, dtype=torch.bfloat16) * 0.1
    expert_offsets = _build_routing(T_total, E, K)
    return [hidden_states, expert_offsets]
