template <int L>
__device__ __forceinline__ void insert_pair(float v, int idx, float (&vals)[L], int (&idxs)[L]) {
    if (!better_pair(v, idx, vals[L - 1], idxs[L - 1])) {

#pragma unroll

__device__ __forceinline__ void block_best(

__global__ void argmax_kernel(

__device__ __forceinline__ void merge_row(

    const int* __restrict__ tmp_i,
    float* __restrict__ out_v,

template <int K, int THREADS, int ITEMS>
__global__ void topk_kernel(
    const float* __restrict__ x,
    float* __restrict__ tmp_v,
    int* __restrict__ tmp_i,
    int* __restrict__ counters,
    float* __restrict__ out_v,
    int64_t* __restrict__ out_i,
    int n,
    int chunks) {

    int row = block / chunks;
    int chunk = block - row * chunks;
    int tid = threadIdx.x;
    int chunk_start = chunk * THREADS * ITEMS;

    float vals[ITEMS];
    int idxs[ITEMS];
    init_local<ITEMS>(vals, idxs);

    for (int i = 0; i < ITEMS; ++i) {

        if (col < n) {
            insert_pair<ITEMS>(x[row * n + col], col, vals, idxs);
        }

    __shared__ int shared_i[32];
    __shared__ int shared_pos[32];

            tmp_i[tmp_base + j] = best_i;

            tmp_v, tmp_i, out_v, out_i, row, chunks, tid, shared_v, shared_i, shared_pos);

            counters[row] = 0;
        }
    }
}
