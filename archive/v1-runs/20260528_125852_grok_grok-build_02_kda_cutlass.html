"""Custom Triton implementation of Kimi Delta Attention (KDA) chunk forward.

Implements the exact chunk-parallel formulation from the reference (no FLA calls).

- Intra-chunk kernel: per-(chunk, head) program. Computes cumsum g (via exp2), 
  numerically-stable centered decay, lower-triangular Aqk, the (I + L + L^2 + ...) 
  factor via 5-step nilpotent squaring (exact for BT=64), then w/u/kg/qg projections.
- Inter-chunk kernel: per-(V-tile, head) program. Serial walk over NT chunks with 
  the KV state S (128xV fp32) kept entirely in registers (K-split 64+64). Fuses 
  v_new = u - w @ S, o = qg @ S + Aqk @ v_new, and the rank-K state update.
- Whole forward wrapped in CUDA graph (when safe) to eliminate launch overhead on 
  the latency-bound recurrence. Correctness path always uses eager (fresh ptrs).

Specialized for the benchmark shapes: BT=64, K=V=128, bf16 q/k/v/beta, fp32 g.
"""

from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


OP_TYPE = "linear_attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]


_RCP_LN2 = 1.4426950408889634  # 1/ln(2) so exp(x) == exp2(x * _RCP_LN2)


@triton.autotune(
    configs=[
        triton.Config({}, num_warps=4, num_stages=2),
        triton.Config({}, num_warps=8, num_stages=2),
    ],
    key=["BT"],
)
@triton.jit
def _kda_intra_kernel(
    q_ptr,
    k_ptr,
    v_ptr,
    g_ptr,
    beta_ptr,
    Aqk_ptr,
    w_ptr,
    u_ptr,
    kg_ptr,
    qg_ptr,
    g_last_ptr,
    scale,
    T,
    NT,
    H,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    RCP_LN2: tl.constexpr,
):
    i_t = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    offs_t = tl.arange(0, BT)
    offs_k = tl.arange(0, K)
    offs_v = tl.arange(0, V)

    # Base pointers for this chunk/head. Layout: (B, T, H, D) contiguous.
    base_qk = (i_b * T + i_t * BT) * H * K + i_h * K
    base_v = (i_b * T + i_t * BT) * H * V + i_h * V
    base_a = (i_b * T + i_t * BT) * H * BT + i_h * BT
    base_b = (i_b * T + i_t * BT) * H + i_h

    p_q = q_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    p_k = k_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    p_v = v_ptr + base_v + offs_t[:, None] * H * V + offs_v[None, :]
    p_g = g_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    p_beta = beta_ptr + base_b + offs_t * H

    b_q = tl.load(p_q)
    b_k = tl.load(p_k)
    b_v = tl.load(p_v)
    b_g_raw = tl.load(p_g)
    b_beta = tl.load(p_beta).to(tl.float32)

    # In-chunk cumsum in log2 space (g already pre-scaled by caller expectation).
    b_g = tl.cumsum(b_g_raw * RCP_LN2, axis=0)

    # Midpoint centering for numerical stability on long sequences.
    mid = BT // 2
    b_gn = tl.sum(tl.where(offs_t[:, None] == mid, b_g, 0.0), axis=0)
    b_gm = b_g - b_gn[None, :]

    b_exp_pos = tl.exp2(b_gm)
    b_exp_neg = tl.exp2(-b_gm)

    # qg for Aqk (lower triangular incl. diag)
    b_qg = (b_q.to(tl.float32) * b_exp_pos).to(tl.bfloat16)
    b_kg_neg = (b_k.to(tl.float32) * b_exp_neg).to(tl.bfloat16)
    b_Aqk = tl.dot(b_qg, tl.trans(b_kg_neg)) * scale
    m_lower_eq = offs_t[:, None] >= offs_t[None, :]
    b_Aqk = tl.where(m_lower_eq, b_Aqk, 0.0)

    # Akk contribution (strict lower for the nilpotent L)
    b_kg_pos = (b_k.to(tl.float32) * b_exp_pos).to(tl.bfloat16)
    b_Akk = tl.dot(b_kg_pos, tl.trans(b_kg_neg))

    m_strict_lower = offs_t[:, None] > offs_t[None, :]
    b_A = -b_Akk * b_beta[:, None]
    b_A = tl.where(m_strict_lower, b_A, 0.0)

    # (I - L)^{-1} via nilpotent geometric series (L^64 == 0 for BT=64).
    # 5 doublings cover up to L^31; one more iteration reaches full coverage.
    b_I = tl.where(offs_t[:, None] == offs_t[None, :], 1.0, 0.0)
    b_M = b_I + b_A
    b_L_pow = b_A
    for _ in tl.static_range(6):
        b_L_pow = tl.dot(b_L_pow, b_L_pow, input_precision="tf32")
        b_M = tl.dot(b_M, b_I + b_L_pow, input_precision="tf32")

    b_A_final = b_M * b_beta[None, :]

    # Write Aqk (bf16)
    p_Aqk = Aqk_ptr + base_a + offs_t[:, None] * H * BT + offs_t[None, :]
    tl.store(p_Aqk, b_Aqk.to(Aqk_ptr.dtype.element_ty))

    # w = A_final @ (exp(g) * k)
    b_kg_global = (b_k.to(tl.float32) * tl.exp2(b_g)).to(tl.bfloat16)
    b_A_final_bf = b_A_final.to(tl.bfloat16)
    b_w = tl.dot(b_A_final_bf, b_kg_global)
    p_w = w_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    tl.store(p_w, b_w.to(w_ptr.dtype.element_ty))

    # u = A_final @ v
    b_u = tl.dot(b_A_final_bf, b_v)
    p_u = u_ptr + base_v + offs_t[:, None] * H * V + offs_v[None, :]
    tl.store(p_u, b_u.to(u_ptr.dtype.element_ty))

    # g_last for inter-chunk decay
    b_g_last = tl.sum(tl.where(offs_t[:, None] == (BT - 1), b_g, 0.0), axis=0)
    p_g_last = g_last_ptr + (i_b * NT + i_t) * H * K + i_h * K + offs_k
    tl.store(p_g_last, b_g_last)

    # kg = k * exp(g_last - g)  (for state update)
    b_kg_out = b_k.to(tl.float32) * tl.exp2(b_g_last[None, :] - b_g)
    p_kg = kg_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    tl.store(p_kg, b_kg_out.to(kg_ptr.dtype.element_ty))

    # qg = q * exp(g) * scale  (for the S contribution to o)
    b_qg_global = b_q.to(tl.float32) * tl.exp2(b_g) * scale
    p_qg = qg_ptr + base_qk + offs_t[:, None] * H * K + offs_k[None, :]
    tl.store(p_qg, b_qg_global.to(qg_ptr.dtype.element_ty))


@triton.autotune(
    configs=[
        triton.Config({"BV": 16}, num_warps=2, num_stages=2),
        triton.Config({"BV": 16}, num_warps=4, num_stages=2),
        triton.Config({"BV": 32}, num_warps=4, num_stages=2),
    ],
    key=["NT"],
)
@triton.jit
def _kda_fused_h_o_kernel(
    qg_ptr,
    w_ptr,
    u_ptr,
    g_last_ptr,
    kg_ptr,
    Aqk_ptr,
    o_ptr,
    T,
    NT,
    H,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
):
    i_v = tl.program_id(0)
    i_bh = tl.program_id(1)
    i_b = i_bh // H
    i_h = i_bh % H

    offs_t = tl.arange(0, BT)
    offs_v_loc = tl.arange(0, BV)
    offs_v = i_v * BV + offs_v_loc
    # K=128 split into two 64-element tiles (register friendly)
    offs_k0 = tl.arange(0, 64)
    offs_k1 = 64 + tl.arange(0, 64)

    # Persistent KV state S, split across K for register tiling. fp32 accum.
    b_s0 = tl.zeros([64, BV], dtype=tl.float32)
    b_s1 = tl.zeros([64, BV], dtype=tl.float32)

    base_qk = i_b * T * H * K + i_h * K
    base_v = i_b * T * H * V + i_h * V
    base_a = i_b * T * H * BT + i_h * BT

    for i_t in range(NT):
        off_qk = base_qk + i_t * BT * H * K
        off_v = base_v + i_t * BT * H * V
        off_a = base_a + i_t * BT * H * BT

        # Load w (BT, K) split
        p_w0 = w_ptr + off_qk + offs_t[:, None] * H * K + offs_k0[None, :]
        p_w1 = w_ptr + off_qk + offs_t[:, None] * H * K + offs_k1[None, :]
        b_w0 = tl.load(p_w0)
        b_w1 = tl.load(p_w1)

        b_wS = tl.dot(b_w0, b_s0.to(b_w0.dtype))
        b_wS += tl.dot(b_w1, b_s1.to(b_w1.dtype))

        # v_new = u - w @ S
        p_u = u_ptr + off_v + offs_t[:, None] * H * V + offs_v[None, :]
        b_u = tl.load(p_u).to(tl.float32)
        b_v_new = b_u - b_wS

        # o = qg @ S
        p_qg0 = qg_ptr + off_qk + offs_t[:, None] * H * K + offs_k0[None, :]
        p_qg1 = qg_ptr + off_qk + offs_t[:, None] * H * K + offs_k1[None, :]
        b_qg0 = tl.load(p_qg0)
        b_qg1 = tl.load(p_qg1)

        b_o = tl.dot(b_qg0, b_s0.to(b_qg0.dtype))
        b_o += tl.dot(b_qg1, b_s1.to(b_qg1.dtype))

        # o += Aqk @ v_new   (Aqk is lower-triangular incl. diag per intra kernel)
        p_A = Aqk_ptr + off_a + offs_t[:, None] * H * BT + offs_t[None, :]
        b_A = tl.load(p_A)
        b_v_new_bf = b_v_new.to(b_w0.dtype)
        b_o += tl.dot(b_A, b_v_new_bf)

        p_o = o_ptr + off_v + offs_t[:, None] * H * V + offs_v[None, :]
        tl.store(p_o, b_o.to(o_ptr.dtype.element_ty))

        # State decay: S = S * exp(g_last)
        g_last_off = i_b * NT * H * K + i_t * H * K + i_h * K
        b_gl0 = tl.load(g_last_ptr + g_last_off + offs_k0)
        b_gl1 = tl.load(g_last_ptr + g_last_off + offs_k1)
        b_s0 = b_s0 * tl.exp2(b_gl0)[:, None]
        b_s1 = b_s1 * tl.exp2(b_gl1)[:, None]

        # State update: S += kg^T @ v_new
        p_kg0 = kg_ptr + off_qk + offs_t[:, None] * H * K + offs_k0[None, :]
        p_kg1 = kg_ptr + off_qk + offs_t[:, None] * H * K + offs_k1[None, :]
        b_kg0 = tl.load(p_kg0)
        b_kg1 = tl.load(p_kg1)

        b_s0 += tl.dot(tl.trans(b_kg0), b_v_new_bf)
        b_s1 += tl.dot(tl.trans(b_kg1), b_v_new_bf)


def _launch(q, k, v, g, beta, Aqk, w, u, kg, qg, g_last, o, B, T, H, K, V, BT, NT, scale):
    _kda_intra_kernel[(NT, B * H)](
        q, k, v, g, beta,
        Aqk, w, u, kg, qg, g_last,
        scale, T, NT, H,
        K=K, V=V, BT=BT, RCP_LN2=_RCP_LN2,
    )
    _kda_fused_h_o_kernel[lambda m: (triton.cdiv(V, m["BV"]), B * H)](
        qg, w, u, g_last, kg, Aqk, o,
        T, NT, H,
        K=K, V=V, BT=BT,
    )


def _alloc_intermediates(B, T, H, K, V, BT, NT, device, dtype):
    Aqk = torch.empty((B, T, H, BT), device=device, dtype=torch.bfloat16)
    w = torch.empty((B, T, H, K), device=device, dtype=torch.bfloat16)
    u = torch.empty((B, T, H, V), device=device, dtype=torch.bfloat16)
    kg = torch.empty((B, T, H, K), device=device, dtype=torch.bfloat16)
    qg = torch.empty((B, T, H, K), device=device, dtype=torch.bfloat16)
    g_last = torch.empty((B, NT, H, K), device=device, dtype=torch.float32)
    o = torch.empty((B, T, H, V), device=device, dtype=dtype)
    return Aqk, w, u, kg, qg, g_last, o


def _kda_forward_eager(q, k, v, g, beta, scale, chunk_size=64):
    q = q.contiguous()
    k = k.contiguous()
    v = v.contiguous()
    g = g.contiguous()
    beta = beta.contiguous()
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    assert T % BT == 0, f"T must be multiple of chunk_size"
    NT = T // BT
    assert K == 128 and V == 128, "kernel specialized for K=V=128"
    Aqk, w, u, kg, qg, g_last, o = _alloc_intermediates(B, T, H, K, V, BT, NT, q.device, v.dtype)
    _launch(q, k, v, g, beta, Aqk, w, u, kg, qg, g_last, o, B, T, H, K, V, BT, NT, scale)
    return o


class _GraphState:
    __slots__ = ("sig", "graph", "out", "keep")

    def __init__(self):
        self.sig = None
        self.graph = None
        self.out = None
        self.keep = None


class Model(nn.Module):
    """KDA forward (chunk form). No learned parameters; all inputs are activations."""

    def __init__(self, B: int, T: int, H: int, K: int, V: int, chunk_size: int = 64):
        super().__init__()
        self.B = B
        self.T = T
        self.H = H
        self.K = K
        self.V = V
        self.chunk_size = chunk_size
        self.scale = float(K) ** -0.5
        self.register_buffer("_dummy", torch.zeros(1), persistent=False)
        self._gs = _GraphState()
        self._graphs_ok = True

    def forward(self, q: torch.Tensor, k: torch.Tensor, v: torch.Tensor, g: torch.Tensor, beta: torch.Tensor) -> torch.Tensor:
        scale = self.scale
        BT = self.chunk_size
        B, T, H, K = q.shape
        V = v.shape[-1]

        use_graph = (
            self._graphs_ok
            and q.is_cuda
            and K == 128
            and V == 128
            and BT == 64
            and (T % BT == 0)
            and q.is_contiguous()
            and k.is_contiguous()
            and v.is_contiguous()
            and g.is_contiguous()
            and beta.is_contiguous()
        )
        if not use_graph:
            return _kda_forward_eager(q, k, v, g, beta, scale, BT)

        sig = (q.data_ptr(), k.data_ptr(), v.data_ptr(), g.data_ptr(), beta.data_ptr(), B, T, H, V, v.dtype)
        gs = self._gs
        if gs.sig == sig and gs.graph is not None:
            gs.graph.replay()
            return gs.out

        try:
            return self._build_graph(q, k, v, g, beta, B, T, H, K, V, BT, scale, sig)
        except Exception:
            self._graphs_ok = False
            self._gs = _GraphState()
            return _kda_forward_eager(q, k, v, g, beta, scale, BT)

    def _build_graph(self, q, k, v, g, beta, B, T, H, K, V, BT, scale, sig):
        NT = T // BT
        dev = q.device
        Aqk, w, u, kg, qg, g_last, o = _alloc_intermediates(B, T, H, K, V, BT, NT, dev, v.dtype)

        def run():
            _launch(q, k, v, g, beta, Aqk, w, u, kg, qg, g_last, o, B, T, H, K, V, BT, NT, scale)

        # Warmup on side stream for stable autotune + allocator before capture.
        warm = torch.cuda.Stream()
        warm.wait_stream(torch.cuda.current_stream())
        with torch.cuda.stream(warm):
            for _ in range(5):
                run()
        torch.cuda.current_stream().wait_stream(warm)
        torch.cuda.synchronize()

        # Return a correct eager result for *this* call (graph is for future replays only).
        result = o.clone()

        graph = torch.cuda.CUDAGraph()
        with torch.cuda.graph(graph):
            run()

        gs = self._gs
        gs.sig = sig
        gs.graph = graph
        gs.out = o
        gs.keep = (q, k, v, g, beta, Aqk, w, u, kg, qg, g_last)
        return result
