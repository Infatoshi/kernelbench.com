struct GemmKey {
  int64_t M;
  int64_t N;
  int64_t K;
  bool operator==(const GemmKey& o) const {
    return M == o.M && N == o.N && K == o.K;

};

struct GemmKeyHash {
  size_t operator()(const GemmKey& k) const {
    return (size_t(k.M) * 1315423911u) ^ (size_t(k.N) * 2654435761u) ^ size_t(k.K);
  }
};

static cublasStatus_t run_with_cache(
    cublasLtMatmulDesc_t desc,
    cublasLtMatrixLayout_t Adesc,
    cublasLtMatrixLayout_t Bdesc,
    cublasLtMatrixLayout_t Cdesc,

          out_ptr, Cdesc,

__device__ __forceinline__ __nv_bfloat16 fp8_to_bf16_dev(__nv_fp8_e4m3 v) {

__global__ void fp8_to_bf16_kpad_kernel(
    const __nv_fp8_e4m3* __restrict__ src,
    __nv_bfloat16* __restrict__ dst,
    int rows,
    int K_in,
    int K_pad) {
  int row = blockIdx.y;
  if (row >= rows) return;

  int col_base = (blockIdx.x * blockDim.x + threadIdx.x) * 16;
  if (col_base >= K_pad) return;

  __nv_bfloat16 tmp[16];
  const uint8_t* src_u8 = reinterpret_cast<const uint8_t*>(src);

  if (col_base + 16 <= K_in) {
    uint4 pack = *reinterpret_cast<const uint4*>(src_u8 + (size_t)row * K_in + col_base);
    const uint8_t* bytes = reinterpret_cast<const uint8_t*>(&pack);
    #pragma unroll
    for (int i = 0; i < 16; ++i) {
      __nv_fp8_e4m3 v;
      v.__x = bytes[i];
      tmp[i] = fp8_to_bf16_dev(v);
    }
  } else {
    #pragma unroll
    for (int i = 0; i < 16; ++i) {
      int c = col_base + i;
      if (c < K_in) {
        __nv_fp8_e4m3 v;
        v.__x = src_u8[(size_t)row * K_in + c];
        tmp[i] = fp8_to_bf16_dev(v);
      } else {
        tmp[i] = __float2bfloat16(0.0f);
      }
    }
  }

  *reinterpret_cast<uint4*>(dst + (size_t)row * K_pad + col_base) =
      *reinterpret_cast<const uint4*>(tmp);
  *reinterpret_cast<uint4*>(dst + (size_t)row * K_pad + col_base + 8) =
      *reinterpret_cast<const uint4*>(tmp + 8);
}

    """y = x.to(bf16) @ weight.T via custom fp8 cast + cuBLASLt BF16 GEMM."""

    def __init__(self, M: int, N: int, K: int):
        super().__init__()
        self.M, self.N, self.K = M, N, K
        self.K_pad = _next_multiple(K, K_ALIGN)
        self.weight = nn.Parameter(torch.empty(N, K, dtype=torch.bfloat16))
        nn.init.normal_(self.weight, std=0.02)

        self.register_buffer(
            "_weight_padded",
            torch.empty(N, self.K_pad, dtype=torch.bfloat16),
            persistent=False,
        )
        self.register_buffer(
            "_x_bf",
            torch.empty(M, self.K_pad, dtype=torch.bfloat16),
            persistent=False,
        )
        self.register_buffer(
            "_out",
            torch.empty(M, N, dtype=torch.bfloat16),
            persistent=False,
        )
        self._weight_version = -1

    def _materialize_weight_padded(self) -> None:
        if self.K_pad != self.K:
            self._weight_padded.zero_()
        self._weight_padded[:, : self.K].copy_(self.weight.detach())
        self._weight_version = self.weight._version

    def load_state_dict(self, state_dict, strict: bool = True, assign: bool = False):
        result = super().load_state_dict(state_dict, strict=strict, assign=assign)
        self._weight_version = -1
        return result

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if not x.is_cuda:
            return x.to(torch.bfloat16) @ self.weight.to(torch.bfloat16).T
