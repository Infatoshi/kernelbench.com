"""Custom Triton paged-attention decode for packed bf16 KV cache.

The main path is FlashDecoding-style split-K: stage 1 streams a partition of
the paged KV cache for one (batch, kv_head) pair and computes online-softmax
partials for the GQA query heads sharing that KV head; stage 2 merges the
partials.  A single-kernel path is used when the natural block count is already
high enough to avoid the second launch and scratch traffic.
"""
import math

import torch
import torch.nn as nn
import triton
import triton.language as tl

OP_TYPE = "attention"
SUPPORTED_PRECISIONS = ["bf16"]
HARDWARE_REQUIRED = ["RTX_PRO_6000", "H100", "B200"]

BATCH = 8
NUM_HEADS = 32
NUM_KV_HEADS = 8
HEAD_DIM = 128
SEQ_LEN = 1024
PAGE_SIZE = 16


@triton.jit
def _direct_kernel(
    Q,
    KV,
    BT,
    SL,
    OUT,
    SCALE: tl.constexpr,
    H: tl.constexpr,
    Hkv: tl.constexpr,
    D: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    NUM_PAGES: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    BLOCK_N: tl.constexpr,
    PAGES_PER_TILE: tl.constexpr,
):
    b = tl.program_id(0)
    kvh = tl.program_id(1)

    offs_g = tl.arange(0, GROUP_SIZE)
    offs_d = tl.arange(0, D)
    offs_n = tl.arange(0, BLOCK_N)
    q_heads = kvh * GROUP_SIZE + offs_g

    q = tl.load(Q + (b * H + q_heads[:, None]) * D + offs_d[None, :])
    seq_len = tl.load(SL + b)

    m_i = tl.full((GROUP_SIZE,), -float("inf"), tl.float32)
    l_i = tl.zeros((GROUP_SIZE,), tl.float32)
    acc = tl.zeros((GROUP_SIZE, D), tl.float32)

    for page_base in range(0, NUM_PAGES, PAGES_PER_TILE):
        page_ids = page_base + offs_n // PAGE_SIZE
        page_offs = offs_n % PAGE_SIZE
        tok = page_ids * PAGE_SIZE + page_offs
        valid_n = (page_ids < NUM_PAGES) & (tok < seq_len)
        block_ids = tl.load(BT + b * NUM_PAGES + page_ids, mask=page_ids < NUM_PAGES, other=0)

        k_token = block_ids[None, :] * PAGE_SIZE + page_offs[None, :]
        v_token = block_ids[:, None] * PAGE_SIZE + page_offs[:, None]
        k_base = ((k_token * Hkv + kvh) * (2 * D)) + offs_d[:, None]
        v_base = ((v_token * Hkv + kvh) * (2 * D)) + D + offs_d[None, :]

        k = tl.load(KV + k_base, mask=valid_n[None, :], other=0.0, eviction_policy="evict_first")
        v = tl.load(KV + v_base, mask=valid_n[:, None], other=0.0, eviction_policy="evict_first")

        scores = tl.dot(q, k, out_dtype=tl.float32) * SCALE
        scores = tl.where(valid_n[None, :], scores, -float("inf"))

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        alpha = tl.exp(m_i - m_new)
        p = tl.exp(scores - m_new[:, None])
        p = tl.where(valid_n[None, :], p, 0.0)
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v, out_dtype=tl.float32)
        m_i = m_new

    out = acc / l_i[:, None]
    tl.store(OUT + (b * H + q_heads[:, None]) * D + offs_d[None, :], out)


@triton.jit
def _phase1_kernel(
    Q,
    KV,
    BT,
    SL,
    PARTIAL_O,
    PARTIAL_M,
    PARTIAL_L,
    SCALE_LOG2E: tl.constexpr,
    H: tl.constexpr,
    Hkv: tl.constexpr,
    D: tl.constexpr,
    PAGE_SIZE: tl.constexpr,
    PARTITION_SIZE: tl.constexpr,
    MAX_BLOCKS: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    PAD_G: tl.constexpr,
    BLOCK_N: tl.constexpr,
    BLOCK_D: tl.constexpr,
    NUM_PARTITIONS: tl.constexpr,
    N_ITERS: tl.constexpr,
):
    part_id = tl.program_id(0)
    kvh = tl.program_id(1)
    b = tl.program_id(2)

    seq_len = tl.load(SL + b)
    part_start = part_id * PARTITION_SIZE
    part_end = tl.minimum(part_start + PARTITION_SIZE, seq_len)

    offs_g = tl.arange(0, PAD_G)
    offs_d = tl.arange(0, BLOCK_D)
    offs_n = tl.arange(0, BLOCK_N)
    g_mask = offs_g < GROUP_SIZE
    d_mask = offs_d < D

    h_base = kvh * GROUP_SIZE
    q = tl.load(
        Q + ((b * H + h_base + offs_g[:, None]) * D + offs_d[None, :]),
        mask=g_mask[:, None] & d_mask[None, :],
        other=0.0,
    )
    q = (q.to(tl.float32) * SCALE_LOG2E).to(tl.bfloat16)

    neg = -1.0e30
    m_i = tl.full((PAD_G,), neg, tl.float32)
    l_i = tl.zeros((PAD_G,), tl.float32)
    acc = tl.zeros((PAD_G, BLOCK_D), tl.float32)

    for it in tl.static_range(N_ITERS):
        n_start = part_start + it * BLOCK_N
        tok = n_start + offs_n
        valid_n = tok < part_end
        page_idx = tok // PAGE_SIZE
        page_off = tok % PAGE_SIZE
        block_id = tl.load(BT + b * MAX_BLOCKS + page_idx, mask=valid_n, other=0)

        kv_off = ((block_id * PAGE_SIZE + page_off) * Hkv + kvh) * (2 * D)
        k = tl.load(
            KV + kv_off[:, None] + offs_d[None, :],
            mask=valid_n[:, None] & d_mask[None, :],
            other=0.0,
            eviction_policy="evict_first",
        )
        v = tl.load(
            KV + kv_off[:, None] + D + offs_d[None, :],
            mask=valid_n[:, None] & d_mask[None, :],
            other=0.0,
            eviction_policy="evict_first",
        )

        scores = tl.dot(q, tl.trans(k), out_dtype=tl.float32)
        scores = tl.where(valid_n[None, :], scores, neg)

        m_new = tl.maximum(m_i, tl.max(scores, axis=1))
        alpha = tl.exp2(m_i - m_new)
        p = tl.exp2(scores - m_new[:, None])
        p = tl.where(valid_n[None, :], p, 0.0)
        l_i = l_i * alpha + tl.sum(p, axis=1)
        acc = acc * alpha[:, None] + tl.dot(p.to(tl.bfloat16), v, out_dtype=tl.float32)
        m_i = m_new

    ml_off = ((b * Hkv + kvh) * NUM_PARTITIONS + part_id) * GROUP_SIZE + offs_g
    tl.store(PARTIAL_M + ml_off, m_i, mask=g_mask)
    tl.store(PARTIAL_L + ml_off, l_i, mask=g_mask)
    o_off = (
        (((b * Hkv + kvh) * NUM_PARTITIONS + part_id) * GROUP_SIZE + offs_g[:, None]) * D
        + offs_d[None, :]
    )
    tl.store(PARTIAL_O + o_off, acc, mask=g_mask[:, None] & d_mask[None, :])


@triton.jit
def _phase2_kernel(
    PARTIAL_O,
    PARTIAL_M,
    PARTIAL_L,
    OUT,
    H: tl.constexpr,
    Hkv: tl.constexpr,
    GROUP_SIZE: tl.constexpr,
    D: tl.constexpr,
    BLOCK_D: tl.constexpr,
    BLOCK_P: tl.constexpr,
    NUM_PARTITIONS: tl.constexpr,
):
    h = tl.program_id(0)
    b = tl.program_id(1)

    kvh = h // GROUP_SIZE
    g = h % GROUP_SIZE

    offs_p = tl.arange(0, BLOCK_P)
    offs_d = tl.arange(0, BLOCK_D)
    p_mask = offs_p < NUM_PARTITIONS
    d_mask = offs_d < D

    neg = -1.0e30
    ml_off = ((b * Hkv + kvh) * NUM_PARTITIONS + offs_p) * GROUP_SIZE + g
    m_arr = tl.load(PARTIAL_M + ml_off, mask=p_mask, other=neg)
    l_arr = tl.load(PARTIAL_L + ml_off, mask=p_mask, other=0.0)

    m_global = tl.max(m_arr, axis=0)
    weights = tl.exp2(m_arr - m_global)
    weights = tl.where(p_mask, weights, 0.0)
    l_global = tl.sum(weights * l_arr, axis=0)

    o_off = (
        (((b * Hkv + kvh) * NUM_PARTITIONS + offs_p[:, None]) * GROUP_SIZE + g) * D
        + offs_d[None, :]
    )
    o_arr = tl.load(PARTIAL_O + o_off, mask=p_mask[:, None] & d_mask[None, :], other=0.0)
    out = tl.sum(weights[:, None] * o_arr, axis=0) / l_global

    tl.store(OUT + (b * H + h) * D + offs_d, out.to(tl.bfloat16), mask=d_mask)


def _next_pow2(x: int) -> int:
    return 1 << (x - 1).bit_length()


_NUM_SMS = 188


def _pick_partition(batch: int, num_kv_heads: int, seq_len: int, block_n: int) -> tuple[int, int]:
    base_blocks = batch * num_kv_heads
    target_blocks = _NUM_SMS * 4
    if base_blocks >= target_blocks:
        num_partitions = 1
    else:
        num_partitions = (target_blocks + base_blocks - 1) // base_blocks
    partition_size = (seq_len + num_partitions - 1) // num_partitions
    partition_size = ((partition_size + block_n - 1) // block_n) * block_n
    num_partitions = (seq_len + partition_size - 1) // partition_size
    return partition_size, num_partitions

    def __init__(
        self,
        batch: int,
        num_heads: int,
        num_kv_heads: int,
        head_dim: int,
        seq_len: int,
        page_size: int,

        self.page_size = page_size

        self.scale_log2e = self.scale * 1.4426950408889634

            batch, num_kv_heads, seq_len, self.block_n
        )
        self.use_direct = head_dim == 128 and self.group_size == 4 and batch * num_kv_heads >= 128
        self.register_buffer("_dummy", torch.zeros(1, dtype=torch.bfloat16), persistent=False)
        self._scratch = None
        self._graph_cache = {}

    def _alloc_buffers(self, device: torch.device, out_dtype: torch.dtype):
        if self._scratch is not None:
            return self._scratch

        out = torch.empty(self.batch, self.num_heads, self.head_dim, dtype=out_dtype, device=device)
        if self.use_direct:
            self._scratch = (out, None, None, None)
            return self._scratch

            self.batch,
            self.num_kv_heads,
            self.num_partitions,
            self.group_size,
            self.head_dim,
            dtype=torch.float32,
            device=device,
        )
        partial_m = torch.empty(
            self.batch,
            self.num_kv_heads,
            self.num_partitions,
            self.group_size,
            dtype=torch.float32,
            device=device,
        )
        partial_l = torch.empty(
            self.batch,
            self.num_kv_heads,
            self.num_partitions,
            self.group_size,
            dtype=torch.float32,
            device=device,
        )
        self._scratch = (out, partial_o, partial_m, partial_l)
        return self._scratch

            partial_l,

    def forward(
        self,
        query: torch.Tensor,
        kv_cache: torch.Tensor,
        block_table: torch.Tensor,
        seq_lens: torch.Tensor,
    ) -> torch.Tensor:
        buffers = self._alloc_buffers(query.device, query.dtype)
        out = buffers[0]
        key = (query.data_ptr(), kv_cache.data_ptr(), block_table.data_ptr(), seq_lens.data_ptr())

        if graph is None:

            self._graph_cache[key] = graph
        graph.replay()
        return out

def _pick_partition(B: int, Hkv: int, L: int, block_n: int):
    """Choose partition_size to balance block count and per-block work.
