"""Optimized KDA forward (chunk form) for RTX PRO 6000 (SM120 Blackwell).

Vectorized PyTorch implementation: replaces Python loops over chunk positions
with batched matmuls for the A-matrix construction, keeps the sequential
cumulative-invert and inter-chunk recurrence as tight Python loops.
"""
from __future__ import annotations

import torch
import torch.nn as nn


def _kda_fwd(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int = 64,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    NT = T // BT
    device = q.device

    # ---- Cast to fp32 and scale q ----
    q_f = q.float() * scale
    k_f = k.float()
    v_f = v.float()
    g_f = g.float()
    beta_f = beta.float()

    # ---- Reshape to (B, H, N, C, K/V) ----
    q_c = q_f.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4).contiguous()
    k_c = k_f.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4).contiguous()
    v_c = v_f.view(B, NT, BT, H, V).permute(0, 3, 1, 2, 4).contiguous()
    g_c = g_f.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4).contiguous()
    beta_c = beta_f.view(B, NT, BT, H).permute(0, 3, 1, 2).contiguous()

    # ---- Cumsum g within each chunk ----
    g_c = g_c.cumsum(dim=-2)

    # ---- Precompute exp(g) and exp(-g) scaled keys/queries ----
    g_exp = g_c.exp()
    g_neg_exp = (-g_c).exp()
    kg = k_c * g_exp        # k * exp(g)
    kng = k_c * g_neg_exp   # k * exp(-g)
    qg = q_c * g_exp        # q * exp(g)

    # ---- Build Akk via batched matmul ----
    # A[i,c] = sum_d kng[i,d] * kg[c,d] = (kng @ kg^T)[i,c]
    A = torch.matmul(kng, kg.transpose(-1, -2))  # (B,H,N,C,C)

    # Mask upper triangle + diagonal to 0, negate, then left-multiply by diag(beta)
    mask_du = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=0)
    A.masked_fill_(mask_du, 0)
    A.neg_()
    A.mul_(beta_c.unsqueeze(-1))

    # ---- Cumulative invert (forward substitution for (I - A_lower)^{-1}) ----
    for i in range(1, BT):
        A[..., i, :i] += (A[..., i, :, None] * A[..., :, :i]).sum(-2)

    # Add identity, right-multiply by diag(beta)
    eye = torch.eye(BT, dtype=torch.float32, device=device)
    A.add_(eye)
    A.mul_(beta_c.unsqueeze(-2))

    # ---- W and U ----
    w = torch.matmul(A, kg)    # (B,H,N,C,K)
    u = torch.matmul(A, v_c)   # (B,H,N,C,V)

    # ---- Build Aqk via batched matmul ----
    # Aqk[j,c] = sum_d kng[j,d] * qg[c,d] = (kng @ qg^T)[j,c]
    Aqk = torch.matmul(kng, qg.transpose(-1, -2))  # (B,H,N,C,C)
    mask_su = torch.triu(torch.ones(BT, BT, dtype=torch.bool, device=device), diagonal=1)
    Aqk.masked_fill_(mask_su, 0)

    # ---- Inter-chunk recurrence ----
    S = q_c.new_zeros(B, H, K, V)
    o = torch.zeros_like(v_c)

    for i in range(NT):
        u_i = u[:, :, i]
        w_i = w[:, :, i]
        qg_i = qg[:, :, i]
        Aqk_i = Aqk[:, :, i]
        g_i = g_c[:, :, i]
        k_i = k_c[:, :, i]

        v_i = u_i - torch.matmul(w_i, S)
        o[:, :, i] = torch.matmul(qg_i, S) + torch.matmul(Aqk_i, v_i)

        g_last_exp = g_i[:, :, -1].exp().unsqueeze(-1)  # (B,H,K,1)
        S.mul_(g_last_exp)
        kg_shifted = k_i * (g_i[:, :, -1:] - g_i).exp()
        S.add_(torch.matmul(kg_shifted.transpose(-1, -2), v_i))

    return o.permute(0, 2, 3, 1, 4).reshape(B, T, H, V).to(dtype)
