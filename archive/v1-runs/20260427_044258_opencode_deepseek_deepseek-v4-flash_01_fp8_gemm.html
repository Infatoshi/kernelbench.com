#include <cstdint>
#include <cuda_bf16.h>
#include <cuda_fp8.h>
#include <cublas_v2.h>

__global__ void fp8_to_bf16_kernel(
    const __nv_fp8_e4m3* __restrict__ src,
    __nv_bfloat16* __restrict__ dst,
    int n
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) dst[idx] = __nv_bfloat16(src[idx]);
}

extern "C" {

void fp8_gemm_forward(
    unsigned long long A_fp8_ptr,
    unsigned long long B_ptr,
    unsigned long long C_ptr,
    int M, int N, int K
) {
    const __nv_fp8_e4m3* A_fp8 = (const __nv_fp8_e4m3*)(uintptr_t)A_fp8_ptr;
    const __nv_bfloat16* B = (const __nv_bfloat16*)(uintptr_t)B_ptr;
    __nv_bfloat16* C = (__nv_bfloat16*)(uintptr_t)C_ptr;

    __nv_bfloat16* A_bf16;
    cudaMallocAsync(&A_bf16, (size_t)M * K * sizeof(__nv_bfloat16), 0);

    int threads = 256;
    int blocks = (M * K + threads - 1) / threads;
    fp8_to_bf16_kernel<<<blocks, threads>>>(A_fp8, A_bf16, M * K);

    cublasHandle_t handle;
    cublasCreate(&handle);

    float alpha = 1.0f, beta = 0.0f;

    // C(M,N) = A(M,K) @ B(N,K)^T   (all row-major)
    //
    // cuBLAS is column-major. A row-major matrix (M,K) with stride K becomes
    // a column-major matrix (K,M) with leading dim K:
    //   A_col(k,i) = memory[k + i*K] = (A_row[i])[k] = A_row(i,k)
    // op(A) = A_col^T via CUBLAS_OP_T → op(A)(i,k) = A_col(k,i) = A_row(i,k) ✓
    // op(B) = B_col via CUBLAS_OP_N → op(B)(k,j) = B_col(k,j) = B_row(j,k) ✓
    // C(i,j) = sum_k op(A)(i,k) * op(B)(k,j) = sum_k A(i,k)*B(j,k) = (A@B^T)[i,j] ✓

    cublasGemmEx(
        handle,
        CUBLAS_OP_T, CUBLAS_OP_N,
        M, N, K,
        &alpha,
        A_bf16, CUDA_R_16BF, K,
        B,      CUDA_R_16BF, K,
        &beta,
        C,      CUDA_R_16BF, M,
        CUDA_R_32F,
        CUBLAS_GEMM_DEFAULT

    cublasDestroy(handle);
    cudaFreeAsync(A_bf16, 0);
}

}  // extern "C"
