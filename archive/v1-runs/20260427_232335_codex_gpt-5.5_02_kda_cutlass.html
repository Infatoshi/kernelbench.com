@triton.jit
def _state_kernel(
    kg,
    w,
    u,
    g,
    hbuf,
    vnew,
    T: tl.constexpr,
    H: tl.constexpr,
    NT: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
):
    pid_v = tl.program_id(0)
    pid_bh = tl.program_id(1)
    b = pid_bh // H
    hidx = pid_bh - b * H

    offs_v = pid_v * BV + tl.arange(0, BV)
    offs_c = tl.arange(0, BT)
    offs_k = tl.arange(0, 64)
    mask_v = offs_v < V

    s0 = tl.zeros((64, BV), dtype=tl.float32)
    s1 = tl.zeros((64, BV), dtype=tl.float32)

    for n in range(0, NT):
        h_base = (((b * NT + n) * H + hidx) * K) * V
        tl.store(
            hbuf + h_base + offs_k[:, None] * V + offs_v[None, :],
            s0.to(hbuf.dtype.element_ty),
            mask=mask_v[None, :],
        )
        tl.store(
            hbuf + h_base + (offs_k[:, None] + 64) * V + offs_v[None, :],
            s1.to(hbuf.dtype.element_ty),
            mask=mask_v[None, :],
        )

        t = n * BT + offs_c
        w_base = ((b * T + t[:, None]) * H + hidx) * K
        w0 = tl.load(w + w_base + offs_k[None, :])
        w1 = tl.load(w + w_base + offs_k[None, :] + 64)
        projected = tl.dot(w0, s0.to(w0.dtype)) + tl.dot(w1, s1.to(w1.dtype))

        uv_base = ((b * T + t[:, None]) * H + hidx) * V
        u_block = tl.load(u + uv_base + offs_v[None, :], mask=mask_v[None, :], other=0.0)
        vn = u_block.to(tl.float32) - projected
        tl.store(
            vnew + uv_base + offs_v[None, :],
            vn.to(vnew.dtype.element_ty),
            mask=mask_v[None, :],
        )

        last_t = n * BT + BT - 1
        g_base = ((b * T + last_t) * H + hidx) * K
        g0 = tl.load(g + g_base + offs_k).to(tl.float32)
        g1 = tl.load(g + g_base + offs_k + 64).to(tl.float32)
        s0 *= tl.exp(g0)[:, None]
        s1 *= tl.exp(g1)[:, None]

        kg_base = ((b * T + t[:, None]) * H + hidx) * K
        kg0 = tl.load(kg + kg_base + offs_k[None, :])
        kg1 = tl.load(kg + kg_base + offs_k[None, :] + 64)
        vn_tc = vn.to(kg0.dtype)
        s0 += tl.dot(tl.trans(kg0), vn_tc)
        s1 += tl.dot(tl.trans(kg1), vn_tc)


@triton.jit
def _out_kernel(
    q,
    g,
    amat,
    hbuf,
    vnew,
    out,
    scale: tl.constexpr,
    T: tl.constexpr,
    H: tl.constexpr,
    NT: tl.constexpr,
    K: tl.constexpr,
    V: tl.constexpr,
    BT: tl.constexpr,
    BV: tl.constexpr,
):
    pid_v = tl.program_id(0)
    n = tl.program_id(1)
    pid_bh = tl.program_id(2)
    b = pid_bh // H
    hidx = pid_bh - b * H

    offs_v = pid_v * BV + tl.arange(0, BV)
    offs_c = tl.arange(0, BT)
    offs_k = tl.arange(0, 64)
    mask_v = offs_v < V
    t = n * BT + offs_c

    qg_base = ((b * T + t[:, None]) * H + hidx) * K
    q0 = tl.load(q + qg_base + offs_k[None, :])
    q1 = tl.load(q + qg_base + offs_k[None, :] + 64)
    g0 = tl.load(g + qg_base + offs_k[None, :]).to(tl.float32)
    g1 = tl.load(g + qg_base + offs_k[None, :] + 64).to(tl.float32)
    q0 = (q0 * tl.exp(g0)).to(q0.dtype)
    q1 = (q1 * tl.exp(g1)).to(q1.dtype)

    h_base = (((b * NT + n) * H + hidx) * K) * V
    hs0 = tl.load(
        hbuf + h_base + offs_k[:, None] * V + offs_v[None, :],
        mask=mask_v[None, :],
        other=0.0,
    )
    hs1 = tl.load(
        hbuf + h_base + (offs_k[:, None] + 64) * V + offs_v[None, :],
        mask=mask_v[None, :],
        other=0.0,
    )
    acc = tl.dot(q0, hs0) + tl.dot(q1, hs1)
    acc *= scale

    a_base = ((b * T + t[:, None]) * H + hidx) * BT
    a = tl.load(amat + a_base + offs_c[None, :])
    uv_base = ((b * T + t[:, None]) * H + hidx) * V
    vn = tl.load(vnew + uv_base + offs_v[None, :], mask=mask_v[None, :], other=0.0)
    acc += tl.dot(a, vn)

    tl.store(
        out + uv_base + offs_v[None, :],
        acc.to(out.dtype.element_ty),
        mask=mask_v[None, :],
    )


def _forward_impl(
    q: torch.Tensor,
    k: torch.Tensor,
    v: torch.Tensor,
    g: torch.Tensor,
    beta: torch.Tensor,
    scale: float,
    chunk_size: int,
) -> torch.Tensor:
    dtype = v.dtype
    B, T, H, K = q.shape
    V = v.shape[-1]
    BT = chunk_size
    NT = T // BT
    if K != 128 or V != 128 or BT != 64:
        raise RuntimeError("this kernel is specialized for K=V=128 and chunk_size=64")

    gf = g.float()
    bf = beta.float()

    q_bhnck = q.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4).contiguous()
    k_bhnck = k.view(B, NT, BT, H, K).permute(0, 3, 1, 2, 4).contiguous()
    v_bhncv = v.view(B, NT, BT, H, V).permute(0, 3, 1, 2, 4).contiguous()
    g_bhnck = (
        gf.view(B, NT, BT, H, K)
        .permute(0, 3, 1, 2, 4)
        .contiguous()
        .cumsum(-2)
    )
    b_bhnc = bf.view(B, NT, BT, H).permute(0, 3, 1, 2).contiguous()

    flat = B * H * NT
    tc_setup = T >= 2048 and B * H >= 8
    k_flat_raw = k_bhnck.reshape(flat, BT, K)
    g_flat = g_bhnck.reshape(flat, BT, K)
    v_flat_raw = v_bhncv.reshape(flat, BT, V)
    beta_flat = b_bhnc.reshape(flat, BT)

    eg = g_flat.exp()
    if tc_setup:
        kg_exp = (k_flat_raw.float() * eg).to(dtype)
        kg_inv = (k_flat_raw.float() * (-g_flat).exp()).to(dtype)
        v_flat = v_flat_raw
        kk = torch.bmm(kg_exp, kg_inv.transpose(1, 2)).float()
    else:
        k_flat = k_flat_raw.float()
        v_flat = v_flat_raw.float()
        kg_exp = k_flat * eg
        kg_inv = k_flat * (-g_flat).exp()
        kk = torch.bmm(kg_exp, kg_inv.transpose(1, 2))
    lower_strict = torch.tril(
        torch.ones(BT, BT, dtype=torch.bool, device=q.device),
        diagonal=-1,
    )
    lmat = (-kk * beta_flat[:, :, None]).masked_fill(~lower_strict, 0.0)
    eye = torch.eye(BT, dtype=torch.float32, device=q.device)
    rhs = torch.diag_embed(beta_flat)
    amat_full = torch.linalg.solve_triangular(
        eye.expand(flat, BT, BT) - lmat,
        rhs,
        upper=False,
        unitriangular=True,
    )

    if tc_setup:
        amat_mm = amat_full.to(dtype)
        w = torch.bmm(amat_mm, kg_exp)
        u = torch.bmm(amat_mm, v_flat)
    else:
        w = torch.bmm(amat_full, kg_exp)
        u = torch.bmm(amat_full, v_flat)

    q_scaled = (q_bhnck.float() * scale).reshape(flat, BT, K)
    if tc_setup:
        aqk = torch.bmm((q_scaled * eg).to(dtype), kg_inv.transpose(1, 2)).float()
    else:
        aqk = torch.bmm(q_scaled * eg, kg_inv.transpose(1, 2))
    lower = torch.tril(torch.ones(BT, BT, dtype=torch.bool, device=q.device))
    aqk = aqk.masked_fill(~lower, 0.0)

    w_bthk = (
        w.reshape(B, H, NT, BT, K)
        .permute(0, 2, 3, 1, 4)
        .reshape(B, T, H, K)
        .contiguous()
        .to(dtype)
    )
    u_bthv = (
        u.reshape(B, H, NT, BT, V)
        .permute(0, 2, 3, 1, 4)
        .reshape(B, T, H, V)
        .contiguous()
        .to(dtype)
    )
    aqk_bthc = (
        aqk.reshape(B, H, NT, BT, BT)
        .permute(0, 2, 3, 1, 4)
        .reshape(B, T, H, BT)
        .contiguous()
        .to(dtype)
    )

    g_last = g_bhnck[:, :, :, -1:, :]
    kg_state = (
        (k_bhnck.float() * (g_last - g_bhnck).exp())
        .permute(0, 2, 3, 1, 4)
        .reshape(B, T, H, K)
        .contiguous()
        .to(dtype)
    )
    g_bthk = (
        g_bhnck.permute(0, 2, 3, 1, 4)
        .reshape(B, T, H, K)
        .contiguous()
    )

    hbuf = torch.empty((B, NT, H, K, V), device=q.device, dtype=dtype)
    vnew = torch.empty_like(v)
    out = torch.empty_like(v)
    grid_state = (triton.cdiv(V, 64), B * H)
    _state_kernel[grid_state](
        kg_state,
        w_bthk,
        u_bthv,
        g_bthk,
        hbuf,
        vnew,
        T,
        H,
        NT,
        K,
        V,
        BT,
        BV=64,
        num_warps=4,
        num_stages=3,
    )
    grid_out = (triton.cdiv(V, 64), NT, B * H)
    _out_kernel[grid_out](
        q,
        g_bthk,
        aqk_bthc,
        hbuf,
        vnew,
        out,
        scale,
        T,
        H,
        NT,
        K,
        V,
        BT,
        BV=64,
        num_warps=4,
        num_stages=3,
    )
    return out
