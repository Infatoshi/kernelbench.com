"""Grouped GEMM + fused SwiGLU for MoE up-projection using Triton.

Per expert e, computes:
    h_e = silu(x_e @ W_gate[e]) * (x_e @ W_up[e])
where x_e = hidden_states[offsets[e]:offsets[e+1]].

Two GEMMs share the same input tile x_e; fused SwiGLU avoids writing
intermediate gate/up results to DRAM.
"""
from __future__ import annotations

import torch
import torch.nn as nn
import triton
import triton.language as tl


@triton.autotune(
    configs=[
        triton.Config({'BM': 64, 'BN': 128, 'BK': 32}, num_warps=4, num_stages=3),
        triton.Config({'BM': 64, 'BN': 128, 'BK': 64}, num_warps=4, num_stages=2),
        triton.Config({'BM': 128, 'BN': 64, 'BK': 32}, num_warps=4, num_stages=3),
        triton.Config({'BM': 128, 'BN': 64, 'BK': 64}, num_warps=4, num_stages=2),
        triton.Config({'BM': 128, 'BN': 128, 'BK': 32}, num_warps=8, num_stages=2),
        triton.Config({'BM': 64, 'BN': 64, 'BK': 32}, num_warps=4, num_stages=3),
        triton.Config({'BM': 64, 'BN': 64, 'BK': 64}, num_warps=4, num_stages=3),
        triton.Config({'BM': 128, 'BN': 128, 'BK': 64}, num_warps=8, num_stages=2),
    ],
    key=[],
)
@triton.jit
def _grouped_gemm_swiglu_kernel(
    # Pointers
    hidden_ptr, W_gate_ptr, W_up_ptr, out_ptr, offsets_ptr,
    # Strides
    stride_hm, stride_hh,
    stride_wge, stride_wgh, stride_wgi,
    stride_wue, stride_wuh, stride_wui,
    stride_om, stride_oi,
    # Compile-time dimensions (enables loop unrolling + software pipelining)
    H: tl.constexpr,
    I: tl.constexpr,
    # Tile sizes (autotuned)
    BM: tl.constexpr,
    BN: tl.constexpr,
    BK: tl.constexpr,
):
    # Compute grid decomposition at compile time
    n_blocks_n: tl.constexpr = (I + BN - 1) // BN

    expert_idx = tl.program_id(0)
    tile_id = tl.program_id(1)

    n_block = tile_id % n_blocks_n
    m_block = tile_id // n_blocks_n

    # Expert row range
    start = tl.load(offsets_ptr + expert_idx).to(tl.int32)
    end = tl.load(offsets_ptr + expert_idx + 1).to(tl.int32)
    m_size = end - start

    m_off = m_block * BM
    if m_off >= m_size:
        return

    n_off = n_block * BN

    # Index arrays
    rows = start + m_off + tl.arange(0, BM)
    cols_n = n_off + tl.arange(0, BN)

    # Boundary masks
    m_mask = (m_off + tl.arange(0, BM)) < m_size
    n_mask = cols_n < I

    # Float32 accumulators for gate and up projections
    acc_gate = tl.zeros((BM, BN), dtype=tl.float32)
    acc_up = tl.zeros((BM, BN), dtype=tl.float32)

    # ---- GEMM loop over K=H (fully unrolled when H is constexpr) ----
    for k_start in range(0, H, BK):
        k_offs = k_start + tl.arange(0, BK)

        # Load x tile (BM, BK) — shared by both GEMMs
        x = tl.load(
            hidden_ptr + rows[:, None] * stride_hm + k_offs[None, :] * stride_hh,
            mask=m_mask[:, None],
            other=0.0,
        )

        # Load W_gate tile (BK, BN)
        wg = tl.load(
            W_gate_ptr + expert_idx * stride_wge
            + k_offs[:, None] * stride_wgh + cols_n[None, :] * stride_wgi,
            mask=n_mask[None, :],
            other=0.0,
        )

        # Load W_up tile (BK, BN)
        wu = tl.load(
            W_up_ptr + expert_idx * stride_wue
            + k_offs[:, None] * stride_wuh + cols_n[None, :] * stride_wui,
            mask=n_mask[None, :],
            other=0.0,
        )

        acc_gate = tl.dot(x, wg, acc_gate)
        acc_up = tl.dot(x, wu, acc_up)

    # ---- Fused SwiGLU epilogue: silu(gate) * up ----
    result = (acc_gate * tl.sigmoid(acc_gate)) * acc_up

    # Store output tile
    tl.store(
        out_ptr + rows[:, None] * stride_om + cols_n[None, :] * stride_oi,
        result.to(tl.bfloat16),
        mask=m_mask[:, None] & n_mask[None, :],
    )


def _launch(
    hidden_states: torch.Tensor,
    W_gate: torch.Tensor,
    W_up: torch.Tensor,
    expert_offsets: torch.Tensor,
) -> torch.Tensor:
    T_perm, H = hidden_states.shape
    E = W_gate.shape[0]
    I = W_gate.shape[2]

    out = torch.empty(T_perm, I, dtype=torch.bfloat16, device=hidden_states.device)

    offsets_cpu = expert_offsets.cpu()

    def grid_fn(meta):
        BM = meta['BM']
        BN = meta['BN']
        nbn = triton.cdiv(I, BN)
        max_mb = 1
        for e in range(E):
            m_e = offsets_cpu[e + 1].item() - offsets_cpu[e].item()
            mb = triton.cdiv(m_e, BM)
            if mb > max_mb:
                max_mb = mb
        return (E, max_mb * nbn)

    _grouped_gemm_swiglu_kernel[grid_fn](
        hidden_states, W_gate, W_up, out, expert_offsets,
        hidden_states.stride(0), hidden_states.stride(1),
        W_gate.stride(0), W_gate.stride(1), W_gate.stride(2),
        W_up.stride(0), W_up.stride(1), W_up.stride(2),
        out.stride(0), out.stride(1),
        H=H, I=I,
    )
    return out

def _build_routing(T_total: int, E: int, K: int, device: str = "cpu") -> torch.Tensor:
    T_perm = T_total * K
    base = T_perm // E
